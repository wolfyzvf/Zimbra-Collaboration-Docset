 24 		 * @constructor

 27 		 * @param {Object} s Optional settings object for the control manager.

 29 		ControlManager : function(ed, s) {
 30 			var t = this, i;
 31 
 32 			s = s || {};
 33 			t.editor = ed;
 34 			t.controls = {};
 35 			t.onAdd = new tinymce.util.Dispatcher(t);
 36 			t.onPostRender = new tinymce.util.Dispatcher(t);
 37 			t.prefix = s.prefix || ed.id + '_';
 38 			t._cls = {};
 39 
 40 			t.onPostRender.add(function() {
 41 				each(t.controls, function(c) {
 42 					c.postRender();
 43 				});
 44 			});
 45 		},

 92 		/**
 93 		 * Adds a control to the control collection inside the manager.
 94 		 *
 95 		 * @method add
 96 		 * @param {tinymce.ui.Control} Control instance to add to collection.
 97 		 * @return {tinymce.ui.Control} Control instance that got passed in.
 98 		 */
 99 		add : function(c) {
100 			var t = this;
101 
102 			if (c) {
103 				t.controls[c.id] = c;
104 				t.onAdd.dispatch(c, t);
105 			}
106 
107 			return c;
108 		},

110 		/**
111 		 * Creates a control by name, when a control is created it will automatically add it to the control collection.
112 		 * It first ask all plugins for the specified control if the plugins didn't return a control then the default behavior
113 		 * will be used.
114 		 *
115 		 * @method createControl
116 		 * @param {String} name Control name to create for example "separator".
117 		 * @return {tinymce.ui.Control} Control instance that got created and added.
118 		 */
119 		createControl : function(name) {
120 			var ctrl, i, l, self = this, editor = self.editor, factories, ctrlName;
121 
122 			// Build control factory cache
123 			if (!self.controlFactories) {
124 				self.controlFactories = [];
125 				each(editor.plugins, function(plugin) {
126 					if (plugin.createControl) {
127 						self.controlFactories.push(plugin);
128 					}
129 				});
130 			}
131 
132 			// Create controls by asking cached factories
133 			factories = self.controlFactories;
134 			for (i = 0, l = factories.length; i < l; i++) {
135 				ctrl = factories[i].createControl(name, self);
136 
137 				if (ctrl) {
138 					return self.add(ctrl);
139 				}
140 			}
141 
142 			// Create sepearator
143 			if (name === "|" || name === "separator") {
144 				return self.createSeparator();
145 			}
146 
147 			// Create control from button collection
148 			if (editor.buttons && (ctrl = editor.buttons[name])) {
149 				return self.createButton(name, ctrl);
150 			}
151 
152 			return self.add(ctrl);
153 		},

155 		/**
156 		 * Creates a drop menu control instance by id.
157 		 *
158 		 * @method createDropMenu
159 		 * @param {String} id Unique id for the new dropdown instance. For example "some menu".
160 		 * @param {Object} s Optional settings object for the control.
161 		 * @param {Object} cc Optional control class to use instead of the default one.
162 		 * @return {tinymce.ui.Control} Control instance that got created and added.
163 		 */
164 		createDropMenu : function(id, s, cc) {
165 			var t = this, ed = t.editor, c, bm, v, cls;
166 
167 			s = extend({
168 				'class' : 'mceDropDown',
169 				constrain : ed.settings.constrain_menus
170 			}, s);
171 
172 			s['class'] = s['class'] + ' ' + ed.getParam('skin') + 'Skin';
173 			if (v = ed.getParam('skin_variant'))
174 				s['class'] += ' ' + ed.getParam('skin') + 'Skin' + v.substring(0, 1).toUpperCase() + v.substring(1);
175 
176 			s['class'] += ed.settings.directionality == "rtl" ? ' mceRtl' : '';
177 
178 			id = t.prefix + id;
179 			cls = cc || t._cls.dropmenu || tinymce.ui.DropMenu;
180 			c = t.controls[id] = new cls(id, s);
181 			c.onAddItem.add(function(c, o) {
182 				var s = o.settings;
183 
184 				s.title = ed.getLang(s.title, s.title);
185 
186 				if (!s.onclick) {
187 					s.onclick = function(v) {
188 						if (s.cmd)
189 							ed.execCommand(s.cmd, s.ui || false, s.value);
190 					};
191 				}
192 			});
193 
194 			ed.onRemove.add(function() {
195 				c.destroy();
196 			});
197 
198 			// Fix for bug #1897785, #1898007
199 			if (tinymce.isIE) {
200 				c.onShowMenu.add(function() {
201 					// IE 8 needs focus in order to store away a range with the current collapsed caret location
202 					ed.focus();
203 
204 					bm = ed.selection.getBookmark(1);
205 				});
206 
207 				c.onHideMenu.add(function() {
208 					if (bm) {
209 						ed.selection.moveToBookmark(bm);
210 						bm = 0;
211 					}
212 				});
213 			}
214 
215 			return t.add(c);
216 		},

218 		/**
219 		 * Creates a list box control instance by id. A list box is either a native select element or a DOM/JS based list box control. This
220 		 * depends on the use_native_selects settings state.
221 		 *
222 		 * @method createListBox
223 		 * @param {String} id Unique id for the new listbox instance. For example "styles".
224 		 * @param {Object} s Optional settings object for the control.
225 		 * @param {Object} cc Optional control class to use instead of the default one.
226 		 * @return {tinymce.ui.Control} Control instance that got created and added.
227 		 */
228 		createListBox : function(id, s, cc) {
229 			var t = this, ed = t.editor, cmd, c, cls;
230 
231 			if (t.get(id))
232 				return null;
233 
234 			s.title = ed.translate(s.title);
235 			s.scope = s.scope || ed;
236 
237 			if (!s.onselect) {
238 				s.onselect = function(v) {
239 					ed.execCommand(s.cmd, s.ui || false, v || s.value);
240 				};
241 			}
242 
243 			s = extend({
244 				title : s.title,
245 				'class' : 'mce_' + id,
246 				scope : s.scope,
247 				control_manager : t
248 			}, s);
249 
250 			id = t.prefix + id;
251 
252 
253 			function useNativeListForAccessibility(ed) {
254 				return ed.settings.use_accessible_selects && !tinymce.isGecko
255 			}
256 
257 			if (ed.settings.use_native_selects || useNativeListForAccessibility(ed))
258 				c = new tinymce.ui.NativeListBox(id, s);
259 			else {
260 				cls = cc || t._cls.listbox || tinymce.ui.ListBox;
261 				c = new cls(id, s, ed);
262 			}
263 
264 			t.controls[id] = c;
265 
266 			// Fix focus problem in Safari
267 			if (tinymce.isWebKit) {
268 				c.onPostRender.add(function(c, n) {
269 					// Store bookmark on mousedown
270 					Event.add(n, 'mousedown', function() {
271 						ed.bookmark = ed.selection.getBookmark(1);
272 					});
273 
274 					// Restore on focus, since it might be lost
275 					Event.add(n, 'focus', function() {
276 						ed.selection.moveToBookmark(ed.bookmark);
277 						ed.bookmark = null;
278 					});
279 				});
280 			}
281 
282 			if (c.hideMenu)
283 				ed.onMouseDown.add(c.hideMenu, c);
284 
285 			return t.add(c);
286 		},

288 		/**
289 		 * Creates a button control instance by id.
290 		 *
291 		 * @method createButton
292 		 * @param {String} id Unique id for the new button instance. For example "bold".
293 		 * @param {Object} s Optional settings object for the control.

295 		 * @return {tinymce.ui.Control} Control instance that got created and added.

297 		createButton : function(id, s, cc) {
298 			var t = this, ed = t.editor, o, c, cls;
299 
300 			if (t.get(id))
301 				return null;
302 
303 			s.title = ed.translate(s.title);
304 			s.label = ed.translate(s.label);
305 			s.scope = s.scope || ed;
306 
307 			if (!s.onclick && !s.menu_button) {
308 				s.onclick = function() {
309 					ed.execCommand(s.cmd, s.ui || false, s.value);
310 				};
311 			}
312 
313 			s = extend({
314 				title : s.title,
315 				'class' : 'mce_' + id,
316 				unavailable_prefix : ed.getLang('unavailable', ''),
317 				scope : s.scope,
318 				control_manager : t
319 			}, s);
320 
321 			id = t.prefix + id;
322 
323 			if (s.menu_button) {
324 				cls = cc || t._cls.menubutton || tinymce.ui.MenuButton;
325 				c = new cls(id, s, ed);
326 				ed.onMouseDown.add(c.hideMenu, c);
327 			} else {
328 				cls = t._cls.button || tinymce.ui.Button;
329 				c = new cls(id, s, ed);
330 			}
331 
332 			return t.add(c);
333 		},
