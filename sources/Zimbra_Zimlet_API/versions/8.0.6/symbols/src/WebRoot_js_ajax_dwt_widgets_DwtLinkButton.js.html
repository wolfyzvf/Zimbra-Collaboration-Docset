 24 DwtLinkButton = function(params) {
 25 	params.className = params.className || "ZButtonLink";
 26 	DwtButton.call(this, params);
 27 };
 28 
 29 
 30 DwtLinkButton.prototype = new DwtButton;
 31 DwtLinkButton.prototype.constructor = DwtLinkButton;
 32 
 33 DwtLinkButton.prototype.TEMPLATE = "dwt.Widgets#ZLinkButton";
 34

 38 DwtLinkButton.prototype._dropDownHovImg = null; //same as above
 39 
 40 DwtLinkButton.prototype.toString =
 41 function() {
 42 	return "DwtLinkButton";
 43 };
