 16 /**
 17  * Creates an empty mobile device object.
 18  * @constructor
 19  * @class
 20  * This class represents  mobile device.
 21  * 
 22  * @author Parag Shah
 23  *
 24  * @param	{Hash}		params		a hash of parameters
 25  * @param	{Date}		params.lastPolicyUpdate		the last policy update time
 26  * @param	{Date}		params.firstReqReceived		the first request received time
 27  * @param	{Date}		params.remoteWipeAckTime		the remote wipe acknowledged time
 28  * @param	{Date}		params.remoteWipeReqTime		the remote wipe requested time
 29  * @param	{constant}	params.status			the status (see <code>ZmMobileDevice.STATUS_</code>)
 30  */
 31 ZmMobileDevice = function(params) {
 32 	this.lastPolicyUpdate = params.lastPolicyUpdate;
 33 	this.firstReqReceived = params.firstReqReceived;
 34 	this.remoteWipeAckTime = params.remoteWipeAckTime;
 35 	this.remoteWipeReqTime = params.remoteWipeReqTime;
 36 	this.status = params.status;
 37 	this.provisionable = params.provisionable;
 38 	this.protocol = params.protocol;
 39 	this.ua = params.ua;
 40 	this.type = params.type;
 41 	this.id = params.id;
 42 };

 76 /**
 77  * Gets the status.
 78  * 
 79  * @return	{constant}	the status (see <code>ZmMobileDevice.STATUS_</code> constants)
 80  */
 81 ZmMobileDevice.prototype.getStatus =
 82 function() {
 83 	return (!this.provisionable && this.status == ZmMobileDevice.STATUS_NEED_PROVISION)
 84 		? ZmMobileDevice.STATUS_OK : this.status;
 85 };
 86 
 87 /**
 88  * Gets the status string.
 89  * 
 90  * @return	{String}	the status string
 91  */
 92 ZmMobileDevice.prototype.getStatusString =
 93 function() {
 94 	var status = this.getStatus();
 95 
 96 	switch (status) {
 97 		case ZmMobileDevice.STATUS_NEED_PROVISION:			return ZmMsg.mobileStatusNeedProvision;
 98 		case ZmMobileDevice.STATUS_OK:						return ZmMsg.mobileStatusOk;
 99 		case ZmMobileDevice.STATUS_SUSPENDED:				return ZmMsg.mobileStatusSuspended;
100 		case ZmMobileDevice.STATUS_REMOTE_WIPE_REQUESTED:	return ZmMsg.mobileStatusWipe;
101 		case ZmMobileDevice.STATUS_REMOTE_WIPE_COMPLETE:	return ZmMsg.mobileStatusWipeComplete;
102 	}
103 	return "";
104 };

146 ZmMobileDevice.prototype.doAction =
147 function(id, callback) {
148 	var request;
149 	switch (id) {
150 		case ZmOperation.MOBILE_REMOVE: 		request = "RemoveDeviceRequest"; break;
151 		case ZmOperation.MOBILE_RESUME_SYNC:	request = "ResumeDeviceRequest"; break;
152 		case ZmOperation.MOBILE_SUSPEND_SYNC:	request = "SuspendDeviceRequest"; break;
153 		case ZmOperation.MOBILE_WIPE:			request = "RemoteWipeRequest"; break;
154 		case ZmOperation.MOBILE_CANCEL_WIPE:	request = "CancelPendingRemoteWipeRequest"; break;
155 	}
156 
157 	if (request) {
158 		var soapDoc = AjxSoapDoc.create(request, "urn:zimbraSync");
159 		var node = soapDoc.set("device");
160 		node.setAttribute("id", this.id);
161 
162 		var respCallback = new AjxCallback(this, this._handleDoAction, callback);
163 		appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:true, callback:respCallback});
164 	}
165 };
166 
167 ZmMobileDevice.prototype._handleDoAction =
168 function(callback, results) {
169 	var resp = results.getResponse();
170 	for (var i in resp) {
171 		var device = resp[i].device && resp[i].device[0];
172 		if (device && device.id == this.id) {
173 			this.status = device.status;
174 			if (device.lastPolicyUpdate) {
175 				this.lastPolicyUpdate = device.lastPolicyUpdate;
176 			}
177 			if (device.firstReqReceived) {
178 				this.firstReqReceived = device.firstReqReceived;
179 			}
180 			if (device.remoteWipeAckTime) {
181 				this.remoteWipeAckTime = device.remoteWipeAckTime;
182 			}
183 			if (device.remoteWipeReqTime) {
184 				this.remoteWipeReqTime = device.remoteWipeReqTime;
185 			}
186 		}
187 	}
188 
189 	if (callback) {
190 		callback.run();
191 	}
192 };
