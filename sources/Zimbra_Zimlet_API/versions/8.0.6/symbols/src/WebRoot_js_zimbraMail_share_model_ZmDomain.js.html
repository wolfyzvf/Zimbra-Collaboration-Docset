 23 /**
 24  * Creates a domain
 25  * @class
 26  * This class represents a domain.
 27  * 
 28  * @param	{String}	name	the name
 29  * @param	{Object}	parent	the parent
 30  * @param	{String}	headerFlags		header flags
 31  * 
 32  * @extends	ZmModel
 33  */
 34 ZmDomain = function(name, parent, headerFlags) {
 35 	
 36 	ZmModel.call(this);
 37 
 38 	this.name = name.toLowerCase();
 39 	this.parent = parent;
 40 	this._parseHeaderFlags(headerFlags);
 41 	this._subdomains = {};
 42 }
 43 
 44 ZmDomain.prototype = new ZmModel;
 45 ZmDomain.prototype.constructor = ZmDomain;
 46 
 47 /**
 48  * Compares two domains by name.
 49  * 
 50  * @param	{ZmDomain}	a		the first domain
 51  * @param	{ZmDomain}	b		the second domain
 52  * @return	{int}	0 if the domains match; 1 if "a" is before "b"; -1 if "b" is before "a"
 53  */
 54 ZmDomain.sortCompare = 
 55 function(a, b) {
 56 	var check = ZmOrganizer.checkSortArgs(a, b);
 57 	if (check != null) return check;
 58 
 59 	if (a.name < b.name) return -1;
 60 	if (a.name > b.name) return 1;
 61 	return 0;
 62 };
 63 
 64 /**
 65  * Returns a string representation of the object.
 66  * 
 67  * @return		{String}		a string representation of the object
 68  */
 69 ZmDomain.prototype.toString = 
 70 function() {
 71 	return "ZmDomain";
 72 };
 73 
 74 /**
 75  * Gets the sub-domain.
 76  * 
 77  * @param	{String}	name		the name
 78  * @return	{ZmDomain}	the sub-domain
 79  */
 80 ZmDomain.prototype.getSubDomain =
 81 function(name) {
 82 	return this._subdomains[name];
 83 };
 84 
 85 /**
 86  * Gets the sub-domains.
 87  * 
 88  * 
 89  */
 90 ZmDomain.prototype.getSubDomains =
 91 function() {
 92 	return this._subdomains;
 93 };
 94 
 95 /**
 96  * Gets the sub-domains.
 97  * 
 98  * @return	{Array}	an array of {@link ZmDomain} objects (sorted)
 99  */
100 ZmDomain.prototype.getSortedSubDomains = 
101 function() {
102 	if (this._sorted) {
103 		return this._sorted;
104 	}
105 	this._sorted = [];
106 	for (var d in this._subdomains) {
107 		this._sorted.push(this._subdomains[d]);
108 	}
109 	this._sorted.sort(ZmDomain.sortCompare);
110 	return this._sorted;
111 };
112 
113 /**
114  * Adds the sub-domain.
115  * 
116  * @param	{String}	name		the name
117  * @param	{String}	headerFlags		the header flags
118  * @return	{ZmDomain}	the newly created sub-domain
119  */
120 ZmDomain.prototype.addSubDomain =
121 function(name, headerFlags) {
122 	name = name.toLowerCase();
123 	var sd = this._subdomains[name];
124 	if (sd) {
125 		return sd;
126 	}
127 		
128 	sd = new ZmDomain(name, this, headerFlags);
129 	this._subdomains[name] = sd;
130 
131 	if (this._sorted) {
132 		delete this._sorted;
133 	}
134 
135 	return sd;
136 };
137 
138 /**
139  * @private
140  */
141 ZmDomain.prototype._parseHeaderFlags =
142 function(flags) {
143 	this.hasFrom = (flags.indexOf("f") != -1);
144 	this.hasTo = (flags.indexOf("t") != -1);
145 	this.hasCc = (flags.indexOf("c") != -1);
146 };
