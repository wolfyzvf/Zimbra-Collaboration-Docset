 16 /**
 17 * show history of the status window
 18 * @param parent			the element that created this view
 19  * @private
 20 */
 21 ZmQuickReminderDialog = function(parent, reminderController, calController) {
 22 
 23 	// init custom buttons
 24 	var selectId = Dwt.getNextId();
 25 	// call base class
 26 	DwtDialog.call(this, {parent:parent, standardButtons: [DwtDialog.OK_BUTTON]});
 27 
 28 	this._reminderController = reminderController;
 29 	this._calController = calController;
 30 
 31 	this.setContent(this._contentHtml(selectId));
 32 	this.setTitle(ZmMsg.currentMeetings);
 33 };
 34 
 35 ZmQuickReminderDialog.prototype = new DwtDialog;
 36 ZmQuickReminderDialog.prototype.constructor = ZmQuickReminderDialog;
 37 
 38 
 39 ZmQuickReminderDialog.SOON = -AjxDateUtil.MSEC_PER_FIFTEEN_MINUTES;
 40

 49 ZmQuickReminderDialog.prototype.popup =
 50 function() {
 51 	DwtDialog.prototype.popup.call(this);

 54 ZmQuickReminderDialog.prototype.initialize =
 55 function(list) {
 56 	this._list = list.clone();
 57 	this._apptData = {};
 58 
 59 	var html = [];
 60 	var idx = 0;
 61 	var size = list.size();
 62 
 63 	html[idx++] = "<table cellpadding=0 cellspacing=0 border=0 width=100%>";
 64 	for (var i = 0; i < size; i++) {
 65 		var appt = list.get(i);
 66 		var uid = appt.getUniqueId(true);
 67 		var data = this._apptData[uid] = {appt:appt};
 68 		idx = this._addAppt(html, idx, appt, data, (i > 0));
 69 	}
 70     if(size == 0) {
 71         html[idx++] = '<tr name="rdsep">';
 72 		html[idx++] = '<td colspan=3><div>' + ZmMsg.noMeetingsFound + '</div></td>';
 73 		html[idx++] = '</tr>';
 74     }
 75 	html[idx++] = "</table>";
 76 
 77 	
 78 	if (this._openButtons) {
 79 		for (var id in this._openButtons) {
 80 			this._openButtons[id].dispose();
 81 		}
 82 	}
 83 	this._openButtons = {};
 84 
 85 	var openListener = new AjxListener(this, this._openButtonListener);
 86 	var div = document.getElementById(this._listId);
 87 	div.innerHTML = html.join("");
 88 
 89 	for (var i = 0; i < size; i++) {
 90 		var appt = list.get(i);
 91 		var uid = appt.getUniqueId(true);
 92 		var data = this._apptData[uid];
 93 
 94 		// open button
 95 		var openBtn = this._openButtons[data.openBtnId] = new DwtButton({parent:this, className:"DwtToolbarButton", parentElement:data.openBtnId});
 96 		openBtn.setImage(appt.otherAttendees ? "ApptMeeting" : "Appointment");
 97 		openBtn.setText(ZmMsg.viewAppointment);
 98 		openBtn.addSelectionListener(openListener);
 99 		openBtn.apptUid = uid;
100 
101 		this._updateDelta(data);
102 	}
103 };
104 
105 ZmQuickReminderDialog.prototype._contentHtml =
106 function(selectId) {
107 	this._listId = Dwt.getNextId();
108 	return ["<div class='ZmQuickReminderDialog' id='", this._listId, "'>"].join("");
109 };
110 
111 ZmQuickReminderDialog.prototype._updateDelta = 
112 function(data) {
113 	var td = document.getElementById(data.deltaId);
114 	if (td) {
115 		var startDelta = this._computeDelta(data.appt);
116 
117 		if (startDelta >= 0) 							td.className = 'ZmReminderOverdue';
118 		else if (startDelta > ZmQuickReminderDialog.SOON)	td.className = 'ZmReminderSoon';
119 		else											td.className = 'ZmReminderFuture';
120 
121 		td.innerHTML = this._formatDeltaString(startDelta);
122 	}
123 };

125 ZmQuickReminderDialog.prototype._addAppt =
126 function(html, idx, appt, data, needSep) {
127 
128 	data.openBtnId = Dwt.getNextId();
129 	data.deltaId = Dwt.getNextId();
130 	data.rowId = Dwt.getNextId();
131 
132 	var calName = (appt.folderId != ZmOrganizer.ID_CALENDAR && this._calController)
133 		? this._calController.getCalendarName(appt.folderId) : null;
134 
135 	var calendar = appCtxt.getById(appt.folderId);
136 
137     var apptLabel = appt.isUpcomingEvent ? " (" + ZmMsg.upcoming + ")" : ""
138 
139 	var params = {
140 		needSep: needSep,
141 		rowId: data.rowId,
142 		calName: calName,
143 		accountName: (appCtxt.multiAccounts && calendar && calendar.getAccount().getDisplayName()),
144 		location: appt.getLocation(),
145 		apptIconHtml: (AjxImg.getImageHtml(appt.otherAttendees ? "ApptMeeting" : "Appointment")),
146 		organizer: appt.otherAtt ? appt.organizer : null,
147 		reminderName: (AjxStringUtil.htmlEncode(appt.name + apptLabel)),
148 		durationText: (AjxStringUtil.trim(this._getDurationText(appt))),
149 		deltaId: data.deltaId,
150 		openBtnId: data.openBtnId
151 	};
152 	html[idx++] = AjxTemplate.expand("calendar.Calendar#ReminderDialogRow", params);
153 	return idx;
154 };

156 ZmQuickReminderDialog.prototype._openButtonListener =
157 function(ev) {
158 
159 	var obj = DwtControl.getTargetControl(ev);
160 	var data = this._apptData[obj.apptUid];
161 	var appt = data ? data.appt : null;
162 	if (appt) {
163 		AjxDispatcher.require(["CalendarCore", "Calendar"]);
164 
165 		var cc = AjxDispatcher.run("GetCalController");
166 
167 		// the give appt object is a ZmCalBaseItem. We need a ZmAppt
168 		var newAppt = new ZmAppt();
169 		for (var i in appt) {
170 			if (!AjxUtil.isFunction(appt[i])) {
171 				newAppt[i] = appt[i];
172 			}
173 		}
174 		var callback = new AjxCallback(cc, cc._showAppointmentDetails, newAppt);
175 		newAppt.getDetails(null, callback, null, null, true);
176         this.popdown();
177 	}
178 };

180 ZmQuickReminderDialog.prototype._getDurationText =
181 function(appt) {
182 	var isMultiDay = appt.isMultiDay();
183 	var start = appt.startDate;
184 	var endTime = appt.getEndTime();
185 	var end = new Date(endTime);
186 
187 	if (appt.isAllDayEvent()) {
188 		end = new Date(endTime - (isMultiDay ? 2 * AjxDateUtil.MSEC_PER_HOUR : 0));
189 		var pattern = isMultiDay ? ZmMsg.apptTimeAllDayMulti : ZmMsg.apptTimeAllDay;
190 		return AjxMessageFormat.format(pattern, [start, end]);
191 	}
192 	var pattern = isMultiDay ? ZmMsg.apptTimeInstanceMulti : ZmMsg.apptTimeInstance;
193 	return AjxMessageFormat.format(pattern, [start, end, ""]);
194 };

196 ZmQuickReminderDialog.prototype._computeDelta =
197 function(appt) {
198 	return ((new Date()).getTime() - appt.getStartTime());
199 };

201 ZmQuickReminderDialog.prototype._formatDeltaString =
202 function(deltaMSec) {
203 	var prefix = deltaMSec < 0 ? "In" : "OverdueBy";
204 	deltaMSec = Math.abs(deltaMSec);
205 
206 	// calculate parts
207 	var years =  Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 365));
208 	if (years != 0)
209 		deltaMSec -= years * AjxDateUtil.MSEC_PER_DAY * 365;
210 	var months = Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 30.42));
211 	if (months > 0)
212 		deltaMSec -= Math.floor(months * AjxDateUtil.MSEC_PER_DAY * 30.42);
213 	var days = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_DAY);
214 	if (days > 0)
215 		deltaMSec -= days * AjxDateUtil.MSEC_PER_DAY;
216 	var hours = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_HOUR);
217 	if (hours > 0)
218 		deltaMSec -= hours * AjxDateUtil.MSEC_PER_HOUR;
219 	var mins = Math.floor(deltaMSec / 60000);
220 	if (mins > 0)
221 		deltaMSec -= mins * 60000;
222 	var secs = Math.floor(deltaMSec / 1000);
223 	if (secs > 30 && mins < 59) mins++;
224 
225 	var secs = 0;
226 
227 	// determine message
228 	var amount;
229 	if (years > 0) {
230 		amount = "Years";
231 		if (years <= 3 && months > 0) {
232 			amount = "YearsMonths";
233 		}
234 	} else if (months > 0) {
235 		amount = "Months";
236 		if (months <= 3 && days > 0) {
237 			amount = "MonthsDays";
238 		}
239 	} else if (days > 0) {
240 		amount = "Days";
241 		if (days <= 2 && hours > 0) {
242 			amount = "DaysHours";
243 		}
244 	} else if (hours > 0) {
245 		amount = "Hours";
246 		if (hours < 5 && mins > 0) {
247 			amount = "HoursMinutes";
248 		}
249 	} else {
250 		amount = "Minutes";
251 	}
252 
253 	// format message
254 	var key = ["reminder",prefix,amount].join("");
255 	var args = [deltaMSec, years, months, days, hours, mins, secs];
256 	return AjxMessageFormat.format(ZmMsg[key], args);
257 };
