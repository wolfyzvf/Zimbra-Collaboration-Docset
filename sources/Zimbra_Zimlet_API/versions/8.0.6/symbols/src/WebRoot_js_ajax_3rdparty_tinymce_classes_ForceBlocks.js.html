 11 tinymce.ForceBlocks = function(editor) {
 12 	var settings = editor.settings, dom = editor.dom, selection = editor.selection, blockElements = editor.schema.getBlockElements();
 13 
 14 	function addRootBlocks() {
 15 		var node = selection.getStart(), rootNode = editor.getBody(), rng, startContainer, startOffset, endContainer, endOffset, rootBlockNode, tempNode, offset = -0xFFFFFF, wrapped, isInEditorDocument;
 16 
 17 		if (!node || node.nodeType !== 1 || !settings.forced_root_block)
 18 			return;
 19 
 20 		// Check if node is wrapped in block
 21 		while (node && node != rootNode) {
 22 			if (blockElements[node.nodeName])
 23 				return;
 24 
 25 			node = node.parentNode;
 26 		}
 27 
 28 		// Get current selection
 29 		rng = selection.getRng();
 30 		if (rng.setStart) {
 31 			startContainer = rng.startContainer;
 32 			startOffset = rng.startOffset;
 33 			endContainer = rng.endContainer;
 34 			endOffset = rng.endOffset;
 35 		} else {
 36 			// Force control range into text range
 37 			if (rng.item) {
 38 				node = rng.item(0);
 39 				rng = editor.getDoc().body.createTextRange();
 40 				rng.moveToElementText(node);
 41 			}
 42 
 43 			isInEditorDocument = rng.parentElement().ownerDocument === editor.getDoc();
 44 			tmpRng = rng.duplicate();
 45 			tmpRng.collapse(true);
 46 			startOffset = tmpRng.move('character', offset) * -1;
 47 
 48 			if (!tmpRng.collapsed) {
 49 				tmpRng = rng.duplicate();
 50 				tmpRng.collapse(false);
 51 				endOffset = (tmpRng.move('character', offset) * -1) - startOffset;
 52 			}
 53 		}
 54 
 55 		// Wrap non block elements and text nodes
 56 		node = rootNode.firstChild;
 57 		while (node) {
 58 			if (node.nodeType === 3 || (node.nodeType == 1 && !blockElements[node.nodeName])) {
 59 				if (!rootBlockNode) {
 60 					rootBlockNode = dom.create(settings.forced_root_block);
 61 					node.parentNode.insertBefore(rootBlockNode, node);
 62 					wrapped = true;
 63 				}
 64 
 65 				tempNode = node;
 66 				node = node.nextSibling;
 67 				rootBlockNode.appendChild(tempNode);
 68 			} else {
 69 				rootBlockNode = null;
 70 				node = node.nextSibling;
 71 			}
 72 		}
 73 
 74 		if (wrapped) {
 75 			if (rng.setStart) {
 76 				rng.setStart(startContainer, startOffset);
 77 				rng.setEnd(endContainer, endOffset);
 78 				selection.setRng(rng);
 79 			} else {
 80 				// Only select if the previous selection was inside the document to prevent auto focus in quirks mode
 81 				if (isInEditorDocument) {
 82 					try {
 83 						rng = editor.getDoc().body.createTextRange();
 84 						rng.moveToElementText(rootNode);
 85 						rng.collapse(true);
 86 						rng.moveStart('character', startOffset);
 87 
 88 						if (endOffset > 0)
 89 							rng.moveEnd('character', endOffset);
 90 
 91 						rng.select();
 92 					} catch (ex) {
 93 						// Ignore
 94 					}
 95 				}
 96 			}
 97 
 98 			editor.nodeChanged();
 99 		}
100 	};
101 
102 	// Force root blocks
103 	if (settings.forced_root_block) {
104 		editor.onKeyUp.add(addRootBlocks);
105 		editor.onNodeChange.add(addRootBlocks);
106 	}
107 };
