201 		/**
202 		 * Renderes the editor/adds it to the page.
203 		 *
204 		 * @method render
205 		 */
206 		render : function(nst) {
207 			var t = this, s = t.settings, id = t.id, sl = tinymce.ScriptLoader;
208 
209 			// Page is not loaded yet, wait for it
210 			if (!Event.domLoaded) {
211 				Event.add(window, 'ready', function() {
212 					t.render();
213 				});
214 				return;
215 			}
216 
217 			tinyMCE.settings = s;
218 
219 			// Element not found, then skip initialization
220 			if (!t.getElement())
221 				return;
222 
223 			// Is a iPad/iPhone and not on iOS5, then skip initialization. We need to sniff 
224 			// here since the browser says it has contentEditable support but there is no visible caret.
225 			if (tinymce.isIDevice && !tinymce.isIOS5)
226 				return;
227 
228 			// Add hidden input for non input elements inside form elements
229 			if (!/TEXTAREA|INPUT/i.test(t.getElement().nodeName) && s.hidden_input && DOM.getParent(id, 'form'))
230 				DOM.insertAfter(DOM.create('input', {type : 'hidden', name : id}), id);
231 
232 			// Hide target element early to prevent content flashing
233 			if (!s.content_editable) {
234 				t.orgVisibility = t.getElement().style.visibility;
235 				t.getElement().style.visibility = 'hidden';
236 			}
237 
238 			/**
239 			 * Window manager reference, use this to open new windows and dialogs.
240 			 *
241 			 * @property windowManager
242 			 * @type tinymce.WindowManager
243 			 * @example
244 			 * // Shows an alert message
245 			 * tinyMCE.activeEditor.windowManager.alert('Hello world!');
246 			 * 
247 			 * // Opens a new dialog with the file.htm file and the size 320x240
248 			 * // It also adds a custom parameter this can be retrieved by using tinyMCEPopup.getWindowArg inside the dialog.
249 			 * tinyMCE.activeEditor.windowManager.open({
250 			 *    url : 'file.htm',
251 			 *    width : 320,
252 			 *    height : 240
253 			 * }, {
254 			 *    custom_param : 1
255 			 * });
256 			 */
257 			if (tinymce.WindowManager)
258 				t.windowManager = new tinymce.WindowManager(t);
259 
260 			if (s.encoding == 'xml') {
261 				t.onGetContent.add(function(ed, o) {
262 					if (o.save)
263 						o.content = DOM.encode(o.content);
264 				});
265 			}
266 
267 			if (s.add_form_submit_trigger) {
268 				t.onSubmit.addToTop(function() {
269 					if (t.initialized) {
270 						t.save();
271 						t.isNotDirty = 1;
272 					}
273 				});
274 			}
275 
276 			if (s.add_unload_trigger) {
277 				t._beforeUnload = tinyMCE.onBeforeUnload.add(function() {
278 					if (t.initialized && !t.destroyed && !t.isHidden())
279 						t.save({format : 'raw', no_events : true});
280 				});
281 			}
282 
283 			tinymce.addUnload(t.destroy, t);
284 
285 			if (s.submit_patch) {
286 				t.onBeforeRenderUI.add(function() {
287 					var n = t.getElement().form;
288 
289 					if (!n)
290 						return;
291 
292 					// Already patched
293 					if (n._mceOldSubmit)
294 						return;
295 
296 					// Check page uses id="submit" or name="submit" for it's submit button
297 					if (!n.submit.nodeType && !n.submit.length) {
298 						t.formElement = n;
299 						n._mceOldSubmit = n.submit;
300 						n.submit = function() {
301 							// Save all instances
302 							tinymce.triggerSave();
303 							t.isNotDirty = 1;
304 
305 							return t.formElement._mceOldSubmit(t.formElement);
306 						};
307 					}
308 
309 					n = null;
310 				});
311 			}
312 
313 			// Load scripts
314 			function loadScripts() {
315 				if (s.language && s.language_load !== false)
316 					sl.add(tinymce.baseURL + '/langs/' + s.language + '.js');
317 
318 				if (s.theme && typeof s.theme != "function" && s.theme.charAt(0) != '-' && !ThemeManager.urls[s.theme])
319 					ThemeManager.load(s.theme, 'themes/' + s.theme + '/editor_template' + tinymce.suffix + '.js');
320 
321 				each(explode(s.plugins), function(p) {
322 					if (p &&!PluginManager.urls[p]) {
323 						if (p.charAt(0) == '-') {
324 							p = p.substr(1, p.length);
325 							var dependencies = PluginManager.dependencies(p);
326 							each(dependencies, function(dep) {
327 								var defaultSettings = {prefix:'plugins/', resource: dep, suffix:'/editor_plugin' + tinymce.suffix + '.js'};
328 								dep = PluginManager.createUrl(defaultSettings, dep);
329 								PluginManager.load(dep.resource, dep);
330 							});
331 						} else {
332 							// Skip safari plugin, since it is removed as of 3.3b1
333 							if (p == 'safari') {
334 								return;
335 							}
336 							PluginManager.load(p, {prefix:'plugins/', resource: p, suffix:'/editor_plugin' + tinymce.suffix + '.js'});
337 						}
338 					}
339 				});
340 
341 				// Init when que is loaded
342 				sl.loadQueue(function() {
343 					if (!t.removed)
344 						t.init();
345 				});
346 			};
347 
348 			loadScripts();
349 		},

351 		/**
352 		 * Initializes the editor this will be called automatically when
353 		 * all plugins/themes and language packs are loaded by the rendered method.
354 		 * This method will setup the iframe and create the theme and plugin instances.
355 		 *
356 		 * @method init
357 		 */
358 		init : function() {
359 			var n, t = this, s = t.settings, w, h, mh, e = t.getElement(), o, ti, u, bi, bc, re, i, initializedPlugins = [];
360 
361 			tinymce.add(t);
362 
363 			s.aria_label = s.aria_label || DOM.getAttrib(e, 'aria-label', t.getLang('aria.rich_text_area'));
364 
365 			/**
366 			 * Reference to the theme instance that was used to generate the UI. 
367 			 *
368 			 * @property theme
369 			 * @type tinymce.Theme
370 			 * @example
371 			 * // Executes a method on the theme directly
372 			 * tinyMCE.activeEditor.theme.someMethod();
373 			 */
374 			if (s.theme) {
375 				if (typeof s.theme != "function") {
376 					s.theme = s.theme.replace(/-/, '');
377 					o = ThemeManager.get(s.theme);
378 					t.theme = new o();
379 
380 					if (t.theme.init)
381 						t.theme.init(t, ThemeManager.urls[s.theme] || tinymce.documentBaseURL.replace(/\/$/, ''));
382 				} else {
383 					t.theme = s.theme;
384 				}
385 			}
386 
387 			function initPlugin(p) {
388 				var c = PluginManager.get(p), u = PluginManager.urls[p] || tinymce.documentBaseURL.replace(/\/$/, ''), po;
389 				if (c && tinymce.inArray(initializedPlugins,p) === -1) {
390 					each(PluginManager.dependencies(p), function(dep){
391 						initPlugin(dep);
392 					});
393 					po = new c(t, u);
394 
395 					t.plugins[p] = po;
396 
397 					if (po.init) {
398 						po.init(t, u);
399 						initializedPlugins.push(p);
400 					}
401 				}
402 			}
403 			
404 			// Create all plugins
405 			each(explode(s.plugins.replace(/\-/g, '')), initPlugin);
406 
407 			// Setup popup CSS path(s)
408 			if (s.popup_css !== false) {
409 				if (s.popup_css)
410 					s.popup_css = t.documentBaseURI.toAbsolute(s.popup_css);
411 				else
412 					s.popup_css = t.baseURI.toAbsolute("themes/" + s.theme + "/skins/" + s.skin + "/dialog.css");
413 			}
414 
415 			if (s.popup_css_add)
416 				s.popup_css += ',' + t.documentBaseURI.toAbsolute(s.popup_css_add);
417 
418 			/**
419 			 * Control manager instance for the editor. Will enables you to create new UI elements and change their states etc.
420 			 *
421 			 * @property controlManager
422 			 * @type tinymce.ControlManager
423 			 * @example
424 			 * // Disables the bold button
425 			 * tinyMCE.activeEditor.controlManager.setDisabled('bold', true);
426 			 */
427 			t.controlManager = new tinymce.ControlManager(t);
428 
429 			// Enables users to override the control factory
430 			t.onBeforeRenderUI.dispatch(t, t.controlManager);
431 
432 			// Measure box
433 			if (s.render_ui && t.theme) {
434 				t.orgDisplay = e.style.display;
435 
436 				if (typeof s.theme != "function") {
437 					w = s.width || e.style.width || e.offsetWidth;
438 					h = s.height || e.style.height || e.offsetHeight;
439 					mh = s.min_height || 100;
440 					re = /^[0-9\.]+(|px)$/i;
441 
442 					if (re.test('' + w))
443 						w = Math.max(parseInt(w, 10) + (o.deltaWidth || 0), 100);
444 
445 					if (re.test('' + h))
446 						h = Math.max(parseInt(h, 10) + (o.deltaHeight || 0), mh);
447 
448 					// Render UI
449 					o = t.theme.renderUI({
450 						targetNode : e,
451 						width : w,
452 						height : h,
453 						deltaWidth : s.delta_width,
454 						deltaHeight : s.delta_height
455 					});
456 
457 					// Resize editor
458 					DOM.setStyles(o.sizeContainer || o.editorContainer, {
459 						width : w,
460 						height : h
461 					});
462 
463 					h = (o.iframeHeight || h) + (typeof(h) == 'number' ? (o.deltaHeight || 0) : '');
464 					if (h < mh)
465 						h = mh;
466 				} else {
467 					o = s.theme(t, e);
468 
469 					// Convert element type to id:s
470 					if (o.editorContainer.nodeType) {
471 						o.editorContainer = o.editorContainer.id = o.editorContainer.id || t.id + "_parent";
472 					}
473 
474 					// Convert element type to id:s
475 					if (o.iframeContainer.nodeType) {
476 						o.iframeContainer = o.iframeContainer.id = o.iframeContainer.id || t.id + "_iframecontainer";
477 					}
478 
479 					// Use specified iframe height or the targets offsetHeight
480 					h = o.iframeHeight || e.offsetHeight;
481 
482 					// Store away the selection when it's changed to it can be restored later with a editor.focus() call
483 					if (isIE) {
484 						t.onInit.add(function(ed) {
485 							ed.dom.bind(ed.getBody(), 'beforedeactivate keydown', function() {
486 								ed.lastIERng = ed.selection.getRng();
487 							});
488 						});
489 					}
490 				}
491 
492 				t.editorContainer = o.editorContainer;
493 			}
494 
495 			// Load specified content CSS last
496 			if (s.content_css) {
497 				each(explode(s.content_css), function(u) {
498 					t.contentCSS.push(t.documentBaseURI.toAbsolute(u));
499 				});
500 			}
501 
502 			// Content editable mode ends here
503 			if (s.content_editable) {
504 				e = n = o = null; // Fix IE leak
505 				return t.initContentBody();
506 			}
507 
508 			// User specified a document.domain value
509 			if (document.domain && location.hostname != document.domain)
510 				tinymce.relaxedDomain = document.domain;
511 
512 			t.iframeHTML = s.doctype + '<html><head xmlns="http://www.w3.org/1999/xhtml">';
513 
514 			// We only need to override paths if we have to
515 			// IE has a bug where it remove site absolute urls to relative ones if this is specified
516 			if (s.document_base_url != tinymce.documentBaseURL)
517 				t.iframeHTML += '<base href="' + t.documentBaseURI.getURI() + '" />';
518 
519 			// IE8 doesn't support carets behind images setting ie7_compat would force IE8+ to run in IE7 compat mode.
520 			if (s.ie7_compat)
521 				t.iframeHTML += '<meta http-equiv="X-UA-Compatible" content="IE=7" />';
522 			else
523 				t.iframeHTML += '<meta http-equiv="X-UA-Compatible" content="IE=edge" />';
524 
525 			t.iframeHTML += '<meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />';
526 
527 			// Load the CSS by injecting them into the HTML this will reduce "flicker"
528 			for (i = 0; i < t.contentCSS.length; i++) {
529 				t.iframeHTML += '<link type="text/css" rel="stylesheet" href="' + t.contentCSS[i] + '" />';
530 			}
531 
532 			t.contentCSS = [];
533 
534 			bi = s.body_id || 'tinymce';
535 			if (bi.indexOf('=') != -1) {
536 				bi = t.getParam('body_id', '', 'hash');
537 				bi = bi[t.id] || bi;
538 			}
539 
540 			bc = s.body_class || '';
541 			if (bc.indexOf('=') != -1) {
542 				bc = t.getParam('body_class', '', 'hash');
543 				bc = bc[t.id] || '';
544 			}
545 
546 			t.iframeHTML += '</head><body id="' + bi + '" class="mceContentBody ' + bc + '" onload="window.parent.tinyMCE.get(\'' + t.id + '\').onLoad.dispatch();"><br></body></html>';
547 
548 			// Domain relaxing enabled, then set document domain
549 			if (tinymce.relaxedDomain && (isIE || (tinymce.isOpera && parseFloat(opera.version()) < 11))) {
550 				// We need to write the contents here in IE since multiple writes messes up refresh button and back button
551 				u = 'javascript:(function(){document.open();document.domain="' + document.domain + '";var ed = window.parent.tinyMCE.get("' + t.id + '");document.write(ed.iframeHTML);document.close();ed.initContentBody();})()';
552 			}
553 
554 			// Create iframe
555 			// TODO: ACC add the appropriate description on this.
556 			n = DOM.add(o.iframeContainer, 'iframe', { 
557 				id : t.id + "_ifr",
558 				src : u || 'javascript:""', // Workaround for HTTPS warning in IE6/7
559 				frameBorder : '0',
560 				allowTransparency : "true",
561 				title : s.aria_label,
562 				style : {
563 					width : '100%',
564 					height : h,
565 					display : 'block' // Important for Gecko to render the iframe correctly
566 				}
567 			});
568 
569 			t.contentAreaContainer = o.iframeContainer;
570 
571 			if (o.editorContainer) {
572 				DOM.get(o.editorContainer).style.display = t.orgDisplay;
573 			}
574 
575 			// Restore visibility on target element
576 			e.style.visibility = t.orgVisibility;
577 
578 			DOM.get(t.id).style.display = 'none';
579 			DOM.setAttrib(t.id, 'aria-hidden', true);
580 
581 			if (!tinymce.relaxedDomain || !u)
582 				t.initContentBody();
583 
584 			e = n = o = null; // Cleanup
585 		},

587 		/**
588 		 * This method get called by the init method ones the iframe is loaded.
589 		 * It will fill the iframe with contents, setups DOM and selection objects for the iframe.
590 		 * This method should not be called directly.
591 		 *
592 		 * @method initContentBody
593 		 */
594 		initContentBody : function() {
595 			var self = this, settings = self.settings, targetElm = DOM.get(self.id), doc = self.getDoc(), html, body, contentCssText;
