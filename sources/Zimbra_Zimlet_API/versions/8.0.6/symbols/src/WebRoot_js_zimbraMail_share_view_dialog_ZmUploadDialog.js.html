 20 /**
 21  * Creates an upload dialog.
 22  * @class
 23  * This class represents an upload dialog.
 24  * 
 25  * @param	{DwtComposite}	shell		the parent
 26  * @param	{String}	className		the class name
 27  *  
 28  * @extends		DwtDialog
 29  */
 30 ZmUploadDialog = function(shell, className) {
 31 	className = className || "ZmUploadDialog";
 32 	var title = ZmMsg.uploadDocs;
 33 	DwtDialog.call(this, {parent:shell, className:className, title:title});
 34 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._upload));
 35 	this._createUploadHtml();
 36     this._showLinkTitleText = false;
 37     this._linkText = {};
 38 }

 99 ZmUploadDialog.prototype.popup =
100 function(folder, callback, title, loc, oneFileOnly, noResolveAction, showNotes, isImage) {

102 	this._uploadCallback = callback;
103 	var aCtxt = ZmAppCtxt.handleWindowOpener();
104 
105     this._supportsHTML5 = AjxEnv.supportsHTML5File && !this._showLinkTitleText && (aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT) != null);
106

126             labelEl.innerHTML = ZmMsg.uploadChooseImage;

128         }
129         else{

131             Dwt.setVisible(labelEl, !oneFileOnly);
132         }

134     var actionRowEl = document.getElementById(id+"_actionRow");

136 		Dwt.setVisible(actionRowEl, !noResolveAction);

138 
139     var notesEl = document.getElementById(id+"_notesTD");
140 	if (notesEl) {
141 		Dwt.setVisible(notesEl, showNotes);

148             if(oneFileOnly){

184 ZmUploadDialog.prototype.uploadFiles =
185 function(files,uploadForm,folder) {
186 
187     if (files.length == 0) {
188 		return;
189 	}
190     this._uploadFolder = folder;
191     var callback = new AjxCallback(this, this._uploadSaveDocs, [files]);
192 
193     var uploadMgr = appCtxt.getUploadManager();
194 	  window._uploadManager = uploadMgr;
195 
196     try {
197 		uploadMgr.execute(callback, uploadForm);
198 	} catch (ex) {
199 		if (ex.msg) {
200 			this._popupErrorDialog(ex.msg);
201 		} else {
202 			this._popupErrorDialog(ZmMsg.unknownError);
203 		}
204 	}

207 // Protected methods
208 ZmUploadDialog.prototype._upload = function(){
209 	var form = this._uploadForm;
210 	var files = [];
211     this._linkText = {};
212 	var elements = form.elements;
213 	for (var i = 0; i < elements.length; i++) {
214 		var element = form.elements[i];
215 		if (element.name != ZmUploadDialog.UPLOAD_FIELD_NAME) continue;
216 		if (!element.value) continue;
217 		if (!this._checkExtension(element.value)) {
218 			var params = [ this._extensions.join(", ") ];
219 			var message = AjxMessageFormat.format(ZmMsg.errorNotAllowedFile, params);
220 			this._popupErrorDialog(message);
221 			return;
222 		}
223         this._msgInfo.innerHTML = "";
224         var notes = this.getNotes();
225         if(this._supportsHTML5){
226             if(this._validateSize()){
227                 var f = element.files; 
228                 for(var j=0; j<f.length; j++){
229                     files.push({name:f[j].name, fullname: f[j].name, notes: notes});
230                 }
231             }else{
232 	            var aCtxt = ZmAppCtxt.handleWindowOpener();
233                 this._msgInfo.innerHTML = AjxMessageFormat.format(ZmMsg.attachmentSizeError, AjxUtil.formatSize(aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT)));;
234                 return;
235             }
236         }else{
237             var file = {
238                 fullname: element.value,
239                 name: element.value.replace(/^.*[\\\/:]/, ""),
240                 notes: notes
241             };
242             files.push(file);
243         }
244         if(this._showLinkTitleText) {
245             var id = element.id;
246             id = id.replace("_input", "") + "_titleinput";
247             var txtElement = document.getElementById(id);
248             if(txtElement) {
249                 this._linkText[file.name] = txtElement.value;
250             }
251         }
252 
253 	}
254 	if (files.length == 0) {
255 		return;
256 	}
257     
258 	this.setButtonEnabled(DwtDialog.OK_BUTTON, false);
259 	this.setButtonEnabled(DwtDialog.CANCEL_BUTTON, false);
260 
261 	var callback = new AjxCallback(this, this._uploadSaveDocs, [files]);	
262 	var uploadMgr = appCtxt.getUploadManager();
263 	window._uploadManager = uploadMgr;
264 	try {
265 		uploadMgr.execute(callback, this._uploadForm);
266 	} catch (ex) {
267 		if (ex.msg) {
268 			this._popupErrorDialog(ex.msg);
269 		} else {
270 			this._popupErrorDialog(ZmMsg.unknownError);
271 		}
272 	}
273 };

275 ZmUploadDialog.prototype._checkExtension = function(filename) {
276 	if (!this._extensions) return true;
277 	var ext = filename.replace(/^.*\./,"").toUpperCase();
278 	for (var i = 0; i < this._extensions.length; i++) {
279 		if (this._extensions[i] == ext) {
280 			return true;
281 		}
282 	}
283 	return false;
284 };

286 ZmUploadDialog.prototype._validateSize =
287 function(){
288 
289     var atts = document.getElementsByName(ZmUploadDialog.UPLOAD_FIELD_NAME);
290     var file, size;
291 	for (var i = 0; i < atts.length; i++){
292         file = atts[i].files;
293         if(!file || file.length == 0) continue;
294         for(var j=0; j<file.length;j++){
295             var f = file[j];
296             size = f.size || f.fileSize /*Safari*/;
297 	        var aCtxt = ZmAppCtxt.handleWindowOpener();
298             if(size > aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT)){
299                 return false;
300             }
301         }
302     }
303 	return true;        
304 };

315 ZmUploadDialog.prototype._uploadSaveDocs = function(files, status, guids) {
316 	if (status != AjxPost.SC_OK) {
317 		var message = AjxMessageFormat.format(ZmMsg.uploadError, status);
318 		if(status == '413') {
319 			message = ZmMsg.errorAttachmentTooBig;
320 		}
321 		this._popupErrorDialog(message);
322 	} else {
323 		guids = guids.split(",");
324 		for (var i = 0; i < files.length; i++) {
325 			DBG.println("guids["+i+"]: "+guids[i]+", files["+i+"]: "+files[i]);
326 			files[i].guid = guids[i];
327 		}
328 		if (this._uploadFolder) {
329 			this._uploadSaveDocs2(files, status, guids);
330 		}
331 		else {
332 			this._finishUpload(files, status, guids);
333 		}
334 	}
335 };

337 ZmUploadDialog.prototype._uploadSaveDocs2 =
338 function(files, status, guids) {
339 	// create document wrappers
340     var request = [];
341     var foundOne = false;
342 	for (var i = 0; i < files.length; i++) {
343          var file = files[i];
344 		if (file.done) { continue; }
345 		foundOne = true;
346 
347         var  SaveDocumentRequest = {
348             _jsns: "urn:zimbraMail",
349             requestId: i,
350             doc: {}
351         }
352 
353         var doc = SaveDocumentRequest.doc;
354         if(file.id){
355             doc.id = file.id;
356             doc.ver = file.version;
357         }else{
358             doc.l = this._uploadFolder.id;
359         }
360         if(file.notes){
361             doc.desc = file.notes;
362         }
363         doc.upload = {
364             id: file.guid
365         }
366         request.push(SaveDocumentRequest);
367     }
368 
369     if (foundOne) {
370         var json = {
371             BatchRequest: {
372                 _jsns: "urn:zimbra",
373                 onerror: "continue",
374                 SaveDocumentRequest: ( (request.length == 1) ? request[0] : request )
375             }
376         };
377 		var callback = new AjxCallback(this, this._uploadSaveDocsResponse, [ files, status, guids ]);
378 		var params = {
379 			jsonObj: json,
380 			asyncMode:true,
381 			callback:callback
382 		};
383 		var appController = appCtxt.getAppController();
384 		appController.sendRequest(params);
385 	}
386 	else {
387 		this._finishUpload(files); //this calls the callback of the client - e.g. ZmSignatureEditor.prototype._imageUploaded since _uploadSaveDocsResponse is not called in this case, we still need the client callback since the user chose the "old" version of the image
388 	}
389 };

391 ZmUploadDialog.prototype._uploadSaveDocsResponse =
392 function(files, status, guids, response) {
393 	var resp = response && response._data && response._data.BatchResponse;
394 
395 	// mark successful uploads
396 	if (resp && resp.SaveDocumentResponse) {
397 		for (var i = 0; i < resp.SaveDocumentResponse.length; i++) {
398 			var saveDocResp = resp.SaveDocumentResponse[i];
399 			files[saveDocResp.requestId].done = true;
400 			files[saveDocResp.requestId].name = saveDocResp.doc[0].name;
401             files[saveDocResp.requestId].id   = saveDocResp.doc[0].id;
402             files[saveDocResp.requestId].ver   = saveDocResp.doc[0].ver;
403             files[saveDocResp.requestId].version   = saveDocResp.doc[0].ver;
404 		}
405 	}
406 
407 	// check for conflicts
408 	var conflicts = [];
409 	if (resp && resp.Fault) {
410 		var errors = [], mailboxQuotaExceeded=false, isItemLocked=false;
411 		for (var i = 0; i < resp.Fault.length; i++) {
412 			var fault = resp.Fault[i];
413 			var error = fault.Detail.Error;
414 			var code = error.Code;
415 			var attrs = error.a;
416             isItemLocked = (code==ZmCsfeException.LOCKED);
417 			if (code == ZmCsfeException.MAIL_ALREADY_EXISTS ||
418 				code == ZmCsfeException.MODIFY_CONFLICT) {
419 				var file = files[fault.requestId];
420 				for (var p in attrs) {
421 					var attr = attrs[p];
422 					switch (attr.n) {
423                         case "itemId" : { file.id = attr._content; break }
424 						case "id": { file.id = attr._content; break; }
425 						case "ver": { file.version = attr._content; break; }
426 						case "name": { file.name = attr._content; break; }
427 					}
428 				}
429                 file.version = file.version || 1;
430 				conflicts.push(file);
431 			}else {
432 				DBG.println("Unknown error occurred: "+code);
433                 if(code == ZmCsfeException.MAIL_QUOTA_EXCEEDED){
434                     mailboxQuotaExceeded = true;
435                 }
436 				errors[fault.requestId] = fault;
437 			}
438 		}
439 		// TODO: What to do about other errors?
440         if(mailboxQuotaExceeded){
441             this._popupErrorDialog(ZmMsg.errorQuotaExceeded);
442             return;
443         }
444         else if(isItemLocked){
445             this._popupErrorDialog(ZmMsg.errorItemLocked);
446             return;
447         }
448         else if(code==ZmCsfeException.SVC_PERM_DENIED){
449             this._popupErrorDialog(ZmMsg.errorPermissionDenied);
450             this.popdown();
451             return;
452         }
453 	}
454 
455 	// dismiss dialog
456 	this.popdown();
457 
458 	// resolve conflicts
459 	var conflictCount = conflicts.length;
460 	var action = this._conflictAction || this._selector.getValue();
461 	if (conflictCount > 0 && action == ZmUploadDialog.ACTION_ASK) {
462         var dialog = appCtxt.getUploadConflictDialog();
463 		dialog.popup(this._uploadFolder, conflicts,
464                              new AjxCallback(this, this._uploadSaveDocs2, [ files, status, guids ]));
465     }
466 
467 	// keep mine
468 	else if (conflictCount > 0 && action == ZmUploadDialog.ACTION_KEEP_MINE) {
469         if(this._conflictAction){
470             this._shieldSaveDocs(files, status, guids);
471         }else{
472 		    this._uploadSaveDocs2(files, status, guids);
473         }
474 	}
475 	// perform callback
476 	else if (this._uploadCallback) {
477 		this._finishUpload(files, status, guids);
478 	}else{
479         this._conflictAction = null;
480     }
481 };

503 ZmUploadDialog.prototype._finishUpload = function(files, status, guids) {
504 	var filenames = [];
505 	for (var i in files) {
506         var name = files[i].name;
507         if(this._linkText[name]) {
508             files[i].linkText = this._linkText[name]; 
509         }
510 		filenames.push(name);
511 	}
512 	this._uploadCallback.run(this._uploadFolder, filenames, files);
513     this._conflictAction = null;
514 };

593 ZmUploadDialog.prototype._handleFileSize =
594 function(inputEl, sizeEl){
595 
596     var files = inputEl.files;
597     if(!files) return;
598 
599     var sizeStr = [], className, totalSize =0;
600     for(var i=0; i<files.length;i++){
601         var file = files[i];
602         var size = file.size || file.fileSize /*Safari*/;
603 	    var aCtxt = ZmAppCtxt.handleWindowOpener();
604         if(size > aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT))
605             className = "RedC";
606         totalSize += size;
607     }
608 
609     if(sizeEl) {
610         sizeEl.innerHTML = "  ("+AjxUtil.formatSize(totalSize, true)+")";
611         if(className)
612             Dwt.addClass(sizeEl, "RedC");
613         else
614             Dwt.delClass(sizeEl, "RedC");
615     }
616     
617 };

619 ZmUploadDialog._removeHandler = function(event) {
620 	var span = DwtUiEvent.getTarget(event || window.event);
621 	var cell = span.parentNode;
622 	var row = cell.parentNode;
623 
624     var endRow = row;
625 
626     if(span.id) {
627        var id = span.id;
628        id = id.replace("_remove", "") + "_titleinput";
629        var txtInput = document.getElementById(id);
630        if(txtInput) {
631            var txtCell = txtInput.parentNode;
632            var txtRow = txtCell.parentNode;
633            endRow = txtRow;
634        }
635     }
636     
637 	if (row.previousSibling == null && endRow.nextSibling == null) {
638 		var comp = DwtControl.findControl(span);
639 		comp._addFileInputRow();
640 	}
641 
642     if(endRow != row) {
643         endRow.parentNode.removeChild(endRow);
644     }
645 
646 	row.parentNode.removeChild(row);
647 };

655 ZmUploadDialog.prototype._createUploadHtml = function() {
656 	var id = this._htmlElId;
657 	var aCtxt = ZmAppCtxt.handleWindowOpener();
658 	if (!aCtxt) {
659 		//hack if ZmAppCtxt is not defined; not sure why that would be the case.
660 		aCtxt = appCtxt;
661 		if (window.opener) {
662 			try {
663 				aCtxt = window.opener.appCtxt;
664 			}
665 			catch (ex) {
666 				aCtxt = appCtxt;
667 			}
