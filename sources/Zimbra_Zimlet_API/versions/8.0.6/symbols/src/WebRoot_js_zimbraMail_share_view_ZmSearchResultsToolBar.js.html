 16 /**
 17  * @class
 18  * This class represents a search toolbar that shows up above search results. It can be
 19  * used to refine the search results. Each search term is contained within a bubble that
 20  * can easily be removed.
 21  * 
 22  * @param {hash}			params		a hash of parameters:
 23  * @param {DwtComposite}	parent		the parent widget
 24  * @param {string}			id			an explicit ID to use for the control's HTML element
 25  * 
 26  * @extends		ZmSearchToolBar
 27  * 
 28  * @author Conrad Damon
 29  */
 30 ZmSearchResultsToolBar = function(params) {
 31 
 32 	params.posStyle = Dwt.ABSOLUTE_STYLE;
 33 	params.className = "ZmSearchResultsToolBar";
 34 	this._controller = params.controller;
 35 
 36 	ZmSearchToolBar.apply(this, arguments);
 37 	
 38 	this._bubbleId = {};
 39 	this._origin = ZmId.SEARCHRESULTS;
 40 	
 41 	this.initAutocomplete();
 42 };
 43 
 44 ZmSearchResultsToolBar.prototype = new ZmSearchToolBar;
 45 ZmSearchResultsToolBar.prototype.constructor = ZmSearchResultsToolBar;
 46 
 47 ZmSearchResultsToolBar.prototype.isZmSearchResultsToolBar = true;
 48 ZmSearchResultsToolBar.prototype.toString = function() { return "ZmSearchResultsToolBar"; };
 49 
 50 
 51 ZmSearchResultsToolBar.prototype.TEMPLATE = "share.Widgets#ZmSearchResultsToolBar";
 52 
 53 ZmSearchResultsToolBar.prototype._createHtml =
 54 function() {
 55 
 56 	this.getHtmlElement().innerHTML = AjxTemplate.expand(this.TEMPLATE, {id:this._htmlElId});
 57 
 58 	var idContext = this._controller.getCurrentViewId();
 59 	
 60 	this._label = document.getElementById(this._htmlElId + "_label");
 61 	this._iconDiv = document.getElementById(this._htmlElId + "_icon");
 62 	
 63 	// add search input field
 64 	var inputFieldCellId = this._htmlElId + "_inputFieldCell";
 65 	var inputFieldCell = document.getElementById(inputFieldCellId);
 66 	if (inputFieldCell) {
 67 		var aifParams = {
 68 			parent:					this,
 69 			strictMode:				false,
 70 			id:						DwtId.makeId(ZmId.WIDGET_INPUT, idContext),
 71 			bubbleAddedCallback:	this._bubbleChange.bind(this),
 72 			bubbleRemovedCallback:	this._bubbleChange.bind(this),
 73 			noOutsideListening:		true
 74 		}
 75 		var aif = this._searchField = new ZmAddressInputField(aifParams);
 76 		aif.reparentHtmlElement(inputFieldCell);
 77 		
 78 		var inputEl = this._searchField.getInputElement();
 79 		inputEl.className = "search_results_input";
 80 	}
 81 	
 82 	// add search button
 83 	this._button[ZmSearchToolBar.SEARCH_BUTTON] = ZmToolBar.addButton({
 84 				parent:		this, 
 85 				tdId:		"_searchButton",
 86 				buttonId:	ZmId.getButtonId(idContext, ZmId.SEARCHRESULTS_SEARCH),
 87 				lbl:		ZmMsg.search,
 88 				tooltip:	ZmMsg.searchTooltip
 89 			});
 90 
 91 	// add save search button if saved-searches enabled
 92 	this._button[ZmSearchToolBar.SAVE_BUTTON] = ZmToolBar.addButton({
 93 				parent:		this, 
 94 				setting:	ZmSetting.SAVED_SEARCHES_ENABLED,
 95 				tdId:		"_saveButton",
 96 				buttonId:	ZmId.getButtonId(idContext, ZmId.SEARCHRESULTS_SAVE),
 97 				lbl:		ZmMsg.save,
 98 				tooltip:	ZmMsg.saveSearchTooltip
 99 			});
100 };

102 // TODO: use the main search toolbar's autocomplete list - need to manage location callback
103 ZmSearchResultsToolBar.prototype.initAutocomplete =
104 function() {
105 	if (!this._acList) {
106 		this._acList = new ZmAutocompleteListView(this._getAutocompleteParams());
107 		this._acList.handle(this.getSearchField(), this._searchField._htmlElId);
108 	}
109 	this._searchField.setAutocompleteListView(this._acList);
110 };
111 
112 ZmSearchResultsToolBar.prototype._getAutocompleteParams =
113 function() {
114 	var params = ZmSearchToolBar.prototype._getAutocompleteParams.apply(this, arguments);
115 	params.options = { addrBubbles: true, noBubbleParse: true };
116 	return params;
117 };
118 
119 ZmSearchResultsToolBar.prototype.setSearch =
120 function(search) {
121 	this._settingSearch = true;
122 	this._searchField.clear(true);
123 	var tokens = search.getTokens();
124 	if (tokens && tokens.length) {
125 		for (var i = 0, len = tokens.length; i < len; i++) {
126 			var token = tokens[i], prevToken = tokens[i - 1], nextToken = tokens[i + 1];
127 			var showAnd = (prevToken && prevToken.op == ZmParsedQuery.GROUP_CLOSE) || (nextToken && nextToken.op == ZmParsedQuery.GROUP_OPEN);
128 			var text = token.toString(showAnd);
129 			if (text) {
130 				var bubble = this._searchField.addBubble({address:text, noParse:true});
131 				this._bubbleId[text] = bubble.id;
132 			}
133 		}
134 	}
135 	this._settingSearch = false;
136 };
137 
138 ZmSearchResultsToolBar.prototype.setLabel =
139 function(text, showError) {
140 	this._label.innerHTML = text;
141 	this._iconDiv.style.display = showError ? "inline-block" : "none";
142 };
143 
144 // Don't let the removal or addition of a bubble when we're setting up trigger a search loop.
145 ZmSearchResultsToolBar.prototype._bubbleChange =
146 function(bubble, added) {
147 	if (!this._settingSearch) {
148 		var pq = new ZmParsedQuery(bubble.address);
149 		var tokens = pq.getTokens();
150 		// don't run search if a conditional was added or removed
151 		if (!(tokens && tokens[0] && tokens[0].type == ZmParsedQuery.COND)) {
152 			// use timer to let content of search bar get set - if a search term is autocompleted,
153 			// the bubble is added before the text it replaces is removed
154 			setTimeout(this._handleEnterKeyPress.bind(this), 10);
155 		}
156 	}
157 };
158 
159 ZmSearchResultsToolBar.prototype._handleEnterKeyPress =
160 function(ev) {
161 	this.setLabel(ZmMsg.searching);
162 	ZmSearchToolBar.prototype._handleEnterKeyPress.apply(this, arguments);
163 };

165 /**
166  * Adds a bubble for the given search term.
167  * 
168  * @param {ZmSearchToken}	term		search term
169  * @param {boolean}			skipNotify	if true, don't trigger a search
170  * @param {boolean}			addingCond	if true, user clicked to add a conditional term
171  */
172 ZmSearchResultsToolBar.prototype.addSearchTerm =
173 function(term, skipNotify, addingCond) {
174 
175 	var text = term.toString(addingCond);
176 	var index;
177 	if (addingCond) {
178 		var bubbleList = this._searchField._getBubbleList();
179 		var bubbles = bubbleList.getArray();
180 		for (var i = 0; i < bubbles.length; i++) {
181 			if (bubbleList.isSelected(bubbles[i])) {
182 				index = i;
183 				break;
184 			}
185 		}
186 	}
187 
188 	var bubble = this._searchField.addBubble({
189 				address:	text,
190 				addClass:	term.type,
191 				skipNotify:	skipNotify,
192 				index:		index
193 			});
194 	if (bubble) {
195 		this._bubbleId[text] = bubble.id;
196 		return bubble.id;
197 	}
198 };
199 
200 /**
201  * Removes the bubble with the given search term.
202  * 
203  * @param {ZmSearchToken|string}	term		search term or bubble ID
204  * @param {boolean}					skipNotify	if true, don't trigger a search
205  */
206 ZmSearchResultsToolBar.prototype.removeSearchTerm =
207 function(term, skipNotify) {
208 	if (!term) { return; }
209 	var text = term.toString();
210 	var id = term.isZmSearchToken ? this._bubbleId[text] : term;
211 	if (id) {
212 		this._searchField.removeBubble(id, skipNotify);
213 		delete this._bubbleId[text];
214 	}
215 };
