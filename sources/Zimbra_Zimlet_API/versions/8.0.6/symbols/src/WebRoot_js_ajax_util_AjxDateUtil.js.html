 66 AjxDateUtil._init =
 67 function() {
 68 	AjxDateUtil._dateFormat = AjxDateFormat.getDateInstance(AjxDateFormat.SHORT).clone();
 69 	var segments = AjxDateUtil._dateFormat.getSegments();
 70 	for (var i = 0; i < segments.length; i++) {
 71 		if (segments[i] instanceof AjxDateFormat.YearSegment) {
 72 			segments[i] = new AjxDateFormat.YearSegment(AjxDateUtil._dateFormat, "yyyy");
 73 		}
 74 	}
 75 	AjxDateUtil._dateTimeFormat = 
 76 		new AjxDateFormat(AjxDateUtil._dateFormat.toPattern() + " " + AjxDateFormat.getTimeInstance(AjxDateFormat.SHORT));
 77 
 78 	AjxDateUtil._dateFormatNoYear = new AjxDateFormat(AjxMsg.formatDateMediumNoYear);
 79 };

 83 /* return true if the specified date (yyyy|yy, m (0-11), d (1-31))

 86 AjxDateUtil.validDate =
 87 function(y, m, d) {
 88 	var date = new Date(y, m, d);
 89 	var year = y > 999 ? date.getFullYear() : date.getYear();
 90 	return date.getMonth() == m && date.getDate() == d && year == y;
 91 };

124 /**
125  * rolls the month/year. If the day of month in the date passed in is greater
126  * then the max day in the new month, set it to the max. The date passed in is
127  * modified and also returned.
128  */
129 AjxDateUtil.roll = 
130 function(date, field, offset) {
131 	var d = date.getDate();
132 	 // move back to first day before rolling in case previous
133 	 // month/year has less days
134 
135 	if (field == AjxDateUtil.MONTH) {
136 		date.setDate(1);	
137 		date.setMonth(date.getMonth() + offset);
138 		var max = AjxDateUtil.daysInMonth(date.getFullYear(), date.getMonth());
139 		date.setDate(Math.min(d, max));		
140 	} else if (field == AjxDateUtil.YEAR) {
141 		date.setDate(1);		
142 		date.setFullYear(date.getFullYear() + offset);
143 		var max = AjxDateUtil.daysInMonth(date.getFullYear(), date.getMonth());
144 		date.setDate(Math.min(d, max));		
145 	} else if (field == AjxDateUtil.WEEK) {
146 		date.setDate(date.getDate() + 7*offset);
147 	} else if (field == AjxDateUtil.DAY) {
148 		date.setDate(date.getDate() + offset);
149 	} else if (field == AjxDateUtil.TWO_WEEKS) {
150 		date.setDate(date.getDate() + 14*offset);
151 	} else {
152 		return date;
153 	}
154 	return date;
155 };

157 /**
158  * checks whether given date is derived from DST shift
159  */
160 AjxDateUtil.isDayShifted =
161 function(date) {
162     var refDate = new Date(date.getTime());
163 
164     //advance it by 1 day and reset to beginning of the day
165     refDate.setDate(refDate.getDate() +1)
166     refDate.setHours(0,0,0,0);
167 
168     //if DST has no effect the advanced time should differ from given time
169     return refDate.getTime() == date.getTime();
170 };

201 // Returns a string describing the duration, which is in milliseconds.
202 AjxDateUtil.computeDuration =
203 function(duration, brief, simplified) {
204 	// bug fix #2203 - if delta is less than zero, dont bother computing
205 	if (duration < 0) return null;
206 
207 	var years =  Math.floor(duration / (AjxDateUtil.MSEC_PER_DAY * 365));
208 	if (years != 0)
209 		duration -= years * AjxDateUtil.MSEC_PER_DAY * 365;
210 	var months = Math.floor(duration / (AjxDateUtil.MSEC_PER_DAY * 30.42));
211 	if (months > 0)
212 		duration -= Math.floor(months * AjxDateUtil.MSEC_PER_DAY * 30.42);
213 	var days = Math.floor(duration / AjxDateUtil.MSEC_PER_DAY);
214 	if (days > 0)
215 		duration -= days * AjxDateUtil.MSEC_PER_DAY;
216 	var hours = Math.floor(duration / AjxDateUtil.MSEC_PER_HOUR);
217 	if (hours > 0) 
218 		duration -= hours * AjxDateUtil.MSEC_PER_HOUR;
219 	var mins = Math.floor(duration / 60000);
220 	if (mins > 0)
221 		duration -= mins * 60000;
222 	var secs = Math.floor(duration / 1000);
223 
224 	var formatter = brief ? AjxDurationFormatConcise : AjxDurationFormatVerbose;
225 	if (years > 0) {
226 		return simplified
227             ? formatter.formatYears(years)
228             : formatter.formatYears(years, months);
229 	} else if (months > 0) {
230 		return simplified
231             ? formatter.formatMonths(months)
232             : formatter.formatMonths(months, days);
233 	} else if (days > 0) {
234 		return simplified
235             ? formatter.formatDays(days)
236             : formatter.formatDays(days, hours);
237 	} else if (hours > 0) {
238 		return simplified
239             ? formatter.formatHours(hours)
240             : formatter.formatHours(hours, mins);
241 	} else if (mins > 0) {
242 		return simplified
243             ? formatter.formatMinutes(mins)
244             : formatter.formatMinutes(mins, secs);
245 	} else {
246 		return formatter.formatSeconds(secs);
247 	}
248 };

276 AjxDateUtil.computeDateStr =
277 function(now, dateMSec) {
278 	if (dateMSec == null)
279 		return "";
280 
281 	var date = new Date(dateMSec);
282 	if (now.getTime() - dateMSec < AjxDateUtil.MSEC_PER_DAY &&
283 		now.getDay() == date.getDay()) {
284 		return AjxDateUtil.computeTimeString(date);
285 	}
286 
287 	if (now.getFullYear() == date.getFullYear()) {
288 		return AjxDateUtil._dateFormatNoYear.format(date);
289 	}
290 
291 	return AjxDateUtil.simpleComputeDateStr(date);
292 };

299 // Example output: "Today, 9:44 AM" "Yesterday, 12:22 PM" "Sun, 1/11/01 1:11 PM"
300 AjxDateUtil.computeWordyDateStr =
301 function(now, dateMSec) {
302 	if (dateMSec == null) {
303 		return "";
304 	}
305 
306 	var date = new Date(dateMSec);
307 	if (now.getTime() - dateMSec < AjxDateUtil.MSEC_PER_DAY && now.getDay() == date.getDay()) {
308 		if (!AjxDateUtil._wordyDateToday) {
309 			AjxDateUtil._wordyDateToday = new AjxDateFormat(AjxMsg.formatWordyDateToday);
310 		}
311 		return AjxDateUtil._wordyDateToday.format(date);
312 	} else if ((now.getTime() - dateMSec) < (2 * AjxDateUtil.MSEC_PER_DAY) && (now.getDay() - 1) == date.getDay()) {
313 		if (!AjxDateUtil._wordyDateYesterday) {
314 			AjxDateUtil._wordyDateYesterday = new AjxDateFormat(AjxMsg.formatWordyDateYesterday);
315 		}
316 		return AjxDateUtil._wordyDateYesterday.format(date);
317 	} else {
318 		if (!AjxDateUtil._wordyDate) {
319 			AjxDateUtil._wordyDate = new AjxDateFormat(AjxMsg.formatWordyDate);
320 		}
321 		return AjxDateUtil._wordyDate.format(date);
322 	}
323 };

326  * in compliance with the locale of the user ie. dd/mm/yy or mm/dd/yy etc.
327  * Also for date strings like 1/32/2000 (that roll over to 2/1/2000), false is returned.

330 function(str){
331         if(!str) {return false};
332         var dateValue = AjxDateUtil.getSimpleDateFormat().parse(str);
333         if (!dateValue) {return false};
334         var dateValueStr = AjxDateUtil.simpleComputeDateStr(dateValue);

417 AjxDateUtil.getFirstDayOfWeek =
418 function (dt, startOfWeek) {
419     startOfWeek = startOfWeek || 0;
420     var dayOfWeekIndex = dt.getDay();
421     var dayOfWeek = (dayOfWeekIndex - startOfWeek + 7) % 7;
422     dt.setDate(dt.getDate() - dayOfWeek);
423     return dt;
424 };

426 AjxDateUtil.getLastDayOfWeek =
427 function (dt, startOfWeek) {
428     startOfWeek = startOfWeek || 0;
429     var dayOfWeekIndex = dt.getDay();
430     var dayOfWeek = (dayOfWeekIndex - startOfWeek + 7) % 7;
431     dt.setDate(dt.getDate() - dayOfWeek + 6);
432     dt.setHours(23, 59, 59, 999);
433     return dt;
434 };

436 AjxDateUtil.getWeekNumber =
437 function(date, firstDayOfWeek, janDate, isISO8601WeekNum) {
438 
439     // Setup Defaults
440     firstDayOfWeek = firstDayOfWeek || 0;
441     janDate = janDate || AjxDateUtil.WEEK_ONE_JAN_DATE;
442     date = date || new Date();
443 
444     date.setHours(12,0,0,0);
445     var targetDate = date,
446             startOfWeek,
447             endOfWeek;
448 
449     if (targetDate.getDay() === firstDayOfWeek) {
450         startOfWeek = targetDate;
451     } else {
452         startOfWeek = AjxDateUtil.getFirstDayOfWeek(targetDate, firstDayOfWeek);
453     }
454 
455     var startYear = startOfWeek.getFullYear(),
456             startTime = startOfWeek.getTime();
457 
458     // DST shouldn't be a problem here, math is quicker than setDate();
459     endOfWeek = new Date(startOfWeek.getTime() + 6*AjxDateUtil.MSEC_PER_DAY);
460 
461     var weekNum;
462 
463     if(!isISO8601WeekNum) {
464         if (startYear !== endOfWeek.getFullYear() && endOfWeek.getDate() >= janDate) {
465             weekNum = 1;
466         } else {
467             var weekOne = (new Date(startYear, 0, janDate));
468             weekOne.setHours(12,0,0,0);
469             var weekOneDayOne = AjxDateUtil.getFirstDayOfWeek(weekOne, firstDayOfWeek);
470 
471             // Round days to smoothen out 1 hr DST diff
472             var daysDiff  = Math.round((targetDate.getTime() - weekOneDayOne.getTime())/AjxDateUtil.MSEC_PER_DAY);
473 
474             // Calc. Full Weeks
475             var rem = daysDiff % 7;
476             var weeksDiff = (daysDiff - rem)/7;
477             weekNum = weeksDiff + 1;
478         }
479         return weekNum;
480     }else {
481 
482         var newYear = new Date(date.getFullYear(),0,1);
483         var day = newYear.getDay() - 1;
484         day = (day >= 0 ? day : day + 7);
485         var dayOftheYear = Math.floor((date.getTime()-newYear.getTime() - (date.getTimezoneOffset()-newYear.getTimezoneOffset())*60000)/AjxDateUtil.MSEC_PER_DAY) + 1;
486 
487         if(day < 4)
488         {
489             weekNum = Math.floor((dayOftheYear+day-1)/7) + 1;
490             if(weekNum > 52)
491             {
492                 var nxtYear = new Date(date.getFullYear() + 1,0,1);
493                 var nxtDay = nxtYear.getDay() - 1;
494                 nxtDay = nxtDay >= 0 ? nxtDay : nxtDay + 7;
495                 weekNum = nxtDay < 4 ? 1 : 53;
496             }
497         }else {
498             weekNum = Math.floor((dayOftheYear+day -1 )/7);
499             if(weekNum == 0)
500             {
501                 var prevYear = new Date(date.getFullYear()-1,0,1);
502                 var prevDay = prevYear.getDay()-1;
503                 prevDay = (prevDay >= 0 ? prevDay : prevDay + 7);
504                 weekNum = ( prevDay==3 || ( AjxDateUtil.isLeapYear(prevYear.getFullYear()) && prevDay==2 ) ) ? 53 : 52;
505             }
506         }
507         return weekNum;
508     }
509 };

511 AjxDateUtil.getTimeStr = 
512 function(date, format) {
513 	var s = format;
514 	s = s.replace(/%d/g, AjxDateUtil._getDate(date, true));				// zero padded day of the month
515 	s = s.replace(/%D/g, AjxDateUtil._getDate(date, false));			// day of the month without padding
516 	s = s.replace(/%w/g, AjxDateUtil._getWeekday(date));				// day of the week
517 	s = s.replace(/%M/g, AjxDateUtil._getMonthName(date));				// full month name
518 	s = s.replace(/%t/g, AjxDateUtil._getMonthName(date, true));		// abbr. month name
519 	s = s.replace(/%n/g, AjxDateUtil._getMonth(date, true));		    // zero padded month
520 	s = s.replace(/%Y/g, AjxDateUtil._getFullYear(date));				// full year
521 	s = s.replace(/%h/g, AjxDateUtil._getHoursStr(date, false, false));	// non-padded hours
522 	s = s.replace(/%H/g, AjxDateUtil._getHoursStr(date, true, false ));	// padded hours
523 	s = s.replace(/%m/g, AjxDateUtil._getMinutesStr(date));				// padded minutes
524 	s = s.replace(/%s/g, AjxDateUtil._getSecondsStr(date));				// padded seconds
525 	s = s.replace(/%P/g, AjxDateUtil._getAMPM(date, true));				// upper case AM PM
526 	s = s.replace(/%p/g, AjxDateUtil._getAMPM(date, false));			// lower case AM PM
527 	return s;
528 };

538 AjxDateUtil.roundTimeMins = 
539 function(date, roundTo) {
540 	var mins = date.getMinutes();
541 	var hours = date.getHours();
542 	if (mins != 0 && roundTo){
543 		mins = (Math.ceil( (mins/roundTo) )) * roundTo;
544 		if (mins == 60) {
545 			mins = 0;
546 			hours++;
547 		}
548 		date.setMinutes(mins);
549 		date.setHours(hours);
550 	}
551 	return date;
552 };

592 		}

599 AjxDateUtil.parseServerTime = 
600 function(serverStr, date, noSpecialUtcCase) {
601 	if (serverStr.charAt(8) == 'T') {
602 		var hh = parseInt(serverStr.substr(9,2), 10);
603 		var mm = parseInt(serverStr.substr(11,2), 10);
604 		var ss = parseInt(serverStr.substr(13,2), 10);

656     var m = /^(Z)|^(\+|\-)(\d\d):(\d\d)/.exec(s.substr(i));
657     if (m) {

659         if (m[1]) o.timezone = offset;
660         else {

664             o.timezone -= offset;
665         }
666     }
667     return i + (m ? m[0].length : 0);

1048 AjxDateUtil.calculate =
1049 function(rule, date) {
1050 	// initialize
1051 	if (!AjxDateUtil.__calculate_initialized) {
1052 		AjxDateUtil.__calculate_initialized = true;
1053 		AjxDateUtil.__calculate_init();
1054 	}
1055 
1056 	var now = date || new Date;
1057 	rule = rule.replace(/^\s*|\s*$/, "").replace(/\s*=\s*/g,"=").replace(/\s*,\s*/g,",");
1058 	var a = rule.split(/\s+/g);
1059 	var s, m, plusminus, number, type, amount, weekord, daynum;
1060 	for (var i = 0; i < a.length; i++) {
1061 		s = a[i];
1062 		// comment
1063 		if (s.match(AjxDateUtil.RE_COMMENT)) {
1064 			break;
1065 		}
1066 		// context date
1067 		if (s.match(AjxDateUtil.RE_NOW)) {
1068 			date = new Date(now.getTime());
1069 			continue;
1070 		}
1071 		// add
1072 		if (m = s.match(AjxDateUtil.RE_ADD_NUMBER)) {
1073 			plusminus = m[1];
1074 			number = AjxDateUtil.__calculate_parseInt(m[2]);
1075 			type = a[++i];
1076 			amount = plusminus == '+' ? number : number * -1;
1077 			AjxDateUtil.__calculate_add(date, type, amount);
1078 			continue;
1079 		}
1080 		// set
1081 		if (m = s.match(AjxDateUtil.RE_SET)) {
1082 			AjxDateUtil.__calculate_set(date, m[1], m[2]);
1083 			continue;
1084 		}
1085 		// try to parse as a date
1086 		date = AjxDateFormat.parse("yyyyy-MM-dd", s);
1087 		if (!date && (date = AjxDateFormat.parse("yyyy-MM-dd'T'hh:mm:ss'Z'", s))) {
1088 			date.setMinutes(date.getMinutes() - date.getTimezoneOffset());
1089 		}
1090 		if (!date) date = AjxDateFormat.parse("yyyy-MM-dd'T'HH:mm:ss", s);
1091 		if (!date) throw "invalid date pattern: \""+s+"\"";
1092 	}
1093 	return date;
1094 };
