 16 /**
 17  * This class holds all of the known timezone rules. Each timezone
 18  * is represented by an object that has a unique identifier and the
 19  * offset to UTC in standard time. If the timezone defines daylight
 20  * savings time, then additional information is provided (e.g. when
 21  * the DST takes effect and it's offset to UTC).
 22  * <p>
 23  * Both the standard and daylight information are specified as objects
 24  * with the following properties:
 25  * <dl>
 26  * <dt> offset
 27  *   <dd> The offset to UTC (in minutes)
 28  * <dt> mon
 29  *   <dd> The transition month of the year (January = 1, ...).
 30  * <dt> week
 31  *   <dd> The transition week of the month (First = 1, ..., Fourth = 4,
 32  *        Last = -1).
 33  * <dt> wkday
 34  *   <dd> The transition day of the week (Sunday = 1, ...).
 35  * <dt> mday
 36  *   <dd> The transition day of the month (1, ... 31).
 37  * <dt> hour
 38  *   <dd> The transition hour (midnight = 0, noon = 12, ...).
 39  * <dt> min
 40  *   <dd> The transition minute.
 41  * <dt> sec
 42  *   <dt> The transition second (which is usually 0).
 43  * </dl>
 44  *
 45  * <h5>Notes</h5>
 46  * <ul>
 47  * <li> Timezones with no DST only specify an id and a standard info
 48  *      object with a single "offset" property.
 49  * <li> Timezones with a DST <em>must</em> provide standard and
 50  *      daylight info objects.
 51  * <li> If timezone has DST, then the following properties of the
 52  *      standard and daylight info objects are <em>required</em>:
 53  *      offset, month, hour, min, sec.
 54  * <li> Transition dates are specified in only one of the following ways:
 55  *   <ul>
 56  *   <li> by specifying a specific date of the year (e.g. March 10);
 57  *   <li> or by specifying the day of a specific week within some
 58  *        month (e.g. Second Wednesday, Last Saturday, etc).
 59  *   </ul>
 60  * <li> If the transition date is specified as a specific date of the
 61  *      year, then the following field in the standard and/or daylight
 62  *      info objects are <em>required</em>: mday.
 63  * <li> If the transition date is specified as the day of a specific
 64  *      week, then the following fields in the standard and/or daylight
 65  *      info objects are <em>required</em>: week, wkday.
 66  * </ul>
 67  *
 68  * <h5>Examples</h5>
 69  * <dl>
 70  * <dt> Timezone with no DST
 71  *   <dd>
 72  *   <pre>
 73  *   var timezone = { clientId: "My Timezone", standard: { offset: -480 } };
 74  *   </pre>
 75  * <dt> America/Los_Angeles, 2007
 76  *   <dd>
 77  *   <pre>
 78  *   var standard = {
 79  *     offset: -480,
 80  *     mon: 11, week: 1, wkday: 1,
 81  *     hour: 2, min: 0, sec: 0    
 82  *   };
 83  *   var daylight = {
 84  *     offset: -420,
 85  *     mon: 3, week: 2, wkday: 1,
 86  *     hour: 2, min: 0, sec: 0
 87  *   };
 88  *   var timezone = { clientId: "My Timezone",
 89  *                    standard: standard, daylight: daylight };
 90  *   </pre>
 91  * <dt> Custom US/Pacific using 11 Mar 2007 and 2 Dec 2007
 92  *   <dd>
 93  *   <pre>
 94  *   var standard = {
 95  *     offset: -480,
 96  *     mon: 11, mday: 2,
 97  *     hour: 2, min: 0, sec: 0
 98  *   };
 99  *   var daylight = {
100  *     offset: -420,
101  *     mon: 3, mday: 11,
102  *     hour: 2, min: 0, sec: 0
103  *   };
104  *   var timezone = { clientId: "My Timezone",
105  *                    standard: standard, daylight: daylight };
106  *   </pre>
107  * </dl>
108  * <p>
109  * <strong>Note:</strong>
110  * Specifying a transition date using a specific date of the year
111  * <em>should</em> be avoided.
112  *
113  * <hr>
114  *
115  * <p>
116  * This class stores mappings between client and server identifiers for
117  * timezones as well as attempting to guess the default timezone. The 
118  * application can override this value, through a user preference perhaps, 
119  * by setting the <code>DEFAULT</code> property's value. The default 
120  * timezone is specified using the client identifier.
121  *
122  * @class
123  */

130 AjxTimezone.convertTimezone = function(date, fromClientId, toClientId) {
131 	if (fromClientId == toClientId) {
132 		return date;
133 	}
134 	var offset1 = AjxTimezone.getOffset(toClientId, date);
135 	var offset2 = AjxTimezone.getOffset(fromClientId, date);
136 	//returning a new Date object since we might not always want to modify the parameter Date object
137 	return new Date(date.getTime() + (offset1 - offset2) * 60 * 1000);
138 };

141 AjxTimezone.getTransition = function(onset, year) {
142 	var trans = [ year || new Date().getFullYear(), onset.mon, 1 ];
143 	if (onset.mday) {
144 		trans[2] = onset.mday;
145 	}
146 	else if (onset.wkday) {
147 		var date = new Date(year, onset.mon - 1, 1, onset.hour, onset.min, onset.sec);
148 
149 		// last wkday of month
150 		if (onset.week == -1) {
151 			// NOTE: This creates a date of the *last* day of specified month by
152 			//       setting the month to *next* month and setting day of month
153 			//       to zero (i.e. the day *before* the first day).
154 			var last = new Date(new Date(date.getTime()).setMonth(onset.mon, 0));
155 			var count = last.getDate();
156 			var wkday = last.getDay() + 1;
157 			var adjust = wkday >= onset.wkday ? wkday - onset.wkday : 7 - onset.wkday - wkday;
158 			trans[2] = count - adjust;
159 		}
160 
161 		// Nth wkday of month
162 		else {
163 			var wkday = date.getDay() + 1;
164 			var adjust = onset.wkday == wkday ? 1 :0;
165 			trans[2] = onset.wkday + 7 * (onset.week - adjust) - wkday + 1;
166 		}
167 	}
168 	return trans;
169 };

205 AjxTimezone.addWkDayTransition = function(onset) {
206 	var trans = onset.trans;
207 	var mon = trans[1];
208 	var monDay = trans[2];
209 	var week = Math.floor((monDay - 1) / 7);
210 	var date = new Date(trans[0], trans[1] - 1, trans[2], 12, 0, 0);
211 
212 	// NOTE: This creates a date of the *last* day of specified month by
213 	//       setting the month to *next* month and setting day of month
214 	//       to zero (i.e. the day *before* the first day).
215 	var count = new Date(new Date(date.getTime()).setMonth(mon - 1, 0)).getDate();
216 	var last = count - monDay < 7;
217 
218 	// set onset values
219 	onset.mon =  mon;
220 	onset.week = last ? -1 : week + 1;
221 	onset.wkday = date.getDay() + 1;
222 	onset.hour = trans[3];
223 	onset.min = trans[4];
224 	onset.sec = trans[5];
225 	return onset;
226 };

228 AjxTimezone.createTransitionDate = function(onset) {
229     var date = new Date(AjxTimezoneData.TRANSITION_YEAR, onset.mon - 1, 1, 12, 0, 0);
230     if (onset.mday) {
231         date.setDate(onset.mday);
232     }
233     else if (onset.week == -1) {
234         date.setMonth(date.getMonth() + 1, 0);
235         for (var i = 0; i < 7; i++) {
236             if (date.getDay() + 1 == onset.wkday) {
237                 break;
238             }
239             date.setDate(date.getDate() - 1);
240         }
241     }
242     else {
243         for (var i = 0; i < 7; i++) {
244             if (date.getDay() + 1 == onset.wkday) {
245                 break;
246             }
247             date.setDate(date.getDate() + 1);
248         }
249         date.setDate(date.getDate() + 7 * (onset.week - 1));
250     }
251     var trans = [ date.getFullYear(), date.getMonth() + 1, date.getDate() ];
252     return trans;
253 };

320 AjxTimezone.addRule = function(rule) {
321     var serverId = rule.serverId;
322     var clientId = rule.clientId;
323 
324     AjxTimezone._CLIENT2SERVER[clientId] = serverId;
325     AjxTimezone._SERVER2CLIENT[serverId] = clientId;
326     AjxTimezone._SHORT_NAMES[clientId] = AjxTimezone._generateShortName(rule.standard.offset);
327     AjxTimezone._CLIENT2RULE[clientId] = rule;
328 
329     var array = rule.daylight ? AjxTimezone.DAYLIGHT_RULES : AjxTimezone.STANDARD_RULES;
330     array.push(rule);
331 };

333 AjxTimezone.getRule = function(clientId, tz) {
334 	var rule = AjxTimezone._CLIENT2RULE[clientId];
335     if (!rule) {
336         // try to find the rule treating the clientId as the serverId
337         clientId = AjxTimezone._SERVER2CLIENT[clientId];
338         rule = AjxTimezone._CLIENT2RULE[clientId];
339     }
340     if (!rule && tz) {
341         var names = [ "standard", "daylight" ];
342         var rules = tz.daylight ? AjxTimezone.DAYLIGHT_RULES : AjxTimezone.STANDARD_RULES;
343         for (var i = 0; i < rules.length; i++) {
344             rule = rules[i];
345 
346             var found = true;
347             for (var j = 0; j < names.length; j++) {
348                 var name = names[j];
349                 var onset = rule[name];
350                 if (!onset) continue;
351 			
352 				var breakOuter = false;
353 
354                 for (var p in tz[name]) {
355                     if (tz[name][p] != onset[p]) {
356                         found = false;
357                         breakOuter = true;
358                         break;
359                     }
360                 }
361                 
362                 if(breakOuter){
363                 	break;
364                 }
365             }
366             if (found) {
367                 return rule;
368             }
369         }
370         return null;
371     }
372 
373     return rule;
374 };

376 AjxTimezone.getOffset = function(clientId, date) {
377 	var rule = AjxTimezone.getRule(clientId || AjxTimezone.DEFAULT);
378 	if (rule && rule.daylight) {
379 		var year = date.getFullYear();
380 
381 		var standard = rule.standard, daylight  = rule.daylight;
382 		var stdTrans = AjxTimezone.getTransition(standard, year);
383 		var dstTrans = AjxTimezone.getTransition(daylight, year);
384 
385 		var month    = date.getMonth()+1, day = date.getDate();
386 		var stdMonth = stdTrans[1], stdDay = stdTrans[2];
387 		var dstMonth = dstTrans[1], dstDay = dstTrans[2];
388 
389 		// northern hemisphere
390 		var isDST = false;
391 		if (dstMonth < stdMonth) {
392 			isDST = month > dstMonth && month < stdMonth;
393 			isDST = isDST || (month == dstMonth && day >= dstDay);
394 			isDST = isDST || (month == stdMonth && day <  stdDay);
395 		}
396 
397 		// sorthern hemisphere
398 		else {
399 			isDST = month < stdMonth || month > dstMonth;
400 			isDST = isDST || (month == dstMonth && day >=  dstDay);
401 			isDST = isDST || (month == stdMonth && day < stdDay);
402 		}
403 
404 		return isDST ? daylight.offset : standard.offset;
405 	}
406 	return rule ? rule.standard.offset : -(new Date().getTimezoneOffset());
407 };

413 AjxTimezone.getAbbreviatedZoneChoices = function() {
414 	if (AjxTimezone._ABBR_ZONE_OPTIONS) {
415 		var count = AjxTimezone._ABBR_ZONE_OPTIONS.length;
416 		var total = AjxTimezone.STANDARD_RULES.length + AjxTimezone.DAYLIGHT_RULES.length;
417 		if (count != total) {
418 			AjxTimezone._ABBR_ZONE_OPTIONS = null;
419 		}
420 	}
421 	if (!AjxTimezone._ABBR_ZONE_OPTIONS) {
422 		AjxTimezone._ABBR_ZONE_OPTIONS = [];
423 		for (var clientId in AjxTimezone._CLIENT2SERVER) {
424 			var rule = AjxTimezone._CLIENT2RULE[clientId];
425 			var serverId = rule.serverId;
426 			var option = {
427 				displayValue: AjxTimezone.getMediumName(clientId),
428 				value: serverId,
429 				// these props used by sort comparator
430 				standard: rule.standard,
431 				serverId: serverId, //In _BY_OFFSET, the attribute name is serverId.
432                 clientId: clientId
433 			};
434 			AjxTimezone._ABBR_ZONE_OPTIONS.push(option);
435 		}
436 		AjxTimezone._ABBR_ZONE_OPTIONS.sort(AjxTimezone._BY_OFFSET);
437 	}
438 	return AjxTimezone._ABBR_ZONE_OPTIONS;
439 };

459 				// these props used by sort comparator
460 				standard: rule.standard,
461 				serverId: serverId, //In _BY_OFFSET, the attribute name is serverId.

464 			AjxTimezone._MATCHING_ZONE_OPTIONS.push(option);
465 		}

467 	}
468 	return AjxTimezone._MATCHING_ZONE_OPTIONS;

471 AjxTimezone._BY_OFFSET = function(arule, brule) {
472 	// sort by offset and then by name
473 	var delta = arule.standard.offset - brule.standard.offset;
474 	if (delta == 0) {
475 		var aname = arule.serverId;
476 		var bname = brule.serverId;
477 		if (aname < bname) delta = -1;
478 		else if (aname > bname) delta = 1;
479 	}
480 	return delta;
481 };

534 /**
535  * One problem with firefox, is if the timezone on the machine changes,
536  * the browser isn't updated. You have to restart firefox for it to get the 
537  * new machine timezone.
538  * <p>
539  * <strong>Note:</strong>
540  * It looks like the current versions of FF always reflect the current
541  * timezone w/o needing to restart the browser.
542  * timezonePreference - optional value used to decide timezone rule in case of conflict 
543  */
544 AjxTimezone._guessMachineTimezone = 
545 function(timezonePreference) {
546 	var dec1 = new Date(AjxTimezoneData.TRANSITION_YEAR, 11, 1, 0, 0, 0);
547 	var jun1 = new Date(AjxTimezoneData.TRANSITION_YEAR, 5, 1, 0, 0, 0);
548 	var dec1offset = -dec1.getTimezoneOffset();
549 	var jun1offset = -jun1.getTimezoneOffset();
550 
551     AjxTimezone.MATCHING_RULES = [];
552     AjxTimezone.TIMEZONE_CONFLICT = false;
553     var matchingRules = [];
554     var matchingRulesMap = {};
555     var offsetMatchingRules = [];
556     var daylightMatchingFound = false;
557 
558     // if the offset for jun is the same as the offset in december,
559 	// then we have a timezone that doesn't deal with daylight savings.
560 	if (jun1offset == dec1offset) {
561 		var rules = AjxTimezone.STANDARD_RULES;
562  		for (var i = 0; i < rules.length ; ++i ) {
563             var rule = rules[i];
564             if (rule.standard.offset == jun1offset) {
565 				 if(!matchingRulesMap[rule.serverId]) {
566                      matchingRules.push(rule);
567                      matchingRulesMap[rule.serverId] = true;
568                  }
569                  AjxTimezone.MATCHING_RULES.push(rule);
570 			}
571 		}
572 	}
573 
574     // we need to find a rule that matches both offsets
575     else {
576 		var rules = AjxTimezone.DAYLIGHT_RULES;
577 		var dst = Math.max(dec1offset, jun1offset);
578 		var std = Math.min(dec1offset, jun1offset);
579         var now = new Date();
580         var currentOffset = -now.getTimezoneOffset();
581         for (var i = 0; i < rules.length ; ++i ) {
582 			var rule = rules[i];
583 			if (rule.standard.offset == std && rule.daylight.offset == dst) {
584                 var strans = rule.standard.trans;
585                 var dtrans = rule.daylight.trans;
586 
587                 var s0 = new Date(strans[0], strans[1]-1, strans[2]-1);
588                 var s1 = new Date(strans[0], strans[1]-1, strans[2]+2);
589                 var d0 = new Date(dtrans[0], dtrans[1]-1, dtrans[2]-1);
590                 var d1 = new Date(dtrans[0], dtrans[1]-1, dtrans[2]+2);
591                 if (-s1.getTimezoneOffset() == std && -d1.getTimezoneOffset() == dst &&
592                     -s0.getTimezoneOffset() == dst && -d0.getTimezoneOffset() == std) {
593                     if(!matchingRulesMap[rule.serverId]) {
594                         matchingRules.push(rule);
595                         matchingRulesMap[rule.serverId] = true;
596                     }                    
597                     daylightMatchingFound = true;
598                 }
599             }
600             //used for conflict resolution when server rules are wrong 
601             if (rule.standard.offset == currentOffset || rule.daylight.offset == currentOffset) {
602                     AjxTimezone.MATCHING_RULES.push(rule);
603             }
604 		}
605 	}
606 
607     //when there is a timezone conflict use the preference to find better match
608     if((matchingRules.length > 0) && timezonePreference != null) {
609         var rules = matchingRules; 
610         for(var i in rules) {
611             if(rules[i].serverId == timezonePreference) {
612                 return rules[i];
613             }
614         }
615     }
616 
617     if(matchingRules.length > 0) {
618         // resolve conflict, if possible
619         if (matchingRules.length > 1) {
620             matchingRules.sort(AjxTimezone.__BY_SCORE);
621             if (matchingRules[0].score != matchingRules[1].score) {
622                 matchingRules.length = 1;
623             }
624         }
625         // mark if conflict and return best guess
626         AjxTimezone.TIMEZONE_CONFLICT = (matchingRules.length > 1);  
627         return matchingRules[0];        
628     }
629 
630     if((AjxTimezone.MATCHING_RULES.length > 0) && timezonePreference != null) {
631         var rules = AjxTimezone.MATCHING_RULES; 
632         for(var i in rules) {
633             if(rules[i].serverId == timezonePreference) {
634                 return rules[i];
635             }
636         }
637     }
638 
639     // generate default rule
640     return AjxTimezone._generateDefaultRule();
641 };

643 AjxTimezone.__BY_SCORE = function(a, b) {
644     return b.score - a.score;
645 };

647 // Thanks to Jiho for this new, improved logic for generating the timezone rule.
648 AjxTimezone._generateDefaultRule = function() {
649 	var byMonth = 0;
650 	var byDate = 1;
651 	var byHour = 2;
652 	var byMinute = 3;
653 	var bySecond = 4;
654 
655 	// Sweep the range between d1 and d2 looking for DST transitions.
656 	// Iterate the range by "by" unit.  When a transition is detected,
657 	// sweep the range between before/after dates by increasingly
658 	// smaller unit, month then date then hour then minute then finally second.
659 	function sweepRange(d1, d2, by, rule) {
660 		var upperBound = d2.getTime();
661 		var d = new Date();
662 		d.setTime(d1.getTime());
663 		var prevD = new Date();
664 		prevD.setTime(d.getTime());
665 		var prevOffset = d1.getTimezoneOffset() * -1;
666 
667 		// initialize rule
668 		if (!rule) {
669 			rule = {
670 				clientId: AjxTimezone.AUTO_DETECTED,
671 				autoDetected: true
672 			};
673 		}
674 
675 		// perform sweep
676 		while (d.getTime() <= upperBound) {
677 			// Increment by the right unit.
678 			if (by == byMonth) {
679 				d.setUTCMonth(d.getUTCMonth() + 1);
680 			}
681 			else if (by == byDate) {
682 				d.setUTCDate(d.getUTCDate() + 1);
683 			}
684 			else if (by == byHour) {
685 				d.setUTCHours(d.getUTCHours() + 1);
686 			}
687 			else if (by == byMinute) {
688 				d.setUTCMinutes(d.getUTCMinutes() + 1);
689 			}
690 			else if (by == bySecond) {
691 				d.setUTCSeconds(d.getUTCSeconds() + 1);
692 			}
693 			else {
694 				return rule;
695 			}
696 
697 			var offset = d.getTimezoneOffset() * -1;
698 			if (offset != prevOffset) {
699 				if (by < bySecond) {
700 					// Drill down.
701 					rule = sweepRange(prevD, d, by + 1, rule);
702 				}
703 				else {
704 					// Tricky:
705 					// Initialize a Date object whose UTC fields are set to prevD's local fields.
706 					// Then add 1 second to get UTC version of onset time.  We want to work in UTC
707 					// to prevent the date object from experiencing the DST jump when we add 1 second.
708 					var trans = new Date();
709 					trans.setUTCFullYear(prevD.getFullYear(), prevD.getMonth(), prevD.getDate());
710 					trans.setUTCHours(prevD.getHours(), prevD.getMinutes(),     prevD.getSeconds() + 1);
711 
712 					var onset = rule[prevOffset < offset ? "daylight" : "standard"] = {
713 						offset: offset,
714 						trans: [
715 							trans.getUTCFullYear(), trans.getUTCMonth() + 1, trans.getUTCDate(),    // yyyy-MM-dd
716 							trans.getUTCHours(),    trans.getUTCMinutes(),   trans.getUTCSeconds()  //   HH:mm:ss
717 						]
718 					};
719 					AjxTimezone.addWkDayTransition(onset);
720 					return rule;
721 				}
722 			}
723 
724 			prevD.setTime(d.getTime());
725 			prevOffset = offset;
726 		}
727 
728 		return rule;
729 	}
730 
731 	// Find DST transitions between yyyy/07/71 00:00:00 and yyyy+1/06/30 23:59:59.
732 	// We can detect transition on/around 12/31 and 01/01.  Assume no one will
733 	// transition on/around 6/30 and 07/01.
734 	var d1 = new Date();
735 	var d2 = new Date();
736 
737 	// set sweep start to yesterday
738 	var year = d1.getFullYear();
739 	d1.setUTCFullYear(year, d1.getMonth(), d1.getDate() - 1);
740 	d1.setUTCHours(0, 0, 0, 0);
741 
742 	// set sweep end to tomorrow + 1 year
743 	d2.setTime(d1.getTime());
744 	d2.setUTCFullYear(year + 1, d1.getMonth(), d1.getDate() + 1);
745 
746 	// case 1: no onset returned -> TZ doesn't use DST
747 	// case 2: two onsets returned -> TZ uses DST
748 	// case 3: only one onset returned -> mid-year policy change -> simplify and assume it's non-DST
749 	// case 4: three or more onsets returned -> shouldn't happen
750 	var rule = sweepRange(d1, d2, byMonth);
751 
752 	// handle case 1 and 3
753 	if (!rule.daylight || !rule.standard) {
754 		rule.standard = { offset: d1.getTimezoneOffset() * -1 };
755 		delete rule.daylight;
756 	}
757 
758 	// now that standard offset is determined, set serverId
759 	rule.serverId = ["(GMT",AjxTimezone._generateShortName(rule.standard.offset, true),") ",AjxTimezone.AUTO_DETECTED].join("");
760 
761 	// bug 33800: guard against inverted daylight/standard onsets
762 	if (rule.daylight && rule.daylight.offset < rule.standard.offset) {
763 		var onset = rule.daylight;
764 		rule.daylight = rule.standard;
765 		rule.standard = onset;
766 	}
767 
768 	// add generated rule to proper list
769 	//AjxTimezoneData.TIMEZONE_RULES.unshift(rule);
770 	//var rules = rule.daylight ? AjxTimezone.DAYLIGHT_RULES : AjxTimezone.STANDARD_RULES;
771 	//rules.unshift(rule);
772 
773 	return rule;
774 };

776 AjxTimezone._generateShortName = function(offset, period) {
777 	if (offset == 0) return "";
778 	var sign = offset < 0 ? "-" : "+";
779 	var stdOffset = Math.abs(offset);
780 	var hours = Math.floor(stdOffset / 60);
781 	var minutes = stdOffset % 60;
782 	hours = hours < 10 ? '0' + hours : hours;
783 	minutes = minutes < 10 ? '0' + minutes : minutes;
784 	return [sign,hours,period?".":"",minutes].join("");
785 };
