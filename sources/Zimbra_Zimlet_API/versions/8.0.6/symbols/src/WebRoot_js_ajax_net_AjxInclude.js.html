 17 /**
 18  * Utility function (NOT object!) that loads external files in a specified
 19  * order (thus it'll work fine if scripts have dependencies, you just need to
 20  * pass them all in the includes array).  If a callback is given, it will be
 21  * called when all the scripts in the includes array (and possibly dependencies
 22  * generated by them using document.write) are loaded.
 23  *
 24  * Note that scripts are loaded asynchronously -- this function returns
 25  * immediately, but there's no guarantee that scripts have finished loading by
 26  * that time.  DO pass a callback if you want to execute code that rely on
 27  * those scripts.
 28  *
 29  * @param {array}	includes 	An array of strings or object; each is the URL to
 30  *                              some script OR an object with a required src property
 31  *                              and an optional id property. The id, if specified,
 32  *                              will be attached to the SCRIPT tag so that the caller
 33  *                              can access the element later.
 34  * @param {string}	baseurl 	the URL that will be prepended to scripts with a
 35  *                            relative address (not starting with http://, ftp:// or /)
 36  * @param {AjxCallback}	callback the callback will be called when all scripts were processed
 37  *
 38  * @author Mihai Bazon
 39  * 
 40  * @private
 41  */
 42 AjxInclude = function(includes, baseurl, callback, proxy) {
 43 	var head = document.getElementsByTagName("head")[0];
 44 
 45 	function loadNextScript(script) {
 46 		if (AjxEnv.isIE && script && !/loaded|complete/.test(script.readyState))
 47 			return;
 48 		if (script) {
 49 			// Clear the event handler so IE won't leak.  (Did you know
 50 			// that Microsoft knew about the mem. leak bug in 1998
 51 			// and by the end of 2005 it's still not fixed? :-p
 52 			// http://www.bazon.net/mishoo/home.epl?NEWS_ID=1281 )
 53 			script[AjxInclude.eventName] = null;
 54 		}
 55 		var scripts = AjxInclude.dwhack_scripts.length > 0
 56 			? AjxInclude.dwhack_scripts
 57 			: includes;
 58 		window.status = "";
 59 		if (scripts.length > 0) {
 60             var object = scripts.shift();
 61 			var fullurl = typeof object == "string" ? object : object.src;
 62 			var orig = fullurl;
 63 			if (!/^((https?|ftps?):\x2f\x2f|\x2f)/.test(fullurl)) {
 64 				if (baseurl)
 65 					fullurl = baseurl + fullurl;
 66 				if (cacheKillerVersion)
 67 					fullurl += "?v=" + cacheKillerVersion;
 68 			} else if (proxy && fullurl.indexOf('/') != 0) {
 69  				// fully qualified URL-s will go through our proxy
 70  				fullurl = proxy + AjxStringUtil.urlEncode(fullurl);
 71 			}
 72 			var script = document.createElement("script");
 73 			var handler = AjxCallback.simpleClosure(loadNextScript, null, script);
 74 			if (AjxEnv.isIE) {
 75 				script.attachEvent("onreadystatechange", handler);
 76 				script.attachEvent("onerror", handler);
 77 			} else {
 78 				script.addEventListener("load", handler, true);
 79 				script.addEventListener("error", handler, true);
 80 			}
 81 			script.type = "text/javascript";
 82 			script.src = fullurl;
 83 			window.status = "Loading script: " + orig;
 84 
 85             var oscript = null;
 86             if (object.id) {
 87                 script.id = object.id;
 88                 oscript = document.getElementById(object.id);
 89             }
 90             if (oscript) {
 91                 oscript.parentNode.replaceChild(script, oscript);
 92             }
 93             else {
 94                 head.appendChild(script);
 95             }
 96 		} else if (includes.length == 0) {
 97 			script = null;
 98 			head = null;
 99 			if (callback)
100 				callback.run();
101 		}
102 	};
103 
104 	loadNextScript(null);
105 };

107 // The document.write hack.  If files are present in this array, they will be
108 // favored by AjxInclude (see inner function loadNextScript).  I originally
109 // tried to make the function below an inner function too, but this doesn't
110 // work because the whole mess is asynchronous (think multiple Zimlets loading
111 // external files that call document.write).
112 AjxInclude.dwhack_scripts = [];
113 document.write = document.writeln = function() {
114 	// let's assume there may be more arguments
115 	var a = [];
116 	for (var i = 0; i < arguments.length; ++i)
117 		a[i] = arguments[i];
118 	var str = a.join("");
119 	if (/<script[^>]+src=([\x22\x27])(.*?)\1/i.test(str)) {
120 		// we have a <script>, let's add it to our includes list. :-)
121 		AjxInclude.dwhack_scripts.push(RegExp.$2);
122 	}
123 	// If it's not a script, we can't do anything...  The idea is that the
124 	// original document.write would mess all our HTML anyway, so we can't
125 	// call it.  If scripts rely on it to insert elements, well, that's too
126 	// bad--they won't work.  For this reason we don't even care to save
127 	// the original functions.
128 };
129 
130 
131 // XXX: DO NOT REMOVE - PREVENTS MEM LEAK IN IE
132 if (AjxEnv.isIE) {
133 	AjxInclude._removeWriteln = function() {
134 		document.write = document.writeln = null;
135 		window.detachEvent("onunload", AjxInclude._removeWriteln);
136 	};
137 	window.attachEvent("onunload", AjxInclude._removeWriteln);
138 }
