 16 /**
 17  * @class
 18  * This class provides a central area for managing email recipient fields. It is not a control,
 19  * and does not exist within the widget hierarchy.
 20  * 
 21  * @param {hash}		params						a hash of params:
 22  * @param {function}	resetContainerSizeMethod	callback for when size needs to be adjusted
 23  * @param {function}	enableContainerInputs		callback for enabling/disabling input fields
 24  * @param {function}	reenter						callback to enable design mode
 25  * @param {AjxListener}	contactPopdownListener		listener called when contact picker pops down
 26  * @param {string}		contextId					ID of owner (used for autocomplete list)
 27  */
 28 ZmRecipients = function(params) {
 29 
 30     this._useAcAddrBubbles = appCtxt.get(ZmSetting.USE_ADDR_BUBBLES) && appCtxt.get(ZmSetting.CONTACTS_ENABLED);
 31 	this._divId			= {};
 32 	this._buttonTdId	= {};
 33 	this._fieldId		= {};
 34 	this._using			= {};
 35 	this._button		= {};
 36 	this._field			= {};
 37 	this._divEl			= {};
 38     if (this._useAcAddrBubbles) {
 39         this._addrInputField = {};
 40     }
 41 
 42     this._resetContainerSize = params.resetContainerSizeMethod;
 43     this._enableContainerInputs = params.enableContainerInputs;
 44     this._reenter = params.reenter;
 45     this._contactPopdownListener = params.contactPopdownListener;
 46 	this._contextId = params.contextId;
 47 
 48     this._bubbleOps = {};
 49     this._bubbleOps[AjxEmailAddress.TO]  = ZmOperation.MOVE_TO_TO;
 50     this._bubbleOps[AjxEmailAddress.CC]  = ZmOperation.MOVE_TO_CC;
 51     this._bubbleOps[AjxEmailAddress.BCC] = ZmOperation.MOVE_TO_BCC;
 52     this._opToField = {};
 53     this._opToField[ZmOperation.MOVE_TO_TO]  = AjxEmailAddress.TO;
 54     this._opToField[ZmOperation.MOVE_TO_CC] = AjxEmailAddress.CC;
 55     this._opToField[ZmOperation.MOVE_TO_BCC] = AjxEmailAddress.BCC;
 56 };
 57 
 58 ZmRecipients.OP = {};
 59 ZmRecipients.OP[AjxEmailAddress.TO]		= ZmId.CMP_TO;
 60 ZmRecipients.OP[AjxEmailAddress.CC]		= ZmId.CMP_CC;
 61 ZmRecipients.OP[AjxEmailAddress.BCC]	= ZmId.CMP_BCC;
 62 
 63 ZmRecipients.ADDR_SETTING = {};
 64 ZmRecipients.ADDR_SETTING[AjxEmailAddress.BCC]= ZmSetting.SHOW_BCC;
 65 
 66 ZmRecipients.BAD = "_bad_addrs_";

 74 ZmRecipients.prototype.createRecipientIds =
 75 function(htmlElId, typeStr) {
 76     var ids = {};
 77     var components = ["row", "picker", "control", "cell"];
 78     for (var i = 0; i < components.length; i++) {
 79         ids[components[i]] = [htmlElId, typeStr, components[i]].join("_")
 80     }
 81     return ids;
 82 }

 85 ZmRecipients.prototype.createRecipientHtml =
 86 function(parent, viewId, htmlElId, fieldNames, bccToggleId) {
 87 
 88 	this._fieldNames = fieldNames;
 89 	var contactsEnabled = appCtxt.get(ZmSetting.CONTACTS_ENABLED);
 90 	var galEnabled = appCtxt.get(ZmSetting.GAL_ENABLED);
 91 
 92 	// init autocomplete list
 93 	if (contactsEnabled || galEnabled || appCtxt.isOffline) {
 94 		var params = {
 95 			dataClass:		appCtxt.getAutocompleter(),
 96 			matchValue:		ZmAutocomplete.AC_VALUE_FULL,
 97 			compCallback:	this._acCompHandler.bind(this),
 98 			keyUpCallback:	this._acKeyupHandler.bind(this),
 99 			options:		{addrBubbles:this._useAcAddrBubbles},
100 			contextId:		this._contextId
101 		};
102 		this._acAddrSelectList = new ZmAutocompleteListView(params);
103 	}
104 
105 	var isPickerEnabled = contactsEnabled || galEnabled || appCtxt.multiAccounts;
106 
107 	this._pickerButton = {};
108 
109 	// process compose fields
110 	for (var i = 0; i < fieldNames.length; i++) {
111 		var type = fieldNames[i];
112 		var typeStr = AjxEmailAddress.TYPE_STRING[type];
113 
114 		// save identifiers
115 		var ids = this.createRecipientIds(htmlElId, typeStr);
116 		this._divId[type] = ids.row;
117 		this._buttonTdId[type] = ids.picker;
118 		var inputId = this._fieldId[type] = ids.control;
119 
120 		// save field elements
121 		this._divEl[type] = document.getElementById(this._divId[type]);
122 		var aifId;
123 		if (this._useAcAddrBubbles) {
124 			var aifParams = {
125 				parent:								parent,
126 				autocompleteListView:				this._acAddrSelectList,
127 				bubbleAddedCallback:				(new AjxCallback(this, this._bubblesChangedCallback)),
128 				bubbleRemovedCallback:				(new AjxCallback(this, this._bubblesChangedCallback)),
129 				bubbleMenuCreatedCallback:			(new AjxCallback(this, this._bubbleMenuCreated)),
130 				bubbleMenuResetOperationsCallback:	(new AjxCallback(this, this._bubbleMenuResetOperations)),
131 				inputId:							inputId,
132 				type:								type
133 			}
134 			var aif = this._addrInputField[type] = new ZmAddressInputField(aifParams);
135 			aifId = aif._htmlElId;
136 			aif.reparentHtmlElement(ids.cell);
137 		}
138 
139 		// save field control
140 		this._field[type] = document.getElementById(this._fieldId[type]);
141 		if (this._field[type]) {
142 			this._field[type].addrType = type;
143 			if (!this._useAcAddrBubbles) {
144 				this._setEventHandler(this._fieldId[type], "onFocus");
145 			}
146 		}
147 
148 		// create picker
149 		if (isPickerEnabled) {
150 
151 			// bug 78318 - if GAL enabled but not contacts, we need some things defined to handle GAL search
152 			if (!contactsEnabled) {
153 				appCtxt.getAppController()._createApp(ZmApp.CONTACTS);
154 			}
155 
156 			var pickerId = this._buttonTdId[type];
157 			var pickerEl = document.getElementById(pickerId);
158 			if (pickerEl) {
159 				var buttonId = ZmId.getButtonId(viewId, ZmRecipients.OP[type]);
160 				var button = this._pickerButton[type] = new DwtButton({parent:parent, id:buttonId});
161 				button.setText(pickerEl.innerHTML);
162 				button.replaceElement(pickerEl);
163 
164 				button.addSelectionListener(this.addressButtonListener.bind(this));
165 				button.addrType = type;
166 
167 				// autocomplete-related handlers
168 				if (contactsEnabled || appCtxt.isOffline) {
169 					this._acAddrSelectList.handle(this._field[type], aifId);
170 				} else {
171 					this._setEventHandler(this._fieldId[type], "onKeyUp");
172 				}
173 
174 				this._button[type] = button;
175 			}
176 		}
177 	}
178 
179 	// Toggle BCC
180 	if (bccToggleId){
181 		this._toggleBccEl = document.getElementById(bccToggleId);
182 		if (this._toggleBccEl) {
183 			Dwt.setHandler(this._toggleBccEl, DwtEvent.ONCLICK, AjxCallback.simpleClosure(this._toggleBccField, this));
184 		}
185 	}
186 }

188 ZmRecipients.prototype.reset =
189 function() {
190 
191 	// reset To/CC/BCC fields
192 	for (var i = 0; i < this._fieldNames.length; i++) {
193 		var type = this._fieldNames[i];
194 		var textarea = this._field[type];
195 		textarea.value = "";
196 		this._adjustAddrHeight(textarea, true);
197 		if (this._useAcAddrBubbles) {
198 			var addrInput = this._addrInputField[type];
199 			if (addrInput) {
200 				addrInput.clear();
201 			}
202 		}
203 	}

257 // Adds the given addresses to the form. If we're using address bubbles, we need to add each
258 // address separately in case it's a DL.
259 ZmRecipients.prototype.addAddresses =
260 function(type, addrVec, used) {
261 
262 	var addrAdded = false;
263 	used = used || {};
264 	var addrList = [];
265 	var addrs = AjxUtil.toArray(addrVec);
266 	if (addrs && addrs.length) {
267 		for (var i = 0, len = addrs.length; i < len; i++) {
268 			var addr = addrs[i];
269 			var email = addr.isAjxEmailAddress ? addr && addr.getAddress() : addr;
270 			if (!email) { continue; }
271 			email = email.toLowerCase();
272 			if (!used[email]) {
273 				if (this._useAcAddrBubbles) {
274 					this.setAddress(type, addr);	// add the bubble now
275 				} else {
276 					addrList.push(addr);
277 				}
278 				used[email] = true;
279 				addrAdded = true;
280 			}
281 		}
282 		if (!this._useAcAddrBubbles) {
283 			// calls implicit toString() on each addr object
284 			var addrStr = addrList.join(AjxEmailAddress.SEPARATOR);
285 			this.setAddress(type, addrStr);
286 		}
287 	}
288 	return addrAdded;
289 };

292 /**
293  * Sets an address field.
294  *
295  * @param type	the address type
296  * @param addr	the address string
297  *
298  * XXX: if addr empty, check if should hide field
299  *
300  * @private
301  */
302 ZmRecipients.prototype.setAddress =
303 function(type, addr) {
304 
305 	addr = addr || "";
306 
307 	var addrStr = addr.isAjxEmailAddress ? addr.toString() : addr;
308 
309 	//show first, so focus works on IE.
310 	if (addrStr.length && !this._using[type]) {
311 		this._using[type] = true;
312 		this._showAddressField(type, true);
313 	}
314 
315 	if (this._useAcAddrBubbles) {
316 		var addrInput = this._addrInputField[type];
317 		if (!addrStr) {
318 			addrInput.clear();
319 		}
320 		else {
321 			if (addr.isAjxEmailAddress) {
322 				var match = {isDL: addr.isGroup && addr.canExpand, email: addrStr};
323 				addrInput.addBubble({address:addrStr, match:match, skipNotify:true, noFocus:true});
324 			}
325 			else {
326 				this._setAddrFieldValue(type, addrStr);
327 			}
328 		}
329 	}
330 	else {
331 		this._setAddrFieldValue(type, addrStr);
332 	}
333 
334 	// Use a timed action so that first time through, addr textarea
335 	// has been sized by browser based on content before we try to
336 	// adjust it (bug 20926)
337 	AjxTimedAction.scheduleAction(new AjxTimedAction(this,
338 		function() {
339 			this._adjustAddrHeight(this._field[type]);
340 		}), 0);
341 };

375 // Grab the addresses out of the form. Optionally, they can be returned broken
376 // out into good and bad addresses, with an aggregate list of the bad ones also
377 // returned. If the field is hidden, its contents are ignored.
378 ZmRecipients.prototype.collectAddrs =
379 function() {
380 
381 	var addrs = {};
382 	addrs[ZmRecipients.BAD] = new AjxVector();
383 	for (var i = 0; i < this._fieldNames.length; i++) {
384 		var type = this._fieldNames[i];
385 		if (!this._using[type]) { continue; }
386 
387 		var val = this.getAddrFieldValue(type);
388 		if (val.length == 0) { continue; }
389 		val = val.replace(/[; ,]+$/, "");	// ignore trailing (and possibly extra) separators
390 		var result = AjxEmailAddress.parseEmailString(val, type, false);
391 		if (result.all.size() == 0) { continue; }
392 		addrs.gotAddress = true;
393 		addrs[type] = result;
394 		if (result.bad.size()) {
395 			addrs[ZmRecipients.BAD].addList(result.bad);
396 			if (!addrs.badType) {
397 				addrs.badType = type;
398 			}
399 		}
400 	}
401 	return addrs;
402 };

405 ZmRecipients.prototype.getAddrFieldValue =
406 function(type) {
407 
408 	var val = "";
409 	if (this._useAcAddrBubbles) {
410 		var addrInput = this._addrInputField[type];
411 		if (addrInput) {
412 			val = addrInput.getValue();
413 		}
414 	}
415 	else {
416 		val = AjxStringUtil.trim(this._field[type].value)
417 	}
418 
419 	return val;
420 };

433 // Address buttons invoke contact picker
434 ZmRecipients.prototype.addressButtonListener =
435 function(ev, addrType) {
436 	var obj = ev ? DwtControl.getTargetControl(ev) : null;
437 	if (this._enableContainerInputs) {
438 		this._enableContainerInputs(false);
439 	}
440 
441 	if (!this._contactPicker) {
442 		AjxDispatcher.require("ContactsCore");
443 		var buttonInfo = [];
444         for (var i = 0; i < this._fieldNames.length; i++) {
445             buttonInfo[i] = { id: this._fieldNames[i],
446                               label : ZmMsg[AjxEmailAddress.TYPE_STRING[this._fieldNames[i]]]};
447         }
448 		this._contactPicker = new ZmContactPicker(buttonInfo);
449 		this._contactPicker.registerCallback(DwtDialog.OK_BUTTON, this._contactPickerOkCallback, this);
450 		this._contactPicker.registerCallback(DwtDialog.CANCEL_BUTTON, this._contactPickerCancelCallback, this);
451 	}
452 
453 	var curType = obj ? obj.addrType : addrType;
454 	var addrList = {};
455 	var addrs = !this._useAcAddrBubbles && this.collectAddrs();
456 	for (var i = 0; i < this._fieldNames.length; i++) {
457 		var type = this._fieldNames[i];
458 		addrList[type] = this._useAcAddrBubbles ? this._addrInputField[type].getAddresses(true) :
459 				   								  addrs[type] && addrs[type].good.getArray();
460 	}
461 	if (this._contactPopdownListener) {
462 		this._contactPicker.addPopdownListener(this._contactPopdownListener);
463 	}
464 	var str = (this._field[curType].value && !(addrList[curType] && addrList[curType].length))
465 		? this._field[curType].value : "";
466 
467 	var account;
468 	if (appCtxt.multiAccounts && this._fromSelect) {
469 		var addr = this._fromSelect.getSelectedOption().addr;
470 		account = appCtxt.accountList.getAccountByEmail(addr.address);
471 	}
472 	this._contactPicker.popup(curType, addrList, str, account);
473 };

480 // Show address field
481 ZmRecipients.prototype._showAddressField =
482 function(type, show, skipNotify, skipFocus) {
483 	this._using[type] = show;
484 	Dwt.setVisible(this._divEl[type], show);
485 	this._setAddrFieldValue(type, "");	 // bug fix #750 and #3680
486 	this._field[type].noTab = !show;
487 	var setting = ZmRecipients.ADDR_SETTING[type];
488 	if (setting) {
489 		appCtxt.set(setting, show, null, false, skipNotify);
490 	}
491 	if ((type == AjxEmailAddress.BCC) && this._toggleBccEl) {
492 		Dwt.setInnerHtml(this._toggleBccEl, show ? ZmMsg.hideBCC : ZmMsg.showBCC );
493 	}
494 	if (this._resetContainerSize) {
495 		this._resetContainerSize();
496 	}
497 };

507 ZmRecipients.prototype._acKeyupHandler =
508 function(ev, acListView, result, element) {
509 	var key = DwtKeyEvent.getCharCode(ev);
510 	// process any printable character or enter/backspace/delete keys
511 	if (result && element && (ev.inputLengthChanged ||
512 		(key == 3 || key == 13 || key == 8 || key == 46 ||
513 		(AjxEnv.isMac && key == 224)))) // bug fix #24670
514 	{
515 		element.value = element.value && element.value.replace(/;([^\s])/g, function(all, group){return "; "+group}) || ""; // Change ";" to "; " if it is not succeeded by a whitespace
516 		this._adjustAddrHeight(element);
517 	}
518 };

520 ZmRecipients.prototype._adjustAddrHeight =
521 function(textarea, skipResetBodySize) {
522 
523 	if (this._useAcAddrBubbles || !textarea) { return; }
524 
525 	if (textarea.value.length == 0) {
526 		textarea.style.height = "21px";
527 
528 		if (AjxEnv.isIE) {
529 			// for IE use overflow-y
530 			textarea.style.overflowY = "hidden";
531 		} else {
532 			textarea.style.overflow = "hidden";
533 		}
534 
535 		if (!skipResetBodySize && this._resetContainerSize) {
536 			this._resetContainerSize();
537 		}
538 
539 		return;
540 	}
541 
542 	var sh = textarea.scrollHeight;
543 	if (sh > textarea.clientHeight) {
544 		var taHeight = parseInt(textarea.style.height) || 0;
545 		if (taHeight <= 65) {
546 			if (sh >= 65) {
547 				sh = 65;
548 				if (AjxEnv.isIE)
549 					textarea.style.overflowY = "scroll";
550 				else
551 					textarea.style.overflow = "auto";
552 			}
553 			textarea.style.height = sh + 13;
554 			if (this._resetContainerSize) {
555 				this._resetContainerSize();
556 			}
557 		} else {
558 			if (AjxEnv.isIE) {
559 				// for IE use overflow-y
560 				textarea.style.overflowY = "scroll";
561 			}
562 			else {
563 				textarea.style.overflow = "auto";
564 			}
565 
566 			textarea.scrollTop = sh;
567 		}
568 	}
569 };

573 /**
574  * a callback that's called when bubbles are added or removed, since we need to resize the msg body in those cases.
575  */
576 ZmRecipients.prototype._bubblesChangedCallback =
577 function() {
578 	if (!this._useAcAddrBubbles) { return; }
579 	if (this._resetContainerSize) {
580 		this._resetContainerSize(); // body size might change due to change in size of address field (due to new bubbles).
581 	}
582 };

584 ZmRecipients.prototype._bubbleMenuCreated =
585 function(addrInput, menu) {
586 
587 	if (!this._useAcAddrBubbles) { return; }
588 
589 	this._bubbleActionMenu = menu;
590     if (this._fieldNames.length > 1) {
591         menu.addOp(ZmOperation.SEP);
592         var listener = new AjxListener(this, this._bubbleMove);
593 
594         for (var i = 0; i < this._fieldNames.length; i++) {
595             var type = this._fieldNames[i];
596             var op = this._bubbleOps[type];
597             menu.addOp(op);
598             menu.addSelectionListener(op, listener);
599         }
600     }
601 };
602 
603 ZmRecipients.prototype._bubbleMenuResetOperations =
604 function(addrInput, menu) {
605 	var sel = addrInput.getSelection();
606     for (var i = 0; i < this._fieldNames.length; i++) {
607         var type = this._fieldNames[i];
608 		var op = this._bubbleOps[type];
609 		menu.enable(op, sel.length > 0 && (type != addrInput.type));
610 	}
611 };
612 
613 ZmRecipients.prototype._bubbleMove =
614 function(ev) {
615 
616 	var sourceInput = ZmAddressInputField.menuContext.addrInput;
617 	var op = ev && ev.item && ev.item.getData(ZmOperation.KEY_ID);
618 	var type = this._opToField[op];
619 	var targetInput = this._addrInputField[type];
620 	if (sourceInput && targetInput) {
621 		var sel = sourceInput.getSelection();
622 		if (sel.length) {
623 			for (var i = 0; i < sel.length; i++) {
624 				var bubble = sel[i];
625 				this._showAddressField(type, true);
626 				targetInput.addBubble({bubble:bubble});
627 				sourceInput.removeBubble(bubble.id);
628 			}
629 		}
630 	}
631 };

634 ZmRecipients.prototype._setAddrFieldValue =
635 function(type, value) {
636 
637 	if (this._useAcAddrBubbles) {
638 		var addrInput = this._addrInputField[type];
639 		if (addrInput) {
640 			addrInput.setValue(value, true);
641 		}

687 // Transfers addresses from the contact picker to the compose view.
688 ZmRecipients.prototype._contactPickerOkCallback =
689 function(addrs) {
690 
691 	if (this._enableContainerInputs) {
692 		this._enableContainerInputs(true);
693 	}
694 	for (var i = 0; i < this._fieldNames.length; i++) {
695 		var type = this._fieldNames[i];
696 		this.setAddress(type, "");
697         // If there was only one button, the picker will just return the list of selections,
698         // not a list per button type
699         var typeAddrs = (this._fieldNames.length == 1) ? addrs :  addrs[type];
700 		var addrVec = this._expandAddrs(typeAddrs);
701 		this.addAddresses(type, addrVec);
702 	}
703 
704 	//I still need this here since REMOVING stuff with the picker does not call removeBubble in the ZmAddresInputField.
705 	//Also - it's better to do it once than for every bubble in this case. user might add many addresses with the picker
706 	this._bubblesChangedCallback();
707 
708 	if (this._contactPopdownListener) {
709 		this._contactPicker.removePopdownListener(this._contactPopdownListener);
710 	}
711 	this._contactPicker.popdown();
712 	if (this._reenter) {
713 		this._reenter();
714 	}
715 };

717 // Expands any addresses that are groups
718 ZmRecipients.prototype._expandAddrs =
719 function(addrs) {
720 	var addrsNew = [];
721 	var addrsArray = (addrs instanceof AjxVector) ? addrs.getArray() : addrs;
722 	if (addrsArray && addrsArray.length) {
723 		for (var i = 0; i < addrsArray.length; i++) {
724 			var addr = addrsArray[i];
725 			if (addr) {
726 				if (addr.isGroup && !(addr.__contact && addr.__contact.isDL)) {
727 					var members = addr.__contact ? addr.__contact.getGroupMembers().good.getArray() :
728 												   AjxEmailAddress.split(addr.address);
729 					addrsNew = addrsNew.concat(members);
730 				}
731 				else {
732 					addrsNew.push(addr);
733 				}
734 			}
735 		}
736 	}
737 	return AjxVector.fromArray(addrsNew);
738 };
