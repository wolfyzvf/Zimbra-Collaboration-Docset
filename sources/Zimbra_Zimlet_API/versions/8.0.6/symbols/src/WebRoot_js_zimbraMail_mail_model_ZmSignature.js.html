 16 /**
 17  * Creates a signature.
 18  * @class
 19  * This class represents a signature.
 20  * 
 21  * 
 22  */
 23 ZmSignature = function(id) {
 24 	this.id = id;
 25 };
 26 
 27 ZmSignature.prototype.toString = function() {
 28 	return "ZmSignature";
 29 };
 30

 55 ZmSignature.createFromJson =
 56 function(object) {
 57 	var signature = new ZmSignature(object.id);
 58 	signature.setFromJson(object);
 59 	return signature;
 60 };

 62 //
 63 // Public methods

 65 /**
 66  * Creates the signature.
 67  * 
 68  * @param	{AjxCallback}		callback		the callback
 69  * @param	{AjxCallback}		errorCallback		the error callback
 70  * @param	{ZmBatchCommand}		batchCmd		the batch command
 71  */
 72 ZmSignature.prototype.create =
 73 function(callback, errorCallback, batchCmd) {
 74 	var respCallback = callback ? new AjxCallback(this, this._handleCreateResponse, [callback]) : null;
 75 	var resp = this._sendRequest("CreateSignatureRequest", false, respCallback, errorCallback, batchCmd);
 76 	if (!callback && !batchCmd) {
 77 		this._handleCreateResponse(callback, resp);
 78 	}
 79 };
 80 
 81 /**
 82  * Saves the signature.
 83  * 
 84  * @param	{AjxCallback}		callback		the callback
 85  * @param	{AjxCallback}		errorCallback		the error callback
 86  * @param	{ZmBatchCommand}		batchCmd		the batch command
 87  */
 88 ZmSignature.prototype.save =
 89 function(callback, errorCallback, batchCmd) {
 90 	var respCallback = callback ? new AjxCallback(this, this._handleModifyResponse, [callback]) : null;
 91 	var resp = this._sendRequest("ModifySignatureRequest", false, respCallback, errorCallback, batchCmd);
 92 	if (!callback && !batchCmd) {
 93 		this._handleModifyResponse(callback, resp);
 94 	}
 95 };

 97 /**
 98  * Deletes the signature.
 99  * 
100  * @param	{AjxCallback}		callback		the callback
101  * @param	{AjxCallback}		errorCallback		the error callback
102  * @param	{ZmBatchCommand}		batchCmd		the batch command
103  */
104 ZmSignature.prototype.doDelete =
105 function(callback, errorCallback, batchCmd) {
106 	var respCallback = callback ? new AjxCallback(this, this._handleDeleteResponse, [callback]) : null;
107 	var resp = this._sendRequest("DeleteSignatureRequest", true, respCallback, errorCallback, batchCmd);
108 	if (!callback && !batchCmd) {
109 		this._handleDeleteResponse(callback, resp);
110 	}
111 };

113 /**
114  * Sets the signature from JSON object.
115  * 
116  * @param	{Object}	object		the object
117  */
118 ZmSignature.prototype.setFromJson =
119 function(object) {
120 
121 	this.name = object.name || this.name;
122 	var c = object.content;
123     if (c) {
124 		var sig = c[0]._content ? c[0] : c[1];
125 		this.contentType = sig.type || this.contentType;
126 		this.value = sig._content || this.value;
127     }
128 	if (object.cid) {
129 		this.contactId = object.cid[0]._content;
130 	}
131 };

154 /**
155  * @param outputType	[string]	(Optional) Formats the resulting
156  *									signature text to the specified
157  *									content-type. If not specified,
158  *									the signature text is returned in
159  *									the original format.
160  *
161  * @private
162  */
163 ZmSignature.prototype.getValue =
164 function(outputType) {
165 	
166     var isHtml = this.contentType == ZmMimeTable.TEXT_HTML;
167 	var value = this.value;
168 
169 	var type = outputType || this.contentType;
170 	if (type != this.contentType) {
171         value = isHtml ? AjxStringUtil.convertHtml2Text(value) : AjxStringUtil.convertToHtml(value);
172 	}
173 
174     return value;
175 };

182 ZmSignature.prototype._sendRequest =
183 function(method, idOnly, respCallback, errorCallback, batchCmd) {
184 
185 /*
186 	var jsonObj = {};
187 	var request = jsonObj[method] = {_jsns:"urn:zimbraAccount"};
188 	var sig = request.signature = {};
189 	if (this.id) {
190 		sig.id = this.id;
191 	}
192 	if (!idOnly) {
193 		sig.name = this.name;
194 		sig.cid = this.contactId || null;
195 		sig.content = [];
196 		sig.content.push({_content:this.value, type:this.contentType});
197 
198         // Empty the other content type
199         var emptyType = (this.contentType == ZmMimeTable.TEXT_HTML) ? ZmMimeTable.TEXT_PLAIN : ZmMimeTable.TEXT_HTML;
200 		sig.content.push({_content:"", type:emptyType});
201 	}
202 */
203 
204 	var soapDoc = AjxSoapDoc.create(method, "urn:zimbraAccount");
205 	var signatureEl = soapDoc.set("signature");
206 	if (this.id) {
207 		signatureEl.setAttribute("id", this.id);
208 	}
209 	if (!idOnly) {
210 		signatureEl.setAttribute("name", this.name);
211 		if (this.contactId || (method == "ModifySignatureRequest")) {
212 			soapDoc.set("cid", this.contactId || null, signatureEl);
213 		}
214 		var contentEl = soapDoc.set("content", this.value, signatureEl);
215 		contentEl.setAttribute("type", this.contentType);
216 
217         //Empty the other content type
218         var emptyType = (this.contentType == ZmMimeTable.TEXT_HTML) ? ZmMimeTable.TEXT_PLAIN : ZmMimeTable.TEXT_HTML;
219         contentEl = soapDoc.set("content", "", signatureEl);
220 		contentEl.setAttribute("type", emptyType);
221 
222 	}
223 
224 	if (batchCmd) {
225 		batchCmd.addNewRequestParams(soapDoc, respCallback, errorCallback);
226 		return;
227 	}
228 
229 	var appController = appCtxt.getAppController();
230 	var params = {
231 		soapDoc:		soapDoc,
232 		asyncMode:		Boolean(respCallback),
233 		callback:		respCallback,
234 		errorCallback:	errorCallback
235 	}
236 	return appController.sendRequest(params);
237 };

239 ZmSignature.prototype._handleCreateResponse =
240 function(callback, resp) {
241 	// save id
242 	this.id = resp._data.CreateSignatureResponse.signature[0].id;
243 
244 	// add to global hash
245 	var signatures = appCtxt.getSignatureCollection();
246 	signatures.add(this);
247 
248 	if (callback) {
249 		callback.run();
250 	}
251 };
252 
253 ZmSignature.prototype._handleModifyResponse = function(callback, resp) {
254 	// promote settings to global signature
255 	var signatures = appCtxt.getSignatureCollection();
256 	var signature = signatures.getById(this.id);
257 	signature.name = this.name;
258 	signature.value = this.value;
259     signature.contentType = this.contentType;
260 	signatures._notify(ZmEvent.E_MODIFY, { item: signature });
261 
262 	if (callback) {
263 		callback.run();
264 	}
265 };
266 
267 ZmSignature.prototype._handleDeleteResponse = function(callback, resp) {
268 	// remove from global hash
269 	var signatures = appCtxt.getSignatureCollection();
270 	signatures.remove(this);
271 
272 	if (callback) {
273 		callback.run();
274 	}
275 };
