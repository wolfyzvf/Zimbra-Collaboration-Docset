 16 /**
 17  * Creates a form.
 18  * @class
 19  * This class represents a form.
 20  * 
 21  * @param	{hash}	params		a hash of parameters
 22  * 
 23  * @extends		DwtComposite
 24  * 
 25  * @private
 26  */
 27 DwtForm = function(params) {
 28 	if (arguments.length == 0) return;
 29 	params = Dwt.getParams(arguments, DwtForm.PARAMS);
 30 	params.className = params.className || "DwtForm";
 31 	DwtComposite.apply(this, arguments);
 32 	this.setScrollStyle(DwtControl.SCROLL);
 33 
 34 	// data
 35 	this._tabGroup = new DwtTabGroup(this._htmlElId);
 36 
 37 	// context
 38 	this._context = {
 39 		set: AjxCallback.simpleClosure(this.set, this),
 40 		get: AjxCallback.simpleClosure(this.get, this)
 41 	};
 42 
 43 	// construct form
 44 	this._dirty = {};
 45 	this._ignore = {};
 46 	this._invalid = {};
 47 	this._errorMessages = {};
 48 	this.setModel(params.model);
 49 	this.setForm(params.form);
 50 	this.reset();
 51 };
 52 DwtForm.prototype = new DwtComposite;
 53 DwtForm.prototype.constructor = DwtForm;
 54 
 55 DwtForm.prototype.toString = function() {
 56 	return "DwtForm";
 57 };

 69 /**
 70  * Sets the value.
 71  * 
 72  * @param	{string}	id	the id
 73  * @param	{string}	value		the value
 74  * @param	{boolean}	force		if <code>true</code>, to force update
 75  */
 76 DwtForm.prototype.setValue = function(id, value, force) {
 77 	if (typeof id != "string") id = String(id);
 78 	if (id.match(/\./) || id.match(/\[/)) {
 79 		var parts = id.replace(/\[(\d+)\](\.)?/,".$1$2").split(".");
 80 		var control = this.getControl(parts[0]);
 81 		if (Dwt.instanceOf(control, "DwtForm")) {
 82 			control.setValue(parts.slice(1).join("."), value, force);
 83 		}
 84 		return;
 85 	}
 86 	var item = this._items[id];
 87 	if (!item) return;
 88 	if (!force && value == item.value) return;
 89 	this._setModelValue(id, value);
 90 	this._setControlValue(id, value);
 91 };

 93 /**
 94  * Gets the value.
 95  * 
 96  * @param	{string}	id	the id
 97  * @param	{string}	defaultValue		the default value
 98  * @return	{string}	the value
 99  */
100 DwtForm.prototype.getValue = function(id, defaultValue) {
101 	if (typeof id != "string") id = String(id);
102 	if (id.match(/\./) || id.match(/\[/)) {
103 		var parts = id.replace(/\[(\d+)\](\.)?/,".$1$2").split(".");
104 		var control = this.getControl(parts[0]);
105 		if (Dwt.instanceOf(control, "DwtForm")) {
106 			return control.getValue(parts.slice(1).join("."));
107 		}
108 		return null;
109 	}
110 	var item = this._items[id];
111 	if (!item) return;
112 	if (item.getter) {
113 		return this._call(item.getter) || defaultValue;
114 	}
115 	var value = this._getControlValue(id);
116     if (value == null) value = item.value;
117 
118     //added <|| ""> because ... if value="" than it always returns defaultValue which could be undefined.
119 	return value || defaultValue || "";
120 };

156 DwtForm.prototype.setLabel = function(id, label) {
157 	var item = this._items[id];

163 	if (control.setText) { control.setText(label); return; }

182 DwtForm.prototype.setVisible = function(id, visible) {
183 	// set the form's visibility
184 	if (arguments.length == 1) {
185 		DwtComposite.prototype.setVisible.call(this, arguments[0]);
186 		return;
187 	}
188 	// set control's visibility
189 	var item = this._items[id];
190 	var control = item && item.control;
191 	if (!control) return;
192 	if (control.setVisible) {
193 		control.setVisible(visible);
194 	}
195 	else {
196 		Dwt.setVisible(control, visible);
197 	}
198 	// if there's a corresponding "*_row" element
199 	var el = document.getElementById([this._htmlElId, id, "row"].join("_"));
200 	if (el) {
201 		Dwt.setVisible(el, visible);
202 	}
203 };

205 DwtForm.prototype.isVisible = function(id) {
206 	// this form's visibility
207 	if (arguments.length == 0) {
208 		return DwtComposite.prototype.isVisible.call(this);
209 	}
210 	// control's visibility
211 	var item = this._items[id];
212 	var control = item && item.control;
213 	if (!control) return false;
214 	if (control.getVisible) return control.getVisible();
215 	if (control.isVisible) return control.isVisible();
216 	return  Dwt.getVisible(control);

245  * 
246  * @param	{string}	id 		the id
247  * @return	{boolean}	<code>true</code> if the item is enabled
248  */

250 	// this form enabled?
251 	if (arguments.length == 0) {
252 		return DwtComposite.prototype.isEnabled.call(this);

254 	// the control enabled?
255 	var item = this._items[id];
256 	var control = item && item.control;
257 	if (!control) return false;
258 	if (control.isEnabled) return control.isEnabled();
259 	if (control.getEnabled) return control.getEnabled();
260 	return  !control.disabled;

263 /**
264  * Sets the valid flag.
265  * 
266  * @param	{string}	id 		the id
267  * @param	{boolean}	valid		if <code>true</code>, the item is valid
268  */
269 DwtForm.prototype.setValid = function(id, valid) {
270 	if (typeof(id) == "boolean") {
271 		valid = arguments[0];
272 		for (id in this._items) {
273 			this.setValid(id, valid);
274 		}
275 		return;
276 	}
277 	if (valid) {
278 		delete this._invalid[id]; 
279 	}
280 	else {
281 		this._invalid[id] = true;
282 	}
283 };

285 /**
286  * Checks if the item is valid.
287  * 
288  * @param	{string}	id 		the id
289  * @return	{boolean}	<code>true</code> if the item is valid
290  */
291 DwtForm.prototype.isValid = function(id) {
292 	if (arguments.length == 0 || AjxUtil.isUndefined(id)) {
293 		for (var id in this._invalid) {
294 			return false;
295 		}
296 		return true;
297 	}
298 	return !(id in this._invalid);
299 };

340 DwtForm.prototype.setDirty = function(id, dirty, skipNotify) {
341 	if (typeof id == "boolean") {
342 		dirty = arguments[0];
343 		for (id in this._items) {
344 			this.setDirty(id, dirty, true);
345 		}
346 		if (!skipNotify && this._ondirty) {
347 			this._call(this._ondirty, ["*"]);
348 		}
349 		return;
350 	}
351 	if (dirty) {
352 		this._dirty[id] = true;
353 	}
354 	else {
355 		delete this._dirty[id]; 
356 	}
357 	if (!skipNotify && this._ondirty) {
358 		var item = this._items[id];
359 		if (!item.ignore || !this._call(item.ignore)) {
360 			this._call(this._ondirty, [id]);
361 		}
362 	}
363 };
364 DwtForm.prototype.isDirty = function(id) {
365 	if (arguments.length == 0) {
366 		for (var id in this._dirty) {
367 			var item = this._items[id];
368 			if (item.ignore && this._call(item.ignore)) {
369 				continue;
370 			}
371 			return true;
372 		}
373 		return false;
374 	}
375 	var item = this._items[id];
376 	return item.ignore && this._call(item.ignore) ? false : id in this._dirty;
377 };
378 DwtForm.prototype.getDirtyItems = function() {
379 	// NOTE: This avoids needing a closure
380 	DwtForm.__acceptDirtyItem.form = this;
381 	return AjxUtil.keys(this._dirty, DwtForm.__acceptDirtyItem);
382 };
383 DwtForm.__acceptDirtyItem = function(id) {
384 	var form = arguments.callee.form;
385 	var item = form._items[id];
386 	return !item.ignore || !form._call(item.ignore);
387 };

425 DwtForm.prototype.validate = function(id) {
426 	if (arguments.length == 0) {
427 		this.setValid(true);
428 		for (var id in this._items) {
429 			this._validateItem(id);
430 		}
431 		return this.isValid();
432 	}
433 	return this._validateItem(id);
434 };
435 
436 DwtForm.prototype._validateItem = function(id) {
437 	if (!id) return true;
438 	var item = this._items[id];
439 	if (!item) return true;
440 	try {
441 		var value = this.getValue(id);
442 		var outcome = item.validator ? item.validator(value) : ((item.control && item.control.validator) ? item.control.validator(value) : true);
443 		// the validator may return false to signify that the validation failed (but preferably throw an error with a message)
444 		// it may return true to signify that the field validates
445 		// It also may return a string or hash (truthy) that we may put into the value field (for normalization of data; e.g. if 13/10/2009 is transformed to 1/10/2010 by the validator)
446 		this.setValid(id, Boolean(outcome) || outcome === "");
447 		if (AjxUtil.isString(outcome) || AjxUtil.isObject(outcome)) {
448 			this._setControlValue(id, outcome); // Set display value
449 			item.value = item.setter ? this._call(item.setter, [outcome]) : outcome; // Set model value
450 			var dirty = !Boolean(this._call(item.equals, [item.value,item.ovalue]));
451 			this.setDirty(id, dirty);
452 		}
453 	}
454 	catch (e) {
455 		this.setErrorMessage(id, AjxUtil.isString(e) ? e : e.message);
456 		this.setValid(id, false);
457 	}
458 	return !(id in this._invalid);
459 };

461 DwtForm.prototype.reset = function(useCurrentValues) {
462 	// init state
463 	this._dirty = {};
464 	this._ignore = {};
465 	this._invalid = {};
466 	for (var id in this._items) {
467 		var item = this._items[id];
468 		if (item.control instanceof DwtForm) {
469 			item.control.reset(useCurrentValues);
470 		}
471 		var itemDef = this._items[id].def;
472 		if (!itemDef) continue;
473 		this._initControl(itemDef, useCurrentValues);
474 	}
475 	// update values
476 	this.update();
477 	for (var id in this._items) {
478 		var item = this._items[id];
479 		item.ovalue = item.value;
480 	}
481 	// clear state
482 	this.validate();
483     this.setDirty(false);
484 	// call handler
485 	if (this._onreset) {
486 		this._call(this._onreset);
487 	}
488 };
489 
490 DwtForm.prototype.update = function() {
491 	// update all the values first
492 	for (var id in this._items) {
493 		var item = this._items[id];
494 		if (item.control instanceof DwtForm) {
495 			item.control.update();
496 		}
497 		if (item.getter) {
498 			this.setValue(id, this._call(item.getter));
499 		}
500 	}
501 	// now set visible/enabled/ignore based on values
502 	for (var id in this._items) {
503 		var item = this._items[id];
504 		if (item.visible) {
505 			this.setVisible(id, Boolean(this._call(item.visible)));
506 		}
507 		if (item.enabled) {
508 			this.setEnabled(id, Boolean(this._call(item.enabled)));
509 		}
510 		if (item.ignore) {
511 			this.setIgnore(id, Boolean(this._call(item.ignore)));
512 		}
513 	}
514 	// call handler
515 	if (this._onupdate) {
516 		this._call(this._onupdate);
517 	}
518 };

524 DwtForm.prototype._setModelValue = function(id, value) {
525 	var item = this._items[id];
526 	item.value = item.setter ? this._call(item.setter, [value]) : value;
527 	var dirty = !Boolean(this._call(item.equals, [item.value,item.ovalue]));
528 	this.setDirty(id, dirty);
529 	this.validate(id);
530 	return dirty;
531 };
532 
533 DwtForm.prototype._setControlValue = function(id, value) {
534 	var control = this._items[id].control;
535 	if (control) {
536 		// TODO: display value
537 		if (control instanceof DwtCheckbox || control instanceof DwtRadioButton) {
538 			control.setSelected(value);
539 			return;
540 		}
541 		if (control instanceof DwtMenuItem && control.isStyle(DwtMenuItem.CHECK_STYLE)) {
542 			control.setChecked(value, true);
543 			return;
544 		}
545 		if (control.setSelectedValue) { control.setSelectedValue(value); return; }
546 		if (control.setValue) { control.setValue(value); return; }
547 		if (control.setText && !(control instanceof DwtButton)) { control.setText(value); return; }
548 		if (!(control instanceof DwtControl)) {
549 			// TODO: support other native form elements like select
550 			if (control.type == "checkbox" || control == "radio") {
551 				control.checked = value;
552 			}
553 			else {
554 				// TODO: handle error setting form input value
555 				control.value = value;
556 			}
557 			return;
558 		}
559 	}
560 };
561 DwtForm.prototype._getControlValue = function(id) {
562 	var control = this._items[id].control;
563 	if (control) {
564 		if (control instanceof DwtCheckbox || control instanceof DwtRadioButton) {
565 			return control.isSelected();
566 		}
567 		if (control.getSelectedValue) return control.getSelectedValue();
568 		if (control.getValue) return control.getValue();
569 		if (control.getText && !(control instanceof DwtButton)) return control.getText();
570 		if (!(control instanceof DwtControl)) {
571 			if (control.type == "checkbox" || control == "radio") return control.checked;
572 			return control.value;
573 		}
574 	}
575 };

597 	this._createHtmlFromTemplate(templateId || this.TEMPLATE, { id: this._htmlElId });

600 DwtForm.prototype._createHtmlFromTemplate = function(templateId, data) {
601 	DwtComposite.prototype._createHtmlFromTemplate.apply(this, arguments);
602 
603 	// initialize state
604 	var tabIndexes = [];
605 	this._items = {};
606 	this._tabGroup.removeAllMembers();
607 	this._onupdate = null;
608 	this._onreset = null;
609 	this._ondirty = null;
610 
611 	// create form
612 	var form = this.form;
613 	if (form && form.items) {
614 		// create controls
615 		this._registerControls(form.items, null, tabIndexes);
616 		// create handlers
617 		this._onupdate = DwtForm.__makeFunc(form.onupdate);
618 		this._onreset = DwtForm.__makeFunc(form.onreset);
619 		this._ondirty = DwtForm.__makeFunc(form.ondirty);
620 	}
621 
622 	// add links to list of tabIndexes
623 	var links = this.getHtmlElement().getElementsByTagName("A");
624 	for (var i = 0; i < links.length; i++) {
625 		var link = links[i];
626 		if (!link.href || link.getAttribute("notab") == "true") continue;
627         var controlId = link.id && link.id.substr(this.getHTMLElId().length+1);
628 		if (this._items[controlId]) continue;
629 		tabIndexes.push({
630 			tabindex:	link.getAttribute("tabindex") || Number.MAX_VALUE,
631 			control:	link
632 		});
633 	}
634 
635 	// add controls to tab group
636 	tabIndexes.sort(DwtForm.__byTabIndex);
637 	for (var i = 0; i < tabIndexes.length; i++) {
638 		var control = tabIndexes[i].control;
639 		var member = (control.getTabGroupMember && control.getTabGroupMember()) || control;
640 		this._tabGroup.addMember(member);
641 	}
642 };

644 DwtForm.prototype._registerControls = function(itemDefs, parentDef,
645                                                tabIndexes, params,
646                                                parent, defaultType) {
647 	for (var i = 0; i < itemDefs.length; i++) {
648 		this._registerControl(itemDefs[i], parentDef, tabIndexes, params, parent, defaultType);
649 	}
650 };

652 DwtForm.prototype._registerControl = function(itemDef, parentDef,
653                                               tabIndexes, params,
654                                               parent, defaultType) {
655 	// create item entry
656 	var id = itemDef.id || [this._htmlElId, Dwt.getNextId()].join("_");
657 	var item = this._items[id] = {
658 		id:			id, // for convenience
659 		def:		itemDef,
660 		parentDef:	parentDef,
661 		equals:		DwtForm.__makeFunc(itemDef.equals) || DwtForm.__equals,
662 		getter:		DwtForm.__makeGetter(itemDef),
663 		setter:		DwtForm.__makeSetter(itemDef),
664 		value:		itemDef.value,
665 		visible:	DwtForm.__makeFunc(itemDef.visible),
666 		enabled:	DwtForm.__makeFunc(itemDef.enabled),
667 		validator:	DwtForm.__makeFunc(itemDef.validator),
668 		ignore:		DwtForm.__makeFunc(itemDef.ignore),
669 		control:	itemDef.control
670 	};
671 	// NOTE: This is used internally for indexing of rows
672 	if (itemDef.aka) {
673 		this._items[id].aka = itemDef.aka;
674 		this._items[itemDef.aka] = item;
675 	}
676 
677 	// is control already created?
678 	var control = item.control;
679 	if (control) {
680 		return control;
681 	}
682 
683 	// create control
684 	parent = parent || this;

1341 	if (!itemDef) return;
1342 
1343 	if (index == null) index = this._rowCount;
1344 
1345 	// move other rows "up"
1346 	for (var i = this._rowCount - 1; i >= index; i--) {

1368 		rowsEl.insertBefore(rowEl, rowsEl.childNodes[index]);
1369 	}
1370 
1371 	// create controls
1372 	var tabIndexes = [];
1373 	var control = this._registerControl(itemDef, null, tabIndexes);
1374

1429 DwtFormRows.prototype.removeRow = function(indexOrId) {
1430 	if (this._rowCount <= this._minRows) {
1431 		return;
1432 	}
1433 
1434 	// delete item at specified index
1435 	var item = this._items[indexOrId];
1436 	if (item.control instanceof DwtControl) {
1437 		this.removeChild(item.control);
1438 	}
1439 	delete this._items[item.aka];
1440 	this._deleteItem(item.id);
1441 
1442 	// delete add item
1443 	var addItem = this._items[item._addId];
1444 	if (addItem) {
1445 		this.removeChild(addItem.control);
1446 		this._deleteItem(addItem.id);
1447 	}
1448 
1449 	// delete remove item

1451 	if (removeItem) {
1452 		this.removeChild(removeItem.control);
1453 		this._deleteItem(removeItem.id);
1454 	}
1455 
1456 	// shift everything down one, removing old last row
1457 	var fromIndex = Number(item.aka);
1458 	for (var i = fromIndex + 1; i < this._rowCount; i++) {
1459 		var oindex = i, nindex = i-1;
1460 		this._items[nindex] = this._items[oindex];
1461 		this._items[nindex].aka = String(nindex);
1462 		this._setControlIds(this._items[nindex].id, this._items[nindex].aka);
1463 	}
1464 	this._deleteItem(String(--this._rowCount));
1465 
1466 	// remove row element
1467 	var rowEl = item._rowEl;
1468 	rowEl.parentNode.removeChild(rowEl);
1469 	delete item._rowEl;

1473 	this._rowsTabGroup.removeMember(tabGroup);
1474 
1475 	// update display and notify handler
1476 	this.update();

1479 	}
