 17  * 
 18  * 
 19  * @private
 20  */
 21 DwtXFormDialog = function(xformDef, xmodelDef, parent, className, title, standardButtons, extraButtons, zIndex, mode, loc) {
 22 	if (arguments.length == 0) return;
 23 	className = className || "DwtXFormDialog";
 24 	DwtDialog.call(this, parent, className, title, standardButtons, extraButtons, zIndex, mode, loc);
 25 	 
 26 	this._xform = new XForm(xformDef, new XModel(xmodelDef), null, this);
 27 	this._xform.addListener(DwtEvent.XFORMS_FORM_DIRTY_CHANGE, new AjxListener(this, this._handleXFormDirty));
 28 	
 29 	this.setView(this._xform);
 30 
 31 	if (this._button[DwtDialog.OK_BUTTON]) {	
 32 		this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 33 	}
 34 	if (this._button[DwtDialog.CANCEL_BUTTON]) {	
 35 		this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._handleCancelButton));
 36 	}
 37 	if (this._button[DwtDialog.YES_BUTTON]) {	
 38 		this.setButtonListener(DwtDialog.YES_BUTTON, new AjxListener(this, this._handleYesButton));
 39 	}
 40 	if (this._button[DwtDialog.NO_BUTTON]) {	
 41 		this.setButtonListener(DwtDialog.NO_BUTTON, new AjxListener(this, this._handleNoButton));
 42 	}
 43 }
 44 DwtXFormDialog.prototype = new DwtDialog;
 45 DwtXFormDialog.prototype.constructor = DwtXFormDialog;
 46 
 47 // Data
 48 
 49 DwtXFormDialog.prototype._xform;
 50 DwtXFormDialog.prototype._xformInitialized = false;
 51 
 52 // Public methods
 53 
 54 DwtXFormDialog.prototype.setInstance = function(instance) { 
 55 	this._xform.setInstance(instance);
 56 }
 57 DwtXFormDialog.prototype.getInstance = function() {
 58 	return this._xform.getInstance();
 59 }
 60 
 61 DwtXFormDialog.prototype.popup = function(loc) {
 62 	this._initDialog();
 63 	
 64 	// make sure that form represents current data and show
 65 	this._xform.setIsDirty(true);
 66 	this._xform.refresh();
 67 	if (this._button[DwtDialog.OK_BUTTON]) {
 68 		this.setButtonEnabled(DwtDialog.OK_BUTTON, false);
 69 	}
 70 	if (this._button[DwtDialog.YES_BUTTON]) {
 71 		this.setButtonEnabled(DwtDialog.YES_BUTTON, false);
 72 	}
 73 	DwtDialog.prototype.popup.call(this, loc);
 74 }
 75 
 76 // Protected methods
 77 
 78 DwtXFormDialog.prototype._initDialog = function() {
 79 	// initialize form
 80 	if (!this._xformInitialized) {
 81 		this._xform.draw();
 82 		this._xformInitialized = true;
 83 	}
 84 }
 85 
 86 DwtXFormDialog.prototype._handleXFormDirty = function(event) {
 87 	if (this._button[DwtDialog.OK_BUTTON]) {
 88 		this.setButtonEnabled(DwtDialog.OK_BUTTON, true);
 89 	}
 90 	if (this._button[DwtDialog.YES_BUTTON]) {
 91 		this.setButtonEnabled(DwtDialog.YES_BUTTON, true);
 92 	}
 93 }
 94 
 95 DwtXFormDialog.prototype._handleOkButton = function(event) {
 96 	this.popdown();
 97 	this.setInstance(null);
 98 }
 99 DwtXFormDialog.prototype._handleCancelButton = DwtXFormDialog.prototype._handleOkButton;
100 
101 DwtXFormDialog.prototype._handleYesButton = DwtXFormDialog.prototype._handleOkButton;
102 DwtXFormDialog.prototype._handleNoButton = DwtXFormDialog.prototype._handleOkButton;
