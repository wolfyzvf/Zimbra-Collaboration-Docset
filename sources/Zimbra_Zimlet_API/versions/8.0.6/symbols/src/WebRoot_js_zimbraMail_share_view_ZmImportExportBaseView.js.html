 52 ZmImportExportBaseView.prototype.update = function() {
 53 	DwtForm.prototype.update.apply(this, arguments);
 54 
 55 	// update type hint
 56 	var type = this.getValue("TYPE", ZmImportExportController.TYPE_TGZ);
 57 	this.setValue("TYPE_HINT", this.TYPE_HINTS[type]);
 58 };
 59 
 60 ZmImportExportBaseView.prototype.setValue = function(name, value) {
 61     var ovalue = this.getValue(name);
 62 	DwtForm.prototype.setValue.apply(this, arguments);
 63 	if (name == "TYPE" && value != ovalue) {
 64 		var type = value;
 65 		this._initSubType(type);
 66 		var isTgz = type == ZmImportExportController.TYPE_TGZ;
 67 		this.setEnabled("ADVANCED", isTgz);
 68 		if (this.getValue("ADVANCED") && !isTgz) {
 69 			this.setValue("ADVANCED", false);
 70 			this.update();
 71 		}
 72 		var folder;
 73 		switch (type) {
 74 			case ZmImportExportController.TYPE_CSV: {
 75 				// TODO: Does this work for child accounts w/ fully-qualified ids?
 76 				folder = appCtxt.getById(ZmOrganizer.ID_ADDRBOOK);
 77 				break;
 78 			}
 79 			case ZmImportExportController.TYPE_ICS: {
 80 				folder = appCtxt.getById(ZmOrganizer.ID_CALENDAR);
 81 				break;
 82 			}
 83 			case ZmImportExportController.TYPE_TGZ: {
 84 				folder = appCtxt.getById(ZmOrganizer.ID_ROOT);
 85 				break;
 86 			}
 87 		}
 88 		this._setFolderButton(folder);
 89 	}
 90 };

 98 ZmImportExportBaseView.prototype._initSubType = function(type) {
 99 	var select = this.getControl("SUBTYPE");
100 	if (!select) return;
101 
102 	var options = this._getSubTypeOptions(type);
103 	if (!options || options.length == 0) return;
104 
105 	select.clearOptions();
106 	for (var i = 0; i < options.length; i++) {
107 		select.addOption(options[i]);
108 	}
109 	select.setSelectedValue(options[0].value);
110 };
111 
112 ZmImportExportBaseView.prototype._getSubTypeOptions = function(type) {
113 	if (!ZmImportExportBaseView.prototype.TGZ_OPTIONS) {
114 		ZmImportExportBaseView.prototype.TGZ_OPTIONS = [
115 			{ displayValue: ZmMsg["zimbra-tgz"],			value: "zimbra-tgz" }
116 		];
117 		ZmImportExportBaseView.prototype.CSV_OPTIONS = [];
118 		var formats = appCtxt.get(ZmSetting.AVAILABLE_CSVFORMATS);
119 		for (var i = 0; i < formats.length; i++) {
120 			var format = formats[i];
121 			if (format) {
122 				ZmImportExportBaseView.prototype.CSV_OPTIONS.push(
123 					{ displayValue: ZmMsg[format] || format, value: format }
124 				);
125 			}
126 		}
127 		ZmImportExportBaseView.prototype.ICS_OPTIONS = [
128 			{ displayValue: ZmMsg["zimbra-ics"],			value: "zimbra-ics" }
129 		];
130 	}
131 	var options;
132 	switch (type) {
133 		case ZmImportExportController.TYPE_TGZ: {
134 			options = this.TGZ_OPTIONS;
135 			break;
136 		}
137 		case ZmImportExportController.TYPE_CSV: {
138 			options = this.CSV_OPTIONS;
139 			break;
140 		}
141 		case ZmImportExportController.TYPE_ICS: {
142 			options = this.ICS_OPTIONS;
143 			break;
144 		}
145 	}
146 	return options;
147 };

157 ZmImportExportBaseView.prototype._folderButton_onclick = function() {
158 	// init state
159 	if (!this._handleFolderDialogOkCallback) {
160 		this._handleFolderDialogOkCallback = new AjxCallback(this, this._handleFolderDialogOk);
161 	}
162 
163 	if (!this._TREES) {
164 		this._TREES = {};
165 		this._TREES[ZmImportExportController.TYPE_TGZ] = [];
166         for (var org in ZmOrganizer.VIEWS) {
167             if (org == ZmId.APP_VOICE){
168                 continue;  //Skip voice folders for import/export (Bug: 72269)
169             }
170             var settingId = ZmApp.SETTING[ZmOrganizer.APP2ORGANIZER_R[org]];
171             if (settingId == null || appCtxt.get(settingId)) {
172                 this._TREES[ZmImportExportController.TYPE_TGZ].push(org);
173             }
174         }
175 		this._TREES[ZmImportExportController.TYPE_CSV] = appCtxt.get(ZmSetting.CONTACTS_ENABLED) ? [ZmOrganizer.ADDRBOOK] : [];
176 		this._TREES[ZmImportExportController.TYPE_ICS] = appCtxt.get(ZmSetting.CALENDAR_ENABLED) ? [ZmOrganizer.CALENDAR] : [];
177 	}
178 
179 	// pop-up dialog
180 	var dialog = appCtxt.getChooseFolderDialog();
181 	dialog.registerCallback(DwtDialog.OK_BUTTON, this._handleFolderDialogOkCallback);
182 	var type = this.getValue("TYPE") || ZmImportExportController.TYPE_TGZ;
183 	var acctName = appCtxt.multiAccounts ? appCtxt.getActiveAccount().name : "";
184 	var params = {
185 		treeIds:		this._TREES[type],
186 		overviewId:		dialog.getOverviewId([this.toString(), type, acctName].join("_")),
187 		description:	"",
188 		skipReadOnly:	true,
189 		omit:			{},
190 		forceSingle:	true,
191 		hideNewButton:	(this instanceof ZmExportView)
192 	};
193 	params.omit[ZmOrganizer.ID_TRASH] = true;
194 	dialog.popup(params);
195 };
196 
197 ZmImportExportBaseView.prototype._handleFolderDialogOk = function(folder) {
198 	appCtxt.getChooseFolderDialog().popdown();
199 	this._setFolderButton(folder);
200 	return true;
201 };
202 
203 ZmImportExportBaseView.prototype._setFolderButton = function(folder) {
204 	// NOTE: Selecting a header is the same as "all folders"
205 	this._folderId = folder ? folder.id : -1;
206 	if (folder) {
207 		var isRoot = folder.nId == ZmOrganizer.ID_ROOT;
208 		this.setLabel("FOLDER_BUTTON", isRoot ? ZmMsg.allFolders : AjxStringUtil.htmlEncode(folder.name));
209 	}
210 	else {
211 		this.setLabel("FOLDER_BUTTON", ZmMsg.browse);
212 	}
213 };

242 ZmImportExportDataTypes.prototype.setValue = function(value) {
243 	// NOTE: Special case "" as *all* types -- this will include things
244 	//       like conversations, etc, that are implicitly added if no
245 	//       data types are specified for import/export.
246 	if (value == "") {
247 		var children = this.getChildren();
248 		for (var i = 0; i < children.length; i++) {
249 			var checkbox = children[i];
250 			checkbox.setSelected(true);
251 		}
252 		return;
253 	}
254 
255 	// return only those types that are checked
256 	var types = value ? value.split(",") : [];
257 	var type = {};
258 	for (var i = 0; i < types.length; i++) {
259 		type[types[i]] = true;
260 	}
261 	var children = this.getChildren();
262 	for (var i = 0; i < children.length; i++) {
263 		var checkbox = children[i];
264 		var selected = true;
265 		var types = checkbox.getValue().split(",");
266 		for (var j = 0; j < types.length; j++) {
267 			if (!type[types[j]]) {
268 				selected = false;
269 				break;
270 			}
271 		}
272 		checkbox.setSelected(selected);
273 	}
274 };
275 
276 ZmImportExportDataTypes.prototype.getValue = function() {
277 	// NOTE: Special case "" as *all* types. 
278 	if (this.isAllSelected()) {
279 		return "";
280 	}
281 
282 	var types = [];
283 	var children = this.getChildren();
284 	for (var i = 0; i < children.length; i++) {
285 		var checkbox = children[i];
286 		if (checkbox.isSelected()) {
287 			types.push(checkbox.getValue());
288 		}
289 	}
290 	return types.join(",");
291 };

293 /**
294  * Checks if "all" is selected.
295  * 
296  * @return	{Boolean}	<code>true</code> if all is selected
297  */
298 ZmImportExportDataTypes.prototype.isAllSelected = function() {
299 	var children = this.getChildren();
300 	for (var i = 0; i < children.length; i++) {
301 		var checkbox = children[i];
302 		if (!checkbox.isSelected()) {
303 			return false;
304 		}
305 	}
306 	return true;
307 };

320 ZmImportExportDataTypes.prototype._createHtml = function(templateId) {
321 	this._createHtmlFromTemplate(templateId || this.TEMPLATE, {id:this._htmlElId});
322 };
323 
324 ZmImportExportDataTypes.prototype._createHtmlFromTemplate =
325 function(templateId, data) {
326 	// get number of checkboxes
327 	data.count = 0;
328 	for (var appName in ZmApp.ORGANIZER) {
329 		var orgType = ZmApp.ORGANIZER[appName];
330 		var views = ZmOrganizer.VIEWS[orgType];
331 		if (!views || views.length == 0) continue;
332 		data.count++;
333 	}
334 
335 	// create cells
336 	DwtComposite.prototype._createHtmlFromTemplate.call(this, templateId, data);
337 
338 	// create checkboxes and place in cells
339 	var i = 0;
340 	for (var appName in ZmApp.ORGANIZER) {
341 		var orgType = ZmApp.ORGANIZER[appName];
342 		var views = ZmOrganizer.VIEWS[orgType];
343 		if (!views || views.length == 0) continue;
344 
345 		var checkbox = new ZmImportExportDataTypeCheckbox({parent:this,checked:true});
346 		checkbox.setImage(ZmApp.ICON[appName]);
347 		checkbox.setText(ZmMsg[ZmApp.NAME[appName]] || ZmApp.NAME[appName] || appName);
348 		// NOTE: I know it's the default join string but I prefer
349 		//       explicit behavior.
350 		checkbox.setValue(views.join(","));
351 		checkbox.replaceElement(data.id+"_cell_"+i);
352 
353 		this._tabGroup.addMember(checkbox);
354 
355 		i++;
356 	}
357 };
