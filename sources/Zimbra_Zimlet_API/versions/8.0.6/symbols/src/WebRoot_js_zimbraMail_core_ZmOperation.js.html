 22 /**
 23  * Creates the operation object.
 24  * @class
 25  * This class provides the idea of an "operation", which is a user-initiated action
 26  * exposed through a button or menu item. Many operations (such as Delete) are shared
 27  * across applications/controllers. An operation gets defined by specifying its name,
 28  * tool tip, and image. Then controllers can simply select which operations they'd like
 29  * to support.
 30  * <br/>
 31  * <br/>
 32  * The two primary clients of this class are {@link ZmButtonToolBar} and {@link ZmActionMenu}. Clients 
 33  * should support {@link #createOp} and {@link #getOp} methods. See the two aforementioned clients for
 34  * examples.
 35  * 
 36  * @author Conrad Damon
 37  */
 38 ZmOperation = function() {};
 39 
 40 // Special operations
 41 ZmOperation.NONE 					= "NONE";		// no operations or menu items
 42 ZmOperation.SEP 					= "SEP";		// separator
 43 ZmOperation.SPACER 					= "SPACER";		// spacer (toolbar)
 44 ZmOperation.FILLER 					= "FILLER";		// filler (toolbar)
 45 
 46 // suffix for disabled image
 47 ZmOperation.DIS = "Dis";
 48 
 49 // text and icons displayed for operation
 50 ZmOperation.SETUP = {};
 51 
 52 // special-purpose operations
 53 ZmOperation.SETUP[ZmOperation.NONE]		= {};	// means no operations (as opposed to a default set)
 54 ZmOperation.SETUP[ZmOperation.SEP]		= {};	// a thin vertical or horizontal bar
 55 ZmOperation.SETUP[ZmOperation.SPACER]	= {};	// empty space of a given size
 56 ZmOperation.SETUP[ZmOperation.FILLER]	= {};	// expandable space (for right-align in toolbars)
 57 
 58 // preconditions for operations - no automatic checking is done, so a client
 59 // of this class has to check them on its own if it wants
 60 ZmOperation.SETTING = {};
 61 
 62 // code to run after an operation has been created - typically used to create
 63 // a menu for a button
 64 ZmOperation.CALLBACK = {};
 65 
 66 /**
 67  * Defines the aspects of an operation, and the ID that refers to it.
 68  * 
 69  * @param {String}	op		the name of the operation
 70  * @param {String}	text		the msg key for button or menu item text
 71  * @param {String}	tooltip	the msg key for tooltip text
 72  * @param {String}	image		the icon class for the button or menu item
 73  * @param {String}	disImage	the disabled version of image; defaults to image + "Dis"
 74  * @param {constant}	setting	the setting which acts as a precondition for this operation
 75  * @param {AjxCallback}	callback	the callback to run after this operation has been created
 76  */
 77 ZmOperation.registerOp =
 78 function(op, params, setting, callback) {
 79 	ZmOperation[op] = op;
 80 	ZmOperation.SETUP[op] = params || {};
 81 	if (setting)	{ ZmOperation.SETTING[op]	= setting; }
 82 	if (callback)	{ ZmOperation.CALLBACK[op]	= callback; }
 83 };

214 /**
215  * Creates operation descriptors for the given operation IDs,
216  * then creates the appropriate widget for each operation based on the type of
217  * the parent. If it's a toolbar, then buttons are created. If it's a menu, menu items are
218  * created.
219  * <p>
220  * To override or add properties to a particular operation, pass in a hash of properties and
221  * values as a value in overrides, with the operation ID as the key.
222  * </p>
223  *
224  * @param {DwtComposite}	parent		the containing widget (toolbar or menu)
225  * @param {Array}	operations	a list of operation IDs
226  * @param {Hash}	overrides		a hash of overrides by op ID
227  *
228  * @returns	{Hash}	a hash of operations by ID
229  */
230 ZmOperation.createOperations =
231 function(parent, operations, overrides) {
232 	var obj = new ZmOperation();
233 	return obj._createOperations(parent, operations, overrides);
234 }
235 
236 /**
237  * Done through an object so that we can have more than one invocation going
238  * without sharing memory (eg, creating New submenu).
239  * 
240  * @private
241  */
242 ZmOperation.prototype._createOperations =
243 function(parent, operations, overrides) {
244 	if (operations == ZmOperation.NONE) {
245 		operations = null;
246 	}
247 	overrides = overrides || {};
248 
249 	var opHash = {};
250 	if (operations && operations.length) {
251 		for (var i = 0; i < operations.length; i++) {
252 			var id = operations[i];
253 			ZmOperation.defineOperation(id, overrides[id]);
254 			ZmOperation.addOperation(parent, id, opHash, null, overrides[id] && overrides[id].htmlElId);
255 		}
256 	}
257 
258 	return opHash;
259 };
260 
261 /**
262 * Creates an operation descriptor. The ID of an existing operation can be passed
263 * in to use as a base, with overridden properties passed in a hash. A new operation
264 * can be defined by passing its properties in a hash.
265 *
266 * @param {String}	baseId		the ID of an existing operation
267 * @param {Hash}	overrides	the property overrides for the operation
268 */
269 ZmOperation.defineOperation =
270 function(baseId, overrides) {
271 	var id = (overrides && overrides.id) || (baseId && baseId.id) || baseId || Dwt.getNextId();
272 	var textKey = (overrides && overrides.textKey) || ZmOperation.getProp(baseId, "textKey");
273 	var text = textKey && ZmMsg[textKey];
274 	var tooltipKey = (overrides && overrides.tooltipKey) || ZmOperation.getProp(baseId, "tooltipKey");
275 	var tooltip = tooltipKey && ZmMsg[tooltipKey];
276 	var image = ZmOperation.getProp(baseId, "image");
277 	var showImageInToolbar = ZmOperation.getProp(baseId, "showImageInToolbar");
278 	var showTextInToolbar = ZmOperation.getProp(baseId, "showTextInToolbar");
279 	var disImage = ZmOperation.getProp(baseId, "disImage");
280 	var enabled = (overrides && (overrides.enabled !== false));
281 	var style = ZmOperation.getProp(baseId, "style");
282 	var shortcut = ZmOperation.getProp(baseId, "shortcut");
283 
284     var opDesc = {id:id, text:text, image:image, showImageInToolbar:showImageInToolbar, showTextInToolbar:showTextInToolbar, disImage:disImage, enabled:enabled,
285 				  tooltip:tooltip, style:style, shortcut:shortcut};
286 	if (overrides) {
287 		for (var i in overrides) {
288 			opDesc[i] = overrides[i];
289 		}
290 	}
291 
292 	ZmOperation._operationDesc[id] = opDesc;
293 	
294 	return opDesc;
295 };

297 /**
298  * Gets the value of a given property for a given operation.
299  *
300  * @param {String}	id		the operation ID
301  * @param {String}	prop	the name of an operation property
302  * 
303  * @return	{Object}	the value
304  */
305 ZmOperation.getProp =
306 function(id, prop) {
307 	var value = null;
308 	var setup = ZmOperation.SETUP[id];
309 	if (setup) {
310 		value = setup[prop];
311 		if (!value && (prop == "disImage") && setup.image) {
312 			value = setup.image;
313 		}
314 	}
315 
316 	return value;
317 };

319 /**
320  * Checks if the operation is a separator or spacer.
321  * 
322  * @param	{String}	id		the id
323  * @return	{Boolean}	<code>true</code> if the operation is a spacer
324  */
325 ZmOperation.isSep =
326 function(id) {
327 	return (id == ZmOperation.SEP || id == ZmOperation.SPACER || id == ZmOperation.FILLER);
328 };

330 /**
331  * Adds the operation.
332  * 
333  * @param {DwtComposite}	parent		the containing widget (toolbar or menu)
334  * @param	{String}		id		the id
335  * @param	{Hash}		opHash		a hash
336  * @param	{String}	[index]		the index
337  */
338 ZmOperation.addOperation =
339 function(parent, id, opHash, index, htmlElId) {
340 
341 	var opDesc = ZmOperation._operationDesc[id] || ZmOperation.defineOperation(id);
342 
343 	if (id == ZmOperation.SEP) {
344         if (parent instanceof DwtMenu) {
345             parent.createSeparator(index);
346         }
347         else {
348             parent.addSeparator(null, index);
349         }
350     } else if (id == ZmOperation.SPACER) {	// toolbar only
351 		parent.addSpacer(null, index);
352 	} else if (id == ZmOperation.FILLER) {	// toolbar only
353 		parent.addFiller(null, index);
354 	} else {
355 		if (index != null) {
356 			opDesc.index = index;
357 		}
358 		opHash[id] = parent.createOp(id, opDesc, htmlElId);
359 	}
360 	var callback = ZmOperation.CALLBACK[id];
361 	if (callback) {
362 		if (callback.isAjxCallback) {
363 			callback.run(opHash[id]);
364 		} else {
365 			callback(opHash[id]);
366 		}
367 	}
368 };

370 /**
371  * Adds a deferred menu.
372  * 
373  * @param	{function}	addMenuFunc		the add menu function
374  * @param {DwtComposite}	parent		the containing widget (toolbar or menu)
375  */
376 ZmOperation.addDeferredMenu =
377 function(addMenuFunc, parent /* ... */) {
378     var args = [parent];
379     for (var i = 2; i < arguments.length; i++) {
380         args.push(arguments[i]);
381     }
382 	var callback = new AjxCallback(null, addMenuFunc, args);
383 	parent.setMenu(callback);
384 };

386 /**
387  * Removes the operation.
388  * 
389  * @param {DwtComposite}	parent		the containing widget (toolbar or menu)
390  * @param	{String}	id		the id
391  * @param	{Hash}	opHash		a hash
392  */
393 ZmOperation.removeOperation =
394 function(parent, id, opHash) {
395 	parent.getOp(id).dispose();
396 	delete opHash[id];
397 }

399 /**
400  * Replaces the attributes of one operation with those of another, wholly or in part.
401  *
402  * @param {DwtComposite}	parent		the parent widget
403  * @param {String}	oldOp		the ID of operation to replace
404  * @param {String}	newOp		the ID of new operation to get replacement attributes from
405  * @param {String}	text		the new text (overrides text of newOp)
406  * @param {String}	image		the new image (overrides image of newOp)
407  * @param {String}	disImage	the new disabled image (overrides that of newOp)
408  */
409 ZmOperation.setOperation =
410 function(parent, oldOp, newOp, text, image, disImage) {
411 	var op = parent.getOp(oldOp);
412 	if (!op) return;
413 
414 	op.setText(text ? text : ZmMsg[ZmOperation.getProp(newOp, "textKey")]);
415 	op.setImage(image ? image : ZmOperation.getProp(newOp, "image"));
416 };

418 /**
419  * Takes a list of operations and removes any who have a corresponding setting that's
420  * not set. Also deals with the fact that you don't want a separator or a spacer unless
421  * there's stuff on either side of it.
422  * 
423  * @param	{Array}	list		a list of {ZmOperation} objects
424  * @return	{Array}		a list of {ZmOperation} objects
425  */
426 ZmOperation.filterOperations =
427 function(list) {
428 	var newList = [];
429 	if (!(list && list.length)) { return newList; }
430 	
431 	// remove disabled operations
432 	for (var i = 0; i < list.length; i++) {
433 		var op = list[i];
434 		if (!op) { continue; }
435 		var setting = ZmOperation.SETTING[op];
436 		if (!setting || appCtxt.get(setting)) {
437 			newList.push(op);
438 		}
439 	}
440 	// reduce multiple consecutive separators to the first one
441 	var newList1 = [];
442 	var gotSep = false;
443 	for (var i = 0; i < newList.length; i++) {
444 		var op = newList[i];
445 		if (op == ZmOperation.SEP || op == ZmOperation.SPACER) {
446 			if (!gotSep) {
447 				newList1.push(op);
448 			}
449 			gotSep = true;
450 		} else {
451 			newList1.push(op);
452 			gotSep = false;
453 		}
454 	}
455 	// remove separator at beginning or end
456 	if (newList1 && newList1.length) {
457 		if (newList1[0] == ZmOperation.SEP || newList1[0] == ZmOperation.SPACER) {
458 			newList1.shift();
459 		}
460 		var i = newList1.length - 1;
461 		if (newList1[i] == ZmOperation.SEP || newList1[i] == ZmOperation.SPACER || newList1[i] == ZmOperation.FILLER) {
462 			newList1.pop();
463 		}
464 	}
465 	
466 	return newList1;
467 };

469 /**
470  * Adds a "New" submenu. Overrides are used because we don't want "New" at the
471  * beginning of each label.
472  *
473  * @param {DwtComposite}		parent		the parent widget
474  * @return	{ZmActionMenu}	the menu
475  */
476 ZmOperation.addNewMenu =
477 function(parent) {
478 	var list = ZmOperation.NEW_ITEM_OPS;
479 	list.push(ZmOperation.SEP);
480 	list = list.concat(ZmOperation.NEW_ORG_OPS);
481 
482 	var overrides = {};
483 	for (var i = 0; i < list.length; i++) {
484 		var op = list[i];
485 		var htmlElId = parent._htmlElId + "_" + op;
486 		overrides[op] = {htmlElId: htmlElId};
487 		var textKey = ZmOperation.NEW_ITEM_KEY[op] || ZmOperation.NEW_ORG_KEY[op];
488 		if (textKey) {
489 			overrides[op].textKey = textKey;
490 		}
491 	}
492 
493 	var menu = new ZmActionMenu({parent:parent, menuItems:list, overrides:overrides});
494 	parent.setMenu(menu);
495 
496 	return menu;
497 };

568 /**
569  * Gets the tooltip for the operation with the given ID. If the operation has a shortcut associated
570  * with it, a shortcut hint is appended to the end of the tooltip.
571  *
572  * @param {String}	id		the operation ID
573  * @param {String}	keyMap	the key map (for resolving shortcut)
574  * @param {String}	tooltip	the tooltip override
575  * @return	{String}	the tool tip
576  */
577 ZmOperation.getToolTip =
578 function(id, keyMap, tooltip) {
579 	var opDesc = ZmOperation._operationDesc[id] || ZmOperation.defineOperation(id);
580 	tooltip = tooltip || opDesc.tooltip;
581 	var sc = tooltip && opDesc.shortcut && appCtxt.getShortcutHint(keyMap, opDesc.shortcut);
582 	return sc ? [tooltip, sc].join("") : tooltip;
583 };
