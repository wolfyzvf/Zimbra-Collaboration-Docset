 16 /**
 17  * @overview
 18  * This file contains the command handler class. The following commands are supported:
 19  *
 20  * 		$set:debug {1,2,3}					set debug level to 1, 2, or 3
 21  * 		$set:debug t						toggle debug timing on/off
 22  * 		$set:debugtrace {on,off}			turn offline debug trace on/off
 23  * 		$set:instant_notify					show whether instant notify is on or off
 24  * 		$set:instant_notify {on,off}		turn instant notify on/off
 25  *		$set:poll [N]						set poll interval to N ms (unless doing instant notify)
 26  * 		$set:noop							send a poll (get notifications)
 27  * 		$set:rr								refresh reminders
 28  * 		$set:rh								run reminder housekeeping
 29  * 		$set:toast							show sample toast messages
 30  * 		$set:get version					show client version
 31  * 		$set:expire							expire session; refresh block will come on next response
 32  * 		$set:refresh						force immediate return of a refresh block
 33  * 		$set:relogin						logout the user
 34  * 		$set:alert							issue a test sound alert
 35  * 		$set:alert {browser,desktop,app} N	issue a test alert in given context in N seconds
 36  * 		$set:leak {begin,report,end}		manage leak detection
 37  * 		$set:tabs							show tab groups (in debug window)
 38  * 		$set:ymid [id]						set Yahoo IM user to id
 39  * 		$set:log [type]						dump log contents for type
 40  * 		$set:log [type]	[size]				set number of msgs to keep for type
 41  * 		$set:compose						compose msg based on mailto: in query string
 42  * 		$set:error							show error dialog
 43  * 		$set:modify [setting] [value]		set setting to value, then optionally restart
 44  * 		$set:clearAutocompleteCache			clear contacts autocomplete cache
 45  *
 46  * TODO: should probably I18N the alert messages
 47  */

 59 /**
 60  * Executes the command.
 61  * 
 62  * @param	{String}	cmdStr		the command
 63  * @param	{ZmSearchController}	searchController	the search controller
 64  * @param	{Object}	[cmdArg1]		command arguments
 65  */
 66 ZmClientCmdHandler.prototype.execute =
 67 function(cmdStr, searchController) {
 68 
 69 	if (!cmdStr) { return; }
 70 
 71 	cmdStr = AjxStringUtil.trim(cmdStr, true);
 72 	
 73 	if (cmdStr == "") { return; }
 74 	
 75 	var argv = cmdStr.split(/\s/);
 76 	var arg0 = argv[0];
 77 
 78 	if (arg0 != "modify") {
 79 		cmdStr = cmdStr.toLowerCase();
 80 	}
 81 
 82 	var func = this["execute_" + arg0];
 83 	if (func) {
 84 		var args = [].concat(cmdStr, searchController, argv);
 85 		return func.apply(this, args);
 86 	}
 87 };

 97 /**
 98  * Executes the command with debug.
 99  * 
100  * @param	{String}	cmdStr		the command
101  * @param	{ZmSearchController}	searchController	the search controller
102  * @param	{Object}	[cmdArg1]		command arguments
103  */
104 ZmClientCmdHandler.prototype.execute_debug =
105 function(cmdStr, searchController, cmdName, cmdArg1 /* ..., cmdArgN */) {
106 	if (!cmdArg1 || !this._dbg) { return; }
107 	if (cmdArg1 == "t") {
108 		var on = this._dbg._showTiming;
109 		var newState = on ? "off" : "on";
110 		this._alert("Turning timing info " + newState);
111 		this._dbg.showTiming(!on);
112 	} else {
113 		this._dbg.setDebugLevel(cmdArg1);
114 		this._alert("Setting debug level to: " + cmdArg1);
115 	}
116 };

121  * @param	{String}	cmdStr		the command

123  * @param	{Object}	[cmdArg1]		command arguments
124  */
125 ZmClientCmdHandler.prototype.execute_debugtrace =
126 function(cmdStr, searchController, cmdName, cmdArg1 /* ..., cmdArgN */) {
127 	if (!cmdArg1) return;
128 
129 	var val;
130 	if (cmdArg1 == "on") {
131 		val = true;
132 	} else if (cmdArg1 == "off") {
133 		val = false;
134 	}
135 
136 	if (val != undefined) {
137 		appCtxt.set(ZmSetting.OFFLINE_DEBUG_TRACE, val, null, null, true);
138 		this._alert("Debug trace is " + cmdArg1);
139 	}
140 };

176 	}

287 /**
288  * Executes the alert command.
289  * 
290  * @param	{String}	cmdStr		the command
291  * @param	{ZmSearchController}	searchController	the search controller
292  * @param	{Object}	[cmdArg1]		command arguments
293  * @param	{Object}	[cmdArg2]		command arguments
294  */
295 ZmClientCmdHandler.prototype.execute_alert =
296 function(cmdStr, searchController, cmdName, cmdArg1, cmdArg2 /* ..., cmdArgN */) {
297 	//  $set:alert [sound/browser/app] [delay in seconds]
298 	function doIt() {
299 		if (cmdArg1 == "browser") {
300 			AjxDispatcher.require("Alert");
301 			ZmBrowserAlert.getInstance().start("Alert Test!");
302 		} else if (cmdArg1 == "desktop") {
303 			AjxDispatcher.require("Alert");
304 			ZmDesktopAlert.getInstance().start("Title!", "Message!");
305 		} else if (cmdArg1 == "app") {
306 			appCtxt.getApp(ZmApp.MAIL).startAlert();
307 		} else {
308 			AjxDispatcher.require("Alert");
309 			ZmSoundAlert.getInstance().start();
310 		}
311 	}
312 	setTimeout(doIt, Number(cmdArg2) * 1000);
313 };

420 /**
421  * Executes the modify command.
422  * 
423  * @param	{String}	cmdStr		the command

425  * @param	{Object}	[cmdArg1]		command arguments

429 function(cmdStr, searchController, cmdName, cmdArg1, cmdArg2 /* ..., cmdArgN */) {

444 	dialog.setMessage(ZmMsg.accountChangeRestart, DwtMessageDialog.WARNING_STYLE);
445 	dialog.popup();
446 };

462 /**
463  * Executes the getCharWidth command.
464  *
465  * @param	{String}	cmdStr		the command
466  * @param	{ZmSearchController}	searchController	the search controller
467  * @param	{Object}	[cmdArg1]		command arguments
468  * @param	{Object}	[cmdArg2]		command arguments
469  */
470 ZmClientCmdHandler.prototype.execute_getCharWidth =
471 function(cmdStr, searchController, cmdName, cmdArg1, cmdArg2 /* ..., cmdArgN */) {
472 	var cla = appCtxt.getApp(ZmApp.CONTACTS).getContactList().getArray();
473 	for (var i = 0; i < cla.length; i++) {
474 		ZmClientCmdHandler._testWidth(cla[i]._attrs["firstLast"] || cla[i]._attrs["company"] || "");
475 	}
476 	var text = [];
477 	text.push("Avg char width: " + ZmClientCmdHandler.WIDTH / ZmClientCmdHandler.CHARS);
478 	text.push("Avg bold char width: " + ZmClientCmdHandler.BWIDTH / ZmClientCmdHandler.CHARS);
479 	var w = ZmClientCmdHandler._testWidth(AjxStringUtil.ELLIPSIS);
480 	text.push("Ellipsis width: " + w.w);
481 	w = ZmClientCmdHandler._testWidth(", ");
482 	text.push("Comma + space width: " + w.w);
483 	alert(text.join("\n"));
484 };

486 ZmClientCmdHandler.CHARS = 0;
487 ZmClientCmdHandler.WIDTH = 0;
488 ZmClientCmdHandler.BWIDTH = 0;
489 
490 ZmClientCmdHandler._testWidth =
491 function(str) {
492 	var div = document.createElement("DIV");
493 	div.style.position = Dwt.ABSOLUTE_STYLE;
494 	var shellEl = appCtxt.getShell().getHtmlElement();
495 	shellEl.appendChild(div);
496 	Dwt.setLocation(div, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
497 	div.innerHTML = str;
498 	var size = Dwt.getSize(div);
499 	ZmClientCmdHandler.CHARS += str.length;
500 	ZmClientCmdHandler.WIDTH += size.x;
501 	div.style.fontWeight = "bold";
502 	var bsize = Dwt.getSize(div);
503 	ZmClientCmdHandler.BWIDTH += bsize.x;
504 	shellEl.removeChild(div);
505 	return {w:size.x, bw:bsize.x};
506 };
