188 ZmTasksApp.prototype.handleOp =
189 function(op, params) {
190 	switch (op) {
191 		case ZmOperation.NEW_TASK: {
192             params = params || {};
193 			var loadCallback = new AjxCallback(this, this._handleLoadNewTask, [params]);
194 			AjxDispatcher.require(["TasksCore", "Tasks"], false, loadCallback, null, true);
195 			break;
196 		}
197 		case ZmOperation.NEW_TASK_FOLDER: {
198 			var loadCallback = new AjxCallback(this, this._handleLoadNewTaskFolder);
199 			AjxDispatcher.require(["TasksCore", "Tasks"], false, loadCallback, null, true);
200 			break;
201 		}
202 	}
203 };

208 	AjxDispatcher.run("GetTaskController").show((new ZmTask(null, null, params && params.folderId)));

211 ZmTasksApp.prototype._handleLoadNewTaskFolder =
212 function() {
213 	appCtxt.getAppViewMgr().popView(true, ZmId.VIEW_LOADING);	// pop "Loading..." page
214 	var dialog = appCtxt.getNewTaskFolderDialog();
215 	if (!this._newTaskFolderCb) {

217 	}

221 /**
222  * Checks for the creation of a tasks folder or a mount point to one.
223  *
224  * @param {Hash}	creates		a hash of create notifications
225  * @param	{Boolean}	force	if <code>true</code>, force the create
226  * 
227  */
228 ZmTasksApp.prototype.createNotify =
229 function(creates, force) {
230 	if (!creates["folder"] && !creates["task"] && !creates["link"]) { return; }
231 	if (!force && this._deferNotifications("create", creates)) { return; }
232 
233 	for (var name in creates) {
234 		var list = creates[name];
235 		if (!list) { continue; }
236 
237 		for (var i = 0; i < list.length; i++) {
238 			var create = list[i];
239 			if (appCtxt.cacheGet(create.id)) { continue; }
240 	
241 			if (name == "folder") {
242 				this._handleCreateFolder(create, ZmOrganizer.TASKS);
243 			} else if (name == "link") {
244 				this._handleCreateLink(create, ZmOrganizer.TASKS);
245 			} else if (name == "task") {
246 				// bug fix #29833 - always attempt to process new tasks
247 				var taskList = AjxDispatcher.run("GetTaskListController").getList();
248 				if (taskList) {
249 					taskList.notifyCreate(create);
250 				}
251 			}
252 		}
253 	}
254 };

296 ZmTasksApp.prototype._handleLoadShowSearchResults =
297 function(results, callback, searchResultsController) {
298 	var folderId = results && results.search && results.search.isSimple() && results.search.folderId;
299 	var sessionId = searchResultsController ? searchResultsController.getCurrentViewId() : ZmApp.MAIN_SESSION;
300 	var controller = AjxDispatcher.run("GetTaskListController", sessionId, searchResultsController);
301 	controller.show(results, folderId);
302 	this._setLoadedTime(this.toString(), new Date());
303 	if (callback) {
304 		callback.run(controller);
305 	}
306 };

378 /**
379  * Performs a search.
380  * 
381  * @param	{ZmFolder}		folder		the folder
382  * @param	{Date}			startDate	the start date
383  * @param	{Date}			endDate		the end date
384  * @param	{AjxCallback}	callback	the callback
385  * @param	{String}		accountName	the account name
386  */
387 ZmTasksApp.prototype.search =
388 function(folder, startDate, endDate, callback, accountName) {
389     var query = folder ? folder.createQuery() : "";
390     query = query || (appCtxt.isExternalAccount() ? "inid:" + this.getDefaultFolderId() : "in:tasks");
391 
392 	var params = {
393 		query:			query,
394 		types:			[ZmItem.TASK],
395 		limit:			this.getLimit(),
396 		searchFor:		ZmItem.TASK,
397 		callback:		callback,
398 		accountName:	(accountName || (folder && folder.getAccount().name))
399 	};
400 	var sc = appCtxt.getSearchController();
401 	sc.searchAllAccounts = false;
402 	sc.search(params);
403 };

416 /**
417  * Gets the list of checked calendar ids. If calendar packages are not loaded,
418  * gets the list from deferred folder ids.
419  *
420  * @param	{Boolean}		localOnly	if <code>true</code>, use local calendar only
421  * @return	{Array}	an array of ids
422  */
423 ZmTasksApp.prototype.getTaskFolderIds =
424 function(localOnly) {
425 	var folderIds = [];
426 	if (AjxDispatcher.loaded("TasksCore")) {
427 		folderIds = AjxDispatcher.run("GetTaskListController").getTaskFolderIds(localOnly);
428 	} else {
429 		// will be used in reminder dialog
430 		this._folderNames = {};
431 		for (var i = 0; i < this._deferredFolders.length; i++) {
432 			var params = this._deferredFolders[i];
433 			//var str = (params && params.obj && params.obj.f) ? params.obj.f : "";
434 			//if (str && (str.indexOf(ZmOrganizer.FLAG_CHECKED) != -1)) {
435 				if (localOnly && params.obj.zid != null) {
436 					continue;
437 				}
438 				folderIds.push(params.obj.id);
439 				// _folderNames are used when deferred folders are not created
440 				// and calendar name is required. example: calendar name
441 				// requirement in reminder module
442 				this._folderNames[params.obj.id] = params.obj.name;
443 			//}
444 		}
445 	}
446 	return folderIds;
447 };

479 /**
480  * Creates a new button with a reminder options as its menu.
481  *
482  * @param	{DwtComposite}	parent						the parent
483  * @param	{String}	buttonId 					the button id to fetch inside DOM and append DwtButton to
484  * @param	{AjxListener}	buttonListener			the listener to call when date button is pressed
485  * @param	{AjxListener}	menuSelectionListener	the listener to call when date is selected in {@link DwtCalendar}
486  */
487 ZmTasksApp.createpCompleteButton =
488 function(parent, buttonId, buttonListener, menuSelectionListener) {
489 	// create button
490 	var pCompleteButton = new DwtButton({parent:parent});
491 	pCompleteButton.addDropDownSelectionListener(buttonListener);
492 	pCompleteButton.setData(Dwt.KEY_ID, buttonId);
493 	pCompleteButton.setSize("25");
494 
495 	// create menu for button
496 	var pCompleteMenu = new DwtMenu({parent:pCompleteButton, style:DwtMenu.DROPDOWN_STYLE});
497 	pCompleteMenu.setSize("100");
498 	pCompleteButton.setMenu(pCompleteMenu, true);
499 
500     var formatter = new AjxMessageFormat(AjxMsg.percentageString);
501 	for (var i = 0; i <= 100; i += ZmTask.PCOMPLETE_INT) {
502 		var mi = new DwtMenuItem({parent: pCompleteMenu, style: DwtMenuItem.NO_STYLE});
503 		mi.setText((formatter.format(i)));
504 		mi.setData("value", i);
505 		if(menuSelectionListener) mi.addSelectionListener(menuSelectionListener);
506 	}
507     
508 	// reparent and cleanup
509 	pCompleteButton.reparentHtmlElement(buttonId);
510 	delete buttonId;
511 
512 	return pCompleteButton;
513 };
