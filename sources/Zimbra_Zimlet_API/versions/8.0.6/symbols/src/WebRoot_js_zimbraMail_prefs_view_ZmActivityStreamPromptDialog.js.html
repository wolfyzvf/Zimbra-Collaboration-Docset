 16 ZmActivityStreamPromptDialog = function() {
 17 	var extraButtons = new DwtDialog_ButtonDescriptor(ZmActivityStreamPromptDialog.ADD_ADVANCED_BUTTON, ZmMsg.advanced, DwtDialog.ALIGN_LEFT);
 18 	DwtDialog.call(this, {parent:appCtxt.getShell(), className:"ZmActivityStreamPromptDialog", title:"Activity Stream Rules",
 19 						  standardButtons:[DwtDialog.OK_BUTTON, DwtDialog.CANCEL_BUTTON], extraButtons: [extraButtons]});	
 20 	// set content
 21 	this.setContent(this._contentHtml());
 22 	this._initialize();
 23 	
 24 	var okButton = this.getButton(DwtDialog.OK_BUTTON);
 25 	okButton.setText(ZmMsg.save);
 26 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._saveListener));
 27 	
 28 	var advancedButton = this.getButton(ZmActivityStreamPromptDialog.ADD_ADVANCED_BUTTON);
 29 	this.setButtonListener(ZmActivityStreamPromptDialog.ADD_ADVANCED_BUTTON, new AjxListener(this, this._advancedListener));
 30 };
 31 
 32 ZmActivityStreamPromptDialog.prototype = new DwtDialog;
 33 ZmActivityStreamPromptDialog.prototype.constructor = ZmActivityStreamPromptDialog;
 34 ZmActivityStreamPromptDialog.ADD_ADVANCED_BUTTON = ++DwtDialog.LAST_BUTTON;

 36 ZmActivityStreamPromptDialog.prototype._contentHtml = 
 37 function() {   
 38 	return "<div style='width: 400px' id='ACTIVITYSTREAM_PROMPT_FORM'>" + ZmMsg.activityStreamPrompt + "</div>";				
 39 };

 41 ZmActivityStreamPromptDialog.prototype._initialize = 
 42 function() {
 43 	var params = {};
 44 	params.parent = this;
 45 	params.template = "prefs.Pages#ActivityStreamPrompt";
 46 	params.form = {
 47 		items: [
 48 			{ id: "SENTTO", type: "DwtCheckbox", label: ZmMsg.to + ":", value: "to"},
 49 			{ id: "TO", type: "DwtInputField", value: "", cols: 30},
 50 			{ id: "RECEIVED", type: "DwtCheckbox", label: ZmMsg.receivedFrom, value: "received"},
 51 			{ id: "FROM", type: "DwtInputField", value: "", cols: 30},
 52 			{ id: "SUBJECT", type: "DwtCheckbox", label: ZmMsg.subjectContains, value: "subject"},
 53 			{ id: "CONTAINS", type: "DwtInputField", value: "", cols: 30}
 54 		]
 55 	};
 56 	this._activityStreamForm = new DwtForm(params);
 57 	var activityStreamForm = document.getElementById("ACTIVITYSTREAM_PROMPT_FORM");
 58 	this._activityStreamForm.appendElement(activityStreamForm);
 59 	this._activityStreamForm.getControl("SUBJECT").setSelected(false);
 60 };

 62 ZmActivityStreamPromptDialog.prototype._handleResponseLoadRules = 
 63 function() {
 64 	this._activityRule = this._rules.getRuleByName(ZmMsg.activityStreamsRule);	
 65 };
 66 
 67 /**
 68  * Checks to see if new condition is being added before popping up dialog
 69  * @param skip {Boolean} true to skip new condition check
 70  */
 71 ZmActivityStreamPromptDialog.prototype.popup = 
 72 function(skip) {
 73 	this._rules = AjxDispatcher.run("GetFilterRules");
 74 	var callback = new AjxCallback(this, this._handleResponseLoadRules);
 75 	this._rules.loadRules(true, callback); // make sure rules are loaded (for when we save)
 76 	if (skip || this._isNewCondition(this._getActivityStreamRule())) {	
 77 		DwtDialog.prototype.popup.call(this);
 78 	}
 79 };

 81 /**
 82  * sets form fields
 83  * @param item  {ZmMailMsg} mail message
 84  */
 85 ZmActivityStreamPromptDialog.prototype.setFields = 
 86 function(item) {
 87 	this._subject = item.subject;
 88 	var msg = item.type == ZmId.ITEM_CONV ? item.getFirstHotMsg() : item;
 89 	if (msg) {
 90 		this._from = msg.getMsgSender();
 91 	}
 92 	else if (item.participants) {
 93 		var arr = item.participants.getArray();
 94 		for (var i=0; i<arr.length; i++) {
 95 			if (arr[i].getType() == "FROM") {
 96 				this._from = arr[i].getAddress();
 97 			}
 98 		}
 99 	}
100 
101 
102 	var arr = msg._addrs && msg._addrs["TO"] && msg._addrs["TO"].getArray();
103     this._to = (arr.length == 1) ? arr[0].getAddress() : "";
104 
105 	if (this._subject) {
106 		var subjectField = this._activityStreamForm.getControl("CONTAINS");
107 		subjectField.setValue(this._subject);
108 	}
109 	
110 	if (this._from) {
111 		var fromField = this._activityStreamForm.getControl("FROM");
112 		fromField.setValue(this._from);
113 	}
114     var toField = this._activityStreamForm.getControl("TO");
115 	toField.setValue(this._to);
116 
117 };

119 ZmActivityStreamPromptDialog.prototype._saveListener =
120 function() {
121 	var foundCondition = this._setConditions(this._activityRule);		
122 	if (foundCondition) {
123 		this._rules.saveRules(null, true);
124 	}
125 	this.popdown();
126 };
127 
128 ZmActivityStreamPromptDialog.prototype._setConditions = 
129 function(rule) {
130 	var received = this._activityStreamForm.getControl("RECEIVED");
131     var sentto = this._activityStreamForm.getControl("SENTTO");
132 	var subject = this._activityStreamForm.getControl("SUBJECT");
133 	var foundCondition = false;
134 	
135 	if (received && received.isSelected() && rule) {
136 		var from = this._activityStreamForm.getControl("FROM");
137 		if (from) {
138 			rule.addCondition(ZmFilterRule.TEST_ADDRESS, ZmFilterRule.OP_CONTAINS, from.getValue(), ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_FROM]);
139 			foundCondition = true;
140 		}
141 	}
142 
143     if (sentto && sentto.isSelected() && rule) {
144 		var to = this._activityStreamForm.getControl("TO");
145 		if (to) {
146 			rule.addCondition(ZmFilterRule.TEST_ADDRESS, ZmFilterRule.OP_CONTAINS, to.getValue(), ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_TO]);
147 			foundCondition = true;
148 		}
149 	}
150 		
151 	if (subject && subject.isSelected() && rule) {
152 		var contains = this._activityStreamForm.getControl("CONTAINS");
153 		if (contains) {
154 			rule.addCondition(ZmFilterRule.TEST_HEADER, ZmFilterRule.OP_CONTAINS, contains.getValue(), ZmFilterRule.C_HEADER_VALUE[ZmFilterRule.C_SUBJECT]);
155 			foundCondition = true;
156 		}
157 	}
158 	
159 	return foundCondition;
160 };

162 ZmActivityStreamPromptDialog.prototype._advancedListener =
163 function() {
164 	this.popdown(); //popdown existing 
165 	var filterRuleDialog = appCtxt.getFilterRuleDialog();
166 	this._setConditions(this._activityRule);
167 	filterRuleDialog.popup(this._activityRule, true);
168 };

170 /**
171  * Determine if user has already created an activity stream condition with subject or email value.
172  * @param  activityRule {ZmFilterRule} the activity stream rule to determine if condition already exists
173  * @return {boolean} true this is a new condition or false condition with subject or email exists
174  */
175 ZmActivityStreamPromptDialog.prototype._isNewCondition =
176 function(activityRule) {
177 	var newCondition = activityRule ? true : false;   //if we don't have an activity rule don't prompt user
178 	var conditionData = {};
179 	var header = "";
180 	var contains = -1;
181 	if (this._subject && activityRule) {
182 		var headerTest = activityRule.conditions[ZmFilterRule.TEST_HEADER] || [];
183 		for (var i=0; i<headerTest.length && newCondition; i++) {
184 			conditionData = headerTest[i];
185 			header = conditionData.header;
186 			contains = conditionData.value ? this._subject.indexOf(conditionData.value) : -1;		
187 			newCondition = !(header == ZmFilterRule.C_HEADER_VALUE[ZmFilterRule.C_SUBJECT] && contains != -1);    
188 		}
189 	}
190 	
191 	if (this._from && activityRule && newCondition) {
192 		var addressTest = activityRule.conditions[ZmFilterRule.TEST_ADDRESS] || [];
193 		for (var i=0; i<addressTest.length && newCondition; i++) {
194 			conditionData = addressTest[i];
195 			header = conditionData.header;
196 			contains = conditionData.value ? this._from.indexOf(conditionData.value) : -1;
197 			newCondition = !(header == ZmFilterRule.C_FROM.toLowerCase() && contains != -1); 
198 		}
199 	}
200 	
201 	return newCondition;
202 };

204 ZmActivityStreamPromptDialog.prototype._getActivityStreamRule = 
205 function() {
206 	return this._activityStreamRule || this._rules.getRuleByName(ZmMsg.activityStreamsRule);
