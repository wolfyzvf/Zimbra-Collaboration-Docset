 20 /**
 21  * Creates a prompt dialog.
 22  * @class
 23  * This class represents a prompt dialog.
 24  * 
 25  * @param	{Hash}	args	a hash of arguments
 26  * @param	{Array}	args.buttons		an array of buttons (default is [{@link DwtDialog.OK_BUTTON}, {@link DwtDialog.CANCEL_BUTTON}])
 27  * @param	{String}	args.password	the password
 28  *  
 29  * @extends		ZmDialog
 30  * 
 31  * @see		ZmPromptDialog.getInstance
 32  * @see		ZmPromptDialog.getPasswordInstance
 33  */
 34 ZmPromptDialog = function(args) {
 35 	args.buttons = args.buttons || [DwtDialog.OK_BUTTON, DwtDialog.CANCEL_BUTTON];
 36 	this._password = args.password;
 37 	ZmDialog.call(this, args);
 38 
 39 	this._labelFieldId = this._htmlElId + "_label";
 40 	this._nameFieldId = this._htmlElId + "_name";
 41 	this._setNameField(this._nameFieldId);
 42 };
 43 
 44 ZmPromptDialog.prototype = new ZmDialog;
 45 ZmPromptDialog.prototype.constructor = ZmPromptDialog;
 46 
 47 ZmPromptDialog.prototype.toString =
 48 function() {
 49 	return "ZmPromptDialog";
 50 };
 51 
 52 /**
 53  * Gets an instance of the prompt dialog.
 54  * 
 55  * @return	{ZmPromptDialog}		the dialog
 56  */
 57 ZmPromptDialog.getInstance =
 58 function() {
 59 	if (!ZmPromptDialog._INSTANCE) {
 60 		ZmPromptDialog._INSTANCE = new ZmPromptDialog({ parent:appCtxt.getShell() });
 61 	}
 62 	return ZmPromptDialog._INSTANCE;
 63 };
 64 
 65 /**
 66  * Gets an instance of the prompt dialog.
 67  * 
 68  * @return	{ZmPromptDialog}	the dialog
 69  */
 70 ZmPromptDialog.getPasswordInstance =
 71 function() {
 72 	if (!ZmPromptDialog._PASSWORD_INSTANCE) {
 73 		ZmPromptDialog._PASSWORD_INSTANCE = new ZmPromptDialog({ parent:appCtxt.getShell(), password: true });
 74 	}
 75 	return ZmPromptDialog._PASSWORD_INSTANCE;
 76 };
 77 
 78 /**
 79  * Pops-up the dialog.
 80  *  
 81  * @param {Hash}	params				a hash of parameters
 82  * @param {String}      params.title					the dialog box title
 83  * @param {String}	params.label					the label next to the dialog's input field
 84  * @param {String}	params.value					the initial value of input field
 85  * @param {AjxCallback}	params.callback			the callback to run when ok button is pressed
 86  */
 87 ZmPromptDialog.prototype.popup =
 88 function(params) {
 89 	this.setTitle(params.title);
 90 	Dwt.byId(this._labelFieldId).innerHTML = params.label;
 91 	var nameElement = Dwt.byId(this._nameFieldId);
 92 	nameElement.innerHTML = params.value || "";
 93 	this._resetCallbacks();
 94 	this.registerCallback(DwtDialog.OK_BUTTON, params.callback);	
 95 	DwtDialog.prototype.popup.call(this);
 96 	if (nameElement.focus) {
 97 		nameElement.focus();
 98 	}
 99 };
100 
101 ZmPromptDialog.prototype._contentHtml =
102 function() {
103 	return AjxTemplate.expand("share.Dialogs#ZmPromptDialog", { id: this._htmlElId, type: this._password ? "password" : "text" });
104 };
105 
106 ZmPromptDialog.prototype._okButtonListener =
107 function(ev) {
108 	var results = this._getPromptData();
109 	if (results)
110 		DwtDialog.prototype._buttonListener.call(this, ev, results);
111 };
112 
113 ZmPromptDialog.prototype._getPromptData =
114 function() {
115 	return {
116 		value: AjxStringUtil.trim(this._nameField.value),
117 		dialog: this
118 	};
119 };
120 
121 ZmPromptDialog.prototype._enterListener =
122 function(ev) {
123 	var results = this._getPromptData();
124 	if (results)
125 		this._runEnterCallback(results);
126 };
127 
128 ZmPromptDialog.prototype._getTabGroupMembers =
129 function() {
130 	return [this._nameField, this._colorButton];
131 };
