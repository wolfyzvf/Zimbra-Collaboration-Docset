582 /**
583  * Checks for the creation of a calendar or a mount point to one, or an
584  * appointment.
585  *
586  * @param {Hash}	creates	a hash of create notifications
587  * 
588  * @private
589  */
590 ZmCalendarApp.prototype.createNotify =
591 function(creates, force) {
592 	if (!creates["folder"] && !creates["appt"] && !creates["link"]) { return; }
593 	if (!force && !this._noDefer && this._deferNotifications("create", creates)) { return; }
594 
595 	var ctlr = AjxDispatcher.run("GetCalController");
596 	for (var name in creates) {
597 		var list = creates[name];
598 		for (var i = 0; i < list.length; i++) {
599 			var create = list[i];
600 			if (appCtxt.cacheGet(create.id)) { continue; }
601 
602 			if (name == "folder") {
603 				this._handleCreateFolder(create, ZmOrganizer.CALENDAR);
604 			} else if (name == "link") {
605 				this._handleCreateLink(create, ZmOrganizer.CALENDAR);
606 			} else if (name == "appt") {
607 				ctlr.notifyCreate(create);
608 			}
609 
610 			if ((name == "folder" || name == "link") && ctlr) {
611 				ctlr._updateCheckedCalendars();
612 			}
613 		}
614 	}
615 };

696 ZmCalendarApp.prototype._handleLoadLaunch =
697 function(params, callback) {
698 	var cc = AjxDispatcher.run("GetCalController");
699 	var view = cc.getDefaultViewType();
700 	var sd = null;
701 
702 	params = params || {};
703 	if (params.qsParams) {
704 		var viewArg = params.qsParams.view;
705 		if (viewArg) {
706 			var viewId = ZmCalendarApp.VIEW_FOR_SETTING[viewArg];
707 			if (viewId) {
708 				view = viewId;
709 				var date = params.qsParams.date;
710 				if (date) {
711 					date = AjxDateUtil.parseServerDateTime(date);
712 					if (date && !isNaN(date)) {
713 						sd = new Date((date).setHours(0,0,0,0));
714 					}
715 				}
716 			}
717 		}
718 	}
719 
720 	if (appCtxt.get(ZmSetting.CONTACTS_ENABLED)) {
721 		this.initResources();
722 	}
723 
724 	cc.show(view, sd);
725 	this._setLoadedTime(this.toString(), new Date());
726 	if (callback) {
727 		callback.run();
728 	}
729 };

1185 /**
1186  * Parses the given string and return reminder info containing units and exact value
1187  *
1188  * @param reminderString 	reminder string eg. "20 minutes before"
1189  * 
1190  * @private
1191  */
1192 ZmCalendarApp.parseReminderString =
1193 function(reminderString) {
1194 	var reminderFormats = {};
1195 	reminderFormats[ZmMsg.apptRemindNDaysBefore]	= ZmCalItem.REMINDER_UNIT_DAYS;
1196 	reminderFormats[ZmMsg.apptRemindNMinutesBefore]	= ZmCalItem.REMINDER_UNIT_MINUTES;
1197 	reminderFormats[ZmMsg.apptRemindNHoursBefore]	= ZmCalItem.REMINDER_UNIT_HOURS;
1198 	reminderFormats[ZmMsg.apptRemindNWeeksBefore]	= ZmCalItem.REMINDER_UNIT_WEEKS;
1199 
1200 	reminderString = AjxStringUtil.trim(reminderString);
1201 	var formattedString = reminderString;
1202 	var reminderValue = formattedString.replace(/\D/g, "");
1203 	reminderValue = AjxStringUtil.trim(reminderValue);
1204 
1205 	// junk content returns empty reminder (None)
1206 	if (reminderValue == "") {
1207 		return {
1208 			reminderValue: "",
1209 			reminderUnits: ZmCalItem.REMINDER_NONE
1210 		};
1211 	}
1212 	if (reminderValue.indexOf(" ") >= 0) {
1213 		reminderValue = reminderValue.split(" ")[0];
1214 	}
1215 
1216 	// look for standard reminder formats strings
1217 	for (var pattern in  reminderFormats) {
1218 		var formattedContent = ZmCalendarApp.__formatLabel(pattern, reminderValue);
1219 		if(formattedContent != "" && formattedContent.toLowerCase() == reminderString.toLowerCase()) {
1220             //Fix for bug: 80651 - set and return object to determine before snooze
1221 			return  {reminderValue: reminderValue, reminderUnits: reminderFormats[pattern], before: true};
1222 		}
1223 	}
1224 
1225 	var reminderHours = parseInt(reminderValue);
1226 
1227     // parse the reminder string for singular units like minute, hour, day etc
1228 	var remUnitStrings = {};
1229 	remUnitStrings[ZmCalItem.REMINDER_UNIT_MINUTES] = AjxMsg.minute;
1230 	remUnitStrings[ZmCalItem.REMINDER_UNIT_HOURS] = AjxMsg.hour;
1231 	remUnitStrings[ZmCalItem.REMINDER_UNIT_DAYS] = AjxMsg.day;
1232 	remUnitStrings[ZmCalItem.REMINDER_UNIT_WEEKS] = AjxMsg.week;
1233 
1234 	//look for matching units
1235     // default unit is hours
1236 	var reminderUnits = ZmCalItem.REMINDER_UNIT_HOURS;
1237 
1238 	for(var i in remUnitStrings) {
1239 		if(formattedString.indexOf(remUnitStrings[i]) >= 0) {
1240 			reminderUnits = i;
1241             return {reminderValue: reminderHours ? reminderHours : 0,  reminderUnits: reminderUnits};
1242 		}
1243 	}
1244 
1245     // parse the reminder string for plural units like minutes, hours, days etc
1246     var remUnitPluralStrings = {};
1247     remUnitPluralStrings[ZmCalItem.REMINDER_UNIT_MINUTES] = AjxMsg.minutes;
1248     remUnitPluralStrings[ZmCalItem.REMINDER_UNIT_HOURS] = AjxMsg.hours;
1249     remUnitPluralStrings[ZmCalItem.REMINDER_UNIT_DAYS] = AjxMsg.days;
1250     remUnitPluralStrings[ZmCalItem.REMINDER_UNIT_WEEKS] = AjxMsg.weeks;
1251 
1252     for(var i in remUnitPluralStrings) {
1253         if(formattedString.indexOf(remUnitPluralStrings[i]) >= 0) {
1254             reminderUnits = i;
1255             return {reminderValue: reminderHours ? reminderHours : 0,  reminderUnits: reminderUnits};
1256         }
1257     }
1258     // fall back to hours if no matching string found
1259 	return {reminderValue: reminderHours ? reminderHours : 0,  reminderUnits: reminderUnits};
1260 };
