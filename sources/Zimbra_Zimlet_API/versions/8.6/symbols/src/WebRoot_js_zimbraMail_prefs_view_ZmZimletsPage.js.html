 22 /**
 23  * Creates a Zimlets preference page.
 24  * @constructor
 25  * @class ZmZimletsPage
 26  * This class represents a page that allows the user to enable/disable available
 27  * zimlets. User can see all the simlets those are enabled by admin for his account.
 28  * Out of these available zimlets user can choose some or all for his account.
 29  *
 30  * @author Rajendra Patil
 31  *
 32  * @param {DwtControl}	parent			the containing widget
 33  * @param {Object}	section			the page
 34  * @param {ZmPrefController}	controller		the prefs controller
 35  * 
 36  * @extends		ZmPreferencesPage
 37  * @private
 38  */
 39 ZmZimletsPage = function(parent, section, controller) {
 40 	ZmPreferencesPage.call(this, parent, section, controller);
 41 };

 72 ZmZimletsPage.prototype.reset =
 73 function(){
 74 	var arr = this.getZimletsArray();

 77 	}
 78 	this.showMe();

 81 ZmZimletsPage.prototype.showMe =
 82 function(deferred){
 83 	ZmPreferencesPage.prototype.showMe.call(this);
 84 	var zimlets = this.getZimlets();
 85 	if (zimlets.size() === 0) {
 86 		this._zimlets = null; //otherwise it would stay cached and would not update when this method is calledback.
 87 		if (!deferred) {
 88 			appCtxt.getAppController().addListener(ZmAppEvent.POST_STARTUP, new AjxListener(this, this.showMe, [true]));
 89 		}
 90 		return;
 91 	}
 92 	if (this._listView) {
 93 		var s = this._listView.getSelection();
 94 		this._listView.set(zimlets.clone());
 95 		if (s && s[0]) {
 96 			this._listView.setSelection(s[0]);
 97 		}
 98 	}
 99 };

154 ZmZimletsPage.prototype._handleZimletUpload =
155 function(status, aid) {
156 	if (status == 200) {
157 		this._deployZimlet(aid);
158 	}
159 	else {
160 		var msg = (status == AjxPost.SC_NO_CONTENT)
161 			? ZmMsg.zimletUploadError
162 			: (AjxMessageFormat.format(ZmMsg.zimletUploadStatus, status));
163         appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_CRITICAL);
164 
165 		this._uploadButton.setEnabled(true);
166 	}
167 };

169 ZmZimletsPage.prototype._getAdminAuth =
170 function(callback) {
171 	// first, make sure we have a valid admin password (parsed from location)
172 	var pword;
173 	var searches = document.location.search.split("&");
174 	for (var i = 0; i < searches.length; i++) {
175 		var idx = searches[i].indexOf("at=");
176 		if (idx != -1) {
177 			pword = searches[i].substring(idx+3);
178 			break;
179 		}
180 	}
181 
182 	// for dev build
183 	if (!pword) {
184 		pword = "@install.key@";
185 	}
186 
187 	var soapDoc = AjxSoapDoc.create("AuthRequest", "urn:zimbraAdmin");
188 	soapDoc.set("name", "zimbra");
189 	soapDoc.set("password", pword);
190 
191 	var params = {
192 		soapDoc: soapDoc,
193 		noSession: true,
194 		asyncMode: true,
195 		noAuthToken: true,
196 		skipAuthCheck: true,
197 		serverUri: ZmZimletsPage.ADMIN_URI,
198 		callback: (new AjxCallback(this, this._handleResponseAuthenticate, [callback]))
199 	};
200 	(new ZmCsfeCommand()).invoke(params);
201 };

203 ZmZimletsPage.prototype._handleResponseAuthenticate =
204 function(callback, result) {
205 	if (result.isException()) {
206 		this._handleZimletDeployError();
207 		return;
208 	}
209 
210 	// set the admin auth cookie (expires when browser closes)
211 	this._adminAuthToken = result.getResponse().Body.AuthResponse.authToken[0]._content;
212 	AjxCookie.setCookie(document, ZmZimletsPage.ADMIN_COOKIE_NAME, this._adminAuthToken, null, "/service/upload");
213 
214 	if (callback) {
215 		callback.run();
216 	}
217 };

219 ZmZimletsPage.prototype._deployZimlet =
220 function(aid, action) {
221 	var dialog = appCtxt.getCancelMsgDialog();
222 	dialog.reset();
223 	dialog.setMessage(ZmMsg.zimletDeploying, DwtMessageDialog.INFO_STYLE);
224 	dialog.registerCallback(DwtDialog.CANCEL_BUTTON, new AjxCallback(this, this._handleZimletCancel, [dialog]));
225 	dialog.popup();
226 
227 	var soapDoc = AjxSoapDoc.create("DeployZimletRequest", "urn:zimbraAdmin");
228 	var method = soapDoc.getMethod();
229 	method.setAttribute("action", (action || "deployLocal"));
230 	method.setAttribute("flush", "1");
231 	method.setAttribute("synchronous", "1");
232 	var content = soapDoc.set("content");
233 	content.setAttribute("aid", aid);
234 
235 	var params = {
236 		soapDoc: soapDoc,
237 		asyncMode: true,
238 		skipAuthCheck: true,
239 		callback: (new AjxCallback(this, this._deployZimletResponse, [dialog, aid])),
240 		serverUri: ZmZimletsPage.ADMIN_URI,
241 		authToken: this._adminAuthToken
242 	};
243 
244 	(new ZmCsfeCommand()).invoke(params);
245 };

247 ZmZimletsPage.prototype._deployZimletResponse =
248 function(dialog, aid, result) {
249     
250     // remove Admin auth key
251      AjxCookie.deleteCookie(document, ZmZimletsPage.ADMIN_COOKIE_NAME, "/service/upload");
252 
253 	if (result.isException()) {
254 		dialog.popdown();
255 		this._uploadButton.setEnabled(true);
256 
257 		this._controller.popupErrorDialog(ZmMsg.zimletDeployError, result.getException());
258 		return;
259 	}
260 
261 	var status = result.getResponse().Body.DeployZimletResponse.progress[0].status;
262 	if (status == ZmZimletsPage.ZIMLET_UPLOAD_STATUS_PENDING) {
263 		if (this._checkStatusCount++ > ZmZimletsPage.ZIMLET_MAX_CHECK_STATUS) {
264 			this._handleZimletDeployError();
265 		} else {
266 			AjxTimedAction.scheduleAction(new AjxTimedAction(this, this._deployZimlet, [aid, "status"]), 1500);
267 		}
268 	} else {
269 		dialog.popdown();
270 
271 		if (status == ZmZimletsPage.ZIMLET_UPLOAD_STATUS_FAIL) {
272 			this._handleZimletDeployError();
273 		}
274 		else {
275 			this._uploadButton.setEnabled(true);
276 
277 			var settings = appCtxt.getSettings(appCtxt.accountList.mainAccount);
278 			settings._showConfirmDialog(ZmMsg.zimletDeploySuccess, settings._refreshBrowserCallback.bind(settings), DwtMessageDialog.INFO_STYLE);
279 		}
280 	}
281 };

321 ZmZimletsPage.prototype._undeployZimletResponse =
322 function(zimletName, result) {
323 
324     // remove admin auth key
325     AjxCookie.deleteCookie(document, ZmZimletsPage.ADMIN_COOKIE_NAME, "/service/upload");
326     
327 	if (result.isException()) {
328 		this._controller.popupErrorDialog(ZmMsg.zimletUndeployError, result.getException());
329 		return;
330 	}
331 
332 	// remove the uninstalled zimlet from the listview
333     var zimletsCtxt = this.getZimletsCtxt();
334 	var zimlet = zimletsCtxt.getPrefZimletByName(zimletName);
335 	if (zimlet) {
336 		zimletsCtxt.removePrefZimlet(zimlet);
337 		this._listView.set(zimletsCtxt.getZimlets().clone());
338 	}
339 
340 	// prompt user to restart client
341 	var settings = appCtxt.getSettings(appCtxt.accountList.mainAccount);
342 	settings._showConfirmDialog(ZmMsg.zimletUndeploySuccess, settings._refreshBrowserCallback.bind(settings), DwtMessageDialog.INFO_STYLE);
343 };

345 ZmZimletsPage.prototype.addCommand  =
346 function(batchCommand) {
347 	var soapDoc = AjxSoapDoc.create("ModifyZimletPrefsRequest", "urn:zimbraAccount");
348 	// LDAP supports multi-valued attrs, so don't serialize list
349 	var zimlets = this.getZimletsArray();
350 	var settingsObj = appCtxt.getSettings();
351 	var setting = settingsObj.getSetting(ZmSetting.CHECKED_ZIMLETS);
352 	var checked = [];
353 	for (var i = 0; i < zimlets.length; i++) {
354 		if (zimlets[i].active) {
355 			checked.push(zimlets[i].name);
356 		}
357 		var node = soapDoc.set("zimlet", null);
358 		node.setAttribute("name", zimlets[i].name);
359 		node.setAttribute("presence", (zimlets[i].active ? "enabled" : "disabled"));
360 	}
361 	setting.setValue(checked);
362 	batchCommand.addNewRequestParams(soapDoc);
363 };

394 ZmZimletsPage.prototype.isDirty =
395 function() {
396 	var dirty = false;
397 	var arr = this.getZimletsArray();
398 	var dirtyZimlets = [];
399 
400 	var printZimlet = function(zimlet) {
401 		if (AjxUtil.isArray(zimlet)) {
402 			return AjxUtil.map(zimlet, printZimlet).join("\n");
403 		}
404 		return [zimlet.name," (from ",zimlet._origStatus," to ",zimlet.active,")"].join("");
405 	}
406 
407 	for (var i = 0; i < arr.length; i++) {
408 		if (arr[i]._origStatus != arr[i].active) {
409 			dirty = true;
410 			dirtyZimlets.push(arr[i]);
411 		}
412 	}
413 
414 	if (dirty) {
415 		AjxDebug.println(AjxDebug.PREFS, "Dirty preferences:\n" + "Dirty zimlets:\n" + printZimlet(dirtyZimlets));
416 	}
417 	return dirty;
418 };

446 ZmZimletsPage._getZimlets =
447 function() {
448 	var allz = appCtxt.get(ZmSetting.ZIMLETS) || [];
449 	var zimlets = new ZmPrefZimlets();
450     var zimletsLoaded = appCtxt.getZimletMgr().isLoaded();
451 	for (var i = 0; i <  allz.length; i++) {
452 		var name = allz[i].zimlet[0].name;
453 		if (allz[i].zimletContext[0].presence == "mandatory") {
454 			continue; // skip mandatory zimlets to be shown in prefs
455 		}
456 		var desc = allz[i].zimlet[0].description;
457 		var label = allz[i].zimlet[0].label || name.replace(/^.*_/,"");
458         if (zimletsLoaded) {
459             desc = ZmZimletContext.processMessage(name, desc);
460             label = ZmZimletContext.processMessage(name, label);
461         }
462 		var isEnabled = allz[i].zimletContext[0].presence == "enabled";
463 		zimlets.addPrefZimlet(new ZmPrefZimlet(name, isEnabled, desc, label));
464 	}
465 	zimlets.sortByName();
466 	return zimlets;
467 };

526 ZmPrefZimletListView.prototype._handleZimletsLoaded = function(evt) {
527     this._zimletsLoaded = true;
528     var array = this.parent.getZimletsArray();
529     for (var i = 0; i < array.length; i++) {
530         var item = array[i];
531         var label = item.label || item.name.replace(/^.*_/,"");
532         item.label = ZmZimletContext.processMessage(item.name, label);
533         item.desc = ZmZimletContext.processMessage(item.name, item.desc);
534         this.setCellContents(item, ZmPrefZimletListView.COL_NAME, AjxStringUtil.htmlEncode(item.label));
535         this.setCellContents(item, ZmPrefZimletListView.COL_DESC, AjxStringUtil.htmlEncode(item.desc));
536     }
537 };

580 ZmPrefZimletListView.prototype._getCellContents =
581 function(html, idx, item, field, colIdx, params) {
582 	if (field == ZmPrefZimletListView.COL_ACTIVE) {
583 		html[idx++] = "<input name='checked_zimlets' type='checkbox' ";
584 		html[idx++] = item.active ? "checked " : "";
585 		html[idx++] = "id='";
586 		html[idx++] = item.name;
587 		html[idx++] = "_zimletCheckbox' _name='";
588 		html[idx++] = item.name;
589 		html[idx++] = "' _flvId='";
590 		html[idx++] = this._internalId;
591 		html[idx++] = "' onchange='ZmPrefZimletListView._activeStateChange'>";
592 	}
593 	else if (field == ZmPrefZimletListView.COL_DESC) {
594         var desc = this._zimletsLoaded ? item.desc : ZmMsg.loading;
595         html[idx++] = "<div id='";
596         html[idx++] = this._getCellId(item, ZmPrefZimletListView.COL_DESC);
597         html[idx++] = "'>";
598 		html[idx++] = AjxStringUtil.stripTags(desc, true);
599         html[idx++] = "</div>";
600 	}
601 	else if (field == ZmPrefZimletListView.COL_NAME) {
602         html[idx++] = "<div id='";
603         html[idx++] = this._getCellId(item, ZmPrefZimletListView.COL_NAME);
604         html[idx++] = "' title='";
605         html[idx++] = item.name;
606         html[idx++] = "'>";
607 		html[idx++] = AjxStringUtil.stripTags(item.getNameWithoutPrefix(!this._zimletsLoaded), true);
608         html[idx++] = "</div>";
609 	}
610 	else if (field == ZmPrefZimletListView.COL_ACTION) {
611 		html[idx++] = "<a href='javascript:;' onclick='ZmPrefZimletListView.undeployZimlet(";
612 		html[idx++] = '"' + item.name + '"';
613 		html[idx++] = ");'>";
614 		html[idx++] = ZmMsg.uninstall;
615 		html[idx++] = "</a>";
616 	}
617 
618 	return idx;
619 };

703 /**
704  *
705  * @param desc true for desc sorting, false or empty otherwise
706  */
707 ZmPrefZimlets.prototype.sortByName =
708 function(desc) {
709 	var r = 0;
710     var zimletsLoaded = appCtxt.getZimletMgr().isLoaded();
711 	this._vector.sort(function(a,b) {
712 		var aname = a.getNameWithoutPrefix(!zimletsLoaded).toLowerCase();
713 		var bname = b.getNameWithoutPrefix(!zimletsLoaded).toLowerCase();
714 
715 		if (aname == bname) {
716 			r = 0;
717 		} else if (aname > bname) {
718 			r = 1;
719 		} else {
720 			r = -1;
721 		}
722 		return (desc ? -r : r);
723 	});
724 };
