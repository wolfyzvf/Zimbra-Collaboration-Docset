 22 /**
 23  * Drag and Drop Event handler

 25  * @author Hem Aravind

 30 ZmDragAndDrop = function(parent) {
 31     this._view = parent;
 32     this._controller = parent._controller;
 33     this._element = parent.getHtmlElement();
 34     this._initialize();
 35 };
 36 
 37 ZmDragAndDrop.prototype.constructor = ZmDragAndDrop;
 38 
 39 /**
 40 * @return	{boolean}	true if drag and drop is supported
 41 */
 42 ZmDragAndDrop.isSupported = function() {
 43 
 44     //Refer https://github.com/Modernizr/Modernizr/issues/57#issuecomment-4187079 Drag and Drop support
 45     var div = document.createElement('div'),
 46         dragSupport = (('draggable' in div) || ('ondragstart' in div && 'ondrop' in div)),
 47         isSupported = dragSupport && !!window.FileReader;
 48 
 49     if (AjxEnv.isSafari4up && dragSupport) {
 50         isSupported = true;
 51     }
 52 
 53     ZmDragAndDrop.isSupported = function() {
 54         return isSupported;
 55     };
 56 
 57     if (isSupported) {
 58         ZmDragAndDrop.MESSAGE_SIZE_LIMIT = appCtxt.get(ZmSetting.MESSAGE_SIZE_LIMIT);
 59         ZmDragAndDrop.ATTACHMENT_URL = appCtxt.get(ZmSetting.CSFE_ATTACHMENT_UPLOAD_URI)+"?fmt=extended,raw";
 60     }
 61 
 62     return ZmDragAndDrop.isSupported();
 63 };
 64 
 65 /**
 66  * @return	{boolean} 	true if attachment size exceeded and shows the warning dialog
 67  */
 68 ZmDragAndDrop.isAttachmentSizeExceeded = function(files, showDialog) {
 69     var j,
 70         filesLength,
 71 		size,
 72         file;
 73 
 74     if (!files) {
 75         return false;
 76     }
 77 
 78     for (j = 0 , size = 0, filesLength = files.length; j < filesLength; j++) {
 79         file = files[j];
 80         if (file) {
 81 			//Check the total size of the files we upload this time (we don't know the previously uploaded files total size so we do the best we can).
 82 			//NOTE - we compare to the MTA message size limit since there's no limit on specific attachments.
 83             size += file.size || file.fileSize /*Safari*/ || 0;
 84             //Showing Error dialog if the attachment size is exceeded
 85             if ((-1 /* means unlimited */ != ZmDragAndDrop.MESSAGE_SIZE_LIMIT) &&
 86                 (size > ZmDragAndDrop.MESSAGE_SIZE_LIMIT)) {
 87                 if (showDialog) {
 88                     var msgDlg = appCtxt.getMsgDialog();
 89                     var errorMsg = AjxMessageFormat.format(ZmMsg.attachmentSizeError, AjxUtil.formatSize(ZmDragAndDrop.MESSAGE_SIZE_LIMIT));
 90                     msgDlg.setMessage(errorMsg, DwtMessageDialog.WARNING_STYLE);
 91                     msgDlg.popup();
 92                 }
 93                 return true;
 94             }
 95         }
 96     }
 97     return false;
 98 };

100 ZmDragAndDrop.prototype._initialize = function () {
101 	if (!ZmDragAndDrop.isSupported() && this._element && this._element.id) {
102 		var tooltip = document.getElementById(this._element.id + ZmId.CMP_DND_TOOLTIP);
103 		if (tooltip) {
104 			tooltip.style.display = "none";
105 			tooltip.innerHTML = "";
106 		}
107 	}
108     if (!this._view || !this._controller || !this._element || !ZmDragAndDrop.isSupported()) {
109         return;
110     }
111     this._addHandlers(this._element);
112     this._dndTooltipEl = document.getElementById(this._element.id + ZmId.CMP_DND_TOOLTIP);
113     this._setToolTip();
114 };
115 
116 ZmDragAndDrop.prototype._addHandlers = function(el) {
117     Dwt.setHandler(el,"ondragover",this._onDragOver.bind(this));
118     Dwt.setHandler(el,"ondrop", this._onDrop.bind(this));
119 };
120 
121 ZmDragAndDrop.prototype._setToolTip = function(){
122     if (!this._dndTooltipEl) {
123         return;
124     }
125     if (this._view._attachCount > 0 || this._dndFilesLength > 0){
126         this._dndTooltipEl.style.display = "none";
127         this._dndTooltipEl.innerHTML = "";
128     } else {
129         this._dndTooltipEl.innerHTML = ZmMsg.dndTooltip;
130         this._dndTooltipEl.style.display = "block";
131     }
132 };
133 
134 ZmDragAndDrop.prototype._onDragOver = function(ev) {
135     ZmDragAndDrop._stopEvent(ev);
136 };
137 
138 ZmDragAndDrop.prototype._onDrop = function(ev, isEditorDND) {
139     var dt,
140         files,
141         file,
142         j,
143         filesLength;
144 
145     if (!ev || (this._view && this._view._disableAttachments === true) ) {
146         return;
147     }
148 
149     dt = ev.dataTransfer;
150     if (!dt) {
151         return;
152     }
153 
154     files = dt.files;
155     if (!files) {
156         return;
157     }
158 
159     if (files.length) {
160         ZmDragAndDrop._stopEvent(ev);
161     }
162 
163 	//just re-use code from the my computer option as it should be exactly the same case from now on.
164 	this._view._submitMyComputerAttachments(files, null, isEditorDND, ev);
165 };
166 
167 ZmDragAndDrop._stopEvent = function(ev) {
168 	if (!ZmDragAndDrop.containFiles(ev)) {
169 		return;
170 	}
171 	if (ev.preventDefault) {
172 		ev.preventDefault();
173 	}
174 	if (ev.stopPropagation) {
175 		ev.stopPropagation();
176 	}
177 };
178 
179 ZmDragAndDrop.containFiles =
180 function(ev, type) {
181 	var typesArray = ev && ev.dataTransfer && ev.dataTransfer.types;
182     if (!typesArray) {
183 		return false;
184 	}
185 	type = type || "Files";
186 	for (var i = 0; i < typesArray.length; i++) {
187 		if (typesArray[i] === type) {
188 			return true;
189 		}
190 	}
191 	return false;
192 };
