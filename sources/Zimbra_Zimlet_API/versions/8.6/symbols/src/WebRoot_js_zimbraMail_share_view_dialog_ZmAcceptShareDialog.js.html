 26 /**
 27  * Creates an "accept share" dialog.
 28  * @class
 29  * This class represents an "accept share" dialog.
 30  * 
 31  * @param	{DwtControl}	parent		the parent
 32  * @param	{String}	className		the class name
 33  * 
 34  * @extends		DwtDialog
 35  */
 36 ZmAcceptShareDialog = function(parent, className) {
 37 	className = className || "ZmAcceptShareDialog";
 38 	DwtDialog.call(this, {parent:parent, className:className, title:ZmMsg.acceptShare,
 39 						  standardButtons:[DwtDialog.YES_BUTTON, DwtDialog.NO_BUTTON], id: "ZmAcceptShare"});
 40 	this.setButtonListener(DwtDialog.YES_BUTTON, new AjxListener(this, this._handleYesButton));
 41 	this.setButtonListener(DwtDialog.NO_BUTTON, new AjxListener(this, this._handleNoButton));
 42 	
 43 	this.setView(this._createView());
 44 	
 45 	// create formatters
 46 	this._headerFormatter = new AjxMessageFormat(ZmMsg.acceptShareHeader);
 47 	this._detailsFormatter = new AjxMessageFormat(ZmMsg.acceptShareDetails);
 48 };
 49 
 50 ZmAcceptShareDialog.prototype = new DwtDialog;
 51 ZmAcceptShareDialog.prototype.constructor = ZmAcceptShareDialog;
 52 
 53 // Constants
 54 
 55 ZmAcceptShareDialog._ACTIONS = {};
 56 ZmAcceptShareDialog._ACTIONS[ZmShare.ROLE_NONE]		= ZmMsg.acceptShareDetailsNone;
 57 ZmAcceptShareDialog._ACTIONS[ZmShare.ROLE_VIEWER]	= ZmMsg.acceptShareDetailsViewer;
 58 ZmAcceptShareDialog._ACTIONS[ZmShare.ROLE_MANAGER]	= ZmMsg.acceptShareDetailsManager;
 59 ZmAcceptShareDialog._ACTIONS[ZmShare.ROLE_ADMIN]	= ZmMsg.acceptShareDetailsAdmin;
 60 
 61 // Public methods
 62 
 63 /**
 64  * Pops-up the dialog.
 65  * 
 66  * @param	{ZmShare}		share		the share
 67  * @param	{String}		fromAddr	the from address
 68  */
 69 ZmAcceptShareDialog.prototype.popup =
 70 function(share, fromAddr) {
 71 
 72 	this._share = share;
 73 	this._fromAddr = fromAddr;
 74 	this._headerEl.innerHTML = this._headerFormatter.format([share.grantor.name || share.grantor.email, AjxStringUtil.htmlEncode(share.link.name)]);
 75 
 76 	var role = ZmShare._getRoleFromPerm(share.link.perm);
 77 	var params = [
 78 		ZmShare.getRoleName(role),
 79 		ZmAcceptShareDialog._ACTIONS[role]   // TODO: Be able to generate custom perms list
 80 	];
 81 	this._detailsEl.innerHTML = this._detailsFormatter.format(params);
 82 	this._questionEl.innerHTML = "<b>" + ZmMsg.acceptShareQuestion + "</b>";
 83 
 84 	var namePart = share.grantor.name || (share.grantor.email && share.grantor.email.substr(0, share.grantor.email.indexOf('@')));
 85 	this._nameEl.value = ZmShare.getDefaultMountpointName(namePart, share.link.name);
 86 
 87 	this._reply.setReplyType(ZmShareReply.NONE);
 88 	this._reply.setReplyNote("");
 89 
 90 	var orgType = ZmOrganizer.TYPE[share.link.view];
 91 	var icon = null;
 92 	var orgClass = ZmOrganizer.ORG_CLASS[orgType];
 93 	if (orgClass) {
 94 		var orgPackage = ZmOrganizer.ORG_PACKAGE[orgType];
 95 		if (orgPackage) {
 96 			AjxDispatcher.require(orgPackage);
 97 			//to fix bug 55320 - got rid of the calling getIcon on the prototype hack - that caused isRemote to set _isRemote on the prototype thus causing every object to have it by default set.
 98 			var sample = new window[orgClass]({}); //get a sample object just for the icon
 99 			sample._isRemote = true; //hack - so it would get the remote version of the icon
100 			icon = sample.getIcon();
101 		}
102 	}
103 	this._color.setImage(icon);
104 	this._color.setValue(ZmOrganizer.DEFAULT_COLOR[orgType]);
105 	
106 	DwtDialog.prototype.popup.call(this);
107 };
108 
109 /**
110  * Sets the accept listener.
111  * 
112  * @param	{AjxListener}		listener		the listener
113  */
114 ZmAcceptShareDialog.prototype.setAcceptListener =
115 function(listener) {
116 	this.removeAllListeners(ZmAcceptShareDialog.ACCEPT);
117 	if (listener) {
118 		this.addListener(ZmAcceptShareDialog.ACCEPT, listener);
119 	}
120 };
121 
122 // Protected methods
123 
124 ZmAcceptShareDialog.prototype._handleYesButton =
125 function(ev) {
126 	var replyType = this._reply.getReplyType();
127 	var notes = (replyType == ZmShareReply.QUICK) ? this._reply.getReplyNote(): "";
128 	var callback = new AjxCallback(this, this._yesButtonCallback, [ev]);
129 	this._share.accept(this._nameEl.value, this._color.getValue(), replyType, notes, callback, this._fromAddr);
130 };
131 
132 ZmAcceptShareDialog.prototype._yesButtonCallback =
133 function(ev) {
134 	// notify accept listener and clear
135 	this.notifyListeners(ZmAcceptShareDialog.ACCEPT, ev);
136 	this.setAcceptListener(null);
137 	this.popdown();
138 };
139 
140 ZmAcceptShareDialog.prototype._handleNoButton =
141 function(ev) {
142 	this.popdown();
143 };
144 
145 ZmAcceptShareDialog.prototype._getSeparatorTemplate =
146 function() {
147 	return "";
148 };
149 
150 ZmAcceptShareDialog.prototype._createView =
151 function() {
152 	var view = new DwtComposite(this);
153 
154 	this._headerEl = document.createElement("DIV");
155 	this._headerEl.style.marginBottom = "0.5em";
156 	this._detailsEl = document.createElement("DIV");
157 	this._detailsEl.style.marginBottom = "1em";
158 	this._detailsEl.id = "ZmAcceptShare_details";
159 	this._questionEl = document.createElement("DIV");
160 	this._questionEl.style.marginBottom = "0.5em";
161 	this._questionEl.id = "ZmAcceptShare_questions";
162 	this._nameEl = document.createElement("INPUT");
163 	this._nameEl.style.width = "20em";
164 	this._nameEl.id = "ZmAcceptShare_name";
165 	var nameElement = this._nameEl;
166 
167 	this._color = new ZmColorButton({parent:this, id: "ZmAcceptShare_color"});
168 
169 	var props = this._propSheet = new DwtPropertySheet(view);
170 	var propsEl = props.getHtmlElement();
171 	propsEl.style.marginBottom = "0.5em";
172 	propsEl.id = "ZmAcceptShare_props";
173 	props.addProperty(ZmMsg.nameLabel, nameElement);
174 	props.addProperty(ZmMsg.colorLabel, this._color);
175 
176 	this._reply = new ZmShareReply(view);
177 
178 	var settings = document.createElement("DIV");
179 	settings.style.marginLeft = "1.5em";
180 	settings.id = "ZmAcceptShare_settings";
181 	settings.appendChild(propsEl);
182 	settings.appendChild(this._reply.getHtmlElement());	
183 
184 	var el = view.getHtmlElement();
185 	el.appendChild(this._headerEl);
186 	el.appendChild(this._detailsEl);
187 	el.appendChild(this._questionEl);
188 	el.appendChild(settings);
189 
190 	this._tabGroup.addMember(this._color.getTabGroupMember());
191 	this._tabGroup.addMember(this._propSheet.getTabGroupMember());
192 	this._tabGroup.addMember(this._reply.getTabGroupMember());
193 
194 	return view;
195 };
