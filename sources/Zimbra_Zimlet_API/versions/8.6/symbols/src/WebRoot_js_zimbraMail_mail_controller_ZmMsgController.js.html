 69 /**
 70  * Displays a message in the single-pane view.
 71  *
 72  * @param {ZmMailMsg}			msg					the message to display
 73  * @param {ZmListController}	parentController	the controller that called this method
 74  * @param {AjxCallback}			callback			the client callback
 75  * @param {Boolean}				markRead			if <code>true</code>, mark msg read
 76  * @param {Boolean}				hidePagination		if <code>true</code>, hide the pagination buttons
 77  */
 78 ZmMsgController.prototype.show = 
 79 function(msg, parentController, callback, markRead, hidePagination, forceLoad, noTruncate) {
 80 	this.setMsg(msg);
 81 	this._parentController = parentController;
 82 	//if(msg.list) {
 83         this.setList(msg.list);
 84     //}
 85 	if (!msg._loaded || forceLoad) {
 86 		var respCallback = new AjxCallback(this, this._handleResponseShow, [callback, hidePagination]);
 87 		if (msg._loadPending) {
 88 			// override any local callback if we're being launched by double-pane view,
 89 			// so that multiple GetMsgRequest's aren't made
 90 			msg._loadCallback = respCallback;
 91 		} else {
 92 			markRead = markRead || (appCtxt.get(ZmSetting.MARK_MSG_READ) == ZmSetting.MARK_READ_NOW);
 93 			msg.load({callback:respCallback, markRead:markRead, forceLoad:forceLoad, noTruncate:noTruncate});
 94 		}
 95 	} else {
 96 		// May have been explicitly marked as unread
 97 		var marked = false;
 98 		if (msg.isUnread && (appCtxt.get(ZmSetting.MARK_MSG_READ) != ZmSetting.MARK_READ_NONE)) {
 99 			if (msg.list) {
100 				// Need to mark it on the server
101 				marked = true;
102 				var markCallback =  this._handleResponseShow.bind(this, callback, hidePagination);
103 				msg.list.markRead({items: msg, value: true, callback: markCallback, noBusyOverlay: true});
104 			}  else {
105 				msg.markRead();
106 			}
107 		}
108 		if (!marked) {
109 			this._handleResponseShow(callback, hidePagination);
110 		}
111 	}
112 };
113 
114 ZmMsgController.prototype._handleResponseShow = 
115 function(callback, hidePagination, result) {
116 	this._showMsg();
117 	this._showNavToolBarButtons(this._currentViewId, !hidePagination);
118 	if (callback && callback.run) {
119 		callback.run(this, this._view[this._currentViewId]);
120 	}
121 };

171 ZmMsgController.prototype.handleKeyAction =
172 function(actionCode) {
173 	DBG.println(AjxDebug.DBG3, "ZmMsgController.handleKeyAction");
174 	
175 	switch (actionCode) {
176 		case ZmKeyMap.CANCEL:
177 			this._backListener();
178 			break;
179 
180 		case ZmKeyMap.NEXT_PAGE:
181 			this._goToMsg(this._currentViewId, true);
182 			break;
183 
184 		case ZmKeyMap.PREV_PAGE:
185 			this._goToMsg(this._currentViewId, false);
186 			break;
187 
188 		// switching view not supported here
189 		case ZmKeyMap.VIEW_BY_CONV:
190 		case ZmKeyMap.VIEW_BY_MSG:
191 			break;
192 		
193 		default:
194 			if (ZmMsgController.ALLOWED_SHORTCUT[actionCode]) {
195 				return ZmMailListController.prototype.handleKeyAction.call(this, actionCode);
196 			}
197 	}
198 	return true;
199 };

252 ZmMsgController.prototype._initializeView =
253 function(view) {
254 	if (!this._view[view]) {
255 		var params = {

257 			id:			ZmId.getViewId(ZmId.VIEW_MSG, null, view),
258 			posStyle:	Dwt.ABSOLUTE_STYLE,
259 			mode:		ZmId.VIEW_MSG,
260 			controller:	this
261 		};
262 		this._view[view] = new ZmMailMsgView(params);
263 		this._view[view].addInviteReplyListener(this._inviteReplyListener);
264 		this._view[view].addShareListener(this._shareListener);
265 		this._view[view].addSubscribeListener(this._subscribeListener);
266 	}

299 ZmMsgController.prototype._resetNavToolBarButtons =
300 function(view) {
301 	view = view || this.getCurrentViewId();
302 	if (!this._navToolBar[view]) { return; }
303 	// NOTE: we purposely do not call base class here!
304 	if (!appCtxt.isChildWindow) {
305 		var list = this._msg.list && this._msg.list.getVector();
306 
307 		this._navToolBar[view].enable(ZmOperation.PAGE_BACK, (list && (list.get(0) != this._msg)));
308 
309 		var bEnableForw = list && (this._msg.list.hasMore() || (list.getLast() != this._msg));
310 		this._navToolBar[view].enable(ZmOperation.PAGE_FORWARD, bEnableForw);
311 
312 		this._navToolBar[view].setToolTip(ZmOperation.PAGE_BACK, ZmMsg.previousMessage);
313 		this._navToolBar[view].setToolTip(ZmOperation.PAGE_FORWARD, ZmMsg.nextMessage);
314 	}
315 };

317 ZmMsgController.prototype._showNavToolBarButtons =
318 function(view, show) {
319 	var toolbar = this._navToolBar[view];
320 	if (!toolbar) { return; }
321 	if (!appCtxt.isChildWindow) {
322 		toolbar.getButton(ZmOperation.PAGE_BACK).setVisible(show);
323 		toolbar.getButton(ZmOperation.PAGE_FORWARD).setVisible(show);
324 	}
325 };

438 ZmMsgController.prototype._printListener =
439 function(ev) {
440     var ids = [];
441     var item = this._msg;
442     var id;
443     var showImages;
444     // always extract out the msg ids from the conv
445     if (item.toString() == "ZmConv") {
446         // get msg ID in case of virtual conv.
447         // item.msgIds.length is inconsistent, so checking if conv id is negative.
448         if (appCtxt.isOffline && item.id.split(":")[1]<0) {
449             id = item.msgIds[0];
450         } else {
451             id = "C:" + item.id;
452         }
453         var msgList = item.getMsgList();
454         for(var j=0; j<msgList.length; j++) {
455             if(msgList[j].showImages) {
456                 showImages = true;
457                 break;
458             }
459         }
460     } else {
461         id = item.id;
462         // Fix for bug: 84261, bug: 85363. partId is present if original message is present as an attachment.
463         var part = item.partId;
464         if (part) {
465             id += "&part=" + part;
466         }
467 
468         if (item.showImages) {
469             showImages = true;
470         }
471     }
472     var url = "/h/printmessage?id=" + id + "&tz=" + AjxTimezone.getServerId(AjxTimezone.DEFAULT);
473     if (appCtxt.get(ZmSetting.DISPLAY_EXTERNAL_IMAGES) || showImages) {
474         url += "&xim=1";
475     }
476     if (appCtxt.isOffline) {
477         var acctName = item.getAccount().name;
478         url+="&acct=" + acctName ;
479     }
480     window.open(appContextPath+url, "_blank");
481 };
