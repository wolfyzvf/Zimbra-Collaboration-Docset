 26 /**
 27  * Creates a share reply widget.
 28  * @class
 29  * This class implements a share reply query box and additional input 
 30  * controls to allow the user to specify the reply type and quick reply 
 31  * note, if wanted. This control can be used from within the various
 32  * share dialogs to add reply capabilities.
 33  * 
 34  * @param	{DwtControl}	parent		the parent
 35  * @param	{String}		className	the class name
 36  * @param	{Array}	options			an array of options
 37  * @extends		DwtComposite
 38  */
 39 ZmShareReply = function(params) {
 40 
 41 	params = Dwt.getParams(arguments, ZmShareReply.PARAMS);
 42 
 43 	params.className = params.className || "ZmShareReply";
 44 	params.id = "ZmShareReply";
 45 	DwtComposite.call(this, params);
 46 	this._tabGroup = new DwtTabGroup(this.toString());
 47 	this._initControl(params);
 48 };
 49 
 50 ZmShareReply.PARAMS = [ 'parent', 'className', 'options' ];
 51 
 52 ZmShareReply.prototype = new DwtComposite;
 53 ZmShareReply.prototype.constructor = ZmShareReply;
 54 //ZmShareReply.prototype.isFocusable = true;
 55 
 56 // Constants
 57 /**
 58  * Defines the "none" reply type.
 59  */
 60 ZmShareReply.NONE		= 0;
 61 /**
 62  * Defines the "standard" reply type.
 63  */
 64 ZmShareReply.STANDARD	= 1;
 65 /**
 66  * Defines the "quick" reply type.
 67  */
 68 ZmShareReply.QUICK		= 2;
 69 
 70 ZmShareReply.DEFAULT_OPTIONS = [
 71 	ZmShareReply.NONE, ZmShareReply.STANDARD, ZmShareReply.QUICK
 72 ];
 73 
 74 ZmShareReply.EXTERNAL_USER_OPTIONS = [
 75 	ZmShareReply.STANDARD, ZmShareReply.QUICK
 76 ];
 77 
 78 ZmShareReply._LABELS = {};
 79 ZmShareReply._LABELS[ZmShareReply.NONE]		= ZmMsg.sendNoMailAboutShare;
 80 ZmShareReply._LABELS[ZmShareReply.STANDARD] = ZmMsg.sendStandardMailAboutShare;
 81 ZmShareReply._LABELS[ZmShareReply.QUICK]	= ZmMsg.sendStandardMailAboutSharePlusNote;
 82 
 83 // Public methods
 84 
 85 /**
 86  * Sets the reply type.
 87  * 
 88  * @param	{constant}	type		the type
 89  */
 90 ZmShareReply.prototype.setReplyType =
 91 function(type) {
 92 	this._replyType.setSelectedValue(type);
 93 	Dwt.setVisible(this._replyStandardMailNoteEl, type == ZmShareReply.STANDARD || type == ZmShareReply.QUICK);
 94 	Dwt.setVisible(this._replyNoteEl, type == ZmShareReply.QUICK);
 95 };
 96 
 97 /**
 98  * Gets the reply type.
 99  * 
100  * @return	{constant}		the reply type
101  */
102 ZmShareReply.prototype.getReplyType =
103 function() {
104 	return this._replyType.getValue();
105 };
106 
107 /**
108  * Sets the reply note.
109  * 
110  * @param	{String}	note		the note
111  */
112 ZmShareReply.prototype.setReplyNote =
113 function(note) {
114 	this._replyNoteEl.value = note;
115 };
116 
117 /**
118  * Gets the reply note.
119  * 
120  * @return	{String}		the reply note
121  */
122 ZmShareReply.prototype.getReplyNote =
123 function() {
124 	return this._replyNoteEl.value;
125 };
126 
127 /**
128  * Sets the reply options.
129  * 
130  * @param	{Array}	options		an array of options
131  */
132 ZmShareReply.prototype.setReplyOptions =
133 function(options) {
134 	if (this._replyOptions == options) return;
135 
136 	this._replyOptions = options;
137 	this._replyType.clearOptions();
138 
139 	for (var i = 0; i < options.length; i++) {
140 		var value = options[i];
141 		this._replyType.addOption(ZmShareReply._LABELS[value], false, value);
142 	}
143 };
144 
145 /**
146  * Gets the reply options.
147  * 
148  * @return	{Array}	an array of options
149  */
150 ZmShareReply.prototype.getReplyOptions =
151 function() {
152 	return this._replyOptions;
153 };
154 
155 // Protected methods
156 
157 ZmShareReply.prototype._handleReplyType =
158 function(event) {
159 	var type = this._replyType.getValue();
160 	this.setReplyType(type);
161 };
162 
163 ZmShareReply.prototype._initControl = function(params) {
164 
165 	this._replyType = new DwtSelect({
166 		parent:   this,
167 		id:       "ZmShareReplySelect",
168 		legendId: params.legendId
169 	});
170     var options = params.options || ZmShareReply.DEFAULT_OPTIONS;
171     this.setReplyOptions(options);
172 	this._replyType.addChangeListener(this._handleReplyType.bind(this));
173 
174 	var doc = document;
175 	this._replyTypeEl = doc.createElement("DIV");
176 	this._replyTypeEl.style.paddingBottom = "0.5em";
177 	this._replyTypeEl.appendChild(this._replyType.getHtmlElement());
178 	
179 	this._replyStandardMailNoteEl = doc.createElement("DIV");
180 	this._replyStandardMailNoteEl.style.paddingBottom = "0.125em";
181 	this._replyStandardMailNoteEl.style.width = "30em";
182 	this._makeFocusable(this._replyStandardMailNoteEl);
183 	this._replyStandardMailNoteEl.innerHTML = ZmMsg.sendMailAboutShareNote;
184 	
185 	var div = doc.createElement("DIV");
186 	this._replyNoteEl = doc.createElement("TEXTAREA");
187 	this._replyNoteEl.cols = 50;
188 	this._replyNoteEl.rows = 4;
189 	div.appendChild(this._replyNoteEl);
190 	
191 	this._replyControlsEl = doc.createElement("DIV");
192 	this._replyControlsEl.style.marginLeft = "1.5em";
193 	this._replyControlsEl.appendChild(this._replyTypeEl);
194 	this._replyControlsEl.appendChild(this._replyStandardMailNoteEl);
195 	this._replyControlsEl.appendChild(div);
196 
197 	// append controls
198 	var element = this.getHtmlElement();
199 	element.appendChild(this._replyControlsEl);
200 	this._tabGroup.addMember(this._replyType);
201 	this._tabGroup.addMember(this._replyStandardMailNoteEl);
202 	this._tabGroup.addMember(this._replyNoteEl);
203 };
204 
205 ZmShareReply.prototype.getTabGroupMember = function(){
206 	return this._tabGroup;
207 };
