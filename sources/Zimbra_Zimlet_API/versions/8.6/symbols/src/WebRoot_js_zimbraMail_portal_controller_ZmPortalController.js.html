 27 /**
 28  * Creates the portal controller.
 29  * @class
 30  * This class represents the portal controller.
 31  * 
 32  * @param	{DwtComposite}	container	the containing element
 33  * @param	{ZmPortalApp}	app			the application
 34  * @param	{constant}		type		controller type
 35  * @param	{string}		sessionId	the session id
 36  * 
 37  * @extends		ZmListController
 38  */
 39 ZmPortalController = function(container, app, type, sessionId) {
 40 	if (arguments.length == 0) { return; }
 41 	ZmListController.apply(this, arguments);
 42 
 43     // TODO: Where does this really belong? Answer: in ZmPortalApp
 44     ZmOperation.registerOp(ZmId.OP_PAUSE_TOGGLE, {textKey:"pause", image:"Pause", style: DwtButton.TOGGLE_STYLE});
 45 
 46     this._listeners[ZmOperation.REFRESH] = new AjxListener(this, this._refreshListener);
 47     this._listeners[ZmOperation.PAUSE_TOGGLE] = new AjxListener(this, this._pauseListener);
 48 }
 49 ZmPortalController.prototype = new ZmListController;
 50 ZmPortalController.prototype.constructor = ZmPortalController;
 51 
 52 ZmPortalController.prototype.isZmPortalController = true;
 53 ZmPortalController.prototype.toString = function() { return "ZmPortalController"; };
 54 
 55 //
 56 // Public methods
 57 //
 58 
 59 ZmPortalController.prototype.getDefaultViewType = function() {
 60 	return ZmId.VIEW_PORTAL;
 61 };
 62 
 63 ZmPortalController.prototype.show = function() {
 64 	ZmListController.prototype.show.call(this);
 65 	this._setup(this._currentViewId);
 66 
 67 	var elements = this.getViewElements(this._currentViewId, this._listView[this._currentViewId]);
 68 
 69 	this._setView({ view:		this._currentViewId,
 70 					viewType:	this._currentViewType,
 71 					elements:	elements,
 72 					isAppView:	true});
 73 };
 74 
 75 /**
 76  * Sets the paused flag for the portlets.
 77  * 
 78  * @param	{Boolean}	paused		if <code>true</code>, pause the portlets
 79  */
 80 ZmPortalController.prototype.setPaused = function(paused) {
 81     var view = this._listView[this._currentViewId];
 82     var portletIds = view && view.getPortletIds();
 83     if (portletIds && portletIds.length > 0) {
 84         var portletMgr = appCtxt.getApp(ZmApp.PORTAL).getPortletMgr();
 85         for (var i = 0; i < portletIds.length; i++) {
 86             var portlet = portletMgr.getPortletById(portletIds[i]);
 87             portlet.setPaused(paused);
 88         }
 89     }
 90 };

 96 ZmPortalController.prototype._getToolBarOps = function() {
 97 	return [ ZmOperation.REFRESH /*, ZmOperation.PAUSE_TOGGLE*/ ];

 99 
100 ZmPortalController.prototype._createNewView = function(view) {
101 	return new ZmPortalView(this._container, this, this._dropTgt);
102 };
103 
104 ZmPortalController.prototype._setViewContents = function(view) {
105 	this._listView[view].set();
106 };
107 
108 // listeners
109 
110 ZmPortalController.prototype._refreshListener = function() {
111     this._app.refreshPortlets();
112 };
113 
114 ZmPortalController.prototype._pauseListener = function(event) {
115     var toolbar = this._toolbar[this._currentViewId];
116 
117     // en/disable refresh button
118     var button = toolbar && toolbar.getButton(ZmOperation.REFRESH);
119     var paused = event.item.isToggled();
120     if (button) {
121         button.setEnabled(!paused);
122     }
123 
124     // pause portlets appearing on portal page
125     this.setPaused(paused);
126 };
127 
128 ZmPortalController.prototype._resetOperations = function(parent, num) {
129 //    ZmListController.prototype._resetOperations.call(parent, num);
130     parent.enable(this._getToolBarOps(), true);
131 };
