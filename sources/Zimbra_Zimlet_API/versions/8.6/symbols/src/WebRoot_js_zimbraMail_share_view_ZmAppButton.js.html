 28 /**
 29  * @class
 30  * This class represents a button that behaves like a "tab" button, designed specifically for the row of
 31  * applications buttons at the top of the Zimbra Web Client interface.
 32  * <p>
 33  * Limitations:
 34  * <ul>
 35  * <li>cannot have a menu</li>
 36  * <li>does not support enabled/disabled</li>
 37  * </ul>
 38  * </p>
 39  * 
 40  * @author Conrad Damon
 41  * 
 42  * @param	{Hash}		params		a hash of parameters
 43  * 
 44  * @extends		DwtButton
 45  */
 46 ZmAppButton = function(params) {
 47 
 48 	if (arguments.length == 0) {
 49 		return;
 50 	}
 51 
 52     params.style = params.style ? params.style : DwtLabel.IMAGE_LEFT;
 53 	params.posStyle = DwtControl.RELATIVE_STYLE;
 54     DwtButton.call(this, params);
 55 
 56 	if (params.image) {
 57 		this.setImage(params.image);
 58 	}
 59 	else {
 60 		if (params.leftImage) {
 61 			this.setImage(	params.leftImage, DwtLabel.LEFT);
 62 		}
 63 		if (params.rightImage) {
 64 			this.setImage(params.rightImage, DwtLabel.RIGHT);
 65 		}
 66 	}
 67 
 68 	if (params.hoverImage) {
 69 		this.setHoverImage(params.hoverImage);
 70 	}
 71 	else {
 72 		if (params.leftHoverImage) {
 73 			this.setHoverImage(params.leftHoverImage, DwtLabel.LEFT);
 74 		}
 75 		if (params.rightHoverImage) {
 76 			this.setHoverImage(params.rightHoverImage, DwtLabel.RIGHT);
 77 		}
 78 	}
 79     this.setText(params.text);
 80 };
 81 
 82 ZmAppButton.prototype = new DwtButton;
 83 ZmAppButton.prototype.constructor = ZmAppButton;
 84 ZmAppButton.prototype.role = "tab";
 85 
 86 /**
 87  * Returns a string representation of the object.
 88  * 
 89  * @return		{String}		a string representation of the object
 90  */
 91 ZmAppButton.prototype.toString =
 92 function() {
 93 	return "ZmAppButton";
 94 };

102 //
103 // Public methods
104 //
105 ZmAppButton.prototype.setSelected =
106 function(selected) {
107     this.isSelected = selected;
108     this.setDisplayState(selected ? DwtControl.SELECTED : DwtControl.NORMAL);
109 };
110 
111 /**
112  * Sets the display state.
113  * 
114  * @param	{String}	state		the display state
115  * @see		DwtControl
116  */
117 ZmAppButton.prototype.setDisplayState =
118 function(state) {
119     if (this.isSelected && state != DwtControl.SELECTED) {
120         state = [DwtControl.SELECTED, state].join(" ");
121     }
122     DwtButton.prototype.setDisplayState.call(this, state);
123 };
124 
125 ZmAppButton.prototype.handleKeyAction =
126 function(actionCode, ev) {
127 
128 	switch (actionCode) {
129 
130 		case DwtKeyMap.SELECT:
131 			if (this.isListenerRegistered(DwtEvent.SELECTION)) {
132 				var selEv = DwtShell.selectionEvent;
133 				selEv.item = this;
134 				this.notifyListeners(DwtEvent.SELECTION, selEv);
135 			}
136 			break;
137 
138 		default:
139 			return false;
140 	}
141 	return true;
142 };
