 21 /**
 22  * Appointment list view.
 23  */
 24 ZmApptListView = function(parent, posStyle, controller, dropTgt) {
 25     if (arguments.length == 0) return;
 26     var params = Dwt.getParams(arguments, ZmApptListView.PARAMS);
 27     params.headerList = this._getHeaderList();
 28     params.view = params.view || ZmId.VIEW_CAL_TRASH;
 29     ZmListView.call(this, params);
 30     this._bSortAsc = true;
 31     this._defaultSortField = ZmItem.F_DATE;
 32     this.setDragSource(params.controller._dragSrc);
 33 };
 34 ZmApptListView.prototype = new ZmListView;

 60 ZmApptListView.prototype.refresh = function() {
 61     if (this.needsRefresh()) {
 62         var selection = this.getSelection();
 63         this.set(this.getApptList());
 64         if (selection) {
 65             this.setSelectedItems(selection);
 66         }
 67         this.setNeedsRefresh(false);
 68     }
 69 };
 70 
 71 ZmApptListView.prototype.needsRefresh = function() {
 72     var controller = this._controller;
 73     return controller.getCurrentView().needsRefresh();
 74 };
 75 
 76 ZmApptListView.prototype.setNeedsRefresh = function(needsRefresh) {
 77     var controller = this._controller;
 78     if(controller.getCurrentView().setNeedsRefresh){
 79         return controller.getCurrentView().setNeedsRefresh(needsRefresh);
 80     }
 81     return null;
 82 };

105 ZmApptListView.prototype._getToolTip =
106 function(params) {
107 	var tooltip, field = params.field, item = params.item;
108 	if (field && (field == ZmItem.F_SELECTION || field == ZmItem.F_TAG)) {
109 		tooltip = ZmListView.prototype._getToolTip.apply(this, arguments);
110 	} else if (item.getToolTip) {
111 		tooltip = item.getToolTip(this._controller);
112 	}
113 	return tooltip;
114 };

116 ZmApptListView.prototype._sortList = function(list, column) {
117 	ZmApptListView.sortByAsc = this._bSortAsc;
118 
119 	switch (column) {
120 		case ZmItem.F_SUBJECT:	list.sort(ZmApptListView._sortSubject); break;
121 		case ZmItem.F_STATUS:	list.sort(ZmApptListView._sortStatus); break;
122 		case ZmItem.F_FOLDER:	list.sort(ZmApptListView._sortFolder); break;
123 		case ZmItem.F_DATE:		list.sort(ZmApptListView._sortDate); break;
124 	}
125 };
126 
127 ZmApptListView.prototype._sortColumn = function(columnItem, bSortAsc) {
128 	this._defaultSortField = columnItem._field;
129 
130 	var list = this.getList();
131 	list = list && list.clone();
132 	if (list) {
133 		this._sortList(list, columnItem._field);
134 		this.set(list, null, true);
135 	}
136 };

138 ZmApptListView.prototype._getHeaderToolTip = function(field, itemIdx) {
139 	switch (field) {
140 		case ZmItem.F_LOCATION: return ZmMsg.location;
141 		case ZmItem.F_FOLDER:	return ZmMsg.calendar;
142 		case ZmItem.F_DATE:		return ZmMsg.date;
143         case ZmItem.F_RECURRENCE:return ZmMsg.recurrence;       
144 	}
145 	return ZmListView.prototype._getHeaderToolTip.call(this, field, itemIdx);
146 };

148 ZmApptListView.prototype._getHeaderList = function() {
149 	var hList = [];
150 
151 	if (appCtxt.get(ZmSetting.SHOW_SELECTION_CHECKBOX)) {
152 		hList.push(new DwtListHeaderItem({field:ZmItem.F_SELECTION, icon:"CheckboxUnchecked", width:ZmListView.COL_WIDTH_ICON, name:ZmMsg.selection}));
153 	}
154 	if (appCtxt.get(ZmSetting.TAGGING_ENABLED)) {
155 		hList.push(new DwtListHeaderItem({field:ZmItem.F_TAG, icon:"Tag", width:ZmListView.COL_WIDTH_ICON, name:ZmMsg.tag}));
156 	}
157 	hList.push(new DwtListHeaderItem({field:ZmItem.F_ATTACHMENT, icon:"Attachment", width:ZmListView.COL_WIDTH_ICON, name:ZmMsg.attachment}));
158 	hList.push(new DwtListHeaderItem({field:ZmItem.F_SUBJECT, text:ZmMsg.subject, noRemove:true, sortable:ZmItem.F_SUBJECT}));
159 	hList.push(new DwtListHeaderItem({field:ZmItem.F_LOCATION, text:ZmMsg.location, width:ZmApptListView.COL_WIDTH_LOCATION, resizeable:true}));
160 	hList.push(new DwtListHeaderItem({field:ZmItem.F_STATUS, text:ZmMsg.status, width:ZmApptListView.COL_WIDTH_STATUS, resizeable:true, sortable:ZmItem.F_STATUS}));
161 	hList.push(new DwtListHeaderItem({field:ZmItem.F_FOLDER, text:ZmMsg.calendar, width:ZmApptListView.COL_WIDTH_FOLDER, resizeable:true, sortable:ZmItem.F_FOLDER}));
162 	hList.push(new DwtListHeaderItem({field:ZmItem.F_RECURRENCE, icon:"ApptRecur", width:ZmListView.COL_WIDTH_ICON, name:ZmMsg.recurrence}));
163 	hList.push(new DwtListHeaderItem({field:ZmItem.F_DATE, text:ZmMsg.startDate, width:ZmApptListView.COL_WIDTH_DATE, sortable:ZmItem.F_DATE}));
164 
165 	return hList;
166 };

177 ZmApptListView.prototype.set = function(apptList, skipMiniCalUpdate, skipSort) {
178     this._apptList = apptList;
179 	if (!skipSort) {
180 		if ((this._defaultSortField != ZmItem.F_DATE) ||
181 			(this._defaultSortField == ZmItem.F_DATE && !this._bSortAsc))
182 		{
183 			this._sortList(apptList, this._defaultSortField);
184 		}
185 	}
186 	ZmListView.prototype.set.call(this, apptList, this._defaultSortField);
187     this._resetColWidth();
188     //Does not make sense but required to make the scrollbar appear
189     var size = this.getSize();
190     this._listDiv.style.height = (size.y - DwtListView.HEADERITEM_HEIGHT)+"px";
191 };

203 ZmApptListView.prototype._getCellId = function(item, field) {
204 	if (field == ZmItem.F_SUBJECT || field == ZmItem.F_DATE || field == ZmItem.F_LOCATION || field == ZmItem.F_STATUS || field == ZmItem.F_FOLDER) {

209 ZmApptListView.prototype._getCellContents = function(htmlArr, idx, appt, field, colIdx, params) {
210 	if (field == ZmItem.F_RECURRENCE) {
211 		var icon;
212 		if (appt.isException) {
213 			icon = "ApptExceptionIndicator";
214 		}
215         else if (appt.isRecurring()) {
216 			icon = "ApptRecur";
217 		}
218 		idx = this._getImageHtml(htmlArr, idx, icon, this._getFieldId(appt, field));
219 
220 	}
221     else if (field == ZmItem.F_SUBJECT) {
222 		htmlArr[idx++] = AjxStringUtil.htmlEncode(appt.getName(), true);
223 		if (appCtxt.get(ZmSetting.SHOW_FRAGMENTS) && appt.fragment) {
224 			htmlArr[idx++] = this._getFragmentSpan(appt);
225 		}
226 
227 	}
228     else if (field == ZmItem.F_LOCATION) {
229 		htmlArr[idx++] = AjxStringUtil.htmlEncode(appt.getLocation(), true);
230 
231 	}
232     else if (field == ZmItem.F_STATUS) {
233 		if (appt.otherAttendees) {
234 			htmlArr[idx++] = appt.getParticipantStatusStr();
235 		}
236 
237 	}
238     else if (field == ZmItem.F_FOLDER) {
239 		var calendar = appt.getFolder();
240         var rgb = calendar.rgb || ZmOrganizer.COLOR_VALUES[calendar.color||ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.CALENDAR]]; 
241 		var colors = ZmCalBaseView._getColors(rgb);
242 		var subs = {
243             folder: calendar,
244 			folderColor: colors.standard.header.bgcolor,
245 			folderName: calendar.getName(),
246             id: Dwt.getNextId()
247 		};
248 		htmlArr[idx++] = AjxTemplate.expand("calendar.Calendar#ListViewFolder", subs);
249 
250 	}
251     else if (field == ZmItem.F_DATE) {
252 		htmlArr[idx++] = (appt.isAllDayEvent())
253 			? AjxMessageFormat.format(ZmMsg.apptDateTimeAllDay, [appt.startDate])
254 			: AjxMessageFormat.format(ZmMsg.apptDateTime, [appt.startDate, appt.startDate]);
255 
256 	}
257     else {
258 		idx = ZmListView.prototype._getCellContents.apply(this, arguments);
259 	}
260 
261 	return idx;
262 };

268 ZmApptListView._sortSubject = function(a, b) {
269     // Bug fix # 80458 - Convert the subject line to lower case and compare
270     var aVal = a.getName().toLowerCase();
271     var bVal = b.getName().toLowerCase();
272 
273 	if (aVal < bVal)		{ return ZmApptListView.sortByAsc ? -1 : 1; }
274 	else if (aVal > bVal)	{ return ZmApptListView.sortByAsc ? 1 : -1; }
275 	else 					{ return 0; }
276 };
277 
278 ZmApptListView._sortStatus = function(a, b) {
279 	if (!a.otherAttendees)	{ return ZmApptListView.sortByAsc ? -1 : 1; }
280 	if (!b.otherAttendees)	{ return ZmApptListView.sortByAsc ? 1 : -1; }
281 
282 	var aVal = a.getParticipantStatusStr();
283 	var bVal = b.getParticipantStatusStr();
284 
285 	if (aVal < bVal)		{ return ZmApptListView.sortByAsc ? -1 : 1; }
286 	else if (aVal > bVal)	{ return ZmApptListView.sortByAsc ? 1 : -1; }
287 	else 					{ return 0; }
288 };
289 
290 ZmApptListView._sortFolder = function(a, b) {
291 	var aVal = a.getFolder().getName();
292 	var bVal = b.getFolder().getName();
293 
294 	if (aVal < bVal)		{ return ZmApptListView.sortByAsc ? -1 : 1; }
295 	else if (aVal > bVal)	{ return ZmApptListView.sortByAsc ? 1 : -1; }
296 	else 					{ return 0; }
297 };
298 
299 ZmApptListView._sortDate = function(a, b) {
300 	var aVal = a.startDate.getTime();
301 	var bVal = b.startDate.getTime();
302 
303 	if (aVal < bVal)		{ return ZmApptListView.sortByAsc ? -1 : 1; }
304 	else if (aVal > bVal)	{ return ZmApptListView.sortByAsc ? 1 : -1; }
305 	else 					{ return 0; }
306 };
