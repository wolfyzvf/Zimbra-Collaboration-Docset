 24 var ZeroClipboard = {
 25 	
 26 	version: "1.0.7",
 27 	clients: {}, // registered upload clients on page, indexed by id
 28 	moviePath: 'ZeroClipboard.swf', // URL to movie
 29 	nextId: 1, // ID of next movie

 31 	$: function(thingy) {
 32 		// simple DOM lookup utility function
 33 		if (typeof(thingy) == 'string') thingy = document.getElementById(thingy);
 34 		if (!thingy.addClass) {
 35 			// extend element with a few useful methods
 36 			thingy.hide = function() { this.style.display = 'none'; };
 37 			thingy.show = function() { this.style.display = ''; };
 38 			thingy.addClass = function(name) { this.removeClass(name); this.className += ' ' + name; };
 39 			thingy.removeClass = function(name) {
 40 				var classes = this.className.split(/\s+/);
 41 				var idx = -1;
 42 				for (var k = 0; k < classes.length; k++) {
 43 					if (classes[k] == name) { idx = k; k = classes.length; }
 44 				}
 45 				if (idx > -1) {
 46 					classes.splice( idx, 1 );
 47 					this.className = classes.join(' ');
 48 				}
 49 				return this;
 50 			};
 51 			thingy.hasClass = function(name) {
 52 				return !!this.className.match( new RegExp("\\s*" + name + "\\s*") );
 53 			};
 54 		}
 55 		return thingy;
 56 	},

 58 	setMoviePath: function(path) {
 59 		// set path to ZeroClipboard.swf
 60 		this.moviePath = path;
 61 	},

 63 	dispatch: function(id, eventName, args) {
 64 		// receive event from flash movie, send to client		
 65 		var client = this.clients[id];
 66 		if (client) {
 67 			client.receiveEvent(eventName, args);
 68 		}
 69 	},
 70 	
 71 	register: function(id, client) {
 72 		// register new client to receive events
 73 		this.clients[id] = client;
 74 	},
 75 	
 76 	getDOMObjectPosition: function(obj, stopObj) {
 77 		// get absolute coordinates for dom element
 78 		var info = {
 79 			left: 0, 
 80 			top: 0, 
 81 			width: obj.width ? obj.width : obj.offsetWidth, 
 82 			height: obj.height ? obj.height : obj.offsetHeight
 83 		};
 84 
 85 		while (obj && (obj != stopObj)) {
 86 			info.left += obj.offsetLeft;
 87 			info.top += obj.offsetTop;
 88 			obj = obj.offsetParent;
 89 		}
 90 
 91 		return info;
 92 	},
 93 	
 94 	Client: function(elem) {
 95 		// constructor for new simple upload client
 96 		this.handlers = {};
 97 		
 98 		// unique ID
 99 		this.id = ZeroClipboard.nextId++;
100 		this.movieId = 'ZeroClipboardMovie_' + this.id;
101 		
102 		// register client with singleton to receive flash events
103 		ZeroClipboard.register(this.id, this);
104 		
105 		// create movie
106 		if (elem) this.glue(elem);
107 	}
108 };

110 ZeroClipboard.Client.prototype = {
111 	
112 	id: 0, // unique ID for us
113 	ready: false, // whether movie is ready to receive events or not
114 	movie: null, // reference to movie object
115 	clipText: '', // text to copy to clipboard
116 	handCursorEnabled: true, // whether to show hand cursor, or default pointer cursor
117 	cssEffects: true, // enable CSS mouse effects on dom container
118 	handlers: null, // user event handlers
119 	
120 	glue: function(elem, appendElem, stylesToAdd) {
121 		// glue to DOM element
122 		// elem can be ID or actual DOM element object
123 		this.domElement = ZeroClipboard.$(elem);
124 		
125 		// float just above object, or zIndex 99 if dom element isn't set
126 		var zIndex = 99;
127 		if (this.domElement.style.zIndex) {
128 			zIndex = parseInt(this.domElement.style.zIndex, 10) + 1;
129 		}
130 		
131 		if (typeof(appendElem) == 'string') {
132 			appendElem = ZeroClipboard.$(appendElem);
133 		}
134 		else if (typeof(appendElem) == 'undefined') {
135 			appendElem = document.getElementsByTagName('body')[0];
136 		}
137 		
138 		// find X/Y position of domElement
139 		var box = ZeroClipboard.getDOMObjectPosition(this.domElement, appendElem);
140 		
141 		// create floating DIV above element
142 		this.div = document.createElement('div');
143 		var style = this.div.style;
144 		style.position = 'absolute';
145 		style.left = '' + box.left + 'px';
146 		style.top = '' + box.top + 'px';
147 		style.width = '' + box.width + 'px';
148 		style.height = '' + box.height + 'px';
149 		style.zIndex = zIndex;
150 		
151 		if (typeof(stylesToAdd) == 'object') {
152 			for (addedStyle in stylesToAdd) {
153 				style[addedStyle] = stylesToAdd[addedStyle];
154 			}
155 		}
156 		
157 		// style.backgroundColor = '#f00'; // debug
158 		
159 		appendElem.appendChild(this.div);
160 		
161 		this.div.innerHTML = this.getHTML( box.width, box.height );
162 	},
163 	
164 	getHTML: function(width, height) {
165 		// return HTML for movie
166 		var html = '';
167 		var flashvars = 'id=' + this.id + 
168 			'&width=' + width + 
169 			'&height=' + height;
170 			
171 		if (navigator.userAgent.match(/MSIE/)) {
172 			// IE gets an OBJECT tag
173 			var protocol = location.href.match(/^https/i) ? 'https://' : 'http://';
174 			html += '<object classid="clsid:d27cdb6e-ae6d-11cf-96b8-444553540000" codebase="'+protocol+'download.macromedia.com/pub/shockwave/cabs/flash/swflash.cab#version=9,0,0,0" width="'+width+'" height="'+height+'" id="'+this.movieId+'" align="middle"><param name="allowScriptAccess" value="always" /><param name="allowFullScreen" value="false" /><param name="movie" value="'+ZeroClipboard.moviePath+'" /><param name="loop" value="false" /><param name="menu" value="false" /><param name="quality" value="best" /><param name="bgcolor" value="#ffffff" /><param name="flashvars" value="'+flashvars+'"/><param name="wmode" value="transparent"/></object>';
175 		}
176 		else {
177 			// all other browsers get an EMBED tag
178 			html += '<embed id="'+this.movieId+'" src="'+ZeroClipboard.moviePath+'" loop="false" menu="false" quality="best" bgcolor="#ffffff" width="'+width+'" height="'+height+'" name="'+this.movieId+'" align="middle" allowScriptAccess="always" allowFullScreen="false" type="application/x-shockwave-flash" pluginspage="http://www.macromedia.com/go/getflashplayer" flashvars="'+flashvars+'" wmode="transparent" />';
179 		}
180 		return html;
181 	},
182 	
183 	hide: function() {
184 		// temporarily hide floater offscreen
185 		if (this.div) {
186 			this.div.style.left = '-2000px';
187 		}
188 	},
189 	
190 	show: function() {
191 		// show ourselves after a call to hide()
192 		this.reposition();
193 	},
194 	
195 	destroy: function() {
196 		// destroy control and floater
197 		if (this.domElement && this.div) {
198 			this.hide();
199 			this.div.innerHTML = '';
200 			
201 			var body = document.getElementsByTagName('body')[0];
202 			try { body.removeChild( this.div ); } catch(e) {;}
203 			
204 			this.domElement = null;
205 			this.div = null;
206 		}
207 	},
208 	
209 	reposition: function(elem) {
210 		// reposition our floating div, optionally to new container
211 		// warning: container CANNOT change size, only position
212 		if (elem) {
213 			this.domElement = ZeroClipboard.$(elem);
214 			if (!this.domElement) this.hide();
215 		}
216 		
217 		if (this.domElement && this.div) {
218 			var box = ZeroClipboard.getDOMObjectPosition(this.domElement);
219 			var style = this.div.style;
220 			style.left = '' + box.left + 'px';
221 			style.top = '' + box.top + 'px';
222 		}
223 	},
224 	
225 	setText: function(newText) {
226 		// set text to be copied to clipboard
227 		this.clipText = newText;
228 		if (this.ready) this.movie.setText(newText);
229 	},
230 	
231 	addEventListener: function(eventName, func) {
232 		// add user event listener for event
233 		// event types: load, queueStart, fileStart, fileComplete, queueComplete, progress, error, cancel
234 		eventName = eventName.toString().toLowerCase().replace(/^on/, '');
235 		if (!this.handlers[eventName]) this.handlers[eventName] = [];
236 		this.handlers[eventName].push(func);
237 	},
238 	
239 	setHandCursor: function(enabled) {
240 		// enable hand cursor (true), or default arrow cursor (false)
241 		this.handCursorEnabled = enabled;
242 		if (this.ready) this.movie.setHandCursor(enabled);
243 	},
244 	
245 	setCSSEffects: function(enabled) {
246 		// enable or disable CSS effects on DOM container
247 		this.cssEffects = !!enabled;
248 	},
249 	
250 	receiveEvent: function(eventName, args) {
251 		// receive event from flash
252 		eventName = eventName.toString().toLowerCase().replace(/^on/, '');
253 				
254 		// special behavior for certain events
255 		switch (eventName) {
256 			case 'load':
257 				// movie claims it is ready, but in IE this isn't always the case...
258 				// bug fix: Cannot extend EMBED DOM elements in Firefox, must use traditional function
259 				this.movie = document.getElementById(this.movieId);
260 				if (!this.movie) {
261 					var self = this;
262 					setTimeout( function() { self.receiveEvent('load', null); }, 1 );
263 					return;
264 				}
265 				
266 				// firefox on pc needs a "kick" in order to set these in certain cases
267 				if (!this.ready && navigator.userAgent.match(/Firefox/) && navigator.userAgent.match(/Windows/)) {
268 					var self = this;
269 					setTimeout( function() { self.receiveEvent('load', null); }, 100 );
270 					this.ready = true;
271 					return;
272 				}
273 				
274 				this.ready = true;
275 				this.movie.setText( this.clipText );
276 				this.movie.setHandCursor( this.handCursorEnabled );
277 				break;
278 			
279 			case 'mouseover':
280 				if (this.domElement && this.cssEffects) {
281 					this.domElement.addClass('hover');
282 					if (this.recoverActive) this.domElement.addClass('active');
283 				}
284 				break;
285 			
286 			case 'mouseout':
287 				if (this.domElement && this.cssEffects) {
288 					this.recoverActive = false;
289 					if (this.domElement.hasClass('active')) {
290 						this.domElement.removeClass('active');
291 						this.recoverActive = true;
292 					}
293 					this.domElement.removeClass('hover');
294 				}
295 				break;
296 			
297 			case 'mousedown':
298 				if (this.domElement && this.cssEffects) {
299 					this.domElement.addClass('active');
300 				}
301 				break;
302 			
303 			case 'mouseup':
304 				if (this.domElement && this.cssEffects) {
305 					this.domElement.removeClass('active');
306 					this.recoverActive = false;
307 				}
308 				break;
309 		} // switch eventName
310 		
311 		if (this.handlers[eventName]) {
312 			for (var idx = 0, len = this.handlers[eventName].length; idx < len; idx++) {
313 				var func = this.handlers[eventName][idx];
314 			
315 				if (typeof(func) == 'function') {
316 					// actual function reference
317 					func(this, args);
318 				}
319 				else if ((typeof(func) == 'object') && (func.length == 2)) {
320 					// PHP style object + method, i.e. [myObject, 'myMethod']
321 					func[0][ func[1] ](this, args);
322 				}
323 				else if (typeof(func) == 'string') {
324 					// name of function
325 					window[func](this, args);
326 				}
327 			} // foreach event handler defined
328 		} // user defined handler for event
329 	}
330 	
331 };
