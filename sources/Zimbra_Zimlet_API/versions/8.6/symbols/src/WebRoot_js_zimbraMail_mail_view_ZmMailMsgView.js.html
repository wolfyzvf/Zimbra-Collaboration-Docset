179 ZmMailMsgView.prototype.set =
180 function(msg, force, dayViewCallback) {
181 	
182 	if (!force && this._msg && msg && !msg.force && (this._msg == msg)) { return; }
183 
184 	var oldMsg = this._msg;
185 	this.reset();
186 	var contentDiv = this._getContainer();
187 	this._msg = this._item = msg;
188 
189 	if (!msg) {
190 		if (this._inviteMsgView) {
191 			this._inviteMsgView.resize(true); //make sure the msg preview pane takes the entire area, in case we were viewing an invite. (since then it was resized to allow for day view) - bug 53098
192 		}
193 		contentDiv.innerHTML = (this._controller.getList().size()) ? AjxTemplate.expand("mail.Message#viewMessage") : "";
194 		return;
195 	}
196 
197 	msg.force = false;
198 	var respCallback = this._handleResponseSet.bind(this, msg, oldMsg, dayViewCallback);
199 	this._renderMessage(msg, contentDiv, respCallback);
200 };

215 ZmMailMsgView.prototype.__hasMountpoint2 =
216 function(organizer, zid, rid) {
217 	if (organizer.zid == zid && organizer.rid == rid)
218 		return true;
219 
220 	if (organizer.children) {
221 		var children = organizer.children.getArray();
222 		for (var i = 0; i < children.length; i++) {
223 			var found = this.__hasMountpoint2(children[i], zid, rid);
224 			if (found) {
225 				return true;
226 			}
227 		}
228 	}
229 	return false;
230 };

232 ZmMailMsgView.prototype.highlightObjects =
233 function(origText) {
234 	if (origText != null) {
235 		// we get here only for text messages; it's a lot
236 		// faster to call findObjects on the whole text rather
237 		// than parsing the DOM.
238 		DBG.timePt("START - highlight objects on-demand, text msg.");
239 		this._lazyCreateObjectManager();
240 		var html = this._objectManager.findObjects(origText, true, null, true);
241 		html = html.replace(/^ /mg, " ")
242 			.replace(/\t/g, "<pre style='display:inline;'>\t</pre>")
243 			.replace(/\n/g, "<br>");
244 		var container = this.getContentContainer();
245 		container.innerHTML = html;
246 		DBG.timePt("END - highlight objects on-demand, text msg.");
247 	} else {
248 		this._processHtmlDoc();
249 	}

458 ZmMailMsgView.prototype._handleResponseSet =
459 function(msg, oldMsg, dayViewCallback) {
460 
461 	var bubblesCreated = false;
462 	if (this._inviteMsgView) {
463 		// always show F/B view if in stand-alone message view otherwise, check if reading pane is on
464 		if (this._inviteMsgView.isActive() && (this._controller.isReadingPaneOn() || (this._controller.isZmMsgController))) {
465 			bubblesCreated = true;
466 			appCtxt.notifyZimlets("onMsgView", [msg, oldMsg, this]);
467 			this._inviteMsgView.showMoreInfo(this._createBubbles.bind(this), dayViewCallback);
468 		}
469 		else {
470 			// resize the message view without F/B view
471 			this._inviteMsgView.resize(true);
472 		}
473     }
474 
475 	this._setTags(msg);
476 	// Remove listener for current msg if it exists
477 	if (oldMsg) {
478 		oldMsg.removeChangeListener(this._changeListener);
479 	}
480 	msg.addChangeListener(this._changeListener);
481 
482 	if (msg.cloneOf) {
483 		msg.cloneOf.addChangeListener(this._changeListener);
484 	}
485 	if (oldMsg && oldMsg.cloneOf) {
486 		oldMsg.cloneOf.removeChangeListener(this._changeListener);
487 	}
488 
489 	// reset scroll view to top most
490 	var htmlElement = this.getHtmlElement();
491 	htmlElement.scrollTop = 0;
492 	if (htmlElement.scrollTop != 0 && this._usingIframe) {
493 		/* situation that happens only on Chrome, without repro steps - bug 55775/57090 */
494 		AjxDebug.println(AjxDebug.SCROLL, "scrollTop not set to 0. scrollTop=" + htmlElement.scrollTop + " offsetHeight=" + htmlElement.offsetHeight + " scrollHeight=" + htmlElement.scrollHeight + " browser=" + navigator.userAgent);
495 		AjxDebug.dumpObj(AjxDebug.SCROLL, htmlElement.outerHTML);
496 		/*
497 			trying this hack for solution -
498 			explanation: The scroll bar does not appear if the scrollHeight of the div is bigger than the total height of the iframe and header together (i.e. if htmlElement.scrollHeight >= htmlElement.offsetHeight)
499 			If the scrollbar does not appear it's set to, and stays 0 when the scrollbar reappears due to resizing the iframe in _resetIframeHeight (which is later, I think always on timer).
500 			So what I do here is set the height of the iframe to very small (since the default is 150px), so the scroll bar disappears.
501 			it will reappear when we reset the size in _resetIframeHeight. I hope this will solve the issue.
502 		*/
503 		var iframe = this.getIframe();
504 		if (iframe) {
505 			iframe.style.height = "1px";
506 			AjxDebug.println(AjxDebug.SCROLL, "scrollTop after resetting it with the hack =" + htmlElement.scrollTop);
507 		}
508 
509 	}
510 
511 	if (!bubblesCreated) {
512 		this._createBubbles();
513 		appCtxt.notifyZimlets("onMsgView", [msg, oldMsg, this]);
514 	}
515 
516 	if (!msg.isDraft && msg.readReceiptRequested) {
517 		this._controller.sendReadReceipt(msg);
518 	}
519 };

521 // This is needed for Gecko only: for some reason, clicking on a local link will
522 // open the full Zimbra chrome in the iframe :-( so we fake a scroll to the link
523 // target here. (bug 7927)
524 ZmMailMsgView.__localLinkClicked =
525 function(msgView, ev) {
526 	// note that this function is called in the context of the link ('this' is an A element)
527 	var id = this.getAttribute("href");
528 	var el = null;
529 	var doc = this.ownerDocument;
530 
531 	if (id.substr(0, 1) == "#") {
532 		id = id.substr(1);
533 		el = doc.getElementById(id);
534 		if (!el) {
535 			try {
536 				el = doc.getElementsByName(id)[0];
537 			} catch(ex) {}
538 		}
539 		if (!el) {
540 			id = decodeURIComponent(id);
541 			el = doc.getElementById(id);
542 			if (!el) {
543 				try {
544 					el = doc.getElementsByName(id)[0];
545 				} catch(ex) {}
546 			}
547 		}
548 	}
549 
550 	// attempt #1: doesn't work at all -- we're not scrolling with the IFRAME :-(
551 	// 		if (el) {
552 	// 			var pos = Dwt.getLocation(el);
553 	// 			doc.contentWindow.scrollTo(pos.x, pos.y);
554 	// 		}
555 
556 	// attempt #2: works pretty well, but the target node will showup at the bottom of the frame
557 	// 		var foo = doc.createElement("a");
558 	// 		foo.href = "#";
559 	// 		foo.innerHTML = "foo";
560 	// 		el.parentNode.insertBefore(foo, el);
561 	// 		foo.focus();
562 
563 	// the final monstrosity: scroll the containing DIV
564 	// (that is the whole msgView).  Note we have to take
565 	// into account the headers, "display images", etc --
566 	// so we add iframe.offsetTop/Left.
567 	if (el) {
568 		var div = msgView.getHtmlElement();
569 		var iframe = msgView.getIframe();
570 		var pos = Dwt.getLocation(el);
571 		div.scrollTop = pos.y + iframe.offsetTop - 20; // fuzz factor necessary for unknown reason :-(
572 		div.scrollLeft = pos.x + iframe.offsetLeft;
573 	}
574 	if (ev) {
575 		ev.stopPropagation();
576 		ev.preventDefault();
577 	}
578 	return false;
579 };

581 ZmMailMsgView.prototype.hasValidHref =
582 function (node) {
583 	// Bug 22958: IE can throw when you try and get the href if it doesn't like
584 	// the value, so we wrap the test in a try/catch.
585 	// hrefs formatted like http://www.name@domain.com can cause this to happen.
586 	try {
587 		var href = node.href;
588 		return ZmMailMsgView._URL_RE.test(href) || ZmMailMsgView._MAILTO_RE.test(href);
589 	} catch (e) {
590 		return false;

639 ZmMailMsgView.prototype._checkImgInAttachments =
640 function(img) {
641     if (!this._msg) { return; }

652 	}

674 			}

677 			if (filename == att.fileName) {

680 			}
681 		}

764 /**
765  * Reverses the work of the (server-side) defanger, so that images are displayed.
766  * 
767  * @param {ZmMailMsg}	msg			mail message
768  * @param {Element}		elem		element to be checked (img)
769  * @param {string}		aname		attribute name
770  * @param {boolean}		external	if true, look only for external images
771  * 
772  * @return	true if the image is external
773  */
774 ZmMailMsgView.__unfangInternalImage =
775 function(msg, elem, aname, external) {
776 	
777 	var avalue, pnsrc;
778 	try {
779 		if (external) {
780 			avalue = elem.getAttribute("df" + aname);
781 		}
782 		else {
783 			pnsrc = avalue = elem.getAttribute("pn" + aname);
784 			avalue = avalue || elem.getAttribute(aname);
785 		}
786 	}
787 	catch(e) {
788 		AjxDebug.println(AjxDebug.DATA_URI, "__unfangInternalImage: couldn't access attribute " + aname);
789 	}
790 
791 	if (avalue) {
792 		if (avalue.substr(0,4) == "cid:") {
793 			var cid = "<" + AjxStringUtil.urlComponentDecode(avalue.substr(4)) + ">"; // Parse percent-escaping per bug #52085 (especially %40 to @)
794 			avalue = msg.getContentPartAttachUrl(ZmMailMsg.CONTENT_PART_ID, cid);
795 			if (avalue) {
796 				elem.setAttribute(aname, avalue);
797 			}
798 			return false;
799 		} else if (avalue.substring(0,4) == "doc:") {
800 			avalue = [appCtxt.get(ZmSetting.REST_URL), ZmFolder.SEP, avalue.substring(4)].join('');
801 			if (avalue) {
802 				elem.setAttribute(aname, avalue);
803 				return false;
804 			}
805 		} else if (pnsrc) { // check for content-location verison
806 			avalue = msg.getContentPartAttachUrl(ZmMailMsg.CONTENT_PART_LOCATION, avalue);
807 			if (avalue) {
808 				elem.setAttribute(aname, avalue);
809 				return false;
810 			}
811 		} else if (avalue.substring(0,5) == "data:") {
812 			return false;
813 		}
814 		return true;	// not recognized as inline img
815 	}
816 	return false;
817 };
