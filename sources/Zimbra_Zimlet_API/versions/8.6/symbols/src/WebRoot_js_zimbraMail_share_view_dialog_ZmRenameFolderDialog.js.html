 26 /**
 27  * Creates a rename folder dialog.
 28  * @class
 29  * This class represents a rename folder dialog.
 30  * 
 31  * @param	{DwtComposite}	parent		the parent
 32  * @param	{String}	className		the class name
 33  *  
 34  * @extends		ZmDialog
 35  */
 36 ZmRenameFolderDialog = function(parent, className) {
 37 
 38 	ZmDialog.call(this, {parent:parent, className:className, title:ZmMsg.renameFolder, id:"RenameFolderDialog"});
 39 
 40 	this._setNameField(this._nameFieldId);
 41 };
 42 
 43 ZmRenameFolderDialog.prototype = new ZmDialog;
 44 ZmRenameFolderDialog.prototype.constructor = ZmRenameFolderDialog;
 45 
 46 ZmRenameFolderDialog.prototype.toString = 
 47 function() {
 48 	return "ZmRenameFolderDialog";
 49 };
 50 
 51 /**
 52  * Pops-up the dialog.
 53  * 
 54  * @param	{ZmFolder}		folder		the folder
 55  * @param	{Object}		[source]	(not used)
 56  */
 57 ZmRenameFolderDialog.prototype.popup =
 58 function(folder, source) {
 59 	ZmDialog.prototype.popup.call(this);
 60 	var title = (folder.type == ZmOrganizer.SEARCH) ? ZmMsg.renameSearch : ZmMsg.renameFolder;
 61 	this.setTitle(title + ': ' + folder.getName(false, ZmOrganizer.MAX_DISPLAY_NAME_LENGTH));
 62 	this._nameField.value = folder.getName(false, null, true);
 63 	this._folder = folder;
 64 };
 65 
 66 ZmRenameFolderDialog.prototype._contentHtml = 
 67 function() {
 68 	this._nameFieldId = this._htmlElId + "_name";
 69 	var subs = {id:this._htmlElId, newLabel:ZmMsg.newName};
 70 	return AjxTemplate.expand("share.Dialogs#ZmRenameDialog", subs);
 71 };
 72 
 73 ZmRenameFolderDialog.prototype._okButtonListener =
 74 function(ev) {
 75 	var results = this._getFolderData();
 76 	if (results) {
 77 		DwtDialog.prototype._buttonListener.call(this, ev, results);
 78 	}
 79 };
 80 
 81 ZmRenameFolderDialog.prototype._getFolderData =
 82 function() {
 83 	// check name for presence and validity
 84 	var name = AjxStringUtil.trim(this._nameField.value);
 85 	var msg = ZmFolder.checkName(name, this._folder.parent);
 86 
 87 	// make sure another folder with this name doesn't already exist at this level
 88 	if (!msg) {
 89 		var folder = this._folder.parent.getByName(name);
 90         var folderType = appCtxt.getFolderTree(appCtxt.getActiveAccount()).getFolderTypeByName(name);
 91         if (folder && (folder.id != this._folder.id)) {
 92 			msg = AjxMessageFormat.format(ZmMsg.errorAlreadyExists, [name,ZmMsg[folderType.toLowerCase()]]);
 93 		}
 94 	}
 95 
 96 	return (msg ? this._showError(msg) : [this._folder, name]);
 97 };
 98 
 99 ZmRenameFolderDialog.prototype._enterListener =
100 function(ev) {
101 	var results = this._getFolderData();
102 	if (results) {
103 		this._runEnterCallback(results);
104 	}
105 };
