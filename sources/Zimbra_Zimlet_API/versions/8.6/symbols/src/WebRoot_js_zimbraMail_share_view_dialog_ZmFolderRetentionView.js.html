 40 // Data to populate and process the custom units and values
 41 ZmFolderRetentionView._CustomUnitsToDays = { year: 366, month:31, week:7, day:1};
 42 ZmFolderRetentionView._CustomUnits = [
 43     { id:"day",   label: ZmMsg.days.toLowerCase(),   days: ZmFolderRetentionView._CustomUnitsToDays.day},
 44     { id:"week",  label: ZmMsg.weeks.toLowerCase(),  days: ZmFolderRetentionView._CustomUnitsToDays.week},
 45     { id:"month", label: ZmMsg.months.toLowerCase(), days: ZmFolderRetentionView._CustomUnitsToDays.month},
 46     { id:"year",  label: ZmMsg.years.toLowerCase(),  days: ZmFolderRetentionView._CustomUnitsToDays.year}];

 69 ZmFolderRetentionView.prototype._handleFolderChange =
 70 function() {
 71     // Read the policies from the server, and add 'Custom'
 72     if (this._initialized) {
 73         this._dataBindComponents(this._organizer, ZmOrganizer.RETENTION_KEEP);
 74         this._dataBindComponents(this._organizer, ZmOrganizer.RETENTION_PURGE);
 75     } else {
 76         var systemPolicies = new ZmSystemRetentionPolicy();
 77         systemPolicies.getPolicies(this._processSystemPolicies.bind(this));
 78         this._setBusy(true);
 79     }
 80 }

 82 ZmFolderRetentionView.prototype._processSystemPolicies =
 83 function(systemKeepPolicies, systemPurgePolicies) {
 84 
 85     this._populatePolicySelect(ZmOrganizer.RETENTION_KEEP,  systemKeepPolicies);
 86     this._populatePolicySelect(ZmOrganizer.RETENTION_PURGE, systemPurgePolicies);
 87 
 88     this._dataBindComponents(this._organizer, ZmOrganizer.RETENTION_KEEP);
 89     this._dataBindComponents(this._organizer, ZmOrganizer.RETENTION_PURGE);
 90 
 91     this._initialized = true;
 92     this._setBusy(false);
 93 }

 96 ZmFolderRetentionView.prototype._dataBindComponents =
 97 function(organizer, policyElement) {
 98     var components = this._components[policyElement];
 99     components.policyEnable.checked = false;
100 
101     var policy = organizer.getRetentionPolicy(policyElement);
102     if (policy) {
103         // The organizer has a retention policy
104         components.policyEnable.checked = true;
105 
106         if (policy.type == "user") {
107             // Custom policy defined.
108             components.policySelect.selectedIndex = components.policySelect.options.length-1;
109             // parseInt will discard the unit
110             var lifetime = parseInt(policy.lifetime);
111             // In case someone used SOAP to specify a custom policy, convert it
112             // to days (which is the smallest unit we can handle via the UI).
113             var conversionFactor = 1;
114             // Intervals taken from DateUtil.java
115             var interval = policy.lifetime.slice(policy.lifetime.length-1);
116             switch (interval) {
117                 case  "d": conversionFactor =  1; break;
118                 case  "h": conversionFactor = 24; break;
119                 case  "m": conversionFactor = AjxDateUtil.MINUTES_PER_DAY; break;
120                 case  "s": conversionFactor = AjxDateUtil.SECONDS_PER_DAY; break;
121                 case "ms": conversionFactor = AjxDateUtil.MSEC_PER_DAY;    break;
122                 default  : conversionFactor = AjxDateUtil.SECONDS_PER_DAY; break;
123             }
124             var days = Math.floor((lifetime-1)/conversionFactor) + 1;
125 
126             // Convert lifetime to the best fit for unit and amount.  Start with year,
127             // proceed to smaller units.  If the amount in days is evenly divisible by the
128             // days for a unit, use that unit
129             for (var i = ZmFolderRetentionView._CustomUnits.length-1; i >= 0; i--) {
130                 if ((days >= ZmFolderRetentionView._CustomUnits[i].days) &&
131                     ((days % ZmFolderRetentionView._CustomUnits[i].days) == 0)) {
132                     components.customUnit.selectedIndex = i;
133                     components.customValue.value = days/ZmFolderRetentionView._CustomUnits[i].days;
134                     break;
135                 }
136             }
137         } else {
138             // System policy, find the match in the policy selection pull down
139             for (var i = 0; i < components.policySelect.options.length; i++) {
140                 if (components.policySelect.options[i].value == policy.id) {
141                     components.policySelect.selectedIndex = i;
142                     break;
143                 }
144             }
145             // Reset custom fields to their defaults
146             components.customUnit.selectedIndex   = 0;
147             components.customValue.value          = "";
148         }
149     }
150     if (!components.policyEnable.checked) {
151         // No policy of this type (keep/purge) reset the selects and input fields
152         components.policySelect.selectedIndex = 0;
153         // Default to the largest unit to reduce the chance of an inadvertent tiny deletion period
154         components.customUnit.selectedIndex   =  components.customUnit.options.length-1;
155         components.customValue.value          = "";
156     }
157     this._handleSelectionChange(policyElement);
158     this._handleEnableClick(policyElement);
159 }

162 /**  doSave will be invoked for each tab view.
163  *
164  * @param	{BatchCommand}	batchCommand	Accumulates updates from all tabs
165  * @param	{Object}	    saveState		Accumulates error messages and indication of any update
166  */
167 ZmFolderRetentionView.prototype.doSave =
168 function(batchCommand, saveState) {
169 	if (!this._handleErrorCallback) {
170 		this._handleErrorCallback = new AjxCallback(this, this._handleError);
171 	}
172 
173     var organizer = this._organizer;
174 
175     var initialErrorCount = saveState.errorMessage.length;
176     var newRetentionPolicy = { };
177     // Create policy objects from the UI fields, attach to the newRetentionPolicy variable
178     this._createPolicy(newRetentionPolicy, ZmOrganizer.RETENTION_KEEP,  saveState);
179     this._createPolicy(newRetentionPolicy, ZmOrganizer.RETENTION_PURGE, saveState);
180 
181     if (initialErrorCount == saveState.errorMessage.length) {
182         var keepPolicy  = organizer.getRetentionPolicy(ZmOrganizer.RETENTION_KEEP);
183         var purgePolicy = organizer.getRetentionPolicy(ZmOrganizer.RETENTION_PURGE);
184         if (organizer.policiesDiffer(keepPolicy,  newRetentionPolicy.keep) ||
185             organizer.policiesDiffer(purgePolicy, newRetentionPolicy.purge)) {
186             // Retention policy has changed
187             batchCommand.add(new AjxCallback(organizer, organizer.setRetentionPolicy,
188                              [newRetentionPolicy, null, this._handleErrorCallback]));
189             saveState.commandCount++;
190         }
191     }
192 
193 };

196 // Create a retention policy object from UI components for processing by the
197 // organizer.setRetentionPolicy
198 ZmFolderRetentionView.prototype._createPolicy =
199 function(retentionPolicy, policyElement, saveState) {
200     var components = this._components[policyElement];
201 
202     var policy;
203     if (components.policyEnable.checked) {
204         // A keep or Purge retention policy is defined
205         var policySelection = components.policySelect.options[components.policySelect.selectedIndex].value;
206         var policyType;
207         if (policySelection == "custom") {
208             policyType = "user";
209             var unit   = components.customUnit.options[components.customUnit.selectedIndex].value;
210             // Parse the custom value field to get the number of units
211             var invalidCustomValue = false;
212             var amountText = AjxStringUtil.trim(components.customValue.value);
213             if (!amountText) {
214                 invalidCustomValue = true;
215             } else {
216                 var amount = 0;
217                 var nonNumericIndex = amountText.search(/\D/);
218                 if (nonNumericIndex == -1) {
219                     amount = parseInt(amountText);
220                 }
221 
222                 if (amount <= 0) {
223                     invalidCustomValue = true;
224                 } else {
225                     var daysPerUnit = ZmFolderRetentionView._CustomUnitsToDays[unit];
226                     var lifetime = (amount * daysPerUnit).toString() + "d";
227                     policy = {type:"user", lifetime:lifetime};
228                 }
229             }
230             if (invalidCustomValue) {
231                 var  errorMessage = (policyElement == ZmOrganizer.RETENTION_KEEP) ?
232                                      ZmMsg.retentionKeepLifetimeAmount : ZmMsg.retentionPurgeLifetimeAmount;
233                 saveState.errorMessage.push(errorMessage);
234             }
235         } else {
236             policy = {type:"system", id:policySelection};
237         }
238     }
239     retentionPolicy[policyElement] = policy;
240 }

243 ZmFolderRetentionView.prototype._populatePolicySelect =
244 function(policyElement, systemPolicies) {
245     var components = this._components[policyElement];
246     components.policies = systemPolicies ? systemPolicies : [];
247 	
248 	var sorted = {};
249 	for (var i=0; i< components.policies.length; i++) {
250 		sorted[components.policies[i].name] =  components.policies[i].id;
251 	}
252 	
253 	var sortedKeys = AjxUtil.getHashKeys(sorted);
254 	sortedKeys.push(ZmMsg.custom);  //append custom to the end
255 	sorted[ZmMsg.custom] = "custom";
256     components.policySelect.options.length = 0;
257     for (var i = 0; i < sortedKeys.length; i++) {
258         components.policySelect.options[i] = new Option(sortedKeys[i], sorted[sortedKeys[i]]);
259     }
260 }

263 ZmFolderRetentionView.prototype._handleEnableClick =
264 function(policyElement) {
265     var components = this._components[policyElement];
266     var disabled = !components.policyEnable.checked;
267     components.policySelect.disabled = disabled;
268     components.customValue.disabled  = disabled;
269     components.customUnit.disabled   = disabled;
270 }
271 
272 ZmFolderRetentionView.prototype._handleSelectionChange = function(policyElement) {
273 
274     var components = this._components[policyElement],
275         policySelect   = components.policySelect,
276 	    selectedOption = policySelect.options[policySelect.selectedIndex],
277         policySelection = selectedOption.value,
278         visible = (policySelection == "custom");
279 
280     // Show hide the custom unit and values fields based on whether the policy
281     // selected is a system defined policy, or custom
282     components.customValue.style.visibility = visible ? "visible" : "hidden";
283     components.customUnit.style.visibility  = visible ? "visible" : "hidden";
284 
285 	// accessibility
286 	policySelect.setAttribute('aria-label', AjxMessageFormat.format(ZmMsg.policyTypeLabel, selectedOption.innerHTML));
287 };

289 ZmFolderRetentionView.prototype._createView =
290 function() {
291 	// Create html elements
292     this._id = Dwt.getNextId();
293     var params = {
294         id: this._id
295     }
296 
297     var container = this.getHtmlElement();
298     container.innerHTML = AjxTemplate.expand("share.Dialogs#ZmFolderRetentionView", params);
299 
300     this._components = {};
301     this._setupComponents(ZmOrganizer.RETENTION_KEEP,  this._components);
302     this._setupComponents(ZmOrganizer.RETENTION_PURGE, this._components);
303 
304     this._createBusyOverlay(container);
305     this._contentEl = container;
306 };

308 ZmFolderRetentionView.prototype._setupComponents =
309 function(policyElement, allComponents) {
310 
311     var components = {};
312     allComponents[policyElement] = components;
313 
314     components.policyEnable = document.getElementById(this._id + "_" + policyElement + "Checkbox");
315     components.policySelect = document.getElementById(this._id + "_" + policyElement);
316     components.customValue  = document.getElementById(this._id + "_" + policyElement + "Value");
317     components.customUnit   = document.getElementById(this._id + "_" + policyElement + "Unit");
318 
319     for (var i = 0; i < ZmFolderRetentionView._CustomUnits.length; i++) {
320         var unit = ZmFolderRetentionView._CustomUnits[i];
321         components.customUnit.options[i] = new Option(unit.label, unit.id);
322     }
323 
324 	this._tabGroup.addMember(components.policyEnable);
325 	this._tabGroup.addMember(components.policySelect);
326 	this._tabGroup.addMember(components.customValue);
327 	this._tabGroup.addMember(components.customUnit);
328 
329     Dwt.setHandler(components.policyEnable, "onclick", this._handleEnableClick.bind(this, policyElement));
330     Dwt.setHandler(components.policySelect, "onclick", this._handleSelectionChange.bind(this, policyElement));
331 }
