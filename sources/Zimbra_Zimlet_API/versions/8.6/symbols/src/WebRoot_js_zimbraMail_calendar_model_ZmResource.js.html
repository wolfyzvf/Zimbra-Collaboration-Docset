 22 /**
 23  * Creates a resource.
 24  * @class
 25  * This class represents a resource.
 26  * 
 27  * @param	{String}	id		the id
 28  * @param	{ZmList}	list	the list
 29  * @param	{constant}	resType		the resource type
 30  * 
 31  * @extends		ZmContact
 32  * @see		ZmCalBaseItem
 33  */
 34 ZmResource = function(id, list, resType) {
 35 	id = id ? id : Dwt.getNextId();
 36 	ZmContact.call(this, id, list, ZmItem.RESOURCE);
 37 	
 38 	this.resType = resType;
 39 };

 44 ZmResource.F_mail			= "email";

 49 ZmResource.ATTR_LOCATION	= "Location";
 50 ZmResource.ATTR_EQUIPMENT	= "Equipment";
 51 
 52 /**
 53 * Creates a resource from an XML node.
 54 *
 55 * @param node		a "calresource" XML node
 56 * @param args		args to pass to the constructor
 57 */
 58 ZmResource.createFromDom =
 59 function(node, args) {
 60 	var resource = new ZmResource(node.id, args.list);
 61 	resource._loadFromDom(node);
 62 	resource.resType = (resource.getAttr(ZmResource.F_type) == ZmResource.ATTR_LOCATION) ?
 63 						ZmCalBaseItem.LOCATION : ZmCalBaseItem.EQUIPMENT;
 64 	if (!resource.list) {
 65 		var calApp = appCtxt.getApp(ZmApp.CALENDAR);
 66 		resource.list = (resource.resType == ZmCalBaseItem.LOCATION) ? calApp.getLocations() :
 67 																   calApp.getEquipment();
 68 	}
 69 	
 70 	return resource;
 71 };

 81 /**
 82  * Checks if the resource is a location.
 83  * 
 84  * @return	{Boolean}	<code>true</code> if is location
 85  */
 86 ZmResource.prototype.isLocation =
 87 function() {
 88 	return (this.resType == ZmCalBaseItem.LOCATION);
 89 };

 91 /**
 92  * Gets the resource email.
 93  * 
 94  * @return	{String}	the email
 95  */
 96 ZmResource.prototype.getEmail =
 97 function() {
 98 	var attr = this.getAttr(ZmResource.F_mail);
 99 	return attr instanceof Array ? attr[0] : attr;
100 };
101 
102 /**
103  * Gets the resource full name.
104  * 
105  * @return	{String}	the full name
106  */
107 ZmResource.prototype.getFullName =
108 function() {
109 	return ( this.getAttr(ZmResource.F_name)
110             || this.getAttr(ZmResource.F_locationName) );
111 };
112 
113 /**
114  * Initializes from an email address.
115  *
116  * @param {AjxEmailAddress|String}	email	an email address object an email string
117  */
118 ZmResource.prototype.initFromEmail =
119 function(email) {
120 	if (email instanceof AjxEmailAddress) {
121 		this.setAttr(ZmResource.F_mail, email.getAddress());
122 		this.setAttr(ZmResource.F_name, email.getName());
123 	} else {
124 		this.setAttr(ZmResource.F_mail, email);
125 	}
126 };
127 
128 ZmResource.prototype.getAttendeeText =
129 function(type, shortForm) {
130 	var text = "";
131 	var name = this.getFullName();
132 	var email = this._lookupEmail || this.getEmail();
133 	if (shortForm) {
134 		text = name || email || "";
135 	} else {
136 		var e = new AjxEmailAddress(email, null, name);
137 		text = e.toString();
138 	}
139 	return text;
140 };
