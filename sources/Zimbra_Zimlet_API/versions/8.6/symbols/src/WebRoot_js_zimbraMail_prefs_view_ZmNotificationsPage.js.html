 74 ZmNotificationsPage.prototype.showMe = function() {
 75 	// default processing
 76 	var initialize = !this.hasRendered;
 77 	ZmPreferencesPage.prototype.showMe.apply(this, arguments);
 78 
 79 	// setup controls
 80 	var status = this._getValidatedDevice() ?
 81 			ZmNotificationsPageForm.CONFIRMED : ZmNotificationsPageForm.UNCONFIRMED;
 82 	this._form.setValue("DEVICE_EMAIL_CODE_STATUS_VALUE", status);
 83 	this._form.setValue("EMAIL", this._getEmailAddress());
 84 	this._form.update();
 85 
 86 	// load SMS data, if needed
 87 	if (initialize && this._form.getControl("DEVICE_EMAIL_REGION") != null) {
 88 		var locid = window.appRequestLocaleId ? "&locid=" + window.appRequestLocaleId : "";
 89 		var includes = [
 90 			[   appContextPath,
 91 				"/res/ZmSMS.js",
 92 				"?v=",cacheKillerVersion,
 93 				locid,
 94 				appDevMode ? "&debug=1" : ""
 95 			].join("")
 96 		];
 97 		var baseurl = null;
 98 		var callback = new AjxCallback(this, this._smsDataLoaded);
 99 		var proxy = null;
100 		AjxInclude(includes, baseurl, callback, proxy);
101 	}
102 };

160 ZmNotificationsPage.prototype._createControls = function() {
161 
162 	//cleanup
163 	this._cleanup();
164 
165 	// the following part is to do stuff that ZmPreferencesPage does (setting the origValue of the preference),
166 	// but simplified for this case (since we are not using all the stuff from ZmPreferencesPage, we are using DwtForm instead)
167 	var email = this._getEmailAddress();
168 	var settings = appCtxt.getSettings();
169 	var emailPrefId = ZmSetting.CAL_EMAIL_REMINDERS_ADDRESS;
170 	var pref = settings.getSetting(emailPrefId);
171 	// save the current value (for checking later if it changed)
172 	pref.origValue = email;
173 
174 	// create form controls
175 	this._form = this._setupCustomForm();
176 	this._form.reparentHtmlElement(this.getContentHtmlElement());
177 
178 	// make it look like email notification pref control is present
179 	// NOTE: This is needed so that the default pref saving works.
180 	// NOTE: This is used in conjunction with set/getFormValue.
181 	if (this._form.getControl("EMAIL")) {
182 		this._prefPresent = {};
183 		this._prefPresent[emailPrefId] = true;
184 	}
185 
186 	// finish setup
187 	this.setVisible(true);
188 	this.hasRendered = true;
189 };

259 ZmNotificationsPageForm.prototype.setSMSData = function(data) {
260 	this._smsData = data;
261 
262 	// setup regions control
263 	var regionMap = this._defineRegions(data);
264 	var button = this.getControl("DEVICE_EMAIL_REGION");
265 	var menu = this.__createRegionsMenu(button, regionMap);
266 	if (menu == null) {
267 		// always have at least one region
268 		menu = new DwtMenu({parent:button});
269 		this.__createRegionMenuItem(menu, ZmNotificationsPageForm.UNKNOWN, ZmMsg.unknown);
270 	}
271 	menu.addSelectionListener(this._regionSelectionListener);
272 	button.setMenu(menu);
273 
274 	// set default region
275 	this.setRegion(data.defaultRegionId);
276 };

278 ZmNotificationsPageForm.prototype.setRegion = function(regionId) {
279 	// decorate the region button
280 	var region = ZmNotificationsPage.REGIONS[regionId] || { id:ZmNotificationsPageForm.UNKNOWN, label:ZmMsg.unknown };
281 	var button = this.getControl("DEVICE_EMAIL_REGION");
282 	button.setText(region.label);
283 	button.setImage(region.image);
284 	button.setData(Dwt.KEY_ID, region.id);
285 
286 	// update the form
287 	this.setCarriers(regionId);
288 	this.setValue("DEFAULT_EMAIL_REGION", regionId);
289 	if (regionId == this._smsData.defaultRegionId && this._smsData.defaultCarrierId) {
290 		this.setValue("DEVICE_EMAIL_CARRIER", this._smsData.defaultCarrierId);
291 	}
292 	this.update();

295 ZmNotificationsPageForm.prototype.setCarriers = function(regionId) {
296 	var select = this.getControl("DEVICE_EMAIL_CARRIER");
297 	select.clearOptions();
298 	var region = ZmNotificationsPage.REGIONS[regionId] || {};
299 	var carriers = ZmNotificationsPageForm.__getRegionCarriers(region, true);
300 	carriers.sort(ZmNotificationsPageForm.__byLabel);
301 	for (var i = 0; i < carriers.length; i++) {
302 		var carrier = carriers[i];
303 		var image = carrier.image || carrier.region.image;
304 		select.addOption({displayValue:carrier.label, value:carrier.id, image:image});
305 	}
306 	select.addOption({displayValue:ZmMsg.custom, value:ZmNotificationsPageForm.CUSTOM, image:null});
307 };

309 ZmNotificationsPageForm.prototype.isCustom = function() {
310 	// user selected "Custom"
311 	var carrierId = this.getValue("DEVICE_EMAIL_CARRIER");
312 	if (carrierId == ZmNotificationsPageForm.CUSTOM) return true;
313 
314 	// any entry w/o an email pattern is also custom
315 	var carrier = ZmNotificationsPage.CARRIERS[carrierId];
316 	var hasPattern = Boolean(carrier && carrier.pattern);
317 	return !hasPattern;
318 };

320 ZmNotificationsPageForm.prototype.getEmailAddress = function() {
321 	if (this.isCustom()) {
322 		var number = this.getValue("DEVICE_EMAIL_CUSTOM_NUMBER");
323 		var address = this.getValue("DEVICE_EMAIL_CUSTOM_ADDRESS");
324 		return number && address ? [number,address].join("@") : "";
325 	}
326 
327 	var phone = ZmNotificationsPageForm.normalizePhoneNumber(this.getValue("DEVICE_EMAIL_PHONE"));
328 	var carrier = ZmNotificationsPage.CARRIERS[this.getValue("DEVICE_EMAIL_CARRIER")];
329 	return phone ? AjxMessageFormat.format(carrier.pattern, [phone]) : "";
330 };

332 ZmNotificationsPageForm.prototype.getCodeStatus = function() {
333 	// is there anything to do?
334 	var control = this.getControl("DEVICE_EMAIL_CODE_STATUS");
335 	if (!control) return "";
336 
337 	// remove other status class names
338 	var controlEl = control.getHtmlElement();
339 	Dwt.delClass(controlEl, ZmNotificationsPageForm.CONFIRMED);
340 	Dwt.delClass(controlEl, ZmNotificationsPageForm.PENDING);
341 	Dwt.delClass(controlEl, ZmNotificationsPageForm.UNCONFIRMED);
342 
343 	// add appropriate class name
344 	var status = this.get("DEVICE_EMAIL_CODE_STATUS_VALUE");
345 	Dwt.addClass(controlEl, status);
346 
347 	// format status text
348 	if (status == ZmNotificationsPageForm.CONFIRMED) {
349 		var email = this._getValidatedDevice();
350 		if (email) {
351 			var pattern = ZmMsg.deviceEmailNotificationsVerificationStatusConfirmed;
352 			return AjxMessageFormat.format(pattern, [email]);
353 		}
354 		// default back to unconfirmed
355 		Dwt.delClass(controlEl, status, ZmNotificationsPageForm.UNCONFIRMED);
356 	}
357 	return status == ZmNotificationsPageForm.PENDING ?
358 			ZmMsg.deviceEmailNotificationsVerificationStatusPending :
359 			ZmMsg.deviceEmailNotificationsVerificationStatusUnconfirmed
360 			;
361 };

363 ZmNotificationsPageForm.prototype.getPhoneHint = function() {
364 	var carrierId = this.getValue("DEVICE_EMAIL_CARRIER");
365 	var carrier = ZmNotificationsPage.CARRIERS[carrierId];
366 	var carrierHint = carrier && carrier.hint;
367 	var email = this.getEmailAddress();
368 	var emailHint = email && AjxMessageFormat.format(ZmMsg.deviceEmailNotificationsPhoneNumber, [email]);
369 	if (carrierHint || emailHint) {
370 		if (!carrierHint) return emailHint;
371 		if (!emailHint) return carrierHint;
372 		return AjxMessageFormat.format(ZmMsg.deviceEmailNotificationsCarrierEmailHint, [emailHint,carrierHint]);
373 	}
374 	return "";
375 };

377 ZmNotificationsPageForm.normalizePhoneNumber = function(phone) {
378 	if (phone) {
379 		phone = phone.replace(/[a-z]/gi, ZmNotificationsPageForm.__letters2Numbers);
380 		phone = phone.replace(/[^+#\*0-9]/g, "");
381 	}
382 	return phone;
383 };

459 ZmNotificationsPageForm.prototype._defineRegions = function(data) {
460 	// define regions
461 	var regionMap = {};
462 	for (var id in data) {
463 		// do we care about this entry?
464 		if (!id.match(/^region_/)) continue;
465 
466 		// take apart message key
467 		var parts = id.split(/[_\.]/);
468 		var prop = parts[parts.length - 1];
469 
470 		// set region info
471 		var regions = regionMap;
472 		var regionIds = parts[1].split("/"), region;
473 		for (var i = 0; i < regionIds.length; i++) {
474 			var regionId = regionIds[i];
475 			if (!regions[regionId]) {
476 				regions[regionId] = ZmNotificationsPage.REGIONS[regionId] = { id: regionId };
477 			}
478 			region = regions[regionId];
479 			if (i < regionIds.length - 1 && !region.regions) {
480 				region.regions = {};
481 			}
482 			regions = region.regions;
483 		}
484 
485 		// store property
486 		region[prop] = data[id];
487 	}
488 
489 	// define carriers
490 	this._defineCarriers(data);
491 
492 	return regionMap;
493 };

495 ZmNotificationsPageForm.prototype._defineCarriers = function(data) {
496 	for (var id in data) {
497 		// do we care about this entry?
498 		if (!id.match(/^carrier_/)) continue;
499 
500 		// take apart message key
501 		var s = id.split(/[_\.]/);
502 		var prop = s[s.length - 1];
503 
504 		// set carrier info
505 		var carrierId = s.slice(1, 3).join("_");
506 		var carrier = ZmNotificationsPage.CARRIERS[carrierId];
507 		if (!carrier) {
508 			var regionId = s[1];
509 			var region = ZmNotificationsPage.REGIONS[regionId];
510 			if (!region.carriers) {
511 				region.carriers = {};
512 			}
513 			carrier = region.carriers[carrierId] = ZmNotificationsPage.CARRIERS[carrierId] = {
514 				id: carrierId, region: region
515 			};
516 		}
517 
518 		// store property
519 		carrier[prop] = data[id];
520 	}
521 	return ZmNotificationsPage.CARRIERS;
522 };

590 ZmNotificationsPageForm.prototype._handleValidateCodeResponse = function(resp) {
591 	var success = AjxUtil.get(resp.getResponse(), "VerifyCodeResponse", "success") == "1";
592 	var params = {
593 		msg: success ?
594 				ZmMsg.deviceEmailNotificationsVerificationCodeValidateSuccess :
595 				ZmMsg.deviceEmailNotificationsVerificationCodeValidateFailure,
596 		level: success ? ZmStatusView.LEVEL_INFO : ZmStatusView.LEVEL_CRITICAL
597 	};
598 	appCtxt.setStatusMsg(params);
599 
600 	// NOTE: Since the preference values only come in at launch time,
601 	// NOTE: manually set the confirmed email address so that we can
602 	// NOTE: display the correct confirmed code status text
603 	if (success) {
604 		appCtxt.set(ZmSetting.CAL_DEVICE_EMAIL_REMINDERS_ADDRESS, this.getEmailAddress());

648 ZmNotificationsPageForm.prototype.__createRegionsMenu = function(parent, regionMap, parentRegion) {
649 	var regions = AjxUtil.values(regionMap, ZmNotificationsPageForm.__acceptRegion);
650 	if (regions.length == 0) return null;
651 
652 	var menu = new DwtMenu({parent:parent});
653 
654 	regions.sort(ZmNotificationsPageForm.__byLabel);
655 	for (var i = 0; i < regions.length; i++) {
656 		var region = regions[i];
657 		// add entry for this region
658 		var image = parent instanceof DwtMenuItem ? region.image : null;
659 		var menuItem = this.__createRegionMenuItem(menu, region.id, region.label, image);
660 		// add sub-regions
661 		var subMenu = region.regions && this.__createRegionsMenu(menuItem, region.regions, region);
662 		if (subMenu && subMenu.getItemCount() > 0) {
663 			subMenu.addSelectionListener(this._regionSelectionListener);
664 			menuItem.setMenu(subMenu);
665 		}
666 	}
667 
668 	// NOTE: Since only the call to this method *within* this method
669 	// NOTE: passes in a parentRegion, we'll only add a general entry for
670 	// NOTE: the parent region at a sub-level and never at the top level.
671 	if (parentRegion) {
672 		var hasCarriers = parentRegion.carriers && parentRegion.carriers.length > 0;
673 		var menuItemCount = menu.getItemCount();
674 		if (hasCarriers || menuItemCount > 1) {
675 			this.__createRegionMenuItem(menu, parentRegion.id, parentRegion.label, parentRegion.image, 0);
676 			if (menuItemCount > 1) {
677 				new DwtMenuItem({parent:menu,style:DwtMenuItem.SEPARATOR_STYLE,index:1});
678 			}
679 		}
680 	}
681 
682 	return menu;
683 };

693 ZmNotificationsPageForm.__acceptRegion = function(regionId, regionMap) {
694 	var region = regionMap[regionId];
695 	var hasRegions = false;
696 	for (var id in region.regions) {
697 		if (ZmNotificationsPageForm.__acceptRegion(id, region.regions)) {
698 			hasRegions = true;
699 			break;
700 		}
701 	}
702 	var hasCarriers = ZmNotificationsPageForm.__getRegionCarriers(region).length > 0;
703 	return hasRegions || hasCarriers;
704 };

706 ZmNotificationsPageForm.__getRegionCarriers = function(region, recurse) {
707 	if (region.carriers && !(region.carriers instanceof Array)) {
708 		region.carriers = AjxUtil.values(region.carriers);
709 		region.carriers.sort(ZmNotificationsPageForm.__byLabel);
710 	}
711 	var carriers = region.carriers || [];
712 	if (recurse && region.regions) {
713 		for (var regionId in region.regions) {
714 			carriers = carriers.concat(ZmNotificationsPageForm.__getRegionCarriers(region.regions[regionId], true));
715 		}
716 	}
717 	return carriers;
718 };
