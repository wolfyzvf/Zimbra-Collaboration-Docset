 64 /**
 65  * Shows the contact.
 66  *
 67  * @param	{ZmContact}	contact		the contact
 68  * @param	{Boolean}	isDirty		<code>true</code> to mark the contact as dirty
 69  * @param	{Boolean}	isBack		<code>true</code> in case of DL, we load (or reload) all the DL info, so we have to call back here. isBack indicates this is after the reload so we can continue.
 70  */
 71 ZmContactController.prototype.show =
 72 function(contact, isDirty, isBack) {
 73 	if (contact.id && contact.isDistributionList() && !isBack) {
 74 		//load the full DL info available for the owner, for edit.
 75 		var callback = this.show.bind(this, contact, isDirty, true); //callback HERE
 76 		contact.clearDlInfo();
 77 		contact.gatherExtraDlStuff(callback);
 78 		return;
 79 	}
 80 
 81 	this._contact = contact;
 82 	if (isDirty) {
 83 		this._contactDirty = true;
 84 	}
 85 	this.setList(contact.list);
 86 
 87 	if (!this.getCurrentToolbar()) {
 88 		this._initializeToolBar(this._currentViewId);
 89 	}
 90 	this._resetOperations(this.getCurrentToolbar(), 1); // enable all buttons
 91 
 92 	this._createView(this._currentViewId);
 93 
 94 	this._setViewContents();
 95 	this._initializeTabGroup(this._currentViewId);
 96 	this._app.pushView(this._currentViewId);
 97 	this.updateTabTitle();
 98 };

100 ZmContactController.prototype._createView =
101 function(viewId) {
102 	if (this._contactView) {
103 		return;
104 	}
105 	var view = this._contactView = this._createContactView();
106 	//Note - I store this in this._view just to be consistent with certain calls such as for ZmBaseController.prototype._initializeTabGroup. Even though there's no real reason to keep an array of views per type since each controller would only have one view and therefor one type
107 	this._view[viewId] = view;
108 
109 	var callbacks = {};
110 		callbacks[ZmAppViewMgr.CB_PRE_HIDE] = this._preHideCallback.bind(this);
111 		callbacks[ZmAppViewMgr.CB_PRE_UNLOAD] = this._preUnloadCallback.bind(this);
112 		callbacks[ZmAppViewMgr.CB_POST_SHOW] = this._postShowCallback.bind(this);
113 	var elements = this.getViewElements(null, view, this._toolbar[viewId]);
114 
115 	this._app.createView({	viewId:		viewId,
116 							viewType:	this._currentViewType,
117 							elements:	elements, 
118 							hide:		this._elementsToHide,
119 							controller:	this,
120 							callbacks:	callbacks,
121 							tabParams:	this._getTabParams()});
122 };

124 ZmContactController.prototype._postShowCallback =
125 function() {
126 	//have to call it since it's overridden in ZmBaseController to do nothing.
127 	ZmController.prototype._postShowCallback.call(this);
128 	if (this._contactView.postShow) {
129 		this._contactView.postShow();
130 	}
131 };

154 ZmContactController.prototype.updateTabTitle =
155 function() {
156 	var	tabTitle = this._contactView._getFullName(true);
157 	if (!tabTitle) {
158 		tabTitle = this._getDefaultTabText();

161 
162 	appCtxt.getAppViewMgr().setTabTitle(this._currentViewId, tabTitle);

172 ZmContactController.prototype.handleKeyAction =
173 function(actionCode) {
174 	DBG.println("ZmContactController.handleKeyAction");
175 	switch (actionCode) {
176 
177 		case ZmKeyMap.SAVE:
178 			var tb = this.getCurrentToolbar();
179 			var saveButton = tb.getButton(ZmOperation.SAVE);
180 			if (!saveButton.getEnabled()) {
181 				break;
182 			}
183 			this._saveListener();
184 			break;
185 
186 		case ZmKeyMap.CANCEL:
187 			this._cancelListener();
188 			break;
189 	}
190 	return true;
191 };

245 /**
246  * @private
247  */
248 ZmContactController.prototype._initializeToolBar =
249 function(view) {
250 	ZmListController.prototype._initializeToolBar.call(this, view);
251 
252 	var tb = this._toolbar[view];
253 
254 	// change the cancel button to "close" if editing existing contact
255 	var cancelButton = tb.getButton(ZmOperation.CANCEL);
256 	if (this._contact.id == undefined || (this._contact.isGal && !this._contact.isDistributionList())) {
257 		cancelButton.setText(ZmMsg.cancel);
258 	} else {
259 		cancelButton.setText(ZmMsg.close);
260 	}
261 
262 	var saveButton = tb.getButton(ZmOperation.SAVE);
263 	if (saveButton) {
264 		saveButton.setToolTipContent(ZmMsg.saveContactTooltip);
265 	}
266 
267 	appCtxt.notifyZimlets("initializeToolbar", [this._app, tb, this, view], {waitUntilLoaded:true});
268 };

321 /**
322  * @private
323  */
324 ZmContactController.prototype._resetOperations =
325 function(parent, num) {
326 	if (!parent) return;
327 	if (!this._contact.id) {
328 		// disble all buttons except SAVE and CANCEL
329 		parent.enableAll(false);
330 		parent.enable([ZmOperation.SAVE, ZmOperation.CANCEL], true);
331 	}
332 	else if (this._contact.isGal) {
333 		//GAL item or DL.
334 		parent.enableAll(false);
335 		parent.enable([ZmOperation.SAVE, ZmOperation.CANCEL], true);
336 		//for editing a GAL contact - need to check special case for DLs that are owned by current user and if current user has permission to delete on this domain.
337 		var deleteAllowed = ZmContactList.deleteGalItemsAllowed([this._contact]);
338 		parent.enable(ZmOperation.DELETE, deleteAllowed);
339 	} else if (this._contact.isReadOnly()) {
340 		parent.enableAll(true);
341 		parent.enable(ZmOperation.TAG_MENU, false);
342 	} else {
343 		ZmListController.prototype._resetOperations.call(this, parent, num);
344 	}
345 };

347 /**
348  * @private
349  */
350 ZmContactController.prototype._saveListener =
351 function(ev, bIsPopCallback) {
352 	var fileAsChanged = false;
353 	var view = this._contactView;
354 	if (view instanceof DwtForm)
355 		view.validate();
356 	if (!view.isValid()) {
357 		var invalidItems = view.getInvalidItems();
358 		for (var i=0; i<invalidItems.length; i++) {
359 			msg = view.getErrorMessage(invalidItems[i]);
360 			if (AjxUtil.isString(msg)) {
361 				//var msg = ZmMsg.errorSaving + (ex ? (":<p>" + ex) : ".");
362 				msg = msg ? AjxMessageFormat.format(ZmMsg.errorSavingWithMessage, msg) : ZmMsg.errorSaving;
363 				var ed = appCtxt.getMsgDialog();
364 				if (ed) {
365 					ed.setMessage(msg, DwtMessageDialog.CRITICAL_STYLE);
366 					ed.popup();
367 				} else {
368 					appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_CRITICAL);
369 				}
370 				return;
371 			}
372 		}
373 		return;
374 	}
375 
376 	var mods = view.getModifiedAttrs();
377 	view.enableInputs(false);
378 
379 	var contact = view.getContact();
380 	if (mods && AjxUtil.arraySize(mods) > 0) {
381 
382 		// bug fix #22041 - when moving betw. shared/local folders, dont modify
383 		// the contact since it will be created/deleted into the new folder
384 		var newFolderId = mods[ZmContact.F_folderId];
385 		var newFolder = newFolderId ? appCtxt.getById(newFolderId) : null;
386 		if (contact.id != null && newFolderId &&
387 			(contact.isShared() || (newFolder && newFolder.link)) && !contact.isGal)
388 		{
389 			// update existing contact with new attrs
390 			for (var a in mods) {
391 				if (a != ZmContact.F_folderId && a != ZmContact.F_groups) {
392 					contact.attr[a] = mods[a];
393 				}
394 			}
395 			// set folder will do the right thing for this shared contact
396 			contact._setFolder(newFolderId);
397 		}
398 		else
399 		{
400 			if (contact.id && (!contact.isGal || contact.isDistributionList())) {
401 				if (view.isEmpty()) { //If contact empty, alert the user
402 					var ed = appCtxt.getMsgDialog();
403 					ed.setMessage(ZmMsg.emptyContactSave, DwtMessageDialog.CRITICAL_STYLE);
404 					ed.popup();
405 					view.enableInputs(true);
406 					bIsPopCallback = true;
407 				} else {
408 					var contactFileAsBefore = ZmContact.computeFileAs(contact);
409 					var contactFileAsAfter = ZmContact.computeFileAs(AjxUtil.hashUpdate(AjxUtil.hashCopy(contact.getAttrs()), mods, true));
410 					this._doModify(contact, mods);
411 					if (contactFileAsBefore.toLowerCase()[0] !== contactFileAsAfter.toLowerCase()[0]) {
412 						fileAsChanged = true;
413 					}
414 				}
415 			} else {
416 				var isEmpty = true;
417 				for (var a in mods) {
418 					if (mods[a]) {
419 						isEmpty = false;
420 						break;
421 					}
422 				}
423 				if (isEmpty) {
424 					var msg = this._isGroup()
425 						? ZmMsg.emptyGroup
426 						: ZmMsg.emptyContact;
427 					appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_WARNING);
428 				}
429 				else {
430 					if (contact.isDistributionList()) {
431 						contact.create(mods);
432 					}
433 					else {
434 						var clc = AjxDispatcher.run("GetContactListController");
435 						var list = (clc && clc.getList()) || new ZmContactList(null);
436 						fileAsChanged = true;
437 						this._doCreate(list, mods);
438 					}
439 				}
440 			}
441 		}
442 	} else {
443 		if (contact.isDistributionList()) {
444 			//in this case, we need to pop the view since we did not call the server to modify the DL.
445 			this.popView();
446 		}
447 		// bug fix #5829 - differentiate betw. an empty contact and saving
448 		//                 an existing contact w/o editing
449 		if (view.isEmpty()) {
450 			var msg = this._isGroup()
451 				? ZmMsg.emptyGroup
452 				: ZmMsg.emptyContact;
453 			appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_WARNING);
454 		} else {
455 			var msg = contact.isDistributionList()
456 				? ZmMsg.dlSaved
457 				: this._isGroup()
458 				? ZmMsg.groupSaved
459 				: ZmMsg.contactSaved;
460 			appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_INFO);
461 		}
462 	}
463 
464 	if (!bIsPopCallback && !contact.isDistributionList()) {
465 		//in the DL case it might fail so wait to pop the view when we receive success from server.
466 		this.popView();
467 	}
468 	else {
469 		view.enableInputs(true);
470 	}
471 	if (fileAsChanged) // bug fix #45069 - if the contact is new, change the search to "all" instead of displaying contacts beginning with a specific letter
472 		ZmContactAlphabetBar.alphabetClicked(null);
473 
474     return true;
475 };

507 /**
508  * @private
509  */
510 ZmContactController.prototype._doDelete =
511 function(items, hardDelete, attrs, skipPostProcessing) {
512 	ZmListController.prototype._doDelete.call(this, items, hardDelete, attrs);
513 	if (items.isDistributionList()) { //items === this._contact here
514 		//do not pop the view as we are not sure the user will confirm the hard delete
515 		return;
516 	}
517 	appCtxt.getApp(ZmApp.CONTACTS).updateIdHash(items, true);
518 
519 	if (!skipPostProcessing) {
520 		// disable input fields (to prevent blinking cursor from bleeding through)
521 		this._contactView.enableInputs(false);
522 		this._app.popView(true);
523 	}
524 };

526 /**
527  * @private
528  */
529 ZmContactController.prototype._preHideCallback =
530 function(view, force) {
531 	ZmController.prototype._preHideCallback.call(this);
532 
533 	if (force) return true;
534 
535 	var view = this._contactView;
536 	if (!view.isDirty()) {
537 		view.cleanup();
538 		return true;
539 	}
540 
541 	var ps = this._popShield = appCtxt.getYesNoCancelMsgDialog();
542 	ps.reset();
543 	ps.setMessage(ZmMsg.askToSave, DwtMessageDialog.WARNING_STYLE);
544 	ps.registerCallback(DwtDialog.YES_BUTTON, this._popShieldYesCallback, this);
545 	ps.registerCallback(DwtDialog.NO_BUTTON, this._popShieldNoCallback, this);
546 	ps.popup(view._getDialogXY());
547 
548 	return false;
549 };
