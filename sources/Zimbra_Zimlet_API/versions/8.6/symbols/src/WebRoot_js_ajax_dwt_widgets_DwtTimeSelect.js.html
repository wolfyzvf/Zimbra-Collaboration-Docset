 50 DwtTimeSelect.getDateFromFields =
 51 function(hours, minutes, ampm, date) {
 52 	hours = Number(hours);
 53 	if (ampm) {
 54 		if (ampm == "AM" || ampm === 0) {
 55 			hours = (hours == 12) ? 0 : hours;
 56 		} else if (ampm == "PM" || ampm == 1) {
 57 			hours = (hours < 12) ? hours + 12 : hours;
 58 		}
 59 	}
 60 	date = date ? date : new Date();
 61 	date.setHours(hours, Number(minutes), 0, 0);
 62 	return date;
 63 };

 65 DwtTimeSelect.parse =
 66 function(timeString) {
 67     var date;
 68 	var lTimeString = timeString.toLowerCase();
 69 	if (lTimeString === AjxMsg.midnight.toLowerCase() || lTimeString === AjxMsg.noon.toLowerCase()) {
 70 		date = new Date();
 71 		date.setMinutes(0);
 72 		date.setSeconds(0);
 73 		date.setHours(lTimeString === AjxMsg.noon.toLowerCase() ? 12 : 0);
 74 	} else {
 75 		var timeFormatter = AjxDateFormat.getTimeInstance(AjxDateFormat.SHORT);    
 76 		date = timeFormatter.parse(timeString) || AjxDateFormat.parseTime(timeString);
 77 	}
 78     return date;
 79 };
 80 
 81 DwtTimeSelect.format =
 82 function(date) {
 83 	if (date.getHours() == 0 && date.getMinutes() == 0) {
 84 		return AjxMsg.midnight;
 85 	} else if (date.getHours() == 12 && date.getMinutes() == 0) {
 86 		return AjxMsg.noon;
 87 	} else {
 88 		return AjxDateFormat.getTimeInstance(AjxDateFormat.SHORT).format(date);
 89 	}
 90 };

 92 /**
 93 * Adjust an appt's start or end based on changes to the other one. If the user changes
 94 * the start time, change the end time so that the appt duration is maintained. If the
 95 * user changes the end time, we leave things alone.
 96 *
 97 * @param ev					[Event]				UI event from a DwtSelect
 98 * @param startSelect		[DwtTimeSelect]		start time select
 99 * @param endSelect			[DwtTimeSelect]		end time select
100 * @param startDateField		[element]			start date field
101 * @param endDateField		[element]			end date field
102 */
103 DwtTimeSelect.adjustStartEnd =
104 function(ev, startSelect, endSelect, startDateField, endDateField) {
105 	var select = ev._args.selectObj;
106 	var startDate = AjxDateUtil.simpleParseDateStr(startDateField.value);
107 	var endDate = AjxDateUtil.simpleParseDateStr(endDateField.value);
108 	var startDateOrig = startDateField.value;
109 	var endDateOrig = endDateField.value;
110 	if (select.id == DwtTimeSelect.START) {
111 		var hours = (select.compId == DwtTimeSelect.HOUR) ? ev._args.oldValue : startSelect.getHours();
112 		var minutes = (select.compId == DwtTimeSelect.MINUTE) ? ev._args.oldValue : startSelect.getMinutes();
113 		var ampm = (select.compId == DwtTimeSelect.AMPM) ? ev._args.oldValue : startSelect.getAmPm();
114 		var oldStartDateMs = DwtTimeSelect.getDateFromFields(hours, minutes, ampm, startDate).getTime();
115 		var newStartDateMs = DwtTimeSelect.getDateFromFields(startSelect.getHours(), startSelect.getMinutes(), startSelect.getAmPm(), startDate).getTime();
116 		var oldEndDateMs = DwtTimeSelect.getDateFromFields(endSelect.getHours(), endSelect.getMinutes(), endSelect.getAmPm(), endDate).getTime();
117 		var delta = oldEndDateMs - oldStartDateMs;
118 		if (!delta) return null;
119 		var newEndDateMs = newStartDateMs + delta;
120 		var newEndDate = new Date(newEndDateMs);
121 		endSelect.set(newEndDate);
122 		endDateField.value = AjxDateUtil.simpleComputeDateStr(newEndDate);
123 		if (endDateField.value != endDateOrig) {
124 			return endDateField;
125 		}
126 	} else {
127 		return null;
128 	}
129 };

131 /**
132  * Returns true if the start date/time is before the end date/time.
133  *
134  * @param ss				[DwtTimeSelect]		start time select
135  * @param es				[DwtTimeSelect]		end time select
136  * @param startDateField	[element]			start date field
137  * @param endDateField		[element]			end date field
138  */
139 DwtTimeSelect.validStartEnd =
140 function(startDateField, endDateField, ss, es) {
141 	var startDate = AjxDateUtil.simpleParseDateStr(startDateField.value);
142 	var endDate = AjxDateUtil.simpleParseDateStr(endDateField.value);
143 
144     if (startDate && endDate) {
145         if((startDate.valueOf() > endDate.valueOf())){
146             return false;
147         }
148         // bug fix #11329 - dont allow year to be more than the earth will be around :]
149 		if (startDate.getFullYear() > 9999 || endDate.getFullYear() > 9999) {
150 			return false;
151 		}
152         if(ss && es){
153             var startDateMs = DwtTimeSelect.getDateFromFields(ss.getHours(), ss.getMinutes(), ss.getAmPm(), startDate).getTime();
154             var endDateMs = DwtTimeSelect.getDateFromFields(es.getHours(), es.getMinutes(), es.getAmPm(), endDate).getTime();
155             if (startDateMs > endDateMs) {
156                 return false;
157             }
158         }
159     } else {
160 		return false;
161 	}
162 	return true;
163 };

174 * Sets the time select according to the given date.
175 *
176 * @param date	[Date]		a Date object
177 */
178 DwtTimeSelect.prototype.set = 
179 function(date) {
180 
181 	var hourIdx = 0, minuteIdx = 0, amPmIdx = 0;
182 	var isLocale24Hour = this.isLocale24Hour();
183 
184 	var hours = date.getHours();
185 	if (!isLocale24Hour && hours > 12) {
186 		hourIdx = hours - 13;
187 	} else if (!isLocale24Hour && hours == 0) {
188 		hourIdx = this.getHourSelectSize() - 1;
189 	} else {
190 		hourIdx = isLocale24Hour ? hours : hours - 1;
191 	}
192 
193 	minuteIdx = Math.floor(date.getMinutes() / 5);
194 
195 	if (!isLocale24Hour) {
196 		amPmIdx = (date.getHours() >= 12) ? 1 : 0;
197 	}
198 
199 	this.setSelected(hourIdx, minuteIdx, amPmIdx);
200 };

287 DwtTimeSelect.prototype._createSelects =
288 function() {
289 	this._hourSelectId = Dwt.getNextId();
290 	this._minuteSelectId = Dwt.getNextId();
291 	this._amPmSelectId = Dwt.getNextId();
292 
293 	// get the time formatter for the user's locale
294 	var timeFormatter = AjxDateFormat.getTimeInstance(AjxDateFormat.SHORT);
295 	var hourSegmentIdx = 0;
296 	var minuteSegmentIdx = 0;
297 
298 	var html = [];
299 	var i = 0;
300 
301 	html[i++] = "<table border=0 cellpadding=0 cellspacing=0><tr>";
302 
303 	// walk time formatter's segments array to render each segment part in the right order
304 	for (var j = 0; j < timeFormatter._segments.length; j++) {
305 		var segmentStr = timeFormatter._segments[j]._s;
306 
307 		if (timeFormatter._segments[j] instanceof AjxFormat.TextSegment) {
308 			var trimStr = AjxStringUtil.trim(segmentStr);
309 			if (trimStr.length) {
310 				html[i++] = "<td class='TextPadding ZmFieldLabel'>"
311 				html[i++] = segmentStr;
312 				html[i++] = "</td>";
313 			}
314 		} else if (segmentStr.charAt(0) == "h" || segmentStr.charAt(0) == "H") {
315 			hourSegmentIdx = j;
316 			html[i++] = "<td width=42 id='"
317 			html[i++] = this._hourSelectId;
318 			html[i++] = "'></td>";
319 		} else if (segmentStr.charAt(0) == "m") {
320 			minuteSegmentIdx = j;
321 			html[i++] = "<td width=42 id='"
322 			html[i++] = this._minuteSelectId;
323 			html[i++] = "'></td>";
324 		} else if (segmentStr == "a") {	
325 			this._isLocale24Hour = false;
326 			html[i++] = "<td width=42 id='"
327 			html[i++] = this._amPmSelectId;
328 			html[i++] = "'></td>";
329 		}
330 	}
331 	
332 	html[i++] = "</tr></table>";
333 
334 	// append html template to DOM
335 	this.getHtmlElement().innerHTML = html.join("");
336 
337 	// init vars for adding hour DwtSelect
338 	var now = new Date();
339 	var start = this._isLocale24Hour ? 0 : 1;
340 	var limit = this._isLocale24Hour ? 24 : 13;
341 
342 	// create new DwtSelect for hour slot
343 	this._hourSelect = new DwtSelect({parent:this});
344 	this._hourSelect.id = this.id;
345 	this._hourSelect.compId = DwtTimeSelect.HOUR;
346 	for (var i = start; i < limit; i++) {
347 		now.setHours(i);
348 		var label = timeFormatter._segments[hourSegmentIdx].format(now);
349 		this._hourSelect.addOption(label, false, i);
350 	}
351 	this._hourSelect.reparentHtmlElement(this._hourSelectId);
352 	delete this._hourSelectId;
353 
354 	// create new DwtSelect for minute slot
355 	this._minuteSelect = new DwtSelect({parent:this});
356 	this._minuteSelect.id = this.id;
357 	this._minuteSelect.compId = DwtTimeSelect.MINUTE;
358 	for (var i = 0; i < 60; i = i + 5) {
359 		now.setMinutes(i);
360 		var label = timeFormatter._segments[minuteSegmentIdx].format(now);
361 		this._minuteSelect.addOption(label, false, i);
362 	}
363 	this._minuteSelect.reparentHtmlElement(this._minuteSelectId);
364 	delete this._minuteSelectId;
365 
366 	// if locale is 12-hour time, add AM|PM DwtSelect
367 	if (!this._isLocale24Hour) {
368 		this._amPmSelect = new DwtSelect({parent:this});
369 		this._amPmSelect.id = this.id;
370 		this._amPmSelect.compId = DwtTimeSelect.AMPM;
371 		this._amPmSelect.addOption(I18nMsg["periodAm"], false, "AM");
372 		this._amPmSelect.addOption(I18nMsg["periodPm"], false, "PM");
373 		this._amPmSelect.reparentHtmlElement(this._amPmSelectId);
374 		delete this._amPmSelectId;
375 	}
376 };

426 /**
427 * Adjust an appt's start or end based on changes to the other one. If the user changes
428 * the start time, change the end time so that the appt duration is maintained. If the
429 * user changes the end time, we leave things alone.
430 *
431 * @param ev					[Event]				UI event from a DwtSelect
432 * @param startSelect		[DwtTimeInput]		start time select
433 * @param endSelect			[DwtTimeInput]		end time select
434 * @param startDateField		[element]			start date field
435 * @param endDateField		[element]			end date field
436 * @param dateInfo		    [object]			date info used to calculate the old time before changing this
437 * @param id		            [string]			an ID which got changed 
438 */
439 DwtTimeInput.adjustStartEnd =
440 function(ev, startSelect, endSelect, startDateField, endDateField, dateInfo, id) {
441     var startDate = AjxDateUtil.simpleParseDateStr(startDateField.value);
442     var endDate = AjxDateUtil.simpleParseDateStr(endDateField.value);
443     var startDateOrig = startDateField.value;
444     var endDateOrig = endDateField.value;
445     if (id == DwtTimeInput.START) {
446         var timeStr = dateInfo ? dateInfo.startTimeStr : startSelect.getTimeString();
447         var oldStartDateMs = DwtTimeInput.getDateFromFields(timeStr, startDate).getTime();
448         var newStartDateMs = DwtTimeInput.getDateFromFields(startSelect.getTimeString(), startDate).getTime();
449         var oldEndDateMs = DwtTimeInput.getDateFromFields(endSelect.getTimeString(), endDate).getTime();
450 
451         var delta = oldEndDateMs - oldStartDateMs;
452         if (!delta) return null;
453 
454         var newEndDateMs = newStartDateMs + delta;
455         var newEndDate = new Date(newEndDateMs);
456 
457         startSelect.set(new Date(newStartDateMs));
458         endSelect.set(newEndDate);
459         endDateField.value = AjxDateUtil.simpleComputeDateStr(newEndDate);
460 
461         if (endDateField.value != endDateOrig) {
462             return endDateField;
463         }
464     } else if (id == DwtTimeInput.END){
465         var timeStr = dateInfo ? dateInfo.endTimeStr : endSelect.getTimeString();
466         var oldEndDateMs = DwtTimeInput.getDateFromFields(timeStr, endDate).getTime();
467         var newEndDateMs = DwtTimeInput.getDateFromFields(endSelect.getTimeString(), endDate).getTime();
468         var oldStartDateMs = DwtTimeInput.getDateFromFields(startSelect.getTimeString(), startDate).getTime();
469 
470         var delta = oldEndDateMs - oldStartDateMs;
471         if (!delta) return null;
472 
473         //adjust start date only when the end date falls earlier than start date
474         if(newEndDateMs < oldStartDateMs) {
475             var newStartDateMs = newEndDateMs - delta;
476             var newStartDate = new Date(newStartDateMs);
477 
478             startSelect.set(newStartDate);
479             endSelect.set(new Date(newEndDateMs));
480             startDateField.value = AjxDateUtil.simpleComputeDateStr(newStartDate);
481             endDateField.value = AjxDateUtil.simpleComputeDateStr(new Date(newEndDateMs));
482         }
483 
484         if (startDateField.value != startDateOrig) {
485             return startDateField;
486         }
487 
488     } else {
489         return null;
490     }
491 };

635 DwtTimeInput.prototype.correctTimeString =
636 function(val, originalDate) {
637 
638     var segments = val.split(":");
639 
640     if(!segments) return originalDate;
641 
642     var hrs = (segments.length && segments[0] != null) ? parseInt(segments[0].replace(/\D/g, "")) : null;
643     var mins = (segments.length > 1 && segments[1]!= null) ? parseInt(segments[1].replace(/\D/g, "")) : 0;
644 
645     if(!hrs) hrs = (hrs == 0) ? 0 : originalDate.getHours();
646     if(!mins) mins = 0;
647 
648     originalDate.setHours(hrs, mins, 0, 0);
649 
650     return originalDate;
651 
652 };

667 DwtTimeInput.prototype._timeButtonListener =
668 function(ev) {
669 	var timeFormatter = AjxDateFormat.getTimeInstance(AjxDateFormat.SHORT);
670 	var defaultTopMenuItem;
671     if(!this._menuItemsAdded) {
672         var j,
673             k,
674             mi,
675             smi,
676             text,
677             maxMinutesItem,
678             minutesSelectMenu,
679             now = new Date(),
680             timeSelectButton = this._timeSelectBtn,
681             menuSelectionListener = new AjxListener(this, this._timeSelectionListener);
682 
683         for (j = 0; j < 24; j++) {
684             now.setHours(j);
685             now.setMinutes(0);
686 
687             mi = new DwtMenuItem({parent: this._hoursSelectMenu, style: DwtMenuItem.NO_STYLE});
688             text = timeFormatter.format(now); // Regular formatter, returns the I18nMsg formatted time
689             this.putTimeIndex(text, j);
690 
691 			if (j==0 || j==12) {
692                 text = DwtTimeSelect.format(now); // Specialized formatter, returns AjxMsg.midnight for midnight and AjxMsg.noon for noon
693                 this.putTimeIndex(text, j); // Both should go in the indexer
694             }
695 
696             mi.setText(text);
697             mi.setData("value", j*60);
698             if (menuSelectionListener) mi.addSelectionListener(menuSelectionListener);
699             if (j == DwtTimeInput.DEFAULT_TOP_ROW) defaultTopMenuItem = mi;
700 
701             maxMinutesItem = 60 / this._interval;
702             minutesSelectMenu = new DwtMenu({parent:mi, style:DwtMenu.DROPDOWN_CENTERV_STYLE, layout:DwtMenu.LAYOUT_CASCADE, maxRows:maxMinutesItem, congruent: true});
703             mi.setMenu(minutesSelectMenu, true);
704             mi.setSelectableWithSubmenu(true);
705 			minutesSelectMenu.dontStealFocus(true);
706 			for (k = 1; k < maxMinutesItem; k++) {
707                 now.setMinutes(k*this._interval);
708                 smi = new DwtMenuItem({parent: minutesSelectMenu, style: DwtMenuItem.NO_STYLE});
709                 smi.setText(timeFormatter.format(now));
710                 smi.setData("value", j*60 + k*this._interval);
711                 if (menuSelectionListener) smi.addSelectionListener(menuSelectionListener);
712             }
713         }
714         this._hoursSelectMenu.setWidth(timeSelectButton.getW() + this._timeSelectInput.getW());
715 
716         this._menuItemsAdded = true;
717     }
718 	ev.item.popup();
719 	if (defaultTopMenuItem) {
720 		this._hoursSelectMenu.scrollToItem(defaultTopMenuItem);
721 	}
722 	this._scrollToValue(timeFormatter.format(this.getValue()));
723 };
