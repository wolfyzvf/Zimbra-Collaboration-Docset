 29 /**
 30  * @class
 31  * Creates an overview. An overview is a {@link DwtComposite} that holds tree views.
 32  * 
 33  * @author Conrad Damon
 34  *
 35  * @param {Hash}	params 				a hash of parameters
 36  * @param	{String}	params.id 	the id for the HTML element
 37  * @param	{String}	params.overviewId 	the overview id
 38  * @param	{String}	params.containerId 	the overview container id (multi-account)
 39  * @param	{Array}	params.treeIds an array of organizer types that may be displayed in this overview
 40  * @param	{ZmZimbraAccount}	params.account		the account this overview belongs to
 41  * @param	{DwtControl}	params.parent			the containing widget
 42  * @param	{String}	params.overviewClass		the class name for overview DIV
 43  * @param	{constant}	params.posStyle				the positioning style for overview DIV
 44  * @param	{constant}	params.scroll				the scrolling style for overview DIV
 45  * @param	{Boolean}	params.selectionSupported <code>true</code> left-click selection is supported
 46  * @param	{Boolean}	params.actionSupported		<code>true</code> if right-click action menus are supported
 47  * @param	{Boolean}	params.dndSupported			<code>true</code> if drag-and-drop is supported
 48  * @param	{String}	params.headerClass			the class name for header item
 49  * @param	{Boolean}	params.showUnread			if <code>true</code>, unread counts will be shown
 50  * @param	{Boolean}	params.showNewButtons		if <code>true</code>, tree headers may have buttons for creating new organizers
 51  * @param	{constant}	params.treeStyle			the default display style for tree views
 52  * @param	{Boolean}	params.isCheckedByDefault	the default state for "checked" display style
 53  * @param	{Boolean}	params.noTooltips			if <code>true</code>, do not show toolt ips for tree items
 54  * @param	{Boolean}	params.skipImplicit			if <code>true</code>, do not save implicit prefs of expanded/collapsed node status for this overview (see ZmDialog.prototype._setOverview)
 55  * @param	{Boolean}	params.dynamicWidth			if <code>true</code>, the width is dynamic, i.e. the width is auto instead of fixed. Used for ZmDolderChooser so far.
 56  * @param {ZmOverviewController}	controller			the overview controller
 57  * 
 58  * @extends	DwtComposite
 59  */
 60 ZmOverview = function(params, controller) {
 61 
 62 	var overviewClass = params.overviewClass ? params.overviewClass : "ZmOverview";
 63 	params.id = params.id || ZmId.getOverviewId(params.overviewId);
 64 	DwtComposite.call(this, {parent:params.parent, className:overviewClass, posStyle:params.posStyle, id:params.id});
 65 
 66 	this._controller = controller;
 67 
 68 	this.setScrollStyle(params.scroll || Dwt.SCROLL_Y);
 69 
 70 	this.overviewId			= params.overviewId;
 71 	this.containerId		= params.containerId;
 72 	this.account			= params.account;
 73 	this.selectionSupported	= params.selectionSupported;
 74 	this.actionSupported	= params.actionSupported;
 75 	this.dynamicWidth		= params.dynamicWidth;
 76 	this.dndSupported		= params.dndSupported;
 77 	this.headerClass		= params.headerClass;
 78 	this.showUnread			= params.showUnread;
 79 	this.showNewButtons		= params.showNewButtons;
 80 	this.treeStyle			= params.treeStyle;
 81 	this.isCheckedByDefault = params.isCheckedByDefault;
 82 	this.noTooltips			= params.noTooltips;
 83 	this.isAppOverview		= params.isAppOverview;
 84 	this.skipImplicit 		= params.skipImplicit;
 85 	this.appName            = params.appName;
 86 
 87 	this._treeIds			= [];
 88 	this._treeHash			= {};
 89 	this._treeParents		= {};
 90 
 91 	// Create a parent div for each overview tree.
 92 	var doc = document;
 93 	var element = this.getHtmlElement();
 94 	if (params.treeIds) {
 95 		for (var i = 0, count = params.treeIds.length; i < count; i++) {
 96 			var div = doc.createElement("DIV");
 97 			var treeId = params.treeIds[i];
 98 			this._treeParents[treeId] = div.id = [this.overviewId, treeId].join("-parent-");
 99 			element.appendChild(div);
100 		}
101 	}
102 
103 	if (this.dndSupported) {
104 		this._scrollableContainerId = this.containerId || this.overviewId;
105 		var container = this.containerId ? document.getElementById(this.containerId) : this.getHtmlElement();
106 		var params = {container:container, threshold:15, amount:5, interval:10, id:this._scrollableContainerId};
107 		this._dndScrollCallback = new AjxCallback(null, DwtControl._dndScrollCallback, [params]);
108 	}
109 
110 	this.setAttribute('aria-label', ZmMsg.overviewLabel);
111 };

113 ZmOverview.prototype = new DwtComposite;

115 ZmOverview.prototype.role = "navigation";

138 /**
139  * Displays the given list of tree views in this overview.
140  *
141  * @param {Array}	treeIds		an array of organizer ids
142  * @param {Hash}	omit		the hash of organizer ids to ignore
143  */
144 ZmOverview.prototype.set =
145 function(treeIds, omit) {
146 	if (treeIds && treeIds.length) {
147 		for (var i = 0; i < treeIds.length; i++) {
148 			this.setTreeView(treeIds[i], omit);
149 		}
150 	}
151 };
152 
153 /**
154  * Sets the given tree view. Its tree controller is responsible for using the appropriate
155  * data tree to populate the tree view. The tree controller will be lazily created if
156  * necessary. The tree view is cleared before it is set. The tree view inherits options
157  * from this overview.
158  * 
159  * @param {String}	treeId	the organizer ID
160  * @param {Hash}	omit	a hash of organizer ids to ignore
161  */
162 ZmOverview.prototype.setTreeView =
163 function(treeId, omit) {
164 	// check for false since setting precondition is optional (can be null)
165 	if (appCtxt.get(ZmOrganizer.PRECONDITION[treeId]) === false) { return; }
166 
167 	AjxDispatcher.require(ZmOrganizer.ORG_PACKAGE[treeId]);
168 	var treeController = this._controller.getTreeController(treeId);
169 	if (!treeController) { return; }
170 	if (this._treeHash[treeId]) {
171 		treeController.clearTreeView(this.overviewId);
172 	} else {
173 		this._treeIds.push(treeId);
174 	}
175 	var params = {
176 		overviewId:		this.overviewId,
177 		omit:			omit,
178 		showUnread:		this.showUnread,
179 		account:		this.account
180 	};
181 	this._treeHash[treeId] = treeController.show(params); // render tree view
182 };

205 /**
206  * Searches the tree views for the tree item whose data object has the given ID and type.
207  * 
208  * @param {int}	id			the id to look for
209  * @param {constant}	type			the item must also have this type
210  * @return	{Object}	the item or <code>null</code> if not found
211  */
212 ZmOverview.prototype.getTreeItemById =
213 function(id, type) {
214 	if (!id) { return null; }
215 	for (var i = 0; i < this._treeIds.length; i++) {
216 		var treeView = this._treeHash[this._treeIds[i]];
217 		if (treeView) {
218 			var item = treeView.getTreeItemById && treeView.getTreeItemById(id);
219 			if (item && (!type || (this._treeIds[i] == type))) {
220 				return item;
221 			}
222 		}
223 	}
224 	return null;
225 };
226 
227 /**
228 * Returns the first selected item within this overview.
229 * 
230 * @param	{Boolean}	typeOnly	if <code>true</code>, return the type only
231 * @return	{Object}	the item (or type if <code>typeOnly</code>) or <code>null</code> if not found
232 */
233 ZmOverview.prototype.getSelected =
234 function(typeOnly) {
235 	for (var i = 0; i < this._treeIds.length; i++) {
236 		var treeView = this._treeHash[this._treeIds[i]];
237 		if (treeView) {
238 			var item = treeView.getSelected();
239 			if (item) {
240 				return typeOnly ? treeView.type : item;
241 			} // otherwise continue with other treeviews to look for selected item
242 		}
243 	}
244 	return null;
245 };

247 ZmOverview.prototype.getTabGroupMember =
248 function() {
249 	var tabGroup = new DwtTabGroup();
250 	if (this.getZIndex() > Dwt.Z_HIDDEN) {
251 		tabGroup.addMember(this.getSelected());
252 	}
253 	return tabGroup;
254 };

256 ZmOverview.prototype.deselectAllTreeViews =
257 function() {
258 	for (var i = 0; i < this._treeIds.length; i++) {
259 		var treeView = this._treeHash[this._treeIds[i]];
260 		if (treeView) {
261 			treeView.deselectAll();
262 		}
263 	}
264 };

267 /**
268  * Selects the item with the given ID within the given tree in this overview.
269  *
270  * @param {String}	id	the item id
271  * @param {constant}	type	the tree type
272  */
273 ZmOverview.prototype.setSelected =
274 function(id, type) {
275 	var ti, treeView;
276 	if (type) {
277 		treeView = this._treeHash[type];
278 		ti = treeView && treeView.getTreeItemById(id);
279 	} else {
280 		for (var type in this._treeHash) {
281 			treeView = this._treeHash[type];
282 			ti = treeView && treeView.getTreeItemById(id);
283 			if (ti) { break; }
284 		}
285 	}
286 
287 	if (ti && (this._selectedTreeItem != ti)) {
288 		treeView.setSelected(id, true, true);
289 	}
290 	this.itemSelected(ti);
291 };
292 
293 /**
294  * Given a tree item, de-selects all items in the overview's
295  * other tree views, enforcing single selection within the overview.
296  * Passing a null argument will clear selection in all tree views.
297  *
298  * @param {DwtTreeItem}	treeItem		the tree item
299  */
300 ZmOverview.prototype.itemSelected =
301 function(treeItem) {
302 	if (appCtxt.multiAccounts && treeItem) {
303 		var name = this.overviewId.substring(0, this.overviewId.indexOf(":"));
304 		var container = this._controller.getOverviewContainer(name);
305 		if (container) {
306 			container.deselectAll(this);
307 		}
308 	}
309 
310 	if (this._selectedTreeItem && (this._selectedTreeItem._tree != (treeItem && treeItem._tree))) {
311 		this._selectedTreeItem._tree.deselectAll();
312 	}
313 
314 	this._selectedTreeItem = treeItem;
315 };
316 
317 /**
318  * Clears the tree views.
319  */
320 ZmOverview.prototype.clear =
321 function() {
322 	for (var i = 0; i < this._treeIds.length; i++) {
323 		var treeId = this._treeIds[i];
324 		if (this._treeHash[treeId]) {
325 			var treeController = this._controller.getTreeController(treeId);
326 			treeController.clearTreeView(this.overviewId);
327 			delete this._treeHash[treeId];
328 		}
329 	}
330 };
331 
332 ZmOverview.prototype.clearSelection =
333 function() {
334 	if (this._selectedTreeItem) {
335 		this._selectedTreeItem._tree.deselectAll();
336 	}
337 };

348 /**
349  * @private
350  */
351 ZmOverview.prototype._focus =
352 function() {
353 	var item = this._selectedTreeItem;
354 	if (!item) {
355 		var tree = this._treeHash[this._treeIds[0]];
356 		if (tree) {
357 			item = tree._getNextTreeItem(true);
358 		}
359 	}
360 
361     if (item) {
362         item.focus();
363         item._tree.setSelection(item, false, true);
364     }
365 };
366 
367 /**
368  * @private
369  */
370 ZmOverview.prototype._blur =
371 function() {
372 	var item = this._selectedTreeItem;
373 	if (item) {
374 		item._blur();
375 	}
376 };

378 /**
379  * Returns the next/previous selectable tree item within this overview, starting with the
380  * tree immediately after/before the given one. Used to handle tree item selection that
381  * spans trees.
382  *
383  * @param {Boolean}	next		if <code>true</code>, look for next item as opposed to previous item
384  * @param {ZmTreeView}	tree		the tree that we are just leaving
385  *
386  * @private
387  */
388 ZmOverview.prototype._getNextTreeItem =
389 function(next, tree) {
390 
391 	for (var i = 0; i < this._treeIds.length; i++) {
392 		if (this._treeHash[this._treeIds[i]] == tree) {
393 			break;
394 		}
395 	}
396 
397 	var nextItem = null;
398 	var idx = next ? i + 1 : i - 1;
399 	tree = this._treeHash[this._treeIds[idx]];
400 	while (tree) {
401 		nextItem = DwtTree.prototype._getNextTreeItem.call(tree, next);
402 		if (nextItem) {
403 			break;
404 		}
405 		idx = next ? idx + 1 : idx - 1;
406 		tree = this._treeHash[this._treeIds[idx]];
407 	}
408 
409 	return nextItem;
410 };
