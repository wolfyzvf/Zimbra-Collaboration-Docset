116     // Inject element with style element and some CSS rules
117     injectElementWithStyles = function( rule, callback, nodes, testnames ) {
118 
119       var style, ret, node,
120           div = document.createElement('div'),
121           // After page load injecting a fake body doesn't work so check if body exists
122           body = document.body,
123           // IE6 and 7 won't return offsetWidth or offsetHeight unless it's in the body element, so we fake it.
124           fakeBody = body ? body : document.createElement('body');
125 
126       if ( parseInt(nodes, 10) ) {
127           // In order not to give false positives we create a node for each test
128           // This also allows the method to scale for unspecified uses
129           while ( nodes-- ) {
130               node = document.createElement('div');
131               node.id = testnames ? testnames[nodes] : mod + (nodes + 1);
132               div.appendChild(node);
133           }
134       }
135 
136       // <style> elements in IE6-9 are considered 'NoScope' elements and therefore will be removed
137       // when injected with innerHTML. To get around this you need to prepend the 'NoScope' element
138       // with a 'scoped' element, in our case the soft-hyphen entity as it won't mess with our measurements.
139       // msdn.microsoft.com/en-us/library/ms533897%28VS.85%29.aspx
140       // Documents served as xml will throw if using ­ so use xml friendly encoded version. See issue #277
141       style = ['­','<style id="s', mod, '">', rule, '</style>'].join('');
142       div.id = mod;
143       // IE6 will false positive on some tests due to the style element inside the test div somehow interfering offsetHeight, so insert it into body or fakebody.
144       // Opera will act all quirky when injecting elements in documentElement when page is served as xml, needs fakebody too. #270
145       (body ? div : fakeBody).innerHTML += style;
146       fakeBody.appendChild(div);
147       if ( !body ) {
148           //avoid crashing IE8, if background image is used
149           fakeBody.style.background = "";
150           docElement.appendChild(fakeBody);
151       }
152 
153       ret = callback(div, rule);
154       // If this is done after page load we don't want to remove the body so check if body exists
155       !body ? fakeBody.parentNode.removeChild(fakeBody) : div.parentNode.removeChild(div);
156 
157       return !!ret;
158

188     //
189     // isEventSupported determines if a given element supports the given event
190     // kangax.github.com/iseventsupported/
191     //
192     // The following results are known incorrects:
193     //   Modernizr.hasEvent("webkitTransitionEnd", elem) // false negative
194     //   Modernizr.hasEvent("textInput") // in Webkit. github.com/Modernizr/Modernizr/issues/333
195     //   ...
196     isEventSupported = (function() {
197 
198       var TAGNAMES = {
199         'select': 'input', 'change': 'input',
200         'submit': 'form', 'reset': 'form',
201         'error': 'img', 'load': 'img', 'abort': 'img'
202       };
203 
204       function isEventSupported( eventName, element ) {
205 
206         element = element || document.createElement(TAGNAMES[eventName] || 'div');
207         eventName = 'on' + eventName;
208 
209         // When using `setAttribute`, IE skips "unload", WebKit skips "unload" and "resize", whereas `in` "catches" those
210         var isSupported = eventName in element;
211 
212         if ( !isSupported ) {
213           // If it has no `setAttribute` (i.e. doesn't implement Node interface), try generic element
214           if ( !element.setAttribute ) {
215             element = document.createElement('div');
216           }
217           if ( element.setAttribute && element.removeAttribute ) {
218             element.setAttribute(eventName, '');
219             isSupported = is(element[eventName], 'function');
220 
221             // If property was created, "remove it" (by setting value to `undefined`)
222             if ( !is(element[eventName], 'undefined') ) {
223               element[eventName] = undefined;
224             }
225             element.removeAttribute(eventName);
226           }
227         }
228 
229         element = null;
230         return isSupported;
231       }
232       return isEventSupported;
233     })(),

327     // testProps is a generic CSS / DOM property test.
328 
329     // In testing support for a given CSS property, it's legit to test:
330     //    `elem.style[styleName] !== undefined`
331     // If the property is supported it will return an empty string,
332     // if unsupported it will return undefined.
333 
334     // We'll take advantage of this quick test and skip setting a style
335     // on our modernizr element, but instead just testing undefined vs
336     // empty string.
337 
338     // Because the testing of the CSS property names (with "-", as
339     // opposed to the camelCase DOM properties) is non-portable and
340     // non-standard but works in WebKit and IE (but not Gecko or Opera),
341     // we explicitly reject properties with dashes so that authors
342     // developing in WebKit or IE first don't end up with
343     // browser-specific content by accident.
344 
345     function testProps( props, prefixed ) {
346         for ( var i in props ) {
347             var prop = props[i];
348             if ( !contains(prop, "-") && mStyle[prop] !== undefined ) {
349                 return prefixed == 'pfx' ? prop : true;
350             }
351         }
352         return false;
353     }

356     // TODO :: add testDOMProps
357     /**
358      * testDOMProps is a generic DOM property test; if a browser supports
359      *   a certain property, it won't return undefined for it.
360      */
361     function testDOMProps( props, obj, elem ) {
362         for ( var i in props ) {
363             var item = obj[props[i]];
364             if ( item !== undefined) {
365 
366                 // return the property name as a string
367                 if (elem === false) return props[i];
368 
369                 // let's bind a function
370                 if (is(item, 'function')){
371                   // default to autobind unless override
372                   return item.bind(elem || obj);
373                 }
374 
375                 // return the unbound function or obj or value
376                 return item;
377             }
378         }
379         return false;
380     }

383     /**
384      * testPropsAll tests a list of DOM properties we want to check against.
385      *   We specify literally ALL possible (known and/or likely) properties on
386      *   the element including the non-vendor prefixed one, for forward-
387      *   compatibility.
388      */
389     function testPropsAll( prop, prefixed, elem ) {
390 
391         var ucProp  = prop.charAt(0).toUpperCase() + prop.slice(1),
392             props   = (prop + ' ' + cssomPrefixes.join(ucProp + ' ') + ucProp).split(' ');
393 
394         // did they call .prefixed('boxSizing') or are we just testing a prop?
395         if(is(prefixed, "string") || is(prefixed, "undefined")) {
396           return testProps(props, prefixed);
397 
398         // otherwise, they called .prefixed('requestAnimationFrame', window[, elem])
399         } else {
400           props = (prop + ' ' + (domPrefixes).join(ucProp + ' ') + ucProp).split(' ');
401           return testDOMProps(props, prefixed, elem);
402         }
403     }
