 22 /**
 23  * Creates a signature collection.
 24  * @class
 25  * This class represents a signature collection.
 26  * 
 27  * @extends		ZmModel
 28  */
 29 ZmSignatureCollection = function() {
 30 	ZmModel.call(this, ZmEvent.S_SIGNATURE);
 31 	this._idMap = {};
 32 	this._nameMap= {};
 33 	this._size = 0;
 34 };
 35 
 36 ZmSignatureCollection.prototype = new ZmModel;
 37 ZmSignatureCollection.prototype.constructor = ZmSignatureCollection;
 38 
 39 ZmSignatureCollection.prototype.toString =
 40 function() {
 41 	return "ZmSignatureCollection";
 42 };
 43 
 44 //
 45 // Public methods
 46 //
 47 /**
 48  * Adds the signature.
 49  * 
 50  * @param	{ZmSignature}	signature		the signature to add
 51  */
 52 ZmSignatureCollection.prototype.add =
 53 function(signature) {
 54 	if (!this._idMap[signature.id]) {
 55 		this._idMap[signature.id] = signature;
 56 		this._nameMap[signature.name] = signature;
 57 		this._size++;
 58 		this._notify(ZmEvent.E_CREATE, { item: signature });
 59 	}
 60 };
 61 
 62 /**
 63  * Removes the signature.
 64  * 
 65  * @param	{ZmSignature}	signature		the signature to remove
 66  */
 67 ZmSignatureCollection.prototype.remove =
 68 function(signature) {
 69 	if (this._idMap[signature.id]) {
 70 		delete this._idMap[signature.id];
 71 		delete this._nameMap[signature.name];
 72 		this._size--;
 73 		this._notify(ZmEvent.E_DELETE, { item: signature });
 74 	}
 75 };
 76 
 77 /**
 78  * Gets the count of signatures.
 79  * 
 80  * @return	{int}		the size
 81  */
 82 ZmSignatureCollection.prototype.getSize =
 83 function() {
 84 	return this._size;
 85 };
 86 
 87 /**
 88  * Gets the signatures.
 89  * 
 90  * @return	{Array}	an array of {@link ZmSignature} objects
 91  */
 92 ZmSignatureCollection.prototype.getSignatures =
 93 function(sort) {
 94 
 95 	var signatures = AjxUtil.values(this._idMap);
 96 	if (sort) {
 97 		signatures.sort(ZmSignatureCollection.BY_NAME);
 98 	}
 99 	return signatures;
100 };
101 
102 ZmSignatureCollection.prototype.getSignatureOptions =
103 function() {
104 	// collect signatures
105 	var signatures = [];
106 	for (var id in this._idMap) {
107 		signatures.push(this._idMap[id]);
108 	}
109 	signatures.sort(ZmSignatureCollection.BY_NAME);
110 
111 	// create options
112 	var options = [];
113 	//In Web Client offline mode signature having vCard will be suppressed
114 	var isWebClientOffline = appCtxt.isWebClientOffline();
115 	for (var i = 0; i < signatures.length; i++) {
116 		var signature = signatures[i];
117 		if (isWebClientOffline && signature.contactId) {
118 			continue;
119 		}
120 		options.push(new DwtSelectOptionData(signature.id, signature.name));
121 	}
122 	options.push(new DwtSelectOptionData("", ZmMsg.signatureDoNotAttach));
123 	return options;
124 };
125 
126 /**
127  * Gets the signature by id.
128  * 
129  * @param	{String}	id		the signature
130  * @return	{ZmSignature} the signature
131  */
132 ZmSignatureCollection.prototype.getById =
133 function(id) {
134 	return this._idMap[id];
135 };
136 
137 /**
138  * Gets the signature by name.
139  * 
140  * @param	{String}	name		the signature
141  * @return	{ZmSignature} the signature
142  */
143 ZmSignatureCollection.prototype.getByName =
144 function(name) {
145 	var lname = name.toLowerCase();
146 	for (var key in this._nameMap) {
147 		if (key.toLowerCase() == lname) {
148 			return this._nameMap[key];
149 		}
150 	}
151 };
152 
153 ZmSignatureCollection.prototype.initialize =
154 function(data) {
155 	if (this._size) return;
156 
157 	var signatures = data.signature;
158 	if (!signatures) return;
159 
160 	for (var i = 0; i < signatures.length; i++) {
161 		var signature = ZmSignature.createFromJson(signatures[i]);
162 		this.add(signature);
163 	}
164 };
165 
166 //
167 // Static functions
168 //
169 
170 ZmSignatureCollection.BY_NAME =
171 function(a, b) {
172 	return a.name.localeCompare(b.name);
173 };
