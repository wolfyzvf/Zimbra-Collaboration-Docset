 11 /**
 12  * This class handles the undo/redo history levels for the editor. Since the build in undo/redo has major drawbacks a custom one was needed.
 13  *
 14  * @class tinymce.UndoManager
 15  */
 16 define("tinymce/UndoManager", [
 17 	"tinymce/Env",
 18 	"tinymce/util/Tools",
 19 	"tinymce/html/SaxParser"
 20 ], function(Env, Tools, SaxParser) {
 21 	var trim = Tools.trim, trimContentRegExp;
 22 
 23 	trimContentRegExp = new RegExp([
 24 		'<span[^>]+data-mce-bogus[^>]+>[\u200B\uFEFF]+<\\/span>', // Trim bogus spans like caret containers
 25 		'\\s?data-mce-selected="[^"]+"' // Trim temporaty data-mce prefixed attributes like data-mce-selected
 26 	].join('|'), 'gi');
 27 
 28 	return function(editor) {
 29 		var self = this, index = 0, data = [], beforeBookmark, isFirstTypedCharacter, locks = 0;
 30 
 31 		/**
 32 		 * Returns a trimmed version of the editor contents to be used for the undo level. This
 33 		 * will remove any data-mce-bogus="all" marked elements since these are used for UI it will also
 34 		 * remove the data-mce-selected attributes used for selection of objects and caret containers.
 35 		 * It will keep all data-mce-bogus="1" elements since these can be used to place the caret etc and will
 36 		 * be removed by the serialization logic when you save.
 37 		 *
 38 		 * @private
 39 		 * @return {String} HTML contents of the editor excluding some internal bogus elements.
 40 		 */
 41 		function getContent() {
 42 			var content = editor.getContent({format: 'raw', no_events: 1});
 43 			var bogusAllRegExp = /<(\w+) [^>]*data-mce-bogus="all"[^>]*>/g;
 44 			var endTagIndex, index, matchLength, matches, shortEndedElements, schema = editor.schema;
 45 
 46 			content = content.replace(trimContentRegExp, '');
 47 			shortEndedElements = schema.getShortEndedElements();
 48 
 49 			// Remove all bogus elements marked with "all"
 50 			while ((matches = bogusAllRegExp.exec(content))) {
 51 				index = bogusAllRegExp.lastIndex;
 52 				matchLength = matches[0].length;
 53 
 54 				if (shortEndedElements[matches[1]]) {
 55 					endTagIndex = index;
 56 				} else {
 57 					endTagIndex = SaxParser.findEndTag(schema, content, index);
 58 				}
 59 
 60 				content = content.substring(0, index - matchLength) + content.substring(endTagIndex);
 61 				bogusAllRegExp.lastIndex = index - matchLength;
 62 			}
 63 
 64 			return trim(content);
 65 		}
 66 
 67 		function addNonTypingUndoLevel(e) {
 68 			self.typing = false;
 69 			self.add({}, e);
 70 		}
 71 
 72 		// Add initial undo level when the editor is initialized
 73 		editor.on('init', function() {
 74 			self.add();
 75 		});
 76 
 77 		// Get position before an execCommand is processed
 78 		editor.on('BeforeExecCommand', function(e) {
 79 			var cmd = e.command;
 80 
 81 			if (cmd != 'Undo' && cmd != 'Redo' && cmd != 'mceRepaint') {
 82 				self.beforeChange();
 83 			}
 84 		});
 85 
 86 		// Add undo level after an execCommand call was made
 87 		editor.on('ExecCommand', function(e) {
 88 			var cmd = e.command;
 89 
 90 			if (cmd != 'Undo' && cmd != 'Redo' && cmd != 'mceRepaint') {
 91 				addNonTypingUndoLevel(e);
 92 			}
 93 		});
 94 
 95 		editor.on('ObjectResizeStart', function() {
 96 			self.beforeChange();
 97 		});
 98 
 99 		editor.on('SaveContent ObjectResized blur', addNonTypingUndoLevel);
100 		editor.on('DragEnd', addNonTypingUndoLevel);
101 
102 		editor.on('KeyUp', function(e) {
103 			var keyCode = e.keyCode;
104 
105 			if ((keyCode >= 33 && keyCode <= 36) || (keyCode >= 37 && keyCode <= 40) || keyCode == 45 || keyCode == 13 || e.ctrlKey) {
106 				addNonTypingUndoLevel();
107 				editor.nodeChanged();
108 			}
109 
110 			if (keyCode == 46 || keyCode == 8 || (Env.mac && (keyCode == 91 || keyCode == 93))) {
111 				editor.nodeChanged();
112 			}
113 
114 			// Fire a TypingUndo event on the first character entered
115 			if (isFirstTypedCharacter && self.typing) {
116 				// Make the it dirty if the content was changed after typing the first character
117 				if (!editor.isDirty()) {
118 					editor.isNotDirty = !data[0] || getContent() == data[0].content;
119 
120 					// Fire initial change event
121 					if (!editor.isNotDirty) {
122 						editor.fire('change', {level: data[0], lastLevel: null});
123 					}
124 				}
125 
126 				editor.fire('TypingUndo');
127 				isFirstTypedCharacter = false;
128 				editor.nodeChanged();
129 			}
130 		});
131 
132 		editor.on('KeyDown', function(e) {
133 			var keyCode = e.keyCode;
134 
135 			// Is caracter positon keys left,right,up,down,home,end,pgdown,pgup,enter
136 			if ((keyCode >= 33 && keyCode <= 36) || (keyCode >= 37 && keyCode <= 40) || keyCode == 45) {
137 				if (self.typing) {
138 					addNonTypingUndoLevel(e);
139 				}
140 
141 				return;
142 			}
143 
144 			// If key isn't shift,ctrl,alt,capslock,metakey
145 			if ((keyCode < 16 || keyCode > 20) && keyCode != 224 && keyCode != 91 && !self.typing) {
146 				self.beforeChange();
147 				self.typing = true;
148 				self.add({}, e);
149 				isFirstTypedCharacter = true;
150 			}
151 		});
152 
153 		editor.on('MouseDown', function(e) {
154 			if (self.typing) {
155 				addNonTypingUndoLevel(e);
156 			}
157 		});
158 
159 		// Add keyboard shortcuts for undo/redo keys
160 		editor.addShortcut('ctrl+z', '', 'Undo');
161 		editor.addShortcut('ctrl+y,ctrl+shift+z', '', 'Redo');
162 
163 		editor.on('AddUndo Undo Redo ClearUndos', function(e) {
164 			if (!e.isDefaultPrevented()) {
165 				editor.nodeChanged();
166 			}
167 		});
168 
169 		self = {
170 			// Explose for debugging reasons
171 			data: data,
172 
173 			/**
174 			 * State if the user is currently typing or not. This will add a typing operation into one undo
175 			 * level instead of one new level for each keystroke.
176 			 *
177 			 * @field {Boolean} typing
178 			 */
179 			typing: false,
180 
181 			/**
182 			 * Stores away a bookmark to be used when performing an undo action so that the selection is before
183 			 * the change has been made.
184 			 *
185 			 * @method beforeChange
186 			 */
187 			beforeChange: function() {
188 				if (!locks) {
189 					beforeBookmark = editor.selection.getBookmark(2, true);
190 				}
191 			},
192 
193 			/**
194 			 * Adds a new undo level/snapshot to the undo list.
195 			 *
196 			 * @method add
197 			 * @param {Object} level Optional undo level object to add.
198 			 * @param {DOMEvent} Event Optional event responsible for the creation of the undo level.
199 			 * @return {Object} Undo level that got added or null it a level wasn't needed.
200 			 */
201 			add: function(level, event) {
202 				var i, settings = editor.settings, lastLevel;
203 
204 				level = level || {};
205 				level.content = getContent();
206 
207 				if (locks || editor.removed) {
208 					return null;
209 				}
210 
211 				lastLevel = data[index];
212 				if (editor.fire('BeforeAddUndo', {level: level, lastLevel: lastLevel, originalEvent: event}).isDefaultPrevented()) {
213 					return null;
214 				}
215 
216 				// Add undo level if needed
217 				if (lastLevel && lastLevel.content == level.content) {
218 					return null;
219 				}
220 
221 				// Set before bookmark on previous level
222 				if (data[index]) {
223 					data[index].beforeBookmark = beforeBookmark;
224 				}
225 
226 				// Time to compress
227 				if (settings.custom_undo_redo_levels) {
228 					if (data.length > settings.custom_undo_redo_levels) {
229 						for (i = 0; i < data.length - 1; i++) {
230 							data[i] = data[i + 1];
231 						}
232 
233 						data.length--;
234 						index = data.length;
235 					}
236 				}
237 
238 				// Get a non intrusive normalized bookmark
239 				level.bookmark = editor.selection.getBookmark(2, true);
240 
241 				// Crop array if needed
242 				if (index < data.length - 1) {
243 					data.length = index + 1;
244 				}
245 
246 				data.push(level);
247 				index = data.length - 1;
248 
249 				var args = {level: level, lastLevel: lastLevel, originalEvent: event};
250 
251 				editor.fire('AddUndo', args);
252 
253 				if (index > 0) {
254 					editor.isNotDirty = false;
255 					editor.fire('change', args);
256 				}
257 
258 				return level;
259 			},
260 
261 			/**
262 			 * Undoes the last action.
263 			 *
264 			 * @method undo
265 			 * @return {Object} Undo level or null if no undo was performed.
266 			 */
267 			undo: function() {
268 				var level;
269 
270 				if (self.typing) {
271 					self.add();
272 					self.typing = false;
273 				}
274 
275 				if (index > 0) {
276 					level = data[--index];
277 
278 					// Undo to first index then set dirty state to false
279 					if (index === 0) {
280 						editor.isNotDirty = true;
281 					}
282 
283 					editor.setContent(level.content, {format: 'raw'});
284 					editor.selection.moveToBookmark(level.beforeBookmark);
285 
286 					editor.fire('undo', {level: level});
287 				}
288 
289 				return level;
290 			},
291 
292 			/**
293 			 * Redoes the last action.
294 			 *
295 			 * @method redo
296 			 * @return {Object} Redo level or null if no redo was performed.
297 			 */
298 			redo: function() {
299 				var level;
300 
301 				if (index < data.length - 1) {
302 					level = data[++index];
303 
304 					editor.setContent(level.content, {format: 'raw'});
305 					editor.selection.moveToBookmark(level.bookmark);
306 
307 					editor.fire('redo', {level: level});
308 				}
309 
310 				return level;
311 			},
312 
313 			/**
314 			 * Removes all undo levels.
315 			 *
316 			 * @method clear
317 			 */
318 			clear: function() {
319 				data = [];
320 				index = 0;
321 				self.typing = false;
322 				editor.fire('ClearUndos');
323 			},
324 
325 			/**
326 			 * Returns true/false if the undo manager has any undo levels.
327 			 *
328 			 * @method hasUndo
329 			 * @return {Boolean} true/false if the undo manager has any undo levels.
330 			 */
331 			hasUndo: function() {
332 				// Has undo levels or typing and content isn't the same as the initial level
333 				return index > 0 || (self.typing && data[0] && getContent() != data[0].content);
334 			},
335 
336 			/**
337 			 * Returns true/false if the undo manager has any redo levels.
338 			 *
339 			 * @method hasRedo
340 			 * @return {Boolean} true/false if the undo manager has any redo levels.
341 			 */
342 			hasRedo: function() {
343 				return index < data.length - 1 && !this.typing;
344 			},
345 
346 			/**
347 			 * Executes the specified function in an undo transation. The selection
348 			 * before the modification will be stored to the undo stack and if the DOM changes
349 			 * it will add a new undo level. Any methods within the transation that adds undo levels will
350 			 * be ignored. So a transation can include calls to execCommand or editor.insertContent.
351 			 *
352 			 * @method transact
353 			 * @param {function} callback Function to execute dom manipulation logic in.
354 			 */
355 			transact: function(callback) {
356 				self.beforeChange();
357 
358 				try {
359 					locks++;
360 					callback();
361 				} finally {
362 					locks--;
363 				}
364 
365 				self.add();
366 			}
367 		};
368 
369 		return self;
370 	};
371 });
