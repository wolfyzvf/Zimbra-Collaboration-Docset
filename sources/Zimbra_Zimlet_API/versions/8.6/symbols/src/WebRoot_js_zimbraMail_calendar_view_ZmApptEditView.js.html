 23  * Creates a new calendar item edit view.

 40 ZmApptEditView = function(parent, attendees, controller, dateInfo) {
 41 
 42 	var idParams = {
 43 		skinComponent:  ZmId.SKIN_APP_MAIN,
 44 		app:            ZmId.APP_CALENDAR,
 45 		componentType:  ZmId.WIDGET_VIEW,
 46 		componentName:  ZmId.VIEW_APPOINTMENT
 47 	};
 48 
 49 	var domId = ZmId.create(idParams, "An appointment editing view");
 50 
 51 	ZmCalItemEditView.call(this, parent, attendees, controller, dateInfo, null, "ZmApptEditView", domId);
 52 
 53 	// cache so we dont keep calling appCtxt
 54 	this.GROUP_CALENDAR_ENABLED = appCtxt.get(ZmSetting.GROUP_CALENDAR_ENABLED);
 55 
 56 	this._attTypes = [];
 57 	if (this.GROUP_CALENDAR_ENABLED) {
 58 		this._attTypes.push(ZmCalBaseItem.PERSON);
 59 	}
 60 	this._attTypes.push(ZmCalBaseItem.LOCATION);
 61 	if (appCtxt.get(ZmSetting.GAL_ENABLED) && this.GROUP_CALENDAR_ENABLED) {
 62 		this._attTypes.push(ZmCalBaseItem.EQUIPMENT);
 63 	}
 64     this._locationTextMap = {};
 65     this._attendeePicker = {};
 66     this._pickerButton = {};
 67 
 68     //used to preserve original attendees while forwarding appt
 69     this._fwdApptOrigAttendees = [];
 70     this._attendeesHashMap = {};
 71 
 72     // Store Appt form values.
 73     this._apptFormValue = {};
 74     this._showAsValueChanged  = false;
 75 
 76     this._locationExceptions  = null;
 77     this._alteredLocations    = null;
 78     this._enableResolveDialog = true;
 79 
 80     this._locationConflict    = false;
 81     this._locationStatusMode = ZmApptEditView.LOCATION_STATUS_NONE;
 82 
 83     var app = appCtxt.getApp(ZmApp.CALENDAR);
 84     // Each ApptEditView must now have its own copy of the FreeBusyCache.  The cache will
 85     // now hold FreeBusy info that is unique to an appointment, in that the Server provides
 86     // Free busy info that excludes the current appointment.  So the cache information cannot
 87     // be shared across appointments.
 88     //this._fbCache = app.getFreeBusyCache();
 89     AjxDispatcher.require(["MailCore", "CalendarCore"]);
 90     this._fbCache = new ZmFreeBusyCache(app);
 91 
 92     this._customRecurDialogCallback = this._recurChangeForLocationConflict.bind(this);
 93 };

191 ZmApptEditView.prototype.initializeLocationConflictCheck =
192 function(appt) {
193     // Create a 'Location-only' clone of the appt, for use with the
194     // resource conflict calls
195     ZmLocationAppt.prototype = appt;
196     ZmLocationRecurrence.prototype = appt.getRecurrence();
197     this._locationConflictAppt = new ZmLocationAppt();
198     this._locationConflictAppt._recurrence = new ZmLocationRecurrence();
199     this._locationConflictAppt._attendees[ZmCalBaseItem.LOCATION] =
200         appt._attendees[ZmCalBaseItem.LOCATION];
201     this._locationConflictAppt._attendees[ZmCalBaseItem.PERSON]	  = [];
202     this._locationConflictAppt._attendees[ZmCalBaseItem.EQUIPMENT]= [];
203 
204     this._processLocationCallback = this.processLocationConflicts.bind(this);
205     this._noLocationCallback =
206         this.setLocationStatus.bind(this, ZmApptEditView.LOCATION_STATUS_NONE);
207     this.parent.setLocationConflictCallback(this.updatedLocationsConflictChecker.bind(this));
208 
209     this._getRecurrenceSearchResponseCallback =
210         this._getExceptionSearchResponse.bind(this, this._locationConflictAppt);
211     this._getRecurrenceSearchErrorCallback =
212         this._getExceptionSearchError.bind(this, this._locationConflictAppt);
213 
214     if (!this._pendingLocationRequest &&
215          this._scheduleAssistant && this._scheduleAssistant.isInitialized()) {
216         // Trigger an initial location check - the appt may have been saved
217         // with a location that has conflicts.  Only do it if no pending
218         // request and the assistant is initialized (location preferences
219         // are loaded). If !initialized, the locationConflictChecker will
220         // be run when preferences are loaded.
221         this.locationConflictChecker();
222     }
223 }

288 ZmApptEditView.prototype._doExceptionSearchRequest =
289 function() {
290     var numRecurrence = this.getNumLocationConflictRecurrence();
291     var startDate = new Date(this._calItem.startDate);
292     var endTime = ZmApptComposeController.getCheckResourceConflictEndTime(
293         this._locationConflictAppt, startDate, numRecurrence);
294 
295     var jsonObj = {SearchRequest:{_jsns:"urn:zimbraMail"}};
296     var request = jsonObj.SearchRequest;
297 
298     request.sortBy = "dateasc";
299     request.limit = numRecurrence.toString();
300     // AjxEnv.DEFAULT_LOCALE is set to the browser's locale setting in the case
301     // when the user's (or their COS) locale is not set.
302     request.locale = { _content: AjxEnv.DEFAULT_LOCALE };
303     request.calExpandInstStart = startDate.getTime();
304     request.calExpandInstEnd   = endTime;
305     request.types = ZmSearch.TYPE[ZmItem.APPT];
306     request.query = {_content:'item:"' + this._calItem.id.toString() + '"'};
307     var accountName = appCtxt.multiAccounts ? appCtxt.accountList.mainAccount.name : null;
308 
309     var params = {
310         jsonObj:       jsonObj,
311         asyncMode:     true,
312         callback:      this._getExceptionSearchResponse.bind(this),
313         errorCallback: this._getExceptionSearchError.bind(this),
314         noBusyOverlay: true,
315         accountName:   accountName
316     };
317     appCtxt.getAppController().sendRequest(params);
318 }

320 ZmApptEditView.prototype._getExceptionSearchResponse =
321 function(result) {
322 	if (!result) { return; }
323 
324 	var resp;
325     var appt;
326 	try {
327 		resp = result.getResponse();
328 	} catch (ex) {
329 		return;
330 	}
331 
332     // See ZmApptCache.prototype.processSearchResponse
333     var rawAppts = resp.SearchResponse.appt;
334     this._apptExceptionList = new ZmApptList();
335     this._apptExceptionList.loadFromSummaryJs(rawAppts);
336     this._apptExceptionLookup = {};
337 
338     this._locationExceptions = {}
339     for (var i = 0; i < this._apptExceptionList.size(); i++) {
340         appt = this._apptExceptionList.get(i);
341         this._apptExceptionLookup[appt.startDate.getTime()] = appt;
342         if (appt.isException) {
343             // Found an exception, store its location info, using its start date as the key
344             var location = appt._attendees[ZmCalBaseItem.LOCATION];
345             if (!location || (location.length == 0)) {
346                 location = this.getAttendeesFromString(ZmCalBaseItem.LOCATION, appt.location, false);
347                 location = location.getArray();
348             }
349             this._locationExceptions[appt.startDate.getTime()] = location;
350         }
351     }
352     this._enableResolveDialog = true;
353 
354     // Now find the conflicts
355     this._runLocationConflictChecker();
356 };

364 // Callback executed when the CheckResourceConflictRequest completes.
365 // Store the conflict instances (if any) and update the status field
366 ZmApptEditView.prototype.processLocationConflicts =
367 function(inst) {
368     this._inst = inst;
369     var len = inst ? inst.length : 0,
370     locationStatus = ZmApptEditView.LOCATION_STATUS_NONE;
371     for (var i = 0; i < len; i++) {
372         if (this._inst[i].usr) {
373             // Conflict exists for this instance
374             if (this._locationExceptions && this._locationExceptions[this._inst[i].s]) {
375                 // Assume that an existing exception (either persisted to the DB, or set via
376                 // the current use of the resolve dialog) means that the instance conflict is resolved
377                 locationStatus = ZmApptEditView.LOCATION_STATUS_RESOLVED;
378             } else {
379                 // No exception for the instance, using default location which has a conflict
380                 locationStatus = ZmApptEditView.LOCATION_STATUS_CONFLICT;
381                 break;
382             }
383         }
384     }
385 
386     this.setLocationStatus(locationStatus);
387 }

389 ZmApptEditView.prototype.setLocationStatus =
390 function(locationStatus, currentLocationConflict) {
391     var className = "";
392     var statusMessage = "";
393     var linkMessage = "";
394     var msgVisible = false;
395     var linkVisible = false;
396     var statusText = "";
397 
398     if (locationStatus != ZmApptEditView.LOCATION_STATUS_UNDEFINED) {
399         this._locationStatusMode = locationStatus;
400     }
401     if (currentLocationConflict !== undefined) {
402         this._locationConflict  = currentLocationConflict;
403     }
404 
405     // Manage the location suggestion line beneath the location field.
406     switch (this._locationStatusMode) {
407         case ZmApptEditView.LOCATION_STATUS_NONE:
408              // No recurrence conflicts or nothing to check - display based on current conflict flag
409              if (this._locationConflict) {
410                  statusMessage = AjxImg.getImageHtml("Warning_12", "display:inline-block;padding-right:4px;") +
411                                  ZmMsg.locationCurrentConflicts;
412                  className     = "ZmLocationStatusConflict";
413                  msgVisible    = true;
414              } else {
415                  msgVisible    = false;
416              }
417              break;
418         case ZmApptEditView.LOCATION_STATUS_VALIDATING:
419              // The conflict resource check is in progress, show a busy spinner
420              className     = "ZmLocationStatusValidating";
421              // Don't incorporate currentConflict flag - just show validating; It will update upon completion
422              statusMessage = AjxImg.getImageHtml("Wait_16", "display:inline-block;padding-right:4px;") +
423                              ZmMsg.validateLocation;
424              msgVisible    = true;
425              linkVisible   = false;
426              break;
427         case ZmApptEditView.LOCATION_STATUS_CONFLICT:
428              // Unresolved recurrence conflicts - show the 'Resolve Conflicts' link
429              className     = "ZmLocationStatusConflict";
430              statusText    = this._locationConflict ? ZmMsg.locationCurrentAndRecurrenceConflicts :
431                                                       ZmMsg.locationRecurrenceConflicts;
432              statusMessage = AjxImg.getImageHtml("Warning_12", "display:inline-block;padding-right:4px;") +
433                              statusText;
434              linkMessage   = ZmMsg.resolveConflicts;
435              msgVisible    = true;
436              linkVisible   = true;
437              break;
438         case ZmApptEditView.LOCATION_STATUS_RESOLVED:
439              // Resolved conflicts - show the 'View Resolutions' link
440              className     = "ZmLocationStatusResolved";
441              statusMessage = this._locationConflict ? ZmMsg.locationRecurrenceResolvedButCurrentConflict :
442                              ZmMsg.locationRecurrenceConflictsResolved;
443              linkMessage   = ZmMsg.viewResolutions;
444              msgVisible    = true;
445              linkVisible   = true;
446              break;
447         default: break;
448     }
449 
450     Dwt.setVisible(this._locationStatus, msgVisible);
451     if (!this._enableResolveDialog) {
452         // Unable to read the exeptions, prevent the use of the resolve dialog
453         linkVisible = false;
454     }
455 
456     // NOTE: Once CreateAppt/ModifyAppt SOAP API changes are completed (Bug 56464), enable
457     //       the display of the resolve links and the use of the resolve dialog
458     // *** NOT DONE ***
459     linkVisible = false;
460 
461     Dwt.setVisible(this._locationStatusAction, linkVisible);
462     Dwt.setInnerHtml(this._locationStatus, statusMessage);
463     Dwt.setInnerHtml(this._locationStatusAction, linkMessage);
464     this._locationStatus.className = className;
465 }

567 ZmApptEditView.prototype.isValid =
568 function() {
569 	var errorMsg = [];
570 
571 	// check for required subject
572 	var subj = AjxStringUtil.trim(this._subjectField.getValue());
573 
574     //bug: 49990 subject can be empty while proposing new time
575 	if ((subj && subj.length) || this._isProposeTime) {
576 		var allDay = this._allDayCheckbox.checked;
577 		if (!DwtTimeInput.validStartEnd(this._startDateField, this._endDateField, (allDay ? null : this._startTimeSelect), (allDay ? null : this._endTimeSelect))) {
578 				errorMsg.push(ZmMsg.errorInvalidDates);
579 		}
580 
581 	} else {
582 		errorMsg.push(ZmMsg.errorMissingSubject);
583 	}
584     if (this._reminderSelectInput) {
585         var reminderString = this._reminderSelectInput.getValue();
586         var reminderInfo = ZmCalendarApp.parseReminderString(reminderString);
587         if (reminderInfo.reminderValue > ZmApptEditView.REMINDER_MAX_VALUE[reminderInfo.reminderUnits]) {
588             errorMsg.push(ZmMsg.errorInvalidReminderValue);
589         }
590     }
591 	if (errorMsg.length > 0) {
592 		throw errorMsg.join("<br>");
593 	}
594 
595 	return true;
596 };

1735 /**
1736  * One-stop shop for setting address field content. The input may be either a DwtInputField or a
1737  * ZmAddressInputField. The address(es) passed in may be a string, an array, or an AjxVector. The
1738  * latter two types may have a member type of string, AjxEmailAddress, or ZmContact/ZmResource.
1739  * 
1740  * @param addrInput
1741  * @param addrs
1742  * @param type
1743  * @param shortForm
1744  * 
1745  * @private
1746  */
1747 ZmApptEditView.prototype._setAddresses =
1748 function(addrInput, addrs, type, shortForm) {
1749 
1750 	// non-person attendees are shown in short form by default
1751 	shortForm = (shortForm || (type && type != ZmCalBaseItem.PERSON));
1752 
1753 	// if we get a string with multiple email addresses, split it
1754 	if (typeof addrs == "string" && (addrs.indexOf(ZmAppt.ATTENDEES_SEPARATOR) != -1)) {
1755 		var result = AjxEmailAddress.parseEmailString(addrs, type);
1756 		addrs = result.good;
1757 	}
1758 
1759 	if (addrs.isAjxVector) {
1760 		//todo - why aren't we using ZmRecipients way more here? We probably could use a refactoring to unite this code with the
1761 		//mail compose recipients case - same thing as attendees, more or less.
1762 		addrs = ZmRecipients.expandAddrs(addrs);  //expand groups to their individual emails (not DLs).
1763 	}
1764 
1765 	// make sure we have an array to deal with
1766 	addrs = (addrs instanceof AjxVector) ? addrs.getArray() : (typeof addrs == "string") ? [addrs] : addrs;
1767 
1768 	addrInput.clear();
1769 	if (addrs && addrs.length) {
1770         var len = addrs.length;
1771 		for (var i = 0; i < len; i++) {
1772 			var addr = addrs[i];
1773 			if (addr) {
1774 				var addrStr, email, match;
1775 				if (typeof addr == "string") {
1776 					addrStr = addr;
1777 				}
1778 				else if (addr.isAjxEmailAddress) {
1779 					addrStr = addr.toString(shortForm);
1780 					match = {isDL: addr.isGroup && addr.canExpand, email: addrStr};
1781 				}
1782 				else if (addr instanceof ZmContact) {
1783 					email = addr.getEmail(true);
1784                     //bug: 57858 - give preference to lookup email address if its present
1785                     //bug:60427 to show display name format the lookupemail
1786                     addrStr = addr.getLookupEmail() ? (new AjxEmailAddress(addr.getLookupEmail(),null,addr.getFullNameForDisplay())).toString() : ZmApptViewHelper.getAttendeesText(addr, type);
1787                     match = {isDL: addr.isGroup() && addr.canExpand, email: addrStr};
1788 				}
1789 				addrInput.addBubble({address:addrStr, match:match, skipNotify:true});
1790 			}
1791 		}
1792 	}
1793 };
