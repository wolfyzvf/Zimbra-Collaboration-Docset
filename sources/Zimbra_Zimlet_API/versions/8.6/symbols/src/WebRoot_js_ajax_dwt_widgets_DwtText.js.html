 23 /**
 24  * Creates a text control.
 25  * @constructor
 26  * @class
 27  * This class represents a container for a piece of text.
 28  * 
 29  * @author Ross Dargahi
 30  * 
 31  * @param {hash}	params		a hash of parameters
 32  * @param {DwtComposite}      parent	the parent widget
 33  * @param {string}      className		CSS class
 34  * @param {constant}      posStyle		the positioning style (see {@link DwtControl})
 35  * @param {string}      id			an explicit ID to use for the control's HTML element
 36  * 
 37  * @extends		DwtControl
 38  */
 39 DwtText = function(params) {
 40 	if (arguments.length == 0) return;
 41 	params = Dwt.getParams(arguments, DwtText.PARAMS);
 42 	params.className = params.className || "DwtText";
 43 	DwtControl.call(this, params);
 44 
 45 	// we start out empty, so suppress focus
 46 	this.noTab = true;
 47 };
 48 
 49 DwtText.PARAMS = ["parent", "className", "posStyle"];
 50 
 51 DwtText.prototype = new DwtControl;
 52 DwtText.prototype.constructor = DwtText;
 53 
 54 DwtText.prototype.isFocusable = true;
 55 
 56 DwtText.prototype.toString =
 57 function() {
 58 	return "DwtText";
 59 };
 60 
 61 DwtText.prototype.getTabGroupMember = function() {
 62 	return null;
 63 };
 64 
 65 /**
 66  * Sets the text.
 67  * 
 68  * @param	{string}	text		the text
 69  */
 70 DwtText.prototype.setText =
 71 function(text) {
 72 	// only appear in tab order when we have text
 73 	this.noTab = !text;
 74 
 75 	if (!this._textNode) {
 76 		 this._textNode = document.createTextNode(text);
 77 		 this.getHtmlElement().appendChild(this._textNode);
 78 	} else {
 79 		try { // IE mysteriously throws an error sometimes, but still does the right thing
 80 			this._textNode.data = text;
 81 		} catch (e) {}
 82 	}
 83 };
 84 
 85 /**
 86  * Gets the text.
 87  * 
 88  * @return	{string}	the text
 89  */
 90 DwtText.prototype.getText =
 91 function() {
 92 	return this._textNode ? this._textNode.data : "";
 93 };
 94 
 95 /**
 96  * Gets the text node.
 97  * 
 98  * @return	{Object}	the node
 99  */
100 DwtText.prototype.getTextNode =
101 function() {
102 	return this._textNode;
103 };
