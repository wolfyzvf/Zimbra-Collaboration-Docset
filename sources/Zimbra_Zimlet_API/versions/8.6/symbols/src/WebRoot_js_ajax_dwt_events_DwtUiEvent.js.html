 28 DwtUiEvent = function(init) {
 29 	if (arguments.length == 0) return;
 30 	DwtEvent.call(this, true);
 31 	this.reset();
 32 }

 40 DwtUiEvent.prototype.reset =
 41 function() {
 42 	this.dwtObj = null
 43 	this.altKey = false;
 44 	this.ctrlKey = false;
 45 	this.metaKey = false;
 46 	this.shiftKey = false;
 47 	this.target = null;
 48 	this.type = null;
 49 	this.docX = -1;
 50 	this.docY = -1;
 51 	this.elementX = -1;
 52 	this.elementY = -1;
 53 	this.ersatz = false; // True means this event was manufactured
 54 	this._stopPropagation = false;
 55 	this._returnValue = true;
 56 	this._dontCallPreventDefault = false; // True means to allow the the (unusual) situation in Firefox where we
 57 	                                      // want the event handler to return false without calling preventDefault().
 58 }

 60 /**
 61  * Pass caller's "this" as 'target' if using IE and the ev may have come from another window. The target
 62  * will be used to get to the window that generated the event, so the event can be found.
 63  */
 64 DwtUiEvent.getEvent =
 65 function(ev, target) {
 66 	ev = ev || window.event;
 67 	if (ev) { return ev; }
 68 
 69 	// get event from iframe in IE; see http://www.outofhanwell.com/blog/index.php?cat=25
 70 	if (target) {
 71 		DBG.println(AjxDebug.DBG3, "getEvent: Checking other window for event");
 72 		var pw = (target.ownerDocument || target.document || target).parentWindow;
 73 		return pw ? pw.event : null;
 74 	}
 75 }

 77 /**
 78  * Returns the target element of the event.
 79  * 
 80  * @param ev				[Event]		DHTML event
 81  * @param useRelatedTarget	[boolean]*	if true, return element that was related to this event;
 82  * 										for a MOUSEOVER or MOUSEOUT event, that's the element
 83  * 										moved from/to.
 84  */
 85 DwtUiEvent.getTarget =
 86 function(ev, useRelatedTarget)  {
 87 	ev = DwtUiEvent.getEvent(ev);
 88 	if (!ev) { return null; }
 89 	if (!useRelatedTarget) {
 90 		if (ev.target) {
 91 			// if text node (like on Safari) return parent
 92 			return (ev.target.nodeType == 3) ? ev.target.parentNode : ev.target;
 93 		} else if (ev.srcElement) {		// IE
 94 			return ev.srcElement;
 95 		}
 96 	} else {
 97 		if (ev.relatedTarget) {
 98 			return ev.relatedTarget;
 99 		} else if (ev.toElement) {		// IE
100 			return ev.toElement;
101 		} else if (ev.fromElement) {	// IE
102 			return ev.fromElement;
103 		}
104 	}
105 	return null;
106 }

108 /**
109  * Returns the first element with a value for the given property, working its way up the element chain.
110  *
111  * @param ev				[Event]		DHTML event
112  * @param prop				[string]	the name of a property
113  * @param useRelatedTarget	[boolean]*	if true, return element that was related to this event;
114  * @param value				[string]*	expected value of given property
115  */
116 DwtUiEvent.getTargetWithProp =
117 function(ev, prop, useRelatedTarget, value)  {
118 	var htmlEl = DwtUiEvent.getTarget(ev, useRelatedTarget);
119 	while (htmlEl) {
120 		var elValue = Dwt.getAttr(htmlEl, prop);
121 		if (elValue != null && elValue !== "" && (!value || (elValue == value))) {
122 			return htmlEl;
123 		}
124 		htmlEl = htmlEl.parentNode;
125 	}
126 	return null;
127 }

129 /**
130  * Returns the first element with the given class name, working its way up the element chain.
131  *
132  * @param ev				[Event]		DHTML event
133  * @param className			[string]	the requested class name
134  * @param useRelatedTarget	[boolean]*	if true, return element that was related to this event;
135  */
136 DwtUiEvent.getTargetWithClass =
137 function(ev, className, useRelatedTarget)  {
138 	var htmlEl = DwtUiEvent.getTarget(ev, useRelatedTarget);
139 	while (htmlEl && htmlEl.nodeType === 1) {
140 		if (Dwt.hasClass(htmlEl, className)) {
141 			return htmlEl;
142 		}
143 		htmlEl = htmlEl.parentNode;
144 	}
145 	return null;
146 }
147 
148 /**
149  * Returns the first element with values for all of the given properties, working its way up the element chain.
150  *
151  * @param ev				[Event]		DHTML event
152  * @param props				[array]		a list of property names (strings)
153  */
154 DwtUiEvent.getTargetWithProps =
155 function(ev, props)  {
156 	var htmlEl = DwtUiEvent.getTarget(ev);
157 	while (htmlEl) {
158 		var okay = true;
159 		for (var i in props) {
160 			var val = Dwt.getAttr(htmlEl, props[i]);
161 			if (val == null || val === "") {
162 				htmlEl = htmlEl.parentNode;
163 				okay = false;
164 				break;
165 			}
166 		}
167 		if (okay)
168 			return htmlEl;
169 	}
170 	return null;
171 }

173 DwtUiEvent.copy = 
174 function(dest, src) {
175 	dest.altKey = src.altKey;
176 	dest.ctrlKey = src.ctrlKey;
177 	dest.metaKey = src.metaKey;
178 	dest.shiftKey = src.shiftKey;
179 	dest.target = src.target;
180 	dest.type = src.type;
181 	dest.dwtObj = src.dwtObj;
182 	dest.docX = src.docX;
183 	dest.docY = src.docY;
184 	dest.elementX = src.elementX;
185 	dest.elementY = src.elementY;
186 	dest.ersatz = src.ersatz;
187 	dest._stopPropagation = src._stopPropagation;
188 	dest._returnValue = src._returnValue;
189 }

191 /**
192  * Copies properties from the native DHTML event to this DWT event object. The target
193  * control can be optionally fetched by providing true as the second argument.
194  * 
195  * @param ev	[Event]				DHTML event
196  * @param obj	[DwtControl|true]	if true, the target object will be fetched; otherwise
197  * 									used to set target object if present
198  */
199 DwtUiEvent.prototype.setFromDhtmlEvent =
200 function(ev, obj) {
201 	ev = DwtUiEvent.getEvent(ev);
202 	if (!ev) { return; }
203 	this.altKey = ev.altKey;
204 	this.ctrlKey = ev.ctrlKey;
205 	this.metaKey = ev.metaKey;
206 	this.shiftKey = ev.shiftKey;
207 	this.type = ev.type;
208 	this.target = DwtUiEvent.getTarget(ev);
209 	this.dwtObj = (obj === true) ? DwtControl.getTargetControl(ev) : obj;
210 
211 	// Compute document coordinates
212 	if (ev.pageX != null) {
213 		this.docX = ev.pageX;
214 		this.docY = ev.pageY;
215 	} else if (ev.clientX != null) {
216 		this.docX = ev.clientX + document.body.scrollLeft - document.body.clientLeft;
217 		this.docY = ev.clientY + document.body.scrollTop - document.body.clientTop;
218 		if (document.body.parentElement) {
219 				var bodParent = document.body.parentElement;
220 				this.docX += bodParent.scrollLeft - bodParent.clientLeft;
221 				this.docY += bodParent.scrollTop - bodParent.clientTop;
222 		}
223 	}
224 	// Compute Element coordinates
225 	if (ev.offsetX != null) {
226 		this.elementX = ev.offsetX;
227 		this.elementY = ev.offsetY;
228 	} else if (!AjxEnv.isWebKitBased && ev.layerX != null) {
229 		this.elementX = ev.layerX;
230 		this.elementY = ev.layerY;
231 	} else { // fail hard for others
232 		this.elementX = Dwt.DEFAULT;
233 		this.elementY = Dwt.DEFAULT;
234 	}
235 	
236 	this.ersatz = false;
237 	return ev;
238 }

240 DwtUiEvent.prototype.setToDhtmlEvent =
241 function(ev) {
242 	DwtUiEvent.setBehaviour(ev, this._stopPropagation, this._returnValue, this._dontCallPreventDefault);
243 }
244 
245 DwtUiEvent.setBehaviour =
246 function(ev, stopPropagation, allowDefault, dontCallPreventDefault) {
247 	var dhtmlEv = DwtUiEvent.getEvent(ev);
248 	DwtUiEvent.setDhtmlBehaviour(dhtmlEv, stopPropagation, allowDefault, dontCallPreventDefault);
249 };
250 
251 DwtUiEvent.setDhtmlBehaviour =
252 function(dhtmlEv, stopPropagation, allowDefault, dontCallPreventDefault) {
253 
254 	dhtmlEv = DwtUiEvent.getEvent(dhtmlEv);
255 	if (!dhtmlEv) { return; }
256 
257 	// stopPropagation is referring to the function found in Mozilla's event object
258 	if (dhtmlEv.stopPropagation != null) {
259 		if (stopPropagation)
260 			dhtmlEv.stopPropagation();
261 		if (!allowDefault && !dontCallPreventDefault)
262 			dhtmlEv.preventDefault();
263 	} else {
264 		// IE only..
265 		dhtmlEv.returnValue = allowDefault;
266 		dhtmlEv.cancelBubble = stopPropagation;
267 	}
268 };

270 /**
271  * @deprecated
272  * Use DwtControl.getTargetControl() instead.
273  * 
274  * Returns a control (DWT object) based on the event, by finding the event target and using
275  * its reference to a DWT object in the element's "dwtObj" expando property.
276  * 
277  * @param ev				[Event]		DHTML event
278  * @param useRelatedTarget	[boolean]*	if true, return element that was related to this event;
279  */
280 DwtUiEvent.getDwtObjFromEvent =
281 function(ev, useRelatedTarget) {
282 	var htmlEl = DwtUiEvent.getTargetWithProp(ev, "dwtObj", useRelatedTarget);
283 	return htmlEl ? Dwt.getObjectFromElement(htmlEl) : null;
284 };
285 
286 /**
287  * @deprecated
288  * Instead, do something like this:
289  * 		var htmlEl = DwtUiEvent.getTargetWithProp(ev, "myProp");
290  * 		var obj = DwtControl.findControl(htmlEl);
291  * 
292  * Returns a control (DWT object) based on the event, by finding the event target with the
293  * given property and using its reference to a DWT object.
294  * 
295  * @param ev				[Event]		DHTML event
296  * @param useRelatedTarget	[boolean]*	if true, return element that was related to this event;
297  */
298 DwtUiEvent.getDwtObjWithProp =
299 function(ev, prop) {
300 	var htmlEl = DwtUiEvent.getTargetWithProps(ev, ["dwtObj", prop]);
301 	return htmlEl ? Dwt.getObjectFromElement(htmlEl) : null;
302 };
