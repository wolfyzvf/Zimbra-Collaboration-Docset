 22 /**
 23  * @overview
 24  * This file contains the search application class.
 25  */
 26 
 27 /**
 28  * Creates and initializes the search application.
 29  * @constructor
 30  * @class
 31  * The search app manages user-initiated searches.
 32  *
 33  * @param	{DwtControl}	container		the container
 34  * @param	{ZmController}	parentController	the parent window controller (set by the child window)
 35  * 
 36  * @author Conrad Damon
 37  * 
 38  * @extends		ZmApp
 39  */
 40 ZmSearchApp = function(container, parentController) {
 41 
 42 	ZmApp.call(this, ZmApp.SEARCH, container, parentController);
 43 };
 44 
 45 ZmSearchApp.prototype = new ZmApp;
 46 ZmSearchApp.prototype.constructor = ZmSearchApp;
 47 
 48 ZmSearchApp.prototype.isZmSearchApp = true;
 49 ZmSearchApp.prototype.toString = function() {	return "ZmSearchApp"; };
 50 
 51 ZmApp.SEARCH					= ZmId.APP_SEARCH;
 52 ZmApp.CLASS[ZmApp.SEARCH]		= "ZmSearchApp";
 53 ZmApp.SETTING[ZmApp.SEARCH]		= ZmSetting.SEARCH_ENABLED;
 54 
 55 ZmSearchApp.CONTROLLER_CLASS = "ZmSearchResultsController";
 56 
 57 ZmSearchApp.prototype.getSearchResultsController =
 58 function(sessionId, appName) {
 59 	return this.getSessionController({
 60 				controllerClass:	ZmSearchApp.CONTROLLER_CLASS,
 61 				sessionId:			sessionId,
 62 				appName:			appName
 63 			});
 64 };
 65 
 66 // override so we don't try to set overview panel content
 67 ZmSearchApp.prototype.activate =
 68 function(active) {
 69 	this._active = active;
 70 };
 71 
 72 // Not hooked up for activate, but it will be called after displaying the search results
 73 ZmSearchApp.prototype.resetWebClientOfflineOperations =
 74 function(searchResultsController) {
 75 	ZmApp.prototype.resetWebClientOfflineOperations.apply(this);
 76 	if (!searchResultsController) {
 77 		var controllerType = this.getTypeFromController(ZmSearchApp.CONTROLLER_CLASS);
 78 		var sessionId = this.getCurrentSessionId(controllerType);
 79 		searchResultsController = this.getSearchResultsController(sessionId);
 80 	}
 81 	// Only Save affected currently
 82 	var searchResultsToolBar = searchResultsController && searchResultsController._toolbar;
 83 	var saveButton = searchResultsToolBar && searchResultsToolBar.getButton(ZmSearchToolBar.SAVE_BUTTON);
 84 	if (saveButton) {
 85 		saveButton.setEnabled(!appCtxt.isWebClientOffline());
 86 	}
 87 };
