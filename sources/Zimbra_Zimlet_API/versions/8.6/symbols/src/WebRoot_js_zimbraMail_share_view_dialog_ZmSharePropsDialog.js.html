 26 /**
 27  * Creates a share properties dialog.
 28  * @class
 29  * This class represents a share properties dialog.
 30  * 
 31  * @param	{DwtComposite}	shell		the parent
 32  * @param	{String}	className		the class name
 33  *  
 34  * @extends		DwtDialog
 35  */
 36 ZmSharePropsDialog = function(shell, className) {
 37 	className = className || "ZmSharePropsDialog";
 38 	DwtDialog.call(this, {parent:shell, className:className, title:ZmMsg.shareProperties, id:"ShareDialog"});
 39 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 40 
 41 	var aifParams = {
 42 		parent:		this,
 43 		inputId:	"ShareDialog_grantee"
 44 	}
 45 
 46 	this._grantee = new ZmAddressInputField(aifParams);
 47 	this._grantee.setData(Dwt.KEY_OBJECT, this);
 48 	Dwt.associateElementWithObject(this._grantee, this);
 49 
 50 	this._granteeInput = this._grantee.getInputElement();
 51 	this._granteeInputId = this._grantee._htmlElId;
 52 	Dwt.associateElementWithObject(this._granteeInput, this);
 53 
 54 	// create auto-completer
 55 	if (appCtxt.get(ZmSetting.CONTACTS_ENABLED) || appCtxt.get(ZmSetting.GAL_ENABLED)) {
 56 		var params = {
 57 			dataClass:		appCtxt.getAutocompleter(),
 58 			options:		{massDLComplete:true},
 59 			matchValue:		ZmAutocomplete.AC_VALUE_EMAIL,
 60 			keyUpCallback:	this._acKeyUpListener.bind(this),
 61 			contextId:		this.toString()
 62 		};
 63 		this._acAddrSelectList = new ZmAutocompleteListView(params);
 64 		this._acAddrSelectList.handle(this._granteeInput, this._granteeInputId);
 65 		this._grantee.setAutocompleteListView(this._acAddrSelectList);
 66 	}
 67 
 68 	// set view
 69 	this.setView(this._createView());
 70 };

114 ZmSharePropsDialog.prototype.popup =
115 function(mode, object, share) {
116 
117 	this._shareMode = mode;
118 	this._object = object;
119 	this._share = share;
120 
121 	this._nameEl.innerHTML = AjxStringUtil.htmlEncode(object.name);
122 	this._typeEl.innerHTML = ZmMsg[ZmOrganizer.FOLDER_KEY[this._object.type]] || ZmMsg.folder;
123 	// TODO: False until server handling of the flag is added
124 	//if (object.type == ZmOrganizer.FOLDER) {
125 	if (false) {
126 		this._markReadEl.innerHTML = object.globalMarkRead ? ZmMsg.sharingDialogGlobalMarkRead :
127                                                              ZmMsg.sharingDialogPerUserMarkRead;
128 		this._props.setPropertyVisible(this._markReadId, true)
129 	} else {
130 		this._props.setPropertyVisible(this._markReadId, false)
131 	}
132 
133 	var isNewShare = (this._shareMode == ZmSharePropsDialog.NEW);
134 	var isUserShare = share ? share.isUser() || share.isGroup() : true;
135 	var isGuestShare = share ? share.isGuest() : false;
136 	var isPublicShare = share ? share.isPublic() : false;
137 	var supportsPublic = object.supportsPublicAccess();
138 	var externalEnabled = appCtxt.get(ZmSetting.SHARING_EXTERNAL_ENABLED);
139 	var publicEnabled = appCtxt.get(ZmSetting.SHARING_PUBLIC_ENABLED);
140 
141 	this._userRadioEl.checked = isUserShare;
142 	this._userRadioEl.disabled = !isNewShare;
143 	this._guestRadioEl.checked = isGuestShare;
144 	this._guestRadioEl.disabled = !(externalEnabled && isNewShare  && supportsPublic);
145 	this._publicRadioEl.checked = isPublicShare;
146 	this._publicRadioEl.disabled = !(publicEnabled && isNewShare && supportsPublic && (object.type !== ZmOrganizer.FOLDER));
147 
148 	var type = this._getType(isUserShare, isGuestShare, isPublicShare);
149 	this._handleShareWith(type);
150 
151 	var grantee = "", password  = "";
152 	if (share) {
153 		if (isGuestShare) {
154 			grantee = share.grantee.id;
155 			password = share.link.pw;
156 		} else {
157 			grantee = (share.grantee.name || ZmMsg.userUnknown);
158 			password = share.grantee.id;
159 		}
160 	}
161 	this._grantee.clear();
162 	this._grantee.setValue(grantee, true);
163 	this._grantee.setEnabled(isNewShare);
164 
165 	// Make all the properties visible so that their elements are in the
166 	// document. Otherwise, we won't be able to get a handle on them to perform
167 	// operations.
168 	this._props.setPropertyVisible(this._shareWithOptsId, true);
169 	//this._shareWithOptsProps.setPropertyVisible(this._passwordId, true);
170 	this._props.setPropertyVisible(this._shareWithBreakId, true);
171 
172 	//this._passwordButton.setVisible(!isNewShare);
173 	//this._shareWithOptsProps.setPropertyVisible(this._passwordId, isGuestShare);
174 	//this._passwordInput.setValue(password, true);
175 
176 	if (this._inheritEl) {
177 		this._inheritEl.checked = share ? share.link.inh : isNewShare;
178 	}
179 
180 	var perm = share && share.link.perm;
181 
182 	if (perm != null) {
183 		perm = perm.replace(/-./g, "");
184 		this._privateEl.checked = (perm.indexOf(ZmShare.PERM_PRIVATE) != -1);
185 		perm = perm.replace(/p/g, "");
186 		var role = ZmShare._getRoleFromPerm(perm);
187 		var radioEl = this._radioElByRole[role];
188 		if (radioEl) {
189 			radioEl.checked = true;
190 		}
191 	}
192 
193 	this._privatePermissionEnabled = object.supportsPrivatePermission();
194 	this._privatePermission.setVisible(object.supportsPrivatePermission());
195 
196 	if (perm == null || (perm == this._viewerRadioEl.value)) {
197 		this._viewerRadioEl.checked = true;
198 	} else if (perm == this._noneRadioEl.value) {
199 		this._noneRadioEl.checked = true;
200 	} else if (perm == this._managerRadioEl.value) {
201 		this._managerRadioEl.checked = true;
202 	} else if (perm == this._adminRadioEl.value) {
203 		this._adminRadioEl.checked = true;
204 	}
205 
206 	// Force a reply if new share
207 	this._reply.setReplyType(ZmShareReply.STANDARD);
208 	this._reply.setReplyNote("");
209 
210 	this._populateUrls();
211 
212 	var size = this.getSize();
213 	Dwt.setSize(this._granteeInput, 0.6*size.x);
214 	//Dwt.setSize(this._passwordInput.getInputElement(), 0.6*size.x);
215 
216 	DwtDialog.prototype.popup.call(this);
217 	this.setButtonEnabled(DwtDialog.OK_BUTTON, false);
218 	if (isNewShare) {
219 		this._userRadioEl.checked = true;
220 		this._grantee.focus();
221 	}
222 
223 	if (appCtxt.multiAccounts) {
224 		var acct = object.account || appCtxt.accountList.mainAccount;
225 		this._acAddrSelectList.setActiveAccount(acct);
226 	}

229 ZmSharePropsDialog.prototype._populateUrls =
230 function() {
231 
232     var acct, restUrl;
233     if (appCtxt.multiAccounts) {
234         acct = this._object.getAccount();
235         restUrl = this._object.getRestUrl(acct);
236     } else {
237         restUrl = this._object.getRestUrl();
238     }    
239 	if (appCtxt.isOffline) {
240 		var remoteUri = appCtxt.get(ZmSetting.OFFLINE_REMOTE_SERVER_URI, null, acct);
241 		restUrl = remoteUri + restUrl.substring((restUrl.indexOf("/",7)));
242 	}
243 	var url = AjxStringUtil.htmlEncode(restUrl).replace(/&/g,'%26');
244 	var text = url;
245 	if (text.length > 50) {
246 		var length = text.length - 50;
247 		var index = (text.length - length) / 2;
248 		text = text.substr(0, index) + "..." + text.substr(index + length);
249 	}
250 
251 	var proto = (location.protocol === ZmSetting.PROTO_HTTPS) ? "webcals:" : "webcal:";
252     var webcalURL = proto + url.substring((url.indexOf("//")));
253     var webcalText = webcalURL;
254     if (webcalText.length > 50) {
255 		var length = webcalText.length - 50;
256 		var index = (webcalText.length - length) / 2;
257 		webcalText = webcalText.substr(0, index) + "..." + webcalText.substr(index + length);
258 	}
259 
260 	var isRestFolder = this._object.type != ZmOrganizer.FOLDER;
261 	this._urlGroup.setVisible(isRestFolder);
262 	if (isRestFolder) {
263 		if (this._object.type == ZmOrganizer.CALENDAR) {
264 			this._urlEl.innerHTML = [
265 				"<div>", ZmMsg.ics, ":    ",
266 					'<a target=_new id="SharePropsURL_ICS" href="',url,'">',text,"</a>",
267 				"</div>",
268 				"<div>", ZmMsg.view, ":  ",
269 					'<a target=_new id="SharePropsURL_view" href="',url,'.html">',text,".html</a>",
270 				"</div>",
271                 "<div>", ZmMsg.outlookURL, ":  ",
272 					'<a target=_new id="SharePropsURL_Outlook" href="',webcalURL,'">',webcalText,"</a>",
273 				"</div>"
274 			].join("");
275 		} else {
276 			this._urlEl.innerHTML = [
277 				"<div style='padding-left:2em;'>",
278 					'<a target=_new id="SharePropsURL" href="',url,'">',text,"</a>",
279 				"</div>"
280 			].join("");
281 		}
282 	}
283 };

315 ZmSharePropsDialog.prototype._handleOkButton =
316 function(event) {
317 	var isUserShare = this._userRadioEl.checked;
318 	var isGuestShare = this._guestRadioEl.checked;
319 	var isPublicShare = this._publicRadioEl.checked;
320 	var shareWithMyself = false;
321 
322 	var parsedEmailsFromText = AjxEmailAddress.parseEmailString(this._granteeInput.value);
323 	var goodEmailsFromText = parsedEmailsFromText.good.getArray();
324 	var goodEmailsFromBubbles =  this._grantee.getAddresses();
325 
326 	var goodEmails = goodEmailsFromBubbles.concat(goodEmailsFromText);
327 	var badEmails = parsedEmailsFromText.bad.getArray();
328 
329 	// validate input
330 	if (!isPublicShare) {
331 		var error;
332 		if (badEmails.length) {
333 			error = AjxMessageFormat.format(AjxMsg.invalidEmailAddrValue, AjxStringUtil.htmlEncode(this._granteeInput.value));
334 		}
335 		else if (!goodEmails.length) {
336 			error = AjxMsg.valueIsRequired;
337 		}
338 
339 		if (error) {
340 			var dialog = appCtxt.getErrorDialog();
341 			dialog.setMessage(error);
342 			dialog.popup(null, true);
343 
344 			if (!goodEmails.length) {
345 				return;
346 			}
347 		}
348 	}
349 
350     var replyType = this._reply.getReplyType();
351     if (replyType != ZmShareReply.NONE) {
352         var notes = (replyType == ZmShareReply.QUICK) ? this._reply.getReplyNote() : "";
353     }
354 
355 	var shares = [];
356 	if (this._shareMode == ZmSharePropsDialog.NEW) {
357 		var type = this._getType(isUserShare, isGuestShare, isPublicShare);
358 		if (!isPublicShare) {
359 			for (var i = 0; i < goodEmails.length; i++) {
360 				// bug fix #26428 - exclude me from list of addresses
361 				var addr = goodEmails[i];
362 				//bug#66610: allow Calendar Sharing with addresses present in zimbraAllowFromAddress
363 				var allowLocal;
364 				var excludeAllowFromAddress = true;
365 				if (appCtxt.isMyAddress(addr, allowLocal, excludeAllowFromAddress)) {
366 					shareWithMyself = true;
367 					continue;
368 				}
369 
370 				var share = this._setUpShare();
371 				share.grantee.name = addr;
372 				share.grantee.type = type;
373 				shares.push(share);
374 			}
375 		} else {
376 			var share = this._setUpShare();
377 			share.grantee.type = type;
378 			shares.push(share);
379 		}
380 	} else {
381 		shares.push(this._setUpShare(this._share)); // editing perms on a share
382 	}
383 	
384 	// Since we may be sharing with multiple users, use a batch command
385 	var accountName = appCtxt.multiAccounts ? this._object.getAccount().name : null;
386 	var batchCmd = new ZmBatchCommand(null, accountName);
387 	var perm = this._getPermsFromRole();
388 	//var pw = isGuestShare && this._passwordInput.getValue();
389 	if (shares && shares.length == 0 && shareWithMyself) {
390 		var msgDlg = appCtxt.getMsgDialog(true);
391 		msgDlg.setMessage(ZmMsg.sharingErrorWithSelf,DwtMessageDialog.INFO_STYLE);
392 		msgDlg.setTitle(ZmMsg.sharing);
393 		msgDlg.popup();
394 		return;
395 	}
396 	for (var i = 0; i < shares.length; i++) {
397 		var share = shares[i];
398 		if (perm != share.link.perm) {
399 			var cmd = new AjxCallback(share, share.grant,
400 			                          [perm, null, notes,
401 			                           replyType, this._shareMode]);
402 			batchCmd.add(cmd);
403 		}
404 	}
405 	if (batchCmd.size() > 0) {
406 		var respCallback = !isPublicShare
407 			? (new AjxCallback(this, this._handleResponseBatchCmd, [shares])) : null;
408 		batchCmd.run(respCallback);
409 	}
410 	
411 	this.popdown();
412 };

414 ZmSharePropsDialog.prototype._handleResponseBatchCmd =
415 function(shares, result) {
416 
417 
418     var response = result.getResponse();
419     var batchResponse = response.BatchResponse;
420 
421     //bug:67698 Do not send notification on failed share
422     if(batchResponse.Fault){
423        appCtxt.setStatusMsg(ZmMsg.shareNotCreated,ZmStatusView.LEVEL_WARNING);
424        return false;
425     }
426     else{
427         if (!shares || (shares && shares.length == 0)) { return; }
428         var ignore = this._getFaultyEmails(result);
429         var replyType = this._reply.getReplyType();
430         if (replyType != ZmShareReply.NONE) {
431             var notes = (replyType == ZmShareReply.QUICK) ? this._reply.getReplyNote() : "";
432             var guestnotes;
433             var batchCmd;
434 
435             if (shares.length > 1) {
436                 var accountName = appCtxt.multiAccounts ? this._object.getAccount().name : null;
437                 batchCmd = new ZmBatchCommand(false, accountName, true);
438             }
439 
440             for (var i = 0; i < shares.length; i++) {
441                 var share = shares[i];
442                 var email = share.grantee.email || share.grantee.id;
443                 if (!email) {
444                     // last resort: check if grantee name is a valid email address
445                     if (AjxEmailAddress.isValid(share.grantee.name))
446                         email = share.grantee.name;
447                 }
448 
449                 if (!email || (email && ignore[email])) { continue; }
450 
451                 var addrs = new AjxVector();
452                 var addr = new AjxEmailAddress(email, AjxEmailAddress.TO);
453                 addrs.add(addr);
454 
455                 var tmpShare = new ZmShare({object:share.object});
456 
457                 tmpShare.grantee.id = share.grantee.id;
458                 tmpShare.grantee.email = email;
459                 tmpShare.grantee.name = share.grantee.name;
460 
461                 // REVISIT: What if you have delegated access???
462                 if (tmpShare.object.isRemote()) {
463                     tmpShare.grantor.id = tmpShare.object.zid;
464                     tmpShare.grantor.email = tmpShare.object.owner;
465                     tmpShare.grantor.name = tmpShare.grantor.email;
466                     tmpShare.link.id = tmpShare.object.rid;
467                     tmpShare.link.name = tmpShare.object.oname || tmpShare.object.name;
468                 } else {
469                     // bug: 50936  get setting for respective account
470                     // to prevent sharing the default account unintentionally
471                     tmpShare.grantor.id = appCtxt.get(ZmSetting.USERID, null, this._object.getAccount());
472                     tmpShare.grantor.email = appCtxt.get(ZmSetting.USERNAME, null, this._object.getAccount());
473                     tmpShare.grantor.name = appCtxt.get(ZmSetting.DISPLAY_NAME, null, this._object.getAccount()) || tmpShare.grantor.email;
474                     tmpShare.link.id = tmpShare.object.id;
475                     tmpShare.link.name = tmpShare.object.name;
476                 }
477                 // If folder is not synced before sharing, link ID might have changed in ZD.
478                 // Always get from response.
479                 if(appCtxt.isOffline) {
480                     var linkId = this.getLinkIdfromResp(result);
481                     if(linkId) {
482                         tmpShare.link.id =  [tmpShare.grantor.id, linkId].join(":");
483                     }
484                 }
485 
486                 tmpShare.link.perm = share.link.perm;
487                 tmpShare.link.view = ZmOrganizer.getViewName(tmpShare.object.type);
488                 tmpShare.link.inh = this._inheritEl ? this._inheritEl.checked : true;
489 
490                 if (this._guestRadioEl.checked) {
491                     if (!this._guestFormatter) {
492                         this._guestFormatter = new AjxMessageFormat(ZmMsg.shareCalWithGuestNotes);
493                     }
494 
495                     var url = share.object.getRestUrl();
496                     url = url.replace(/&/g,'%26');
497                     if (appCtxt.isOffline) {
498                         var remoteUri = appCtxt.get(ZmSetting.OFFLINE_REMOTE_SERVER_URI);
499                         url = remoteUri + url.substring((url.indexOf("/",7)));
500                     }
501 
502                     //bug:34647 added webcal url for subscribing to outlook/ical on a click
503                     var webcalURL = "webcals:" + url.substring((url.indexOf("//")));
504 
505                     //var password = this._passwordInput.getValue();
506                     guestnotes = this._guestFormatter.format([url, webcalURL, email, "", notes]);
507                 }
508                 tmpShare.notes = guestnotes || notes;
509 
510                 /*
511                     tmpShare.sendMessage(this._shareMode, addrs, null, batchCmd);
512                 */
513             }
514             if (batchCmd)
515                 batchCmd.run();
516 
517             var shareMsg = (this._shareMode==ZmSharePropsDialog.NEW)?ZmMsg.shareCreatedSubject:ZmMsg.shareModifiedSubject;
518             appCtxt.setStatusMsg(shareMsg);
519 
520         }
521     }
522 };

524 ZmSharePropsDialog.prototype.getLinkIdfromResp =
525 function(result){
526 
527     if (!result) { return; }
528     var resp = result.getResponse().BatchResponse.FolderActionResponse || [];
529     if (resp.length > 0 && resp[0].action) {
530         return resp[0].action.id;
531     } else {
532         return null;
533     }
534 };

536 // HACK: grep the Faults in BatchResponse and sift out the bad emails
537 ZmSharePropsDialog.prototype._getFaultyEmails =
538 function(result) {
539 
540 	if (!result) { return; }
541 	var noSuchAccount = "no such account: ";
542 	var bad = {};
543 	var fault = result.getResponse().BatchResponse.Fault || [];
544 	for (var i = 0; i < fault.length; i++) {
545 		var reason = fault[i].Reason.Text;
546 		if (reason.indexOf(noSuchAccount) == 0) {
547 			bad[reason.substring(noSuchAccount.length)] = true;
548 		}
549 	}
550 	return bad;
551 };

589 ZmSharePropsDialog._enableFieldsOnEdit =
590 function(dialog) {
591 	var isEdit = dialog._mode == ZmSharePropsDialog.EDIT;
592 
593 	var isUserShare = dialog._userRadioEl.checked;
594 	var isPublicShare = dialog._publicRadioEl.checked;
595 	var isGuestShare = dialog._guestRadioEl.checked;
596 
597 	dialog._privatePermission.setVisible(dialog._privatePermissionEnabled && !dialog._noneRadioEl.checked && !isPublicShare);
598 
599 	var hasEmail = AjxStringUtil.trim(dialog._grantee.getValue()) != "";
600 	//var hasPassword = AjxStringUtil.trim(dialog._passwordInput.getValue()) != "";
601 
602 	var enabled = isEdit ||
603 				  isPublicShare ||
604 				  (isUserShare && hasEmail) ||
605 				  (isGuestShare && hasEmail);
606 	dialog.setButtonEnabled(DwtDialog.OK_BUTTON, enabled);
607 };

618 ZmSharePropsDialog.prototype._handleShareWith = function(type) {
619 	var isUserShare = type == ZmShare.TYPE_USER;
620 	var isGuestShare = type == ZmShare.TYPE_GUEST;
621 	var isPublicShare = type == ZmShare.TYPE_PUBLIC;
622 
623     // TODO - Currently external sharing is enabled for briefcase only.
624     var guestRadioLabelEl = document.getElementById("LblShareWith_external");
625 
626     if (appCtxt.getCurrentApp().getName() === ZmId.APP_BRIEFCASE) {
627         this._rolesGroup.setVisible(isUserShare || isGuestShare);
628         guestRadioLabelEl.innerText = ZmMsg.shareWithExternalGuest;
629     }
630     else {
631 	    this._rolesGroup.setVisible(isUserShare);
632         guestRadioLabelEl.innerText = ZmMsg.shareWithGuest;
633     }
634 	this._messageGroup.setVisible(!isPublicShare);
635 	this._privatePermission.setVisible(this._privatePermissionEnabled && !isPublicShare);
636 
637     var adminRadioRow = document.getElementById("ShareRole_Row_" + ZmShare.ROLE_ADMIN);
638 
639     if (isGuestShare) {
640         this._reply && this._reply.setReplyOptions(ZmShareReply.EXTERNAL_USER_OPTIONS);
641         adminRadioRow.style.display = 'none';
642     }
643     else {
644         this._reply && this._reply.setReplyOptions(ZmShareReply.DEFAULT_OPTIONS);
645         this._reply.setReplyType(ZmShareReply.STANDARD);
646         adminRadioRow.style.display = '';
647     }
648 	this._props.setPropertyVisible(this._shareWithOptsId, !isPublicShare);
649 	//this._shareWithOptsProps.setPropertyVisible(this._passwordId, isGuestShare);
650 	this._props.setPropertyVisible(this._shareWithBreakId, !isPublicShare);
651     this._setAutoComplete(isGuestShare);
652 
653 	if (!isUserShare) {
654 		this._viewerRadioEl.checked = true;
655 	}
656 };

658 /**
659  * Returns a perms string based on the user's selection of a role and privacy.
660  */
661 ZmSharePropsDialog.prototype._getPermsFromRole =
662 function() {
663 	var role = ZmShare.ROLE_NONE;
664 	if (this._viewerRadioEl.checked) {
665 		role = ZmShare.ROLE_VIEWER;
666 	}
667 	if (this._managerRadioEl.checked) {
668 		role = ZmShare.ROLE_MANAGER;
669 	}
670 	if (this._adminRadioEl.checked) {
671 		role = ZmShare.ROLE_ADMIN;
672 	}
673 	var perm = ZmShare.ROLE_PERMS[role];
674 	if (perm && this._privatePermissionEnabled && this._privateEl.checked) {
675 		perm += ZmShare.PERM_PRIVATE;
676 	}
677 	return perm;
678 };
