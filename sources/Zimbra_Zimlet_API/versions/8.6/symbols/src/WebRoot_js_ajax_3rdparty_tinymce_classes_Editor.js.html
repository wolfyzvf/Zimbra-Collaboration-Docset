277 	Editor.prototype = {
278 		/**
279 		 * Renderes the editor/adds it to the page.
280 		 *
281 		 * @method render
282 		 */
283 		render: function() {
284 			var self = this, settings = self.settings, id = self.id, suffix = self.suffix;
285 
286 			function readyHandler() {
287 				DOM.unbind(window, 'ready', readyHandler);
288 				self.render();
289 			}
290 
291 			// Page is not loaded yet, wait for it
292 			if (!Event.domLoaded) {
293 				DOM.bind(window, 'ready', readyHandler);
294 				return;
295 			}
296 
297 			// Element not found, then skip initialization
298 			if (!self.getElement()) {
299 				return;
300 			}
301 
302 			// No editable support old iOS versions etc
303 			if (!Env.contentEditable) {
304 				return;
305 			}
306 
307 			// Hide target element early to prevent content flashing
308 			if (!settings.inline) {
309 				self.orgVisibility = self.getElement().style.visibility;
310 				self.getElement().style.visibility = 'hidden';
311 			} else {
312 				self.inline = true;
313 			}
314 
315 			var form = self.getElement().form || DOM.getParent(id, 'form');
316 			if (form) {
317 				self.formElement = form;
318 
319 				// Add hidden input for non input elements inside form elements
320 				if (settings.hidden_input && !/TEXTAREA|INPUT/i.test(self.getElement().nodeName)) {
321 					DOM.insertAfter(DOM.create('input', {type: 'hidden', name: id}), id);
322 					self.hasHiddenInput = true;
323 				}
324 
325 				// Pass submit/reset from form to editor instance
326 				self.formEventDelegate = function(e) {
327 					self.fire(e.type, e);
328 				};
329 
330 				DOM.bind(form, 'submit reset', self.formEventDelegate);
331 
332 				// Reset contents in editor when the form is reset
333 				self.on('reset', function() {
334 					self.setContent(self.startContent, {format: 'raw'});
335 				});
336 
337 				// Check page uses id="submit" or name="submit" for it's submit button
338 				if (settings.submit_patch && !form.submit.nodeType && !form.submit.length && !form._mceOldSubmit) {
339 					form._mceOldSubmit = form.submit;
340 					form.submit = function() {
341 						self.editorManager.triggerSave();
342 						self.isNotDirty = true;
343 
344 						return form._mceOldSubmit(form);
345 					};
346 				}
347 			}
348 
349 			/**
350 			 * Window manager reference, use this to open new windows and dialogs.
351 			 *
352 			 * @property windowManager
353 			 * @type tinymce.WindowManager
354 			 * @example
355 			 * // Shows an alert message
356 			 * tinymce.activeEditor.windowManager.alert('Hello world!');
357 			 *
358 			 * // Opens a new dialog with the file.htm file and the size 320x240
359 			 * // It also adds a custom parameter this can be retrieved by using tinyMCEPopup.getWindowArg inside the dialog.
360 			 * tinymce.activeEditor.windowManager.open({
361 			 *    url: 'file.htm',
362 			 *    width: 320,
363 			 *    height: 240
364 			 * }, {
365 			 *    custom_param: 1
366 			 * });
367 			 */
368 			self.windowManager = new WindowManager(self);
369 
370 			if (settings.encoding == 'xml') {
371 				self.on('GetContent', function(e) {
372 					if (e.save) {
373 						e.content = DOM.encode(e.content);
374 					}
375 				});
376 			}
377 
378 			if (settings.add_form_submit_trigger) {
379 				self.on('submit', function() {
380 					if (self.initialized) {
381 						self.save();
382 					}
383 				});
384 			}
385 
386 			if (settings.add_unload_trigger) {
387 				self._beforeUnload = function() {
388 					if (self.initialized && !self.destroyed && !self.isHidden()) {
389 						self.save({format: 'raw', no_events: true, set_dirty: false});
390 					}
391 				};
392 
393 				self.editorManager.on('BeforeUnload', self._beforeUnload);
394 			}
395 
396 			// Load scripts
397 			function loadScripts() {
398 				var scriptLoader = ScriptLoader.ScriptLoader;
399 
400 				if (settings.language && settings.language != 'en' && !settings.language_url) {
401 					settings.language_url = self.editorManager.baseURL + '/langs/' + settings.language + '.js';
402 				}
403 
404 				if (settings.language_url) {
405 					scriptLoader.add(settings.language_url);
406 				}
407 
408 				if (settings.theme && typeof settings.theme != "function" &&
409 					settings.theme.charAt(0) != '-' && !ThemeManager.urls[settings.theme]) {
410 					var themeUrl = settings.theme_url;
411 
412 					if (themeUrl) {
413 						themeUrl = self.documentBaseURI.toAbsolute(themeUrl);
414 					} else {
415 						themeUrl = 'themes/' + settings.theme + '/theme' + suffix + '.js';
416 					}
417 
418 					ThemeManager.load(settings.theme, themeUrl);
419 				}
420 
421 				if (Tools.isArray(settings.plugins)) {
422 					settings.plugins = settings.plugins.join(' ');
423 				}
424 
425 				each(settings.external_plugins, function(url, name) {
426 					PluginManager.load(name, url);
427 					settings.plugins += ' ' + name;
428 				});
429 
430 				each(settings.plugins.split(/[ ,]/), function(plugin) {
431 					plugin = trim(plugin);
432 
433 					if (plugin && !PluginManager.urls[plugin]) {
434 						if (plugin.charAt(0) == '-') {
435 							plugin = plugin.substr(1, plugin.length);
436 
437 							var dependencies = PluginManager.dependencies(plugin);
438 
439 							each(dependencies, function(dep) {
440 								var defaultSettings = {
441 									prefix: 'plugins/',
442 									resource: dep,
443 									suffix: '/plugin' + suffix + '.js'
444 								};
445 
446 								dep = PluginManager.createUrl(defaultSettings, dep);
447 								PluginManager.load(dep.resource, dep);
448 							});
449 						} else {
450 							PluginManager.load(plugin, {
451 								prefix: 'plugins/',
452 								resource: plugin,
453 								suffix: '/plugin' + suffix + '.js'
454 							});
455 						}
456 					}
457 				});
458 
459 				scriptLoader.loadQueue(function() {
460 					if (!self.removed) {
461 						self.init();
462 					}
463 				});
464 			}
465 
466 			loadScripts();
467 		},
468 
469 		/**
470 		 * Initializes the editor this will be called automatically when
471 		 * all plugins/themes and language packs are loaded by the rendered method.
472 		 * This method will setup the iframe and create the theme and plugin instances.
473 		 *
474 		 * @method init
475 		 */
476 		init: function() {
477 			var self = this, settings = self.settings, elm = self.getElement();
478 			var w, h, minHeight, n, o, Theme, url, bodyId, bodyClass, re, i, initializedPlugins = [];
479 
480 			self.rtl = this.editorManager.i18n.rtl;
481 			self.editorManager.add(self);
482 
483 			settings.aria_label = settings.aria_label || DOM.getAttrib(elm, 'aria-label', self.getLang('aria.rich_text_area'));
484 
485 			/**
486 			 * Reference to the theme instance that was used to generate the UI.
487 			 *
488 			 * @property theme
489 			 * @type tinymce.Theme
490 			 * @example
491 			 * // Executes a method on the theme directly
492 			 * tinymce.activeEditor.theme.someMethod();
493 			 */
494 			if (settings.theme) {
495 				if (typeof settings.theme != "function") {
496 					settings.theme = settings.theme.replace(/-/, '');
497 					Theme = ThemeManager.get(settings.theme);
498 					self.theme = new Theme(self, ThemeManager.urls[settings.theme]);
499 
500 					if (self.theme.init) {
501 						self.theme.init(self, ThemeManager.urls[settings.theme] || self.documentBaseUrl.replace(/\/$/, ''), self.$);
502 					}
503 				} else {
504 					self.theme = settings.theme;
505 				}
506 			}
507 
508 			function initPlugin(plugin) {
509 				var Plugin = PluginManager.get(plugin), pluginUrl, pluginInstance;
510 
511 				pluginUrl = PluginManager.urls[plugin] || self.documentBaseUrl.replace(/\/$/, '');
512 				plugin = trim(plugin);
513 				if (Plugin && inArray(initializedPlugins, plugin) === -1) {
514 					each(PluginManager.dependencies(plugin), function(dep) {
515 						initPlugin(dep);
516 					});
517 
518 					pluginInstance = new Plugin(self, pluginUrl, self.$);
519 
520 					self.plugins[plugin] = pluginInstance;
521 
522 					if (pluginInstance.init) {
523 						pluginInstance.init(self, pluginUrl);
524 						initializedPlugins.push(plugin);
525 					}
526 				}
527 			}
528 
529 			// Create all plugins
530 			each(settings.plugins.replace(/\-/g, '').split(/[ ,]/), initPlugin);
531 
532 			// Measure box
533 			if (settings.render_ui && self.theme) {
534 				self.orgDisplay = elm.style.display;
535 
536 				if (typeof settings.theme != "function") {
537 					w = settings.width || elm.style.width || elm.offsetWidth;
538 					h = settings.height || elm.style.height || elm.offsetHeight;
539 					minHeight = settings.min_height || 100;
540 					re = /^[0-9\.]+(|px)$/i;
541 
542 					if (re.test('' + w)) {
543 						w = Math.max(parseInt(w, 10), 100);
544 					}
545 
546 					if (re.test('' + h)) {
547 						h = Math.max(parseInt(h, 10), minHeight);
548 					}
549 
550 					// Render UI
551 					o = self.theme.renderUI({
552 						targetNode: elm,
553 						width: w,
554 						height: h,
555 						deltaWidth: settings.delta_width,
556 						deltaHeight: settings.delta_height
557 					});
558 
559 					// Resize editor
560 					if (!settings.content_editable) {
561 						h = (o.iframeHeight || h) + (typeof(h) == 'number' ? (o.deltaHeight || 0) : '');
562 						if (h < minHeight) {
563 							h = minHeight;
564 						}
565 					}
566 				} else {
567 					o = settings.theme(self, elm);
568 
569 					// Convert element type to id:s
570 					if (o.editorContainer.nodeType) {
571 						o.editorContainer = o.editorContainer.id = o.editorContainer.id || self.id + "_parent";
572 					}
573 
574 					// Convert element type to id:s
575 					if (o.iframeContainer.nodeType) {
576 						o.iframeContainer = o.iframeContainer.id = o.iframeContainer.id || self.id + "_iframecontainer";
577 					}
578 
579 					// Use specified iframe height or the targets offsetHeight
580 					h = o.iframeHeight || elm.offsetHeight;
581 				}
582 
583 				self.editorContainer = o.editorContainer;
584 			}
585 
586 			// Load specified content CSS last
587 			if (settings.content_css) {
588 				each(explode(settings.content_css), function(u) {
589 					self.contentCSS.push(self.documentBaseURI.toAbsolute(u));
590 				});
591 			}
592 
593 			// Load specified content CSS last
594 			if (settings.content_style) {
595 				self.contentStyles.push(settings.content_style);
596 			}
597 
598 			// Content editable mode ends here
599 			if (settings.content_editable) {
600 				elm = n = o = null; // Fix IE leak
601 				return self.initContentBody();
602 			}
603 
604 			self.iframeHTML = settings.doctype + '<html><head>';
605 
606 			// We only need to override paths if we have to
607 			// IE has a bug where it remove site absolute urls to relative ones if this is specified
608 			if (settings.document_base_url != self.documentBaseUrl) {
609 				self.iframeHTML += '<base href="' + self.documentBaseURI.getURI() + '" />';
610 			}
611 
612 			// IE8 doesn't support carets behind images setting ie7_compat would force IE8+ to run in IE7 compat mode.
613 			if (!Env.caretAfter && settings.ie7_compat) {
614 				self.iframeHTML += '<meta http-equiv="X-UA-Compatible" content="IE=7" />';
615 			}
616 
617 			self.iframeHTML += '<meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />';
618 
619 			// Load the CSS by injecting them into the HTML this will reduce "flicker"
620 			for (i = 0; i < self.contentCSS.length; i++) {
621 				var cssUrl = self.contentCSS[i];
622 				self.iframeHTML += '<link type="text/css" rel="stylesheet" href="' + cssUrl + '" />';
623 				self.loadedCSS[cssUrl] = true;
624 			}
625 
626 			bodyId = settings.body_id || 'tinymce';
627 			if (bodyId.indexOf('=') != -1) {
628 				bodyId = self.getParam('body_id', '', 'hash');
629 				bodyId = bodyId[self.id] || bodyId;
630 			}
631 
632 			bodyClass = settings.body_class || '';
633 			if (bodyClass.indexOf('=') != -1) {
634 				bodyClass = self.getParam('body_class', '', 'hash');
635 				bodyClass = bodyClass[self.id] || '';
636 			}
637 
638 			if (settings.content_security_policy) {
639 				self.iframeHTML += '<meta http-equiv="Content-Security-Policy" content="' + settings.content_security_policy + '" />';
