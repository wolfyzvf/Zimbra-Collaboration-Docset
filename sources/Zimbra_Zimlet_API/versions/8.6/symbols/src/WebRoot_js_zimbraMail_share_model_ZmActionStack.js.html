 26 /**
 27  * @class
 28  * Creates a stack of undoable actions (ZmAction objects)
 29  *
 30  * @param	{int}	[maxLength]		The maximum size of the stack. Defaults to 0, meaning no limit
 31  *
 32  * Adding actions to a full stack will pop the oldest actions off
 33  */
 34 ZmActionStack = function(maxLength) {
 35 	this._stack = [];
 36 	this._pointer = -1;
 37 	this._maxLength = maxLength || 0; // 0 means no limit
 38 };
 39 
 40 ZmEvent.S_ACTION			= "ACTION";
 41 
 42 ZmActionStack.validTypes	= [ZmId.ORG_FOLDER, ZmId.ITEM_MSG, ZmId.ITEM_CONV,
 43                                ZmId.ITEM_CONTACT, ZmId.ITEM_GROUP,
 44                                ZmId.ITEM_BRIEFCASE, ZmId.ORG_BRIEFCASE,
 45                                ZmId.ITEM_TASK, ZmId.ORG_TASKS
 46                               ]; // Set ZmActionStack.validTypes to false to allow all item types
 47 
 48 ZmActionStack.prototype.toString = function() {
 49 	return "ZmActionStack";
 50 };
 51 
 52 /**
 53  * Logs a raw action, interpreting the params and creates a ZmAction object that is pushed onto the stack and returned
 54  * 
 55  * @param {String}	op			operation to perform. Currently supported are "move", "trash", "spam" and "!spam"
 56  * @param {Hash}	[attrs] 	attributes for the operation. Pretty much the same as what the backend expects, e.g. "l" for the destination folderId of a move
 57 
 58  * @param {String}	[items] 	array of items to perform the action for. Valid types are specified in ZmActionStack.validTypes. Only one of [items],[item],[ids] or [id] should be specified; the first one found is used, ignoring the rest.
 59  * @param {String}	[item] 		item to perform the action for, if there is only one item. Accomplishes the same as putting the item in an array and giving it as [items]
 60  * @param {String}	[ids] 		array of ids of items to perform the action for.
 61  * @param {String}	[id] 		id of item to perform the action for, if there is only one. Accomplishes the same as putting the id in an array and giving it as [ids].
 62  */
 63 ZmActionStack.prototype.logAction = function(params) {
 64 
 65 	var op = params.op,
 66 	    items = [];
 67 
 68 	if (params.items) {
 69 		for (var i = 0; i < params.items.length; i++) {
 70 			var item = params.items[i];
 71 			if (item && this._isValidType(item.type)) {
 72 				items.push(item);
 73 			}
 74 		}
 75 	}
 76 	else if (params.item) {
 77 		if (params.item && this._isValidType(params.item.type)) {
 78 			items.push(params.item);
 79 		}
 80 	}
 81 	else if (params.ids) {
 82 		for (var i = 0; i < params.ids.length; i++) {
 83 			var item = appCtxt.getById(params.ids[i]);
 84 			if (item && this._isValidType(item.type)) {
 85 				items.push(item);
 86 			}
 87 		}
 88 	}
 89 	else if (params.id) {
 90 		var item = appCtxt.getById(params.id);
 91 		if (item && this._isValidType(item.type)) {
 92 			items.push(item);
 93 		}
 94 	}
 95 
 96 	var attrs = params.attrs;
 97 
 98 	// for a conv, create a list of undoable msg moves so msgs can be restored to their disparate original folders
 99 	for (var i = 0; i < items.length; i++) {
100 		var item = items[i];
101 		if (item.type === ZmItem.CONV) {
102 			var tcon = attrs && attrs.tcon;
103 			for (var msgId in item.msgFolder) {
104 				var folderId = item.msgFolder[msgId],
105 					tconCode = ZmFolder.TCON_CODE[folderId];
106 
107 				// if tcon kept us from moving a msg, no need to undo it
108 				if (!tcon || tcon.indexOf(tconCode) === -1) {
109 					items.push({
110 						isConvMsg:  true,
111 						id:         msgId,
112 						type:       ZmItem.MSG,
113 						folderId:   folderId,
114 						list:       { type: ZmItem.MSG }    // hack to expose item.list.type
115 					});
116 				}
117 			}
118 		}
119 	}
120 
121 	var multi = items.length > 1;
122 
123 	var action = null;
124 	var folderId;
125 	switch (op) {
126 		case "trash":
127 			folderId = ZmFolder.ID_TRASH;
128 			break;
129 		case "spam":
130 			folderId = ZmFolder.ID_SPAM;
131 			break;
132 		case "move":
133 		case "!spam":
134 			folderId = attrs.l;
135 			break;
136 	}
137 
138 	var folder = appCtxt.getById(folderId);
139 	if (folder && !folder.isRemote()) { // Enable undo only when destination folder exists (it should!!) and is not remote (bug #51656)
140 		switch (op) {
141 			case "trash":
142 			case "move":
143 			case "spam":
144 			case "!spam":
145 				for (var i = 0; i < items.length; i++) {
146 					var item = items[i];
147 					var moveAction;
148 				
149 					if (item instanceof ZmItem) {
150 						if (!item.isShared()) { // Moving shared items is not undoable
151 							moveAction = new ZmItemMoveAction(item, item.getFolderId(), folderId, op);
152 						}
153 					}
154 					else if (item instanceof ZmOrganizer) {
155 						if (!item.isRemote()) { // Moving remote organizers is not undoable
156 							moveAction = new ZmOrganizerMoveAction(item, item.parent.id, folderId, op);
157 						}
158 					}
159 					else if (item.isConvMsg) {
160 						if (!appCtxt.isRemoteId(item.id)) {
161 							moveAction = new ZmItemMoveAction(item, item.folderId, folderId, op);
162 						}
163 					}
164 					if (moveAction) {
165 						if (multi) {
166 							if (!action) action = new ZmCompositeAction(folderId);
167 							action.addAction(moveAction);
168 						} else {
169 							action = moveAction;
170 						}
171 					}
172 				}
173 				break;
174 		}
175 		if (action) {
176 			this._push(action);
177 		}
178 	}
179 
180 	return action;
181 };

183 /**
184  * Returns whether there are actions that can be undone
185  */
186 ZmActionStack.prototype.canUndo = function() {
187 	return this._pointer >= 0;
188 };
189 
190 /**
191  * Returns whether there are actions that can be redone
192  */
193 ZmActionStack.prototype.canRedo = function() {
194 	return this._pointer < this._stack.length - 1;
195 };
196 
197 
198 /**
199  * Returns whether the next undo action has completed
200  */
201 ZmActionStack.prototype.actionIsComplete = function() {
202 	return this.canUndo() && this._current().getComplete();
203 };
204 
205 /**
206  * Attaches a completion callback to the current action
207  */
208 ZmActionStack.prototype.onComplete = function(callback) {
209 	var action = this._current();
210 	if (action) {
211 		action.onComplete(callback);
212 	}
213 };
214 
215 /**
216  * Undoes the current action (if applicable) and moves the internal pointer
217  */
218 ZmActionStack.prototype.undo = function() {
219 	if (this.canUndo()) {
220 		var action = this._pop();
221 		action.undo();
222 	}
223 };
224 
225 /**
226  * Redoes the current action (if applicable) and moves the internal pointer
227  */
228 ZmActionStack.prototype.redo = function() {
229 	if (this.canRedo()) {
230 		var action = this._stack[++this._pointer];
231 		action.redo();
232 	}
233 };
234 
235 /**
236  * Puts an action into the stack at the current position
237  * If we're not at the top of the stack (ie. undoes have been performed), we kill all later actions (so redoing the undone actions is no longer possible)
238  */
239 ZmActionStack.prototype._push = function(action) {
240 	if (action && action instanceof ZmAction) {
241 		var next = this._pointer + 1;
242 		while (this._maxLength && next>=this._maxLength) {
243 			// Stack size is reached, shift off actions until we're under the limit
244 			this._stack.shift();
245 			next--;
246 		}
247 		this._stack[next] = action;
248 		this._stack.length = next+1; // Kill all actions after pointer
249 		this._pointer = next;
250 	}
251 };
252 
253 /**
254  * Returns the action at the current position and moves the pointer
255  */
256 ZmActionStack.prototype._pop = function() {
257 	return this.canUndo() ? this._stack[this._pointer--] : null;
258 };
259 
260 /**
261  * Returns the action at the current position, does not move the pointer
262  */
263 ZmActionStack.prototype._current = function() {
264 	return this.canUndo() ? this._stack[this._pointer] : null;
265 };
266 
267 /**
268  * Returns true if the given type is valid.
269  */
270 ZmActionStack.prototype._isValidType = function(type) {
271 	return !ZmActionStack.validTypes || AjxUtil.indexOf(ZmActionStack.validTypes, type) !== -1;
272 };
