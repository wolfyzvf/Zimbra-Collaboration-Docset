 27 /**
 28  * Creates a mountpoint organizer.
 29  * @class
 30  * This class represents a mountpoint organizer. This class can be used to represent generic
 31  * mountpoints in an overview tree but is mostly used as a utility to create mountpoints.
 32  * 
 33  * @param	{Hash}	params		a hash of parameters
 34  * 
 35  * @extends		ZmOrganizer
 36  */
 37 ZmMountpoint = function(params) {
 38 	params.type = ZmOrganizer.MOUNTPOINT;
 39 	ZmOrganizer.call(this, params);
 40 	this.view = params.view;
 41 }
 42 
 43 ZmMountpoint.prototype = new ZmOrganizer;
 44 ZmMountpoint.prototype.constructor = ZmMountpoint;
 45 
 46 // Constants
 47 ZmMountpoint.__CREATE_PARAMS = AjxUtil.arrayAsHash(["l", "name", "zid", "rid", "owner", "path", "view", "color", "rgb", "f"]);
 48 
 49 
 50 // Public Methods
 51 
 52 /**
 53  * Returns a string representation of the object.
 54  * 
 55  * @return		{String}		a string representation of the object
 56  */
 57 ZmMountpoint.prototype.toString =
 58 function() {
 59 	return "ZmMountpoint";
 60 };
 61 
 62 /**
 63  * Creates the mountpoint.
 64  * 
 65  * @param {Hash}	params		a hash of parameters
 66  * @param	{String}	params.name		the name
 67  */
 68 ZmMountpoint.create =
 69 function(params, callback) {
 70 	var soapDoc = AjxSoapDoc.create("CreateMountpointRequest", "urn:zimbraMail");
 71 
 72 	var linkNode = soapDoc.set("link");
 73 	for (var p in params) {
 74 		if (!(p in ZmMountpoint.__CREATE_PARAMS)) continue;
 75 		linkNode.setAttribute(p, params[p]);
 76 	}
 77 
 78 	var errorCallback = new AjxCallback(null, ZmMountpoint._handleCreateError, params.name);
 79 	appCtxt.getAppController().sendRequest({soapDoc:soapDoc,
 80 											asyncMode:true,
 81 											callback:callback,
 82 											errorCallback:errorCallback});
 83 };
 84 
 85 /**
 86  * @private
 87  */
 88 ZmMountpoint._handleCreateError =
 89 function(name, response) {
 90 
 91 	var msg;
 92 	if (response.code == ZmCsfeException.SVC_PERM_DENIED || response.code == ZmCsfeException.MAIL_NO_SUCH_FOLDER) {
 93 		msg = ZmCsfeException.getErrorMsg(response.code);
 94 	} else if (response.code == ZmCsfeException.MAIL_ALREADY_EXISTS) {
 95         var type = appCtxt.getFolderTree(appCtxt.getActiveAccount()).getFolderTypeByName(name);
 96 		msg = AjxMessageFormat.format(ZmMsg.errorAlreadyExists, [name,type.toLowerCase()]);
 97 	}
 98 	if (msg) {
 99 		appCtxt.getAppController().popupErrorDialog(msg, null, null, true);
100 		return true;
101 	}
102 };
