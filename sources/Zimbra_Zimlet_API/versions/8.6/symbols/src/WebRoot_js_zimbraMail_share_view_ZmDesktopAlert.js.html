 22 /**
 23  * Singleton alert class that alerts the user by popping up a message on the desktop.
 24  * @class
 25  * @private
 26  */
 27 ZmDesktopAlert = function() {
 28     if (window.webkitNotifications) {
 29         this.useWebkit = true;
 30 	} else if (window.Notification) {
 31 		this.useNotification = true;
 32     } else if (appCtxt.isOffline && window.platform && (AjxEnv.isWindows || AjxEnv.isMac)) {
 33         this.usePrism = true;
 34     }
 35 };
 36 
 37 ZmDesktopAlert.prototype = new ZmAlert;
 38 ZmDesktopAlert.prototype.constructor = ZmDesktopAlert;
 39 
 40 ZmDesktopAlert.prototype.toString =
 41 function() {
 42 	return "ZmDesktopAlert";
 43 };
 44 
 45 ZmDesktopAlert.getInstance =
 46 function() {
 47 	return ZmDesktopAlert.INSTANCE = ZmDesktopAlert.INSTANCE || new ZmDesktopAlert();
 48 };
 49 
 50 /**
 51  * Returns text to show in a prefs page next to the checkbox to enable this type of alert.
 52  */
 53 ZmDesktopAlert.prototype.getDisplayText =
 54 function() {
 55     if (this.useWebkit || this.useNotification) {
 56        return ZmMsg.showPopup;
 57     } else if (this.usePrism) {
 58 		return AjxEnv.isMac ? ZmMsg.showPopupMac : ZmMsg.showPopup;
 59 	}
 60 };
 61 
 62 ZmDesktopAlert.prototype.start =
 63 function(title, message, sticky) {
 64     if (this.useWebkit) {
 65         var allowedCallback = this._showWebkitNotification.bind(this, title, message, sticky);
 66         this._checkWebkitPermission(allowedCallback);
 67 	} else if (this.useNotification) {
 68 		var notificationCallback = this._showNotification.bind(this, title, message, sticky);
 69 		this._checkNotificationPermission(notificationCallback);
 70     } else if (this.usePrism) {
 71 		if (AjxEnv.isMac) {
 72 			try {
 73 				window.platform.showNotification(title, message, "resource://webapp/icons/default/launcher.icns");
 74 			} catch (err) {}
 75 		}
 76 		else if (AjxEnv.isWindows) {
 77 			try {
 78 				window.platform.icon().showNotification(title, message, 5);
 79 			} catch (err) {}
 80 		}
 81 	}
 82 };

 84 /* Checks if we have permission to use webkit notifications. If so, or when the user
 85  * grants permission, allowedCallback is called.
 86  */
 87 ZmDesktopAlert.prototype._checkWebkitPermission =
 88 function(allowedCallback) {
 89     var allowed = window.webkitNotifications.checkPermission() == 0;
 90     if (allowed) {
 91         allowedCallback();
 92     } else if (!ZmDesktopAlert.requestedPermission) {
 93         ZmDesktopAlert.requestedPermission = true; // Prevents multiple permission requests in one session.
 94         window.webkitNotifications.requestPermission(this._checkWebkitPermission.bind(this, allowedCallback));
 95     }
 96 };
 97 
 98 ZmDesktopAlert.prototype._showWebkitNotification =
 99 function(title, message, sticky) {
100 	sticky = sticky || false;
101     // Icon: I chose to use the favIcon because it's already overridable by skins.
102     // It's a little ugly though.
103     // change for bug#67359: Broken notification image in chrome browser
104     // //var icon = window.favIconUrl;
105 	var icon = skin.hints.notificationBanner;
106     var popup = window.webkitNotifications.createNotification(icon, title, message);
107     popup.show();
108 	popup.onclick = function() {popup.cancel();};
109     if (sticky) {
110         if (!ZmDesktopAlert.notificationArray) {
111             ZmDesktopAlert.notificationArray = [];
112         }
113         ZmDesktopAlert.notificationArray.push(popup);
114     }
115     else {
116         // Close the popup after 5 seconds.
117         setTimeout(popup.cancel.bind(popup), 5000);
118     }
119 };
120 
121 /* Checks if we have permission to use the notification api. If so, or when the user
122  * grants permission, allowedCallback is called.
123  */
124 ZmDesktopAlert.prototype._checkNotificationPermission = function(allowedCallback) {
125 	var allowed = window.Notification.permission === 'granted';
126 	if (allowed) {
127 		allowedCallback();
128 	} else if (!ZmDesktopAlert.requestedPermission) {
129 		ZmDesktopAlert.requestedPermission = true; // Prevents multiple permission requests in one session.
130 		// Currently, cannot directly call requestPermission.  Re-test when Chrome 37 is released
131 		//window.Notification.requestPermission(this._checkNotificationPermission.bind(this, allowedCallback));
132 		var requestCallback = this._checkNotificationPermission.bind(this, allowedCallback);
133 		this.requestRequestPermission(requestCallback);
134 	}
135 };
136 
137 // Chrome Notification only allows requesting permission in response to a user action, not a programmatic call.
138 // The issue may be fixed in Chrome 37, whenever that comes out.  See:
139 //   https://code.google.com/p/chromium/issues/detail?id=274284
140 ZmDesktopAlert.prototype.requestRequestPermission = function(requestCallback) {
141 	var msgDialog = appCtxt.getYesNoMsgDialog();
142 	var callback = 	this._doRequestPermission.bind(this, msgDialog, requestCallback);
143 	msgDialog.registerCallback(DwtDialog.YES_BUTTON, callback);
144 	msgDialog.setMessage(ZmMsg.notificationPermission, DwtMessageDialog.INFO_STYLE);
145 	msgDialog.popup();
146 };
147 
148 ZmDesktopAlert.prototype._doRequestPermission = function(msgDialog, requestCallback) {
149 	msgDialog.popdown();
150 	window.Notification.requestPermission(requestCallback);
151 }
152 
153 ZmDesktopAlert.prototype._showNotification = function(title, message, sticky) {
154 	var icon = skin.hints.notificationBanner;
155 
156 	var popup = new Notification(title, { body: message, icon: icon});
157 	//popup.show();
158 	popup.onclick = function() {popup.close();};
159 	if (sticky) {
160 		if (!ZmDesktopAlert.notificationArray) {
161 			ZmDesktopAlert.notificationArray = [];
162 		}
163 		ZmDesktopAlert.notificationArray.push(popup);
164 	}
165 	else {
166 		// Close the popup after 5 seconds.
167 		setTimeout(popup.close.bind(popup), 5000);
168 	}
169 };
170 
171 ZmDesktopAlert.prototype._notifyServiceCallback =
172 function(title, message, service) {
173 	try {
174 		service.show({ title: title, message: message }, function(){});
175 	} catch (err) {}
176 };

178 /**
179  * Closes desktop notification if any during onbeforeunload event
180  */
181 ZmDesktopAlert.closeNotification =
182 function() {
183     var notificationArray = ZmDesktopAlert.notificationArray,
184         popup;
185 
186     if (notificationArray) {
187         while (popup = notificationArray.pop()) {
188             //notifications may be already closed by the user go for try catch
189             try {
190                 popup.cancel();
191             }
192             catch (e) {
193             }
194         }
195     }
196 };
