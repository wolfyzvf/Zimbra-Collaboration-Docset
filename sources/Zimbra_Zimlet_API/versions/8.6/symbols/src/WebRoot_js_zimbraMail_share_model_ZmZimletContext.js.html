 27 /**
 28  * Creates the Zimlet context.
 29  * @class
 30  * This class represents the Zimlet context.
 31  * 
 32  * @param	{String}	id		the id
 33  * @param	{ZmZimletBase}	zimlet	the Zimlet
 34  * 
 35  */
 36 ZmZimletContext = function(id, zimlet) {
 37 
 38 	// sanitize JSON here
 39 	this.json = ZmZimletContext.sanitize(zimlet, "zimlet", ZmZimletContext.RE_ARRAY_ELEMENTS);
 40 
 41 	this.id = id;
 42 	this.icon = "ZimbraIcon";
 43 	this.ctxt = zimlet.zimletContext;
 44 	this.config = zimlet.zimletConfig;
 45 	zimlet = zimlet.zimlet[0];
 46 	/**
 47 	 * The zimlet name.
 48 	 * @type String
 49 	 */
 50 	this.name = zimlet.name;
 51 	this._url = this.ctxt[0].baseUrl;
 52 	this.priority = this.ctxt[0].priority;
 53 	/**
 54 	 * The zimlet description.
 55 	 * @type String
 56 	 */
 57 	this.description = zimlet.description;
 58 	/**
 59 	 * The zimlet version.
 60 	 * @type String
 61 	 */
 62 	this.version = zimlet.version;
 63 	this.label = zimlet.label;
 64 	this.includes = this.json.zimlet.include || [];
 65 	this.includes.push([appContextPath, "/messages/", this.name, ".js?v=", cacheKillerVersion].join(""));
 66 	this.includeCSS = this.json.zimlet.includeCSS;
 67 
 68 	if (zimlet.serverExtension && zimlet.serverExtension[0].hasKeyword) {
 69 		this.keyword = zimlet.serverExtension[0].hasKeyword;
 70 	}
 71 
 72 	DBG.println(AjxDebug.DBG2, "Zimlets - context: " + this.name);
 73 
 74 	this.targets = {};
 75 	var targets = (zimlet.target || "main").split(" ");
 76 	for (var i = 0; i < targets.length; i++) {
 77 		this.targets[targets[i]] = true;
 78 	}
 79 
 80 	this._contentActionMenu = null;
 81 	if (zimlet.contentObject) {
 82 		this.contentObject = zimlet.contentObject[0];
 83 		if (this.contentObject.type) {
 84 			this.type = this.contentObject.type;
 85 		}
 86 		if (this.contentObject.contextMenu) {
 87 			if (this.contentObject.contextMenu instanceof Array) {
 88 				this.contentObject.contextMenu = this.contentObject.contextMenu[0];
 89 			}
 90 			this._contentActionMenu = new AjxCallback(this, this._makeMenu,[this.contentObject.contextMenu.menuItem]);
 91 		}
 92 	}
 93 
 94 	this._panelActionMenu = null;
 95 	if (zimlet.zimletPanelItem && !appCtxt.isChildWindow) {
 96 		this.zimletPanelItem = zimlet.zimletPanelItem[0];
 97 		if (this.zimletPanelItem.label) {
 98 			this.zimletPanelItem.label = this.process(this.zimletPanelItem.label);
 99 		}
100 		if (this.zimletPanelItem.toolTipText && this.zimletPanelItem.toolTipText[0]) {
101 			this.zimletPanelItem.toolTipText = this.process(this.zimletPanelItem.toolTipText[0]._content);
102 		}
103 		if (this.zimletPanelItem.icon) {
104 			this.icon = this.zimletPanelItem.icon;
105 		}
106 		if (this.zimletPanelItem.contextMenu) {
107 			if (this.zimletPanelItem.contextMenu instanceof Array) {
108 				this.zimletPanelItem.contextMenu = this.zimletPanelItem.contextMenu[0];
109 			}
110 			this._panelActionMenu = new AjxCallback(this, this._makeMenu, [this.zimletPanelItem.contextMenu.menuItem]);
111 		}
112 		if (this.zimletPanelItem.onClick instanceof Array) {
113 			this.zimletPanelItem.onClick = this.zimletPanelItem.onClick[0];
114 		}
115 		if (this.zimletPanelItem.onDoubleClick instanceof Array) {
116 			this.zimletPanelItem.onDoubleClick = this.zimletPanelItem.onDoubleClick[0];
117 		}
118 	}
119 
120 	if (zimlet.handlerObject) {
121 		this.handlerObject = zimlet.handlerObject[0]._content;
122 	}
123 
124 	var portlet = zimlet.portlet && zimlet.portlet[0];
125 	if (portlet) {
126 		portlet = ZmZimletContext.sanitize(portlet);
127 		portlet.portletProperties = (portlet.portletProperties && portlet.portletProperties.property) || {};
128 		this.portlet = portlet;
129 	}
130 
131 	this.userProperties = zimlet.userProperties ? zimlet.userProperties[0] : [];
132 	this._propsById = {};
133 	if (zimlet.userProperties) {
134 		this._translateUserProp();
135 	}
136 
137 	if (this.config) {
138 		if (this.config instanceof Array ||
139 			(appCtxt.isChildWindow && this.config.length && this.config[0])) {
140 
141 			this.config = this.config[0];
142 		}
143 		this._translateConfig();
144 	}
145 
146 	this._handleMenuItemSelected = new AjxListener(this, this._handleMenuItemSelected);
147 };

170 /**
171  * This function creates a 'sane' JSON object, given one returned by the
172  * Zimbra server.
173  *<p>
174  * It will basically remove unnecessary arrays and create String objects for
175  * those tags that have text data, so that we don't need to dereference lots of
176  * arrays and use _content. It does the job that the server should do.  *grin*
177  * </p>
178  * <b>WARNING</b>: usage of an attribute named "length" may give sporadic
179  * results, since we convert tags that have text content to Strings.
180  *
181  * @param obj -- array or object, whatever was given by server
182  * @param tag -- the tag of this object, if it's an array
183  * @param wantarray_re -- RegExp that matches tags that must remain an array
184  *
185  * @return -- sanitized object
186  * 
187  * @private
188  */
189 ZmZimletContext.sanitize =
190 function(obj, tag, wantarray_re) {
191 	function doit(obj, tag) {
192 		var cool_json, val, i;
193 		if (obj instanceof DwtControl) { //Don't recurse into DwtControls, causes too much recursion
194 			return obj;
195 		}
196 		else if (obj instanceof Array || AjxUtil.isArray1(obj)) {
197 			if (obj.length == 1 && !(wantarray_re && wantarray_re.test(tag))) {
198 				cool_json = doit(obj[0], tag);
199 			} else {
200 				cool_json = [];
201 				for (i = 0; i < obj.length; ++i) {
202 					cool_json[i] = doit(obj[i], tag);
203 				}
204 			}
205 		}
206 		else if (obj && typeof obj == "object") {
207 			if (obj._content) {
208 				cool_json = new String(obj._content);
209 			} else {
210 				cool_json = {};
211 			}
212 			for (i in obj) {
213 				cool_json[i] = doit(obj[i], i);
214 			}
215 		} else {
216 			cool_json = obj;
217 		}
218 		return cool_json;
219 	}
220 	return doit(obj, tag);
221 };

233 /**
234  * <strong>Note:</strong>
235  * This method is called by ZmZimletMgr#_finished_loadIncludes.
236  * 
237  * @private
238  */
239 ZmZimletContext.prototype._finished_loadIncludes =
240 function() {
241     // localize messages
242     this.label = this.label && this.processMessage(this.label);
243     this.description = this.description && this.processMessage(this.description);
244 
245 	var CTOR = this.handlerObject ? window[this.handlerObject] : ZmZimletBase;
246 	if (!CTOR) {
247 		DBG.println("zimlet handler not defined ("+this.handlerObject+")");
248 		return;
249 	}
250 	this.handlerObject = new CTOR();
251 	if (!this.handlerObject._init) {
252 		var msg = [
253 			"ERROR - Zimlet handler (",
254 			this.name,
255 			") not defined. ",
256 			"Make sure the Zimlet name and handlerObject defined in ",
257 			this.name,
258 			".xml are different."
259 		].join("");
260 		DBG.println(AjxDebug.DBG1, msg);
261 	}
262 	this.handlerObject._init(this, DwtShell.getShell(window));
263 	if (this.contentObject) {
264 		appCtxt.getZimletMgr().registerContentZimlet(this.handlerObject, this.type, this.priority);
265 	}
266 	this.handlerObject.init();
267 	this.handlerObject._zimletContext = this;
268 	// If it has an _id then we need to make sure the treeItem is up-to-date now
269 	// that the i18n files have loaded.
270 	if (this._id) {
271 		var acct = appCtxt.isOffline ? appCtxt.accountList.mainAccount : null;
272 		var tree = appCtxt.getZimletTree(acct);
273 		if (tree) {
274 			var zimletItem = tree.getById(this._id);
275 			zimletItem.resetNames();
276 		}
277 	}
278 
279 	// initialize portlets
280 	if (appCtxt.get(ZmSetting.PORTAL_ENABLED) && !appCtxt.isChildWindow) {
281 		var params = {
282 			name: "Portal",
283 			callback: new AjxCallback(this, this._finished_loadIncludes2)
284 		};
285 		DBG.println("------------------- REQUIRING Portal (ZmZimletContext)");
286 		AjxPackage.require(params);
287 	}
288 
289 	DBG.println(AjxDebug.DBG2, "Zimlets - init() complete: " + this.name);
290 };

337 /**
338  * Calls the handler.
339  * 
340  * @param	{String}	funcname		the function
341  * @param	{Hash}		args			the arguments
342  * @return	{Object}	the results or <code>null</code> for none
343  * 
344  * @private
345  */
346 ZmZimletContext.prototype.callHandler =
347 function(funcname, args) {
348 	if (this.handlerObject) {
349 		var f = this.handlerObject[funcname];
350 		if (typeof f == "function") {
351 			if (typeof args == "undefined") {
352 				args = [];
353 			}
354 			else if (!(args instanceof Array)) {
355 				args = [args];
356 			}
357 			return f.apply(this.handlerObject, args);
358 		}
359 	}
360 	return null;
361 };

413 /**
414  * @private
415  */
416 ZmZimletContext.prototype._translateConfig =
417 function() {
418 	if (!this.config) { return; }
419 
420 	if (this.config.global && this.config.global[0]) {
421 		var prop = this.config.global[0].property;
422 		this.config.global = {};
423 		for (var i in prop) {
424 			this.config.global[prop[i].name] = prop[i]._content;
425 		}
426 	}
427 	if (this.config.local && this.config.local[0]) {
428 		var propLocal = this.config.local[0].property;
429 		this.config.local = {};
430 		for (var j in propLocal) {
431 			this.config.local[propLocal[j].name] = propLocal[j]._content;
432 		}
433 	}
434 };

436 /**
437  * Gets the configuration value.
438  * 
439  * @param	{String}	name		the config key name
440  * @return	{Object}	the config value or <code>null</code> if not set
441  */
442 ZmZimletContext.prototype.getConfig =
443 function(name) {
444 
445 	var config = (this.config && this.config.length && this.config[0]) ? this.config[0] : this.config;
446 	if (!config) { return; }
447 
448 	if (config.local && config.local[name]) {
449 		return config.local[name];
450 	}
451 
452 	if (config.global && config.global[name]) {
453 		return config.global[name];
454 	}
455 
456 	return null;
457 };

479 	if (menu == null || (menu instanceof ZmActionMenu) == false)

482 	var items = menu.getMenuItems();
483 	for (menuId in items) {

492 /**
493  * @private
494  */
495 ZmZimletContext.prototype._makeMenu =
496 function(obj) {
497 	var menu = new ZmActionMenu({parent:DwtShell.getShell(window), menuItems:ZmOperation.NONE});
498 	for (var i = 0; i < obj.length; ++i) {
499 		var data = obj[i];
500 		if (!data.id) {
501 			menu.createSeparator();
502 		} else {
503 			var params = {image:data.icon, text:this.process(data.label),disImage:data.disabledIcon};
504 			var item = menu.createMenuItem(data.id, params);
505 			item.setData("xmlMenuItem", data);
506 			item.addSelectionListener(this._handleMenuItemSelected);
507 			if (data.menuItem) {
508 				item.setMenu(this._makeMenu(data.menuItem));
509 			}
510 		}
511 	}
512 	return menu;
513 };

528 /**
529  * @private
530  */
531 ZmZimletContext.prototype.process =
532 function(str, obj, props) {
533 	if (obj) {
534 		str = this.processString(str, obj);
535 	}
536 	str = this.processMessage(str); 
537 	str = this.replaceObj(ZmZimletContext.RE_SCAN_PROP, str, props || this._propsById);
538 	str = this.replaceObj(ZmZimletContext.RE_SCAN_APP, str, ZmZimletContext.APP);
539 	str = str.replace(ZmZimletContext.__RE_SCAN_SETTING, ZmZimletContext.__replaceSetting);
540 	return str;
541 };
542 
543 /**
544  * @private
545  */
546 ZmZimletContext.prototype.processString =
547 function(str, obj) {
548 	return this.replaceObj(ZmZimletContext.RE_SCAN_OBJ, str, obj);
549 };

551 /**
552  * @private
553  */
554 ZmZimletContext.processMessage = function(name, str) {
555 	// i18n files load async so if not defined skip translation
556 	if (!window[name]) {
557 		DBG.println(AjxDebug.DBG2, "processMessage no messages: " + str);
558 		return str;
559 	}
560 	var props = window[name];
561 	return ZmZimletContext.replaceObj(ZmZimletContext.RE_SCAN_MSG, str, props);
562 };
563 
564 /**
565  * @private
566  */
567 ZmZimletContext.replaceObj = function(re, str, obj) {
568 	return String(str).replace(re,
569 		function(str, p1, prop) {
570 			var txt = p1;
571 			if (obj instanceof Array && obj.length > 1) {
572 				for(var i=0; i < obj.length; i++) {
573 					if(txt) {txt += ",";}
574 					var o = obj[i];
575 					if (o[prop] instanceof Object) {
576 						txt += o[prop].value;  // user prop
577 					} else {
578 						txt += o[prop];   // string
579 					}
580 				}
581 			} else {
582 				if (typeof obj[prop] != "undefined") {
583 					if (obj[prop] instanceof Object) {
584 						txt += obj[prop].value;  // user prop
585 					} else {
586 						txt += obj[prop];   // string
587 					}
588 				} else {
589 					txt += "(UNDEFINED - str '" + str + "' obj '" + obj + "')";
590 				}
591 			}
592 			return txt;
593 		});
594 };

618 /**
619  * @private
620  */
621 ZmZimletContext.prototype.makeURL =
622 function(actionUrl, obj, props) {
623 	// All URL's to have REST substitutions
624 	var url = this.process(actionUrl.target, obj, props);
625 	var param = [];
626 	if (actionUrl.param) {
627 		var a = actionUrl.param;
628 		for (var i = 0; i < a.length; ++i) {
629 			// trim whitespace as it's almost certain that the
630 			// developer didn't intend it.
631 			var val = AjxStringUtil.trim(a[i]._content || a[i]);
632 			val = this.process(val, obj, props);
633 			param.push([ AjxStringUtil.urlEncode(a[i].name),
634 				     "=",
635 				     AjxStringUtil.urlEncode(val) ].join(""));
636 		}
637 		var startChar = actionUrl.paramStart || '?';
638 		var joinChar = actionUrl.paramJoin || '&';
639 		url = [ url, startChar, param.join(joinChar) ].join("");
640 	}
641 	return url;
642 };

644 /**
645  * If there already is a paintable canvas to use, as in the case of tooltip,
646  * pass it to 'div' parameter.  otherwise a canvas (window, popup, dialog) will
647  * be created to display the contents from the url.
648  *
649  * @param actionUrl
650  * @param canvas
651  * @param obj
652  * @param div
653  * @param x
654  * @param y
655  * 
656  * @private
657  */
658 ZmZimletContext.prototype.handleActionUrl =
659 function(actionUrl, canvas, obj, div, x, y) {
660 	var url = this.makeURL(actionUrl, obj);
661 	var xslt = actionUrl.xslt && this.getXslt(actionUrl.xslt);
662 
663 	// need to use callback if the paintable canvas already exists, or if it
664 	// needs xslt transformation.
665 	if (div || xslt) {
666 		if (!div) {
667 			canvas = this.handlerObject.makeCanvas(canvas, null, x, y);
668 			div = document.getElementById("zimletCanvasDiv");
669 		}
670 		url = ZmZimletBase.PROXY + AjxStringUtil.urlComponentEncode(url);
671 		AjxRpc.invoke(null, url, null, new AjxCallback(this, this._rpcCallback, [xslt, div]), true);
672 	} else {
673 		this.handlerObject.makeCanvas(canvas, url, x, y);
674 	}
675 };

864 /**
865  * Gets the xslt.
866  * 
867  * @param	{String}	url		the URL
868  * @return	{AjxXslt}	the xslt
869  */
870 ZmZimletContext.prototype.getXslt =
871 function(url) {
872 	if (!this._xslt) {
873 		this._xslt = {};
874 	}
875 	var realurl = this.getUrl() + url;
876 	if (!this._xslt[realurl]) {
877 		this._xslt[realurl] = AjxXslt.createFromUrl(realurl);
878 	}
879 	return this._xslt[realurl];
880 };

882 /**
883  * @private
884  */
885 ZmZimletContext.prototype._rpcCallback =
886 function(xslt, canvas, result) {
887 	var html, resp = result.xml;
888 	if (!resp) {
889 		var doc = AjxXmlDoc.createFromXml(result.text);
890 		resp = doc.getDoc();
891 	}
892 	// TODO:  instead of changing innerHTML, maybe append
893 	// the dom tree to the canvas.
894 	if (xslt) {
895 		html = xslt.transformToString(resp);
896 		// If we don't have HTML at this point, we probably have a HTML fragment.
897 		if (!html) {
898 			html = result.text;
899 		}
900 		
901 	} else {
902 		html = resp.innerHTML;
903 	}
904 	canvas.innerHTML = html;
905 };

907 /**
908  * @private
909  */
910 ZmZimletContext._getMsgBody =
911 function(o) {
912 	//If message is not loaded let developer take care of it
913 	if (!o._loaded) {
914 		return "";
915 	}
916 	var part = o.getBodyPart(ZmMimeTable.TEXT_PLAIN) || o.getBodyPart(ZmMimeTable.TEXT_HTML);
917 	var content = part && part.getContent();
918 	if (content && (part.contentType == ZmMimeTable.TEXT_HTML)) {
919 		var div = document.createElement("div");
920 		div.innerHTML = content;
921 		content = AjxStringUtil.convertHtml2Text(div);
922 	}
923 	return content || "";
924 };
