 23 /**
 24  * Creates a Tree widget.
 25  * @constructor
 26  * @class
 27  * This class implements a tree widget. Tree widgets may contain one or more DwtTreeItems.
 28  *
 29  * @author Ross Dargahi
 30  * 
 31  * @param {hash}	params				a hash of parameters
 32  * @param  {DwtComposite}     params.parent			the parent widget
 33  * @param  {DwtTree.SINGLE_STYLE|DwtTree.MULTI_STYLE|DwtTree.CHECKEDITEM_STYLE}     params.style 	the tree style
 34  * @param  {string}     params.className				the CSS class
 35  * @param  {constant}     params.posStyle				the positioning style (see {@link DwtControl})
 36  * @param  {boolean}     params.isCheckedByDefault	default checked state if tree styles is "checked"
 37  * 
 38  * @extends		DwtComposite
 39  */
 40 DwtTree = function(params) {
 41 	if (arguments.length == 0) { return; }
 42 	params = Dwt.getParams(arguments, DwtTree.PARAMS);
 43 	params.className = params.className || "DwtTree";
 44 	DwtComposite.call(this, params);
 45 
 46 	var events = [DwtEvent.ONMOUSEDOWN, DwtEvent.ONMOUSEUP, DwtEvent.ONDBLCLICK];
 47 	if (!AjxEnv.isIE) {
 48 		events = events.concat([DwtEvent.ONMOUSEOVER, DwtEvent.ONMOUSEOUT]);
 49 	}
 50 	this._setEventHdlrs(events);
 51 
 52 	var style = params.style;
 53 	if (!style) {
 54 		this._style = DwtTree.SINGLE_STYLE;
 55 	} else {
 56 		if (style == DwtTree.CHECKEDITEM_STYLE) {
 57 			style |= DwtTree.SINGLE_STYLE;
 58 		}
 59 		this._style = style;
 60 	}
 61 	this.isCheckedStyle = ((this._style & DwtTree.CHECKEDITEM_STYLE) != 0);
 62 	this.isCheckedByDefault = params.isCheckedByDefault;
 63 
 64 	this._selectedItems = new AjxVector();
 65 	this._selEv = new DwtSelectionEvent(true);
 66 	this._selByClickEv = new DwtSelectionEvent(true);
 67 	this._selByClickEv.clicked = true;
 68 	this._selByEnterEv = new DwtSelectionEvent(true);
 69 	this._selByEnterEv.enter = true;
 70 };

201 	var a = this._selectedItems.getArray();
202 	var sz = this._selectedItems.size();
203 	for (var i = 0; i < sz; i++) {

205 			a[i]._setSelected(false);

211 	this._selectedItems.removeAll();

233 DwtTree.prototype.setSelection =
234 function(treeItem, skipNotify, kbNavEvent, noFocus) {
235 	if (!treeItem) { return; }
236 
237 	// Remove currently selected items from the selection list. if <treeItem> is in that list, then note it and return
238 	// after we are done processing the selected list
239 	var a = this._selectedItems.getArray();
240 	var sz = this._selectedItems.size();
241 	var da;
242 	var j = 0;
243 	var alreadySelected = false;
244 	for (var i = 0; i < sz; i++) {
245 		if (a[i] == treeItem) {
246 			alreadySelected = true;
247 		} else {
248 			a[i]._setSelected(false);
249 			this._selectedItems.remove(a[i]);
250 			if (da == null) {
251 				da = new Array();
252 			}
253 			da[j++] = a[i];
254 		}
255 	}
256 
257 	if (da && !skipNotify) {
258 		this._notifyListeners(DwtEvent.SELECTION, da, DwtTree.ITEM_DESELECTED, null, this._selEv, kbNavEvent);
259 	}
260 
261 	if (alreadySelected) { return; }
262 	this._selectedItems.add(treeItem);
263 
264 	// Expand all parent nodes, and then set item selected
265 	this._expandUp(treeItem);
266 	if (treeItem._setSelected(true, noFocus) && !skipNotify) {
267 		this._notifyListeners(DwtEvent.SELECTION, [treeItem], DwtTree.ITEM_SELECTED, null, this._selEv, kbNavEvent);
268 	}
269 };

295 // Expand parent chain from given item up to root
296 DwtTree.prototype._expandUp =
297 function(item) {
298 	var parent = item.parent;
299 	while (parent instanceof DwtTreeItem) {
300 		parent.setExpanded(true);
301 		parent.setVisible(true);
302 		parent = parent.parent;
303 	}
304 };

306 DwtTree.prototype._addItem =
307 function(item, index) {
308 	this._children.add(item, index);
309 	var thisHtmlElement = this._getContainerElement();
310 	var numChildren = thisHtmlElement.childNodes.length;
311 	if (index == null || index > numChildren) {
312 		thisHtmlElement.appendChild(item.getHtmlElement());
313 	} else {
314 		thisHtmlElement.insertBefore(item.getHtmlElement(), thisHtmlElement.childNodes[index]);	
315 	}
316 };

320 DwtTree.prototype.sort =
321 function(cmp) {
322     var children = this.getItems();
323     children.sort(cmp);
324     var fragment = document.createDocumentFragment();
325     AjxUtil.foreach(children, function(item, i){
326         fragment.appendChild(item.getHtmlElement());
327         item._index = i;
328     });
329     this._getContainerElement().appendChild(fragment);
330 };

332 DwtTree.prototype.removeChild =
333 function(child) {
334 	this._children.remove(child);
335 	this._selectedItems.remove(child);
336     var childEl = child.getHtmlElement();
337     if (childEl.parentNode) {
338         childEl.parentNode.removeChild(childEl);
339     }
340 };

342 /**
343  * Returns the next (or previous) tree item relative to the currently selected
344  * item, in top-to-bottom order as the tree appears visually. Items such as
345  * separators that cannot be selected are skipped.
346  * </p><p>
347  * If there is no currently selected item, return the first or last item. If we go past
348  * the beginning or end of the tree, return null.
349  * </p><p>
350  * For efficiency, a flattened list of the visible and selectable tree items is maintained.
351  * It will be cleared on any change to the tree's display, then regenerated when it is
352  * needed.
353  *
354  * @param {boolean}	next		if <code>true</code>, return next tree item; otherwise, return previous tree item
355  * 
356  * @private
357  */
358 DwtTree.prototype._getNextTreeItem =
359 function(next) {
360 
361 	var sel = this.getSelection();
362 	var curItem = (sel && sel.length) ? sel[0] : null;
363 
364 	var nextItem = null, idx = -1;
365 	var list = this.getTreeItemList(true);
366 	if (curItem) {
367 		for (var i = 0, len = list.length; i < len; i++) {
368 			var ti = list[i];
369 			if (ti == curItem) {
370 				idx = next ? i + 1 : i - 1;
371 				break;
372 			}
373 		}
374 		nextItem = list[idx]; // if array index out of bounds, nextItem is undefined
375 	} else {
376 		// if nothing is selected yet, return the first or last item
377 		if (list && list.length) {
378 			nextItem = next ? list[0] : list[list.length - 1];
379 		}
380 	}
381 	return nextItem;
382 };

384 DwtTree.prototype._getFirstTreeItem =
385 function() {
386 	var a = this.getTreeItemList(true);
387 	if (a && a.length > 0) {
388 		return a[0];
389 	}
390 	return null;
391 };

393 DwtTree.prototype._getLastTreeItem =
394 function() {
395 	var a = this.getTreeItemList(true);
396 	if (a && a.length > 0) {
397 		return a[a.length - 1];
398 	}
399 	return null;
400 };

402 /**
403  * Creates a flat list of this tree's items, going depth-first.
404  *
405  * @param {boolean}	visible		if <code>true</code>, only include visible/selectable items
406  * @return	{array}	an array of {@link DwtTreeItem} objects
407  */
408 DwtTree.prototype.getTreeItemList =
409 function(visible) {
410 	return this._addToList([], visible);
411 };
412 
413 DwtTree.prototype._addToList =
414 function(list, visible, treeItem) {
415 	if (treeItem && !treeItem._isSeparator &&
416 		(!visible || (treeItem.getVisible() && treeItem._selectionEnabled))) {
417 
418 		list.push(treeItem);
419 	}
420 	if (!treeItem || !visible || treeItem._expanded) {
421 		var parent = treeItem || this;
422 		var children = parent.getChildren ? parent.getChildren() : [];
423 		for (var i = 0; i < children.length; i++) {
424 			this._addToList(list, visible, children[i]);
425 		}
426 	}
427 	return list;
428 };

430 DwtTree.prototype._deselect =
431 function(item) {
432 	if (this._selectedItems.contains(item)) {
433 		this._selectedItems.remove(item);
434 		item._setSelected(false);
435 		this._notifyListeners(DwtEvent.SELECTION, [item], DwtTree.ITEM_DESELECTED, null, this._selEv);
436 	}

439 DwtTree.prototype._itemActioned =
440 function(item, ev) {
441 	if (this._actionedItem && !this._actionedItem.isDisposed()) {
442 		this._actionedItem._setActioned(false);
443 		this._notifyListeners(DwtEvent.SELECTION, [this._actionedItem], DwtTree.ITEM_DESELECTED, ev, this._selEv);
444 	}
445 	this._actionedItem = item;
446 	item._setActioned(true);
447 	this._notifyListeners(DwtEvent.SELECTION, [item], DwtTree.ITEM_ACTIONED, ev, this._selEv);
448 };

455 DwtTree.prototype._itemClicked =
456 function(item, ev) {
457 	var i;
458 	var a = this._selectedItems.getArray();
459 	var numSelectedItems = this._selectedItems.size();
460 	if (this._style & DwtTree.SINGLE_STYLE || (!ev.shiftKey && !ev.ctrlKey)) {
461 		if (numSelectedItems > 0) {
462 			for (i = 0; i < numSelectedItems; i++) {
463 				a[i]._setSelected(false);
464 			}
465 			// Notify listeners of deselection
466 			this._notifyListeners(DwtEvent.SELECTION, this._selectedItems.getArray(), DwtTree.ITEM_DESELECTED, ev, this._selByClickEv);
467 			this._selectedItems.removeAll();
468 		}
469 		this._selectedItems.add(item);
470 		if (item._setSelected(true)) {
471 			this._notifyListeners(DwtEvent.SELECTION, [item], DwtTree.ITEM_SELECTED, ev, this._selByClickEv);
472 		}
473 	} else {
474 		if (ev.ctrlKey) {
475 			if (this._selectedItems.contains(item)) {
476 				this._selectedItems.remove(item);
477 				item._setSelected(false);
478 				this._notifyListeners(DwtEvent.SELECTION, [item], DwtTree.ITEM_DESELECTED, ev, this._selByClickEv);
479 			} else {
480 				this._selectedItems.add(item);
481 				if (item._setSelected(true)) {
482 					this._notifyListeners(DwtEvent.SELECTION, [item], DwtTree.ITEM_SELECTED, ev, this._selByClickEv);
483 				}
484 			}
485 		} else {
486 			// SHIFT KEY
487 		}
488 	}
489 };

503 DwtTree.prototype._itemCollapsed =
504 function(item, ev, skipNotify) {
505 	var i;
506 	if (!skipNotify) {
507 		this._notifyListeners(DwtEvent.TREE, [item], DwtTree.ITEM_COLLAPSED, ev, DwtShell.treeEvent);
508 	}
509 	var setSelection = false;
510 	var a = this._selectedItems.getArray();
511 	var numSelectedItems = this._selectedItems.size();
512 	var da;
513 	var j = 0;
514 	for (i = 0; i < numSelectedItems; i++) {
515 		if (a[i]._isChildOf(item)) {
516 			setSelection = true;
517 			if (da == null) {
518 				da = new Array();
519 			}
520 			da[j++] = a[i];
521 			a[i]._setSelected(false);
522 			this._selectedItems.remove(a[i]);
523 		}		
524 	}
525 
526 	if (da) {
527 		this._notifyListeners(DwtEvent.SELECTION, da, DwtTree.ITEM_DESELECTED, ev, this._selEv);
528 	}
529 
530 	if (setSelection && !this._selectedItems.contains(item)) {
531 		if (item._setSelected(true)) {
532 			this._selectedItems.add(item);
533 			this._notifyListeners(DwtEvent.SELECTION, [item], DwtTree.ITEM_SELECTED, ev, this._selEv);
534 		}
535 	}
536 };

538 DwtTree.prototype._notifyListeners =
539 function(listener, items, detail, srcEv, destEv, kbNavEvent) {
540 	if (this.isListenerRegistered(listener)) {
541 		if (srcEv) {
542 			DwtUiEvent.copy(destEv, srcEv);
543 		}
544 		destEv.items = items;
545 		if (items.length == 1) {
546 			destEv.item = items[0];
547 		}
548 		destEv.detail = detail;
549 		destEv.kbNavEvent = kbNavEvent;
550 		this.notifyListeners(listener, destEv);
551 		if (listener == DwtEvent.SELECTION) {
552 			this.shell.notifyGlobalSelection(destEv);
553 		}
554 	}
