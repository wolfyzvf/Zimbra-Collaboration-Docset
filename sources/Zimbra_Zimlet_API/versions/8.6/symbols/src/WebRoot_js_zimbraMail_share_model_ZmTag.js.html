 27 /**
 28  * Creates a tag
 29  * @class
 30  * This class represents a tag.
 31  * 
 32  * @param	{Hash}	params		a hash of parameters
 33  * @extends	ZmOrganizer
 34  */
 35 ZmTag = function(params) {
 36 	params.type = ZmOrganizer.TAG;
 37 	ZmOrganizer.call(this, params);
 38 	this.notLocal = params.notLocal;
 39 };

 75 /**
 76  * Tags come from back end as a flat list, and we manually create a root tag, so all tags
 77  * have the root as parent. If tags ever have a tree structure, then this should do what
 78  * ZmFolder does (recursively create children).
 79  * 
 80  * @private
 81  */
 82 ZmTag.createFromJs =
 83 function(parent, obj, tree, sorted, account) {
 84 	var tag;
 85 	var nId = ZmOrganizer.normalizeId(obj.id);
 86 	if (nId < ZmOrganizer.FIRST_USER_ID[ZmOrganizer.TAG]) { return; }
 87 	tag = tree.getById(obj.id);
 88 	if (tag) { return tag; }
 89 
 90 	var params = {
 91 		id: obj.id,
 92 		name: obj.name,
 93 		color: ZmTag.checkColor(obj.color),
 94 		rgb: obj.rgb,
 95 		parent: parent,
 96 		tree: tree,
 97 		numUnread: obj.u,
 98 		account: account
 99 	};
100 	tag = new ZmTag(params);
101 	var index = sorted ? ZmOrganizer.getSortIndex(tag, ZmTag.sortCompare) : null;
102 	parent.children.add(tag, index);
103 
104 	var tagNameMap = parent.getTagNameMap();
105 	tagNameMap[obj.name] = tag;
106 
107 	return tag;
108 };
109 
110 ZmTag.createNotLocalTag =
111 function(name) {
112 	//cache so we don't create many objects in case many items are tagged by non local tags.
113 	var cache = ZmTag.notLocalCache = ZmTag.notLocalCache || [];
114 	var tag = cache[name];
115 	if (tag) {
116 		return tag;
117 	}
118 	tag = new ZmTag({notLocal: true, id: "notLocal_" + name, name: name});
119 	cache[name] = tag;
120 	return tag;
121 };

123 /**
124  * Compares the tags by name.
125  * 
126  * @param	{ZmTag}	tagA		the first tag
127  * @param	{ZmTag}	tagB		the second tag
128  * @return	{int}	0 if the tag names match (case-insensitive); 1 if "a" is before "b"; -1 if "b" is before "a"
129  */
130 ZmTag.sortCompare = 
131 function(tagA, tagB) {
132 	var check = ZmOrganizer.checkSortArgs(tagA, tagB);
133 	if (check != null) return check;
134 
135 	if (tagA.name.toLowerCase() > tagB.name.toLowerCase()) return 1;
136 	if (tagA.name.toLowerCase() < tagB.name.toLowerCase()) return -1;
137 	return 0;
138 };

140 /**
141  * Checks the tag name.
142  * 
143  * @param	{String}	name		the name
144  * @return	{String}	<code>null</code> if the name is valid or a error message
145  */
146 ZmTag.checkName =
147 function(name) {
148 	var msg = ZmOrganizer.checkName(name);
149 	if (msg) { return msg; }
150 
151 	if (name.indexOf('\\') == 0) {
152 		return AjxMessageFormat.format(ZmMsg.errorInvalidName, AjxStringUtil.htmlEncode(name));
153 	}
154 
155 	return null;
156 };
157 
158 /**
159  * Checks the color.
160  * 
161  * @param	{String}	color	the color
162  * @return	{Number}	the valid color
163  */
164 ZmTag.checkColor =
165 function(color) {
166 	color = Number(color);
167 	return ((color != null) && (color >= 0 && color <= ZmOrganizer.MAX_COLOR)) ? color : ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
168 };

170 ZmTag.getIcon = function(color) {
171     var object = { getIcon:ZmTag.prototype.getIcon, getColor:ZmTag.prototype.getColor, color:color };
172     if (String(color).match(/^#/)) {
173         object.rgb = color;
174         object.color = null;
175     }
176     return ZmTag.prototype.getIconWithColor.call(object);
177 }

179 /**
180  * Creates a tag.
181  * 
182  * @param	{Hash}	params	a hash of parameters
183  */
184 ZmTag.create =
185 function(params) {
186 	var request = {_jsns: "urn:zimbraMail"};
187 	var jsonObj = {CreateTagRequest: request};
188 	request.tag = {name: params.name}
189 
190     if (params.rgb) {
191         request.tag.rgb = params.rgb;
192     }
193     else {
194         request.tag.color = ZmOrganizer.checkColor(params.color) || ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
195     }
196 	var errorCallback = new AjxCallback(null, ZmTag._handleErrorCreate, params);
197 	appCtxt.getAppController().sendRequest({
198 			jsonObj: jsonObj,
199 			asyncMode: true,
200 			errorCallback: errorCallback,
201 			accountName: params.accountName
202 	});
203 };
204 
205 /**
206  * @private
207  */
208 ZmTag._handleErrorCreate =
209 function(params, ex) {
210 	if (ex.code == ZmCsfeException.MAIL_INVALID_NAME) {
211 		var msg = AjxMessageFormat.format(ZmMsg.errorInvalidName, AjxStringUtil.htmlEncode(params.name));
212 		var msgDialog = appCtxt.getMsgDialog();
213 		msgDialog.setMessage(msg, DwtMessageDialog.CRITICAL_STYLE);
214 		msgDialog.popup();
215 		return true;
216 	}
217 	return false;
218 };

220 /**
221  * Gets the icon.
222  * 
223  * @return	{String}	the icon or <code>null</code> for no icon
224  */
225 ZmTag.prototype.getIcon = 
226 function() {
227 	if (this.notLocal) {
228 		return "TagShared";
229 	}
230 	
231 	return (this.id == ZmOrganizer.ID_ROOT) ? null : "Tag";
232 };

265 ZmTag.prototype.notifyCreate =
266 function(obj) {
267 	var child = ZmTag.createFromJs(this, obj, this.tree, true);
268 	child._notify(ZmEvent.E_CREATE);
269 };

272 ZmTag.prototype.notifyModify =
273 function(obj) {
274 	if (obj.name) {
275 		//this is a rename - update the tagNameMap
276 		var oldName = this.name;
277 		var nameMap = this.parent.getTagNameMap();
278 		delete nameMap[oldName];
279 		nameMap[obj.name] = this;
280 		//we don't change the name on this ZmTag object here, it is done in ZmOrganizer.prototype.notifyModify
281 	}
282 	ZmOrganizer.prototype.notifyModify.call(this, obj);
283 };
284 
285 
286 ZmTag.prototype.notifyDelete =
287 function() {
288 	var nameMap = this.parent.getTagNameMap();
289 	delete nameMap[this.name];  //remove from name map
290 	
291 	ZmOrganizer.prototype.notifyDelete.call(this);
292 };

305 ZmTag.prototype.getByNameOrRemote =
306 function(name) {
307 	var tagNameMap = this.getTagNameMap();
308 	var tag = tagNameMap[name];
309 	if (tag) {
310 		return tag;
311 	}
312 	return ZmTag.createNotLocalTag(name);
313 };
