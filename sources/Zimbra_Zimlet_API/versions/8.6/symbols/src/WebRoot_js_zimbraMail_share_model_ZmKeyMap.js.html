 27 /**
 28  * Creates a key map for the ZCS application.
 29  * @class
 30  * This class maps keys to actions for the ZCS application. There is a global key map
 31  * with bindings that apply to any key not handled by the current controller; these
 32  * global bindings apply across applications (mail, contacts, etc). Key bindings that
 33  * are context-dependent are tied to a particular controller. If that controller has
 34  * control, then those bindings will be used.
 35  * <br/>
 36  * <br/>
 37  * Bindings are passed in via the <code>ZmKeys</code> object, which is populated from a properties
 38  * file. The identifiers used in the properties file must match those used here.
 39  * 
 40  * @author Ross Dargahi
 41  * @author Conrad Damon
 42  * 
 43  * @extends		DwtKeyMap
 44  */
 45 ZmKeyMap = function() {
 46 	
 47 	ZmKeyMap._setPreconditions();
 48 	DwtKeyMap.call(this);
 49 	this._load(this._map, ZmKeys);
 50 
 51 	if (appCtxt.get(ZmSetting.DEV) && this._map[ZmKeyMap.MAP_GLOBAL]) {
 52 		this._map[ZmKeyMap.MAP_GLOBAL]["Alt+Shift+D,0"] = ZmKeyMap.DBG_NONE;
 53 		this._map[ZmKeyMap.MAP_GLOBAL]["Alt+Shift+D,1"] = ZmKeyMap.DBG_1;
 54 		this._map[ZmKeyMap.MAP_GLOBAL]["Alt+Shift+D,2"] = ZmKeyMap.DBG_2;
 55 		this._map[ZmKeyMap.MAP_GLOBAL]["Alt+Shift+D,3"] = ZmKeyMap.DBG_3;
 56 		this._map[ZmKeyMap.MAP_GLOBAL]["Alt+Shift+D,T"] = ZmKeyMap.DBG_TIMING;
 57 	}
 58 };

283 /**
284  * Checks if this map is valid. A map may have a precondition,
285  * which is either a setting that must be true, or a function that returns
286  * true.
287  *
288  * @param {String}	mapName	the name of map
289  * @return	{Boolean}	<code>true</code> if the map is valid
290  * 
291  * @private
292  */
293 ZmKeyMap.prototype._checkMap =
294 function(mapName) {
295 	var result;
296 	var mapPre = ZmKeyMap.MAP_PRECONDITION[mapName];
297 	if (!mapPre) {
298 		result = true;
299 	} else if (typeof mapPre == "string" || typeof mapPre == "number") {
300 		result = appCtxt.get(mapPre);
301 	} else if (typeof mapPre == "function") {
302 		result = mapPre();
303 	}
304 	this._checkedMap[mapName] = result;
305 	return result;
306 };
307 
308 /**
309  * Checks if this action is valid. A map or an action may have a precondition,
310  * which is either a setting that must be true, or a function that returns
311  * true.
312  *
313  * @param {String} mapName	the name of map
314  * @param {String} action	the action to check
315  * @return	{Boolean}	<code>true</code> if the action is valid
316  * 
317  * @private
318  */
319 ZmKeyMap.prototype._checkAction =
320 function(mapName, action) {
321 	if ((this._checkedMap[mapName] === false) ||
322 		(!this._checkedMap[mapName] && !this._checkMap(mapName))) { return false; }
323 	var mapPre = ZmKeyMap.ACTION_PRECONDITION[mapName];
324 	if (!mapPre) { return true; }
325 	var pre = mapPre[action];
326 	if (!pre) { return true; }
327 	if (typeof pre == "string" || typeof pre == "number") {
328 		return appCtxt.get(pre);
329 	} else if (typeof pre == "function") {
330 		return pre();
331 	}
332 	return true;
333 };
