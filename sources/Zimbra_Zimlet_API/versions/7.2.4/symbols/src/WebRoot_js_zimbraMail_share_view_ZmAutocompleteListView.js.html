 21 /**
 22  * Creates a new autocomplete list. The list isn't populated or displayed until some
 23  * autocompletion happens. Takes a data class and loader, so that when data is needed (it's
 24  * loaded lazily), the loader can be called on the data class.
 25  * @class
 26  * This class implements autocomplete functionality. It has two main parts: matching data based
 27  * on keystroke events, and displaying/managing the list of matches. This class is theoretically
 28  * neutral concerning the data that gets matched (as long as its class has an <code>autocompleteMatch()</code>
 29  * method), and the field that it's being called from.
 30  * <p>
 31  * The data class's <code>autocompleteMatch()</code> method should returns a list of matches, where each match is
 32  * an object with the following properties:</p>
 33  * <table border="1" width="50%">
 34  * <tr><td width="15%">data</td><td>the object being matched</td></tr>
 35  * <tr><td>text</td><td>the text to display for this object in the list</td></tr>
 36  * <tr><td>[key1]</td><td>a string that may be used to replace the typed text</td></tr>
 37  * <tr><td>[keyN]</td><td>a string that may be used to replace the typed text</td></tr>
 38  * </table>
 39  * </p><p>
 40  * The calling client also specifies the key in the match result for the string that will be used
 41  * to replace the typed text (also called the "completion string"). For example, the completion 
 42  * string for matching contacts could be a full address, or just the email.
 43  * </p><p>
 44  * The client may provide additional key event handlers in the form of callbacks. If the callback
 45  * explicitly returns true or false, that's what the event handler will return.
 46  * </p>
 47  * 
 48  * @author Conrad Damon
 49  *
 50  * @param {Hash}	params			a hash of parameters:
 51  * @param	{String}		matchValue			the name of field in match result to use for completion
 52  * @param	{function}		dataClass			the class that has the data loader
 53  * @param	{function}		dataLoader			a method of dataClass that returns data to match against
 54  * @param	{DwtComposite}	parent				the control that created this list (defaults to shell)
 55  * @param	{String}		className			the CSS class
 56  * @param	{Array}			delims				the list of delimiters (which separate tokens such as addresses)
 57  * @param	{Array}			delimCodes			the list of delimiter key codes
 58  * @param	{String}		separator			the separator (gets added to the end of a match)
 59  * @param	{AjxCallback}	compCallback		the callback into client to notify it that completion happened
 60  * @param	{AjxCallback}	keyDownCallback		the additional client ONKEYDOWN handler
 61  * @param	{AjxCallback}	keyPressCallback	the additional client ONKEYPRESS handler
 62  * @param	{AjxCallback}	keyUpCallback		the additional client ONKEYUP handler
 63  * @param	{AjxCallback}	enterCallback		the client handler for Enter key
 64  * @param	{Hash}			options				the additional options for the data class
 65  * 
 66  * @extends		DwtComposite
 67  */
 68 ZmAutocompleteListView = function(params) {
 69 
 70 	if (arguments.length == 0) { return; }
 71 
 72 	var className = params.className ? params.className : "ZmAutocompleteListView";
 73 	DwtComposite.call(this, params.parent || appCtxt.getShell(), className, DwtControl.ABSOLUTE_STYLE);
 74 
 75 	this._dataClass = this._dataAPI = params.dataClass;
 76 	this._dataLoader = params.dataLoader;
 77 	this._dataLoaded = false;
 78 	this._matchValue = params.matchValue;
 79 	this._separator = (params.separator != null) ? params.separator : AjxEmailAddress.SEPARATOR;
 80     this._options = params.options || {};
 81 
 82 	this._callbacks = {};
 83 	for (var i = 0; i < ZmAutocompleteListView.CALLBACKS.length; i++) {
 84 		this._setCallbacks(ZmAutocompleteListView.CALLBACKS[i], params);
 85 	}
 86 
 87 	this._isDelim = AjxUtil.arrayAsHash(params.delims || ZmAutocompleteListView.DELIMS);
 88 	this._isDelimCode = AjxUtil.arrayAsHash(params.delimCodes || ZmAutocompleteListView.DELIM_CODES);
 89 	if (!params.delims && !params.delimCodes) {
 90 		this._isDelim[','] = this._isDelimCode[188] = appCtxt.get(ZmSetting.AUTOCOMPLETE_ON_COMMA); 
 91 		var listener = new AjxListener(this, this._settingChangeListener);
 92 		var aoc = appCtxt.getSettings().getSetting(ZmSetting.AUTOCOMPLETE_ON_COMMA);
 93 		if (aoc) {
 94 			aoc.addChangeListener(listener);
 95 		}
 96 	}
 97 
 98     // mouse event handling
 99 	this._setMouseEventHdlrs();
100 	this.addListener(DwtEvent.ONMOUSEDOWN, new AjxListener(this, this._mouseDownListener));
101 	this.addListener(DwtEvent.ONMOUSEOVER, new AjxListener(this, this._mouseOverListener));
102 	this._addSelectionListener(new AjxListener(this, this._listSelectionListener));
103 	this._outsideListener = new AjxListener(null, ZmAutocompleteListView._outsideMouseDownListener);
104 
105 	// only trigger matching after a sufficient pause
106 	this._acInterval = appCtxt.get(ZmSetting.AC_TIMER_INTERVAL);
107 	this._acAction = new AjxTimedAction(null, this._autocompleteAction);
108 	this._acActionId = -1;
109 
110 	// for managing focus on Tab in Firefox
111 	if (AjxEnv.isGeckoBased) {
112 		this._focusAction = new AjxTimedAction(null, this._autocompleteFocus);
113 	}
114 
115 	this._origClass = "acRow";
116 	this._selClass = "acRow-selected";
117 	this._showLinkTextClass = "LinkText";
118 	this._hideLinkTextClass = "LinkText-hide";
119 	this._hideSelLinkTextClass = "LinkText-hide-selected";
120 
121 	this._done = {};
122 	this.setVisible(false);
123 	this.setScrollStyle(Dwt.SCROLL);
124 	this.reset();
125 };

157 /**
158  * Handles the on key down event.
159  * 
160  * @param	{Event}	event		the event
161  */
162 ZmAutocompleteListView.onKeyDown =
163 function(ev) {
164 	ev = DwtUiEvent.getEvent(ev);
165 	var result = null;
166 	var key = DwtKeyEvent.getCharCode(ev);
167 	DBG.println("ac", ev.type + ": " + key);
168 	if (key == 38 || key == 40) {
169 		ZmAutocompleteListView.__geckoKeyCode = null;
170 		result = ZmAutocompleteListView._onKeyUp(ev);
171 	} else {
172 		result = ZmAutocompleteListView._onKeyDown(ev);
173 	}
174 	var element = DwtUiEvent.getTargetWithProp(ev, "_aclvId");
175 	var aclv = element && DwtControl.ALL_BY_ID[element._aclvId];
176 	if (aclv) {
177 		aclv._inputValue = element.value;
178 		var cbResult = aclv._runCallbacks(ZmAutocompleteListView.CB_KEYDOWN, element && element.id, [ev, aclv, result, element]);
179 		result = (cbResult === true || cbResult === false) ? cbResult : result;
180 	}
181     if (AjxEnv.isFirefox){
182        ZmAutocompleteListView.clearTimer();
183        ZmAutocompleteListView.timer =  new AjxTimedAction(this, ZmAutocompleteListView.onKeyUp, [ev]);
184        AjxTimedAction.scheduleAction(ZmAutocompleteListView.timer, 300)
185     }
186 	return result;
187 };

189 /**
190  * Handles the on key press event.
191  * 
192  * @param	{Event}	event		the event
193  */
194 ZmAutocompleteListView.onKeyPress =
195 function(ev) {
196 	ev = DwtUiEvent.getEvent(ev);
197 	DwtKeyEvent.geckoCheck(ev);
198 	var result = null;
199 	var key = DwtKeyEvent.getCharCode(ev);
200 	DBG.println("ac", ev.type + ": " + key);
201 	if (AjxEnv.isGeckoBased && (key == 38 || key == 40)) {
202 		if (ZmAutocompleteListView.__geckoKeyCode) {
203 			result = ZmAutocompleteListView._onKeyUp(ev);
204 		} else {
205 			ZmAutocompleteListView.__geckoKeyCode = key;
206 		}
207 	}
208 	var element = DwtUiEvent.getTargetWithProp(ev, "_aclvId");
209 	var aclv = element && DwtControl.ALL_BY_ID[element._aclvId];
210 
211 	if (aclv) {
212 		var cbResult = aclv._runCallbacks(ZmAutocompleteListView.CB_KEYPRESS, element && element.id, [ev, aclv, result, element]);
213 		result = (cbResult === true || cbResult === false) ? cbResult : result;
214 	}
215 
216 	return (result != null) ? result : ZmAutocompleteListView._echoKey(true, ev);
217 };

243 /**
244 * "onkeydown" handler for catching Tab and Esc keys. We don't want to let the browser
245 * handle this event for those (which it will do before we get the keyup event).
246 *
247 * @param ev		the key event
248 * 
249 * @private
250 */
251 ZmAutocompleteListView._onKeyDown =
252 function(ev) {
253 	var key = DwtKeyEvent.getCharCode(ev);
254 	// don't echo enter key if list view is visible, since in that case it's
255 	// just a selection mechanism
256 	if (key == 3 || key == 13) {
257 		var element = DwtUiEvent.getTargetWithProp(ev, "_aclvId");
258 		if (!element) {
259 			return ZmAutocompleteListView._echoKey(true, ev);
260 		}
261 		var aclv = DwtControl.ALL_BY_ID[element._aclvId];
262 		if (aclv && aclv.getVisible()) {
263 			return ZmAutocompleteListView._echoKey(false, ev);
264 		}
265 	}
266 	if (key == 9 || key == 27) {
267 		return ZmAutocompleteListView._onKeyUp(ev);
268 	} else {
269 		return ZmAutocompleteListView._echoKey(true, ev);
270 	}
271 };
272 
273 /**
274 * "onkeyup" handler for performing autocompletion. The reason it's an "onkeyup" handler is that neither 
275 * "onkeydown" nor "onkeypress" arrives after the form field has been updated.
276 *
277 * @param ev		the key event
278 * 
279 * @private
280 */
281 ZmAutocompleteListView._onKeyUp =
282 function(ev) {
283 
284 	var element = DwtUiEvent.getTargetWithProp(ev, "_aclvId");
285 	if (!element) {
286 		return ZmAutocompleteListView._echoKey(true, ev);
287 	}
288 
289 	var aclv = DwtControl.ALL_BY_ID[element._aclvId];
290 	var key = DwtKeyEvent.getCharCode(ev);
291 	aclv._hasCompleted = false;
292 
293 	// Tab/Esc handled in keydown for IE
294 	if (AjxEnv.isIE && ev.type == "keyup" && (key == 9 || key == 27)) {
295 		return ZmAutocompleteListView._echoKey(true, ev);
296 	}
297 
298 	var value = element.value;
299 	DBG.println(AjxDebug.DBG3, ev.type + " event, key = " + key + ", value = " + value);
300 	ev.inputChanged = (value != aclv._inputValue);
301 
302 	// reset timer on any address field key activity
303 	if (aclv._acActionId != -1 && !DwtKeyMap.IS_MODIFIER[key]) {
304 		AjxTimedAction.cancelAction(aclv._acActionId);
305 		aclv._acActionId = -1;
306 	}
307 	
308 	// Figure out what this handler should return. If it returns true, the browser will
309 	// handle the key event. That usually means it just echoes the typed character, but
310 	// it could do something like change focus (eg tab). We let the browser handle input
311 	// characters, and anything weird that we don't want to deal with. The only keys we
312 	// don't let the browser handle are ones that control the features of the autocomplete
313 	// list.
314 
315 	if (DwtKeyMap.IS_MODIFIER[key] || DwtKeyMapMgr.hasModifier(ev)) {
316 		return ZmAutocompleteListView._echoKey(true, ev);
317 	}
318 	// if the field is empty, clear the list
319 	if (!value) {
320 		aclv.reset();
321 		if (key != 13 && key != 3) { // If the key is ENTER, we want to continue and eventually call the enter-listeners
322 			return ZmAutocompleteListView._echoKey(true, ev);
323 		}
324 	}
325 	if (key == 37 || key == 39) { // left/right arrow key
326 		return ZmAutocompleteListView._echoKey(true, ev);
327 	}
328 	// Pass tab/esc through if there's no list
329 	if ((key == 9 || key == 27) && !aclv.size()) {
330 		return ZmAutocompleteListView._echoKey(true, ev);
331 	}
332 
333 	// if the user types a single delimiting character with the list showing, do completion
334 	var isDelim = (aclv.getVisible() && (!ev.shiftKey && aclv._isDelimCode[key]));
335 
336 	DBG.println(AjxDebug.DBG3, "key = " + key + ", isDelim: " + isDelim);
337 	if (isDelim || (key == 27 || (aclv.getVisible() && (key == 38 || key == 40)))) {
338 		aclv.handleAction(key, isDelim);
339 		// In Firefox, focus shifts on Tab even if we return false (and stop propagation and prevent default),
340 		// so make sure the focus stays in this element.
341 		if (AjxEnv.isGeckoBased && key == 9) {
342 			aclv._focusAction.args = [ element ];
343 			AjxTimedAction.scheduleAction(aclv._focusAction, 0);
344 		}
345 		if (key == 13 || key == 3) {
346 			var result = aclv._runCallbacks(ZmAutocompleteListView.CB_ENTER, element && element.id, [ev]);
347 			return (result != null) ? result : ZmAutocompleteListView._echoKey(true, ev);
348 		}
349 		return ZmAutocompleteListView._echoKey(false, ev);
350 	}
351 
352 	// skip if it's some weird character
353 	if (!ev.inputChanged && (key != 3 && key != 13 && key != 9 && key != 8 && key != 46)) {
354 		return ZmAutocompleteListView._echoKey(true, ev);
355 	}
356     ZmAutocompleteListView.clearTimer();
357 	if (key == 13 || key == 3) {
358 		if (aclv._options.addrBubbles && aclv._dataAPI.isComplete && aclv._dataAPI.isComplete(value)) {
359 				DBG.println(AjxDebug.DBG3, "got a Return, found an addr: " + value);
360 				aclv._runCallbacks(ZmAutocompleteListView.CB_ADDR_FOUND, element.id, [aclv, value, "\n"]);
361 		} else {
362 			// Treat as regular selection
363 			var selEv = DwtShell.selectionEvent;
364 			DwtUiEvent.copy(selEv, ev);
365 			selEv.detail = 0;
366 			aclv.notifyListeners(DwtEvent.SELECTION, selEv);
367 		}
368 		aclv.reset();
369 		var result = aclv._runCallbacks(ZmAutocompleteListView.CB_ENTER, element && element.id, [ev]);
370 		return (result != null) ? result : ZmAutocompleteListView._echoKey(true, ev);
371 	}
372 
373 	// regular input, schedule autocomplete
374 	var ev1 = new DwtKeyEvent();
375 	DwtKeyEvent.copy(ev1, ev);
376 	ev1.aclv = aclv;
377 	ev1.element = element;
378 	aclv._acAction.obj = aclv;
379 	aclv._acAction.args = [ ev1 ];
380 	DBG.println(AjxDebug.DBG2, "scheduling autocomplete");
381 	aclv._autocompleting = true;
382 
383 	aclv._acActionId = AjxTimedAction.scheduleAction(aclv._acAction, aclv._acInterval);
384 	
385 	return ZmAutocompleteListView._echoKey(true, ev);
386 };

470 /**
471  * Autocomplete typed text. Should be called by a handler for a keyboard event.
472  *
473  * @param {Element}	element	the element (some sort of text field) doing autocomplete
474  * @param {Object}	loc		where to popup the list, if appropriate
475  * 
476  * @private
477  */
478 ZmAutocompleteListView.prototype.autocomplete =
479 function(info) {
480 	DBG.println(AjxDebug.DBG3, "ZmAutocompleteListView: autocomplete");
481 	if (!info || (info.text == "undefined")) { return; }
482 	if (this._dataLoader && !this._dataLoaded) {
483 		this._data = this._dataLoader.call(this._dataClass);
484 		this._dataAPI = this._data;
485 		this._dataLoaded = true;
486 	}
487 
488 	// The callback into ourself emulates a while loop, and is here to support
489 	// async calls into _autocomplete(), which may result in a server request
490 	// (for example, when matching against the GAL)
491 	var callback = new AjxCallback(this, this.autocomplete);
492 	if (info.start < info.text.length) {
493 		var chunk = this._nextChunk(info.text, info.start);
494 		this._autocomplete(chunk, callback);
495 	} else if (info.change) {
496 		// quick completion was done
497 		this._autocompleting = false;
498 		DBG.println(AjxDebug.DBG2, "autocomplete, new text: " + info.text + "; element: " + this._element.value);
499 		this._updateField(info.text, info.match);
500 	}
501 };

504  * Resets the state of the autocomplete list.
505  */
506 ZmAutocompleteListView.prototype.reset =
507 function() {
508 
509 	this._matches = null;
510 	this._selected = null;

513 	this._forgetLink		= {};
514 	this._expandLink		= {};

517 	if (this._memberListView) {
518 		this._memberListView.show(false);
519 	}
520 	this.setWaiting(false);

523 /**
524 * Checks the given key to see if it's used to control the autocomplete list in some way.
525 * If it does, the action is taken and the key won't be echoed into the input area.
526 *
527 * The following keys are action keys:
528 *	38 40		up/down arrows (list selection)
529 *	27			escape (hide list)
530 *
531 * The following keys are delimiters (trigger completion):
532 *	3 13		return
533 *	9			tab
534 *	59 186		semicolon
535 *	188			comma
536 *
537 * @param key		a numeric key code
538 * @param isDelim	true if a single delimiter key was typed
539 * 
540 * @private
541 */
542 ZmAutocompleteListView.prototype.handleAction =
543 function(key, isDelim) {
544 
545 	DBG.println(AjxDebug.DBG2, "autocomplete handleAction for key " + key + " / " + isDelim);
546 
547 	if (isDelim) {
548 		this._update(true);
549 	} else if (key == 38 || key == 40) {
550 		// handle up and down arrow keys
551 		if (this.size() <= 1) { return; }
552 		if (key == 40) {
553 			this._setSelected(ZmAutocompleteListView.NEXT);
554 		} else if (key == 38) {
555 			this._setSelected(ZmAutocompleteListView.PREV);
556 		}
557 	} else if (key == 27) {
558 		this.reset(); // ESC hides the list
559 	}
560 };

562 /**
563  * Sets the waiting status.
564  * 
565  * @param	{Boolean}	on		if <code>true</code>, turn waiting "on"
566  * 
567  */
568 ZmAutocompleteListView.prototype.setWaiting =
569 function(on) {
570 
571 	if (!on && !this._waitingDiv) { return; }
572 
573 	var div = this._waitingDiv;
574 	if (!div) {

641 /*
642  * Finds the next chunk of text in a string that we should try to autocomplete, by reading
643  * until it hits some sort of address delimiter (or runs out of text). If the chunk is a
644  * valid email address, it's skipped since there's no point in trying to autocomplete it.
645  * 
646  * @param text	[string]	text to scan
647  * @param start	[int]		index to start at
648  */
649 ZmAutocompleteListView.prototype._nextChunk =
650 function(text, start) {
651 
652 	while (text.charAt(start) == ' ') {	// ignore leading space
653 		start++;
654 	}
655 	var insideQuotes = false;
656 	for (var i = start; i < text.length; i++) {
657 		var c = text.charAt(i);
658 		if (c == '"') {
659 			// skip text and delimiters that are quoted
660 			c = text.charAt(++i);
661 			while (i < text.length && c != '"') {
662 				c = text.charAt(++i);
663 			}
664 		}
665 		var isDelim = this._isDelim[c];
666 		if (isDelim || (this._options.addrBubbles && c == ' ')) {
667 			var chunk = text.substring(start, i);
668 			if (this._dataAPI.isComplete && this._dataAPI.isComplete(chunk)) {
669 				DBG.println(AjxDebug.DBG3, "skipping completed chunk: " + chunk);
670 				if (this._runCallbacks(ZmAutocompleteListView.CB_ADDR_FOUND, this._element && this._element.id, [this, chunk, c])) {
671 					return null;
672 				}
673 				if (isDelim) {
674 					start = i + 1;
675 				}
676 				while (text.charAt(start) == ' ') {	// ignore leading space
677 					start++;
678 				}
679 			} else if (isDelim) {
680 				return {text: text, str: chunk, start: start, end: i, delim: isDelim};
681 			}
682 		}
683 	}
684 	var chunk = text.substring(start, i);
685 
686 	return {text: text, str: chunk, start: start, end: i, delim: false};
687 };

689 // Looks for matches for a string and either displays them in a list, or does the completion
690 // immediately (if the string was followed by a delimiter). The chunk object that we get has
691 // information that allows us to do the replacement if we are performing completion.
692 ZmAutocompleteListView.prototype._autocomplete =
693 function(chunk, callback) {
694 	DBG.println(AjxDebug.DBG3, "ZmAutocompleteListView: _autocomplete");
695 
696 	if (!chunk || !(this._dataAPI && this._dataAPI.autocompleteMatch)) { return; }
697 	var str = AjxStringUtil.trim(chunk.str);
698 
699 	// if string is empty or already a delimited address, no reason to look for matches
700 	if (!(str && str.length) || (this._done[str])) {
701 		if (callback) {
702 			callback.run({text: chunk.text, start: chunk.end + 1});
703 			return;
704 		} else {
705 			return {text: chunk.text, start: chunk.end + 1};
706 		}
707 	}
708 
709 	//start and end of the chunk we are autocompleting, i.e. the range we might replace.
710 	this._start = chunk.start;
711 	this._end = chunk.end;
712 
713 	var text = chunk.text;
714 	var start = chunk.end; // move beyond the current chunk
715 
716 	this._removeAll();
717 
718 	var respCallback = new AjxCallback(this, this._handleResponseAutocomplete, [str, chunk, text, start, callback]);
719 	this._dataAPI.autocompleteMatch(str, respCallback, this, this._options, this._activeAccount);
720 };

722 ZmAutocompleteListView.prototype._handleResponseAutocomplete =
723 function(str, chunk, text, start, callback, list) {
724 
725 	var retValue;
726 	var change = false;
727 	this._autocompleting = false;
728 
729 	if (!retValue) {
730 		if (list && list.length) {
731 			DBG.println(AjxDebug.DBG2, "matches found: " + list.length);
732 			// done now in case of quick complete
733 			this._set(list); // populate the list view
734 		} else {
735 			this._showNoResults();
736 		}
737 
738 		// if text ends in a delimiter, complete immediately without showing the list
739 		var match;
740 		if (chunk && chunk.delim && (chunk.end == chunk.text.length - 1)) {
741 			DBG.println(AjxDebug.DBG2, "performing quick completion");
742 			var result = this._complete(text, str, true);
743 			if (result) {
744 				text = result.text;
745 				start = result.start;
746 				match = result.match;
747 				change = true;
748 			}
749 		} else if (list && list.length) {
750 			// show the list, unless this is not the first time we're displaying results for
751 			// this autocomplete string (eg GAL), and the user selected a result from the
752 			// first time we showed the list (bug 28886)
753 			if (!this._hasCompleted) {
754 				this.show(true);
755 			}
756 		}
757 	
758 		retValue = {text: text, start: start, match: match, change: change};
759 	}
760 	
761 	if (callback) {
762 		callback.run(retValue);
763 	}
764 };

766 /**
767  * Replaces the current chunk of text with the selected match (even if the list
768  * is not visible), or with replacement text returned by the data class. If the
769  * given text ends with a delimiter, "quick complete" is triggered, which does
770  * the replacement without showing the list.
771  * 
772  * @param text		[string]		current input text
773  * @param str		[string]*		current chunk
774  * @param hasDelim	[boolean]*		current chunk ends with a delimiter
775  * @param match		[object]*		forced match object
776  */
777 ZmAutocompleteListView.prototype._complete =
778 function(text, str, hasDelim, match) {
779 
780 	DBG.println(AjxDebug.DBG3, "ZmAutocompleteListView: _complete: selected is " + this._selected);
781 	match = match || this._matchHash[this._selected];
782 	if (!match)	{ return; }
783 
784 	var start = this._start;
785 	var end = hasDelim ? this._end + 1 : this._end;
786 	DBG.println(AjxDebug.DBG2, "update replace range: " + start + " - " + end);
787 	var value = this._getCompletionValue(match);
788 	var newText = this._options.addrBubbles ? value :
789 				  [text.substring(0, start), value, this._separator, text.substring(end, text.length)].join("");
790 	this._start = this._end = newText.length; //update in case this is called multiple times from ZmDLAutocompleteListView for selecitng all items from a DL
791 	if (value) {
792 		this._done[value] = true;
793 	}
794 	DBG.display(AjxDebug.DBG2, newText);
795 	return {text: newText, start: start + value.length + this._separator.length, match: match};
796 };

798 ZmAutocompleteListView.prototype._getCompletionValue =
799 function(match) {
800 	var value = "";
801 	if (this._matchValue instanceof Array) {
802 		for (var i = 0, len = this._matchValue.length; i < len; i++) {
803 			if (match[this._matchValue[i]]) {
804 				value = match[this._matchValue[i]];
805 				break;
806 			}
807 		}
808 	} else {
809 		value = match[this._matchValue] || "";
810 	}
811 	return value;
812 };

814 // Resets the value of an element to the given text.
815 ZmAutocompleteListView.prototype._updateField =
816 function(text, match, ev) {
817 
818 	var el = this._element;
819 	if (this._options.addrBubbles) {
820 		var addrInput = el && el._aifId && DwtControl.ALL_BY_ID[el._aifId];

824 			}

1264 /**
1265  * Displays a second popup list with the members of the given distribution list.
1266  *
1267  * @param {string}			email		address of a distribution list
1268  * @param {string}			textId		ID of link text
1269  * @param {string}			rowId		ID or list view row
1270  * @param {DwtMouseEvent}	ev			mouse event
1271  * @param {DwtPoint}		loc			location to popup at; default is right of parent ACLV
1272  * @param {Element}			element		input element
1273  */
1274 ZmAutocompleteListView.prototype.expandDL =
1275 function(email, textId, rowId, ev, loc, element) {
1276 
1277 	if (!this._dataAPI.expandDL) { return; }
1278 
1279 	var mlv = this._memberListView;
1280 	if (mlv && mlv.getVisible() && textId && this._curExpanded == textId) {
1281 		// User has clicked "Collapse" link
1282 		mlv.show(false);
1283 		this._curExpanded = null;
1284 		this._setExpandText(textId, false);
1285 	} else {
1286 		// User has clicked "Expand" link
1287 		if (mlv && mlv.getVisible()) {
1288 			// expanding a DL while another one is showing
1289 			this._setExpandText(this._curExpanded, false);
1290 			mlv.show(false);
1291 		}
1292 		var contactsApp = appCtxt.getApp(ZmApp.CONTACTS);
1293 		var contact = contactsApp.getContactByEmail(email);
1294 		if (!contact) {
1295 			contact = new ZmContact(null);
1296 			contact.initFromEmail(email);	// don't cache, since it's not a real contact (no ID)
1297 		}
1298 		contact.isDL = true;
1299 		if (textId && rowId) {
1300 			this._curExpanded = textId;
1301 			this._setExpandText(textId, true);
1302 		}
1303 		this._dataAPI.expandDL(contact, 0, new AjxCallback(this, this._handleResponseExpandDL, [contact, loc, textId]));
1304 	}
1305 	this._element = element || this._element;
1306 	if (this._element) {
1307 		this._element.focus();
1308 	}
1309 };

1370 ZmAutocompleteListView.prototype._runCallbacks =
1371 function(type, inputId, args) {
1372 
1373 	var result = null;
1374 	var list = this._callbacks[type];
1375 	if (list && list.length) {
1376 		for (var i = 0; i < list.length; i++) {
1377 			var cbObj = list[i];
1378 			if (inputId && cbObj.inputId && (inputId != cbObj.inputId)) { continue; }
1379 			var r = AjxCallback.prototype.run.apply(cbObj.callback, args);
1380 			if (r === true || r === false) {
1381 				result = (result == null) ? r : result && r;
1382 			}
1383 		}
1384 	}
1385 	return result;
1386 };
