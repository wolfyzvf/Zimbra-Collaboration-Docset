 21 /**
 22  * Creates an address book dialog.
 23  * @class
 24  * This class represents the address book dialog.
 25  * 
 26  * @param	{DwtControl}	parent		the parent
 27  * @param	{String}	className		the class name
 28  * @param	{String}	title		the dialog title
 29  * @param	{constant}	type		the type
 30  * 
 31  * @extends		ZmNewOrganizerDialog
 32  */
 33 ZmNewAddrBookDialog = function(parent, className) {
 34 	var title = ZmMsg.createNewAddrBook;
 35 	var type = ZmOrganizer.ADDRBOOK;
 36 	ZmNewOrganizerDialog.call(this, parent, className, title, type);
 37 }
 38 
 39 ZmNewAddrBookDialog.prototype = new ZmNewOrganizerDialog;
 40 ZmNewAddrBookDialog.prototype.constructor = ZmNewAddrBookDialog;
 41 
 42 
 43 // Public methods
 44 
 45 /**
 46  * Returns a string representation of the object.
 47  * 
 48  * @return		{String}		a string representation of the object
 49  */
 50 ZmNewAddrBookDialog.prototype.toString =
 51 function() {
 52 	return "ZmNewAddrBookDialog";
 53 };

 56 // Protected methods

 58 /**
 59  * overload since we always want to init the color to grey.
 60  * 
 61  * @private
 62  */
 63 ZmNewAddrBookDialog.prototype._initColorSelect =
 64 function() {
 65 	var option = this._colorSelect.getOptionWithValue(ZmOrganizer.DEFAULT_COLOR[this._organizerType]);
 66 	this._colorSelect.setSelectedOption(option);
 67 };
 68 
 69 /**
 70  * overload so we dont show this.
 71  * 
 72  * @private
 73  */
 74 ZmNewAddrBookDialog.prototype._createRemoteContentHtml =

 76 	return idx;
 77 };

 79 /**
 80  * @private
 81  */
 82 ZmNewAddrBookDialog.prototype._setupFolderControl =
 83 function(){
 84     ZmNewOrganizerDialog.prototype._setupFolderControl.call(this);
 85     if(this._omit) this._omit[ZmFolder.ID_TRASH] = true;
 86 };
