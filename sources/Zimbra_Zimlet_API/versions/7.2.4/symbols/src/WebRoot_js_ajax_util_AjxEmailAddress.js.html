 16 /**
 17  * Creates a new emal address, either by parsing an email string or from component parts.
 18  * @constructor
 19  * @class
 20  * This class represents an email address and defines some related constants. The class does not attempt full compliance
 21  * with RFC2822, so there are limitations for some of the edge cases.
 22  *
 23  * @author Conrad Damon
 24  * 
 25  * @param {string}	address		an email string, or just the address portion
 26  * @param {constant}	type		from, to, cc, bcc, or reply-to
 27  * @param {string}	name		the personal name portion
 28  * @param {string}	dispName	a brief display version of the name
 29  * @param {boolean}	isGroup		if <code>true</code>, the address param is really a list of email addresses
 30  * 
 31  */
 32 AjxEmailAddress = function(address, type, name, dispName, isGroup) {
 33 	this.address = address;
 34 	this.name = this._setName(name);
 35 	this.dispName = dispName;
 36 	this.type = type || AjxEmailAddress.TO;
 37 	this.isGroup = isGroup;
 38 	this.canExpand = false;
 39     this.isAjxEmailAddress = true;
 40 };

103 AjxEmailAddress.addrAnglePat = /(\s*<(((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*"(([^\\"])|(\\([^\x0A\x0D])))+"\s*))\@((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*\[(\s*(([^\[\]\\])|(\\([^\x0A\x0D])))+)*\s*\]\s*)))>\s*)/;
104 AjxEmailAddress.addrAngleQuotePat = /(\s*<'(((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*"(([^\\"])|(\\([^\x0A\x0D])))+"\s*))\@((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*\[(\s*(([^\[\]\\])|(\\([^\x0A\x0D])))+)*\s*\]\s*)))'>\s*)/;
105 // use addrPat to validate strings as email addresses
106 AjxEmailAddress.addrPat = /(((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*"(([^\\"])|(\\([^\x0A\x0D])))+"\s*))\@((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*\[(\s*(([^\[\]\\])|(\\([^\x0A\x0D])))+)*\s*\]\s*)))/;
107 // use addrPat1 to parse email addresses - pattern is lenient in that it will allow the following:
108 // 		"Joe Smith" joe@x.com
109 //		"Joe Smith"joe@x.com
110 // (RFC822 wants the address part to be in <> if preceded by name part)
111 AjxEmailAddress.addrPat1 = /(^|"|\s)(((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*"(([^\\"])|(\\([^\x0A\x0D])))+"\s*))\@((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*\[(\s*(([^\[\]\\])|(\\([^\x0A\x0D])))+)*\s*\]\s*)))/;
112 // pattern below is for account part of address (before @)
113 AjxEmailAddress.accountPat = /((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*"(([^\\"])|(\\([^\x0A\x0D])))+"\s*))/;
114 // Pattern below hangs on an unclosed comment, so use simpler one if parsing for comments
115 //AjxEmailAddress.commentPat = /(\s*\((\s*(([^()\\])|(\\([^\x0A\x0D]))|(\s*\((\s*(([^()\\])|(\\([^\x0A\x0D]))|(\s*\((\s*(([^()\\])|(\\([^\x0A\x0D]))|(\s*\((\s*(([^()\\])|(\\([^\x0A\x0D]))|(\s*\((\s*(([^()\\])|(\\([^\x0A\x0D]))|)+)*\s*\)\s*))+)*\s*\)\s*))+)*\s*\)\s*))+)*\s*\)\s*))+)*\s*\)\s*)/;
116 AjxEmailAddress.commentPat = /\((.*)\)/g;
117 AjxEmailAddress.phrasePat = /(((\s*[^\x00-\x1F\x7F()<>\[\]:;@\"\s]+\s*)|(\s*"(([^\\"])|(\\([^\x0A\x0D])))+"\s*))+)/;
118 AjxEmailAddress.boundAddrPat = /(\s*<?(((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*"(([^\\"])|(\\([^\x0A\x0D])))+"\s*))\@((\s*([^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+(\.[^\x00-\x1F\x7F()<>\[\]:;@\,."\s]+)*)\s*)|(\s*\[(\s*(([^\[\]\\])|(\\([^\x0A\x0D])))+)*\s*\]\s*)))>?\s*)$/;

120 /**
121  * Parses an email address string into its component parts. The parsing is adapted from the perl module 
122  * <a href="http://search.cpan.org/~cwest/Email-Address-1.2/lib/Email/Address.pm">Email::Address</a>. Check that out if you
123  * want to know how the gory regexes that do the parsing were built. They are based on RFC2822, but don't represent a full 
124  * implementation. We don't really need or want that, since we don't want to be overly restrictive or bloated. It was easier
125  * to just use the resulting regexes from the Perl module, rather than go through all the rigmarole of building them up from
126  * atoms.
127  * <p>
128  * If the address parses successfully, the current object's properties will be set.
129  * </p>
130  * 
131  * @param	{string}	str		the string to parse
132  * @return	{AjxEmailAddress}	the email address or <code>null</code>
133  */
134 AjxEmailAddress.parse =
135 function(str) {
136 	var addr, name;
137 	var str = AjxStringUtil.trim(str);
138 	var prelimOkay = AjxEmailAddress._prelimCheck(str);
139 	if (!(prelimOkay && str.match(AjxEmailAddress.addrPat))) {
140 		DBG.println(AjxDebug.DBG2, "mailbox match failed: " + str);
141 		return null;
142 	}
143 
144 	// Note: It would be nice if you could get back the matching parenthesized subexpressions from replace,
145 	// then we wouldn't have to do both a match and a replace. The parsing works by removing parts after it
146 	// finds them.
147 	
148 	// First find the address (and remove it)
149 	var parts = str.match(AjxEmailAddress.addrAngleQuotePat) || str.match(AjxEmailAddress.addrAnglePat);
150 	if (parts && parts.length) {
151 		addr = parts[2];
152 		str = str.replace(AjxEmailAddress.addrAnglePat, '');
153 	} else {
154 		parts = str.match(AjxEmailAddress.addrPat1);
155 		if (parts && parts.length) {
156 			if (parts[1] == '"') {
157 				return null;	// unmatched quote
158 			}
159             //AjxEmailAddress.addrPat recognizes the email better than using parts[0] from AjxEmailAddress.addrPat1
160             addr = str.match(AjxEmailAddress.addrPat);
161             addr = (addr && addr.length && addr[0] != "") ? AjxStringUtil.trim(addr[0]) : parts[0];
162 			if (addr && addr.indexOf("..") != -1) {
163 				return null;
164 			}
165 			str = str.replace(AjxEmailAddress.addrPat, '');
166 		}
167 	}
168  	if (!addr) {
169 		return null;
170 	}
171 	//Invalidate if address matches any of AjxEmailAddress.customInvalidEmailPats
172 	for(var i = 0; i< AjxEmailAddress.customInvalidEmailPats.length; i++) {
173 	   var match = addr.match(AjxEmailAddress.customInvalidEmailPats[i]);
174 		if(match) {
175 			return null;
176 		}
177 	}
178 
179 	// What remains is the name
180 	parts = str.match(AjxEmailAddress.phrasePat);
181 	if (parts) {
182 		name = AjxStringUtil.trim(parts[0]);
183 
184 		// Trim off leading and trailing quotes, but leave escaped quotes and unescape them
185 		name = name.replace(/\\"/g,""");
186 		name = AjxStringUtil.trim(name, null, "\"");
187 		name = name.replace(/"/g,"\"");
188 	}
189 	
190 	return new AjxEmailAddress(addr, null, name);
191 };

193 /**
194  * Parses a string with one or more addresses and parses it. An object with lists of good addresses, bad
195  * addresses, and all addresses is returned. Strict RFC822 validation (at least as far as it goes in the
196  * regexes we have) is optional. If it's off, we'll retry a failed address after quoting the personal part.
197  *
198  * @param {string}	emailStr	an email string with one or more addresses
199  * @param {constant}	type		address type of the string
200  * @param {boolean}	strict		if <code>true</code>, do strict checking
201  * @return	{hash}		the good/bad/all addresses
202  */
203 AjxEmailAddress.parseEmailString =
204 function(emailStr, type, strict) {
205 	var good = new AjxVector();
206 	var bad = new AjxVector();
207 	var all = new AjxVector();
208 	var addrList = AjxEmailAddress.split(emailStr);
209 	for (var i = 0; i < addrList.length; i++) {
210 		var addrStr = AjxStringUtil.trim(addrList[i]);
211 		if (addrStr) {
212 			var addr = AjxEmailAddress.parse(addrStr);
213 			if (!addr && !strict) {
214 				var temp = addrStr;
215 				var parts = temp.match(AjxEmailAddress.addrAnglePat);
216 				if (parts && parts.length) {
217 					var name = temp.replace(AjxEmailAddress.addrAnglePat, '');
218 					var newAddr = ['"', name, '" ', parts[0]].join("");
219 					addr = AjxEmailAddress.parse(newAddr);
220 					if (addr) {
221 						addr.name = name; // reset name to original unquoted form
222 					}
223 				}
224 			}
225 			if (addr) {
226 				addr.type = type;
227 				good.add(addr);
228 				all.add(addr);
229 			} else {
230 				bad.add(addrStr);
231 				all.add(new AjxEmailAddress(addrStr));
232 			}
233 		}
234 	}
235 	return {good: good, bad: bad, all: all};
236 };

248 AjxEmailAddress._prelimCheck =
249 function(str) {
250 	// Do preliminary check for @ since we don't support local addresses, and as workaround for Mozilla bug
251 	// https://bugzilla.mozilla.org/show_bug.cgi?id=225094
252 	// Also check for . since we require FQDN
253 	var atIndex = str.indexOf('@');
254 	var dotIndex = str.lastIndexOf('.');
255 	return ((atIndex != -1) && (dotIndex != -1) && (dotIndex > atIndex) && (dotIndex != str.length - 1));
256 };

258 /**
259  * Splits a string into (possible) email address strings based on delimiters. Tries to
260  * be flexible about what it will accept. The following delimiters are recognized, under
261  * the following conditions:
262  *
263  * <ul>
264  * <li><i>return</i> -- always</li>
265  * <li><i>semicolon</i> -- must not be inside quoted or comment text</li>
266  * <li><i>comma</i> -- must not be inside quoted or comment text, and must follow an address (which may be in angle brackets)</li>
267  * <li><i>space</i> -- can only separate plain addresses (no quoted or comment text)</li>
268  * </ul>
269  * 
270  * The requirement that a comma follow an address allows us to be lenient when a mailer
271  * doesn't quote the friendly part, so that a string such as the one below is split correctly:
272  * <code>Smith, John <jsmith@aol.com></code>
273  *
274  * @param {string}	str	the string to be split
275  * @return	{array}	the list of {String} addresses
276  */
277 AjxEmailAddress.split =
278 function(str) {
279 	str = AjxStringUtil.trim(str);
280 	// first, construct a list of ranges to ignore because they are quoted or comment text
281 	var ignore = [];
282 	var pos = 0, startPos = 0;
283 	var prevCh = "", startCh = "";
284 	var inside = false;
285 	while (pos < str.length) {
286 		var ch = str.charAt(pos);
287 		if ((ch == '"' || ch == '(') && prevCh != "\\") {
288 			inside = true;
289 			startCh = ch;
290 			startPos = pos;
291 			pos++;
292 			while (inside && pos < str.length) {
293 				var ch = str.charAt(pos);
294 				if (((startCh == '"' && ch == '"') || (startCh == '(' && ch == ')')) && (prevCh != "\\")) {
295 					ignore.push({start: startPos, end: pos});
296 					inside = false;
297 				}
298 				pos++;
299 				prevCh = ch;
300 			}
301 		} else {
302 			pos++;
303 		}
304 		prevCh = ch;
305 	}
306 	if (ignore.length) {
307 		AjxEmailAddress.IS_DELIM[" "] = false;
308 	}
309 	
310 	// Progressively scan the string for delimiters. Once an email string has been found, continue with
311 	// the remainder of the original string.
312 	startPos = 0;
313 	var addrList = [];
314 	while (startPos < str.length) {
315 		var sub = str.substring(startPos, str.length);
316 		pos = 0;
317 		var delimPos = sub.length;
318 		while ((delimPos == sub.length) && (pos < sub.length)) {
319 			var ch = sub.charAt(pos);
320 			if (AjxEmailAddress.IS_DELIM[ch]) {
321 				var doIgnore = false;
322 				if (ch != "\n") {
323 					for (var i = 0; i < ignore.length; i++) {
324 						var range = ignore[i];
325 						var absPos = startPos + pos;
326 						doIgnore = (absPos >= range.start && absPos <= range.end);
327 						if (doIgnore) break;
328 					}
329 				}
330 				if (!doIgnore) {
331 					var doAdd = true;
332 					var test = sub.substring(0, pos);
333 					if (ch == "," || ch == " ") {
334 						// comma/space allowed as non-delimeter outside quote/comment,
335 						// so we make sure it follows an actual address
336 
337 						//bug 77630 - instead of this ==> test.match(AjxEmailAddress.boundAddrPat);
338 						//I do the following, to be more lenient in detecting attempts for an email address, even if
339 						//it's an illegal address, so it is not taken mistakingly as the name of the next entry after the comma
340 						doAdd = test.indexOf('.') > -1 && test.indexOf('@') > -1;
341 					}
342 					if (doAdd) {
343 						addrList.push(test);
344 						delimPos = pos;
345 						startPos += test.length + 1;
346 					}
347 				}
348 				// strip extra delimeters
349 				ch = str.charAt(startPos);
350 				while ((startPos < str.length) && AjxEmailAddress.IS_DELIM[ch]) {
351 					startPos++;
352 					ch = str.charAt(startPos);
353 				}
354 				pos++;
355 			} else {
356 				pos++;
357 			}
358 		}
359 		if (delimPos == sub.length) {
360 			addrList.push(sub);
361 			startPos += sub.length + 1;
362 		}
363 	}
364 	AjxEmailAddress.IS_DELIM[" "] = true;
365 
366 	return addrList;
367 };

369 /**
370  * Returns a string representation of this object.
371  * 
372  * @param {boolean}		shortForm	if true, return a brief version (name if available, otherwise email)
373  * @param {boolean}		forceUnescape	if true, name will not be in quotes and any quotes inside the name will be unescaped (e.g. "John \"JD\" Doe" <jd@zimbra.com> becomes John "JD" Doe <jd@zimbra.com>
374  * 
375  * @return	{string}		a string representation of this object
376  */
377 AjxEmailAddress.prototype.toString =
378 function(shortForm, forceUnescape) {
379 
380 	if (this.name) {
381 		var name = this.name;
382 		if (!shortForm && !forceUnescape) {
383 			name = name.replace(/\\+"/g, '"');	// unescape double quotes (avoid double-escaping)
384 			name = name.replace(/"/g,'\\"');  // escape quotes
385 		}
386 		var buffer = (shortForm || forceUnescape) ? [name] : ['"', name, '"'];
387 		if (this.address && !shortForm) {
388 			buffer.push(" <", this.address, ">");
389 		}
390 		return buffer.join("");	// quote friendly part
391 	} else {
392 		return this.address;
393 	}
394 };

495 AjxEmailAddress.prototype._setName =
496 function(name) {
497 	if (!name) return "";
498 	
499 	// remove wrapping single quotes from name if present
500 	if (name && name.charAt(0) == "'" && name.charAt(name.length - 1) == "'")
501 		name = name.substring(1, name.length - 1);
502 		
503 	return name;

506 AjxEmailAddress.sortCompareByAddress =
507 function(a, b) {
508
