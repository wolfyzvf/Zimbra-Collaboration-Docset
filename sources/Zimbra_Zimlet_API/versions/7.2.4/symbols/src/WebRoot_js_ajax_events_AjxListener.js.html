 17 /**
 18 * Creates a new listener.
 19 * @constructor
 20 * @class
 21 * This class represents a listener, which is a function to be called in response to an event.
 22 * A listener is a slightly specialized callback: it has a {@link #handleEvent} method, and it does not
 23 * return a value.
 24 *
 25 * @author Ross Dargahi
 26 * 
 27 * @param {Object}	obj	the object to call the function from
 28 * @param {function}	func	the listener function
 29 * @param {primative|array}	args   the default arguments
 30 * 
 31 * @extends		AjxCallback
 32 */
 33 AjxListener = function(obj, method, args) {
 34 	AjxCallback.call(this, obj, method, args);
 35 }
 36 
 37 AjxListener.prototype = new AjxCallback();
 38 AjxListener.prototype.constructor = AjxListener;
 39 
 40 AjxListener.prototype.toString = 
 41 function() {
 42 	return "AjxListener";
 43 }
 44 
 45 /**
 46 * Invoke the listener function.
 47 *
 48 * @param {AjxEvent}		ev		the event object that gets passed to an event handler
 49 */
 50 AjxListener.prototype.handleEvent =
 51 function(ev) {
 52 	return this.run(ev);
 53 }
