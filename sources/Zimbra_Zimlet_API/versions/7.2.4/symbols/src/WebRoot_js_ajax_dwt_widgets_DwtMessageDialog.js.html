 21 /**
 22  * @class
 23  * 
 24  * Creates a new message dialog. This class represents a reusable message dialog box.
 25  * Messages can be informational, warning, or critical.
 26  * 
 27  * @author Ross Dargahi
 28  * 
 29  * @param {hash}		params			a hash of parameters
 30  * @param {DwtComposite}	params.parent		the parent widget (the shell)
 31  * @param {string}	params.className 		the CSS class
 32  * @param {array}	params.buttons				the buttons to show. Defaults to {@link DwtDialog.OK_BUTTON} button
 33  * @param {array}	params.extraButtons	  	a list of {@link DwtDialog_ButtonDescriptor} objects describing custom buttons to add to the dialog
 34  * 
 35  * @extends	DwtDialog
 36  */
 37 DwtMessageDialog = function(params) {
 38 	if (arguments.length == 0) { return; }
 39 	params = Dwt.getParams(arguments, DwtMessageDialog.PARAMS);
 40 	this._msgCellId = Dwt.getNextId("MessageDialog_");
 41 	params.standardButtons = params.buttons || [DwtDialog.OK_BUTTON];
 42 	DwtDialog.call(this, params);
 43 	
 44 	this.setContent(this._contentHtml());
 45 	this._msgCell = document.getElementById(this._msgCellId);
 46 	this.addEnterListener(new AjxListener(this, this._enterListener));
 47 };
 48 
 49 DwtMessageDialog.PARAMS = ["parent", "className", "buttons", "extraButtons", "id"];
 50 
 51 DwtMessageDialog.prototype = new DwtDialog;
 52 DwtMessageDialog.prototype.constructor = DwtMessageDialog;
 53 
 54 /**
 55  * Defines the "critical" style.
 56  */
 57 DwtMessageDialog.CRITICAL_STYLE = 1;
 58 /**
 59  * Defines the "info" style.
 60  */
 61 DwtMessageDialog.INFO_STYLE = 2;
 62 /**
 63  * Defines the "warning" style.
 64  */
 65 DwtMessageDialog.WARNING_STYLE = 3;
 66 
 67 DwtMessageDialog.TITLE = {};
 68 DwtMessageDialog.TITLE[DwtMessageDialog.CRITICAL_STYLE] = AjxMsg.criticalMsg;
 69 DwtMessageDialog.TITLE[DwtMessageDialog.INFO_STYLE] = AjxMsg.infoMsg;
 70 DwtMessageDialog.TITLE[DwtMessageDialog.WARNING_STYLE] = AjxMsg.warningMsg;
 71 
 72 DwtMessageDialog.ICON = {};
 73 DwtMessageDialog.ICON[DwtMessageDialog.CRITICAL_STYLE] = "Critical_32";
 74 DwtMessageDialog.ICON[DwtMessageDialog.INFO_STYLE] = "Information_32";
 75 DwtMessageDialog.ICON[DwtMessageDialog.WARNING_STYLE] = "Warning_32";

 78 // Public methods
 79 
 80 /**
 81  * Returns a string representation of the object.
 82  * 
 83  * @return		{string}		a string representation of the object
 84  */
 85 DwtMessageDialog.prototype.toString = 
 86 function() {
 87 	return "DwtMessageDialog";
 88 };
 89 
 90 /**
 91 * Sets the message style (info/warning/critical) and content.
 92 *
 93 * @param {string}	msgStr		the message text
 94 * @param {constant}	style		the style (see <code>DwtMessageDialog.*_STYLE</code> constants)
 95 * @param {string}	title		the dialog box title
 96 */
 97 DwtMessageDialog.prototype.setMessage =
 98 function(msgStr, style, title) {
 99 	style = style || DwtMessageDialog.INFO_STYLE;
100 	title = title || DwtMessageDialog.TITLE[style];
101 	this.setTitle(title);
102 	if (msgStr) {
103         var html = [];
104 		var i = 0;
105 		html[i++] = "<table cellspacing=0 cellpadding=0 border=0 width=100% height=100%><tr><td valign='top'>";
106 		html[i++] =  AjxImg.getImageHtml(DwtMessageDialog.ICON[style], null, "id='" +  this._msgCellId + "_Image'");
107 		html[i++] = "</td><td class='DwtMsgArea' id='" +  this._msgCellId +"_Msg'>";
108 		html[i++] = msgStr;
109 		html[i++] = "</td></tr></table>";
110 		this._msgCell.innerHTML = html.join("");
111 	} else {
112 		this._msgCell.innerHTML = "";
113 	}
114 };
115 
116 DwtMessageDialog.prototype.setSize =
117 function(width, height) {
118 	var msgCell = document.getElementById(this._msgCellId);
119 	if (msgCell && (width || height)) {
120 		Dwt.setSize(msgCell, width, height);
121 	}
122 };
123 
124 /**
125  * Resets the message dialog. This should be performed to "reuse" the dialog.
126  * 
127  */
128 DwtMessageDialog.prototype.reset = 
129 function() {
130 	this._msgCell.innerHTML = "";
131 	DwtDialog.prototype.reset.call(this);
132 };
133 
134 /**
135  * Handles the dialog key action. If the user hits the "Esc" key and no "Cancel" button is present,
136  * the key action is treated it as a press of the "OK" button.
137  * 
138  * @param	{DwtKeyMap}		actionCode	the key action code
139  * @param	{DwtKeyEvent}	ev	the key event
140  * 
141  * @private
142  */
143 DwtMessageDialog.prototype.handleKeyAction =
144 function(actionCode, ev) {
145 	return DwtDialog.prototype.handleKeyAction.call(this, actionCode, ev);
146 };

148 // Private methods
149 
150 /**
151  * @private
152  */
153 DwtMessageDialog.prototype._contentHtml = 
154 function() {
155 	return "<div id='" + this._msgCellId + "' class='DwtMsgDialog'></div>";
156 };
157 
158 /**
159  * @private
160  */
161 DwtMessageDialog.prototype._enterListener =
162 function(ev) {
163 	this._runEnterCallback();
164 };
