 16 /**
 17  * @param {hash} params Constructor parameters.
 18  * @param {DwtComposite} params.parent Parent control.
 19  * @param {string} params.image Item image to display next to each color choice.
 20  * @param {boolean} params.hideNone True to hide the "None" option.
 21  * @param {boolean} params.hideNoFill True to hide the no-fill/use-default option.
 22  */
 23 ZmColorMenu = function(params) {
 24     if (arguments.length == 0) return;
 25     params.className = params.className || "ZmColorMenu DwtMenu";
 26     DwtMenu.call(this, params);
 27     this._hideNone = params.hideNone;
 28     this._hideNoFill = params.hideNoFill;
 29     this.setImage(params.image);
 30     this._populateMenu();
 31 };
 32 ZmColorMenu.prototype = new DwtMenu;
 33 ZmColorMenu.prototype.constructor = ZmColorMenu;
 34 
 35 ZmColorMenu.prototype.toString = function() {
 36     return "ZmColorMenu";
 37 };

 49 ZmColorMenu.prototype.setImage = function(image) {
 50     if (this._image != image) {
 51         this._image = image;
 52         var children = this.getChildren();
 53         for (var i = 0; i < children.length; i++) {
 54             var child = children[i];
 55             var color = child.getData(ZmColorMenu.__KEY_COLOR);
 56             var icon = image && color ?[image,color].join(",color=") : image;
 57             child.setImage(icon);
 58         }
 59     }
 60 };
 61 ZmColorMenu.prototype.getImage = function() {
 62     return this._image;
 63 };
 64 
 65 ZmColorMenu.prototype.getTextForColor = function(color) {
 66     color = String(color).toLowerCase();
 67     if (!color.match(/^#/)) color = ZmOrganizer.COLOR_VALUES[color];
 68     var children = this.getChildren();
 69     for (var i = 0; i < children.length; i++) {
 70         var mi = children[i];
 71         if (mi.getData(ZmColorMenu.__KEY_COLOR) == color) {
 72             return mi.getText();
 73         }
 74     }
 75     return ZmMsg.custom;
 76 };

 78 ZmColorMenu.prototype.showMoreColors = function() {
 79     if (this.parent && this.parent.getMenu() == this) {
 80         var menu = this.parent.getMenu();
 81         var moreMenu = this._getMoreColorMenu();
 82         this.parent.setMenu(moreMenu);
 83         if (menu.isPoppedUp()) {
 84             var loc = menu.getLocation();
 85             menu.popdown();
 86             moreMenu.popup(0, loc.x, loc.y);
 87         }
 88         else {
 89             this.parent.popup();
 90         }
 91     }
 92 };
 93 
 94 ZmColorMenu.prototype.showLessColors = function() {
 95     if (this.parent && this.parent.getMenu() != this) {
 96         var menu = this.parent.getMenu();
 97         this.parent.setMenu(this);
 98         if (menu.isPoppedUp()) {
 99             var loc = menu.getLocation();
100             menu.popdown();
101             this.popup(0, loc.x, loc.y);
102         }
103         else {
104             this.parent.popup();
105         }
106     }
107 };

126 ZmColorMenu.prototype._populateMenu = function() {
127     var list = ZmOrganizer.COLOR_VALUES;
128     for (var id = 0; id < list.length; id++) {
129         var color = ZmOrganizer.COLOR_VALUES[id];
130         if (!color && this._hideNone) continue;
131         var image = this._image && color ? [this._image,color].join(",color=") : this._image;
132         var text = ZmOrganizer.COLOR_TEXT[id];
133         var menuItem = new DwtMenuItem({parent:this});
134         menuItem.setImage(image);
135         menuItem.setText(text);
136         menuItem.setData(ZmOperation.MENUITEM_ID, id);
137         menuItem.setData(ZmColorMenu.__KEY_COLOR, color);
138     }
139     var callback = new AjxCallback(this, this.showMoreColors); 
140     var showMoreItem = new ZmColorMenuItem({parent:this,callback:callback});
141     showMoreItem.setText(ZmMsg.colorsShowMore);
142 };
143 
144 ZmColorMenu.prototype._getMoreColorMenu = function() {
145     if (!this._moreMenu) {
146         var callback = new AjxCallback(this, this.showLessColors);
147         this._moreMenu = new ZmMoreColorMenu({parent:this.parent,callback:callback,hideNoFill:this._hideNoFill});
148     }
149     return this._moreMenu;
150 };

156 ZmMoreColorMenu = function(params) {
157     params.style = DwtMenu.COLOR_PICKER_STYLE;
158     DwtMenu.call(this, params);
159     this._colorPicker = new DwtColorPicker({parent:this,hideNoFill:params.hideNoFill});
160     this._colorPicker.getData = this.__DwtColorPicker_getData; // HACK
161     var showLessItem = new ZmColorMenuItem({parent:this,callback:params.callback});
162     showLessItem.setText(ZmMsg.colorsShowLess);
163 };

171 /**
172  * <strong>Note:</strong>
173  * This method is run in the context of the color picker!
174  *
175  * @private
176  */
177 ZmMoreColorMenu.prototype.__DwtColorPicker_getData = function(key) {
178     // HACK: This is to fake the color picker as a menu item whose
179     // HACK: id is the selected color.
180     if (key == ZmOperation.MENUITEM_ID) {
181         return this.getInputColor();
182     }
183     return DwtColorPicker.prototype.getData.apply(this, arguments);
184 };

190 /**
191  * A custom menu item class for the "More colors..." and
192  * "Fewer colors..." options which should not leave space for
193  * an image next to the text. A sub-class is also needed so
194  * that we can avoid the default handling of the item click.
195  *
196  * @param params
197  */
198 ZmColorMenuItem = function(params) {
199     DwtMenuItem.call(this, params);
200     this.callback = params.callback;
201     // HACK: This is needed because we no-op the add/removeSelectionListener
202     // HACK: methods so that external people can't register listeners but we
203     // HACK: still want to handle a true selection to call the callback.
204     DwtMenuItem.prototype.addSelectionListener.call(this, new AjxListener(this, this.__handleItemSelect));
205 };
206 ZmColorMenuItem.prototype = new DwtMenuItem;
207 ZmColorMenuItem.prototype.constructor = ZmColorMenuItem;
208 
209 ZmColorMenuItem.prototype.toString = function() {
210     return "ZmColorMenuItem";
211 };
