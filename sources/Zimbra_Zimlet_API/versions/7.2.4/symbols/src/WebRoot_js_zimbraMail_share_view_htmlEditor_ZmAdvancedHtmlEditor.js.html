 67 ZmAdvancedHtmlEditor.prototype.setSize =
 68 function(x, y) {
 69 	var editor = this.getEditor();
 70 	var bodyField = this.getBodyField();
 71 
 72 	// FUDGE: we must substract borders and paddings - yuck.
 73 	var delta = this._mode == DwtHtmlEditor.HTML ? 10 : 8;
 74 
 75 	x -= delta + 4;
 76 	y -= delta; // subtract fudge factor
 77 
 78 	// bug fix #6786 - normalize width/height if less than zero
 79 	if (x < 0) { x = 0; }
 80 	if (y < 0) { y = 0; }
 81 
 82 	var editorContainer = document.getElementById(this._bodyTextAreaId + "_tbl");
 83 	if (editor && editorContainer) {
 84 		editorContainer.style.height = y + "px";
 85 		editorContainer.style.width = "100%";
 86 	}
 87 
 88 	if (this._mode == DwtHtmlEditor.HTML && this.isTinyMCE)
 89 		y -= 26;
 90 
 91 	bodyField.style.width = x + 5 + "px";
 92 	bodyField.style.height = y + "px";
 93 
 94 	//todo: handle spellcheck ids
 95 };

129 ZmAdvancedHtmlEditor.prototype.getContent =
130 function(insertFontStyle, onlyInnerContent) {
131 
132     this.discardMisspelledWords();
133     
134 	var field = this.getContentField();
135 
136 	if (this._mode == DwtHtmlEditor.HTML) {
137 		var editor = this.getEditor();
138         var params = {};
139         params.format ='raw';
140 		var content = editor ? editor.getContent(params) : (this._pendingContent || "");
141         if(content == '<br mce_bogus="1">' || content == '<br mce_bogus="1"/>') {
142             content = '';
143         }        
144 		content = this._embedHtmlContent(content, insertFontStyle, onlyInnerContent);
145 		return content;
146 	}
147 
148 	return field.value;
149 };

151 ZmAdvancedHtmlEditor.prototype._embedHtmlContent =
152 function(html, insertFontStyle, onlyInnerContent) {
153 	if (!insertFontStyle && !onlyInnerContent) {
154 		return [ "<html><body>", html, "</body></html>" ].join("");
155 	}
156 
157 	if (onlyInnerContent) {
158 		var cont = [];
159 		var idx = 0;
160 
161 		if (insertFontStyle) {
162 			cont[idx++] = "<div";
163 			cont[idx++] = " style='font-family:";
164 			cont[idx++] = appCtxt.get(ZmSetting.COMPOSE_INIT_FONT_FAMILY);
165 			cont[idx++] = "; font-size: ";
166 			cont[idx++] = appCtxt.get(ZmSetting.COMPOSE_INIT_FONT_SIZE);
167 			cont[idx++] = "; color: ";
168 			cont[idx++] = appCtxt.get(ZmSetting.COMPOSE_INIT_FONT_COLOR);
169 			cont[idx++] = ";'>";
170 			cont[idx++] = html;
171 			cont[idx++] = "</div>";
172 		} else {
173 			cont[idx++] = html;
174 		}
175 
176 		return cont.join("");
177 	}
178 
179 	var p_style = "<style type='text/css'>p { margin: 0; }</style>"; // bug 3264
180 
181 	if (insertFontStyle) {
182 		html = this._getFontStyle(html);
183 	}
184 	return [
185 		"<html><head>",
186 		p_style,
187 		"</head><body>",
188 		html,
189 		"</body></html>"
190 	].join("");
191 };

239 ZmAdvancedHtmlEditor.prototype._convertHtml2Text =
240 function() {
241 	var editor = this.getEditor();
242 	var doc = editor && editor.getDoc();
243 
244 	return (doc && doc.body)
245 		? this._convertHtml2TextContinue(doc.body) : "";
246 };

248 ZmAdvancedHtmlEditor.prototype._convertHtml2TextContinue =
249 function(domRoot) {
250 	var text = [];
251 	var idx = 0;
252 	var ctxt = {};
253 
254 	AjxStringUtil._traverse(domRoot, text, idx, AjxStringUtil._NO_LIST, 0, 0, ctxt);
255 
256 	// tinymce always inserts <p> tag which creates unwanted new lines in format
257 	// conversion
258 	if (text.length && text[0] == "\n\n") {
259 		text[0] = "";
260 	}
261 
262 	return text.join("");
263 };

265 ZmAdvancedHtmlEditor.prototype.moveCaretToTop =
266 function() {
267 	var focused = document.activeElement;
268 	if (this._mode == DwtHtmlEditor.TEXT) {
269 		var control = this.getContentField();
270 		if (control.createTextRange) { // IE
271 			var range = control.createTextRange();
272 			range.collapse(true);
273 			range.select();
274 		} else if (control.setSelectionRange) { // FF
275 			control.setSelectionRange(0, 0);
276 		}
277 	} else {
278 		this._moveCaretToTopHtml(true);
279 	}
280     try{
281 	    if (focused) focused.focus();
282     }catch(ex){
283         // do nothing
284     }
285 };

287 ZmAdvancedHtmlEditor.prototype._moveCaretToTopHtml =
288 function(tryOnTimer) {
289 	var editor = this.getEditor();
290 	if (!editor) { return; }
291 
292 	var body = editor.getDoc().body;
293 	var success = false;
294 	if (AjxEnv.isIE) {
295 		if (body) {
296 			body.createTextRange().collapse(true);
297 			success = true;
298 		}
299 	} else {
300 		var selection = editor.selection ? editor.selection.getSel() : "";
301 		if (selection) {
302 			selection.collapse(body,0);
303 			success = true;
304 		}
305 	}
306 	if (!success && tryOnTimer) {
307 		var action = new AjxTimedAction(this, this._moveCaretToTopHtml);
308 		AjxTimedAction.scheduleAction(action, DwtHtmlEditor._INITDELAY + 1);
309 	}
310 };

361 ZmAdvancedHtmlEditor.prototype.initTinyMCEEditor =
362 function(parent, posStyle, content, mode, withAce) {
363 
364 	var params = {
365 		parent: parent,
366 		posStyle: posStyle,
367 		mode: mode,
368 		content: content,
369 		withAce: withAce,
370 		className:"ZmHtmlEditor"
371 	};
372 	this._editorContainer = new ZmEditorContainer(params);
373 	var htmlEl = this._editorContainer.getHtmlElement();
374 
375 	//textarea on which html editor is constructed
376 	var id = this._bodyTextAreaId = this._editorContainer.getHTMLElId() + "_content";
377 	var textEl = document.createElement("textarea");
378 	textEl.setAttribute("id", id);
379 	textEl.setAttribute("name", id);
380 	textEl.className = "DwtHtmlEditorTextArea";
381 	htmlEl.appendChild(textEl);
382 	this._textAreaId = id;
383 
384 	Dwt.setHandler(textEl, DwtEvent.ONFOCUS, AjxCallback.simpleClosure(this.setFocusStatus, this, true, true));
385 	Dwt.setHandler(textEl, DwtEvent.ONBLUR, AjxCallback.simpleClosure(this.setFocusStatus, this, false, true));
386 	this._editorContainer.setFocusMember(textEl);
387 
388 	if (content != null) {
389 		this.setPendingContent(content);
390 	}
391 
392 	if (!window.tinyMCE) {
393 		var callback = new AjxCallback(this, this.initEditorManager, [id, mode, content]);
394 		var data = {
395 			name: "tiny_mce",
396 			path: appContextPath + ZmAdvancedHtmlEditor.TINY_MCE_PATH + "/tiny_mce.js",
397 			extension: ".js",
398 			method: AjxPackage.METHOD_XHR_SYNC,
399 			async: false,
400 			callback: null,
401 			scripts: [],
402 			basePath: appContextPath + ZmAdvancedHtmlEditor.TINY_MCE_PATH
403 		};
404 
405 		AjxPackage.require(data);
406 		this.initEditorManager(id, mode, content);
407 	} else {
408 		this.initEditorManager(id, mode, content);
409 	}
410 };

470 ZmAdvancedHtmlEditor.prototype.initEditorManager =
471 function(id, mode, content) {
472 
473 	var obj = this;
474 
475 	function handleContentLoad(ed) {
476 		obj.onLoadContent(ed);
477 		obj.initDefaultFontSize(ed);
478 	};
479 
480 	function onTinyMCEEditorInit(ed) {
481 		obj.initDefaultFontSize(ed);
482 		tinymce.dom.Event.add(ed.getWin(), 'focus', function(e) {
483 			obj.setFocusStatus(true);
484 		});
485 		tinymce.dom.Event.add(ed.getWin(), 'blur', function(e) {
486 			obj.setFocusStatus(false);
487 		});
488 
489 		var ec = obj.getEditorContainer();
490 		ec.setFocusMember(ed.getWin());
491 
492 		obj._editorInitialized = true;
493 	};
494 
495 	function onEditorKeyPress(ed, e) {
496 		return obj._handleEditorKeyEvent(e, ed);
497 	};
498 
499 	var urlParts = AjxStringUtil.parseURL(location.href);
500 
501 	//important: tinymce doesn't handle url parsing well when loaded from REST URL - override baseURL/baseURI to fix this
502 	tinymce.baseURL = appContextPath + ZmAdvancedHtmlEditor.TINY_MCE_PATH + "/";
503 
504 	if (tinymce.EditorManager) {
505 		tinymce.EditorManager.baseURI = new tinymce.util.URI(urlParts.protocol + "://" + urlParts.authority + tinymce.baseURL);
506 	}
507 
508 	if (tinymce.dom) {
509 		tinymce.DOM = new tinymce.dom.DOMUtils(document, {process_html : 0});
510 	}
511 
512 	if (tinymce.dom && tinymce.dom.Event) {
513 		tinymce.dom.Event.domLoaded = true;
514 	}
515 
516 	var locale = appCtxt.get(ZmSetting.LOCALE_NAME);
517 	var editorCSS = appContextPath + "/css/editor_ui.css?v=" + window.cacheKillerVersion + "&skin=" + appCurrentSkin + "&locale=" + locale;
518 
519     var fonts = [];
520 	var KEYS = [ "fontFamilyIntl", "fontFamilyBase" ];
521 	var i, j, key, value, name;
522 	for (j = 0; j < KEYS.length; j++) {
523 		for (i = 1; value = AjxMsg[KEYS[j]+i+".css"]; i++) {
524 			if (value.match(/^#+$/)) break;
525 			value = value.replace(/,\s/g,",");
526 			name = AjxMsg[KEYS[j]+i+".display"];
527 			fonts.push(name+"="+value);
528 		}
529 	}
530 
531 	tinyMCE.init({
532 		// General options
533 		mode :  (mode == DwtHtmlEditor.HTML)? "exact" : "none",
534 		elements:  id,
535 		plugins : "table,ztable,inlinepopups,zcontextmenu,fullscreen,zbreakquote",
536 		theme : "advanced",
537 		theme_advanced_buttons1 : "fontselect,fontsizeselect,formatselect,justifyleft,justifycenter,justifyright,justifyfull,separator,bullist,numlist,outdent,indent,separator,bold,italic,underline,separator,forecolor,backcolor,separator,link,ztablecontrols,fullscreen",
538 		theme_advanced_buttons2: "",
539 		theme_advanced_buttons3: "",
540 		theme_advanced_buttons4: "",
541 		theme_advanced_toolbar_location : "top",
542 		theme_advanced_toolbar_align : "left",
543 		theme_advanced_resizing : true,
544         theme_advanced_fonts : fonts.join(";"),
545 		convert_urls : false,
546 		verify_html : false,
547 		gecko_spellcheck : true,
548 		force_br_newlines : true,
549 		forced_root_block : '',
550 		force_p_newlines : false,
551 		content_css : false,
552 		editor_css: editorCSS,
553 		inline_styles: false,
554 		setup : function(ed) {
555 			ed.onLoadContent.add(handleContentLoad);
556 			ed.onInit.add(onTinyMCEEditorInit);
557 			ed.onKeyPress.add(onEditorKeyPress);
558 
559 		}
560 	});
561 
562 	this._editor = this.getEditor();
563 	this._iFrameId = this._bodyTextAreaId + "_ifr";
564 };

566 ZmAdvancedHtmlEditor.prototype.setMode =
567 function(mode, convert, convertor) {
568 
569     this.discardMisspelledWords();
570 
571 	if (mode == this._mode || (mode != DwtHtmlEditor.HTML && mode != DwtHtmlEditor.TEXT)) {	return;	}
572 
573 	this._mode = mode;
574 	var editor = this.getEditor();
575 	if (mode == DwtHtmlEditor.HTML) {
576 		var textArea = this.getContentField();
577 		var content = convert ? AjxStringUtil.convertToHtml(textArea.value, true) : textArea.value;
578 		if (editor && editor.getDoc()) {
579 			var doc = editor.getDoc();
580 			doc.body.innerHTML = content;
581 			this._pendingContent = content;
582 			//important: tinymce expects html markup in textarea so it might treat email
583 			//address in <user1@testdomain.com> as tag
584 			textArea.value = "";
585 			this._editorContainer.setFocusMember(editor.getWin());
586 		} else {
587 			this._pendingContent = content;
588 		}
589 		tinyMCE.execCommand('mceToggleEditor', false, this._bodyTextAreaId);
590 	} else {
591 		var textArea = this.getContentField();
592 		var doc = editor.getDoc();
593 		var textContent = convert ? this._convertHtml2Text(convertor) : doc.innerHTML;
594 
595 		tinyMCE.execCommand('mceToggleEditor', false, this._bodyTextAreaId);
596 		textArea.value = textContent;
597 		this._editorContainer.setFocusMember(textArea);
598 	}
599 };

659 ZmAdvancedHtmlEditor.prototype.insertDefaultCSS =
660 function(doc, selectorText, ruleText) {
661 	var sheet, styleElement;
662 	if (doc.createStyleSheet) {
663 		sheet = doc.createStyleSheet();
664 	} else {
665 		styleElement = doc.createElement("style");
666 		doc.getElementsByTagName("head")[0].appendChild(styleElement);
667 		sheet = styleElement.styleSheet ? styleElement.styleSheet : styleElement.sheet;
668 	}
669 
670 	if (!sheet && styleElement) {
671 		//remove braces
672 		ruleText = ruleText.replace(/^\{?([^\}])/, "$1");
673 		styleElement.innerHTML = selectorText + ruleText;
674 	} else if (sheet.addRule) {
675 		//remove braces
676 		ruleText = ruleText.replace(/^\{?([^\}])/, "$1");
677 		DBG.println("ruleText:" + ruleText + ",selector:" + selectorText);
678 		sheet.addRule(selectorText, ruleText);
679 	} else if (sheet.insertRule) {
680 		//need braces

1194 ZmAdvancedHtmlEditor.prototype.highlightMisspelledWords =
1195 function(words, keepModeDiv) {
1196 	this.discardMisspelledWords(keepModeDiv);
1197 
1198 	var word, style, doc, body, self = this,
1199 		spanIds     = {},
1200 		wordIds     = {},
1201 		regexp      = [ "([^A-Za-z0-9']|^)(" ],
1202 		suggestions = {};
1203 
1204 	// preparations: initialize some variables that we then save in
1205 	// this._spellCheck (the current spell checker context).
1206 	for (var i = 0; i < words.length; ++i) {
1207 		word = words[i].word;
1208 		if (!suggestions[word]) {
1209 			i && regexp.push("|");
1210 			regexp.push(word);
1211 			var a = words[i].suggestions.split(/\s*,\s*/);
1212 			if (!a[a.length-1])
1213 				a.pop();
1214 			suggestions[word] = a;
1215 			if (suggestions[word].length > 5)
1216 				suggestions[word].length = 5;
1217 		}
1218 	}
1219 	regexp.push(")([^A-Za-z0-9']|$)");
1220 	regexp = new RegExp(regexp.join(""), "gm");
1221 
1222 	function hiliteWords(text, textWhiteSpace) {
1223 		text = textWhiteSpace
1224 			? AjxStringUtil.convertToHtml(text)
1225 			: AjxStringUtil.htmlEncode(text);
1226 
1227 		var m;
1228 
1229 		regexp.lastIndex = 0;
1230 		while (m = regexp.exec(text)) {
1231 			var str = m[0];
1232 			var prefix = m[1];
1233 			var word = m[2];
1234 			var suffix = m[3];
1235 
1236 			var id = Dwt.getNextId();
1237 			spanIds[id] = word;
1238 			if (!wordIds[word])
1239 				wordIds[word] = [];
1240 			wordIds[word].push(id);
1241 
1242 			var repl = [
1243 				prefix,
1244 				'<span word="',
1245 				word, '" id="', id, '" class="ZM-SPELLCHECK-MISSPELLED">',
1246 				word, '</span>',
1247 				suffix
1248 				].join("");
1249 			text = [
1250 				text.substr(0, m.index),
1251 				repl,
1252 				text.substr(m.index + str.length)
1253 			].join("");
1254 
1255 			// All this crap necessary because the suffix
1256 			// must be taken into account at the next
1257 			// match and JS regexps don't have look-ahead
1258 			// constructs (except \b, which sucks).  Oh well.
1259 			regexp.lastIndex = m.index + repl.length - suffix.length;
1260 		}
1261 		return text;
1262 	};
1263 
1264 	var doc;
1265 
1266 	// having the data, this function will parse the DOM and replace
1267 	// occurrences of the misspelled words with <span
1268 	// class="ZM-SPELLCHECK-MISSPELLED">word</span>
1269 	rec = function(node) {
1270 		switch (node.nodeType) {
1271 			case 1: /* ELEMENT */
1272 				for (var i = node.firstChild; i; i = rec(i));
1273 				node = node.nextSibling;
1274 				break;
1275 			case 3: /* TEXT */
1276 				if (!/[^\s\xA0]/.test(node.data)) {
1277 					node = node.nextSibling;
1278 					break;
1279 				}
1280 				// for correct handling of whitespace we should
1281 				// not mess ourselves with leading/trailing
1282 				// whitespace, thus we save it in 2 text nodes.
1283 				var a = null, b = null;
1284 
1285 				var result = /^[\s\xA0]+/.exec(node.data);
1286 				if (result) {
1287 					// a will contain the leading space
1288 					a = node;
1289 					node = node.splitText(result[0].length);
1290 				}
1291 				result = /[\s\xA0]+$/.exec(node.data);
1292 				if (result) {
1293 					// and b will contain the trailing space
1294 					b = node.splitText(node.data.length - result[0].length);
1295 				}
1296 
1297 				var text = hiliteWords(node.data, false);
1298 				text = text.replace(/^ +/, " ").replace(/ +$/, " ");
1299 				var div = doc.createElement("div");
1300 				div.innerHTML = text;
1301 
1302 				// restore whitespace now
1303 				if (a) {
1304 					div.insertBefore(a, div.firstChild);
1305 				}
1306 				if (b) {
1307 					div.appendChild(b);
1308 				}
1309 
1310 				var p = node.parentNode;
1311 				while (div.firstChild) {
1312 					p.insertBefore(div.firstChild, node);
1313 				}
1314 				div = node.nextSibling;
1315 				p.removeChild(node);
1316 				node = div;
1317 				break;
1318 			default :
1319 				node = node.nextSibling;
1320 		}
1321 		return node;
1322 	};
1323 
1324 	if (this._mode == DwtHtmlEditor.HTML) {
1325 		// HTML mode; See the "else" branch for the TEXT mode--code differs
1326 		// quite a lot.  We should probably implement separate functions as
1327 		// this already becomes long.
1328 
1329 		doc = this._getIframeDoc();
1330 		body = doc.body;
1331 
1332 		// load the spell check styles, if not already there.
1333 		this._loadExternalStyle("/css/spellcheck.css");
1334 
1335 		body.style.display = "none";	// seems to have a good impact on speed,
1336 										// since we may modify a lot of the DOM
1337 		if (!AjxEnv.isIE) {
1338 			body.normalize();
1339 		} else {
1340 			body.innerHTML = body.innerHTML;
1341 		}
1342 		rec(body);
1343 		if (!AjxEnv.isIE) {
1344 			body.normalize();
1345 		} else {
1346 			body.innerHTML = body.innerHTML;
1347 		}
1348 		body.style.display = ""; // redisplay the body
1349 
1350 		var ed = this.getEditor();
1351 		ed.onContextMenu.add(this._handleEditorEvent, this);
1352 		ed.onMouseUp.add(this._handleEditorEvent, this);
1353 
1354 		//this._registerEditorEventHandler(doc, "contextmenu");
1355 	}
1356 	else { // TEXT mode
1357 		var textArea = document.getElementById(this._textAreaId);
1358 		var scrollTop = textArea.scrollTop;
1359 		var size = Dwt.getSize(textArea);
1360 		textArea.style.display = "none";
1361 		var div = document.createElement("div");
1362 		div.className = "TextSpellChecker";
1363 		this._spellCheckDivId = div.id = Dwt.getNextId();
1364 		div.style.overflow = "auto";
1365 		if (!AjxEnv.isIE) {
1366 			// FIXME: we substract borders/padding here.  this sucks.
1367 			size.x -= 4;
1368 			size.y -= 6;
1369 		}
1370 		div.style.width = size.x + "px";
1371 		div.style.height = size.y + "px";
1372 
1373 		div.innerHTML = AjxStringUtil.convertToHtml(this.getContent());
1374 		doc = document;
1375 		rec(div);
1376 
1377 		textArea.parentNode.insertBefore(div, textArea);
1378 		div.scrollTop = scrollTop;
1379 		div.oncontextmenu = div.onclick
1380 			= function(ev) { self._handleSpellCheckerEvents(ev || window.event); };
1381 	}
1382 
1383 	this._spellCheckShowModeDiv();
1384 
1385 	// save the spell checker context
1386 	this._spellCheck = {
1387 		suggestions: suggestions,
1388 		spanIds: spanIds,
1389 		wordIds: wordIds
1390 	};
1391 };
