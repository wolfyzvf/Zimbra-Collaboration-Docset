 70 ZmApptRecurDialog.prototype.initialize = 
 71 function(startDate, endDate, repeatType, appt) {
 72 
 73     this._startDate = new Date(startDate.getTime());
 74 	this._endDate = new Date(endDate.getTime());
 75     this._origRefDate = startDate;
 76     // based on repeat type, setup the repeat type values
 77 	var repeatType = repeatType || ZmRecurrence.DAILY;
 78 	this._repeatSelect.setSelectedValue(repeatType);
 79 	this._setRepeatSection(repeatType);
 80 
 81 	// dont bother initializing if user is still mucking around
 82 	if (this._saveState)
 83 		return;
 84 
 85 	var startDay = this._startDate.getDay();
 86 	var startDate = this._startDate.getDate();
 87 	var startMonth = this._startDate.getMonth();
 88 
 89 	// reset time based fields
 90 	this._endByField.setValue(AjxDateUtil.simpleComputeDateStr(this._startDate));
 91     this._weeklySelectButton._selected = startDay;
 92     this._weeklySelectButton.setDisplayState(DwtControl.SELECTED);
 93 
 94     var formatter = new AjxMessageFormat(ZmMsg.recurWeeklyEveryWeekday);
 95     var dayFormatter = formatter.getFormatsByArgumentIndex()[0];
 96     this._weeklySelectButton.setText(dayFormatter.format(this._origRefDate));
 97 
 98     this._weeklyCheckboxes[startDay].checked = true;
 99 	this._monthlyDayField.setValue(startDate);
100 	this._monthlyWeekdaySelect.setSelected(startDay);
101 	this._yearlyDayField.setValue(startDate);
102 	this._yearlyMonthSelect.setSelected(startMonth);
103 	this._yearlyWeekdaySelect.setSelected(startDay);
104 	this._yearlyMonthSelectEx.setSelected(startMonth);
105 
106 	// if given appt object, means user is editing existing appointment's recur rules
107 	if (appt) {
108 		this._populateForEdit(appt);
109 	}
110 };

165 /**
166  * Sets custom weekly values.
167  * 
168  * @param	{ZmAppt}	appt		the appointment
169  */
170 ZmApptRecurDialog.prototype.setCustomWeeklyValues =
171 function(appt) {
172 	var recur = appt._recurrence;
173 	recur.repeatWeeklyDays = []
174 	recur.repeatCustom = "1";
175     recur._startDate = new Date(this._origRefDate);
176 	var value = this._getRadioOptionValue(this._weeklyRadioName);
177     var currentDay = recur._startDate.getDay();
178 	if (value == "1") {
179 		recur.repeatCustomCount = 1;
180         var startDay = this._weeklySelectButton._selected;
181         switch(startDay){
182             case 7: //Separator
183                     break;
184             case 8: //Mon, wed, Fri
185                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.MONDAY]);
186                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.WEDNESDAY]);
187                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.FRIDAY]);
188                     startDay = AjxDateUtil.MONDAY;
189                     while(startDay < currentDay){
190                         startDay += 2;
191                     }
192                     break;
193             case 9: //Tue, Thu
194                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.TUESDAY]);
195                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.THURSDAY]);
196                     startDay = AjxDateUtil.TUESDAY;
197                     while(startDay < currentDay){
198                         startDay += 2;
199                     }
200                     break;
201             case 10: //Sat, Sunday
202                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.SATURDAY]);
203                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.SUNDAY]);
204                     startDay = currentDay == AjxDateUtil.SUNDAY? currentDay:AjxDateUtil.SATURDAY;
205                     break;
206             default:
207                     recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[this._weeklySelectButton._selected/*getValue()*/]);
208                     break;
209         }
210         recur._startDate = AjxDateUtil.getDateForNextDay(new Date(this._origRefDate),startDay);
211         //recur._endDate = recur._startDate;
212     } else {
213 		recur.repeatCustomCount = Number(this._weeklyField.getValue());
214         var selectedDays = [];
215         for (var i = 0; i < this._weeklyCheckboxes.length; i++) {
216 			if (this._weeklyCheckboxes[i].checked){
217                 selectedDays.push(i);
218                 recur.repeatWeeklyDays.push(ZmCalItem.SERVER_WEEK_DAYS[i]);
219             }
220         }
221         var startDay = currentDay;
222         for(var i =0; i < selectedDays.length;i++){
223             var startDay = selectedDays[i];
224             if(startDay >= currentDay) { //In past
225                 break;
226             }
227         }
228         recur._startDate = AjxDateUtil.getDateForNextDay(new Date(this._origRefDate),startDay);
229     }
230 };

235  * @param	{ZmAppt}	appt		the appointment

237 ZmApptRecurDialog.prototype.setCustomMonthlyValues =
238 function(appt) {
239 	var recur = appt._recurrence;
240 	recur.repeatCustom = "1";
241    	var value = this._getRadioOptionValue(this._monthlyRadioName);
242     recur._startDate = new Date(this._origRefDate);
243 	if (value == "1") {
244 		recur.repeatCustomType = "S";
245 		recur.repeatCustomCount = this._monthlyMonthField.getValue();
246 		recur.repeatMonthlyDayList = [this._monthlyDayField.getValue()];
247         recur.repeatCustomMonthDay = this._monthlyDayField.getValue();
248         recur._startDate.setDate(recur.repeatCustomMonthDay);
249         var today = new Date(this._origRefDate); //Reference date...
250         var diff = (today - recur._startDate);
251         if(diff >= AjxDateUtil.MSEC_PER_DAY || today.getDate() > recur._startDate.getDate()){ // was in the past, so let's use the next date
252             recur._startDate.setMonth(recur._startDate.getMonth()+1);
253         }
254 
255     } else {
256 		recur.repeatCustomType = "O";
257 		recur.repeatCustomCount = this._monthlyMonthFieldEx.getValue();
258 		recur.repeatBySetPos = this._monthlyDaySelect.getValue();
259 		recur.repeatCustomDayOfWeek = ZmCalItem.SERVER_WEEK_DAYS[this._monthlyWeekdaySelect.getValue()];
260         recur.repeatCustomDays = this.getWeekdaySelectValue(this._monthlyWeekdaySelect);
261 
262         if(recur.repeatBySetPos==-1){ // Last day
263             var lastDate = new Date(this._origRefDate);
264             lastDate.setDate(AjxDateUtil.daysInMonth(lastDate.getFullYear(),lastDate.getMonth())); //Date is now last date of this month
265             var lastDayDate = this.getPossibleStartDate(this._monthlyWeekdaySelect.getValue(), lastDate, recur.repeatBySetPos);
266 
267             //Check if it is already paased
268             var today = new Date(this._origRefDate);
269             var diff = (today - lastDayDate);
270             var isInPast = today.getTime() > lastDayDate.getTime();
271             if(diff >= AjxDateUtil.MSEC_PER_DAY || isInPast ){ //In the past
272                 // Go for next month
273                 lastDate.setMonth(lastDate.getMonth()+1);
274                 recur._startDate = this.getPossibleStartDate(this._monthlyWeekdaySelect.getValue(), lastDate, recur.repeatBySetPos);                              
275             }else{
276                  recur._startDate = lastDayDate;
277             }
278         }else{
279             var first = new Date(this._origRefDate);
280             first.setDate(1);  
281             recur._startDate = this.getPossibleStartDate(this._monthlyWeekdaySelect.getValue(), first, recur.repeatBySetPos); //AjxDateUtil.getDateForNextDay(first,this.getFirstWeekDayOffset(this._monthlyWeekdaySelect),recur.repeatBySetPos);
282              //Check if it is already paased
283             var today = new Date(this._origRefDate);
284             var diff = (today - recur._startDate);
285             var isInPast = today.getTime() > recur._startDate.getTime();
286             if(diff >= AjxDateUtil.MSEC_PER_DAY || isInPast){ //In the past
287                 // Go for next month, find the date as per rule
288                 first.setMonth(first.getMonth() + 1);//Next month
289                 recur._startDate = this.getPossibleStartDate(this._monthlyWeekdaySelect.getValue(), first, recur.repeatBySetPos);
290             }
291         }
292     }
293 };

295 /**
296  * Sets custom yearly values.
297  * 
298  * @param	{ZmAppt}	appt		the appointment
299  */
300 ZmApptRecurDialog.prototype.setCustomYearlyValues =
301 function(appt) {
302 	appt._recurrence.repeatCustom = "1";
303     var recur = appt._recurrence;
304     recur._startDate = new Date(this._origRefDate);
305 	var value = this._getRadioOptionValue(this._yearlyRadioName);
306 
307 	if (value == "1") {
308 		appt._recurrence.repeatCustomType = "S";
309 		appt._recurrence.repeatCustomMonthDay = this._yearlyDayField.getValue();
310 		appt._recurrence.repeatYearlyMonthsList = this._yearlyMonthSelect.getValue() + 1;
311         //Create date out of it
312         var d  = new Date(this._origRefDate);
313         d.setDate(appt._recurrence.repeatCustomMonthDay);
314         d.setMonth(this._yearlyMonthSelect.getValue());
315         //Try to judge, if this date is in future
316         var today = new Date(this._origRefDate);
317         var diff = (today - d);
318         var isInPast = today.getTime() > d.getTime();
319         if( diff >= AjxDateUtil.MSEC_PER_DAY || isInPast){ //In the past
320             d.setFullYear(d.getFullYear()+1);
321         }
322         appt._recurrence._startDate = d;
323         appt._recurrence._endDate = d;
324     } else {
325 		appt._recurrence.repeatCustomType = "O";
326 		appt._recurrence.repeatBySetPos = this._yearlyDaySelect.getValue();
327 		appt._recurrence.repeatCustomDayOfWeek = ZmCalItem.SERVER_WEEK_DAYS[this._yearlyWeekdaySelect.getValue()];
328         appt._recurrence.repeatCustomDays = this.getWeekdaySelectValue(this._yearlyWeekdaySelect);
329 
330         appt._recurrence.repeatYearlyMonthsList = this._yearlyMonthSelectEx.getValue() + 1;
331         var d = new Date(this._origRefDate);
332         d.setMonth(this._yearlyMonthSelectEx.getValue());
333         //Check if date is in past
334         if(appt._recurrence.repeatBySetPos < 0){ // we want last day
335             d.setDate(AjxDateUtil.daysInMonth(d.getFullYear(),d.getMonth()));
336         }else{
337             d.setDate(1);
338         }
339         var dt = this.getPossibleStartDate(this._yearlyWeekdaySelect.getValue(), d, appt._recurrence.repeatBySetPos);
340 
341         var today = new Date(this._origRefDate);
342         var diff = (today -dt);
343         var isInPast = today.getTime() > dt.getTime();
344         if(diff >= AjxDateUtil.MSEC_PER_DAY || isInPast){ // In the past
345             d.setFullYear(d.getFullYear()+1);
346             if(appt._recurrence.repeatBySetPos < 0){ // we want last day
347                 d.setDate(AjxDateUtil.daysInMonth(d.getFullYear(),d.getMonth()));
348             }else{
349                 d.setDate(1);
350             }
351         }
352         appt._recurrence._startDate = this.getPossibleStartDate(this._yearlyWeekdaySelect.getValue(), d, appt._recurrence.repeatBySetPos);
353         appt._recurrence._endDate = appt._recurrence._startDate;
354     }

368 ZmApptRecurDialog.prototype.isValid = 
369 function() {
370 	var valid = true;
371 
372 	// ONLY for the selected options, check if their fields are valid
373 	var repeatValue = this._repeatSelect.getValue();
374 
375 	if (repeatValue == ZmRecurrence.DAILY) {
376 		if (this._dailyFieldRadio.checked)
377 			valid = this._dailyField.isValid();
378 		if (!valid)
379 			this._dailyField.blur();
380 	} else if (repeatValue == ZmRecurrence.WEEKLY) {
381 		if (this._weeklyFieldRadio.checked) {
382 			valid = this._weeklyField.isValid();
383 			if (valid) {
384 				valid = false;
385 				for (var i=0; i<this._weeklyCheckboxes.length; i++) {
386 					if (this._weeklyCheckboxes[i].checked) {
387 						valid = true;
388 						break;
389 					}
390 				}
391 			}
392 			// weekly section is special - force a focus if valid to clear out error
393 			this._weeklyField.focus();
394 			this._weeklyField.blur();
395 		}
396 	} else if (repeatValue == ZmRecurrence.MONTHLY) {
397 		if (this._monthlyDefaultRadio.checked) {
398 			valid = this._monthlyMonthField.isValid() && this._monthlyDayField.isValid();
399 			if (!valid) {
400 				this._monthlyMonthField.blur();
401 				this._monthlyDayField.blur();
402 			}
403 		} else {
404 			valid = this._monthlyMonthFieldEx.isValid();
405 			if (!valid)
406 				this._monthlyMonthFieldEx.blur();
407 		}
408 	} else if (repeatValue == ZmRecurrence.YEARLY) {
409 		if (this._yearlyDefaultRadio.checked)
410 			valid = this._yearlyDayField.isValid();
411 		if (!valid)
412 			this._yearlyDayField.blur();
413 	}
414 
415 	// check end section
416 	if (valid) {
417 		if (this._endAfterRadio.checked) {
418 			valid = this._endIntervalField.isValid();
419 			if (!valid)
420 				this._endIntervalField.blur();
421 		} else if (this._endByRadio.checked) {
422 			valid = this._endByField.isValid();
423 			if (!valid)
424 				this._endByField.blur();
425 		}
426 	}
427 
428 	return valid;
429 };

552 ZmApptRecurDialog.prototype._createRepeatSections = 
553 function() {
554 	var sectionDiv = document.getElementById(this._repeatSectionId);
555 	if (sectionDiv) {
556 		var div = document.createElement("div");

1608 function(weekDayVal, lastDate, repeatBySetPos) {
1609 
1610     //weekday select might contain normal weekdays, day, weekend values also
1611     if(ZmCalItem.SERVER_WEEK_DAYS[weekDayVal]) {
1612         return AjxDateUtil.getDateForThisDay(lastDate, weekDayVal, repeatBySetPos); //Last day of next month/year
1613     }else if(weekDayVal == ZmRecurrence.RECURRENCE_DAY) {
1614         var dayOffset = ((repeatBySetPos==-1)? 0 : (repeatBySetPos-1));
1615         lastDate.setDate(lastDate.getDate() + dayOffset);
1616         return lastDate;
1617     }else if(weekDayVal == ZmRecurrence.RECURRENCE_WEEKDAY) {
1618         return AjxDateUtil.getDateForThisWorkWeekDay(lastDate, repeatBySetPos);
1619     }else if(weekDayVal == ZmRecurrence.RECURRENCE_WEEKEND) {
1620         var lastSunday = AjxDateUtil.getDateForThisDay(lastDate, AjxDateUtil.SUNDAY, repeatBySetPos);
1621         var lastSaturday = AjxDateUtil.getDateForThisDay(lastDate, AjxDateUtil.SATURDAY, repeatBySetPos);
1622         //nearest possible weekend
1623         if(repeatBySetPos < 0) {
1624             return (lastSaturday.getTime() > lastSunday.getTime()) ? lastSaturday : lastSunday;
1625         }else {
1626             return (lastSaturday.getTime() > lastSunday.getTime()) ? lastSunday : lastSaturday;                        
1627         }
