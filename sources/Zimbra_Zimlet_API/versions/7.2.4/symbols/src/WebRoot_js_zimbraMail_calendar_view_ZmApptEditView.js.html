 16 /**
 17  * Creates a new calendar item edit view.
 18  * @constructor
 19  * @class
 20  * This is the main screen for creating/editing an appointment. It provides
 21  * inputs for the various appointment details.
 22  *
 23  * @author Parag Shah
 24  *
 25  * @param {DwtControl}	parent			some container
 26  * @param {Hash}	attendees			attendees/locations/equipment
 27  * @param {Object}	dateInfo			a hash of date info
 28  * @param {ZmController}	controller		the compose controller for this view
 29  * 
 30  * @extends		ZmCalItemEditView
 31  * 
 32  * @private

 34 ZmApptEditView = function(parent, attendees, controller, dateInfo) {
 35 
 36 	ZmCalItemEditView.call(this, parent, attendees, controller, dateInfo, null, "ZmApptEditView");
 37 
 38 	// cache so we dont keep calling appCtxt
 39 	this.GROUP_CALENDAR_ENABLED = appCtxt.get(ZmSetting.GROUP_CALENDAR_ENABLED);
 40 
 41 	this._attTypes = [];
 42 	if (this.GROUP_CALENDAR_ENABLED) {
 43 		this._attTypes.push(ZmCalBaseItem.PERSON);
 44 	}
 45 	this._attTypes.push(ZmCalBaseItem.LOCATION);
 46 	if (appCtxt.get(ZmSetting.GAL_ENABLED) && this.GROUP_CALENDAR_ENABLED) {
 47 		this._attTypes.push(ZmCalBaseItem.EQUIPMENT);
 48 	}
 49     this._locationTextMap = {};
 50     this._attendeePicker = {};
 51     this._pickerButton = {};
 52 
 53 	this._useAcAddrBubbles = appCtxt.get(ZmSetting.USE_ADDR_BUBBLES);
 54 
 55     //used to preserve original attendees while forwarding appt
 56     this._fwdApptOrigAttendees = [];
 57     this._attendeesHashMap = {};
 58 
 59     // Store Appt form values.
 60     this._apptFormValue = {};
 61     this._showAsValueChanged = false;
 62 };

221 ZmApptEditView.prototype.isValid =
222 function() {
223 	var errorMsg = [];
224 
225 	// check for required subject
226 	var subj = AjxStringUtil.trim(this._subjectField.getValue());
227 
228     //bug: 49990 subject can be empty while proposing new time
229 	if ((subj && subj.length) || this._isProposeTime) {
230 		var allDay = this._allDayCheckbox.checked;
231 		if (!ZmTimeInput.validStartEnd(this._startDateField, this._endDateField, (allDay ? null : this._startTimeSelect), (allDay ? null : this._endTimeSelect))) {
232 				errorMsg.push(ZmMsg.errorInvalidDates);
233 		}
234 
235 	} else {
236 		errorMsg.push(ZmMsg.errorMissingSubject);
237 	}
238     if (this._reminderSelectInput) {
239         var reminderString = this._reminderSelectInput.getValue();
240         var reminderInfo = ZmCalendarApp.parseReminderString(reminderString);
241         if (reminderInfo.reminderValue > ZmApptEditView.REMINDER_MAX_VALUE[reminderInfo.reminderUnits]) {
242             errorMsg.push(ZmMsg.errorInvalidReminderValue);
243         }
244     }
245 	if (errorMsg.length > 0) {
246 		throw errorMsg.join("<br>");
247 	}
248 
249 	return true;
250 };

419 ZmApptEditView.prototype._populateForSave =
420 function(calItem) {
421 
422     if (!calItem) {
423         return null;
424     }
425 
426     ZmCalItemEditView.prototype._populateForSave.call(this, calItem);
427 
428     if(this.isOrganizer() && this.isKeyInfoChanged()) this.resetParticipantStatus();
429 
430     //Handle Persona's
431     var identity = this.getIdentity();
432     if(identity){
433        calItem.identity = identity; 
434        calItem.sentBy = (identity && identity.getField(ZmIdentity.SEND_FROM_ADDRESS));
435     }
436 
437 	calItem.freeBusy = this._showAsSelect.getValue();
438 	calItem.privacy = this._privateCheckbox.checked ? ZmApptEditView.PRIVACY_OPTION_PRIVATE : ZmApptEditView.PRIVACY_OPTION_PUBLIC;
439 
440 	// set the start date by aggregating start date/time fields
441 	var startDate = AjxDateUtil.simpleParseDateStr(this._startDateField.value);
442 	var endDate = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
443 	if (this._allDayCheckbox.checked) {
444 		calItem.setAllDayEvent(true);
445         if(AjxDateUtil.isDayShifted(startDate)) {
446             AjxDateUtil.rollToNextDay(startDate);
447             AjxDateUtil.rollToNextDay(endDate);
448         }
449 	} else {
450 		calItem.setAllDayEvent(false);
451 		startDate = this._startTimeSelect.getValue(startDate);
452 		endDate = this._endTimeSelect.getValue(endDate);
453 	}
454 	calItem.setStartDate(startDate, true);
455 	calItem.setEndDate(endDate, true);
456 	if (Dwt.getVisibility(this._tzoneSelectStartElement)) {
457         calItem.timezone = this._tzoneSelectStart.getValue();
458         if (Dwt.getVisibility(this._tzoneSelectEndElement)) {
459             calItem.setEndTimezone(this._tzoneSelectEnd.getValue());
460         }else {
461             calItem.setEndTimezone(this._tzoneSelectStart.getValue());
462         }
463     }
464 
465     // set attendees
466     for (var t = 0; t < this._attTypes.length; t++) {
467         var type = this._attTypes[t];
468         calItem.setAttendees(this._attendees[type].getArray(), type);
469     }
470 
471     var calLoc = AjxStringUtil.trim(this._attInputField[ZmCalBaseItem.LOCATION].getValue());
472      //bug 44858, trimming ';' so that ;; does not appears in outlook, 
473 	calItem.location = AjxStringUtil.trim(calLoc, false, ';');
474 
475 	// set any recurrence rules LAST
476 	this._getRecurrence(calItem);
477 
478     calItem.isForward = this._isForward;
479     calItem.isProposeTime = this._isProposeTime;
480 
481     if(this._isForward)  {
482         var addrs = this._collectForwardAddrs();
483         var a = {};
484         if (addrs[AjxEmailAddress.TO] && addrs[AjxEmailAddress.TO].good) {
485             a[AjxEmailAddress.TO] = addrs[AjxEmailAddress.TO].good.getArray();
486         }        
487         calItem.setForwardAddress(a[AjxEmailAddress.TO]);
488     }
489 
490 	return calItem;
491 };

517 ZmApptEditView.prototype._populateForEdit =
518 function(calItem, mode) {
519 
520 	ZmCalItemEditView.prototype._populateForEdit.call(this, calItem, mode);
521 
522     var enableTimeSelection = !this._isForward;
523     var enableApptDetails = !this._isForward && !this._isProposeTime;
524 
525 	this._showAsSelect.setSelectedValue(calItem.freeBusy);
526     this._showAsSelect.setEnabled(enableApptDetails);
527 
528 	// reset the date/time values based on current time
529 	var sd = new Date(calItem.startDate.getTime());
530 	var ed = new Date(calItem.endDate.getTime());
531 
532     var isNew = (mode == ZmCalItem.MODE_NEW || mode == ZmCalItem.MODE_NEW_FROM_QUICKADD);
533 	var isAllDayAppt = calItem.isAllDayEvent();
534 	if (isAllDayAppt) {
535 		this._allDayCheckbox.checked = true;
536 		this._showTimeFields(false);
537         this.updateShowAsField(true);
538         this._showAsSelect.setSelectedValue(calItem.freeBusy);
539         this._showAsSelect.setEnabled(enableApptDetails);
540 
541 		// set time anyway to current time and default duration (in case user changes mind)
542 		var now = AjxDateUtil.roundTimeMins(new Date(), 30);
543 		this._startTimeSelect.set(now);
544 
545 		now.setTime(now.getTime() + ZmCalViewController.DEFAULT_APPOINTMENT_DURATION);
546 		this._endTimeSelect.set(now);
547 
548 		// bug 9969: HACK - remove the all day durtion for display
549 		if (!isNew && !calItem.draftUpdated && ed.getHours() == 0 && ed.getMinutes() == 0 && ed.getSeconds() == 0 && sd.getTime() != ed.getTime()) {
550 			ed.setHours(-12);
551 		}
552 	} else {
553 		this._showTimeFields(true);
554 		this._startTimeSelect.set(calItem.startDate);
555 		this._endTimeSelect.set(calItem.endDate);
556 	}
557 	this._startDateField.value = AjxDateUtil.simpleComputeDateStr(sd);
558 	this._endDateField.value = AjxDateUtil.simpleComputeDateStr(ed);
559 
560 	this._initTzSelect();
561 	this._resetTimezoneSelect(calItem, isAllDayAppt);
562 
563     //need to capture initial time set while composing/editing appt
564     ZmApptViewHelper.getDateInfo(this, this._dateInfo);
565 
566     this._startTimeSelect.setEnabled(enableTimeSelection);
567     this._endTimeSelect.setEnabled(enableTimeSelection);
568     this._startDateButton.setEnabled(enableTimeSelection);
569     this._endDateButton.setEnabled(enableTimeSelection);
570 
571     this._fwdApptOrigAttendees = [];
572 
573     //editing an appt should exclude the original appt time for FB calculation
574     this._fbExcludeInfo = {};
575 
576     var showScheduleView = false;
577 	// attendees
578 	var attendees = calItem.getAttendees(ZmCalBaseItem.PERSON);
579 	if (attendees && attendees.length) {
580 		if (this.GROUP_CALENDAR_ENABLED) {
581 			var people = calItem.getAttendees(ZmCalBaseItem.PERSON);
582 			var reqAttendees = ZmApptViewHelper.filterAttendeesByRole(people, ZmCalItem.ROLE_REQUIRED);
