134 ZmAttachmentsView.prototype._createFilterTrees = function(){
135 
136     this._tree = {};
137 
138     var treeListener = new AjxListener(this._controller, this._controller._filterTreeListener);
139 
140     this._createTree(ZmAttachmentsController.CATEGORY_BYTYPE, treeListener);
141     this._createTree(ZmAttachmentsController.CATEGORY_BYTIME, treeListener);
142     this._createOverviewTree(ZmAttachmentsController.CATEGORY_BYFOLDER, ZmOrganizer.FOLDER, treeListener);
143     this._createOverviewTree(ZmAttachmentsController.CATEGORY_BYTAG, ZmOrganizer.TAG, treeListener);
144 
145 };

147 ZmAttachmentsView.prototype._createOverviewTree = function(type, treeType, selectionListener){
148 
149     var treeIds = [];
150     treeIds.push(treeType);
151 
152     var omit = {};
153     if(treeType == ZmOrganizer.FOLDER){
154         omit[ZmFolder.ID_ATTACHMENTS] = true;
155     }
156 
157     var params = {
158 		overviewId: Dwt.getNextId(),
159 		parent: this,
160 		treeIds: treeIds,
161         overviewClass: "DwtTree ZmFilterTree"
162     };
163     var overview = appCtxt.getOverviewController().createOverview(params);
164 	overview.set(treeIds, omit);
165 
166     var treeView = this._tree[type] = overview.getTreeView(treeType);
167 
168     var rootItem = treeView.getTreeItemById(ZmOrganizer.ID_ROOT);
169     rootItem.setExpanded(true);
170     rootItem.enableSelection(false);
171     rootItem.setText(ZmMsg[ZmAttachmentsController.CATEGORY_MSGKEY[type]]);
172 
173     //this._newHeaderItem(treeView, {msgKey:ZmAttachmentsController.CATEGORY_MSGKEY[type]});
174 
175     this._filtersContainer.appendChild(overview.getHtmlElement());
176 
177     treeView.addSelectionListener(selectionListener);
178 
179     return treeView;
180 };

182 ZmAttachmentsView.prototype._createTree = function(type, selectionListener){
183 
184     var tree = this._tree[type]  =  new DwtTree({parent:this, className:"DwtTree ZmFilterTree"});
185 
186     var root = this._newHeaderItem(tree, {msgKey:ZmAttachmentsController.CATEGORY_MSGKEY[type]});
187     tree.addSeparator();
188 
189     var categoryItems = ZmAttachmentsController.CATEGORY_ITEMS[type];
190     for(var i=0; i<categoryItems.length; i++){
191         this._newItem(root, categoryItems[i]);
192     }
193     tree.addSeparator();
194     tree.addSeparator();
195 
196     this._filtersContainer.appendChild(tree.getHtmlElement());
197 
198     tree.addSelectionListener(selectionListener);
199 
200     root.setExpanded(true);
201 
202     return tree;
203 };

254 ZmAttachmentsListView.prototype.isValidType = function(attach){
255 
256     if(this._types[0] == "any") return true;
257 
258     if(this._types[0] == "image"){
259         if(attach.ct.indexOf("image") == 0){
260             return true;
261         }
262     }else{
263         for(var i=0; i<this._types.length; i++){
264             if(attach.ct == this._types[i]) return true;
265         }
266     }
267 
268     return false;
269 };

271 ZmAttachmentsListView.prototype._createItemHtml =
272 function(msg) {
273 
274     var hrefRoot = appCtxt.get(ZmSetting.CSFE_MSG_FETCHER_URI) + "&loc=" + AjxEnv.DEFAULT_LOCALE + "&id=" + msg.id + "&part="
275 
276     var htmlEl = [];
277 
278     var atts = msg.attachments;
279     for (var i = 0; i < atts.length; i++) {
280 
281         var attach = atts[i];
282 
283         if(!this.isValidType(attach)) continue;
284 
285         var name = attach.name || attach.filename || (ZmMsg.unknown + " <" + attach.ct + ">");
286         if(name.length>14){
287 		    name = name.substring(0,14)+"...";
288 	    }
289 
290         var mimeInfo = ZmMimeTable.getInfo(attach.ct);
291         var icon = icon = "Img" + ( mimeInfo ? mimeInfo.imageLarge : "UnknownDoc_48");
292 
293         var div = document.createElement("div");
294         div.className = "ZmBriefcaseItem";
295 
296         var div1 = document.createElement("div");
297         div1.className = "ZmThumbnailItem";
298 
299         var div2 = document.createElement("div");
300         div2.className = icon+" ZmThumbnailIcon";
301 
302         div1.appendChild(div2);
303         div.appendChild(div1);
304 
305         var div2 = document.createElement("div");
306         div2.className = "ZmThumbnailName";
307 
308         var span = document.createElement("span");
309         span.innerHTML = ["<a href='",(hrefRoot + attach.part),"' target='_blank'>",name,"</a>"].join("");
310 
311         div2.appendChild(span);
312         div.appendChild(div2);
313 
314         this.associateItemWithElement(msg, div);
315 
316         htmlEl.push(div);
317 
318     }
319 
320     return htmlEl;
321 
322 };

324 ZmAttachmentsListView.prototype._addRow =
325 function(row, index) {
326 	if (!row) { return; }
327 
328 	// bug fix #1894 - check for childNodes length otherwise IE barfs
329 	var len = this._parentEl.childNodes.length;
330 
331     if (index != null && len > 0 && index != len) {
332         var childNodes = this._parentEl.childNodes;
333         this._parentEl.insertBefore(row, childNodes[index]);
334     } else {
335 		this._parentEl.appendChild(row);
336 	}
337 };
