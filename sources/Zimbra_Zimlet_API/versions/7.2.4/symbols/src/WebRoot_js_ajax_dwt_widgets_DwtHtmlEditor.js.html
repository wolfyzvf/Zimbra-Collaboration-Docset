 16 /**
 17  * @constructor
 18  * @class
 19  * Html Editor
 20  *
 21  * @author Ross Dargahi
 22  *
 23  * @param {hash}	params			a hash of parameters
 24  * @param {DwtComposite}      params.parent			the parent widget
 25  * @param {string}      params.className			the CSS class
 26  * @param {constant}      params.posStyle			the positioning style (see {@link DwtControl})
 27  * @param  {string}     params.content		the content (this can be HTML)
 28  * @param  {DwtHtmlEditor.HTML|DwtHtmlEditor.TEXT}     params.mode			the mode
 29  *
 30  * @extends		DwtComposite
 31  */
 32 DwtHtmlEditor = function(params) {
 33 	if (arguments.length == 0) { return; }
 34 	params = Dwt.getParams(arguments, DwtHtmlEditor.PARAMS);
 35 	params.className = params.className || "DwtHtmlEditor";
 36 	DwtComposite.call(this, params);
 37 
 38 	this._mode = (params.mode == DwtHtmlEditor.HTML && this.isHtmlEditingSupported())
 39 		? params.mode : DwtHtmlEditor.TEXT;
 40 
 41 	this.__eventClosure = AjxCallback.simpleClosure(this.__eventClosure, this);
 42 
 43 	// init content
 44 	this._pendingContent = params.content || "";
 45 	this._htmlModeInited = false;
 46 
 47 	this._initialize();
 48 };

161 DwtHtmlEditor.prototype.focus =
162 function(tryCount) {
163 	DBG.println(AjxDebug.DBG1, "DwtHtmlEditor.prototype.focus");
164 	if (!AjxEnv.isIE && this._mode == DwtHtmlEditor.TEXT) {
165 		document.getElementById(this._textAreaId).focus();
166 	} else {
167 		try {
168 			if (!this._htmlModeInited) {
169 				tryCount = tryCount || 0;
170 				if (tryCount < 10) {
171 					setTimeout(AjxCallback.simpleClosure(this.focus, this, tryCount + 1), DwtHtmlEditor._INITDELAY);
172 				}
173 				return;
174 			}
175 			if (AjxEnv.isSafari) {
176 				document.getElementById(this._iFrameId).focus();
177 			} else {
178 				this._getIframeWin().focus();
179 			}
180 			// Hack to fix IE focusing bug
181 			if (AjxEnv.isIE) {
182 				if (this._currInsPt) {
183 					if (this._currInsPt.text.length <= 1) {
184 						this._currInsPt.collapse(false);
185 					}
186 					this._currInsPt.select();
187 				}
188 			}
189 		} catch (ex) {
190 			// whateva
191 		}
192 	}
193 }

195 // Moves the caret to the top of the editor.
196 DwtHtmlEditor.prototype.moveCaretToTop =
197 function() {
198 	var focused = document.activeElement;
199 	if (this._mode == DwtHtmlEditor.TEXT) {
200 		var control = document.getElementById(this._textAreaId);
201 		if (control.createTextRange) { // IE
202 			var range = control.createTextRange();
203 			range.collapse(true);
204 			range.select();
205 		} else if (control.setSelectionRange) { // FF
206 			control.setSelectionRange(0, 0);
207 		}
208 	} else {
209 		this._moveCaretToTopHtml(true);
210 	}
211 	if (focused) {
212 		try {
213 			focused.focus();
214 		} catch (e) {}
215 	}
216 };

218 DwtHtmlEditor.prototype._moveCaretToTopHtml =
219 function(tryOnTimer) {
220 	var body = this._getIframeDoc().body;
221 	var success = false;
222 	if (AjxEnv.isIE) {
223 		if (body) {
224 			body.createTextRange().collapse(true);
225 			success = true;
226 		}
227 	} else {
228 		var selection = this._getSelection();
229 		if (selection) {
230 			selection.collapse(body,0);
231 			success = true;
232 		}
233 	}
234 	if (!success && tryOnTimer) {
235 		var action = new AjxTimedAction(this, this._moveCaretToTopHtml);
236 		AjxTimedAction.scheduleAction(action, DwtHtmlEditor._INITDELAY + 1);
237 	}
238 };

286  * @return	{string}	the content

288 DwtHtmlEditor.prototype.getContent =
289 function() {
290 	if (this._mode == DwtHtmlEditor.HTML) {
291 		var iframeDoc = this._getIframeDoc();
292 		var html = iframeDoc && iframeDoc.body ? (this._getIframeDoc().body.innerHTML) : "";
293 		return this._embedHtmlContent(html);
294 	} else {
295 		var textArea = document.getElementById(this._textAreaId);
296 		return textArea && textArea.value;
297 	}

308  * @param	{string}	content		the content
309  */
310 DwtHtmlEditor.prototype.setContent =
311 function(content) {
312 	if (AjxEnv.isIE) {
313 		this._currInsPt = null; // reset insertion pointer, bug 11623
314 	}
315 
316 	content = content || "";
317 	AjxDebug.println(AjxDebug.REPLY, "DwtHtmlEditor::setContent - Content length: " + content.length);
318 	if (this._mode == DwtHtmlEditor.HTML) {
319 		// If the html editor is initialized then go ahead and set the content, else let
320 		// _finishHtmlModeInit run before we try setting the content
321 		this._pendingContent = content;
322 		AjxDebug.println(AjxDebug.REPLY, "HTML mode, initialized: " + this._htmlModeInited);
323 		if (this._htmlModeInited) {
324 			this._setContentOnTimer();
325 		} // ELSE: _finishHtmlModeInit is about to run and it'll use _pendingContent
326 	} else {
327 		AjxDebug.println(AjxDebug.REPLY, "Text mode, set textarea value");
328 		document.getElementById(this._textAreaId).value = content;
329 	}
330 }

344 function(src, dontExecCommand, width, height) {
345     if (dontExecCommand) {
346         var doc = this._getIframeDoc();
347 	    var img = doc.createElement("img");
348 
349         //avoid mixed content security warning with IE
350         if (src && src.indexOf("cid:") == 0) {
351             img.setAttribute("dfsrc", src);
352         }
353         else {
354             img.setAttribute("src", src);
355         }
356         
357         if (width) {
358 			img.width = width;
359 		}
360 		else {
361 			img.removeAttribute('width');
362 		}
363 		if (height) {
364 			img.height = height;
365 		}
366 		else {
367 			img.removeAttribute('height');
368 		}
369 		img.style.border = "0"; // set border explicitly to 0, since otherwise a border is added implicitly in case it's wrapped with a link 
370 		var df = doc.createDocumentFragment();
371 	    df.appendChild(img);
372 	    this._insertNodeAtSelection(df);
373     }
374     else {
375         this._execCommand(DwtHtmlEditor.IMAGE, src);
376     }
377 };

392 /** Inserts a table at the current cursor position
393  *
394  * @param {number}	rows	the number of rows in the table
395  * @param {number}	cols	the number of columns in the table
396  * @param {string}	width		the width of the table. The units should be part of this value (for example: "100%" or "10px")
397  * @param {number}	cellSpacing	the cell spacing in the table
398  * @param {number}	cellPadding 	the cell padding in the table
399  * @param {constant}	alignment the alignment of the table. This is one of valid alignment values for the HTML
400  *    <table> element
401  *    
402  * @private
403  */
404 DwtHtmlEditor.prototype.insertTable =
405 function(rows, cols, width, cellSpacing, cellPadding, alignment) {
406 	if (this._mode != DwtHtmlEditor.HTML)
407 		return;
408 
409 	var doc = this._getIframeDoc();
410 	var table = doc.createElement("table");
411 	//table.className = "DwtHtmlEditor-Table";
412     /* This border style will solve problem while editing table properties */
413     table.style.borderRight = "1px solid #000";
414     table.style.borderLeft = "1px solid #000";
415     table.style.borderTop = "1px solid #000";
416     table.style.borderBottom = "1px solid #000";
417 
418     if (width != null) table.style.width = width;
419 	else table.style.width = "100%";
420 
421 	table.style.textAlign = "left";
422 	table.style.verticalAlign = "middle";
423 
424 	if (alignment != null) table.align = alignment.toLowerCase();
425 
426 	if (cellSpacing != null) table.cellSpacing = cellSpacing;
427 	else table.cellSpacing = 0;
428 
429 	if (cellPadding != null) table.cellPadding = cellPadding;
430 	else table.cellPadding = 0;
431 
432 	if (!AjxEnv.isIE)
433 		table.style.border = "1px solid #000";
434 	table.style.borderCollapse = "collapse";
435 
436 	var tdWidth;
437 	tdWidth = Math.floor(100 / cols) + "%";
438 	// tdWidth = null;
439 
440 	var tbody = doc.createElement("tbody");
441 	table.appendChild(tbody);
442 	// table.style.tableLayout = "fixed";
443 
444 	for (var i = 0; i < rows; i++) {
445 		var tr = doc.createElement("tr");
446 		tbody.appendChild(tr);
447 		for (var j = 0; j < cols; j++) {
448 			var td = doc.createElement("td");
449 			if (i == 0 && tdWidth)
450 				td.style.width = tdWidth;
451 			if (AjxEnv.isGeckoBased) {
452 				td.appendChild(doc.createElement("br"));
453 			} else if (AjxEnv.isIE) {
454 				td.innerHTML = " ";
455 			}
456 			if (AjxEnv.isIE)
457 				// since in IE we can't select
458 				// multiple cells anyway, and since
459 				// it's buggy if we only assign half
460 				// of the borders.... :-(
461 				td.style.border = "1px solid #000";
462 			else
463 				td.style.borderTop = td.style.borderLeft = "1px solid #000";
464 			tr.appendChild(td);
465 		}
466 	}
467 
468 	var p = doc.createElement("br");
469 	var df = doc.createDocumentFragment();
470 	df.appendChild(p);
471 	df.appendChild(table);
472 	df.appendChild(p.cloneNode(true));
473 
474 	this._insertNodeAtSelection(df);
475 	this.selectNodeContents(table.rows[0].cells[0], true);
476 	return table;
477 };

479 DwtHtmlEditor.prototype.applyTableProperties =
480 function(table, props) {
481 	var doc = this._getIframeDoc();
482 	var all_cells_props = [];
483 	for (var i in props) {
484 		var val = AjxStringUtil.trim(props[i].toString());
485 		var is_set = val != "";
486 		switch (i) {
487 		    case "caption":
488 				var caption = table.getElementsByTagName("caption");
489 				caption = caption.length > 0 ? caption[0] : null;
490 				if (is_set && !caption) {
491 					caption = doc.createElement("caption");
492 					table.insertBefore(caption, table.firstChild);
493 				}
494 				if (!is_set && caption)
495 					caption.parentNode.removeChild(caption);
496 				if (caption)
497 					caption.innerHTML = val;
498 				break;
499 
500 		    case "summary":
501 		    case "align":
502 		    case "cellSpacing":
503 		    case "cellPadding":
504 				if (!is_set)
505 					table.removeAttribute(i, 0);
506 				else
507 					table[i] = val;
508 				break;
509 
510 		    case "borderWidth":
511 		    case "borderStyle":
512 		    case "borderColor":
513 				// save these attributes to apply them to each cell
514 				table.style[i] = val;
515 				all_cells_props.push([ i, val ]);
516 				break;
517 
518 		    default :
519 				// TODO: remove unnecessary style (when is_set is false)
520 				table.style[i] = val;
521 				break;
522 		}
523 	}
524 	if (all_cells_props.length > 0) {
525 		var tds = table.getElementsByTagName("td");
526 		for (var i = tds.length; --i >= 0;) {
527 			var td = tds[i];
528 			for (var j = all_cells_props.length; --j >= 0;)
529 				td.style[all_cells_props[j][0]] = all_cells_props[j][1];
530 		}
531 	}
532 	if (AjxEnv.isGeckoBased)
533 		this._forceRedraw();
534 };

536 DwtHtmlEditor.prototype.applyCellProperties = function(table, cells, props) {
537 	var last_row = true;
538 	for (var row_i = cells.length; --row_i >= 0;) {
539 		var row = cells[row_i];
540 		if (!row) {
541 			continue;
542 		}
543 		var first_row = (row_i == 0);
544 		var last_col = true;
545 		for (var cell_i = row.length; --cell_i >= 0;) {
546 			var first_col = (cell_i == 0);
547 			var td = row[cell_i];
548 
549 			if (props.backgroundColor != null)
550 				td.style.backgroundColor = props.backgroundColor;
551 
552 			if (props.color != null)
553 				td.style.color = props.color;
554 
555 			if (props.textAlign != null)
556 				td.style.textAlign = props.textAlign;
557 
558 			if (props.verticalAlign != null)
559 				td.style.verticalAlign = props.verticalAlign;
560 
561 			if (props.width != null) {
562 				if (props.width)
563 					td.style.width = props.width + "px";
564 				else
565 					td.style.width = "";
566 			}
567 
568 			if (props.height != null) {
569 				if (props.height)
570 					td.style.height = props.height + "px";
571 				else
572 					td.style.height = "";
573 			}
574 
575 			if (props.vertPadding != null) {
576 				if (props.vertPadding)
577 					td.style.paddingTop = td.style.paddingBottom = props.vertPadding + "px";
578 				else
579 					td.style.paddingTop = td.style.paddingBottom = "";
580 			}
581 
582 			if (props.horizPadding != null) {
583 				if (props.horizPadding)
584 					td.style.paddingLeft = td.style.paddingRight = props.horizPadding + "px";
585 				else
586 					td.style.paddingLeft = td.style.paddingRight = "";
587 			}
588 
589 			var borders = props.borders, b;
590 
591 			// horiz. borders
592 
593 			b = borders[DwtHtmlEditor.BORDER_TOP];
594 			if (b != null && first_row) {
595 				td.style.borderTop = b.width + " " + b.style + " " + b.color;
596 			}
597 
598  			b = borders[DwtHtmlEditor.BORDER_MIDDLE];
599  			if (b != null) {
600 				b = b.width + " " + b.style + " " + b.color;
601 				if (!last_row)
602 					td.style.borderBottom = b;
603 				if (!first_row)
604 					td.style.borderTop = b;
605  			}
606 
607 			b = borders[DwtHtmlEditor.BORDER_BOTTOM];
608 			if (b != null && last_row) {
609 				td.style.borderBottom = b.width + " " + b.style + " " + b.color;
610 			}
611 
612 			// vertical borders
613 
614 			b = borders[DwtHtmlEditor.BORDER_LEFT];
615 			if (b != null && first_col) {

1105 DwtHtmlEditor.prototype._splitCells = function(td) {
1106 	var table = td;
1107 	while (table && table.nodeName.toLowerCase() != "table")
1108 		table = table.parentNode;
1109 	var mozbr = AjxEnv.isGeckoBased ? "<br />" : "";
1110 	function splitRow(td) {
1111 		var n = td.rowSpan;
1112 		var nc = td.colSpan;
1113 		td.rowSpan = 1;
1114 		tr = td.parentNode;
1115 		var itr = tr.rowIndex;
1116 		var trs = tr.parentNode.rows;
1117 		var index = td.cellIndex;
1118 		while (--n > 0) {
1119 			tr = trs[++itr];
1120 			var otd = td.cloneNode(false);
1121 			otd.removeAttribute("rowspan");
1122 			otd.colSpan = td.colSpan;
1123 			otd.innerHTML = mozbr;
1124 			tr.insertBefore(otd, tr.cells[index]);
1125 		}
1126 	};
1127 
1128 	function splitCol(td) {
1129 		var nc = td.colSpan;
1130 		td.colSpan = 1;
1131 		tr = td.parentNode;
1132 		var ref = td.nextSibling;
1133 		while (--nc > 0) {
1134 			var otd = td.cloneNode(false);
1135 			otd.removeAttribute("colspan");
1136 			otd.rowSpan = td.rowSpan;
1137 			otd.innerHTML = mozbr;
1138 			tr.insertBefore(otd, ref);
1139 		}
1140 	};

1142 	function splitCell(td) {
1143 		var nc = td.colSpan;
1144 		splitCol(td);
1145 		var items = td.parentNode.cells;
1146 		var index = td.cellIndex;
1147 		while (nc-- > 0) {
1148 			splitRow(items[index++]);
1149 		}
1150 	};
1151 	splitCell(td);

1154 DwtHtmlEditor.prototype.doTableOperation =
1155 function(cmd, params) {
1156 	var table = params.table || this.getNearestElement("table");
1157 	var td = params.td || this.getNearestElement("td");
1158 	var cellIndex, rowIndex, tr;
1159 	if (td) {
1160 		cellIndex = td.cellIndex;
1161 		tr = td.parentNode;
1162 		rowIndex = tr.rowIndex;
1163 	} else {
1164 		// Bug 8974: as usual, IE is weird to say the least.
1165 		cellIndex = 0;
1166 		tr = this.getNearestElement("tr");
1167 		if (tr)
1168 			rowIndex = tr.rowIndex;
1169 	}
1170 	var cells = params.cells;
1171 	while (true) {
1172 		switch (cmd) {
1173 		    case "insertRowAbove":
1174 			DwtHtmlEditor.table_fixCells(
1175 				DwtHtmlEditor.table_insertRow(td));
1176 			break;
1177 
1178 		    case "insertRowUnder":
1179 			DwtHtmlEditor.table_fixCells(
1180 				DwtHtmlEditor.table_insertRow(td, true));
1181 			break;
1182 
1183 		    case "insertColumnBefore":
1184 			DwtHtmlEditor.table_fixCells(
1185 				DwtHtmlEditor.table_insertCol(td));
1186 			break;
1187 
1188 		    case "insertColumnAfter":
1189 			DwtHtmlEditor.table_fixCells(
1190 				DwtHtmlEditor.table_insertCol(td, true));
1191 			break;
1192 
1193 		    case "deleteRow":
1194 			for (var i = 0; i < cells.length; ++i) {
1195 				var td = DwtHtmlEditor.table_deleteRow(cells[i][0]);
1196 				if (td) {
1197 					this.selectNodeContents(td, true);
1198 					this.focus();
1199 				}
1200 			}
1201 			break;
1202 
1203 		    case "deleteColumn":
1204 			for (var i = 0; i < cells[0].length; ++i) {
1205 				var td = DwtHtmlEditor.table_deleteCol(cells[0][i]);
1206 				if (td) {
1207 					this.selectNodeContents(td, true);
1208 					this.focus();
1209 				}
1210 			}
1211 			break;
1212 
1213 		    case "mergeCells":
1214 			td = cells[0][0];
1215 			var html = [ td.innerHTML.replace(/<br>$/i, "") ];
1216 			for (var i = 0; i < cells.length; ++i) {
1217 				var row = cells[i];
1218 				for (var j = 0; j < row.length; ++j) {
1219 					if (i || j) {
1220 						html.push(row[j].innerHTML.replace(/<br>$/i, ""));
1221 						row[j].parentNode.removeChild(row[j]);
1222 					}
1223 				}
1224 			}
1225 			td.colSpan = cells[0].length;
1226 			td.rowSpan = cells.length;
1227 			html = html.join(" ");
1228 			if (AjxEnv.isGeckoBased)
1229 				html += "<br/>";
1230 			td.innerHTML = html;
1231 			this.selectNodeContents(td, true);
1232 			break;
1233 
1234 		    case "splitCells":
1235 			this._splitCells(td);
