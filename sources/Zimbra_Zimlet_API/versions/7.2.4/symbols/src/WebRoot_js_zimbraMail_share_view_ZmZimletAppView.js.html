 16 /**
 17  * @extends		DwtComposite
 18  * @private
 19  */
 20 ZmZimletAppView = function(parent, controller) {
 21 	DwtComposite.call(this, {parent:parent, posStyle:DwtControl.ABSOLUTE_STYLE });
 22 };
 23 ZmZimletAppView.prototype = new DwtComposite;
 24 ZmZimletAppView.prototype.constructor = ZmZimletAppView;
 25 
 26 ZmZimletAppView.prototype.toString = function() {
 27 	return "ZmZimletAppView";
 28 };
 29 
 30 //
 31 // Public methods
 32 //
 33 
 34 ZmZimletAppView.prototype.setContent = function(html) {
 35 	var el = this.getHtmlElement();
 36 	el.innerHTML = html;
 37 };
 38 
 39 ZmZimletAppView.prototype.setView = function(view) {
 40 	var el = this.getHtmlElement();
 41 	el.innerHTML = "";
 42 	view.reparent(this);
 43 };
