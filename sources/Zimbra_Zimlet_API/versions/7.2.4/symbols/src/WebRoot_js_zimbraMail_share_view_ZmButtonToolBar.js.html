 20 /**
 21  * Creates a toolbar with the given buttons.
 22  * @class
 23  * This class represents a toolbar that contains buttons.
 24  * It can be easily created using a set of standard operations, and/or custom buttons
 25  * can be provided. This class is designed for use with items ({@link ZmItem}), so it can for
 26  * example contain a button with a tab submenu. See also {@link ZmActionMenu}.
 27  *
 28  * @author Conrad Damon
 29  *
 30  * @param {Hash}	params			a hash of parameters
 31  * @param	       {DwtComposite}	params.parent		the containing widget
 32  * @param	{Array}	params.buttons			a list of operation IDs
 33  * @param	{constant}	params.posStyle			the positioning style
 34  * @param	{String}	params.className			the CSS class name
 35  * @param	{Stirng}	params.buttonClassName	the CSS class name for buttons
 36  * @param	{Hash}	params.overrides			a hash of overrides by op ID
 37  * @param	{constant}	params.context			the vcontextID (used to generate button IDs)
 38  * @param	{constant}	params.toolbarType		the toolbar type (used to generate button IDs)
 39  * @param	{ZmController}	params.controller		the owning controller
 40  * 
 41  * @extends		ZmToolBar
 42  */
 43 ZmButtonToolBar = function(params) {
 44 	if (arguments.length == 0) return;
 45 
 46     params.className = params.className || "ZToolbar";
 47     params.id = params.context ? ZmId.getToolbarId(params.context, params.toolbarType) : null;
 48     ZmToolBar.call(this, params);
 49 	
 50 	this._context = params.context;
 51 	this._toolbarType = params.toolbarType;
 52 	this._buttonStyle = params.buttonClassName;
 53 
 54 	// standard buttons default to New/Tag/Print/Delete
 55 	var buttons = params.buttons;
 56 	if (!buttons) {
 57 		buttons = [ZmOperation.NEW_MENU, ZmOperation.TAG_MENU, ZmOperation.PRINT, ZmOperation.DELETE];
 58 	} else if (buttons == ZmOperation.NONE) {
 59 		buttons = null;
 60 	}
 61 	// weed out disabled ops, save list of ones that make it
 62 	/**
 63 	 * The operation list property.
 64 	 * @type Array
 65 	 */
 66 	this.opList = ZmOperation.filterOperations(buttons);
 67 	this._buttons = ZmOperation.createOperations(this, this.opList, params.overrides);
 68 
 69 	this._createPrecedenceList();
 70 	
 71 	this._inited = true;
 72 };
 73 
 74 ZmButtonToolBar.prototype = new ZmToolBar;
 75 ZmButtonToolBar.prototype.constructor = ZmButtonToolBar;

 84 /**
 85  * Creates a button and adds its operation ID as data.
 86  * 
 87  * @param {String}	id			the name of the operation
 88  * @param {Hash}	params		a hash of parameters
 89  * @param {String}	params.text			a button text
 90  * @param {String}	params.tooltip		a button tooltip text
 91  * @param {String}	params.image			a icon class for the button
 92  * @param {String}	params.disImage		a disabled version of icon
 93  * @param {Boolean}	params.enabled		if <code>true</code>, button is enabled
 94  * @param {String}	params.className		the CSS class name
 95  * @param {String}	params.style			thebutton style
 96  * @param {int} params.index			the position at which to add the button
 97  */
 98 ZmButtonToolBar.prototype.createOp =
 99 function(id, params) {
100 	params.className = this._buttonStyle;
101 	var b;
102 	if (id == ZmOperation.TEXT) {
103 		var id;
104 		if (this._context) {
105 			var context = this._toolbarType ? [this._context, this._toolbarType].join("_") : this._context;
106 			id = [ZmId.WIDGET, AjxStringUtil.toMixed(context, "_", true), AjxStringUtil.toMixed(id, "_")].join("");
107 		}
108 		params.textClassName = params.textClassName || "ZWidgetTitle";
109 		b = new DwtText({parent:this, className:params.textClassName, id:id});
110 	} else {
111 		params.id = this._context ? ZmId.getButtonId(this._context, id, this._toolbarType) : null;
112 		params.textPrecedence = ZmOperation.getProp(id, "textPrecedence");
113 		params.iconPrecedence = ZmOperation.getProp(id, "iconPrecedence");
114 		b = this.createButton(id, params);
115 	}
116 	b.setData(ZmOperation.KEY_ID, id);
117 
118 	return b;
119 };
120 
121 /**
122  * Adds the operation.
123  * 
124  * @param	{String}	id		the id
125  * @param	{int}		index	the index
126  */
127 ZmButtonToolBar.prototype.addOp =
128 function(id, index) {
129 	if(this.getOp(id)) {
130 		return;
131 	}
132 	ZmOperation.addOperation(this, id, this._buttons, index);
133 	AjxUtil.arrayAdd(this.opList, id, index);
134 };
135 
136 /**
137  * Removes the operation.
138  * 
139  * @param	{String}	id		the id
140  * 
141  * @see ZmOperation
142  */
143 ZmButtonToolBar.prototype.removeOp =
144 function(id) {
145 	ZmOperation.removeOperation(this, id, this._buttons);
146 	AjxUtil.arrayRemove(this.opList, id);
147 };
148 
149 /**
150  * Gets the button.
151  *
152  * @param {constant}	id		the button
153  * @return	{DwtButton}	the button
154  * 
155  * @see ZmOperation
156  */
157 ZmButtonToolBar.prototype.getOp =
158 function(id) {
159 	return this.getButton(id);
160 };
161 
162 /**
163  * Gets the menu tag sub-menu (if any).
164  * 
165  * @return	{ZmTagMenu}		the menu
166  */
167 ZmButtonToolBar.prototype.getTagMenu =
168 function() {
169 	var button = this.getButton(ZmOperation.TAG_MENU);
170 	if (button) {
171 		return button.getMenu();
172 	}
173 };

179 // Returns the ID for the given button.

182 	return button.getData(ZmOperation.KEY_ID);
