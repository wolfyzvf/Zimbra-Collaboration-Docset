 20 /**
 21  * @class
 22  * Creates a stack of undoable actions (ZmAction objects)
 23  *
 24  * @param	{int}	[maxLength]		The maximum size of the stack. Defaults to 0, meaning no limit
 25  *
 26  * Adding actions to a full stack will pop the oldest actions off
 27  */
 28 ZmActionStack = function(maxLength) {
 29 	this._stack = [];
 30 	this._pointer = -1;
 31 	this._maxLength = maxLength || 0; // 0 means no limit
 32 };
 33 
 34 ZmEvent.S_ACTION			= "ACTION";
 35 
 36 ZmActionStack.validTypes	= [ZmId.ORG_FOLDER, ZmId.ITEM_MSG, ZmId.ITEM_CONV,
 37                                ZmId.ITEM_CONTACT, ZmId.ITEM_GROUP,
 38                                ZmId.ITEM_BRIEFCASE, ZmId.ORG_BRIEFCASE,
 39                                ZmId.ITEM_TASK, ZmId.ORG_TASKS
 40                               ]; // Set ZmActionStack.validTypes to false to allow all item types
 41 
 42 ZmActionStack.prototype.toString = function() {
 43 	return "ZmActionStack";
 44 };
 45 
 46 /**
 47  * Logs a raw action, interpreting the params and creates a ZmAction object that is pushed onto the stack and returned
 48  * 
 49  * @param {String}	op			operation to perform. Currently supported are "move", "trash", "spam" and "!spam"
 50  * @param {Hash}	[attrs] 	attributes for the operation. Pretty much the same as what the backend expects, e.g. "l" for the destination folderId of a move
 51 
 52  * @param {String}	[items] 	array of items to perform the action for. Valid types are specified in ZmActionStack.validTypes. Only one of [items],[item],[ids] or [id] should be specified; the first one found is used, ignoring the rest.
 53  * @param {String}	[item] 		item to perform the action for, if there is only one item. Accomplishes the same as putting the item in an array and giving it as [items]
 54  * @param {String}	[ids] 		array of ids of items to perform the action for.
 55  * @param {String}	[id] 		id of item to perform the action for, if there is only one. Accomplishes the same as putting the id in an array and giving it as [ids].
 56  */
 57 ZmActionStack.prototype.logAction = function(params) {
 58 	var op = params.op;
 59 	var items = [];
 60 	if (params.items) {
 61 		for (var i=0; i<params.items.length; i++) {
 62 			var item = params.items[i];
 63 			if (item && (!ZmActionStack.validTypes || AjxUtil.indexOf(ZmActionStack.validTypes, item.type)!=-1)) {
 64 				items.push(item);
 65 			}
 66 		}
 67 	} else if (params.item) {
 68 		if (params.item && (!ZmActionStack.validTypes || AjxUtil.indexOf(ZmActionStack.validTypes, params.item.type)!=-1)) {
 69 			items.push(params.item);
 70 		}
 71 	} else if (params.ids) {
 72 		for (var i=0; i<params.ids.length; i++) {
 73 			var item = appCtxt.getById(params.ids[i]);
 74 			if (item && (!ZmActionStack.validTypes || AjxUtil.indexOf(ZmActionStack.validTypes, item.type)!=-1)) {
 75 				items.push(item);
 76 			}
 77 		}
 78 	} else if (params.id) {
 79 		var item = appCtxt.getById(params.id);
 80 		if (item && (!ZmActionStack.validTypes || AjxUtil.indexOf(ZmActionStack.validTypes, item.type)!=-1)) {
 81 			items.push(item);
 82 		}
 83 	}
 84 	for (var i=0; i<items.length; i++) {
 85 		if (items[i] instanceof ZmConv) { // For conversation moves, also log the messages within
 86 			var msgs = items[i].getMsgList();
 87 			for (var j=0; j<msgs.length; j++) {
 88 				if (AjxUtil.indexOf(msgs[j])==-1) {
 89 					items.push(msgs[j]);
 90 				}
 91 			}
 92 		}
 93 	}
 94 	var attrs = params.attrs;
 95 	var multi = items.length>1;
 96 
 97 	var action = null;
 98 	var folderId;
 99 	switch (op) {
100 		case "trash":
101 			folderId = ZmFolder.ID_TRASH;
102 			break;
103 		case "spam":
104 			folderId = ZmFolder.ID_SPAM;
105 			break;
106 		case "move":
107 		case "!spam":
108 			folderId = attrs.l;
109 			break;
110 	}
111 
112 	var folder = appCtxt.getById(folderId);
113 	if (folder && !folder.isRemote()) { // Enable undo only when destination folder exists (it should!!) and is not remote (bug #51656)
114 		switch (op) {
115 			case "trash":
116 			case "move":
117 			case "spam":
118 			case "!spam":
119 				for (var i=0; i<items.length; i++) {
120 					var item = items[i];
121 					var moveAction;
122 				
123 					if (item instanceof ZmItem) {
124 						if (!item.isShared()) // Moving shared items is not undoable
125 							moveAction = new ZmItemMoveAction(item, item.getFolderId(), folderId, op);
126 					} else if (item instanceof ZmOrganizer) {
127 						if (!item.isRemote()) // Moving remote organizers is not undoable
128 							moveAction = new ZmOrganizerMoveAction(item, item.parent.id, folderId, op);
129 					}
130 					if (moveAction) {
131 						if (multi) {
132 							if (!action) action = new ZmCompositeAction();
133 							action.addAction(moveAction);
134 						} else {
135 							action = moveAction;
136 						}
137 					}
138 				}
139 				break;
140 		}
141 		if (action) {
142 			this._push(action);
143 		}
144 	}
145 
146 	return action;
147 };
148 
149 /**
150  * Returns whether there are actions that can be undone
151  */
152 ZmActionStack.prototype.canUndo = function() {
153 	return this._pointer >= 0;
154 };
155 
156 /**
157  * Returns whether there are actions that can be redone
158  */
159 ZmActionStack.prototype.canRedo = function() {
160 	return this._pointer < this._stack.length - 1;
161 };
162 
163 /**
164  * Undoes the current action (if applicable) and moves the internal pointer
165  */
166 ZmActionStack.prototype.undo = function() {
167 	if (this.canUndo()) {
168 		var action = this._pop();
169 		action.undo();
170 	}
171 };
172 
173 /**
174  * Redoes the current action (if applicable) and moves the internal pointer
175  */
176 ZmActionStack.prototype.redo = function() {
177 	if (this.canRedo()) {
178 		var action = this._stack[++this._pointer];
179 		action.redo();
180 	}
181 };
182 
183 /**
184  * Puts an action into the stack at the current position
185  * If we're not at the top of the stack (ie. undoes have been performed), we kill all later actions (so redoing the undone actions is no longer possible)
186  */
187 ZmActionStack.prototype._push = function(action) {
188 	if (action && action instanceof ZmAction) {
189 		var next = this._pointer + 1;
190 		while (this._maxLength && next>=this._maxLength) {
191 			// Stack size is reached, shift off actions until we're under the limit
192 			this._stack.shift();
193 			next--;
194 		}
195 		this._stack[next] = action;
196 		this._stack.length = next+1; // Kill all actions after pointer
197 		this._pointer = next;
198 	}
199 };
200 
201 /**
202  * Returns the action at the current position and moves the pointer
203  */
204 ZmActionStack.prototype._pop = function() {
205 	return this.canUndo() ? this._stack[this._pointer--] : null;
206 };
