 20 /**
 21  * Creates the assistant.
 22  * @class
 23  * This class represents an assistant.
 24  * 
 25  * @param	{String}	title		the title
 26  * @param	{String}	command		the command
 27  * @param	{String}	commandSummary		the summary
 28  */
 29 ZmAssistant = function(title, command, commandSummary) {
 30 	if (arguments.length == 0) { return; }
 31 	this._objectManager = new ZmObjectManager();
 32     try {
 33         if (window.ZmDate10ObjectHandler) {
 34 		     this._objectManager.addHandler(new ZmDate10ObjectHandler(), ZmObjectManager.DATE, 100);
 35         }
 36     } catch(e) {
 37 
 38     }
 39     this._fields = {};
 40 	this._title = title;
 41 	this._commandSummary = commandSummary ? commandSummary : title;
 42 	this._command = command;
 43 };

 47 ZmAssistant.SPACE = "[\\s\\u00a0\\u2000-\\u200B\\u202F\\u3000]";
 48 ZmAssistant.NONSPACE = "[^\\s\\u00a0\\u2000-\\u200B\\u202F\\u3000]";

 50 ZmAssistant.SPACES = ZmAssistant.SPACE+"*";

 53 ZmAssistant.LEADING_SPACE = "^"+ZmAssistant.SPACES;
 54 ZmAssistant.TRAILING_SPACE = ZmAssistant.SPACES+"$";

 56 ZmAssistant.RE_trimLeading = new RegExp(ZmAssistant.LEADING_SPACE);
 57 ZmAssistant.RE_trimTrailing = new RegExp(ZmAssistant.LEADING_SPACE);
 58 ZmAssistant.RE_normalizeSpaces = new RegExp(ZmAssistant.SPACE+"+","g");
 59 
 60 ZmAssistant.trimLeading = function(s) { return s.replace(ZmAssistant.RE_trimLeading,""); };
 61 ZmAssistant.trimTrailing = function(s) { return s.replace(ZmAssistant.RE_trimTrailing,""); };
 62 ZmAssistant.trim = function(s) { return ZmAssistant.trimLeading(ZmAssistant.trimTrailing(s)); };
 63 ZmAssistant.normalize = function(s) { return s.replace(ZmAssistant.RE_normalizeSpaces," "); };
 64 ZmAssistant.split = function(s) { return s.split(ZmAssistant.RE_normalizeSpaces); };

 66 ZmAssistant._handlers = {};
 67 ZmAssistant._commands = [];
 68 ZmAssistant._allCommands = []; // inclde . commands

 70 /**
 71  * Registers the handler.
 72  * 
 73  * @param	{ZmAssistant}		handler		the handler
 74  * @param	{String}			[name]		the name
 75  */
 76 ZmAssistant.register = 
 77 function(handler, name) {
 78 	if (name == null) name = handler.getCommand();
 79 	if (name in ZmAssistant._handlers) return; // need to alert/error out and/or deal with dups!
 80 	ZmAssistant._handlers[name] = handler;
 81 	ZmAssistant._allCommands.push(name);
 82 	ZmAssistant._allCommands.sort();
 83 	if (name.substring(0,1) != ".") {
 84 		ZmAssistant._commands.push(name);
 85 		ZmAssistant._commands.sort();		
 86 	}
 87 };

 89 /**
 90  * Matches the word.
 91  * 
 92  * @param	{String}	word		the word
 93  * @param	{Array}		words		an array of words
 94  * @return	{Array} an array of matched words or an empty array for none
 95  */
 96 ZmAssistant.matchWord = 
 97 function(word, words) {
 98 	if (words == null) words = ZmAssistant._allCommands;
 99 	var i;
100 	var matched = [];
101 	for (i in words) {
102 		var n = words[i];
103 		if (n == word) return [n];
104 		else if (n.substring(0, word.length) == word) {
105 			matched.push(n);
106 		}
107 	}
108 	return matched;
109 };

232 ZmAssistant.__RE_matchTimeMatch1 = new RegExp([
233     ZmAssistant.SPACES,
234     "(?:(?:@|at|\\-)",ZmAssistant.SPACES,")?(\\d+):(\\d\\d)(?:",ZmAssistant.SPACES,"(AM|PM))?",
235     ZmAssistant.SPACES
236 ].join(""),"i");
237 ZmAssistant.__RE_matchTimeMatch2 = new RegExp([
238     ZmAssistant.SPACES,
239     "(?:(?:@|at|\\-)",ZmAssistant.SPACES,")?(\\d+)(AM|PM)",
240     ZmAssistant.SPACES
241 ].join(""),"i");

243 ZmAssistant.prototype._matchTime =
244 function(args) {
245 	var hour, minute, ampm = null;
246 	var match1 = args.match(ZmAssistant.__RE_matchTimeMatch1);
247 	var match2 = args.match(ZmAssistant.__RE_matchTimeMatch2);
248 	// take the first match
249 	if (match1 && match2) {
250 		if  (match1.index < match2.index) match2 = null;
251 		else match1 = null;
252 	}
253 	if (match1) {
254 		hour = parseInt(match1[1]);
255 		minute = parseInt(match1[2]);
256 		if (match1[3]) ampm = match1[3].toLowerCase();
257 		args = args.replace(match1[0], " ");
258 	} else if (match2) {
259 		hour = parseInt(match2[1]);
260 		minute = 0;
261 		ampm = match2[2].toLowerCase();	
262 		args = args.replace(match2[0], " ");
263 	} else {
264 		return null;
265 	}
266 
267 	if (ampm == 'pm' && hour < 12) hour += 12;
268 	else if (ampm == 'am' && hour == 12) hour = 0;
269 
270 	return {hour: hour, minute: minute, args: args };
271 };

276 ZmAssistant.__RE_matchTypedObjectBrackets = new RegExp([
277     ZmAssistant.SPACES,"\\[([^\\]]*)\\]?",ZmAssistant.SPACES
278 ].join(""), "i");

280     "\\b(",ZmAssistant.WORD,"):",ZmAssistant.SPACES,"$"
281 ].join(""), "i");

283 ZmAssistant.prototype._matchTypedObject =
284 function(args, objType, obj) {
285 	
286 	var match;
287 	var matchIndex = 0;
288 	
289 	if (objType == ZmAssistant._BRACKETS) {
290 		match = args.match(ZmAssistant.__RE_matchTypedObjectBrackets);
291 		matchIndex = 1;
292 	} else {
293 		match = this._objectManager.findMatch(args, objType);
294 	}
295 	if (!match) return null;
296 
297 	var type = obj.defaultType;
298 	var matchType = null;
299 	if (match.index > 0) {
300 		// check for a type
301 		var targs = args.substring(0, match.index);
302 		matchType = targs.match(ZmAssistant.__RE_matchTypedObjectType);
303 		if (matchType) {
304 			type = matchType[1].toLowerCase();
305 		}
306 	}
307 	if (matchType) {
308 		args = args.replace(matchType[0]+match[0], " ");
309 	} else {
310 		args = args.replace(match[0], " ");
311 	}
312 
313 	if (obj.aliases) {
314 		var real = obj.aliases[type];
315 		if (real) type = real;
316 	}
317 	
318 
319 	return {data: match[matchIndex], args: args, type: type};
320 };

322 ZmAssistant.prototype._clearField = 
323 function(title) {
324 	var fieldData = this._fields[title];
325 	if (fieldData) {
326 		var rowEl = document.getElementById(fieldData.rowId);
327 		if (rowEl) rowEl.parentNode.removeChild(rowEl);
328 		delete this._fields[title];
329 	}
330 }

332 ZmAssistant.prototype._setOptField = 
333 function(title, value, isDefault, htmlEncode, afterRowTitle, titleAlign) {
334 	if (value && value != "") {
335 		return this._setField(title, value, isDefault, htmlEncode, afterRowTitle, titleAlign);
336 	} else {
337 		this._clearField(title);
338 		return -1;
339 	}
340 }

342 ZmAssistant.prototype._setField = 
343 function(title, value, isDefault, htmlEncode, desiredRowIndex, titleAlign) {
344 	var cname =  isDefault ? "ZmAsstFieldDefValue" : "ZmAsstField";
345 	var rowIndex = -1;
346 	var fieldData = this._fields[title];
347 	if (htmlEncode) value = AjxStringUtil.htmlEncode(value);
348 	if (fieldData) {
349 		var rowEl = document.getElementById(fieldData.rowId);
350 		if (rowEl) rowIndex = rowEl.rowIndex;
351 		var divEl = document.getElementById(fieldData.id);
352 		divEl.innerHTML = value;
353 		divEl.className = cname;
354 	} else {
355 		var id = Dwt.getNextId();
356 		if (desiredRowIndex != null) rowIndex = desiredRowIndex;
357 		var tableEl = document.getElementById(this._tableId);
358 		var row = tableEl.insertRow(rowIndex);
359 		row.id = Dwt.getNextId();
360 		var cell1 = row.insertCell(-1);
361 		cell1.vAlign = titleAlign ? titleAlign : "top";
362 		cell1.className = 'ZmAsstFieldLabel';
363 		cell1.innerHTML = AjxStringUtil.htmlEncode(title+":");
364 		var cell2 = row.insertCell(-1);
365 		cell2.innerHTML = "<div id='"+id+"' class='"+cname+"'>"+value+"</div></td>";
366 		this._fields[title] = { id: id, rowId: row.id };
367 		rowIndex = row.rowIndex;
368 	}
369 	return rowIndex;
370 };

372 ZmAssistant.prototype._clearFields =

374 	for (var field in this._fields) {
375 		this._clearField(field);
376 	}
