 20 /**
 21  * Creates an attachment dialog.
 22  * @class
 23  * This class represents an attachment dialog.
 24  * 
 25  * @param	{DwtControl}	shell		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtDialog
 29  */
 30 ZmAttachDialog = function(shell, className) {
 31 
 32 	className = className || "ZmAttachDialog";
 33 	DwtDialog.call(this, {parent:shell, className:className, title:ZmMsg.attachFile});
 34 
 35 	// Initialize
 36 	this._createBaseHtml();
 37 
 38 	// Ok and Cancel Actions
 39 	this._defaultCancelCallback = new AjxCallback(this, this._defaultCancelListener);
 40 	this._cancelListeners = {};
 41 
 42 	this._defaultOkCallback = new AjxCallback(this, this._defaultOkListener);
 43 	this._okListeners = {};
 44 
 45 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, function() {
 46 		this._cancelButtonListener();
 47 	}));
 48 
 49 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, function() {
 50 		this._okButtonListener();
 51 	}));
 52 
 53 	this._tabKeys = {};
 54 
 55 	var okButton = this.getButton(DwtDialog.OK_BUTTON);
 56 	okButton.setText(ZmMsg.attach);
 57 
 58 	// Add Default MyComputer tab
 59 	this._addMyComputerTab();
 60     if (appCtxt.multiAccounts || appCtxt.get(ZmSetting.BRIEFCASE_ENABLED)) {
 61         this._addBriefcaseViewTab();
 62     }
 63 };

277 // Upload Utitlity Methods
278 ZmAttachDialog.prototype.uploadFiles =
279 function() {
280     this.setFooter('');
281 	var tabKey = this._tabView.getCurrentTab();
282 	var tabView = this._tabView.getTabView(tabKey);
283     if(tabView && tabView.validate){
284         var valid = tabView.validate();
285         if(!valid.status){
286             this.setFooter(valid.error);
287             return false;
288         }
289     }
290 	if (tabView && tabView.gotAttachments()) {
291 		this.upload(this._uploadCallback, tabView.uploadForm);
292 	} else {
293 		this.setFooter(ZmMsg.attachSelectMessage);
294 	}
295 };

315 /**
316  * Uploads the attachments.
317  * 
318  * @param	{AjxCallback}		callback		the callback
319  * @param	{Object}			uploadForm		the upload form
320  */
321 ZmAttachDialog.prototype.upload =
322 function(callback, uploadForm) {
323 	if (!callback) {
324 		callback = false;
325 	}
326 	this.setButtonEnabled(DwtDialog.OK_BUTTON, false);
327 	this.setButtonEnabled(DwtDialog.CANCEL_BUTTON, true);
328 	this.setFooter(ZmMsg.attachingFiles);
329 	this._cancelUpload = false;
330 	this._processUpload(callback, uploadForm);
331 };
332 
333 ZmAttachDialog.prototype._processUpload =
334 function(callback, uploadForm) {
335 	var ajxCallback = new AjxCallback(this, this._uploadDoneCallback, [callback]);
336 	var um = appCtxt.getUploadManager();
337 	window._uploadManager = um;
338 
339 	try {
340 		um.execute(ajxCallback, uploadForm);
341 	} catch (ex) {
342 		ajxCallback.run();
343 	}
344 };
345 
346 ZmAttachDialog.prototype._uploadDoneCallback =
347 function(callback, status, attId) {
348 	if (this._cancelUpload) { return; }
349 
350 	this.setButtonEnabled(DwtDialog.CANCEL_BUTTON, true);
351 
352 	if (status == AjxPost.SC_OK) {
353 		this.setFooter(ZmMsg.attachingFilesDone);
354 		if (callback) {
355 			callback.run(status, attId);
356 		}
357 	} else if (status == AjxPost.SC_UNAUTHORIZED) {
358 		// auth failed during att upload - let user relogin, continue with compose action
359 		var ex = new AjxException("401 response during attachment upload", ZmCsfeException.SVC_AUTH_EXPIRED);
360 		appCtxt.getAppController()._handleException(ex, {continueCallback:callback});
361 	} else {
362 		// bug fix #2131 - handle errors during attachment upload.
363 		var msg = AjxMessageFormat.format(ZmMsg.errorAttachment, (status || AjxPost.SC_NO_CONTENT));
364 
365 		switch (status) {
366 			// add other error codes/message here as necessary
367 			case AjxPost.SC_REQUEST_ENTITY_TOO_LARGE:	msg += " " + ZmMsg.errorAttachmentTooBig + "<br><br>"; break;
368 			default:									msg += " "; break;
369 		}
370 		var dialog = appCtxt.getMsgDialog();
371 		dialog.setMessage(msg, DwtMessageDialog.CRITICAL_STYLE);
372 		dialog.popup();
373 
374 		this.setFooter(ZmMsg.attachingFilesError);
375 	}
376 
377 	this.setButtonEnabled(DwtDialog.OK_BUTTON, true);
378 };

558 // Attachments
559 ZmMyComputerTabViewPage.prototype._addAttachmentField =
560 function() {
561 	if (this._attachCount >= ZmMyComputerTabViewPage.MAX_NO_ATTACHMENTS) { return; }
562 
563 	this._attachCount++;
564 
565 	var row = this.attachmentTable.insertRow(-1);
566 	var cell = row.insertCell(-1);
567 	var fieldId = Dwt.getNextId();
568 
569 	var subs = {
570 		id: fieldId,
571 		uploadName: ZmMyComputerTabViewPage.UPLOAD_FIELD_NAME
572 	};
573 	cell.innerHTML = AjxTemplate.expand("share.Dialogs#ZmAttachDialog-MyComputerTab-AddAttachment", subs);
574 
575 	var removeEl = document.getElementById(fieldId+"_remove");   
576 	removeEl.onclick = AjxCallback.simpleClosure(this._removeAttachmentField, this, row);
577 
578     var inputId = fieldId+"_input";
579 	if (this._focusElId == -1) {
580 		this._focusElId = inputId;
581 	}    
582     var inputEl = document.getElementById(inputId);
583     var sizeEl = document.getElementById(fieldId+"_size");
584 
585     //HTML5
586     if(ZmAttachDialog.supportsHTML5){
587         Dwt.setHandler(inputEl, "onchange", AjxCallback.simpleClosure(this._handleFileSize, this, inputEl, sizeEl));
588     }
589 
590 	// trap key presses in IE for input field so we can ignore ENTER key (bug 961)
591 	if (AjxEnv.isIE) {
592 		inputEl.onkeydown = AjxCallback.simpleClosure(this._handleKeys, this);
593 	}
594 };

596 ZmMyComputerTabViewPage.prototype._handleFileSize =
597 function(inputEl, sizeEl){
598 
599     var files = inputEl.files;
600     if(!files) return;
601 
602     var sizeStr = [], className, totalSize =0;
603     for(var i=0; i<files.length;i++){
604         var file = files[i];
605         var size = file.size || file.fileSize /*Safari*/;
606         if(size > appCtxt.get(ZmSetting.ATTACHMENT_SIZE_LIMIT))
607             className = "RedC";
608         totalSize += size;
609     }
610 
611     if(sizeEl) {
612         sizeEl.innerHTML = "  ("+AjxUtil.formatSize(totalSize, true)+")";
613         if(className)
614             Dwt.addClass(sizeEl, "RedC");
615         else
616             Dwt.delClass(sizeEl, "RedC");
617     }
618 };

682 ZmMyComputerTabViewPage.prototype._cleanTable =

696 ZmMyComputerTabViewPage.prototype._validateFileSize =
697 function(){
698 
699     var atts = document.getElementsByName(ZmMyComputerTabViewPage.UPLOAD_FIELD_NAME);
700     var file, size;
701 	for (var i = 0; i < atts.length; i++){
702         file = atts[i].files;
703         if(!file || file.length == 0) continue;
704         for(var j=0; j<file.length;j++){
705             var f = file[j];
706             size = f.size || f.fileSize /*Safari*/;
707             if(size > appCtxt.get(ZmSetting.ATTACHMENT_SIZE_LIMIT)){
708                 return false;
709             }
710         }
711     }
712 	return true;
713 };
