 21 /**
 22  * Creates the task tree controller.
 23  * @class
 24  * This class represents the task tree controller.
 25  * 
 26  * @extends		ZmFolderTreeController
 27  */
 28 ZmTaskTreeController = function() {
 29 
 30 	ZmFolderTreeController.call(this, ZmOrganizer.TASKS);
 31 
 32 	this._listeners[ZmOperation.NEW_TASK_FOLDER] = new AjxListener(this, this._newListener);
 33 	this._listeners[ZmOperation.SHARE_TASKFOLDER] = new AjxListener(this, this._shareTaskFolderListener);
 34 	this._listeners[ZmOperation.BROWSE] = new AjxListener(this, function(){ appCtxt.getSearchController().fromBrowse(""); });
 35 
 36 	this._eventMgrs = {};
 37 };
 38 
 39 ZmTaskTreeController.prototype = new ZmFolderTreeController;
 40 ZmTaskTreeController.prototype.constructor = ZmTaskTreeController;

 54 /**
 55  * Displays the tree of this type.
 56  *
 57  * @param {Hash}	params		a hash of parameters
 58  * @param	{constant}	params.overviewId		the overview ID
 59  * @param	{Boolean}	params.showUnread		if <code>true</code>, unread counts will be shown
 60  * @param	{Object}	params.omit				a hash of organizer IDs to ignore
 61  * @param	{Object}	params.include			a hash of organizer IDs to include
 62  * @param	{Boolean}	params.forceCreate		if <code>true</code>, tree view will be created
 63  * @param	{String}	params.app				the app that owns the overview
 64  * @param	{Boolean}	params.hideEmpty		if <code>true</code>, don't show header if there is no data
 65  * @param	{Boolean}	params.noTooltips	if <code>true</code>, don't show tooltips for tree items
 66  */
 67 ZmTaskTreeController.prototype.show = function(params) {
 68 	params.include = params.include || {};
 69     params.include[ZmFolder.ID_TRASH] = true;
 70     params.showUnread = false;
 71     return ZmFolderTreeController.prototype.show.call(this, params);
 72 };

 74 ZmTaskTreeController.prototype.resetOperations =
 75 function(parent, type, id) {
 76 	var deleteText = ZmMsg.del;
 77 	var folder = appCtxt.getById(id);
 78     var isShareVisible = true;
 79 
 80 	parent.enableAll(true);
 81 	if (folder) {
 82         if (folder.isSystem()) {
 83             parent.enable([ZmOperation.DELETE, ZmOperation.RENAME_FOLDER], false);
 84         } else if (folder.link && !folder.isAdmin()) {
 85             isShareVisible = false;
 86         }
 87         if (appCtxt.isOffline) {
 88             var acct = folder.getAccount();
 89             isShareVisible = !acct.isMain && acct.isZimbraAccount;
 90         }
 91         parent.enable([ZmOperation.SHARE_TASKFOLDER], isShareVisible);
 92         parent.enable(ZmOperation.SYNC, folder.isFeed());
 93 	}
 94 
 95     parent.enable(ZmOperation.EMPTY_FOLDER,((folder.numTotal > 0) || (folder.children && (folder.children.size() > 0))));
 96     var nId = ZmOrganizer.normalizeId(id);
 97     var isTrash = nId == ZmOrganizer.ID_TRASH;
 98 	this.setVisibleIfExists(parent, ZmOperation.EMPTY_FOLDER, isTrash);
 99 
100 	parent.enable(ZmOperation.EDIT_PROPS, !isTrash);
101 	var emptyFolderOp = parent.getOp(ZmOperation.EMPTY_FOLDER);
102 	if (emptyFolderOp) {
103 		emptyFolderOp.setText(ZmMsg.emptyTrash);
104 	}
105 
106 	this._enableRecoverDeleted(parent, isTrash);
107 
108 	var op = parent.getOp(ZmOperation.DELETE);
109 	if (op) {
110 		op.setText(deleteText);
111 	}
112 
113 	// we always enable sharing in case we're in multi-mbox mode
114 	// But no sharing for trash folder
115 	this._resetButtonPerSetting(parent, ZmOperation.SHARE_TASKFOLDER, !isTrash && appCtxt.get(ZmSetting.SHARING_ENABLED));
116 };

161 ZmTaskTreeController.prototype._shareTaskFolderListener =
162 function(ev) {
163 	if (!this._sharingPossible()) {
164 		return;
165 	}
166 	
167 	this._pendingActionData = this._getActionedOrganizer(ev);
168 	appCtxt.getSharePropsDialog().popup(ZmSharePropsDialog.NEW, this._pendingActionData);
169 };
170 
171 ZmTaskTreeController.prototype._deleteListener =
172 function(ev) {
173 	var organizer = this._getActionedOrganizer(ev);
174 	var callback = new AjxCallback(this, this._deleteListener2, [organizer]);
175 	var message = AjxMessageFormat.format(ZmMsg.confirmDeleteTaskFolder, AjxStringUtil.htmlEncode(organizer.name));
176 
177 	appCtxt.getConfirmationDialog().popup(message, callback);
178 };

199 /**
200  * Gets the task Folders.
201  *
202  * @param	{String}	overviewId		the overview id
203  * @param   {boolean}   includeTrash    True to include trash, if checked.
204  * @return	{Array}		an array of {@link ZmCalendar} objects
205  */
206 ZmTaskTreeController.prototype.getTaskFolders =
207 function(overviewId, includeTrash) {
208 	var tasks = [];
209 	var items = this._getItems(overviewId);
210 	for (var i = 0; i < items.length; i++) {
211 		var item = items[i];
212 		if (item._isSeparator) { continue; }
213 		var task = item.getData(Dwt.KEY_OBJECT);
214         if (task && task.id == ZmOrganizer.ID_TRASH && !includeTrash && task.type) continue;
215 		if (task) tasks.push(task);
216 	}
217 
218 	return tasks;
219 };
220 
221 
222 ZmTaskTreeController.prototype._getItems =
223 function(overviewId) {
224 	var treeView = this.getTreeView(overviewId);
225 	if (treeView) {
226 		var account = appCtxt.multiAccounts ? treeView._overview.account : null;
227 		if (!appCtxt.get(ZmSetting.TASKS_ENABLED, null, account)) { return []; }
228 
229 		var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT, account);
230 		var root = treeView.getTreeItemById(rootId);
231 		if (root) {
232 			var totalItems = [];
233 			this._getSubItems(root, totalItems);
234 			return totalItems;
235 		}
236 	}
237 	return [];
238 };
239 
240 ZmTaskTreeController.prototype._getSubItems =
241 function(root, totalItems) {
242 	if (!root || (root && root._isSeparator)) { return; }
243 
244 	var items = root.getItems();
245 	for (var i in items) {
246 		var item = items[i];
247 		if (item && !item._isSeparator) {
248 			totalItems.push(item);
249 			this._getSubItems(item, totalItems);
250 		}
251 	}
252 };
