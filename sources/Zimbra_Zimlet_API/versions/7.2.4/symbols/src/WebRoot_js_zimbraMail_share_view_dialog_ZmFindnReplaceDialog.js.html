 20 /**
 21  * Creates a "find-and-replace" dialog.
 22  * @class
 23  * This class represents a "find-and-replace" dialog.
 24  * 
 25  * @param	{DwtControl}	shell		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtDialog
 29  */
 30 ZmFindnReplaceDialog = function(shell, className) {
 31 	className = className || "ZmFindnReplaceDialog";
 32 	
 33 	var findBtn = new DwtDialog_ButtonDescriptor(ZmFindnReplaceDialog.FIND_BUTTON, 
 34 													  ZmMsg.find, DwtDialog.ALIGN_LEFT);
 35 	var replaceBtn = new DwtDialog_ButtonDescriptor(ZmFindnReplaceDialog.REPLACE_BUTTON, 
 36 													ZmMsg.replace, DwtDialog.ALIGN_LEFT);
 37 	var replaceAllBtn = new DwtDialog_ButtonDescriptor(ZmFindnReplaceDialog.REPLACE_ALL_BUTTON, 
 38 													ZmMsg.replaceAll, DwtDialog.ALIGN_LEFT);
 39 	
 40 	DwtDialog.call(this, {parent:shell, className:className, title:ZmMsg.findNReplaceTitle,
 41 						  standardButtons:[DwtDialog.CANCEL_BUTTON], extraButtons:[findBtn,replaceBtn,replaceAllBtn]});
 42 
 43 	this._findId = Dwt.getNextId();
 44 	this._replaceId = Dwt.getNextId();
 45 	this._dirId = Dwt.getNextId();
 46 	this._dirIdUp = Dwt.getNextId();
 47 	this._dirIdDown = Dwt.getNextId();
 48 	this._caseId = Dwt.getNextId();
 49 	this._wholeWordId = Dwt.getNextId();
 50 	this._messageId = Dwt.getNextId();
 51 	
 52 	var numberOfCols =1;
 53 		var html = [
 54          "<table><tr><td>",
 55          "<div style='padding:2px;' id='",this._messageId,"' style='padding-left:30px;'></div>",
 56          "<table border='0'>",
 57          "<tr><td class='Label' align='left'>",ZmMsg.findWhatLabel,"</td>",
 58          "<td colspan=2 id='",this._findId ,"'></td></tr>",
 59          "<tr><td class='Label' align='left'>",ZmMsg.replaceWithLabel,"</td>",
 60          "<td colspan=2 id='",this._replaceId ,"'></td></tr>",   
 61          "<tr><td class='Label' align='left'>",ZmMsg.directionLabel,"</td>",
 62          "<td colspan=2 id='",this._dirId ,"' align='left'>",
 63          	"<table cellpadding='3'><tr>",
 64          	"<td><input type='radio' id='",this._dirIdUp,"' name='",this._dirId,"' value='up'></td>","<td class='Label'>",ZmMsg.upLabel,"</td>",
 65          	"<td><input type='radio' id='",this._dirIdDown,"' name='",this._dirId,"' value='down' checked></td>","<td class='Label'>",ZmMsg.downLabel,"</td>",
 66          	"</tr></table>",
 67          "</td></tr>",        
 68          "<tr><td colspan='3'>",
 69 		   	"<table cellpadding='3'><tr>",
 70 		   	"<td class='Label' align='right'><input type='checkbox' id='",this._caseId,"'>",
 71 		   	"<td class='Label' align='left'>",ZmMsg.caseSensitive,"</td>",
 72          	"</tr></table>",
 73 		 "</td></tr>",		         
 74          "</table>",          
 75          "</td></tr></table>"].join("");
 76      
 77 	this.setContent(html);
 78 	// set view
 79 	this.setView(this._createView());
 80 	this.registerCallback(ZmFindnReplaceDialog.FIND_BUTTON, this._handleFindButton, this);
 81 	this.registerCallback(ZmFindnReplaceDialog.REPLACE_BUTTON, this._handleReplaceButton, this);
 82 	this.registerCallback(ZmFindnReplaceDialog.REPLACE_ALL_BUTTON, this._handleReplaceAllButton, this);
 83 };

 95 /**
 96  * Pops-up the dialog.
 97  * 
 98  * @param	{Hash}		editorInfo		a hash of editor info
 99  * @param	{AjxCallback}	callback	the callback
100  */
101 ZmFindnReplaceDialog.prototype.popup =
102 function(editorInfo, callback) {
103 	this._editorInfo = editorInfo || {};
104 	this._callback = callback;
105     var findVal = "";
106     if(editorInfo) {
107         var editor = editorInfo.editor;
108         findVal = editor._getSelectedText();
109         if(findVal) {
110             findVal = AjxStringUtil.trim(findVal.toString());
111         }
112     }
113     this._findInput.setValue(findVal);
114     this._replaceInput.setValue("");    
115     DwtDialog.prototype.popup.call(this);
116 };
117 
118 ZmFindnReplaceDialog.prototype.popdown =
119 function() {
120 	if (this._acPageList) {
121 		this._acPageList.show(false);
122 	}
123 	DwtDialog.prototype.popdown.call(this);
124 };

126 ZmFindnReplaceDialog._handleKeyPress = function(ev){
127     var inputField = DwtControl.getTargetControl(ev);
128 	var charCode = DwtKeyEvent.getCharCode(ev);
129 	if (charCode == 13 || charCode == 3) {		
130 		var dialog = inputField.parent.parent;
131 		dialog.replaceAction('none',true);
132 	    return false;
133 	}
134 	return true;	
135 };

139 ZmFindnReplaceDialog.prototype._createView =
140 function() {
141 
142 	var view = new DwtComposite(this);
143 	var inputParams = {
144 		parent: view,
145 		type: DwtInputField.STRING,
146 		validationStyle: DwtInputField.CONTINUAL_VALIDATION
147 	}
148 
149 	// create common DWT controls
150 	this._findInput = new DwtInputField(inputParams);
151 	this._findInput.reparentHtmlElement(this._findId);
152 	this._replaceInput = new DwtInputField(inputParams);
153 	this._replaceInput.reparentHtmlElement(this._replaceId);
154 
155 	Dwt.setHandler(this._findInput.getInputElement(), DwtEvent.ONKEYPRESS, ZmFindnReplaceDialog._handleKeyPress);
156 	// create properties
157 	
158 	return view;
159 };
160 
161 ZmFindnReplaceDialog.prototype._handleFindButton =
162 function(event) {
163 	this.replaceAction('none',true);
164 };
165 
166 ZmFindnReplaceDialog.prototype._handleReplaceButton =
167 function() {
168 	this.replaceAction('current',false);
169 };
170 
171 ZmFindnReplaceDialog.prototype._handleReplaceAllButton =
172 function() {
173 	this.replaceAction('all',false);
174 };

189 ZmFindnReplaceDialog.prototype.replaceAction =
190 function(mode,findOnly)
191 {
192 	var findVal = this._findInput.getValue();
193 	var replaceVal = (findOnly? null : this._replaceInput.getValue());
194 	var radioBtns = document.getElementById(this._dirIdUp);
195 	var casesensitiveVal = false;
196 	var backwardsVal = false;
197 	if(radioBtns && radioBtns.checked){
198 		backwardsVal = true;
199 	}
200 	this._caseCheckbox = document.getElementById(this._caseId);
201 	
202 	if(this._caseCheckbox && this._caseCheckbox.checked) {
203 		casesensitiveVal = true;
204 	}	
205 	
206 	var params = {
207 			searchstring: findVal,
208 			replacestring: replaceVal,
209 			replacemode : mode,
210 			casesensitive : casesensitiveVal,
211 			backwards : backwardsVal
212 		};
213 	if(this._editorInfo.editor){
214 		var editor = this._editorInfo.editor;
215 		if(AjxEnv.iIE){
216 		editor.focus();
217 		}
218 		editor.searchnReplace(params);			
219 	}
220 	
221 	if (this._callback) {		
222 		this._callback.run(params);		
223 	}
224 };
