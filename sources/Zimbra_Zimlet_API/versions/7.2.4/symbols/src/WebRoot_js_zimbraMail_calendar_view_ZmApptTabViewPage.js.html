 15 /**
 16  * Creates a new appointment tab.
 17  * @constructor
 18  * @class
 19  * This is the main screen for creating/editing an appointment. It provides inputs
 20  * for the various appointment details.
 21  *
 22  * @author Parag Shah
 23  *
 24  * @param {DwtComposite}	parent			the appt compose view
 25  * @param {Hash}	attendees			the attendees/locations/equipment
 26  * @param {ZmApptComposeController}	controller		the appt compose controller
 27  * @param {Hash}	dateInfo			the hash of date info
 28  * 
 29  * @extends		DwtTabViewPage
 30  * @private
 31  */
 32 ZmApptTabViewPage = function(parent, attendees, controller, dateInfo) {
 33 	if (arguments.length == 0) return;
 34 
 35 	DwtTabViewPage.call(this, parent);
 36 
 37 	this._controller = controller;
 38 	this._editView = new ZmApptEditView(this, attendees, controller, dateInfo);
 39 };
 40 
 41 ZmApptTabViewPage.prototype = new DwtTabViewPage;
 42 ZmApptTabViewPage.prototype.constructor = ZmApptTabViewPage;
 43 
 44 ZmApptTabViewPage.prototype.toString =
 45 function() {
 46 	return "ZmApptTabViewPage";
 47 };
 48 
 49 
 50 // Public
 51 
 52 ZmApptTabViewPage.prototype.showMe =
 53 function() {
 54 	if (!this._editView.isRendered()) return;
 55 
 56 	this._editView.show();
 57 	this.parent.tabSwitched(this._tabKey);
 58 	this._controller._setComposeTabGroup(true);
 59 };
 60 
 61 ZmApptTabViewPage.prototype.tabBlur =
 62 function(useException) {
 63 	this._editView.blur();
 64 };
 65 
 66 ZmApptTabViewPage.prototype.getEditView =
 67 function() {
 68 	return this._editView;
 69 };
 70 
 71 ZmApptTabViewPage.prototype.initialize =
 72 function(appt, mode, isDirty, apptComposeMode) {
 73 	this._editView.initialize(appt, mode, isDirty, apptComposeMode);
 74 };
 75 
 76 ZmApptTabViewPage.prototype.isDirty =
 77 function(excludeAttendees) {
 78 	return this._editView.isDirty(excludeAttendees);
 79 };
 80 
 81 ZmApptTabViewPage.prototype.cleanup =
 82 function() {
 83 	this._editView.cleanup();
 84 };
 85 
 86 ZmApptTabViewPage.prototype.createHtml =
 87 function() {
 88 	this._editView.createHtml();
 89 };
 90 
 91 ZmApptTabViewPage.prototype.resize =
 92 function(newWidth, newHeight) {
 93     if(newHeight)
 94         newHeight = newHeight - 35;
 95 
 96 	this._editView.resize(newWidth, newHeight);
 97 };
 98 
 99 ZmApptTabViewPage.prototype.isValid =
100 function() {
101 	return this._editView.isValid();
102 };
103 
104 ZmApptTabViewPage.prototype.getTabPage =
105 function(id) {
106 	return this.parent.getTabPage(id);
107 };
