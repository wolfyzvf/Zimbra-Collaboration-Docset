 21 /**
 22  * Creates the search result
 23  * @class
 24  * This class represents a search result.
 25  * 
 26  * @param	{ZmSearch}	search		the search
 27  */
 28 ZmSearchResult = function(search) {
 29 	if (!search) { return; }
 30 	this._results = {};
 31 	this.search = search;
 32 };
 33 
 34 /**
 35  * Returns a string representation of the object.
 36  * 
 37  * @return		{String}		a string representation of the object
 38  */
 39 ZmSearchResult.prototype.toString = 
 40 function() {
 41 	return "ZmSearchResult";
 42 };
 43 
 44 /**
 45  * @private
 46  */
 47 ZmSearchResult.prototype.dtor = 
 48 function() {
 49 	for (var type in this._results) {
 50 		if (this._results[type].clear) {
 51 			this._results[type].clear();
 52 			this._results[type] = null;
 53 		}
 54 	}
 55 	this._results = null;
 56 };
 57 
 58 /**
 59  * Gets the results.
 60  * 
 61  * @param	{constant}	type	the type
 62  * @return	{Array}	an array of results
 63  */
 64 ZmSearchResult.prototype.getResults =
 65 function(type) {
 66 
 67 	type = type || this.type;
 68 	if (!this._results) {
 69 		// probably got an exception - return an empty list
 70 		return ZmItem.RESULTS_LIST[type](this.search);
 71 	}
 72 	if (type == ZmItem.MIXED) {
 73 		// mail list - lazy way to make it easy to do mail ops on items
 74 		var list = new ZmMailList(ZmItem.MIXED, this.search);
 75 		for (var type in this._results) {
 76 			var results = this._results[type];
 77 			if (results && results.size()) {
 78 				var a = results.getArray();
 79 				for (var j = 0; j < a.length; j++) {
 80 					list.add(a[j]);
 81 				}
 82 			}
 83 		}
 84 		return list;
 85 	} else if (this.search.idsOnly) {
 86 		return this._results;
 87 	} else {
 88 		// if we don't have results for the requested type, the search was probably for the wrong type
 89 		return this._results[type] ? this._results[type] : type && ZmItem.RESULTS_LIST[type](this.search);
 90 	}
 91 };

 93 /**
 94  * Gets the attribute.
 95  * 
 96  * @param	{String}	name		the attribute name
 97  * @return	{Object}	the attribute
 98  */
 99 ZmSearchResult.prototype.getAttribute = 
100 function(name) {
101 	return this._respEl ? this._respEl[name] : null;
102 };

104 /**
105  * Sets the response.
106  * 
107  * @private
108  */
109 ZmSearchResult.prototype.set =
110 function(respEl) {
111 
112 	if (!this.search) { return; }
113 
114 	this._respEl = respEl;
115 
116 	// <match> objects are returned for autocomplete search, not items; let caller handle them
117 	if (this.search.isAutocompleteSearch) { return; }
118 
119 	var foundType = {};
120 	var numTypes = 0;
121 	var currentType, defaultType;
122 	var isGalSearch = this.search.isGalSearch;
123 	
124 	var _st = new Date();
125 	var count = 0;
126 	if (isGalSearch || this.search.isCalResSearch) {
127 		// process JS eval result for SearchGalRequest
128 		currentType = defaultType = isGalSearch ? ZmItem.CONTACT : ZmItem.RESOURCE;
129 		var data = isGalSearch ? respEl.cn : respEl.calresource;
130 		if (data) {
131 			if (!this._results[currentType]) {
132 				// create list as needed - may invoke package load
133 				this._results[currentType] = ZmItem.RESULTS_LIST[currentType](this.search);
134 			}
135 			for (var j = 0; j < data.length; j++) {
136 				this._results[currentType].addFromDom(data[j]);
137 			}
138 
139 			count = data.length;
140 		}
141 	} else if (this.search.idsOnly) {
142 		this._results = respEl.hit || [];
143 		return;
144 	} else {
145 		// process JS eval result for SearchResponse
146 		var types = this.search.types.getArray();
147 		defaultType = types[0];
148 
149 		// bug fix #44232 - resolve default type if none provided
150 		if (!defaultType) {
151 			var allTypes = AjxUtil.values(ZmList.NODE);
152 			for (var i = 0; i < allTypes.length; i++) {
153 				var t = allTypes[i];
154 				if (respEl[t]) {
155 					defaultType = ZmList.ITEM_TYPE[t];
156 					if (types && types.length == 0) {
157 						types = [defaultType];
158 					}
159 					break;
160 				}
161 			}
162 		}
163 
164 		if (!defaultType) {
165 			var curApp = appCtxt.getCurrentAppName();
166 			var types = ZmApp.SEARCH_TYPES[curApp];
167 			defaultType = types && types.length && types[0];
168 		}
169 
170 		for (var i = 0; i < types.length; i++) {
171 			var type = types[i];
172 			var data = respEl[ZmList.NODE[type]];
173 
174 			// A chat isa message. Futz with the types to deal with this.
175 			// (Eventually we'll avoid this problem by displying chat history in im app.)
176 			if (!data && (type == ZmItem.MSG)) {
177 				data = respEl["chat"];
178 			}
179 
180 			// do a bunch of sanity checks
181 			if (data && data.length) {
182 				count += data.length;
183 				if (!this._results[type]) {
184 					// create list as needed - may invoke package load
185 					this._results[type] = ZmItem.RESULTS_LIST[type](this.search);
186 				}
187 				for (var j = 0; j < data.length; j++) {
188 					var item = data[j];
189 					item._type = type;
190 					this._results[type].addFromDom(item);
191 				}
192 
193 				if (!foundType[type]) {
194 					foundType[type] = true;
195 					numTypes++;
196 					currentType = type;
197 				}
198 			}
199 		}
200 	}
201 	if (!count && defaultType) {
202 		this._results[defaultType] = ZmItem.RESULTS_LIST[defaultType](this.search);
203 	}
204 	if ((isGalSearch || this.search.isGalAutocompleteSearch) && this._results[ZmItem.CONTACT]) {
205 		this._results[ZmItem.CONTACT].setIsGal(true);
206 	}
207 	if (this.search.isGalAutocompleteSearch) {
208 		this.isTokenized = (this._respEl.tokenizeKey != null);
209 	}
210 	
211 	var _en = new Date();
212 	DBG.println(AjxDebug.DBG1, "TOTAL PARSE TIME for " + count + " NODES: " + (_en.getTime() - _st.getTime()));
213 
214 	currentType = currentType || defaultType;
215 	if (numTypes <= 1) {
216 		this.type = currentType;
217 	} else if (numTypes == 2 && (foundType[ZmItem.PAGE] || foundType[ZmItem.DOCUMENT])) {
218 		this.type = ZmItem.PAGE;	// notebook search can return either/both
219 	} else {
220 		this.type = appCtxt.get(ZmSetting.MIXED_VIEW_ENABLED) ? ZmItem.MIXED : currentType;
221 	}
222 
223 	return this.type;
224 };

226 /**
227  * @private
228  */
229 ZmSearchResult._sortGalResults =
230 function(a, b) {
231 	var af = a.getFileAs && a.getFileAs().toLowerCase();
232 	var bf = b.getFileAs && b.getFileAs().toLowerCase();
233 	return af < bf ? -1 : (af > bf ? 1 : 0);
234 };
