 59 /**
 60  * Converts the given text to mixed-case. The input text is one or more words
 61  * separated by spaces. The output is a single word in mixed (or camel) case.
 62  * 
 63  * @param {string}	text		text to convert
 64  * @param {string|RegEx}	sep		text separator (defaults to any space)
 65  * @param {boolean}	camel		if <code>true</code>, first character of result is lower-case
 66  * @return	{string}	the resulting string
 67  */
 68 AjxStringUtil.toMixed =
 69 function(text, sep, camel) {
 70 	if (!text || (typeof text != "string")) { return ""; }
 71 	sep = sep || /\s+/;
 72 	var wds = text.split(sep);
 73 	var newText = [];
 74 	newText.push(camel ? wds[0].toLowerCase() : wds[0].substring(0, 1).toUpperCase() + wds[0].substring(1).toLowerCase());
 75 	for (var i = 1; i < wds.length; i++) {
 76 		newText.push(wds[i].substring(0, 1).toUpperCase() + wds[i].substring(1).toLowerCase());
 77 	}
 78 	return newText.join("");
 79 };

 94 /**
 95  * Removes white space from the beginning and end of a string, optionally compressing internal white space. By default, white
 96  * space is defined as a sequence of  Unicode whitespace characters (\s in regexes). Optionally, the user can define what
 97  * white space is by passing it as an argument.
 98  *
 99  * <p>TODO: add left/right options</p>
100  *
101  * @param {string}	str      	the string to trim
102  * @param {boolean}	compress 	whether to compress internal white space to one space
103  * @param {string}	space    	a string that represents a user definition of white space
104  * @return	{string}	a trimmed string
105  */
106 AjxStringUtil.trim =
107 function(str, compress, space) {
108 
109 	if (!str) {return "";}
110 
111 	var trim_re = AjxStringUtil.TRIM_RE;
112 
113 	var compress_re = AjxStringUtil.COMPRESS_RE;
114 	if (space) {
115 		trim_re = new RegExp("^" + space + "+|" + space + "+$", "g");
116 		compress_re = new RegExp(space + "+", "g");
117 	} else {
118 		space = " ";
119 	}
120 	str = str.replace(trim_re, '');
121 	if (compress) {
122 		str = str.replace(compress_re, space);
123 	}
124 
125 	return str;
126 };

165 /**
166 * Splits a string, ignoring delimiters that are in quotes or parentheses. Comma
167 * is the default split character, but the user can pass in a string of multiple
168 * delimiters. It can handle nested parentheses, but not nested quotes.
169 *
170 * <p>TODO: handle escaped quotes</p>
171 *
172 * @param {string} str	the string to split
173 * @param {string}	[dels]	an optional string of delimiter characters
174 * @return	{array}	an array of strings
175 */
176 AjxStringUtil.split =
177 function(str, dels) {
178 
179 	if (!str) {return [];}
180 	var i = 0;
181 	dels = dels ? dels : ',';
182 	var isDel = new Object();
183 	if (typeof dels == 'string') {
184 		isDel[dels] = 1;
185 	} else {
186 		for (i = 0; i < dels.length; i++) {
187 			isDel[dels[i]] = 1;
188 		}
189 	}
190 
191 	var q = false;
192 	var p = 0;
193 	var start = 0;
194 	var chunk;
195 	var chunks = [];
196 	var j = 0;
197 	for (i = 0; i < str.length; i++) {
198 		var c = str.charAt(i);
199 		if (c == '"') {
200 			q = !q;
201 		} else if (c == '(') {
202 			p++;
203 		} else if (c == ')') {
204 			p--;
205 		} else if (isDel[c]) {
206 			if (!q && !p) {
207 				chunk = str.substring(start, i);
208 				chunks[j++] = chunk;
209 				start = i + 1;
210 			}
211 		}
212 	}
213 	chunk = str.substring(start, str.length);
214 	chunks[j++] = chunk;
215 
216 	return chunks;
217 };

240 /**
241  * Wraps text to the given length and optionally quotes it. The level of quoting in the
242  * source text is preserved based on the prefixes. Special lines such as email headers
243  * always start a new line.
244  *
245  * @param {hash}	params	a hash of parameters
246  * @param {string}      params.text 				the text to be wrapped
247  * @param {number}      [params.len=80]				the desired line length of the wrapped text, defaults to 80
248  * @param {string}      [params.pre]				an optional string to prepend to each line (useful for quoting)
249  * @param {string}      [params.before]				text to prepend to final result
250  * @param {string}      [params.after]				text to append to final result
251  * @param {boolean}		[params.preserveReturns]	if true, don't combine small lines
252  *
253  * @return	{string}	the wrapped/quoted text
254  */
255 AjxStringUtil.wordWrap =
256 function(params) {
257 
258 	if (!(params && params.text)) { return ""; }
259 
260 	var text = params.text;
261 	var before = params.before || "", after = params.after || "";
262 
263 	// For HTML, just surround the content with the before and after, which is
264 	// typically a block-level element that puts a border on the left
265 	if (params.htmlMode) {
266 		return [before, text, after].join("");
267 	}
268 
269 	var len = params.len || 80;
270 	var pre = params.pre || "";
271 	var eol = "\n";
272 
273 	text = AjxStringUtil.trim(text, false, "[\t ]");
274 	text = text.replace(/\r\n/g, eol);
275 	var lines = text.split(eol);
276 	var words = [];
277 
278 	// Divides lines into words. Each word is part of a hash that also has
279 	// the word's prefix, whether it's a paragraph break, and whether it's
280 	// special (cannot be wrapped into a previous line)
281 	for (var l = 0, llen = lines.length; l < llen; l++) {
282 		var line = lines[l];
283 		// get this line's prefix
284 		var m = line.match(/^([\s>\|]+)/);
285 		var prefix = m ? m[1] : "";
286 		if (prefix) {
287 			line = line.substr(prefix.length);
288 		}
289 		var wds = AjxStringUtil.splitKeepLeadingWhitespace(line);
290 		if (wds && wds[0] && wds[0].length) {
291 			var isSpecial = AjxStringUtil.MSG_SEP_RE.test(line) || AjxStringUtil.COLON_RE.test(line) ||
292 							AjxStringUtil.HDR_RE.test(line) || AjxStringUtil.SIG_RE.test(line);
293 			for (var w = 0, wlen = wds.length; w < wlen; w++) {
294 				var lastWord = params.preserveReturns && (w == wlen - 1);
295 				words.push({w:wds[w], p:prefix, special:(isSpecial && w == 0), lastWord:lastWord});
296 			}
297 		} else {
298 			words.push({para:true, p:prefix});	// paragraph marker
299 		}
300 	}
301 
302 	// Take the array of words and put them back together. We break for a new line
303 	// when we hit the max line length, change prefixes, or hit a special word.
304 	var max = params.len || 72;
305 	var addPrefix = params.pre || "";
306 	var apl = addPrefix.length;
307 	var result = "", curLen = 0, wds = [], curP = null;
308 	for (var i = 0, len = words.length; i < len; i++) {
309 		var word = words[i];
310 		var w = word.w, p = word.p;
311 		var sp = wds.length ? 1 : 0;
312 		var pl = (curP === null) ? 0 : curP.length;
313 		if (word.para) {
314 			// paragraph break - output what we have, add a blank line
315 			if (wds.length) {
316 				result += addPrefix + (curP || "") + wds.join("").replace(/^ /,"") + eol;
317 			}
318 			result += addPrefix + p + eol;
319 			wds = [];
320 			curLen = 0;
321 			curP = null;
322 		} else if ((apl + pl + curLen + sp + w.length <= max) && (p == curP || curP === null) && !word.special) {
323 			// still room left on the current line, add the word
324 			wds.push(w);
325 			curLen += w.length;
326 			curP = p;
327 			if (word.lastWord && words[i + 1]) {
328 				words[i + 1].special = true;
329 			}
330 		} else {
331 			// output what we have and start a new line
332 			if (wds.length) {
333 				result += addPrefix + (curP || "") + wds.join("").replace(/^ /,"") + eol;
334 			}
335 			wds = [w];
336 			curLen = w.length;
337 			curP = p;
338 			if (word.lastWord && words[i + 1]) {
339 				words[i + 1].special = true;
340 			}
341 		}
342 	}
343 
344 	// handle last line
345 	if (wds.length) {
346 		result += addPrefix + curP + wds.join("").replace(/^ /,"") + eol;
347 	}
348 
349 	return [before, result, after].join("");
350 };

352 /**
353  * Quotes text with the given quote character. For HTML, surrounds the text with the
354  * given strings. Does no wrapping.
355  *
356  * @param {hash}	params	a hash of parameters
357  * @param {string}      params.text 				the text to be wrapped
358  * @param {string}      [params.pre]				prefix for quoting
359  * @param {string}      [params.before]				text to prepend to final result
360  * @param {string}      [params.after]				text to append to final result
361  *
362  * @return	{string}	the quoted text
363  */
364 AjxStringUtil.quoteText =
365 function(params) {
366 
367 	if (!(params && params.text)) { return ""; }
368 
369 	var text = params.text;
370 	var before = params.before || "", after = params.after || "";
371 
372 	// For HTML, just surround the content with the before and after, which is
373 	// typically a block-level element that puts a border on the left
374 	if (params.htmlMode || !params.pre) {
375 		return [before, text, after].join("");
376 	}
377 
378 	var len = params.len || 80;
379 	var pre = params.pre || "";
380 	var eol = "\n";
381 
382 	text = AjxStringUtil.trim(text);
383 	text = text.replace(/\n\r/g, eol);
384 	var lines = text.split(eol);
385 	var result = [];
386 
387 	for (var l = 0, llen = lines.length; l < llen; l++) {
388 		var line = AjxStringUtil.trim(lines[l]);
389 		result.push(pre + line + eol);
390 	}
391 
392 	return before + result.join("") + after;
393 };

577 /**
578  * Converts the string to HTML.
579  * 
580  * @param	{string}	str		the string
581  * @return	{string}	the resulting string
582  */
583 AjxStringUtil.convertToHtml =
584 function(str, quotePrefix) {
585 	if (!str) {return "";}
586 
587 	if (quotePrefix) {
588 		// Convert a section of lines prefixed with > or |
589 		// to a section encapsuled in <blockquote> tags
590 		var prefix_re = /^(>|\|\s+)/;
591 		var lines = str.split(/\r?\n/);
592 		var level = 0;
593 		for (var i=0; i<lines.length; i++) {
594 			var line = lines[i];
595 			if (line.length > 0) {
596 				var lineLevel = 0;
597 				while (line.match(prefix_re)) { // Remove prefixes while counting how many there are on the line
598 					line = line.replace(prefix_re,"");
599 					lineLevel++;
600 				}
601 				while (lineLevel > level) { // If the lineLevel has changed since the last line, add blockquote start or end tags, and adjust level accordingly
602 					line = "<blockquote>" + line;
603 					level++;
604 				}
605 				while (lineLevel < level) {
606 					line = line + "</blockquote>";
607 					level--;
608 				}
609 			}
610 			lines[i] = line;
611 		}
612 		while (level > 0) {
613 			lines.push("</blockquote>");
614 			level--;
615 		}
616 
617 		str = lines.join("\n");
618 		str = str.replace(/&/mg, "&");
619 		
620 		str = str.replace(/<(?!\/?blockquote)/mg, "<"); // Replace "<" only if it is not followed by "blockquote"
621 
622 		str = str.replace(/(\/?blockquote)?>/mg, function($0, $1) { // Replace ">" only if it is not preceded by "blockquote"
623 			return $1 ? $0 : '>';
624 		});
625 
626 	} else {
627 		str = str
628 			.replace(/&/mg, "&")
629 			.replace(/</mg, "<")
630 			.replace(/>/mg, ">")
631 	}
632 		
633 
634 	str = str
635 		.replace(/  /mg, "  ")

772 /**
773  * Forces a string to wrap at "pixelWidth" using "className" on hidden 'AjxStringUtil._calcDIV'.
774  * Returns "origString" with "<br>" tags inserted to force wrapping.
775  * Breaks string on embedded space characters, EOL ("/n") and "<br>" tags when possible.
776  *
777  * @returns		"origString" with "<br>" tags inserted to force wrapping.
778  * 
779  * @private
780  */
781 AjxStringUtil.wrap =
782 function(origString, pixelWidth, className) {
783 	var calcDIV = AjxStringUtil.calcDIV();
784 	if (arguments.length == 3) calcDIV.className = className;
785 
786 	var newString = "";
787 	var newLine = "";
788 	var textRows = origString.split("/n");
789 	for (var trCount = 0; trCount < textRows.length; trCount++) {
790 		if (trCount != 0) {
791 			newString += newLine + "<br>";
792 			newLine = "";
793 		}
794 		htmlRows = textRows[trCount].split("<br>");
795 		for (var hrCount=0; hrCount<htmlRows.length; hrCount++) {
796 			if (hrCount != 0) {
797 				newString += newLine + "<br>";
798 				newLine = "";
799 			}
800 			words = htmlRows[hrCount].split(" ");
801 			var wCount=0;
802 			while (wCount<words.length) {
803 				calcDIV.innerHTML = newLine + " " + words[wCount];
804 				var newLinePixels = calcDIV.offsetWidth;
805 				if (newLinePixels > pixelWidth) {
806 					// whole "words[wCount]" won't fit on current "newLine" - insert line break, avoid incrementing "wCount"
807 					calcDIV.innerHTML = words[wCount];
808 					newLinePixels = newLinePixels - calcDIV.offsetWidth;
809 					if ( (newLinePixels >= pixelWidth) || (calcDIV.offsetWidth <= pixelWidth) ) {
810 						// either a) excess caused by <space> character or b) will fit completely on next line
811 						// so just break without incrementing "wCount" and append next time
812 						newString += newLine + "<br>";
813 						newLine = "";
814 					}
815 					else { // must break "words[wCount]"
816 						var keepLooping = true;
817 						var atPos = 0;
818 						while (keepLooping) {
819 							atPos++;
820 							calcDIV.innerHTML = newLine + " " + words[wCount].substring(0,atPos);
821 							keepLooping = (calcDIV.offsetWidth <= pixelWidth);
822 						}
823 						atPos--;
824 						newString += newLine + words[wCount].substring(0,atPos) + "<br>";
825 						words[wCount] = words[wCount].substr(atPos);
826 						newLine = "";
827 					}
828 				} else { // doesn't exceed pixelWidth, append to "newLine" and increment "wCount"
829 					newLine += " " + words[wCount];
830 					wCount++;
831 				}
832 			}
833 		}
834 	}
835 	newString += newLine;
836 	return newString;
837 };

857 /**
858  * Returns a list of chunks of top-level content in a message body. Top-level
859  * content is what was actually typed by the sender. We attempt to exclude quoted
860  * content and signatures.
861  *
862  * The following lines/blocks (and variants) and any text after them are ignored:
863  *
864  * 		----- Original Message -----
865  *
866  * 		----- Forwarded Message -----
867  *
868  *		--
869  *		some signature text
870  *
871  *		______________________________		|
872  *											| Outlook 2003 does this
873  *		From:								|
874  *
875  * Lines that begin with a prefix character (">" or "|") are ignored. The
876  * following lines/blocks are ignored if they precede a line that begins with a
877  * prefix character:
878  *
879  * 		Fred Flintstone <fred@bedrock.org> wrote:
880  *
881  * 		Fred Flintstone <fred@bedrock.org> wrote:
882  * 		[snipped]
883  *
884  * Since quoted text may be interleaved with original text, we may return several
885  * chunks of original text. That is so they may be separated when they are quoted.
886  *
887  * @param {string}	text		a message body
888  * 
889  * @private
890  */
891 AjxStringUtil.getTopLevel =
892 function(text, eol, htmlMode) {
893 
894 	if (!text) { return [""]; }
895 	var isHtml = /<br|<div/i.test(text);
896 	var split = isHtml ? AjxStringUtil.HTML_BR_RE : AjxStringUtil.SPLIT_RE;
897 	var eol = isHtml ? '<br>' : '\n';
898 	text = AjxStringUtil._trimBlankLines(text, split, eol, isHtml);
899 	var lines = text.split(split);
900 	var len = lines.length;
901 	var i = 0, start = 0;
902 	var chunks = [];
903 	var skipping = false;
904 
905 	while (i < len) {
906 		var wasSkipping = skipping;
907 		var skip = AjxStringUtil._linesToSkip(lines, i, htmlMode);
908 		skipping = (skip > 0);
909 		if (wasSkipping && !skipping) {
910 			start = i;
911 		} else if (!wasSkipping && skipping && i > start) {
912 			var chunk = AjxStringUtil._trimBlankLines(lines.slice(start, i).join(eol), split, eol, htmlMode);
913 			if (chunk && chunk.length) {
914 				chunks.push(chunk);
915 			}
916 		}
917 		i += skipping ? skip : 1;
918 	}
919 
920 	if (!skipping && i > start) {
921 		var chunk = AjxStringUtil._trimBlankLines(lines.slice(start, i).join(eol), split, eol, htmlMode);
922 		if (chunk && chunk.length) {
923 			chunks.push(chunk);
924 		}
925 	}
926 
927 	return chunks;
928 };

930 /**
931  * Starting at a given line, returns the number of lines that should be skipped because
932  * they are quoted (or signature) content.
933  * 
934  * @param {array}	lines		lines of text
935  * @param {number}	i			index of current line
936  * @param {boolean}	htmlMode	if <code>true</code>, text is HTML
937  * 
938  * @private
939  */
940 AjxStringUtil._linesToSkip =
941 function(lines, i, htmlMode) {
942 	var len = lines.length;
943 	var skip = 0;
944 	var start = i;
945 	var line = htmlMode ? AjxStringUtil.stripTags(lines[i]) : lines[i];
946 	
947 	if (AjxStringUtil.MSG_SEP_RE.test(line)) {
948 		skip = len - i;
949 	} else if (AjxStringUtil.PREFIX_RE.test(line)) {
950 		while (i < lines.length && (AjxStringUtil.PREFIX_RE.test(line) || AjxStringUtil.BLANK_RE.test(line))) {
951 			i++;
952 			line = htmlMode ? AjxStringUtil.stripTags(lines[i]) : lines[i];
953 		}
954 		skip = i - start;
955 	} else if (AjxStringUtil.HTML_QUOTE_PRE_RE.test(line) && line.indexOf(AjxStringUtil.HTML_QUOTE_COLOR) != -1) {
956 		while (i < lines.length && (!AjxStringUtil.HTML_QUOTE_POST_RE.test(line))) {
957 			i++;
958 			line = lines[i];
959 		}
960 		if (i < lines.length) {
961 			lines[i] = lines[i].replace(AjxStringUtil.HTML_QUOTE_POST_RE, "");
962 		}
963 		skip = i - start;
964 	} else if (AjxStringUtil.COLON_RE.test(line)) {
965 		var idx = AjxStringUtil._nextNonBlankLineIndex(lines, i + 1, htmlMode);
966 		var line1 = (idx == -1) ? null : htmlMode ? AjxStringUtil.stripTags(lines[idx]) : lines[idx];
967 		if (line1 && AjxStringUtil.PREFIX_RE.test(line1)) {
968 			skip = idx - i;
969 		} else {
970 			if (idx != -1) {
971 				idx = AjxStringUtil._nextNonBlankLineIndex(lines, idx + 1, htmlMode);
972 			}
973 			var line2 = (idx == -1) ? null : htmlMode ? AjxStringUtil.stripTags(lines[idx]) : lines[idx];
974 			if (line2 && AjxStringUtil.BRACKET_RE.test(line1) && AjxStringUtil.PREFIX_RE.test(line2)) {
975 				skip = idx - i;
976 			}
977 		}
978 	} else if (AjxStringUtil.LINE_RE.test(line)) {
979 		var idx = AjxStringUtil._nextNonBlankLineIndex(lines, i + 1, htmlMode);
980 		var line1 = (idx == -1) ? null : htmlMode ? AjxStringUtil.stripTags(lines[idx]) : lines[idx];
981 		if (line1 && AjxStringUtil.HDR_RE.test(line1)) {
982 			skip = len - i;
983 		}
984 	} else if (AjxStringUtil.SIG_RE.test(line)) {
985 		skip = len - i;
986 	}
987 	return skip;
988 };

1080 AjxStringUtil._traverse =
1081 function(el, text, idx, listType, listLevel, bulletNum, ctxt, convertor, onlyOneNewLinePerP) {
1082 
1083 	var nodeName = el.nodeName.toLowerCase();
1084 
1085 	var result = null;
1086 	if (convertor && convertor[nodeName]) {
1087 		result = convertor[nodeName](el,ctxt);
1088 	}
1089 
1090 	if (result != null) {
1091 		text[idx++] = result;
1092 	} else if (nodeName == "#text") {
1093 		if (el.nodeValue.search(AjxStringUtil._NON_WHITESPACE) != -1) {
1094 			if (ctxt.lastNode == "ol" || ctxt.lastNode == "ul") {
1095 				text[idx++] = "\n";
1096 			}
1097 			if (ctxt.isPreformatted) {
1098 				text[idx++] = AjxStringUtil.trim(el.nodeValue) + " ";
1099 			} else {
1100 				text[idx++] = AjxStringUtil.trim(el.nodeValue.replace(AjxStringUtil._LF, " "), true) + " ";
1101 			}
1102 		}
1103 	} else if (nodeName == "p") {
1104 		text[idx++] = onlyOneNewLinePerP ? "\n" : "\n\n";
1105 	} else if (listType == AjxStringUtil._NO_LIST && (nodeName == "br" || nodeName == "hr")) {
1106 		text[idx++] = "\n";
1107 	} else if (nodeName == "ol" || nodeName == "ul") {
1108 		text[idx++] = "\n";
1109 		if (el.parentNode.nodeName.toLowerCase() != "li" && ctxt.lastNode != "br" && ctxt.lastNode != "hr") {
1110 			text[idx++] = "\n";
1111 		}
1112 		listType = (nodeName == "ol") ? AjxStringUtil._ORDERED_LIST : AjxStringUtil._UNORDERED_LIST;
1113 		listLevel++;
1114 		bulletNum = 0;
1115 	} else if (nodeName == "li") {
1116 		for (var i = 0; i < listLevel; i++) {
1117 			text[idx++] = AjxStringUtil._INDENT;
1118 		}
1119 		if (listType == AjxStringUtil._ORDERED_LIST) {
1120 			text[idx++] = bulletNum + ". ";
1121 		} else {
1122 			text[idx++] = "\u002A "; // TODO ZmMsg.bullet
1123 		}
1124 	} else if (nodeName == "img") {
1125 		if (el.alt && el.alt != "") {
1126 			text[idx++] = el.alt;
1127 		}
1128 	} else if (nodeName == "tr" && el.parentNode.firstChild != el) {
1129 		text[idx++] = "\n";
1130 	} else if (nodeName == "td" && el.parentNode.firstChild != el) {
1131 		text[idx++] = "\t";
1132 	} else if (nodeName == "div") {
1133 		text[idx++] = "\n";
1134 	} else if (nodeName == "blockquote") {
1135 		text[idx++] = "\n\n";
1136 	} else if (nodeName == "pre") {
1137 		ctxt.isPreformatted = true;
1138 	} else if (nodeName == "#comment" ||
1139 			   nodeName == "script" ||
1140 			   nodeName == "select" ||
1141 			   nodeName == "style") {
1142 		return idx;
1143 	}
1144 
1145 	var childNodes = el.childNodes;
1146 	var len = childNodes.length;
1147 	for (var i = 0; i < len; i++) {
1148 		var tmp = childNodes[i];
1149 		if (tmp.nodeType == 1 && tmp.tagName.toLowerCase() == "li") {
1150 			bulletNum++;
1151 		}
1152 		idx = this._traverse(tmp, text, idx, listType, listLevel, bulletNum, ctxt, convertor, onlyOneNewLinePerP);
1153 	}
1154 
1155 	if (convertor && convertor["/"+nodeName]) {
1156 		text[idx++] = convertor["/"+nodeName](el);
1157 	}
1158 
1159 	if (nodeName == "h1" || nodeName == "h2" || nodeName == "h3" || nodeName == "h4"
1160 		|| nodeName == "h5" || nodeName == "h6") {
1161 			text[idx++] = "\n";
1162 			ctxt.list = false;
1163 	} else if (nodeName == "pre") {
1164 		ctxt.isPreformatted = false;
1165 	} else if (nodeName == "li") {
1166 		if (!ctxt.list) {
1167 			text[idx++] = "\n";
1168 		}
1169 		ctxt.list = false;
1170 	} else if (nodeName == "ol" || nodeName == "ul") {
1171 		ctxt.list = true;
1172 	} else if (nodeName != "#text") {
1173 		ctxt.list = false;
1174 	}
1175 
1176 	ctxt.lastNode = nodeName;
1177 	return idx;
1178 };
