 16 /**
 17  * Creates the import/export page.
 18  * @class
 19  * This class represents the import/export page.
 20  * 
 21  * @param {DwtControl}	parent			the containing widget
 22  * @param {Object}	section			the page
 23  * @param {ZmPrefController}	controller		the prefs controller
 24  * 
 25  * @extends	ZmPreferencesPage
 26  * 
 27  * @private
 28  */
 29 ZmFilterPage = function(parent, section, controller) {
 30 	ZmPreferencesPage.apply(this, arguments);
 31 };
 32 
 33 ZmFilterPage.prototype = new ZmPreferencesPage;
 34 ZmFilterPage.prototype.constructor = ZmFilterPage;
 35 
 36 ZmFilterPage.prototype.toString =
 37 function () {
 38     return "ZmFilterPage";
 39 };
 40 
 41 ZmFilterPage.prototype._createControls =
 42 function() {
 43 	this._tabView = new DwtTabView({parent:this, posStyle:Dwt.STATIC_STYLE});
 44 	this._tabView.reparentHtmlElement(this._htmlElId+"_tabview");
 45 	var incomingController = this._controller.getIncomingFilterRulesController();
 46 	this._tabView.addTab(ZmMsg.incomingMessageFilters, incomingController.getFilterRulesView());
 47 	var outgoingController = this._controller.getOutgoingFilterRulesController();
 48 	this._tabView.addTab(ZmMsg.outgoingMessageFilters, outgoingController.getFilterRulesView());
 49 	this.setVisible(true);
 50 	
 51 	this.hasRendered = true;
 52 }
 53 
 54 ZmFilterPage.prototype.getTabView =
 55 function () {
 56 	return this._tabView;
 57 };
 58 
 59 //
 60 // Protected methods
 61 //
 62 
 63 ZmFilterPage.prototype._setupCustom = function(id, setup, value) {
 64 	if (id == "FILTER_TABS") {
 65 		return this.getTabView();
 66 	}
 67 	return ZmPreferencesPage.prototype._setupCustom.apply(this, arguments);
 68 };
