 16 /**
 17  * Alerts of an event by playing a sound.
 18  * @private
 19  */
 20 ZmSoundAlert = function() {
 21 	this.html5AudioEnabled = ZmSoundAlert.isHtml5AudioEnabled();
 22 	this.enabled = this.html5AudioEnabled || AjxPluginDetector.detectQuickTime() || AjxPluginDetector.detectWindowsMedia();
 23 	if (this.enabled) {
 24 		var element = this._element = document.createElement("DIV");
 25 		element.style.position = 'relative';
 26 		element.style.top = '-1000px';
 27 		element.style.left = '-1000px';
 28 		document.body.appendChild(this._element);
 29 	} else {
 30 		DBG.println("No QuickTime or Windows Media plugin detected. Sound alerts are disabled.")
 31 	}
 32 };
 33 
 34 //this tests for wav file support. we might want to refactor it for various file types if needed later
 35 ZmSoundAlert.isHtml5AudioEnabled =
 36 function() {
 37 
 38 	try {
 39 		var audio = new Audio("");
 40 		if (!audio.canPlayType) {
 41 			return false;
 42 		}
 43 	}
 44 	catch (e) {
 45 		return false;
 46 	}
 47 
 48 	// canPlayType(type) returns: "", "no", "maybe" or "probably"
 49 	var canPlayWav = audio.canPlayType("audio/wav");
 50 	return (canPlayWav !== "no" && canPlayWav !== "");
 51 
 52 }
 53 
 54 ZmSoundAlert.prototype.toString =
 55 function() {
 56 	return "ZmSoundAlert";
 57 };
 58 
 59 ZmSoundAlert.getInstance =
 60 function() {
 61 	return ZmSoundAlert.INSTANCE = ZmSoundAlert.INSTANCE || new ZmSoundAlert();
 62 };
 63 
 64 ZmSoundAlert.prototype.start =
 65 function(soundFile) {
 66 	if (!this.enabled) {
 67 		return;
 68 	}
 69 
 70 	var time = new Date().getTime();
 71 	if (this._lastTime && ((time - this._lastTime) < 5000)) {
 72 		return;
 73 	}
 74 	this._lastTime = time;
 75 
 76 	soundFile = soundFile || "/public/sounds/im/alert.wav";
 77 	var url = appContextPath + soundFile;
 78 
 79 	var html;
 80 	if (this.html5AudioEnabled) {
 81 		html = '<audio src="' + url + '" autoplay="yes"></audio>';
 82 	}
 83 	else {
 84 		var embedId = Dwt.getNextId();
 85 		var htmlArr = [
 86 			"<object CLASSID='CLSID:6BF52A52-394A-11d3-B153-00C04F79FAA6' type='audio/wav'>",
 87 			"<param name='url' value='", url, "'>",
 88 			"<param name='autostart' value='true'>",
 89 			"<param name='controller' value='true'>",
 90 			"<embed id='", embedId, "' src='", url, "' controller='false' autostart='true' type='audio/wav'/>",
 91 			"</object>"
 92 		];
 93 		html = htmlArr.join("");
 94 	}
 95 	this._element.innerHTML = html;
 96 
 97 	if (!this.html5AudioEnabled && AjxEnv.isFirefox && AjxEnv.isWindows) {
 98 		// The quicktime plugin steals focus and breaks our keyboard nav.
 99 		// The best workaround I've found for this is to blur the embed
100 		// element, and that only works after the sound plays.
101 		//
102 		// Unfortunately it seems that on a slow connection this prevents
103 		// the sound from playing. I'm hoping this is less bad than killing
104 		// keyboard focus.
105 		//
106 		// Mozilla bug: https://bugzilla.mozilla.org/show_bug.cgi?id=78414
107 		if (this._blurActionId) {
108 			AjxTimedAction.cancelAction(this._blurActionId);
109 			this._blurActionId = null;
110 		}
111 		this._blurEmbedTimer(embedId, 0);
112 	}
113 };
114 
115 ZmSoundAlert.prototype._blurEmbedTimer =
116 function(embedId, tries) {
117 	var action = new AjxTimedAction(this, this._blurEmbed, [embedId, tries]);
118 	this._blurActionId = AjxTimedAction.scheduleAction(action, 500);
119 };
120 
121 ZmSoundAlert.prototype._blurEmbed =
122 function(embedId, tries) {
123 	this._blurActionId = null;
124 
125 	// Take focus from the embed.
126 	var embedEl = document.getElementById(embedId);
127 	if (embedEl && embedEl.blur) {
128 		embedEl.blur();
129 	}
130 
131 	// Force focus to the keyboard manager's focus obj.
132 	var focusObj = appCtxt.getKeyboardMgr().getFocusObj();
133 	if (focusObj && focusObj.focus) {
134 		focusObj.focus();
135 	}
136 
137 	// Repeat hack.
138 	if (tries < 2) {
139 		this._blurEmbedTimer(embedId, tries + 1);
140 	}
141 };
