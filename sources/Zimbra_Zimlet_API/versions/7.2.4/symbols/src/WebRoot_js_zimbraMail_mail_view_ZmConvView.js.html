 16 ZmConvView = function(params) {
 17 
 18 	params.className = "ZmConvView";
 19 	params.posStyle = Dwt.ABSOLUTE_STYLE;
 20 	params.mode = ZmId.VIEW_CONV;
 21 	ZmDoublePaneView.call(this, params);
 22 
 23 	this._changeListener = new AjxListener(this, this._convChangeListener);
 24 	
 25 	// add change listener to tree view to catch empty trash action
 26 	var folderTree = appCtxt.getFolderTree();
 27 	if (folderTree) {
 28 		folderTree.addChangeListener(new AjxListener(this, this._folderChangeListener));
 29 	}
 30 	
 31 	// Add a change listener to taglist to track tag color changes
 32 	this._tagList = appCtxt.getTagTree();
 33 	if (this._tagList) {
 34 		this._tagList.addChangeListener(new AjxListener(this, this._tagChangeListener));
 35 	}
 36 	
 37 	this._controller = params.controller;
 38 }

 60 ZmConvView.prototype.setItem =
 61 function(conv) {
 62 	if (!(conv instanceof ZmConv)) { return; }
 63 		
 64 	// Remove and re-add listeners for current conversation if it exists
 65 	if (this._conv) {
 66 		this._conv.removeChangeListener(this._changeListener);
 67 	}
 68 	this._conv = conv;
 69 	conv.addChangeListener(this._changeListener);
 70 
 71 	this._mailListView.set(conv.msgs, ZmItem.F_DATE);
 72 	this.isStale = false;
 73 	this._setSubject(conv.subject);
 74 	this._setTags(conv);
 75 	
 76 	// display "hot" message (or newest if no search performed)
 77 	var offset = this._mailListView.offset;
 78 	var hot = this._conv.getFirstHotMsg({offset:offset, limit:this._mailListView.getLimit(offset)});
 79 	this._mailListView.setSelection(hot);
 80 };

112 // override since we need to take summary into consideration
113 ZmConvView.prototype._resetSize = 
114 function(newWidth, newHeight, force) {
115 
116 	if (newWidth <= 0 || newHeight <= 0) { return; }
117 	if (!force && newWidth == this._lastResetWidth && newHeight == this._lastResetHeight) { return; }
118 
119 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
120 
121 	var summaryHeight = this._summary.getHtmlElement().offsetHeight;
122 	if (this.isMsgViewVisible()) {
123 		var sash = this.getSash();
124 		var sashSize = sash.getSize();
125 		var sashThickness = readingPaneOnRight ? sashSize.x : sashSize.y;
126 		if (readingPaneOnRight) {
127 			var listViewWidth = this._vertSashX || Math.floor(newWidth / 2.5);
128 			this._mailListView.setLocation(0, summaryHeight);
129 			this._mailListView.resetSize(listViewWidth, newHeight - summaryHeight);
130 			sash.setLocation(listViewWidth, 0);
131 			this._msgView.setBounds(listViewWidth + sashThickness, 0,
132 									newWidth - (listViewWidth + sashThickness), newHeight);
133 		} else {
134 			var listViewHeight = 0, upperHeight, rowsHeight = 0;
135 			if (this._horizSashY) {
136 				listViewHeight = this._horizSashY - summaryHeight;
137 				upperHeight = summaryHeight + listViewHeight;
138 			} else {
139 				// set height of MLV based on number of msgs in conv - since that number is usually small,
140 				// we can give the MV more space
141 				var list = this._mailListView.getList();
142 				if (list && list.size() > 0) {
143 					var threshold = Math.min(list.size() + 1, 6);
144 					var div = document.getElementById(this._mailListView._getItemId(list.get(0)));
145 					if (div) {
146 						var rowsHeight = Dwt.getSize(div).y * threshold;
147 						listViewHeight = rowsHeight + DwtListView.HEADERITEM_HEIGHT;
148 						upperHeight = this._minUpperHeight = summaryHeight + listViewHeight;
149 					}
150 				}
151 			}
152 			if (summaryHeight && listViewHeight && upperHeight) {
153 				this._mailListView.setLocation(0, summaryHeight);
154 				this._mailListView.resetSize(newWidth, listViewHeight);
155 				sash.setLocation(0, upperHeight);
156 				var mvHeight = Math.max((newHeight - (upperHeight + sashThickness)), 0);
157 				this._msgView.setBounds(0, upperHeight + sashThickness, newWidth, mvHeight);
158 			}
159 		}
160 	} else {
161 		this._mailListView.resetSize(newWidth, newHeight - summaryHeight);
162 	}
163 
164 	if (AjxEnv.isIE && this._subjectDiv) {
165 		Dwt.setSize(this._subjectDiv, newWidth - 95);
166 	}
167 	
168 	this._mailListView._resetColWidth();
169 };

171 ZmConvView.prototype._sashCallback =
172 function(delta) {
173 
174 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
175 
176 	if (delta > 0) {
177 		if (readingPaneOnRight) {
178 			// moving sash right
179 			this._summary.setSize(this._summary.getSize().x + delta, Dwt.DEFAULT);
180 			this._mailListView.resetSize(this._mailListView.getSize().x + delta, Dwt.DEFAULT);
181 			this._msgView.setBounds(this._msgView.getLocation().x + delta, Dwt.DEFAULT,
182 									this._msgView.getSize().x - delta, Dwt.DEFAULT);
183 		} else {
184 			// moving sash down
185 			var newMsgViewHeight = this._msgView.getSize().y - delta;
186 			var minMsgViewHeight = this._msgView.getMinHeight();
187 			// make sure msg header remains visible
188 			if (newMsgViewHeight > minMsgViewHeight) {
189 				this._mailListView.resetSize(Dwt.DEFAULT, this._mailListView.getSize().y + delta);
190 				this._msgView.setSize(Dwt.DEFAULT, newMsgViewHeight);
191 				this._msgView.setLocation(Dwt.DEFAULT, this._msgView.getLocation().y + delta);
192 			} else {
193 				delta = 0;
194 			}
195 		}
196 	} else {
197 		var absDelta = Math.abs(delta);
198 		if (readingPaneOnRight) {
199 			// moving sash left
200 			this._summary.setSize(this._summary.getSize().x - absDelta, Dwt.DEFAULT);
201 			this._mailListView.resetSize(this._mailListView.getSize().x - absDelta, Dwt.DEFAULT);
202 			this._msgView.setBounds(this._msgView.getLocation().x - absDelta, Dwt.DEFAULT,
203 									this._msgView.getSize().x + absDelta, Dwt.DEFAULT);
204 		} else {
205 			// moving sash up
206 			// make sure summary and MLV remain visible
207 			if (this._horizSashY - absDelta > this._minUpperHeight) {
208 				this._mailListView.resetSize(Dwt.DEFAULT, this._mailListView.getSize().y - absDelta);
209 				this._msgView.setSize(Dwt.DEFAULT, this._msgView.getSize().y + absDelta);
210 				this._msgView.setLocation(Dwt.DEFAULT, this._msgView.getLocation().y - absDelta);
211 			} else {
212 				delta = 0;
213 			}
214 		}
215 	}
216 
217 	if (delta) {
218 		this._mailListView._resetColWidth();
219 		if (readingPaneOnRight) {
220 			this._vertSashX = this._vertMsgSash.getLocation().x;
221 		} else {
222 			this._horizSashY = this._horizMsgSash.getLocation().y;
223 		}
224 	}
225 
226 	return delta;
227 };

229 ZmConvView.prototype._initHeader =
230 function() {
231 	this._summary = new DwtComposite(this, "Summary", Dwt.RELATIVE_STYLE);
232 
233 	var tagDivId = appCtxt.get(ZmSetting.TAGGING_ENABLED) ? (this._htmlElId + "_tagDiv") : null;
234 	var subs = { id: this._htmlElId, tagDivId: tagDivId };
235 
236 	this._summary.getHtmlElement().innerHTML = AjxTemplate.expand("mail.Message#MessageListHeader", subs);
237 
238 	// add the close button
239 	var closeButtonId = this._htmlElId + "_closeBtnCell";
240 	var closeButton = new DwtButton({parent:this, className:"DwtToolbarButton", parentElement:closeButtonId});
241 	closeButton.setImage("Close");
242 	closeButton.setText(ZmMsg.close);
243 	closeButton.addSelectionListener(new AjxListener(this, this._closeButtonListener));
244 	
245 	// save the necessary DOM objects we just created
246 	this._subjectDiv = document.getElementById(this._htmlElId + "_subjDiv");
247 
248 	if (tagDivId) {
249 		this._tagDiv = document.getElementById(tagDivId);
250 		Dwt.setSize(this._tagDiv, Dwt.DEFAULT, ZmConvView._TAGLIST_HEIGHT);
251 		Dwt.setVisible(this._tagDiv, false);
252 	}
253 };

262 ZmConvView.prototype._setTags =
263 function(conv) {
264 	if (!appCtxt.get(ZmSetting.TAGGING_ENABLED)) return;
265 
266 	var numTags = conv.tags.length;
267 	var origVis = Dwt.getVisible(this._tagDiv);
268 	var newVis = (numTags > 0);
269 	if (origVis != newVis) {
270 		Dwt.setVisible(this._tagDiv, newVis);
271 		var sz = this.getSize();
272 		this._resetSize(sz.x, sz.y, true);
273 		if (!newVis) {
274 			this._tagDiv.innerHTML = "";
275 			return;
276 		}
277 	}
278 	
279 	if (!numTags) return;
280 		
281 	var ta = new Array();
282 	for (var j = 0; j < numTags; j++)
283 		ta[j] = this._tagList.getById(conv.tags[j]);
284 
285 	if (ta.length == 0)	return;
286 	ta.sort(ZmTag.sortCompare);
287 
288 	var html = new Array();
289 	var i = 0;
290 	html[i++] = "<table cellspacing=0 cellpadding=0 border=0 width=100%>";
291 	html[i++] = "<tr><td style='overflow:hidden'>";
292 	
293 	for (var j = 0; j < ta.length; j++) {
294 		var tag = ta[j];
295 		if (!tag) continue;
296 		var anchorId = [this._tagDiv.id, ZmConvView._TAG_ANCHOR, tag.id].join("");
297 		var imageId = [this._tagDiv.id, ZmDoublePaneView._TAG_IMG, tag.id].join("");
298 
299 		html[i++] = "<a href='javascript:;' onclick='ZmConvView._tagClick(\"";
300 		html[i++] = this._htmlElId;
301 		html[i++] = '","';
302 		html[i++] = tag.id;
303 		html[i++] = "\"); return false;' id='";
304 		html[i++] = anchorId;
305 		html[i++] = "'>";
306 		html[i++] = "<table style='display:inline; vertical-align:middle;' border=0 cellspacing=0 cellpadding=0><tr><td>";
307 		html[i++] = AjxImg.getImageHtml(tag.getIconWithColor(), null, ["id='", imageId, "'"].join(""));
308 		html[i++] = "</td></tr></table>";
309 		html[i++] = AjxStringUtil.htmlEncodeSpace(tag.name);
310 		html[i++] = "</a>";
311 	}
312 	html[i++] = "</td></tr></table>";
313 	this._tagDiv.innerHTML = html.join("");
314 };

319 ZmConvView.prototype._convChangeListener =
320 function(ev) {
321 	if (ev.type != ZmEvent.S_CONV)
322 		return;
323 	var fields = ev.getDetail("fields");
324 	if (ev.event == ZmEvent.E_TAGS || ev.event == ZmEvent.E_REMOVE_ALL) {
325 		this._setTags(this._conv);
326 	} else if (ev.event == ZmEvent.E_MODIFY && (fields && fields[ZmItem.F_ID])) {
327 		this._controller._convId = this._conv.id;
328 	}
329 };

331 ZmConvView.prototype._folderChangeListener = 
332 function(ev) {
333 	if (ev.event == ZmEvent.E_DELETE &&
334 	    ev.source instanceof ZmFolder && 
335 		ev.source.id == ZmFolder.ID_TRASH && 
336 		this._conv.msgs) 
337 	{
338 		// user emptied trash folder.. search for any msgs in trash and remove from list view
339 		var list = this._conv.msgs.getArray();
340 		var len = list.length; // save original length
341 		for (var i = 0; i < list.length; i++) {
342 			var folder = appCtxt.getById(list[i].folderId);
343 			if (folder.isInTrash()) {
344 				this._mailListView.removeItem(list[i]);
345 				this._conv.msgs.remove(list[i], true);
346 				i--;
347 			}
348 		}
349 		
350 		this._controller._resetNavToolBarButtons(this._controller._getViewType());
351 		
352 		if (len != this._conv.numMsgs) {
353 			// allow CLV to update its msg count if its been changed
354 			var fields = {};
355 			fields[ZmItem.F_SIZE] = true;
356 			this._conv.list._notify(ZmEvent.E_MODIFY, {items: [this._conv], fields: fields});
357 			// reset selection to first msg
358 			this._mailListView.setSelection(this._conv.msgs.getVector().get(0));
359 		}
360 	}
361 };
362 
363 ZmConvView.prototype._tagChangeListener =
364 function(ev) {
365 	if (ev.type != ZmEvent.S_TAG)
366 		return;
367 
368 	var fields = ev.getDetail("fields");
369 	if (ev.event == ZmEvent.E_MODIFY && (fields && fields[ZmOrganizer.F_COLOR])) {
370 		var tag = ev.getDetail("organizers")[0];
371 		var img = document.getElementById(this._tagDiv.id +  ZmDoublePaneView._TAG_IMG + tag.id);
372 		if (img)
373 			AjxImg.setImage(img, tag.getIconWithColor());
374 	}
375 	
376 	if (ev.event == ZmEvent.E_DELETE || ev.event == ZmEvent.MODIFY)
377 		this._setTags(this._conv);
378 };

385 // mimics handling by ZmConvListController.prototype._listSelectionListener
386 ZmConvView.prototype._staleHandler =
387 function() {
388 
389 	var ctlr = this._controller._parentController || AjxDispatcher.run("GetConvListController");
390 	this._conv._loaded = false;	// force request to be made
391 	var respCallback = new AjxCallback(ctlr, ctlr._handleResponseListSelectionListener, this._conv);
392 	AjxDispatcher.run("GetConvController").show(ctlr._activeSearch, this._conv, ctlr, respCallback, true);
393 };

403 ZmConvView.prototype.handleRemoveAttachment =
404 function(oldMsgId, newMsg) {
405 	var ac = window.parentAppCtxt || window.appCtxt;
406 	// cache this actioned ID so we can reset selection to it once the CREATE
407 	// notifications have been processed.
408 	ac.getApp(ZmApp.MAIL).getMailListController().actionedMsgId = newMsg.id;
409 	var msgView = this.getMsgView();
410 	if (msgView) {
411 		var mailListView = this.getMailListView();
412 		if (mailListView) {
413 			var mailList = mailListView.getList();
414 			var pos = mailList.indexOf(msgView._msg);
415 			mailList.replace(pos, newMsg);
416 			mailListView.set(mailList.clone(), mailListView._sortByString);
417 			mailListView.setSelection(newMsg);
418 		}
419 		if (this._conv) {
420 			//hack to avoid popview if only one message in conversation; delete notification will cause view to be popped otherwise
421 			this._conv.addMsg(newMsg);
422 			this._conv.numMsgs = this._conv.msgIds.length;
423 		}
424 		msgView.handleRemoveAttachment(oldMsgId, newMsg);
425 	}
426 };
