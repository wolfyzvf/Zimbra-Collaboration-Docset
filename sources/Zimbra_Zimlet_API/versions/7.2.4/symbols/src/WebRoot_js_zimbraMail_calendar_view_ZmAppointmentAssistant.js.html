 16 /**
 17  * Creates an appointment assistant.
 18  * @class
 19  * This class represents an appointment assistant.
 20  * 
 21  * @extends		ZmAssistant
 22  */
 23 ZmAppointmentAssistant = function() {
 24 	ZmAssistant.call(this, ZmMsg.createNewAppt, ZmMsg.ASST_CMD_APPOINTMENT, ZmMsg.ASST_CMD_SUM_APPOINTMENT);
 25 };
 26 
 27 ZmAppointmentAssistant.prototype = new ZmAssistant();
 28 ZmAppointmentAssistant.prototype.constructor = ZmAppointmentAssistant;
 29 
 30 ZmAppointmentAssistant.prototype.initialize =
 31 function(dialog) {
 32 	ZmAssistant.prototype.initialize.call(this, dialog);
 33 	this._apptData = {};
 34 };
 35 
 36 ZmAppointmentAssistant.prototype.okHandler =
 37 function(dialog) {
 38 	if (!this._apptData.subject) {
 39 		dialog.messageDialog(ZmMsg.errorMissingSubject, DwtMessageDialog.CRITICAL_STYLE);
 40 	} else {
 41 		var appt = this.getAppt();
 42 		appt.save();// TODO: callback, etc.
 43 		appCtxt.setStatusMsg(ZmMsg.apptCreated);
 44 		return true;
 45 	}
 46 };
 47 
 48 ZmAppointmentAssistant.prototype.extraButtonHandler =
 49 function(dialog) {
 50 	var calApp = appCtxt.getApp(ZmApp.CALENDAR);
 51 	var cc = calApp.getCalController();
 52 	cc.newAppointment(this.getAppt(), ZmCalItem.MODE_NEW_FROM_QUICKADD, true); // dirty bit
 53 	return true;
 54 };
 55 
 56 ZmAppointmentAssistant.prototype.getHelp =
 57 function() {
 58 	return ZmMsg.ASST_APPT_HELP;
 59 };
 60 
 61 /**
 62  * Gets the appointment.
 63  * 
 64  * @return	{ZmAppt}		the appointment
 65  */
 66 ZmAppointmentAssistant.prototype.getAppt =
 67 function() {
 68 	var appt = new ZmAppt();
 69 	appt.setStartDate(this._apptData.startDate);
 70 	appt.setEndDate(this._apptData.endDate ? this._apptData.endDate : this._apptData.startDate);
 71 	appt.setAllDayEvent(this._apptData.startTime == null);
 72 
 73 	if (this._apptData.location) appt.setAttendees(AjxEmailAddress.split(this._apptData.location), ZmCalBaseItem.LOCATION);
 74 	if (this._apptData.notes) appt.setTextNotes(this._apptData.notes);
 75 	if (this._apptData.subject) appt.setName(this._apptData.subject);
 76 	return appt;
 77 };
 78 
 79 ZmAppointmentAssistant.__RE_handleQuotes = new RegExp([ZmAssistant.SPACES,"\"([^\"]*)\"?",ZmAssistant.SPACES].join(""));
 80 ZmAppointmentAssistant.__RE_handleBrackets = new RegExp([ZmAssistant.SPACES,"\\[([^\\]]*)\\]?",ZmAssistant.SPACES].join(""));
 81 ZmAppointmentAssistant.__RE_handleParens = new RegExp([ZmAssistant.SPACES,"\\(([^)]*)\\)?",ZmAssistant.SPACES].join(""));
 82 
 83 /**
 84  * 
 85  * (...)                 matched as notes, stripped out
 86  * [...]                 matched as location, stripped out
 87  * {date-spec}           first matched pattern is "start date", second is "end date"
 88  * {time-spec}           first matched pattern is "start time", second is "end time"
 89  * repat {repeat-spec}   recurrence rule
 90  * calendar {cal-name}   calendar to add appt to
 91  * invite {e1,e2,e3}     email addresses to invite (ideally would auto-complete)
 92  * subject "..."         explicit subject
 93  * 
 94  * everything renaming is the subject, unless subject was explicit
 95  * 
 96  * example:
 97  * 
 98  * lunch 12:30 PM next friday with satish (to discuss future release) [CPK, Palo Alto]
 99  * 
100  * "12:30 PM" matched as a time, saved as "start time" * 
101  * "next friday" matches a date, so is stripped out and saved as "start date"
102  * (...) matched as notes, stripped out and saved as "notes"
103  * [...] matched as location
104  * 
105  * everything left "lunch with satish" is taken as subject
106  * 
107  * 
108  * @private
109  */
110 ZmAppointmentAssistant.prototype.handle =
111 function(dialog, verb, args) {
112 	dialog._setOkButton(AjxMsg.ok, true, true, false);
113 	dialog._setExtraButton(ZmMsg.moreDetails, true, true, false);
114 	
115 	var adata = this._apptData = {};
116 	
117 	adata.startDate = new Date();
118 	adata.endDate = null;
119 
120 	var match;
121 	
122 	match = args.match(ZmAppointmentAssistant.__RE_handleQuotes);
123 	if (match) {
124 		adata.subject = match[1];
125 		args = args.replace(match[0], " ");
126 	}
127 
128 	match = args.match(ZmAppointmentAssistant.__RE_handleBrackets);
129 	if (match) {
130 		adata.location = match[1];
131 		args = args.replace(match[0], " ");
132 	}
133 
134 	match = args.match(ZmAppointmentAssistant.__RE_handleParens);
135 	if (match) {
136 		adata.notes = match[1];
137 		args = args.replace(match[0], " ");
138 	}
139 
140 	adata.startDate.setMinutes(0);
141 	adata.startTime = this._matchTime(args);
142 	if (adata.startTime) {
143 		adata.startDate.setHours(adata.startTime.hour, adata.startTime.minute);
144 		args = adata.startTime.args;
145 	}
146 
147 	// look for an end time
148 	adata.endTime = this._matchTime(args);
149 	if (adata.endTime) {
150 		args = adata.endTime.args;
151 	}
152 
153 	// look for start date
154 	match = this._objectManager.findMatch(args, ZmObjectManager.DATE);
155 	if (match) {
156 		args = args.replace(match[0], " ");
157 		adata.startDate = match.context.date;
158 		if (adata.startTime) adata.startDate.setHours(adata.startTime.hour, adata.startTime.minute);
159 	}
160 	
161 	// look for end date
162 	match = this._objectManager.findMatch(args, ZmObjectManager.DATE);
163 	if (match) {
164 		args = args.replace(match[0], " ");
165 		adata.endDate = match.context.date;
166 		if (adata.endTime != null) adata.endDate.setHours(adata.endTime.hour, adata.endTime.minute);
167 		else if (adata.startTime != null) adata.endDate.setHours(adata.startTime.hour, adata.startTime.minute);
168 	} else {
169 		if (adata.endTime) {
170 			adata.endDate = new Date(adata.startDate.getTime());
171 			if (adata.endTime != null) adata.endDate.setHours(adata.endTime.hour, adata.endTime.minute);			
172 		} else if (adata.startTime) {
173 			adata.endDate = new Date(adata.startDate.getTime() + 1000 * 60 * 60);
174 		}
175 	}
176 
177 	if (adata.subject == null) {
178 		adata.subject = ZmAssistant.normalize(ZmAssistant.trim(args));
179 	}
180 
181 	var subStr = AjxStringUtil.convertToHtml(adata.subject == "" ? ZmMsg.ASST_APPT_subject : adata.subject);
182 	var locStr = AjxStringUtil.convertToHtml(adata.location == null ? ZmMsg.ASST_APPT_location : adata.location);
183 	var notesStr = AjxStringUtil.convertToHtml(adata.notes == null ? ZmMsg.ASST_APPT_notes : adata.notes);
184 	var subjectIndex = this._setField("* "+ZmMsg.subject, subStr, adata.subject == "", false);
185 	this._setDateFields(adata.startDate, adata.startTime, adata.endDate, adata.endTime, subjectIndex);
186 	this._setField(ZmMsg.location, locStr, adata.location == null, false);	
187 	this._setField(ZmMsg.notes, notesStr, adata.notes == null, false);
188 
189 	var cc = appCtxt.getApp(ZmApp.CALENDAR).getCalController();
190 	var agenda = cc.getDayToolTipText(adata.startDate, true);
191 	this._setField(ZmMsg.agenda, agenda, false, false);
192 	return;
193 };

195 ZmAppointmentAssistant.prototype._setDateFields = 
196 function(startDate, startTime, endDate, endTime, rowIndex) {
197 	var startDateValue = DwtCalendar.getDateFullFormatter().format(startDate);
198 	var sameDay = false;
199 	var html = new AjxBuffer();
200 	html.append("<table border=0 cellpadding=0 cellspacing=0>");
201 	html.append("<tr>");
202 	html.append("<td>", AjxStringUtil.htmlEncode(startDateValue), "</td>");
203 	if (startTime) {
204 		var startTimeValue = AjxDateUtil.computeTimeString(startDate);
205 		html.append("<td></td><td> </td><td>@</td><td> </td>");
206 		html.append("<td>", AjxStringUtil.htmlEncode(startTimeValue), "</td>");
207 		sameDay = endDate && endDate.getFullYear() == startDate.getFullYear() && 
208 			endDate.getMonth() == startDate.getMonth() && endDate.getDate() == startDate.getDate();
209 		if (sameDay) {
210 			var endTimeValue = AjxDateUtil.computeTimeString(endDate);
211 			html.append("<td> - </td>");
212 			html.append("<td>", AjxStringUtil.htmlEncode(endTimeValue), "</td>");
213 		}
214 	}
215 	html.append("</tr></table>");	
216 	var doEnd = (endDate && !sameDay);
217 	
218 	if (doEnd) {
219 		this._clearField(ZmMsg.time);
220 		rowIndex = this._setField(ZmMsg.startTime, html.toString(), false, false, rowIndex+1);
221 		
222 		html.clear();
223 		var endDateValue = DwtCalendar.getDateFullFormatter().format(endDate);
224 			html.append("<table border=0 cellpadding=0 cellspacing=0>");		
225 		html.append("<tr>");
226 		html.append("<td>", AjxStringUtil.htmlEncode(endDateValue), "</td>");
227 		if (startTime) { // display end time if a startTime was specified
228 			var endTimeValue = AjxDateUtil.computeTimeString(endDate);
229 			html.append("<td></td><td> </td><td>@</td><td> </td>");
230 			html.append("<td>", AjxStringUtil.htmlEncode(endTimeValue), "</td>");
231 		}
232 		html.append("</tr></table>");
233 		this._setField(ZmMsg.endTime, html.toString(), false, false, rowIndex+1);
234 		
235 	} else {
236 		this._setField(ZmMsg.time, html.toString(), false, false, rowIndex+1);
237 		this._clearField(ZmMsg.startTime);
238 		this._clearField(ZmMsg.endTime);		
239 	}
240 };
