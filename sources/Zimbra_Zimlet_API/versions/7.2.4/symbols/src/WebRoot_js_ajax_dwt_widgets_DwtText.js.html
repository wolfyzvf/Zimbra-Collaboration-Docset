 17 /**
 18  * Creates a text control.
 19  * @constructor
 20  * @class
 21  * This class represents a container for a piece of text.
 22  * 
 23  * @author Ross Dargahi
 24  * 
 25  * @param {hash}	params		a hash of parameters
 26  * @param {DwtComposite}      parent	the parent widget
 27  * @param {string}      className		CSS class
 28  * @param {constant}      posStyle		the positioning style (see {@link DwtControl})
 29  * @param {string}      id			an explicit ID to use for the control's HTML element
 30  * 
 31  * @extends		DwtControl
 32  */
 33 DwtText = function(params) {
 34 	if (arguments.length == 0) return;
 35 	params = Dwt.getParams(arguments, DwtText.PARAMS);
 36 	params.className = params.className || "DwtText";
 37 	DwtControl.call(this, params);
 38 };
 39 
 40 DwtText.PARAMS = ["parent", "className", "posStyle"];
 41 
 42 DwtText.prototype = new DwtControl;
 43 DwtText.prototype.constructor = DwtText;
 44 
 45 DwtText.prototype.toString =
 46 function() {
 47 	return "DwtText";
 48 };
 49 
 50 DwtText.prototype.getTabGroupMember = function() {
 51 	return null;
 52 };
 53 
 54 /**
 55  * Sets the text.
 56  * 
 57  * @param	{string}	text		the text
 58  */
 59 DwtText.prototype.setText =
 60 function(text) {
 61 	if (!this._textNode) {
 62 		 this._textNode = document.createTextNode(text);
 63 		 this.getHtmlElement().appendChild(this._textNode);
 64 	} else {
 65 		try { // IE mysteriously throws an error sometimes, but still does the right thing
 66 			this._textNode.data = text;
 67 		} catch (e) {}
 68 	}
 69 };
 70 
 71 /**
 72  * Gets the text.
 73  * 
 74  * @return	{string}	the text
 75  */
 76 DwtText.prototype.getText =
 77 function() {
 78 	return this._textNode ? this._textNode.data : "";
 79 };
 80 
 81 /**
 82  * Gets the text node.
 83  * 
 84  * @return	{Object}	the node
 85  */
 86 DwtText.prototype.getTextNode =
 87 function() {
 88 	return this._textNode;
 89 };
