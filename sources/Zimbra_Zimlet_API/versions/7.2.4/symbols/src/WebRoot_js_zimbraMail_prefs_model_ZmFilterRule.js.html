 16 /**
 17  * Creates an empty filter rule. Conditions and actions will need to be added.
 18  * @constructor
 19  * @class
 20  * ZmFilterRule represents a filter rule. A rule includes one or more conditions
 21  * and one or more actions.
 22  *
 23  * @author Conrad Damon
 24  *
 25  * @param {String}	name			the rule name
 26  * @param {Boolean}	active		if <code>true</code>, if the rule is enabled
 27  * @param {Object}	filterActions	the filter action data as raw json object
 28  * @param {Object}	filterTests	the filter conditions data as raw json object
 29  * 
 30  */
 31 ZmFilterRule = function(name, active, filterActions, filterTests) {
 32 	/**
 33      * The name of the filter rule.
 34      * @type String
 35      */
 36 	this.name = name;
 37 	/**
 38 	 * The filter rule actions.
 39 	 * @type	Object
 40 	 */
 41 	this.actions = filterActions || {};
 42 	/**
 43 	 * The filter rules conditions.
 44 	 * @type	Object
 45 	 */
 46 	this.conditions = filterTests || {};
 47 	this.active = (active !== false);
 48 	if (!filterTests) {
 49 		this.setGroupOp();
 50 	}
 51 
 52 	this.id = ZmFilterRule._nextId++;
 53 };

255 /**
256  * This defines a hash of conditions. Each condition is a hash of parameters. The key of the hash
257  * is also known as the condition "subject". It is the field of an email message that 
258  * the condition is tested against.
259  * 
260  * <p>
261  * The condition parameters are:
262  * <ul>
263  * <li><b>subjectMod</b>	Type of input widget for the subjectModifier, which is a specifier or 
264  *				modifier for the subject (such as which address to look at)</li>
265  * <li><b>smOptions</b>		List of possible values for the subjectModifier ({@link ZmFilterRule.TYPE_SELECT})</li>
266  * <li><b>ops</b>			Type of input widget for choosing the comparator</li>
267  * <li><b>opsOptions</b>	List of possible comparators for this subject ({@link ZmFilterRule.TYPE_SELECT} type)</li>
268  * <li><b>value</b>			Type of input widget for specifying the value</li>
269  * <li><b>vOptions</b>		List of possible values ({@link ZmFilterRule.TYPE_SELECT} type)</li>
270  * <li><b>valueMod</b>		Type of input widget for the valueModifier, which is a specifier or 
271  *				modifier for the value (such as units for size)</li>
272  * <li><b>vmOptions</b>		List of possible values for the valueModifier ({@link ZmFilterRule.TYPE_SELECT} type)</li>
273  * </ul>
274  */

615 /**
616  * Checks if the if the rule is enabled.
617  * 
618  * @return	{Boolean}	<code>true</code> if the rule is enabled
619  */
620 ZmFilterRule.prototype.hasValidAction =
621 function() {
622 	for (var i in this.actions) {
623 		var actionIndex = ZmFilterRule.A_VALUE_MAP[i];
624 		var actionCfg = ZmFilterRule.ACTIONS[actionIndex];
625 		if ((actionIndex != ZmFilterRule.A_STOP && (ZmFilterRule.checkPreconditions(actionCfg))) || actionIndex == ZmFilterRule.A_FORWARD) {
626 			return true;
627 		}
628 	}
629 	return false;
630 };

635 ZmFilterRule.getConditionData =
636 function(testType, comparator, value, subjectMod, caseSensitive) {
637 	var conditionData = {};
638 
639 	// add subject modifier
640 	if (subjectMod &&
641 		(testType == ZmFilterRule.TEST_HEADER ||
642 		 testType == ZmFilterRule.TEST_HEADER_EXISTS ||
643 		 testType == ZmFilterRule.TEST_ADDRBOOK ||
644 		 testType == ZmFilterRule.TEST_MIME_HEADER))
645 	{
646 		conditionData.header = subjectMod;
647 	}
648 
649 	// normalize negative operator and add comparator
650 	var negativeOp;
651 	switch (comparator) {
652 		case ZmFilterRule.OP_NOT_IS:		negativeOp = ZmFilterRule.OP_IS; break;
653 		case ZmFilterRule.OP_NOT_CONTAINS:	negativeOp = ZmFilterRule.OP_CONTAINS; break;
654 		case ZmFilterRule.OP_NOT_MATCHES:	negativeOp = ZmFilterRule.OP_MATCHES; break;
655 		case ZmFilterRule.OP_NOT_EXISTS:	negativeOp = ZmFilterRule.OP_EXISTS; break;
656 		case ZmFilterRule.OP_NOT_UNDER:		negativeOp = ZmFilterRule.OP_UNDER; break;
657 		case ZmFilterRule.OP_NOT_OVER:		negativeOp = ZmFilterRule.OP_OVER; break;
658 		case ZmFilterRule.OP_NOT_BEFORE:	negativeOp = ZmFilterRule.OP_BEFORE; break;
659 		case ZmFilterRule.OP_NOT_AFTER:		negativeOp = ZmFilterRule.OP_AFTER; break;
660 		case ZmFilterRule.OP_NOT_IN:		negativeOp = ZmFilterRule.OP_IN; break;
661         case ZmFilterRule.OP_NOT_REPLIED:   negativeOp = ZmFilterRule.OP_IS_REPLIED; break;
662         case ZmFilterRule.OP_NOT_REQUESTED: negativeOp = ZmFilterRule.OP_IS_REQUESTED; break;
663         case ZmFilterRule.OP_NOT_READRECEIPT: negativeOp = ZmFilterRule.OP_CONTAINS; break;
664 	}
665 	if (negativeOp) {
666 		conditionData.negative = "1";
667 	}
668 
669 	var compType = ZmFilterRule.COMP_TEST_MAP[testType];
670 	if (compType) {
671 		conditionData[compType] = ZmFilterRule.OP_VALUE[negativeOp || comparator];
672 	}
673 
674 	// add data value
675 	if (value) {
676 		switch (testType) {
677 			case ZmFilterRule.TEST_ADDRBOOK:	conditionData.folderPath = value; break;
678 			case ZmFilterRule.TEST_SIZE:		conditionData.s = value; break;
679 			case ZmFilterRule.TEST_DATE:		conditionData.d = value; break;
680 			default:							conditionData.value = value; break;
681 		}
682 	}
683 
684 	if (testType == ZmFilterRule.TEST_INVITE) {
685 	    conditionData.method = [{_content:ZmFilterRule.OP_VALUE[negativeOp || comparator]}];
686 	}
687 	if (caseSensitive != null) {
688 		conditionData.caseSensitive = caseSensitive;
689 	}
690 
691 	return conditionData;
692 };

717 ZmFilterRule.checkPreconditions =
718 function(obj) {
719     if (!ZmFilterRule.__preConditionsInitialized) {
720         ZmFilterRule.__preConditionsInitialized = true;
721         ZmFilterRule._setPreconditions();
722     }
723 
724 	var pre = obj && obj.precondition;
725 	if (!pre) { return true; }
726 
727 	var conds = AjxUtil.toArray(pre);
728 	for (var i = 0; i < conds.length; i++) {
729 		if (!appCtxt.get(conds[i])) {
730 			return false;
731 		}
732 	}
733 	return true;
734 };

751 		case ZmFilterRule.OP_UNDER:		negativeOp = ZmFilterRule.OP_NOT_UNDER; break;
752 		case ZmFilterRule.OP_OVER:		negativeOp = ZmFilterRule.OP_NOT_OVER; break;
753 		case ZmFilterRule.OP_BEFORE:	negativeOp = ZmFilterRule.OP_NOT_BEFORE; break;
754 		case ZmFilterRule.OP_AFTER:		negativeOp = ZmFilterRule.OP_NOT_AFTER; break;
755 		case ZmFilterRule.OP_IN:		negativeOp = ZmFilterRule.OP_NOT_IN; break;
756         case ZmFilterRule.OP_IS_REPLIED:   negativeOp = ZmFilterRule.OP_NOT_REPLIED; break;
757         case ZmFilterRule.OP_IS_REQUESTED: negativeOp = ZmFilterRule.OP_NOT_REQUESTED; break;
758         case ZmFilterRule.OP_IS_READRECEIPT: negativeOp = ZmFilterRule.OP_NOT_CONTAINS; break;
