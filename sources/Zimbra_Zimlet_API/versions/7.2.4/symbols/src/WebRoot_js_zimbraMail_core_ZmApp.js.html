 23 /**
 24  * Creates the application.
 25  * @class
 26  * This object represents a Zimbra Application. This class is a base class for application classes.
 27  * "App" is a useful abstraction for a set of related functionality, such as mail,
 28  * address book, or calendar. Looked at another way, an app is a collection of one or more controllers.
 29  * 
 30  * @param	{String}	name		the application name
 31  * @param	{DwtControl}	container	the control that contains components
 32  * @param	{ZmController}	parentController	the parent window controller (set by the child window)
 33  *
 34  */
 35 ZmApp = function(name, container, parentController) {
 36 
 37 	if (arguments.length == 0) return;
 38 	
 39 	this._name = name;
 40 	this._container = container;
 41 	this._parentController = parentController;
 42 	this._active = false;
 43 	this.currentSearch = null;
 44 
 45 	this._deferredFolders = [];
 46 	this._deferredFolderHash = {};
 47 	this._deferredNotifications = [];
 48 
 49 	this._sessionController		= {};
 50 	this._sessionId				= {};
 51 	this._curSessionId			= {};
 52 
 53 	ZmApp.DROP_TARGETS[name] = {};
 54 
 55 	this._defineAPI();
 56 	if (!parentController) {
 57 		this._registerSettings();
 58 	}
 59 	this._registerOperations();
 60 	this._registerItems();
 61 	this._registerOrganizers();
 62 	if (!parentController) {
 63 		this._setupSearchToolbar();
 64 	}
 65 	this._registerApp();
 66 
 67 };

137 /**
138  * Registers and stores information about an app. Note: Setting a value that evaluates to
139  * false (such as 0 or an empty string) will not do anything.
140  *
141  * @param {constant}	app				the app ID
142  * @param {Hash}	params			a hash of parameters
143  * @param params.mainPkg			[string]	main package that contains the app
144  * @param params.nameKey			[string]	msg key for app name
145  * @param params.icon				[string]	name of app icon class
146  * @param params.textPrecedence	[int]		order for removing button text
147  * @param params.imagePrecedence	[int]		order for removing button image
148  * @param params.chooserTooltipKey	[string]	msg key for app tooltip
149  * @param params.viewTooltipKey	[string]	msg key for app view menu tooltip
150  * @param params.defaultSearch		[constant]	type of item to search for in the app
151  * @param params.organizer			[constant]	main organizer for this app
152  * @param params.overviewTrees		[array]		list of tree IDs to show in overview
153  * @param params.hideZimlets		[boolean]	if true, hide Zimlet tree in overview
154  * @param params.assistants		[hash]		hash of assistant class names and required packages
155  * @param params.searchTypes		[array]		list of types of saved searches to show in overview
156  * @param params.gotoActionCode	[constant]	key action for jumping to this app
157  * @param params.newActionCode		[constant]	default "new" action code
158  * @param params.actionCodes		[hash]		keyboard actions mapped to operations
159  * @param params.newItemOps		[hash]		IDs of operations that create a new item, and their text keys
160  * @param params.newOrgOps			[hash]		IDs of operations that create a new organizer, and their text keys
161  * @param params.qsViews			[array]		list of views to handle in query string
162  * @param params.chooserSort		[int]		controls order of apps in app chooser toolbar
163  * @param params.defaultSort		[int]		controls order in which app is chosen as default start app
164  * @param params.trashViewOp		[constant]	menu choice for "Show Only ..." in Trash view
165  * @param params.upsellUrl			[string]	URL for content of upsell
166  *
167  * @private
168  */
169 ZmApp.registerApp =
170 function(app, params) {
171 
172 	if (params.mainPkg)				{ ZmApp.MAIN_PKG[app]			= params.mainPkg; }
173 	if (params.nameKey)				{ ZmApp.NAME[app]				= params.nameKey; }
174 	if (params.icon)				{ ZmApp.ICON[app]				= params.icon; }
175 	if (params.textPrecedence)		{ ZmApp.TEXT_PRECEDENCE[app]	= params.textPrecedence; }
176 	if (params.imagePrecedence)		{ ZmApp.IMAGE_PRECEDENCE[app]	= params.imagePrecedence; }
177 	if (params.chooserTooltipKey)	{ ZmApp.CHOOSER_TOOLTIP[app]	= params.chooserTooltipKey; }
178 	if (params.viewTooltipKey)		{ ZmApp.VIEW_TOOLTIP[app]		= params.viewTooltipKey; }
179 	if (params.defaultSearch)		{ ZmApp.DEFAULT_SEARCH[app]		= params.defaultSearch; }
180 	if (params.organizer)			{ ZmApp.ORGANIZER[app]			= params.organizer; }
181 	if (params.overviewTrees)		{ ZmApp.OVERVIEW_TREES[app]		= params.overviewTrees; }
182 	if (params.hideZimlets)			{ ZmApp.HIDE_ZIMLETS[app]		= params.hideZimlets; }
183 	if (params.assistants)			{ ZmApp.ASSISTANTS[app]			= params.assistants; }
184 	if (params.searchTypes) 		{ ZmApp.SEARCH_TYPES[app]		= params.searchTypes; }
185 	if (params.gotoActionCode)		{ ZmApp.GOTO_ACTION_CODE[app]	= params.gotoActionCode; }
186 	if (params.newActionCode)		{ ZmApp.NEW_ACTION_CODE[app]	= params.newActionCode; }
187 	if (params.qsViews)				{ ZmApp.QS_VIEWS[app]			= params.qsViews; }
188 	if (params.chooserSort)			{ ZmApp.CHOOSER_SORT[app]		= params.chooserSort; }
189 	if (params.defaultSort)			{ ZmApp.DEFAULT_SORT[app]		= params.defaultSort; }
190 	if (params.trashViewOp)			{ ZmApp.TRASH_VIEW_OP[app]		= params.trashViewOp; }
191 	if (params.upsellUrl)			{ ZmApp.UPSELL_URL[app]			= params.upsellUrl; }
192 
193 	if (params.searchTypes) {
194 		ZmApp.SEARCH_TYPES_R[app] = {};
195 		for (var i = 0; i < params.searchTypes.length; i++) {
196 			ZmApp.SEARCH_TYPES_R[app][params.searchTypes[i]] = true;
197 		}
198 	}
199 
200 	if (params.gotoActionCode) {
201 		ZmApp.GOTO_ACTION_CODE_R[params.gotoActionCode] = app;
202 	}
203 
204 	if (params.actionCodes) {
205 		for (var ac in params.actionCodes) {
206 			if (!ac) { continue; }
207 			ZmApp.ACTION_CODES_R[ac] = app;
208 			ZmApp.ACTION_CODES[ac] = params.actionCodes[ac];
209 		}
210 	}
211 
212     var appEnabled = appCtxt.get(ZmApp.SETTING[app]);
213 	if (params.newItemOps && appEnabled) {
214 		for (var op in params.newItemOps) {
215 			if (!op) { continue; }
216 			ZmApp.OPS_R[op] = app;
217 			ZmOperation.NEW_ITEM_OPS.push(op);
218 			ZmOperation.NEW_ITEM_KEY[op] = params.newItemOps[op];
219 		}
220 	}
221 	if (params.newOrgOps && appEnabled) {
222 		for (var op in params.newOrgOps) {
223 			if (!op) { continue; }
224 			ZmApp.OPS_R[op] = app;
225 			ZmOperation.NEW_ORG_OPS.push(op);
226 			ZmOperation.NEW_ORG_KEY[op] = params.newOrgOps[op];
227 		}
228 	}
229 
230 	if (params.qsViews) {
231 		for (var i = 0; i < params.qsViews.length; i++) {
232 			ZmApp.QS_VIEWS[params.qsViews[i]] = app;
233 		}
234 	}
235 };

379 /**
380  * Adds a deferred folder.

382  * @param	{Hash}	params		a hash of parameters
383  */
384 ZmApp.prototype.addDeferredFolder =
385 function(params) {
386 	var id = params.obj && params.obj.id;
387 	if (id && !this._deferredFolderHash[id]) {
388 		this._deferredFolders.push(params);
389 		this._deferredFolderHash[id] = true;
390 		appCtxt.cacheSetDeferred(id, this._name);

394 /**
395  * Gets the remote folder ids.
396  *
397  * @param	{Object}	account		the account
398  * @return	{Array}		an array of {String} ids
399  */
400 ZmApp.prototype.getRemoteFolderIds =
401 function(account) {
402 	// XXX: optimize by caching this list? It would have to be cleared anytime
403 	// folder structure changes
404 	var list = [];
405 	if (appCtxt.getOverviewController(true)) {
406 		var type = ZmApp.ORGANIZER[this.getName()];
407 
408 		// first, make sure there aren't any deferred folders that need to be created
409 		if (this._deferredFolders.length) {
410 			this._createDeferredFolders(type);
411 		}
412 
413 		var tree = appCtxt.getFolderTree(account);
414 		var folders = tree ? tree.getByType(type) : [];
415 		for (var i = 0; i < folders.length; i++) {
416 			var folder = folders[i];
417 			if (folder.isRemote()) {
418 				list.push(folder.id);
419 			}
420 		}
421 	}
422 	return list;
423 };

501 /**
502  * Resets the current overview, preserving expansion.
503  *
504  * @param {String}		overviewId	the id of overview to reset
505  */
506 ZmApp.prototype.resetOverview =
507 function(overviewId) {
508 	var overview = overviewId ? appCtxt.getOverviewController().getOverview(overviewId) : this.getOverview();
509 	if (overview) {
510 		var expIds = [];
511 		var treeIds = overview.getTreeViews(), len = treeIds.length;
512 		for (var i = 0; i < len; i++) {
513 			var treeId = treeIds[i];
514 			var treeView = overview.getTreeView(treeId);
515 			var items = treeView.getTreeItemList();
516 			var len1 = items.length;
517 			for (var j = 0; j < len1; j++) {
518 				var treeItem = items[j];
519 				if (treeItem._expanded) {
520 					expIds.push(treeItem._htmlElId);
521 				}
522 			}
523 		}
524 		overview.clear();
525 		overview.set(this._getOverviewTrees());
526 		len = expIds.length;
527 		for (var i = 0; i < len; i++) {
528 			var treeItem = DwtControl.fromElementId(expIds[i]);
529 			if (treeItem && !treeItem._expanded) {
530 				treeItem.setExpanded(true);
531 			}
532 		}
533 	}
534 };

547 /**
548  * Returns a hash of params with the standard overview options.
549  *
550  * @private
551  */
552 ZmApp.prototype._getOverviewParams =
553 function() {
554 	// Get the sorted list of overview trees.
555 	var treeIds = [];
556 	for (var id in ZmOverviewController.CONTROLLER) {
557 		treeIds.push(id);
558 	}
559 	var sortFunc = function(a, b) {
560 		return (ZmOrganizer.DISPLAY_ORDER[a] || 9999) - (ZmOrganizer.DISPLAY_ORDER[b] || 9999);
561 	};
562 	treeIds.sort(sortFunc);
563 
564 	return {
565 		posStyle:			Dwt.ABSOLUTE_STYLE,
566 		selectionSupported:	true,
567 		actionSupported:	true,
568 		dndSupported:		true,
569 		showUnread:			true,
570 		showNewButtons:		true,
571 		isAppOverview:		true,
572 		treeIds:			treeIds,
573 		appName:			this._name,
574 		account:			appCtxt.getActiveAccount()
575 	};
576 };

578 /**
579  * Returns the list of trees to show in the overview for this app. Don't show
580  * Folders unless mail is enabled. Other organizer types won't be created unless
581  * their apps are enabled, so we don't need to check for them.
582  *
583  * @private
584  */
585 ZmApp.prototype._getOverviewTrees =
586 function() {
587 	var list = ZmApp.OVERVIEW_TREES[this._name] || [];
588 	var newList = [];

622 ZmApp.prototype.getSessionController =
623 function(type, controllerClass, sessionId) {
624 
625 	if (!this._sessionController[type]) {
626 		this._sessionController[type] = {};
627 		this._sessionId[type] = 1;
628 	}
629 
630 	if (sessionId && this._sessionController[type][sessionId]) {
631 		return this._sessionController[type][sessionId];
632 	}
633 
634 	var controllers = this._sessionController[type];
635 	var controller;
636 	for (var id in controllers) {
637 		if (controllers[id].inactive) {
638 			controller = controllers[id];
639 			break;
640 		}
641 	}
642 
643 	sessionId = controller ? controller.sessionId : this._sessionId[type]++;
644 
645 	if (!controller) {
646 		var ctlrClass = eval(controllerClass);
647 		controller = this._sessionController[type][sessionId] = new ctlrClass(this._container, this);
648 	}
649 	controller.setSessionId(type, sessionId);
650 	this._curSessionId[type] = sessionId;
651 	controller.inactive = false;
652 
653 	return controller;
654 };

723 			if (ctlr) {

727 					ZmTreeController.DROP_SOURCES[orgType] = [];
728 				}
729 				ZmTreeController.DROP_SOURCES[orgType].push(className);
730 			}
731 		}
732 	}

801 /**
802  * Defer notifications if this app's main package has not been loaded.
803  * Returns true if notifications were deferred.
804  *
805  * @param type	[string]	type of notification (delete, create, or modify)
806  * @param data	[array]		list of notifications
807  *
808  * TODO: revisit use of MAIN_PKG, it's hokey

811  */
812 ZmApp.prototype._deferNotifications =
813 function(type, data) {

824 /**
825  * @private
826  */
827 ZmApp.prototype._handleDeferredNotifications =
828 function() {
829 	var dns = this._deferredNotifications;
830 	for (var i = 0; i < dns.length; i++) {
831 		var dn = dns[i];
832 		if (dn.type == "delete") {
833 			this.deleteNotify(dn.data, true);
834 		} else if (dn.type == "create") {
835 			this.createNotify(dn.data, true);
836 		} else if (dn.type == "modify") {
837 			this.modifyNotify(dn.data, true);
838 		}
839 	}

842 /**
843  * Notify change listeners with a list of notifications, rather than a single
844  * item, so that they can optimize. For example, a list view can wait to
845  * fix its alternation of dark and light rows until after all the moved ones
846  * have been taken out, rather than after the removal of each row.
847  *
848  * @param mods	{Array}		list of notification objects
849  */
850 ZmApp.prototype._batchNotify =
851 function(mods) {
852 
853 	if (!(mods && mods.length >= ZmApp.BATCH_NOTIF_LIMIT)) { return; }
854 
855 	var notifs = {}, item, gotOne = false;
856 	for (var i = 0, len = mods.length; i < len; i++) {
857 		var mod = mods[i];
858 		item = appCtxt.cacheGet(mod.id);
859 		if (item) {
860 			var ev = item.notifyModify(mod, true);
861 			if (ev) {
862 				if (!notifs[ev]) {
863 					notifs[ev] = [];
864 				}
865 				mod.item = item;
866 				notifs[ev].push(mod);
867 				gotOne = true;
868 			}
869 		}
870 	}
871 
872 	if (!gotOne || !item) { return; }
873 
874 	var list = item.list;
875 	if (!list) { return; }
876 	list._evt.batchMode = true;
877 	list._evt.item = item;	// placeholder - change listeners like it to be there
878 	list._evt.items = null;
879 	for (var ev in notifs) {
880 		var details = {notifs:notifs[ev]};
881 		list._notify(ev, details);
882 	}
883 };
