 16 /**
 17  * Creates an empty list of mail items.
 18  * @constructor
 19  * @class
 20  * This class represents a list of mail items (conversations, messages, or
 21  * attachments). We retain a handle to the search that generated the list for
 22  * two reasons: so that we can redo the search if necessary, and so that we
 23  * can get the folder ID if this list represents folder contents.
 24  *
 25  * @author Conrad Damon
 26  * 
 27  * @param type		type of mail item (see ZmItem for constants)
 28  * @param search	the search that generated this list
 29  */
 30 ZmMailList = function(type, search) {
 31 
 32 	ZmList.call(this, type, search);
 33 
 34 	this.convId = null; // for msg list within a conv
 35 
 36 	// mail list can be changed via folder or tag action (eg "Mark All Read")
 37 	var folderTree = appCtxt.getFolderTree();
 38 	if (folderTree) {
 39 		this._folderChangeListener = new AjxListener(this, this._folderTreeChangeListener);
 40 		folderTree.addChangeListener(this._folderChangeListener);
 41 	}
 42 };

 52 /**
 53  * Override so that we can specify "tcon" attribute for conv move - we don't want
 54  * to move messages in certain system folders as a side effect. Also, we need to
 55  * update the UI based on the response if we're moving convs, since the 
 56  * notifications only tell us about moved messages. This method should be called
 57  * only in response to explicit action by the user, in which case we want to
 58  * remove the conv row(s) from the list view (even if the conv still matches the
 59  * search).
 60  *
 61  * @param {Hash}	params		a hash of parameters
 62  *        items			[array]			a list of items to move
 63  *        folder		[ZmFolder]		destination folder
 64  *        attrs			[hash]			additional attrs for SOAP command
 65  *        callback		[AjxCallback]*	callback to run after each sub-request
 66  *        finalCallback	[AjxCallback]*	callback to run after all items have been processed
 67  *        count			[int]*			starting count for number of items processed
 68  *        fromFolderId  [String]*       optional folder to represent when calculating tcon. If unspecified, use current search folder nId
 69  *        
 70  * @private
 71  */
 72 ZmMailList.prototype.moveItems =
 73 function(params) {
 74 
 75 	if (this.type != ZmItem.CONV) {
 76 		return ZmList.prototype.moveItems.apply(this, arguments);
 77 	}
 78 
 79 	params = Dwt.getParams(arguments, ["items", "folder", "attrs", "callback", "finalCallback", "noUndo", "actionText", "fromFolderId"]);
 80 	params.items = AjxUtil.toArray(params.items);
 81 
 82 	var params1 = AjxUtil.hashCopy(params);
 83 	delete params1.fromFolderId;
 84 
 85 	params1.attrs = {};
 86 	var tcon = this._getTcon(params.items, params.fromFolderId);
 87 	if (tcon) {
 88 		params1.attrs.tcon = tcon;
 89 	}
 90 	params1.attrs.l = params.folder.id;
 91 	params1.action = (params.folder.id == ZmFolder.ID_TRASH) ? "trash" : "move";
 92     if (params1.folder.id == ZmFolder.ID_TRASH) {
 93         params1.actionText = params.actionText || ZmMsg.actionTrash;
 94     } else {
 95         params1.actionText = params.actionText || ZmMsg.actionMove;
 96         params1.actionArg = params1.folder.getName(false, false, true);
 97     }
 98 	params1.callback = new AjxCallback(this, this._handleResponseMoveItems, [params]);
 99 
100 	if (appCtxt.multiAccounts) {
101 		// Reset accountName for multi-account to be the respective account if we're
102 		// moving a draft out of Trash.
103 		// OR,
104 		// check if we're moving to or from a shared folder, in which case, always send
105 		// request on-behalf-of the account the item originally belongs to.
106         var folderId = params.items[0].getFolderId && params.items[0].getFolderId();
107 
108         // on bulk delete, when the second chunk loads try to get folderId from the item id.
109         if (!folderId) {
110             var itemId = params.items[0] && params.items[0].id;
111             folderId = itemId && appCtxt.getById(itemId) && appCtxt.getById(itemId).folderId;
112         }
113         var fromFolder = folderId && appCtxt.getById(folderId);
114 		if ((params.items[0].isDraft && params.folder.id == ZmFolder.ID_DRAFTS) ||
115 			(params.folder.isRemote()) || (fromFolder && fromFolder.isRemote()))
116 		{
117 			params1.accountName = params.items[0].getAccount().name;
118 		}
119 	}
120 
121 	this._itemAction(params1);
122 };

124 /**
125  * Marks items as "spam" or "not spam". If they're marked as "not spam", a target folder
126  * may be provided.
127  * @param {Hash}	params		a hash of parameters
128  *        items			[array]			a list of items
129  *        markAsSpam	[boolean]		if true, mark as "spam"
130  *        folder		[ZmFolder]		destination folder
131  *        childWin		[window]*		the child window this action is happening in
132  *        closeChildWin	[boolean]*		is the child window closed at the end of the action?
133  *        callback		[AjxCallback]*	callback to run after each sub-request
134  *        finalCallback	[AjxCallback]*	callback to run after all items have been processed
135  *        count			[int]*			starting count for number of items processed
136  * @private

138 ZmMailList.prototype.spamItems = 
139 function(params) {
140 
141 	var items = params.items = AjxUtil.toArray(params.items);
142 
143 	if (appCtxt.multiAccounts) {
144 		var accounts = this._filterItemsByAccount(items);
145 		this._spamAccountItems(accounts, params);
146 	} else {
147 		this._spamItems(params);
148 	}
149 };
150 
151 ZmMailList.prototype._spamAccountItems =
152 function(accounts, params) {
153 	var items;
154 	for (var i in accounts) {
155 		items = accounts[i];
156 		break;
157 	}
158 
159 	if (items) {
160 		delete accounts[i];
161 
162 		params.accountName = appCtxt.accountList.getAccount(i).name;
163 		params.items = items;
164 		params.callback = new AjxCallback(this, this._spamAccountItems, [accounts, params]);
165 
166 		this._spamItems(params);
167 	}
168 };

170 ZmMailList.prototype._spamItems =
171 function(params) {
172 	params = Dwt.getParams(arguments, ["items", "markAsSpam", "folder", "childWin"]);
173 
174 	var params1 = AjxUtil.hashCopy(params);
175 
176 	params1.action = params.markAsSpam ? "spam" : "!spam";
177 	params1.attrs = {};
178 	var tcon = this._getTcon(params.items);
179 	if (tcon) {
180 		params1.attrs.tcon = tcon;
181 	}
182 	if (params.folder) {
183 		params1.attrs.l = params.folder.id;
184 	}
185 	params1.actionText = params.markAsSpam ? ZmMsg.actionMarkAsJunk : ZmMsg.actionMarkAsNotJunk;
186 
187 	params1.callback = new AjxCallback(this, this._handleResponseSpamItems, params);
188 	this._itemAction(params1);
189 };

191 ZmMailList.prototype._handleResponseSpamItems =
192 function(params, result) {
193 
194 	var movedItems = result.getResponse();
195 	var summary;
196 	if (movedItems && movedItems.length) {
197 		var folderId = params.markAsSpam ? ZmFolder.ID_SPAM : (params.folder ? params.folder.id : ZmFolder.ID_INBOX);
198 		this.moveLocal(movedItems, folderId);
199 		var convs = {};
200 		for (var i = 0; i < movedItems.length; i++) {
201 			var item = movedItems[i];
202 			if (item.cid) {
203 				var conv = appCtxt.getById(item.cid);
204 				if (conv) {
205 					if (!convs[conv.id])
206 						convs[conv.id] = {conv:conv,msgs:[]};
207 					convs[conv.id].msgs.push(item);
208 				}
209 			}
210 			var details = {oldFolderId:item.folderId, fields:{}};
211 			details.fields[ZmItem.F_FRAGMENT] = true;
212 			item.moveLocal(folderId);
213 		}
214 
215 		for (var id in convs) {
216 			if (convs.hasOwnProperty(id)) {
217 				var conv = convs[id].conv;
218 				var msgs = convs[id].msgs;
219 				conv.updateFragment(msgs);
220 			}
221 		}
222 		//ZmModel.notifyEach(movedItems, ZmEvent.E_MOVE);
223 		
224 		var item = movedItems[0];
225 		var list = item.list;
226 		list._evt.batchMode = true;
227 		list._evt.item = item;	// placeholder
228 		list._evt.items = movedItems;
229 		list._notify(ZmEvent.E_MOVE, details);
230 
231 		if (params.actionText) {
232 			summary = ZmList.getActionSummary(params.actionText, params.numItems, params.type, params.actionArg);
233 		}
234 
235 		if (params.childWin) {
236 			params.childWin.close();
237 		}
238 	}
239 	params.actionSummary = summary;
240 	if (params.callback) {
241 		params.callback.run(result);
242 	}
243 };

245 /**
246  * Override so that delete of a conv in Trash doesn't hard-delete its msgs in
247  * other folders. If we're in conv mode in Trash, we add a constraint of "t",
248  * meaning that the action is only applied to items (msgs) in the Trash.
249  *
250  * @param {Hash}		params		a hash of parameters
251  * @param  {Array}     params.items			list of items to delete
252  * @param {Boolean}      params.hardDelete	whether to force physical removal of items
253  * @param {Object}      params.attrs			additional attrs for SOAP command
254  * @param {window}       params.childWin		the child window this action is happening in
255  *        
256  * @private
257  */
258 ZmMailList.prototype.deleteItems =
259 function(params) {
260 
261 	params = Dwt.getParams(arguments, ["items", "hardDelete", "attrs", "childWin"]);
262 
263 	if (this.type == ZmItem.CONV) {
264 		var searchFolder = this.search ? appCtxt.getById(this.search.folderId) : null;
265 		if (searchFolder && searchFolder.isHardDelete()) {
266 			var instantOn = appCtxt.getAppController().getInstantNotify();
267 			if (instantOn) {
268 				// bug fix #32005 - disable instant notify for ops that might take awhile
269 				appCtxt.getAppController().setInstantNotify(false);
270 				params.errorCallback = new AjxCallback(this, this._handleErrorDeleteItems);
271 			}
272 
273 			params.attrs = params.attrs || {};
274 			params.attrs.tcon = ZmFolder.TCON_CODE[searchFolder.nId];
275 			params.action = "delete";
276             params.actionText = ZmMsg.actionDelete;
277 			params.callback = new AjxCallback(this, this._handleResponseDeleteItems, instantOn);
278 			return this._itemAction(params);
279 		}
280 	}
281 	ZmList.prototype.deleteItems.call(this, params);
282 };
283 
284 ZmMailList.prototype._handleResponseDeleteItems =
285 function(instantOn, result) {
286 	var deletedItems = result.getResponse();
287 	if (deletedItems && deletedItems.length) {
288 		this.deleteLocal(deletedItems);
289 		for (var i = 0; i < deletedItems.length; i++) {
290 			deletedItems[i].deleteLocal();
291 		}
292 		// note: this happens before we process real notifications
293 		ZmModel.notifyEach(deletedItems, ZmEvent.E_DELETE);
294 	}
295 
296 	if (instantOn) {
297 		appCtxt.getAppController().setInstantNotify(true);
298 	}
299 };

301 ZmMailList.prototype._handleErrorDeleteItems =

303 	appCtxt.getAppController().setInstantNotify(true);

306 /**
307  * Only make the request for items whose state will be changed.
308  *
309  * @param {Hash}		params		a hash of parameters
310  *
311  *        items			[array]				a list of items to mark read/unread
312  *        value			[boolean]			if true, mark items read
313  *        callback		[AjxCallback]*		callback to run after each sub-request
314  *        finalCallback	[AjxCallback]*		callback to run after all items have been processed
315  *        count			[int]*				starting count for number of items processed
316  *        
317  * @private
318  */
319 ZmMailList.prototype.markRead =
320 function(params) {
321 
322 	var items = AjxUtil.toArray(params.items);
323 
324 	var items1;
325 	if (items[0] && items[0] instanceof ZmItem) {
326 		items1 = [];
327 		for (var i = 0; i < items.length; i++) {
328 			var item = items[i];
329 			if ((item.type == ZmItem.CONV && item.hasFlag(ZmItem.FLAG_UNREAD, params.value)) || (item.isUnread == params.value)) {
330 				items1.push(item);
331 			}
332 		}
333 	} else {
334 		items1 = items;
335 	}
336 
337 	if (items1.length) {
338 		params.items = items1;
339 		params.op = "read";
340 		if (items1.length > 1) {
341         	params.actionText = params.value ? ZmMsg.actionMarkRead : ZmMsg.actionMarkUnread;
342 		}
343 		this.flagItems(params);
344 	}
345 };

347 // set "force" flag to true on actual hard deletes, so that msgs
348 // in a conv list are removed
349 ZmMailList.prototype.deleteLocal =
350 function(items) {
351 	for (var i = 0; i < items.length; i++) {
352 		this.remove(items[i], true);
353 	}
354 };

356 // When a conv or msg is moved to Trash, it is marked read by the server.
357 ZmMailList.prototype.moveLocal =
358 function(items, folderId) {
359 	ZmList.prototype.moveLocal.call(this, items, folderId);
360 	if (folderId != ZmFolder.ID_TRASH) { return; }
361 
362 	var flaggedItems = [];
363 	for (var i = 0; i < items.length; i++) {
364 		if (items[i].isUnread) {
365 			items[i].flagLocal(ZmItem.FLAG_UNREAD, false);
366 			flaggedItems.push(items[i]);
367 		}
368 	}
369 	ZmModel.notifyEach(flaggedItems, ZmEvent.E_FLAGS, {flags:[ZmItem.FLAG_UNREAD]});
370 };

372 ZmMailList.prototype.notifyCreate = 
373 function(convs, msgs) {
374 
375 	var createdItems = [];
376 	var newConvs = [];
377 	var newMsgs = [];
378 	var flaggedItems = [];
379 	var modifiedItems = [];
380 	var newConvId = {};
381 	var fields = {};
382 	var sortBy = this.search ? this.search.sortBy : null;
383 	var sortIndex = {};
384 	if (this.type == ZmItem.CONV) {
385 		// handle new convs first so we can set their fragments later from new msgs
386 		for (var id in convs) {
387 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: handling conv create " + id);
388 			if (this.getById(id)) {
389 				AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: conv already exists " + id);
390 				continue;
391 			}
392 			newConvId[id] = convs[id];
393 			var conv = convs[id];
394 			if (this.search && this.search.matches && this.search.matches(conv) && !conv.ignoreJunkTrash()) {
395 				if (!appCtxt.multiAccounts ||
396 					(appCtxt.multiAccounts && (this.search.isMultiAccount() || conv.getAccount() == appCtxt.getActiveAccount()))) 
397 				{
398 					// a new msg for this conv matches current search
399 					conv.list = this;
400 					newConvs.push(conv);
401 					AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: conv added " + id);
402                     appCtxt.setNotifyDebug("Handling NOTIFY: notifyCreate ZmMailList --- New conv added");
403 				}
404 				else {
405 					AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: conv failed account checks " + id);
406 				}
407 			}
408 			else {
409 				// debug info for bug 47589
410 				var query = this.search ? this.search.query : "";
411 				var ignore = conv.ignoreJunkTrash();
412 				AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: conv does not match search '" + query + "' or was ignored (" + ignore + "); match function:");
413 				var matchFunc = (this.search && this.search.matches) || "";
414 				AjxDebug.println(AjxDebug.NOTIFY, matchFunc.toString());
415 				if (!conv) {
416 					AjxDebug.println(AjxDebug.NOTIFY, "conv is null!");
417 				}
418 				else {
419 					var folders = AjxUtil.keys(conv.folders);
420 					AjxDebug.println(AjxDebug.NOTIFY, "conv spans " + folders.length + " folder(s): " + folders.join(" "));
421 				}
422 			}
423 		}
424 
425 		// a new msg can hand us a new conv, and update a conv's info
426 		for (var id in msgs) {
427 			var msg = msgs[id];
428 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: CLV handling msg create " + id);
429 			var cid = msg.cid;
430 			var matchFunc = this.search && this.search.matches;
431 			var msgMatches =  matchFunc && matchFunc(msg) && !msg.ignoreJunkTrash();
432 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: CLV msg matches: " + msgMatches);
433 			var isActiveAccount = (!appCtxt.multiAccounts || (appCtxt.multiAccounts && msg.getAccount() == appCtxt.getActiveAccount()));
434 			var conv = newConvId[cid] || this.getById(cid);
435 			if (msgMatches && isActiveAccount) {
436 				if (!conv) {
437 					// msg will have _convCreateNode if it is 2nd msg and caused promotion of virtual conv;
438 					// the conv node will have proper count and subject
439 					var args = {list:this};
440 					if (msg._convCreateNode) {
441 						if (msg._convCreateNode._newId) {
442 							msg._convCreateNode.id = msg._convCreateNode._newId;
443 						}
444 						conv = ZmConv.createFromDom(msg._convCreateNode, args);
445 					} else {
446 						conv = appCtxt.getById(cid) || ZmConv.createFromMsg(msg, args);
447 					}
448 					newConvId[cid] = conv;
449 					conv.folders[msg.folderId] = true;
450 					newConvs.push(conv);
451                     appCtxt.setNotifyDebug("Handling NOTIFY: in ZmMailList - notifyCreate - New message becomes a conv");
452 				}
453 				conv.list = this;
454 			}
455 			// make sure conv's msg list is up to date
456 			if (conv && !(conv.msgs && conv.msgs.getById(id))) {
457 				if (!conv.msgs) {
458 					conv.msgs = new ZmMailList(ZmItem.MSG);
459 					conv.msgs.addChangeListener(conv._listChangeListener);
460 				}
461 				msg.list = conv.msgs;
462 				if (!msg.isSent && msg.isUnread) {
463 					conv.isUnread = true;
464 					flaggedItems.push(conv);
465 				}
466 				// if the new msg matches current search, update conv date, fragment, and sort order
467 				if (msgMatches) {
468 					msg.inHitList = true;
469 				}
470 				if (msgMatches || (!matchFunc && !msg.isSent)) {
471 					if (conv.fragment != msg.fragment) {
472 						conv.fragment = msg.fragment;
473 						fields[ZmItem.F_FRAGMENT] = true;
474 					}
475 					if (conv.date != msg.date) {
476 						conv.date = msg.date;
477 						// recalculate conv's sort position since we changed its date
478 						fields[ZmItem.F_DATE] = true;
479 					}
480 					// conv gained a msg, may need to be moved to top/bottom
481 					if (!newConvId[conv.id] && this._vector.contains(conv)) {
482 						fields[ZmItem.F_INDEX] = true;
483 					}
484 					modifiedItems.push(conv);
485 				}
486 				AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: conv list accepted msg " + id);
487 				newMsgs.push(msg);
488 			}
489 		}
490 	} else if (this.type == ZmItem.MSG) {
491 		// add new msg to list
492 		for (var id in msgs) {
493 			var msg = msgs[id];
494 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: handling msg create " + id);
495 			if (this.getById(id)) {
496 				if (this.search.matches && this.search.matches && this.search.matches(msg) && !msg.ignoreJunkTrash()) {
497 					var query = this.search ? this.search.query : "";
498 					var ignore = msg.ignoreJunkTrash();
499 					AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: msg does not match search '" + query + "' or was ignored (" + ignore + ")");
500 					msg.list = this; // Even though we have the msg in the list, it sometimes has its list wrong.
501 				}
502 				continue;
503 			}
504 			if (this.convId) { // MLV within CV
505 				if (msg.cid == this.convId && !this.getById(msg.id)) {
506 					msg.list = this;
507 					AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: msg list (CV) accepted msg " + id);
508 					newMsgs.push(msg);
509 				}
510 			} else { // MLV (traditional)
511 				if (this.search.matches && this.search.matches && this.search.matches(msg) && !msg.ignoreJunkTrash()) {
512 					msg.list = this;
513 					AjxDebug.println(AjxDebug.NOTIFY, "ZmMailList: msg list (TV) accepted msg " + id);
514 					newMsgs.push(msg);
515 				}
516 			}
517 		}
518         if (window.isNotifyDebugOn && newMsgs.length > 1) {
519             appCtxt.setNotifyDebug("Handling NOTIFY: notifyCreate ZmMailList --- New message added to list");
520         }
521 	}
522 
523 	// sort item list in reverse so they show up in correct order when processed (oldest appears first)
524 	if (newConvs.length > 1) {
525 		ZmMailItem.sortBy = sortBy;
526 		newConvs.sort(ZmMailItem.sortCompare);
527 		newConvs.reverse();
528 	}
529 
530 	this._sortAndNotify(newConvs, sortBy, ZmEvent.E_CREATE);
531 	this._sortAndNotify(newMsgs, sortBy, ZmEvent.E_CREATE);
532 	ZmModel.notifyEach(flaggedItems, ZmEvent.E_FLAGS, {flags:[ZmItem.FLAG_UNREAD]});
533 	this._sortAndNotify(modifiedItems, sortBy, ZmEvent.E_MODIFY, {fields:fields});
534 	this._sortAndNotify(newMsgs, sortBy, ZmEvent.E_MODIFY, {fields:fields});
535 };

537 /**
538 * Convenience method for adding messages to a conv on the fly. The specific use case for
539 * this is when a virtual conv becomes real. We basically add the new message(s) to the
540 * old (virtual) conv's message list.
541 *
542 * @param msgs		hash of messages to add
543 */
544 ZmMailList.prototype.addMsgs =
545 function(msgs) {
546 	var addedMsgs = [];
547 	for (var id in msgs) {
548 		var msg = msgs[id];
549 		if (msg.cid == this.convId) {
550 			this.add(msg, 0);
551 			msg.list = this;
552 			addedMsgs.push(msg);
553 		}
554 	}

586 /**
587  * Gets the first msg in the list which was marked by the server as matching
588  * the search used to create the list.
589  * 
590  * @param {int}	offset	the starting point within list
591  * @param {int}	limit		the ending point within list
592  * @return	{ZmMailMsg}		the message
593  */
594 ZmMailList.prototype.getFirstHit =
595 function(offset, limit) {
596 	if (this.type != ZmItem.MSG) { return null; }
597 
598 	var msg = null;	
599 	offset = offset || 0;
600 	limit = limit || appCtxt.get(ZmSetting.CONVERSATION_PAGE_SIZE);
601 	var numMsgs = this.size();
602 
603 	if (numMsgs > 0 && offset >= 0 && offset < numMsgs) {
604 		var end = (offset + limit > numMsgs) ? numMsgs : offset + limit;
605 		var list = this.getArray();
606 		for (var i = offset; i < end; i++) {
607 			if (list[i].inHitList) {
608 				msg = list[i];
609 				break;
610 			}
611 		}
612 		if (!msg) {
613 			msg = list[0];	// no hot messages, use first msg
614 		}
615 	}
616 	
617 	return msg;
618 };

620 /**
621  * Returns the insertion point for the given item into this list. If we're not sorting by
622  * date, returns 0 (the item will be inserted at the top of the list).
623  *
624  * @param item		[ZmMailItem]	a mail item
625  * @param sortBy		[constant]		sort order
626  */
627 ZmMailList.prototype._getSortIndex =
628 function(item, sortBy) {
629 	if (!sortBy || (sortBy != ZmSearch.DATE_DESC && sortBy != ZmSearch.DATE_ASC)) {
630 		return 0;
631 	}
632 	
633 	var itemDate = parseInt(item.date);
634 	var a = this.getArray();
635 	for (var i = 0; i < a.length; i++) {
636 		var date = parseInt(a[i].date);
637 		if ((sortBy == ZmSearch.DATE_DESC && (itemDate >= date)) ||
638 			(sortBy == ZmSearch.DATE_ASC && (itemDate <= date))) {
639 			return i;
640 		}
641 	}
642 	return i;
643 };

645 ZmMailList.prototype._sortAndNotify =
646 function(items, sortBy, event, details) {
647 
648 	if (!(items && items.length)) { return; }
649 
650 	var itemType = items[0] && items[0].type;
651 	if ((this.type == ZmItem.MSG) && (itemType == ZmItem.CONV)) { return; }
652 
653 	details = details || {};
654 	var doSort = ((event == ZmEvent.E_CREATE) || (details.fields && details.fields[ZmItem.F_DATE]));
655 	for (var i = 0; i < items.length; i++) {
656 		var item = items[i];
657 		if (doSort) {
658 			var doAdd = (itemType == this.type);
659 			var sortIndex = 0;
660 			if (this.type == ZmItem.CONV && itemType == ZmItem.MSG) {
661 				var conv = this.getById(item.cid);
662 				if (conv) {
663 					sortIndex = conv.msgs._getSortIndex(item, conv._sortBy);
664 					if (event == ZmEvent.E_CREATE) {
665 						conv.addMsg(item, sortIndex);
666 					}
667 				}
668 			} else {
669 				sortIndex = this._getSortIndex(item, sortBy);
670 			}
671 			if (event != ZmEvent.E_CREATE) {
672 				// if date changed, re-insert item into correct slot
673 				if (sortIndex != this.indexOf(item)) {
674 					this.remove(item);
675 				} else {
676 					doAdd = false;
677 				}
678 			}
679 			if (doAdd) {
680 				this.add(item, sortIndex);
681 			}
682 			details.sortIndex = sortIndex;
683 		}
684 		item._notify(event, details);
685 	}
686 };

688 ZmMailList.prototype._getTcon =
689 function(items, nId) {
690 	var chars = [];
691 	var folders = [ZmFolder.ID_TRASH, ZmFolder.ID_SPAM, ZmFolder.ID_SENT];
692     var id;
693     if(!nId){
694         var searchFolder = this.search && appCtxt.getById(this.search.folderId);
695         if(searchFolder){
696             nId = searchFolder.isRemote() ? searchFolder.rid : searchFolder.nId;
697             id = searchFolder.id;
698         }
699     }
700 
701 	for (var i = 0; i < folders.length; i++) {
702 		var folderId = folders[i];
703         var folder;
704         // get folder object from qualified Ids for multi-account
705         if (appCtxt.multiAccounts) {
706             var acct  = items && items[0].getAccount && items[0].getAccount();
707             var acctId = acct ? acct.id : appCtxt.getActiveAccount().id;
708             var fId = [acctId, ":", folderId].join("");
709             folder = appCtxt.getById(fId);
710         } else {
711             folder = appCtxt.getById(folderId);
712         }
713         var nFolder = (id) ? appCtxt.getById(id) : appCtxt.getById(nId);
714         // if nId is undefined send the default tcon [-tjs].
715 		if (!nId || (nId != folderId && folder && nFolder && !nFolder.isChildOf(folder))) {
716 			chars.push(ZmFolder.TCON_CODE[folderId]);
717 		}
718 	}
719 	return (chars.length) ?  ("-" + chars.join("")) : "";
720 };

722 // If this list is the result of a search that is constrained by the read
723 // status, and the user has marked all read in a folder, redo the search.
724 ZmMailList.prototype._folderTreeChangeListener = 
725 function(ev) {
726 	if (this.size() == 0) { return; }
727 
728 	var flag = ev.getDetail("flag");
729 	var view = appCtxt.getCurrentViewId();
730 	var ctlr = appCtxt.getCurrentController();
731 
732 	if (ev.event == ZmEvent.E_FLAGS && (flag == ZmItem.FLAG_UNREAD)) {
733 		if (this.type == ZmItem.CONV) {
734 			if (view == ZmId.VIEW_CONVLIST && ctlr._currentSearch.hasUnreadTerm) {
735 				this._redoSearch(ctlr);
736 			}
737 		} else if (this.type == ZmItem.MSG) {
738 			if (view == ZmId.VIEW_TRAD && ctlr._currentSearch.hasUnreadTerm) {
739 				this._redoSearch(ctlr);
740 			} else {
741 				var on = ev.getDetail("state");
742 				var organizer = ev.getDetail("item");
743 				var flaggedItems = [];
744 				var list = this.getArray();
745 				for (var i = 0; i < list.length; i++) {
746 					var msg = list[i];
747 					if ((organizer.type == ZmOrganizer.FOLDER && msg.folderId == organizer.id) ||
748 						(organizer.type == ZmOrganizer.TAG && msg.hasTag(organizer.id))) {
749 						msg.isUnread = on;
750 						flaggedItems.push(msg);
751 					}
752 				}
753 				ZmModel.notifyEach(flaggedItems, ZmEvent.E_FLAGS, {flags:[flag]});
754 			}
755 		}
756 	} else {
757 		ZmList.prototype._folderTreeChangeListener.call(this, ev);
758 	}
759 };
760 
761 ZmMailList.prototype._tagTreeChangeListener = 
762 function(ev) {
763 	if (this.size() == 0) return;
764 
765 	var flag = ev.getDetail("flag");
766 	if (ev.event == ZmEvent.E_FLAGS && (flag == ZmItem.FLAG_UNREAD)) {
767 		this._folderTreeChangeListener(ev);
768 	} else {
769 		ZmList.prototype._tagTreeChangeListener.call(this, ev);
770 	}
771 };
