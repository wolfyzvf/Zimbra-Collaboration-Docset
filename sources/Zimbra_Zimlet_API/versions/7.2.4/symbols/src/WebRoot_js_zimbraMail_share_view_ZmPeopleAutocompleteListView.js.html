 51 // Creates the list and its member elements based on the matches we have. Each match becomes a
 52 // row. The first match is automatically selected.
 53 ZmPeopleAutocompleteListView.prototype._set =
 54 function(list) {
 55 	var table = this._getTable();
 56 	this._matches = list;
 57 
 58 	for (var i = 0; i < list.length; i++) {
 59 		var match = list[i];
 60 		if (match && (match.text || match.icon)) {
 61 			var rowId = match.id = this._getId("Row", i);
 62 			this._matchHash[rowId] = match;
 63 		}
 64 
 65 		var rowId = this._getId("Row", i);
 66 		var contact = match.item;
 67 		var data = {
 68 			id: this._htmlElId,
 69 			rowId: rowId,
 70 			fullName: contact.getFullName(),
 71 			title: contact.getAttr(ZmContact.F_jobTitle),
 72 			email: contact.getEmail(),
 73             phone: contact.getAttr(ZmContact.F_workPhone),
 74             photoFileName: contact.getAttr("photoFileName")
 75 		};
 76 
 77         // zimlet support
 78         appCtxt.notifyZimlets("onPeopleSearchData", [data]);
 79         
 80 
 81 		var rowHtml = AjxTemplate.expand("share.Widgets#ZmPeopleAutocompleteListView", data);
 82 
 83         var row = Dwt.parseHtmlFragment(rowHtml, true);
 84         var tbody = document.createElement("tbody");
 85         tbody.appendChild(row);
 86 		var rowEl = table.appendChild(tbody);
 87 
 88         if (data.email){
 89             var emailTxt = new DwtText({parent:this, parentElement:rowId + "-email", index:0, id:"NewMsg", className:"FakeAnchor"});
 90             emailTxt.isLinkText = true;
 91             emailTxt.setText(data.email);
 92             emailTxt.addListener(DwtEvent.ONMOUSEDOWN, new AjxListener(this, this._peopleItemListener));
 93             emailTxt.addListener(DwtEvent.ONMOUSEOVER, new AjxListener(this, this.peopleItemMouseOverListener));
 94             emailTxt.addListener(DwtEvent.ONMOUSEOUT, new AjxListener(this, this.peopleItemMouseOutListener));
 95         }
 96 
 97         if (data.fullName){
 98             var nameTxt = new DwtText({parent:this, parentElement:rowId + "-fullName", index:0, id:"NewContact", className:"ZmPeopleSearch-fullname"});
 99             nameTxt.isLinkText = true;
100             nameTxt.setText(data.fullName);
101             nameTxt.addListener(DwtEvent.ONMOUSEDOWN, new AjxListener(this, this._peopleItemListener));
102             nameTxt.addListener(DwtEvent.ONMOUSEOVER, new AjxListener(this, this.peopleItemMouseOverListener));
103             nameTxt.addListener(DwtEvent.ONMOUSEOUT, new AjxListener(this, this.peopleNameMouseOutListener));
104         }
105 		Dwt.associateElementWithObject(row, contact, "contact");
106         // ask zimlets if they want to make data into links
107 		appCtxt.notifyZimlets("onPeopleSearchShow", [this, contact, rowId]);
108 
109         if (i==0)
110             this._setSelected(rowId);
111 
112 	}
113 
114 	//leave this out as part of bug 50692
115 	/*
116 	     //fetch free/busy info for all results;
117 	    if (list.length > 0) {
118 		AjxTimedAction.scheduleAction(new AjxTimedAction(this, this._getFreeBusyInfo, [list]), 100);
119 	} */
120 };

143 ZmPeopleAutocompleteListView.prototype._setSelected =
144 function(id) {
145     if (id && typeof id == "string") {
146         id = id.split("-")[0]; 
147     }
148 
149     if (id == ZmPeopleAutocompleteListView.NO_RESULTS || id == this.getHtmlElement().id) { return; }
150 
151 	if (id == ZmAutocompleteListView.NEXT || id == ZmAutocompleteListView.PREV) {
152 		var table = document.getElementById(this._tableId);
153 		var rows = table && table.rows;
154 		id = this._getRowId(rows, id, rows.length);
155 		if (!id) { return; }
156 	}
157 
158 	var rowEl = document.getElementById(id);
159 	if (rowEl) {
160 		this._activeContact = Dwt.getObjectFromElement(rowEl, "contact");
161 	}
162 
163 	ZmAutocompleteListView.prototype._setSelected.apply(this, arguments);
164 };

166 ZmPeopleAutocompleteListView.prototype._getFreeBusyInfo =
167 function(list) {
168 	var emailList = [];
169 	var emailHash = {};
170 	for (var i = 0; i < list.length; i++) {
171 		var match = list[i];
172 		emailList.push(match.email);
173 		emailHash[match.email] = match.id;
174 	}
175 
176 	var now = new Date();
177 	var jsonObj = {GetFreeBusyRequest:{_jsns:"urn:zimbraMail"}};
178 	var request = jsonObj.GetFreeBusyRequest;
179 	request.s = now.getTime();
180 	request.e = now.getTime() + (5*60*1000); // next 5 mins
181 	request.name = emailList.join(",");
182 
183 	return appCtxt.getAppController().sendRequest({
184 		jsonObj: jsonObj,
185 		asyncMode: true,
186 		callback: (new AjxCallback(this, this._handleFreeBusyResponse, [emailHash])),
187 		noBusyOverlay: true
188 	});
189 };
190 
191 ZmPeopleAutocompleteListView.prototype._handleFreeBusyResponse =
192 function(emailHash, result) {
193 	if (!this.getVisible()) { return; }
194 
195 	var fb = result.getResponse().GetFreeBusyResponse.usr;
196 	for (var i = 0; i < fb.length; i++) {
197 		var id = fb[i].id;
198 		var el = id && (document.getElementById(emailHash[id] + "-freebusy"));
199         var td = document.createElement("td");
200         td.innerHTML="- ";
201         td.className="ZmPeopleSearch-busy";
202         el.parentNode.insertBefore(td, el);
203         var text = new DwtText({parent:this, parentElement:el, index:1, id:"NewAppt", className:"FakeAnchor"});
204         text.isLinkText = true;
205         text.addListener(DwtEvent.ONMOUSEDOWN, new AjxListener(this, this._peopleItemListener));
206         text.addListener(DwtEvent.ONMOUSEOVER, new AjxListener(this, this.peopleItemMouseOverListener));
207         text.addListener(DwtEvent.ONMOUSEOUT, new AjxListener(this, this.peopleItemMouseOutListener));
208 
209 		if (el && fb[i].b) {
210             text.setText("Busy");
211 		}else if(el) {
212             text.setText("Available");
213         }
214 	}
215 };

217 ZmPeopleAutocompleteListView.prototype._removeAll =
218 function() {
219 	var table = this._getTable();
220 	for (var i = table.rows.length - 1; i >= 0; i--) {
221 		var row = table.rows[i];
222 		var contact = Dwt.getObjectFromElement(row, "contact");
223 		if (contact) {
224 			Dwt.disassociateElementFromObject(row, contact, "contact");
225 		}
226 	}
227 
228 	this._activeContact = null;
229 
230 	ZmAutocompleteListView.prototype._removeAll.apply(this, arguments);
231 };

238 ZmPeopleAutocompleteListView.prototype._peopleItemListener =
239  function(ev){
240     if (!this._activeContact) {
241         return;
242     }
243 
244     var target = DwtUiEvent.getTargetWithProp(ev, "id");
245     var action = "";
246     if (target && target.id)
247         action = target.id.split("_")[0]; //ids are inserted by DwtText, clean up as necessary
248 
249     switch (action){
250         case "NewMsg":
251             var params = {action:ZmOperation.NEW_MESSAGE, toOverride: new AjxEmailAddress(this._activeContact.getEmail(),
252             AjxEmailAddress.TO, this._activeContact.getFullName())};
253 	        AjxDispatcher.run("Compose", params);
254             break;
255 
256         case "NewAppt":
257             AjxDispatcher.require(["CalendarCore", "Calendar", "CalendarAppt"]);
258 			var cc = AjxDispatcher.run("GetCalController");
259 			var appt = cc.newApptObject((new Date()));
260 			appt.setAttendees([this._activeContact.getEmail()], ZmCalBaseItem.PERSON);
261 			cc.newAppointment(appt);
262 			break;
263 
264         case "NewContact":
265             AjxDispatcher.require(["ContactsCore", "Contacts"]);
266             var cc = AjxDispatcher.run("GetContactListController");
267             var list = new ZmContactList((new ZmSearch()), true);
268             list.add(this._activeContact);
269 	        cc.show(list, true);
270             break;
271     }
272 
273     this.show(false);
274 
275  };

296 ZmPeopleAutocompleteListView._outsideMouseDownListener =
297 function(ev) {
298 	var curList = ZmAutocompleteListView._activeAcList;
299 	if (curList) {
300 		var obj = DwtControl.getTargetControl(ev);
301 		if (obj && obj.parent && obj.parent == curList._toolbar) {
302 			return;
303 		}
304 	}
305 
306 	ZmAutocompleteListView._outsideMouseDownListener(ev);
307     ZmPeopleAutocompleteListView.prototype._listSelectionListener(ev);
308 
309 
310 };
311 
312 ZmPeopleAutocompleteListView.prototype._addMouseDownListener =
313 function() {
314 	DwtEventManager.addListener(DwtEvent.ONMOUSEDOWN, ZmPeopleAutocompleteListView._outsideMouseDownListener);
315 	this.shell._setEventHdlrs([DwtEvent.ONMOUSEDOWN]);
316 	this.shell.addListener(DwtEvent.ONMOUSEDOWN, this._outsideListener);
317 };
318 
319 ZmPeopleAutocompleteListView.prototype._removeMouseDownListener =
320 function() {
321 	DwtEventManager.removeListener(DwtEvent.ONMOUSEDOWN, ZmPeopleAutocompleteListView._outsideMouseDownListener);
322 	this.shell._setEventHdlrs([DwtEvent.ONMOUSEDOWN], true);
323 	this.shell.removeListener(DwtEvent.ONMOUSEDOWN, this._outsideListener);
324 };

326 /*
327     Display message to user that more results are available than fit in the current display
328     @param {int}    availHeight available height of display
329  */
330 ZmPeopleAutocompleteListView.prototype._showMoreResultsText =
331 function (availHeight) {
332       var rowNum = this._getNumberofAllowedRows(availHeight);
333       var textPos = rowNum > 1 ? rowNum-1 : 0;
334       var rowEl = this._getTable().rows[textPos];
335       var rowCell = Dwt.parseHtmlFragment(AjxTemplate.expand("share.Widgets#ZmPeopleAutocompleteListView-MoreResults"), true);
336       rowEl.parentNode.insertBefore(rowCell, rowEl);
337       //remove rows below text so they are not displayed
338       this._removeRows(rowNum);
339 };
340 
341 /* remove rows from bottom to index number */
342 ZmPeopleAutocompleteListView.prototype._removeRows =
343 function(idx) {
344 	this._matches = null;
345 	var table = this._getTable();
346 	for (var i = table.rows.length - 1; i >= 0 && i >= idx; i--) {
347 		var row = table.rows[i];
348 		if (row != this._waitingRow) {
349 			table.deleteRow(i);
350 		}
351 	}
352 };
353 
354 /*
355     Get the number of rows within the available height
356     @param {int}    availHeight available height for display
357     @return {int}   return the number of rows
358  */
359 ZmPeopleAutocompleteListView.prototype._getNumberofAllowedRows =
360 function(availHeight) {
361    var rowCount = 0;
362    var totalHeight = 0;
363    for(var i = 0; i< this._getTable().rows.length; i++){
364        var row = this._getTable().rows[i];
365        totalHeight += Dwt.getSize(row).y;
366        if (totalHeight < availHeight){
367            rowCount++;
368        } else {
369            break;
370        }
371    }
372 
373     return rowCount;
