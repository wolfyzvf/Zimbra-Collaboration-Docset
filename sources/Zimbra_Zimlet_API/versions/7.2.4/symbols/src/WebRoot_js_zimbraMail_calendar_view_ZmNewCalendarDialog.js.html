 16 ZmNewCalendarDialog = function(parent, className) {
 17 	var title = ZmMsg.createNewCalendar;
 18 	var type = ZmOrganizer.CALENDAR;
 19 	ZmNewOrganizerDialog.call(this, parent, className, title, type);
 20 };
 21 
 22 ZmNewCalendarDialog.prototype = new ZmNewOrganizerDialog;
 23 ZmNewCalendarDialog.prototype.constructor = ZmNewCalendarDialog;
 24 
 25 ZmNewCalendarDialog.prototype.toString = 
 26 function() {
 27 	return "ZmNewCalendarDialog";
 28 };
 29 
 30 // Public methods
 31 
 32 ZmNewCalendarDialog.prototype.reset =
 33 function(account) {
 34 	ZmNewOrganizerDialog.prototype.reset.apply(this, arguments);
 35 	this._excludeFbCheckbox.checked = false;
 36 };
 37 
 38 // Protected methods
 39 
 40 ZmNewCalendarDialog.prototype._getRemoteLabel =
 41 function() {
 42 	return ZmMsg.addRemoteAppts;
 43 };
 44 
 45 ZmNewCalendarDialog.prototype._createExtraContentHtml =
 46 function(html, idx) {
 47 	idx = this._createFreeBusyContentHtml(html, idx);
 48 	return ZmNewOrganizerDialog.prototype._createExtraContentHtml.call(this, html, idx);
 49 };
 50 
 51 ZmNewCalendarDialog.prototype._createFreeBusyContentHtml =
 52 function(html, idx) {
 53 	this._excludeFbCheckboxId = this._htmlElId + "_excludeFbCheckbox";
 54 	html[idx++] = AjxTemplate.expand("share.Dialogs#ZmNewCalDialogFreeBusy", {id:this._htmlElId});
 55 	return idx;
 56 };
 57 
 58 // NOTE: new calendar dialog doesn't show overview
 59 ZmNewCalendarDialog.prototype._createFolderContentHtml = 
 60 function(html, idx) {
 61 	return idx;
 62 };
 63 
 64 ZmNewCalendarDialog.prototype._setupExtraControls =
 65 function() {
 66 	ZmNewOrganizerDialog.prototype._setupExtraControls.call(this);
 67 	this._setupFreeBusyControl();
 68 };
 69 
 70 ZmNewCalendarDialog.prototype._setupFreeBusyControl =
 71 function() {
 72 	this._excludeFbCheckbox = document.getElementById(this._excludeFbCheckboxId);
 73 };
 74 
 75 /*
 76 *   Overwritten the parent class method to include application specific params.
 77 */
 78 ZmNewCalendarDialog.prototype._setupColorControl =
 79 function() {
 80     var el = document.getElementById(this._colorSelectId);
 81 	this._colorSelect = new ZmColorButton({parent:this,parentElement:el,hideNone:true});
 82 };
 83 
 84 /** 
 85  * Checks the input for validity and returns the following array of values:
 86  * <ul>
 87  * <li> parentFolder
 88  * <li> name
 89  * <li> color
 90  * <li> URL
 91  * <li> excludeFB
 92  * </ul>
 93  * 
 94  * @private
 95  */
 96 ZmNewCalendarDialog.prototype._getFolderData =
 97 function() {
 98 	var data = ZmNewOrganizerDialog.prototype._getFolderData.call(this);
 99 	if (data) {
100 		data.f = this._excludeFbCheckbox.checked ? "b#" : "#";
101 	}
102 	return data;
103 };

105 /**
106  * @Override Added for tabindexing checkboxes.
107  * 
108  * @private
109  */
110 //For bug 21985
111 ZmNewCalendarDialog.prototype._getTabGroupMembers =
112 function() {
113 	var list = ZmNewOrganizerDialog.prototype._getTabGroupMembers.call(this);
114     if (this._excludeFbCheckbox) {
115 		list.push(this._excludeFbCheckbox);
116 	}
117     if (this._remoteCheckboxField) {
118 		list.push(this._remoteCheckboxField);
119 	}
120     return list;
121 };
