 20 AjxCookie = function() {
 21 }
 22 
 23 AjxCookie.prototype.toString = 
 24 function() {
 25 	return "AjxCookie";
 26 }
 27 
 28 AjxCookie.getCookie = 
 29 function(doc, name) {
 30 	var arg = name + "=";
 31 	var alen = arg.length;
 32 	var clen = doc.cookie.length;
 33 	var cookie = doc.cookie;
 34 	var i = 0;
 35 	while (i < clen) {
 36 		var j = i + alen;
 37 		if (cookie.substring(i, j) == arg) {
 38 			var endstr = cookie.indexOf (";", j);
 39 			if (endstr == -1)
 40 				endstr = cookie.length;
 41 			return unescape(cookie.substring(j, endstr));
 42 		}
 43 		i = cookie.indexOf(" ", i) + 1;
 44 		if (i == 0) 
 45 			break; 
 46 	}
 47   return null;
 48 }
 49 
 50 AjxCookie.setCookie = 
 51 function(doc, name, value, expires, path, domain, secure) {
 52 	doc.cookie = name + "=" + escape (value) +
 53 		((expires) ? "; expires=" + expires.toGMTString() : "") +
 54 		((path) ? "; path=" + path : "") +
 55 		((domain) ? "; domain=" + domain : "") +
 56 		((secure) ? "; secure" : "");
 57 }
 58 
 59 AjxCookie.deleteCookie = 
 60 function (doc, name, path, domain) {
 61 	doc.cookie = name + "=" +
 62 	((path) ? "; path=" + path : "") +
 63 	((domain) ? "; domain=" + domain : "") + "; expires=Fri, 31 Dec 1999 23:59:59 GMT";
 64 }
 65 
 66 AjxCookie.areCookiesEnabled = 
 67 function (doc) {
 68 	var name = "ZM_COOKIE_TEST";
 69 	var value = "Zimbra";
 70 	AjxCookie.setCookie(doc, name, value);
 71 	var cookie = AjxCookie.getCookie(doc, name);
 72 	AjxCookie.deleteCookie(doc, name);
 73 	return cookie == value;
 74 }
