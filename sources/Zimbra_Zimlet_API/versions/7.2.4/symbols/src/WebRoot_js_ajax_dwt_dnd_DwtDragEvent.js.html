 16 /**
 17  * @constructor
 18  * @class
 19  * DwtDragEvent is generated by the Drag and Drop framework when a drag operation is
 20  * in process. The drag event is dispatched to the registered {@link DwtDragSource} instance.
 21  * 
 22  * @author Ross Dargahi
 23  * 
 24  * @see DwtDragSource
 25  */
 26 DwtDragEvent = function() {
 27 	/**
 28 	 * Type of drag operation. One of:
 29 	 * <ul>
 30 	 * <li>{@link DwtDragEvent.DRAG_START}</li>
 31 	 * <li>{@link DwtDragEvent.SET_DATA}</li>
 32 	 * <li>{@link DwtDragEvent.DRAG_END}</li>
 33 	 * </ul>
 34 	 */
 35 	this.operation = null;
 36 	
 37 	/**
 38 	 * Drag source control
 39 	 * @type DwtControl
 40 	 */
 41 	this.srcControl = null;
 42 	
 43 	/**
 44 	 * Action being performed. One of:
 45 	 * <ul>
 46 	 * <li>{@link Dwt.DND_DROP_NONE}</li>
 47 	 * <li>{@link Dwt.DND_DROP_COPY}</li>
 48 	 * <li>{@link Dwt.DND_DROP_MOVE}</li>
 49 	 * </ul>
 50 	 */
 51 	this.action = null;
 52 	
 53 	/**
 54 	 * Whether the DnD framework should perform the operation. The application is
 55 	 * responsible for setting this value based on whatever business logic it is
 56 	 * implementing
 57 	 * @type boolean
 58 	 */
 59 	this.doIt = false;
 60 	
 61 	/**
 62 	 * Drag source data. This is the application data associated with the item being dragged.
 63 	 */
 64 	this.srcData = null;
 65 };
 66 
 67 /**
 68  * Drag initialization.
 69  */
 70 DwtDragEvent.DRAG_INIT = "INIT";
 71 
 72 /**
 73  * Drag is starting.
 74  */
 75 DwtDragEvent.DRAG_START = "START";
 76 
 77 /**
 78  * Set the <code>srcData</code> field of the event.
 79  */
 80 DwtDragEvent.SET_DATA = "SET_DATA";
 81 
 82 /**
 83  * Drag movement has occurred.
 84  */
 85 DwtDragEvent.DRAG_MOVE = "MOVE";
 86 
 87 /**
 88  * Drag has ended.
 89  */
 90 DwtDragEvent.DRAG_END = "END";
 91 
 92 /**
 93  * Drag canceled (i.e. dropped on invalid target).
 94  */
 95 DwtDragEvent.DRAG_CANCEL = "CANCEL";
