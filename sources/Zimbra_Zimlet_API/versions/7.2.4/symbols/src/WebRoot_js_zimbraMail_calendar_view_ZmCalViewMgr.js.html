 16 /**
 17  * Creates the calendar view manager.
 18  * @class
 19  * This class represents the calendar view manager.
 20  * 
 21  * @param {DwtShell}	parent			the element that created this view
 22  * @param {ZmController}		controller		the controller
 23  * @param {DwtDropTarget}	dropTgt			the drop target
 24  * 
 25  * @extends		DwtComposite
 26  */
 27 ZmCalViewMgr = function(parent, controller, dropTgt) {
 28 
 29 	DwtComposite.call(this, {parent:parent, className:"ZmCalViewMgr", posStyle:Dwt.ABSOLUTE_STYLE});
 30 	this.addControlListener(new AjxListener(this, this._controlListener));
 31 
 32 	this._controller = controller;
 33 	this._dropTgt = dropTgt;
 34     this._showNewScheduleView = appCtxt.get(ZmSetting.FREE_BUSY_VIEW_ENABLED);
 35 	// View hash. Holds the various views e.g. day, month, week, etc...
 36 	this._views = {};
 37 	this._date = new Date();
 38 	this._viewFactory = {};
 39 	this._viewFactory[ZmId.VIEW_CAL_DAY]		= ZmCalDayView;
 40 	this._viewFactory[ZmId.VIEW_CAL_WORK_WEEK]	= ZmCalWorkWeekView;
 41 	this._viewFactory[ZmId.VIEW_CAL_WEEK]		= ZmCalWeekView;
 42 	this._viewFactory[ZmId.VIEW_CAL_MONTH]		= ZmCalMonthView;
 43 	this._viewFactory[ZmId.VIEW_CAL_LIST]		= ZmCalListView;
 44     this._viewFactory[ZmId.VIEW_CAL_FB]	        = ZmCalNewScheduleView;
 45     this._viewFactory[ZmId.VIEW_CAL_SCHEDULE]	= ZmCalScheduleView;
 46 
 47 	this._viewFactory[ZmId.VIEW_CAL_APPT]		= ZmApptView;
 48     this._viewFactory[ZmId.VIEW_CAL_TRASH]		= ZmApptListView;
 49 };

137 ZmCalViewMgr.prototype.createView =
138 function(viewName) {
139 	var view = new this._viewFactory[viewName](this, DwtControl.ABSOLUTE_STYLE, this._controller, this._dropTgt);
140 
141 	if (viewName != ZmId.VIEW_CAL_APPT && viewName != ZmId.VIEW_CAL_TRASH) {
142 		view.addTimeSelectionListener(new AjxListener(this, this._viewTimeSelectionListener));
143 		view.addDateRangeListener(new AjxListener(this, this._viewDateRangeListener));
144 		view.addViewActionListener(new AjxListener(this, this._viewActionListener));
145 	}
146 	this._views[viewName] = view;
147     if (viewName == ZmId.VIEW_CAL_TRASH) {
148         var controller = this._controller;
149         view.addSelectionListener(new AjxListener(controller, controller._listSelectionListener));
150         view.addActionListener(new AjxListener(controller, controller._listActionListener));
151     }
152 	return view;
153 };

159 ZmCalViewMgr.prototype.getSelTrashCount = function() {
160 
161     var folders  = this._controller.getCheckedCalendars(true);
162     this._multiAccTrashQuery = [];
163     for (var i=0; i< folders.length; i++) {
164         if (folders[i].nId == ZmOrganizer.ID_TRASH) {
165             this._multiAccTrashQuery.push(['inid:', '"', folders[i].getAccount().id, ':', ZmOrganizer.ID_TRASH, '"'].join(""));
166         }
167     }
168     return this._multiAccTrashQuery.length;
169 };
170 
171 ZmCalViewMgr.prototype.setSubContentVisible = function(visible) {
172 
173     if (appCtxt.multiAccounts) {
174         var selCount = this.getSelTrashCount();
175         // if no trash is checked
176         if (selCount < 1) {
177             this._subContentShown = false;
178             this._subContentInitialized = true;
179             this._controller.setCurrentListView(null);
180         } else {
181         // if more one or more trash is checked
182             this._subContentShown = true;
183         }
184     } else if (this._subContentShown != visible) {
185         this._subContentShown = visible;
186         if (!visible) {
187             this._controller.setCurrentListView(null);
188         }
189     }
190     this._layout();
191 };
192 
193 ZmCalViewMgr.prototype._createSubContent = function() {
194     if (!this._subContentShown) return null;
195     if (this._subContentInitialized) return this._list;
196 
197     this._subContentInitialized = true;
198 
199     this._sash = new DwtSash({parent:this,posStyle:Dwt.ABSOLUTE_STYLE,style:DwtSash.VERTICAL_STYLE});
200     this._sash.registerCallback(this._handleSashAdjustment, this);
201     this._list = this.createView(ZmId.VIEW_CAL_TRASH);
202     this._list.set(new AjxVector([]));
203 
204     this._populateTrashListView(this._list);
205     return this._list;
206 };
207 
208 ZmCalViewMgr.prototype._handleSashAdjustment = function(delta) {
209     // sash moved too far up
210     var sashLocation = this._sash.getLocation();
211     if (sashLocation.y + delta < ZmCalViewMgr.MIN_CONTENT_SIZE) {
212         delta = ZmCalViewMgr.MIN_CONTENT_SIZE - sashLocation.y;
213     }
214 
215     // sash moved to0 far down
216     else {
217         var size = this.getSize();
218         if (sashLocation.y + delta > size.y - ZmCalViewMgr.MIN_CONTENT_SIZE) {
219             delta = size.y - ZmCalViewMgr.MIN_CONTENT_SIZE - sashLocation.y;
220         }
221     }
222 
223     // adjust sub-content
224     if (delta != 0) {
225         var listSize = this._list.getSize();
226         this._list.setSize(listSize.x, listSize.y - delta);
227         this._layoutControls(true);
228     }
229 
230     return delta;
231 };

233 ZmCalViewMgr.prototype._populateTrashListView = function(listView) {
234     var params = {
235         searchFor:ZmItem.APPT,
236         limit:20,
237         types:AjxVector.fromArray([ZmId.ITEM_APPOINTMENT]),
238         forceSearch: true,
239 //        noRender: true,
240         callback: new AjxCallback(this, this._populateTrashListViewResults, [listView])
241     };
242 
243     if (appCtxt.multiAccounts) {
244         params.query = this._multiAccTrashQuery.join(" OR ");
245         params.account = appCtxt.accountList.mainAccount.name;
246     } else {
247         params.query = "inid:"+ZmOrganizer.ID_TRASH;
248     }
249     var search = new ZmSearch(params);
250     search.execute(params);
251 };
252 
253 ZmCalViewMgr.prototype._populateTrashListViewResults = function(listView, results) {
254     var data = results && results._data;
255     var apptList = data && data._results && data._results.APPT;
256     listView.set(apptList || new AjxVector([]));
257 };

289 ZmCalViewMgr.prototype.setView =
290 function(viewName) {
291 	if (viewName != this._currentViewName) {
292 		if (this._currentViewName) {
293 			this._views[this._currentViewName].setLocation(Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
294 		}
295 		var view = this._views[viewName];
296 		this._currentViewName = viewName;
297 
298 		if (viewName != ZmId.VIEW_CAL_APPT) {
299 			var vd = view.getDate();
300 			if (vd == null || (view.getDate().getTime() != this._date.getTime())) {
301 				view.setDate(this._date, this._duration, true);
302 			}
303 		}
304 		this._layout();
305 	}
306 };

308 ZmCalViewMgr.prototype._layout =
309 function() {
310     // create sub-content, if needed
311     var showSubContent = this._subContentShown;
312     if (showSubContent && !this._subContentInitialized) {
313         this._createSubContent();
314 
315         // NOTE: The list maintains its size so we can toggle back and forth
316         var size = this.getSize();
317         this._list.setSize(null, size.y / 3);
318     }
319     // Always re-populate trash list for multi-accounts
320     if (appCtxt.multiAccounts && this._subContentInitialized) {
321         this._populateTrashListView(this._list);
322     }
323 
324     // show sub-content
325     if (this._sash) {
326         this._sash.setVisible(showSubContent);
327         this._list.setVisible(showSubContent);
328     }
329 
330     // layout the controls
331     this._layoutControls();
332 };
333 
334 ZmCalViewMgr.prototype._layoutControls = function(skipSash) {
335     // size sub-content
336     var size = this.getSize();
337     var contentHeight = size.y;
338     if (this._subContentShown) {
339         var listSize = this._list.getSize();
340         var sashSize = this._sash.getSize();
341         var subContentHeight = listSize.y + sashSize.y;
342 
343         contentHeight -= subContentHeight;
344 
345         if (!skipSash) {
346             this._sash.setBounds(0, contentHeight, size.x, sashSize.y);
347         }
348         this._list.setBounds(0, contentHeight+sashSize.y, size.x, listSize.y);
349     }
350 
351     // size content
352     var view = this._views[this._currentViewName];
353     view.setBounds(0, 0, size.x, contentHeight);
354 
355     //need to reset layout for time view renderings
356     if (view instanceof ZmCalBaseView) view.layoutView();
357 };

359 ZmCalViewMgr.prototype._controlListener =
360 function(ev) {
361 	if (ev.oldHeight != ev.newHeight ||
362 		ev.oldWidth != ev.newWidth)
363 	{
364 		this._layout();
365 	}
366 };
