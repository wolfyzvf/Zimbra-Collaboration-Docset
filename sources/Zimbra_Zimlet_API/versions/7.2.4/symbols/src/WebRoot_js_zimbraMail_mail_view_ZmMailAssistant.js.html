 56 ZmMailAssistant.prototype.okHandler =
 57 function(dialog) {
 58 	var bad = new AjxVector();	
 59 	var good = new AjxVector();
 60 	var msg = this.getMessage(bad, good);
 61 
 62 	var cd = this._confirmDialog = appCtxt.getOkCancelMsgDialog();
 63 	cd.reset();
 64 	
 65 	var subject = AjxStringUtil.trim(msg.subject);
 66 	if ((subject == null || subject == "") && !this._noSubjectOkay) {
 67     	cd.setMessage(ZmMsg.compSubjectMissing, DwtMessageDialog.WARNING_STYLE);
 68 		cd.registerCallback(DwtDialog.OK_BUTTON, this._noSubjectOkCallback, this, dialog);
 69 		cd.registerCallback(DwtDialog.CANCEL_BUTTON, this._noSubjectCancelCallback, this, dialog);
 70 	    cd.popup();
 71 		return false;
 72 	}
 73 
 74 	if (good.size() == 0) {
 75 		dialog.messageDialog(ZmMsg.noAddresses, DwtMessageDialog.CRITICAL_STYLE);
 76 		return false;
 77 	}
 78 
 79 	if (bad.size() > 0 && !this._badAddrsOkay) {
 80 	    var badAddrs = AjxStringUtil.htmlEncode(bad.toString(AjxEmailAddress.SEPARATOR));
 81 	    var message = AjxMessageFormat.format(ZmMsg.compBadAddresses, badAddrs);
 82     	cd.setMessage(message, DwtMessageDialog.WARNING_STYLE);
 83 		cd.registerCallback(DwtDialog.OK_BUTTON, this._badAddrsOkCallback, this, dialog);
 84 		cd.registerCallback(DwtDialog.CANCEL_BUTTON, this._badAddrsCancelCallback, this, dialog);
 85 	    cd.popup();
 86 		return false;
 87 	} else {
 88 		this._badAddrsOkay = false;
 89 	}
 90 
 91 	var respCallback = new AjxCallback(this, this._handleResponseSendMsg, [dialog]);
 92 	var errorCallback = new AjxCallback(this, this._handleErrorSendMsg, [dialog]);
 93 	
 94 	msg.send(false, respCallback, errorCallback);
 95 	// need to popdown in handle response instead of returning true..
 96 	return false;
 97 };

 99 ZmMailAssistant.prototype._handleResponseSendMsg =
100 function(dialog) {
101 	if (!appCtxt.isOffline) { // see bug #29372
102 		appCtxt.setStatusMsg(ZmMsg.messageSent);
103 	}
104 	dialog.popdown();
105 };
106 
107 ZmMailAssistant.prototype._handleErrorSendMsg =
108 function(dialog, ex) {
109 	var msg = null;
110 	if (ex.code == ZmCsfeException.MAIL_SEND_ABORTED_ADDRESS_FAILURE) {
111 		var invalid = ex.getData(ZmCsfeException.MAIL_SEND_ADDRESS_FAILURE_INVALID);
112 		var invalidMsg = (invalid && invalid.length) ? AjxMessageFormat.format(ZmMsg.sendErrorInvalidAddresses,
113 														AjxStringUtil.htmlEncode(invalid.join(", "))) : null;
114 		msg = ZmMsg.sendErrorAbort + "<br/>" + invalidMsg;
115 	} else if (ex.code == ZmCsfeException.MAIL_SEND_PARTIAL_ADDRESS_FAILURE) {
116 		var invalid = ex.getData(ZmCsfeException.MAIL_SEND_ADDRESS_FAILURE_INVALID);
117 		msg = (invalid && invalid.length) ? AjxMessageFormat.format(ZmMsg.sendErrorPartial,
118 											AjxStringUtil.htmlEncode(invalid.join(", "))) : ZmMsg.sendErrorAbort;
119 	}
120 	if (msg) {
121 		dialog.messageDialog(msg, DwtMessageDialog.CRITICAL_STYLE);
122 		return true;
123 	} else {
124 		return false;
125 	}
126 };

158 ZmMailAssistant.prototype._getAddrs =
159 function(msg, type, value, bad, good) {
160 	if (value == null || value == "") return;
161 	var result = AjxEmailAddress.parseEmailString(value, type, false)
162 	if (result.bad.size() > 0 && bad != null) bad.addList(result.bad);
163 	if (result.all.size() > 0) 	{
164 		msg.setAddresses(type, result.all);
165 		if (good != null) good.addList(result.all);
166 	}
167 };

169 // bad is a vector that gets filled with bad addresses
170 ZmMailAssistant.prototype.getMessage =
171 function(bad, good) { 
172 	var msg = new ZmMailMsg();
173 	var body = new ZmMimePart();
174 	body.setContentType(ZmMimeTable.TEXT_PLAIN);
175 	var bodyText = this._mailFields.body != null ? this._mailFields.body : "";
176 	var identity = appCtxt.getIdentityCollection().defaultIdentity;
177 	if (Boolean(identity.signature)) {
178 		var sig = identity.signature;
179 		if (sig != null && sig != "") {
180 			bodyText = bodyText + "\n" + 
181 				((identity.signatureStyle == ZmSetting.SIG_INTERNET) ? "--\n" : "") +
182 				sig + "\n";
183 		}
184 	}
185 	body.setContent(bodyText);
186 	msg.setTopPart(body);
187 	if (this._mailFields.subject) msg.setSubject(this._mailFields.subject);
188 	this._getAddrs(msg, AjxEmailAddress.TO, this._mailFields.to, bad, good);
189 	this._getAddrs(msg, AjxEmailAddress.CC, this._mailFields.cc, bad, good);
190 	this._getAddrs(msg, AjxEmailAddress.BCC, this._mailFields.bcc, bad, good);		
191 	return msg;
192 }

194 ZmMailAssistant.__RE_handleSubject = new RegExp([ZmAssistant.SPACES,"\"([^\"]*)\"?",ZmAssistant.SPACES].join(""));
195 ZmMailAssistant.__RE_handleKeyValue = new RegExp([
196     "((",ZmAssistant.WORD,")(:",ZmAssistant.SPACES,")(.*?)",ZmAssistant.SPACES,")(",ZmAssistant.WORD,":|$)"
197 ].join(""),"m");

199 ZmMailAssistant.prototype.handle =
200 function(dialog, verb, args) {
201 	dialog._setOkButton(AjxMsg.ok, true, true); //, true, "NewMessage");
202 	var match;
203 	this._mailFields = {};	
204 
205 	if (!this._mailFields.subject) {
206 		match = args.match(ZmMailAssistant.__RE_handleSubject);
207 		if (match) {
208 			this._mailFields.subject = match[1];
209 			args = args.replace(match[0], " ");
210 		}
211 	}
212 
213 	while(match = args.match(ZmMailAssistant.__RE_handleKeyValue)) {
214 		var k = match[2];
215 		var v = match[4];
216 		var strip = match[1];
217 		var field = ZmMailAssistant._FIELDS[k];
218 		if (field) {
219 			if (field.key == 'body') {
220 				strip = match[2] + match[3];
221 				v = ZmAssistant.trimLeading(args.replace(strip,""));
222 				args = "";
223 			}
224 			if (v == null) v = "";
225 			this._mailFields[field.key] = v;
226 		}
227 		if (args) args = args.replace(strip,"");
228 	}
229 
230 	var index = -1, ri;
231 
232 	for (var i=0; i < ZmMailAssistant._FIELD_ORDER.length; i++) {	
233 		var key = ZmMailAssistant._FIELD_ORDER[i];
234 		var field = ZmMailAssistant._FIELDS[key];
235 		var value = this._mailFields[key]; 
236 		if (value != null) {
237 			value = field.multiLine ? AjxStringUtil.convertToHtml(value) : AjxStringUtil.htmlEncode(value);
238 		}
239 		if (field.defaultValue || value != null) {
240 			//var useDefault = (value == null || value == "");
241 			var useDefault = (value == null);
242 			ri = this._setField(field.field, useDefault ? field.defaultValue : value, useDefault, false, index+1);
243 		} else {
244 			ri = this._setOptField(field.field, value, false, false, index+1);
245 		}
246 		index = Math.max(index, ri);
247 	}
248 };

250 ZmMailAssistant.prototype._getConfirmDialog = 
251 function(dialog) {
252 	if (!this._confirmDialog) {
253 		this._confirmDialog = new DwtMessageDialog({parent:dialog.shell, buttons:[DwtDialog.OK_BUTTON, DwtDialog.CANCEL_BUTTON]});
254 	}
255 	return this._confirmDialog;
