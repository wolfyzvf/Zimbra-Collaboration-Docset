 22 /**
 23  * @class
 24  * A composite may contain other controls. All controls that need to contain child controls
 25  * (such as menus, trees) should inherit from this class.
 26  * 
 27  * @param {hash}	params		a hash of parameters
 28  * @param {DwtComposite}	params.parent	the parent widget
 29  * @param {string}	params.className		the CSS class
 30  * @param {constant}	params.posStyle		the positioning style
 31  * @param {boolean}	params.deferred		if <code>true</code>, postpone initialization until needed
 32  * @param {string}	params.id			an explicit ID to use for the control's HTML element
 33  * @param {number}	params.index 		the index at which to add this control among parent's children
 34  * 
 35  * @extends	DwtControl
 36  */
 37 DwtComposite = function(params) {
 38 	if (arguments.length == 0) { return; }
 39 	params = Dwt.getParams(arguments, DwtComposite.PARAMS);
 40 	
 41 	params.className = params.className || "DwtComposite";
 42 	DwtControl.call(this, params);
 43 
 44 	/**
 45 	 * Vector of child elements
 46 	 * @type AjxVector
 47 	 */
 48 	this._children = new AjxVector();
 49 }
 50 
 51 DwtComposite.PARAMS = DwtControl.PARAMS.concat();
 52 
 53 DwtComposite.prototype = new DwtControl;
 54 DwtComposite.prototype.constructor = DwtComposite;
 55 
 56 /**
 57  * Pending elements hash (i.e. elements that have not yet been realized).
 58  * @private
 59  */
 60 DwtComposite._pendingElements = new Object();
 61 
 62 DwtComposite.prototype.toString = 
 63 function() {
 64 	return "DwtComposite";
 65 }

 67 /**
 68  * Disposes of the control. This method will remove the control from under the
 69  * control of it's parent and release any resources associate with the component.
 70  * The method will also notify any event listeners on registered {@link DwtEvent.DISPOSE} event type.
 71  * 
 72  * <p>
 73  * In the case of {@link DwtComposite} this method will also dispose of all of the composite's
 74  * children.
 75  * 
 76  * <p> 
 77  * Subclasses may override this method to perform their own dispose functionality but
 78  * should generally call the parent <code>dispose()</code> method.
 79  * 
 80  * @see DwtControl#isDisposed
 81  * @see DwtControl#addDisposeListener
 82  * @see DwtControl#removeDisposeListener
 83  */
 84 DwtComposite.prototype.dispose =
 85 function() {
 86 	if (this._disposed) return;
 87 
 88 	var children = this._children.getArray();
 89 	while (children.length > 0)
 90         children.pop().dispose();
 91 
 92 	DwtControl.prototype.dispose.call(this);
 93 }
 94 
 95 /**
 96  * Gets the children of this composite.
 97  * 
 98  * @return	{array}		an array of {@link DwtControl} objects
 99  */
100 DwtComposite.prototype.getChildren =
101 function() {
102 	return this._children.getArray().slice(0);
103 }
104 
105 /**
106  * Gets the number of children of this composite.
107  * 
108  * @return {number} 		the number of composite children
109  */
110 DwtComposite.prototype.getNumChildren =
111 function() {
112 	return this._children.size();
113 }
114 
115 /**
116  * Removes all of the composite children.
117  * 
118  */
119 DwtComposite.prototype.removeChildren =
120 function() {
121 	var a = this._children.getArray();
122 	while (a.length > 0)
123 		a[0].dispose();
124 }

126 /**
127  * Clears the composite HTML element of content and removes
128  * all composite children by calling <code>removeChildren</code>.
129  * 
130  * @see #removeChildren
131  */
132 DwtComposite.prototype.clear =
133 function() {
134 	this.removeChildren();
135 	this.getHtmlElement().innerHTML = "";
136 }

138 /**
139 * Adds the given child control to this composite at the index (if specified).
140 *
141 * @param {DwtControl} child		the child control to add
142 * @param {number}	index		the index at which to add the child (may be <code>null</code>)
143 */
144 DwtComposite.prototype.addChild =
145 function(child, index) {
146 	this._children.add(child, index);
147 	
148 	// check for a previously removed element
149 	var childHtmlEl = child.getHtmlElement();
150 	childHtmlEl.setAttribute("parentId", this._htmlElId);
151 	if (this instanceof DwtShell && this.isVirtual()) {
152 		// If we are operating in "virtual shell" mode, then children of the shell's html elements
153 		// are actually parented to the body
154 		document.body.appendChild(childHtmlEl);
155 	} else {
156 		child.reparentHtmlElement(child.__parentElement || this.getHtmlElement(), index);
157 		child.__parentElement = null; // don't keep the reference to element, if any
158 	}
159 };
160 
161 /**
162 * Removes the specified child control from this control. A removed child is no longer retrievable via
163 * <code>getHtmlElement()</code>, so there is an option to save a reference to the removed child. 
164 * That way it can be added later using <code>addChild()</code>.
165 *
166 * @param {DwtConrol} child		the child control to remove
167 * @see #addChild
168 */
169 DwtComposite.prototype.removeChild =
170 function(child) {
171 	DBG.println(AjxDebug.DBG3, "DwtComposite.prototype.removeChild: " + child._htmlElId + " - " + child.toString());
172 	// Make sure that the child is initialized. Certain children (such as DwtTreeItems)
173 	// can be created in a deferred manner (i.e. they will only be initialized if they
174 	// are becoming visible.
175 	if (child.isInitialized()) {
176 		this._children.remove(child);
177 		// Sometimes children are nested in arbitrary HTML so we elect to remove them
178 		// in this fashion rather than use this.getHtmlElement().removeChild(child.getHtmlElement()
179 		var childHtmlEl = child.getHtmlElement();
180         if (childHtmlEl) {
181 			childHtmlEl.removeAttribute("parentId");
182 			if (childHtmlEl.parentNode) {
183 				var el = childHtmlEl.parentNode.removeChild(childHtmlEl);
184 			}
185 		}
186 	}
187 }

189 /**
190  * Allows the user to use the mouse to select text on the control.
191  * 
192  * @private
193  */
194 DwtComposite.prototype._setAllowSelection =
195 function() {
196 	if (!this._allowSelection) {
197 		this._allowSelection = true;
198 		this.addListener(DwtEvent.ONMOUSEDOWN, new AjxListener(this, this._mouseDownListener));
199 		this.addListener(DwtEvent.ONCONTEXTMENU, new AjxListener(this, this._contextMenuListener));
200 	}
201 };

203 /**
204  * Sets whether to prevent the browser from allowing text selection.
205  * 
206  * @see DwtControl#preventSelection
207  * @private
208  */
209 DwtComposite.prototype.preventSelection = 
210 function(targetEl) {
211 	return this._allowSelection ? false : DwtControl.prototype.preventSelection.call(this, targetEl);
212 };
213 
214 /**
215  * Determines whether to prevent the browser from displaying its context menu.
216  * 
217  * @see DwtControl#preventContextMenu
218  * @private
219  */
220 DwtComposite.prototype.preventContextMenu =
221 function(target) {
222 	if (!this._allowSelection) {
223 		return DwtControl.prototype.preventContextMenu.apply(this, arguments);
224 	}
225 	
226 	var bObjFound = target ? (target.id.indexOf("OBJ_") == 0) : false;
227 	var bSelection = false;
228 
229 	// determine if anything has been selected (IE and mozilla do it differently)
230 	if (document.selection) {			// IE
231 		bSelection = document.selection.type == "Text";
232 	} else if (getSelection()) {		// mozilla
233 		bSelection = getSelection().toString().length > 0;
234 	}
235 
236 	// if something has been selected and target is not a custom object,
237 	return (bSelection && !bObjFound) ? false : true;
238 };

240 /**
241  * Handles focus control when the mouse button is released
242  * 
243  * @see DwtControl#_focusByMouseUpEvent
244  * @private
245  */
246 DwtComposite.prototype._focusByMouseUpEvent =
247 function()  {
248 	if (!this._allowSelection) {
249 		DwtControl.prototype._focusByMouseUpEvent.apply(this, arguments);
250 	}
251 	// ...Else do nothing....
252 	// When text is being selected, we don't want the superclass
253 	// to give focus to the keyboard input control.
254 };

256 /**
257  * Event listener that is only registered when this control allows selection
258  * 
259  * @see _allowSelection
260  * @private
261  */
262 DwtComposite.prototype._mouseDownListener =
263 function(ev) {
264 	if (ev.button == DwtMouseEvent.LEFT) {
265 		// reset mouse event to propagate event to browser (allows text selection)
266 		ev._stopPropagation = true;
267 		ev._returnValue = true;
268 	}
269 };
270 
271 /**
272  * Event listener that is only registered when this control allows selection
273  * 
274  * @see _allowSelection
275  * @private
276  */
277 DwtComposite.prototype._contextMenuListener =
278 function(ev) {
279 	// reset mouse event to propagate event to browser (allows context menu)
280 	ev._stopPropagation = false;
281 	ev._returnValue = true;
282 };
