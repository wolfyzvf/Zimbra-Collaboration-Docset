 22 /**
 23  * Creates a new, empty list controller. This controller must be initialized before it can be used.
 24  * @class
 25  * This class is a base class for any controller that manages lists of items (for example, mail or
 26  * contacts). It consolidates handling of list functionality (for example, selection) and of common
 27  * operations such as tagging and deletion. Operations may be accessed by the user through
 28  * either the tool bar or an action menu. The public method {@link #show} gets everything going,
 29  * and then the controller just handles events.
 30  *
 31  * <p>Support is also present for handling multiple views (for example, contacts).</p>
 32  *
 33  * <p>Controllers for single items may extend this class, since the functionality needed is
 34  *  virtually the same. An item can be thought of as the degenerate form of a list.</p>
 35  *
 36  * @author Conrad Damon
 37  *
 38  * @param {DwtControl}		container	the containing shell
 39  * @param {ZmApp}		app		the containing application
 40  * 
 41  * @extends		ZmController
 42  */
 43 ZmListController = function(container, app) {
 44 
 45 	if (arguments.length == 0) { return; }
 46 	ZmController.call(this, container, app);
 47 
 48 	this._toolbar = {};			// ZmButtonToolbar (one per view)
 49 	this._navToolBar = {};		// ZmNavToolBar (one per view)
 50 	this._listView = {};		// ZmListView (one per view)
 51 	this._tabGroups = {};		// DwtTabGroup (one per view)
 52 	this._list = null;			// ZmList (the data)
 53 	this._actionMenu = null; 	// ZmActionMenu
 54 	this._actionEv = null;
 55 	this._activeSearch = null;
 56 
 57 	this._tagList = appCtxt.getTagTree();
 58 	if (this._tagList) {
 59 		this._tagChangeLstnr = new AjxListener(this, this._tagChangeListener);
 60 		this._tagList.addChangeListener(this._tagChangeLstnr);
 61 	}
 62 
 63 	// create a listener for each operation
 64 	this._listeners = {};
 65 	this._listeners[ZmOperation.NEW_MENU] = new AjxListener(this, this._newListener);
 66 	this._listeners[ZmOperation.TAG_MENU] = new AjxListener(this, this._tagButtonListener);
 67 	this._listeners[ZmOperation.TAG] = new AjxListener(this, this._tagListener);
 68 	this._listeners[ZmOperation.PRINT] = new AjxListener(this, this._printListener);
 69 	this._listeners[ZmOperation.DELETE]  = new AjxListener(this, this._deleteListener);
 70 	this._listeners[ZmOperation.CLOSE] = new AjxListener(this, this._backListener);
 71 	this._listeners[ZmOperation.MOVE]  = new AjxListener(this, this._moveListener);
 72 	this._listeners[ZmOperation.SEARCH] = new AjxListener(this, this._participantSearchListener);
 73 	this._listeners[ZmOperation.BROWSE] = new AjxListener(this, this._participantBrowseListener);
 74 	this._listeners[ZmOperation.NEW_MESSAGE] = new AjxListener(this, this._participantComposeListener);
 75 	if (window.ZmImApp) {
 76 		this._listeners[ZmOperation.IM] = ZmImApp.getImMenuItemListener();
 77 	}
 78 	this._listeners[ZmOperation.CONTACT] = new AjxListener(this, this._participantContactListener);
 79 	this._listeners[ZmOperation.VIEW] = new AjxListener(this, this._viewMenuItemListener);
 80 
 81 	this._menuPopdownListener = new AjxListener(this, this._menuPopdownActionListener);
 82 
 83 	this._dropTgt = new DwtDropTarget("ZmTag");
 84 	this._dropTgt.markAsMultiple();
 85 	this._dropTgt.addDropListener(new AjxListener(this, this._dropListener));
 86 
 87 	this._itemCountText = {};
 88 	this._continuation = {count:0, totalItems:0};
 89 };

223 /**
224  * Handles the key action.
225  * 
226  * @param	{constant}	actionCode		the action code
227  * @return	{Boolean}	<code>true</code> if the action is handled
228  */
229 ZmListController.prototype.handleKeyAction =
230 function(actionCode) {
231 	DBG.println(AjxDebug.DBG3, "ZmListController.handleKeyAction");
232 	var listView = this._listView[this._currentView];
233 
234 	switch (actionCode) {
235 
236 		case DwtKeyMap.DBLCLICK:
237 			return listView.handleKeyAction(actionCode);
238 
239 		case ZmKeyMap.DEL:
240 			var tb = this._toolbar[this._currentView];
241 			var button = tb && (tb.getButton(ZmOperation.DELETE) || tb.getButton(ZmOperation.DELETE_MENU));
242 			if (button && button.getEnabled()) {
243 				this._doDelete(this.getSelection());
244 			}
245 			break;
246 
247 		case ZmKeyMap.FLAG:
248 			this._doFlag(this.getSelection());
249 			break;
250 
251 		case ZmKeyMap.MOVE:
252 			this._moveListener();
253 			break;
254 
255 		case ZmKeyMap.NEXT_PAGE:
256 			var ntb = this._navToolBar[this._currentView];
257 			var button = ntb ? ntb.getButton(ZmOperation.PAGE_FORWARD) : null;
258 			if (button && button.getEnabled()) {
259 				this._paginate(this._currentView, true);
260 			}
261 			break;
262 
263 		case ZmKeyMap.PREV_PAGE:
264 			var ntb = this._navToolBar[this._currentView];
265 			var button = ntb ? ntb.getButton(ZmOperation.PAGE_BACK) : null;
266 			if (button && button.getEnabled()) {
267 				this._paginate(this._currentView, false);
268 			}
269 			break;
270 
271 		case ZmKeyMap.PRINT:
272 			if (appCtxt.get(ZmSetting.PRINT_ENABLED)) {
273 				this._printListener();
274 			}
275 			break;
276 
277 		case ZmKeyMap.TAG:
278 			var items = this.getSelection();
279 			if (items && items.length && (appCtxt.getTagTree().size() > 0)) {
280 				var dlg = appCtxt.getPickTagDialog();
281 				ZmController.showDialog(dlg, new AjxCallback(this, this._tagSelectionCallback, [items, dlg]));
282 			}
283 			break;
284 
285 		case ZmKeyMap.UNTAG:
286 			if (appCtxt.get(ZmSetting.TAGGING_ENABLED)) {
287 				var items = this.getSelection();
288 				if (items && items.length) {
289 					this._doRemoveAllTags(items);
290 				}
291 			}
292 			break;
293 
294 		default:
295 			return ZmController.prototype.handleKeyAction.call(this, actionCode);
296 	}
297 	return true;
298 };

392 /**
393  * Initializes the toolbar buttons and listeners.
394  * 
395  * @private
396  */
397 ZmListController.prototype._initializeToolBar =
398 function(view) {
399 	if (this._toolbar[view]) { return; }
400 
401 	var buttons = this._getToolBarOps();
402 	if (!buttons) { return; }
403 
404 	var tb = this._toolbar[view] = new ZmButtonToolBar({parent:this._container, buttons:buttons, context:view, controller:this,
405 														refElementId:ZmId.SKIN_APP_TOP_TOOLBAR});
406 
407 	var button;
408 	for (var i = 0; i < tb.opList.length; i++) {
409 		button = tb.opList[i];
410 		if (this._listeners[button]) {
411 			tb.addSelectionListener(button, this._listeners[button]);
412 		}
413 	}
414 
415 	button = tb.getButton(ZmOperation.PRINT);
416 	if (button) {
417 		button.setText(null);
418 	}
419 
420 	button = tb.getButton(ZmOperation.MOVE);
421 	if (button) {
422 		button.setText(null);
423 	}
424 
425 	button = tb.getButton(ZmOperation.NEW_MENU);
426 	if (button) {
427 		var listener = new AjxListener(tb, ZmListController._newDropDownListener);
428 		button.addDropDownSelectionListener(listener);
429 		tb._ZmListController_this = this;
430 		tb._ZmListController_newDropDownListener = listener;
431 	}
432 
433 	button = tb.getButton(ZmOperation.TAG_MENU);
434 	if (button) {
435 		button.noMenuBar = true;
436 		this._setupTagMenu(tb);
437 	}
438 
439 	appCtxt.notifyZimlets("initializeToolbar", [this._app, tb, this, view], {waitUntilLoaded:true});
440 };

503 /**
504  * Creates the desired application view.
505  *
506  * @param params		[hash]			hash of params:
507  *        view			[constant]		view ID
508  *        elements		[array]			array of view components
509  *        isAppView		[boolean]*		this view is a top-level app view
510  *        clear			[boolean]*		if true, clear the hidden stack of views
511  *        pushOnly		[boolean]*		don't reset the view's data, just swap the view in
512  *        isTransient	[boolean]*		this view doesn't go on the hidden stack
513  *        stageView		[boolean]*		stage the view rather than push it
514  *        tabParams		[hash]*			button params; view is opened in app tab instead of being stacked
515  *        
516  * @private
517  */
518 ZmListController.prototype._setView =
519 function(params) {
520 
521 	var view = params.view;
522 
523 	// create the view (if we haven't yet)
524 	if (!this._appViews[view]) {
525 		// view management callbacks
526 		var callbacks = {};
527 		callbacks[ZmAppViewMgr.CB_PRE_HIDE]		= this._preHideCallback ? new AjxCallback(this, this._preHideCallback) : null;
528 		callbacks[ZmAppViewMgr.CB_PRE_UNLOAD]	= this._preUnloadCallback ? new AjxCallback(this, this._preUnloadCallback) : null;
529 		callbacks[ZmAppViewMgr.CB_POST_HIDE]	= this._postHideCallback ? new AjxCallback(this, this._postHideCallback) : null;
530 		callbacks[ZmAppViewMgr.CB_PRE_SHOW]		= this._preShowCallback ? new AjxCallback(this, this._preShowCallback) : null;
531 		callbacks[ZmAppViewMgr.CB_POST_SHOW]	= this._postShowCallback ? new AjxCallback(this, this._postShowCallback) : null;
532 
533 		params.callbacks = callbacks;
534 		params.viewId = view;
535 		this._app.createView(params);
536 		this._appViews[view] = 1;
537 	}
538 
539 	// populate the view
540 	if (!params.pushOnly) {
541 		this._setViewContents(view);
542 	}
543 
544 	// push the view
545 	if (params.stageView) {
546 		this._app.stageView(view);
547 	} else {
548 		return (params.clear ? this._app.setView(view) : this._app.pushView(view));
549 	}
550 };

609 /**
610  * Create some new thing, via a dialog. If just the button has been pressed (rather than
611  * a menu item), the action taken depends on the app.
612  *
613  * @param {DwtUiEvent}	ev		the ui event
614  * @param {constant}	op		the operation ID
615  * @param {Boolean}		newWin	<code>true</code> if in a separate window
616  * 
617  * @private
618  */
619 ZmListController.prototype._newListener =
620 function(ev, op, params) {
621 	if (!ev && !op) { return; }
622 	op = op || ev.item.getData(ZmOperation.KEY_ID);
623 	if (!op || op == ZmOperation.NEW_MENU) {
624 		op = this._defaultNewId;
625 	}
626 
627 	var appName = ZmApp.OPS_R[op];
628 	var app = appName && appCtxt.getApp(appName);
629 	if (app) {
630 		params = params || {};
631 		params.ev = ev;
632 		if (!params.folderId && appCtxt.multiAccounts) {
633 			var tree = appCtxt.getFolderTree(appCtxt.getActiveAccount() || appCtxt.accountList.defaultAccount);
634 			var folderId = app.getDefaultFolderId && app.getDefaultFolderId(op);
635 			if (folderId) {
636 				params.folderId = tree.getById(folderId).id;
637 			}
638 		}
639 		app.handleOp(op, params);
640 	} else {
641 		ZmController.prototype._newListener.call(this, ev, op);
642 	}
643 };

660 /**
661  * Tag/untag items.
662  * 
663  * @private
664  */
665 ZmListController.prototype._tagListener =
666 function(ev, items) {
667 
668 	var curView = appCtxt.getAppViewMgr().getCurrentViewId();
669 	if (curView == this._getViewType() || curView == ZmId.VIEW_MIXED) {
670 		var tagEvent = ev.getData(ZmTagMenu.KEY_TAG_EVENT);

928 /**
929  * The list view as a whole is the drop target, since it's the lowest-level widget. Still, we
930  * need to find out which item got dropped onto, so we get that from the original UI event
931  * (a mouseup). The header is within the list view, but not an item, so it's not a valid drop
932  * target. One drawback of having the list view be the drop target is that we can't exercise
933  * fine-grained control on what's a valid drop target. If you enter via an item and then drag to
934  * the header, it will appear to be valid.
935  * 
936  * @protected
937  */
938 ZmListController.prototype._dropListener =
939 function(ev) {
940 	var view = this._listView[this._currentView];
941 	var div = view.getTargetItemDiv(ev.uiEvent);
942 	var item = view.getItemFromElement(div);
943 
944 	// only tags can be dropped on us
945 	var data = ev.srcData.data;
946 	if (ev.action == DwtDropEvent.DRAG_ENTER) {
947 		ev.doIt = (item && (item instanceof ZmItem) && !item.isShared() && this._dropTgt.isValidTarget(data));
948         // Bug: 44488 - Don't allow dropping tag of one account to other account's item
949         if (appCtxt.multiAccounts) {
950            var listAcctId = item ? item.getAccount().id : null;
951            var tagAcctId = (data.account && data.account.id) || data[0].account.id;
952            if (listAcctId != tagAcctId) {
953                ev.doIt = false;
954            }
955         }
956 		DBG.println(AjxDebug.DBG3, "DRAG_ENTER: doIt = " + ev.doIt);
957 		if (ev.doIt) {
958 			view.dragSelect(div);
959 		}
960 	} else if (ev.action == DwtDropEvent.DRAG_DROP) {
961 		view.dragDeselect(div);
962 		var items = [item];
963 		var sel = this.getSelection();
964 		if (sel.length) {
965 			var vec = AjxVector.fromArray(sel);
966 			if (vec.contains(item)) {
967 				items = sel;
968 			}
969 		}
970 		this._doTag(items, data, true);
971 	} else if (ev.action == DwtDropEvent.DRAG_LEAVE) {
972 		view.dragDeselect(div);
973 	} else if (ev.action == DwtDropEvent.DRAG_OP_CHANGED) {
974 		// nothing
975 	}
976 };

1122  *
1123  * @param {Array}	items		a list of items to move
1124  * @param {ZmFolder}	folder		the destination folder
1125  * @param {Object}	attrs		the additional attrs for SOAP command
1126  * @param {Boolean}		isShiftKey	<code>true</code> if forcing a copy action
1127  * @private
1128  */
1129 ZmListController.prototype._doMove =
1130 function(items, folder, attrs, isShiftKey) {
1131 
1132 	items = AjxUtil.toArray(items);
1133 	if (!items.length) { return; }
1134 
1135 	var move = [];
1136 	var copy = [];
1137 	if (items[0] instanceof ZmItem) {
1138 		for (var i = 0; i < items.length; i++) {
1139 			var item = items[i];
1140 			if (!item.folderId || (item.folderId != folder.id || (attrs && attrs.op == "recover"))) {
1141 				if (!this._isItemMovable(item, isShiftKey, folder)) {
1142 					copy.push(item);
1143 				} else {
1144 					move.push(item);
1145 				}
1146 			}
1147 		}
1148 	} else {
1149 		move = items;
1150 	}
1151 
1152 	var params = {folder:folder, attrs:attrs};
1153     params.errorCallback = new AjxCallback(this, this._actionErrorCallback);
1154 
1155 	var allDoneCallback = new AjxCallback(this, this._checkItemCount);
1156 	if (move.length) {
1157 		params.items = move;
1158 		var list = this._setupContinuation(this._doMove, [folder, attrs, isShiftKey], params, allDoneCallback);
1159 
1160 		if (folder.isInTrash()) {
1161 			var anyScheduled = false;
1162 			var mItems = AjxUtil.toArray(move);
1163 			for (var i=0, cnt=mItems.length; i<cnt; i++) {
1164 				if (mItems[i] && mItems[i].isScheduled) {
1165 					anyScheduled = true;
1166 					break;
1167 				}
1168 			}

1172 			} else {
1173 				list.moveItems(params);
1174 			}
1175 		} else {
1176 			list.moveItems(params);
1177 		}
1178 	}
1179 
1180 	if (copy.length) {
1181 		params.items = copy;
1182 		var list = this._setupContinuation(this._doMove, [folder, attrs, isShiftKey], params, allDoneCallback);

1465 /**
1466  * Gets next or previous page of items. The set of items may come from the
1467  * cached list, or from the server (using the current search as a base).
1468  * <p>
1469  * The loadIndex is the index'd item w/in the list that needs to be loaded -
1470  * initiated only when user is in CV and pages a conversation that has not
1471  * been loaded yet.</p>
1472  * <p>
1473  * Note that this method returns a value even though it may make an
1474  * asynchronous SOAP request. That's possible as long as no caller
1475  * depends on the results of that request. Currently, the only caller that
1476  * looks at the return value acts on it only if no request was made.</p>
1477  *
1478  * @param {constant}	view		the current view
1479  * @param {Boolean}	forward		if <code>true</code>, get next page rather than previous
1480  * @param {int}		loadIndex	the index of item to show
1481  * @param {int}	limit		the number of items to fetch
1482  * 
1483  * @private
1484  */
1485 ZmListController.prototype._paginate =
1486 function(view, forward, loadIndex, limit) {
1487 
1488 	var needMore = false;
1489 	var lv = this._listView[view];
1490 	var offset, max;
1491 
1492     limit = limit || lv.getLimit(offset);
1493 
1494 	if (lv._isPageless) {
1495 		offset = this._list.size();
1496 		needMore = true;
1497 	} else {
1498 		offset = lv.getNewOffset(forward);
1499 		needMore = (offset + limit > this._list.size());
1500 		this.currentPage = this.currentPage + (forward ? 1 : -1);
1501 		this.maxPage = Math.max(this.maxPage, this.currentPage);
1502 	}
1503 
1504 	// see if we're out of items and the server has more
1505 	if (needMore && this._list.hasMore()) {
1506 		lv.offset = offset; // cache new offset
1507 		if (lv._isPageless) {
1508 			max = limit;
1509 		} else {
1510 			// figure out how many items we need to fetch
1511 			var delta = (offset + limit) - this._list.size();
1512 			max = delta < limit && delta > 0 ? delta : limit;
1513 			if (max < limit) {
1514 				offset = ((offset + limit) - max) + 1;
1515 			}
1516 		}
1517 
1518 		// handle race condition - user has paged quickly and we don't want
1519 		// to do second fetch while one is pending
1520 		if (this._searchPending) { return false;	}
1521 
1522 		// figure out if this requires cursor-based paging
1523 		var list = lv.getList();
1524 		var lastItem = list && list.getLast();
1525 		var lastSortVal = (lastItem && lastItem.id) ? lastItem.sf : null;
1526 		var lastId = lastSortVal ? lastItem.id : null;
1527 
1528 		this._setItemCountText(ZmMsg.loading);
1529 
1530 		// get next page of items from server; note that callback may be overridden
1531 		this._searchPending = true;
1532 		var respCallback = new AjxCallback(this, this._handleResponsePaginate, [view, false, loadIndex, offset]);
1533 		this._search(view, offset, max, respCallback, true, lastId, lastSortVal);
1534 		return false;
1535 	} else if (!lv._isPageless) {
1536 		lv.offset = offset; // cache new offset
1537 		this._resetOperations(this._toolbar[view], 0);
1538 		this._resetNavToolBarButtons(view);
1539 		this._setViewContents(view);
1540 		this._resetSelection();
1541 		return true;
1542 	}
1543 };

1545 /**
1546  * Updates the list and the view after a new page of items has been retrieved.
1547  *
1548  * @param {constant}	view				the current view
1549  * @param {Boolean}	saveSelection			if <code>true</code>, maintain current selection
1550  * @param {int}	loadIndex				the index of item to show
1551  * @param {ZmCsfeResult}	result			the result of SOAP request
1552  * @param {Boolean}	ignoreResetSelection	if <code>true</code>, do not reset selection
1553  * 
1554  * @private
1555  */
1556 ZmListController.prototype._handleResponsePaginate =
1557 function(view, saveSelection, loadIndex, offset, result, ignoreResetSelection) {
1558 
1559 	var searchResult = result.getResponse();
1560 
1561 	// update "more" flag
1562 	this._list.setHasMore(searchResult.getAttribute("more"));
1563 
1564 	this._cacheList(searchResult, offset);
1565 
1566 	var lv = this._listView[this._currentView];
1567 	var num = lv._isPageless ? this.getSelectionCount() : 0;
1568 	this._resetOperations(this._toolbar[view], num);
1569 
1570 	// remember selected index if told to
1571 	var selItem = saveSelection ? this.getSelection()[0] : null;
1572 	var selectedIdx = selItem ? lv.getItemIndex(selItem) : -1;
1573 
1574 	var items = searchResult && searchResult.getResults().getArray();
1575 	if (lv._isPageless && items && items.length) {
1576 		lv._itemsToAdd = items;
1577 	} else {
1578 		lv._itemsToAdd = null;
1579 	}
1580 	var wasEmpty = (lv._isPageless && (lv.size() == 0));
1581 
1582 	this._setViewContents(view);
1583 
1584 	// add new items to selection if all results selected, in a way that doesn't call deselectAll()
1585 	if (lv.allSelected) {
1586 		for (var i = 0, len = items.length; i < len; i++) {
1587 			lv.selectItem(items[i], true);
1588 			lv.setSelectionCbox(items[i], false);
1589 		}
1590 		lv.setSelectionHdrCbox(true);
1591 		DBG.println("scr", "pagination - selected more items: " + items.length);
1592 		DBG.println("scr", "items selected: " + this.getSelectionCount());
1593 	}
1594 	this._resetNavToolBarButtons(view);
1595 
1596 	// bug fix #5134 - some views may not want to reset the current selection
1597 	if (!ignoreResetSelection && !lv._isPageless) {
1598 		this._resetSelection(selectedIdx);
1599 	} else if (wasEmpty) {
1600 		lv._setNextSelection();
1601 	}
1602 
1603 	if (!appCtxt.isChildWindow) {
1604 		appCtxt.getAppController().focusContentPane();
1605 	}
1606 	this._searchPending = false;
1607 };

1966 										  {actionCallback:actionCallback, allDoneCallback:allDoneCallback});
1967 	params.count = this._continuation.count;
1968 	params.idsOnly = true;
1969 
1970 	var items = params.items || [];
1971 	var item = items[0];
1972 	var list = (item && item.list) || this._list;
1973 	if (list && (list.type == ZmItem.MIXED) && item._mixedType) {
1974 		list.type = item._mixedType;
1975 	}
1976 	if (!this._continuation.lastItem) {
1977 		this._continuation.lastItem = list.getVector().getLast();
1978 		this._continuation.totalItems = list.size();
1979 	}
1980 
1981 	return list;
1982 };

1984 /**
1985  * See if we are performing an action on all items, including ones that match the current search
1986  * but have not yet been retrieved. If so, keep doing searches and performing the action on the
1987  * results, until there are no more results.
1988  *
1989  * The arguments in the action callback should be those after the initial 'items' argument. The
1990  * array of items retrieved by the search is prepended to the callback's argument list before it
1991  * is run.
1992  *
1993  * @param {Hash}	params			a hash of parameters
1994  * @param {AjxCallback}	actionCallback		the callback with action to be performed on search results
1995  * @param {AjxCallback} allDoneCallback		the callback to run when we're all done
1996  * @param {Hash}	actionParams		the params from <code>ZmList._itemAction</code>, added when this is called
1997  * 
1998  * @private
1999  */
2000 ZmListController.prototype._continueAction =
2001 function(params, actionParams) {
2002 
2003 	var lv = this._listView[this._currentView];
2004 	var cancelled = actionParams && actionParams.cancelled;
2005 	var contResult = this._continuation.result;
2006 	var hasMore = contResult ? contResult.getAttribute("more") : this._list.hasMore();
2007 	DBG.println("sa", "lv.allSelected: " + lv.allSelected + ", hasMore: " + hasMore);
2008 	if (lv.allSelected && hasMore && !cancelled) {
2009 		var cs = this._currentSearch;
2010 		var limit = ZmListController.CONTINUATION_SEARCH_ITEMS;
2011 		var searchParams = {
2012 			query:		this.getSearchString(),
2013 			queryHint:	this.getSearchStringHint(),
2014 			types:		cs.types,
2015 			sortBy:		cs.sortBy,
2016 			limit:		limit,
2017 			idsOnly:	true
2018 		};
2019 
2020 		var list = contResult ? contResult.getResults() : this._list.getArray();
2021 		var lastItem = this._continuation.lastItem;
2022 		if (!lastItem) {
2023 			lastItem = list && list[list.length - 1];
2024 		}
2025 		if (lastItem) {
2026 			searchParams.lastId = lastItem.id;
2027 			searchParams.lastSortVal = lastItem.sf;
2028 			DBG.println("sa", "***** continuation search: " + searchParams.query + " --- " + [lastItem.id, lastItem.sf].join("/"));
2029 		} else {
2030 			searchParams.offset = limit + (this._continuation.search ? this._continuation.search.offset : 0);
2031 		}
2032 
2033 		this._continuation.count = actionParams.numItems;
2034 		if (!this._continuation.totalItems) {
2035 			this._continuation.totalItems = list.length;
2036 		}
2037 
2038 		this._continuation.search = new ZmSearch(searchParams);
2039 		var respCallback = new AjxCallback(this, this._handleResponseContinueAction, [params.actionCallback]);
2040 		appCtxt.getSearchController().redoSearch(this._continuation.search, true, null, respCallback);
2041 	} else {
2042 		DBG.println("sa", "end of continuation");
2043 		if (contResult) {
2044 			if (lv.allSelected) {
2045 				// items beyond page were acted on, give user a total count
2046 				if (actionParams.actionText) {
2047 					actionParams.actionSummary = ZmList.getActionSummary(actionParams.actionText, this._continuation.totalItems, contResult.type, actionParams.actionArg);
2048 				}
2049 				lv.deselectAll();
2050 			}
2051 			this._continuation = {count:0, totalItems:0};
2052 		}
2053 		if (params.allDoneCallback) {
2054 			params.allDoneCallback.run();
2055 		}
2056 
2057 		ZmList.killProgressDialog(actionParams.actionSummary, actionParams.actionLogItem, actionParams.closeChildWin);
2058 	}
2059 };

2061 /**
2062  * @private
2063  */
2064 ZmListController.prototype._handleResponseContinueAction =
2065 function(actionCallback, result) {
2066 
2067 	this._continuation.result = result.getResponse();
2068 	var items = this._continuation.result.getResults();
2069 	DBG.println("sa", "continuation search results: " + items.length);
2070 	if (items.length) {
2071 		this._continuation.lastItem = items[items.length - 1];
2072 		this._continuation.totalItems += items.length;
2073 		DBG.println("sa", "continuation last item: " + this._continuation.lastItem.id);
2074 		actionCallback.args = actionCallback.args || [];
2075 		actionCallback.args.unshift(items);
2076 		DBG.println("sa", "calling continuation action on search results");
2077 		actionCallback.run();
2078 	} else {
2079 		DBG.println(AjxDebug.DBG1, "Continuation with empty search results!");
2080 	}
2081 };
