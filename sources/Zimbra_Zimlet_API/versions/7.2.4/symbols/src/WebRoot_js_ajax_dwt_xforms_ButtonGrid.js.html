 17 /**
 18  * @constructor
 19  * @class 
 20  * This class is an example of putting a custom widget in a {@link XForm}.
 21  * 
 22  * @param	{array}	attributes		the attributes
 23  * @private
 24  */
 25 ButtonGrid = function(attributes) {
 26 	XFG.assignUniqueId(this, "__BUTTON_GRID__");
 27 
 28 	// copy any props passed in into the object
 29 	for (var prop in attributes) {
 30 		this[prop] = attributes[prop];
 31 	}
 32 	// handle any props that need special care
 33 	if (this.onChange) this.setOnChange(this.onChange);
 34 
 35 	// initialize to an empty array for value
 36 	this.value = [];
 37 	
 38 	
 39 }
 40 var BGP = ButtonGrid.prototype;
 41 BGP.choices = null;
 42 BGP.numCols = 4;
 43 BGP.cssClass = "xform_button_grid_medium";
 44 BGP.onChange = null;
 45 BGP.multiple = true;
 46 BGP.addBracketingCells = false;
 47 
 48 BGP.setOnChange = function (onChange) {
 49 	if (onChange && typeof onChange == "string") {
 50 		onChange = new Function("value, event", onChange);	
 51 	}
 52 	this.onChange = onChange;
 53 }
 54 
 55 
 56 BGP.getValue = function() {
 57 	return this.value;
 58 }
 59 
 60 BGP.setValue = function(value) {
 61 	if (value == null) value = [];
 62 	if (typeof value == "string") value = value.split(",");
 63 	this.value = value;
 64 	
 65 	this.showValue();
 66 }
 67 
 68 BGP.showValue = function() {
 69 	var value = this.value;
 70 	if (value == null) value = "";
 71 
 72 	// assumes value is a comma-delimited string or an array
 73 	if (typeof value == "string") value = value.split(",");
 74 	
 75 	// hack up value to make searching for a particular option value easier
 76 	var uniqueStartStr = "{|[", uniqueEndStr = "]|}";
 77 	value = uniqueStartStr + value.join(uniqueEndStr + uniqueStartStr) + uniqueEndStr;
 78 	var choices = this.choices;
 79 	for (var i = 0; i < choices.length; i++) {
 80 		var element = XFG.getEl(this.getButtonId(i));
 81 		var isPresent = (value.indexOf(uniqueStartStr + choices[i].value + uniqueEndStr) > -1);
 82 		if (isPresent) {
 83 			XFG.showSelected(element);
 84 		} else {
 85 			XFG.hideSelected(element);
 86 		}
 87 	}	
 88 }
 89 
 90 BGP.toggleValue = function(value, element) {
 91 	if (this.multiple) {
 92 		if (this.valueIsSelected(value)) {
 93 			this.deselectValue(value, element);
 94 		} else {
 95 			this.selectValue(value, element);
 96 		}
 97 	} else {
 98 		this.value = value;
 99 	}
100 	return this.value;
101 }
102 
103 BGP.valueIsSelected = function(value) {
104 	for (var i = 0; i < this.value.length; i++) {
105 		if (this.value[i] == value) return true;
106 	}
107 }
108 
109 BGP.selectValue = function(value, element) {
110 	if (!this.valueIsSelected(value)) {
111 		this.value.push(value);
112 	}
113 	if (element) XFG.showSelected(element);
114 }
115 
116 BGP.deselectValue = function(value, element) {
117 	if (this.valueIsSelected(value)) {
118 		for (var i = 0; i < this.value.length; i++) {
119 			if (this.value[i] == value) {
120 				this.value = this.value.slice(0, i).concat(this.value.slice(i+1, this.value.length));
121 			}
122 		}
123 	}
124 	if (element) XFG.hideSelected(element);
125 }
126 
127 BGP.getButtonId = function (btnNum) {
128 	return this.__id + "_button_" + btnNum;
129 }
130 
131 
132 BGP.onButtonClick = function(choiceValue, element, event) {
133 	var newValue = this.toggleValue(choiceValue, element);
134 	if (this.onChange) {
135 		this.onChange(newValue, event);
136 	}
137 }

139 BGP.getHTML = function () {
140 	if (this.choices == null) return (this.__HTMLOutput = null);
141 
142 	var buffer = new AjxBuffer();
143 	
144 	// write HTML for this element
145 	var buttonCssClass = this.cssClass + "_button";
146 	buffer.append("<table class=\"", this.cssClass, "_table\">");
147 	var i = 0;
148 	var numRows = Math.ceil(this.choices.length / this.numCols);
149 	for (var r = 0; r < numRows; r++) {
150 		buffer.append("<tr>\r");
151 		if (this.addBracketingCells) {
152 			buffer.append("\t<td width=50%><div class=", this.cssClass + "_start></div></td>");
153 		}
154 		for (var c = 0; c < this.numCols; c++) {
155 			var choice = this.choices[i];
156 			if (typeof choice == "string") {
157 				choice = this.choices[i] = {value:choice, label:choice};
158 			}
159 			buffer.append("<td class=", this.cssClass + "_td ><div id=", this.getButtonId(i), " class=", buttonCssClass, //(this.valueIsSelected(choice.value) ? "_selected" : ""),
160 								" onclick=\"XFG.cacheGet('", this.__id, "').onButtonClick('",choice.value,"',this,event);\">", 
161 								choice.label,
162 						"</div></td>");
163 			i++;
164 			if (i >= this.choices.length) break;
165 		}
166 		if (this.addBracketingCells) {
167 			buffer.append("<td width=50%><div class=", this.cssClass + "_end></div></td>");
168 		}
169 		buffer.append("</tr>");
170 	}
171 	buffer.append("</table>");
172 	this.__HTMLOutput = buffer.toString();
173 	return this.__HTMLOutput;
174 }
175 
176 
177 BGP.insertIntoXForm = function (form, item, element) {
178 	element.innerHTML = this.getHTML();
179 }
180 
181 BGP.updateChoicesHTML = function(labels) {
182 	var i = 0;
183 	for (var r = 0; r < numRows; r++) {
184 		for (var c = 0; c < this.numCols; c++) {
185 			var btn = document.getElementById(this.getButtonId(i));
186 			if(btn) {
187 				btn.innerHTML =labels[i].label;
188 			}
189 			i++;
190 			if (i >= labels.length) break;
191 		}
192 	}
193 }
194 
195 BGP.updateInXForm = function (form, item, value, element) {
196 	var valueStr = (value instanceof Array ? value.join(",") : value);
197 	if (!form.forceUpdate && this.__lastDisplayValue == valueStr) return;
198 	
199 	this.setValue(value);
200 	this.__lastDisplayValue = valueStr;
201 }
