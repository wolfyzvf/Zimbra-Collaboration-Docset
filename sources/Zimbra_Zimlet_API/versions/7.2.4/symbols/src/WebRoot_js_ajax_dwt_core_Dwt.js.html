284 /**
285  * @deprecated
286  * The association between an element and a control is now via DwtControl.ALL_BY_ID,
287  * where the unique element ID is a key to the control. The association is made when
288  * the control is initialized.
289  * 
290  * This method builds an indirect association between a DOM object and a JavaScript
291  * object. This indirection is important to prevent memory leaks (particularly in IE) by
292  * not directly creating a circular reference between a DOM object
293  *
294  * @param {DOMElement} domElement the DOM element (typically an HTML element)
295  * @param {Object} jsObject the JavaScript object
296  * 
297  * @private
298  */
299 Dwt.associateElementWithObject =
300 function(domElement, jsObject, attrName) {
301 	domElement[attrName||"dwtObj"] = jsObject.__internalId = AjxCore.assignId(jsObject);
302 };
303 
304 /**
305  * @deprecated
306  * The association will be broken by the control when it is disposed.
307  * 
308  * This method breaks the indirect association between a DOM object and a JavaScript
309  * object that was created by the <code>Dwt.associateElementWithObject</code>method
310  *
311  * @param {DOMElement} domElement the DOM element (typically an HTML element)
312  * @param {Object} jsObject the JavaScript object
313  * 
314  * @private
315  */
316 Dwt.disassociateElementFromObject =
317 function(domElement, jsObject, attrName) {
318 	if (domElement){
319 		domElement.removeAttribute(attrName||"dwtObj");
320 	}
321 	if (jsObject.__internalId){
322 		AjxCore.unassignId(jsObject.__internalId);
323 	}
324 };

336 /**
337  * Finds an ancestor element with a value for the given attr.
338  * 
339  * @param {DOMElement} domElement the DOM element (typically an HTML element)
340  * @param {string}	attrName	the attribute name
341  * @param {DOMElement} the DOM element
342  */
343 Dwt.findAncestor =
344 function(domElement, attrName) {
345 	var attr = Dwt.getAttr(domElement, attrName);
346 	while (domElement && (attr == null || attr == "")) {
347 		domElement = domElement.parentNode;
348 		attr = Dwt.getAttr(domElement, attrName);
349 	}
350 	return domElement;
351 };

353 /**
354  * Returns true if el1 is an ancestor (in the parent chain) of el2, or if
355  * el1 and el2 are the same element.
356  *
357  * @param {DOMElement}	el1
358  * @param {DOMElement}	el2
359  */
360 Dwt.isAncestor =
361 function(el1, el2) {
362 
363 	if (el1 == el2) {
364 		return true;
365 	}
366 
367 	var el = el2;
368 	while (el) {
369 		el = el.parentNode;
370 		if (el == el1) {
371 			return true;
372 		}
373 	}
374 	return false;
375 };

377 Dwt.setHandler =
378 function(htmlElement, event, func) {
379 	if (event == DwtEvent.ONMOUSEWHEEL && AjxEnv.isGeckoBased) {
380 		Dwt.clearHandler(htmlElement, event);
381 	}
382 	htmlElement[event] = func;
383 	if (event == DwtEvent.ONMOUSEWHEEL && AjxEnv.isGeckoBased) {
384 		htmlElement.addEventListener("DOMMouseScroll", func, true);
385 	}
386 };
387 
388 Dwt.clearHandler =
389 function(htmlElement, event) {
390 	if (event == DwtEvent.ONMOUSEWHEEL && AjxEnv.isGeckoBased) {
391 		if (htmlElement[event]) {
392 			var func = htmlElement[event];
393 			htmlElement.removeEventListener("DOMMouseScroll", func, true);
394 		}
395 	}
396 	htmlElement[event] = null;
397 };

409 /**
410  * Gets the bounds of an HTML element.
411  *
412  * @param {HTMLElement} htmlElement		the HTML element
413  *
414  * @return {DwtRectangle}	the elements bounds
415  *
416  * @see #setBounds
417  * @see #getLocation
418  * @see #getSize
419  */
420 Dwt.getBounds =
421 function(htmlElement, rect) {
422 	var tmpPt = DwtPoint.tmp;
423 
424 	Dwt.getLocation(htmlElement, tmpPt);
425 	var locX = tmpPt.x;
426 	var locY = tmpPt.y;
427 
428 	Dwt.getSize(htmlElement, tmpPt);
429 
430 	if (!rect) {
431 		return new DwtRectangle(locX, locY, tmpPt.x, tmpPt.y);
432 	} else {
433 		rect.set(locX, locY, tmpPt.x, tmpPt.y);
434 		return rect;
435 	}
436 };

500 function(htmlElement, point) {
501 	point = point || new DwtPoint(0, 0);

503 	if (htmlElement.style.position == Dwt.ABSOLUTE_STYLE) {
504 		point.set(parseInt(DwtCssStyle.getProperty(htmlElement, "left")),

512 /**
513  * Sets the location of an HTML element. The position type of the element must
514  * be absolute or else an exception is thrown. To only set one of the coordinates,
515  * pass in a value of {@link Dwt.DEFAULT} for the coordinate for which the value is
516  * not to be set
517  *
518  * @param {HTMLElement} htmlElement the absolutely positioned HTML element
519  * @param {number|string} x the x coordinate of the element (for example: 10, "10px", {@link Dwt.DEFAULT})
520  * @param {number|string} y the y coordinate of the element (for example: 10, "10px", {@link Dwt.DEFAULT})
521  *
522  * @throws DwtException
523  *
524  * @see #getLocation
525  * @see #setBounds
526  * @see #setSize
527  */
528 Dwt.setLocation =
529 function(htmlElement, x, y) {
530 	var position = htmlElement.style.position;
531 	if (position != Dwt.ABSOLUTE_STYLE && position != Dwt.RELATIVE_STYLE && position != Dwt.FIXED_STYLE) {
532 		DBG.println(AjxDebug.DBG1, "Cannot position static widget " + htmlElement.className);
533 		throw new DwtException("Static widgets may not be positioned", DwtException.INVALID_OP, "Dwt.setLocation");
534 	}
535 	if (x = Dwt.__checkPxVal(x)) {
536 		htmlElement.style.left = x;
537 	}
538 	if (y = Dwt.__checkPxVal(y)) {
539 		htmlElement.style.top = y;
540 	}
541 };

553 /**
554  * Returns <code>htmlElement</code>'s scroll style. The scroll style determines the element's
555  * behaviour when content overflows its boundaries. Possible values are:
556  * <ul>
557  * <li><i>Dwt.CLIP</i> - Clip on overflow</li>
558  * <li><i>Dwt.VISIBLE</i> - Allow overflow to be visible</li>
559  * <li><i>Dwt.SCROLL</i> - Automatically create scrollbars if content overflows</li>
560  * <li><i>Dwt.FIXED_SCROLL</i> - Always have scrollbars whether content overflows or not</li>
561  * </ul>
562  *
563  * @param {HTMLElement} htmlElement HTML element
564  *
565  * @return {Dwt.CLIP|Dwt.VISIBLE|Dwt.SCROLL|Dwt.FIXED_SCROLL}	the elements scroll style
566  */
567 Dwt.getScrollStyle =
568 function(htmlElement) {
569 	var overflow =  DwtCssStyle.getProperty(htmlElement, "overflow");
570 
571 	if (overflow == "hidden")		{ return Dwt.CLIP; }
572 	if (overflow =="auto")			{ return Dwt.SCROLL; }
573 	if (overflow =="scroll")		{ return Dwt.FIXED_SCROLL; }
574 
575 	if (overflow == '') {
576 		var overflowX = DwtCssStyle.getProperty(htmlElement, "overflowX");
577 		var overflowY = DwtCssStyle.getProperty(htmlElement, "overflowY");
578 
579 		if (overflowX == 'scroll')	{ return Dwt.SCROLL_X; }
580 		if (overflowY == 'scroll')	{ return Dwt.SCROLL_Y; }
581 	}
582 	return Dwt.VISIBLE;
583 };

585 /**
586  * Sets the <code>htmlElement</code>'s scroll style. The scroll style determines the elements's
587  * behaviour when content overflows its div's boundaries. Possible values are:
588  * <ul>
589  * <li><i>Dwt.CLIP</i> - Clip on overflow</li>
590  * <li><i>Dwt.VISIBLE</i> - Allow overflow to be visible</li>
591  * <li><i>Dwt.SCROLL</i> - Automatically create scrollbars if content overflows</li>
592  * <li><i>Dwt.FIXED_SCROLL</i> - Always have scrollbars whether content overflows or not</li>
593  * </ul>
594  *
595  * @param {HTMLElement} htmlElement HTML element
596  * @param {Dwt.CLIP|Dwt.VISIBLE|Dwt.SCROLL|Dwt.FIXED_SCROLL}	scrollStyle		the elements scroll style
597  */
598 Dwt.setScrollStyle =
599 function(htmlElement, scrollStyle) {
600 	if (scrollStyle == Dwt.CLIP)
601 		htmlElement.style.overflow = "hidden";
602 	else if (scrollStyle == Dwt.SCROLL)
603 		htmlElement.style.overflow = "auto";
604 	else if (scrollStyle == Dwt.FIXED_SCROLL)
605 		htmlElement.style.overflow = "scroll";
606 	else if (scrollStyle == Dwt.SCROLL_Y) {
607 		htmlElement.style.overflowX = "hidden";
608 		htmlElement.style.overflowY = "auto";
609 	} else if (scrollStyle == Dwt.SCROLL_X) {
610 		htmlElement.style.overflowY = "hidden";
611 		htmlElement.style.overflowX = "auto";
612 	} else {
613 		htmlElement.style.overflow = "visible";
614 	}
615 };

617 // Note: in FireFox, offsetHeight includes border and clientHeight does not;
618 // may want to look at clientHeight for FF
619 Dwt.getSize =
620 function(htmlElement, point) {
621 	var p;
622 	if (!point) {
623 		p = new DwtPoint(0, 0);
624 	} else {
625 		p = point;
626 		p.set(0, 0);
627 	}
628 
629 	if (!htmlElement) { return p; }
630 
631 	p.x = htmlElement.offsetWidth;
632 	if (p.x != null) {
633 		p.y = htmlElement.offsetHeight;
634 	} else if (htmlElement.clip && htmlElement.clip.width != null) {
635 		p.x = parseInt(htmlElement.clip.width);
636 		p.y = parseInt(htmlElement.clip.height);
637 	} else if (htmlElement.style && htmlElement.style.pixelWidth != null) {
638 		p.x = parseInt(htmlElement.style.pixelWidth);
639 		p.y = parseInt(htmlElement.style.pixelHeight);
640 	}
641 	return p;
642 };

644 Dwt.setSize =
645 function(htmlElement, width, height) {
646 	if (!htmlElement.style) { return; }
647 
648 	if (width == Dwt.CLEAR) {
649 		htmlElement.style.width = null;
650 	} else if (width = Dwt.__checkPxVal(width, true)) {
651 		htmlElement.style.width = width;
652 	}
653 
654 	if (height == Dwt.CLEAR) {
655 		htmlElement.style.height = null;
656 	} else if (height = Dwt.__checkPxVal(height, true)) {
657 		htmlElement.style.height = height;
658 	}
659 };

676 Dwt.toDocumentFragment =
677 function(html, id) {
678 	var div = AjxStringUtil.calcDIV();
679 	div.innerHTML = html;
680 
681 	var fragment = document.createDocumentFragment();
682 	var container = id && document.getElementById(id);
683 	if (container) {
684 		fragment.appendChild(container);
685 	}
686 	else {
687 		for (var child = div.firstChild; child; child = div.firstChild) {
688 			fragment.appendChild(child);
689 		}
690 	}
691 	return fragment;
692 };

694 Dwt.getAttr =
695 function(htmlEl, attr, recursive) {
696 	// test for tagName so we dont try to eval non-html elements (i.e. document)
697 	if (!recursive) {
698 		return htmlEl && htmlEl.tagName
699 			? (htmlEl.getAttribute(attr) || htmlEl[attr])
700 			: null;
701 	} else {
702 		while (htmlEl) {
703 			if (Dwt.getAttr(htmlEl, attr) != null) {
704 				return htmlEl;
705 			}
706 			htmlEl = htmlEl.parentNode;
707 		}
708 		return null;
709 	}
710 };

718 Dwt.setVisible =
719 function(htmlElement, visible) {
720 	if (visible) {
721 		if (htmlElement.nodeName.match(/tr/i)) {
722 			htmlElement.style.display = Dwt.DISPLAY_TABLE_ROW;
723 		}
724 		else if (htmlElement.nodeName.match(/td|th/i)) {
725 			htmlElement.style.display = Dwt.DISPLAY_TABLE_CELL;
726 		}
727 		else {
728 			htmlElement.style.display = htmlElement.getAttribute("x-display") ||
729 										Dwt.DISPLAY_BLOCK;
730 		}
731 	} else {
732 		var display = DwtCssStyle.getComputedStyleObject(htmlElement).display;
733 		if (display != "none") {
734 			htmlElement.setAttribute("x-display", display);
735 		}
736 		htmlElement.style.display = Dwt.DISPLAY_NONE;
737 	}
738 };

814 Dwt.toWindow =
815 function(htmlElement, x, y, containerElement, dontIncScrollTop, point) {
816 	var p;
817 	if (!point) {
818 		p = new DwtPoint(x, y);
819 	} else {
820 		p = point;
821 		p.set(x, y);
822 	}
823 
824 	var offsetParent = htmlElement;
825 	while (offsetParent && offsetParent != containerElement) {
826 		p.x += offsetParent.offsetLeft - offsetParent.scrollLeft;
827 		p.y += offsetParent.offsetTop;
828 		if (!dontIncScrollTop) {
829 			var scrollTop = AjxEnv.isOpera ? offsetParent.pageYOffset : offsetParent.scrollTop;
830 			if (scrollTop) {
831 				p.y -= scrollTop;
832 			}
833 			var parentNode = offsetParent.parentNode;
834 			while (parentNode != offsetParent.offsetParent && parentNode != containerElement) {
835 				scrollTop = AjxEnv.isOpera ? parentNode.pageYOffset : parentNode.scrollTop;
836 				if (scrollTop) {
837 					p.y -= scrollTop;
838 				}
839 				parentNode = parentNode.parentNode;
840 			}
841 		}
842 		offsetParent = offsetParent.offsetParent;
843 	}
844 	return p;
845 };

1458 /**
1459  * Returns true if the two elements overlap.
1460  * 
1461  * @param el1
1462  * @param el2
1463  */
1464 Dwt.doOverlap =
1465 function(el1, el2) {
1466 
1467 	if (!el1 || !el2) { return false; }
1468 
1469 	var loc1 = Dwt.getLocation(el1), loc2 = Dwt.getLocation(el2);
1470 	var size1 = Dwt.getSize(el1), size2 = Dwt.getSize(el2);
1471 	var left1 = loc1.x, left2 = loc2.x, top1 = loc1.y, top2 = loc2.y;
1472 	var right1 = left1 + size1.x, right2 = left2 + size2.x;
1473 	var bottom1 = top1 + size1.y, bottom2 = top2 + size2.y;
1474 
1475 	return !(left1 > right2 || right1 < left2 || top1 > bottom2 || bottom1 < top2);
1476 };
