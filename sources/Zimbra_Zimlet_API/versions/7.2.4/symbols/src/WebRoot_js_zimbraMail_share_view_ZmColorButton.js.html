 16 ZmColorButton = function(params) {
 17     if (arguments.length == 0) return;
 18     DwtButton.call(this, params);
 19     var menu = new ZmColorMenu({parent:this,hideNone:params.hideNone});
 20     menu.addSelectionListener(new AjxListener(this, this._handleSelection));
 21     this.setMenu(menu);
 22     this._colorMenu = menu;
 23 };
 24 ZmColorButton.prototype = new DwtButton;
 25 ZmColorButton.prototype.constructor = ZmColorButton;
 26 
 27 ZmColorButton.prototype.toString = function() {
 28     return "ZmColorButton";
 29 };
 30 
 31 //
 32 // Public methods
 33 //
 34 
 35 ZmColorButton.prototype.setImage = function(image, skipMenu) {
 36     DwtButton.prototype.setImage.apply(this, arguments);
 37     if (!skipMenu) {
 38         this._colorMenu.setImage(image);
 39     }
 40 };
 41 
 42 ZmColorButton.prototype.setValue = function(color) {
 43 	var standardColorCode = ZmOrganizer.getStandardColorNumber(color);
 44 	if(standardColorCode != -1) {
 45 	 this._color = standardColorCode;
 46 	} else {
 47     this._color = color;
 48 	}
 49     var image = this.getImage();
 50     if (image) {
 51         image = image.replace(/,.*$/,"");
 52         this.setImage(this._color?[image,this._color].join(",color="):image, true);
 53     }
 54     this.setText(this._colorMenu.getTextForColor(this._color));
 55 };
 56 
 57 
 58 ZmColorButton.prototype.getValue = function() {
 59     return this._color;
 60 };
 61 
 62 //
 63 // Protected methods
 64 //
 65 
 66 ZmColorButton.prototype._handleSelection = function(evt) {
 67     this.setValue(evt.item.getData(ZmOperation.MENUITEM_ID)); 
 68 };
