285 ZmAccountsPage.prototype.setAccount =
286 function(account, skipUpdate, ignoreProvider) {
287 	// keep track of changes made to current account
288 	if (this._currentAccount) {
289 		if (!skipUpdate) {
290 			this._setAccountFields(this._currentAccount, this._currentSection);
291 		}
292 		this._tabGroup.removeMember(this._currentSection.tabGroup);
293 		this._currentAccount = null;
294 		this._currentSection = null;
295 	}
296 
297 	// toggle delete button
298 	if (this._deleteButton) {
299 		var isEnabled = (appCtxt.isOffline)
300 			? (account && account.type == ZmAccount.TYPE_PERSONA)
301 			: (account && account.type != ZmAccount.TYPE_ZIMBRA);
302 		this._deleteButton.setEnabled(isEnabled);
303 	}
304 
305 	// intialize sections
306 	for (var type in this._sectionDivs) {
307 		Dwt.setVisible(this._sectionDivs[type], false);
308 	}
309 
310 	// HACK: Attempt to get around an IE update issue.
311 	setTimeout(AjxCallback.simpleClosure(this._setAccount2, this, account, skipUpdate, ignoreProvider),0);
312 };

314 ZmAccountsPage.prototype._setAccount2 =
315 function(account, skipUpdate, ignoreProvider) {
316 	// NOTE: hide all sections first, then show the specific section b/c some
317 	// sections use same div. This avoids double inititalization in that case.
318 	var isExternal = account instanceof ZmDataSource;
319 	var provider = !ignoreProvider && isExternal && account.getProvider();
320 	var div = (provider && this._sectionDivs[provider.id]) || this._getSectionDiv(account);
321     if (div) {
322         this._currentAccount = account;
323         Dwt.setVisible(div, true);
324         if (appCtxt.isOffline) {
325             // bug 48014 - add "Use this persona" section for offline
326             if(account.type == ZmAccount.TYPE_PERSONA) {
327                 this._currentSection = ZmAccountsPage.SECTIONS["PERSONA"];
328                 this._setPersona(account, this._currentSection);
329             } else {
330                 this._currentSection = ZmAccountsPage.SECTIONS["PRIMARY"];
331                 this._setZimbraAccount(account, this._currentSection);
332             }
333         } else {
334             switch (account.type) {
335                 case ZmAccount.TYPE_POP:
336                 case ZmAccount.TYPE_IMAP: {
337                     this._currentSection = provider && ZmAccountsPage.SECTIONS[provider.id];
338                     this._currentSection = this._currentSection || ZmAccountsPage.SECTIONS["EXTERNAL"];
339                     this._setExternalAccount(account, this._currentSection);
340                     if (ignoreProvider) {
341                         this._setControlValue("PROVIDER", this._currentSection, "");
342                     }
343                     if (!skipUpdate) {
344                         var password = this._getControlObject("PASSWORD", this._currentSection);
345                         if (password) {
346                             password.setShowPassword(false);
347                         }
348                     }
349                     break;
350                 }
351                 case ZmAccount.TYPE_PERSONA: {
352                     this._currentSection = ZmAccountsPage.SECTIONS["PERSONA"];
353                     this._setPersona(account, this._currentSection);
354                     break;
355                 }
356                 default: {
357                     this._currentSection = ZmAccountsPage.SECTIONS["PRIMARY"];
358                     this._setZimbraAccount(account, this._currentSection);
359                     break;
360                 }
361             }
362         }
363 		if (!this._tabGroup.contains(this._currentSection.tabGroup)) {
364 			this._tabGroup.addMember(this._currentSection.tabGroup);
365 		}
366 	}
367 
368 	// update list cells
369 	this._updateList(account);
370 
371 	var control = this._currentSection && this._currentSection.controls[isExternal ? "EMAIL" : "NAME"];
372 
373 	// When a hidden field is applied focus(), IE throw's an exception.
374 	// Thus checking for isActive()
375 	if (control && this.isActive()) {
376 		control.focus();
377 	}
378 };

398 ZmAccountsPage.prototype.reset =
399 function(useDefaults) {
400 	ZmPreferencesPage.prototype.reset.apply(this, arguments);
401 
402 	// clear current list of accounts
403 	this._accounts.removeAll();
404 	this._deletedAccounts = [];
405 	this._currentAccount = null;
406 	this._currentSection = null;
407 
408 	var usedIdentities = {};
409 
410 	// add zimbra accounts (i.e. family mboxes)
411 	var mboxes = appCtxt.accountList.getAccounts();
412 	var active = appCtxt.getActiveAccount();
413 	for (var j in mboxes) {
414 		var acct = mboxes[j];
415 		var ident = acct.getIdentity();
416 		// NOTE: We create proxies of all of the account objects so that we can
417 		//       store temporary values while editing.
418 		if (active.isMain || acct == active) {
419 			this._accounts.add(ZmAccountsPage.__createProxy(acct));
420 			if (ident) usedIdentities[ident.id] = true;
421 		}
422 	}
423 	// add data sources unless we're in offline mode
424 	if (!appCtxt.isOffline && active.isMain) {
425 		var dataSourceCollection = appCtxt.getDataSourceCollection();
426 		var dataSources = dataSourceCollection.getItems(); // TODO: rename getItems or rename getIdentities
427 		for (var i = 0; i < dataSources.length; i++) {
428 			var datasource = dataSources[i];
429 			var ident = datasource.getIdentity();
430 			delete datasource.password;
431 			this._accounts.add(ZmAccountsPage.__createProxy(datasource));
432 			if (ident) usedIdentities[ident.id] = true;
433 		}
434 	}
435 
436 	// add identities/personas
437 	var identityCollection = appCtxt.getIdentityCollection();
438 	var identities = identityCollection.getIdentities();
439 	for (var i = 0; i < identities.length; i++) {
440 		var identity = identities[i];
441 		if (identity.isDefault || identity.isFromDataSource) continue;
442 		var id = identity.id;
443 		if (usedIdentities[id]) continue;
444 		usedIdentities[id] = true;
445 		var persona = new ZmPersona(identity);
446 		this._accounts.add(ZmAccountsPage.__createProxy(persona), null, true);
447 	}
448 
449 	// initialize list view
450 	this._accounts.sort(ZmAccountsPage.__ACCOUNT_COMPARATOR);
451 	var account = this._accounts.get(0);
452 	this._resetAccountListView(account);
453 	this.setAccount(account);
454 };

458 ZmAccountsPage.prototype.isDirty =
459 function() {
460 	// make sure that the current object proxy is up-to-date
461 	this._setAccountFields(this._currentAccount, this._currentSection, true);
462 
463 	var dirty = this._deletedAccounts.length > 0;
464 	if (!dirty) {
465 		var accounts = this._accounts.getArray();
466 		for (var i = 0; i < accounts.length; i++) {

468 			if (account._new || account._dirty || account._visibleDirty) {
469 				dirty = true;
470 				break;
471 			}
472 		}
473 	}
474 	return dirty;
475 };

477 /**
478  * Does minimal checking:
479  * <li>bug 21104: persona name and the associated display value for the From address
480  * <li>bug 950: email addresses
481  */
482 ZmAccountsPage.prototype.validate =
483 function() {
484 	var accounts = this._accounts.getArray();
485 	for (var i = 0; i < accounts.length; i++) {
486 		var account = accounts[i];
487 		var type = account.type;
488 		var isPrimary = type == ZmAccount.TYPE_ZIMBRA;
489 		var isExternal = type == ZmAccount.TYPE_POP || type == ZmAccount.TYPE_IMAP;
490 		var isPersona = type == ZmAccount.TYPE_PERSONA;
491 
492 		// bug 21104
493 		if (isPersona && (account._new || account._dirty) && 
494 			!(account.identity && account.identity.name)) {
495 			this._errorMsg = ZmMsg.invalidPersonaName;
496 			return false;
497 		}
498 		// bug 950
499 		if (isExternal && !this.__validateEmail(this.__getAccountValue(account, "EMAIL"))) {
500 			return false;
501 		}
502 		if (this.__getIdentityValue(account, "REPLY_TO") &&
503 			!this.__validateEmail(this.__getIdentityValue(account, "REPLY_TO_EMAIL"))) {
504 			return false;
505 		}
506 		if (isExternal && !this.__validateEmail(this.__getIdentityValue(account, "FROM_EMAIL"))) {
507 			return false;
508 		}
509 		if (isPersona && this.__getIdentityValue(account, "WHEN_SENT_TO") &&
510 			!this.__validateEmailList(this.__getIdentityValue(account, "WHEN_SENT_TO_LIST"))) {
511 			return false;
512 		}
513 	}
514 	return true;
515 };

566 ZmAccountsPage.prototype.addCommand =
567 function(batchCmd) {
568 	// make sure that the current object proxy is up-to-date
569 	// NOTE: This is already done so don't do it again or else we'll
570 	// NOTE: lose the folderId from the create/rename folder op.
571 	//this._setAccountFields(this._currentAccount, this._currentSection);
572 
573 	// delete accounts
574 	for (var i = 0; i < this._deletedAccounts.length; i++) {
575 		var callback = null;
576 		var account = this._deletedAccounts[i];
577 		var folderId = account.folderId;
578 		if (folderId == ZmAccountsPage.DOWNLOAD_TO_FOLDER || folderId != ZmAccountsPage.DOWNLOAD_TO_INBOX) {
579 			var root = appCtxt.getById(ZmOrganizer.ID_ROOT);
580 			var name = account.getName();
581 			var folder = root.getByName(name);
582 			if (folder && !folder.isSystem()) {
583 				callback = new AjxCallback(this, this._promptToDeleteFolder, [folder]);
584 			}
585 		}
586 		this._deletedAccounts[i].doDelete(callback, null, batchCmd);
587 	}
588 
589 	// for multi-account mbox, check if user changed visible flag on subaccounts
590 	if (appCtxt.accountList.size(true) > 1) {
591 		this._saveVisibleAccounts(batchCmd);
592 	}
593 
594 	// modify existing accounts
595 	var newAccounts = [];
596 	var accounts = this._accounts.getArray();
597 	for (var i = 0; i < accounts.length; i++) {
598 		var account = accounts[i];
599 		if (account._new) {
600 			newAccounts.push(account);
601 			continue;
602 		}
603 
604 		if (account._dirty) {
605 			var callback = new AjxCallback(this, this._handleSaveAccount, [account]);
606 			account.save(callback, null, batchCmd);
607 		}
608 	}
609 
610 	// add new accounts
611 	for (var i = 0; i < newAccounts.length; i++) {
612 		var account = newAccounts[i];
613 		var callback = new AjxCallback(this, this._handleCreateAccount, [account]);
614 		account.create(callback, null, batchCmd);
615 	}
616

901 function(id, section) {
902 	var control = section.controls[id];

904 	if (!setup) return null;
905 	if (!control) {

929 	}

931 		value = !control.isSelected();
932 	}
933 	else {
934 		switch (setup.displayContainer) {
935 			case ZmPref.TYPE_STATIC: {
936 				value = control.getText();
937 				break;
938 			}
939 			case ZmPref.TYPE_CHECKBOX: {
940 				value = control.isSelected();
941 				if (setup.options) {

949 			}

951 			case ZmPref.TYPE_SELECT: {

953 				break;
954 			}
955 			case ZmPref.TYPE_COMBOBOX: {
956 				value = control.getValue() || control.getText();
957 				break;
958 			}
959 		}
960 	}

1048 ZmAccountsPage.prototype._valuesEqual =
1049 function(ovalue, nvalue) {
1050 	if (AjxUtil.isArray(ovalue) && AjxUtil.isArray(nvalue)) {
1051 		if (ovalue.length != nvalue.length) {
1052 			return false;
1053 		}
1054 		var oarray = [].concat(ovalue).sort();
1055 		var narray = [].concat(nvalue).sort();
1056 		for (var i = 0; i < oarray.length; i++) {
1057 			if (oarray[i] != narray[i]) {
1058 				return false;
1059 			}
1060 		}
1061 		return true;
1062 	}
1063 	return ovalue == nvalue;

1276 /*

1278  * @param accounts	array of account objects
1279  * @param unique	boolean: if true, addresses will be included in output only if they are not already present. Defaults to true
1280  * @param valid		boolean: if true, performs a validation check on the address and only includes it if it passes. Defaults to true
1281  * @param addresses	optional array of addresses (as strings) to append to. Defaults to an empty array
1282 */
1283 ZmAccountsPage.prototype._getAddressesFromAccounts = function(accounts, addresses, unique, valid) {
1284 	if (!AjxUtil.isArray(addresses))
1285 		addresses = [];
1286 	for (var i=0; i<accounts.length; i++) {
1287 		var account = accounts[i];
1288 		if (account.isMain || account.enabled) {
1289 			var address = account.getEmail();
1290 			if (!AjxUtil.isEmpty(address) && (!valid || AjxEmailAddress.isValid(address)) && (!unique || AjxUtil.indexOf(addresses, address, false) == -1)) // Make sure we are not adding an empty address and that we are not adding the address twice
1291 				addresses.push(address);
1292 		}
1293 	}
1294 	return addresses;

1805 ZmAccountsPage.prototype._preSaveCreateFolders =
1806 function(continueCallback) {
1807 	var batchCmd;
1808 	var root = appCtxt.getById(ZmOrganizer.ID_ROOT);
1809 	var accounts = this._accounts.getArray();
1810 	for (var i = 0; i < accounts.length; i++) {
1811 		var account = accounts[i];
1812 		if (account.type == ZmAccount.TYPE_POP || account.type == ZmAccount.TYPE_IMAP) {
1813 			if (account.folderId != ZmOrganizer.ID_INBOX) {
1814 				var name = AjxStringUtil.trim(account.getName());
1815 				if (!batchCmd) { batchCmd = new ZmBatchCommand(false); }
1816 
1817 				// avoid folder create if it already exists
1818 				var folder = root.getByName(name, true);
1819 				if (folder) {
1820 					var folderId = folder.id;
1821 					if (folderId != ZmOrganizer.ID_INBOX && Number(folderId) < 256) {
1822 						var params = {
1823 							msg: AjxMessageFormat.format(ZmMsg.accountNameReserved, [AjxStringUtil.htmlEncode(name)]),
1824 							level: ZmStatusView.LEVEL_CRITICAL
1825 						};
1826 						appCtxt.setStatusMsg(params);
1827 						continueCallback.run(false);
1828 						return;
1829 					}
1830 					// if there already is a folder by this name in Trash, rename the trashed folder
1831 					else if (folder.isInTrash()) {
1832 						folder.rename(folder.name+"_");
1833 					}
1834 					else {
1835 						account.folderId = folder.id;
1836 						continue;
1837 					}
1838 				}
1839 
1840 				// this means user modified name of the folder, so let's rename it
1841 				folder = account._object_ && appCtxt.getById(account._object_.folderId);
1842 				if (folder && Number(folder.id) >= 256) {
1843 					if (folder.name != name) {
1844 						var soapDoc = AjxSoapDoc.create("FolderActionRequest", "urn:zimbraMail");
1845 						var actionNode = soapDoc.set("action");
1846 						actionNode.setAttribute("op", "rename");
1847 						actionNode.setAttribute("id", folder.id);
1848 						actionNode.setAttribute("name", name);
1849 
1850 						var callback = new AjxCallback(this, this._handleRenameFolderResponse, [account, folder]);
1851 						batchCmd.addNewRequestParams(soapDoc, callback, callback);
1852 					}
1853 					else {
1854 						account.folderId = folder.id;
1855 					}
1856 				} else {
1857 					var soapDoc = AjxSoapDoc.create("CreateFolderRequest", "urn:zimbraMail");
1858 					var folderEl = soapDoc.set("folder");
1859 					folderEl.setAttribute("l", ZmOrganizer.ID_ROOT);
1860 					folderEl.setAttribute("name", name);
1861 					folderEl.setAttribute("fie", "1"); // fetch-if-exists
1862 
1863 					var callback = new AjxCallback(this, this._handleCreateFolderResponse, [account]);
1864 					batchCmd.addNewRequestParams(soapDoc, callback, callback);
1865 				}
1866 			}
1867 		}
1868 	}
1869 
1870 	// continue
1871 	if (batchCmd && batchCmd.size() > 0) {
1872 		// HACK: Don't know a better way to set an error condition
1873 		this.__hack_preSaveSuccess = true;
1874 		var callback = new AjxCallback(this, this._preSaveFinish, [continueCallback]);
1875 		batchCmd.run(callback);
1876 	}
1877 	else {
1878 		this._preSaveFinish(continueCallback);
1879 	}
1880 };
