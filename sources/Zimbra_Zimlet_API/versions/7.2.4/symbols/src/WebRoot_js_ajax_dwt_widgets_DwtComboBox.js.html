 16 /**
 17  * Creates a combo box.
 18  * @constructor
 19  * @class
 20  * This class represents a combo box.
 21  *
 22  * @author Dave Comfort
 23  * 
 24  * @param {hash}	params		a hash of parameters
 25  * @param {DwtComposite}      parent		the parent widget
 26  * @param {hash}	inputParams		params for the input (see {@link DwtInputField})
 27  * @param {string}      className		the CSS class
 28  * @param {constant}      posStyle		the positioning style (see {@link DwtControl})
 29  * 
 30  * @extends		DwtComposite
 31  */
 32 DwtComboBox = function(params) {
 33     if (arguments.length == 0) { return; }
 34 	params = Dwt.getParams(arguments, DwtComboBox.PARAMS);
 35     params.className = params.className || "DwtComboBox";
 36     DwtComposite.call(this, params);
 37     
 38     this.input = null;
 39     this._button = null;
 40     
 41     this._textToValue = {}; // Map of text strings to their values.
 42     this._valueToText = {};
 43     this._valueToItem = {};
 44 	this._size = 0;
 45 
 46     this._hasMenuCallback = true;
 47 	this._menuItemListenerObj = new AjxListener(this, this._menuItemListener);
 48 
 49     this._inputParams = params.inputParams;
 50     this._createHtml();
 51 };
 52 
 53 DwtComboBox.PARAMS = ["parent", "inputParams", "className", "posStyle", "dialog"];

 95 /**
 96  * Adds an entry to the combo box list.
 97  * 
 98  * @param {string}	text		the user-visible text for the entry
 99  * @param {string}	value		the value for the entry
100  * @param {boolean}	selected	if <code>true</code>, the entry is selected
101  */
102 DwtComboBox.prototype.add =
103 function(text, value, selected) {
104 	this._textToValue[text] = value;
105     this._valueToText[value] = text;
106     if (!this._hasMenuCallback) {
107 		var menu = this._button.getMenu();
108     	this._createMenuItem(menu, text);
109 	}
110 	if (selected) {
111 		this.setText(text);
112 	}
113 	this._size++;
114 	this._updateButton();
115 };
116 
117 /**
118  * Removes the specified value from the list.
119  *
120  * @param	{string}	value		the value
121  */
122 DwtComboBox.prototype.remove = function(value) {
123     var item = this._valueToItem[value];
124     if (item) {
125         this._button.getMenu().removeChild(item);
126         var text = this._valueToText[value];
127         delete this._textToValue[text];
128         delete this._valueToText[value];
129         delete this._valueToItem[value];
130         if (this.getText() == text) {
131             this.setText("");
132         }
133 		this._size--;
134 		this._updateButton();
135 	}
136 };
137 
138 /**
139  * Removes all the items in the list.
140  * 
141  */
142 DwtComboBox.prototype.removeAll = function() {
143     this._button.setMenu(new AjxCallback(this, this._createMenu), true);
144     this._hasMenuCallback = true;
145 
146     this._textToValue = {};
147     this._valueToText = {};
148     this._valueToItem = {};
149 	this._size = 0;
150 	this._updateButton();
151 };
152 
153 /**
154  * Gets the value of the currently selected entry. If the entry
155  * is one that was not added via the add method (that is, if it was
156  * typed in by the user) then <code>null</code> is returned.
157  * 
158  * @return	{string}	the value
159  */
160 DwtComboBox.prototype.getValue =
161 function() {
162 	var text = this.getText();
163 	return this._textToValue[text];
164 };
165 
166 /**
167  * Sets the value.
168  * 
169  * @param	{string}	value		the value
170  */
171 DwtComboBox.prototype.setValue = function(value) {
172 	var text = this._valueToText[value];
173 	this.setText(text || value);
174 };

176 /**

178  *

181 DwtComboBox.prototype.getText =
182 function() {

197 function(enabled) {

199 		DwtComposite.prototype.setEnabled.call(this, enabled);

201 		this._button.setEnabled(enabled);
202     }

213 DwtComboBox.prototype._createMenu =
214 function() {
215     var menu = new DwtMenu({parent:this});
216     for (var i in this._textToValue) {
217     	var item = this._createMenuItem(menu, i);
218         var value = this._textToValue[i];
219         this._valueToItem[value] = item;
220     }
221 	this._hasMenuCallback = false;
222 	return menu;
223 };
224 
225 DwtComboBox.prototype._createMenuItem =
226 function(menu, text) {
227 	var item = new DwtMenuItem({parent:menu});
228 	item.setText(text);
229 	item.addSelectionListener(this._menuItemListenerObj);
230 	if (!this._menuWidth) {
231 		this._menuWidth = this.getW() - 10; // 10 is some fudge factor that lines up the menu right.
232 	}
233     item.getHtmlElement().style.minWidth = this._menuWidth;
234     return item;
235 };
236 
237 DwtComboBox.prototype._menuItemListener =
238 function(ev) {
239 	var menuItem = ev.dwtObj;
240 	var ovalue = this.input.getValue();
241 	var nvalue = menuItem.getText();
242 	this.input.setValue(nvalue);
243 
244 	// notify our listeners
245 	var event = DwtUiEvent.getEvent(ev);
246 	event._args = { selectObj: this, newValue: nvalue, oldValue: ovalue };
247 	this.notifyListeners(DwtEvent.ONCHANGE, event);
248 
249 	var input = this.input.getInputElement();
250 	input.focus();
251 	input.select();
252 };
253 
254 DwtComboBox.prototype._handleKeyDown = function(ev) {
255 	this.__ovalue = this.input.getValue();
256 	return true;
257 };
258 
259 DwtComboBox.prototype._handleKeyUp = function(ev) {
260 	// propagate event to DwtInputField
261 	DwtInputField._keyUpHdlr(ev);
262 	// notify our listeners
263 	var event = DwtUiEvent.getEvent(ev);
264 	event._args = { selectObj: this, newValue: this.input.getValue(), oldValue: this.__ovalue };
265 	this.notifyListeners(DwtEvent.ONCHANGE, event);
266 	return true;
267 };
268 
269 DwtComboBox.prototype._updateButton =
270 function() {
271 	this._button.setVisible(this._size > 0);
272 };
273 
274 DwtComboBox.prototype._createHtml = function(templateId) {
275     var data = { id: this._htmlElId };
276     this._createHtmlFromTemplate(templateId || this.TEMPLATE, data);
277 };
278 
279 DwtComboBox.prototype._createHtmlFromTemplate = function(templateId, data) {
280     DwtComposite.prototype._createHtmlFromTemplate.call(this, templateId, data);
281 
282 	var inputParams = this._inputParams || {};
283 	inputParams.parent = this;
284 	inputParams.size = inputParams.size || 40;
285 	delete this._inputParams;
286     
287     this.input = new DwtInputField(inputParams);
288     this.input.replaceElement(data.id + "_input");
289 	this.input.setHandler(DwtEvent.ONKEYDOWN, AjxCallback.simpleClosure(this._handleKeyDown, this));
290 	this.input.setHandler(DwtEvent.ONKEYUP, AjxCallback.simpleClosure(this._handleKeyUp, this));
291 
292     this._button = new DwtComboBoxButton({parent:this});
293 	this._button.setMenu(new AjxListener(this, this._createMenu), true);
294     this._button.replaceElement(data.id + "_button");
295 	this._updateButton();
296 
297 	this._tabGroup = new DwtTabGroup(this._htmlElId);
298 	this._tabGroup.addMember(this.input);
299 	this._tabGroup.addMember(this._button);
300 };
301 
302 /**
303  * The input field inherits the id for accessibility purposes.
304  * 
305  * @private
306  */
307 DwtComboBox.prototype._replaceElementHook =
308 function(oel, nel, inheritClass, inheritStyle) {
309 	DwtComposite.prototype._replaceElementHook.apply(this, arguments);
310 	// set input settings
311 	if (oel.size) {
312 		this.input.getInputElement().size = oel.size;
313 	}
314 	if (oel.title) {
315 		this.input.setHint(oel.title);
316 	}
317 };

323 /**
324  * DwtComboBoxButton: Stylizable button just for use in combo boxes.
325  * 
326  * @param {hash}	params		a hash of parameters
327  * @param {DwtComposite}       params.parent		the parent widget
328  * @param	{string}       params.className		the CSS class
329  * 
330  * @extends		DwtButton
331  * @private
332  */
333 DwtComboBoxButton = function(params) {
334 	params = Dwt.getParams(arguments, DwtComboBoxButton.PARAMS);
335 	params.posStyle = Dwt.RELATIVE_STYLE;
336 	DwtButton.call(this, params);
337 }
