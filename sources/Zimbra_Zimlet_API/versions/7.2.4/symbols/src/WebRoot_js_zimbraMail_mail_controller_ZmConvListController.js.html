 16 /**
 17  * Creates a new, empty conversation list controller.
 18  * @constructor
 19  * @class
 20  * This class manages the conversations mail view. Conversations are listed, and any
 21  * conversation with more than one message is expandable. Expanding a conversation
 22  * shows its messages in the list just below it.
 23  *
 24  * @author Conrad Damon
 25  *
 26  * @param {ZmComposite}	container	the containing shell
 27  * @param {ZmMailApp}	mailApp			the containing app
 28  * 
 29  * @extends		ZmDoublePaneController
 30  */
 31 ZmConvListController = function(container, mailApp) {
 32 	ZmDoublePaneController.call(this, container, mailApp);
 33 	this._msgControllerMode = ZmId.VIEW_CONVLIST;
 34 };

 74 /**
 75  * Handles switching the order of messages within expanded convs.
 76  *
 77  * @param view		[constant]*		the id of the new order
 78  * @param force		[boolean]		if true, always redraw view
 79  */
 80 ZmConvListController.prototype.switchView =
 81 function(view, force) {
 82 
 83 	if (view == ZmSearch.DATE_DESC || view == ZmSearch.DATE_ASC) {
 84 		if ((appCtxt.get(ZmSetting.CONVERSATION_ORDER) != view) || force) {
 85 			appCtxt.set(ZmSetting.CONVERSATION_ORDER, view);
 86 			this._mailListView.redoExpansion();
 87 		}
 88 		if (this._toolbar && this._toolbar[this._currentView])
 89 			this._toolbar[this._currentView].adjustSize();
 90 	} else {
 91 		ZmDoublePaneController.prototype.switchView.apply(this, arguments);
 92 	}
 93 };

101 ZmConvListController.prototype.handleKeyAction =
102 function(actionCode) {
103 	DBG.println(AjxDebug.DBG3, "ZmConvListController.handleKeyAction");
104 	
105 	var mlv = this._mailListView;
106 	switch (actionCode) {
107 
108 		case ZmKeyMap.EXPAND:
109 		case ZmKeyMap.COLLAPSE:
110 			if (mlv.getSelectionCount() != 1) { return false; }
111 			var item = mlv.getItemFromElement(mlv._kbAnchor);
112 			if (!item) { return false; }
113 			if ((actionCode == ZmKeyMap.EXPAND) != mlv._expanded[item.id]) {
114 				mlv._expandItem(item);
115 			}
116 			break;
117 
118 		case ZmKeyMap.TOGGLE:
119 			if (mlv.getSelectionCount() != 1) { return false; }
120 			var item = mlv.getItemFromElement(mlv._kbAnchor);
121 			if (!item) { return false; }
122 			if (item.type == ZmItem.CONV && !mlv._isExpandable(item)) {
123 				return DwtListView.prototype.handleKeyAction.call(mlv, DwtKeyMap.DBLCLICK);
124 			} else {
125 				mlv._expandItem(item);
126 			}
127 			break;
128 
129 		case ZmKeyMap.EXPAND_ALL:
130 			mlv._expandAll(true);
131 			break;
132 
133 		case ZmKeyMap.COLLAPSE_ALL:
134 			mlv._expandAll(false);
135 			break;
136 
137 		case ZmKeyMap.NEXT_UNREAD_MSG:
138 		case ZmKeyMap.PREV_UNREAD_MSG:
139 			this.lastListAction = actionCode;
140 			var selItem, noBump = false;
141 			if (mlv.getSelectionCount() == 1) {
142 				var sel = mlv.getSelection();
143 				selItem = sel[0];
144 				if (selItem && mlv._isExpandable(selItem)) {
145 					noBump = true;
146 				}
147 			}
148 
149 		case ZmKeyMap.FIRST_UNREAD_MSG:
150 		case ZmKeyMap.LAST_UNREAD_MSG:
151 			var item = (selItem && selItem.type == ZmItem.MSG && noBump) ? selItem :
152 					   this._getUnreadItem(ZmMailListController.ACTION_CODE_WHICH[actionCode], null, noBump);
153 			if (!item) { return; }
154 			if (!mlv._expanded[item.id] && mlv._isExpandable(item)) {
155 				var callback = new AjxCallback(this, this._handleResponseExpand, [actionCode]);
156 				if (item.type == ZmItem.MSG) {
157 					this._expand({conv:appCtxt.getById(item.cid), msg:item, offset:mlv._msgOffset[item.id], callback:callback});
158 				} else {
159 					this._expand({conv:item, callback:callback});
160 				}
161 			} else if (item) {
162 				this._selectItem(mlv, item);
163 			}
164 			break;
165 
166 		// need to invoke DwtListView method directly since our list view no-ops DBLCLICK
167 		case DwtKeyMap.DBLCLICK:
168 			return DwtListView.prototype.handleKeyAction.apply(mlv, arguments);
169 
170 		default:
171 			return ZmDoublePaneController.prototype.handleKeyAction.call(this, actionCode);
172 	}
173 	return true;
174 };

176 ZmConvListController.prototype._handleResponseExpand =
177 function(actionCode) {
178 	var unreadItem = this._getUnreadItem(ZmMailListController.ACTION_CODE_WHICH[actionCode], ZmItem.MSG);
179 	if (unreadItem) {
180 		this._selectItem(this._mailListView, unreadItem);
181 	}

184 /**
185  * Override to handle paging among msgs within an expanded conv.
186  * 
187  * TODO: handle msg paging (current item is expandable msg)
188  * 
189  * @private
190  */
191 ZmConvListController.prototype.pageItemSilently =
192 function(currentItem, forward) {
193 	if (!currentItem) { return; }
194 	if (currentItem.type == ZmItem.CONV) {
195 		ZmMailListController.prototype.pageItemSilently.apply(this, arguments);
196 		return;
197 	}
198 	
199 	var conv = appCtxt.getById(currentItem.cid);
200 	if (!(conv && conv.msgs)) { return; }
201 	var found = false;
202 	var list = conv.msgs.getArray();
203 	for (var i = 0, count = list.length; i < count; i++) {
204 		if (list[i] == currentItem) {
205 			found = true;
206 			break;
207 		}
208 	}
209 	if (!found) { return; }
210 	
211 	var msgIdx = forward ? i + 1 : i - 1;
212 	if (msgIdx >= 0 && msgIdx < list.length) {
213 		var msg = list[msgIdx];
214 		var clv = this._listView[this._currentView];
215 		clv.emulateDblClick(msg);
216 	}
217 };

278 ZmConvListController.prototype._listSelectionListener =
279 function(ev) {
280 	var item = ev.item;
281 	if (!item) { return; }
282 	if (ev.field == ZmItem.F_EXPAND && this._mailListView._isExpandable(item)) {
283 		this._toggle(item, false);
284 	} else {
285 		var handled = ZmDoublePaneController.prototype._listSelectionListener.apply(this, arguments);
286 		if (!handled) {
287 			if (ev.detail == DwtListView.ITEM_DBL_CLICKED) {
288 				var respCallback = new AjxCallback(this, this._handleResponseListSelectionListener, item);
289 				if (item.type == ZmItem.MSG) {
290 					AjxDispatcher.run("GetMsgController", item && item.nId).show(item, this._msgControllerMode, respCallback, true);
291 				} else {
292 					AjxDispatcher.run("GetConvController").show(this._activeSearch, item, this, respCallback, true);
293 				}
294 			}
295 		}
296 	}
297 };

307 /**
308  * Returns the first matching msg in the conv, if available. No request will
309  * be made to the server if the conv has not been loaded.
310  */
311 ZmConvListController.prototype.getMsg =
312 function(params) {
313 	var sel = this._listView[this._currentView].getSelection();
314 	var item = (sel && sel.length) ? sel[0] : null;
315 	if (item) {
316 		if (item.type == ZmItem.CONV) {
317 			return item.getFirstHotMsg(params);
318 		} else if (item.type == ZmItem.MSG) {
319 			return ZmDoublePaneController.prototype.getMsg.apply(this, arguments);
320 		}
321 	}
322 	return null;
323 };

325 /**
326  * Returns the first matching msg in the conv. The conv will be loaded if
327  * necessary.
328  */
329 ZmConvListController.prototype._getLoadedMsg =
330 function(params, callback) {
331 	params = params || {};
332 	var sel = this._listView[this._currentView].getSelection();

339 		} else if (item.type == ZmItem.MSG) {
340 			ZmDoublePaneController.prototype._getLoadedMsg.apply(this, arguments);
341 		}
342 	} else {
343 		callback.run();
344 	}

360 ZmConvListController.prototype._toggle =
361 function(item, getFirstMsg) {
362 	if (this._mailListView._expanded[item.id]) {
363 		this._collapse(item);
364 	} else {
365 		var conv = item, msg = null, offset = 0;
366 		if (item.type == ZmItem.MSG) {
367 			conv = appCtxt.getById(item.cid);
368 			msg = item;
369 			offset = this._mailListView._msgOffset[item.id];
370 		}
371 		this._expand({conv:conv, msg:msg, offset:offset, getFirstMsg:getFirstMsg});
372 	}
373 };

375 /**
376  * Expands the given conv or msg, performing a search to get items if necessary.
377  *
378  * @param params		[hash]			hash of params:
379  *        conv			[ZmConv]		conv to expand
380  *        msg			[ZmMailMsg]		msg to expand (get next page of msgs for conv)
381  *        offset		[int]			index of msg in conv
382  *        getFirstMsg	[boolean]		if true, fetch body of first msg
383  *        callback		[AjxCallback]	callback to run when done
384  */
385 ZmConvListController.prototype._expand =
386 function(params) {
387 
388 	var conv = params.conv;
389 	var offset = params.offset || 0;
390 	var respCallback = new AjxCallback(this, this._handleResponseLoadItem, [params]);
391 	var pageWasCached = false;
392 	if (offset) {
393 		if (this._paginateConv(conv, offset, respCallback)) {
394 			// page was cached, callback won't be run
395 			this._handleResponseLoadItem(params, new ZmCsfeResult(conv.msgs));
396 		}
397 	} else if (!conv._loaded) {
398 		// no msgs have been loaded yet
399 		var getFirstMsg = (params.getFirstMsg === false) ? false : this.isReadingPaneOn();
400 		conv.load({getFirstMsg:getFirstMsg}, respCallback);
401 	} else {
402 		// re-expanding first page of msgs
403 		this._handleResponseLoadItem(params, new ZmCsfeResult(conv.msgs));
404 	}
405 };

417 /**
418  * Adapted from ZmListController::_paginate
419  */
420 ZmConvListController.prototype._paginateConv =
421 function(conv, offset, callback) {
422 	var list = conv.msgs;
423 	// see if we're out of msgs and the server has more
424 	var limit = appCtxt.get(ZmSetting.CONVERSATION_PAGE_SIZE);
425 	if (offset && list && ((offset + limit > list.size()) && list.hasMore())) {
426 		// figure out how many items we need to fetch
427 		var delta = (offset + limit) - list.size();
428 		var max = delta < limit && delta > 0 ? delta : limit;
429 		if (max < limit) {
430 			offset = ((offset + limit) - max) + 1;
431 		}
432 		var respCallback = new AjxCallback(this, this._handleResponsePaginateConv, [conv, offset, callback]);
433 		var getFirstMsg = this.isReadingPaneOn();
434 		conv.load({offset:offset, limit:limit, getFirstMsg:getFirstMsg}, respCallback);
435 		return false;
436 	} else {
437 		return true;
438 	}
439 };
440 
441 ZmConvListController.prototype._handleResponsePaginateConv =
442 function(conv, offset, callback, result) {
443 	if (!conv.msgs) { return; }
444 	var searchResult = result.getResponse();
445 	conv.msgs.setHasMore(searchResult.getAttribute("more"));
446 	var newList = searchResult.getResults(ZmItem.MSG).getVector();
447 	conv.msgs.cache(offset, newList);
448 	if (callback) {
449 		callback.run(result);
450 	}
451 };

463 // Actions
464 //
465 // Since a selection might contain both convs and msgs, we need to split them up and
466 // invoke the action for each type separately.
467 
468 /**
469  * Takes the given list of items (convs and msgs) and splits it into one list of each
470  * type. Since an action applied to a conv is also applied to its msgs, we remove any
471  * msgs whose owning conv is also in the list.
472  */
473 ZmConvListController.prototype._divvyItems =
474 function(items) {
475 	var convs = [], msgs = [];
476 	var convIds = {};
477 	for (var i = 0; i < items.length; i++) {
478 		var item = items[i];
479 		if (item.type == ZmItem.CONV) {
480 			convs.push(item);
481 			convIds[item.id] = true;
482 		} else {
483 			msgs.push(item);
484 		}
485 	}
486 	var msgs1 = [];
487 	for (var i = 0; i < msgs.length; i++) {
488 		if (!convIds[msgs[i].cid]) {
489 			msgs1.push(msgs[i]);
490 		}
491 	}
492 	var lists = {};
493 	lists[ZmItem.MSG] = msgs1;	
494 	lists[ZmItem.CONV] = convs;
495 	
496 	return lists;
497 };

516 			}
517 		}
518 	}

535 /**
536  * Splits the given items into two lists, one of convs and one of msgs, and
537  * applies the given method and args to each.
538  *
539  * @param items		[array]			list of convs and/or msgs
540  * @param method	[string]		name of function to call in parent class
541  * @param args		[array]			additional args to pass to function
542  */
543 ZmConvListController.prototype._applyAction =
544 function(items, method, args) {
545 	args = args ? args : [];
546 	var lists = this._divvyItems(items);
547 	var hasMsgs = false;
548 	if (lists[ZmItem.MSG] && lists[ZmItem.MSG].length) {
549 		args.unshift(lists[ZmItem.MSG]);
550 		ZmDoublePaneController.prototype[method].apply(this, args);
551 		hasMsgs = true;
552 	}
553 	if (lists[ZmItem.CONV] && lists[ZmItem.CONV].length) {
554 		hasMsgs ? args[0] = lists[ZmItem.CONV] : args.unshift(lists[ZmItem.CONV])
555 		ZmDoublePaneController.prototype[method].apply(this, args);
556 	}
557 };
