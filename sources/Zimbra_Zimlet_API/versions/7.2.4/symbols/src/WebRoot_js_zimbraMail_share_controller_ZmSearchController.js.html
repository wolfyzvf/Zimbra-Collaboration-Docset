 22 /**
 23  * Creates a search controller.
 24  * @class
 25  * This class represents the search controller.
 26  * 
 27  * @param {DwtControl}	container	the top-level container
 28  * @extends	ZmController
 29  */
 30 ZmSearchController = function(container) {
 31 
 32 	ZmController.call(this, container);
 33 
 34 	this._inited = false;
 35 
 36 	// default menu values
 37 	this._searchFor = appCtxt.get(ZmSetting.MAIL_ENABLED) ? ZmId.SEARCH_MAIL : ZmId.SEARCH_ANY;
 38 	this._contactSource = ZmItem.CONTACT;
 39 	this._results = null;
 40 
 41 	if (appCtxt.get(ZmSetting.SEARCH_ENABLED)) {
 42 		this._setView();
 43 	}
 44 };

 72 /**
 73  * Performs a search by date.
 74  * 
 75  * @param	{Date}	d		the date or <code>d</code> for now
 76  * @param	{String}	searchFor	the search for string
 77  */
 78 ZmSearchController.prototype.dateSearch =
 79 function(d, searchFor) {
 80 	d = d || new Date();
 81     var formatter = AjxDateFormat.getDateInstance(AjxDateFormat.SHORT);
 82     var date = formatter.format(d);
 83 	var groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
 84 	var query = "date:" + date;
 85 	this.search({query:query, types:[groupBy], searchFor: searchFor});
 86 };

 88 /**
 89  * Performs a search by from address.
 90  * 
 91  * @param	{String}	address		the from address
 92  */
 93 ZmSearchController.prototype.fromSearch =
 94 function(address) {
 95 	// always search for mail when doing a "from: <address>" search
 96 	var groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
 97 	var query = address instanceof Array ? address.concat() : [ address ];
 98 	for (var i = 0; i < query.length; i++) {
 99 		query[i] = ["from:(", query[i], ")"].join("");
100 	}
101 
102     this.search({query:query.join(" OR "), types:[groupBy]});
103 };

106 /**
107  * Performs a search by to address.
108  *
109  * @param	{String}	address		the to address
110  */
111 ZmSearchController.prototype.toSearch =
112 function(address) {
113 	// always search for mail when doing a "tocc: <address>" search
114 	var groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
115 	var query = address instanceof Array ? address.concat() : [ address ];
116 	for (var i = 0; i < query.length; i++) {
117 		query[i] = ["tocc:(", query[i], ")"].join("");
118 	}
119     if (this.currentSearch && this.currentSearch.folderId == ZmFolder.ID_SENT)
120         this.search({query:"in:sent AND (" + query.join(" OR ") + ")", types:[groupBy]});
121     else
122 	    this.search({query:query.join(" OR "), types:[groupBy]});
123 };

176 /**
177  * Shows or hides the Advanced Search panel, which contains various pickers.
178  * Since it may require loading the "Browse" package, callers should use a 
179  * callback to run subsequent code. By default, the display of the panel is
180  * toggled.
181  * 
182  * @param {Boolean}	forceShow		if <code>true</code>, show panel
183  * @param {Boolean}	forceHide		if <code>true</code>, hide panel (takes precenence over forceShow)
184  * @param {AjxCallback}	callback		the callback to run after display is done
185  */
186 ZmSearchController.prototype.showBrowseView =
187 function(forceShow, callback, forceHide) {
188 	if (!this._browseViewController) {
189 		if (forceHide) {
190 			return; //it was never displayed yet anyway so no need to hide
191 		}
192 		var loadCallback = new AjxCallback(this, this._handleLoadShowBrowseView, [forceShow, forceHide, callback]);
193 		AjxDispatcher.require("Browse", false, loadCallback, null, false);
194 		return;
195 	}
196 	this._doShowBrowseView(forceShow, forceHide, callback);
197 };

203 function(forceShow, forceHide, callback) {
204 	this._browseViewController = new ZmBrowseController(this.searchPanel);
205 	this._doShowBrowseView(forceShow, forceHide, callback);
206 };

208 /**
209  * @private
210  */
211 ZmSearchController.prototype._doShowBrowseView =
212 function(forceShow, forceHide, callback) {
213 	var bvc = this._browseViewController;
214 	bvc.setBrowseViewVisible(!forceHide && (forceShow || !bvc.getBrowseViewVisible()));
215 	if (callback) {
216 		callback.run(bvc.getBrowseView());
217 	}
218 };

279 function() {

288 	}
289 
290 	this._createTabGroup();
291 	this._tabGroup.addMember(this._searchToolBar.getSearchField());
292 	var buttons = this._searchToolBar.getButtons();
293 	for (var i = 0; i < buttons.length; i++) {
294 		this._tabGroup.addMember(buttons[i]);
295 	}
296 
297 	// Register keyboard callback for search field
298 	this._searchToolBar.registerCallback(this._searchFieldCallback, this);
299 
300 	// Button listeners
301 	this._searchToolBar.addSelectionListener(ZmSearchToolBar.SEARCH_BUTTON, new AjxListener(this, this._searchButtonListener));
302 	if (appCtxt.get(ZmSetting.BROWSE_ENABLED)) {
303 		this._searchToolBar.addSelectionListener(ZmSearchToolBar.BROWSE_BUTTON, new AjxListener(this, this._browseButtonListener));

306 		this._searchToolBar.addSelectionListener(ZmSearchToolBar.SAVE_BUTTON, new AjxListener(this, this._saveButtonListener));
307 	}

329 /**
330  * Performs a search and displays the results.
331  *
332  * @param {Hash}	params		a hash of parameters
333  * @param {String}	params.query	the search string
334  * @param {constant}	params.searchFor	the semantic type to search for
335  * @param {Array}	params.types		the item types to search for
336  * @param {constant}	params.sortBy		the sort constraint
337  * @param {int}	params.offset	the starting point in list of matching items
338  * @param {int}	params.limit	the maximum number of items to return
339  * @param {int}	params.searchId	the ID of owning search folder (if any)
340  * @param {int}	params.prevId	the ID of last items displayed (for pagination)
341  * @param {constant}	params.prevSortBy	previous sort order (for pagination)
342  * @param {Boolean}	params.noRender	if <code>true</code>, results will not be passed to controller
343  * @param {Boolean}	params.noClear	if <code>true</code>, previous results will not be destroyed
344  * @param {Boolean}	params.userText	if <code>true</code>, text was typed by user into search box
345  * @param {AjxCallback}	params.callback		the async callback
346  * @param {AjxCallback}	params.errorCallback	the async callback to run if there is an exception
347  * @param	{Object}	params.response		the canned JSON response (no request will be made)
348  * 
349  */
350 ZmSearchController.prototype.search =
351 function(params) {
352 
353 	if (!params.query && !params.queryHint) { // What to do when the search field is empty?
354 		var appName;
355 		switch (params.searchFor) {
356 			case ZmId.SEARCH_MAIL:
357 				appName = ZmApp.MAIL;
358 				break;
359 			case ZmId.SEARCH_GAL:
360 				// Do not search in GAL when query is empty
361 				return;
362 			case ZmId.SEARCH_ANY:
363 				params.query = "is:anywhere";
364 				break;
365 			case ZmItem.APPT:
366 				break;
367 			default:
368 				// Get the app of the item type being searched
369 				appName = ZmItem.APP[params.searchFor];
370 				break;
371 		}
372 		if (appName) {
373 			// Get the "main" folder of the app related to the searched item type
374 			var organizerName = ZmOrganizer.APP2ORGANIZER[appName];
375 			var defaultFolder = organizerName && ZmOrganizer.DEFAULT_FOLDER[organizerName];
376 			var folder = defaultFolder && appCtxt.getById(defaultFolder);
377 			if (folder) {
378 				params.query = "in:" + folder._systemName;
379 			}
380 		}
381 		if (params.query) {
382 			params.userText = false;
383 		} else if (params.searchFor != ZmItem.APPT) { // Appointment searches without query are ok, all others should fail
384 			return;
385 		}
386 	}
387 
388 	// if the search string starts with "$set:" then it is a command to the client
389 	if (params.query && (params.query.indexOf("$set:") == 0 || params.query.indexOf("$cmd:") == 0)) {
390 		appCtxt.getClientCmdHandler().execute((params.query.substr(5)), this);
391 		return;
392 	}
393 
394 	params.searchAllAccounts = this.searchAllAccounts;
395 	var respCallback = new AjxCallback(this, this._handleResponseSearch, [params.callback]);
396 	this._doSearch(params, params.noRender, respCallback, params.errorCallback);
397 };

405 		callback.run(result);
406 	}
407 };

409 /**
410  * Performs the given search. It takes a ZmSearch, rather than constructing one out of the currently selected menu
411  * choices. Aside from re-executing a search, it can be used to perform a canned search.
412  *
413  * @param {ZmSearch}	search		the search object
414  * @param {Boolean}		noRender		if <code>true</code>, results will not be passed to controller
415  * @param {Object}	changes		the hash of changes to make to search
416  * @param {AjxCallback}	callback		the async callback
417  * @param {AjxCallback}	errorCallback	the async callback to run if there is an exception
418  */
419 ZmSearchController.prototype.redoSearch =
420 function(search, noRender, changes, callback, errorCallback) {
421 
422 	var params = {};
423 	params.query		= search.query;
424 	params.queryHint	= search.queryHint;
425 	params.types		= search.types;
426 	params.sortBy		= search.sortBy;
427 	params.offset		= search.offset;
428 	params.limit		= search.limit;
429 	params.prevId		= search.prevId;
430 	params.prevSortBy	= search.prevSortBy;
431 	params.fetch		= search.fetch;
432 	params.searchId		= search.searchId;
433 	params.lastSortVal	= search.lastSortVal;
434 	params.endSortVal	= search.endSortVal;
435 	params.lastId		= search.lastId;
436 	params.soapInfo		= search.soapInfo;
437 	params.accountName	= search.accountName;
438 	params.searchFor	= this._searchFor;
439 	params.idsOnly		= search.idsOnly;
440 	params.inDumpster   = search.inDumpster;
441 
442 	if (changes) {
443 		for (var key in changes) {
444 			params[key] = changes[key];
445 		}
446 	}
447 
448 	this._doSearch(params, noRender, callback, errorCallback);
449 };

487 /**
488  * Gets the item types. Assembles a list of item types to return based on a search menu value (which can
489  * be passed in).
490  *
491  * @param {Hash}	params			a hash of arguments for the search
492  * @param {String}	params.searchFor	the string for.
493  * @return	{AjxVector}		a list of types
494  * 
495  * @see		#search
496  */
497 ZmSearchController.prototype.getTypes =
498 function(params) {
499 	var types = new AjxVector();
500 	var searchFor = params.searchFor || this._searchFor;
501 
502 	var groupBy;
503 	if ((searchFor == ZmId.SEARCH_MAIL || searchFor == ZmId.SEARCH_ANY) &&
504 		appCtxt.get(ZmSetting.MAIL_ENABLED))
505 	{
506 		groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
507 	}
508 
509 	if (searchFor == ZmId.SEARCH_MAIL) {
510 		types.add(groupBy);
511 	} else if (searchFor == ZmId.SEARCH_ANY) {
512 		if (appCtxt.get(ZmSetting.MAIL_ENABLED) && groupBy)	{ types.add(groupBy); }
513 		if (appCtxt.get(ZmSetting.CONTACTS_ENABLED))		{ types.add(ZmItem.CONTACT); }
514 		if (appCtxt.get(ZmSetting.CALENDAR_ENABLED))		{ types.add(ZmItem.APPT); }
515 		if (appCtxt.get(ZmSetting.TASKS_ENABLED))			{ types.add(ZmItem.TASK); }
516 		if (appCtxt.get(ZmSetting.NOTEBOOK_ENABLED))		{ types.add(ZmItem.PAGE); }
517 		if (appCtxt.get(ZmSetting.BRIEFCASE_ENABLED))		{ types.add(ZmItem.BRIEFCASE_ITEM); }
518 	} else {
519 		types.add(searchFor);
520 		if (searchFor == ZmItem.PAGE) {
521 			types.add(ZmItem.DOCUMENT);
522 		}
523 	}
524 
525 	return types;
526 };

528 /**
529  * Selects the appropriate item in the overview based on the search. Selection only happens
530  * if the search was a simple search for a folder, tag, or saved search. A check is done to
531  * make sure that item is not already selected, so selection should only occur for a query
532  * manually run by the user.
533  *
534  * @param {ZmSearch}	searchObj		the current search
535  */
536 ZmSearchController.prototype.updateOverview =
537 function(searchObj) {
538 	var search = searchObj || appCtxt.getCurrentSearch();
539 
540 	var id, type;
541 	if (search && (search.singleTerm || search.searchId)) {
542 		if (search.searchId) {
543 			id = this._getNormalizedId(search.searchId);
544 			type = ZmOrganizer.SEARCH;
545 		} else if (search.folderId) {
546 			id = this._getNormalizedId(search.folderId);
547 			var folderTree = appCtxt.getFolderTree();
548 			var folder = folderTree && folderTree.getById(id);
549 			type = folder ? folder.type : ZmOrganizer.FOLDER;
550             if (search.searchFor == ZmItem.TASK) {
551                 type = ZmOrganizer.TASKS;
552             }
553 		} else if (search.tagId) {
554 			id = this._getNormalizedId(search.tagId);
555 			type = ZmOrganizer.TAG;
556 		}
557 		var app = appCtxt.getCurrentApp();
558 		var overview = app && app.getOverview();
559 		if (overview) {
560 			overview.setSelected(id, type);
561 		}
562 	}
563 };

565 /**
566  * @private
567  */
568 ZmSearchController.prototype._getSuitableSortBy =
569 function(types) {
570 	var sortBy;
571 
572 	if (types.size() == 1) {
573 		var type = types.get(0);
574 		var viewType;
575 		switch (type) {
576 			case ZmItem.CONV:		viewType = ZmId.VIEW_CONVLIST; break;
577 			case ZmItem.MSG:		viewType = ZmId.VIEW_TRAD; break;
578 			case ZmItem.CONTACT:	viewType = ZmId.VIEW_CONTACT_SIMPLE; break;
579 			case ZmItem.APPT:		viewType = ZmId.VIEW_CAL; break;
580 			case ZmItem.TASK:		viewType = ZmId.VIEW_TASKLIST; break;
581 			case ZmId.SEARCH_GAL:	viewType = ZmId.VIEW_CONTACT_SIMPLE; break;
582 			case ZmItem.BRIEFCASE_ITEM:	viewType = ZmId.VIEW_BRIEFCASE_DETAIL; break;
583 			// more types go here as they are suported...
584 		}
585 
586 		if (viewType) {
587 			sortBy = appCtxt.get(ZmSetting.SORTING_PREF, viewType);
588 		}
589 	}
590 
591 	return sortBy;
592 };

594 /**
595  * Performs the search.
596  *
597  * @param {Hash}	params		a hash of params for the search
598  * @param {String}	params.searchFor	the search for
599  * @param {String}	params.query	the search query
600  * @param {String}	params.userText	the user text
601  * @param {Array}	params.type		an array of types
602  * @param {boolean} params.forceSearch     Ignores special processing and just executes the search.
603  * @param {Boolean}	noRender		if <code>true</code>, the search results will not be rendered
604  * @param {AjxCallback}	callback		the callback
605  * @param {AjxCallback}	errorCallback	the error callback
606  * 
607  * @see	#search
608  * 
609  * @private
610  */
611 ZmSearchController.prototype._doSearch =
612 function(params, noRender, callback, errorCallback) {
613 
614 	var searchFor = this._searchFor = params.searchFor || this._searchFor;
615 	appCtxt.notifyZimlets("onSearch", [params.query]);
616 
617 	if (this._searchToolBar) {
618 		var value = (appCtxt.get(ZmSetting.SHOW_SEARCH_STRING) || params.userText)
619 			? params.query : null;
620 		this._searchToolBar.setSearchFieldValue(value || "");
621 
622 		// bug: 42512 - deselect global inbox if searching via search toolbar
623 		if (appCtxt.multiAccounts && params.userText && this.searchAllAccounts) {
624 			appCtxt.getCurrentApp().getOverviewContainer().deselectAll();
625 		}
626 	}
627 
628 	// get types from search type if not passed in explicitly
629 	var types = params.types || this.getTypes(params);
630 	if (types instanceof Array) { // convert array to AjxVector if necessary
631 		types = AjxVector.fromArray(types);
632 	}
633 	if (searchFor == ZmId.SEARCH_MAIL) {
634 		params = appCtxt.getApp(ZmApp.MAIL).getSearchParams(params);
635 	}
636 
637 	if (searchFor == ZmItem.TASK) {
638 		var tlc = AjxDispatcher.run("GetTaskListController");
639 		params.allowableTaskStatus = (tlc) ? tlc.getAllowableTaskStatus() : null;
640 	}
641 
642 	// if the user explicitly searched for all types, force mixed view
643 	var isMixed = (searchFor == ZmId.SEARCH_ANY);
644 
645 	if (params.searchAllAccounts && !params.queryHint) {
646 		params.queryHint = appCtxt.accountList.generateQuery(null, types);
647 		params.accountName = appCtxt.accountList.mainAccount.name;
648 	}
649 	else if (this._inclSharedItems) {
650 		// a query hint is part of the query that the user does not see
651 		params.queryHint = isMixed
652 			? ZmSearchController.QUERY_ISREMOTE
653 			: ZmSearchController.generateQueryForShares(types.getArray());
654 	}
655 
656 	// only set contact source if we are searching for contacts
657 	params.contactSource = (types.contains(ZmItem.CONTACT) || types.contains(ZmId.SEARCH_GAL))
658 		? this._contactSource : null;
659 	if (params.contactSource == ZmId.SEARCH_GAL) {
660 		params.expandDL = true;
661 	}
662 
663 	// find suitable sort by value if not given one (and if applicable)
664 	params.sortBy = params.sortBy || this._getSuitableSortBy(types);
665 	params.types = types;
666 	var search = new ZmSearch(params);
667 
668 	var args = [search, noRender, isMixed, callback, params.noUpdateOverview, params.noClear];
669 	var respCallback = new AjxCallback(this, this._handleResponseDoSearch, args);
670 	if (!errorCallback) {
671 		errorCallback = new AjxCallback(this, this._handleErrorDoSearch, [search, isMixed]);
672 	}
673 
674 	// calendar searching is special so hand it off if necessary
675 	if (searchFor == ZmItem.APPT && !params.forceSearch) {
676 		var controller = AjxDispatcher.run("GetCalController");
677 		if (controller && types.contains(ZmItem.APPT)) {
678 			controller.handleUserSearch(params, respCallback);
679 		} else {
680             search.execute({callback:respCallback, errorCallback:errorCallback});            
681         }
682 	} else {
683 		search.execute({callback:respCallback, errorCallback:errorCallback});
684 	}
685 };

687 /**
688  * Takes the search result and hands it to the appropriate controller for display.
689  *
690  * @param {ZmSearch}	search			contains search info used to run search against server
691  * @param {Boolean}		noRender		<code>true</code> to skip rendering results
692  * @param {Boolean}	isMixed				<code>true</code> if in mixed mode
693  * @param {AjxCallback}	callback		the callback to run after processing search response
694  * @param {Boolean}	noUpdateOverview	<code>true</code> to skip updating the overview
695  * @param {Boolean}		noClear			if <code>true</code>, previous results will not be destructed
696  * @param {ZmCsfeResult}	result			the search results
697  */
698 ZmSearchController.prototype._handleResponseDoSearch =
699 function(search, noRender, isMixed, callback, noUpdateOverview, noClear, result) {
700 
701 	var results = result && result.getResponse();
702 	if (!results) { return; }
703 
704 	if (!results.type) {
705 		results.type = search.types.get(0);
706 	}
707 
708 	if (results.type == ZmItem.APPT) {
709 		this._results = new ZmSearchResult(search);
710 	} else {
711 		this.currentSearch = search;
712 		DBG.timePt("execute search", true);
713 
714 		// bug fix #34776 - don't show search results if user is in the composer
715 		if (!noRender) {
716 			this._showResults(results, search, isMixed, noUpdateOverview, noClear);
717 		}
718 	}
719 
720 	if (callback) {
721 		callback.run(result);
722 	}
723 };

725 /**
726  * @private
727  */
728 ZmSearchController.prototype._showResults =
729 function(results, search, isMixed, noUpdateOverview, noClear) {
730 	// allow old results to dtor itself
731 	if (!noClear && this._results && (this._results.type == results.type) && this._results.dtor) {
732 		this._results.dtor();
733 	}
734 	this._results = results;
735 
736 	DBG.timePt("handle search results");
737 
738 	// determine if we need to default to mixed view
739 	if (appCtxt.get(ZmSetting.SAVED_SEARCHES_ENABLED)) {
740 		var saveBtn = this._searchToolBar && this._searchToolBar.getButton(ZmSearchToolBar.SAVE_BUTTON);
741 		if (saveBtn) {
742 			saveBtn.setEnabled(this._contactSource != ZmId.SEARCH_GAL);
743 		}
744 	}
745 
746 	// show results based on type - may invoke package load
747 	var resultsType = isMixed ? ZmItem.MIXED : results.type;
748 	var loadCallback = new AjxCallback(this, this._handleLoadShowResults, [results, search, noUpdateOverview]);
749 	var app = appCtxt.getApp(ZmItem.APP[resultsType]) || appCtxt.getCurrentApp();
750 	app.currentSearch = search;
751 	app.currentQuery = search.query;
752 	app.showSearchResults(results, loadCallback);
753 };

755 /**
756  * @private
757  */
758 ZmSearchController.prototype._handleLoadShowResults =
759 function(results, search, noUpdateOverview) {
760 	appCtxt.setCurrentList(results.getResults(results.type));
761 	if (!noUpdateOverview) {
762 		this.updateOverview(search);
763 	}
764 	DBG.timePt("render search results");
765 };

767 /**
768  * Handle a few minor errors where we show an empty result set and issue a
769  * status message to indicate why the query failed. Those errors are: no such
770  * folder, no such tag, and bad query. If it's a "no such folder" error caused
771  * by the deletion of a folder backing a mountpoint, we pass it along for
772  * special handling by ZmZimbraMail.
773  * 
774  * @private
775  */
776 ZmSearchController.prototype._handleErrorDoSearch =
777 function(search, isMixed, ex) {
778 	DBG.println(AjxDebug.DBG1, "Search exception: " + ex.code);
779 
780 	if (ex.code == ZmCsfeException.MAIL_NO_SUCH_TAG ||
781 		ex.code == ZmCsfeException.MAIL_QUERY_PARSE_ERROR ||
782 		ex.code == ZmCsfeException.MAIL_TOO_MANY_TERMS ||
783 		(ex.code == ZmCsfeException.MAIL_NO_SUCH_FOLDER && !(ex.data.itemId && ex.data.itemId.length)))
784 	{
785 		var msg = ex.getErrorMsg();
786 		appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_WARNING);
787 		var results = new ZmSearchResult(search);
788 		results.type = search.types ? search.types.get(0) : null;
789 		this._showResults(results, search, isMixed);
790 		return true;
791 	}
792 	return false;
793 };

795 /**
796  * Provides a string to add to the query when the search includes shared items.
797  * 
798  * @param types		[array]		list of item types
799  * 
800  * @private
801  */
802 ZmSearchController.generateQueryForShares =
803 function(types, account) {
804 	var ac = window.parentAppCtxt || window.appCtxt;
805 	var list = [];
806 	for (var j = 0; j < types.length; j++) {
807 		var type = types[j];
808 		var app = ac.getApp(ZmItem.APP[type]);
809 		if (app) {
810 			var ids = app.getRemoteFolderIds(account);
811 			for (var i = 0; i < ids.length; i++) {
812 				var id = ids[i];
813 				var idText = AjxUtil.isNumeric(id) ? id : ['"', id, '"'].join("");
814 				list.push("inid:" + idText);
815 			}
816 		}
817 	}
818 
819 	if (list.length > 0) {
820 		list.push("is:local");
821 		return list.join(" OR ");
822 	}
823 
824 	return null;
825 };

892 /**
893  * @private
894  */
895 ZmSearchController.prototype._searchMenuListener =
896 function(ev, id) {
897 	var btn = this._searchToolBar.getButton(ZmSearchToolBar.SEARCH_MENU_BUTTON);
898 	if (!btn) { return; }
899 
900 	var menu = btn.getMenu();
901 	var item = ev ? ev.item : (menu.getItemById(ZmSearchToolBar.MENUITEM_ID, id));
902 
903 	if (!item || (!!(item._style & DwtMenuItem.SEPARATOR_STYLE))) { return; }
904 	id = item.getData(ZmSearchToolBar.MENUITEM_ID);
905 
906 	var selItem = menu.getSelectedItem();
907 	var sharedMI = menu.getItemById(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_SHARED);
908 
909 	// enable shared menu item if not a gal search
910 	if (id == ZmId.SEARCH_GAL) {
911 		this._contactSource = ZmId.SEARCH_GAL;
912 		if (sharedMI) {
913 			sharedMI.setChecked(false, true);
914 			sharedMI.setEnabled(false);
915 		}
916 	} else {
917 		if (sharedMI) {
918 			// we allow user to check "Shared Items" for appointments since it
919 			// is based on whats checked in their tree view
920 			if (id == ZmItem.APPT || id == ZmId.SEARCH_CUSTOM) {
921 				sharedMI.setChecked(false, true);
922 				sharedMI.setEnabled(false);
923 			} else {
924 				sharedMI.setEnabled(true);
925 			}
926 		}
927 		this._contactSource = ZmItem.CONTACT;
928 	}
929 	this._inclSharedItems = sharedMI && sharedMI.getChecked();
930 
931 	// search all accounts? Only applies to multi-account mbox
932 	var allAccountsMI = menu.getItemById(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_ALL_ACCOUNTS);
933 	if (allAccountsMI) {
934 		if (id == ZmItem.APPT) {
935 			this.resetSearchAllAccounts();
936 			allAccountsMI.setEnabled(false);
937 		} else {
938 			allAccountsMI.setEnabled(true);
939 			this.searchAllAccounts = allAccountsMI && allAccountsMI.getChecked();
940 		}
941 	}
942 
943 	if (id == ZmId.SEARCH_SHARED) {
944 		var icon = this.searchAllAccounts
945 			? allAccountsMI.getImage() : selItem.getImage();

947 		if (this._inclSharedItems) {
948 			var selItemId = selItem && selItem.getData(ZmSearchToolBar.MENUITEM_ID);
949 			icon = selItemId
950 				? ((ZmSearchToolBar.SHARE_ICON[selItemId]) || item.getImage())
951 				: item.getImage();
952 		}
953 
954 		btn.setImage(icon);
955 	}
956 	else if (id == ZmId.SEARCH_ALL_ACCOUNTS) {
957 		var icon = (this.searchAllAccounts && !this._inclSharedItems)
958 			? item.getImage()
959 			: (this._inclSharedItems) ? this._getSharedImage(selItem) : selItem.getImage();
960 		btn.setImage(icon);
961 	}
962 	else {
963 		// only set search for if a "real" search-type menu item was clicked
964 		this._searchFor = id;
965 		var icon = item.getImage();
966 
967 		if (this._inclSharedItems) {
968 			icon = this._getSharedImage(selItem);
969 		}
970 		else if (this.searchAllAccounts) {
971 			icon = allAccountsMI.getImage();
972 		}
973 
974 		btn.setImage(icon);
975 		btn.setText(item.getText());
976 	}
977

979 	var tooltip = ZmMsg[ZmSearchToolBar.TT_MSG_KEY[id]];
980 	if (id == ZmId.SEARCH_MAIL) {
981 		var groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
982 		tooltip = ZmMsg[ZmSearchToolBar.TT_MSG_KEY[groupBy]];
983 	}
984 	btn.setToolTipContent(tooltip);
985 };
