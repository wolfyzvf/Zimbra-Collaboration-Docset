178 ZmAttachmentsController.prototype._initToolBar =
179 function(){
180 
181     if (this._toolbar) return;
182 
183     this._listeners[ZmOperation.RESET] = new AjxListener(this, this._reset);
184     this._listeners[ZmOperation.CANCEL] = new AjxListener(this, this._cancel);
185     //this._listeners[ZmOperation.SEND] = new AjxListener(this, this._sendMail);
186 
187     var buttons = [ZmOperation.RESET, /*ZmOperation.SEND,*/ ZmOperation.CANCEL];
188 
189 
190 	var className = appCtxt.isChildWindow ? "ZmAppToolBar_cw" : "ZmAppToolBar";
191 	this._toolbar = new ZmButtonToolBar({parent:this._container, buttons:buttons, className:className+" ImgSkin_Toolbar",
192 										 context:ZmId.VIEW_ATTACHMENTS});
193 
194 	for (var i = 0; i < this._toolbar.opList.length; i++) {
195 		var button = this._toolbar.opList[i];
196 		if (this._listeners[button]) {
197 			this._toolbar.addSelectionListener(button, this._listeners[button]);
198 		}
199 	}	
200 };

202 ZmAttachmentsController.prototype._reset = function(ev){
203      this._attachmentsView._resetFilterTrees();
204      this.search({query:this._getQuery()});
205 };
206 
207 ZmAttachmentsController.prototype._cancel = function(){
208     this._app.popView();
209 };

211 ZmAttachmentsController.prototype.initAttachmentsView = function(){
212 
213     if (this._attachmentsView) return;
214 
215     this._attachmentsView =  new ZmAttachmentsView(this._container, this);
216 	this._initToolBar();
217 
218     var elements = {};
219     elements[ZmAppViewMgr.C_TOOLBAR_TOP] = this._toolbar;
220 	elements[ZmAppViewMgr.C_APP_CONTENT] = this._attachmentsView;
221 
222     this._app.createView({viewId:ZmId.VIEW_ATTACHMENTS, elements:elements, isTransient:true});
223 
224 };

230     this._app.pushView(ZmId.VIEW_ATTACHMENTS);

232     this._reset();

236 ZmAttachmentsController.prototype.search = function(params){
237 
238     params.query  = params.query || ZmAttachmentsController.DEFAULT_QUERY;
239     params.offset = 0;
240     params.limit  = 50;
241     params.sortBy = ZmSearch.DATE_DESC;
242     params.fetch  = "all";
243     params.searchFor = ZmItem.MSG;
244 
245     var types = new AjxVector();
246     types.add(ZmItem.MSG);
247     params.types  = types;
248 
249     params.noBusyOverlay = true;
250     params.noRender = true;
251     params.callback = new AjxCallback(this, this._handleSearchResponse);
252 
253     var sc = appCtxt.getSearchController();
254     sc.search(params);
255 
256 };
257 
258 ZmAttachmentsController.prototype._handleSearchResponse = function(results){
259 
260     results = results.getResponse().getResults(ZmItem.MSG);
261 
262     this._attachmentsView.set(results);
263 
264 };

266 ZmAttachmentsController.prototype._filterTreeListener = function(ev){
267 
268     if(ev.detail == DwtTree.ITEM_SELECTED){
269         this.search({query:this._getQuery()});
270     }
271 };

274 ZmAttachmentsController.prototype._getQuery = function(){
275 
276     var query = "";
277 
278     var queryString = "";
279 
280     var byTypeItem = this._attachmentsView.getSelectedItem(ZmAttachmentsController.CATEGORY_BYTYPE);
281     if(byTypeItem){
282 
283         var subQuery = [];
284         var ct = byTypeItem.ct;
285         if(byTypeItem.id == ZmAttachmentsController.SUBCATEGORY_BYTYPE_ALL){
286             subQuery.push("any");
287         }else{
288             for(var i=0; i<ct.length; i++){
289                 subQuery.push('"'+ct[i]+'"');
290             }
291         }        
292         subQuery = subQuery.join(" OR ");
293 
294         query += byTypeItem.cmd + "("+ subQuery+")";
295 
296         queryString += '<strong>' + ZmMsg[byTypeItem.msgKey] + '</strong> attachments <br>';
297 
298     }else{
299         queryString += '<strong>' + ZmMsg['all'] + '</strong> attachments <br>';
300     }
301 
302     var byFolderItem = this._attachmentsView.getSelection(ZmAttachmentsController.CATEGORY_BYFOLDER);
303     if(byFolderItem){
304          var folder = byFolderItem.getData(Dwt.KEY_OBJECT);
305          query += ' ' + folder.createQuery();
306 
307          queryString += 'in <strong>' + folder.getName() + '</strong>'+' folder<br>';
308     }
309 
310     var byTagItem = this._attachmentsView.getSelection(ZmAttachmentsController.CATEGORY_BYTAG);
311     if(byTagItem){
312         var tag = byTagItem.getData(Dwt.KEY_OBJECT);
313         query += ' ' + ZmAttachmentsController.TAG_QUERY_CMD + '"' + tag.name + '"';
314 
315         queryString += 'with tag <strong>' + tag.getName() + '</strong><br>';
316     }
317 
318     var byTimeItem = this._attachmentsView.getSelectedItem(ZmAttachmentsController.CATEGORY_BYTIME);
319     if(byTimeItem){
320 
321         var date = new Date();
322         var offset = -1;
323         if(byTimeItem.id == ZmAttachmentsController.SUBCATEORY_BYTIME_TODAY){
324             date.setDate(date.getDate() + offset);
325         }else if(byTimeItem.id == ZmAttachmentsController.SUBCATEORY_BYTIME_WEEK){
326             date.setDate(date.getDate()+7*offset);
327         }else if(byTimeItem.id == ZmAttachmentsController.SUBCATEORY_BYTIME_MONTH){
328             d = date.getDate();
329             date.setDate(1);
330             date.setMonth(date.getMonth() + offset);
331             var max = AjxDateUtil.daysInMonth(date.getFullYear(), date.getMonth());
332 		    date.setDate(Math.min(d, max));
333         }else if(byTimeItem.id == ZmAttachmentsController.SUBCATEORY_BYTIME_YEAR){
334             d = date.getDate();
335             date.setDate(1);
336             date.setFullYear(date.getFullYear() + offset);
337 		    var max = AjxDateUtil.daysInMonth(date.getFullYear(), date.getMonth());
338 		    date.setDate(Math.min(d, max));
339         }
340 
341         var formatter = AjxDateFormat.getDateInstance(AjxDateFormat.SHORT);
342         var afterDate = formatter.format(date);
343 
344         if(byTimeItem.cmd) {
345             query += ' ' + byTimeItem.cmd + '"' +  afterDate  +'"';
346         }
347 
348         queryString += 'during <strong>' + ZmMsg[byTimeItem.msgKey] + '</strong><br>';
349 
350     }
351 
352     //if(queryString == "") queryString = '<strong>' + ZmMsg['all'] + '</strong> attachments<br>';
353 
354     this._attachmentsView.setQueryString(queryString);
355 
356     return (query != "") ? query : ZmAttachmentsController.DEFAULT_QUERY;
357 };
