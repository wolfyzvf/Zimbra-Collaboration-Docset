 16 ZmCalendarAssistant = function() {
 17 	ZmAssistant.call(this, ZmMsg.openCalendar, ZmMsg.ASST_CMD_CALENDAR, ZmMsg.ASST_CMD_SUM_CALENDAR);
 18 };
 19 
 20 ZmCalendarAssistant.prototype = new ZmAssistant();
 21 ZmCalendarAssistant.prototype.constructor = ZmCalendarAssistant;
 22 
 23 ZmCalendarAssistant.prototype.okHandler =
 24 function(dialog) {
 25     var calMgr = appCtxt.getCalManager();
 26     calMgr.getMiniCalendar();
 27     var cc = calMgr.getCalViewController();
 28 	cc.setDate(this._startDate);
 29 	// need to call twice due to cal view controller bug
 30 	cc.show(this._view);
 31 	cc.show(this._view);	
 32 	return true;
 33 };
 34 
 35 ZmCalendarAssistant.prototype.getHelp =
 36 function() {
 37 	return ZmMsg.ASST_CALENDAR_HELP;
 38 };
 39 
 40 ZmCalendarAssistant.prototype.handle =
 41 function(dialog, verb, args) {
 42 	
 43 	this._startDate = null;
 44 	// look for start date
 45 	var match = this._objectManager.findMatch(args, ZmObjectManager.DATE);
 46 	if (match) {
 47 		args = args.replace(match[0], " ");
 48 		this._startDate = match.context.date;
 49 	}
 50 
 51 	match = args.match(RegExp("\\b("
 52 							  +ZmMsg.viewCalForDay+"|"
 53 							  +ZmMsg.viewCalForWeek+"|"
 54 							  +ZmMsg.viewCalForWorkWeek+"|"
 55 							  +ZmMsg.viewCalForList+"|"
 56 							  +ZmMsg.viewCalForMonth+")\\b"));
 57 	var view = (match) ? match[1] : null;
 58 	var icon;
 59 	switch (view) {
 60 		case ZmMsg.viewCalForDay:
 61 			icon = 'DayView';
 62 			this._view = ZmId.VIEW_CAL_DAY;
 63 			view = ZmMsg.viewDay;
 64 			break;
 65 		case ZmMsg.viewCalForWeek:
 66 			icon = 'WeekView';
 67 			this._view = ZmId.VIEW_CAL_WEEK;
 68 			view = ZmMsg.viewWeek;
 69 			break;
 70 		case ZmMsg.viewCalForWorkWeek:
 71 			icon = 'WorkWeekView';
 72 			this._view = ZmId.VIEW_CAL_WORK_WEEK;
 73 			view = ZmMsg.viewWorkWeek;
 74 			break;
 75 		case ZmMsg.viewCalForMonth:
 76 			icon = 'MonthView';
 77 			this._view = ZmId.VIEW_CAL_MONTH;
 78 			view = ZmMsg.viewMonth;
 79 			break;
 80 		case ZmMsg.viewCalForList:
 81 			icon = 'TasksListView';
 82 			this._view = ZmId.VIEW_CAL_LIST;
 83 			view = ZmMsg.viewList;
 84 			break;
 85 		default:
 86 			icon = "CalendarApp";
 87 			this._view = null;
 88 	}
 89 
 90 	dialog._setOkButton(AjxMsg.ok, true, true);
 91 	
 92 	if (this._startDate == null) this._startDate = new Date();
 93 	var startDateValue = DwtCalendar.getDateFullFormatter().format(this._startDate);
 94 
 95 	this._setField(ZmMsg.goToDate, startDateValue, false, true);
 96 	this._setField(ZmMsg.view, view == null ? ZmMsg.calAssistDefaultView : view, view == null, true);
 97 };
