 22 /**
 23  * 
 24  * @class
 25  * This class represents an IM address and is used to centralize the representation of IM addresses,
 26  * since they will be used in both contacts, IM and probably in a Zimlet that recognizes them in email.
 27  * 
 28  */
 29 ZmImAddress = {
 30 
 31     IM_SERVICES : [
 32             { label: ZmMsg.zimbraTitle , value: "local" },
 33             { label: ZmMsg.yahoo       , value: "yahoo" },
 34             { label: ZmMsg.msn         , value: "msn" },
 35             { label: ZmMsg.aol         , value: "aol" },
 36             { label: ZmMsg.other       , value: "other" }
 37     ],
 38 
 39     REGEXP : [], // it's built at load-time below
 40 
 41     parse : function(addr) {
 42         var m = ZmImAddress.REGEXP.exec(addr);
 43         if (m) {
 44             return { service    : m[1],
 45                 screenName : m[2] };
 46         }
 47                 // undef if unknown
 48     },
 49 
 50     make : function(service, screenName) {
 51         var addr = service + "://" + screenName;
 52                 // check if it's acceptable
 53         if (ZmImAddress.parse(addr)) {
 54             return addr;
 55         }
 56         return "";
 57     },
 58 
 59     display : function(id) {
 60         var addr = ZmImAddress.parse(id);
 61         if (addr) {
 62             var a = ZmImAddress.IM_SERVICES, i = 0, s;
 63             while (s = a[i++]) {
 64                 if (s.value == addr.service)
 65                     break;
 66             }
 67             if (s)
 68                 return addr.screenName + " (" + s.label + ")";
 69         }
 70         return id;
 71     }
 72 
 73 };

 75 // HACK for bug 23603
 76 if (/^SmartZone/.test(ZmMsg.zimbraTitle)) {

 80 for (var i = 0; i < ZmImAddress.IM_SERVICES.length; ++i)
 81         ZmImAddress.REGEXP.push(ZmImAddress.IM_SERVICES[i].value);
 82 
 83 ZmImAddress.REGEXP = new RegExp("^(" + ZmImAddress.REGEXP.join("|") + ")://([^\\s]+)$", "i");
