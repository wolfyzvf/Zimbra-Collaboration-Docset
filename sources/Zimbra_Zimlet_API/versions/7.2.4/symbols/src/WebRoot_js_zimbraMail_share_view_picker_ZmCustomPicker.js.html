 20 /**
 21  * Creates a custom picker control.
 22  * @class
 23  * This class represents a custom picker control.
 24  * 
 25  * @param		{DwtControl}	parent		the parent
 26  * 
 27  * @extends		ZmPicker
 28  * @see			ZmPicker.CUSTOM
 29  */
 30 ZmCustomPicker = function(parent) {
 31 
 32 	ZmPicker.call(this, parent);
 33 
 34     this.setTitle(ZmMsg.search);
 35     this.setImage("Search");
 36 }
 37 
 38 ZmCustomPicker.prototype = new ZmPicker;
 39 ZmCustomPicker.prototype.constructor = ZmCustomPicker;
 40 
 41 ZmPicker.CTOR[ZmPicker.CUSTOM] = ZmCustomPicker;
 42 
 43 ZmCustomPicker.prototype.toString = 
 44 function() {
 45 	return "ZmCustomPicker";
 46 }
 47 
 48 ZmCustomPicker.prototype._setupPicker =
 49 function(parent) {
 50 	var picker = new DwtComposite(parent);
 51 	var size = 24;
 52 	var fieldId = Dwt.getNextId();
 53 	var html = new Array(20);
 54 	var i = 0;
 55 	html[i++] = "<p>" + ZmMsg.addSearch + "</p>";
 56 	html[i++] = "<center><table cellpadding='2' cellspacing='0' border='0'>";
 57 	html[i++] = "<tr valign='middle'>";
 58 	html[i++] = "<td align='right' nowrap>" + ZmMsg.searchLabel + "</td>";
 59 	html[i++] = "<td align='left' nowrap><input type='text' autocomplete='off' nowrap size='" + size + "' id='" + fieldId + "'/></td>";
 60 	html[i++] = "</tr>";
 61 	html[i++] = "</table></center>";
 62 	picker.getHtmlElement().innerHTML = html.join("");
 63 
 64 	var field = this._field = document.getElementById(fieldId);
 65 	Dwt.setHandler(field, DwtEvent.ONCHANGE, ZmCustomPicker._onChange);
 66 	field._picker = this;
 67 }
 68 
 69 ZmCustomPicker._onChange = 
 70 function(ev) {
 71 	var element = DwtUiEvent.getTarget(ev);
 72 	var picker = element._picker;
 73 	var charCode = DwtKeyEvent.getCharCode(ev);
 74 	if (charCode == 13 || charCode == 3) {
 75 		picker.execute();
 76 	    return false;
 77 	} else {
 78 		picker._updateQuery();
 79 		return true;
 80 	}	
 81 }
 82 
 83 ZmCustomPicker.prototype._updateQuery = 
 84 function() {
 85 	var val = this._field.value;
 86 	if (val)
 87 		this.setQuery("(" + this._field.value + ")");
 88 	else
 89 		this.setQuery("");
 90 }
