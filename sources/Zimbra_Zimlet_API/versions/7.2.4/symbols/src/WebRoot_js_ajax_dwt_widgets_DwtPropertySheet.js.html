 17  * 
 18  * @private
 19  */
 20 DwtPropertySheet = function(parent, className, posStyle, labelSide) {
 21 	if (arguments.length == 0) return;
 22 	className = className || "DwtPropertySheet";
 23 	DwtComposite.call(this, {parent:parent, className:className, posStyle:posStyle});
 24 
 25 	this._labelSide = labelSide || DwtPropertySheet.DEFAULT;
 26 
 27 	this._propertyIdCount = 0;
 28 	this._propertyList = [];
 29 	this._propertyMap = {};
 30 	
 31 	this._tableEl = document.createElement("TABLE");
 32 	this._tableEl.border = 0;
 33 	this._tableEl.cellSpacing = 3;
 34 	this._tableEl.cellPadding = 0;
 35 	
 36 	var element = this.getHtmlElement();
 37 	element.appendChild(this._tableEl);
 38 }
 39 
 40 DwtPropertySheet.prototype = new DwtComposite;
 41 DwtPropertySheet.prototype.constructor = DwtPropertySheet;
 42 
 43 DwtPropertySheet.prototype.toString = 
 44 function() {
 45 	return "DwtPropertySheet";
 46 }
 47 
 48 // Constants
 49 
 50 DwtPropertySheet.RIGHT = "right";
 51 DwtPropertySheet.LEFT = "left";
 52 DwtPropertySheet.DEFAULT = DwtPropertySheet.LEFT;
 53 
 54 // Data
 55 
 56 DwtPropertySheet.prototype._labelSide;
 57 
 58 DwtPropertySheet.prototype._labelCssClass = "Label";
 59 DwtPropertySheet.prototype._valueCssClass = "Field";
 60 
 61 DwtPropertySheet.prototype._tableEl;
 62 
 63 DwtPropertySheet.prototype._propertyIdCount;
 64 DwtPropertySheet.prototype._propertyList;
 65 DwtPropertySheet.prototype._propertyMap;
 66 
 67 // Public methods
 68 
 69 /**
 70  * Adds a property.
 71  *
 72  * @param label [string] The property label. The value is used to set the
 73  *				inner HTML of the property label cell.
 74  * @param value The property value. If the value is an instance of DwtControl
 75  *				the element returned by <code>getHtmlElement</code> is used;
 76  *				if the value is an instance of Element, it is added directly;
 77  * 				anything else is set as the inner HTML of the property value
 78  *				cell.
 79  * @param required [boolean] Determines if the property should be marked as
 80  *				   required. This is denoted by an asterisk next to the label.
 81  */
 82 DwtPropertySheet.prototype.addProperty = function(label, value, required) {
 83 	var index = this._tableEl.rows.length;
 84 
 85 	var row = this._tableEl.insertRow(index);
 86 	row.vAlign = this._vAlign ? this._vAlign : "top";
 87 
 88 	if (this._labelSide == DwtPropertySheet.LEFT) {
 89 		this._insertLabel(row, label, required);
 90 		this._insertValue(row, value, required);
 91 	}
 92 	else {
 93 		this._insertValue(row, value, required);
 94 		this._insertLabel(row, label, required);
 95 	}
 96 	
 97 	var id = this._propertyIdCount++;
 98 	var property = { id: id, index: index, row: row, visible: true };
 99 	this._propertyList.push(property);
100 	this._propertyMap[id] = property;
101 	return id;
102 };
103 
104 DwtPropertySheet.prototype._insertLabel = function(row, label, required) {
105 	var labelCell = row.insertCell(-1);
106 	labelCell.className = this._labelCssClass;
107 	if (this._labelSide != DwtPropertySheet.LEFT) {
108 		labelCell.width = "100%";
109 		labelCell.style.textAlign = "left";
110 	}
111 	labelCell.innerHTML = label;
112 	if (required) {
113 		var asterisk = this._tableEl.ownerDocument.createElement("SUP");
114 		asterisk.innerHTML = "*";
115 		labelCell.insertBefore(asterisk, labelCell.firstChild);
116 	}
117 };
118 
119 DwtPropertySheet.prototype._insertValue = function(row, value, required) {
120 	var valueCell = row.insertCell(-1);
121 	valueCell.className = this._valueCssClass;
122 	if (!value) {
123 		valueCell.innerHTML = " ";
124 	} else if (value instanceof DwtControl) {
125 		valueCell.appendChild(value.getHtmlElement());
126 	}
127 	/**** NOTE: IE says Element is undefined
128 	else if (value instanceof Element) {
129 	/***/
130 	else if (value.nodeType == AjxUtil.ELEMENT_NODE) {
131 	/***/
132 		valueCell.appendChild(value);
133 	} else {
134 		valueCell.innerHTML = String(value);
135 	}
136 };
137 
138 DwtPropertySheet.prototype.removeProperty = function(id) {
139 	var prop = this._propertyMap[id];
140 	if (prop.visible) {
141 		var propIndex = prop.index;
142 		var tableIndex = this.__getTableIndex(propIndex);
143 		var row = this._tableEl.rows[tableIndex];
144 		row.parentNode.removeChild(row);
145 	}
146 
147 	prop.row = null;
148 	for (var i = index + 1; i < this._propertyList.length; i++) {
149 		var prop = this._propertyList[i];
150 		prop.index--;
151 	}
152 	this._propertyList.splice(index, 1);
153 	delete this._propertyMap[id];
154 };
155 
156 DwtPropertySheet.prototype.setPropertyVisible = function(id, visible) {
157 	var prop = this._propertyMap[id];
158 	if (prop.visible != visible) {
159 		prop.visible = visible;
160 		var propIndex = prop.index;
161 		if (visible) {
162 			var tableIndex = this.__getTableIndex(propIndex);
163 			var row = this._tableEl.insertRow(tableIndex);
164 			DwtPropertySheet.__moveChildNodes(prop.row, row);
165 			prop.row = row;
166 		}
167 		else {
168 			var row = prop.row;
169 			if (row && row.parentNode) {
170 				row.parentNode.removeChild(row);
171 			}
172 		}
173 	}
174 };
175 
176 DwtPropertySheet.prototype.__getTableIndex = function(propIndex) {
177 	var tableIndex = 0;
178 	for (var i = 0; i < propIndex; i++) {
179 		var prop = this._propertyList[i];
180 		if (prop.visible) {
181 			tableIndex++;
182 		}
183 	}
184 	return tableIndex;
185 };
186 
187 DwtPropertySheet.__moveChildNodes = function(srcParent, destParent) {
188 	if (srcParent === destParent) return;
189 	var srcChild = srcParent.firstChild;
190 	while (srcChild != null) {
191 		destParent.appendChild(srcChild);
192 		srcChild = srcParent.firstChild;
193 	}
194 };
