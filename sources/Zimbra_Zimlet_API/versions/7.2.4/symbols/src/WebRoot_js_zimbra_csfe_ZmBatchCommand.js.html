 21 /**
 22  * Creates an empty batch command. Use the {@link #add} method to add commands to it,
 23  * and {@link #run} to invoke it.
 24  * @class
 25  * This class represent a batch command, which is a collection of separate 
 26  * requests. Each command is a callback with a method, arguments, and (usually) an
 27  * object on which to call the method. Normally, when the command is run, it creates
 28  * a SOAP document or JSON object which it hands to the app controller's <code>sendRequest()</code>
 29  * method. It may also pass a response callback and/or an error callback.
 30  * <p>
 31  * Instead of calling sendRequest(), the command should hand the batch command its SOAP
 32  * document or JSON object, response callback, and error callback. The last argument that
 33  * the command receives is a reference to the batch command; that's how it knows it's in batch mode.
 34  * </p>
 35  * <p>
 36  * After all commands have been added to the batch command, call its run() method. That will
 37  * create a BatchRequest out of the individual commands' requests and send it to the
 38  * server. Each subrequest gets an ID. When the BatchResponse comes back, it is broken into
 39  * individual responses. If a response indicates success (it is a <code>*Response</code>), the corresponding
 40  * response callback is called with the result. If the response is a fault, the corresponding
 41  * error callback is called with the exception.
 42  * </p>
 43  * <p>
 44  * A command does not have to be the method that generates a SOAP document or JSON object.
 45  * It can be higher-level. Just make sure that the reference to the batch command gets passed down to it.
 46  * </p>
 47  * @author Conrad Damon
 48  * 
 49  * @param {Boolean}	continueOnError	if <code>true</code>, the batch request continues processing when a subrequest fails (defaults to <code>true</code>)
 50  * @param {String}	accountName		the account name to run this batch command as.
 51  * @param {Boolean}	useJson			if <code>true</code>, send JSON rather than XML
 52  */
 53 ZmBatchCommand = function(continueOnError, accountName, useJson) {
 54 	
 55 	this._onError = (continueOnError === false) ? ZmBatchCommand.STOP : ZmBatchCommand.CONTINUE;
 56 	this._accountName = accountName;
 57 	this._useJson = useJson;
 58 
 59 	this.curId = 0;
 60     this._cmds = [];
 61 	this._requests = [];
 62 	this._respCallbacks = [];
 63 	this._errorCallbacks = [];
 64 };
 65 
 66 /**
 67  * Returns a string representation of the object.
 68  * 
 69  * @return		{String}		a string representation of the object
 70  */
 71 ZmBatchCommand.prototype.toString =
 72 function() {
 73 	return "ZmBatchCommand";
 74 };

 92 /**
 93  * Sets the sensitive flag. This indicates that this batch command
 94  * contains a request with sensitive data. Note: There is no way to unset
 95  * this value for the batch command.
 96  * 
 97  * @param	{Boolean}	sensitive		<code>true</code> to set command as sensitive
 98  */
 99 ZmBatchCommand.prototype.setSensitive = function(sensitive) {
100 	this._sensitive = this._sensitive || sensitive;
101 };
102 
103 /**
104  * Checks if the command is sensitive.
105  * 
106  * @return	{Boolean}	<code>true</code> if the command is sensitive
107  */
108 ZmBatchCommand.prototype.isSensitive = function() {
109 	return this._sensitive;
110 };

112 /**
113  * Adds a command to the list of commands to run as part of this batch request.
114  * 
115  * @param {AjxCallback}	cmd		the command
116  */
117 ZmBatchCommand.prototype.add =
118 function(cmd) {
119 	this._cmds.push(cmd);
120 };

122 /**
123  * Gets the number of commands that are part of this batch request.
124  * 
125  * @return	{int}	the size
126  */
127 ZmBatchCommand.prototype.size =
128 function() {
129 	return this.curId || this._cmds.length;
130 };
131 
132 /**
133  * Runs the batch request. For each individual request, either a response or an
134  * error callback will be called.
135  * 
136  * @param {AjxCallback}		callback		the callback to run after entire batch request has completed
137  * @param {AjxCallback}		errorCallback	the error callback called if anything fails.
138  *										The error callbacks arguments are all
139  *										of the exceptions that occurred. Note:
140  *										only the first exception is passed if
141  *										this batch command's onError is set to
142  *										stop.
143  */
144 ZmBatchCommand.prototype.run =
145 function(callback, errorCallback) {
146 
147 	// Invoke each command so that it hands us its SOAP doc, response callback,
148 	// and error callback
149 	for (var i = 0; i < this._cmds.length; i++) {
150 		var cmd = this._cmds[i];
151 		cmd.run(this);
152 		this.curId++;
153 	}
154 
155 	var params = {
156 		sensitive:		this._sensitive,
157 		asyncMode:		true,
158 		callback:		new AjxCallback(this, this._handleResponseRun, [callback, errorCallback]),
159 		errorCallback:	errorCallback,
160 		accountName:	this._accountName
161 	};
162 
163 	// Create the BatchRequest
164 	if (this._useJson) {
165 		var jsonObj = {BatchRequest:{_jsns:"urn:zimbra", onerror:this._onError}};
166 		var batchRequest = jsonObj.BatchRequest;
167 		var size = this.size();
168 		if (size && this._requests.length) {
169 			for (var i = 0; i < size; i++) {
170 				var request = this._requests[i];
171 				request.requestId = i;
172 				var methodName = ZmCsfeCommand.getMethodName(request);
173 				if (!batchRequest[methodName]) {
174 					batchRequest[methodName] = [];
175 				}
176 				request[methodName].requestId = i;
177 				batchRequest[methodName].push(request[methodName]);
178 			}
179 			params.jsonObj = jsonObj;
180 		}
181 	}
182 	else {
183 		var batchSoapDoc = AjxSoapDoc.create("BatchRequest", "urn:zimbra");
184 		batchSoapDoc.setMethodAttribute("onerror", this._onError);
185 		// Add each command's request element to the BatchRequest, and set its ID
186 		var size = this.size();
187 		if (size > 0) {
188 			for (var i = 0; i < size; i++) {
189 				var soapDoc = this._requests[i];
190 				var reqEl = soapDoc.getMethod();
191 				reqEl.setAttribute("requestId", i);
192 				var node = batchSoapDoc.adoptNode(reqEl);
193 				batchSoapDoc.getMethod().appendChild(node);
194 			}
195 			params.soapDoc = batchSoapDoc;
196 		}
197 	}
198 
199 	// Issue the BatchRequest *but* only when there's something to request
200 	if (params.jsonObj || params.soapDoc) {
201 		appCtxt.getAppController().sendRequest(params);
202 	}
203 	else if (callback) {
204 		callback.run();
205 	}
206 };
207 
208 /**
209  * @private
210  */
211 ZmBatchCommand.prototype._handleResponseRun =
212 function(callback, errorCallback, result) {
213 	var batchResponse = result.getResponse();
214 	if (!batchResponse.BatchResponse) {
215 		DBG.println(AjxDebug.DBG1, "Missing batch response!");
216 		return;
217 	}
218 	// NOTE: In case the order of the requests is significant, we process
219 	//       the responses in the same order.
220 	var responses = [];
221 	for (var method in batchResponse.BatchResponse) {
222 		if (method.match(/^_/)) continue;
223 
224 		var methodResponses = batchResponse.BatchResponse[method];
225 		for (var i = 0; i < methodResponses.length; i++) {
226 			responses[methodResponses[i].requestId] = { method: method, resp: methodResponses[i] };
227 		}
228 	}
229 	var exceptions = [];
230 	for (var i = 0; i < responses.length; i++) {
231 		var response = responses[i];
232 		try {
233 			this._processResponse(response.method, response.resp);
234 		}
235 		catch (ex) {
236 			exceptions.push(ex);
237 			if (this._onError == ZmBatchCommand.STOP) {
238 				break;
239 			}
240 		}
241 	}
242 	if (exceptions.length > 0 && errorCallback) {
243 		errorCallback.run.apply(errorCallback, exceptions);
244 	}
245 	else if (callback) {
246 		callback.run(result);
247 	}
248 };

250 /**
251  * Adds the given command parameters to the batch command, as part of a command's
252  * invocation. Should be called by a function that was added via {@link #add} earlier; that
253  * function should pass the request object.
254  * 
255  * @param {AjxSoapDoc|Object}	request		a SOAP document or JSON object with the command's request
256  * @param {AjxCallback}	respCallback	the next callback in chain for async request
257  * @param {AjxCallback}		errorCallback	the callback to run if there is an exception
258  * 
259  * @see		#add
260  */
261 ZmBatchCommand.prototype.addRequestParams =
262 function(request, respCallback, errorCallback) {
263 	this._requests[this.curId] = request;
264 	this._respCallbacks[this.curId] = respCallback;
265 	this._errorCallbacks[this.curId] = errorCallback;
266 };
267 
268 /**
269  * Adds the given command parameters to the batch command, as part of a command's
270  * invocation. Should be called without a previous {@link #add} command, when the request
271  * object can immediately generate its request object.
272  * 
273  * @param {AjxSoapDoc|object}	request		a SOAP document or JSON object with the command's request
274  * @param {AjxCallback}	respCallback	the next callback in chain for async request
275  * @param {AjxCallback}	errorCallback	the callback to run if there is an exception
276  * 
277  * @see		#add
278  */
279 ZmBatchCommand.prototype.addNewRequestParams =
280 function(request, respCallback, errorCallback) {
281     this.addRequestParams(request, respCallback, errorCallback);
282     this.curId++;
283 };

285 /**
286  * Each type of request will return an array of <code>*Response</code> elements. There may also be
287  * an array of Fault elements. Each element has an ID, so we can match it to its
288  * response or error callback, and run whichever is appropriate.
289  * 
290  * @private
291  */
292 ZmBatchCommand.prototype._processResponse =
293 function(method, resp) {
294 	var id = resp.requestId;
295 
296 	// handle error
297 	if (method == "Fault") {
298 		var ex = ZmCsfeCommand.faultToEx(resp, "ZmBatchCommand.prototype.run");
299 		if (this._errorCallbacks[id]) {
300 			var handled = this._errorCallbacks[id].run(ex);
301 			if (!handled) {
302 				appCtxt.getAppController()._handleException(ex);
303 			}
304 		}
305 		throw ex;
306 	}
307 
308 	// process response callback
309 	if (this._respCallbacks[id]) {
310 		var data = {};
311 		data[method] = resp;
312 		var result = new ZmCsfeResult(data);
313 		this._respCallbacks[id].run(result, resp);
314 	}
315 };
