 22 /**
 23  * Constructor
 24  * @class
 25  * This class is responsible for providing unique, predictable IDs for HTML elements.
 26  * That way, code outside the client can easily locate particular elements.
 27  * <br/>
 28  * <br/>
 29  * Not every element that has an associated JS object will have a known ID. Those are
 30  * allocated only for elements it would be useful to locate: major components of the UI,
 31  * toolbars, buttons, views, menus, some menu items, some selects, and miscellaneous other
 32  * components.
 33  * <br/>
 34  * <br/>
 35  * Unless noted otherwise, a getElementById() on any of the non-skin IDs will return a DIV. One exception
 36  * is input fields. The ID is given to the DwtInputField's actual INPUT, rather than to the
 37  * DIV that contains it. Most other exceptions are table-related: TABLE, TR, and TD.
 38  * <br/>
 39  * <br/>
 40  * There is a simple naming scheme for the IDs themselves. Each ID starts with a "z" followed
 41  * by one to a few letters that indicate the type of object (widget) represented by the element:
 42  * 
 43  * <ul>
 44  * <li><b>z</b> a component that is not a special-purpose widget listed below</li>
 45  * <li><b>ztb</b> 		a toolbar (see {@link ZmId.WIDGET})</li>
 46  * <li><b>zb</b> 		a button (see {@link ZmId.WIDGET_TOOLBAR})</li>
 47  * <li><b>zi</b> 		an input field (see {@link ZmId.WIDGET_INPUT})</li>
 48  * <li><b>zm</b> 		a menu (see {@link ZmId.WIDGET_MENU})</li>
 49  * <li><b>zmi</b> 		a menu item (see {@link ZmId.WIDGET_MENU_ITEM})</li>
 50  * <li><b>zs</b> 		a select (see {@link ZmId.WIDGET_SELECT})</li>
 51  * <li><b>zov</b> 		an overview (see {@link ZmId.WIDGET_OVERVIEW})</li>
 52  * <li><b>zt</b> 		a tree (see {@link ZmId.WIDGET_TREE})</li>
 53  * <li><b>zti</b> 		a tree item (see {@link ZmId.WIDGET_TREE_ITEM})</li>
 54  * <li><b>ztih</b> 	a tree item header (see {@link ZmId.WIDGET_TREE_ITEM_HDR})</li>
 55  *</ul>
 56  *
 57  * The skin defines its own container IDs starting with "skin_", which we provide constants for here.
 58  * 
 59  * @author Conrad Damon
 60  */
 61 ZmId = function() {};

331 ZmId.getTreeItemId =
332 function(overviewId, organizerId, type) {
333 	if (!organizerId && !type) { return; }

336 	} else {
337 		return DwtId._makeId(ZmId.WIDGET_TREE_ITEM, overviewId, organizerId);
338 	}

358 	var id = DwtId._makeId(ZmId.WIDGET_VIEW, context, viewId);

360 };

587 /*
588  * ------------
589  * Message view
590  * ------------
591  * 
592  * A message view displays an email message. There are several different instances of message views, which
593  * makes it necessary to include a context (owning view) to be able to identify each one of them.
594  * 
595  * The function to use is:
596  * 
597  * 		ZmId.getViewId(ZmId.VIEW_MSG, component, context)
598  * 
599  * Since message views are not singletons, a context is always necessary. Omit the component only when getting
600  * an ID for a message view itself.
601  * 
602  * To get the ID for a message view, pass the constant for the message view as well as the context, which can be
603  * ZmId.VIEW_CONVLIST, ZmId.VIEW_CONV, ZmId.VIEW_MSG, or ZmId.VIEW_TRAD:
604  * 
605  * 		ZmId.getViewId(ZmId.VIEW_MSG, null, ZmId.VIEW_TRAD)
606  * 
607  * There are also many components within a message view which are useful to retrieve. To get the ID for a
608  * message view component, pass the component ID (ZmId.MV_*):
609  * 
610  * 		ZmId.getViewId(ZmId.VIEW_MSG, ZmId.MV_HDR_TABLE_TOP_ROW, ZmId.VIEW_TRAD)
611  * 		ZmId.getViewId(ZmId.VIEW_MSG, ZmId.MV_ATT_LINKS, ZmId.VIEW_TRAD)
612  * 
613  * 		var bodyId = ZmId.getViewId(ZmId.VIEW_MSG, ZmId.MV_MSG_BODY, ZmId.VIEW_TRAD)
614  * 
615  * will return the ID for the DIV containing the msg body iframe. To get the ID of the IFRAME element
616  * itself, pass that ID as the context for the IFRAME:
617  * 
618  * 		var iframeId = DwtId.getIframeId(bodyId);
619  * 
620  * For buttons within msg view, pass the context and operation as usual, and add the identifier for
621  * message view (which distinguishes its buttons from, say, those on the VIEW_TRAD toolbar).
622  * 
623  * 		ZmId.getButtonId(ZmId.VIEW_MSG, ZmId.OP_CLOSE, ZmId.VIEW_CONV)
624  * 		ZmId.getButtonId(ZmId.VIEW_MSG, ZmId.OP_EXPAND, ZmId.VIEW_TRAD)
625  */
