 22 /**
 23  * Default constructor for helper class.
 24  * @class
 25  * Miscellaneous contacts-related utility functions. So far, mostly things that
 26  * {@link ZmContactPicker} and {@link ZmGroupView} both need to perform a contacts search and
 27  * display results in a list view.
 28  *
 29  * @author Conrad Damon
 30  */
 31 ZmContactsHelper = function() {};
 32 
 33 /**
 34  * Performs a contact search (in either personal contacts or in the GAL) and populates
 35  * the source list view with the results.
 36  *
 37  * @param	{Hash}	params		a hash of parameters
 38  * @param {Object}	params.obj			the object that is doing the search
 39  * @param {String}	params.query			the query string to search on
 40  * @param {String}	params.queryHint		the query hint (i.e. searching shared folders)
 41  * @param {Boolean}	params.ascending		if <code>true</code>, sort in ascending order
 42  * @param {int}	params.lastId		the ID of last item displayed (for pagination)
 43  * @param {String}	params.lastSortVal	the value of sort field for above item
 44  * @param {AjxCallback}	params.respCallback	the callback to call once response comes back from server
 45  * @param {AjxCallback}	params.errorCallback	the callback to call if error returned from server
 46  * @param {String}	params.accountName	the account to make search request on behalf of
 47  * @param {Array}   params.conds	the conds to restrict the search by (array of {attr:"", op:"", value:""} hashes)
 48  */
 49 ZmContactsHelper.search =
 50 function(params) {
 51 	var o = params.obj;
 52 	if (o._searchButton) {
 53 		o._searchButton.setEnabled(false);
 54 	}
 55 
 56 	params.sortBy = params.ascending ? ZmSearch.NAME_ASC : ZmSearch.NAME_DESC;
 57 	params.types = AjxVector.fromArray([ZmItem.CONTACT]);
 58 	params.offset = params.offset || 0;
 59 	params.limit = ZmContactsApp.SEARCHFOR_MAX;
 60 	params.contactSource = o._contactSource;
 61 	params.field = "contact";
 62 
 63 	var search = new ZmSearch(params);
 64 	search.execute({callback:params.respCallback, errorCallback:params.errorCallback});
 65 };

 67 /**
 68  * Take the contacts and create a list of their email addresses (a contact may have more than one)
 69  * 
 70  * @private
 71  */
 72 ZmContactsHelper._processSearchResponse = 
 73 function(resp) {
 74 	var vec = resp.getResults(ZmItem.CONTACT);
 75 
 76 	// Take the contacts and create a list of their email addresses (a contact may have more than one)
 77 	var list = [];
 78 	var a = vec.getArray();
 79 	for (var i = 0; i < a.length; i++) {
 80 		var contact = a[i];
 81 		if (contact.isGroup() && !contact.isDL) {
 82 			var members = contact.getGroupMembers().good.toString(AjxEmailAddress.SEPARATOR);
 83 			ZmContactsHelper._addContactToList(list, contact, members, true);
 84 		} else {
 85 			var emails = contact.isGal ? [contact.getEmail()] : contact.getEmails();
 86 			for (var j = 0; j < emails.length; j++) {
 87 				ZmContactsHelper._addContactToList(list, contact, emails[j]);
 88 			}
 89 		}
 90 	}
 91 	
 92 	return list;
 93 };
 94 
 95 /**
 96  * @private
 97  */
 98 ZmContactsHelper._addContactToList = 
 99 function(list, contact, addr, isGroup) {
100 	
101 	var fileAs = contact.getFileAs();
102 	var name = (fileAs != addr) ? fileAs : "";
103 	var email = new AjxEmailAddress(addr, null, name, null, isGroup);
104 	email.id = Dwt.getNextId();
105 	email.__contact = contact;
106 	email.icon = contact.getIcon();
107 	if (contact.isDL) {
108 		email.isGroup = true;
109 		email.canExpand = contact.canExpand;
110 		var ac = window.parentAppCtxt || window.appCtxt;
111 		ac.setIsExpandableDL(addr, email.canExpand);
112 	}
113 	list.push(email);
114 };

116 /**
117  * The items are AjxEmailAddress objects
118  * 
119  * @private
120  */
121 ZmContactsHelper._getEmailField =
122 function(html, idx, item, field, colIdx) {
123 	if (field == ZmItem.F_TYPE) {
124 		html[idx++] = AjxImg.getImageHtml(item.icon);
125 	} else if (field == ZmItem.F_NAME) {
126 		html[idx++] = "<nobr>";
127 		html[idx++] = AjxStringUtil.htmlEncode(item.name);
128 		html[idx++] = "</nobr>";
129 	} else if (field == ZmItem.F_EMAIL) {
130 		html[idx++] = AjxStringUtil.htmlEncode(item.address);
131 	} else if (field == ZmItem.F_DEPARTMENT) {
132 		if (item.__contact) {
133 			html[idx++] = AjxStringUtil.htmlEncode(ZmContact.getAttr(item.__contact, ZmContact.F_department));
134 		}
135 	}
136 	return idx;
137 };
