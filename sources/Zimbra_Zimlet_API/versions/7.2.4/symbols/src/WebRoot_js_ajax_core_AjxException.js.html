 16 /**
 17  * Creates an exception.
 18  * @constructor
 19  * @class
 20  * This is the base class for all exceptions in the Zimbra Ajax Toolkit.
 21  * 
 22  * @author Ross Dargahi
 23  * 
 24  * @param {string} 		[msg]		the human readable message
 25  * @param {constant} 	[code]	any error or fault code
 26  * @param {string} 		[method] 	the name of the method throwing the exception
 27  * @param {string} 		[detail]		any additional detail
 28  */
 29 AjxException = function(msg, code, method, detail) {
 30 
 31 	if (arguments.length == 0) { return; }
 32 	
 33 	/** 
 34 	 * Human readable message, if applicable.
 35 	 */
 36 	this.msg = msg;
 37 	
 38 	/** 
 39 	 * Error or fault code, if applicable.
 40 	 */
 41 	this.code = code;
 42 	
 43 	/**
 44 	 * Name of the method throwing the exception, if applicable.
 45 	 */
 46 	this.method = method;
 47 	
 48 	/**
 49 	 * Any additional detail.
 50 	 */
 51 	this.detail = detail;
 52 };
 53 
 54 /**
 55  * Returns a string representation of the object.
 56  * 
 57  * @return		{string}		a string representation of the object
 58  */
 59 AjxException.prototype.toString = 
 60 function() {
 61 	return "AjxException";
 62 };
 63 
 64 /**
 65  * Dumps the exception.
 66  * 
 67  * @return {string}	the state of the exception
 68  */
 69 AjxException.prototype.dump = 
 70 function() {
 71 	return "AjxException: msg=" + this.msg + " code=" + this.code + " method=" + this.method + " detail=" + this.detail;
 72 };

124 AjxException.defaultScriptErrorHandler =
125 function(ex) {
126 	alert(ex);
127 };
128 
129 AjxException.setScriptErrorHandler =
130 function(func) {
131 	AjxException.scriptErrorHandler = func;
132 };
133 
134 AjxException.reportScriptError =
135 function(ex) {
136 	if (AjxException.reportScriptErrors && AjxException.scriptErrorHandler && !(ex instanceof AjxException)) {
137 		AjxException.scriptErrorHandler(ex);
138 	}
139 	throw ex;
140 };
141 
142 AjxException.reportScriptErrors = false;
143 AjxException.scriptErrorHandler = AjxException.defaultScriptErrorHandler;
