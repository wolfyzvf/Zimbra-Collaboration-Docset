 16 /**
 17  * Creates the portal application.
 18  * @class
 19  * This class represents the portal application.
 20  * 
 21  * @param	{DwtControl}	container		the container
 22  * @param	{ZmPortalController}	parentController		the controller
 23  * 
 24  * @extends		ZmApp
 25  */
 26 ZmPortalApp = function(container, parentController) {
 27 	ZmApp.call(this, ZmApp.PORTAL, container, parentController);
 28 }
 29 
 30 ZmPortalApp.prototype = new ZmApp;
 31 ZmPortalApp.prototype.constructor = ZmPortalApp;

 77 /**
 78  * Refreshes the portlets.
 79  * 
 80  */
 81 ZmPortalApp.prototype.refreshPortlets = function() {
 82     var mgr = this.getPortletMgr();
 83     var portlets = mgr.getPortlets();
 84     for (var id in portlets) {
 85         portlets[id].refresh();
 86     }
 87 };

 89 ZmPortalApp.prototype.launch =
 90 function(params, callback) {
 91     var loadCallback = new AjxCallback(this, this._handleLoadLaunch, [params, callback]);
 92 	AjxDispatcher.require("Portal", true, loadCallback, null, true);
 93 };
 94 
 95 ZmPortalApp.prototype._handleLoadLaunch = function(params, callback) {
 96 	var controller = this.getPortalController();
 97 	controller.show();
 98     ZmApp.prototype.launch.call(this, params, callback);
 99 };

101 ZmPortalApp.prototype.activate = function(active) {
102 	var controller = this.getPortalController();
103 	controller.setPaused(!active);

107 /**
108  * Gets the portal manifest.
109  * 
110  * @param	{AjxCallback}		callback		the callback to call after the manifest is loaded
111  * @return	{Object}		the manifest
112  */
113 ZmPortalApp.prototype.getManifest = function(callback) {
114     if (!this._manifest) {
115         // load the portal manifest
116         var portalName = appCtxt.get(ZmSetting.PORTAL_NAME);
117         if (portalName) {
118             var timestamp = new Date().getTime(); 
119             var params = {
120                 url: [ window.appContextPath,"/portals/",portalName,"/manifest.xml?v=",timestamp ].join(""),
121                 callback: callback ? new AjxCallback(this, this._handleLoadManifest, [callback]) : null
122             };
123             var req = AjxLoader.load(params);
124             if (!callback) {
125                 this._handleLoadManifest(callback, req);
126             }
127         }
128     }
129 	else if (callback) {
130 		callback.run(this._manifest);
131 	}
132 	return this._manifest;
133 };
134 
135 ZmPortalApp.prototype._handleLoadManifest = function(callback, req) {
136     var e;
137     if (req.status == 200 && req.responseXML) {
138         try {
139             // serialize manifest into JSON and evaluate
140             var json = new AjxJsonSerializer(true).serialize(req.responseXML);
141             eval("this._manifest = "+json);
142 
143             // further minimize the object structure
144             var portalDef = this._manifest.portal ;
145             var portletsDef = portalDef && portalDef.portlets;
146             if (portletsDef && !(portletsDef.portlet instanceof Array)) {
147                 portletsDef.portlet = [ portletsDef.portlet ];
148             }
149             portalDef.portlets = portletsDef.portlet;
150 
151             if (portalDef.portlets) {
152                 for (var i = 0; i < portalDef.portlets.length; i++) {
153                     var portletDef = portalDef.portlets[i];
154                     var propertyDef = portletDef.property;
155                     if (propertyDef && !(propertyDef instanceof Array)) {
156                         propertyDef = [ propertyDef ];
157                     }
158                     portletDef.properties = propertyDef;
159                     delete portletDef.property;
160                 }
161             }
162         }
163         catch (e) {
164             DBG.println(e);
165         }
166     }
167     else {
168         e = ""
169     }
170 
171     if (!this._manifest) {
172         this._manifest = { error: e };
173     }
174     
175     // callback
176     if (callback) {
177         callback.run(this._manifest);
178     }
179 };

181 /**
182  * Gets the portal controller.
183  * 
184  * @return	{ZmPortalController}	the controller
185  */
186 ZmPortalApp.prototype.getPortalController = function() {
187 	AjxDispatcher.require("Portal");
188 	if (!this._portalController) {
189 		this._portalController = new ZmPortalController(this._container, this);
190 	}
191 	return this._portalController;
192 };

203     }
204     return this._portletMgr;
205 };

221 ZmPortalApp.prototype._getOverviewApp =
222 function() {
223 	if (!this._overviewApp) {
224 		var apps = [];
225 		for (var name in ZmApp.CHOOSER_SORT) {
226 			apps.push({ name: name, sort: ZmApp.CHOOSER_SORT[name] });
227 		}
228 		apps.sort(ZmPortalApp.__BY_SORT);
229 
230 		var appName = null;
231 		for (var i = 0; i < apps.length; i++) {
232 			var app = apps[i];
233 			if (app.name == this._name) { continue; }
234 			if (appCtxt.getApp(app.name).isUpsell) { continue; }
235 
236 			appName = app.name;
237 			break;
238 		}
239 		this._overviewApp = appCtxt.getApp(appName);
240 	}
241 	return this._overviewApp;
242 };
