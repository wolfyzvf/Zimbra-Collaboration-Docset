 20 /**
 21  * Creates a new search dialog.
 22  * @class
 23  * This class represents a new search dialog.
 24  * 
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		ZmDialog
 29  */
 30 ZmNewSearchDialog = function(parent, className) {
 31 
 32 	ZmDialog.call(this, {parent:parent, className:className, title:ZmMsg.saveSearch, id:"SaveSearch"});
 33 
 34 	this._omit = {};
 35 	this._omit[ZmFolder.ID_SPAM] = true;
 36 	this._omit[ZmFolder.ID_DRAFTS] = true;
 37 	this._setNameField(this._nameFieldId);
 38 };
 39 
 40 ZmNewSearchDialog.prototype = new ZmDialog;
 41 ZmNewSearchDialog.prototype.constructor = ZmNewSearchDialog;
 42 
 43 ZmNewSearchDialog.prototype.toString = 
 44 function() {
 45 	return "ZmNewSearchDialog";
 46 };
 47 
 48 /**
 49  * Pops-up the dialog.
 50  * 
 51  * @param	{Hash}	params		a hash of parameters
 52  * @param	{String}	params.search		the search
 53  * @param	{Boolean}	params.showOverview	if <code>true</code>, make the overview visible
 54  * 
 55  */
 56 ZmNewSearchDialog.prototype.popup =
 57 function(params) {
 58 	var account = appCtxt.multiAccounts ? appCtxt.getActiveAccount() : null;
 59 	var overviewId = this._curOverviewId = (account ? ([this.toString(),account.name].join("-")) : this.toString());
 60 	var overviewParams = {
 61 		account: account,
 62 		treeIds: [ZmOrganizer.FOLDER, ZmOrganizer.SEARCH],
 63 		fieldId: this._folderTreeCellId,
 64 		omit: this._omit,
 65 		overviewId: overviewId
 66 	};
 67 	var ov = this._setOverview(overviewParams, true);
 68 	this._folderTreeView = ov.getTreeView(ZmOrganizer.FOLDER);
 69 	this._searchTreeView = ov.getTreeView(ZmOrganizer.SEARCH);
 70 	this._search = params.search;
 71 	this._searchTreeView.setSelected(appCtxt.getFolderTree(account).root, true);
 72 	this._isGlobalSearch = appCtxt.multiAccounts && appCtxt.getSearchController().searchAllAccounts;
 73 
 74 	if (appCtxt.multiAccounts) {
 75 		this._searchTreeView.setVisible(true);
 76 		this._makeOverviewVisible(overviewId);
 77 	}
 78 
 79 	var overviewDiv = document.getElementById(this._overviewDivId);
 80 	if (overviewDiv) {
 81 		Dwt.setVisible(overviewDiv, (params.showOverview && !this._isGlobalSearch));
 82 	}
 83 
 84 	ZmDialog.prototype.popup.call(this);
 85 };
 86 
 87 ZmNewSearchDialog.prototype._contentHtml = 
 88 function() {
 89 	this._nameFieldId 		= this._htmlElId + "_nameField";
 90 	this._folderTreeCellId 	= this._htmlElId + "_folderTreeCell";
 91 	this._overviewDivId 	= this._htmlElId + "_overviewDiv";
 92 
 93 	return (AjxTemplate.expand("share.Dialogs#NewSearch", {id: this._htmlElId}));
 94 };
 95 
 96 ZmNewSearchDialog.prototype._okButtonListener =
 97 function(ev) {
 98 	var results = this._getSearchData();
 99 	if (results) {
100 		DwtDialog.prototype._buttonListener.call(this, ev, results);
101 	}
102 };
103 
104 ZmNewSearchDialog.prototype._getSearchData =
105 function() {
106 
107 	// make sure a parent was selected
108 	var parentFolder = this._isGlobalSearch
109 		? appCtxt.getById(ZmOrganizer.ID_ROOT)
110 		: this._overview[this._curOverviewId].getSelected();
111 
112 	// check name for presence and validity
113 	var name = AjxStringUtil.trim(this._nameField.value);
114 	var msg = ZmFolder.checkName(name, parentFolder);
115 
116 	// make sure parent doesn't already have a child by this name
117 	if (!msg && parentFolder.hasChild(name)) {
118 		msg = ZmMsg.folderOrSearchNameExists;
119 	}
120 		
121 	// if we're creating a top-level search, check for conflict with top-level folder
122 	if (!msg && (parentFolder.id == ZmOrganizer.ID_ROOT) && appCtxt.getFolderTree().root.hasChild(name)) {
123 		msg = ZmMsg.folderOrSearchNameExists;
124 	}
125 
126 	if (msg) {
127 		return this._showError(msg);
128 	} else {
129 		return {
130 			parent: parentFolder,
131 			isGlobal: this._isGlobalSearch,
132 			name: name,
133 			search: this._search
134 		};
135 	}
136 };
137 
138 ZmNewSearchDialog.prototype._enterListener =
139 function(ev) {
140 	var results = this._getSearchData();
141 	if (results) {
142 		this._runEnterCallback(results);
143 	}
144 };
145 
146 ZmNewSearchDialog.prototype._getTabGroupMembers =
147 function() {
148 	var list = [this._nameField];
149 	if (this._overview[this._curOverviewId]) {
150 		list.push(this._overview[this._curOverviewId]);
151 	}
152 	return list;
153 };
