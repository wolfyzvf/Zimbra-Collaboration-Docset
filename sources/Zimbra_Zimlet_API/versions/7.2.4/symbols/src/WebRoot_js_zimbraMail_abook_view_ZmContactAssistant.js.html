 21 /**
 22  * Creates a contact assistant.
 23  * @class
 24  * This class represents a contact assistant.
 25  * 
 26  * @extends	ZmAssistant
 27  */
 28 ZmContactAssistant = function() {
 29 	ZmAssistant.call(this, ZmMsg.createNewContact, ZmMsg.ASST_CMD_CONTACT, ZmMsg.ASST_CMD_SUM_CONTACT);
 30 	this._commandIndex = {};	
 31 	this._sortedFields = [];
 32 	for (var i in ZmContactAssistant._CONTACT_FIELDS) {
 33 		var f = ZmContactAssistant._CONTACT_FIELDS[i];
 34 		this._sortedFields.push(f.scmd);
 35 		this._commandIndex[f.scmd.toLowerCase()] = i;
 36 		this._commandIndex[f.lcmd.toLowerCase()] = i;		
 37 	}
 38 	this._sortedFields.sort();
 39 };

 54 	}

130 ZmContactAssistant.prototype.getHelp =
131 function() {
132 	if (this._helpCommand == null) {
133 		var html = new AjxBuffer();
134 		html.append("<div class='ZmAsstContactHelp'>");
135 		html.append(ZmMsg.ASST_CONTACT_HELP);
136 		html.append("<table cellspacing=1 cellpadding=2 border=0>");
137 		html.append("<tr class=ZmAsstContactHelpHeader><td>", "Short Name", "</td><td>", "Long Name", "</td><td>", "Contact Field", "</td></tr>");
138 		var cmds = ZmAssistant.getHandlerCommands();
139 		for (var i in this._sortedFields) {
140 			var field = this._lookupField(this._sortedFields[i]);
141 			if (field.noHelp) continue;
142 			html.append("<tr><td><b>", AjxStringUtil.htmlEncode(field.scmd), "</b></td><td><b>", AjxStringUtil.htmlEncode(field.lcmd), "</b></td><td>", AjxStringUtil.htmlEncode(field.field),"</td></tr>");
143 		}
144 		html.append("</table></div>");
145 		this._helpCommand = html.toString();
146 	}
147 	return this._helpCommand;
148 };

162 /**
163  * @private
164  */
165 ZmContactAssistant.prototype._parseAddress =
166 function(addr, type) {
167 	var fields = ZmContactAssistant._ADDR_FIELDS[type];
168 	var values= addr.split(";");
169 	for (var i = 0; i < values.length; i++) {
170 		var v = ZmAssistant.trim(values[i]);
171 		this._contactFields[fields[i]] = (i == 2) ? v.toUpperCase() : this._capitalize(v);
172 	}
173 };

175 /**
176  * Gets the contact.
177  * 
178  * @return	{ZmContact}	the contact
179  */
180 ZmContactAssistant.prototype.getContact =
181 function() {
182 	var contact = new ZmContact(null);
183 	for (var key in this._contactFields) {
184 		if (this._contactFields[key]) {
185 			contact.setAttr(key, this._contactFields[key]);
186 		}
187 	}
188 	return contact;
189 };

191 /**
192  * @private
193  */
194 ZmContactAssistant.prototype._capitalize =
195 function(v) {
196 	var words = ZmAssistant.split(v);
197 	for (var i=0; i < words.length; i++) words[i] = words[i].substring(0,1).toUpperCase() + words[i].substring(1);
198 	return words.join(" ");
199 };

201 ZmContactAssistant.__RE_handleNotes = new RegExp([ZmAssistant.SPACES,"\\(([^)]*)\\)?",ZmAssistant.SPACES].join(""));
202 ZmContactAssistant.__RE_handleKeyValue = new RegExp([
203     "((",ZmAssistant.WORD,"):",ZmAssistant.SPACES,"(.*?)",ZmAssistant.SPACES,")",
204     "(",ZmAssistant.WORD,":|$)"
205 ].join(""),"m");

207 /**
208  * @private
209  */
210 ZmContactAssistant.prototype.handle =
211 function(dialog, verb, args) {
212 	dialog._setOkButton(AjxMsg.ok, true, true); // true, "NewContact");
213 	dialog._setExtraButton(ZmMsg.moreDetails, true, true);
214 
215 	var match, i;
216 	this._contactFields = {};
217 
218 	// check address first, since we grab any fields quoted with [], objects in them won't be matched later
219 	for (i = 0; i < ZmContactAssistant._CONTACT_OBJECT_ORDER.length; i++) {
220 		var objType = ZmContactAssistant._CONTACT_OBJECT_ORDER[i];
221 		var obj = ZmContactAssistant._CONTACT_OBJECTS[objType];
222 		while (match = this._matchTypedObject(args, objType, obj)) {
223 			var field = this._lookupField(match.type);
224 			var type = field ? field.key : match.type;
225 			this._contactFields[type] = field.capitalize ? this._capitalize(match.data) : match.data;
226 			args = match.args;
227 		}
228 	}
229 
230 	if (!this._contactFields.notes) {
231 		match = args.match(ZmContactAssistant.__RE_handleNotes);
232 		if (match) {
233 			this._contactFields.notes = match[1];
234 			args = args.replace(match[0], " ");
235 		}
236 	}
237 
238 	while(match = args.match(ZmContactAssistant.__RE_handleKeyValue)) {
239 		var k = match[2];
240 		var v = match[3];
241 		var field = this._lookupField(k);
242 		if (field) {
243 			if (v == null) v = "";
244 			this._contactFields[field.key] = (field.capitalize) ? this._capitalize(v) : v;
245 		}
246 		args = args.replace(match[1],"");
247 	}
248 
249 	var fullName = ZmAssistant.normalize(ZmAssistant.trim(args));
250 
251 	if (this._contactFields[ZmContact.F_firstName] || this._contactFields[ZmContact.F_middleName] || this._contactFields[ZmContact.F_lastName]) {
252 		fullName = null;
253 		if (this._contactFields[ZmContact.F_firstName]) fullName = this._contactFields[ZmContact.F_firstName];
254 		if (this._contactFields[ZmContact.F_middleName]) {
255 			if (fullName) fullName += " ";
256 			fullName += this._contactFields[ZmContact.F_middleName];
257 		}
258 		if (this._contactFields[ZmContact.F_lastName]) {
259 			if (fullName) fullName += " ";
260 			fullName += this._contactFields[ZmContact.F_lastName];
261 		}
262 	} else {
263         fullName = this._capitalize(fullName);
264         var parts = ZmAssistant.split(fullName);
265 		switch (parts.length) {
266 			case 0:
267 				break;
268 			case 1:
269 				this._contactFields[ZmContact.F_firstName] = parts[0];
270 				break;
271 			case 2:
272 				this._contactFields[ZmContact.F_firstName] = parts[0];
273 				this._contactFields[ZmContact.F_lastName] = parts[1];
274 				break;
275 			default:
276 				this._contactFields[ZmContact.F_firstName] = parts.shift();
277 				this._contactFields[ZmContact.F_middleName] = parts.shift();
278 				this._contactFields[ZmContact.F_lastName] = parts.join(" ");
279 				break;
280 		}
281 	}
282 
283 	if ("wa" in this._contactFields) {
284 		this._parseAddress(this._contactFields.wa, "work");
285 		delete this._contactFields.wa;
286 	}
287 	if ("ha" in this._contactFields) {
288 		this._parseAddress(this._contactFields.ha, "home");
289 		delete this._contactFields.ha;
290 	}
291 	if ("oa" in this._contactFields) {
292 		this._parseAddress(this._contactFields.oa, "other");
293 		delete this._contactFields.oa;
294 	}
295 
296 	var index, ri;
297 
298 	index = this._setField(ZmMsg.AB_FIELD_fullName, fullName == "" ? ZmMsg.ASST_CONTACT_fullName : fullName, fullName == "", true);
299 
300 	for (var i=0; i < ZmContactAssistant._CONTACT_FIELDS.length; i++) {
301 		var field = ZmContactAssistant._CONTACT_FIELDS[i];
302 		if (!field || field.dontShow) continue;		
303 		var value = this._contactFields[field.key];
304 
305 		if (value != null) {
306 			value = field.multiLine ? AjxStringUtil.convertToHtml(value) : AjxStringUtil.htmlEncode(value);
307 		}
308 		if (field.defaultValue || value != null) {
309 			var useDefault = (value == null);
310 			ri = this._setField(field.field, useDefault ? field.defaultValue : value, useDefault, false, index+1);
311 		} else {
312 			ri = this._setOptField(field.field, value, false, false, index+1);
313 		}
314 		index = Math.max(index, ri);
315 	}
316 	return;
317 };
