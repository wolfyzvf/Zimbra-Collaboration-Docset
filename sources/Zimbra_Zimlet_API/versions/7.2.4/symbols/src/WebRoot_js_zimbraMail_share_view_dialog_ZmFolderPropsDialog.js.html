 20 /**
 21  * Creates a folder properties dialog.
 22  * @class
 23  * This class represents a folder properties dialog.
 24  * 
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtDialog
 29  */
 30 ZmFolderPropsDialog = function(parent, className) {
 31 	className = className || "ZmFolderPropsDialog";
 32 	var extraButtons;
 33 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
 34 		extraButtons = [
 35 			new DwtDialog_ButtonDescriptor(ZmFolderPropsDialog.ADD_SHARE_BUTTON, ZmMsg.addShare, DwtDialog.ALIGN_LEFT)
 36 		];
 37 	}
 38 
 39 	DwtDialog.call(this, {parent:parent, className:className, title:ZmMsg.folderProperties, extraButtons:extraButtons});
 40 
 41 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
 42 		this.registerCallback(ZmFolderPropsDialog.ADD_SHARE_BUTTON, this._handleAddShareButton, this);
 43 	}
 44 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 45 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._handleCancelButton));
 46 
 47 	this._folderChangeListener = new AjxListener(this, this._handleFolderChange);
 48 
 49 	var view = this._createView();
 50 	this.setView(view);
 51 };

 74 ZmFolderPropsDialog.prototype.popup =
 75 function(organizer) {
 76 	this._organizer = organizer;
 77 	organizer.addChangeListener(this._folderChangeListener);
 78 	var colorCode = 0;
 79 	if (this._color) {
 80         var icon = organizer.getIcon(); 
 81         this._color.setImage(icon);
 82 		if(ZmOrganizer.COLOR_VALUES[organizer.color] && (organizer.rgb != ZmOrganizer.COLOR_VALUES[organizer.color])) {
 83 			colorCode = organizer.rgb;
 84 		} else {
 85 			colorCode = organizer.color;
 86 		}
 87         var defaultColorCode = ZmOrganizer.DEFAULT_COLOR[organizer.type],
 88             defaultColor = ZmOrganizer.COLOR_VALUES[defaultColorCode],
 89             colorMenu = this._color.getMenu(),
 90             moreColorMenu;
 91         if(colorMenu) {
 92             moreColorMenu = (colorMenu.toString() == "ZmMoreColorMenu") ? colorMenu : colorMenu._getMoreColorMenu();
 93             if(moreColorMenu) moreColorMenu.setDefaultColor(defaultColor);
 94         }
 95         this._color.setValue(colorCode);
 96         this._color.setEnabled(organizer.id != ZmFolder.ID_DRAFTS);
 97 	}
 98 
 99 	this._handleFolderChange();
100 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
101 		var isVisible = (!organizer.link || organizer.isAdmin());
102 		this.setButtonVisible(ZmFolderPropsDialog.ADD_SHARE_BUTTON, isVisible);
103 	}
104 
105 	DwtDialog.prototype.popup.call(this);
106 
107 	if (this._nameInputEl.style.display != "none") {
108 		this._nameInputEl.focus();
109 	}
110 };

112 ZmFolderPropsDialog.prototype.popdown =
113 function() {
114 	if (this._organizer) {
115 		this._organizer.removeChangeListener(this._folderChangeListener);
116 		this._organizer = null;
117 	}
118 	DwtDialog.prototype.popdown.call(this);
119 };

144 ZmFolderPropsDialog.prototype._handleResendShare =
145 function(event, share) {
146 
147 	AjxDispatcher.require("Share");
148 	share = share || Dwt.getObjectFromElement(DwtUiEvent.getTarget(event));
149 
150 	// create share info
151 	var tmpShare = new ZmShare({object:share.object});
152 	tmpShare.grantee.id = share.grantee.id;
153 	tmpShare.grantee.email = (share.grantee.type == "guest") ? share.grantee.id : share.grantee.name;
154 	tmpShare.grantee.name = share.grantee.name;
155     if (tmpShare.object.isRemote()) {
156 		tmpShare.grantor.id = tmpShare.object.zid;
157 		tmpShare.grantor.email = tmpShare.object.owner;
158 		tmpShare.grantor.name = tmpShare.grantor.email;
159 		tmpShare.link.id = tmpShare.object.rid;
160 	} else {
161 		tmpShare.grantor.id = appCtxt.get(ZmSetting.USERID);
162 		tmpShare.grantor.email = appCtxt.get(ZmSetting.USERNAME);
163 		tmpShare.grantor.name = appCtxt.get(ZmSetting.DISPLAY_NAME) || tmpShare.grantor.email;
164 		tmpShare.link.id = tmpShare.object.id;
165 	}
166 
167 	tmpShare.link.name = share.object.name;
168 	tmpShare.link.view = ZmOrganizer.getViewName(share.object.type);
169 	tmpShare.link.perm = share.link.perm;
170 
171 	if (share.grantee.type == "guest") {
172 		if (!this._guestFormatter) {
173 			this._guestFormatter = new AjxMessageFormat(ZmMsg.shareWithGuestNotes);
174 		}
175 		var url = share.object.getRestUrl();
176 		var username = tmpShare.grantee.email;
177 		var password = share.link.pw;
178 
179 		if (password && username) {
180 			tmpShare.notes = this._guestFormatter.format([url, username, password]);
181 		}
182 	}
183 
184 	tmpShare.sendMessage(ZmShare.NEW);
185 	appCtxt.setStatusMsg(ZmMsg.resentShareMessage);
186 
187 	return false;
188 };

196 ZmFolderPropsDialog.prototype._handleOkButton =
197 function(event) {
198 	if (!this._handleErrorCallback) {
199 		this._handleErrorCallback = new AjxCallback(this, this._handleError);
200 		this._handleRenameErrorCallback = new AjxCallback(this, this._handleRenameError);
201 	}
202 
203 	// rename folder
204 	var callback = new AjxCallback(this, this._handleColor);
205 	var organizer = this._organizer;
206 	if (!organizer.isSystem() && !organizer.isDataSource()) {
207 		var name = this._nameInputEl.value;
208 		if (organizer.name != name) {
209 			var error = ZmOrganizer.checkName(name);
210 			if (error) {
211 				var dialog = appCtxt.getMsgDialog();
212 				dialog.setMessage(error, DwtMessageDialog.WARNING_STYLE);
213 				dialog.popup();
214 				return;
215 			}
216 			organizer.rename(name, callback, this._handleRenameErrorCallback);
217 			return;
218 		}
219 	}
220 
221 	// else, start by changing color
222 	callback.run(null);
223 };

225 ZmFolderPropsDialog.prototype._handleColor =
226 function(response) {
227 	// change color
228 	var callback = new AjxCallback(this, this._handleFreeBusy);
229 	var organizer = this._organizer;
230 	var color = this._color.getValue() || ZmOrganizer.DEFAULT_COLOR[organizer.type];
231 	if (this._isColorChanged(color, organizer.color, organizer.rgb)) {
232         if (String(color).match(/^#/)) {
233             organizer.setRGB(color, callback, this._handleErrorCallback);
234         }
235         else {
236             organizer.setColor(color, callback, this._handleErrorCallback);
237         }
238         return;
239 	}
240 	// else, change free/busy
241 	callback.run(response);
242 };
243 
244 ZmFolderPropsDialog.prototype._isColorChanged =
245 function(color, oColor, rgb) {
246     return !((String(color).match(/^#/) && color == rgb) || oColor == color);
247 };

249 ZmFolderPropsDialog.prototype._handleFreeBusy =
250 function(response) {
251 	// set free/busy
252 	var callback = new AjxCallback(this, this.popdown);
253 	var organizer = this._organizer;
254 	if (Dwt.getVisible(this._excludeFbEl) && organizer.setFreeBusy) {
255 		var excludeFreeBusy = this._excludeFbCheckbox.checked;
256 		if (organizer.excludeFreeBusy != excludeFreeBusy) {
257 			organizer.setFreeBusy(excludeFreeBusy, callback, this._handleErrorCallback);
258 			return;
259 		}
260 	}
261 
262 	// else, popdown
263 	callback.run(response);
264 };

271 ZmFolderPropsDialog.prototype._handleRenameError =
272 function(response) {
273 	var value = this._nameInputEl.value;
274 	var msg;
275 	var noDetails = false;
276 	if (response.code == ZmCsfeException.MAIL_ALREADY_EXISTS) {
277 		msg = AjxMessageFormat.format(ZmMsg.errorAlreadyExists, [value]);
278 	} else if (response.code == ZmCsfeException.MAIL_IMMUTABLE) {
279 		msg = AjxMessageFormat.format(ZmMsg.errorCannotRename, [value]);
280 	} else if (response.code == ZmCsfeException.SVC_INVALID_REQUEST) { 
281 		msg = response.msg; // triggered on an empty name
282 	} else if (response.code == ZmCsfeException.MAIL_INVALID_NAME) {
283 		//I add this here despite checking upfront using ZmOrganizer.checkName, since there might be more restrictions for different types of organizers. so just in case the server still returns an error in the name.
284 		msg = AjxMessageFormat.format(ZmMsg.invalidName, [AjxStringUtil.htmlEncode(value)]);
285 		noDetails = true;
286 	}
287 	appCtxt.getAppController().popupErrorDialog(msg, noDetails ? null : response.msg, null, true);
288 	return true;
289 };

296 ZmFolderPropsDialog.prototype._handleFolderChange =
297 function(event) {
298 	var organizer = this._organizer;
299 	
300 	if (organizer.isSystem() || organizer.isDataSource()) {
301 		this._nameOutputEl.innerHTML = AjxStringUtil.htmlEncode(organizer.name);
302 		this._nameOutputEl.style.display = "block";
303 		this._nameInputEl.style.display = "none";
304 	}
305 	else {
306 		this._nameInputEl.value = organizer.name;
307 		this._nameInputEl.style.display = "block";
308 		this._nameOutputEl.style.display = "none";
309 	}
310 	this._ownerEl.innerHTML = AjxStringUtil.htmlEncode(organizer.owner);
311 	this._typeEl.innerHTML = ZmMsg[ZmOrganizer.FOLDER_KEY[organizer.type]] || ZmMsg.folder;
312 
313 	if (this._color) {
314 		var colorCode = 0;
315 		if(ZmOrganizer.COLOR_VALUES[organizer.color] && (organizer.rgb != ZmOrganizer.COLOR_VALUES[organizer.color])) {
316 			colorCode = organizer.rgb;
317 		} else {
318 			colorCode = organizer.color;
319 		}
320 		this._color.setValue(colorCode);
321 		var isVisible = (organizer.type != ZmOrganizer.FOLDER ||
322 						 (organizer.type == ZmOrganizer.FOLDER && appCtxt.get(ZmSetting.MAIL_FOLDER_COLORS_ENABLED)));
323 		this._props.setPropertyVisible(this._colorId, isVisible);
324 	}
325 	this._excludeFbCheckbox.checked = organizer.excludeFreeBusy;
326 
327 	var showPerm = organizer.isMountpoint;
328 	if (showPerm) {
329 		AjxDispatcher.require("Share");
330 		var share = ZmShare.getShareFromLink(organizer);
331 		var role = share && share.link && share.link.role;
332 		this._permEl.innerHTML = ZmShare.getRoleActions(role);
333 	}
334 
335 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
336 		this._populateShares(organizer);
337 	}
338 
339     var url = organizer.url ? AjxStringUtil.htmlEncode(organizer.url).replace(/&/g,'%26') : null;
340 	var urlDisplayString = url;
341 	if (urlDisplayString) {
342         urlDisplayString = AjxStringUtil.clipByLength(urlDisplayString,50);
343     }
344 
345     if(urlDisplayString){
346         urlDisplayString = ['<a target=_new href="',url,'">',urlDisplayString,'</a>'].join("");
347     }
348 
349     this._urlEl.innerHTML = urlDisplayString || "";
350 
351 	this._props.setPropertyVisible(this._ownerId, organizer.owner != null);
352 
353 	this._props.setPropertyVisible(this._urlId, organizer.url);
354 	this._props.setPropertyVisible(this._permId, showPerm);
355 
356 	Dwt.setVisible(this._excludeFbEl, !organizer.link && (organizer.type == ZmOrganizer.CALENDAR));
357 };

359 ZmFolderPropsDialog.prototype._populateShares =
360 function(organizer) {
361 
362 	this._sharesGroup.setContent("");
363 
364 	var displayShares = this._getDisplayShares(organizer);
365 
366 	var getFolder = false;
367 	if (displayShares.length) {
368 		for (var i = 0; i < displayShares.length; i++) {
369 			var share = displayShares[i];
370 			if (!(share.grantee && share.grantee.name)) {
371 				getFolder = true;
372 			}
373 		}
374 	}
375 
376 	if (getFolder) {
377 		var respCallback = new AjxCallback(this, this._handleResponseGetFolder, [displayShares]);
378 		organizer.getFolder(respCallback);
379 	} else {
380 		this._handleResponseGetFolder(displayShares);
381 	}
382 
383 
384 	this._sharesGroup.setVisible(displayShares.length > 0);
385 };
386 
387 ZmFolderPropsDialog.prototype._getDisplayShares =
388 function(organizer) {
389 
390 	var shares = organizer.shares;
391 	var displayShares = [];
392 	if ((!organizer.link || organizer.isAdmin()) && shares && shares.length > 0) {
393 		AjxDispatcher.require("Share");
394 		var userZid = appCtxt.accountList.mainAccount.id;
395 		// if a folder was shared with us with admin rights, a share is created since we could share it;
396 		// don't show any share that's for us in the list
397 		for (var i = 0; i < shares.length; i++) {
398 			var share = shares[i];
399 			if (share.grantee) {
400 				var granteeId = share.grantee.id;
401 				if ((share.grantee.type != ZmShare.TYPE_USER) || (share.grantee.id != userZid)) {
402 					displayShares.push(share);
403 				}
404 			}
405 		}
406 	}
407 
408 	return displayShares;
409 };

411 ZmFolderPropsDialog.prototype._handleResponseGetFolder =
412 function(displayShares, organizer) {
413 
414 	if (organizer) {
415 		displayShares = this._getDisplayShares(organizer);
416 	}
417 
418 	if (displayShares.length) {
419 		var table = document.createElement("TABLE");
420 		table.border = 0;
421 		table.cellSpacing = 0;
422 		table.cellPadding = 3;
423 		for (var i = 0; i < displayShares.length; i++) {
424 			var share = displayShares[i];
425 			var row = table.insertRow(-1);
426 			var nameEl = row.insertCell(-1);
427 			nameEl.style.paddingRight = "15px";
428 			var nameText = share.grantee && share.grantee.name;
429 			if (share.isAll()) {
430 				nameText = ZmMsg.shareWithAll;
431 			} else if (share.isPublic()) {
432 				nameText = ZmMsg.shareWithPublic;
433 			} else if (share.isGuest()){
434                 nameText = nameText || (share.grantee && share.grantee.id);
435 			}
436 			nameEl.innerHTML = AjxStringUtil.htmlEncode(nameText);
437 
438 			var roleEl = row.insertCell(-1);
439 			roleEl.style.paddingRight = "15px";
440 			roleEl.innerHTML = ZmShare.getRoleName(share.link.role);
441 
442 			this.__createCmdCells(row, share);
443 		}
444 		this._sharesGroup.setElement(table);
445 
446 		var width = Dwt.DEFAULT;
447 		var height = displayShares.length > 5 ? ZmFolderPropsDialog.SHARES_HEIGHT : Dwt.CLEAR;
448 
449 		var insetElement = this._sharesGroup.getInsetHtmlElement();
450 		Dwt.setScrollStyle(insetElement, Dwt.SCROLL);
451 		Dwt.setSize(insetElement, width, height);
452 	}
453 
454 	this._sharesGroup.setVisible(displayShares.length > 0);
455 };

457 ZmFolderPropsDialog.prototype.__createCmdCells =
458 function(row, share) {
459 	var type = share.grantee.type;
460 	if (type == ZmShare.TYPE_DOMAIN || !share.link.role) {
461 		var cell = row.insertCell(-1);
462 		cell.colSpan = 3;
463 		cell.innerHTML = ZmMsg.configureWithAdmin;
464 		return;
465 	}
466 
467 	var actions = [ZmShare.EDIT, ZmShare.REVOKE, ZmShare.RESEND];
468 	var handlers = [this._handleEditShare, this._handleRevokeShare, this._handleResendShare];
469 
470 	for (var i = 0; i < actions.length; i++) {
471 		var action = actions[i];
472 		var cell = row.insertCell(-1);
473 
474 		// public shares have no editable fields, and sent no mail
475 		var isAllShare = share.grantee && (share.grantee.type == ZmShare.TYPE_ALL);
476 		if ((isAllShare || share.isPublic()) && (action == ZmShare.EDIT || action == ZmShare.RESEND)) { continue; }
477 
478 		var link = document.createElement("A");
479 		link.href = "#";
480 		link.innerHTML = ZmShare.ACTION_LABEL[action];
481 
482 		Dwt.setHandler(link, DwtEvent.ONCLICK, handlers[i]);
483 		Dwt.associateElementWithObject(link, share);
484 
485 		cell.appendChild(link);
486 	}
487 };

489 ZmFolderPropsDialog.prototype._createView =
490 function() {
491 	var view = new DwtComposite(this);
492 
493 	// create html elements
494 	this._nameOutputEl = document.createElement("SPAN");
495 	this._nameInputEl = document.createElement("INPUT");
496 	this._nameInputEl.style.width = "20em";
497 	this._nameInputEl._dialog = this;
498 	var nameElement = this._nameInputEl;
499 
500 	this._ownerEl = document.createElement("DIV");
501 	this._typeEl = document.createElement("DIV");
502 	this._urlEl = document.createElement("DIV");
503 	this._permEl = document.createElement("DIV");
504 
505 	var nameEl = document.createElement("DIV");
506 	nameEl.appendChild(this._nameOutputEl);
507 	nameEl.appendChild(nameElement);
508 
509 	this._excludeFbCheckbox = document.createElement("INPUT");
510 	this._excludeFbCheckbox.type = "checkbox";
511 	this._excludeFbCheckbox._dialog = this;
512 
513 	this._excludeFbEl = document.createElement("DIV");
514 	this._excludeFbEl.style.display = "none";
515 	this._excludeFbEl.appendChild(this._excludeFbCheckbox);
516 	this._excludeFbEl.appendChild(document.createTextNode(ZmMsg.excludeFromFreeBusy));
517 
518 	// setup properties group
519 	var propsGroup = new DwtGrouper(view);
520 	propsGroup.setLabel(ZmMsg.properties);
521 
522 	this._props = new DwtPropertySheet(propsGroup);
523 	this._color = new ZmColorButton({parent:this});
524 
525 	this._props.addProperty(ZmMsg.nameLabel, nameEl);
526 	this._props.addProperty(ZmMsg.typeLabel, this._typeEl);
527 	this._ownerId = this._props.addProperty(ZmMsg.ownerLabel, this._ownerEl);
528 	this._urlId = this._props.addProperty(ZmMsg.urlLabel, this._urlEl);
529 	this._permId = this._props.addProperty(ZmMsg.permissions, this._permEl);
530 	this._colorId = this._props.addProperty(ZmMsg.colorLabel, this._color);
531 
532 	var propsContainer = document.createElement("DIV");
533 	propsContainer.appendChild(this._props.getHtmlElement());
534 	propsContainer.appendChild(this._excludeFbEl);
535 	
536 	propsGroup.setElement(propsContainer);
537 
538 	// setup shares group
539 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
540 		this._sharesGroup = new DwtGrouper(view);
541 		this._sharesGroup.setLabel(ZmMsg.folderSharing);
542 		this._sharesGroup.setVisible(false);
543 		this._sharesGroup.setScrollStyle(Dwt.SCROLL);
544 	}
545 
546 	// add everything to view and return
547 	var element = view.getHtmlElement();
548 	element.appendChild(propsGroup.getHtmlElement());
549 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
550 		element.appendChild(this._sharesGroup.getHtmlElement());
551 	}
552 
553 	return view;
554 };
