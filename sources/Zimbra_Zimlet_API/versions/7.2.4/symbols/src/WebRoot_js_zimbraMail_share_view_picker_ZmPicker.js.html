 20 /**
 21  * Creates a picker control.
 22  * @class
 23  * This class represents a picker (i.e. "selection") control.
 24  * 
 25  * @param		{DwtControl}	parent		the parent
 26  * @param		{constant}		id		the id
 27  * 
 28  * @extends		DwtComposite
 29  */
 30 ZmPicker = function(parent, id) {
 31 	if (arguments.length == 0) return;
 32 	DwtComposite.call(this, {parent:parent, className:"ZmPicker", posStyle:DwtControl.ABSOLUTE_STYLE});
 33 
 34 	this._header = new DwtToolBar({parent:this});
 35 
 36 	this._label = new DwtLabel(this._header, DwtLabel.IMAGE_LEFT | DwtLabel.ALIGN_LEFT, "ZmPickerLabel");
 37 	this._header.addFiller();
 38 
 39 	this._close = new DwtButton({parent:this._header, style:DwtLabel.IMAGE_LEFT, className:"DwtToolbarButton"});
 40 	this._close.setImage("Close");
 41 	this._close.setToolTipContent(ZmMsg.close);
 42 
 43 	this._picker = new DwtComposite(this, "ZmPickerOverview");
 44 	this._picker.setSize(Dwt.DEFAULT, parent.getH() - this._header.getH());
 45 	this._picker.setScrollStyle(Dwt.SCROLL);
 46 	this._pickerEvent = new ZmEvent(ZmEvent.S_PICKER);
 47 	this._pickerEvent.set(ZmEvent.E_MODIFY, this);
 48 
 49 	this.id = id;
 50     this.setTitle(ZmMsg[ZmPicker.T_MSG_KEY[id]]);
 51     this.setImage(ZmPicker.IMAGE[id]);
 52 	this._setupPicker(this._picker);
 53 };

302 ZmPicker.prototype.execute =
303 function() {
304 	if (this.isListenerRegistered(ZmEvent.L_PICKER)) {
305 		this._pickerEvent.set(ZmEvent.E_LOAD, this);
306 		this.notifyListeners(ZmEvent.L_PICKER, this._pickerEvent);
307 	}
308 };
309 
310 ZmPicker.prototype.setQuery =
311 function(query) {
312 	this._query = query;
313 	if (this.isListenerRegistered(ZmEvent.L_PICKER)) {
314 		this._pickerEvent.set(ZmEvent.E_MODIFY, this);
315 		this.notifyListeners(ZmEvent.L_PICKER, this._pickerEvent);
316 	}
317 };

319 /**
320  * Disposes of the picker.
321  * 
322  */
323 ZmPicker.prototype.dispose =
324 function() {
325 	DwtComposite.prototype.dispose.call(this);
326 	if (this._treeView) {
327 		var opc = appCtxt.getOverviewController();
328 		this._overview.clear();
329 	}
330 };
331 
332 ZmPicker.prototype._setOverview =
333 function(overviewId, parent, types) {
334 	var params = {
335 		overviewId: overviewId,
336 		parent: parent,
337 		headerClass: "DwtTreeItem",
338 		treeStyle: DwtTree.CHECKEDITEM_STYLE,
339 		isCheckedByDefault: false,
340 		treeIds: types,
341 		account: appCtxt.getActiveAccount()
342 	};
343 	var overview = this._overview = appCtxt.getOverviewController().createOverview(params);
344 	overview.set(types);
345 	this._treeView = {};
346 	for (var i = 0; i < types.length; i++) {
347 		var treeView = this._treeView[types[i]] = overview.getTreeView(types[i]);
348 		treeView.addSelectionListener(new AjxListener(this, this._treeListener));
349 	}
350 	if (types.length == 1) {
351 		this._hideRoot(types[0]);
352 	}
353 };
354 
355 ZmPicker.prototype._hideRoot =
356 function(type) {
357 	var ti = this._treeView[type].getTreeItemById(ZmOrganizer.ID_ROOT);
358 	if (!ti) {
359 		var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT);
360 		ti = this._treeView[type].getTreeItemById(rootId);
361 	}
362 	ti.showCheckBox(false);
363 	ti.setExpanded(true);
364 	ti.setVisible(false, true);
365 };
