 16 /**
 17  * Creates a new, empty "traditional view" controller.
 18  * @constructor
 19  * @class
 20  * This class manages the two-pane message view. The top pane contains a list
 21  * view of the messages in the conversation, and the bottom pane contains the current
 22  * message.
 23  *
 24  * @author Parag Shah
 25  * 
 26  * @param {ZmComposite}	container	the containing shell
 27  * @param {ZmMailApp}	mailApp			the containing app
 28  * 
 29  * @extends		ZmDoublePaneController
 30  * 
 31  * @private
 32  */
 33 ZmTradController = function(container, mailApp) {
 34 	ZmDoublePaneController.call(this, container, mailApp);
 35 	this._msgControllerMode = ZmId.VIEW_TRAD;
 36 };
 37 
 38 ZmTradController.prototype = new ZmDoublePaneController;
 39 ZmTradController.prototype.constructor = ZmTradController;
 40 
 41 ZmMailListController.GROUP_BY_ITEM[ZmId.VIEW_TRAD]		= ZmItem.MSG;
 42 ZmMailListController.GROUP_BY_SETTING[ZmId.VIEW_TRAD]	= ZmSetting.GROUP_BY_MESSAGE;
 43 
 44 // view menu
 45 ZmMailListController.GROUP_BY_ICON[ZmId.VIEW_TRAD]		= "MessageView";
 46 ZmMailListController.GROUP_BY_MSG_KEY[ZmId.VIEW_TRAD]	= "byMessage";
 47 ZmMailListController.GROUP_BY_SHORTCUT[ZmId.VIEW_TRAD]	= ZmKeyMap.VIEW_BY_MSG;
 48 ZmMailListController.GROUP_BY_VIEWS.push(ZmId.VIEW_TRAD);
 49 
 50 // Public methods
 51 
 52 ZmTradController.prototype.toString = 
 53 function() {
 54 	return "ZmTradController";
 55 };
 56 
 57 ZmTradController.prototype.show =
 58 function(search) {
 59 	this._list = search.getResults(ZmItem.MSG);
 60 
 61 	// call base class
 62 	ZmDoublePaneController.prototype.show.call(this, search, this._list);
 63 	appCtxt.set(ZmSetting.GROUP_MAIL_BY, ZmSetting.GROUP_BY_MESSAGE);
 64 	this._resetNavToolBarButtons(ZmId.VIEW_TRAD);
 65 };
 66 
 67 // Private methods
 68 
 69 ZmTradController.prototype._createDoublePaneView = 
 70 function() {
 71 	return (new ZmTradView({parent:this._container, posStyle:Dwt.ABSOLUTE_STYLE,
 72 							controller:this, dropTgt:this._dropTgt}));
 73 };
 74 
 75 ZmTradController.prototype._getViewType =
 76 function() {
 77 	return ZmId.VIEW_TRAD;
 78 };
 79 
 80 ZmTradController.prototype._paginate = 
 81 function(view, bPageForward, convIdx, limit) {
 82 	view = view ? view : this._currentView;
 83 	return ZmDoublePaneController.prototype._paginate.call(this, view, bPageForward, convIdx, limit);
 84 };
 85 
 86 ZmTradController.prototype._resetNavToolBarButtons = 
 87 function(view) {
 88 
 89 	ZmDoublePaneController.prototype._resetNavToolBarButtons.call(this, view);
 90 	if (!this._navToolBar[view]) { return; }
 91 
 92 	this._navToolBar[view].setToolTip(ZmOperation.PAGE_BACK, ZmMsg.previousPage);
 93 	this._navToolBar[view].setToolTip(ZmOperation.PAGE_FORWARD, ZmMsg.nextPage);
 94 };
 95 
 96 ZmTradController.prototype._getMoreSearchParams = 
 97 function(params) {
 98 	// OPTIMIZATION: find out if we need to pre-fetch the first hit message
 99 	params.fetch = this.isReadingPaneOn();
100 	params.markRead = true;
101 };
102 
103 ZmTradController.prototype._listSelectionListener =
104 function(ev) {
105 	var item = ev.item;
106 	if (!item) { return; }
107 	var handled = ZmDoublePaneController.prototype._listSelectionListener.apply(this, arguments);
108 	if (!handled && ev.detail == DwtListView.ITEM_DBL_CLICKED) {
109 		var respCallback = new AjxCallback(this, this._handleResponseListSelectionListener, item);
110 		var ctlr = AjxDispatcher.run("GetMsgController", item.nId);
111 		ctlr.show(item, this._msgControllerMode, respCallback, true);
112 	}
113 };
114

117 ZmTradController.prototype._handleResponsePaginate = 
118 function(view, saveSelection, loadIndex, offset, result, ignoreResetSelection) {
119 	// bug fix #5134 - overload to ignore resetting the selection since it is handled by setView
120 	ZmMailListController.prototype._handleResponsePaginate.call(this, view, saveSelection, loadIndex, offset, result, true);
121 };
