 16 /**
 17  * Creates a recurrence object.
 18  * @class
 19  * This class represents a recurrence pattern.
 20  * 
 21  * @param	{ZmCalItem}	calItem		the calendar item
 22  * 
 23  */
 24 ZmRecurrence = function(calItem) {
 25 	this._startDate 			= (calItem && calItem.startDate) ? calItem.startDate : (new Date());
 26 
 27 	// initialize all params (listed alphabetically)
 28 	this.repeatCustom			= "0";  										// 1|0
 29 	this.repeatCustomCount		= 1; 											// ival
 30 	this.repeatCustomDayOfWeek	= "SU"; 										// (DAY|WEEKDAY|WEEKEND) | (SU|MO|TU|WE|TH|FR|SA)
 31 	this.repeatBySetPos	= "1";
 32 	this.repeatCustomMonthDay	= this._startDate.getDate();
 33 	this.repeatCustomType		= "S"; 											// (S)pecific, (O)rdinal
 34 	this.repeatEnd				= null;
 35 	this.repeatEndCount			= 1; 											// maps to "count" (when there is no end date specified)
 36 	this.repeatEndDate			= null; 										// maps to "until"
 37 	this.repeatEndType			= "N";
 38 	this.repeatMonthlyDayList	= null; 										// list of numbers representing days (usually, just one day)
 39 	this.repeatType				= ZmRecurrence.NONE;							// maps to "freq"
 40 	this.repeatWeekday			= false; 										// set to true if freq = "DAI" and custom repeats every weekday
 41 	this.repeatWeeklyDays		= [];	 										// SU|MO|TU|WE|TH|FR|SA
 42 	this.repeatYearlyMonthsList	= 1; 											// list of numbers representing months (usually, just one month)
 43 
 44     this._cancelRecurIds        = {};                                           //list of recurIds to be excluded
 45 };

 86 ZmRecurrence.prototype.setSoap =
 87 function(soapDoc, inv) {
 88 	if (this.repeatType == ZmRecurrence.NONE) return;
 89 
 90 	var recur = soapDoc.set("recur", null, inv);
 91 	var add = soapDoc.set("add", null, recur);
 92 	var rule = soapDoc.set("rule", null, add);
 93 	rule.setAttribute("freq", this.repeatType);
 94 
 95 	var interval = soapDoc.set("interval", null, rule);
 96 	interval.setAttribute("ival", this.repeatCustomCount);
 97 
 98 	if (this.repeatEndDate != null && this.repeatEndType == "D") {
 99 		var until = soapDoc.set("until", null, rule);
100 		until.setAttribute("d", AjxDateUtil.getServerDate(this.repeatEndDate));
101 	} else if (this.repeatEndType == "A"){
102 		var c = soapDoc.set("count",null, rule);
103 		c.setAttribute("num", this.repeatEndCount);
104 	}
105 
106 	if (this.repeatCustom != "1") {
107         this.setExcludes(soapDoc, recur);
108 		return;
109     }
110 
111 	if (this.repeatType == ZmRecurrence.DAILY) {
112 		if (this.repeatWeekday) {
113 			// TODO: for now, handle "every weekday" as M-F
114 			//       eventually, needs to be localized work week days
115 			var bwd = soapDoc.set("byday", null, rule);
116 			for (var i in ZmCalItem.SERVER_WEEK_DAYS) {
117 				var day = ZmCalItem.SERVER_WEEK_DAYS[i];
118 				if (day == "SA" || day == "SU")
119 					continue;
120 				var wkDay = soapDoc.set("wkday", null, bwd);
121 				wkDay.setAttribute("day", day);
122 			}
123 		}
124 	} else if (this.repeatType == ZmRecurrence.WEEKLY) {
125 		var bwd = soapDoc.set("byday", null, rule);
126 		for (var i = 0; i < this.repeatWeeklyDays.length; ++i) {
127 			var wkDay = soapDoc.set("wkday", null, bwd);
128 			wkDay.setAttribute("day", this.repeatWeeklyDays[i]);
129 		}
130 	}
131 	else if (this.repeatType == ZmRecurrence.MONTHLY)
132 	{
133 		if (this.repeatCustomType == "S") {
134 			var bmd = soapDoc.set("bymonthday", null, rule);
135 			bmd.setAttribute("modaylist", this.repeatMonthlyDayList);
136 		} else {
137 			var bwd = soapDoc.set("byday", null, rule);
138 
139             if(this.repeatCustomDays) {
140                 for(var i=0; i < this.repeatCustomDays.length; i++) {
141                     wkDay = soapDoc.set("wkday", null, bwd);
142                     wkDay.setAttribute("day", this.repeatCustomDays[i]);
143                     if(this.repeatCustomOrdinal) {
144                         wkDay.setAttribute("ordwk", this.repeatCustomOrdinal);
145                     }
146                 }
147             }
148 
149             if(this.repeatCustomOrdinal == null) {
150                 var bysetpos = soapDoc.set("bysetpos", null, rule);
151                 bysetpos.setAttribute("poslist", this.repeatBySetPos);
152             }
153         }
154     }
155 	else if (this.repeatType == ZmRecurrence.YEARLY)
156 	{
157 		var bm = soapDoc.set("bymonth", null, rule);
158 		bm.setAttribute("molist", this.repeatYearlyMonthsList);
159 		if (this.repeatCustomType == "O") {
160 			var bwd = soapDoc.set("byday", null, rule);
161 			
162             if(this.repeatCustomDays) {
163                 for(var i=0; i < this.repeatCustomDays.length; i++) {
164                     wkDay = soapDoc.set("wkday", null, bwd);
165 			        //wkDay.setAttribute("ordwk", this.repeatCustomOrdinal);
166                     wkDay.setAttribute("day", this.repeatCustomDays[i]);
167                     if(this.repeatCustomOrdinal) {
168                         wkDay.setAttribute("ordwk", this.repeatCustomOrdinal);
169                     }
170                 }
171             }
172 
173             if(this.repeatCustomOrdinal == null) {
174                 var bysetpos = soapDoc.set("bysetpos", null, rule);
175                 bysetpos.setAttribute("poslist", this.repeatBySetPos);
176             }
177 
178         } else {
179 			var bmd = soapDoc.set("bymonthday", null, rule);
180 			bmd.setAttribute("modaylist", this.repeatCustomMonthDay);
181 		}
182 
183 	}
184 
185     this.setExcludes(soapDoc, recur);
186 };

188 ZmRecurrence.prototype.setExcludes =
189 function(soapDoc, recur) {
190     if(!this._cancelRecurIds) return;
191 
192     var exclude;
193     var dates;
194 
195     for(var i in this._cancelRecurIds) {
196 
197         if(!this._cancelRecurIds[i]) continue;
198 
199         if(!exclude && !dates) {
200             exclude = soapDoc.set("exclude", null, recur);
201             dates = soapDoc.set("dates", null, exclude);
202         }
203 
204         var ridZ = i;
205         var dtval = soapDoc.set("dtval", null, dates);
206         var s = soapDoc.set("s", null, dtval);
207         s.setAttribute("d", ridZ);
208     }
209 };

211 /**
212  * Gets the recurrence blurb.
213  * 
214  * @return	{String}	the blurb text
215  */
216 ZmRecurrence.prototype.getBlurb =
217 function() {
218 	if (this.repeatType == ZmRecurrence.NONE)
219 		return "";
220 
221 	var every = [];
222 	switch (this.repeatType) {
223 		case ZmRecurrence.DAILY: {
224 			if (this.repeatCustom == "1" && this.repeatWeekday) {
225 				every.push(ZmMsg.recurDailyEveryWeekday);
226 			} else if (this.repeatCustomCount == 1) {
227 				every.push(ZmMsg.recurDailyEveryDay);
228 			} else {
229 				var formatter = new AjxMessageFormat(ZmMsg.recurDailyEveryNumDays);
230 				every.push(formatter.format(this.repeatCustomCount));
231 			}
232 			break;
233 		}
234 		case ZmRecurrence.WEEKLY: {
235 			if (this.repeatCustomCount == 1 && this.repeatWeeklyDays.length == 1) {
236 				var dayofweek = AjxUtil.indexOf(ZmCalItem.SERVER_WEEK_DAYS, this.repeatWeeklyDays[0]);
237 				var date = new Date();
238 				date.setDate(date.getDate() - date.getDay() + dayofweek);
239 
240 				var formatter = new AjxMessageFormat(ZmMsg.recurWeeklyEveryWeekday);
241 				every.push(formatter.format(date));
242 			} else {
243 				var weekdays = [];
244 				for (var i = 0; i < this.repeatWeeklyDays.length; i++) {
245 					var dayofweek = AjxUtil.indexOf(ZmCalItem.SERVER_WEEK_DAYS, this.repeatWeeklyDays[i]);
246 					var date = new Date();
247 					date.setDate(date.getDate() - date.getDay() + dayofweek);
248 					weekdays.push(date);
249 				}
250 
251 				var formatter = new AjxMessageFormat(ZmMsg.recurWeeklyEveryNumWeeksDate);
252 				every.push(formatter.format([ this.repeatCustomCount, weekdays, "" ]));
253 			}
254 			break;
255 		}
256 		case ZmRecurrence.MONTHLY: {
257 			if (this.repeatCustomType == "S") {
258 				var count = Number(this.repeatCustomCount);
259 				var date = Number(this.repeatMonthlyDayList[0]);
260 
261 				var formatter = new AjxMessageFormat(ZmMsg.recurMonthlyEveryNumMonthsDate);
262 				every.push(formatter.format([ date, count ]));
263 			} else {
264 				var ordinal = Number(this.repeatCustomOrdinal);
265                 var bysetpos = Number(this.repeatBySetPos);                                
266                 var dayofweek = AjxUtil.indexOf(ZmCalItem.SERVER_WEEK_DAYS, this.repeatCustomDayOfWeek);
267 				var day = new Date();
268 				day.setDate(day.getDate() - day.getDay() + dayofweek);
269                 var count = Number(this.repeatCustomCount);
270 
271                 var days = this.repeatCustomDays.join(",");
272                 var workWeekDays = ZmCalItem.SERVER_WEEK_DAYS.slice(1,6).join(","); 
273                 var weekEndDays = [ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.SUNDAY], ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.SATURDAY]].join(",");
274 
275                 if((ZmCalItem.SERVER_WEEK_DAYS.join(",") == days) || (workWeekDays == days) || (weekEndDays == days)) {
276                     var formatter = new AjxMessageFormat(ZmMsg.recurMonthlyEveryNumMonthsWeekDays);
277                     var dayType = -1;
278                     if(workWeekDays == days) {
279                         dayType = 1;
280                     }else if(weekEndDays == days) {
281                         dayType = 0;
282                     }
283                     every.push(formatter.format([ bysetpos || ordinal, dayType, count ]));
284                 }else {
285                     var day = new Date();
286                     day.setDate(day.getDate() - day.getDay() + dayofweek);
287                     var formatter = new AjxMessageFormat(ZmMsg.recurMonthlyEveryNumMonthsNumDay);
288                     every.push(formatter.format([ bysetpos || ordinal, day, count ]));
289                 }
290 			}
291 			break;
292 		}
293 		case ZmRecurrence.YEARLY: {
294 			if (this.repeatCustomType == "S") {
295 				var month = new Date();
296 				month.setMonth(Number(this.repeatYearlyMonthsList) - 1);
297 				var day = Number(this.repeatCustomMonthDay);
298 
299 				var formatter = new AjxMessageFormat(ZmMsg.recurYearlyEveryDate);
300 				every.push(formatter.format([ month, day ]));
301 			} else {
302 				var ordinal = Number(this.repeatCustomOrdinal);
303                 var bysetpos = Number(this.repeatBySetPos);                
304                 var dayofweek = AjxUtil.indexOf(ZmCalItem.SERVER_WEEK_DAYS, this.repeatCustomDayOfWeek);
305                 var month = new Date();
306                 month.setMonth(Number(this.repeatYearlyMonthsList)-1);
307 
308                 var days = this.repeatCustomDays.join(",");
309                 var workWeekDays = ZmCalItem.SERVER_WEEK_DAYS.slice(1,6).join(",");
310                 var weekEndDays = [ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.SUNDAY], ZmCalItem.SERVER_WEEK_DAYS[AjxDateUtil.SATURDAY]].join(",");
311 
312                 if((ZmCalItem.SERVER_WEEK_DAYS.join(",") == days) || (workWeekDays == days) || (weekEndDays == days)) {
313                     var formatter = new AjxMessageFormat(ZmMsg.recurYearlyEveryMonthWeekDays);
314                     var dayType = -1;
315                     if(workWeekDays == days) {
316                         dayType = 1;
317                     }else if(weekEndDays == days) {
318                         dayType = 0;
319                     }
320                     every.push(formatter.format([ bysetpos || ordinal, dayType, month ]));
321                 }else {
322 
323                     var day = new Date();
324                     day.setDate(day.getDate() - day.getDay() + dayofweek);
325                     var formatter = new AjxMessageFormat(ZmMsg.recurYearlyEveryMonthNumDay);
326                     every.push(formatter.format([ bysetpos || ordinal, day, month ]));
327                 }
328             }
329 			break;
330 		}
331 	}
332 
333 	// start
334 	var start = [];
335 	var formatter = new AjxMessageFormat(ZmMsg.recurStart);
336 	start.push(formatter.format(this._startDate));
337 
338 	// end
339 	var end = [];
340 	switch (this.repeatEndType) {
341 		case "N": {
342 			end.push(ZmMsg.recurEndNone);
343 			break;
344 		}
345 		case "A": {
346 			formatter = new AjxMessageFormat(ZmMsg.recurEndNumber);
347 			end.push(formatter.format(this.repeatEndCount));
348 			break;
349 		}
350 		case "D": {
351 			formatter = new AjxMessageFormat(ZmMsg.recurEndByDate);
352 			end.push(formatter.format(this.repeatEndDate));
353 			break;
354 		}
355 	}
356 
357 	// join all three together
358 	formatter = new AjxMessageFormat(ZmMsg.recurBlurb);
359 	return formatter.format([ every.join(""), start.join(""), end.join("") ]);
360 };

362 ZmRecurrence.prototype.parse =
363 function(recurRules) {
364 	// bug 16513: This array never gets cleaned.
365 	// TODO: Maybe this needs a flag so it doesn't reparse?
366 	this.repeatWeeklyDays = [];
367 
368 	for (var k = 0; k < recurRules.length ; ++k) {
369 		var adds = recurRules[k].add;
370 		if (!adds) continue;
371 
372 		this.repeatYearlyMonthsList = this._startDate.getMonth() + 1;
373 		for (var i = 0; i < adds.length; ++i) {
374 			var rules = adds[i].rule;
375 			if (!rules) continue;
376 
377 			for (var j = 0; j < rules.length; ++j) {
378 				var rule = rules[j];
379 				if (rule.freq) {
380 					this.repeatType = rule.freq.substring(0,3);
381 					if (rule.interval && rule.interval[0].ival) {
382 						this.repeatCustomCount = parseInt(rule.interval[0].ival);
383 						this.repeatCustom = "1";
384 					}
385 				}
386 				if (rule.bymonth) {
387 					this.repeatYearlyMonthsList = rule.bymonth[0].molist;
388 					this.repeatCustom = "1";
389 				}
390 				if (rule.bymonthday) {
391 					if (this.repeatType == ZmRecurrence.YEARLY) {
392 						this.repeatCustomMonthDay = rule.bymonthday[0].modaylist;
393 						this.repeatCustomType = "S";
394 					} else if (this.repeatType == ZmRecurrence.MONTHLY){
395 						this.repeatMonthlyDayList = rule.bymonthday[0].modaylist.split(",");
396 					}
397 					this.repeatCustom = "1";
398 				}
399 				if (rule.byday && rule.byday[0] && rule.byday[0].wkday) {
400 					this.repeatCustom = "1";
401 					var wkday = rule.byday[0].wkday;
402 					if (this.repeatType == ZmRecurrence.WEEKLY || (this.repeatType == ZmRecurrence.DAILY && wkday.length == 5)) {
403 						this.repeatWeekday = this.repeatType == ZmRecurrence.DAILY;
404 						for (var x = 0; x < wkday.length; ++x) {
405 							this.repeatWeeklyDays.push(wkday[x].day);
406 						}
407 					} else {
408 						this.repeatCustomDayOfWeek = wkday[0].day;
409                         var days = [];
410                         for(var i in wkday) {
411                             days.push(wkday[i].day);
412                         }
413                         this.repeatCustomDays = days;                        
414                         this.repeatCustomOrdinal = wkday[0].ordwk;
415                         this.repeatBySetPos  = (rule.bysetpos && (rule.bysetpos.length > 0)) ? rule.bysetpos[0].poslist : null;
416                         //ical sends only ordwk in recurrence rule, we follow outlook behavior in setting repeatbysetpos instead of ordwk
417                         if(this.repeatBySetPos == null && this.repeatCustomOrdinal) {
418                             this.repeatBySetPos  = this.repeatCustomOrdinal; 
419                         }
420                         this.repeatCustomType = "O";
421 					}
422 				}
423 				if (rule.until) {
424 					this.repeatEndType = "D";
425 					this.repeatEndDate = AjxDateUtil.parseServerDateTime(rule.until[0].d);
426 				} else if (rule.count) {
427 					this.repeatEndType = "A";
428 					this.repeatEndCount = rule.count[0].num;
429 				}
430 			}
431 		}
432 	}
433 };

435 ZmRecurrence.prototype.setRecurrenceStartTime =
436 function(startTime) {
437 	
438 	this._startDate.setTime(startTime);
439     this.repeatCustomMonthDay	= this._startDate.getDate();    
440 	
441 	if (this.repeatType == ZmRecurrence.NONE) return;
442 
443 	if (this.repeatCustom != "0")
444 		return;
445 
446  	if (this.repeatType == ZmRecurrence.WEEKLY) {
447 		this.repeatWeeklyDays = [ZmCalItem.SERVER_WEEK_DAYS[this._startDate.getDay()]];
448 	} else if (this.repeatType == ZmRecurrence.MONTHLY) {
449 		this.repeatMonthlyDayList = [this._startDate.getDate()];
450     } else if (this.repeatType == ZmRecurrence.YEARLY) {
451 		this.repeatYearlyMonthsList = this._startDate.getMonth() + 1;	
452 	}
453 };

455 ZmRecurrence.prototype.setRecurrenceRules =
456 function(recurRules, startDate) {
457 
458     if (recurRules)
459         this.parse(recurRules);    
460 
461     if(!startDate) return;
462 
463     if (this.repeatWeeklyDays == null) {
464         this.repeatWeeklyDays = [ZmCalItem.SERVER_WEEK_DAYS[startDate.getDay()]];
465     }
466     if (this.repeatMonthlyDayList == null) {
467         this.repeatMonthlyDayList = [startDate.getDate()];
468     }
469 
470 };

492 ZmRecurrence.prototype.parseExcludeInfo =
493 function(recurInfo) {
494 
495     if (!recurInfo) { return; }
496 
497     for(var i=0; i < recurInfo.length; i++) {
498         var excludeInfo = (recurInfo[i] && recurInfo[i].exclude) ? recurInfo[i].exclude : null;
499         if(!excludeInfo) continue;
500         for(var j=0; j < excludeInfo.length; j++) {
501             var datesInfo = excludeInfo[j].dates;
502             if(datesInfo) this._parseExcludeDates(datesInfo);
503         }
504     }
505 
506 };
507 
508 ZmRecurrence.prototype._parseExcludeDates =
509 function(datesInfo) {
510     for(var j=0; j < datesInfo.length; j++) {
511 
512         var dtval = datesInfo[j].dtval;
513         if(!dtval) continue;
514 
515         for(var k=0; k < dtval.length; k++) {
516             var dinfo = dtval[k];
517             var excludeDate = (dinfo && dinfo.s) ? dinfo.s[0].d : null;
518             if(excludeDate) this.addCancelRecurId(excludeDate);
519         }
520     }
521 };
