 16 /**
 17  * This requires an "owner" which is the object that owns the full set of items, implementing:
 18  * getItemCount() to return the number of items
 19  * getItem(index) to return the item at a given index.
 20  * 
 21  * And optionally implementing
 22  * itemSelectionChanged(item, index, isSelected) which is called
 23  *         for each item that is selected or deselected
 24  * selectionChanged() which is called after a batch of items have
 25  *         been selected or deselected with select()
 26  *
 27  * @private
 28  */
 29 AjxSelectionManager = function(anOwner) {
 30 	this._owner = anOwner;
 31 };
 32 
 33 // -----------------------------------------------------------
 34 // Constants
 35 // -----------------------------------------------------------
 36 
 37 // Actions for select()
 38 AjxSelectionManager.SELECT_ONE_CLEAR_OTHERS = 0;
 39 AjxSelectionManager.TOGGLE_ONE_LEAVE_OTHERS = 1;
 40 AjxSelectionManager.SELECT_TO_ANCHOR = 2;
 41 AjxSelectionManager.DESELECT_ALL = 3;
 42 AjxSelectionManager.SELECT_ALL = 4;
 43 
 44 // -----------------------------------------------------------
 45 // API Methods
 46 // -----------------------------------------------------------
 47 
 48 /**
 49  * returns an AjxVector
 50  */
 51 AjxSelectionManager.prototype.getItems = function() {
 52 	if (this._selectedItems == null) {
 53 		this._selectedItems = this._createItemsCollection();
 54 	}
 55 	return this._selectedItems;
 56 };
 57 
 58 /**
 59  * returns the number of selected items
 60  */	
 61 AjxSelectionManager.prototype.getLength = function() {
 62 	return this.getItems().length;
 63 };
 64 	
 65 /**
 66  * returns the anchor, unless nothing is selected
 67  */
 68 AjxSelectionManager.prototype.getAnchor = function() {
 69 	if (this._anchor == null) {
 70 		var items = this.getItems();
 71 		if (items.length > 0) {
 72 			this._anchor = items[0];
 73 		}
 74 	}
 75 	return this._anchor;
 76 };
 77     
 78 /**
 79  * The cursor probably changes when the users navigates with 
 80  * the keyboard. This returns the item that is currently the cursor,
 81  * and null if nothing is selected.
 82  */
 83 AjxSelectionManager.prototype.getCursor = function() {
 84 	if (this._cursor == null) {
 85 		this._cursor = this.getAnchor();
 86 	}
 87 	return this._cursor;
 88 };
 89     
 90     
 91 /**
 92  * Returns true if the given item is selected.
 93  */
 94 AjxSelectionManager.prototype.isSelected = function(item) {
 95 	return this.getItems().binarySearch(item) != -1;
 96 };
 97     
 98 AjxSelectionManager.prototype.selectOneItem = function(item) {
 99 	this.select(item, AjxSelectionManager.SELECT_ONE_CLEAR_OTHERS);
100 };
101     
102 AjxSelectionManager.prototype.toggleItem = function(item) {
103 	this.select(item, AjxSelectionManager.TOGGLE_ONE_LEAVE_OTHERS);
104 };
105 	
106 AjxSelectionManager.prototype.selectFromAnchorToItem = function(item) {
107 	this.select(item, AjxSelectionManager.SELECT_TO_ANCHOR);
108 };
109     
110 AjxSelectionManager.prototype.deselectAll = function() {
111 	this.select(null, AjxSelectionManager.DESELECT_ALL);
112 };
113 	
114 AjxSelectionManager.prototype.selectAll = function() {
115 	this.select(null, AjxSelectionManager.SELECT_ALL);
116 };
117     
118     
119 /**
120  * This method will notify the owner of any changes by calling
121  * itemSelectionChanged() (if the owner defines it) for each item whose
122  * selection changes and also by calling selectionChanged() (if the
123  * owner defines it) once at the end, if anything changed selection.
124  *
125  */
126 AjxSelectionManager.prototype.select = function(item, action) {
127 	
128 	// Update the anchor and cursor, if necessary
129 	this._setAnchorAndCursor(item, action);
130     
131 	// save off the old set of selected items
132 	var oldItems = this._selectedItems;
133 	var oldItemsCount = (oldItems == null) ? 0 : oldItems.length;
134 	
135 	// create a fresh set of selected items
136 	this._selectedItems = null;
137 	this._selectedItems = this._createItemsCollection();
138 	
139 	// Now update the selection
140 	var itemCount = this._owner.getItemCount();
141 	var needsSort = false;
142 	var selectionChanged = false;
143 	var selecting = false;
144 	for (var i = 0; i < itemCount; ++i) {
145 		var testItem = this._owner.getItem(i);
146 		var oldSelectionExists = this._isItemOldSelection(testItem, oldItems);
147 		var newSelectionExists = oldSelectionExists;
148 		
149 		switch (action) {
150 		case AjxSelectionManager.SELECT_TO_ANCHOR:
151 			if (this._anchor == null) {
152 				// If we have no anchor, let it be the first item
153 				// in the list
154 				this._anchor = testItem;
155 			}
156 			var atEdge = (testItem == this._anchor || testItem == item);
157 			var changed = false;
158 			// mark the beginning of the selection for the iteration
159 			if (!selecting && atEdge) {
160 				selecting = true;
161 				changed = true;
162 			}
163 			newSelectionExists = selecting;
164 			// mark the end of the selection if we're there
165 			if ((!changed || this._anchor == item) 
166 				&& selecting && atEdge) {
167 				selecting = false;
168 			}
169 
170 			break;
171 		case AjxSelectionManager.SELECT_ONE_CLEAR_OTHERS:
172 			newSelectionExists = (testItem == item);
173 			break;
174 		case AjxSelectionManager.TOGGLE_ONE_LEAVE_OTHERS:
175 			if (testItem == item) {
176 				newSelectionExists = !oldSelectionExists ;
177 			}
178 			break;
179 		case AjxSelectionManager.DESELECT_ALL:
180 			newSelectionExists = false;
181 			break;
182 		case AjxSelectionManager.SELECT_ALL:
183 			newSelectionExists = true;
184 			break;
185 		}
186 
187 		if (newSelectionExists) {
188 			this._selectedItems.add(testItem);
189 			needsSort = (this._selectedItems.length > 1);
190 		}
191 
192 		if ( newSelectionExists != oldSelectionExists) {
193 			// Something changed so notify the owner.
194 			if (this._owner.itemSelectionChanged != null) {
195 				this._owner.itemSelectionChanged(testItem, 
196 												 i, newSelectionExists);
197 			}
198 			selectionChanged = true;
199 		}
200 	}
201 	selectionChanged = selectionChanged || (oldItemsCount != 
202 											this._selectedItems.length);
203 
204 	if (needsSort) this._selectedItems.sort();
205 	
206 	if (selectionChanged && this._owner.selectionChanged != null) {
207 		this._owner.selectionChanged(item);
208 	}
209 };

211 /**
212  * Remove an item from the selection managers selected items
213  * collection if it exists.
214  */
215 AjxSelectionManager.prototype.removeItem = function(item) {
216 	if (this._selectedItems) {
217 		var index = this._selectedItems.binarySearch(item);
218 		if (index > -1) this._selectedItems.removeAt(index);
219 	}
220 };

222 // -----------------------------------------------------------
223 // Internal Methods
224 // -----------------------------------------------------------
225 	
226 /**
227  * Creates an array suitable for use as the sorted list of selected
228  * items and returns it.
229  */
230 AjxSelectionManager.prototype._createItemsCollection = function() {
231 	return new AjxVector();
232 };
233 
234 AjxSelectionManager.prototype._isItemOldSelection = function (testItem, oldItems) {
235 	var ret = false;
236 	if (oldItems) {
237 		var oldSelectionIndex = oldItems.binarySearch(testItem);
238 		if (oldSelectionIndex > -1) {
239 			oldItems.removeAt(oldSelectionIndex);
240 		}
241 		ret = (oldSelectionIndex != -1);
242 	}
243 	return ret;
244 };
245 
246 AjxSelectionManager.prototype._setAnchorAndCursor = function (item, action) {
247 	switch (action) {
248 	case AjxSelectionManager.SELECT_TO_ANCHOR:
249 		this._cursor = item;
250 		break;
251 	case AjxSelectionManager.SELECT_ONE_CLEAR_OTHERS:		
252 		this._anchor = item;
253 		this._cursor = item;
254 		break;
255 	case AjxSelectionManager.TOGGLE_ONE_LEAVE_OTHERS:
256 		this._anchor = item;
257 		this._cursor = item;
258 		break;
259 	case AjxSelectionManager.DESELECT_ALL:
260 		this._anchor = null;
261 		this._cursor = null;
262 		break;
263 	case AjxSelectionManager.SELECT_ALL:
264 		return;
265 	}
266 };
