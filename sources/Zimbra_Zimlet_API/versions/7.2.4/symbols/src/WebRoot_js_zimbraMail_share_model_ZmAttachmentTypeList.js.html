 16 /**
 17  * @overview
 18  * 
 19  * This file defines a list of attachment types.
 20  *
 21  */
 22 
 23 /**
 24  * Creates an attachment type list
 25  * @class
 26  * This class represents attachment types.
 27  * 
 28  * @extends	ZmModel
 29  */
 30 ZmAttachmentTypeList = function() {
 31 	ZmModel.call(this, ZmEvent.S_ATT);
 32 };
 33 
 34 ZmAttachmentTypeList.prototype = new ZmModel;
 35 ZmAttachmentTypeList.prototype.constructor = ZmAttachmentTypeList;
 36 
 37 /**
 38  * Returns a string representation of the object.
 39  * 
 40  * @return		{String}		a string representation of the object
 41  */
 42 ZmAttachmentTypeList.prototype.toString = 
 43 function() {
 44 	return "ZmAttachmentTypeList";
 45 };
 46 
 47 /**
 48  * Gets the attachments.
 49  * 
 50  * @return	{Array}	an array of attachments
 51  */
 52 ZmAttachmentTypeList.prototype.getAttachments =
 53 function() {
 54 	return this._attachments;
 55 };
 56 
 57 /**
 58  * Compares attachment type lists by description.
 59  * 
 60  * @param	{ZmAttachmentTypeList}	a			the first entry
 61  * @param	{ZmAttachmentTypeList}	b			the first entry
 62  * @return	{int}	0 if the entries match; 1 if "a" is before "b"; -1 if "b" is before "a"
 63  */
 64 ZmAttachmentTypeList.compareEntry = 
 65 function(a,b) {
 66 	if (a.desc.toLowerCase() < b.desc.toLowerCase())
 67 		return -1;
 68 	if (a.desc.toLowerCase() > b.desc.toLowerCase())
 69 		return 1;
 70 	else
 71 		return 0;
 72 };
 73 
 74 /**
 75  * Loads the attachments.
 76  * 
 77  * @param	{AjxCallback}	callback		the callback to call after load
 78  */
 79 ZmAttachmentTypeList.prototype.load =
 80 function(callback) {
 81 	this._attachments = new Array();
 82 
 83 	var soapDoc = AjxSoapDoc.create("BrowseRequest", "urn:zimbraMail");
 84 	soapDoc.getMethod().setAttribute("browseBy", "attachments");
 85 
 86 	var respCallback = new AjxCallback(this, this._handleResponseLoad, callback);
 87 	appCtxt.getAppController().sendRequest({soapDoc: soapDoc, asyncMode: true, callback: respCallback});
 88 };
 89 
 90 /**
 91  * @private
 92  */
 93 ZmAttachmentTypeList.prototype._handleResponseLoad =
 94 function(callback, result) {
 95 	var att = result.getResponse().BrowseResponse.bd;
 96 	if (att) {
 97 		for (var i = 0; i < att.length; i++) {
 98 			var type = att[i]._content;
 99 			if (!ZmMimeTable.isIgnored(type) && (type.indexOf("/") != -1 || type == "image"))
100 				this._attachments.push(ZmMimeTable.getInfo(type, true));
101 		}
102 		this._attachments.sort(ZmAttachmentTypeList.compareEntry);
103 	}
104 	
105 	if (callback) callback.run(result);
106 };
