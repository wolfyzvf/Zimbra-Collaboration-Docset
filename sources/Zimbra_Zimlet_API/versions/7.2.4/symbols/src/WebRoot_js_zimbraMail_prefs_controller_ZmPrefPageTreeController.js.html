 16 /**
 17  * Creates a preferences page tree controller.
 18  * @class
 19  * This class represents the preferences page tree controller.
 20  * 
 21  * @extends		ZmTreeController
 22  */
 23 ZmPrefPageTreeController = function() {
 24 	ZmTreeController.apply(this, arguments);
 25 };
 26 ZmPrefPageTreeController.prototype = new ZmTreeController;
 27 ZmPrefPageTreeController.prototype.constructor = ZmPrefPageTreeController;
 28 
 29 ZmPrefPageTreeController.prototype.toString =
 30 function() {
 31 	return "ZmPrefPageTreeController";
 32 };

 38 ZmPrefPageTreeController.prototype.show =
 39 function(params) {
 40 	// populate tree
 41 	var app = appCtxt.getApp(ZmApp.PREFERENCES);
 42 	var view = app.getPrefController().getPrefsView();
 43 	var account = params.account;
 44 
 45 	if (appCtxt.multiAccounts && !this._currentAccount) {
 46 		this._currentAccount = account;
 47 	}
 48 
 49 	var tree = new ZmTree(ZmOrganizer.PREF_PAGE);
 50 	var root = tree.root = new ZmPrefPage({id:ZmId.getPrefPageId(0), name:"", tree:tree});
 51 	appCtxt.cacheSet(root.id, root);
 52 
 53 	// create pseudo-organizers
 54 	var organizers = [];
 55 	var count = view.getNumTabs();
 56 	for (var i = 0; i < count; i++) {
 57 		var tabKey = i+1;
 58 		var name = view.getTabTitle(tabKey);
 59 		var section = view.getSectionForTab(tabKey);
 60 		if (!account || this._showSection(account, section.id)) {
 61 			// for multi-account mbox, child accounts only show a select few pref options
 62 			var organizer = ZmPrefPage.createFromSection(section);
 63 			organizer.pageId = tabKey;
 64 			organizer.account = account;
 65 			organizers.push(organizer);
 66 		}
 67 	}
 68 
 69 	// order pages
 70 	for (var i = 0; i < organizers.length; i++) {
 71 		var organizer = organizers[i];
 72 		var section = view.getSectionForTab(organizer.pageId);
 73 		var parentId = section.parentId;
 74 		if (appCtxt.isOffline &&
 75 			(section.id == "SIGNATURES" ||
 76 			 section.id == "ACCOUNTS" ||
 77 			 section.id == "COMPOSING" ||
 78 			 section.id == "FILTERS"))
 79 		{
 80 			parentId = null;
 81 		}
 82 		var parent = (parentId && tree.getById(ZmId.getPrefPageId(parentId))) || root;
 83 		parent.children.add(organizer);
 84 
 85 		organizer.parent = parent;
 86 		organizer.icon = section.icon || parent.getIcon();
 87 	}
 88 
 89 	appCtxt.setTree(tree.type, tree, account);
 90 
 91 	// setup tree view
 92 	var treeView = ZmTreeController.prototype.show.apply(this, arguments);
 93 
 94 	if (!appCtxt.multiAccounts || (appCtxt.multiAccounts && account.isMain)) {
 95 		var page1 = root.children.get(0);
 96 		if (page1) {
 97 			treeView.setSelected(page1, true);
 98 		}
 99 	}
100 	if (!appCtxt.isOffline) {
101 		var hi = treeView.getHeaderItem();
102 		if (hi) {
103 			hi.setExpanded(true, true);
104 		}
105 	}
106 	treeView.addSelectionListener(this._handleTreeItemSelection.bind(this, view));
107 	treeView.addPreSelectionListener(this._handleTreeItemPreSelection.bind(this, view));
108 
109 	return treeView;
110 };
111 
112 ZmPrefPageTreeController.prototype._showSection =
113 function(account, sectionId) {
114 
115 	if (appCtxt.isOffline) {
116 		if (sectionId == "MOBILE") {
117 			return false;
118 		}
119 
120 		if (account.isMain) {
121 			if (sectionId == "FILTERS" ||
122 				sectionId == "SHARING" ||
123 				sectionId == "SIGNATURES" ||
124 				sectionId == "ACCOUNTS" ||
125                 sectionId == "NOTIFICATIONS" ||
126                 sectionId == "TRUSTED_ADDR")
127 			{
128 				return false;
129 			}
130 		}
131 		else {
132 			if (sectionId == "COMPOSING") {
133 				return false;
134 			}
135 			if (!account.isZimbraAccount &&
136 				(sectionId == "MAIL" ||
137 				 sectionId == "SHARING" ||
138 				 sectionId == "CALENDAR" ||
139                  sectionId == "NOTIFICATIONS" ||
140                  sectionId == "TRUSTED_ADDR" ))
141 			{
142 				return false;
143 			}
144 		}
145 
146 	}
147 
148 	return (account.isMain ||
149 			(!account.isMain && (sectionId != "GENERAL" &&
150 								 sectionId != "SHORTCUTS" &&
151 								 sectionId != "PREF_ZIMLETS" &&
152                                  sectionId != "BACKUP" &&
153 								 sectionId != "COMPOSING")
154 			));
155 };

175 ZmPrefPageTreeController.prototype._handleTreeItemSelection =
176 function(tabView, ev) {
177 	if (ev.detail != DwtTree.ITEM_SELECTED || ev.handled) { return; }
178 
179 	var organizer = ev.item.getData(Dwt.KEY_OBJECT);
180 	tabView.switchToTab(organizer && organizer.pageId);
181 };
182 
183 ZmPrefPageTreeController.prototype._handleTreeItemPreSelection =
184 function(tabView, ev) {
185 
186 	var organizer = ev.item.getData(Dwt.KEY_OBJECT);
187 
188 	if (organizer.nId != "PREF_PAGE_FILTERS") {
189 		return true;
190 	}
191 
192 	var serverVersion = appCtxt.get(ZmSetting.OFFLINE_REMOTE_SERVER_VERSION, null, organizer.account);
193 	if (serverVersion && serverVersion.substr(0, 1) > 7) {
194 		var dlg = appCtxt.getMsgDialog();
195 		dlg.reset();
196 		dlg.setMessage(ZmMsg.filtersDisabled, DwtMessageDialog.INFO_STYLE);
197 		dlg.popup();
198 		return false;
199 	}
200 
201 	return true;
202 
203 };

206 ZmPrefPageTreeController.prototype._handleMultiAccountItemSelection =
207 function(ev, overview, treeItem, item) {
208 	if (this._currentAccount != item.account) {
209 		var prefsController = appCtxt.getApp(ZmApp.PREFERENCES).getPrefController();
210 		var prefsView = prefsController.getPrefsView();
211 
212 		this._currentAccount = prefsController._activeAccount = item.account;
213 
214 		if (prefsView.getChangedPrefs(true, true)) {
215 			ev.handled = true;
216 
217 			var dialog = appCtxt.getYesNoCancelMsgDialog();
218 			var args = [ev, overview, treeItem, item, prefsController, dialog];
219 			var yesCallback = new AjxCallback(this, this._savePrefsYes, args);
220 			var noCallback = new AjxCallback(this, this._savePrefsNo, args);
221 			var cancelCallback = new AjxCallback(this, this._savePrefsCancel, dialog);
222 
223 			dialog.reset();
224 			dialog.setMessage(ZmMsg.confirmExitPreferencesChangeAcct, DwtMessageDialog.WARNING_STYLE);
225 			dialog.registerCallback(DwtDialog.YES_BUTTON, yesCallback, this);
226 			dialog.registerCallback(DwtDialog.NO_BUTTON, noCallback, this);
227 			dialog.registerCallback(DwtDialog.CANCEL_BUTTON, cancelCallback, this);
228 			dialog.popup();
229 			return;
230 		}
231 		else {
232 			prefsView.resetOnAccountChange();
233 		}
234 	}
235 
236 	ev.handled = false;
237 	this._handleItemSelection(ev, overview, treeItem, item);
238 };
239 
240 ZmPrefPageTreeController.prototype._savePrefsYes =
241 function(ev, overview, treeItem, item, prefsController, dialog) {
242 	dialog.popdown();
243 
244 	var callback = new AjxCallback(this, this._continueTreeItemSelection, [ev, overview, treeItem, item, prefsController]);
245 	prefsController.save(callback, true);
246 };
247 
248 ZmPrefPageTreeController.prototype._savePrefsNo =
249 function(ev, overview, treeItem, item, prefsController, dialog) {
250 	dialog.popdown();
251 
252 	prefsController.getPrefsView().reset();
253 	this._continueTreeItemSelection(ev, overview, treeItem, item, prefsController);
254 };
255 
256 ZmPrefPageTreeController.prototype._savePrefsCancel =
257 function(dialog) {
258 	dialog.popdown();
259 };
260 
261 ZmPrefPageTreeController.prototype._continueTreeItemSelection =
262 function(ev, overview, treeItem, item, prefsController) {
263 	prefsController.getPrefsView().resetOnAccountChange();
264 
265 	this._handleItemSelection(ev, overview, treeItem, item);
266 	prefsController.getPrefsView().switchToTab(item.pageId);
267 };
