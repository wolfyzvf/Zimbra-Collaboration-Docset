 19 	DwtComposite.call(this, {parent:parent, className:className, posStyle:posStyle});

 24 	this._setMouseEventHdlrs();
 25 	this.setCursor("default");

 27 	this._listenerMouseOver = new AjxListener(this, ZmCalBaseView.prototype._mouseOverListener);
 28 	this._listenerMouseOut = new AjxListener(this, ZmCalBaseView.prototype._mouseOutListener);
 29 	this._listenerDoubleClick = new AjxListener(this, ZmCalBaseView.prototype._doubleClickListener);
 30 	this._listenerMouseDown = new AjxListener(this, ZmCalBaseView.prototype._mouseDownListener);
 31 	this._listenerMouseUp = new AjxListener(this, ZmCalBaseView.prototype._mouseUpListener);
 32 	this._listenerMouseMove = new AjxListener(this, ZmCalBaseView.prototype._mouseMoveListener);
 33 	this.addListener(DwtEvent.ONMOUSEOVER, this._listenerMouseOver);
 34 	this.addListener(DwtEvent.ONMOUSEOUT, this._listenerMouseOut);
 35 	this.addListener(DwtEvent.ONDBLCLICK, this._listenerDoubleClick);
 36 	this.addListener(DwtEvent.ONMOUSEDOWN, this._listenerMouseDown);
 37 	this.addListener(DwtEvent.ONMOUSEUP, this._listenerMouseUp);
 38 	this.addListener(DwtEvent.ONMOUSEMOVE, this._listenerMouseMove);

 40 	this._controller = controller;
 41 	this.view = view;	
 42 	this._evtMgr = new AjxEventMgr();	 
 43 	this._selectedItems = new AjxVector();
 44 	this._selEv = new DwtSelectionEvent(true);
 45 	this._actionEv = new DwtListViewActionEvent(true);	
 46 
 47 	this._normalClass = "appt";

 49 	this._disabledSelectedClass = [this._selectedClass, DwtCssStyle.DISABLED].join("-");

 51 	// the key is the HTML ID of the item's associated DIV; the value is an object

 58 	this._timeRangeEnd = 0;

 88 ZmCalBaseView._getColors = function(color) {
 89 	// generate header and body colors
 90     color = color || ZmOrganizer.COLOR_VALUES[ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.CALENDAR]];
 91 	var hs = { bgcolor: AjxColor.darken(color, ZmCalBaseView.headerColorDelta) };
 92 	var hd = { bgcolor: AjxColor.deepen(hs.bgcolor, ZmCalBaseView.deepenColorAdjustment) };
 93 	var bs = { bgcolor: AjxColor.lighten(color, ZmCalBaseView.bodyColorDelta)  };
 94 	var bd = { bgcolor: AjxColor.deepen(bs.bgcolor, ZmCalBaseView.deepenColorAdjustment) };
 95 
 96 	// ensure enough difference between background and deeper colors
 97 	var cs = AjxColor.components(hs.bgcolor);
 98 	var cd = AjxColor.components(hd.bgcolor);
 99 	var ss = cs[0]+cs[1]+cs[2];
100 	var sd = cd[0]+cd[1]+cd[2];
101 	if (ss/sd > 1 - ZmCalBaseView.deepenThreshold) {
102 		hs.bgcolor = AjxColor.lighten(hd.bgcolor, ZmCalBaseView.deepenThreshold);
103 		bs.bgcolor = AjxColor.lighten(bd.bgcolor, ZmCalBaseView.deepenThreshold);
104 	}
105 
106 	// use light text color for dark backgrounds
107 	hs.color = ZmCalBaseView._isDark(hs.bgcolor) && "#ffffff";
108 	hd.color = ZmCalBaseView._isDark(hd.bgcolor) && "#ffffff";
109 	bs.color = ZmCalBaseView._isDark(bs.bgcolor) && "#ffffff";
110 	bd.color = ZmCalBaseView._isDark(bd.bgcolor) && "#ffffff";
111 
112 	return { standard: { header: hs, body: bs }, deeper: { header: hd, body: bd } };
113 };

124 ZmCalBaseView._isDark =

127 	return c[0]+c[1]+c[2] < ZmCalBaseView.darkThreshold;

146 ZmCalBaseView.parseWorkingHours =
147 function(wHrsString) {
148     if(wHrsString === 0) {
149         return [];
150     }
151 	var userTimeZone = appCtxt.get(ZmSetting.DEFAULT_TIMEZONE),
152         currentTimeZone = AjxTimezone.getServerId(AjxTimezone.DEFAULT),
153         wHrsPerDay = wHrsString.split(','),
154         i,
155         wHrs = [],
156         wDay,
157         w,
158         offset1,
159         offset2,
160         hourMinOffset = 0,
161         idx,
162         startDate = new Date(),
163         endDate = new Date(),
164         hourMin,
165         startDayIdx,
166         endDayIdx,
167         curDayIdx = endDate.getDay(),
168         tf = new AjxDateFormat(ZmCalBaseView.WORK_HOURS_TIME_FORMAT);
169 
170     //Helper inner functions, these functions takes the advantage of the fact that wHrs is available in local scope
171     function isWorkingDay(idx) {
172         return wHrs[idx] && wHrs[idx].isWorkingDay;
173     }
174 
175     function setWorkingDay(idx, startTime, endTime) {
176         if(isWorkingDay(idx)) {
177             addWorkingTime(idx, startTime, endTime);
178         }
179         else {
180             addWorkingDay(idx, startTime, endTime);
181         }
182     }
183 
184     function setNonWorkingDay(idx) {
185         wHrs[idx] = {};
186         wHrs[idx].isWorkingDay = false;
187         wHrs[idx].startTime = ["0000"];
188         wHrs[idx].endTime = ["0000"];
189     }
190 
191     function addWorkingDay(idx, startTime, endTime) {
192         wHrs[idx] = {};
193         wHrs[idx].isWorkingDay = true;
194         wHrs[idx].startTime = [startTime];
195         wHrs[idx].endTime = [endTime];
196     }
197 
198     function addWorkingTime(idx, startTime, endTime) {
199         wHrs[idx].startTime.push(startTime);
200         wHrs[idx].endTime.push(endTime);
201     }
202     
203     if(userTimeZone != currentTimeZone) {
204         offset1 = AjxTimezone.getOffset(AjxTimezone.getClientId(currentTimeZone), startDate);
205         offset2 = AjxTimezone.getOffset(AjxTimezone.getClientId(userTimeZone), startDate);
206         hourMinOffset = offset2 - offset1;
207     }
208     for(i=0; i<wHrsPerDay.length; i++) {
209         wDay = wHrsPerDay[i].split(':');
210         w = {};
211         idx = wDay[0]-1;
212         if(wDay[1] === "N") {
213             if(!isWorkingDay(idx)) {
214                 setNonWorkingDay(idx);
215             }
216             continue;
217         }
218 
219         if(hourMinOffset) {
220             endDate = new Date();
221             startDate = new Date();
222             
223             endDate.setHours(wDay[3]/100, wDay[3]%100);
224             hourMin = endDate.getHours() * 60 + endDate.getMinutes() - hourMinOffset;
225             endDate.setHours(hourMin/60, hourMin%60);
226             endDayIdx = endDate.getDay();
227 
228             startDate.setHours(wDay[2]/100, wDay[2]%100);
229             hourMin = startDate.getHours() * 60 + startDate.getMinutes() - hourMinOffset;
230             startDate.setHours(hourMin/60, hourMin%60);
231             startDayIdx = startDate.getDay();
232 
233             if(startDayIdx == curDayIdx && endDayIdx == curDayIdx) {
234                 //Case 1 working time starts current day and ends on the current day -- IDEAL one :)
235                 setWorkingDay(idx, tf.format(startDate), tf.format(endDate));
236             }
237             else if((endDayIdx == 0 && startDayIdx == 6) ||
238                     (startDayIdx < curDayIdx  && endDayIdx == curDayIdx)) {
239                 //Case 2 working time starts prev day and ends on current day
240                 startDayIdx = idx-1;
241                 if(startDayIdx < 0) {
242                    startDayIdx = 6;
243                 }
244                 setWorkingDay(startDayIdx, tf.format(startDate), "2400");
245                 setWorkingDay(idx, "0000", tf.format(endDate));
246             }
247             else if((startDayIdx == 6 && endDayIdx == 0) || 
248                     (startDayIdx == curDayIdx  && endDayIdx > curDayIdx)) {
249                 //Case 3 working time starts current day and ends on next day
250                 endDayIdx = idx+1;
251                 if(endDayIdx > 6) {
252                    endDayIdx = 0; 
253                 }
254                 setWorkingDay(endDayIdx, "0000", tf.format(endDate));
255                 setWorkingDay(idx, tf.format(startDate), "2400");
256             }
257             else if(startDayIdx < curDayIdx &&
258                     endDayIdx < curDayIdx &&
259                     startDayIdx == endDayIdx) {
260                 //EDGE CASE 1: working time starts and ends on the prev day
261                 startDayIdx = idx-1;
262                 setWorkingDay(startDayIdx, tf.format(startDate), tf.format(endDate));
263                 if(!isWorkingDay(idx)) {
264                     setNonWorkingDay(idx);
265                 }
266             }
267 
268             else if(startDayIdx > curDayIdx &&
269                     endDayIdx > curDayIdx &&
270                     startDayIdx == endDayIdx) {
271                 //EDGE CASE 2: working time starts and ends on the next day
272                 endDayIdx = idx+1;
273                 setWorkingDay(endDayIdx, tf.format(startDate), tf.format(endDate));
274                 if(!isWorkingDay(idx)) {
275                     setNonWorkingDay(idx);
276                 }
277             }            
278         }
279         else {
280             //There is no timezone diff, client and server are in the same timezone
281             setWorkingDay(idx, wDay[2], wDay[3]);
282         }
283 
284     }
285     return wHrs;
286 };

403 ZmCalBaseView.prototype._mouseOutListener = 
404 function(ev) {
405 	var div = this.getTargetItemDiv(ev);
406 	if (!div) { return; }
407 
408 	// NOTE: The DwtListView handles the mouse events on the list items
409 	//		 that have associated tooltip text. Therefore, we must
410 	//		 explicitly null out the tooltip content whenever we handle
411 	//		 a mouse out event. This will prevent the tooltip from
412 	//		 being displayed when we re-enter the listview even though
413 	//		 we're not over a list item.
414 	if (this._getItemData(div, "type") == ZmCalBaseView.TYPE_APPT) {
415 		this.setToolTipContent(null);
416 	}
417 	this._mouseOutAction(ev, div);
418 };

442 	if (this._isReadOnly) { return; }

477 ZmCalBaseView.prototype._doubleClickListener =
478 function(ev) {
479 	var div = this.getTargetItemDiv(ev);
480 	
481 	if (!div) { return;	}
482 		
483 	if (this._getItemData(div, "type") == ZmCalBaseView.TYPE_APPT) {
484 		if (this._evtMgr.isListenerRegistered(DwtEvent.SELECTION)) {
485 			DwtUiEvent.copy(this._selEv, ev);
486 			var item = this.getItemFromElement(div);
487             var orig = item.getOrig();
488             item = orig && orig.isMultiDay() ? orig : item;
489 			this._selEv.item = item;
490 			this._selEv.detail = DwtListView.ITEM_DBL_CLICKED;
491 			this._evtMgr.notifyListeners(DwtEvent.SELECTION, this._selEv);
492 		}
493 	}
494 	return this._doubleClickAction(ev, div);
495 };

497 ZmCalBaseView.prototype._itemClicked =
498 function(clickedEl, ev) {
499 	var i;
500 	var a = this._selectedItems.getArray();
501 	var numSelectedItems = this._selectedItems.size();
502 
503 	// is this element currently in the selected items list?
504 	var bContained = this._selectedItems.contains(clickedEl);
505 
506 	DwtUiEvent.copy(this._selEv, ev);
507 	var item = this._selEv.item = this.getItemFromElement(clickedEl);
508 	var type = this._getItemData(clickedEl, "type");
509 
510 	if (ev.shiftKey && bContained) {
511 		this._selectedItems.remove(clickedEl);
512 		clickedEl.className = this._getStyle(type);
513 		this._selEv.detail = DwtListView.ITEM_DESELECTED;
514 		this._evtMgr.notifyListeners(DwtEvent.SELECTION, this._selEv);
515 	} else if (!bContained) {
516 		// clear out old left click selection(s)
517 		for (i = 0; i < numSelectedItems; i++) {
518 			a[i].className = this._getStyle(type);
519 		}
520 		this._selectedItems.removeAll();
521 
522 		// save new left click selection
523 		this._selectedItems.add(clickedEl);
524 		clickedEl.className = this._getStyle(type, true, !this.getEnabled(), item);
525 		this._selEv.detail = DwtListView.ITEM_SELECTED;
526 		this._evtMgr.notifyListeners(DwtEvent.SELECTION, this._selEv);
527 	}
528 
529 	if (ev.button == DwtMouseEvent.RIGHT) {
530 		DwtUiEvent.copy(this._actionEv, ev);
531 		this._actionEv.item = item;
532 		this._evtMgr.notifyListeners(DwtEvent.ACTION, this._actionEv);
533 	}
534 };

540 ZmCalBaseView.prototype.setSelection =
541 function(item, skipNotify) {
542 	var el = this._getElFromItem(item);
543 	if (el) {
544 		var i;
545 		var a = this._selectedItems.getArray();
546 		var sz = this._selectedItems.size();
547 		for (i = 0; i < sz; i++) {
548 			a[i].className = this._getStyle(this._getItemData(a[i], "type"));
549 		}
550 		this._selectedItems.removeAll();
551 		this._selectedItems.add(el);
552 		el.className = this._getStyle(this._getItemData(el, "type"), true, !this.getEnabled(), item);
553 		if (!skipNotify && this._evtMgr.isListenerRegistered(DwtEvent.SELECTION)) {
554 			var selEv = new DwtSelectionEvent(true);
555 			selEv.button = DwtMouseEvent.LEFT;
556 			selEv.target = el;
557 			selEv.item = item;
558 			selEv.detail = DwtListView.ITEM_SELECTED;
559 			this._evtMgr.notifyListeners(DwtEvent.SELECTION, selEv);
560 		}	
561 	}

673 ZmCalBaseView.prototype.setDate =
674 function(date, duration, roll) {
675 	this._duration = duration;
676 	this._date = new Date(date.getTime());
677 	var d = new Date(date.getTime());
678 	d.setHours(0, 0, 0, 0);
679 	var t = d.getTime();
680 	if (roll || t < this._timeRangeStart || t >= this._timeRangeEnd) {
681 		this._resetList();
682 		this._updateRange();		
683 		this._dateUpdate(true);
684 		this._updateTitle();
685 		
686 		// Notify any listeners
687 		if (this.isListenerRegistered(DwtEvent.DATE_RANGE)) {
688 			if (!this._dateRangeEvent)
689 				this._dateRangeEvent = new DwtDateRangeEvent(true);
690 			this._dateRangeEvent.item = this;
691 			this._dateRangeEvent.start = new Date(this._timeRangeStart);
692 			this._dateRangeEvent.end = new Date(this._timeRangeEnd);
693 			this.notifyListeners(DwtEvent.DATE_RANGE, this._dateRangeEvent);
694 		}
695 	} else {
696 		this._dateUpdate(false);
697 	}
698 };

741 ZmCalBaseView.prototype.set = 
742 function(list) {
743 	this._preSet();
744 	this._selectedItems.removeAll();
745 	var newList = list;
746 	if(list && (list == this._list)) {
747 		newList = list.clone();
748 	}
749 	this._resetList();
750 	this._list = newList;	
751 	if (list) {
752 		var size = list.size();
753 		if (size != 0) {
754 			for (var i=0; i < size; i++) {
755 				var ao = list.get(i);
756 				this.addAppt(ao);
757 			}
758 		}
759 	}
760 	this._postSet(list);
761 };

816 ZmCalBaseView.prototype._resetList =
817 function() {
818 	var list = this.getList();
819 	var size = list ? list.size() : 0;
820 	if (size == 0) return;
821 
822 	for (var i=0; i < size; i++) {
823 		var ao = list.get(i);
824 		var id = this._getItemId(ao);
825 		var appt = document.getElementById(id);
826 		if (appt) {
827 			appt.parentNode.removeChild(appt);
828 			this._data[id] = null;
829 		}
830 	}
831 	list.removeAll();
832 	this.removeAll();
833 };
