 27 DwtPropertyEditor = function(parent, useDwtInputField, className, posStyle, deferred) {
 28 	if (arguments.length > 0) {
 29 		if (!className)
 30 			className = "DwtPropertyEditor";
 31 		DwtComposite.call(this, {parent:parent, className:className, posStyle:posStyle, deferred:deferred});
 32 		this._useDwtInputField = useDwtInputField != null ? useDwtInputField : true;
 33 		this._schema = null;
 34 		this._init();
 35 	}
 36 };

 77 DwtPropertyEditor.prototype._onMouseDown = function(event) {
 78 	var target = event.target;
 79 	var tag = target.tagName.toLowerCase();
 80 	if (tag == "input") {
 81 		event._stopPropagation = false;
 82 		event._returnValue = true;
 83 		return true;
 84 	}
 85 	if (this._currentInputField && !this._currentInputField.onblur()) {
 86 		event._stopPropagation = true;
 87 		event._returnValue = false;
 88 		return false;
 89 	}
 90 	try {
 91 		while (target && tag != "tr") {
 92 			target = target.parentNode;
 93 			tag = target.tagName.toLowerCase();
 94 		}
 95 		if (target && target.__msh_doMouseDown)
 96 			target.__msh_doMouseDown(event);
 97 	} catch(ex) {};
 98 };

100 /**
101  * Call this function to retrieve an object that contains all properties,
102  * indexed by name.  Any "struct" property will map to an object that contains
103  * its child properties.
104  *
105  * For the sample schema below (see comments on initProperties()), we would
106  * retrieve an object like this (dots represent the edited value, or the
107  * initial value if the property wasn't modified):
108  *
109  *  {
110  *    userName : ... ,
111  *    address  : {
112  *                  street   : ... ,
113  *                  country  : ... ,
114  *               },
115  *    age      : ... ,
116  *    birthday : ...
117  *  }
118  */
119 DwtPropertyEditor.prototype.getProperties = function() {
120 	if (this._currentInputField)
121 		// make sure we get the value
122 		this._currentInputField.onblur();
123 	function rec(schema) {
124 		var prop = {}, tmp, n = schema.length;
125 		for (var i = 0; i < n; ++i) {
126 			tmp = schema[i];
127 			if (tmp.type == "struct")
128 				prop[tmp.name] = rec(tmp.children);
129 			else
130 				prop[tmp.name] = tmp.value;
131 		}
132 		return prop;
133 	};
134 	return rec(this._schema);
135 };
136 
137 DwtPropertyEditor.prototype.validateData = function() {
138 	var valid = true;
139 	function rec(schema) {
140 		var tmp, n = schema.length;
141 		for (var i = 0; i < n; ++i) {
142 			tmp = schema[i];
143 			if (tmp.type == "struct")
144 				rec(tmp.children);
145 			else if (!tmp._validate())
146 				valid = false;
147 		}
148 	};
149 	rec(this._schema);
150 	return valid;
151 };

153 /** This function will initialize the Property Editor with a given schema and
154  * property set.
155  *
156  *  @param schema - declares which properties/types are allowed; see below
157  *  @param parent - parent schema, for subproperties
158  *
159  * "schema" is an object that maps property names to property declaration.
160  * Here's an example of what I have in mind:
161  *
162  *  [
163  *    {
164  *      label        : "User Name",
165  *      id           : "userName",
166  *      type         : "string",
167  *      value        : "",
168  *      minLength    : 4,
169  *      maxLength    : 8,
170  *      mustMatch    : /^[a-z][a-z0-9_]+$/i,
171  *      mustNotMatch : /^(admin|root|guest)$/i
172  *    },
173  *    {
174  *      label     : "Address",
175  *      id        : "address",
176  *      type      : "struct",
177  *      children  : [ // this is a nested schema definition
178  *              { label : "Street", id: "street", type: "string" },
179  *              { label  : "Country",
180  *                id     : "country",
181  *                type   : "enum",
182  *                item : [
183  							{ label : "US", value : "US" },
184  							{ label : "UK", value : "UK" },
185  							{ label : "KR", value : "KR" }
186  						 ] 
187  				}
188  *      ]
189  *    },
190  *    {
191  *      label     : "Age",
192  *      id        : "age",
193  *      type      : "integer",
194  *      minValue  : 18,
195  *      maxValue  : 80
196  *    },
197  *    {
198  *      label     : "Birthday",
199  *      id        : "birthday",
200  *      type      : "date",
201  *      minValue  : "YYYY/MM/DD"  // can we restrict the DwtCalendar?
202  *    }
203  *  ]
204  *
205  * The types we will support for now are:
206  *
207  *   - "number" / "integer" : Allows floating point numbers or integers only.
208  *     Properties: "minValue", "maxValue".
209  *
210  *   - "string" : Allows any string to be inserted.  "minLength", "maxLength",
211  *     "mustMatch", "mustNotMatch".
212  *
213  *   - "password" : Same as "string", only it's not displayed.
214  *
215  *   - "enum" : One of the strings listed in the value.
216  *
217  *   - "boolean" : Checkbox that yields string value of "true" and "false".
218  *
219  *   - "struct" : Composite property; doesn't have a value by itself, but has
220  *     child properties (the "children" array) that are defined in the same way
221  *     as a toplevel property.
222  *
223  * All types except "struct" will allow a "value" property which is expected
224  * to be of a valid type that matches all validating properties (such as
225  * minLength, etc.).  The value of this property will be displayed initially
226  * when the widget is constructed.
227  *
228  * Also, all types will support a "readonly" property.
229  */
230 DwtPropertyEditor.prototype.initProperties = function(schema, parent) {
231 	if (parent == null) {
232 		this._schema = schema;
233 		parent = null;
234 	}
235 	for (var i = 0; i < schema.length; ++i)
236 		this._createProperty(schema[i], parent);
237 };

239 DwtPropertyEditor.prototype._createProperty = function(prop, parent) {
240 	var level = parent ? parent._level + 1 : 0;
241 	var tr = this.getTable().firstChild.appendChild(document.createElement("tr"));
242 
243 	// Initialize the "prop" object with some interesting attributes...
244 	prop._parent = parent;
245 	prop._level = level;
246 	prop._rowElId = tr.id = Dwt.getNextId();
247 	prop._propertyEditor = this;
248 
249 	// ... and methods.
250 	for (var i in DwtPropertyEditor._prop_functions)
251 		prop[i] = DwtPropertyEditor._prop_functions[i];
252 
253 	prop._init();
254 
255 	// indent if needed
256 	tr.className = "level-" + level;
257 
258 	if (prop.visible == "false")
259 		tr.className += " invisible";
260 
261 	if (prop.readonly)
262 		tr.className += " readonly";
263 
264 	if (prop.type != "struct") {
265 		tr.className += " " + prop.type;
266 
267 		// this is a simple property, create a label and value cell.
268 		var tdLabel = document.createElement("td");
269 		tdLabel.className = "label";
270 		
271 		if(prop.type=="checkboxgroup"){
272 			tdLabel.className+=" grouplabel";
273 		}
274 		
275 		tr.appendChild(tdLabel);
276 		var html = AjxStringUtil.htmlEncode(prop.label || prop.name);
277 		if (prop.required)
278 			html += "<span class='DwtPropertyEditor-required'>*</span>";
279 		tdLabel.innerHTML = html;
280 		var tdField = document.createElement("td");
281 		tdField.className = "field";
282 		tr.appendChild(tdField);
283 
284 		switch (prop.type) {
285 			case "boolean" : this._createCheckbox(prop, tdField); break;
286 		    case "enum"    : this._createDropDown(prop, tdField); break;
287 		    case "date"    : this._createCalendar(prop, tdField); break;
288    		    case "checkboxgroup"	:	this._createCheckBoxGroup(prop,tdField); break;		    
289 		    default        :
290 			if (this._useDwtInputField)
291 				this._createInputField(prop, tdField);
292 			else {
293 				tdField.innerHTML = prop._makeDisplayValue();
294 				tr.__msh_doMouseDown = DwtPropertyEditor.simpleClosure(prop._edit, prop);
295 			}
296 			break;
297 		}
298 
299 		prop._fieldCellId = tdField.id = Dwt.getNextId();
300 		// prop._labelCellId = tdLabel.id = Dwt.getNextId();
301 
302 		if (tdLabel.offsetWidth > this.maxLabelWidth)
303 			this.maxLabelWidth = tdLabel.offsetWidth;
304 		if (tdField.offsetWidth > this.maxFieldWidth)
305 			this.maxFieldWidth = tdField.offsetWidth;
306 	} else {
307 		var td = document.createElement("td");
308 		td.colSpan = 2;
309 		tr.appendChild(td);
310 		td.className = "label";
311 		tr.className += " expander-collapsed";
312 		td.innerHTML = [ "<div>", AjxStringUtil.htmlEncode(prop.label), "</div>" ].join("");
313 		this.initProperties(prop.children, prop);
314 		tr.__msh_doMouseDown = DwtPropertyEditor.simpleClosure(prop._toggle, prop);
315 	}
316 
317 	// collapsed by default
318 	if (level > 0) {
319 		tr.style.display = "none";
320 		parent._hidden = true;
321 	}
322 };

338 DwtPropertyEditor.prototype._setCurrentMsgDiv = function(div) {
339 	this._currentMsgDiv = div;
340 	this._currentMsgDivTimer = setTimeout(
341 		DwtPropertyEditor.simpleClosure(this._clearMsgDiv, this),
342 		DwtPropertyEditor.MSG_TIMEOUT);
343 };
344 
345 DwtPropertyEditor.prototype._clearMsgDiv = function() {
346 	try {
347 		this._stopMsgDivTimer();
348 	} catch(ex) {};
349 	var div = this._currentMsgDiv;
350 	if (div) {
351 		div.parentNode.removeChild(div);
352 		this._currentMsgDiv = div = null;
353 		this._currentMsgDivTimer = null;
354 	}
355 };

357 DwtPropertyEditor.prototype._stopMsgDivTimer = function() {
358 	if (this._currentMsgDivTimer) {
359 		clearTimeout(this._currentMsgDivTimer);
360 		this._currentMsgDivTimer = null;
361 	}
362 };

456 DwtPropertyEditor.prototype._createDropDown = function(prop, target) {
457 	this._currentFieldCell = target;
458 	var item, sel,
459 		i       = 0,
460 		options = [],
461 		items   = prop.item;
462 	while (item = items[i])
463 		options[i++] = new DwtSelectOption(item.value,
464 						   item.value == prop.value,
465 						   item.label);
466 	prop._select = sel = new DwtSelect({parent:this, options:options});
467 	sel.addChangeListener(new AjxListener(prop, prop._onSelectChange));
468 	sel.addListener(DwtEvent.ONMOUSEDOWN, this._onMouseDown);
469 	this._currentFieldCell = null;
470 };

472 DwtPropertyEditor.prototype._createCalendar = function(prop, target) {
473 	this._currentFieldCell = target;
474 	var btn = new DwtButton({parent:this});
475 	this._currentFieldCell = null;
476 
477 	btn.setText(prop._makeDisplayValue());
478 	var menu = new DwtMenu({parent:btn, style:DwtMenu.CALENDAR_PICKER_STYLE});
479 	menu.setAssociatedObj(btn);
480 	var cal = new DwtCalendar({parent:menu});
481 	var date = new Date();
482 	date.setTime(prop.value);
483 	cal.setDate(date);
484 	cal.setSize(150, "auto");
485 	cal.addSelectionListener(new AjxListener(prop, prop._onCalendarSelect));
486 	btn.setMenu(menu);
487 
488 	prop._dateButton = btn;
489 	prop._dateCalendar = cal;
490 };

493 	"string"    : DwtInputField.STRING,
494 	"password"  : DwtInputField.PASSWORD,

496 	"number"    : DwtInputField.FLOAT

499 DwtPropertyEditor.prototype._createInputField = function(prop, target) {
500 	this._currentFieldCell = target;
501 	var type = DwtPropertyEditor.DWT_INPUT_FIELD_TYPES[prop.type]
502 		|| DwtInputField.STRING;
503 	var field = new DwtInputField({parent: this, type: type, initialValue: prop.value, maxLen: prop.maxLength, rows: prop.rows});
504 	if (type == DwtInputField.INTEGER || type == DwtInputField.FLOAT) {
505 		field.setValidNumberRange(prop.minValue || null,
506 					  prop.maxValue || null);
507 		if (prop.decimals != null)
508 			field.setNumberPrecision(prop.decimals);
509 	}
510 	if (type == DwtInputField.STRING || type == DwtInputField.PASSWORD)
511 		field.setValidStringLengths(prop.minLength, prop.maxLength);
512 	if (prop.required)
513 		field.setRequired();
514 	this._currentFieldCell = null;
515 	prop._inputField = field;
516 	field.setValue(prop.value);
517 	if (prop.readonly)
518 		field.setReadOnly(true);
519 	field.setValidationCallback(new AjxCallback(prop, prop._onDwtInputFieldValidated));
520 };

522 // these will be merged to each prop object that comes in the schema
523 DwtPropertyEditor._prop_functions = {
524 
525 	_init : function() {
526 		this.type != null || (this.type = "string");
527 		this.value != null || (this.value = "");
528 		this._initialVal = this.value;
529 
530 		if (this.type == "date") {
531 			if (!this.value) {
532 // 				var tmp = new Date();
533 // 				tmp.setHours(0);
534 // 				tmp.setMinutes(0);
535 // 				tmp.setSeconds(0);
536 				this.value = new Date().getTime();
537 			}
538 			if (!this.format)
539 				this.format = AjxDateUtil.getSimpleDateFormat().toPattern();
540 		}
541 	},
542 
543 	_modified : function() {
544 		return this._initialVal != this.value;
545 	},
546 
547 	_getRowEl : function() {
548 		return document.getElementById(this._rowElId);
549 	},
550 
551 	_makeDisplayValue : function() {
552 		var val = this._getValue();
553 		switch (this.type) {
554 		    case "password" :
555 			val = val.replace(/./g, "*");
556 			break;
557 		    case "date" :
558 			var date = new Date();
559 			date.setTime(val);
560 			val = AjxDateFormat.format(this.format, date);
561 			break;
562 		}
563 		if (val == "")
564 			val = "<br />";
565 		else
566 			val = AjxStringUtil.htmlEncode(String(val));
567 		return val;
568 	},
569 
570 	_display : function(visible) {
571 		var
572 			c = this.children,
573 			d = visible ? "" : "none";
574 		if (c) {
575 			var i = c.length;
576 			while (--i >= 0) {
577 				c[i]._getRowEl().style.display = d;
578 				if (!visible)
579 					c[i]._display(false);
580 			}
581 			this._hidden = !visible;
582 
583 			// change the class name accordingly
584 			var tr = this._getRowEl();
585 			tr.className = tr.className.replace(
586 				/expander-[^\s]+/,
587 				visible ? "expander-expanded" : "expander-collapsed");
588 		}
589 	},
590 
591 	_toggle : function() { this._display(this._hidden); },
592 
593 	_edit : function() {
594 		// Depending on the type, this should probably create different
595 		// fields for editing.  For instance, in a "date" property we
596 		// would want a calendar, while in a "list" property we would
597 		// want a drop-down select box.
598 
599 		if (this.readonly)
600 			return;
601 
602 		switch (this.type) {
603 		    case "string" :
604 		    case "number" :
605 		    case "integer" :
606 		    case "password" :
607 			setTimeout(
608 				DwtPropertyEditor.simpleClosure(
609 					this._createInputField, this), 50);
610 			break;
611 
612 // 		    default :
613 // 			alert("We don't support this type yet");
614 		}
615 	},
616 
617 	_createInputField : function() {
618 		var	pe     = this._propertyEditor;
619 		var td     = document.getElementById(this._fieldCellId);
620 		var canvas = pe.getRelDiv();
621 		var input  = document.createElement("input");
622 
623 		input.className = "DwtPropertyEditor-input " + this.type;
624 		input.setAttribute("autocomplete", "off");
625 
626 		input.type = this.type == "password"
627 			? "password"
628 			: "text";
629 
630 		var left = td.offsetLeft, top = td.offsetTop;
631 		if (AjxEnv.isGeckoBased) {
632 			--left;
633 			--top;
634 		}
635 		input.style.left = left + "px";
636 		input.style.top = top + "px";
637 		input.style.width = td.offsetWidth + 1 + "px";
638 		input.style.height = td.offsetHeight + 1 + "px";
639 
640 		input.value = this._getValue();
641 
642 		canvas.appendChild(input);
643 		input.focus();
644 
645 		input.onblur = DwtPropertyEditor.simpleClosure(this._saveInput, this);
646 		input.onkeydown = DwtPropertyEditor.simpleClosure(this._inputKeyPress, this);
647 
648 		this._propertyEditor._currentInputField = this._inputField = input;
649 		if (!AjxEnv.isGeckoBased)
650 			input.select();
651 		else
652 			input.setSelectionRange(0, input.value.length);
653 	},
654 
655 	_getValue : function() {
656 		return this.value || "";
657 	},
658 
659 	_checkValue : function(val) {
660 		var empty = val == "";
661 
662 		if (empty) {
663 			if (!this.required)
664 				return val;
665 			this._displayMsg(AjxMsg.valueIsRequired);
666 			return null;
667 		}
668 
669 		if (this.maxLength != null && val.length > this.maxLength) {
670 			this._displayMsg(AjxMessageFormat.format(AjxMsg.stringTooLong, this.maxLength));
671 			return null;
672 		}
673 
674 		if (this.minLength != null && val.length < this.minLength) {
675 			this._displayMsg(AjxMessageFormat.format(AjxMsg.stringTooShort, this.minLength));
676 			return null;
677 		}
678 
679 		if (this.mustMatch && !this.mustMatch.test(val)) {
680 			this._displayMsg(this.msg_mustMatch ||
681 					 DwtPropertyEditor.MSG.mustMatch.replace(
682 						 /REGEXP/, this.mustMatch.toString()));
683 			return null;
684 		}
685 
686 		if (this.mustNotMatch && this.mustNotMatch.test(val)) {
687 			this._displayMsg(this.msg_mustNotMatch ||
688 					 DwtPropertyEditor.MSG.mustNotMatch.replace(
689 						 /REGEXP/, this.mustNotMatch.toString()));
690 			return null;
691 		}
692 
693 		switch (this.type) {
694 		    case "integer" :
695 		    case "number" :
696 			var n = new Number(val);
697 			if (isNaN(n)) {
698 				this._displayMsg(AjxMsg.notANumber);
699 				return null;
700 			}
701 			if (this.type == "integer" && Math.round(n) != n) {
702 				this._displayMsg(AjxMsg.notAnInteger);
703 				return null;
704 			}
705 			if (this.minValue != null && n < this.minValue) {
706 				this._displayMsg(AjxMessageFormat.format(AjxMsg.numberLessThanMin, this.minValue));
707 				return null;
708 			}
709 			if (this.maxValue != null && n > this.maxValue) {
710 				this._displayMsg(AjxMessageFormat.format(AjxMsg.numberMoreThanMax, this.maxValue));
711 				return null;
712 			}
713 			val = n;
714 			if (this.type == "number" && this.decimals != null) {
715 				var str = val.toString();
716 				var pos = str.indexOf(".");
717 				if (pos == -1)
718 					pos = str.length;
719 				val = val.toPrecision(pos + this.decimals);
720 			}
721 			break;
722 		}
723 		return val;
724 	},
725 
726 	_displayMsg : function(msg) {
727 		var x, y, w, h;
728 		var pe  = this._propertyEditor;
729 		var div = pe._currentMsgDiv;
730 
731 		if (!div) {
732 			div = document.createElement("div");
733 			div.className = "DwtPropertyEditor-ErrorMsg";
734 			pe.getRelDiv().appendChild(div);
735 		} else
736 			pe._stopMsgDivTimer();
737 		div.style.visibility = "hidden";
738 		div.innerHTML = AjxStringUtil.htmlEncode(msg);
739 		// position & size
740 		var table = pe.getTable();
741 		w = table.offsetWidth; // padding & border!
742 		if (!AjxEnv.isIE)
743 			w -= 12;
744 		x = table.offsetLeft;
745 		div.style.left = x + "px";
746 		div.style.width = w + "px";
747 		h = div.offsetHeight;
748 		var td = document.getElementById(this._fieldCellId);
749 		y = td.offsetTop + td.offsetHeight;
750 		if (y + h > table.offsetTop + table.offsetHeight)
751 			y = td.offsetTop - h;
752 		div.style.top = y + "px";
753 		div.style.visibility = "";
754 		pe._setCurrentMsgDiv(div);
755 	},
756 
757 	_saveInput : function() {
758 		var input = this._inputField;
759 		var val = this._checkValue(input.value);
760 		if (val != null) {
761 			this._setValue(val);
762 			input.onblur = input.onkeyup = input.onkeydown = input.onkeypress = null;
763 			var td = document.getElementById(this._fieldCellId);
764 			td.innerHTML = this._makeDisplayValue();
765 			this._inputField = null;
766 			this._propertyEditor._currentInputField = null;
767 			this._propertyEditor._clearMsgDiv();
768 			input.parentNode.removeChild(input);
769 			return true;
770 		} else {
771 			if (input.className.indexOf(" DwtPropertyEditor-input-error") == -1)
772 				input.className += " DwtPropertyEditor-input-error";
773 			input.focus();
774 			return false;
775 		}
776 	},
777 
778 	_inputKeyPress : function(ev) {
779 		ev || (ev = window.event);
780 		var input = this._inputField;
781 		if (ev.keyCode == 13) {
782 			this._saveInput();
783 		} else if (ev.keyCode == 27) {
784 			input.value = this._getValue();
785 			this._saveInput();
786 		} else {
787 			this._propertyEditor._clearMsgDiv();
788 			input.className = input.className.replace(/ DwtPropertyEditor-input-error/, "");
789 		}
790 	},
