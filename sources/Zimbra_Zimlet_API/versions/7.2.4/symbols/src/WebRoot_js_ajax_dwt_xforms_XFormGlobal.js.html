 26 XFG.getUniqueId = function (namePrefix) {
 27 	if (namePrefix == null) namePrefix = "__id__";
 28 	var list = XFG.prefixList;
 29 
 30 	// if we've never seen one of these before, call it the name they passed in
 31 	//	(without a number) and set the counter to 1 (so the next one is #2)
 32 	if (list[namePrefix] == null) {
 33 		list[namePrefix] = 1;
 34 		return namePrefix;
 35 	} else {
 36 		list[namePrefix]++;
 37 		return namePrefix + "_" + list[namePrefix];
 38 	}
 39 };

 41 XFG.assignUniqueId = function (object, namePrefix) {
 42 	var id = XFG.getUniqueId(namePrefix);
 43 	object.__id = id;
 44 	XFG.objectCache[id] = object;
 45 };

 51 XFG.createEl = function (id, parentEl, tagName, contents) {
 52 	// create the element
 53 	if (tagName == null) tagName = "div";
 54 	var el = window.document.createElement(tagName);
 55 
 56 	// set its id and contents (if supplied)
 57 	el.id = id;
 58 	if (contents != null) el.innerHTML = contents;
 59 
 60 	// root it under the parent
 61 	if (parentEl == null) {
 62 		parentEl = document.body;
 63 	}
 64 	parentEl.appendChild(el);
 65 	
 66 	return el;
 67 }

 69 XFG.getEl = function (id, frame) {
 70 	// if they passed something other than a string, assume its the element itself
 71 	if (typeof id != "string") return id;
 72 	
 73 	var doc = (doc == null ? document : frame.document);
 74 	var it = doc.getElementById(id);
 75 	if (it == null) it = null;
 76 	return it;
 77 };

 79 XFG.hideEl = function (id,isBlock) {
 80 	var el = (typeof id == "string" ? XFG.getEl(id) : id);
 81 	if (el) {
 82 		if(!isBlock)
 83 			el.style.display = "none";
 84 			
 85 		el.style.visibility = "hidden";
 86 	} else {
 87 		DBG.println("hideEl(", id, "): element not found");
 88 	}
 89 };

 91 XFG.showEl = function (id) {
 92 	var el = (typeof id == "string" ? XFG.getEl(id) : id);
 93 	if (el) {
 94 		if (el.tagName == "TD") {
 95 			if (AjxEnv.isIE) {
 96 				el.style.display = "block";
 97 			} else {
 98 				el.style.display = "table-cell";			
 99 			}
100 		} else {
101 			el.style.display = "block";
102 		}
103 		el.style.visibility = "visible";
104 	} else {
105 		DBG.println("showEl(", id, "): element not found");
106 	}
107 };

161 XFG.getCookie = function (name) {
162 	var value = new RegExp(name + "=([^;]+)").exec(document.cookie);
163 	return (value != null) ? unescape(value[1]) : null;
164 }

174 XFG.valueToString = function (value, skipDerivedProperties, skipMethods, skipPrototypeProperties) {
175 	if (value == null) return "null";
176 
177 	// strings get quotes
178 	if (typeof value == "string") return '"' + value + '"';
179 
180 	// for arrays, list all the objects in it
181 	if (value instanceof Array) {
182 		var buffer = new AjxBuffer();
183 		for (var i = 0; i < value.length; i++) {
184 			buffer.append("        ", XFG.valueToString(value[i], "    ", skipDerivedProperties, skipMethods, skipPrototypeProperties));
185 		}
186 		return "[\n" + buffer.join(",\n") + "\n" + "    ]";
187 	}
188 
189 	// for dates, return the syntax to create a new date (might as well)
190 	if (value instanceof Date) {
191 		return " new Date("+ [value.getFullYear(), value.getMonth(), value.getDate(), value.getHours(), value.getMinutes(), value.getSeconds(), value.getMilliseconds()].join(",") + ")";
192 	}
193 	
194 	if (typeof value == "function") {
195 		return "(function)";
196 	}
197 
198 	// for objects, call
199 	if (typeof value == "object") return XFG.objectToString(value, "    ", skipDerivedProperties, skipMethods, skipPrototypeProperties);
200 	
201 	return value;	
202 }
203 
204 XFG.objectToString = function (object, skipDerivedProperties, skipMethods, skipPrototypeProperties) {
205 
206 	var indentSpacer = "    ";
207 	var buffer = [];
208 	var hasObject = false;
209 	var propCount = 0;
210 	var proto = object.constructor.prototype;
211 
212 	if (proto == null) proto = {};
213 	for (var prop in object) {
214 		var value = object[prop];
215 		if (skipPrototypeProperties && (object[prop] == proto[prop])) continue;
216 		if (skipMethods && value instanceof Function) continue;
217 		
218 		// if we have a derived property, write its id or [object] so we don't recurse too much
219 		if ((prop.indexOf("__") == 0 || prop.indexOf("$") == 0) && value instanceof Object) {
220 			buffer.push(AjxBuffer.concat(prop, ": ", value.toString()));
221 		} else {
222 			hasObject = hasObject || (typeof value == "object");
223 			buffer.push(AjxBuffer.concat(prop, ": ", XFG.valueToString(value, skipDerivedProperties, skipMethods, skipPrototypeProperties)));
224 		}
225 		propCount++;
226 	}
227 	buffer.sort(XFG.sortSpecialLast);
228 	if (hasObject || propCount > 5) {
229 		return "{\n" + indentSpacer + buffer.join(",\n"+ indentSpacer) + "\n" + "}"
230 	} else {
231 		return "{" + indentSpacer + buffer.join(", ") + indentSpacer + "}";
232 	}
233 }
234 
235 XFG.sortSpecialLast = function (a,b) {
236 	var a1 = a.charAt(0);
237 	var b1 = b.charAt(0);
238 	var aIsSpecial = a1 == "_" || a1 == "$";
239 	var bIsSpecial = b1 == "_" || b1 == "$";
240 	if ( !aIsSpecial && !bIsSpecial) {
241 		return (a > b ? 1 : -1)
242 	} else if (aIsSpecial && bIsSpecial) {
243 		return (a > b ? 1 : -1)
244 	} else if (aIsSpecial) {
245 		return 1;
246 	} else {
247 		return -1;
248 	}
249 	
250 }

255 StringBuffer = function() {
256 	this.clear();
257 	if (arguments.length > 0) {
258 		arguments.join = this.buffer.join;
259 		this.buffer[this.buffer.length] = arguments.join("");
260 	}
261 }

263 	return this.buffer.join("");
264 }
265 StringBuffer.prototype.join = function (delim) {
266 	if (delim == null) delim = "";
267 	return this.buffer.join(delim);
268 }
269 StringBuffer.prototype.append = function () {
270 	arguments.join = this.buffer.join;
271 	this.buffer[this.buffer.length] = arguments.join("");
272 }
273 StringBuffer.prototype.join = function (str) {
274 	return this.buffer.join(str);
275 }
276 StringBuffer.prototype.set = function(str) {
277 	this.buffer = [str];
278 }
279 StringBuffer.prototype.clear = function() {
280 	this.buffer = [];
281 }
