 44 ZmSharingPage.prototype.getShares =
 45 function(type, owner, callback) {
 46 
 47 	var jsonObj = {GetShareInfoRequest:{_jsns:"urn:zimbraAccount"}};
 48 	var request = jsonObj.GetShareInfoRequest;
 49 	if (type && type != ZmShare.TYPE_ALL) {
 50 		request.grantee = {type:type};
 51 	}
 52 	if (owner) {
 53 		request.owner = {by:"name", _content:owner};
 54 	}
 55 	var respCallback = new AjxCallback(this, this._handleGetSharesResponse, [callback]);
 56 	appCtxt.getAppController().sendRequest({jsonObj:	jsonObj,
 57 											asyncMode:	true,
 58 											callback:	respCallback});
 59 };
 60 
 61 ZmSharingPage.prototype._handleGetSharesResponse =
 62 function(callback, result) {
 63 
 64 	var resp = result.getResponse().GetShareInfoResponse;
 65 	if (callback) {
 66 		callback.run(resp.share);
 67 	}
 68 };

144 /**
145  * Makes a request to the server for group shares or shares from a particular user.
146  *
147  * @param owner					[string]*		address of account to check for shares from
148  * @param userButtonClicked		[boolean]*		if true, user pressed "Find Shares" button
149  * 
150  * @private
151  */
152 ZmSharingView.prototype.findShares =
153 function(owner, userButtonClicked) {
154 
155 	var errorMsg;
156 	// check if button was actually clicked, since missing owner is fine when form
157 	// goes through rote validation on display
158 	if (userButtonClicked && !owner) {
159 		errorMsg = ZmMsg.sharingErrorOwnerMissing;
160 	} else if (!this._shareForm.validate(ZmSharingView.ID_OWNER)) {
161 		errorMsg = ZmMsg.sharingErrorOwnerSelf;
162 	}
163 	if (errorMsg) {
164 		appCtxt.setStatusMsg({msg: errorMsg, level: ZmStatusView.LEVEL_INFO});
165 		return;
166 	}
167 
168 	var respCallback = new AjxCallback(this, this.showPendingShares);
169 	var type = owner ? null : ZmShare.TYPE_GROUP;
170 	this._curOwner = owner;
171 	var shares = this.parent.getShares(type, owner, respCallback);
172 };
173 /**
174  * Displays a list of shares that have been accepted/mounted by the user.
175  * 
176  * @private
177  */
178 ZmSharingView.prototype.showMounts =
179 function() {
180 
181 	var folderTree = appCtxt.getFolderTree();
182 	var folders = folderTree && folderTree.asList({remoteOnly:true});
183 	if (!folders) { return; }
184 
185 	var ownerHash = {};
186 	for (var i = 0; i < folders.length; i++) {
187 		var folder = folders[i];
188 		if (folder.isMountpoint || folder.link) {
189 			if (folder.owner) {
190 				ownerHash[folder.owner] = true;
191 			}
192 		}
193 	}
194 
195 	var owners = AjxUtil.keys(ownerHash);
196 	if (owners.length > 0) {
197 		var jsonObj = {BatchRequest:{_jsns:"urn:zimbra", onerror:"continue"}};
198 		var br = jsonObj.BatchRequest;
199 		var requests = br.GetShareInfoRequest = [];
200 		for (var i = 0; i < owners.length; i++) {
201 			var req = {_jsns: "urn:zimbraAccount"};
202 			req.owner = {by:"name", _content:owners[i]};
203 			requests.push(req);
204 		}
205 
206 		var respCallback = new AjxCallback(this, this._handleResponseGetShares);
207 		appCtxt.getRequestMgr().sendRequest({jsonObj:jsonObj, asyncMode:true, callback:respCallback});
208 	}
209 };
210 
211 ZmSharingView.prototype._handleResponseGetShares =
212 function(result) {
213 
214 	var mounts = [];
215 	var resp = result.getResponse().BatchResponse.GetShareInfoResponse;
216 	for (var i = 0; i < resp.length; i++) {
217 		var shares = resp[i].share;
218 		if (!(shares && shares.length)) { continue; }
219 		for (var j = 0; j < shares.length; j++) {
220 			var share = ZmShare.getShareFromShareInfo(shares[j]);
221 			if (share.mounted) {
222 				mounts.push(share);
223 			}
224 		}
225 	}
226 
227 	mounts.sort(ZmSharingView.sortCompareShare);
228 	this._mountedShareListView.set(AjxVector.fromArray(mounts));
229 
230 };

232 /**
233  * Displays shares that are pending (have not yet been mounted).
234  *
235  * @param shares	[array]		list of JSON share info objects from GetShareInfoResponse
236  * 
237  * @private
238  */
239 ZmSharingView.prototype.showPendingShares =
240 function(shares) {
241 
242 	var pending = [];
243 	if (shares && shares.length) {
244 		for (var i = 0; i < shares.length; i++) {
245 			// convert share info to ZmShare
246 			var share = ZmShare.getShareFromShareInfo(shares[i]);
247 			if (!share.mounted) {
248 				pending.push(share);
249 			}
250 		}
251 	}
252 	pending.sort(ZmSharingView.sortCompareShare);
253 	this._pendingShareListView.set(AjxVector.fromArray(pending));
254 };

256 /**
257  * Displays grants (folders shared by the user) in a list view. Grants show up as shares
258  * in folders owned by the user.
259  * 
260  * @private
261  */
262 ZmSharingView.prototype.showGrants =
263 function() {
264 
265 	// the grant objects we get in the refresh block don't have grantee names,
266 	// so use GetFolder in a BatchRequest to get them
267 	var batchCmd = new ZmBatchCommand(true, null, true);
268 	var list = appCtxt.getFolderTree().asList();
269 	for (var i = 0; i < list.length; i++) {
270 		var folder = list[i];
271 		if (folder.shares && folder.shares.length) {
272 			for (var j = 0; j < folder.shares.length; j++) {
273 				var share = folder.shares[j];
274 				if (!(share.grantee && share.grantee.name)) {
275 					batchCmd.add(new AjxCallback(folder, folder.getFolder, [null, batchCmd]));
276 					break;
277 				}
278 			}
279 		}
280 	}
281 
282 	if (batchCmd._cmds.length) {
283 		var respCallback = new AjxCallback(this, this._handleResponseGetFolder);
284 		batchCmd.run(respCallback);
285 	} else {
286 		this._handleResponseGetFolder();
287 	}
288 };
289 
290 ZmSharingView.prototype._handleResponseGetFolder =
291 function() {
292 
293 	var shares = [], invalid = [];
294 	var list = appCtxt.getFolderTree().asList();
295 	for (var i = 0; i < list.length; i++) {
296 		var folder = list[i];
297 		if (folder.shares && folder.shares.length) {
298 			for (var j = 0; j < folder.shares.length; j++) {
299 				var share = ZmShare.getShareFromGrant(folder.shares[j]);
300 				if (share.invalid) {
301 					invalid.push(share);
302 				}
303 				shares.push(share);
304 			}
305 		}
306 	}
307 
308 	shares.sort(ZmSharingView.sortCompareGrant);
309 	this._grantListView.set(AjxVector.fromArray(shares));
310 
311 	// an invalid grant is one whose grantee has been removed from the system
312 	// if we have some, ask the user if it's okay to remove them
313 	if (invalid.length) {
314 		invalid.sort(ZmSharingView.sortCompareGrant);
315 		var msgDialog = appCtxt.getOkCancelMsgDialog();
316 		var list = [];
317 		for (var i = 0; i < invalid.length; i++) {
318 			var share = invalid[i];
319 			var path = (share.link && share.link.path);
320 			if (path) {
321 				list.push(["<li>", path, "</li>"].join(""));
322 			}
323 		}
324 		list = AjxUtil.uniq(list);
325 		var listText = list.join("");
326 		msgDialog.setMessage(AjxMessageFormat.format(ZmMsg.granteeGone, listText));
327 		msgDialog.registerCallback(DwtDialog.OK_BUTTON, this._revokeGrantsOk, this, [msgDialog, invalid]);
328 		msgDialog.registerCallback(DwtDialog.CANCEL_BUTTON, this._revokeGrantsCancel, this, msgDialog);
329 		msgDialog.associateEnterWithButton(DwtDialog.OK_BUTTON);
330 		msgDialog.popup(null, DwtDialog.OK_BUTTON);
331 	}
332 };

334 ZmSharingView.prototype._revokeGrantsOk =
335 function(dlg, invalid) {
336 
337 	var batchCmd = new ZmBatchCommand(true, null, true);
338 	var zids = {};
339 	for (var i = 0; i < invalid.length; i++) {
340 		var share = invalid[i];
341 		zids[share.grantee.id] = share.grantee.type;
342 	}
343 
344 	for (var zid in zids) {
345 		batchCmd.add(new AjxCallback(null, ZmShare.revokeOrphanGrants, [zid, zids[zid], null, batchCmd]));
346 	}
347 
348 	if (batchCmd._cmds.length) {
349 		batchCmd.run();
350 	}
351 
352 	dlg.popdown();
353 };

537 /**
538  * Sorts shares in the following order:
539  *   1. by name of owner
540  *   2. by name of group it was shared with, if any
541  *   3. by path of shared folder
542  *   
543  * @private
544  */
545 ZmSharingView.sortCompareShare =
546 function(a, b) {
547 
548 	var ownerA = (a.grantor.name && a.grantor.name.toLowerCase()) || (a.grantor.email && a.grantor.email.toLowerCase()) || "";
549 	var ownerB = (b.grantor.name && b.grantor.name.toLowerCase()) || (b.grantor.email && b.grantor.email.toLowerCase()) || "";
550 	if (ownerA != ownerB) {
551 		return (ownerA > ownerB) ? 1 : -1;
552 	}
553 
554 	var groupA = (a.grantee.type == ZmShare.TYPE_GROUP) ? (a.grantee.name && a.grantee.name.toLowerCase()) : "";
555 	var groupB = (b.grantee.type == ZmShare.TYPE_GROUP) ? (b.grantee.name && b.grantee.name.toLowerCase()) : "";
556 	if (groupA != groupB) {
557 		if (!groupA && groupB) {
558 			return 1;
559 		} else if (groupA && !groupB) {
560 			return -1;
561 		} else {
562 			return (groupA > groupB) ? 1 : -1;
563 		}
564 	}
565 
566 	var pathA = (a.link.name && a.link.name.toLowerCase()) || "";
567 	var pathB = (b.link.name && b.link.name.toLowerCase()) || "";
568 	if (pathA != pathB) {
569 		return (pathA > pathB) ? 1 : -1;
570 	}
571 
572 	return 0;
573 };

575 /**
576  * Sorts shares in the following order:
577  *   1. by name of who it was shared with
578  *   2. by path of shared folder
579  *   
580  * @private
581  */
582 ZmSharingView.sortCompareGrant =
583 function(a, b) {
584 
585 	var granteeA = (a.grantee && a.grantee.name && a.grantee.name.toLowerCase()) || "";
586 	var granteeB = (b.grantee && b.grantee.name && b.grantee.name.toLowerCase()) || "";
587 	if (granteeA != granteeB) {
588 		return (granteeA > granteeB) ? 1 : -1;
589 	}
590 
591 	var pathA = (a.link && a.link.name) || "";
592 	var pathB = (b.link && b.link.name) || "";
593 	if (pathA != pathB) {
594 		return (pathA > pathB) ? 1 : -1;
595 	}
596 
597 	return 0;
598 };

608 /**
609  * Handle modifications to pending shares, which don't have an item to propagate
610  * changes through. The preferences app sends the notifications here.
611  *
612  * @param modifies		[hash]		notifications
613  * 
614  * @private
615  */
616 ZmSharingView.prototype.notifyModify =
617 function(modifies) {
618 
619 	for (var name in modifies) {
620 		if (name == "folder") {
621 			modifies = modifies.folder;
622 			for (var i = 0; i < modifies.length; i++) {
623 				var mod = modifies[i];
624 				var share = this._shareByKey[mod.id];
625 				var ev = new ZmEvent();
626 				if (share) {
627 					var parts = mod.id.split(":");
628 					share.zid = parts[0];
629 					share.rid = parts[1];
630 					ev.ersatz = true;
631 					ev.set(ZmEvent.E_MODIFY);
632 					var fields = {};
633 					if (mod.perm) {
634 						share.setPermissions(mod.perm);
635 						fields[ZmOrganizer.F_PERMS] = true;
636 					}
637 					if (mod.name) {
638 						fields[ZmOrganizer.F_RNAME] = true;
639 					}
640 					if (mod.l) {
641 						ev.set(ZmEvent.E_MOVE);
642 					}
643 					ev.setDetail("share", share);
644 					ev.setDetail("fields", fields);
645 					this._folderTreeChangeListener(ev);
646 					mod._handled = true;
647 				} else if (mod.id.indexOf(":") != -1) {
648 					ev.set(ZmEvent.E_CREATE);
649 				}
650 			}
651 		}
652 	}
653 };

754 ZmSharingListView.prototype._getCellContents =
755 function(html, idx, item, field, colIdx, params) {
756

758 		html[idx++] = item.grantor.name || item.grantor.email;
759 	} else if (field == ZmSharingView.F_WITH) {

764 			html[idx++] = ZmMsg.shareWithAll;
765 		} else if (type == ZmShare.TYPE_GUEST) {
766 			html[idx++] = item.grantee.id;
767 		} else {
768 			html[idx++] = item.grantee.name;
769 		}
770 	} else if (field == ZmSharingView.F_ITEM) {
771 		html[idx++] = item.link.path;
772 	} else if (field == ZmSharingView.F_TYPE) {
773 		html[idx++] = (item.object && item.object.type) ? ZmMsg[ZmOrganizer.FOLDER_KEY[item.object.type]] :
774 					  									  ZmShare._getFolderType(item.link.view);
775 	} else if (field == ZmSharingView.F_ROLE) {
776 		var role = item.link.role || ZmShare._getRoleFromPerm(item.link.perm);

792 ZmSharingListView.prototype._changeListener =
793 function(ev) {
794 
795 	var organizers = ev.getDetail("organizers") || [];
796 	var fields = ev.getDetail("fields") || {};
797 
798 	if (this.type == ZmShare.SHARE) {
799 		var share = ev.getDetail("share");
800 		if (!share) {
801 			var mtpt = organizers[0];
802 			if (!mtpt.link) { return; }
803 			var share = this.sharingView._shareByKey[[mtpt.zid, mtpt.rid].join(":")];
804 			share = ZmShare.getShareFromLink(mtpt, share);	// update share
805 		}
806 		if (!share) { return; }
807 		if (ev.event == ZmEvent.E_CREATE) {
808 			// share accepted, mountpoint created; move from pending to mounted list
809 			if (this.status == ZmSharingView.PENDING) {
810 				this.removeItem(share);
811 			} else if (this.status == ZmSharingView.MOUNTED) {
812 				var index = this._list && this._getIndex(share, this._list.getArray(), ZmSharingView.sortCompareShare);
813 				this.addItem(share, index, true);
814 			}
815 		} else if (ev.event == ZmEvent.E_MODIFY) {
816 			if ((this.status == ZmSharingView.PENDING && share.mounted) ||
817 				(this.status == ZmSharingView.MOUNTED && !share.mounted)) { return; }
818 			if (fields[ZmOrganizer.F_PERMS]) {
819 				var cell = document.getElementById(this._getCellId(share, ZmSharingView.F_ROLE));
820 				if (cell) {
821 					cell.innerHTML = this._getCellContents([], 0, share, ZmSharingView.F_ROLE, null, {returnText:true});
822 				}
823 			}
824 			if ((this.status == ZmSharingView.MOUNTED) && fields[ZmOrganizer.F_NAME]) {
825 				var cell = document.getElementById(this._getCellId(share, ZmSharingView.F_FOLDER));
826 				if (cell) {
827 					cell.innerHTML = this._getCellContents([], 0, share, ZmSharingView.F_FOLDER, null, {returnText:true});
828 				}
829 			}
830 		}
831 		// if a remote folder has been renamed or moved, rerun the search
832 		if (ev.event == ZmEvent.E_MOVE || fields[ZmOrganizer.F_RNAME]) {
833 			if (this.sharingView._curOwner) {
834 				this.sharingView.findShares(this.sharingView._curOwner);
835 			}
836 		}
837 	}
838 
839 	// Any change to a grant (including create or revoke) results in a wholesale replacement of
840 	// the folder's shares, so it's easiest to just redraw the list. Also check for folder rename.
841 	if (this.type == ZmShare.GRANT) {
842 		if ((ev.event = ZmEvent.E_MODIFY && fields[ZmOrganizer.F_SHARES]) ||
843 		    (ev.event = ZmEvent.E_MODIFY && fields[ZmOrganizer.F_NAME] && organizers[0].shares)) {
844 
845 			this.sharingView.showGrants();
846 		}
847 	}
848 };

851  * Adds links for editing, revoking, or resending a grant.
852  *
853  * @param share		[ZmShare]		share
854  * @param html		[array]			HTML content
855  * @param idx		[int]			index
856  *

858  */
859 ZmSharingListView.prototype._addActionLinks =
860 function(share, html, idx) {
861 
862 	var type = share.grantee.type;
863 	var actions = ["edit", "revoke", "resend"];
864 	if (type == ZmShare.TYPE_ALL || type == ZmShare.TYPE_DOMAIN || !share.link.role) {
865 		html[idx++] = ZmMsg.configureWithAdmin;
866 		actions = [];
867 	}
868 
869 	var handlers = ["_handleEditShare", "_handleRevokeShare", "_handleResendShare"]; // handlers in ZmFolderPropsDialog
870 
871 	for (var i = 0; i < actions.length; i++) {
872 
873 		var action = actions[i];
874 
875 		// public shares have no editable fields, and sent no mail
876 		if ((share.isPublic() || share.invalid) && (action == "edit" || action == "resend")) { continue; }
877 
878 		html[idx++] = "<a href='javascript:;' onclick='ZmSharingView._handleShareAction(" + '"' + share.domId + '", "' + handlers[i] + '"' + ");'>" + ZmMsg[action] + "</a> ";
879 	}
880 
881 	return idx;

884 /**
885  * Returns the position of the share in the given list using the given compare function.
886  *
887  * @param share			[ZmShare]		a share
888  * @param list			[array]			list of shares
889  * @param compareFunc	[function]		compare function
890  * 
891  * @private
892  */
893 ZmSharingListView.prototype._getIndex =
894 function(share, list, compareFunc) {
895 
896 	for (var i = 0; i < list.length; i++) {
897 		var result = compareFunc(share, list[i]);
898 		if (result == -1) {
899 			return i;
900 		}
901 	}
902 	return null;
