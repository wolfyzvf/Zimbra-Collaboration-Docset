 21 /**
 22  * Creates a tag
 23  * @class
 24  * This class represents a tag.
 25  * 
 26  * @param	{Hash}	params		a hash of parameters
 27  * @extends	ZmOrganizer
 28  */
 29 ZmTag = function(params) {
 30 	params.type = ZmOrganizer.TAG;
 31 	ZmOrganizer.call(this, params);
 32 };

 67 /**
 68  * Tags come from back end as a flat list, and we manually create a root tag, so all tags
 69  * have the root as parent. If tags ever have a tree structure, then this should do what
 70  * ZmFolder does (recursively create children).
 71  * 
 72  * @private
 73  */
 74 ZmTag.createFromJs =
 75 function(parent, obj, tree, sorted, account) {
 76 	var tag;
 77 	var nId = ZmOrganizer.normalizeId(obj.id);
 78 	if (nId < ZmOrganizer.FIRST_USER_ID[ZmOrganizer.TAG]) { return; }
 79 	tag = tree.getById(obj.id);
 80 	if (tag) { return tag; }
 81 
 82 	var params = {
 83 		id: obj.id,
 84 		name: obj.name,
 85 		color: ZmTag.checkColor(obj.color),
 86 		rgb: obj.rgb,
 87 		parent: parent,
 88 		tree: tree,
 89 		numUnread: obj.u,
 90 		account: account
 91 	};
 92 	tag = new ZmTag(params);
 93 	var index = sorted ? ZmOrganizer.getSortIndex(tag, ZmTag.sortCompare) : null;
 94 	parent.children.add(tag, index);
 95 
 96 	return tag;
 97 };

 99 /**
100  * Compares the tags by name.
101  * 
102  * @param	{ZmTag}	tagA		the first tag
103  * @param	{ZmTag}	tagB		the second tag
104  * @return	{int}	0 if the tag names match (case-insensitive); 1 if "a" is before "b"; -1 if "b" is before "a"
105  */
106 ZmTag.sortCompare = 
107 function(tagA, tagB) {
108 	var check = ZmOrganizer.checkSortArgs(tagA, tagB);
109 	if (check != null) return check;
110 
111 	if (tagA.name.toLowerCase() > tagB.name.toLowerCase()) return 1;
112 	if (tagA.name.toLowerCase() < tagB.name.toLowerCase()) return -1;
113 	return 0;
114 };

116 /**
117  * Checks the tag name.
118  * 
119  * @param	{String}	name		the name
120  * @return	{String}	<code>null</code> if the name is valid or a error message
121  */
122 ZmTag.checkName =
123 function(name) {
124 	var msg = ZmOrganizer.checkName(name);
125 	if (msg) { return msg; }
126 
127 	if (name.indexOf('\\') == 0) {
128 		return AjxMessageFormat.format(ZmMsg.errorInvalidName, AjxStringUtil.htmlEncode(name));
129 	}
130 
131 	return null;
132 };
133 
134 /**
135  * Checks the color.
136  * 
137  * @param	{String}	color	the color
138  * @return	{Number}	the valid color
139  */
140 ZmTag.checkColor =
141 function(color) {
142 	color = Number(color);
143 	return ((color != null) && (color >= 0 && color <= ZmOrganizer.MAX_COLOR)) ? color : ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
144 };
145 
146 ZmTag.getIcon = function(color) {
147     var object = { getIcon:ZmTag.prototype.getIcon, color:color };
148     if (String(color).match(/^#/)) {
149         object.rgb = color;
150         object.color = null;
151     }
152     return ZmTag.prototype.getIconWithColor.call(object);
153 }

155 /**
156  * Creates a tag.
157  * 
158  * @param	{Hash}	params	a hash of parameters
159  */
160 ZmTag.create =
161 function(params) {
162 	var soapDoc = AjxSoapDoc.create("CreateTagRequest", "urn:zimbraMail");
163 	var tagNode = soapDoc.set("tag");
164 	tagNode.setAttribute("name", params.name);
165     if (params.rgb) {
166         tagNode.setAttribute("rgb", params.rgb);
167     }
168     else {
169         var color = ZmOrganizer.checkColor(params.color) || ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
170         tagNode.setAttribute("color", color);
171     }
172 	var errorCallback = new AjxCallback(null, ZmTag._handleErrorCreate, params);
173 	appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:true, errorCallback:errorCallback, accountName:params.accountName});
174 };
175 
176 /**
177  * @private
178  */
179 ZmTag._handleErrorCreate =
180 function(params, ex) {
181 	if (ex.code == ZmCsfeException.MAIL_INVALID_NAME) {
182 		var msg = AjxMessageFormat.format(ZmMsg.errorInvalidName, AjxStringUtil.htmlEncode(params.name));
183 		var msgDialog = appCtxt.getMsgDialog();
184 		msgDialog.setMessage(msg, DwtMessageDialog.CRITICAL_STYLE);
185 		msgDialog.popup();
186 		return true;
187 	}
188 	return false;
189 };

191 /**
192  * Gets the icon.
193  * 
194  * @return	{String}	the icon or <code>null</code> for no icon
195  */
196 ZmTag.prototype.getIcon = 
197 function() {
198 	return (this.id == ZmOrganizer.ID_ROOT) ? null : "Tag";
199 };

201 /**
202  * Creates a query for this tag.
203  * 
204  * @return	{String}	the tag query
205  */
206 ZmTag.prototype.createQuery =
207 function() {
208 	return ['tag:"', this.name, '"'].join("");
209 };

218 /**
219  * @private
220  */
221 ZmTag.prototype.notifyCreate =
222 function(obj) {
223 	var child = ZmTag.createFromJs(this, obj, this.tree, true);
224 	child._notify(ZmEvent.E_CREATE);
225 };
