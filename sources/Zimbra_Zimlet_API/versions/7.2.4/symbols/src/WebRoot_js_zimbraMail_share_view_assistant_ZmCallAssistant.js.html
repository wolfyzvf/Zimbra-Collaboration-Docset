 20 /**
 21  * Creates the call assistant.
 22  * @class
 23  * This class represents a call assistant.
 24  * 
 25  * @extends		ZmAssistant
 26  */
 27 ZmCallAssistant = function() {
 28 	ZmAssistant.call(this, ZmMsg.call, ZmMsg.ASST_CMD_CALL);
 29 };
 30 
 31 ZmCallAssistant.prototype = new ZmAssistant();
 32 ZmCallAssistant.prototype.constructor = ZmCallAssistant;
 33 
 34 ZmCallAssistant.prototype.okHandler =
 35 function(dialog) {
 36 	return true;	//override
 37 };
 38 
 39 ZmCallAssistant.prototype.handle =
 40 function(dialog, verb, args) {
 41 	dialog._setOkButton(AjxMsg.ok, true, true);
 42 };
 43 
 44 ZmCallAssistant.prototype.initialize =
 45 function(dialog) {
 46 	var html = new AjxBuffer();
 47 	html.append("<div>HelloWorld</div>");
 48 	dialog.setAssistantContent(html.toString());
 49 };
 50 
 51 // called when dialog switches away from this assistant
 52 ZmCallAssistant.prototype.finish =
 53 function(dialog) {
 54 
 55 };
