 21 /**
 22  * Creates the briefcase 
 23  * @class
 24  * This class represents a briefcase. A briefcase contains briefcase items.
 25  * 
 26  * @param	{Hash}	params		a hash of parameters
 27  * @param {int}	params.id			the numeric ID
 28  * @param {String}	params.name		the name
 29  * @param {ZmOrganizer}	params.parent		the parent organizer
 30  * @param {ZmTree}	params.tree		the tree model that contains this organizer
 31  * @param {constant}	params.color	the color for this briefcase
 32  * @param {String}	params.owner		the owner of this organizer
 33  * @param {String}	params.oname		the owner's name for this organizer
 34  * @param {String}	[params.zid]		the Zimbra id of owner, if remote share
 35  * @param {String}	[params.rid]		the Remote id of organizer, if remote share
 36  * @param {String}	[params.restUrl]	the REST URL of this organizer.
 37  * 
 38  * @extends		ZmFolder
 39  */
 40 ZmBriefcase = function(params) {
 41 	params.type = ZmOrganizer.BRIEFCASE;
 42 	ZmFolder.call(this, params);
 43 }

 82 ZmBriefcase.prototype.notifyModify =
 83 function(obj) {
 84 	ZmOrganizer.prototype.notifyModify.call(this, obj);
 85 
 86 	var doNotify = false;
 87 	var fields = {};
 88 	if (obj.name != null && this.name != obj.name && !obj._isRemote) {
 89 		this.name = obj.name;
 90 		fields[ZmOrganizer.F_NAME] = true;
 91 		doNotify = true;
 92 	} else if (obj.color != null && this.color != obj.color && !obj._isRemote) {
 93 		this.color = obj.color;
 94 		fields[ZmOrganizer.F_COLOR] = true;
 95 		doNotify = true;
 96 	}
 97 	
 98 	if (doNotify) {
 99 		this._notify(ZmEvent.E_MODIFY, {fields: fields});
100 	}
101 };

117 /**
118  * Comparison function for briefcases.
119  *
120  * @param {ZmBriefcase}	nbA		an item
121  * @param {ZmBriefcase}	nbB		an item
122  * @return	{int} 0 if the items match
123  */
124 ZmBriefcase.sortCompare = 
125 function(nbA, nbB) {
126 	var check = ZmOrganizer.checkSortArgs(nbA, nbB);
127 	if (check != null) return check;
128 
129 	// links appear after personal calendars
130 	if (nbA.link != nbB.link) {
131 		return nbA.link ? 1 : -1;
132 	}
133 	
134 	// sort by calendar name
135 	var nbAName = nbA.name.toLowerCase();
136 	var nbBName = nbB.name.toLowerCase();
137 	if (nbAName < nbBName) return -1;
138 	if (nbAName > nbBName) return 1;
139 	return 0;
140 };

142 /**
143  * Returns true if the given object(s) may be placed in this folder.
144  *
145  * If the object is a folder, check that:
146  * <ul>
147  * <li>We are not the immediate parent of the folder</li>
148  * <li>We are not a child of the folder</li>
149  * <li>We are not Spam or Drafts</li>
150  * <li>We don't already have a child with the folder's name (unless we are in Trash)</li>
151  * <li>We are not moving a regular folder into a search folder</li>
152  * <li>We are not moving a search folder into the Folders container</li>
153  * <li>We are not moving a folder into itself</li>
154  * </ul>
155  *
156  * If the object is an item or a list or items, check that:
157  * <ul>
158  * <li>We are not the Folders container</li>
159  * <li>We are not a search folder</li>
160  * <li>The items aren't already in this folder</li>
161  * <li>A contact can only be moved to Trash</li>
162  * <li>A draft can be moved to Trash or Drafts</li>
163  * <li>Non-drafts cannot be moved to Drafts</li>
164  * </ul>
165  *
166  * @param {Object}	what		the object(s) to possibly move into this briefcase (item or organizer)
167  */
168 ZmBriefcase.prototype.mayContain =
169 function(what, targetFolderType) {
170 
171     if (!what) return true;
172 
173 	var invalid = false;
174     targetFolderType = targetFolderType || this.type;
175 
176     if (what instanceof ZmFolder) { //ZmBriefcase
177          invalid =(
178                     what.parent == this || this.isChildOf(what)
179                  || targetFolderType == ZmOrganizer.SEARCH || targetFolderType == ZmOrganizer.TAG
180                  || (!this.isInTrash() && this.hasChild(what.name))
181                  || (what.id == this.id)
182                  || (this.isRemote() && !this._remoteMoveOk(what))
183                  || (what.isRemote() && !this._remoteMoveOk(what))
184                  ||  what.disallowSubFolder
185                  );
186     } else { //ZmBriefcaseItem
187         var items = AjxUtil.toArray(what);
188 		var item = items[0];
189         if (item.type == ZmItem.BRIEFCASE_ITEM) {
190             invalid = this._checkInvalidFolderItems(items);
191 
192             if (!invalid) {
193                 for (var i = 0; i < items.length; i++) {
194                     if (items[i] instanceof ZmBriefcaseFolderItem && (items[i].id == this.id ||             // Can't move folder items to themselves
195                     		this.isChildOf(items[i].folder))) { // Can't move parent folder to child folder
196                         invalid = true;
197                         break;
198                     }
199                 }
200             }
201             
202             
203             // can't move items to folder they're already in; we're okay if
204             // we have one item from another folder
205             if (!invalid && item.folderId) {
206                 invalid = true;
207                 for (var i = 0; i < items.length; i++) {
208                     var tree = appCtxt.getById(items[i].folderId);
209                     if (tree != this) {
210                         invalid = false;
211                         break;
212                     }
213                 }
214             }
215         } else {
216             invalid = true;
217         }
218         
219         // attachments from mail can be moved inside briefcase
220 		if (item && item.msgId && item.partId) {
221 			invalid = false;
222 		}
223 
224     }
225 
226     if (!invalid && this.link) {
227         invalid = this.isReadOnly();
228     }	
229 
230 	return !invalid;
231 };
232 
233 ZmBriefcase.prototype._checkInvalidFolderItems =
234 function(items, targetFolderType) {
235   var invalid = false;
236   for (var i=0; i<items.length && !invalid; i++) {
237     if (items[i] instanceof ZmBriefcaseFolderItem) {
238         var item = items[i];
239         invalid = (
240            item.parent == this || this.isChildOf(item)
241         || targetFolderType == ZmOrganizer.SEARCH || targetFolderType == ZmOrganizer.TAG
242         || (!this.isInTrash() && this.hasChild(item.name))
243         || (item.id == this.id)
244         || (item.folder && item.folder.isRemote() && !this.isRemote() && !item.folder.rid)
245         || (item.folder && this.isRemote())
246         );
247     }
248   }
249     return invalid;
250 };

262 ZmBriefcase.prototype._generateRestUrl =
263 function() {
264 	var loc = document.location;
265 	var uname = this.getOwner();
266 	var host = loc.host;
267 	var m = uname.match(/^(.*)@(.*)$/);
268 
269 	host = (m && m[2]) || host;
270 
271 	// REVISIT: What about port? For now assume other host uses same port
272 	if (loc.port && loc.port != 80) {
273 		host = host + ":" + loc.port;
274 	}
275 
276 	var searchPath =  this.getSearchPath(true);
277 	var generatedRestURL = [loc.protocol, "//", host, "/service/user/", uname, "/", AjxStringUtil.urlEncode(searchPath)].join("");
278 	var restUrl = this.restUrl;
279 	var oname = this.oname;
280 	var parent = this.parent;
281 	//Get the restUrl and oname from remote share
282 	while (parent) {
283 		if (parent.restUrl) {
284 			restUrl = parent.restUrl;
285 		}
286 		if (parent.oname) {
287 			oname = parent.oname;
288 		}
289 		parent = parent.parent;
290 	}
291 
292 	if (restUrl) {
293 		var index = searchPath.indexOf(oname);  //remove oname from searchPath
294 		if (index != -1) {
295 			searchPath = searchPath.substring(index + oname.length);
296 		}
297 		generatedRestURL = restUrl +  AjxStringUtil.urlEncode(searchPath);
298 	}
299 	return generatedRestURL;
300 };
