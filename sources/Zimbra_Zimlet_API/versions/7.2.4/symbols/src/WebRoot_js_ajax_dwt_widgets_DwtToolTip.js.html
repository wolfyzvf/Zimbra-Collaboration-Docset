 20 DwtToolTip = function(shell, className, dialog) {
 21 	if (arguments.length == 0) return;
 22 	this.shell = shell;
 23 	this._dialog = dialog;
 24 	this._poppedUp = false;
 25 	this.isSticky = false;
 26 	this._div = document.createElement("div");
 27 	this._div.className = className || "DwtToolTip";
 28 	this._div.style.position = DwtControl.ABSOLUTE_STYLE;
 29 	this.shell.getHtmlElement().appendChild(this._div);
 30 	Dwt.setZIndex(this._div, Dwt.Z_HIDDEN);
 31 	Dwt.setLocation(this._div, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
 32 
 33     // create html
 34     // NOTE: This id is ok because there's only ever one instance of a tooltip
 35     var templateId = "dwt.Widgets#"+this._borderStyle;
 36     this._div.innerHTML = AjxTemplate.expand(templateId, "tooltip");
 37 
 38     var params = AjxTemplate.getParams(templateId);
 39     this._borderWidth = Number(params.width);
 40     this._borderHeight = Number(params.height);
 41 
 42     // save reference to content div
 43     this._contentDiv = document.getElementById("tooltipContents");
 44 
 45     Dwt.setHandler(this._div, DwtEvent.ONMOUSEOVER, AjxCallback.simpleClosure(this._mouseOverListener, this));
 46 }

 74 DwtToolTip.prototype.setContent =
 75 function(content, setInnerHTML) {
 76 	this._content = content;
 77 	if(setInnerHTML) {
 78         this._contentDiv.innerHTML = this._content;
 79     }

 82 DwtToolTip.prototype.popup = 
 83 function(x, y, skipInnerHTML, popdownOnMouseOver) {
 84     if (this._popupAction) {
 85         AjxTimedAction.cancelAction(this._popupAction);
 86         this._popupAction = null;
 87     }
 88     this._popdownOnMouseOver = popdownOnMouseOver; // popdownOnMouseOver may be truthy to pop down the tooltip if the mouse hovers over the tooltip. Optionally, it can be an AjxCallback that will be called after popping the tooltip down
 89     if (this._content != null) {
 90 		if(!skipInnerHTML) {
 91             this._contentDiv.innerHTML = this._content;
 92         }
 93 
 94 		this._popupAction = new AjxTimedAction(this, this._positionElement, [x, y]);
 95 		AjxTimedAction.scheduleAction(this._popupAction, 5);
 96 	}
 97 };

 99 /*
100 * setSticky allows making the tooltip not to popdown. 
101 * IMPORTANT: Tooltip is singleton inside Zimbra i.e. only one instance of tooltip is reused by all objects. 
102 * So, it is very important for the code setting tooltip to sticky to have some mechanism to close the tooltip by itself. 
103 * Like have a close-button inside tooltip and when clicked, should set the setSticky(false) and then close the tooltip.
104 * 
105 */
106 DwtToolTip.prototype.setSticky = 
107 function(bool) {
108 	this.isSticky = bool;
109 };
110 
111 DwtToolTip.prototype.popdown = 
112 function() {
113     this._popdownOnMouseOver = false;
114     if (this._popupAction) {
115         AjxTimedAction.cancelAction(this._popupAction);
116         this._popupAction = null;
117     }
118 	if (this._content != null && this._poppedUp) {
119 		Dwt.setLocation(this._div, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
120 		this._poppedUp = false;
121 	}
122 };

128 DwtToolTip.prototype._positionElement = 
129 function(startX, startY) {
130     this._popupAction = null;
131 
132     var element =  this._div;
133 	var baseId = "tooltip";
134 	var dialog = this._dialog;
135 
136 	var WINDOW_GUTTER = 5;
137 	var POPUP_OFFSET_X = 8;
138 	var POPUP_OFFSET_Y = 8;
139 
140     var topPointer = document.getElementById(baseId+'TopPointer');
141     topPointer.style.display = "block";
142 
143     var size = Dwt.getSize(topPointer),
144 		topPointerWidth = size.x,
145 		topPointerHeight = size.y
146 	;
147 
148 	var bottomPointer = document.getElementById(baseId+'BottomPointer');
149     bottomPointer.style.display = "block";
150     
151     size = Dwt.getSize(bottomPointer),
152 		bottomPointerWidth = size.x,
153 		bottomPointerHeight = size.y
154 	;
155 
156 	var pointer = topPointer;
157 
158 	var wdSize = DwtShell.getShell(window).getSize(),
159 		wdWidth = wdSize.x,
160 		wdHeight = wdSize.y
161 	;
162 
163 	var popupSize = Dwt.getSize(element),
164 		popupWidth = popupSize.x,
165 		popupHeight = popupSize.y
166 	;
167 
168 	var topBorderHeight = this._borderHeight,
169 		bottomBorderHeight = this._borderHeight,
170 		leftBorderWidth = this._borderWidth,
171 		rightBorderWidth = this._borderWidth
172 	;
173 
174 	var popupX = startX - popupWidth / 2 - POPUP_OFFSET_X,
175 		popupY
176 	;
177 	
178 	var pointerY,
179 		pointerX,
180 		pointerWidth
181 	;
182 
183 	// top pointer
184 	// NOTE: bottomPointerHeight added sbecause bottom pointer is absolute
185     if (startY + POPUP_OFFSET_Y + topPointerHeight - topBorderHeight + popupHeight < wdHeight - WINDOW_GUTTER) {
186         bottomPointer.style.display = "none";
187 		popupY = startY + POPUP_OFFSET_Y + topPointerHeight - topBorderHeight;
188 		pointerY = topBorderHeight - topPointerHeight;
189 		pointerWidth = topPointerWidth;
190 		pointer = topPointer;
191 	}
192 	
193 	// bottom pointer
194 	else {
195         topPointer.style.display = "none";
196         popupY = startY - POPUP_OFFSET_Y - bottomPointerHeight + bottomBorderHeight - popupHeight;
197 		pointerY = popupHeight - bottomBorderHeight;
198 		pointerWidth = bottomPointerWidth;
199 		pointer = bottomPointer;
200 	}
201 
202 	// make sure popup is wide enough for pointer
203 	var contentEl = document.getElementById(baseId+"Contents");
204 	if (popupWidth - leftBorderWidth - rightBorderWidth < pointerWidth) {
205 		contentEl.width = pointerWidth; // IE
206 		contentEl.style.width = String(pointerWidth)+"px"; // everyone else
207 	} else {
208 		contentEl.width = "auto"; // IE
209 		contentEl.style.width = "auto"; // everyone else
210 	}
211 	
212 	// adjust popup x-location
213 	if (popupX < WINDOW_GUTTER) {
214 		popupX = WINDOW_GUTTER;
215 	}
216 	else if (popupX + popupWidth > wdWidth - WINDOW_GUTTER) {
217 		popupX = wdWidth - WINDOW_GUTTER - popupWidth;
218 	}
219 	
220 	// adjust pointer x-location
221 	pointerX = startX - popupX - pointerWidth / 2;
222 	if (pointerX + pointerWidth > popupWidth - rightBorderWidth) {
223 		pointerX = popupWidth - rightBorderWidth - pointerWidth;
224 	}
225 	if (pointerX < leftBorderWidth) {
226 		pointerX = leftBorderWidth;
227 	}
228 
229     pointer.style.left = pointerX;
230 	pointer.style.top = pointerY;
231 
232 	Dwt.setLocation(element, popupX, popupY);
233 	var zIndex = dialog ? dialog.getZIndex() + Dwt._Z_INC : Dwt.Z_TOOLTIP;
234 	Dwt.setZIndex(element, zIndex);
235     this._poppedUp = true;
236 };
237 
238 DwtToolTip.prototype._mouseOverListener = 
239 function(ev) {
240 	if(this.isSticky) {
241 		return;
242 	}
243     if (this._popdownOnMouseOver && this._poppedUp) {
244         var callback = (this._popdownOnMouseOver instanceof AjxCallback) ? this._popdownOnMouseOver : null;
245         this.popdown();
246         if (callback)
247             callback.run();
248     }
249 };
