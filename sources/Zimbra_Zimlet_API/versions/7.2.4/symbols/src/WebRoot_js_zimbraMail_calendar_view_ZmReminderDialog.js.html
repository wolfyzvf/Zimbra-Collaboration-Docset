 21 ZmReminderDialog = function(parent, reminderController, calController) {
 22 
 23 	// init custom buttons
 24 	var selectId = Dwt.getNextId();
 25 	var html = [];
 26 	var i = 0;
 27 	html[i++] = "<td valign='middle' class='ZmReminderField'>";
 28 	html[i++] = ZmMsg.snoozeAll;
 29 	html[i++] = "</td><td valign='middle' id='";
 30 	html[i++] = selectId;
 31 	html[i++] = "'></td><td valign='middle' id=\"{0}\"></td>";
 32 	
 33 	var snoozeButton = new DwtDialog_ButtonDescriptor(ZmReminderDialog.SNOOZE_BUTTON, ZmMsg.snooze, DwtDialog.ALIGN_LEFT, null, html.join(""));
 34 	var dismissAllButton = new DwtDialog_ButtonDescriptor(ZmReminderDialog.DISMISS_ALL_BUTTON, ZmMsg.dismissAll, DwtDialog.ALIGN_RIGHT);
 35 
 36 	// call base class
 37 	DwtDialog.call(this, {id:"ZmReminderDialog", parent:parent, standardButtons:DwtDialog.NO_BUTTONS, extraButtons:[snoozeButton, dismissAllButton]});
 38 
 39 	this._reminderController = reminderController;
 40 	this._calController = calController;
 41 
 42     this.setContent(this._contentHtml(selectId));
 43     if(this._calController instanceof ZmTaskMgr) {
 44         this.setTitle(ZmMsg.taskReminders);
 45     } else {
 46         this.setTitle(ZmMsg.apptReminders);
 47     }
 48     this.registerCallback(ZmReminderDialog.SNOOZE_BUTTON, this._handleSnoozeButton, this);
 49 	this.registerCallback(ZmReminderDialog.DISMISS_ALL_BUTTON, this._handleDismissAllButton, this);
 50 };

 70 ZmReminderDialog.prototype.popup =
 71 function() {
 72 	DwtDialog.prototype.popup.call(this);
 73 	this._cancelSnooze();
 74 
 75     if (appCtxt.get(ZmSetting.CAL_REMINDER_NOTIFY_BROWSER)) {
 76         AjxPackage.require("Alert");
 77         ZmBrowserAlert.getInstance().start(ZmMsg.reminders);
 78     }
 79 
 80     if (appCtxt.get(ZmSetting.CAL_REMINDER_NOTIFY_SOUNDS)) {
 81         AjxPackage.require("Alert");
 82         ZmSoundAlert.getInstance().start();
 83     }
 84 
 85     if (appCtxt.get(ZmSetting.CAL_REMINDER_NOTIFY_TOASTER)) {
 86         AjxPackage.require("Alert");
 87         var winText = [];
 88         var appts = this._list.getArray();
 89         // only show, at most, five appointment reminders
 90         for (var i = 0; i < appts.length && i < 5; i++) {
 91             var appt = appts[i];
 92             var startDelta = this._computeDelta(appt);
 93             var delta = startDelta ? this._formatDeltaString(startDelta) : "";
 94             var text = [appt.getName(), ", ", this._getDurationText(appt), "\n(", delta, ")"].join("");
 95             if (AjxEnv.isMac) {
 96                 ZmDesktopAlert.getInstance().start(ZmMsg.reminders, text);
 97             } else if (AjxEnv.isWindows) {
 98                 winText.push(text);
 99             }
100         }
101 
102         if (AjxEnv.isWindows && winText.length > 0) {
103             if (appts.length > 5) {
104                 winText.push(ZmMsg.andMore);
105             }
106             ZmDesktopAlert.getInstance().start(ZmMsg.reminders, winText.join("\n"), 5);
107         }
108     }    
109 };

111 ZmReminderDialog.prototype.initialize =
112 function(list) {
113 	this._list = list.clone();
114 	this._apptData = {};
115 
116 	var html = [];
117 	var idx = 0;
118 	var size = list.size();
119 
120     AjxDebug.println(AjxDebug.REMINDER, "---Reminders [" + (new Date().getTime())+ "]---");
121 
122 	html[idx++] = "<table cellpadding=0 cellspacing=0 border=0 width=100%>";
123 	for (var i = 0; i < size; i++) {
124 		var appt = list.get(i);
125 		var uid = appt.getUniqueId(true);
126 		var data = this._apptData[uid] = {appt:appt};
127 		idx = this._addAppt(html, idx, appt, data, (i > 0));
128 	}
129 	html[idx++] = "</table>";
130 
131 	// cleanup before using
132 	if (this._dismissButtons) {
133 		for (var id in this._dismissButtons) {
134 			this._dismissButtons[id].dispose();
135 		}
136 	}
137 	if (this._openButtons) {
138 		for (var id in this._openButtons) {
139 			this._openButtons[id].dispose();
140 		}
141 	}
142 	this._dismissButtons = {};
143 	this._openButtons = {};
144 
145 	var dismissListener = new AjxListener(this, this._dismissButtonListener);
146 	var openListener = new AjxListener(this, this._openButtonListener);
147 	var div = document.getElementById(this._listId);
148 	div.innerHTML = html.join("");
149 
150 	for (var i = 0; i < size; i++) {
151 		var appt = list.get(i);
152 		var uid = appt.getUniqueId(true);
153 		var id = appt.id;
154 		var data = this._apptData[uid];
155 
156         var alarmData = appt.getAlarmData();
157         alarmData = (alarmData && alarmData.length > 0) ? alarmData[0] : {};
158         //bug: 60692 - Add troubleshooting code for late reminders
159         AjxDebug.println(AjxDebug.REMINDER, appt.getReminderName() + " : " + (alarmData.nextAlarm || " NA ") + " / " + (alarmData.alarmInstStart || " NA "));
160 
161 		// dismiss button
162 		var dismissBtn = this._dismissButtons[data.dismissBtnId] = new DwtButton({id:"dismissBtn_" + id, parent:this, className:"DwtToolbarButton", parentElement:data.dismissBtnId});
163 		dismissBtn.setImage("Cancel");
164 		dismissBtn.setText(ZmMsg.dismiss);
165 		dismissBtn.addSelectionListener(dismissListener);
166 		dismissBtn.apptUid = uid;
167 
168 		// open button
169 		var openBtn = this._openButtons[data.openBtnId] = new DwtButton({id:"openBtn_" + id, parent:this, className:"DwtToolbarButton", parentElement:data.openBtnId});
170 		openBtn.setImage(appt.otherAttendees ? "ApptMeeting" : (appt.type == ZmItem.TASK) ? "TasksApp" : "Appointment");
171 		openBtn.setText(ZmMsg.viewAppointment);
172 		openBtn.addSelectionListener(openListener);
173 		openBtn.apptUid = uid;
174 
175 		this._updateDelta(data);
176 	}
177 };

179 ZmReminderDialog.prototype._contentHtml =
180 function(selectId) {
181 
182     this._listId = Dwt.getNextId("ZmReminderDialogContent");//Dwt.getNextId();
183 
184 	var snooze = [1, 5, 10, 15, 30, 45, 60];
185     if(this._select) {
186         this._select.dispose();
187     }
188 	this._select = new DwtSelect({id:"ZmReminderDialog_reminderSelect", parent:this});
189 	var snoozeFormatter = new AjxMessageFormat(ZmMsg.reminderSnoozeMinutes);
190 	for (var i = 0; i < snooze.length; i++) {
191 		var label = snoozeFormatter.format(snooze[i]);
192 		this._select.addOption(label, i==0, snooze[i]);
193 	}
194 	this._select.reparentHtmlElement(selectId);
195 
196 	return ["<div class='ZmReminderDialog' id='", this._listId, "'>"].join("");
197 };

199 ZmReminderDialog.prototype._updateDelta = 
200 function(data) {
201 	var td = document.getElementById(data.deltaId);
202 	if (td) {
203 		var startDelta = this._computeDelta(data.appt);
204 
205 		if (startDelta >= 0) 							td.className = 'ZmReminderOverdue';
206 		else if (startDelta > ZmReminderDialog.SOON)	td.className = 'ZmReminderSoon';
207 		else											td.className = 'ZmReminderFuture';
208 
209 		td.innerHTML = startDelta ? this._formatDeltaString(startDelta) : "";
210 	}
211 };

213 ZmReminderDialog.prototype._addAppt =
214 function(html, idx, appt, data, needSep) {
215 
216     var uid = appt.id;
217 	data.dismissBtnId = "dismissBtnContainer_" + uid;
218 	data.openBtnId = "openBtnContainer_" + uid;
219 	data.deltaId = "delta_" + uid;
220 	data.rowId = "apptRow_" + uid;
221 	data.reminderNameContainerId = "reminderNameContainerId_" + uid;
222 	data.reminderDescContainerId = "reminderDescContainerId_" + uid;
223 
224 	var calName = (appt.folderId != ZmOrganizer.ID_CALENDAR && appt.folderId != ZmOrganizer.ID_TASKS && this._calController)
225 		? this._calController.getCalendarName(appt.folderId) : null;
226 
227 
228 	var calendar = appCtxt.getById(appt.folderId);
229 
230 	var params = {
231 		needSep: needSep,
232 		rowId: data.rowId,
233 		calName: AjxStringUtil.htmlEncode(calName),
234 		accountName: (appCtxt.multiAccounts && calendar && calendar.getAccount().getDisplayName()),
235 		location: (AjxStringUtil.htmlEncode(appt.getReminderLocation())),
236 		apptIconHtml: (AjxImg.getImageHtml(appt.otherAttendees ? "ApptMeeting" : "Appointment")),
237 		organizer: appt.otherAtt ? appt.organizer : null,
238 		reminderName: (AjxStringUtil.htmlEncode(appt.getReminderName())),
239 		durationText: (AjxStringUtil.trim(this._getDurationText(appt))),
240 		deltaId: data.deltaId,
241 		openBtnId: data.openBtnId,
242 		dismissBtnId: data.dismissBtnId,
243         reminderNameContainerId: data.reminderNameContainerId,
244         reminderDescContainerId: data.reminderDescContainerId,
245         type: appt.type ? appt.type : ZmItem.APPT
246 	};
247 	html[idx++] = AjxTemplate.expand("calendar.Calendar#ReminderDialogRow", params);
248 	return idx;
249 };

251 ZmReminderDialog.prototype._openButtonListener =
252 function(ev) {
253 	appCtxt.getAppController().setStatusMsg(ZmMsg.allRemindersAreSnoozed, ZmStatusView.LEVEL_INFO);
254 
255 	this._handleSnoozeButton();
256 
257 	var obj = DwtControl.getTargetControl(ev);
258 	var data = this._apptData[obj.apptUid];
259 	var appt = data ? data.appt : null;
260     var type = appt.type ? appt.type : ZmItem.APPT;
261 	if (appt && type == ZmItem.APPT) {
262 		AjxDispatcher.require(["CalendarCore", "Calendar"]);
263 
264 		var cc = AjxDispatcher.run("GetCalController");
265 
266 		// the give appt object is a ZmCalBaseItem. We need a ZmAppt
267 		var newAppt = new ZmAppt();
268 		for (var i in appt) {
269 			if (!AjxUtil.isFunction(appt[i])) {
270 				newAppt[i] = appt[i];
271 			}
272 		}
273 		var callback = new AjxCallback(cc, cc._showAppointmentDetails, newAppt);
274 		newAppt.getDetails(null, callback, null, null, true);
275 	} else if(appt && type == ZmItem.TASK) {
276         AjxDispatcher.require(["TasksCore", "Tasks"]);
277 
278 		var tlc = AjxDispatcher.run("GetTaskListController");
279 
280 		// the give appt object is a ZmCalBaseItem. We need a ZmAppt
281 		var newTask = new ZmTask();
282 		for (var i in appt) {
283 			if (!AjxUtil.isFunction(appt[i])) {
284 				newTask[i] = appt[i];
285 			}
286 		}
287 		var callback = new AjxCallback(tlc, tlc._editTask, newTask);
288 		newTask.getDetails(null, callback, null, null, true);
289     }
290 };

292 ZmReminderDialog.prototype._dismissButtonListener =
293 function(ev) {
294 	var obj = DwtControl.getTargetControl(ev);
295 	var data = this._apptData[obj.apptUid];
296 	if (!data) { return; }
297 
298 	this._reminderController.dismissAppt(data.appt);
299 
300 	// cleanup HTML
301 	var dismissBtn = this._dismissButtons[data.dismissBtnId];
302 	if (dismissBtn) {
303 		dismissBtn.dispose();
304 		delete this._dismissButtons[data.dismissBtnId];
305 	}
306 	var openBtn = this._openButtons[data.openBtnId];
307 	if (openBtn) {
308 		openBtn.dispose();
309 		delete this._openButtons[data.openBtnId];
310 	}
311 	var row = document.getElementById(data.rowId);
312 	if (row) {
313 		var nextRow = row.nextSibling;
314 		if (nextRow && nextRow.getAttribute("name") == "rdsep") {
315 			nextRow.parentNode.removeChild(nextRow);
316 		}
317 		row.parentNode.removeChild(row);
318 	}
319 
320 	delete this._apptData[obj.apptUid];
321 	this._list.remove(data.appt);
322 
323 	if (this._list.size() == 0) {
324 		this.popdown();
325 	}
326 };

329 function() {

359 ZmReminderDialog.prototype._getDurationText =
360 function(appt) {
361 	var isMultiDay = appt.isMultiDay();
362 	var start = appt._alarmInstStart ? new Date(appt._alarmInstStart) : appt.startDate ? appt.startDate : null;
363 	// bug: 28598 - alarm for recurring appt might still point to old alarm time
364 	// cannot take endTime directly
365 	var endTime = appt._alarmInstStart ? (start.getTime() + appt.getDuration()) : appt.getEndTime();
366 	var end = new Date(endTime);
367 
368     //for task
369     if(appt.type == ZmItem.TASK && !start && !endTime) { return null; }
370 
371 	if (appt.isAllDayEvent()) {
372 		end = appt.type != ZmItem.TASK ? new Date(endTime - (isMultiDay ? 2 * AjxDateUtil.MSEC_PER_HOUR : 0)) : end;
373 		var pattern = isMultiDay ? ZmMsg.apptTimeAllDayMulti : ZmMsg.apptTimeAllDay;
374 		return start ? AjxMessageFormat.format(pattern, [start, end]) : AjxMessageFormat.format(pattern, [end]); //for task
375 	}
376 	var pattern = isMultiDay ? ZmMsg.apptTimeInstanceMulti : ZmMsg.apptTimeInstance;
377 	return AjxMessageFormat.format(pattern, [start, end, ""]);
378 };

380 ZmReminderDialog.prototype._computeDelta =
381 function(appt) {
382 	return (appt.alarmData && appt.alarmData.length > 0)
383 		? (appt.alarmData[0].alarmInstStart ? (new Date()).getTime() - appt.adjustMS(appt.alarmData[0].alarmInstStart, appt.tzo) : appt.getEndTime()  ? (new Date()).getTime() - appt.getEndTime() : null)
384 		: ((new Date()).getTime() - appt.getStartTime());
385 };

387 ZmReminderDialog.prototype._formatDeltaString =
388 function(deltaMSec) {
389 	var prefix = deltaMSec < 0 ? "In" : "OverdueBy";
390 	deltaMSec = Math.abs(deltaMSec);
391 
392 	// calculate parts
393 	var years =  Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 365));
394 	if (years != 0)
395 		deltaMSec -= years * AjxDateUtil.MSEC_PER_DAY * 365;
396 	var months = Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 30.42));
397 	if (months > 0)
398 		deltaMSec -= Math.floor(months * AjxDateUtil.MSEC_PER_DAY * 30.42);
399 	var days = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_DAY);
400 	if (days > 0)
401 		deltaMSec -= days * AjxDateUtil.MSEC_PER_DAY;
402 	var hours = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_HOUR);
403 	if (hours > 0)
404 		deltaMSec -= hours * AjxDateUtil.MSEC_PER_HOUR;
405 	var mins = Math.floor(deltaMSec / 60000);
406 	if (mins > 0)
407 		deltaMSec -= mins * 60000;
408 	var secs = Math.floor(deltaMSec / 1000);
409 	if (secs > 30 && mins < 59) mins++;
410 
411 	var secs = 0;
412 
413 	// determine message
414 	var amount;
415 	if (years > 0) {
416 		amount = "Years";
417 		if (years <= 3 && months > 0) {
418 			amount = "YearsMonths";
419 		}
420 	} else if (months > 0) {
421 		amount = "Months";
422 		if (months <= 3 && days > 0) {
423 			amount = "MonthsDays";
424 		}
425 	} else if (days > 0) {
426 		amount = "Days";
427 		if (days <= 2 && hours > 0) {
428 			amount = "DaysHours";
429 		}
430 	} else if (hours > 0) {
431 		amount = "Hours";
432 		if (hours < 5 && mins > 0) {
433 			amount = "HoursMinutes";
434 		}
435 	} else {
436 		amount = "Minutes";
437 	}
438 
439 	// format message
440 	var key = ["reminder",prefix,amount].join("");
441 	var args = [deltaMSec, years, months, days, hours, mins, secs];
442 	return AjxMessageFormat.format(ZmMsg[key], args);
443 };
