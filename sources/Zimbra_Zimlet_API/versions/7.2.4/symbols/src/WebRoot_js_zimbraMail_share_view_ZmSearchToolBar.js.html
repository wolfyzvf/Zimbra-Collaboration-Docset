 21 ZmSearchToolBar = function(parent, id) {
 22 
 23 	DwtComposite.call(this, {parent:parent, className:"ZmSearchToolbar", id:id});
 24 
 25 	// setup "search all" menu item
 26 	var params = {
 27 		msgKey: "searchAll",
 28 		tooltipKey: "searchForAny",
 29 		icon: "Globe",
 30 		setting: ZmSetting.MIXED_VIEW_ENABLED,
 31 		index: 0,
 32 		id: ZmId.getMenuItemId(ZmId.SEARCH, ZmId.SEARCH_ANY)
 33 	};
 34 	ZmSearchToolBar.addMenuItem(ZmId.SEARCH_ANY, params);
 35 
 36 	// setup "include shared" menu item
 37 	params = {
 38 		msgKey: "searchShared",
 39 		tooltipKey: "searchShared",
 40 		icon: "Group",
 41 		setting: ZmSetting.SHARING_ENABLED,
 42 		id: ZmId.getMenuItemId(ZmId.SEARCH, ZmId.SEARCH_SHARED)
 43 	};
 44 	ZmSearchToolBar.addMenuItem(ZmId.SEARCH_SHARED, params);
 45 
 46 	// setup "all accounts" menu item for multi account
 47 	if (appCtxt.multiAccounts) {
 48 		params = {
 49 			msgKey: "searchAllAccounts",
 50 			icon: "Globe",
 51 			id: ZmId.getMenuItemId(ZmId.SEARCH, ZmId.SEARCH_ALL_ACCOUNTS)
 52 		};
 53 		ZmSearchToolBar.addMenuItem(ZmId.SEARCH_ALL_ACCOUNTS, params);
 54 	}
 55 
 56 	this._createHtml();
 57 };

 82 ZmSearchToolBar.addMenuItem =
 83 function(id, params) {
 84 	if (params.msgKey)		{ ZmSearchToolBar.MSG_KEY[id]		= params.msgKey; }
 85 	if (params.tooltipKey)	{ ZmSearchToolBar.TT_MSG_KEY[id]	= params.tooltipKey; }
 86 	if (params.icon)		{ ZmSearchToolBar.ICON[id]			= params.icon; }
 87 	if (params.shareIcon)	{ ZmSearchToolBar.SHARE_ICON[id]	= params.shareIcon; }
 88 	if (params.setting)		{ ZmSearchToolBar.SETTING[id]		= params.setting; }
 89 	if (params.id)			{ ZmSearchToolBar.ID[id]			= params.id; }
 90 
 91 	// test for null since index value can be zero :)
 92 	if (params.index == null || params.index < 0 || params.index >= ZmSearchToolBar.MENU_ITEMS.length) {
 93 		ZmSearchToolBar.MENU_ITEMS.push(id);
 94 	} else {
 95 		ZmSearchToolBar.MENU_ITEMS.splice(params.index, 0, id);
 96 	}
 97 };

107 ZmSearchToolBar.prototype.removeMenuItem =
108 function(id) {
109 	var idx = 0;
110 	
111 	while (idx < ZmSearchToolBar.MENU_ITEMS.length) {
112 		if ( ZmSearchToolBar.MENU_ITEMS[idx] == id ) { 
113 			break;
114 		}
115 		idx++;
116 	}
117 
118 	if (idx < ZmSearchToolBar.MENU_ITEMS.length) {
119 		var menu = this._searchMenuButton.getMenu();
120 		menu.removeChild(menu.getItemById("_menuItemId",id));
121 		ZmSearchToolBar.MENU_ITEMS.splice(idx,1);
122 		ZmSearchToolBar.MSG_KEY[id]		= "";
123 		ZmSearchToolBar.TT_MSG_KEY[id]	= "";
124 		ZmSearchToolBar.ICON[id]		= "";
125 		ZmSearchToolBar.SHARE_ICON[id]	= "";
126 		ZmSearchToolBar.SETTING[id]		= "";
127 		ZmSearchToolBar.ID[id]			= "";
128 	}
129 	this.dedupSeparators(menu);
130 };

132 // Attempt to dedup separators, and remove any that start or end the menu
133 ZmSearchToolBar.prototype.dedupSeparators =
134 function(menu) {
135 	if (menu == null) {
136 		menu = this._searchMenuButton.getMenu();
137 	}
138 
139 	var children = menu.getItems();
140 	var wasSep = false;
141 	var toRemove = [];
142 	for (mi in children) {
143 		if (!children[mi].__text) {
144 			if (wasSep == true || wasSep == null) {
145 				toRemove.push(children[mi]);
146 			} else {
147 				wasSep = true;
148 			}
149 		} else {
150 			wasSep = false;
151 		}
152 	}
153 	for (mi in toRemove) {
154 		menu.removeChild(toRemove[mi]);
155 	}
156 	if (!children[children.length-1].__text) {	 // No trailing separators
157 		menu.removeChild(children[children.length-1]);
158 	}
159 };

240 ZmSearchToolBar.prototype.createCustomSearchBtn =
241 function(icon, text, listener, id) {
242 	if (!this._customSearchListener) {
243 		this._customSearchListener = new AjxListener(this, this._customSearchBtnListener);
244 	}
245 
246 	// check if custom search should be a button by checking for the Id against the template
247 	var customSearchBtn = document.getElementById(this._htmlElId + "_customSearchButton");
248 	if (customSearchBtn) {
249 		if (!this._customSearchBtn) {
250 			this._customSearchBtn = this._addButton({ tdId:		"_customSearchButton",
251 													  buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_CUSTOM),
252 													  lbl:		text,
253 													  icon:		icon });
254 			this._customSearchBtn.setData("CustomSearchItem", [ icon, text, listener ]);
255 			this._customSearchBtn.addSelectionListener(this._customSearchListener);
256 
257 			// show the separator now that we've added a custom search button
258 			var sep = document.getElementById(this._htmlElId + "_customSearchButtonSep");
259 			if (sep) {
260 				Dwt.setVisible(sep, true);
261 			}
262 		} else {
263 			var menu = this._customSearchBtn.getMenu();
264 			var item;
265 			var params = {parent:menu, enabled:true, style:DwtMenuItem.RADIO_STYLE, radioGroupId:0, id:id};
266 			if (!menu) {
267 				var data = this._customSearchBtn.getData("CustomSearchItem");
268 				menu = new DwtMenu({parent:this._customSearchBtn, className:"ActionMenu", id:ZmId.getMenuId(ZmId.SEARCH, ZmId.SEARCH_CUSTOM)});
269 				this._customSearchBtn.setMenu(menu, false, DwtMenuItem.RADIO_STYLE);
270 				params.imageInfo = data[0];
271 				params.text = data[1];
272 				item = DwtMenuItem.create(params);
273 				item.setData("CustomSearchItem", data);
274 				item.setData(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_CUSTOM);
275 				item.setChecked(true, true);
276 				item.addSelectionListener(this._customSearchListener);
277 			}
278 			params.imageInfo = icon;
279 			params.text = text;
280 			item = DwtMenuItem.create(params);
281 			item.setData("CustomSearchItem", [icon, text, listener] );
282 			item.addSelectionListener(this._customSearchListener);
283 		}
284 	} else {
285 		if (this._searchMenuCreated) {
286 			var menu = this._searchMenuButton.getMenu();
287 			this._createCustomSearchMenuItem(menu, icon, text, listener, id);
288 		} else {
289 			if (!this._customSearchMenuItems) {
290 				this._customSearchMenuItems = [];
291 			}
292 			this._customSearchMenuItems.push({icon:icon, text:text, listener:listener, id:id});
293 		}
294 	}
295 };

297 ZmSearchToolBar.prototype._createCustomSearchMenuItem =
298 function(menu, icon, text, listener, id) {
299 	var mi = menu.getItem(0);
300 	var params = {
301 		parent: menu,
302 		imageInfo: icon,
303 		text: text,
304 		enabled: true,
305 		style: DwtMenuItem.RADIO_STYLE,
306 		radioGroupId: 0,
307 		index: 0,
308 		id: id
309 	};
310 	mi = DwtMenuItem.create(params);
311 	mi.setData("CustomSearchItem", [icon, text, listener]);
312 	mi.setData(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_CUSTOM);
313 	mi.addSelectionListener(this._customSearchListener);
314 
315 	// only add separator if this is the first custom search menu item
316 	if (!(mi && mi.getData("CustomSearchItem"))) {
317 		mi = new DwtMenuItem({parent:menu, style:DwtMenuItem.SEPARATOR_STYLE, index:1});
318 	}
319 };

321 ZmSearchToolBar.prototype._customSearchBtnListener = 
322 function(ev) {
323 	var item = ev.item;
324 	if (!item) { return; }
325 	var data = item.getData("CustomSearchItem");
326 	if (this._customSearchBtn) {
327 		if (item instanceof DwtMenuItem) {
328 			if (ev.detail != DwtMenuItem.CHECKED) { return; }
329 			this._customSearchBtn.setToolTipContent(data[1]);
330 			this._customSearchBtn.setData("CustomSearchItem", data);
331 		}
332 		data[2].run(ev); // call original listener
333 	} else {
334 		this._searchMenuButton.setToolTipContent(data[1]);
335 
336 		var menu = item.parent;
337 		var shareMenuItem = menu ? menu.getItemById(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_SHARED) : null;
338 		if (shareMenuItem) {
339 			shareMenuItem.setChecked(false, true);
340 			shareMenuItem.setEnabled(false);
341 		}
342 
343 		this._searchMenuButton.setImage(data[0]);
344 		this._searchMenuButton.setText(data[1]);
345 	}
346 };

351 ZmSearchToolBar.prototype._createHtml =
352 function() {
353 
354 	this.getHtmlElement().innerHTML = AjxTemplate.expand("share.Widgets#ZmSearchToolBar", {id:this._htmlElId});
355 
356 	// add search input field
357 	var inputFieldId = this._htmlElId + "_inputField";
358 	var inputField = document.getElementById(inputFieldId);
359 	if (inputField) {
360 		this._searchField = new DwtInputField({parent:this, hint:ZmMsg.searchInput, inputId:ZmId.SEARCH_INPUTFIELD});
361 		var inputEl = this._searchField.getInputElement();
362 		inputEl.className = "search_input";
363 		this._searchField.reparentHtmlElement(inputFieldId);
364 		this._searchField.addListener(DwtEvent.ONKEYUP, new AjxListener(this, this._handleKeyUp));
365 		this._searchField._showHint();
366 	}
367 
368 	// add "search types" menu
369 	var searchMenuBtnId = this._htmlElId + "_searchMenuButton";
370 	var searchMenuBtn = document.getElementById(searchMenuBtnId);
371 	if (searchMenuBtn) {
372 		var mailEnabled = appCtxt.get(ZmSetting.MAIL_ENABLED);
373 		this._searchMenuButton = this._addButton({ tdId:		"_searchMenuButton",
374 												   buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_MENU),
375 												   lbl:			mailEnabled ? ZmMsg.searchMail : ZmMsg.searchAll,
376 												   icon:		mailEnabled ? "Message" : "Globe" });
377 		var menu = new AjxCallback(this, this._createSearchMenu);
378 		this._searchMenuButton.setMenu(menu, false, DwtMenuItem.RADIO_STYLE);
379 		this._searchMenuButton.reparentHtmlElement(searchMenuBtnId);
380 	}
381 
382 	// add search button
383 	this._searchButton = this._addButton({ tdId:		"_searchButton",
384 										   buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_SEARCH),
385 										   lbl:			ZmMsg.search,
386 										   icon:		"Search",
387 										   tooltip:		ZmMsg.searchTooltip });
388 
389 	// add save search button if saved-searches enabled
390 	this._saveButton = this._addButton({ setting:	ZmSetting.SAVED_SEARCHES_ENABLED,
391 										 tdId:		"_saveButton",
392 										 buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_SAVE),
393 										 lbl:		ZmMsg.save,
394 										 icon:		"Save",
395 										 type:		"toolbar",
396 										 tooltip:	ZmMsg.saveSearchTooltip });
397 			
398 
399 
400 	// add advanced search button
401 	this._browseButton = this._addButton({ setting:		ZmSetting.BROWSE_ENABLED,
402 										   tdId:		"_advancedButton",
403 										   buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_ADVANCED),
404 										   style:		(DwtLabel.IMAGE_LEFT | DwtLabel.ALIGN_CENTER | DwtButton.TOGGLE_STYLE),
405 										   lbl:			ZmMsg.searchBuilder,
406 										   icon:		"SearchBuilder",
407 										   type:		"toolbar",
408 										   tooltip:		ZmMsg.openSearchBuilder });
409 };

411 ZmSearchToolBar.prototype._createSearchMenu =
412 function() {
413 	var menu = new DwtMenu({parent:this._searchMenuButton, className:"ActionMenu", id:ZmId.getMenuId(ZmId.SEARCH)});
414 	var mi;
415 	if (this._customSearchMenuItems) {
416 		for (var i = 0; i < this._customSearchMenuItems.length; i++) {
417 			var csmi = this._customSearchMenuItems[i];
418 			this._createCustomSearchMenuItem(menu, csmi.icon, csmi.text, csmi.listener);
419 		}
420 	}
421 	var params = {parent:menu, enabled:true, radioGroupId:0};
422 	for (var i = 0; i < ZmSearchToolBar.MENU_ITEMS.length; i++) {
423 		var id = ZmSearchToolBar.MENU_ITEMS[i];
424 
425 		// add separator *before* "shared" menu item
426 		if (id == ZmId.SEARCH_SHARED) {
427 			if (ZmSearchToolBar.MENU_ITEMS.length <= 1) { continue; }
428 			mi = new DwtMenuItem({parent:menu, style:DwtMenuItem.SEPARATOR_STYLE});
429 		}
430 
431 		var setting = ZmSearchToolBar.SETTING[id];
432 		if (setting && !appCtxt.get(setting)) { continue; }
433 
434 		params.style = (id == ZmId.SEARCH_SHARED || id == ZmId.SEARCH_ALL_ACCOUNTS)
435 			? DwtMenuItem.CHECK_STYLE : DwtMenuItem.RADIO_STYLE;
436 		params.imageInfo = ZmSearchToolBar.ICON[id];
437 		params.text = ZmMsg[ZmSearchToolBar.MSG_KEY[id]];
438 		params.id = ZmSearchToolBar.ID[id];
439 		mi = DwtMenuItem.create(params);
440 		mi.setData(ZmSearchToolBar.MENUITEM_ID, id);
441 
442 		// add separator *after* "all" menu item
443 		if (id == ZmId.SEARCH_ANY) {
444 			if (ZmSearchToolBar.MENU_ITEMS.length <= 1) { continue; }
445 			mi = new DwtMenuItem({parent:menu, style:DwtMenuItem.SEPARATOR_STYLE});
446 		}
447 	}
448 
449 	appCtxt.getSearchController()._addMenuListeners(menu);
450 	this._searchMenuCreated = true;
451 
452 	return menu;
453 };

455 /**
456  * Creates a button on the search toolbar.
457  * 
458  * @param params	[hash]		hash of params:
459  *        setting	[const]		setting that must be true for this button to be added
460  *        tdId		[string]	ID of TD that is to contain this button
461  *        buttonId	[string]*	ID of the button
462  *        style		[const]*	button style
463  *        type		[string]*	used to differentiate between regular and toolbar buttons
464  *        lbl		[string]*	button text
465  *        icon		[string]*	button icon
466  *        tooltip	[string]*	button tooltip
467  */
468 ZmSearchToolBar.prototype._addButton =
469 function(params) {
470 	if (params.setting && !appCtxt.get(params.setting)) { return; }
471 
472 	var button;
473 	var tdId = this._htmlElId + (params.tdId || params.buttonId);
474 	var buttonEl = document.getElementById(tdId);
475 	if (buttonEl) {
476 		var btnParams = {parent:this, style:params.style, id:params.buttonId};
477 		button = (params.type && params.type == "toolbar") ? (new DwtToolBarButton(btnParams)) : (new DwtButton(btnParams));
478 		var hint = Dwt.getAttr(buttonEl, "hint");
479 		this._setButtonStyle(button, hint, params.lbl, params.icon);
480 		if (params.tooltip) {
481 			button.setToolTipContent(params.tooltip);
482 		}
483 		button.reparentHtmlElement(tdId);
484 	}
485 
486 	return button;
487 };

489 ZmSearchToolBar.prototype._setButtonStyle =
490 function(button, hint, text, image) {
491 	if (hint == "text") {
492 		button.setText(text);
493 	} else if (hint == "icon") {
494 		button.setImage(image);
495 	} else { // add icon and text if no hint (or unsupported hint) provided
496 		button.setText(text);
497 		button.setImage(image);
498 	}

501 ZmSearchToolBar.prototype._handleKeyUp =
502 function(ev) {
503 	var code = DwtKeyEvent.getCharCode(ev);
504 	if (code == 0x0D) {
505 		return this._handleEnterKeyPress(ev);
506 	}
507 	return true;
508 };

510 ZmSearchToolBar.prototype._handleEnterKeyPress =
511 function(ev) {
512 	var menu = this._searchMenuButton && this._searchMenuButton.getMenu();
513     var item = menu ? menu.getSelectedItem() || menu.getItems()[0] : null
514 	var data = item && item.getData("CustomSearchItem");
515 	if (data) {
516 		data[2].run(ev); // call original listener
517 	} else {
518 		var searchFor = item.getData(ZmSearchToolBar.MENUITEM_ID);
519 		var queryString = this.getSearchFieldValue();
520 		appCtxt.notifyZimlets("onKeyPressSearchField", [queryString]);
521 		this._callback.run(queryString, searchFor);
522 	}
523 	return false;
524 };
