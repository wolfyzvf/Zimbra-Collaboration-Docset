 20 /**
 21  * Creates the version assistant.
 22  * @class
 23  * This class represents a version assistant.
 24  * 
 25  * @extends		ZmAssistant
 26  */
 27 ZmVersionAssistant = function() {
 28 	ZmAssistant.call(this, "Client Version Information", ".version");
 29 };
 30 
 31 ZmVersionAssistant.prototype = new ZmAssistant();
 32 ZmVersionAssistant.prototype.constructor = ZmVersionAssistant;
 33 
 34 ZmVersionAssistant.prototype.handle =
 35 function(dialog, verb, args) {
 36 	dialog._setOkButton(AjxMsg.ok, true, true);
 37 	this._setField("Version", appCtxt.get(ZmSetting.CLIENT_VERSION), false, true);
 38 	this._setField("Release", appCtxt.get(ZmSetting.CLIENT_RELEASE), false, true);
 39 	this._setField("Build Date", appCtxt.get(ZmSetting.CLIENT_DATETIME), false, true);	
 40 };
