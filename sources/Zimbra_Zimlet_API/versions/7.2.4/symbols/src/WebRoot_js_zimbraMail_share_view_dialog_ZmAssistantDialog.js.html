 20 /**
 21  * Creates an assistant dialog.
 22  * @class
 23  * This class represents an assistant dialog.
 24  * 
 25  * @extends		DwtDialog
 26  */
 27 ZmAssistantDialog = function() {
 28 
 29 	var helpButton = new DwtDialog_ButtonDescriptor(ZmAssistantDialog.HELP_BUTTON, ZmMsg.help, DwtDialog.ALIGN_LEFT);
 30 														   
 31 	var extraButton = new DwtDialog_ButtonDescriptor(ZmAssistantDialog.EXTRA_BUTTON, 
 32 														   ZmMsg.moreDetails, DwtDialog.ALIGN_LEFT);														   
 33 														   
 34 	DwtDialog.call(this, {parent:appCtxt.getShell(), className:"ZmAssistantDialog", title:ZmMsg.zimbraAssistant,
 35 						  extraButtons:[helpButton, extraButton]});
 36 
 37 	this.setContent(this._contentHtml());
 38 	this._initContent();
 39 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._okButtonListener));
 40 	this.setButtonListener(ZmAssistantDialog.EXTRA_BUTTON, new AjxListener(this, this._extraButtonListener));	
 41 	this.setButtonListener(ZmAssistantDialog.HELP_BUTTON, new AjxListener(this, this._helpButtonListener));
 42 
 43 	// only trigger matching after a sufficient pause
 44 	this._parseInterval = 75; //appCtxt.get(ZmSetting.AC_TIMER_INTERVAL);
 45 	this._parseTimedAction = new AjxTimedAction(this, this._parseAction);
 46 	this._parseActionId = -1;
 47 
 48 	ZmAssistantDialog.initializeAssistants();
 49 };

 55 /**
 56  * Initializes the assistants.
 57  * 
 58  * @see		ZmDebugAssistant
 59  * @see		ZmVersionAssistant
 60  */
 61 ZmAssistantDialog.initializeAssistants = function() {
 62 	if (!ZmAssistantDialog._handlerInit) {
 63 		ZmAssistant.register(new ZmVersionAssistant());
 64 		ZmAssistant.register(new ZmDebugAssistant());
 65 		for (var i = 0; i < ZmApp.APPS.length; i++) {
 66 			var app = ZmApp.APPS[i];
 67 			var setting = ZmApp.SETTING[app];
 68 			if (!setting || appCtxt.get(setting)) {
 69 				var assistants = ZmApp.ASSISTANTS[app];
 70 				if (assistants) {
 71 					for (var asstName in assistants) {
 72 						var pkg = assistants[asstName];
 73 						AjxDispatcher.require(pkg);
 74 						var asst = eval(asstName);
 75 						ZmAssistant.register(new asst());
 76 					}
 77 				}
 78 			}
 79 		}
 80 	}
 81 };

 88 /**
 89 */
 90 ZmAssistantDialog.prototype.popup =
 91 function() {
 92 	this._commandEl.value = "";
 93 	var commands = ZmAssistant.getHandlerCommands().join(", ");
 94 	this._availableCommands = ZmMsg.ASST_availableCommands+ " " + commands;
 95 	this._setDefault();
 96 	this._setHelpButton(null, true, true);
 97 	DwtDialog.prototype.popup.call(this);
 98 	this._commandEl.focus();
 99 };

101 /*
102 * Returns HTML that forms the basic framework of the dialog.
103 */
104 ZmAssistantDialog.prototype._contentHtml =
105 function() {
106 	var html = new AjxBuffer();
107 	this._contentId = Dwt.getNextId();	
108 	this._commandId = Dwt.getNextId();
109 	this._commandTitleId = Dwt.getNextId();
110 	html.append("<table cellspacing=3 border=0 width=400>");
111 	html.append("<tr><td colspan=3>", ZmMsg.enterCommand, " <span class='ZmAsstField' id='", this._commandTitleId, "'></span></td></tr>");	
112 	html.append("<tr><td colspan=3><div>");
113 	html.append("<textarea rows=", AjxEnv.isIE ? "3" : "2", " style='width:100%' id='",this._commandId,"'>");
114 	html.append("</textarea>");
115 	html.append("</div></td></tr>");
116 	html.append("<tr><td colspan=3><div class=horizSep></div></td></tr>");
117 	html.append("<tr><td colspan=3><div id='", this._contentId, "'></div></td></tr>");
118 	html.append("</table>");	
119 	return html.toString();
120 };

144 ZmAssistantDialog.prototype._initContent =
145 function() {
146 	this._commandEl = document.getElementById(this._commandId);
147 	Dwt.associateElementWithObject(this._commandEl, this);
148 	this._commandEl.onkeyup = ZmAssistantDialog._keyUpHdlr;
149 };
150 
151 ZmAssistantDialog._keyUpHdlr =
152 function(ev) {
153 	var keyEv = DwtShell.keyEvent;
154 	keyEv.setFromDhtmlEvent(ev, true);
155 	var obj = keyEv.dwtObj;
156 	obj._commandUpdated();
157 };
158 
159 ZmAssistantDialog.prototype._commandUpdated =
160 function() {
161 	// reset timer on key activity
162 	if (this._parseActionId != -1) 	AjxTimedAction.cancelAction(this._parseActionId);
163 	this._parseActionId = AjxTimedAction.scheduleAction(this._parseTimedAction, this._parseInterval);
164 }

166 ZmAssistantDialog.__RE_parseActionLeadingWord = new RegExp(["^((?:\\.|",ZmAssistant.NONSPACE,")+)",ZmAssistant.SPACES].join(""));
167 
168 ZmAssistantDialog.prototype._parseAction =
169 function() {
170 	var assistant = null;	
171 	var cmd = ZmAssistant.trimLeading(this._commandEl.value);
172 	var match = cmd.match(ZmAssistantDialog.__RE_parseActionLeadingWord);
173 	if (match) {
174 		var args = cmd.substring(match[0].length);
175 		var mainCommand = match[1];
176 		var commands = ZmAssistant.matchWord(mainCommand);
177 		switch(commands.length) {
178 			case 0:
179 				this._availableCommands = ZmMsg.ASST_availableCommands+ " " + ZmMsg.ASST_no_match;
180 				break;
181 			case 1:
182 				assistant = ZmAssistant.getHandler(commands[0]);
183 				break;
184 			default:
185 				this._availableCommands = ZmMsg.ASST_availableCommands+ " " + commands.join(", ");
186 				break;
187 		}
188 		if (assistant && mainCommand == cmd && this._assistant != assistant) {
189 			this._commandEl.value = assistant.getCommand() + " ";
190 			var len = this._commandEl.value.length;
191 			Dwt.setSelectionRange(this._commandEl, len, len);
192 		}
193 	} else {
194 		this._availableCommands = ZmMsg.ASST_availableCommands+ " " + ZmAssistant.getHandlerCommands().join(", ");
195 	}
196 
197 	if (this._assistant != assistant) {
198 		if (this._assistant != null) this._assistant.finish(this);
199 		this._assistant = assistant;
200 		if (this._assistant) {
201 			// reset help/extra
202 			this._setHelpButton(null, true, true);		
203 			this._setExtraButton(ZmMsg.moreDetails, false, false, true, null);			
204 			this._assistant.initialize(this);
205 			var title = this._assistant.getTitle();
206 			if (title) this._setCommandTitle(title);
207 			var help = this._assistant.getHelp();
208 			this._setHelpButton(null, help != null, true);
209 		}
210 	}
211 
212 	if (this._assistant) this._assistant.handle(this, null, args);
213 	else this._setDefault();
214 };

216 ZmAssistantDialog.prototype._setDefault = 
217 function() {
218 	this.setAssistantContent(this._availableCommands);
219 	this._setOkButton(AjxMsg.ok, false, false, true, null);
220 	this._setHelpButton(null, true, true);		
221 	this._setExtraButton(ZmMsg.moreDetails, false, false, true, null);	
222 	this._setCommandTitle("");
223 };	
224 
225 ZmAssistantDialog.prototype._setCommandTitle =
226 function(title, dontHtmlEncode) {
227 	var titleEl = document.getElementById(this._commandTitleId);
228 	if (titleEl) titleEl.innerHTML = dontHtmlEncode ? title : AjxStringUtil.htmlEncode(title);
229 };

231 ZmAssistantDialog.prototype._setOkButton =
232 function(title, visible, enabled) {
233 	var ok = this.getButton(DwtDialog.OK_BUTTON);
234 	if (title) ok.setText(title);
235 	ok.setEnabled(enabled);
236 	ok.setVisible(visible);
237 };
238 
239 ZmAssistantDialog.prototype._setExtraButton =
240 function(title, visible, enabled) {
241 	var ok = this.getButton(ZmAssistantDialog.EXTRA_BUTTON);
242 	if (title) ok.setText(title);
243 	ok.setEnabled(enabled);
244 	ok.setVisible(visible);
245 };
246 
247 ZmAssistantDialog.prototype._setHelpButton =
248 function(title, visible, enabled) {
249 	var b = this.getButton(ZmAssistantDialog.HELP_BUTTON);
250 	if (title) b.setText(title);
251 	b.setEnabled(enabled);
252 	b.setVisible(visible);
253 };
254 
255 ZmAssistantDialog.prototype.popdown =
256 function() {
257 	DwtDialog.prototype.popdown.call(this);
258 	if (this._assistant != null) this._assistant.finish(this);
259 	this._assistant = null;
260 };
261 
262 ZmAssistantDialog.prototype._okButtonListener =
263 function(ev) {
264 	if (this._assistant && !this._assistant.okHandler(this)) return;
265 	this.popdown();
266 };
267 
268 ZmAssistantDialog.prototype._extraButtonListener =
269 function(ev) {
270 	if (this._assistant && !this._assistant.extraButtonHandler(this)) return;
271 	this.popdown();
272 };
273 
274 ZmAssistantDialog.prototype._helpButtonListener =
275 function(ev) {
276 	if (this._assistant) {
277 		var help = this._assistant.getHelp();
278 		if (help != null) this.messageDialog(help, DwtMessageDialog.INFO_STYLE);
279 	} else {
280 		var html = new AjxBuffer();
281 		html.append(ZmMsg.ASST_HELP);
282 		html.append("<table cellspacing=1 cellpadding=2 border=0>");
283 		var cmds = ZmAssistant.getHandlerCommands();
284 		for (var i=0; i < cmds.length; i++) {
285 			var handler = ZmAssistant.getHandler(cmds[i]);
286 			html.append("<tr><td><b>", AjxStringUtil.htmlEncode(handler.getCommand()), "</b></td><td> </td><td>", AjxStringUtil.htmlEncode(handler.getCommandSummary()),"</td></tr>");
287 		}
288 		html.append("</table>");
289 		this.messageDialog(html.toString(), DwtMessageDialog.INFO_STYLE);
290 	}
291 };

293 
294 ZmAssistantDialog.prototype._handleResponseOkButtonListener =
295 function() {
296 	this.popdown();
297 };

299 /**
300  * Displays a message dialog
301  * 
302  * @param	{String}		message		the message
303  * @param	{constant}		style		the message style
304  * 
305  * @private
306  */
307 ZmAssistantDialog.prototype.messageDialog =
308 function(message, style) {
309 	var msgDialog = appCtxt.getMsgDialog();
310 	msgDialog.reset();
311 	msgDialog.setMessage(message, style);
312 	msgDialog.popup();
313 };
