 16 /**
 17  * Creates a password field.
 18  * @constructor
 19  * @class
 20  * 
 21  * @param	{hash}		params		a hash of parameters
 22  * @param {DwtComposite}      params.parent			the parent widget
 23  * @param {string}      params.initialValue		the initial value of the field
 24  * @param {number}      params.size				size of the input field (in characters)
 25  * @param {number}      params.rows				the number of rows (more than 1 means textarea)
 26  * @param {boolean}      params.forceMultiRow		if <code>true</code>, forces use of textarea even if rows == 1
 27  * @param {number}      params.maxLen			the maximum length (in characters) of the input
 28  * @param {constant}      params.errorIconStyle		the error icon style
 29  * @param {constant}      params.validationStyle	the validation type
 30  * @param  {function}     params.validator			the custom validation function
 31  * @param {Object}      params.validatorCtxtObj		the object context for validation function
 32  * @param {string}      params.className			the CSS class
 33  * @param {constant}      params.posStyle			the positioning style (see {@link DwtControl})
 34  * @param {boolean}      params.required          if <code>true</code>, mark as required.
 35  * @param {string}      params.hint				a hint to display in the input field when the value is empty.
 36  * @param {string}      params.id				an explicit ID to use for the control's DIV element
 37  * @param {string}      params.inputId			an explicit ID to use for the control's INPUT element
 38  * 
 39  * @extends		DwtInputField
 40  */
 41 DwtPasswordField = function(params) {
 42 	if (arguments.length == 0) return;
 43 
 44 	params = params || { parent: DwtShell.getShell(window) };
 45 	params.type = DwtInputField.PASSWORD; 
 46 	DwtInputField.call(this, params);
 47 
 48 	this._tabGroup = new DwtTabGroup(this._htmlElId);
 49 
 50 	// TODO: templatize DwtInputField -- then we don't need to explicitly call _createHtml
 51 	this._createHtml();
 52 };
 53 DwtPasswordField.prototype = new DwtInputField;
 54 DwtPasswordField.prototype.constructor = DwtPasswordField;
 55 
 56 //
 57 // Data
 58 //
 59 
 60 DwtPasswordField.prototype.TEMPLATE = "dwt.Widgets#DwtPasswordField";
 61 
 62 //
 63 // Public methods
 64 //
 65 
 66 DwtPasswordField.prototype.getTabGroupMember = function() {
 67 	return this._tabGroup;
 68 };
 69 
 70 /**
 71  * Shows the password.
 72  * 
 73  * @param	{boolean}	show		if <code>true</code>, show the password
 74  */
 75 DwtPasswordField.prototype.setShowPassword = function(show) {
 76 	this._showCheckbox.setSelected(show);
 77 	this.setInputType(show ? DwtInputField.STRING : DwtInputField.PASSWORD);
 78 };
 79

 81 // Protected methods

 83 
 84 DwtPasswordField.prototype._createHtml = function(templateId) {
 85 	var data = { id: this._htmlElId };
 86 	this._createHtmlFromTemplate(templateId || this.TEMPLATE, data);
 87 };
 88 
 89 DwtPasswordField.prototype._createHtmlFromTemplate =
 90 function(templateId, data) {
 91 	this._tabGroup.removeAllMembers();
 92 
 93 	// save old contents
 94 	var fragment = document.createDocumentFragment();
 95 	var child = this.getHtmlElement().firstChild;
 96 	while (child) {
 97 		var sibling = child.nextSibling;
 98 		fragment.appendChild(child);
 99 		child = sibling;
100 	};
101 
102 	// create HTML and append content
103 	DwtInputField.prototype._createHtmlFromTemplate.apply(this, arguments);
104 	var inputEl = document.getElementById(data.id+"_input");
105 	inputEl.appendChild(fragment);
106 	this._tabGroup.addMember(this.getInputElement());
107 
108 	var showCheckboxEl = document.getElementById(data.id+"_show_password");
109 	if (showCheckboxEl) {
110 		this._showCheckbox = new DwtCheckbox({parent:this});
111 		this._showCheckbox.setText(AjxMsg.showPassword);
112 		this._showCheckbox.addSelectionListener(new AjxListener(this, this._handleShowCheckbox));
113 		this._showCheckbox.replaceElement(showCheckboxEl);
114 		this._tabGroup.addMember(this._showCheckbox);
115 	}
116 };
117 
118 DwtPasswordField.prototype._handleShowCheckbox = function(event) {
119 	this.setShowPassword(event.detail);
120 };
