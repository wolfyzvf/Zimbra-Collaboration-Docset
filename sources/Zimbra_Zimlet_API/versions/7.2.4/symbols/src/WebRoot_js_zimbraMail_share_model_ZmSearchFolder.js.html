 21 /**
 22  * Creates the search folder.
 23  * @class
 24  * This class represents a search folder.
 25  * 
 26  * @param	{Hash}	params		a hash of parameters
 27  * 
 28  * @extends	ZmFolder
 29  */
 30 ZmSearchFolder = function(params) {
 31 	params.type = ZmOrganizer.SEARCH;
 32 	ZmFolder.call(this, params);
 33 	
 34 	if (params.query) {
 35 		var searchParams = {
 36 			query:params.query,
 37 			types:params.types,
 38 			sortBy:params.sortBy,
 39 			searchId:params.id,
 40 			accountName:(params.account && params.account.name)
 41 		};
 42 		this.search = new ZmSearch(searchParams);
 43 	}
 44 };
 45 
 46 ZmSearchFolder.ID_ROOT = ZmOrganizer.ID_ROOT;
 47 
 48 /**
 49  * Creates a search folder.
 50  * 
 51  * @param	{Hash}	params		a hash of parameters
 52  */
 53 ZmSearchFolder.create =
 54 function(params) {
 55 	var soapDoc = AjxSoapDoc.create("CreateSearchFolderRequest", "urn:zimbraMail");
 56 	var searchNode = soapDoc.set("search");
 57 	searchNode.setAttribute("name", params.name);
 58 	searchNode.setAttribute("query", params.search.query);
 59 	if (params.search.types) {
 60 		var a = params.search.types.getArray();
 61 		if (a.length) {
 62 			var typeStr = [];
 63 			for (var i = 0; i < a.length; i++) {
 64 				typeStr.push(ZmSearch.TYPE[a[i]]);
 65 			}
 66 			searchNode.setAttribute("types", typeStr.join(","));
 67 		}
 68 	}
 69 	if (params.search.sortBy) {
 70 		searchNode.setAttribute("sortBy", params.search.sortBy);
 71 	}
 72 
 73 	var accountName;
 74 	if (params.isGlobal) {
 75 		searchNode.setAttribute("f", "g");
 76 		accountName = appCtxt.accountList.mainAccount.name;
 77 	}
 78 
 79 	searchNode.setAttribute("l", params.parent.id);
 80 	appCtxt.getAppController().sendRequest({
 81 		soapDoc: soapDoc,
 82 		asyncMode: true,
 83 		accountName: accountName,
 84 		errorCallback: (new AjxCallback(null, ZmOrganizer._handleErrorCreate, params))
 85 	});
 86 };
 87 
 88 ZmSearchFolder.prototype = new ZmFolder;
 89 ZmSearchFolder.prototype.constructor = ZmSearchFolder;

 99 };

101 /**
102  * Gets the icon.
103  * 
104  * @return	{String}	the icon
105  */
106 ZmSearchFolder.prototype.getIcon = 
107 function() {
108 	return (this.nId == ZmOrganizer.ID_ROOT)
109 		? null
110 		: (this.isOfflineGlobalSearch ? "GlobalSearchFolder" : "SearchFolder");
111 };

119 /**
120  * Returns the organizer with the given ID. Looks in this organizer's tree first.
121  * Since a search folder may have either a regular folder or another search folder
122  * as its parent, we may need to get the parent folder from another type of tree.
123  *
124  * @param {int}	parentId	the ID of the organizer to find
125  * 
126  * @private
127  */
128 ZmSearchFolder.prototype._getNewParent =
129 function(parentId) {
130 	var parent = appCtxt.getById(parentId);
131 	if (parent) {
132 		return parent;
133 	}
134 	
135 	return appCtxt.getById(parentId);
136 };
