 16 /**
 17  * Creates a grid size picker.
 18  * @class
 19  * This class represents a widget that provides the ability to select number of rows/cols similar to
 20  * Word's "insert table" popup menu.  Mainly designed for the "insert table"
 21  * feature in {@link ZmHtmlEditor}, this widget will usually be embedded in a DwtMenu
 22  * with style {@link DwtMenu.GENERIC_WIDGET_STYLE}.
 23  *
 24  * @author Mihai Bazon <mihai@zimbra.com>
 25  * 
 26  * @param	{DwtComposite}	parent 	the parent widget
 27  * @param	{string}	title		the title
 28  * 
 29  * @extends		DwtControl
 30  */
 31 DwtGridSizePicker = function(parent, title) {
 32 	if (arguments.length == 0)
 33 		return;
 34 	DwtControl.call(this, {parent:parent, className:"DwtGridSizePicker"});
 35 	this._title = title;
 36 	if (parent instanceof DwtMenu)
 37 		parent.addPopdownListener(new AjxListener(this, this._reset));
 38 	this._reset();
 39 // 	if (AjxEnv.isIE && parent && parent instanceof DwtMenu) {
 40 // 		var table = document.createElement("table");
 41 // 		table.cellSpacing = table.cellPadding = table.border = 0;
 42 // 		var tbody = document.createElement("tbody");
 43 // 		table.appendChild(tbody);
 44 // 		var tr = document.createElement("tr");
 45 // 		tbody.appendChild(tr);
 46 // 		var td = document.createElement("td");
 47 // 		tr.appendChild(td);
 48 // 		var div = this.getHtmlElement();
 49 // 		var parent = div.parentNode;
 50 // 		parent.insertBefore(table, div);
 51 // 		td.appendChild(div);
 52 // 	}
 53 };
 54 DwtGridSizePicker.prototype = new DwtControl;
 55 DwtGridSizePicker.prototype.constructor = DwtGridSizePicker;

 57 /**
 58  * Adds a selection listener.
 59  * 
 60  * @param	{AjxListener}	listener		the listener
 61  */
 62 DwtGridSizePicker.prototype.addSelectionListener =
 63 function(listener) {
 64 	this.addListener(DwtEvent.SELECTION, listener);
 65 };

 67 /**
 68  * Removes a selection listener.
 69  * 
 70  * @param	{AjxListener}	listener		the listener
 71  */
 72 DwtGridSizePicker.prototype.removeSelectionListener =
 73 function(listener) {
 74 	this.removeListener(DwtEvent.SELECTION, listener);
 75 };

 77 DwtGridSizePicker.prototype._createGrid = function() {
 78 	var el = this.getHtmlElement();
 79 	var html = [ "<table cellpadding='0' border='0' cellspacing='0'>",
 80 		     "<tr class='info'><td colspan='", this.maxCols, "'>", this._title, "</td></tr>" ];
 81 	var row = [ "<tr class='grid'>" ];
 82 	for (var i = 0; i < this.maxCols; ++i) {
 83 		// row.push("<td unselectable='unselectable'> </td>");
 84 		row.push("<td unselectable='unselectable'></td>");
 85 	}
 86 	row.push("</tr>");
 87 	row = row.join("");
 88 	for (var i = 0; i < this.maxRows; ++i)
 89 		html.push(row);
 90 	html.push("</table>");
 91 	html = html.join("");
 92 	el.innerHTML = html;
 93 	el.onmouseover = DwtGridSizePicker._onMouseOver;
 94 	el.onmouseup = DwtGridSizePicker._onSelect;
 95 	// el.firstChild.onmouseout = AjxCallback.simpleClosure(this._onMouseOut, this);
 96 };
 97 
 98 DwtGridSizePicker.prototype._reset = function() {
 99 	this._endRow = -1;
100 	this._endCol = -1;
101 	this.maxRows = 7;
102 	this.maxCols = 7;
103 	this._createGrid();
104 };
105 
106 DwtGridSizePicker._onSelect = function(ev) {
107 	if (AjxEnv.isIE)
108 		ev = window.event;
109 	DwtControl.getTargetControl(ev)._onSelect(ev);
110 };
111 
112 DwtGridSizePicker.prototype._onSelect = function(ev) {
113 	if (AjxEnv.isIE)
114 		ev = window.event;
115 	var dwtev = new DwtUiEvent();
116 	dwtev.setFromDhtmlEvent(ev);
117 	var target = dwtev.target;
118 	if (target.tagName.toLowerCase() == "td") {
119 		var row = target.parentNode.rowIndex;
120 		if (row > 0) {
121 			var col = target.cellIndex;
122 			var selev = DwtShell.selectionEvent;
123 			selev.item = this;
124 			selev.detail = { rows: row, cols: col + 1 };
125 			if (this.parent instanceof DwtMenu)
126 				DwtMenu.closeActiveMenu();
127 			this.notifyListeners(DwtEvent.SELECTION, selev);
128 		}
129 	}
130 };
131 
132 DwtGridSizePicker.prototype._onMouseOut = function(ev) {
133 	if (AjxEnv.isIE)
134 		ev = window.event;
135 	var dwtev = new DwtUiEvent();
136 	dwtev.setFromDhtmlEvent(ev);
137 	var tgt = dwtev.target;
138 	var table = this.getHtmlElement().firstChild;
139 	try {
140 		while (tgt && tgt !== table)
141 			tgt = tgt.parentNode;
142 	} catch(ex) {
143 		tgt = null;
144 	};
145 	if (!tgt) {
146 		this._endRow = this._endCol = -1;
147 		this._update();
148 	}
149 };
150 
151 DwtGridSizePicker._onMouseOver = function(ev) {
152 	if (AjxEnv.isIE)
153 		ev = window.event;
154 	DwtControl.getTargetControl(ev)._onMouseOver(ev);
155 };
156 
157 DwtGridSizePicker.prototype._onMouseOver = function(ev) {
158 	var dwtev = new DwtUiEvent(true);
159 	dwtev.setFromDhtmlEvent(ev);
160 	var target = dwtev.target;
161 	if (this._updateTimeout)
162 		clearTimeout(this._updateTimeout);
163 	this._updateTimeout = setTimeout(
164 		AjxCallback.simpleClosure(
165 			this._updateOnTimeout, this, target),
166 		10);
167 	dwtev.setToDhtmlEvent(ev);
168 };
169 
170 DwtGridSizePicker.prototype._addRow = function() {
171 	var table = this.getHtmlElement().firstChild;
172 	var tr = table.insertRow(-1);
173 	tr.className = "grid";
174 	for (var i = 0; i < this.maxCols; ++i) {
175 		var td = tr.insertCell(-1);
176 		td.unselectable = true;
177 		// td.innerHTML = " ";
178 	}
179 	++this.maxRows;
180 };
181 
182 DwtGridSizePicker.prototype._addCol = function() {
183 	var table = this.getHtmlElement().firstChild;
184 	var rows = table.rows;
185 	for (var i = 1; i <= this.maxRows; ++i) {
186 		var tr = rows[i];
187 		var td = tr.insertCell(-1);
188 		td.unselectable = true;
189 		// td.innerHTML = " ";
190 	}
191 	++table.rows[0].cells[0].colSpan;
192 	++this.maxCols;
193 };
194 
195 DwtGridSizePicker.prototype._updateOnTimeout = function(target) {
196 	var clear = true, row, col;
197 	if (target.tagName.toLowerCase() == "td") {
198 		row = target.parentNode.rowIndex;
199 		if (row > 0) {
200 			col = this._endCol = target.cellIndex;
201 			this._endRow = target.parentNode.rowIndex - 1;
202 			clear = false;
203 		}
204 	}
205 	if (clear)
206 		this._endRow = this._endCol = -1;
207 	this._update();
208 	if (!clear) {
209 		if (row == this.maxRows)
210 			this._addRow();
211 		if (col == this.maxCols - 1)
212 			this._addCol();
213 	}
214 };
215 
216 DwtGridSizePicker.prototype._update = function() {
217 	var endRow = this._endRow;
218 	var endCol = this._endCol;
219 	var table = this.getHtmlElement().firstChild;
220 	var header = table.rows[0].cells[0];
221 	this._rows = endRow + 1;
222 	this._cols = endCol + 1;
223 	if (this._rows > 0 && this._cols > 0)
224 		header.innerHTML = this._rows + " x " + this._cols;
225 	else
226 		header.innerHTML = this._title;
227 	for (var i = 0; i < this.maxRows; ++i) {
228 		var row = table.rows[i+1];
229 		for (var j = 0; j < this.maxCols; ++j) {
230 			var cell = row.cells[j];
231 			var clear = (i > this._endRow) || (j > this._endCol);
232 			if (clear)
233 				Dwt.delClass(cell, "Hovered");
234 			else
235 				Dwt.addClass(cell, "Hovered");
236 		}
237 	}
238 };
