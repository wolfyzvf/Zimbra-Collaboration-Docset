111 /**
112  * Imports user data as specified in the <code>params</code> object.
113  * 
114  * @param {Hash}	params			a hash of parameters
115  * @param {Element}	params.form		the form containing file input field
116  * @param {String}	params.folderId	the folder id for import. If not specified, assumes import to root folder.
117  * @param {String}	params.type		the type (defaults to {@link TYPE_TGZ})
118  * @param {String}	params.subType	the sub-type (defaults to <code>SUBTYPE_DEFAULT[type]</code>)
119  * @param {String}	params.resolve	resolve duplicates: "" (ignore), "modify", "replace", "reset" (defaults to ignore).
120  * @param {String}	params.views	a comma-separated list of views
121  * @param {AjxCallback}	callback	the callback for success
122  * @param {AjxCallback}	errorCallback	the callback for errors
123  *        
124  * @return	{Boolean}	<code>true</code> if the import is successful
125  */
126 ZmImportExportController.prototype.importData =
127 function(params) {
128 	// error checking
129 	params = params || {};
130 	var folderId = params.folderId || -1;
131 	if (folderId == -1) {
132 		var params = {
133 			msg:	ZmMsg.importErrorMissingFolder,
134 			level:	ZmStatusView.LEVEL_CRITICAL
135 		};
136 		appCtxt.setStatusMsg(params);
137 		return false;
138 	}
139 
140 	params.filename = params.form && params.form.elements["file"].value;
141 	if (!params.filename) {
142 		var params = {
143 			msg:	ZmMsg.importErrorMissingFile,
144 			level:	ZmStatusView.LEVEL_CRITICAL
145 		};
146 		appCtxt.setStatusMsg(params);
147 		return false;
148 	}
149 
150 	params.ext = params.filename.replace(/^.*\./,"").toLowerCase();
151 	if (!ZmImportExportController.EXTS_TYPE[params.ext]) {
152 		var params = {
153 			msg:	AjxMessageFormat.format(ZmMsg.importErrorTypeNotSupported, params.ext),
154 			level:	ZmStatusView.LEVEL_CRITICAL
155 		};
156 		appCtxt.setStatusMsg(params);
157 		return false;
158 	}
159 	params.defaultType = params.type || ZmImportExportController.EXTS_TYPE[params.ext] || ZmImportExportController.TYPE_DEFAULT;
160 	var isZimbra = ZmImportExportController.EXTS_TYPE[params.defaultType] == ZmImportExportController.TYPE_TGZ;
161 	var folder = appCtxt.getById(folderId);
162 	if (!isZimbra && folder && folder.nId == ZmOrganizer.ID_ROOT) {
163 		var params = {
164 			msg:	ZmMsg.importErrorRootNotAllowed,
165 			level:	ZmStatusView.LEVEL_CRITICAL
166 		};
167 		appCtxt.setStatusMsg(params);
168 		return false;
169 	}
170 
171 	if (params.resolve == "reset") {
172 		var dialog = appCtxt.getOkCancelMsgDialog();
173 		dialog.registerCallback(DwtDialog.OK_BUTTON, this._confirmImportReset, this, [params]);
174 		dialog.registerCallback(DwtDialog.CANCEL_BUTTON, this._cancelImportReset, this);
175 		var msg = ZmMsg.importResetWarning;
176 		var style = DwtMessageDialog.WARNING_STYLE;
177 		dialog.setMessage(msg, style);
178 		dialog.popup();
179 		return false;
180 	}
181 
182 	// import
183 	return this._doImportData(params);
184 };

186 /**
187  * Exports user data as specified in the <code>params</code> object.
188  *
189  * @param {Hash}	params		a hash of parameters
190  * @param {String}	params.folderId		the folder id for export. If not specified, all folders want to be exported.
191  * @param {String}	params.type			the type (defaults to {@link TYPE_TGZ})
192  * @param {String}	params.subType		the sub-type (defaults to <code>SUBTYPE_DEFAULT[type]</code>)
193  * @param {String}	params.views		a comma-separated list of views
194  * @param {String}	params.filename		the filename for exported file
195  * @param {String}	params.searchFilter	the search filter
196  * @param {Boolean} params.skipMeta		if <code>true</code>, skip export of meta-data
197  * @param {AjxCallback}	callback	the callback for success
198  * @param {AjxCallback}	errorCallback	the callback for errors
199  *        
200  * @return	{Boolean}	<code>true</code> if the export is successful
201  */
202 ZmImportExportController.prototype.exportData = function(params) {
203 	// error checking
204 	params = params || {};
205 	var folderId = params.folderId || -1;
206 	if (folderId == -1) {
207 		var params = {
208 			msg:	ZmMsg.exportErrorMissingFolder,
209 			level:	ZmStatusView.LEVEL_CRITICAL
210 		};
211 		appCtxt.setStatusMsg(params);
212 		return false;
213 	}
214 
215 	var type = params.type = params.type || ZmImportExportController.TYPE_DEFAULT;
216 	var isZimbra = ZmImportExportController.EXTS_TYPE[type] == ZmImportExportController.TYPE_TGZ;
217 	var folder = appCtxt.getById(folderId);
218 	if (!isZimbra && folder && folder.nId == ZmOrganizer.ID_ROOT) {
219 		var params = {
220 			msg:	ZmMsg.exportErrorRootNotAllowed,
221 			level:	ZmStatusView.LEVEL_CRITICAL
222 		};
223 		appCtxt.setStatusMsg(params);
224 		return false;
225 	}
226 
227 	// export
228 	return this._doExportData(params);
229 };

235 /**
236  * @private
237  */
238 ZmImportExportController.prototype._doImportData =
239 function(params) {
240 	if (params.folderId == -1 && params.defaultType != ZmImportExportController.TYPE_TGZ) {
241 		return this._doImportSelectFolder(params);
242 	}
243 	return this._doImport(params);
244 };

250 function(params) {
251 	var dialog = appCtxt.getChooseFolderDialog();
252 	dialog.reset();

263 		dialog.popup({
264 			treeIds:		[ZmOrganizer.ADDRBOOK],
265 			title:			ZmMsg.chooseAddrBook,

268 			skipReadOnly:	true,
269 			hideNewButton:	noNew,
270 			omit:			omit,

297 /**
298  * @private
299  */
300 ZmImportExportController.prototype._doImport =
301 function(params) {
302 	// create custom callback function for this import request
303 	var funcName = "ZmImportExportController__callback__"+Dwt.getNextId("import");
304 	window[funcName] = AjxCallback.simpleClosure(this._handleImportResponse, this, funcName, params);
305 
306 	// generate request url
307 	var folder = params.folderId && appCtxt.getById(params.folderId);
308 	if (folder && folder.nId == ZmOrganizer.ID_ROOT) folder = null;
309 	var path = folder ? folder.getPath(null, null, null, true, true) : "";
310 	var type = params.type || params.ext;
311 	var url = [
312 		"/home/",
313 		encodeURIComponent(appCtxt.get(ZmSetting.USERNAME)),
314 		"/",
315 		encodeURIComponent(path),
316 		"?",
317 		type ? "fmt="+encodeURIComponent(type) : "",
318 		params.views ? "&types="+encodeURIComponent(params.views) : "",
319 		params.resolve ? "&resolve="+encodeURIComponent(params.resolve) : "",
320 		"&callback="+funcName,
321 		"&charset="+appCtxt.getCharset()
322 	].join("");
323 
324 	// initialize form
325 	var form = params.form;
326 	form.action = url;
327 	form.method = "POST";
328 	form.enctype = "multipart/form-data";
329 
330 	// destination iframe
331 	var onload = null;
332 	var onerror = AjxCallback.simpleClosure(this._importError, this, params.errorCallback);
333 	params.iframe = ZmImportExportController.__createIframe(form, onload, onerror);
334 
335 	// import
336 	form.submit();
337 	return true;
338 };

340 /**
341  * @private
342  */
343 ZmImportExportController.prototype._handleImportResponse =
344 function(funcName, params, type, fault1 /* , ... , faultN */) {
345 	// gather error/warning messages
346 	var messages = [];
347 	if (fault1) {
348 		for (var j = 3; j < arguments.length; j++) {
349 			var fault = arguments[j];
350 			var code = fault.Detail.Error.Code;
351 			var message = fault.Reason.Text;
352 			var args = ZmImportExportController.__faultArgs(fault.Detail.Error.a);
353 			if (code == "formatter.UNKNOWN_ERROR") {
354 				args.path = args.path || ["(",ZmMsg.unknown,")"].join("");
355 				args.message = message;
356 			}
357 			var mappings = ZmImportExportController.__FAULT_ARGS_MAPPING[code];
358 			var formatArgs = new Array(mappings ? mappings.length : 0);
359 			for (var i = 0; i < formatArgs.length; i++) {
360 				formatArgs[i] = args[mappings[i]];
361 			}
362 			messages.push(ZMsg[code] ? AjxMessageFormat.format(ZMsg[code], formatArgs) : message);
363 		}
364 	}
365 	// show success or failure
366 	if (type == "fail") {
367 		this._importError(params.errorCallback, messages[0]);
368 	}
369 	else if (type == "warn") {
370 		this._importWarnings(params.callback, messages);
371 	}
372 	else {
373 		this._importSuccess(params.callback);
374 		appCtxt.getAppController().sendNoOp(); //send no-op to refresh
375 	}
376 
377 	// cleanup
378 	try {
379 		delete window[funcName]; // IE fails on this one (bug #57952)
380 	} catch (e) {
381 		if (window[funcName]) {
382 			window[funcName] = undefined;
383 		}
384 	}
385 	var iframe = params.iframe;
386 	setTimeout(function() { // Right now we are actually in the iframe's onload handler, so we defer killing the iframe until we're out of it
387 		iframe.parentNode.removeChild(iframe);
388 	}, 0);
389 };

391 /**

394 ZmImportExportController.__faultArgs =
395 function(array) {
396 	var args = {};
397 	for (var i = 0; array && i < array.length; i++) {
398 		args[array[i].n] = array[i]._content;
399 	}
400 	return args;
401 };

422 /**
423  * @private
424  */
425 ZmImportExportController.prototype._doExportData =
426 function(params) {
427 	var type = params.type;
428 	var isTGZ = type == ZmImportExportController.TYPE_TGZ;
429 	var isCSV = type == ZmImportExportController.TYPE_CSV;
430 	var subType = params.subType || ZmImportExportController.SUBTYPE_DEFAULT[type];
431 
432 	var folder = params.folderId && appCtxt.getById(params.folderId);
433 	if (folder && folder.nId == ZmOrganizer.ID_ROOT) folder = null;
434 	var path = folder ? folder.getPath(null, null, null, true, true) : "";
435 
436 	// generate request URL
437 	var url = [
438 		"/home/",
439 		encodeURIComponent(appCtxt.get(ZmSetting.USERNAME)),
440 		"/",
441 		encodeURIComponent(path)
442 	].join("");
443 
444 	var formParams = { "fmt" : type };
445 	if (isCSV) { formParams[type+"fmt"] = subType; }
446 	var startDate = params.start ? AjxDateUtil.simpleParseDateStr(params.start) : null;
447 	var endDate = params.end ? AjxDateUtil.simpleParseDateStr(params.end) : null;
448 	if (isTGZ && params.views) { formParams["types"] = params.views; }
449     if(startDate) {
450         formParams["start"] = startDate.getTime();
451     }
452     if(endDate) {
453         endDate = AjxDateUtil.roll(endDate, AjxDateUtil.DAY, 1);
454         formParams["end"] = endDate.getTime();
455     }
456 	if (isTGZ && params.searchFilter) { formParams["query"] = params.searchFilter; }
457 	if (params.skipMeta) { formParams["meta"] = "0"; }
458 	if (params.filename) { formParams["filename"] = params.filename; }
459 	formParams.emptyname = ZmMsg.exportEmptyName;
460 	formParams["charset"] = appCtxt.getCharset();
461 
462 	// initialize form
463 	var form = ZmImportExportController.__createForm(url, formParams);
464 
465 	// destination form
466 	var onload = AjxCallback.simpleClosure(this._exportSuccess, this, params.callback);
467 	var onerror = AjxCallback.simpleClosure(this._exportError, this, params.errorCallback);
468 	var iframe = ZmImportExportController.__createIframe(form, onload, onerror);
469 
470 	// export
471 	form.submit();
472 };

486 /**
487  * @private
488  */
489 ZmImportExportController.prototype._importWarnings =
490 function(callback, messages) {
491 	if (callback) {
492 		callback.run(false);
493 	}
494 	// remove duplicates
495 	var msgmap = {};
496 	for (var i = 0; i < messages.length; i++) {
497 		msgmap[messages[i]] = true;
498 	}
499 	messages = AjxUtil.map(AjxUtil.keys(msgmap), AjxStringUtil.htmlEncode);
500 	if (messages.length > 5) {
501 		var count = messages.length - 5;
502 		messages.splice(5, count, AjxMessageFormat.format(ZmMsg.importAdditionalWarnings, [count]));
503 	}
504 	// show warnings
505 	var msglist = [];
506 	for (var i = 0; i < messages.length; i++) {
507 		msglist.push("<li>", messages[i]);
508 	}
509 	var msg = AjxMessageFormat.format(ZmMsg.importSuccessWithWarnings, [ messages.length, msglist.join("") ]);
510 	ZmImportExportController.__showMessage(msg, DwtMessageDialog.WARNING_STYLE);
511 	return true;
512 };

565 /**
566  * @private
567  */
568 ZmImportExportController.__createForm =
569 function(action, params, method) {
570 	var form = document.createElement("FORM");
571 	form.action = action;
572 	form.method = method || "GET";
573 	for (var name in params) {
574 		var value = params[name];
575 		if (!value) continue;
576 		var input = document.createElement("INPUT");
577 		input.type = "hidden";
578 		input.name = name;
579 		input.value = value;
580 		form.appendChild(input);
581 	}
582 	form.style.display = "none";
583 	document.body.appendChild(form);
584 	return form;
585 };

587 /**
588  * @private
589  */
590 ZmImportExportController.__createIframe =
591 function(form, onload, onerror) {
592 	var id = Dwt.getNextId() + "_iframe";
593 	var iframe;
594 	if (AjxEnv.isIE) {
595 		// NOTE: This has to be done because IE doesn't recognize the name
596 		//       attribute if set programmatically. And without that, the
597 		//       form target will cause it to return in a new window which
598 		//       breaks the callback.
599 		var html = [ "<IFRAME id='",id,"' name='",id,"'>" ].join("");
600 		iframe = document.createElement(html);
601 	}
602 	else {
603 		iframe = document.createElement("IFRAME");
604 		iframe.name = iframe.id = id;
605 	}
606 	// NOTE: Event handlers won't be called when iframe hidden.
607 //	iframe.style.display = "none";
608 	iframe.style.position = "absolute";
609 	iframe.style.width = 1;
610 	iframe.style.height = 1;
611 	iframe.style.top = 10;
612 	iframe.style.left = -10;
613 	document.body.appendChild(iframe);
614 	form.target = iframe.name;
615 
616 	iframe.onload = onload;
617 	iframe.onerror = onerror;
618 
619 	return iframe;
620 };
