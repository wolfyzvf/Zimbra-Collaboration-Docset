 16 /**
 17  * Creates the filter rules view.
 18  * @class
 19  * This class represents the filters tab view in preferences application.
 20  * 
 21  * @param	{DwtComposite}	parent		the parent widget
 22  * @param	{ZmController}	controller	the controller
 23  *
 24  * @extends		DwtTabViewPage
 25  * 
 26  * @see		ZmPreferencesApp
 27  */
 28 ZmFilterRulesView = function(parent, controller) {
 29 
 30 	DwtTabViewPage.call(this, parent, "ZmPreferencesPage ZmFilterRulesView", Dwt.STATIC_STYLE);
 31 
 32 	this._controller = controller;
 33 	this._prefsController = AjxDispatcher.run("GetPrefController");
 34 
 35 	this._rules = AjxDispatcher.run(controller.isOutgoing() ? "GetOutgoingFilterRules" : "GetFilterRules");
 36 
 37 	var section = ZmPref.getPrefSectionWithPref(ZmSetting.FILTERS);
 38 	this._title = [ZmMsg.zimbraTitle, controller.getApp().getDisplayName(), section && section.title].join(": ");
 39 
 40 	this._rendered = false;
 41 
 42 	this._tabGroup = new DwtTabGroup(this._htmlElId);
 43 };
 44 
 45 ZmFilterRulesView.prototype = new DwtTabViewPage;
 46 ZmFilterRulesView.prototype.constructor = ZmFilterRulesView;
 47 
 48 ZmFilterRulesView.prototype.toString =
 49 function() {
 50 	return "ZmFilterRulesView";
 51 };
 52 
 53 ZmFilterRulesView.prototype.showMe =
 54 function() {
 55 	Dwt.setTitle(this._title);
 56 	var section = ZmPref.getPrefSectionWithPref(ZmSetting.FILTERS);
 57 
 58 	this._prefsController._resetOperations(this._prefsController._toolbar, section && section.id);
 59     //TODO: We got to optimize / avoid force-rendering logic for multi-account setup
 60 	if (this.hasRendered && !appCtxt.isOffline) { return; }
 61 
 62 	// create the html
 63 	var data = {id:this._htmlElId};
 64 	this.getHtmlElement().innerHTML = AjxTemplate.expand("prefs.Pages#MailFilter", data);
 65 
 66 	// create toolbar
 67 	var toolbarEl = Dwt.byId(data.id + "_toolbar");
 68 	if (toolbarEl) {
 69 		var buttons = this._controller.getToolbarButtons();
 70 		this._toolbar = new ZmButtonToolBar({parent:this, buttons:buttons, posStyle:Dwt.STATIC_STYLE,
 71 											 context:ZmId.VIEW_FILTER_RULES});
 72 		this._toolbar.replaceElement(toolbarEl);
 73 		this._tabGroup.addMember(this._toolbar);
 74 	}
 75 
 76 	// create list view
 77 	var listViewEl = Dwt.byId(data.id + "_list");
 78 	if (listViewEl) {
 79 		this._listView = new ZmFilterListView(this, this._controller);
 80 		this._listView.replaceElement(listViewEl);
 81 		this._tabGroup.addMember(this._listView);
 82 	}
 83 
 84 	// initialize controller
 85 	this._controller.initialize(this._toolbar, this._listView);
 86 
 87 	this.hasRendered = true;
 88 };

138 /**
139  * Creates the filter list view.
140  * @class
141  * This class represents the filter list view.
142  *
143  * @param	{DwtComposite}	parent		the parent widget
144  * @param	{ZmController}	controller	the controller
145  * 
146  * @extends		DwtListView
147  * 
148  * @private
149  */
150 ZmFilterListView = function(parent, controller) {
151 	var headerList = this._getHeaderList();
152 	DwtListView.call(this, {parent:parent, className:"ZmFilterListView", headerList:headerList,
153 							view:ZmId.VIEW_FILTER_RULES});
154 
155 	this._rules = AjxDispatcher.run(controller.isOutgoing() ? "GetOutgoingFilterRules" : "GetFilterRules");
156 
157 	this._controller = controller;
158 	this._rules.addChangeListener(new AjxListener(this, this._changeListener));
159 	this._internalId = AjxCore.assignId(this);
160 };

163 ZmFilterListView.COL_NAME	= "na";
164 ZmFilterListView.CHECKBOX_PREFIX = "_ruleCheckbox";

167 ZmFilterListView.prototype.constructor = ZmFilterListView;

170 function() {
171 	return "ZmFilterListView";

174 /**
175  * Only show rules that have at least one valid action (eg, if the only action
176  * is "tag" and tagging is disabled, don't show the rule).
177  *
178  * @param list
179  * 
180  * @private
181  */
182 ZmFilterListView.prototype.set =
183 function(list) {
184 	this._checkboxIds = [];
185 	var list1 = new AjxVector();
186 	var len = list.size();
187 	for (var i = 0; i < len; i++) {
188 		var rule = list.get(i);
189 		if (rule.hasValidAction()) {
190 			list1.add(rule);
191 		}
192 	}
193 	DwtListView.prototype.set.call(this, list1);
194 
195 	for (var i = 0, len = list1.size(); i < len; i++) {
196 		var rule = list1.get(i);
197 		var checkbox = Dwt.byId(ZmFilterListView._getCheckboxIdForRule(rule));
198 		if (checkbox) {
199 			Dwt.setHandler(checkbox, DwtEvent.ONCHANGE, AjxCallback.simpleClosure(this._activeStateChange, this));
200 		}
201 	}
202 };

204 ZmFilterListView.prototype._getHeaderList =
205 function() {
206 	return [
207 		(new DwtListHeaderItem({field:ZmFilterListView.COL_ACTIVE, text:ZmMsg.active, width:ZmMsg.COLUMN_WIDTH_ACTIVE})),
208 		(new DwtListHeaderItem({field:ZmFilterListView.COL_NAME, text:ZmMsg.filterName}))
209 	];
210 };
211 
212 ZmFilterListView._getCheckboxIdForRule =
213 function(rule) {
214 	return ZmFilterListView.CHECKBOX_PREFIX + rule.id;
215 };
216 
217 ZmFilterListView.prototype._getCellClass =
218 function(item, field, params) {
219 	if (field == ZmFilterListView.COL_ACTIVE) {
220 		return "FilterActiveCell";
221 	}
222 	return DwtListView.prototype._getCellClass.call(this, item, field, params);
223 };
224 
225 ZmFilterListView.prototype._getCellContents =
226 function(html, idx, item, field, colIdx, params) {
227 	if (field == ZmFilterListView.COL_ACTIVE) {
228 		html[idx++] = "<input type='checkbox' ";
229 		html[idx++] = item.active ? "checked " : "";
230 		html[idx++] = "id='";
231 		html[idx++] = ZmFilterListView._getCheckboxIdForRule(item);
232 		html[idx++] = "' _flvId='";
233 		html[idx++] = this._internalId;
234 		html[idx++] = "'>";
235 	} else if (field == ZmFilterListView.COL_NAME) {
236 		html[idx++] = AjxStringUtil.htmlEncode(item.name);
237 	}
238 
239 	return idx;
240 };

242 /**
243  * In general, we just re-display all the rules when anything changes, rather
244  * than trying to update a particular row.
245  *
246  * @param {DwtEvent}	ev		the event
247  * 
248  * @private
249  */
250 ZmFilterListView.prototype._changeListener =
251 function(ev) {
252 	if (ev.type != ZmEvent.S_FILTER) {
253 		AjxDebug.println(AjxDebug.FILTER, "FILTER RULES: ev.type is not S_FILTER; ev.type == " + ev.type);
254 		return;
255 	}
256 	AjxDebug.println(AjxDebug.FILTER, "FILTER RULES: ev.type == " + ev.type);
257 	if (ev.event == ZmEvent.E_MODIFY) {
258 		this._controller.resetListView(ev.getDetail("index"));
259 		AjxDebug.println(AjxDebug.FILTER, "FILTER RULES: MODIFY event, called resetListview");
260 	}
261 };

263 /**
264  * Handles click of 'active' checkbox by toggling the rule's active state.
265  *
266  * @param {DwtEvent}	ev		the event
267  * 
268  * @private
269  */
270 ZmFilterListView.prototype._activeStateChange =
271 function(ev) {
272 	var target = DwtUiEvent.getTarget(ev);
273 	var flvId = target.getAttribute("_flvId");
274 	var flv = AjxCore.objectWithId(flvId);
275 	var ruleId = target.id.substring(13);
276 	var rule = flv._rules.getRuleById(ruleId);
277 	if (rule) {
278 		var active = target.checked;
279 		if (active != rule.active) {
280 			var okCallback = new AjxCallback(flv._rules, flv._rules.setActive, [rule, active]);
281 			var cancelCallback = new AjxCallback(this, function(){target.checked = false;});
282 			if (active) {
283 				this._controller.handleBeforeFilterChange(okCallback, cancelCallback);
284 			} else {
285 				okCallback.run();
286 			}
287 		}
288 	}
289 };

291 /**
292  * Override so that we don't change selection when the 'active' checkbox is clicked.
293  * Also contains a hack for IE for handling a click of the 'active' checkbox, because
294  * the ONCHANGE handler was only getting invoked on every other checkbox click for IE.
295  *
296  * @param {Element}	clickedEl	the list DIV that received the click
297  * @param {DwtEvent}	ev			the click event
298  * @param {constant}	button		the button that was clicked
299  * 
300  * @private
301  */
302 ZmFilterListView.prototype._allowLeftSelection =
303 function(clickedEl, ev, button) {
304 	// We only care about mouse events
305 	if (!(ev instanceof DwtMouseEvent)) { return true; }
306 
307 	var target = DwtUiEvent.getTarget(ev);
308 	var isInput = (target.id.indexOf(ZmFilterListView.CHECKBOX_PREFIX) == 0);
309 	if (isInput) {
310 		this._activeStateChange(ev);
311 	}
312 
313 	return !isInput;
314 };
