137 /**
138  * Fetches a message from the server.
139  *
140  * @param {Hash}			params					a hash of parameters
141  * @param {ZmZimbraMail}	params.sender			the provides access to sendRequest()
142  * @param {int}				params.msgId			the ID of the msg to be fetched.
143  * @param {int}				params.partId 			the msg part ID (if retrieving attachment part, i.e. rfc/822)
144  * @param {int}				params.ridZ   			the RECURRENCE-ID in Z (UTC) timezone
145  * @param {Boolean}			params.getHtml			if <code>true</code>, try to fetch html from the server
146  * @param {Boolean}			params.markRead			if <code>true</code>, mark msg read
147  * @param {AjxCallback}		params.callback			the async callback
148  * @param {AjxCallback}		params.errorCallback	the async error callback
149  * @param {Boolean}			params.noBusyOverlay	if <code>true</code>, do not put up busy overlay during request
150  * @param {Boolean}			params.noTruncate		if <code>true</code>, do not truncate message body
151  * @param {ZmBatchCommand}	params.batchCmd			if set, request gets added to this batch command
152  * @param {String}			params.accountName		the name of the account to send request on behalf of
153  * @param {boolean}			params.needExp			if not <code>false</code>, have server check if addresses are DLs
154  */
155 ZmMailMsg.fetchMsg =
156 function(params) {
157 	var jsonObj = {GetMsgRequest:{_jsns:"urn:zimbraMail"}};
158 	var request = jsonObj.GetMsgRequest;
159 	var m = request.m = {};
160 	m.id = params.msgId;
161 	if (params.partId) {
162 		m.part = params.partId;
163 	}
164 	if (params.markRead) {
165 		m.read = 1;
166 	}
167 	if (params.getHtml) {
168 		m.html = 1;
169 	}
170 	if (params.needExp !== false) {
171 		m.needExp = 1;
172 	}
173 
174 	if (params.ridZ) {
175 		m.ridZ = params.ridZ;
176 	}
177 
178 	for (var hdr in ZmMailMsg.requestHeaders) {
179 		if (!m.header) { m.header = []; }
180 		m.header.push({n:hdr});
181 	}
182 
183 	if (!params.noTruncate) {
184 		m.max = appCtxt.get(ZmSetting.MAX_MESSAGE_SIZE) || ZmMailApp.DEFAULT_MAX_MESSAGE_SIZE;
185 	}
186 
187 	if (params.batchCmd) {
188 		params.batchCmd.addRequestParams(jsonObj, params.callback);
189 	} else {
190 		var newParams = {
191 			jsonObj: jsonObj,
192 			asyncMode: true,
193 			callback: (new AjxCallback(null, ZmMailMsg._handleResponseFetchMsg, [params.callback])),
194 			errorCallback: params.errorCallback,
195 			noBusyOverlay: params.noBusyOverlay,
196 			accountName: params.accountName
197 		};
198 		params.sender.sendRequest(newParams);
199 	}
200 };

202 ZmMailMsg._handleResponseFetchMsg =
203 function(callback, result) {
204 	if (callback instanceof AjxCallback) {
205 		callback.run(result);
206 	}
207 };

218 /**
219  * Gets a vector of addresses of the given type.
220  *
221  * @param {constant}	type		an email address type
222  * @param {Hash}		used		an array of addresses that have been used. If not <code>null</code>,
223  *									then this method will omit those addresses from the
224  * 									returned vector and will populate used with the additional new addresses
225  * @param {Boolean}	addAsContact	if <code>true</code>, emails should be converted to {@link ZmContact} objects
226  * 
227  * @return	{AjxVector}	a vector of email addresses
228  */
229 ZmMailMsg.prototype.getAddresses =
230 function(type, used, addAsContact) {
231 	if (!used) {
232 		return this._addrs[type];
233 	} else {
234 		var a = this._addrs[type].getArray();
235 		var addrs = [];
236 		for (var i = 0; i < a.length; i++) {
237 			var addr = a[i];
238 			var email = addr.getAddress();
239 			if (!used[email]) {
240 				var contact = addr;
241 				if (addAsContact) {
242 					var cl = AjxDispatcher.run("GetContacts");
243 					contact = cl.getContactByEmail(email);
244 					if (contact == null) {
245 						contact = new ZmContact(null);
246 						contact.initFromEmail(addr);
247 					}
248 				}
249 				addrs.push(contact);
250 			}
251 			used[email] = true;
252 		}
253 		return AjxVector.fromArray(addrs);
254 	}
255 };

257 /**
258  * Gets a Reply-To address if there is one, otherwise the From address
259  * unless this message was sent by the user, in which case, it is the To
260  * field (but only in the case of Reply All). A list is returned, since
261  * theoretically From and Reply To can have multiple addresses.
262  * 
263  * @return	{AjxVector}	an array of {@link AjxEmailAddress} objects
264  */
265 ZmMailMsg.prototype.getReplyAddresses =
266 function(mode, aliases, isDefaultIdentity) {
267 
268 	// reply-to has precedence over everything else
269 	var addrVec = this._addrs[AjxEmailAddress.REPLY_TO];
270 	if (!addrVec && this.isInvite() && this.needsRsvp()) {
271 		var invEmail = this.invite.getOrganizerEmail(0);
272 		if (invEmail) {
273 			return AjxVector.fromArray([new AjxEmailAddress(invEmail)]);
274 		}
275 	}
276 
277 	if (!(addrVec && addrVec.size())) {
278 		if (mode == ZmOperation.REPLY_CANCEL || (this.isSent && mode == ZmOperation.REPLY_ALL)) {
279 			addrVec = this.isInvite() ? this._getAttendees() : this._addrs[AjxEmailAddress.TO];
280 		} else {
281 			addrVec = this._addrs[AjxEmailAddress.FROM];
282 			if (aliases && isDefaultIdentity) {
283 				var from = addrVec.get(0);
284 				// make sure we're not replying to ourself
285 				if (from && aliases[from.address]) {
286 					addrVec = this._addrs[AjxEmailAddress.TO];
287 				}
288 			}
289 		}
290 	}
291 	return addrVec;
292 };

294 ZmMailMsg.prototype._getAttendees =
295 function() {
296 	var attendees = this.invite.components[0].at;
297 	var emails = new AjxVector();
298 	for (var i = 0; i < (attendees ? attendees.length : 0); i++) {
299 		var at = attendees[i];
300 		emails.add(new AjxEmailAddress(at.a, null, null, at.d));
301 	}
302 
303 	return emails;
304 };

349 /**
350  * Gets the header string.
351  * 
352  * @param	{constant}	hdr			the header (see <code>ZmMailMsg.HDR_</code> constants)
353  * @param	{boolean}	htmlMode	if true, format as HTML
354  * @return	{String}	the value
355  */
356 ZmMailMsg.prototype.getHeaderStr =
357 function(hdr, htmlMode) {
358 
359 	var key, value;
360 	if (hdr == ZmMailMsg.HDR_DATE) {
361 		if (this.sentDate) {
362 			var formatter = AjxDateFormat.getDateTimeInstance(AjxDateFormat.FULL, AjxDateFormat.MEDIUM);
363 			value = formatter.format(new Date(this.sentDate));
364 		}
365 	} else if (hdr == ZmMailMsg.HDR_SUBJECT) {
366 		value = this.subject;
367 	} else {
368 		var addrs = this.getAddresses(hdr);
369 		value = addrs.toString(", ", true);
370 	}
371 
372 	var key = ZmMailMsg.HDR_KEY[hdr] + ": ";
373 	if (!value) { return; }
374 	if (htmlMode) {
375 		key = "<b>" + key + "</b>";
376 		value = AjxStringUtil.convertToHtml(value);
377 	}
378 
379 	return key + value;
380 };

834 /**
835  * Gets the text part.
836  *
837  */
838 ZmMailMsg.prototype.getTextPart =
839 function() {
840 	var bodyPart = this.getBodyPart();
841 	if (bodyPart && (bodyPart.ct == ZmMimeTable.TEXT_PLAIN || bodyPart.body && ZmMimeTable.isTextType(bodyPart.ct))) {
842 		return bodyPart.content;
843 	}
844     else if (bodyPart && bodyPart.ct == ZmMimeTable.TEXT_CAL) {
845         // NOTE: IE doesn't match multi-line regex, even when explicitly
846         // specifying the "m" attribute.
847         var lines = bodyPart.content.split(/\r\n/);
848         var desc = [];
849         var content = "";
850         for (var i = 0; i < lines.length; i++) {
851             var line = lines[i];
852             if (line.match(/^DESCRIPTION:/)) {
853                 desc.push(line.substr(12));
854                 for (var j = i + 1; j < lines.length; j++) {
855                     line = lines[j];
856                     if (line.match(/^\s+/)) {
857                         desc.push(line.replace(/^\s+/, " "));
858                         continue;
859                     }
860                     break;
861                 }
862                 break;
863             }
864             else if (line.match(/^COMMENT:/)) {
865                 //DESCRIPTION is sent as COMMENT in Lotus notes.
866                 desc.push(line.substr(8));
867                 for (var j = i + 1; j < lines.length; j++) {
868                     line = lines[j];
869                     if (line.match(/^\s+/)) {
870                         desc.push(line.replace(/^\s+/, " "));
871                         continue;
872                     }
873                     break;
874                 }
875                 break;
876             }
877         }
878         if (desc.length > 0) {
879             content = desc.join("");
880             content = content.replace(/\\t/g, "\t");
881             content = content.replace(/\\n/g, "\n");
882             content = content.replace(/\\(.)/g, "$1");
883         }
884         return content;
885     }
886     else if (bodyPart && bodyPart.ct != ZmMimeTable.TEXT_PLAIN && bodyPart.ct != ZmMimeTable.TEXT_HTML) {
887 		// looks like the body of this message is the attachment itself
888 		return "";
889 	}
890 	return -1;
891 };
