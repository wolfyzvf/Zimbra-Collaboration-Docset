 22 /**
 23  * Creates a new search with the given properties.
 24  * @class
 25  * This class represents a search to be performed on the server. It has properties for
 26  * the different search parameters that may be used. It can be used for a regular search,
 27  * or to search within a conversation. The results are returned via a callback.
 28  *
 29  * @param {Hash}		params		a hash of parameters
 30  * @param   {String}	params.query					the query string
 31  * @param	{String}	params.queryHint				the query string that gets appended to the query but not something the user needs to know about
 32  * @param	{AjxVector}	params.types					the item types to search for
 33  * @param	{constant}	params.sortBy					the sort order
 34  * @param	{int}		params.offset					the starting point within result set
 35  * @param	{int}		params.limit					the number of results to return
 36  * @param	{Boolean}	params.getHtml					if <code>true</code>, return HTML part for inlined msg
 37  * @param	{constant}	params.contactSource			where to search for contacts (GAL or personal)
 38  * @param	{Boolean}	params.isGalAutocompleteSearch	if <code>true</code>, autocomplete against GAL
 39  * @param	{constant}	params.galType					the type of GAL autocomplete (account or resource)
 40  * @param	{int}		params.lastId					the ID of last item displayed (for pagination)
 41  * @param	{String}	params.lastSortVal				the value of sort field for above item
 42  * @param	{Boolean}	params.fetch					if <code>true</code>, fetch first hit message
 43  * @param	{int}		params.searchId					the ID of owning search folder (if any)
 44  * @param	{Array}		params.conds					the list of search conditions (<code><SearchCalendarResourcesRequest></code>)
 45  * @param	{Array}		params.attrs					the list of attributes to return (<code><SearchCalendarResourcesRequest></code>)
 46  * @param	{String}	params.field					the field to search within (instead of default)
 47  * @param	{Object}	params.soapInfo					the object with method, namespace, response, and additional attribute fields for creating soap doc
 48  * @param	{Object}	params.response					the canned JSON response (no request will be made)
 49  * @param	{Array}		params.folders					the list of folders for autocomplete
 50  * @param	{Array}		params.allowableTaskStatus		the list of task status types to return (assuming one of the values for "types" is "task")
 51  * @param	{String}	params.accountName				the account name to run this search against
 52  * @param	{Boolean}	params.idsOnly					if <code>true</code>, response returns item IDs only
 53  * @param   {Boolean}   params.inDumpster               if <code>true</code>, search in the dumpster
 54  * @param	{boolean}	params.expandDL					if <code>true</code>, set flag to have server indicate expandability for DLs
 55  */
 56 ZmSearch = function(params) {
 57 
 58 	if (params) {
 59 		for (var p in params) {
 60 			this[p] = params[p];
 61 		}
 62 		this.galType					= this.galType || ZmSearch.GAL_ACCOUNT;
 63 		this.join						= this.join || ZmSearch.JOIN_AND;
 64 
 65 		if (this.query || this.queryHint) {
 66 			if (!this.isGalSearch && !this.isAutocompleteSearch &&
 67 				!this.isGalAutocompleteSearch && !this.isCalResSearch) {
 68 
 69 				this._parseQuery();	// only parse regular searches
 70 			}
 71 			if (this.querySortOrder) {
 72 				this.sortBy = this.querySortOrder;
 73 			}
 74 		}
 75 	}
 76 	this.isGalSearch = false;
 77 	this.isCalResSearch = false;
 78 
 79 	if (ZmSearch._mailEnabled == null) {
 80 		ZmSearch._mailEnabled = appCtxt.get(ZmSetting.MAIL_ENABLED);
 81 		if (ZmSearch._mailEnabled) {
 82 			AjxDispatcher.require("MailCore");
 83 		}
 84 	}
 85 
 86 	if (!(this.types instanceof AjxVector)) {
 87 		this.types = AjxUtil.toArray(this.types);
 88         if (!appCtxt.get(ZmSetting.MAIL_ENABLED)) {
 89             this.types = AjxUtil.arrayAsHash(this.types);
 90             delete this.types[ZmSearch.TYPE[ZmItem.MSG]];
 91             delete this.types[ZmSearch.TYPE[ZmItem.CONV]];
 92             this.types = AjxUtil.keys(this.types);
 93         }
 94         this.types = AjxVector.fromArray(this.types);
 95 	}
 96 
 97     // a search of no types is equivalent to a search of all allowed types
 98     if (this.types.size() == 0) {
 99         this.types = AjxVector.fromArray(AjxUtil.keys(ZmSearch.TYPE));
100     }
101 };

150 ZmSearch.prototype.execute =
151 function(params) {
152 	if (params.batchCmd || this.soapInfo) {
153 		this._executeSoap(params);
154 	} else {
155 		this._executeJson(params);
156 	}

159 /**
160  * Creates a SOAP request that represents this search and sends it to the server.
161  *
162  * @param {Hash}	params		a hash of parameters
163  * @param {AjxCallback}	params.callback		the callback to run when response is received
164  * @param {AjxCallback}	params.errorCallback	the callback to run if there is an exception
165  * @param {ZmBatchCommand}	params.batchCmd		the batch command that contains this request
166  * @param {int}	params.timeout		the timeout value (in seconds)
167  * @param {Boolean}	params.noBusyOverlay	if <code>true</code>, don't use the busy overlay
168  * 
169  * @private
170  */
171 ZmSearch.prototype._executeSoap =
172 function(params) {
173 
174 	this.isGalSearch = (this.contactSource && (this.contactSource == ZmId.SEARCH_GAL));
175 	this.isCalResSearch = (!this.contactSource && this.conds != null);
176     if (appCtxt.isOffline && this.isCalResSearch) {
177         this.isCalResSearch =  appCtxt.isZDOnline();
178     }
179 	if (!this.query && !this.isCalResSearch) return;
180 
181 	var soapDoc;
182 	if (!this.response) {
183 		if (this.isGalSearch) {
184 			// DEPRECATED: Use JSON version
185 			soapDoc = AjxSoapDoc.create("SearchGalRequest", "urn:zimbraAccount");
186 			var method = soapDoc.getMethod();
187 			if (this.galType) {
188 				method.setAttribute("type", this.galType);
189 			}
190 			if (this.expandDL) {
191 				method.setAttribute("needExp", 1);
192 			}
193 			soapDoc.set("name", this.query);
194 			var searchFilterEl = soapDoc.set("searchFilter");
195 			if (this.conds && this.conds.length) {
196 				var condsEl = soapDoc.set("conds", null, searchFilterEl);
197 				this._applySoapCond(this.conds, soapDoc, condsEl);
198 			}
199 		} else if (this.isAutocompleteSearch) {
200 			soapDoc = AjxSoapDoc.create("AutoCompleteRequest", "urn:zimbraMail");
201 			var method = soapDoc.getMethod();
202 			if (this.limit) {
203 				method.setAttribute("limit", this.limit);
204 			}
205 			if (this.expandDL) {
206 				method.setAttribute("needExp", 1);
207 			}
208 			soapDoc.set("name", this.query);
209 		} else if (this.isGalAutocompleteSearch) {
210 			soapDoc = AjxSoapDoc.create("AutoCompleteGalRequest", "urn:zimbraAccount");
211 			var method = soapDoc.getMethod();
212 			method.setAttribute("limit", this._getLimit());
213 			if (this.galType) {
214 				method.setAttribute("type", this.galType);
215 			}
216 			if (this.expandDL) {
217 				method.setAttribute("needExp", 1);
218 			}
219 			soapDoc.set("name", this.query);
220 		} else if (this.isCalResSearch) {
221 			soapDoc = AjxSoapDoc.create("SearchCalendarResourcesRequest", "urn:zimbraAccount");
222 			var method = soapDoc.getMethod();
223 			if (this.attrs) {
224 				var attrs = [].concat(this.attrs);
225 				AjxUtil.arrayRemove(attrs, "fullName");
226 				method.setAttribute("attrs", attrs.join(","));
227 			}
228 			var searchFilterEl = soapDoc.set("searchFilter");
229 			if (this.conds && this.conds.length) {
230 				var condsEl = soapDoc.set("conds", null, searchFilterEl);
231 				this._applySoapCond(this.conds, soapDoc, condsEl);
232 			}
233 		} else {
234 			if (this.soapInfo) {
235 				soapDoc = AjxSoapDoc.create(this.soapInfo.method, this.soapInfo.namespace);
236 				// Pass along any extra soap data. (Voice searches use this to pass user identification.)
237 				for (var nodeName in this.soapInfo.additional) {
238 					var node = soapDoc.set(nodeName);
239 					var attrs = this.soapInfo.additional[nodeName];
240 					for (var attr in attrs) {
241 						node.setAttribute(attr, attrs[attr]);
242 					}
243 				}
244 			} else {
245 				soapDoc = AjxSoapDoc.create("SearchRequest", "urn:zimbraMail");
246 			}
247 			var method = this._getStandardMethod(soapDoc);
248 			if (this.types) {
249 				var a = this.types.getArray();
250 				if (a.length) {
251 					var typeStr = [];
252 					for (var i = 0; i < a.length; i++) {
253 						typeStr.push(ZmSearch.TYPE[a[i]]);
254 					}
255 					method.setAttribute("types", typeStr.join(","));
256 					// special handling for showing participants ("To" instead of "From")
257 					var folder = appCtxt.getById(this.folderId);
258 					if (folder &&
259 						(folder.isUnder(ZmFolder.ID_SENT) ||
260 						folder.isUnder(ZmFolder.ID_DRAFTS) ||
261 						folder.isUnder(ZmFolder.ID_OUTBOX)))
262 					{
263 						method.setAttribute("recip", "1");
264 					}
265 					// if we're prefetching the first hit message, also mark it as read
266 					if (this.fetch) {
267 
268 						method.setAttribute("fetch", ( this.fetch == "all" ) ? "all" : "1");
269 						// and set the html flag if we want the html version
270 						if (this.getHtml) {
271 							method.setAttribute("html", "1");
272 						}
273 					}
274 					if (this.markRead) {
275 						method.setAttribute("read", "1");
276 					}
277 				}
278 			}
279 			if (this.inDumpster) {
280 				method.setAttribute("inDumpster", "1");
281 			}
282 		}
283 	}
284 
285 	var respCallback = new AjxCallback(this, this._handleResponseExecute, [params.callback]);
286 
287 	if (params.batchCmd) {
288 		params.batchCmd.addRequestParams(soapDoc, respCallback);
289 	} else {
290 		return appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:true, callback:respCallback,
291 													   errorCallback:params.errorCallback,
292 													   timeout:params.timeout, noBusyOverlay:params.noBusyOverlay,
293 													   response:this.response});
294 	}
295 };

297 /**
298  * Creates a JSON request that represents this search and sends it to the server.
299  *
300  * @param {Hash}	params		a hash of parameters
301  * @param {AjxCallback}	params.callback		the callback to run when response is received
302  * @param {AjxCallback}	params.errorCallback	the callback to run if there is an exception
303  * @param {ZmBatchCommand}	params.batchCmd		the batch command that contains this request
304  * @param {int}	params.timeout		the timeout value (in seconds)
305  * @param {Boolean}	params.noBusyOverlay	if <code>true</code>, don't use the busy overlay
306  * 
307  * @private
308  */
309 ZmSearch.prototype._executeJson =
310 function(params) {
311 
312 	this.isGalSearch = (this.contactSource && (this.contactSource == ZmId.SEARCH_GAL));
313 	this.isCalResSearch = (!this.contactSource && this.conds != null);
314     if (appCtxt.isOffline && this.isCalResSearch) {
315         this.isCalResSearch = appCtxt.isZDOnline();
316     }
317 	if (!this.query && !this.queryHint && !this.isCalResSearch) { return; }
318 
319 	var jsonObj, request, soapDoc;
320 	if (!this.response) {
321 		if (this.isGalSearch) {
322 			jsonObj = {SearchGalRequest:{_jsns:"urn:zimbraAccount"}};
323 			request = jsonObj.SearchGalRequest;
324 			if (this.galType) {
325 				request.type = this.galType;
326 			}
327 			if (this.expandDL) {
328 				request.needExp = 1;
329 			}
330 			request.name = this.query;
331 
332 			// bug #36188 - add offset/limit for paging support
333 			request.offset = this.offset = (this.offset || 0);
334 			request.limit = this._getLimit();
335 
336 			// bug 15878: see same in ZmSearch.prototype._getStandardMethodJson
337 			request.locale = { _content: AjxEnv.DEFAULT_LOCALE };
338 
339 			if (this.lastId) { // add lastSortVal and lastId for cursor-based paging
340 				request.cursor = {id:this.lastId, sortVal:(this.lastSortVal || "")};
341 			}
342 			if (this.sortBy) {
343 				request.sortBy = this.sortBy;
344 			}
345 			if (this.conds && this.conds.length) {
346 				request.searchFilter = {conds:{}};
347 				request.searchFilter.conds = ZmSearch.prototype._applyJsonCond(this.conds, request);
348 			}
349 		} else if (this.isAutocompleteSearch) {
350 			jsonObj = {AutoCompleteRequest:{_jsns:"urn:zimbraMail"}};
351 			request = jsonObj.AutoCompleteRequest;
352 			if (this.limit) {
353 				request.limit = this.limit;
354 			}
355 			if (this.expandDL) {
356 				request.needExp = 1;
357 			}
358 			request.name = {_content:this.query};
359 		} else if (this.isGalAutocompleteSearch) {
360 			jsonObj = {AutoCompleteGalRequest:{_jsns:"urn:zimbraAccount"}};
361 			request = jsonObj.AutoCompleteGalRequest;
362 			request.limit = this._getLimit();
363 			request.name = this.query;
364 			if (this.galType) {
365 				request.type = this.galType;
366 			}
367 			if (this.expandDL) {
368 				request.needExp = 1;
369 			}
370 		} else if (this.isCalResSearch) {
371 			jsonObj = {SearchCalendarResourcesRequest:{_jsns:"urn:zimbraAccount"}};
372 			request = jsonObj.SearchCalendarResourcesRequest;
373 			if (this.attrs) {
374 				var attrs = [].concat(this.attrs);
375 				request.attrs = attrs.join(",");
376 			}
377             request.offset = this.offset = (this.offset || 0);
378             request.limit = this._getLimit();
379 			if (this.conds && this.conds.length) {
380 				request.searchFilter = {conds:{}};
381 				request.searchFilter.conds = ZmSearch.prototype._applyJsonCond(this.conds, request);
382 			}
383 		} else {
384 			if (this.soapInfo) {
385 				soapDoc = AjxSoapDoc.create(this.soapInfo.method, this.soapInfo.namespace);
386 				// Pass along any extra soap data. (Voice searches use this to pass user identification.)
387 				for (var nodeName in this.soapInfo.additional) {
388 					var node = soapDoc.set(nodeName);
389 					var attrs = this.soapInfo.additional[nodeName];
390 					for (var attr in attrs) {
391 						node.setAttribute(attr, attrs[attr]);
392 					}
393 				}
394 			} else {
395 				jsonObj = {SearchRequest:{_jsns:"urn:zimbraMail"}};
396 				request = jsonObj.SearchRequest;
397 			}
398 			this._getStandardMethodJson(request);
399 			if (this.types) {
400 				var a = this.types.getArray();
401 				if (a.length) {
402 					var typeStr = [];
403 					for (var i = 0; i < a.length; i++) {
404 						typeStr.push(ZmSearch.TYPE[a[i]]);
405 					}
406 					request.types = typeStr.join(",");
407 
408 					// special handling for showing participants ("To" instead of "From")
409 					var folder = appCtxt.getById(this.folderId);
410 					if (folder &&
411 						(folder.isUnder(ZmFolder.ID_SENT) ||
412 						folder.isUnder(ZmFolder.ID_DRAFTS) ||
413 						folder.isUnder(ZmFolder.ID_OUTBOX)))
414 					{
415 						request.recip = 1;
416 					}
417 
418 					// if we're prefetching the first hit message, also mark it as read
419 					if (this.fetch) {
420                         request.fetch = ( this.fetch == "all" ) ? "all" : 1;
421 						// and set the html flag if we want the html version
422 						if (this.getHtml) {
423 							request.html = 1;
424 						}
425 					}
426 
427 					if (this.markRead) {
428 						request.read = 1;
429 					}
430 
431                     if (this.headers) {
432                         for (var hdr in this.headers) {
433                             if (!request.header) { request.header = []; }
434                             request.header.push({n:hdr});
435                         }
436                     }
437 
438 					if (a.length == 1 && a[0] == ZmItem.TASK && this.allowableTaskStatus) {
439 						request.allowableTaskStatus = this.allowableTaskStatus;
440 					}
441                 }
442             }
443 			if (this.inDumpster) {
444 				request.inDumpster = 1;
445 			}
446         }
447     }
448 
449 	var respCallback = new AjxCallback(this, this._handleResponseExecute, [params.callback]);
450 
451 	if (params.batchCmd) {
452 		params.batchCmd.addRequestParams(soapDoc, respCallback);
453 	} else {
454 		var searchParams = {
455 			jsonObj:jsonObj,
456 			soapDoc:soapDoc,
457 			asyncMode:true,
458 			callback:respCallback,
459 			errorCallback:params.errorCallback,
460 			timeout:params.timeout,
461 			noBusyOverlay:params.noBusyOverlay,
462 			response:this.response,
463 			accountName:this.accountName
464 		};
465 		return appCtxt.getAppController().sendRequest(searchParams);
466 	}
467 };

469 ZmSearch.prototype._applySoapCond =
470 function(inConds, soapDoc, condsEl, or) {
471 	if (or || this.join == ZmSearch.JOIN_OR) {
472 		condsEl.setAttribute("or", 1);
473 	}
474 	for (var i = 0; i < inConds.length; i++) {
475 		var c = inConds[i];
476 		if (AjxUtil.isArray(c)) {
477 			var subCondsEl = soapDoc.set("conds", null, condsEl);
478 			this._applySoapCond(c, soapDoc, subCondsEl, true);
479 		} else if (c.attr=="fullName" && c.op=="has") {
480 			var nameEl = soapDoc.set("name", c.value);
481 		} else {
482 			var condEl = soapDoc.set("cond", null, condsEl);
483 			condEl.setAttribute("attr", c.attr);
484 			condEl.setAttribute("op", c.op);
485 			condEl.setAttribute("value", c.value);
486 		}
487 	}
488 };
489 
490 ZmSearch.prototype._applyJsonCond =
491 function(inConds, request, or) {
492 	var outConds = {};
493 	if (or || this.join == ZmSearch.JOIN_OR) {
494 		outConds.or = 1;
495 	}
496 
497 	for (var i = 0; i < inConds.length; i++) {
498 		var c = inConds[i];
499 		if (AjxUtil.isArray(c)) {
500 			if (!outConds.conds)
501 				outConds.conds = [];
502 			outConds.conds.push(this._applyJsonCond(c, request, true));
503 		} else if (c.attr=="fullName" && c.op=="has") {
504 			request.name = {_content: c.value};
505 		} else {
506 			if (!outConds.cond)
507 				outConds.cond = [];
508 			outConds.cond.push({attr:c.attr, op:c.op, value:c.value});
509 		}
510 	}
511 	return outConds;
512 };

514 /**
515  * Converts the response into a {ZmSearchResult} and passes it along.
516  * 
517  * @private
518  */
519 ZmSearch.prototype._handleResponseExecute =
520 function(callback, result) {
521 	var response = result.getResponse();
522 
523 	if      (this.isGalSearch)				{ response = response.SearchGalResponse; }
524 	else if (this.isCalResSearch)			{ response = response.SearchCalendarResourcesResponse; }
525 	else if (this.isAutocompleteSearch)		{ response = response.AutoCompleteResponse; }
526 	else if (this.isGalAutocompleteSearch)	{ response = response.AutoCompleteGalResponse; }
527 	else if (this.soapInfo)					{ response = response[this.soapInfo.response]; }
528 	else									{ response = response.SearchResponse; }
529 
530 	var searchResult = new ZmSearchResult(this);
531 	searchResult.set(response);
532 	result.set(searchResult);
533 
534 	if (callback) {
535 		callback.run(result);
536 	}
537 };

539 /**
540  * Fetches a conversation from the server.
541  *
542  * @param {Hash}		params				a hash of parameters:
543  * @param {String}		params.cid			the conv ID
544  * @param {AjxCallback}	params.callback		the callback to run with result
545  * @param {String}		params.fetchId		the ID of msg to load
546  * @param {Boolean}		params.markRead		if <code>true</code>, mark msg read
547  * @param {Boolean}		params.noTruncate	if <code>true</code>, do not limit size of msg
548  * @param {boolean}		params.needExp		if not <code>false</code>, have server check if addresses are DLs
549  */
550 ZmSearch.prototype.getConv =
551 function(params) {
552 	if ((!this.query && !this.queryHint) || !params.cid) { return; }
553 
554 	var jsonObj = {SearchConvRequest:{_jsns:"urn:zimbraMail"}};
555 	var request = jsonObj.SearchConvRequest;
556 	this._getStandardMethodJson(request);
557 	request.cid = params.cid;
558 	if (params.fetchId) {
559 		request.fetch = params.fetchId;	// fetch content of this msg
560 		if (params.markRead) {
561 			request.read = 1;			// mark that msg read
562 		}
563 		if (this.getHtml) {
564 			request.html = 1;			// get it as HTML
565 		}
566 		if (params.needExp !== false) {
567 			request.needExp = 1;
568 		}
569 		// added headers to the request
570 		if (ZmMailMsg.requestHeaders) {
571 			for (var hdr in ZmMailMsg.requestHeaders) {
572 				if (!request.header) request.header = [];
573 				request.header.push({n:hdr});
574 			}
575 		}
576 	}
577 
578 	if (!params.noTruncate) {
579 		request.max = appCtxt.get(ZmSetting.MAX_MESSAGE_SIZE);
580 	}
581 
582 	var searchParams = {
583 		jsonObj: jsonObj,
584 		asyncMode: true,
585 		callback: (new AjxCallback(this, this._handleResponseGetConv, params.callback)),
586 		accountName:this.accountName
587 	};
588 	appCtxt.getAppController().sendRequest(searchParams);
589 };

798 /**
799  * Parse simple queries so we can do basic matching on new items (determine whether
800  * they match this search query). The following types of query terms are handled:
801  *
802  *    in:[folder]
803  *    tag:[tag]
804  *    is:[flag]
805  *
806  * Those may be joined by conditionals: "and", "or", "not", "-", and the implied "and"
807  * that appears between consecutive terms. The result of the parsing is the creation
808  * of a function that takes an item as its argument and returns true if that item
809  * matches this search. If the parsing fails for any reason, the function is not
810  * created.
811  *
812  * If the query contains a term of "in:" or "tag:" and has no OR conditionals, then
813  * this.folderId or this.tagId will be set.
814  *
815  * Compound terms such as "in:(inbox or sent)" will not result in the creation of
816  * a match function, nor will anything that invokes a text search
817  * (such as "in:inbox xml").
818  * 
819  * @private
820  */
821 ZmSearch.prototype._parseQuery =
822 function() {
823 
824 	var query = this.query || this.queryHint; 
825 	this.hasUnreadTerm = ZmSearch.UNREAD_QUERY_RE.test(query);
826 	this.isAnywhere = ZmSearch.IS_ANYWHERE_QUERY_RE.test(query);
827 
828 	function skipSpace(str, pos) {
829 		while (pos < str.length && str.charAt(pos) == " ") {
830 			pos++;
831 		}
832 		return pos;
833 	}
834 
835 	function getQuotedStr(str, pos) {
836 		var q = str.charAt(pos);
837 		pos++;
838 		var done = false, ch, quoted = "";
839 		while (pos < str.length && !done) {
840 			ch = str.charAt(pos);
841 			if (ch == q) {
842 				done = true;
843 			} else {
844 				quoted += ch;
845 				pos++;
846 			}
847 		}
848 
849 		return done ? {str:quoted, pos:pos + 1} : null;
850 	}
851 
852 
853 	var len = query.length;
854 	var tokens = [], ch, op, word = "", fail = false, eow = false, endOk = true, hasOrTerm = false, bareWord = false;
855 	var pos = skipSpace(query, 0);
856 	while (pos < len) {
857 		ch = query.charAt(pos);
858 		eow = ZmSearch.EOW[ch];
859 
860 		if (ch == ":") {
861 			if (ZmSearch.IS_OP[word]) {
862 				op = word;
863 			} else {
864 				fail = true;
865 			}
866 			word = "";
867 			pos = skipSpace(query, pos + 1);
868 			continue;
869 		}
870 
871 		if (eow) {
872 			if (op && word) {
873 				tokens.push({isTerm:true, op:op, arg:word});
874 				op = word = "";
875 				endOk = true;
876 			} else if (!op) {
877 				if (ZmSearch.COND[word.toLowerCase()]) {
878 					var cond = word.toLowerCase();
879 					tokens.push(ZmSearch.COND[cond]);
880 					endOk = false;
881 					if (cond == "or") {
882 						hasOrTerm = true;
883 					}
884 				} else if (word) {
885 					fail = bareWord = true;
886 				}
887 				word = "";
888 			}
889 		}
890 
891 		if (ch == "'" || ch == '"') {
892 			var results = getQuotedStr(query, pos);
893 			if (results) {
894 				word = results.str;
895 				pos = results.pos;
896 			} else {
897 				fail = true;
898 				word = "";
899                 pos = skipSpace(query, pos + 1);
900 			}
901 		} else if (ch == "(" || ch == ")") {
902 			tokens.push(ch);
903 			pos = skipSpace(query, pos + 1);
904 		} else if (ch == "-" && !word) {
905 			tokens.push("not");
906 			pos = skipSpace(query, pos + 1);
907 			endOk = false;
908 		} else {
909 			if (ch != " ") {
910 				word += ch;
911 			}
912 			pos++;
913 		}
914 	}
915 
916 	// check for term at end
917 	if (!fail && (pos == query.length) && op && word) {
918 		tokens.push({isTerm:true, op:op, arg:word});
919 		endOk = true;
920 	} else if (!op && word) {
921 		fail = bareWord = true;
922 	}
923 
924 	fail = fail || !endOk;
925 
926 	var folderId, tagId;
927 	var func = ["return Boolean("];
928 	for (var i = 0, len = tokens.length; i < len; i++) {
929 		var t = tokens[i];
930 		if (t.isTerm) {
931 			if (t.op == "in" || t.op == "inid") {
932 				folderId = (t.op == "in") ? this._getFolderId(t.arg) : t.arg;
933 				if (folderId) {
934 					func.push("((item.type == ZmItem.CONV) ? item.folders && item.folders['" + folderId +"'] : item.folderId == '" + folderId + "')");
935 				}
936 			} else if (t.op == "tag") {
937 				tagId = this._getTagId(t.arg, true);
938 				if (tagId) {
939 					func.push("item.hasTag('" + tagId + "')");
940 				}
941 			} else if (t.op == "is") {
942 				var test = ZmSearch.FLAG[t.arg];
943 				if (test) {
944 					func.push(test);
945 				}
946 			} else if (t.op == "sort") {
947 				this.querySortOrder = t.arg;
948 			}
949 			var next = tokens[i + 1];
950 			if (next && (next.isTerm || next == ZmSearch.COND["not"] || next == "(")) {
951 				func.push(ZmSearch.COND["and"]);
952 			}
953 		} else {
954 			func.push(t);
955 		}
956 	}
957 	func.push(")");
958 
959 	if (!fail) {
960 		try {
961 			this.matches = new Function("item", func.join(""));
962 		} catch(ex) {}
963 	}
964 
965 	// this way we know if a search was simply "in:foo" or "tag:foo", and nothing else
966 	this.singleTerm = (folderId || tagId) && (tokens.length == 1) && !fail;
967 
968 	// has a text term (word by itself with no operator)
969 	this.hasTextTerm = bareWord;
970 
971 	// the way multi-account searches are done, we set the queryHint *only* so
972 	// set the folderId if it exists for simple multi-account searches
973 	var isMultiAccountSearch = (appCtxt.multiAccounts && this.isMultiAccount() && !this.query && this.queryHint);
974 	if (!hasOrTerm || isMultiAccountSearch) {
975 		this.folderId = folderId;
976 		this.tagId = tagId;
977 	}
978 };
