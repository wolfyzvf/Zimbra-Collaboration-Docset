 20 /**
 21  * Creates a login dialog.
 22  * @class
 23  * This class represents a login dialog.
 24  * 
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtComposite
 29  */
 30 ZmLoginDialog = function(parent, className) {
 31 	
 32     className = className || "LoginScreen";
 33     DwtComposite.call(this, {parent:parent, className:className, posStyle:DwtControl.ABSOLUTE_STYLE});
 34     this._origClassName = className;
 35     this._xparentClassName = className + "-Transparent";
 36     this.setBounds(0, 0, "100%", "100%");
 37     var htmlElement = this.getHtmlElement();
 38     htmlElement.style.zIndex = Dwt.Z_DIALOG + 1; // + 1 to keep appointment reminders underneath login dialog.
 39     htmlElement.className = className;
 40     this.setVisible(false);
 41 
 42 	var params = ZLoginFactory.copyDefaultParams(ZmMsg);
 43 	params.showPanelBorder = true;
 44 	params.showForm = true;
 45 	params.showUserField = true;
 46 	params.showPasswordField = true;
 47 	params.showLicenseMsg = true;
 48 	params.showRememberMeCheckbox = false;
 49 	params.showLogOff = true;
 50 	params.logOffAction = "ZmLoginDialog._loginDiffListener()";
 51 	params.loginAction = "ZmLoginDialog._loginListener(this)";
 52 	params.showButton = true;
 53 
 54 	var html = [];
 55 	var idx = 0;
 56 	html[idx++] = "<table border=0 cellspacing=0 cellpadding=0 style='width:100%; height:100%'><tr><td align='center' valign='center'>";
 57 	html[idx++] = ZLoginFactory.getLoginDialogHTML(params);
 58 	html[idx++] = "</td></tr></table>";
 59 	htmlElement.innerHTML = html.join("");
 60 	
 61 	// Login button is not a DwtButton, so set up reference to this dialog
 62 	var id = ZLoginFactory.LOGIN_BUTTON_ID;
 63 	if (id && DwtControl.ALL_BY_ID) {
 64 		DwtControl.ALL_BY_ID[id] = this;
 65 	}
 66 
 67     this.setReloginMode(false);
 68 };

102 /**
103  * Sets the error message.
104  * 
105  * @param	{String}	errorStr		the error string
106  */
107 ZmLoginDialog.prototype.setError =
108 function(errorStr) {
109 	if (errorStr && errorStr.length) {
110 		ZLoginFactory.showErrorMsg(errorStr);
111 	} else {
112 		ZLoginFactory.hideErrorMsg();
113 	}
114 };

116 /**
117  * Sets the focus.
118  * 
119  * @param	{String}		username		the username
120  * @param	{Boolean}		bReloginMode	if <code>true</code>, relogin mode
121  */
122 ZmLoginDialog.prototype.setFocus =
123 function(username, bReloginMode) {
124 	ZLoginFactory.showUserField(username);
125 	this.setReloginMode(username && username.length && bReloginMode);
126 };
127 
128 ZmLoginDialog.prototype.setVisible = 
129 function(visible, transparentBg) {
130 	DwtComposite.prototype.setVisible.call(this, visible);
131 
132 	// Disable keyboard nav for re-login dialog
133 	var kbm = appCtxt.getKeyboardMgr();
134 	if (kbm) {
135 		if (visible) {
136 		   	if (kbm.isEnabled()) {
137    				kbm.enable(false);
138    				this._kbnavDisabled = true;
139    			}
140 		} else {
141    		    if (this._kbnavDisabled) {
142 		    	kbm.enable(true);
143 		    	this._kbnavDisabled = false;
144 		    }
145 		}
146 	}
147    	
148 	if (!visible) return;
149 		
150 	this.setCursor("default");
151 	if ((transparentBg == null || !transparentBg) && this._className != this._origClassName) {
152 		this.getHtmlElement().className = this._origClassName;
153 		this._className = this._origClassName;
154 	} else if (transparentBg && this._className != this._xparentClassName) {
155 		this.getHtmlElement().className = this._xparentClassName;
156 		this._className = this._xparentClassName;
157 	}
158 
159 	Dwt.setHandler(this.getHtmlElement(), DwtEvent.ONKEYDOWN, ZLoginFactory.handleKeyPress);
160 
161 	var passwordField = ZLoginFactory.get(ZLoginFactory.PASSWORD_ID);
162 	if (passwordField && passwordField.focus) {
163 		passwordField.focus();
164 	}
165 };

172 ZmLoginDialog.prototype.setReloginMode = 
173 function(bReloginMode) {
174 	bReloginMode ? ZLoginFactory.showLogOff() : ZLoginFactory.hideLogOff();
175 	ZLoginFactory.get(ZLoginFactory.USER_ID).disabled = bReloginMode;
176 };
177 
178 ZmLoginDialog.prototype._loginSelListener =
179 function() {
180 	this.setCursor("wait");
181 	var username = ZLoginFactory.get(ZLoginFactory.USER_ID).value;
182 	if (!(username && username.length)) {
183 		this.setError(ZmMsg.enterUsername);
184 		return;
185 	}
186 	if (this._callback) {
187 		var password = ZLoginFactory.get(ZLoginFactory.PASSWORD_ID).value;
188 		this._callback.run(username, password, appCtxt.rememberMe);		
189 	}
190 };
191 
192 ZmLoginDialog._loginListener =
193 function(target) {
194 	// Get the dialog instance.
195 	var element = target;
196 	while (element) {
197 		var object = DwtControl.fromElement(element);
198 		if (object instanceof ZmLoginDialog) {
199 			object._loginSelListener();
200 			break;
201 		}
202 		element = element.parentNode;
203 	}
204 };

206 ZmLoginDialog._loginDiffListener =
207 function(ev) {
208 	DBG.println(AjxDebug.DBG1, "ZmLoginDialog._loginDiffListener : invoking logout");
209 	ZmZimbraMail.logOff();
210 };
