 48 function() {

 72 ZmInviteMsgView.prototype.set =
 73 function(msg) {
 74 	this._msg = msg;
 75 	var invite = this._invite = msg.invite;
 76 
 77 	if (invite && invite.hasAcceptableComponents() &&
 78 		msg.folderId != ZmFolder.ID_SENT)
 79 	{
 80 		if (msg.isInviteCanceled()) {
 81 			//appointment was canceled (not necessarily on this instance, but by now it is canceled. Do not show the toolbar.
 82 			return;
 83 		}
 84 		if (invite.hasCounterMethod()) {
 85 			if (!this._counterToolbar) {
 86 				this._counterToolbar = this._getCounterToolbar();
 87 			}
 88 			this._counterToolbar.reparentHtmlElement(this.parent.getHtmlElement(), 0);
 89 			this._counterToolbar.setVisible(Dwt.DISPLAY_BLOCK);
 90 		}
 91 		else if (!invite.isOrganizer() && invite.hasInviteReplyMethod()) {
 92 			var ac = window.parentAppCtxt || window.appCtxt;
 93 			if (AjxEnv.isIE && this._inviteToolbar) {
 94 				//according to fix to bug 52412 reparenting doestn't work on IE. so I don't reparent for IE but also if the toolbar element exists,
 95 				//I remove it from parent since in the case of double-click message view, it appears multiple times without removing it
 96 				this._inviteToolbar.dispose();
 97 				this._inviteToolbar = null;
 98 			}
 99 
100 			var inviteToolbar = this.getInviteToolbar();
101 
102 			inviteToolbar.reparentHtmlElement(this.parent.getHtmlElement(), 0);
103 			inviteToolbar.setVisible(Dwt.DISPLAY_BLOCK);
104 
105 			// show on-behalf-of info?
106 			this._respondOnBehalfLabel.innerHTML = msg.cif
107 				? AjxMessageFormat.format(ZmMsg.onBehalfOfText, [msg.cif]) : "";
108 			Dwt.setVisible(this._respondOnBehalfLabel, (!!msg.cif));
109 
110 			// logic for showing calendar/folder chooser
111 			var cc = AjxDispatcher.run("GetCalController");
112 			var msgAcct = msg.getAccount();
113 			var calendars = ac.get(ZmSetting.CALENDAR_ENABLED, null, msgAcct)
114 				? cc.getCalendars({includeLinks:true, account:msgAcct, onlyWritable:true}) : [];
115 
116 			if (appCtxt.multiAccounts) {
117 				var accounts = ac.accountList.visibleAccounts;
118 				for (var i = 0; i < accounts.length; i++) {
119 					var acct = accounts[i];
120 					if (acct == msgAcct || !ac.get(ZmSetting.CALENDAR_ENABLED, null, acct)) { continue; }
121 					if (appCtxt.isOffline && acct.isMain) { continue; }
122 
123 					calendars = calendars.concat(cc.getCalendars({includeLinks:true, account:acct, onlyWritable:true}));
124 				}
125 
126 				// always add the local account *last*
127 				if (appCtxt.isOffline) {
128 					calendars.push(appCtxt.getById(ZmOrganizer.ID_CALENDAR));
129 				}
130 			}
131 
132 			var visible = (calendars.length > 1 || appCtxt.multiAccounts);
133 			if (visible) {
134 				this._inviteMoveSelect.clearOptions();
135 				for (var i = 0; i < calendars.length; i++) {
136 					var calendar = calendars[i];
137 					var calAcct = calendar.getAccount();
138 					var icon = appCtxt.multiAccounts ? calAcct.getIcon() : (calendar.getIcon() + ",color=" + calendar.color);
139 					var name = appCtxt.multiAccounts
140 						? ([calendar.name, " (", calAcct.getDisplayName(), ")"].join(""))
141 						: calendar.name;
142 					var isSelected = (calAcct && msgAcct)
143 						? (calAcct == msgAcct && calendar.nId == ZmOrganizer.ID_CALENDAR)
144 						: calendar.nId == ZmOrganizer.ID_CALENDAR;
145                     //bug: 57538 - this invite is intended for owner of shared calendar which should be selected
146                     if(msg.cif && calendar.owner == msg.cif && calendar.rid == ZmOrganizer.ID_CALENDAR) isSelected = true;
147 					var option = new DwtSelectOptionData(calendar.id, name, isSelected, null, icon);
148 					this._inviteMoveSelect.addOption(option);
149 				}
150 
151 				// for accounts that don't support calendar, always set the
152 				// selected calendar to the Local calendar
153 				if (!ac.get(ZmSetting.CALENDAR_ENABLED, null, msgAcct)) {
154 					this._inviteMoveSelect.setSelectedValue(ZmOrganizer.ID_CALENDAR);
155 				}
156 			}
157 			this._inviteMoveLabel.setVisible(visible);
158 			this._inviteMoveSelect.setVisible(visible);
159 		}
160 	}

163 /**
164  * This method does two things:
165  * 1) Checks if invite was responded to with accept/decline/tentative, and if so,
166  *    a GetAppointmentRequest is made to get the status for the other attendees.
167  *
168  * 2) Requests the free/busy status for the start date and renders the day view
169  *    with the results returned.
170  */
171 ZmInviteMsgView.prototype.showMoreInfo =
172 function(callback) {
173 	var apptId = this._invite && this._invite.hasAttendeeResponse() && this._invite.getAppointmentId();
174 	if (apptId) {
175 		var jsonObj = {GetAppointmentRequest:{_jsns:"urn:zimbraMail"}};
176 		var request = jsonObj.GetAppointmentRequest;
177 		request.id = apptId;
178 
179 		appCtxt.getAppController().sendRequest({
180 			jsonObj: jsonObj,
181 			asyncMode: true,
182 			callback: (new AjxCallback(this, this._handleShowMoreInfo, [callback]))
183 		});
184 	}
185 	else {
186 		this._showFreeBusy();
187 		if (callback) {
188 			callback.run();
189 		}
190 	}
191 };

193 ZmInviteMsgView.prototype._handleShowMoreInfo =
194 function(callback, result) {
195 	var appt = result && result.getResponse().GetAppointmentResponse.appt[0];
196 	if (appt) {
197 		var om = this.parent._objectManager;
198 		var html = [];
199 		var idx = 0;
200 		var attendees = appt.inv[0].comp[0].at || [];
201         AjxDispatcher.require(["CalendarCore"]);
202 
203         var options = {};
204 	    options.addrBubbles = appCtxt.get(ZmSetting.USE_ADDR_BUBBLES);
205 	    options.shortAddress = appCtxt.get(ZmSetting.SHORT_ADDRESS);
206 
207 		for (var i = 0; i < attendees.length; i++) {
208 			var at = attendees[i];
209 			var subs = {
210 				icon: ZmCalItem.getParticipationStatusIcon(at.ptst),
211 				attendee: (om ? om.findObjects((new AjxEmailAddress(at.a)), true, ZmObjectManager.EMAIL, false, options) : at.a)
212 			};
213 			html[idx++] = AjxTemplate.expand("mail.Message#InviteHeaderPtst", subs);
214 		}
215 
216 		var ptstEl = document.getElementById(this._ptstId);
217         if(ptstEl)
218             ptstEl.innerHTML = html.join("");
219 	}
220 
221 	if (callback) {
222 		callback.run();
223 	}
224 
225 	this._showFreeBusy();
226 };

228 ZmInviteMsgView.prototype._showFreeBusy =
229 function() {
230 	var ac = window.parentAppCtxt || window.appCtxt;
231 
232 	if (!appCtxt.isChildWindow &&
233 		(ac.get(ZmSetting.CALENDAR_ENABLED) || ac.multiAccounts) &&
234 		(this._invite && this._invite.type != "task"))
235 	{
236 
237 		var inviteDate = this._invite.getServerStartDate(null, true);
238 		if (inviteDate == null) { /* not sure when this happens (probably a bug) but this is defensive check for bug 51754 */
239 			return;
240 		}
241 
242 		var inviteTz = this._invite.getServerStartTimeTz();
243 
244 		inviteDate = AjxTimezone.convertTimezone(inviteDate, AjxTimezone.getClientId(inviteTz), AjxTimezone.DEFAULT);
245 
246 
247 		AjxDispatcher.require(["CalendarCore", "Calendar"]);
248 		var cc = AjxDispatcher.run("GetCalController");
249 
250 		if (!this._dayView) {
251 			// create a new ZmCalDayView under msgview's parent otherwise, we
252 			// cannot position the day view correctly.
253 			this._dayView = new ZmCalDayView(this.parent.parent, DwtControl.ABSOLUTE_STYLE, cc, null, null, null, true, true, this.isRight());
254 			this._dayView.addSelectionListener(new AjxListener(this, this._apptSelectionListener));
255 			this._dayView.setZIndex(Dwt.Z_VIEW); // needed by ZmMsgController's msgview
256 		}
257 
258 		this._dayView.setDisplay(Dwt.DISPLAY_BLOCK);
259 		this._dayView.setDate(inviteDate, 0, false);
260 		this.resize();
261 
262         var acctFolderIds = [].concat(cc.getCheckedCalendarFolderIds()); // create a *copy*
263         if(this._msg.cif) {
264             acctFolderIds = acctFolderIds.concat(cc.getUncheckedCalendarIdsByOwner(this._msg.cif));
265         }
266 		var rt = this._dayView.getTimeRange();
267 		var params = {
268 			start: rt.start,
269 			end: rt.end,
270 			fanoutAllDay: this._dayView._fanoutAllDay(),
271 			callback: (new AjxCallback(this, this._dayResultsCallback, [inviteDate.getHours()])),
272 			accountFolderIds: [acctFolderIds] // pass in array of array
273 		};
274 		cc.apptCache.batchRequest(params);
275 	}
276 };

283 /**
284  * Resizes the view depending on whether f/b is being shown or not.
285  *
286  * @param reset		Boolean		If true, day view is not shown and msgview's bounds need to be "reset"
287  */
288 ZmInviteMsgView.prototype.resize =
289 function(reset) {
290 	if (appCtxt.isChildWindow) { return; }
291 
292 	var isRight = this.isRight();
293 	var grandParentSize = this.parent.parent.getSize();
294 
295 	if (reset) {
296 		isRight
297 			? this.parent.setSize(Dwt.DEFAULT, grandParentSize.y)
298 			: this.parent.setSize(grandParentSize.x, Dwt.DEFAULT);
299 	} else if (this._dayView) {
300 		// bug: 50412 - fix day view for stand-alone message view which is a parent
301 		// of DwtShell and needs to be resized manually.
302 		var padding = 0;
303 		if (this.parent.getController() instanceof ZmMsgController) {
304 			// get the bounds for the app content area so we can position the day view
305 			var appContentBounds = appCtxt.getAppViewMgr()._getContainerBounds(ZmAppViewMgr.C_APP_CONTENT);
306             if (!isRight)
307 			    grandParentSize = {x: appContentBounds.width, y: appContentBounds.height};
308 
309 			// set padding so we can add it to the day view's x-location since it is a child of the shell
310 			padding = appContentBounds.x;
311 		}
312 
313 		var mvBounds = this.parent.getBounds();
314 
315 		/* on IE sometimes the value of top and left is "auto", in which case we get a NaN value here due to parseInt in getLocation. */
316 		/* not sure if 0 is the right value we should use in this case, but it seems to work */
317 		if (isNaN(mvBounds.x)) {
318 			mvBounds.x = 0;
319 		}
320 		if (isNaN(mvBounds.y)) {
321 			mvBounds.y = 0;
322 		}
323 
324 		if (isRight) {
325 			var parentHeight = grandParentSize.y;
326 			var dvHeight = Math.floor(parentHeight / 3);
327 			var mvHeight = parentHeight - dvHeight;
328 
329 			this._dayView.setBounds(mvBounds.x, mvHeight, mvBounds.width, dvHeight);
330             if (this.parent && this.parent instanceof ZmMailMsgView){
331                 var el = this.parent.getHtmlElement();
332                 if (this.mode && this.mode != "MSG") {
333                     if (el){
334                         el.style.height = mvHeight + "px";
335                         Dwt.setScrollStyle(el, Dwt.SCROLL);
336                     }
337                 }
338                 else {
339                     var bodyDiv = this.parent.getMsgBodyElement();
340                     if (bodyDiv) Dwt.setScrollStyle(bodyDiv, Dwt.CLIP);
341                     if (el) {
342                         Dwt.setScrollStyle(el, Dwt.SCROLL);
343                         el.style.height = (mvHeight - ( this._inviteToolbar ? this._inviteToolbar.getYH() : 0 ) + 10) + "px";
344                     }
345                 }
346             }
347 
348 			// don't call DwtControl's setSize() since it triggers control
349 			// listener and leads to infinite loop
350 
351 			Dwt.delClass(this.parent.getHtmlElement(), "RightBorderSeparator");
352 		} else {
353 			var parentWidth = grandParentSize.x;
354 			var dvWidth = Math.floor(parentWidth / 3);
355 			var separatorWidth = 5;
356 			var mvWidth = parentWidth - dvWidth - separatorWidth; 
357 
358 			this._dayView.setBounds(mvWidth + padding + separatorWidth, mvBounds.y, dvWidth, mvBounds.height);
359 			// don't call DwtControl's setSize() since it triggers control
360 			// listener and leads to infinite loop
361 			Dwt.setSize(this.parent.getHtmlElement(), mvWidth, Dwt.DEFAULT);
362 			Dwt.addClass(this.parent.getHtmlElement(), "RightBorderSeparator");
363 		}
364 	}
365 };

367 /**
368  * enables all invite toolbar buttons, except one that matches the current ptst
369  * @param ptst participant status
370  */
371 ZmInviteMsgView.prototype.enableToolbarButtons =
372 function(ptst) {
373 	var disableButtonId;
374 	switch (ptst) {
375 		case ZmCalBaseItem.PSTATUS_ACCEPT:
376 			disableButtonId = ZmOperation.REPLY_ACCEPT;
377 			break;
378 		case ZmCalBaseItem.PSTATUS_DECLINED:
379 			disableButtonId = ZmOperation.REPLY_DECLINE;
380 			break;
381 		case ZmCalBaseItem.PSTATUS_TENTATIVE:
382 			disableButtonId = ZmOperation.REPLY_TENTATIVE;
383 			break;
384 	}
385 	var inviteToolbar = this.getInviteToolbar();
386 
387 	var buttonIds = [ZmOperation.REPLY_ACCEPT, ZmOperation.REPLY_DECLINE, ZmOperation.REPLY_TENTATIVE];
388 	for (var i = 0; i < buttonIds.length; i++) {
389 		var buttonId = buttonIds[i];
390 		inviteToolbar.getButton(buttonId).setEnabled(buttonId != disableButtonId);
391 	}
392 };

422 ZmInviteMsgView.prototype.addSubs =
423 function(subs, sentBy, sentByAddr, obo) {
424     AjxDispatcher.require(["CalendarCore", "Calendar"]);
425 	subs.invite = this._invite;
426 
427 	if (!this._msg.isInviteCanceled() && !subs.invite.isOrganizer() && subs.invite.hasInviteReplyMethod()) {
428 		var yourPtst = this._msg.getPtst();
429 		this.enableToolbarButtons(yourPtst);
430 		if (yourPtst) {
431 			subs.ptstMsg = ZmInviteMsgView.PTST_MSG[yourPtst].msg;
432 			subs.ptstClassName = ZmInviteMsgView.PTST_MSG[yourPtst].className;
433 			subs.ptstIcon = ZmCalItem.getParticipationStatusIcon(yourPtst);
434 		}
435 	}
436 	//ids for updating later
437 	subs.ptstMsgBannerId = this._ptstMsgBannerId = (this.parent._htmlElId + "_ptstMsgBanner");
438 	subs.ptstMsgId = this._ptstMsgId = (this.parent._htmlElId + "_ptstMsg");
439 	subs.ptstMsgIconId = this._ptstMsgIconId = (this.parent._htmlElId + "_ptstMsgIcon");
440 
441 	var isOrganizer = this._invite && this._invite.isOrganizer();
442 	
443 	// counter proposal
444 	if (this._invite.hasCounterMethod() &&
445 		this._msg.folderId != ZmFolder.ID_SENT)
446 	{
447 		subs.counterInvMsg = AjxMessageFormat.format(ZmMsg.counterInviteMsg, [(sentBy && sentBy.name ) ? sentBy.name : sentByAddr]);
448 	}
449 	// if this an action'ed invite, show the status banner
450 	else if (isOrganizer && this._invite.hasAttendeeResponse()) {
451 		var attendee = this._invite.getAttendees()[0];
452 		var ptst = attendee && attendee.ptst;
453 		if (ptst) {
454 			var dispName = attendee.d || attendee.a;
455 			subs.ptstIcon = ZmCalItem.getParticipationStatusIcon(ptst);
456 
457 			switch (ptst) {
458 				case ZmCalBaseItem.PSTATUS_ACCEPT:
459 					subs.ptstMsg = AjxMessageFormat.format(ZmMsg.inviteMsgAccepted, [dispName]);
460 					subs.ptstClassName = "InviteStatusAccept";
461 					break;
462 				case ZmCalBaseItem.PSTATUS_DECLINED:
463 					subs.ptstMsg = AjxMessageFormat.format(ZmMsg.inviteMsgDeclined, [dispName]);
464 					subs.ptstClassName = "InviteStatusDecline";
465 					break;
466 				case ZmCalBaseItem.PSTATUS_TENTATIVE:
467 					subs.ptstMsg = AjxMessageFormat.format(ZmMsg.inviteMsgTentative, [dispName]);
468 					subs.ptstClassName = "InviteStatusTentative";
469 					break;
470 			}
471 		}
472 	}
473 
474     if (isOrganizer && this._invite && this._invite.hasAttendeeResponse() && this._invite.getAppointmentId()){
475         // set an Id for adding more detailed info later
476         subs.ptstId = this._ptstId = (this.parent._htmlElId + "_ptst");
477     }
478 
479     var options = {};
480 	options.addrBubbles = appCtxt.get(ZmSetting.USE_ADDR_BUBBLES);
481 	options.shortAddress = appCtxt.get(ZmSetting.SHORT_ADDRESS);
482 
483 	var om = this.parent._objectManager;
484 	// organizer
485 	var org = new AjxEmailAddress(this._invite.getOrganizerEmail(), null, this._invite.getOrganizerName());
486 	subs.invOrganizer = om ? om.findObjects(org, true, ZmObjectManager.EMAIL, false, options) : org.toString();
487 
488     if(obo) {
489         subs.obo = om ? om.findObjects(obo, true, ZmObjectManager.EMAIL) : obo.toString();
490     }
491 
492 	// sent-by
493 	var sentBy = this._invite.getSentBy();
494 	if (sentBy) {
495 		subs.invSentBy = om ? om.findObjects(sentBy, true, ZmObjectManager.EMAIL, false, options) : sentBy.toString();
496 	}
497 
498     if(this._msg.cif) {
499         subs.intendedForMsg = AjxMessageFormat.format(ZmMsg.intendedForInfo, [this._msg.cif]);
500         subs.intendedForClassName = "InviteIntendedFor";
501     }
502 
503 	// inviteees
504 	var invitees = [];
505     var optInvitees = [];
506 
507 	var list = this._invite.getAttendees();
508 	for (var i = 0; i < list.length; i++) {
509 		var at = list[i];
510 		var attendee = new AjxEmailAddress(at.a, null, at.d);
511         if (at.role == ZmCalItem.ROLE_OPTIONAL) {
512             optInvitees.push(attendee);
513         }
514         else {
515             invitees.push(attendee);
516         }
517 	}
518 	subs.invitees = this.parent.getAddressesFieldHtml(invitees, options, "inv");
519 	subs.optInvitees = this.parent.getAddressesFieldHtml(optInvitees, options, "opt");
520 
521 	// convert to local timezone if necessary
522 	var inviteTz = this._invite.getServerStartTimeTz();
523 	var defaultTz = AjxTimezone.getServerId(AjxTimezone.DEFAULT);
524 
525     var sd = AjxTimezone.convertTimezone(this._invite.getServerStartDate(null, true), AjxTimezone.getClientId(inviteTz), AjxTimezone.DEFAULT);
526 	var ed = AjxTimezone.convertTimezone(this._invite.getServerEndDate(null, true), AjxTimezone.getClientId(inviteTz), AjxTimezone.DEFAULT);
527 
528 	subs.timezone = AjxTimezone.getMediumName(defaultTz);
529 
530 	// duration text
531 	var durText = this._invite.getDurationText(null, null, null, true, sd, ed);
532 	subs.invDate = om ? om.findObjects(durText, true, ZmObjectManager.DATE) : durText;
533 
534 	// recurrence
535 	if (this._invite.isRecurring()) {
536 		var recur = new ZmRecurrence();
537 		recur.setRecurrenceRules(this._invite.getRecurrenceRules(), this._invite.getServerStartDate());
538 		subs.recur = recur.getBlurb();
539 	}

541 	// set changes to the invite

555 function(content, isHtml) {
556 	var sepIdx = content.indexOf(ZmItem.NOTES_SEPARATOR);
557 	if (sepIdx == -1) {
558 		return content;
559 	}
560 	return isHtml

562 		: (content.substring(sepIdx+ZmItem.NOTES_SEPARATOR.length));

601 ZmInviteMsgView.prototype._createInviteToolbar =
602 function() {
603 	var replyButtonIds = [
604 		ZmOperation.INVITE_REPLY_ACCEPT,
605 		ZmOperation.INVITE_REPLY_TENTATIVE,
606 		ZmOperation.INVITE_REPLY_DECLINE
607 	];
608 	var notifyOperationButtonIds = [
609 		ZmOperation.REPLY_ACCEPT_NOTIFY,
610 		ZmOperation.REPLY_TENTATIVE_NOTIFY,
611 		ZmOperation.REPLY_DECLINE_NOTIFY
612 	];
613 	var ignoreOperationButtonIds = [
614 		ZmOperation.REPLY_ACCEPT_IGNORE,
615 		ZmOperation.REPLY_TENTATIVE_IGNORE,
616 		ZmOperation.REPLY_DECLINE_IGNORE
617 	];
618 	var inviteOps = [
619 		ZmOperation.REPLY_ACCEPT,
620 		ZmOperation.REPLY_TENTATIVE,
621 		ZmOperation.REPLY_DECLINE,
622 		ZmOperation.PROPOSE_NEW_TIME
623 	];
624 
625 	var params = {
626 		parent: this.parent,
627 		buttons: inviteOps,
628 		posStyle: DwtControl.STATIC_STYLE,
629 		className: "ZmInviteToolBar",
630 		buttonClassName: "DwtToolbarButton",
631 		context: this.mode,
632 		toolbarType: ZmId.TB_INVITE
633 	};
634 	var tb = new ZmButtonToolBar(params);
635 
636 	var listener = new AjxListener(this, this._inviteToolBarListener);
637 	for (var i = 0; i < tb.opList.length; i++) {
638 		var id = tb.opList[i];
639 
640 		tb.addSelectionListener(id, listener);
641 
642 		if (id == ZmOperation.PROPOSE_NEW_TIME) { continue; }
643 
644 		var button = tb.getButton(id);
645 		var standardItems = [notifyOperationButtonIds[i], replyButtonIds[i], ignoreOperationButtonIds[i]];
646 		var menu = new ZmActionMenu({parent:button, menuItems:standardItems});
647 		standardItems = menu.opList;
648 		for (var j = 0; j < standardItems.length; j++) {
649 			var menuItem = menu.getItem(j);
650 			menuItem.addSelectionListener(listener);
651 		}
652 		button.setMenu(menu);
653 	}
654 
655 	this._respondOnBehalfLabel = tb.addFiller();
656 	tb.addFiller();
657 
658 	// folder picker
659 	var label = this._inviteMoveLabel = new DwtText({parent: tb, className: "DwtText InviteSelectLabel"});
660 	label.setSize(Dwt.DEFAULT, DwtControl.DEFAULT);
661 	label.setText(ZmMsg.calendarLabel);
662 	tb.addSpacer();
663 	this._inviteMoveSelect = new DwtSelect({parent:tb});
664 	tb.addSpacer();
665 
666 	return tb;
667 };
