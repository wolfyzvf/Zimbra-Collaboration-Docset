 17 /**
 18  * Creates a helper class for mouse event capturing.
 19  * @constructor
 20  * @class
 21  *
 22  * @author Ross Dargahi
 23  *
 24  * @param {hash}		params					a hash of parameters:
 25  * @param {Element}		params.targetObj		the target element
 26  * @param {string}		params.id				the ID for this capture instance.
 27  * @param {function}	params.mouseOverHdlr	the browser event handler
 28  * @param {function}	params.mouseDownHdlr	the browser event handler
 29  * @param {function}	params.mouseMoveHdlr	the browser event handler
 30  * @param {function}	params.mouseUpHdlr		the browser event handler
 31  * @param {function}	params.mouseOutHdlr		the browser event handler
 32  * @param {function}	params.mouseWheelHdlr	the browser event handler
 33  * @param {boolean}		params.hardCapture		if <code>true</code>, event propagation is halted at this element (IE only)
 34  *
 35  * @private
 36  */
 37 DwtMouseEventCapture = function(params) {
 38 
 39 	params = Dwt.getParams(arguments, DwtMouseEventCapture.PARAMS);
 40 
 41 	this.targetObj = params.targetObj;
 42 	this._id = params.id;
 43 	this._mouseOverHdlr = params.mouseOverHdlr || DwtMouseEventCapture.emptyHdlr;
 44 	this._mouseDownHdlr = params.mouseDownHdlr || DwtMouseEventCapture.emptyHdlr;
 45 	this._mouseMoveHdlr = params.mouseMoveHdlr || DwtMouseEventCapture.emptyHdlr;
 46 	this._mouseUpHdlr = params.mouseUpHdlr || DwtMouseEventCapture.emptyHdlr;
 47 	this._mouseOutHdlr = params.mouseOutHdlr || DwtMouseEventCapture.emptyHdlr;
 48 	this._mouseWheelHdlr = params.mouseWheelHdlr || DwtMouseEventCapture.emptyHdlr;
 49 	this._hardCapture = (params.hardCapture !== false);
 50 }
 51 
 52 DwtMouseEventCapture.PARAMS = ["targetObj", "id", "mouseOverHdlr", "mouseDownHdlr", "mouseMoveHdlr",
 53 							   "mouseUpHdlr", "mouseOutHdlr", "mouseWheelHdlr", "hardCapture"];
 54 
 55 DwtMouseEventCapture._capturing = false;
 56 
 57 DwtMouseEventCapture.getCaptureObj =
 58 function() {
 59 	return window._mouseEventCaptureObj;
 60 }
 61 
 62 DwtMouseEventCapture.getTargetObj =
 63 function() {
 64 	return window._mouseEventCaptureObj ? window._mouseEventCaptureObj.targetObj : null;
 65 }
 66 
 67 DwtMouseEventCapture.getId =
 68 function() {
 69 	return window._mouseEventCaptureObj ? window._mouseEventCaptureObj._id : null;
 70 }
 71 
 72 DwtMouseEventCapture.prototype.toString = 
 73 function() {
 74 	return "DwtMouseEventCapture";
 75 }
 76 
 77 DwtMouseEventCapture.prototype.capturing =
 78 function() {
 79 	return DwtMouseEventCapture._capturing;
 80 }
 81 
 82 DwtMouseEventCapture.prototype.capture =
 83 function() {
 84 
 85 	if (window._mouseEventCaptureObj) {
 86 		window._mouseEventCaptureObj.release();
 87 	}
 88 
 89 	if (document.body != null && document.body.addEventListener != null) {
 90 		document.body.addEventListener("mouseover", this._mouseOverHdlr, true);
 91 		document.body.addEventListener("mousedown", this._mouseDownHdlr, true);
 92 		document.body.addEventListener("mousemove", this._mouseMoveHdlr, true);
 93 		document.body.addEventListener("mouseup", this._mouseUpHdlr, true);
 94 		document.body.addEventListener("mouseout", this._mouseOutHdlr, true);
 95 		document.body.addEventListener("DOMMouseScroll", this._mouseWheelHdlr, true);
 96 	} else {
 97 		this._savedMouseOverHdlr = document.onmouseover;
 98 		this._savedMouseDownHdlr = document.onmousedown;
 99 		this._savedMouseMoveHdlr = document.onmousemove;
100 		this._savedMouseUpHdlr = document.onmouseup;
101 		this._savedMouseOutHdlr = document.onmouseout;
102 		this._savedMouseWheelHdlr = document.onmousewheel;
103 		document.onmouseover = this._mouseOverHdlr;
104 		document.onmousedown = this._mouseDownHdlr;
105 		document.onmousemove = this._mouseMoveHdlr;
106 		document.onmouseup = this._mouseUpHdlr;
107 		document.onmouseout = this._mouseOutHdlr;
108 		document.onmousewheel = this._mouseWheelHdlr;
109 	}
110 	if (this._hardCapture && document.body && document.body.setCapture) {
111 		document.body.setCapture();
112 	}
113 	window._mouseEventCaptureObj = this;
114 	DwtMouseEventCapture._capturing = true;
115 }
116 
117 
118 DwtMouseEventCapture.prototype.release = 
119 function() {
120 
121 	if (window._mouseEventCaptureObj == null) { return; }
122 
123 	var obj = window._shellCaptureObj;
124 	if (document.body && document.body.addEventListener) {
125 		document.body.removeEventListener("mouseover", this._mouseOverHdlr, true);
126 		document.body.removeEventListener("mousedown", this._mouseDownHdlr, true);
127 		document.body.removeEventListener("mousemove", this._mouseMoveHdlr, true);
128 		document.body.removeEventListener("mouseup", this._mouseUpHdlr, true);
129 		document.body.removeEventListener("mouseout", this._mouseOutHdlr, true);
130 		document.body.removeEventListener("DOMMouseScroll", this._mouseWheelHdlr, true);
131 	} else {
132 		document.onmouseover = this._savedMouseOverHdlr
133 		document.onmousedown = this._savedMouseDownHdlr;
134 		document.onmousemove = this._savedMouseMoveHdlr;
135 		document.onmouseup = this._savedMouseUpHdlr;
136 		document.onmouseout = this._savedMouseOutHdlr;
137 		document.onmousewheel = this._savedMouseWheelHdlr;
138 	}
139 	if (this._hardCapture && document.body && document.body.releaseCapture) {
140 		document.body.releaseCapture();
141 	}
142 	window._mouseEventCaptureObj = null;
143 	DwtMouseEventCapture._capturing = false;
144 }
145 
146 DwtMouseEventCapture.emptyHdlr =
147 function(ev) {
148 	var capObj = DwtMouseEventCapture.getCaptureObj();
149 	var mouseEv = DwtShell.mouseEvent;
150 	mouseEv.setFromDhtmlEvent(ev);	
151 	if (capObj._hardCapture) {
152 		mouseEv._stopPropagation = true;
153 		mouseEv._returnValue = false;
154 		mouseEv.setToDhtmlEvent(ev);
155 		return false;	
156 	} else {
157 		mouseEv._stopPropagation = false;
158 		mouseEv._returnValue = true;
159 		mouseEv.setToDhtmlEvent(ev);
160 		return true;
161 	}	
162 }
