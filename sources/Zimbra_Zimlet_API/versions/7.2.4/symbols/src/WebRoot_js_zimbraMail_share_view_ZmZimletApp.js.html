 22 /**
 23  * @class
 24  * This object represents a Zimlet Application.
 25  * 
 26  * @param	{String}	name		the application name
 27  * @param	{ZmZimlet}	zimlet		the zimlet
 28  * @param	{DwtControl}	container	the container
 29  * 
 30  * @extends	ZmApp
 31  */
 32 ZmZimletApp = function(name, zimlet, container) {
 33 	ZmApp.call(this, name, container);
 34 	this._zimlet = zimlet;
 35 };
 36 
 37 ZmZimletApp.prototype = new ZmApp;
 38 ZmZimletApp.prototype.constructor = ZmZimletApp;
 39 
 40 /**
 41  * Returns a string representation of the object.
 42  * 
 43  * @return		{String}		a string representation of the object
 44  */
 45 ZmZimletApp.prototype.toString =
 46 function() {
 47 	return "ZmZimletApp";
 48 };

 54 /**
 55  * Gets the controller.
 56  * 
 57  * @return	{ZmZimletAppController}		the controller
 58  */
 59 ZmZimletApp.prototype.getController =
 60 function() {
 61 	if (!this._controller) {
 62 		this._controller = new ZmZimletAppController(this.getName(), this._container, this);
 63 	}
 64 	return this._controller;
 65 };
 66 
 67 /**
 68  * Gets the toolbar.
 69  * 
 70  * @return	{ZmToolbar}		the toolbar
 71  */
 72 ZmZimletApp.prototype.getToolbar =
 73 function() {
 74 	return this.getController().getToolbar();
 75 };

 99 // ZmApp methods
100 
101 /**
102  * Launches the application.
103  * 
104  * @param	{Hash}			params			a hash of parameters
105  * @param	{AjxCallback}	callback		the callback
106  */
107 ZmZimletApp.prototype.launch =
108 function(params, callback) {
109 	var isNewViewShown = this.getController().show();
110 	if(!isNewViewShown) {
111 		return;
112 	}
113 	ZmApp.prototype.launch.call(this, params);
114 	if (this._zimlet.appLaunch) {
115 		this._zimlet.appLaunch(this.getName(), params);
116 	}
117 	if (callback) {
118 		callback.run();
119 	}
120 };
121 
122 /**
123  * Activates the application.
124  * 
125  * @param	{Boolean}	active	if <code>true</code>, active; <code>false</code> otherwise
126  * @param	{String}	viewId	the view id
127  */
128 ZmZimletApp.prototype.activate =
129 function(active, viewId) {
130 	ZmApp.prototype.activate.apply(this, arguments);
131 	if (this._zimlet.appActive) {
132 		this._zimlet.appActive(this.getName(), active);
133 	}
134 };
135 
136 /**
137  * Sets the overview tree to display overview content for this application.
138  *
139  * @param {Boolean}	reset		if <code>true</code>, clear the content first
140  */
141 ZmZimletApp.prototype.setOverviewPanelContent =
142 function(reset) {
143 	if (reset) {
144 		this._overviewPanelContent = null;
145 		this._overviewContainer = null;
146 	}
147 
148 	// only set overview panel content if not in full screen mode
149 	var avm = appCtxt.getAppViewMgr();
150 	if (!avm.isFullScreen()) {
151 		avm.setComponent(ZmAppViewMgr.C_TREE, this.getOverviewPanelContent());
152 	}
153 };
