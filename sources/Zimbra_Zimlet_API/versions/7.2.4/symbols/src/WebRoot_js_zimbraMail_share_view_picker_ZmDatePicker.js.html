 20 /**
 21  * Creates a date picker control.
 22  * @class
 23  * This class represents a date picker control.
 24  * 
 25  * @param		{DwtControl}	parent		the parent
 26  * 
 27  * @extends		ZmPicker
 28  * @see			ZmPicker.DATE
 29  */
 30 ZmDatePicker = function(parent) {
 31 	ZmPicker.call(this, parent, ZmPicker.DATE);
 32 };
 33 
 34 ZmDatePicker.prototype = new ZmPicker;
 35 ZmDatePicker.prototype.constructor = ZmDatePicker;
 36 
 37 ZmPicker.CTOR[ZmPicker.DATE] = ZmDatePicker;
 38 
 39 ZmDatePicker.SELECT_OPTIONS = [
 40 	{ label: ZmMsg.isAfter, 	value: "after", 	selected: false },
 41 	{ label: ZmMsg.isBefore, 	value: "before", 	selected: true  },
 42 	{ label: ZmMsg.isOn, 		value: "date", 		selected: false }];
 43 
 44 ZmDatePicker.prototype.toString = 
 45 function() {
 46 	return "ZmDatePicker";
 47 };
 48 
 49 ZmDatePicker.prototype._setupPicker =
 50 function(parent) {
 51 	var picker = new DwtComposite(parent);
 52 	var selectId = Dwt.getNextId();
 53 	var calId = Dwt.getNextId();
 54     
 55     var html = new Array(11);
 56     var i = 0;
 57     html[i++] = "<table cellpadding=3 cellspacing=0 border=0 style='width:100%;'>";
 58     html[i++] = "<tr align='center' valign='middle'><td align='right' nowrap>";
 59     html[i++] = ZmMsg.dateLabel;
 60     html[i++] = "</td><td align='left' id='";
 61     html[i++] = selectId;
 62     html[i++] = "'></td></tr>";
 63     html[i++] = "<tr valign='left'>";
 64     html[i++] = "<td nowrap align='center' colspan='2' id='";
 65     html[i++] = calId;
 66     html[i++] = "'></td></tr></table>";
 67 	picker.getHtmlElement().innerHTML = html.join("");
 68 
 69 	// create and add DwtSelect
 70 	this._select = new DwtSelect({parent:this});
 71 	this._select.addChangeListener(new AjxListener(this, this._dateChangeListener));
 72 	for (var i = 0; i < ZmDatePicker.SELECT_OPTIONS.length; i++) {
 73 		var option = ZmDatePicker.SELECT_OPTIONS[i];
 74 		this._select.addOption(option.label, option.selected, option.value);
 75 	}
 76 	this._select.reparentHtmlElement(selectId);
 77 
 78 	// create and add DwtCalendar
 79 	var firstDay = appCtxt.get(ZmSetting.CAL_FIRST_DAY_OF_WEEK) || 0;
 80 	var cal = this._cal = new DwtCalendar({parent:picker, firstDayOfWeek:firstDay});
 81 	cal.setDate(new Date());
 82 	cal.addSelectionListener(new AjxListener(this, this._calSelectionListener));
 83 	cal.reparentHtmlElement(calId);
 84 
 85 	this._updateQuery();
 86 };
 87 
 88 /**
 89  * Set date for second instance of date picker to 3 months back, select "after".
 90  * 
 91  */
 92 ZmDatePicker.prototype.secondDate =
 93 function() {
 94 	this._select.setSelected(0);
 95 	var date = new Date(this._cal.getDate().getTime());
 96 	AjxDateUtil.roll(date, AjxDateUtil.MONTH, -3);
 97 	this._cal.setDate(date);
 98 	this._updateQuery();
 99 };
100 
101 ZmDatePicker.prototype._updateQuery = 
102 function() {
103 	var d = this._cal.getDate();
104 	if (d) {
105       var query;
106       if(appCtxt.getAppController().getAppViewMgr().getCurrentViewId()==ZmId.VIEW_CAL && this._select.getValue()=="date"){
107         query = [this._select.getValue(),":",d.getTime()].join("");
108       }
109       else{
110         var formatter = AjxDateFormat.getDateInstance(AjxDateFormat.SHORT);
111         var date = formatter.format(d);
112         query = [this._select.getValue(),':"',date,'"'].join("");
113       }
114 
115       this.setQuery(query);
116 
117 	} else {
118 		this.setQuery("");
119 	}
120 	this.execute();
121 };
122 
123 ZmDatePicker.prototype._calSelectionListener =
124 function(ev) {
125 	this._updateQuery();
126 };
127 
128 ZmDatePicker.prototype._dateChangeListener = 
129 function(ev) {
130 	this._updateQuery();
131 };
