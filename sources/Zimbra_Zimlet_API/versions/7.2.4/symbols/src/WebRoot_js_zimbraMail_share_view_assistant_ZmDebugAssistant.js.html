 20 /**
 21  * Creates the debug assistant.
 22  * @class
 23  * This class represents a debug assistant.
 24  * 
 25  * @extends		ZmAssistant
 26  */
 27 ZmDebugAssistant = function() {
 28 	ZmAssistant.call(this, "Debugging Info", ".debug");
 29 	this._dbg = window.DBG;	// trick to fool minimizer regex
 30 };
 31 
 32 ZmDebugAssistant.prototype = new ZmAssistant();
 33 ZmDebugAssistant.prototype.constructor = ZmDebugAssistant;
 34 
 35 ZmDebugAssistant.prototype.okHandler =
 36 function() {
 37 	if (this._newLevel >= 0) {
 38 		this._dbg.setDebugLevel(this._newLevel);
 39 	}
 40 	return true;
 41 };
 42 
 43 ZmDebugAssistant.__RE_handleNumber = new RegExp(["^",ZmAssistant.SPACES,"([0123])",ZmAssistant.SPACES,"$"].join(""));
 44 
 45 ZmDebugAssistant.prototype.handle =
 46 function(dialog, verb, args) {
 47 	var	match = args.match(ZmDebugAssistant.__RE_handleNumber);
 48 	this._newLevel = match ? parseInt(match[1]) : -1;
 49 	var set = this._newLevel >= 0;
 50 	dialog._setOkButton(AjxMsg.ok, true, set);
 51 	this._setField("Current Level", this._dbg.getDebugLevel()+"", false, true);
 52 	this._setField("New Level", !set ? "(0 = none, 1 = minimal, 2 = moderate, 3 = anything goes)" : this._newLevel+"", !set, true);
 53 };
