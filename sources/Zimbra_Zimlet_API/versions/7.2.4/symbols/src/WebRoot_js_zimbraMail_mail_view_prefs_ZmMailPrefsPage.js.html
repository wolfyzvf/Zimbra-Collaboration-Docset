 34 ZmMailPrefsPage.prototype.showMe =
 35 function() {
 36 	ZmPreferencesPage.prototype.showMe.call(this);
 37     if(appCtxt.isOffline){
 38         if(this._initializedAcctId != appCtxt.getActiveAccount().id) {
 39             this._initialized = false;
 40             this._initializedAcctId = appCtxt.getActiveAccount().id;
 41         }
 42     }
 43 	if (!this._initialized) {
 44 		this._initialized = true;
 45 		if (this._blackListControl && this._whiteListControl) {
 46 			var soapDoc = AjxSoapDoc.create("GetWhiteBlackListRequest", "urn:zimbraAccount");
 47 			var callback = new AjxCallback(this, this._handleResponseLoadWhiteBlackList);
 48 			appCtxt.getRequestMgr().sendRequest({soapDoc:soapDoc, asyncMode:true, callback:callback});
 49 		}
 50 	}
 51 };

 84 ZmMailPrefsPage.prototype.addCommand =
 85 function(batchCmd) {
 86 	if (this.isWhiteBlackListDirty()) {
 87 		var soapDoc = AjxSoapDoc.create("ModifyWhiteBlackListRequest", "urn:zimbraAccount");
 88 		this._blackListControl.setSoapContent(soapDoc, "blackList");
 89 		this._whiteListControl.setSoapContent(soapDoc, "whiteList");

102 ZmMailPrefsPage.prototype._setPopDownloadSinceControls =
103 function() {
104 	var popDownloadSinceValue = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE_VALUE);
105 	if (popDownloadSinceValue) {
106 		var value = appCtxt.get(ZmSetting.POP_DOWNLOAD_SINCE);
107 		if (value) {
108 			var date = AjxDateFormat.parse("yyyyMMddHHmmss'Z'", value);
109 			date.setMinutes(date.getMinutes() - date.getTimezoneOffset());
110 			value = date;
111 		}
112 		var pattern = value ? ZmMsg.externalAccessPopCurrentValue : ZmMsg.externalAccessPopNotSet;
113 		popDownloadSinceValue.setText(AjxMessageFormat.format(pattern, value));
114 	}
115 
116 	var popDownloadSince = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE);
117 	if (popDownloadSince) {
118 		popDownloadSince.setSelectedValue(appCtxt.get(ZmSetting.POP_DOWNLOAD_SINCE));
119 	}

147 			? (AjxDateUtil.simpleComputeDateStr(this._formatter.parse(this._endDateVal.value)))

152 		var dateFieldListener = AjxCallback.simpleClosure(this._dateFieldListener, this);

159

162 	}
163 
164 	var cbox = this.getFormObject(ZmSetting.VACATION_MSG_ENABLED);
165 
166 	if (cbox) {
167 		this._handleEnableVacationMsg(cbox);
168 	}
169

171 	var downloadSinceCbox = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE_ENABLED);
172 	if (downloadSinceCbox) {
173 		var downloadSince = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE);

175 			var enabled = downloadSince.getValue() != "";

177 			downloadSince.setEnabled(enabled);

197 ZmMailPrefsPage.prototype._dateButtonListener =
198 function(ev) {

209 ZmMailPrefsPage.prototype._fixAndGetValidDateFromField =
210 function(field) {
211 	var d = AjxDateUtil.simpleParseDateStr(field.value);
212 	if (!d || isNaN(d)) {
213 		d = new Date();
214 		field.value = AjxDateUtil.simpleComputeDateStr(d);
215 	}
216 	return d;
217 };

219 ZmMailPrefsPage.prototype._dateCalSelectionListener =
220 function(ev) {
221 	var parentButton = ev.item.parent.parent;
222 
223 	var newDate = AjxDateUtil.simpleComputeDateStr(ev.detail);
224 
225 	if (parentButton == this._startDateButton) {
226 		this._startDateField.value = newDate;
227 	} else {
228 		if (ev.detail < new Date()) { return; }
229 		this._endDateField.value = newDate;
230 	}
231 
232 	var sd = this._fixAndGetValidDateFromField(this._startDateField);
233 	var ed = this._fixAndGetValidDateFromField(this._endDateField);
234 	
235 	this._fixDates(sd, ed, parentButton == this._endDateButton);
236 
237 	if (this._startDateCheckbox.isSelected()) {
238 		this._startDateVal.value = this._formatter.format(AjxDateUtil.simpleParseDateStr(this._startDateField.value));
239 	}
240 	if (this._endDateCheckbox.isSelected()) {
241 		this._endDateVal.value = this._formatter.format(AjxDateUtil.simpleParseDateStr(this._endDateField.value));
242 	}
243 };
244 
245 ZmMailPrefsPage.prototype._dateFieldListener =
246 function(ev) {
247 	var sd = this._fixAndGetValidDateFromField(this._startDateField);
248 	var ed = this._fixAndGetValidDateFromField(this._endDateField);
249 	this._fixDates(sd, ed, DwtUiEvent.getTarget(ev) == this._endDateField);
250 	this._startDateVal.value = this._formatter.format(AjxDateUtil.simpleParseDateStr(this._startDateField.value));
251 	this._endDateVal.value = this._formatter.format(AjxDateUtil.simpleParseDateStr(this._endDateField.value));
252 };
253 
254 /* Fixes the field values so that end date always is later than or equal to start date
255  * @param startDate	{Date}	The value of the start date field or calendar selection
256  * @param endDate	{Date}	The value of the end date field or calendar selection
257  * @param modifyStart {boolean}	Whether to modify the start date or end date when dates overlap. true for start date, false for end date
258 */
259 
260 ZmMailPrefsPage.prototype._fixDates =
261 function(startDate, endDate, modifyStart) {
262 	if (startDate > endDate) {
263 		// Mismatch; start date is after end date
264 		if (modifyStart) {
265 			// Set them to be equal
266 			this._startDateField.value = AjxDateUtil.simpleComputeDateStr(endDate);
267 		} else {
268 			// Put endDate a bit into the future
269 			this._endDateField.value = AjxDateUtil.simpleComputeDateStr(AjxDateUtil.getDateForNextDay(startDate,AjxDateUtil.FRIDAY));
270 		}
271 	}
272 };

325 ZmMailPrefsPage.prototype._handleEnableVacationMsg =
326 function(cbox, id, evt) {
327 	var textarea = this.getFormObject(ZmSetting.VACATION_MSG);
328 	if (textarea) {
329 		if (id == ZmSetting.START_DATE_ENABLED) {
330 			this._setEnabledStartDate(cbox.isSelected());
331 		} else if (id == ZmSetting.END_DATE_ENABLED) {
332 			this._setEnabledEndDate(cbox.isSelected());
333 		} else {
334 			var enabled = cbox.isSelected();
335 			textarea.setEnabled(enabled);
336 
337 			this._startDateCheckbox.setEnabled(enabled);
338 			this._endDateCheckbox.setEnabled(enabled);
339 
340 			var val = !this._startDateVal.value ? false : true;
341 			this._startDateCheckbox.setSelected(val);
342 
343 			val = !this._endDateVal.value ? false : true;
344 			this._endDateCheckbox.setSelected(val);
345 
346 			this._setEnabledStartDate(enabled && this._startDateCheckbox.isSelected());
347 			this._setEnabledEndDate(enabled && this._endDateCheckbox.isSelected());
348 		}
349 	}
350 };

376 ZmMailPrefsPage.prototype._postSave =
377 function() {
378     var form = this.getFormObject(ZmSetting.POLLING_INTERVAL);
379     if (form && form.getSelectedOption() && form.getSelectedOption().getDisplayValue() == ZmMsg.pollInstant && appCtxt.get(ZmSetting.INSTANT_NOTIFY)
380             && !appCtxt.getAppController().getInstantNotify()){
381         //turn on instant notify if not already on
382         appCtxt.getAppController().setInstantNotify(true);
383     } else{
384         //turn instant notify off if it's on
385         if (appCtxt.getAppController().getInstantNotify())
386             appCtxt.getAppController().setInstantNotify(false);
387     }
388 };

400 ZmWhiteBlackList = function(parent, id, templateId) {
401 	DwtComposite.call(this, {parent:parent});
402 
403 	this._settingId = id;
404 	this._tabGroup = new DwtTabGroup(this._htmlElId);
405     switch(id) {
406         case ZmSetting.MAIL_BLACKLIST:
407             this._max = appCtxt.get(ZmSetting.MAIL_BLACKLIST_MAX_NUM_ENTRIES);
408             break;
409         case ZmSetting.MAIL_WHITELIST:
410             this._max = appCtxt.get(ZmSetting.MAIL_WHITELIST_MAX_NUM_ENTRIES);
411             break;
412         case ZmSetting.TRUSTED_ADDR_LIST:
413             this._max = appCtxt.get(ZmSetting.TRUSTED_ADDR_LIST_MAX_NUM_ENTRIES);
414             break;
415     }
416 	this._setContent(templateId);
417 
418 	this._list = [];
419 	this._add = {};
420 	this._remove = {};
421 };

454 ZmWhiteBlackList.prototype.loadFromJson =
455 function(data) {
456 	if (data) {
457         for (var i = 0; i < data.length; i++) {
458             var content = AjxUtil.isSpecified(data[i]._content) ? data[i]._content : data[i];
459             if(content){
460 			    var item = this._addEmail(content);
461 			    this._list.push(item);
462             }
463 		}
464 	}
465 	this.updateNumUsed();
466 };

468 ZmWhiteBlackList.prototype.setSoapContent =
469 function(soapDoc, method) {
470 	if (!this.isDirty()) { return; }
471 
472 	var methodEl = soapDoc.set(method);
473 
474 	for (var i in this._add) {
475 		var addrEl = soapDoc.set("addr", i, methodEl);
476 		addrEl.setAttribute("op", "+");
477 	}
478 
479 	for (var i in this._remove) {
480 		var addrEl = soapDoc.set("addr", i, methodEl);
481 		addrEl.setAttribute("op", "-");
482 	}
483 };

561 ZmWhiteBlackList.prototype._addListener =
562 function() {
563 	if (this._listView.size() >= this._max) {
564 		var dialog = appCtxt.getMsgDialog();
565 		dialog.setMessage(ZmMsg.errorWhiteBlackListExceeded);
566 		dialog.popup();
567 		return;
568 	}
569 
570 	var val,
571         items = AjxStringUtil.trim(this._inputEl.getValue(), true);
572 	if (items.length) {
573         items = AjxStringUtil.split(items, [',', ';', ' ']);
574         for(var i=0; i<items.length; i++) {
575             val = items[i];
576             if(val) {
577                 this._addEmail(AjxStringUtil.htmlEncode(val));
578                 if (!this._add[val]) {
579                     this._add[val] = true;
580                 }
581             }
582         }
583 		this._inputEl.setValue("", true);
584 		this._inputEl.blur();
585 		this._inputEl.focus();
586 
587 		this.updateNumUsed();
588 	}
589 };

591 ZmWhiteBlackList.prototype._removeListener =
592 function() {
593 	var items = this._listView.getSelection();
594 	for (var i = 0; i < items.length; i++) {
595 		var item = items[i];
596 		this._listView.removeItem(item, true);
597 		var addr = item.toString();
598 		if (this._add[addr]) {
599 			delete this._add[addr];
600 		} else {
601 			this._remove[addr] = true;
602 		}
603 	}
604 
605 	this.updateNumUsed();
606 };
