 20 /**
 21  * Creates a basic picker control.
 22  * @class
 23  * This class represents a basic picker control.
 24  * 
 25  * @param		{DwtControl}	parent		the parent
 26  * 
 27  * @extends		ZmPicker
 28  * @see			ZmPicker.BASIC
 29  */
 30 ZmBasicPicker = function(parent) {
 31 
 32 	ZmPicker.call(this, parent, ZmPicker.BASIC);
 33 	
 34 	var settings = appCtxt.getSettings();
 35 	var listener = new AjxListener(this, this._settingChangeListener);
 36 	settings.getSetting(ZmSetting.SEARCH_INCLUDES_SPAM).addChangeListener(listener);
 37 	settings.getSetting(ZmSetting.SEARCH_INCLUDES_TRASH).addChangeListener(listener);
 38 };
 39 
 40 ZmBasicPicker.prototype = new ZmPicker;
 41 ZmBasicPicker.prototype.constructor = ZmBasicPicker;
 42 
 43 ZmPicker.CTOR[ZmPicker.BASIC] = ZmBasicPicker;
 44 
 45 ZmBasicPicker.prototype.toString = 
 46 function() {
 47 	return "ZmBasicPicker";
 48 };

 50 ZmBasicPicker.prototype._makeRow =
 51 function(text, id) {
 52     var size = 17;
 53     var html = [];
 54     var i = 0;
 55     html[i++] = "<tr valign='middle'>";
 56     html[i++] = "<td align='right' nowrap>";
 57 	html[i++] = AjxMessageFormat.format(ZmMsg.makeLabel, text);
 58 	html[i++] = "</td>";
 59     html[i++] = "<td align='left' nowrap><input type='text' autocomplete='off' nowrap size='";
 60 	html[i++] = size;
 61 	html[i++] = "' id='";
 62 	html[i++] = id;
 63 	html[i++] = "'/></td>";
 64     html[i++] = "</tr>";
 65 
 66 	return html.join("");
 67 };

 69 // TODO: if we really wanted, we could add a prefs listener to update the "also search" checkboxes
 70 ZmBasicPicker.prototype._setupPicker =
 71 function(parent) {
 72     var picker = new DwtComposite(parent);
 73 
 74     var fromId = Dwt.getNextId();
 75     var toId = Dwt.getNextId();
 76     var subjectId = Dwt.getNextId();
 77     var contentId = Dwt.getNextId();
 78     var inTrashId = Dwt.getNextId();
 79     var inSpamId, checked;
 80     
 81 	var html = [];
 82 	var i = 0;
 83 	html[i++] = "<table cellpadding='5' cellspacing='0' border='0' width='100%'>";
 84 	html[i++] = this._makeRow(ZmMsg.from, fromId);
 85 	html[i++] = this._makeRow(ZmMsg.toCc, toId);
 86 	html[i++] = this._makeRow(ZmMsg.subject, subjectId);
 87 	html[i++] = this._makeRow(ZmMsg.content, contentId);
 88 	html[i++] = "</table>";
 89     html[i++] = "<div style='overflow:hidden; padding-left: 20px;'>";
 90 	if (appCtxt.get(ZmSetting.SPAM_ENABLED)) {
 91 		inSpamId = Dwt.getNextId();
 92 		checked = appCtxt.get(ZmSetting.SEARCH_INCLUDES_SPAM) ? " checked" : "";
 93 		/*html[i++] = "<tr valign='middle'>";*/
 94 		html[i++] = "<span><input type='checkbox'";
 95 		html[i++] = checked;
 96 		html[i++] = " id='";
 97 		html[i++] = inSpamId;
 98 		html[i++] = "' /></span>";
 99 		html[i++] = "<span>";
100 		html[i++] = ZmMsg.includeJunk;
101 		html[i++] = "</span>";
102 		html[i++] = "<br>";
103 
104     }
105 	checked = appCtxt.get(ZmSetting.SEARCH_INCLUDES_TRASH) ? " checked" : "";
106 	html[i++] = "<span><input type='checkbox'";
107 	html[i++] = checked;
108 	html[i++] = " id='";
109 	html[i++] = inTrashId;
110 	html[i++] = "' /></span>";
111 	html[i++] = "<span>";
112 	html[i++] = ZmMsg.includeTrash;
113 	html[i++] = "</span>";
114 	html[i++] = "</div>";
115 	picker.getHtmlElement().innerHTML = html.join("");
116 
117 	this._from = this._setupField(fromId);
118 	this._to = this._setupField(toId);
119 	this._subject= this._setupField(subjectId);
120 	this._content = this._setupField(contentId);
121 	if (appCtxt.get(ZmSetting.SPAM_ENABLED)) {
122 		this._inSpam = this._setupCheckbox(inSpamId);
123 	}
124 	this._inTrash = this._setupCheckbox(inTrashId);
125 };

171 ZmBasicPicker.prototype._setupField = 
172 function(id) {
173 	var f = document.getElementById(id);
174 	Dwt.setHandler(f, DwtEvent.ONKEYUP, ZmBasicPicker._onChange);
175 	Dwt.associateElementWithObject(f, this);
176 	return f;
177 };
178 
179 ZmBasicPicker.prototype._setupCheckbox = 
180 function(id) {
181 	var f = document.getElementById(id);
182 	Dwt.setHandler(f, DwtEvent.ONCLICK, ZmBasicPicker._onChange);
183 	Dwt.associateElementWithObject(f, this);
184 	return f;
185 };
186 
187 ZmBasicPicker._onChange =
188 function(ev) {
189 	var element = DwtUiEvent.getTarget(ev);
190 	var picker = Dwt.getObjectFromElement(element);
191 
192 	var charCode = DwtKeyEvent.getCharCode(ev);
193 	if (charCode == 13 || charCode == 3 || charCode == 9) {
194 		picker.execute();
195 	    return false;
196 	} else {
197 		picker._updateQuery();
198 		return true;
199 	}
200 };

202 ZmBasicPicker.prototype._updateQuery = 
203 function() {
204 	var query1 = [];
205 	var from = AjxStringUtil.trim(this._from.value, true);
206 	if (from.length)
207 		query1.push("from:(" + from + ")");
208 	var to = AjxStringUtil.trim(this._to.value, true);
209 	if (to.length)
210 		query1.push("(to:(" + to + ")" + " OR cc:(" + to + "))");
211 	var subject = AjxStringUtil.trim(this._subject.value, true);
212 	if (subject.length)
213 		query1.push("subject:(" + subject + ")");
214 	var content = AjxStringUtil.trim(this._content.value, true);
215 	if (content.length)
216 		query1.push("content:(" + content + ")");
217 	
218 	// Sort out "Check Trash/Spam" pref vs checkbox
219 	var query2 = [];
220 	var checkSpamPref = appCtxt.get(ZmSetting.SEARCH_INCLUDES_SPAM);
221 	var checkTrashPref = appCtxt.get(ZmSetting.SEARCH_INCLUDES_TRASH);
222 	var checkSpamCheckbox = (this._inSpam && this._inSpam.checked);
223 	var checkTrashCheckbox = (this._inTrash && this._inTrash.checked);
224 	
225 	if ((!checkSpamPref && checkSpamCheckbox) && (!checkTrashPref && checkTrashCheckbox)) {
226 		query2.push("is:anywhere");
227 	} else {
228 		if (!checkSpamPref && checkSpamCheckbox)
229 			query2.push("is:anywhere not in:trash");
230 		else if (checkTrashPref && !checkTrashCheckbox)
231 			query2.push("not in:trash");
232 		if (!checkTrashPref && checkTrashCheckbox)
233 			query2.push("is:anywhere not in:junk");
234 		else if (checkSpamPref && !checkSpamCheckbox)
235 			query2.push("not in:junk");
236 	}
237 	
238 	var cbQuery = query2.length ? query2.join(" ") : null;
239 
240 	// if the "search Trash/Spam" checkboxes changed, run the query
241 	var cbChange = (query1.length && ((this._cbQuery || cbQuery) && (this._cbQuery != cbQuery)));
242 	var query = query1.concat(query2);
243 	this.setQuery(query.length ? query.join(" ") : "");
244 	if (cbChange) {
245 		this.execute();
246 	}
247 	this._cbQuery = cbQuery;
248 };
249 
250 ZmBasicPicker.prototype._settingChangeListener =
251 function(ev) {
252 	if (ev.type != ZmEvent.S_SETTING) return;
253 	
254 	var setting = ev.source;
255 	if (setting.id == ZmSetting.SEARCH_INCLUDES_SPAM) {
256 		this._inSpam.checked = appCtxt.get(ZmSetting.SEARCH_INCLUDES_SPAM);
257 		this._updateQuery();
258 	} else	if (setting.id == ZmSetting.SEARCH_INCLUDES_TRASH) {
259 		this._inTrash.checked = appCtxt.get(ZmSetting.SEARCH_INCLUDES_TRASH);
260 		this._updateQuery();
261 	}
262 };
