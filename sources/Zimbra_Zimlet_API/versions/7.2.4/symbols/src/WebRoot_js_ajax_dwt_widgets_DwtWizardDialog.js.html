 17 /**
 18  * Creates a new wizard dialog.
 19  * @constructor
 20  * @class
 21  * This class represents a reusable wizard dialog.
 22  * 
 23  * @param {DwtControl}	parent		the parent control
 24  * @param {string}	className		the CSS class name
 25  * @param {string}	title	the dialog title
 26  * @param {number}	w 		the content area width
 27  * @param {number}	h 		the content area height
 28  * 
 29  * @extends		DwtDialog
 30  * 
 31  * @private
 32  */
 33 DwtWizardDialog = function(parent, className, title, w, h) {
 34 	if (arguments.length == 0) { return; }
 35 	className = className || "DwtDialog";
 36 	
 37 	var nextButton = new DwtDialog_ButtonDescriptor(DwtWizardDialog.NEXT_BUTTON, AjxMsg._next, DwtDialog.ALIGN_RIGHT, new AjxCallback(this, this.goNext));
 38 	var prevButton = new DwtDialog_ButtonDescriptor(DwtWizardDialog.PREV_BUTTON, AjxMsg._prev, DwtDialog.ALIGN_RIGHT, new AjxCallback(this, this.goPrev));
 39 	var finishButton = new DwtDialog_ButtonDescriptor(DwtWizardDialog.FINISH_BUTTON, AjxMsg._finish, DwtDialog.ALIGN_RIGHT, new AjxCallback(this, this.finishWizard));
 40 	DwtDialog.call(this, {parent:parent, className:className, standardButtons:[DwtDialog.CANCEL_BUTTON],
 41 						  extraButtons:[prevButton, nextButton, finishButton]});
 42 
 43 	if (!w) {
 44 		this._contentW = "80ex";
 45 	} else {
 46 		this._contentW = w;
 47 	}
 48 	
 49 	if(!h) {
 50 		this._contentH = "100ex";
 51 	} else {
 52 		this._contentH = h;
 53 	}
 54 	
 55 	this._pages = new Array(); 
 56 	this._pageIx = 1;
 57 	this._currentPage = 1;
 58 	this._progressDiv = document.createElement("div");
 59 	this._progressDiv.style.position = DwtControl.STATIC_STYLE;
 60 	this._pageDiv = document.createElement("div");
 61 	this._pageDiv.className = "DwtWizardDialogPageDiv";
 62 	this._pageDiv.style.width = this._contentW;
 63 	this._pageDiv.style.height = this._contentH;
 64 	this._progressBar = new DwtWizProgressBar(this);
 65 	this._createContentHtml();
 66 	this.setTitle(title);
 67 }

 94 /*
 95  * Makes the dialog visible, and places it. Everything under the dialog will become veiled
 96  * if we are modal.
 97  *
 98  * @param loc	the desired location
 99  *//*
100 DwtWizardDialog.prototype.popup =
101 function(loc) {
102 
103 	var thisZ = this._zIndex;
104 	if (this._mode == DwtDialog.MODAL) {
105 		// place veil under this dialog
106 		var dialogZ = this._shell._veilOverlay.dialogZ;
107 		var currentDialogZ = null;
108 		var veilZ;
109 		if (dialogZ.length)
110 			currentDialogZ = dialogZ[dialogZ.length - 1];
111 		if (currentDialogZ) {
112 			thisZ = currentDialogZ + 2;
113 			veilZ = currentDialogZ + 1;
114 		} else {
115 			thisZ = this._zIndex;
116 			veilZ = Dwt.Z_VEIL;
117 		}
118 		this._shell._veilOverlay.veilZ.push(veilZ);
119 		this._shell._veilOverlay.dialogZ.push(thisZ);
120 		Dwt.setZIndex(this._shell._veilOverlay, veilZ);
121 	}
122 	loc = this._loc = loc || this._loc; // use whichever has a value, local has precedence
123 	var sizeShell = this._shell.getSize();
124 	var sizeThis = this.getSize();
125 	var x, y;
126 	if (loc == null) {
127 		// if no location, go for the middle
128 		x = Math.round((sizeShell.x - sizeThis.x) / 2);
129 		y = Math.round((sizeShell.y - sizeThis.y) / 2);
130 	} else {
131 		x = loc.x;
132 		y = loc.y;
133 	}
134 	// try to stay within shell boundaries
135 	if ((x + sizeThis.x) > sizeShell.x)
136 		x = sizeShell.x - sizeThis.x;
137 	if ((y + sizeThis.y) > sizeShell.y)
138 		y = sizeShell.y - sizeThis.y;
139 	this.setLocation(x, y);
140 	
141 	this.setZIndex(thisZ);
142 }
143 */

145 /*
146 * @param pageKey - key to the page to be shown. 
147 * pageKey is the value returned from @link DwtWizardDialog.prototype.addPage method
148 * This method is called by 
149 *	@link DwtWizardPage.prototype.switchToNextPage 
150 *	and 
151 *	@link DwtWizardPage.prototype.switchToPrevPage
152 */
153 DwtWizardDialog.prototype.goPage = 
154 function(pageKey) {
155 	if(this._pages && this._pages[pageKey]) {
156 		this._currentPage = pageKey;
157 		this._showPage(pageKey);
158 		this._progressBar.showStep(pageKey);
159 		this._pages[pageKey].setActive();
160 	}
161 }
162 
163 DwtWizardDialog.prototype.goNext = 
164 function() {
165 	var nextPageKey = this._currentPage + 1;
166 	if(this._pages && this._pages[this._currentPage]) {
167 		this._pages[this._currentPage].switchToNextPage(nextPageKey);
168 	}
169 
170 }
171 
172 DwtWizardDialog.prototype.goPrev = 
173 function() {
174 	var prevPageKey = this._currentPage - 1;
175 	if(this._pages && this._pages[this._currentPage]) {
176 		this._pages[this._currentPage].switchToPrevPage(prevPageKey);
177 	}
178 }

180 DwtWizardDialog.prototype.finishWizard = 
181 function() {
182 	this.popdown();	
183 }

185 /**
186  * 
187  * @param pageKey  the key for the page, returned from {@link #addPage}
188  * @return {DwtPropertyPage}	the view tab
189  * 
190  * @private
191  */
192 DwtWizardDialog.prototype.getPage =
193 function (pageKey) {
194 	if(this._pages && this._pages[pageKey])
195 		return this._pages[pageKey];
196 	else
197 		return null;
198 }

200 /**
201  * @param {DwtPropertyPage}		wizPage 	the wizard page
202  * @return - the key for the added page. This key can be used to retrieve the tab using {@link #getPage}
203  * 
204  * @private
205 */
206 DwtWizardDialog.prototype.addPage =
207 function (wizPage, stepTitle) {
208 	var pageKey = this._pageIx++;	
209 	this._pages[pageKey] = wizPage;
210 	//add a step to the progress bar
211 /*	if(stepTitle == null)
212  		stepTitle = pageKey;*/
213 	this._progressBar.addStep(pageKey, stepTitle);
214 	//add the page 
215 	this._pageDiv.appendChild(this._pages[pageKey].getHtmlElement());
216 	
217 	if(pageKey==1) //show the first tab 
218 		this._pages[pageKey].setZIndex(DwtWizardDialog.Z_ACTIVE_PAGE); 
219 	else {
220 		//hide all the other tabs
221 		this._pages[pageKey].setZIndex(DwtWizardDialog.Z_HIDDEN_PAGE); 
222 		Dwt.setVisible(this._pages[pageKey].getHtmlElement(), false);
223 	}
224 	return pageKey;
225 }

234 DwtWizardDialog.prototype._createContentHtml =
235 function () {
236 
237 	this._table = document.createElement("table");
238 	this._table.border = 0;
239 	this._table.width = this._contentW;
240 	this._table.cellPadding = this._table.cellSpacing = 0;
241 
242 	this._table.backgroundColor = DwtCssStyle.getProperty(this.parent.getHtmlElement(), "background-color");
243 	
244 	var row1; //_progressBar
245 	var col1;
246 	row1 = this._table.insertRow(0);
247 	row1.align = "left";
248 	row1.vAlign = "middle";
249 	
250 	col1 = row1.insertCell(row1.cells.length);
251 	col1.align = "left";
252 	col1.vAlign = "middle";
253 	col1.noWrap = true;	
254 	col1.width="100%";
255 	col1.className="DwtTabTable";
256 	col1.appendChild(this._progressBar.getHtmlElement());
257 
258 	var rowSep;//separator
259 	var colSep;
260 	rowSep = this._table.insertRow(1);
261 	rowSep.align = "center";
262 	rowSep.vAlign = "middle";
263 	
264 	colSep = rowSep.insertCell(rowSep.cells.length);
265 	colSep.align = "left";
266 	colSep.vAlign = "middle";
267 	colSep.noWrap = true;	
268 	colSep.style.width = this._contentW;
269 	var sepDiv = document.createElement("div");
270 	sepDiv.className = "horizSep";
271 	sepDiv.style.width = this._contentW;
272 	sepDiv.style.height = "5px";
273 	colSep.appendChild(sepDiv);
274 	
275 	var row2; //page
276 	var col2;
277 	row2 = this._table.insertRow(2);
278 	row2.align = "left";
279 	row2.vAlign = "middle";
280 	
281 	col2 = row2.insertCell(row2.cells.length);
282 	col2.align = "left";
283 	col2.vAlign = "middle";
284 	col2.noWrap = true;	
285 	col2.width = this._contentW;
286 	col2.appendChild(this._pageDiv);
287 
288 	this._getContentDiv().appendChild(this._table);
289 }

303 DwtWizardDialog.prototype._showPage = 
304 function(pageKey) {
305 	if(this._pages && this._pages[pageKey]) {
306 		//hide all the tabs
307 		this._hideAllPages();
308 		//make this tab visible
309 		this._pages[pageKey].showMe();
310 		//this._pages[pageKey].setZIndex(DwtWizardDialog.Z_ACTIVE_PAGE);
311 		Dwt.setVisible(this._pages[pageKey].getHtmlElement(), true);
312 	}
313 }
314 
315 DwtWizardDialog.prototype._hideAllPages = 
316 function() {
317 	if(this._pages && this._pages.length) {
318 		for(var curPageKey in this._pages) {
319 			if(this._pages[curPageKey]) {
320 				this._pages[curPageKey].hideMe();
321 				//this._pages[curPageKey].setZIndex(DwtWizardDialog.Z_HIDDEN_PAGE);
322 				Dwt.setVisible(this._pages[curPageKey].getHtmlElement(), false);
323 			}	
324 		}
325 	}
326 }

396 /**
397  * @class DwtWizProgressBar
398  * @constructor
399  * 
400  * @param parent
401  * 
402  * @private
403  */
404 DwtWizProgressBar = function(parent) {
405 	if (arguments.length == 0) return;
406 	DwtComposite.call(this, {parent:parent, className:"DwtWizProgressBar", posStyle:DwtControl.STATIC_STYLE});
407 	this._table = document.createElement("table");
408 	this._table.border = 0;
409 	this._table.cellPadding = 0;
410 	this._table.cellSpacing = 0;
411 	this.getHtmlElement().appendChild(this._table);
412 	this._table.backgroundColor = DwtCssStyle.getProperty(this.parent.getHtmlElement(), "background-color");
413 	this._stepsNumber = 0; //number of steps
414 	this._steps = new Array();
415 	this._lblHeader = new DwtStepLabel(this);
416 	this._lblHeader.setText("Step 0 of 0");
417 	this._lblHeader.setActive(true);
418 }

429 /**
430 * @param stepKey
431 **/
432 DwtWizProgressBar.prototype.showStep = 
433 function(stepKey) {
434 	var szLabelTxt = "Step " + stepKey + " of " + this._stepsNumber;
435 	if(this._steps[stepKey]) {
436 		szLabelTxt = szLabelTxt + ": " + this._steps[stepKey];
437 	}
438 	this._lblHeader.setText(szLabelTxt);
439 }

441 /**
442 * @param stepKey
443 * @param stepNumber
444 **/
445 DwtWizProgressBar.prototype.addStep =
446 function (stepKey, stepTitle) {
447 	this._steps[stepKey] = stepTitle;
448 	return (++this._stepsNumber);
449 }

466 }
