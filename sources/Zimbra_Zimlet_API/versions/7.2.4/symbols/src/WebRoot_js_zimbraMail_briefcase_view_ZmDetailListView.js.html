 21 /**
 22  * Creates the briefcase detail list view.
 23  * @class
 24  * This class represents the briefcase detail list view.
 25  * 
 26  * @param	{ZmControl}		parent		the parent
 27  * @param	{ZmBriefcaseController}	controller		the controller
 28  * @param	{DwtDropTarget}		dropTgt		the drop target
 29  * 
 30  * @extends		ZmBriefcaseBaseView
 31  */
 32 ZmDetailListView = 	function(parent, controller, dropTgt) {
 33 
 34     this._controller = controller;
 35 
 36 	var headerList = this._getHeaderList(parent);
 37 	var params = {parent:parent, className:"ZmBriefcaseDetailListView",
 38 				  view:ZmId.VIEW_BRIEFCASE_DETAIL,
 39 				  controller:controller, headerList:headerList, dropTgt:dropTgt};
 40 	ZmBriefcaseBaseView.call(this, params);
 41 
 42     this.enableRevisionView(true);
 43 
 44     this._expanded = {};
 45     this._itemRowIdList = {};
 46 
 47     this._dragSrc = new DwtDragSource(Dwt.DND_DROP_MOVE);
 48 	this._dragSrc.addDragListener(new AjxListener(this, this._dragListener));
 49 	this.setDragSource(this._dragSrc);
 50 
 51     this._dropTgt = new DwtDropTarget("ZmDetailListView");
 52 	this._dropTgt.markAsMultiple();
 53 	this._dropTgt.addDropListener(new AjxListener(this, this._dropListener));
 54 	this.setDropTarget(this._dropTgt);
 55 };

 93 ZmDetailListView.prototype._changeListener =
 94 function(ev){
 95     if(ev.event == ZmEvent.E_MOVE || ev.event == ZmEvent.E_DELETE){
 96         if (this.getDnDSelection() && this.getDnDSelection instanceof AjxVector)
 97             this.dragDeselect(this.getDnDSelection().get(0));
 98         else if (this.getDnDSelection())
 99             this.dragDeselect(this.getDnDSelection());
100     }
101 
102 
103     ZmBriefcaseBaseView.prototype._changeListener.call(this, ev);
104     if (this._revisionView && ( ev.event == ZmEvent.E_DELETE || ev.event == ZmEvent.E_MOVE )) {
105         var items = ev.getDetail("items") ? ev.getDetail("items") : [this._getItemFromEvent(ev)];
106         for (var i = 0, len = items.length; i < len; i++) {
107 			var item = items[i];
108             this.collapse(item, true);
109         }
110     }
111 };

203 ZmDetailListView.prototype._getCellContents =
204 function(htmlArr, idx, item, field, colIdx, params) {
205 
206 	if (field == ZmItem.F_SELECTION) {
207 		var icon = params.bContained ? "CheckboxChecked" : "CheckboxUnchecked";
208 		idx = this._getImageHtml(htmlArr, idx, icon, this._getFieldId(item, field));
209     } else if (field == ZmItem.F_EXPAND) {
210 		idx = this._getImageHtml(htmlArr, idx, 
211 				this._isExpandable(item) ? (this._expanded[item.id] ? "NodeExpanded" : "NodeCollapsed" )
212 						: null, this._getFieldId(item, field));   
213 	} else if (field == ZmItem.F_TYPE) {
214 		htmlArr[idx++] = AjxImg.getImageHtml(item.getIcon());
215 	} else if (field == ZmItem.F_LOCK) {
216 		idx = this._getImageHtml(htmlArr, idx, (item.locked ? "PadLock" : "Blank_16") , this._getFieldId(item, field)); //AjxImg.getImageHtml(item.locked ? "PadLock" : "Blank_16");
217 	} else if (field == ZmItem.F_VERSION) {
218 		htmlArr[idx++] = item.version;
219 	} else if (field == ZmItem.F_NAME || field == ZmItem.F_SUBJECT) {
220 		htmlArr[idx++] = "<div id='"+this._getFieldId(item, ZmItem.F_NAME)+"'>"+this._getDisplayName(item)+"</div>";
221 	} else if (field == ZmItem.F_FILE_TYPE) {
222         if(item.isFolder){
223             htmlArr[idx++] = ZmMsg.folder;
224         }else{
225             var mimeInfo = item.contentType ? ZmMimeTable.getInfo(item.contentType) : null;
226             htmlArr[idx++] = mimeInfo ? mimeInfo.desc : " ";
227         }
228 	} else if (field == ZmItem.F_SIZE) {
229 	    htmlArr[idx++] = item.isFolder ? ZmMsg.folder : AjxUtil.formatSize(item.size);
230 	} else if (field == ZmItem.F_DATE) {
231 		if (item.modifyDate) {
232 			htmlArr[idx++] = AjxDateUtil.simpleComputeDateStr(item.modifyDate);
233 		}
234 	} else if (field == ZmItem.F_FROM) {
235         var creator = item.modifier || item.creator;
236 		creator = creator ? creator.split("@") : [""];
237 		var cname = creator[0];
238 		var uname = appCtxt.get(ZmSetting.USERNAME);
239 		if (uname) {
240 			var user = uname.split("@");
241 			if (creator[1] != user[1]) {
242 				cname = creator.join("@");
243 			}
244 		}
245 		htmlArr[idx++] = "<span style='white-space: nowrap'>";
246 		htmlArr[idx++] = cname;
247 		htmlArr[idx++] = "</span>";
248 	} else if (field == ZmItem.F_FOLDER) {
249 		var briefcase = appCtxt.getById(item.folderId);
250 		htmlArr[idx++] = briefcase ? briefcase.getName() : item.folderId;
251 	} else if (field == ZmItem.F_SORTED_BY){
252         htmlArr[idx++] = this._getAbridgedContent(item, colIdx);
253     } 
254     else {
255 		idx = ZmListView.prototype._getCellContents.apply(this, arguments);
256 	}
257 
258 	return idx;
259 };

261 ZmDetailListView.prototype._getDisplayName =
262 function(item){
263     var subject;
264     if(item.isRevision){
265         subject = ("      " + item.subject );
266     }else if(parseInt(item.version) > 1){
267         subject = AjxMessageFormat.format(ZmMsg.briefcaseFileVersion, [AjxStringUtil.htmlEncode(item.name), item.version])
268     }
269     return subject || (AjxStringUtil.htmlEncode(item.name));

272 ZmDetailListView.prototype._getAbridgedContent =
273 function(item, colIdx) {
274 
275     var idx=0, html=[];
276 	idx = this._getTable(html, idx, item);
277 	idx = this._getRow(html, idx, item);
278 
279     if(this._revisionView){
280         html[idx++] = "<td style='vertical-align:middle;' width=16 id='" + this._getFieldId(item, ZmItem.F_FOLDER) + "'><center>";
281         idx = this._getCellContents(html, idx, item, ZmItem.F_EXPAND, colIdx);
282         html[idx++] = "</center></td>";
283     }
284 
285 	html[idx++] = "<td style='vertical-align:middle;' width=20 id='" + this._getFieldId(item, ZmItem.F_FOLDER) + "'><center>";
286 	html[idx++] = AjxImg.getImageHtml(item.getIcon());
287 	html[idx++] = "</center></td>";
288 	html[idx++] = "<td style='vertical-align:middle;' width='100%' id='" + this._getFieldId(item, ZmItem.F_NAME) + "'>";
289     html[idx++] = " "+ this._getDisplayName(item);
290 	html[idx++] = "</td>";
291 
292     html[idx++] = "<td style='vertical-align:middle;text-align:left;' width=40 id='" + this._getFieldId(item, ZmItem.F_SIZE) + "'>";
293 	idx = this._getCellContents(html, idx, item, ZmItem.F_SIZE, colIdx);
294 	html[idx++] = "</td>";
295 
296     html[idx++] = "<td style='vertical-align:middle;' width='16' align='right' id='" + this._getFieldId(item, ZmItem.F_TAG)+"'>";
297     idx = this._getImageHtml(html, idx, item.getTagImageInfo(), this._getFieldId(item, ZmItem.F_TAG));
298 	html[idx++] = "</td>";
299 
300 	html[idx++] = "</tr>";
301     html[idx++] = "</table>";
302 
303     html[idx++] = "<table width='100%' border='0' cellpadding=0 cellspacing=0><tr>";
304     html[idx++] = "<td style='text-align:center'>";
305     idx = this._getCellContents(html, idx, item, ZmItem.F_FROM, colIdx);
306     html[idx++] = "<td style='text-align:right;padding-right:50px;'>";

308     html[idx++] = "</td>";

311 	html[idx++] = "</td>";
312     html[idx++] = "</tr></table>";
313

333 ZmDetailListView.prototype.expand =
334 function(item, revisions){
335 
336     if(!item || !revisions || revisions.size() == 0 ) return;
337 
338     this._addRevisionRows(item, revisions);
339        
340     this._setImage(item, ZmItem.F_EXPAND, "NodeExpanded");
341     this._expanded[item.id] = true;
342 };

344 ZmDetailListView.prototype._addRevisionRows =
345 function(item, revisions){
346 
347     var rowIds = this._itemRowIdList[item.id];
348     if (rowIds && rowIds.length && rowIds.length == revisions.size() && this._rowsArePresent(item)){
349         this._showRows(rowIds, true);
350     }else{
351         var index = this._getRowIndex(item);
352         this._itemRowIdList[item.id] = [];
353         for(var i=0; i< revisions.size(); i++){
354             var rev = revisions.get(i);
355             var div = this._createItemHtml(rev);
356             //check if item exists before adding row
357             if (!document.getElementById(div.id))
358                 this._addRow(div, index+i+1);
359             else
360                 this._showRows([div.id],true);
361             this._itemRowIdList[item.id].push(div.id);
362         }
363     }
364     
365 };

367 ZmDetailListView.prototype.collapse =
368 function(item, clear){
369      var rowIds = this._itemRowIdList[item.id];
370      this._showRows(rowIds, false);
371      this._setImage(item, ZmItem.F_EXPAND, "NodeCollapsed");
372 	 this._expanded[item.id] = false;
373      if(clear){
374          this._itemRowIdList[item.id] = null;
375      }
376 };

378 ZmDetailListView.prototype.collapseAll =
379 function(){
380     var list = this.getItemList(), item;
381     for(var id in this._expanded){
382         if(this._expanded[id]){
383             item = list.getById(id);
384             if(item) this.collapse(item);
385         }
386     }
387 };

398    if (rowIds && rowIds.length) {
399         for (var i = 0; i < rowIds.length; i++) {

401             if (row) {
402                 Dwt.setVisible(row, show);
403             }
404         }

408 ZmDetailListView.prototype._rowsArePresent =
409 function(item) {
410 	var rowIds = this._itemRowIdList[item.id];
411 	if (rowIds && rowIds.length) {
412 		for (var i = 0; i < rowIds.length; i++) {
413 			if (document.getElementById(rowIds[i])) {
414 				return true;
415 			}
416 		}
417 	}
418 	this._itemRowIdList[item.id] = [];	// start over
419 	this._expanded[item.id] = false;
420 	return false;
421 };

529 ZmDetailListView.prototype._folderChangeListener =
530 function(ev){
531 
532     // make sure this is current list view
533 	if (appCtxt.getCurrentController() != this._controller) { return; }
534 
535     ZmBriefcaseBaseView.prototype._folderChangeListener.call(this, ev);
536 
537     var organizers = ev.getDetail("organizers");
538 	var organizer = (organizers && organizers.length) ? organizers[0] : ev.source;
539     var currentFolderId = this._controller._folderId;
540 
541     var refresh = false;
542     if (ev.event == ZmEvent.E_CREATE) {
543         if(organizer && currentFolderId == organizer.parent.id)
544             refresh = true;
545     }else if(ev.event == ZmEvent.E_MODIFY) {
546         var fields = ev.getDetail("fields");        
547         if( fields[ZmOrganizer.F_NAME] || fields[ZmOrganizer.F_COLOR] )
548             refresh = true;
549     }else if(ev.event == ZmEvent.E_MOVE || ev.event == ZmEvent.E_DELETE){
550         refresh = true;
551         if (this.getDnDSelection() && this.getDnDSelection instanceof AjxVector)
552             this.dragDeselect(this.getDnDSelection().get(0));
553         else if (this.getDnDSelection())
554             this.dragDeselect(this.getDnDSelection());
555         if(currentFolderId != organizer.id){
556             this.collapseAll();
557         }
558     }
559 
560     if(refresh) {
561         appCtxt.getApp(ZmApp.BRIEFCASE).search({folderId: currentFolderId});
562     }
563         
564 };

566 //drag and drop listeners
567 ZmDetailListView.prototype._dropListener =
568 function(ev) {
569     var data = ev.srcData.data;
570 	var div = this.getTargetItemDiv(ev.uiEvent);
571 	var dropFolder = this.getItemFromElement(div);
572 
573     //handle drag from tree to listview by calling controller
574     if (ev.srcData && ev.srcData.controller != appCtxt.getCurrentController()){
575         appCtxt.getCurrentController()._dropListener(ev);
576         return;
577     }
578 
579 	// only briefcase items can be dropped on us
580 	if (ev.action == DwtDropEvent.DRAG_ENTER) {
581 		ev.doIt = (dropFolder && (dropFolder instanceof ZmBriefcaseFolderItem) && (dropFolder.folder && dropFolder.folder.mayContain(data)));
582 		DBG.println(AjxDebug.DBG3, "DRAG_ENTER: doIt = " + ev.doIt);
583         this.dragSelect(div);
584 	} else if (ev.action == DwtDropEvent.DRAG_DROP) {
585         this.dragDeselect(div);
586 		appCtxt.getCurrentController()._doMove(data, dropFolder.folder);
587 	} else if (ev.action == DwtDropEvent.DRAG_LEAVE) {
588 		view.dragDeselect(div);
589 	} else if (ev.action == DwtDropEvent.DRAG_OP_CHANGED) {
590 		// nothing
591 	}
592 
593 };
