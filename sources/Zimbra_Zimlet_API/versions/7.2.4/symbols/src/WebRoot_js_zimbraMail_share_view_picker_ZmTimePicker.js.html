 20 /**
 21  * Creates a time picker control.
 22  * @class
 23  * This class represents a time picker control.
 24  * 
 25  * @param		{DwtControl}	parent		the parent
 26  * 
 27  * @extends		ZmPicker
 28  * @see			ZmPicker.TIME
 29  */
 30 ZmTimePicker = function(parent) {
 31 
 32 	ZmPicker.call(this, parent, ZmPicker.TIME);
 33 
 34     this._checkedItems = new Object();
 35 }
 36 
 37 ZmPicker.CTOR[ZmPicker.TIME] = ZmTimePicker;
 38 
 39 ZmTimePicker._onClick =
 40 function(ev) {
 41 	var element = DwtUiEvent.getTarget(ev);
 42 	var picker = element._picker;
 43 	if (element.checked)
 44 		picker._checkedItems[element._query] = true;
 45 	else
 46 		delete picker._checkedItems[element._query];
 47 	picker._updateQuery();
 48 }
 49 
 50 ZmTimePicker.prototype = new ZmPicker;
 51 ZmTimePicker.prototype.constructor = ZmTimePicker;
 52 
 53 ZmTimePicker.prototype.toString = 
 54 function() {
 55 	return "ZmTimePicker";
 56 }
 57 
 58 ZmTimePicker.prototype._makeRow =
 59 function(left, leftId, right, rightId) {
 60     var size = 20;
 61     var html = new Array(10);
 62     var i = 0;
 63     html[i++] = "<tr valign='middle'>";
 64     html[i++] = "<td align='left' nowrap><input id='" + leftId + "' type='checkbox' value='" + left + "'></input></td>";
 65     html[i++] = "<td align='left' nowrap>" + left + "</td>";
 66     html[i++] = "<td align='left' nowrap><input id='" + rightId + "' type='checkbox'></input></td>";
 67     html[i++] = "<td align='left' nowrap>" + right + "</td>";
 68     html[i++] = "</tr>";
 69 	return html.join("");
 70 }
 71 
 72 ZmTimePicker.prototype._setupPicker =
 73 function(parent) {
 74     var picker = new DwtComposite(parent);
 75 
 76     var lastHourId = Dwt.getNextId();
 77     var last4HoursId = Dwt.getNextId();
 78     var todayId = Dwt.getNextId();
 79     var yesterdayId = Dwt.getNextId();
 80     var thisWeekId = Dwt.getNextId();
 81     var lastWeekId = Dwt.getNextId();
 82     var thisMonthId = Dwt.getNextId();
 83     var lastMonthId = Dwt.getNextId();
 84     var thisYearId = Dwt.getNextId();
 85     var lastYearId = Dwt.getNextId();
 86 
 87 	var html = new Array(10);
 88 	var i = 0;
 89 	html[i++] = "<table cellpadding='2' cellspacing='0' border='0'>";
 90 	html[i++] = this._makeRow(ZmMsg.P_TIME_LAST_HOUR, lastHourId, ZmMsg.P_TIME_LAST_4_HOURS, last4HoursId);
 91 	html[i++] = this._makeRow(ZmMsg.P_TIME_TODAY, todayId, ZmMsg.P_TIME_YESTERDAY, yesterdayId);
 92 	html[i++] = this._makeRow(ZmMsg.P_TIME_THIS_WEEK, thisWeekId, ZmMsg.P_TIME_LAST_WEEK, lastWeekId);
 93 	html[i++] = this._makeRow(ZmMsg.P_TIME_THIS_MONTH, thisMonthId, ZmMsg.P_TIME_LAST_MONTH, lastMonthId);
 94 	html[i++] = this._makeRow(ZmMsg.P_TIME_THIS_YEAR, thisYearId, ZmMsg.P_TIME_LAST_YEAR, lastYearId);
 95 	html[i++] = "</table>";
 96 	picker.getHtmlElement().innerHTML = html.join("");
 97 
 98 	this._installOnClick(lastHourId, "date:>-60mi");
 99 	this._installOnClick(last4HoursId, "date:>-240mi");
100 	this._installOnClick(todayId, "date:-0day");
101 	this._installOnClick(yesterdayId, "date:yesterday");
102 	this._installOnClick(thisWeekId, "date:-0week");
103 	this._installOnClick(lastWeekId, "date:-1week");
104 	this._installOnClick(thisMonthId, "date:-0month");
105 	this._installOnClick(lastMonthId,"date:-1month"); 
106 	this._installOnClick(thisYearId, "date:-0year"); 
107 	this._installOnClick(lastYearId, "date:-1year"); 
108 }
109 
110 ZmTimePicker.prototype._installOnClick =
111 function(id, query) {
112 	var box = document.getElementById(id);
113 	Dwt.setHandler(box, DwtEvent.ONCLICK, ZmTimePicker._onClick);
114 	box._query = query;
115 	box._picker = this;
116 }
117 
118 ZmTimePicker.prototype._updateQuery = 
119 function() {
120 	var query = new Array();
121 	for (var q in this._checkedItems)
122 		query.push(q);
123 	var str = "";
124 	if (query.length > 1)
125 		str += "(";
126 	str += query.join(" OR ");
127 	if (query.length > 1)
128 		str += ")";
129 	this.setQuery(str);
130 	this.execute();
131 }
