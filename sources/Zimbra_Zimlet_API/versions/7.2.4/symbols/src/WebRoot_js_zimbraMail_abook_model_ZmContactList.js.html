 22 /**
 23  * Create a new, empty contact list.
 24  * @class
 25  * This class represents a list of contacts. In general, the list is the result of a
 26  * search. It may be the result of a <code><GetContactsRequest></code>, which returns all of the user's
 27  * local contacts. That list is considered to be canonical.
 28  * <p>
 29  * Loading of all local contacts has been optimized by delaying the creation of {@link ZmContact} objects until
 30  * they are needed. That has a big impact on IE, and not much on Firefox. Loading a subset
 31  * of attributes did not have much impact on load time, probably because a large majority
 32  * of contacts contain only those minimal fields.</p>
 33  *
 34  * @author Conrad Damon
 35  *
 36  * @param {ZmSearch}	search	the search that generated this list
 37  * @param {Boolean}	isGal		if <code>true</code>, this is a list of GAL contacts
 38  * @param {constant}	type		the item type
 39  * 
 40  * @extends		ZmList
 41  */
 42 ZmContactList = function(search, isGal, type) {
 43 
 44 	if (arguments.length == 0) { return; }
 45 	type = type || ZmItem.CONTACT;
 46 	ZmList.call(this, type, search);
 47 
 48 	this.isGal = (isGal === true);
 49 	this.isCanonical = false;
 50 	this.isLoaded = false;
 51 
 52 	this._app = appCtxt.getApp(ZmApp.CONTACTS);
 53 	this._emailToContact = this._app._byEmail;
 54 	this._imAddressToContact = this._app._byIM;
 55 	this._phoneToContact = this._app._byPhone;
 56 };
 57 
 58 ZmContactList.prototype = new ZmList;
 59 ZmContactList.prototype.constructor = ZmContactList;

 62 // Constants
 63 
 64 // Support for loading user's local contacts from a large string
 65 
 66 ZmContactList.URL = "/Contacts?fmt=cf&t=2&all";	// REST URL for loading user's local contacts
 67 ZmContactList.CONTACT_SPLIT_CHAR	= '\u001E';	// char for splitting string into contacts
 68 ZmContactList.FIELD_SPLIT_CHAR		= '\u001D';	// char for splitting contact into fields
 69 // fields that belong to a contact rather than its attrs
 70 ZmContactList.IS_CONTACT_FIELD = {"id":true, "l":true, "d":true, "fileAsStr":true, "rev":true};

 86 ZmContactList.prototype.addLoadedCallback =
 87 function(callback) {
 88 	if (this.isLoaded) {
 89 		callback.run();
 90 		return;
 91 	}
 92 	if (!this._loadedCallbacks) {
 93 		this._loadedCallbacks = [];
 94 	}
 95 	this._loadedCallbacks.push(callback);
 96 };

 98 /**
 99  * @private
100  */
101 ZmContactList.prototype._finishLoading =
102 function() {
103 	DBG.timePt("done loading " + this.size() + " contacts");
104 	this.isLoaded = true;
105 	if (this._loadedCallbacks) {
106 		var callback;
107 		while (callback = this._loadedCallbacks.shift()) {
108 			callback.run();
109 		}
110 	}
111 };

113 /**
114  * Retrieves the contacts from the back end, and parses the response. The list is then sorted.
115  * This method is used only by the canonical list of contacts, in order to load their content.
116  * <p>
117  * Loading a minimal set of attributes did not result in a significant performance gain.
118  * </p>
119  * 
120  * @private
121  */
122 ZmContactList.prototype.load =
123 function(callback, errorCallback, accountName) {
124 	// only the canonical list gets loaded
125 	this.isCanonical = true;
126 	var respCallback = new AjxCallback(this, this._handleResponseLoad, [callback]);
127 	DBG.timePt("requesting contact list", true);
128 
129 	var params = {asyncMode:true, noBusyOverlay:true, callback:respCallback, errorCallback:errorCallback};
130 	params.restUri = AjxUtil.formatUrl({path:["/home/", (accountName || appCtxt.getUsername()), ZmContactList.URL].join(""), qsReset:true});
131 	DBG.println(AjxDebug.DBG1, "loading contacts from " + params.restUri);
132 	appCtxt.getAppController().sendRequest(params);
133 };

135 /**
136  * @private
137  */
138 ZmContactList.prototype._handleResponseLoad =
139 function(callback, result) {
140 	DBG.timePt("got contact list");
141 
142 	var text = result.getResponse();
143 	if (text) {
144 		var contacts = text.split(ZmContactList.CONTACT_SPLIT_CHAR);
145 		for (var i = 0, len = contacts.length; i < len; i++) {
146 			var fields = contacts[i].split(ZmContactList.FIELD_SPLIT_CHAR);
147 			var contact = {}, attrs = {};;
148 			for (var j = 0, len1 = fields.length; j < len1; j += 2) {
149 				if (ZmContactList.IS_CONTACT_FIELD[fields[j]]) {
150 					contact[fields[j]] = fields[j + 1];
151 				} else {
152 					attrs[fields[j]] = fields[j + 1];
153 				}
154 			}
155 			contact._attrs = attrs;
156 			this._addContact(contact);
157 		}
158 	}
159 	this._finishLoading();
160 
161 	if (callback) {
162 		callback.run();
163 	}
164 };

166 ZmContactList.prototype.add = 
167 function(item, index) {
168 	if (!item.id || !this._idHash[item.id]) {
169 		this._vector.add(item, index);
170 		if (item.id) {
171 			this._idHash[item.id] = item;
172 		}
173 		this._updateHashes(item, true);
174 	}
175 };
176 
177 ZmContactList.prototype.cache = 
178 function(offset, newList) {
179 	var getId = function(){
180 		return this.id;
181 	}
182 	var exists = function(obj) {
183 		return this._vector.containsLike(obj, getId);
184 	}
185 	var unique = newList.sub(exists, this);
186 
187 	this.getVector().merge(offset, unique);
188 	// reparent each item within new list, and add it to ID hash
189 	var list = unique.getArray();
190 	for (var i = 0; i < list.length; i++) {
191 		var item = list[i];
192 		item.list = this;
193 		if (item.id) {
194 			this._idHash[item.id] = item;
195 		}
196 	}
197 };

199 /**
200  * @private
201  */
202 ZmContactList.prototype._addContact =
203 function(contact) {
204 
205 	// note that we don't create a ZmContact here (optimization)
206 	contact.list = this;
207 	this._updateHashes(contact, true);
208 	var fn = [], fl = [];
209 	if (contact._attrs[ZmContact.F_firstName])	{ fn.push(contact._attrs[ZmContact.F_firstName]); }
210 	if (contact._attrs[ZmContact.F_middleName])	{ fn.push(contact._attrs[ZmContact.F_middleName]); }
211 	if (contact._attrs[ZmContact.F_lastName])	{ fn.push(contact._attrs[ZmContact.F_lastName]); }
212 	if (fn.length) {
213 		contact._attrs[ZmContact.X_fullName] = fn.join(" ");
214 	}
215 	if (contact._attrs[ZmContact.F_firstName])	{ fl.push(contact._attrs[ZmContact.F_firstName]); }
216 	if (contact._attrs[ZmContact.F_lastName])	{ fl.push(contact._attrs[ZmContact.F_lastName]); }
217 	contact._attrs[ZmContact.X_firstLast] = fl.join(" ");
218 
219 	this.add(contact);
220 };

222 /**
223  * Converts an anonymous contact object (contained by the JS returned by load request)
224  * into a ZmContact, and updates the containing list if it is the canonical one.
225  *
226  * @param {Object}	contact		a contact
227  * @param {int}	idx		the index of contact in canonical list
228  * 
229  * @private
230  */
231 ZmContactList.prototype._realizeContact =
232 function(contact, idx) {
233 
234 	if (contact instanceof ZmContact) { return contact; }
235 	if (contact && contact.type == ZmItem.CONTACT) { return contact; }	// instanceof often fails in new window
236 
237 	var args = {list:this};
238 	var obj = eval(ZmList.ITEM_CLASS[this.type]);
239 	var realContact = obj && obj.createFromDom(contact, args);
240 
241 	if (this.isCanonical) {
242 		var a = this.getArray();
243 		idx = idx || this._getIndexById(contact.id);
244 		a[idx] = realContact;
245 		this._updateHashes(realContact, true);
246 		this._idHash[contact.id] = realContact;
247 	}
248 
249 	return realContact;
250 };

252 /**
253  * Finds the array index for the contact with the given ID.
254  *
255  * @param {int}	id		the contact ID
256  * @return	{int}	the index
257  * @private
258  */
259 ZmContactList.prototype._getIndexById =
260 function(id) {
261 	var a = this.getArray();
262 	for (var i = 0; i < a.length; i++) {
263 		if (a[i].id == id) {
264 			return i;
265 		}
266 	}
267 	return null;
268 };

270 /**
271  * Override in order to make sure the contacts have been realized. We don't
272  * call realizeContact() since this is not the canonical list.
273  *
274  * @param {int}	offset		the starting index
275  * @param {int}	limit		the size of sublist
276  * @return	{AjxVector}	a vector of {@link ZmContact} objects
277  */
278 ZmContactList.prototype.getSubList =
279 function(offset, limit, folderId) {
280 	if (folderId && this.isCanonical) {
281 		// only collect those contacts that belong to the given folderId if provided
282 		var newlist = [];
283 		var sublist = this.getArray();
284 		var offsetCount = 0;
285 		this.setHasMore(false);
286 
287 		for (var i = 0; i < sublist.length; i++) {
288 			sublist[i] = this._realizeContact(sublist[i], i);
289 			var folder = appCtxt.getById(sublist[i].folderId);
290 			if (folder && folder.nId == ZmOrganizer.normalizeId(folderId)) {
291 				if (offsetCount >= offset) {
292 					if (newlist.length == limit) {
293 						this.setHasMore(true);
294 						break;
295 					}
296 					newlist.push(sublist[i]);
297 				}
298 				offsetCount++;
299 			}
300 		}
301 
302 		return AjxVector.fromArray(newlist);
303 	} else {
304 		var vec = ZmList.prototype.getSubList.call(this, offset, limit);
305 		if (vec) {
306 			var a = vec.getArray();
307 			for (var i = 0; i < a.length; i++) {
308 				a[i] = this._realizeContact(a[i], offset + i);
309 			}
310 		}
311 
312 		return vec;
313 	}
314 };

316 /**
317  * Override in order to make sure the contact has been realized. Canonical list only.
318  *
319  * @param {int}	id		the contact ID
320  * @return	{ZmContact}	the contact or <code>null</code> if not found
321  */
322 ZmContactList.prototype.getById =
323 function(id) {
324 	if (!id || !this.isCanonical) return null;
325 
326 	var contact = this._idHash[id];
327 	return contact ? this._realizeContact(contact) : null;
328 };

330 /**
331  * Gets the contact with the given address, if any (canonical list only).
332  *
333  * @param {String}	address	an email address
334  * @return	{ZmContact}	the contact or <code>null</code> if not found
335  */
336 ZmContactList.prototype.getContactByEmail =
337 function(address) {
338 	if (!address || !this.isCanonical) return null;
339 
340 	var contact = this._emailToContact[address.toLowerCase()];
341 	if (contact) {
342 		contact = this._realizeContact(contact);
343 		contact._lookupEmail = address;	// so caller knows which address matched
344 		return contact;
345 	} else {
346 		return null;
347 	}
348 };

362 /**
363  * Gets information about the contact with the given phone number, if any (canonical list only).
364  *
365  * @param {String}	phone	the phone number
366  * @return	{Hash}	an object with <code>contact</code> = the contact & <code>field</code> = the field with the matching phone number
367  */
368 ZmContactList.prototype.getContactByPhone =
369 function(phone) {
370 	if (!phone || !this.isCanonical) return null;
371 
372 	var digits = this._getPhoneDigits(phone);
373 	var data = this._phoneToContact[digits];
374 	if (data) {
375 		data.contact = this._realizeContact(data.contact);
376 		return data;
377 	} else {
378 		return null;
379 	}

382 /**
383  * Moves a list of items to the given folder.
384  * <p>
385  * This method calls the base class for normal "moves" UNLESS we're dealing w/
386  * shared items (or folder) in which case we must send a CREATE request for the
387  * given folder to the server followed by a hard delete of the shared contact.
388  * </p>
389  *
390  * @param {Hash}	params		a hash of parameters
391  * @param	{Array}       params.items			a list of items to move
392  * @param	{ZmFolder}	params.folder		the destination folder
393  * @param	{Hash}	       params.attrs		the additional attrs for SOAP command
394  * @param	{Boolean}	params.outOfTrash	if <code>true</code>, we are moving contacts out of trash
395  */
396 ZmContactList.prototype.moveItems =
397 function(params) {
398 
399 	params = Dwt.getParams(arguments, ["items", "folder", "attrs", "outOfTrash"]);
400 	params.items = AjxUtil.toArray(params.items);
401 
402 	var moveBatchCmd = new ZmBatchCommand(true, null, true);
403 	var loadBatchCmd = new ZmBatchCommand(true, null, true);
404 	var softMove = [];
405 	var hardMove = [];
406 
407 	// if the folder we're moving contacts to is a shared folder, then dont bother
408 	// checking whether each item is shared or not
409 	if (params.items[0] && params.items[0] instanceof ZmItem) {
410 		for (var i = 0; i < params.items.length; i++) {
411 			var contact = params.items[i];
412 
413 			if (contact.isReadOnly()) { continue; }
414 
415 			if (contact.isShared() || params.folder.link) {
416 				hardMove.push(contact);
417 				if (!contact.isLoaded) {
418 					loadBatchCmd.add(new AjxCallback(contact, contact.load, [null, null]));
419 				}
420 				moveBatchCmd.add(this._getCopyCmd(contact, params.folder));
421 			} else {
422 				softMove.push(contact);
423 			}
424 		}
425 	} else {
426 		softMove = params.items;
427 	}
428 
429 	if (hardMove.length > 0) {
430 		var params1 = {
431 			items: hardMove,
432 			action: "delete",
433 			actionText: ZmMsg.actionMove,
434 			actionArg: params.folder.getName(false, false, true)
435 		};
436 
437 		if (loadBatchCmd.size()) {
438 			var respCallback = new AjxCallback(this, this._handleResponseLoadMove, [moveBatchCmd, params1]);
439 			loadBatchCmd.run(respCallback);
440 		} else {
441 			var deleteCmd = new AjxCallback(this, this._itemAction, [params1]);
442 			moveBatchCmd.add(deleteCmd);
443 
444 			var respCallback = new AjxCallback(this, this._handleResponseMoveBatchCmd);
445 			moveBatchCmd.run(respCallback);
446 		}
447 	}
448 
449 	// for "soft" moves, handle moving out of Trash differently
450 	if (softMove.length > 0) {
451 		var params1 = AjxUtil.hashCopy(params);
452 		params1.attrs = params.attrs || {};
453 		params1.attrs.l = params.folder.id;
454 		params1.action = "move";
455         params1.accountName = appCtxt.multiAccounts && appCtxt.accountList.mainAccount.name;
456         if (params1.folder.id == ZmFolder.ID_TRASH) {
457             params1.actionText = ZmMsg.actionTrash;
458             // bug: 47389 avoid moving to local account's Trash folder.
459             params1.accountName = appCtxt.multiAccounts && params.items[0].getAccount().name;
460         } else {
461             params1.actionText = ZmMsg.actionMove;
462             params1.actionArg = params.folder.getName(false, false, true);
463         }
464 		params1.callback = params.outOfTrash && new AjxCallback(this, this._handleResponseMoveItems, params);
465 
466 		this._itemAction(params1);
467 	}
468 };

470 /**
471  * @private
472  */
473 ZmContactList.prototype._handleResponseMoveBatchCmd =
474 function(result) {
475 	var resp = result.getResponse().BatchResponse.ContactActionResponse;
476 	// XXX: b/c the server does not return notifications for actions done on
477 	//      shares, we manually notify - TEMP UNTIL WE GET BETTER SERVER SUPPORT
478 	var ids = resp[0].action.id.split(",");
479 	for (var i = 0; i < ids.length; i++) {
480 		var contact = appCtxt.cacheGet(ids[i]);
481 		if (contact && contact.isShared()) {
482 			contact.notifyDelete();
483 			appCtxt.cacheRemove(ids[i]);
484 		}
485 	}
486 };

512 function(contact, folder) {

549 ZmContactList.prototype.notifyCreate =
550 function(node) {
551 	var obj = eval(ZmList.ITEM_CLASS[this.type]);
552 	if (obj) {
553 		var item = obj.createFromDom(node, {list:this});
554 		var index = this._sortIndex(item);
555 		// only add if it sorts into this list
556 		var listSize = this.size();
557 		var visible = false;
558 		if (index < listSize || listSize == 0 || (index==listSize && !this._hasMore)) {
559 			this.add(item, index);
560 			this.createLocal(item);
561 			visible = true;
562 		}
563 		this._notify(ZmEvent.E_CREATE, {items: [item], sortIndex: index, visible: visible});
564 	}
565 };

598 /**
599  * Handle modified contact.
600  * 
601  * @private
602  */
603 ZmContactList.prototype.modifyLocal =
604 function(item, details) {
605 	if (details) {
606 		// notify item's list
607 		details.items = [item];
608 		this._notify(ZmEvent.E_MODIFY, details);
609 	}
610 
611 	var contact = details.contact;
612 	if (this.isCanonical) {
613 		// Remove traces of old contact - NOTE: we pass in null for the ID on
614 		// PURPOSE to avoid overwriting the existing cached contact
615 		var oldContact = new ZmContact(null, this);
616 		oldContact.id = details.contact.id;
617 		oldContact.attr = details.oldAttr;
618 		this._updateHashes(oldContact, false);
619 
620 		// add new contact to hashes
621 		this._updateHashes(contact, true);
622 	}
623 
624 	// place in correct position in list
625 	if (details.fileAsChanged) {
626 		this.remove(contact);
627 		var index = this._sortIndex(contact);
628 		var listSize = this.size();
629 		if (index < listSize || listSize == 0 || (index == listSize && !this._hasMore)) {
630 			this.add(contact, index);
631 		}
632 	}
633 
634 	// reset addrbook property
635 	if (contact.addrbook && (contact.addrbook.id != contact.folderId)) {
636 		contact.addrbook = appCtxt.getById(contact.folderId);
637 	}
638 };

650 /**
651  * @private
652  */
653 ZmContactList.prototype._updateHashes =
654 function(contact, doAdd) {
655 
656 	this._app.updateCache(contact, doAdd);
657 
658 	// Update email hash.
659 	for (var index = 0; index < ZmContact.EMAIL_FIELDS.length; index++) {
660 		var field = ZmContact.EMAIL_FIELDS[index];
661 		for (var i = 1; true; i++) {
662 			var aname = ZmContact.getAttributeName(field, i);
663 			var avalue = ZmContact.getAttr(contact, aname);
664 			if (!avalue) break;
665 			if (doAdd) {
666 				this._emailToContact[avalue.toLowerCase()] = contact;
667 			} else {
668 				delete this._emailToContact[avalue.toLowerCase()];
669 			}
670 		}
671 	}
672 
673 	// Update phone hash.
674 	if (appCtxt.get(ZmSetting.VOICE_ENABLED)) {
675 		for (var index = 0; index < ZmContact.PHONE_FIELDS.length; index++) {
676 			var field = ZmContact.PHONE_FIELDS[index];
677 			for (var i = 1; true; i++) {
678 				var aname = ZmContact.getAttributeName(field, i);
679 				var avalue = ZmContact.getAttr(contact, aname);
680 				if (!avalue) break;
681 				var digits = this._getPhoneDigits(avalue);
682 				if (digits) {
683 					if (doAdd) {
684 						this._phoneToContact[avalue] = {contact: contact, field: aname};
685 					} else {
686 						delete this._phoneToContact[avalue];
687 					}
688 				}
689 			}
690 		}
691 	}
692 
693 	// Update IM hash.
694 	if (appCtxt.get(ZmSetting.IM_ENABLED)) {
695 		for (var index = 0; index < ZmContact.IM_FIELDS.length; index++) {
696 			var field = ZmContact.IM_FIELDS[index];
697 			for (var i = 1; true; i++) {
698 				var aname = ZmContact.getAttributeName(field, i);
699 				var avalue = ZmContact.getAttr(contact, aname);
700 				if (!avalue) break;
701 				if (doAdd) {
702 					this._imAddressToContact[avalue.toLowerCase()] = contact;
703 				} else {
704 					delete this._imAddressToContact[avalue.toLowerCase()];
705 				}
706 			}
707 		}
708 	}
709 };

711 /**
712  * Strips all non-digit characters from a phone number.
713  * 
714  * @private
715  */
716 ZmContactList.prototype._getPhoneDigits =
717 function(phone) {
718 	return phone.replace(/[^\d]/g, '');
719 };

721 /**
722  * Returns the position at which the given contact should be inserted in this list.
723  * 
724  * @private
725  */
726 ZmContactList.prototype._sortIndex =
727 function(contact) {
728 	var a = this._vector.getArray();
729 	for (var i = 0; i < a.length; i++) {
730 		if (ZmContact.compareByFileAs(a[i], contact) > 0) {
731 			return i;
732 		}
733 	}
734 	return a.length;
735 };
