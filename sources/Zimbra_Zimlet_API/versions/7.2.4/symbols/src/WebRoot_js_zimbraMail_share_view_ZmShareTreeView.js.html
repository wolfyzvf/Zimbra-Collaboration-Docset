 16 ZmShareTreeView = function(params) {
 17     if (arguments.length == 0) return;
 18     params.className = params.className || "ZmShareTreeView OverviewTree";
 19 	params.id = "ZmShareTreeView";
 20     DwtTree.call(this, params);
 21     // TODO: Why is this tree being set to overflow: visible?!?!
 22     this.getHtmlElement().style.overflow = "auto";
 23 };
 24 ZmShareTreeView.prototype = new DwtTree;
 25 ZmShareTreeView.prototype.constructor = ZmShareTreeView;
 26 
 27 ZmShareTreeView.TREE_INDEX = 0;
 28 
 29 ZmShareTreeView.prototype.toString = function() {
 30     return "ZmShareTreeView";
 31 };
 32 
 33 //
 34 // Public methods
 35 //
 36 
 37 ZmShareTreeView.prototype.set = function(params) {
 38     if (this.root) {
 39         this.removeNode(this.root);
 40     }
 41     this._itemHash = {};
 42 
 43     var root = this.root = params.dataTree.root;
 44     var rootItem = this._createTreeItem(this, root);
 45     var children = root.children.getArray();
 46     for (var i = 0; i < children.length; i++) {
 47         this._createTreeItem(rootItem, children[i]);
 48     }
 49 };
 50 
 51 ZmShareTreeView.prototype.getTreeItemById = function(id) {
 52     return this._itemHash && this._itemHash[id];
 53 };
 54 
 55 // DwtTree methods
 56 
 57 ZmShareTreeView.prototype.removeChild = function(child) {
 58     delete this._itemHash[child.getData(Dwt.KEY_ID)];
 59     DwtTree.prototype.removeChild.apply(this, arguments);
 60 };
 61 
 62 // node manipulation
 63 
 64 ZmShareTreeView.prototype.appendChild = function(newNode, parentNode, index, tooltip) {
 65     // add node
 66     var children = parentNode.children;
 67     index = index || children.size();
 68     children.add(newNode, index);
 69 
 70     // add tree item
 71     var parentTreeItem = this.getTreeItemById(parentNode.id);
 72     return this._createTreeItem(parentTreeItem, newNode, index, tooltip);
 73 };
 74 
 75 ZmShareTreeView.prototype.insertBefore = function(newNode, refNode) {
 76     var parent = refNode.parent;
 77     var index = parent.children.indexOf(refNode);
 78     this.appendChild(newNode, parent, index);
 79 };
 80 
 81 ZmShareTreeView.prototype.replaceNode = function(newNode, oldNode) {
 82     var parent = oldNode.parent;
 83     var children = parent.children;
 84     var index = children.indexOf(oldNode);
 85     var nextNode = children.get(index + 1);
 86     this.removeNode(oldNode);
 87     return nextNode ? this.insertBefore(newNode, nextNode) : this.appendChild(newNode, parent);
 88 };
 89 
 90 ZmShareTreeView.prototype.removeNode = function(oldNode) {
 91     // remove node
 92     if (oldNode.parent) {
 93         oldNode.parent.children.remove(oldNode);
 94     }
 95 
 96     // remove tree item
 97     var treeItem = this.getTreeItemById(oldNode.id);
 98     if (treeItem) {
 99         treeItem.parent.removeChild(treeItem);
100     }
101 };

107 ZmShareTreeView.prototype._createTreeItem = function(parent, organizer, index, tooltip) {
108 	var treeItemId = "ZmShareTreeItem_" + ZmShareTreeView.TREE_INDEX++;
109     var treeItem = new DwtTreeItem({parent:parent, id: treeItemId});
110     treeItem.setText(organizer.name);
111     treeItem.setImage(organizer.getIcon());
112     treeItem.setToolTipContent(tooltip);
113     treeItem.setData(Dwt.KEY_ID, organizer.id);
114     treeItem.setData(Dwt.KEY_OBJECT, organizer);
115     this._itemHash[organizer.id] = treeItem;
116     return treeItem;
117 };
