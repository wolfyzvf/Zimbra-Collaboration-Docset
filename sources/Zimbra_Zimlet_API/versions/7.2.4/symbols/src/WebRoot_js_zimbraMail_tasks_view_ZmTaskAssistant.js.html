 21 /**
 22  * Creates the task assistant.
 23  * @class
 24  * This class represents a task assistant.
 25  * 
 26  * 
 27  * @extends		ZmAssistant
 28  */
 29 ZmTaskAssistant = function() {
 30 	ZmAssistant.call(this, ZmMsg.createNewTask, ZmMsg.ASST_CMD_TASK, ZmMsg.ASST_CMD_SUM_TASK);
 31 };
 32 
 33 ZmTaskAssistant.prototype = new ZmAssistant();
 34 ZmTaskAssistant.prototype.constructor = ZmTaskAssistant;
 35 
 36 /**
 37  * Initializes the assistant.
 38  * 
 39  * @param	{ZmDialog}	dialog		the dialog
 40  */
 41 ZmTaskAssistant.prototype.initialize =
 42 function(dialog) {
 43 	ZmAssistant.prototype.initialize.call(this, dialog);
 44 	this._taskData = {};
 45 };
 46 
 47 ZmTaskAssistant.prototype.okHandler =
 48 function(dialog) {
 49 	if (!this._taskData.subject) {
 50 		dialog.messageDialog(ZmMsg.errorMissingSubject, DwtMessageDialog.CRITICAL_STYLE);
 51 	} else {
 52 		var task = this.getTask();
 53 		task.save();// TODO: callback, etc.
 54 		return true;
 55 	}
 56 };
 57 
 58 ZmTaskAssistant.prototype.extraButtonHandler =
 59 function(dialog) {
 60 	var app = appCtxt.getApp(ZmApp.TASKS);
 61 	var tc = app.getTaskController();
 62 	tc.show(this.getTask(), ZmCalItem.MODE_NEW_FROM_QUICKADD);
 63 	return true;
 64 };
 65 
 66 ZmTaskAssistant.prototype.getHelp =
 67 function() {
 68 	return ZmMsg.ASST_TASK_HELP;
 69 };
 70 
 71 /**
 72  * Gets the task.
 73  * 
 74  * @return	{ZmTask}		the task
 75  */
 76 ZmTaskAssistant.prototype.getTask =
 77 function() {
 78 	var task = new ZmTask();
 79 	if (this._taskData.startDate) task.setStartDate(this._taskData.startDate);
 80 	if (this._taskData.endDate) task.setEndDate(this._taskData.endDate);
 81 	task.setAllDayEvent(true);
 82 
 83 	if (this._taskData.location) task.location = this._taskData.location;
 84 	if (this._taskData.notes) task.setTextNotes(this._taskData.notes);
 85 	if (this._taskData.subject) task.setName(this._taskData.subject);
 86 	return task;
 87 };
 88 
 89 ZmTaskAssistant.__RE_handleSubject = new RegExp([ZmAssistant.SPACES,"\"([^\"]*)\"?",ZmAssistant.SPACES].join(""));
 90 ZmTaskAssistant.__RE_handleLocation = new RegExp([ZmAssistant.SPACES,"\\[([^\\]]*)\\]?",ZmAssistant.SPACES].join(""));
 91 ZmTaskAssistant.__RE_handleNotes = new RegExp([ZmAssistant.SPACES,"\\(([^)]*)\\)?",ZmAssistant.SPACES].join(""));
 92 
 93 /**
 94  *
 95  * (...)                 matched as notes, stripped out
 96  * [...]                 matched as location, stripped out
 97  * {date-spec}           first matched pattern is "start date", second is "end date"
 98  * subject "..."         explicit subject
 99  *
100  * everything renaming is the subject, unless subject was explicit
101  *
102  * example:
103  *
104  * lunch 12:30 PM next friday with satish (to discuss future release) [CPK, Palo Alto]
105  *
106  * "12:30 PM" matched as a time, saved as "start time" *
107  * "next friday" matches a date, so is stripped out and saved as "start date"
108  * (...) matched as notes, stripped out and saved as "notes"
109  * [...] matched as location
110  *
111  * everything left "lunch with satish" is taken as subject
112  * 
113  * @private
114  */
115 ZmTaskAssistant.prototype.handle =
116 function(dialog, verb, args) {
117 	dialog._setOkButton(AjxMsg.ok, true, true, false);
118 	dialog._setExtraButton(ZmMsg.moreDetails, true, true, false);
119 
120 	var tdata = this._taskData = {};
121 
122 	tdata.startDate = new Date();
123 	tdata.endDate = null;
124 
125 	var match;
126 
127 	match = args.match(ZmTaskAssistant.__RE_handleSubject);
128 	if (match) {
129 		tdata.subject = match[1];
130 		args = args.replace(match[0], " ");
131 	}
132 
133 	match = args.match(ZmTaskAssistant.__RE_handleLocation);
134 	if (match) {
135 		tdata.location = match[1];
136 		args = args.replace(match[0], " ");
137 	}
138 
139 	match = args.match(ZmTaskAssistant.__RE_handleNotes);
140 	if (match) {
141 		tdata.notes = match[1];
142 		args = args.replace(match[0], " ");
143 	}
144 
145 	// look for start date
146 	match = this._objectManager.findMatch(args, ZmObjectManager.DATE);
147 	if (match) {
148 		args = args.replace(match[0], " ");
149 		tdata.startDate = match.context.date;
150 		if (tdata.startTime) tdata.startDate.setHours(tdata.startTime.hour, tdata.startTime.minute);
151 	}
152 
153 	// look for end date
154 	match = this._objectManager.findMatch(args, ZmObjectManager.DATE);
155 	if (match) {
156 		args = args.replace(match[0], " ");
157 		tdata.endDate = match.context.date;
158 		if (tdata.endTime != null) tdata.endDate.setHours(tdata.endTime.hour, tdata.endTime.minute);
159 		else if (tdata.startTime != null) tdata.endDate.setHours(tdata.startTime.hour, tdata.startTime.minute);
160 	} else {
161         if (tdata.endTime) {
162 			tdata.endDate = new Date(tdata.startDate.getTime());
163 			if (tdata.endTime != null) tdata.endDate.setHours(tdata.endTime.hour, tdata.endTime.minute);
164 		} else if (tdata.startTime) {
165 			tdata.endDate = new Date(tdata.startDate.getTime() + 1000 * 60 * 60);
166 		}
167         //task cannot get saved only with startDate
168         if(tdata.endDate == null && tdata.startDate != null) {
169             tdata.endDate = tdata.startDate;
170         }
171 	}
172 
173 	if (tdata.subject == null) {
174 		tdata.subject = ZmAssistant.normalize(ZmAssistant.trim(args));
175 	}
176 
177 	var subStr = AjxStringUtil.convertToHtml(tdata.subject == "" ? ZmMsg.ASST_APPT_subject : tdata.subject);
178 	var locStr = AjxStringUtil.convertToHtml(tdata.location == null ? ZmMsg.ASST_APPT_location : tdata.location);
179 	var notesStr = AjxStringUtil.convertToHtml(tdata.notes == null ? ZmMsg.ASST_APPT_notes : tdata.notes);
180 	var subjectIndex = this._setField("* "+ZmMsg.subject, subStr, tdata.subject == "", false);
181 	this._setDateFields(tdata.startDate, tdata.endDate, subjectIndex);
182 	this._setField(ZmMsg.location, locStr, tdata.location == null, false);
183 	this._setField(ZmMsg.notes, notesStr, tdata.notes == null, false);
184 };
185 
186 ZmTaskAssistant.prototype._setDateFields =
187 function(startDate, endDate, rowIndex) {
188 	var startDateValue = DwtCalendar.getDateFullFormatter().format(startDate);
189 	var sameDay = false;
190 	var html = new AjxBuffer();
191 	html.append("<table border=0 cellpadding=0 cellspacing=0>");
192 	html.append("<tr>");
193 	html.append("<td>", AjxStringUtil.htmlEncode(startDateValue), "</td>");
194 	html.append("</tr></table>");
195 	var doEnd = (endDate && !sameDay);
196 
197 	if (doEnd) {
198 		this._clearField(ZmMsg.time);
199 		rowIndex = this._setField(ZmMsg.startTime, html.toString(), false, false, rowIndex+1);
200 
201 		html.clear();
202 		var endDateValue = DwtCalendar.getDateFullFormatter().format(endDate);
203 			html.append("<table border=0 cellpadding=0 cellspacing=0>");
204 		html.append("<tr>");
205 		html.append("<td>", AjxStringUtil.htmlEncode(endDateValue), "</td>");
206 		html.append("</tr></table>");
207 	} else {
208 		this._setField(ZmMsg.time, html.toString(), false, false, rowIndex+1);
209 		this._clearField(ZmMsg.startTime);
210 		this._clearField(ZmMsg.endTime);
211 	}
212 };
