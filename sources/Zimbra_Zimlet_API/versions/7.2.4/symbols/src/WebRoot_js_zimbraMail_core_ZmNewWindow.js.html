 21 /**
 22  * Creates a controller to run <code>ZmNewWindow</code>. Do not call directly, instead use
 23  * the <code>run()</code> factory method.
 24  * @class
 25  * This class is the controller for a window created outside the main client
 26  * window. It is a very stripped down and specialized version of {@link ZmZimbraMail}.
 27  * The child window is single-use; it does not support switching among multiple
 28  * views.
 29  *
 30  * @author Parag Shah
 31  * 
 32  * @extends	ZmController
 33  * 
 34  * @see		#run
 35  */
 36 ZmNewWindow = function() {
 37 
 38 	ZmController.call(this, null);
 39 
 40 	appCtxt.setAppController(this);
 41 
 42 	this._settings = appCtxt.getSettings();
 43 	this._settings.setReportScriptErrorsSettings(AjxException, ZmController.handleScriptError); //must set this for child window since AjxException is fresh for this window. Also must pass AjxException and the handler since we want it to update the one from this child window, and not the parent window
 44 
 45 	this._shell = appCtxt.getShell();
 46 
 47 	// Register keymap and global key action handler w/ shell's keyboard manager
 48 	this._kbMgr = appCtxt.getKeyboardMgr();
 49 	if (appCtxt.get(ZmSetting.USE_KEYBOARD_SHORTCUTS)) {
 50 		this._kbMgr.enable(true);
 51 		this._kbMgr.registerKeyMap(new ZmKeyMap());
 52 		this._kbMgr.pushDefaultHandler(this);
 53 	}
 54 
 55 	this._apps = {};
 56 };

 73 /**
 74  * Sets up new window and then starts it by calling its constructor. It is assumed that the
 75  * CSFE is on the same host.
 76  * 
 77  */
 78 ZmNewWindow.run =
 79 function() {
 80 
 81 	// We're using a custom pkg that includes the mail classes we'll need, so pretend that
 82 	// we've already loaded mail packages so the real ones don't get loaded as well.
 83 	AjxDispatcher.setLoaded("MailCore", true);
 84 	AjxDispatcher.setLoaded("Mail", true);
 85 
 86 	var winOpener = window.opener || window;
 87 
 88 	if (!window.parentController) {
 89 		window.parentController = winOpener._zimbraMail;
 90 	}
 91 
 92 	// Create the global app context
 93 	window.appCtxt = new ZmAppCtxt();
 94 	window.appCtxt.isChildWindow = true;
 95 
 96 	// XXX: DO NOT MOVE THIS LINE
 97 	// redefine ZmSetting from parent window since it loses this info.
 98 	window.parentAppCtxt = winOpener.appCtxt;
 99 	appCtxt.setSettings(parentAppCtxt.getSettings());
100 	appCtxt.isOffline = parentAppCtxt.isOffline;
101 	appCtxt.multiAccounts = parentAppCtxt.multiAccounts;
102 	window.ZmSetting = winOpener.ZmSetting;
103 
104 	ZmOperation.initialize();
105 	ZmApp.initialize();
106 
107 	var shell = new DwtShell({className:"MainShell"});
108 	window.onbeforeunload = ZmNewWindow._confirmExitMethod;
109 	appCtxt.setShell(shell);
110 
111 	// create new window and Go!
112 	var newWindow = new ZmNewWindow();
113     newWindow.startup();
114 };

116 /**
117 * Allows this child window to inform parent it's going away
118 */
119 ZmNewWindow.unload =
120 function(ev) {
121 	if (!window.opener || !window.parentController) { return; }
122 
123 	var command = window.newWindowCommand; //bug 54409 - was using wrong attribute for command in unload
124 	if (command == "compose" || command == "composeDetach"
125 			|| (command == "msgViewDetach" && appCtxt.composeCtlrSessionId)) { //msgViewDetach might turn into a compose session if user hits "reply"/etc
126 		// compose controller adds listeners to parent window's list so we
127 		// need to remove them before closing this window!
128 		var cc = AjxDispatcher.run("GetComposeController", appCtxt.composeCtlrSessionId);
129 		if (cc) {
130 			cc.dispose();
131 		}
132 	}
133 
134 	if (command == "msgViewDetach") {
135 		// msg controller (as a ZmListController) adds listener to tag list
136 		var mc = AjxDispatcher.run("GetMsgController", appCtxt.msgCtlrSessionId);
137 		if (mc) {
138 			mc.dispose();
139 		}
140 	}
141 
142 	if (window.parentController) {
143 		window.parentController.removeChildWindow(window);
144 	}
145 };

147 /**
148  * Presents a view based on a command passed through the window object. Possible commands are:
149  *
150  * <ul>
151  * <li><b>compose</b> compose window launched in child window</li>
152  * <li><b>composeDetach</b> compose window detached from client</li>
153  * <li><b>msgViewDetach</b> msg view detached from client</li>
154  * </ul>
155  * 
156  */
157 ZmNewWindow.prototype.startup =
158 function() {
159 
160 	// get params from parent window b/c of Safari bug #7162
161 	if (window.parentController) {
162 		var childWinObj = window.parentController.getChildWindow(window);
163 		if (childWinObj) {
164 			window.newWindowCommand = childWinObj.command;
165 			window.newWindowParams = childWinObj.params;
166 		}
167 	}
168 
169 	var cmd = window.newWindowCommand;
170 	var params = window.newWindowParams;
171 	if (cmd == "shortcuts") {
172 		var apps = {};
173 		apps[ZmApp.PREFERENCES] = true;
174 		this._createEnabledApps(apps);
175 		this._createView();
176 		return;
177 	}
178 
179 	DBG.println(AjxDebug.DBG1, " ************ Hello from new window!");
180 
181 	if (!this._appViewMgr) {
182 		this._appViewMgr = new ZmAppViewMgr(this._shell, this, true, false);
183 		this._statusView = new ZmStatusView(this._shell, "ZmStatus", Dwt.ABSOLUTE_STYLE, ZmId.STATUS_VIEW);
184 	}
185 
186 	var rootTg = appCtxt.getRootTabGroup();
187 
188 	var apps = {};
189 	apps[ZmApp.MAIL] = true;
190 	apps[ZmApp.CONTACTS] = true;
191 	// only load calendar app if we're dealing with an invite
192 	var msg = (cmd == "msgViewDetach") ? params.msg : null;
193 	if (msg && msg.isInvite()) {
194 		apps[ZmApp.CALENDAR] = true;
195 	}
196 	apps[ZmApp.PREFERENCES] = true;
197     apps[ZmApp.BRIEFCASE] = true;  //Need this for both Compose & Msg View detach window.
198 	this._createEnabledApps(apps);
199 
200 	// inherit parent's identity collection
201 	var parentPrefsApp = parentAppCtxt.getApp(ZmApp.MAIL);
202 	appCtxt.getApp(ZmApp.MAIL)._identityCollection = parentPrefsApp.getIdentityCollection();
203 
204 	// Find target first.
205 	var target;
206 	if (cmd == "compose" || cmd == "composeDetach") {
207 		target = "compose-window";
208 	} else if (cmd == "msgViewDetach") {
209 		target = "view-window";
210 	}
211     
212 	// setup zimlets, Load it first becoz.. zimlets has to get processed first.
213 	if (target) {
214 		var allzimlets = parentAppCtxt.get(ZmSetting.ZIMLETS);
215 		allzimlets = allzimlets || [];
216 		var zimletArray = this._settings._getCheckedZimlets(allzimlets);
217 		if (this._hasZimletsForTarget(zimletArray, target)) {
218 			var zimletMgr = appCtxt.getZimletMgr();
219 			var userProps = this._getUserProps();
220 			var createViewCallback =  new AjxCallback(this, this._createView);
221             appCtxt.setZimletsPresent(true);
222 			zimletMgr.loadZimlets(zimletArray, userProps, target, createViewCallback, true);
223 			return;
224 		}
225 	}
226 
227 	this._createView();
228 };

230 /**
231  * @private
232  */
233 ZmNewWindow.prototype._createView =
234 function() {
235 
236 	var cmd = window.newWindowCommand;
237 	var params = window.newWindowParams;
238 
239 	var rootTg = appCtxt.getRootTabGroup();
240 	var startupFocusItem;
241 
242 	//I null composeCtlrSessionId so it's not kept from irrelevant sessions from parent window.
243 	// (since I set it in every compose session, in ZmMailApp.prototype.compose).
244 	// This is important in case of cmd == "msgViewDetach"
245 	appCtxt.composeCtlrSessionId = null;  
246 	// depending on the command, do the right thing
247 	if (cmd == "compose" || cmd == "composeDetach") {
248 		var cc = AjxDispatcher.run("GetComposeController");	// get a new compose ctlr
249 		appCtxt.composeCtlrSessionId = cc.sessionId;
250 		if (params.action == ZmOperation.REPLY_ALL) {
251 			params.msg = this._deepCopyMsg(params.msg);
252 		}
253 		if (cmd == "compose") {
254 			cc._setView(params);
255 		} else {
256 			AjxDispatcher.require(["CalendarCore"]);
257 			var op = params.action || ZmOperation.NEW_MESSAGE;
258 			if (params.msg && params.msg._mode) {
259 				switch (params.msg._mode) {
260 					case ZmAppt.MODE_DELETE:
261 					case ZmAppt.MODE_DELETE_INSTANCE:
262 					case ZmAppt.MODE_DELETE_SERIES: {
263 						op = ZmOperation.REPLY_CANCEL;
264 						break;
265 					}
266 				}
267 			}
268 			params.action = op;
269 			cc._setView(params);
270 			cc._composeView.setDetach(params);
271 
272 			// bug fix #5887 - get the parent window's compose controller based on its session ID
273 			var parentCC = window.parentController.getApp(ZmApp.MAIL).getComposeController(params.sessionId);
274 			if (parentCC && parentCC._composeView) {
275 				// once everything is set in child window, pop parent window's compose view
276 				parentCC._composeView.reset(true);
277 				parentCC._app.popView(true);
278 			}
279 		}
280 		cc._setComposeTabGroup();
281 		rootTg.addMember(cc.getTabGroup());
282 		startupFocusItem = cc._getDefaultFocusItem();
283 
284 		target = "compose-window";
285 	} else if (cmd == "msgViewDetach") {
286 		//bug 52366 - not sure why only REPLY_ALL causes the problem (and not REPLY for example), but in this case the window is opened first for view. But
287 		//the user might of course click "reply to all" later in the window so I deep copy here in any case.
288 		params.msg = this._deepCopyMsg(params.msg);
289 		
290 		var msgController = AjxDispatcher.run("GetMsgController");
291 		appCtxt.msgCtlrSessionId = msgController.sessionId;
292 		msgController.show(params.msg, params.mode);
293 		rootTg.addMember(msgController.getTabGroup());
294 		startupFocusItem = msgController.getCurrentView();
295 
296 		target = "view-window";
297 	} else if (cmd == "shortcuts") {
298 		var panel = appCtxt.getShortcutsPanel();
299 		panel.popup(params.cols);
300 	}
301 
302 	var kbMgr = appCtxt.getKeyboardMgr();
303 	kbMgr.setTabGroup(rootTg);
304 	kbMgr.grabFocus(startupFocusItem);
305 };

307 /**
308  * HACK: This should go away once we have a cleaner server solution that
309  *       allows us to get just those zimlets for the specified target.
310  *       
311  * @private
312  */
313 ZmNewWindow.prototype._hasZimletsForTarget =
314 function(zimletArray, target) {
315 	var targetRe = new RegExp("\\b"+target+"\\b");
316 	for (var i=0; i < zimletArray.length; i++) {
317 		var zimletObj = zimletArray[i];
318 		var zimlet0 = zimletObj.zimlet[0];
319 		if (targetRe.test(zimlet0.target || "main")) {
320 			return true;
321 		}
322 	}
323 	return false;
324 };

326 /**
327  * @private
328  */
329 ZmNewWindow.prototype._getUserProps =
330 function() {
331 	var userPropsArray = parentAppCtxt.get(ZmSetting.USER_PROPS);
332 
333 	// default to original user props
334 	userPropsArray = userPropsArray ? [].concat(userPropsArray) : [];
335 
336 	// current user props take precedence, if available
337 	var zimletHash = parentAppCtxt.getZimletMgr().getZimletsHash();
338 	var zimletArray = parentAppCtxt.get(ZmSetting.ZIMLETS);
339 	for (var i = 0; i < zimletArray.length; i++) {
340 		var zname = zimletArray[i].zimlet[0].name;
341 		var zimlet = zimletHash[zname];
342 		if (!zimlet || !zimlet.userProperties) continue;
343 		for (var j = 0; j < zimlet.userProperties.length; j++) {
344 			var userProp = zimlet.userProperties[j];
345 			var userPropObj = { zimlet: zname, name: userProp.name, _content: userProp.value };
346 			userPropsArray.push(userPropObj);
347 		}
348 	}
349 
350 	// return user properties
351 	return userPropsArray;
352 };

366 /**
367  * Sends the server requests to the main controller.
368  * 
369  * @param	{Hash}	params		a hash of parameters
370  */
371 ZmNewWindow.prototype.sendRequest =
372 function(params) {
373     // reset onbeforeunload on send
374     window.onbeforeunload = null;
375 	// bypass error callback to get control over exceptions in the childwindow.
376 	params.errorCallback = new AjxCallback(this, this._handleException, [( params.errorCallback || null )]);
377 	return window.parentController ? window.parentController.sendRequest(params) : null;
378 };

380 /**
381  * @private
382  */
383 ZmNewWindow.prototype._handleException =
384 function(errCallback, ex) {
385 	var handled = false;
386 	if (errCallback) {
387 		handled = errCallback.run(ex);
388 	}
389 	if (!handled) {
390 		ZmController.prototype._handleException.apply(this, [ex]);
391 	}
392 	return true;
393 };

395 /**
396  * Popup the error dialog.
397  * 
398  * @param	{String}	msg		the message
399  * @param	{AjxException}	ex	the exception
400  * @param	{Boolean}	noExecReset
401  * @param	{Boolean}	hideReportButton
402  */
403 ZmNewWindow.prototype.popupErrorDialog =
404 function(msg, ex, noExecReset, hideReportButton)  {
405 	// Since ex is from parent window, all the types seems like objects, so need
406 	// to filter the functions
407 	var detailStr;
408 	if (ex instanceof Object || typeof ex == "object") {
409 		var details = [];
410 		ex.msg = ex.msg || msg;
411 		for (var prop in ex) {
412 			if (typeof ex[prop] == "function" ||
413 				(typeof ex[prop] == "object" && ex[prop].apply && ex[prop].call))
414 			{
415 				continue;
416 			}
417 			details.push([prop, ": ", ex[prop], "<br/>\n"].join(""));
418 		}
419 		detailStr = details.join("");
420 	}
421 	ZmController.prototype.popupErrorDialog.call(this, msg, ( detailStr || ex ), noExecReset, hideReportButton);
422 };

513 /**
514  * Instantiates enabled apps. An optional argument may be given limiting the set
515  * of apps that may be created.
516  *
517  * @param {Hash}	apps	the set of apps to create
518  * 
519  * @private
520  */
521 ZmNewWindow.prototype._createEnabledApps =
522 function(apps) {
523 	for (var app in ZmApp.CLASS) {
524 		if (!apps || apps[app]) {
525 			ZmApp.APPS.push(app);
526 		}
527 	}
528 	ZmApp.APPS.sort(function(a, b) {
529 		return ZmZimbraMail.hashSortCompare(ZmApp.LOAD_SORT, a, b);
530 	});
531 
532 	// instantiate enabled apps - this will invoke app registration
533 	for (var i = 0; i < ZmApp.APPS.length; i++) {
534 		var app = ZmApp.APPS[i];
535 		if (app != ZmApp.IM) { // Don't create im app. Seems like the safest way to avoid ever logging in.
536 			var setting = ZmApp.SETTING[app];
537 			if (!setting || appCtxt.get(setting)) {
538 				this._createApp(app);
539 			}
540 		}
541 	}
542 };

556 /**
557  * @private
558  */
559 ZmNewWindow.prototype._deepCopyMsg =
560 function(msg) {
561 	// initialize new ZmSearch if applicable
562 	var newSearch = null;
563 	var oldSearch = msg.list.search;
564 
565 	if (oldSearch) {
566 		newSearch = new ZmSearch();
567 
568 		for (var i in oldSearch) {
569 			if ((typeof oldSearch[i] == "object") || (typeof oldSearch[i] == "function")) { continue; }
570 			newSearch[i] = oldSearch[i];
571 		}
572 
573 		// manually add objects since they are no longer recognizable
574 		newSearch.types = new AjxVector();
575 		var types = oldSearch.types.getArray();
576 		for (var i in types) {
577 			newSearch.types.add(types[i]);
578 		}
579 	}
580 
581 	// initialize new ZmMailList
582 	var newMailList = new ZmMailList(msg.list.type, newSearch);
583 	for (var i in msg.list) {
584 		if ((typeof msg.list[i] == "object") || (typeof msg.list[i] == "function")) { continue; }
585 		newMailList[i] = msg.list[i];
586 	}
587 
588 	// finally, initialize new ZmMailMsg
589 	var newMsg = new ZmMailMsg(msg.id, newMailList);
590 
591 	for (var i in msg) {
592 		if ((typeof msg[i] == "object") || (typeof msg[i] == "function")) { continue; }
593 		newMsg[i] = msg[i];
594 	}
595 
596 	// manually add any objects since they are no longer recognizable
597 	for (var i in msg._addrs) {
598 		var addrs = msg._addrs[i].getArray();
599 		for (var j in addrs) {
600 			newMsg._addrs[i].add(addrs[j]);
601 		}
602 	}
603 
604 	if (msg.attachments && msg.attachments.length > 0) {
605 		for (var i in msg.attachments) {
606 			newMsg.attachments.push(msg.attachments[i]);
607 		}
608 	}
609 
610 	for (var i in msg._bodyParts) {
611 		newMsg._bodyParts.push(msg._bodyParts[i]);
612 	}
613 
614 	if (msg._topPart) {
615 		newMsg._topPart = new ZmMimePart();
616 		for (var i in msg._topPart) {
617 			if ((typeof msg._topPart[i] == "object") || (typeof msg._topPart[i] == "function"))
618 				continue;
619 			newMsg._topPart[i] = msg._topPart[i];
620 		}
621 		var children = msg._topPart.children.getArray();
622 		for (var i in children) {
623 			newMsg._topPart.children.add(children[i]);
624 		}
625 	}
626 
627 	ZmZimbraMail.prototype._registerOrganizers.call(this);
628 	ZmZimbraMail.registerViewsToTypeMap();
629 
630 
631 	if (msg.invite) {
632 		newMsg.invite = msg.invite;
633 	}
634 
635 	if (msg.share) {
636 		newMsg.share = msg.share;
637 	}
638 
639 	return newMsg;
640 };

645 /**
646  * @private
647  */
648 ZmNewWindow._confirmExitMethod =
649 function(ev) {
650 	if (window.parentController &&
651 		(window.newWindowCommand == "compose" || window.newWindowCommand == "composeDetach"))
652 	{
