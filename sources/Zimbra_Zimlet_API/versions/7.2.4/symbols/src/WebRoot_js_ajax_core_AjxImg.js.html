 38 /**
 39  * This method will set the image for <i>parentEl</i>. <i>parentEl</i> should 
 40  * only contain this image and no other children
 41  *
 42  * @param parentEl 		the parent element for the image
 43  * @param imageName 		the name of the image.  The CSS class for the image will be "Img<imageName>".
 44  * @param useParenEl 	if <code>true</code> will use the parent element as the root for the image and will not create an intermediate DIV
 45  * @param _disabled		if <code>true</code>, will append " ZDisabledImage" to the CSS class for the image, 
 46  *							which will make the image partly transparent
 47  */
 48 AjxImg.setImage =
 49 function(parentEl, imageName, useParentEl, _disabled) {
 50 	var origImageName = imageName;
 51     var color, m = imageName && imageName.match(AjxImg.RE_COLOR);
 52 	if (m) {
 53 		imageName = m && m[1];
 54 		color = m && m[2];
 55 	}
 56 
 57 	var className = AjxImg.getClassForImage(imageName, _disabled);
 58 	if (useParentEl) {
 59 		parentEl.className = className;
 60 		return;
 61 	}
 62 	var id = parentEl.firstChild && parentEl.firstChild.id;
 63         
 64 	var overlayName = className+"Overlay";
 65 	var maskName = className+"Mask";
 66 	if (color && window.AjxImgData && AjxImgData[overlayName] && AjxImgData[maskName]) {
 67 		color = (color.match(/^\d$/) ? ZmOrganizer.COLOR_VALUES[color] : color) ||
 68 				ZmOrganizer.COLOR_VALUES[ZmOrganizer.ORG_DEFAULT_COLOR];
 69 		parentEl.innerHTML = AjxImg.getImageHtml(origImageName, null, id ? "id='"+id+"'" : null, false, _disabled);
 70 		return;
 71 	}
 72 
 73 	if (parentEl.firstChild == null || parentEl.firstChild.nodeName.toLowerCase() != "div") {
 74 		var html = [], i = 0;
 75 		html[i++] = "<div";
 76 		if (id) {
 77 			html[i++] = " id='";
 78 			html[i++] = id;
 79 			html[i++] = "'";
 80 		}
 81 		if (className) {
 82 			html[i++] = " class='";
 83 			html[i++] = className;
 84 			html[i++] = "'";
 85 		}
 86 		html[i++] = "></div>";
 87 		parentEl.innerHTML = html.join("");
 88 		return;
 89 	} else if (AjxEnv.isIE) {
 90 		parentEl.firstChild.innerHTML = "";
 91 	}
 92 
 93 	parentEl.firstChild.className = className;
 94 };
 95 
 96 AjxImg.setDisabledImage = function(parentEl, imageName, useParentEl) {
 97 	return AjxImg.setImage(parentEl, imageName, useParentEl, true);
 98 };

100 AjxImg.getClassForImage =
101 function(imageName, disabled) {
102 	var className = imageName ? "Img" + imageName : "";
103 	if (disabled) className += " ZDisabledImage";
104 	return className;
105 };
106 
107 AjxImg.getImageClass =
108 function(parentEl) {
109 	return parentEl.firstChild ? parentEl.firstChild.className : parentEl.className;
110 };

112 AjxImg.getImageElement =
113 function(parentEl) {
114 	return parentEl.firstChild ? parentEl.firstChild : parentEl;
115 };

117 AjxImg.getParentElement =
118 function(imageEl) {
119 	return imageEl.parentNode;
120 };

122 /**
123  * Gets the "image" as an HTML string. 
124  *
125  * @param imageName		the image you want to render
126  * @param styleStr		optional style info (for example, "display:inline")
127  * @param attrStr		optional attributes (for example, "id=X748")
128  * @param wrapInTable	surround the resulting code in a table
129  * @return	{string}	the image string
130  */
131 AjxImg.getImageHtml = 
132 function(imageName, styleStr, attrStr, wrapInTable, _disabled) {
133     styleStr = styleStr || "";
134 	attrStr = attrStr || "";
135 	var pre = wrapInTable ? "<table style='display:inline' cellpadding=0 cellspacing=0 border=0><tr><td align=center valign=bottom>" : "";
136     var html = "";
137 	var post = wrapInTable ? "</td></tr></table>" : "";
138 	if (imageName) {
139         var color, m = imageName.match(AjxImg.RE_COLOR);
140         if (m) {
141             imageName = m && m[1];
142             color = m && m[2];
143         }
144 
145         var className = AjxImg.getClassForImage(imageName, _disabled);
146         var overlayName = className+"Overlay";
147         var maskName = className+"Mask";
148         if (color && window.AjxImgData && AjxImgData[overlayName] && AjxImgData[maskName]) {
149             color = (color.match(/^\d$/) ? ZmOrganizer.COLOR_VALUES[color] : color) ||
150                     ZmOrganizer.COLOR_VALUES[ZmOrganizer.ORG_DEFAULT_COLOR];
151 
152             var overlay = AjxImgData[overlayName], mask = AjxImgData[maskName];
153             if (AjxEnv.isIE && !AjxEnv.isIE9up) {
154                 var clip = "";
155                 var size = [
156                     "width:",overlay.w,";",
157                     "height:",overlay.h,";"
158                 ].join("");
159                 var location = [
160                     "top:",mask.t,";",
161                     "left:",mask.l,";"
162                 ].join("");
163                 if(typeof document.documentMode != 'undefined'){ //IE8 is the first one to define this. IE8 can lie when in compat mode, so we need to really know it's it.
164                     clip = [
165                         'clip:rect(',
166                         (-1*mask.t)-1,'px,',
167                         overlay.w-1,'px,',
168                         (mask.t*-1)+overlay.h-1,'px,',
169                         overlay.l,'px);'
170                     ].join('');
171                 }
172                 var filter = 'filter:mask(color='+color+');';
173                 html = [
174                     // NOTE: Keep in sync with output of ImageMerger.java.
175                     "<div class='IEImage' style='display:inline-block;position:relative;overflow:hidden;",size,styleStr,"' ",attrStr,">",
176                         "<div class='IEImageMask' style='overflow:hidden;position:relative;",size,"'>",
177                             "<img src='",mask.f,"?v=",cacheKillerVersion,"' border=0 style='position:absolute;",location,clip,filter,"'>",
178                         "</div>",
179                         "<div class='IEImageOverlay ",overlayName,"' style='",size,";position:absolute;top:0;left:0;'></div>",
180                     "</div>"
181                 ].join("");
182             }
183 			else if (AjxEnv.isIE9up) {
184 				color = color.replace("#","");
185 				var className = AjxImg.getClassForImage(imageName + "_" + color, _disabled);
186 				html = [
187 					"<div class='", "Img", imageName + "_" + color, "'", styleStr, attrStr, "></div>"
188 				].join("");
189 			}
190             else {
191                 if (!overlay[color]) {
192                     var width = overlay.w, height = overlay.h;
193 
194                     var canvas = document.createElement("CANVAS");
195                     canvas.width = width;
196                     canvas.height = height;
197 
198                     var ctx = canvas.getContext("2d");
199 
200                     ctx.save();
201                     ctx.clearRect(0,0,width,height);
202 
203                     ctx.save();
204 	                var imgId = attrStr;
205 	                if (!imgId) {
206 		                imgId = Dwt.getNextId("CANVAS_IMG_");  //create an imgId in case we need to update the img.src for an element without an id
207 		                attrStr = "id='" + imgId + "'";
208 	                }
209 	                else {
210 		                var match = attrStr.match(/id=[\"\']([^\"\']+)[\"\']+/);
211 		                if (match && match.length > 1) {
212 			                imgId = match[1]; //extract the ID value
213 		                }
214 		                AjxDebug.println(AjxDebug.TAG_ICON, "imgId = " + imgId);
215 	                }
216 	                var maskElement = document.getElementById(maskName);
217 	                var overlayElement = document.getElementById(overlayName);
218 	                if (!maskElement.complete || !overlayElement.complete) {
219 		                AjxDebug.println(AjxDebug.TAG_ICON, "mask status = " + maskElement.complete + " for " + imgId);
220 		                AjxDebug.println(AjxDebug.TAG_ICON, "overlay status = " + overlayElement.complete + " for " + imgId);
221 						var maskImg = new Image();
222 						maskImg.onload = function() {
223 							AjxDebug.println(AjxDebug.TAG_ICON, "mask image loaded");
224 							var overlayImg = new Image();
225 							overlayImg.onload = function() {
226 								AjxImg._drawCanvasImage(ctx, maskImg, overlayImg, mask, overlay, color, width, height)
227 								AjxDebug.println(AjxDebug.TAG_ICON, "overlay image loaded");
228 								var el = document.getElementById(imgId);
229 								if (el) {
230 									AjxDebug.println(AjxDebug.TAG_ICON, "element found for id = " + imgId);
231 									el.src = canvas.toDataURL();
232 									overlay[color] = canvas.toDataURL(); //only save if successful
233 								}
234 								else {
235 									AjxDebug.println(AjxDebug.TAG_ICON, "no element found for id = " + imgId);
236 								}
237 							}
238 							overlayImg.src = document.getElementById(overlayName).src;
239 	                    }
240 	                    maskImg.src = document.getElementById(maskName).src;
241 	                }
242 	                else {
243 		                //image already downloaded
244 		                AjxImg._drawCanvasImage(ctx, maskElement, overlayElement, mask, overlay, color, width, height);
245 		                overlay[color] = canvas.toDataURL();
246 	                }
247                 }
248 
249                 html = [
250                     "<img src='",overlay[color],"' border=0 style='",styleStr,"' ",attrStr,">"
251                 ].join("");
252             }
253         }
254         else {
255             html = [
256                 "<div class='", "Img", imageName, "' style='", styleStr, "' ", attrStr, "></div>"
257             ].join("");
258         }
259 	}
260     else {
261         html = [
262             "<div style='", styleStr, "' ", attrStr, "></div>"
263         ].join("");
264     }
265 	return pre || post ? [pre,html,post].join("") : html;
266 };

268 /**
269  * Gets the "image" as an HTML string.
270  *
271  * @param imageName		the image you want to render
272  * @param styleStr		optional style info (for example, "display:inline")
273  * @param attrStr		optional attributes (for example, "id=X748")
274  * @param label			the text that follows this image
275  * @return	{string}	the image string
276  */
277 AjxImg.getImageSpanHtml =
278 function(imageName, styleStr, attrStr, label) {
279 	var className = AjxImg.getClassForImage(imageName);
280 
281 	var html = [
282         "<span style='white-space:nowrap'>",
283         "<span class='inlineIcon'>",
284         AjxImg.getImageHtml(imageName, styleStr, attrStr),
285         (label || ""),
286         "</span>",
287         "</span>"
288     ];
289 
290 	return html.join("");
291 };

293 /**
294  * Helper method to draw the image using both the mask image and the overlay image
295  * 
296  * @param ctx  {Object} canvas context
297  * @param maskImg   {HtmlElement} mask image object
298  * @param overlayImg {HtmlElement} overlay image object
299  * @param mask  {Object} mask object
300  * @param overlay {Object} overlay object
301  * @param color {String} color for fill
302  * @param width {int} width
303  * @param height {int} height
304  * 
305  * @private
306  */
307 AjxImg._drawCanvasImage = 
308 function(ctx, maskImg, overlayImg, mask, overlay, color, width, height) {
309 	ctx.drawImage(maskImg, mask.l, mask.t);
310 	ctx.globalCompositeOperation = "source-out";
311 	ctx.fillStyle = color;
312 	ctx.fillRect(0, 0, width, height);
313 	ctx.restore();
314 	ctx.drawImage(overlayImg, overlay.l, overlay.t);
315 	ctx.restore();	
316 };
