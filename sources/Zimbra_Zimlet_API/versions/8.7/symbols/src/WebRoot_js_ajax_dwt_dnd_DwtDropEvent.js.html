 24 /**
 25  * @constructor
 26  * @class
 27  * DwtDropEvent is generated by the Drag and Drop framework when a drag n drop operation is
 28  * in process. The drop event is dispatched to the registered {@link DwtDropTarget} instance.
 29  * 
 30  * @author Ross Dargahi
 31  * 
 32  * @see DwtDropTarget
 33  */
 34 DwtDropEvent = function() {
 35 	/**
 36 	 * Type of drag operation. One of:
 37 	 * <ul>
 38 	 * <li>{@link DwtDragEvent.DRAG_START}</li>
 39 	 * <li>{@link DwtDragEvent.SET_DATA}</li>
 40 	 * <li>{@link DwtDragEvent.DRAG_END}</li>
 41 	 * </ul>
 42 	 */
 43 	this.operation = null;
 44 	
 45 	/**
 46 	 * Drop target control.
 47 	 * @type DwtControl
 48 	 * */	
 49 	this.targetControl = null;
 50 	
 51 	/**
 52 	 * Action being performed. One of:
 53 	 * <ul>
 54 	 * <li>{@link Dwt.DND_DROP_NONE}</li>
 55 	 * <li>{@link Dwt.DND_DROP_COPY}</li>
 56 	 * <li>{@link Dwt.DND_DROP_MOVE}</li>
 57 	 * </ul>
 58 	 */
 59 	this.action = null;
 60 	
 61 	/**
 62 	 * Drag source data. This is the application data associated with the item being dragged.
 63 	 */
 64 	this.srcData = null;
 65 
 66 	/**
 67 	 * Whether the DnD framework should perform the operation. The application is
 68 	 * responsible for setting this value based on whatever business logic it is
 69 	 * implementing.
 70 	 * @type boolean
 71 	 */
 72 	this.doIt = false;
 73 }
 74 
 75 /**
 76  * A draggable object has entered the drop target.
 77  */
 78 DwtDropEvent.DRAG_ENTER = 1;
 79 
 80 /**
 81  * A draggable object has left the drop target.
 82  */
 83 DwtDropEvent.DRAG_LEAVE = 2;
 84 
 85 /**
 86  * Drag operation has changed e.g. from move to copy.
 87  */
 88 DwtDropEvent.DRAG_OP_CHANGED = 3;
 89 
 90 /**
 91  * A draggable object has been dropped on the drag target.
 92  */
 93 DwtDropEvent.DRAG_DROP = 4;
