 16 /**
 17  * Creates an identity.
 18  * @class
 19  * This class represents an identity.
 20  * 
 21  * @param	{String}	name		the identity name
 22  * 
 23  */
 24 ZmIdentity = function(name) {
 25 
 26 	this.reset();
 27 	this.name = name;
 28 	this.id = "";
 29 };
 30

 32 function() {
 33 	return "ZmIdentity";
 34 };

103 /**
104  * Creates the identity.
105  * 
106  * @param	{AjxCallback}		callback		the callback

139 /**
140  * Clears this identity's fields.
141  */
142 ZmIdentity.prototype.reset =
143 function() {
144 	for (var field in ZmIdentity.FIELDS) {
145 		var props = ZmIdentity.FIELDS[field];
146 		switch (props.type) {
147 			case ZmSetting.D_STRING:	this[props.name] = "";		break;
148 			case ZmSetting.D_BOOLEAN:	this[props.name] = false;	break;
149 			case ZmSetting.D_LIST:		this[props.name] = [];		break;
150 		}
151 	}
152 };

156 ZmIdentity.prototype._doRequest =
157 function(requestType, respFunction, callback, errorCallback, batchCmd) {
158 
159 	var soapDoc = AjxSoapDoc.create(requestType + "IdentityRequest", "urn:zimbraAccount");
160 	var identityNode = soapDoc.set("identity");
161 
162 	var name = this.isDefault ? ZmIdentity.DEFAULT_NAME : this.name;
163 	if (requestType != "Create" && this.id !== "") {
164 		identityNode.setAttribute("id", this.id);
165 	}
166 	else {
167 		identityNode.setAttribute("name", this.name);
168 	}
169 	if (requestType != "Delete") {
170 		for (var i in ZmIdentity.FIELDS) {
171 			var field = ZmIdentity.FIELDS[i];
172 			if (this.hasOwnProperty(field.name)) {
173 				var value = this.getField(i);
174 				if (field.type == ZmSetting.D_LIST) {
175 					for (var j = 0, count = value.length; j < count; j++) {
176 						if (value[j]) {
177 							var propertyNode = soapDoc.set("a", value[j], identityNode);
178 							propertyNode.setAttribute("name", field.soap);
179 						}
180 					}
181 				} else {
182 					if (field.type == ZmSetting.D_BOOLEAN) {
183 						value = value ? "TRUE" : "FALSE";
184 					}
185 					var isSignature = (i == ZmIdentity.SIGNATURE || i == ZmIdentity.REPLY_SIGNATURE);
186 					var isDisplayName = (i == ZmIdentity.SEND_FROM_DISPLAY || i == ZmIdentity.SET_REPLY_TO_DISPLAY);
187 					var isEmailAddress = (i == ZmIdentity.SET_REPLY_TO_ADDRESS);
188 					if (value || isSignature || isDisplayName || isEmailAddress) {
189 						var propertyNode = soapDoc.set("a", value, identityNode);
190 						propertyNode.setAttribute("name", field.soap);
191 					}
192 				}
193 			}
194 		}
195 	}
196 
197 	var respCallback = new AjxCallback(this, respFunction, [callback]);
198 	if (batchCmd) {
199 		batchCmd.addNewRequestParams(soapDoc, respCallback, errorCallback);
200 		return;
201 	}
202 
203 	var params = {
204 		soapDoc: soapDoc,
205 		asyncMode: Boolean(callback),
206 		callback: respCallback,
207 		errorCallback: errorCallback
208 	};
209 
210 	return appCtxt.getAppController().sendRequest(params);
211 };

213 ZmIdentity.prototype._loadFromDom =
214 function(data) {
215 
216 	this.id = data.id;
217 
218     var props = data._attrs;
219 	if (props) {
220 		for (var i in props) {
221 			var field = ZmIdentity._SOAP[i];
222 			if (field) {
223 				var value = props[i];
224 				if (field.type == ZmSetting.D_BOOLEAN) {
225 					this[field.name] = (value.toString().toUpperCase() == "TRUE");
226 				}
227 				else if (field.type == ZmSetting.D_LIST) {
228 					this[field.name] = AjxUtil.isArray(value) ? value : [value];
229 				}
230 				else {
231 					this[field.name] = value;
232 				}
233 			}
234 		}
235 	}
236 
237     if (data.name) {
238 		if (data.name == ZmIdentity.DEFAULT_NAME) {
239 			this.isDefault = true;
240         }
241 	}
242 };

244 ZmIdentity.prototype._handleCreateResponse =
245 function(callback, result, response) {
246 
247 	this.id = response.identity[0].id;
248 	delete this._new;
249 	delete this._dirty;
250 
251 	var collection = appCtxt.getIdentityCollection();
252 	collection.add(this);
253 	collection._notify(ZmEvent.E_CREATE, { item: this } );
254 
255 	if (callback) {
256 		callback.run(this, result);
257 	}
258 };
259 
260 ZmIdentity.prototype._handleSaveResponse =
261 function(callback, result, response) {
262 
263 	delete this._dirty;
264 
265 	var collection = appCtxt.getIdentityCollection();
266 	collection.remove(this);
267 	collection.add(this);
268 	collection._notify(ZmEvent.E_MODIFY, { item: this } );
269 
270 	if (callback) {
271 		callback.run(this, result);
272 	}
273 };
274 
275 ZmIdentity.prototype._handleDeleteResponse =
276 function(callback, result, response) {
277 
278 	var collection = appCtxt.getIdentityCollection();
279 	collection.remove(this);
280 	collection._notify(ZmEvent.E_DELETE, { item: this } );
281 
282 	if (callback) {
283 		callback.run(this, result);
284 	}
285 };
