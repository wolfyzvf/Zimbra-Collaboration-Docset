 29 	/**
 30 	 * This class enables you to add custom editor commands and it contains
 31 	 * overrides for native browser commands to address various bugs and issues.
 32 	 *
 33 	 * @class tinymce.EditorCommands
 34 	 */
 35 	tinymce.EditorCommands = function(editor) {
 36 		var dom = editor.dom,
 37 			selection = editor.selection,
 38 			commands = {state: {}, exec : {}, value : {}},
 39 			settings = editor.settings,
 40 			formatter = editor.formatter,
 41 			bookmark;

 43 		/**
 44 		 * Executes the specified command.
 45 		 *
 46 		 * @method execCommand
 47 		 * @param {String} command Command to execute.
 48 		 * @param {Boolean} ui Optional user interface state.
 49 		 * @param {Object} value Optional value for command.
 50 		 * @return {Boolean} true/false if the command was found or not.
 51 		 */
 52 		function execCommand(command, ui, value) {
 53 			var func;
 54 
 55 			command = command.toLowerCase();
 56 			if (func = commands.exec[command]) {
 57 				func(command, ui, value);
 58 				return TRUE;
 59 			}
 60 
 61 			return FALSE;
 62 		};
 63 
 64 		/**
 65 		 * Queries the current state for a command for example if the current selection is "bold".
 66 		 *
 67 		 * @method queryCommandState
 68 		 * @param {String} command Command to check the state of.
 69 		 * @return {Boolean/Number} true/false if the selected contents is bold or not, -1 if it's not found.
 70 		 */
 71 		function queryCommandState(command) {
 72 			var func;
 73 
 74 			command = command.toLowerCase();
 75 			if (func = commands.state[command])
 76 				return func(command);
 77 
 78 			return -1;
 79 		};
 80 
 81 		/**
 82 		 * Queries the command value for example the current fontsize.
 83 		 *
 84 		 * @method queryCommandValue
 85 		 * @param {String} command Command to check the value of.
 86 		 * @return {Object} Command value of false if it's not found.
 87 		 */
 88 		function queryCommandValue(command) {
 89 			var func;
 90 
 91 			command = command.toLowerCase();
 92 			if (func = commands.value[command])
 93 				return func(command);
 94 
 95 			return FALSE;
 96 		};
 97 
 98 		/**
 99 		 * Adds commands to the command collection.
100 		 *
101 		 * @method addCommands
102 		 * @param {Object} command_list Name/value collection with commands to add, the names can also be comma separated.
103 		 * @param {String} type Optional type to add, defaults to exec. Can be value or state as well.
104 		 */
105 		function addCommands(command_list, type) {
106 			type = type || 'exec';
107 
108 			each(command_list, function(callback, command) {
109 				each(command.toLowerCase().split(','), function(command) {
110 					commands[type][command] = callback;
111 				});
112 			});
113 		};

118 			queryCommandState : queryCommandState,
119 			queryCommandValue : queryCommandValue,
120 			addCommands : addCommands
121 		});

125 		function execNativeCommand(command, ui, value) {
126 			if (ui === undef)
127 				ui = FALSE;
128 
129 			if (value === undef)
130 				value = null;
131 
132 			return editor.getDoc().execCommand(command, ui, value);
133 		};

252

302 			mceInsertContent : function(command, ui, value) {
303 				var parser, serializer, parentNode, rootNode, fragment, args,
304 					marker, nodeRect, viewPortRect, rng, node, node2, bookmarkHtml, viewportBodyElement;
305 
306 				//selection.normalize();
307 
308 				// Setup parser and serializer
309 				parser = editor.parser;
310 				serializer = new tinymce.html.Serializer({}, editor.schema);
311 				bookmarkHtml = '<span id="mce_marker" data-mce-type="bookmark">\uFEFF</span>';
312 
313 				// Run beforeSetContent handlers on the HTML to be inserted
314 				args = {content: value, format: 'html'};
315 				selection.onBeforeSetContent.dispatch(selection, args);
316 				value = args.content;
317 
318 				// Add caret at end of contents if it's missing
319 				if (value.indexOf('{$caret}') == -1)
320 					value += '{$caret}';
321 
322 				// Replace the caret marker with a span bookmark element
323 				value = value.replace(/\{\$caret\}/, bookmarkHtml);
324 
325 				// Insert node maker where we will insert the new HTML and get it's parent
326 				if (!selection.isCollapsed())
327 					editor.getDoc().execCommand('Delete', false, null);
328 
329 				parentNode = selection.getNode();
330 
331 				// Parse the fragment within the context of the parent node
332 				args = {context : parentNode.nodeName.toLowerCase()};
333 				fragment = parser.parse(value, args);
334 
335 				// Move the caret to a more suitable location
336 				node = fragment.lastChild;
337 				if (node.attr('id') == 'mce_marker') {
338 					marker = node;
339 
340 					for (node = node.prev; node; node = node.walk(true)) {
341 						if (node.type == 3 || !dom.isBlock(node.name)) {
342 							node.parent.insert(marker, node, node.name === 'br');
343 							break;
344 						}
345 					}
346 				}
347 
348 				// If parser says valid we can insert the contents into that parent
349 				if (!args.invalid) {
350 					value = serializer.serialize(fragment);
351 
352 					// Check if parent is empty or only has one BR element then set the innerHTML of that parent
353 					node = parentNode.firstChild;
354 					node2 = parentNode.lastChild;
355 					if (!node || (node === node2 && node.nodeName === 'BR'))
356 						dom.setHTML(parentNode, value);
357 					else
358 						selection.setContent(value);
359 				} else {
360 					// If the fragment was invalid within that context then we need
361 					// to parse and process the parent it's inserted into
362 
363 					// Insert bookmark node and get the parent
364 					selection.setContent(bookmarkHtml);
365 					parentNode = editor.selection.getNode();
366 					rootNode = editor.getBody();
367 
368 					// Opera will return the document node when selection is in root
369 					if (parentNode.nodeType == 9)
370 						parentNode = node = rootNode;
371 					else
372 						node = parentNode;
373 
374 					// Find the ancestor just before the root element
375 					while (node !== rootNode) {
376 						parentNode = node;
377 						node = node.parentNode;
378 					}
379 
380 					// Get the outer/inner HTML depending on if we are in the root and parser and serialize that
381 					value = parentNode == rootNode ? rootNode.innerHTML : dom.getOuterHTML(parentNode);
382 					value = serializer.serialize(
383 						parser.parse(
384 							// Need to replace by using a function since $ in the contents would otherwise be a problem
385 							value.replace(/<span (id="mce_marker"|id=mce_marker).+?<\/span>/i, function() {
386 								return serializer.serialize(fragment);
387 							})
388 						)
389 					);
390 
391 					// Set the inner/outer HTML depending on if we are in the root or not
392 					if (parentNode == rootNode)
393 						dom.setHTML(rootNode, value);
394 					else
395 						dom.setOuterHTML(parentNode, value);
396 				}
397 
398 				marker = dom.get('mce_marker');
399 
400 				// Scroll range into view scrollIntoView on element can't be used since it will scroll the main view port as well
401 				nodeRect = dom.getRect(marker);
402 				viewPortRect = dom.getViewPort(editor.getWin());
403 
404 				// Check if node is out side the viewport if it is then scroll to it
405 				if ((nodeRect.y + nodeRect.h > viewPortRect.y + viewPortRect.h || nodeRect.y < viewPortRect.y) ||
406 					(nodeRect.x > viewPortRect.x + viewPortRect.w || nodeRect.x < viewPortRect.x)) {
407 					viewportBodyElement = tinymce.isIE ? editor.getDoc().documentElement : editor.getBody();
408 					viewportBodyElement.scrollLeft = nodeRect.x;
409 					viewportBodyElement.scrollTop = nodeRect.y - viewPortRect.h + 25;
410 				}
411 
412 				// Move selection before marker and remove it
413 				rng = dom.createRng();
414 
415 				// If previous sibling is a text node set the selection to the end of that node
416 				node = marker.previousSibling;
417 				if (node && node.nodeType == 3) {
418 					rng.setStart(node, node.nodeValue.length);
419 				} else {
420 					// If the previous sibling isn't a text node or doesn't exist set the selection before the marker node
421 					rng.setStartBefore(marker);
422 					rng.setEndBefore(marker);
423 				}
424 
425 				// Remove the marker node and set the new range
426 				dom.remove(marker);
427 				selection.setRng(rng);
428 
429 				// Dispatch after event and add any visual elements needed
430 				selection.onSetContent.dispatch(selection, args);
431 				editor.addVisual();
432 			},

447 			'Indent,Outdent' : function(command) {
448 				var intentValue, indentUnit, value;
449 
450 				// Setup indent level
451 				intentValue = settings.indentation;
452 				indentUnit = /[a-z%]+$/i.exec(intentValue);
453 				intentValue = parseInt(intentValue);
454 
455 				if (!queryCommandState('InsertUnorderedList') && !queryCommandState('InsertOrderedList')) {
456 					// If forced_root_blocks is set to false we don't have a block to indent so lets create a div
457 					if (!settings.forced_root_block && !dom.getParent(selection.getNode(), dom.isBlock)) {
458 						formatter.apply('div');
459 					}
460 
461 					each(selection.getSelectedBlocks(), function(element) {
462 						if (command == 'outdent') {
463 							value = Math.max(0, parseInt(element.style.paddingLeft || 0) - intentValue);
464 							dom.setStyle(element, 'paddingLeft', value ? value + indentUnit : '');
465 						} else
466 							dom.setStyle(element, 'paddingLeft', (parseInt(element.style.paddingLeft || 0) + intentValue) + indentUnit);
467 					});
468 				} else
469 					execNativeCommand(command);
470 			},

507 			mceInsertLink : function(command, ui, value) {
508 				var anchor;
509 
510 				if (typeof(value) == 'string')
511 					value = {href : value};
512 
513 				anchor = dom.getParent(selection.getNode(), 'a');
514 
515 				// Spaces are never valid in URLs and it's a very common mistake for people to make so we fix it here.
516 				value.href = value.href.replace(' ', '%20');
517 
518 				// Remove existing links if there could be child links or that the href isn't specified
519 				if (!anchor || !value.href) {
520 					formatter.remove('link');
521 				}		
522 
523 				// Apply new link to selection
524 				if (value.href) {
525 					formatter.apply('link', value, anchor);
526 				}
527 			},
