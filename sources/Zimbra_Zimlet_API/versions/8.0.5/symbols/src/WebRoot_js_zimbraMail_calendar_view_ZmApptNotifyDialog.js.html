 16 /**
 17 * Simple dialog allowing user to choose between an Instance or Series for an appointment
 18 * @constructor
 19 * @class
 20 *
 21 * @author Parag Shah
 22 * @param parent			the element that created this view
 23 * 
 24 * 
 25 * @extends		DwtDialog
 26 * @private
 27 */
 28 ZmApptNotifyDialog = function(parent) {
 29 
 30 	DwtDialog.call(this, {parent:parent});
 31 
 32 	this.setTitle(ZmMsg.sendUpdateTitle);
 33 	this.setContent(this._setHtml());
 34 	this._cacheFields();
 35 };
 36 
 37 ZmApptNotifyDialog.prototype = new DwtDialog;
 38 ZmApptNotifyDialog.prototype.constructor = ZmApptNotifyDialog;
 39 
 40 // Public methods
 41 
 42 ZmApptNotifyDialog.prototype.toString = 
 43 function() {
 44 	return "ZmApptNotifyDialog";
 45 };
 46 
 47 ZmApptNotifyDialog.prototype.initialize = 
 48 function(appt, attId, addedAttendees, removedAttendees) {
 49 	this._appt = appt;
 50 	this._attId = attId;
 51 	this._defaultRadio.checked = true;
 52 
 53     var aCount = addedAttendees.length;
 54     var rCount = removedAttendees.length;    
 55     Dwt.setSize(Dwt.byId(this._containerId), 275, Dwt.CLEAR);
 56 
 57 	this._addedList.innerHTML = this._getAttedeeHtml(addedAttendees, ZmMsg.added);
 58 	this._removedList.innerHTML = this._getAttedeeHtml(removedAttendees, ZmMsg.removed);
 59 };
 60

 62 ZmApptNotifyDialog.prototype.getAppt = 
 63 function() {
 64 	return this._appt;
 65 };
 66 
 67 // helper method - has no use for this dialog
 68 ZmApptNotifyDialog.prototype.getAttId = 
 69 function() {
 70 	return this._attId;
 71 };
 72 
 73 ZmApptNotifyDialog.prototype.notifyNew = 
 74 function() {
 75 	return this._defaultRadio.checked;
 76 };
 77 
 78 ZmApptNotifyDialog.prototype.addSelectionListener = 
 79 function(buttonId, listener) {
 80 	this._button[buttonId].addSelectionListener(listener);
 81 };

 84 // Private / protected methods
 85 
 86 ZmApptNotifyDialog.prototype._setHtml = 
 87 function() {
 88 	this._defaultRadioId	= Dwt.getNextId();
 89 	this._notifyChoiceName	= Dwt.getNextId();
 90 	this._addedListId		= Dwt.getNextId();
 91 	this._removedListId		= Dwt.getNextId();
 92     this._containerId       = Dwt.getNextId();
 93 
 94 	var html = new Array();
 95 	var i = 0;
 96 
 97 	html[i++] = "<div style='width:275px; overflow: auto;' id='"+this._containerId+"'>";
 98 	html[i++] = ZmMsg.attendeeListChanged;
 99 	html[i++] = "<br><div id='";
100 	html[i++] = this._addedListId;
101 	html[i++] = "'></div>";
102 	html[i++] = "<div id='";
103 	html[i++] = this._removedListId;
104 	html[i++] = "'></div>";
105 	html[i++] = "</div><p>";
106 	html[i++] = "<table align=center border=0 width=1%>";
107 	html[i++] = "<tr><td width=1%><input checked value='1' type='radio' id='";
108 	html[i++] = this._defaultRadioId;
109 	html[i++] = "' name='";
110 	html[i++] = this._notifyChoiceName;
111 	html[i++] = "'></td><td style='white-space:nowrap'>";
112 	html[i++] = "<label for='" + this._defaultRadioId + "'>";
113 	html[i++] = ZmMsg.sendUpdatesNew;
114 	html[i++] = "</label>";
115 	html[i++] = "</td></tr>";
116 	html[i++] = "<tr><td width=1%><input value='2' type='radio' name='";
117 	html[i++] = this._notifyChoiceName;
118 	html[i++] = "'></td><td style='white-space:nowrap'>";
119 	html[i++] = "<label for='" + this._notifyChoiceName + "'>";
120 	html[i++] = ZmMsg.sendUpdatesAll;
121 	html[i++] = "</label>";
122 	html[i++] = "</td></tr>";
123 	html[i++] = "</table>";
124 
125 	return html.join("");
126 };
127 
128 ZmApptNotifyDialog.prototype._getAttedeeHtml = 
129 function(attendeeList, attendeeLabel) {
130 	var html = new Array();
131 	var j = 0;
132 
133 	if (attendeeList.length) {
134 		html[j++] = "<table border=0><tr>";
135 		html[j++] = "<td valign=top>  <b>";
136 		html[j++] = attendeeLabel;
137 		html[j++] = "</b></td><td>";
138 		html[j++] = AjxStringUtil.htmlEncode(attendeeList.join(", "));
139 		html[j++] = "</td></tr></table>";
140 	}
141 	return html.join("");
142 };
143 
144 ZmApptNotifyDialog.prototype._cacheFields = 
145 function() {
146 	this._defaultRadio = document.getElementById(this._defaultRadioId); 		delete this._defaultRadioId;
147 	this._addedList = document.getElementById(this._addedListId); 				delete this._addedListId;
148 	this._removedList = document.getElementById(this._removedListId); 			delete this._removedListId;
149 	
150 };
