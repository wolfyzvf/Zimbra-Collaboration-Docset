 39 var Validator = {
 40 	isEmail : function(s) {
 41 		return this.test(s, '^[-!#$%&\'*+\\./0-9=?A-Z^_`a-z{|}~]+@[-!#$%&\'*+\\/0-9=?A-Z^_`a-z{|}~]+\.[-!#$%&\'*+\\./0-9=?A-Z^_`a-z{|}~]+$');
 42 	},
 43 
 44 	isAbsUrl : function(s) {
 45 		return this.test(s, '^(news|telnet|nttp|file|http|ftp|https)://[-A-Za-z0-9\\.]+\\/?.*$');
 46 	},
 47 
 48 	isSize : function(s) {
 49 		return this.test(s, '^[0-9.]+(%|in|cm|mm|em|ex|pt|pc|px)?$');
 50 	},
 51 
 52 	isId : function(s) {
 53 		return this.test(s, '^[A-Za-z_]([A-Za-z0-9_])*$');
 54 	},
 55 
 56 	isEmpty : function(s) {
 57 		var nl, i;
 58 
 59 		if (s.nodeName == 'SELECT' && s.selectedIndex < 1)
 60 			return true;
 61 
 62 		if (s.type == 'checkbox' && !s.checked)
 63 			return true;
 64 
 65 		if (s.type == 'radio') {
 66 			for (i=0, nl = s.form.elements; i<nl.length; i++) {
 67 				if (nl[i].type == "radio" && nl[i].name == s.name && nl[i].checked)
 68 					return false;
 69 			}
 70 
 71 			return true;
 72 		}
 73 
 74 		return new RegExp('^\\s*$').test(s.nodeType == 1 ? s.value : s);
 75 	},
 76 
 77 	isNumber : function(s, d) {
 78 		return !isNaN(s.nodeType == 1 ? s.value : s) && (!d || !this.test(s, '^-?[0-9]*\\.[0-9]*$'));
 79 	},
 80 
 81 	test : function(s, p) {
 82 		s = s.nodeType == 1 ? s.value : s;
 83 
 84 		return s == '' || new RegExp(p).test(s);
 85 	}
 86 };

 88 var AutoValidator = {
 89 	settings : {
 90 		id_cls : 'id',
 91 		int_cls : 'int',
 92 		url_cls : 'url',
 93 		number_cls : 'number',
 94 		email_cls : 'email',
 95 		size_cls : 'size',
 96 		required_cls : 'required',
 97 		invalid_cls : 'invalid',
 98 		min_cls : 'min',
 99 		max_cls : 'max'
100 	},
101 
102 	init : function(s) {
103 		var n;
104 
105 		for (n in s)
106 			this.settings[n] = s[n];
107 	},
108 
109 	validate : function(f) {
110 		var i, nl, s = this.settings, c = 0;
111 
112 		nl = this.tags(f, 'label');
113 		for (i=0; i<nl.length; i++) {
114 			this.removeClass(nl[i], s.invalid_cls);
115 			nl[i].setAttribute('aria-invalid', false);
116 		}
117 
118 		c += this.validateElms(f, 'input');
119 		c += this.validateElms(f, 'select');
120 		c += this.validateElms(f, 'textarea');
121 
122 		return c == 3;
123 	},
124 
125 	invalidate : function(n) {
126 		this.mark(n.form, n);
127 	},
128 	
129 	getErrorMessages : function(f) {
130 		var nl, i, s = this.settings, field, msg, values, messages = [], ed = tinyMCEPopup.editor;
131 		nl = this.tags(f, "label");
132 		for (i=0; i<nl.length; i++) {
133 			if (this.hasClass(nl[i], s.invalid_cls)) {
134 				field = document.getElementById(nl[i].getAttribute("for"));
135 				values = { field: nl[i].textContent };
136 				if (this.hasClass(field, s.min_cls, true)) {
137 					message = ed.getLang('invalid_data_min');
138 					values.min = this.getNum(field, s.min_cls);
139 				} else if (this.hasClass(field, s.number_cls)) {
140 					message = ed.getLang('invalid_data_number');
141 				} else if (this.hasClass(field, s.size_cls)) {
142 					message = ed.getLang('invalid_data_size');
143 				} else {
144 					message = ed.getLang('invalid_data');
145 				}
146 				
147 				message = message.replace(/{\#([^}]+)\}/g, function(a, b) {
148 					return values[b] || '{#' + b + '}';
149 				});
150 				messages.push(message);
151 			}
152 		}
153 		return messages;
154 	},
155 
156 	reset : function(e) {
157 		var t = ['label', 'input', 'select', 'textarea'];
158 		var i, j, nl, s = this.settings;
159 
160 		if (e == null)
161 			return;
162 
163 		for (i=0; i<t.length; i++) {
164 			nl = this.tags(e.form ? e.form : e, t[i]);
165 			for (j=0; j<nl.length; j++) {
166 				this.removeClass(nl[j], s.invalid_cls);
167 				nl[j].setAttribute('aria-invalid', false);
168 			}
169 		}
170 	},
171 
172 	validateElms : function(f, e) {
173 		var nl, i, n, s = this.settings, st = true, va = Validator, v;
174 
175 		nl = this.tags(f, e);
176 		for (i=0; i<nl.length; i++) {
177 			n = nl[i];
178 
179 			this.removeClass(n, s.invalid_cls);
180 
181 			if (this.hasClass(n, s.required_cls) && va.isEmpty(n))
182 				st = this.mark(f, n);
183 
184 			if (this.hasClass(n, s.number_cls) && !va.isNumber(n))
185 				st = this.mark(f, n);
186 
187 			if (this.hasClass(n, s.int_cls) && !va.isNumber(n, true))
188 				st = this.mark(f, n);
189 
190 			if (this.hasClass(n, s.url_cls) && !va.isAbsUrl(n))
191 				st = this.mark(f, n);
192 
193 			if (this.hasClass(n, s.email_cls) && !va.isEmail(n))
194 				st = this.mark(f, n);
195 
196 			if (this.hasClass(n, s.size_cls) && !va.isSize(n))
197 				st = this.mark(f, n);
198 
199 			if (this.hasClass(n, s.id_cls) && !va.isId(n))
200 				st = this.mark(f, n);
201 
202 			if (this.hasClass(n, s.min_cls, true)) {
203 				v = this.getNum(n, s.min_cls);
204 
205 				if (isNaN(v) || parseInt(n.value) < parseInt(v))
206 					st = this.mark(f, n);
207 			}
208 
209 			if (this.hasClass(n, s.max_cls, true)) {
210 				v = this.getNum(n, s.max_cls);
211 
212 				if (isNaN(v) || parseInt(n.value) > parseInt(v))
213 					st = this.mark(f, n);
214 			}
215 		}
216 
217 		return st;
218 	},
219 
220 	hasClass : function(n, c, d) {
221 		return new RegExp('\\b' + c + (d ? '[0-9]+' : '') + '\\b', 'g').test(n.className);
222 	},
223 
224 	getNum : function(n, c) {
225 		c = n.className.match(new RegExp('\\b' + c + '([0-9]+)\\b', 'g'))[0];
226 		c = c.replace(/[^0-9]/g, '');
227 
228 		return c;
229 	},
230 
231 	addClass : function(n, c, b) {
232 		var o = this.removeClass(n, c);
233 		n.className = b ? c + (o != '' ? (' ' + o) : '') : (o != '' ? (o + ' ') : '') + c;
234 	},
235 
236 	removeClass : function(n, c) {
237 		c = n.className.replace(new RegExp("(^|\\s+)" + c + "(\\s+|$)"), ' ');
238 		return n.className = c != ' ' ? c : '';
239 	},
240 
241 	tags : function(f, s) {
242 		return f.getElementsByTagName(s);
243 	},
244 
245 	mark : function(f, n) {
246 		var s = this.settings;
247 
248 		this.addClass(n, s.invalid_cls);
249 		n.setAttribute('aria-invalid', 'true');
250 		this.markLabels(f, n, s.invalid_cls);
251 
252 		return false;
253 	},
254 
255 	markLabels : function(f, n, ic) {
256 		var nl, i;
257 
258 		nl = this.tags(f, "label");
259 		for (i=0; i<nl.length; i++) {
260 			if (nl[i].getAttribute("for") == n.id || nl[i].htmlFor == n.id)
261 				this.addClass(nl[i], ic);
262 		}
263 
264 		return null;
265 	}
266 };
