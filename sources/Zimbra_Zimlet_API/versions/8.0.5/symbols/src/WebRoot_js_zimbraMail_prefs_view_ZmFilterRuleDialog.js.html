 16 /**
 17  * Creates a dialog for specifying a filter rule. Can be used for either add or edit.
 18  * @constructor
 19  * @class
 20  * This class presents a dialog which a user can use to add or edit a filter rule.
 21  * A filter rule consists of conditions and actions (at least one of each). Different
 22  * types of conditions and actions require different fields to specify them, so they
 23  * are presented in a table in which all columns are not necessarily occupied.
 24  * <p>
 25  * First the HTML is laid out, then DWT objects that are needed for input are plugged
 26  * in.</p>
 27  *
 28  * @author Conrad Damon
 29  * 
 30  * @extends		DwtDialog
 31  */

 33 
 34 	DwtDialog.call(this, {parent:appCtxt.getShell(), className:"ZmFilterRuleDialog", title:ZmMsg.selectAddresses, id: "ZmFilterRuleDialog"});

 44 	this._opsChangeLstnr	= new AjxListener(this, this._opsChangeListener);
 45 	this._dateLstnr			= new AjxListener(this, this._dateListener);

 80 /**
 81  * Pops-up the dialog and displays either a given rule for editing, or a dummy
 82  * rule that is the base for adding a new rule.
 83  *
 84  * @param {ZmFilterRule}	rule				the rule to edit
 85  * @param {Boolean}	editMode			if <code>true</code>, we are editing a rule
 86  * @param {ZmFilterRule}	referenceRule		the rule after which to add new rule
 87  * @param {String}	accountName		the name of the account
 88  */
 89 ZmFilterRuleDialog.prototype.popup =
 90 function(rule, editMode, referenceRule, accountName, outgoing) {
 91 	// always make sure we have the right rules container in case of multi-mbox
 92 	this._rules = AjxDispatcher.run(outgoing ? "GetOutgoingFilterRules" : "GetFilterRules", accountName);
 93 	this._outgoing = outgoing;
 94 	this._rules.loadRules(); // make sure rules are loaded (for when we save)
 95 	this._inputs = {};
 96 	this._rule = rule || ZmFilterRule.getDummyRule();
 97 	this._editMode = editMode;
 98 	this._referenceRule = referenceRule;
 99 	this.setTitle(rule ? ZmMsg.editFilter : ZmMsg.addFilter);
100 
101 	var nameField = Dwt.byId(this._nameInputId);
102 	var name = rule ? rule.name : null;
103 	nameField.value = name || "";
104 
105 	var activeField = Dwt.byId(this._activeCheckboxId);
106 	activeField.checked = (!rule || rule.active);
107 	Dwt.setHandler(activeField, DwtEvent.ONCHANGE, AjxCallback.simpleClosure(this._activeChangeListener, this));
108 
109 	var stopField = Dwt.byId(this._stopCheckboxId);
110 	stopField.checked = (!editMode);
111 
112 	var checkAll = (rule && (rule.getGroupOp() == ZmFilterRule.GROUP_ALL));
113 	this._conditionSelect.setSelectedValue(checkAll ? ZmFilterRule.GROUP_ALL : ZmFilterRule.GROUP_ANY);
114 
115 	this._conditionsTabGroup.removeAllMembers();
116 	this._actionsTabGroup.removeAllMembers();
117 
118 	this._renderTable(this._rule, true, this._conditionsTableId, this._rule.conditions, this._conditionsTabGroup);	// conditions
119 	this._renderTable(this._rule, false, this._actionsTableId, this._rule.actions, this._actionsTabGroup);	// actions
120 	this._addDwtObjects();
121 
122 	DwtDialog.prototype.popup.call(this);
123 
124 	nameField.focus();
125 };

222 	var tabIndex = el.getAttribute("tabindex") || MAX_VALUE - 4;

228 	tabIndexes[tabIndex] = this._conditionsTabGroup;

246 /**
247  * Draws a table of conditions or actions. Returns the ID of the last row added.
248  *
249  * @param {ZmFilterRule}	rule			the source rule
250  * @param {Boolean}	isCondition		if <code>true</code>, we're drawing conditions (as opposed to actions)
251  * @param {String}	tableId		the DWT id representing the parent table
252  * @param {Object}	rowData		the meta data used to figure out which DWT widget to create
253  * @param {DwtTabGroup}	tabGroup		tab group for focus
254  * 
255  * @private
256  */
257 ZmFilterRuleDialog.prototype._renderTable =
258 function(rule, isCondition, tableId, rowData, tabGroup) {
259 	var table = Dwt.byId(tableId);
260 	var row;
261 	for (var i in rowData) {
262 		var data = rowData[i];
263 		if (isCondition && i == "condition") { continue; }
264 
265 		// don't show action if it's disabled
266 		if (!isCondition) {
267 			var actionIndex = ZmFilterRule.A_VALUE_MAP[i];
268 			if (!ZmFilterRule.checkPreconditions(ZmFilterRule.ACTIONS[actionIndex]) && actionIndex != ZmFilterRule.A_FORWARD) { continue; }
269 		}
270 
271 		for (j = 0; j < data.length; j++) {
272 			var rowId = Dwt.getNextId();
273 			this._enterTabScope(rowId);
274 			try {
275 				var html = this._getRowHtml(data[j], i, isCondition, rowId);
276 				if (html) {
277 					row = Dwt.parseHtmlFragment(html, true);
278 					table.tBodies[0].appendChild(row);
279 					tabGroup.addMember(this._getCurrentTabScope());
280 				}
281 			}
282 			finally {
283 				this._exitTabScope();
284 			}
285 		}
286 	}
287 
288 	this._resetOperations(isCondition);
289 
290 	return (row ? row.id : null);
291 };

293 /**
294  * Gets the HTML for a single condition or action row.
295  *
296  * @param {Object}	data			an object containing meta info about the filter rule condition or action
297  * @param {String}	test			the type of test condition (headerTest, sizeTest, bodyTest, etc)
298  * @param {Boolean}	isCondition		if <code>true</code>, we're rendering a condition row
299  * @param {String}	rowId		the unique ID representing this row
300  * 
301  * @private
302  */
303 ZmFilterRuleDialog.prototype._getRowHtml =
304 function(data, test, isCondition, rowId) {
305 	var conf;
306 	if (isCondition) {
307 		conf = this._getConditionFromTest(test, data);
308 	} else {
309 		var actionId = ZmFilterRule.A_VALUE_MAP[test];
310 		conf = ZmFilterRule.ACTIONS[actionId];
311 	}
312 
313 	var html = [];
314 	var i = 0;
315 
316 	this._inputs[rowId] = {};
317 
318 	html[i++] = "<tr id='";
319 	html[i++] = rowId;
320 	html[i++] = "'>";
321 
322 	if (isCondition) {
323 		this._inputs[rowId].isCondition = true;
324 		html[i++] = this._createRowComponent(true, "subject", ZmFilterRule.CONDITIONS_LIST, data, test, rowId);
325 		html[i++] = this._createRowComponent(conf, "subjectMod", conf.smOptions, data, test, rowId);
326 		html[i++] = this._createRowComponent(conf, "ops", conf.opsOptions, data, test, rowId);
327 		html[i++] = this._createRowComponent(conf, "value", conf.vOptions, data, test, rowId);
328 		html[i++] = this._createRowComponent(conf, "valueMod", conf.vmOptions, data, test, rowId);
329 		if (data && data.caseSensitive) {
330 			this._inputs[rowId]["caseSensitive"] = {value: data.caseSensitive}; //save case sensitive value if it exists
331 		}
332 	} else {
333 		if (test == ZmFilterRule.A_NAME_STOP) {
334 			var stopField = Dwt.byId(this._stopCheckboxId);
335 			stopField.checked = true;
336 			return;
337 		}
338 		html[i++] = "<td><table><tr>";
339 		if (conf) {
340 			var options = this._outgoing ? ZmFilterRule.ACTIONS_OUTGOING_LIST : ZmFilterRule.ACTIONS_LIST;
341 			html[i++] = this._createRowComponent(false, "name", options, data, test, rowId);
342 			html[i++] = this._createRowComponent(conf, "param", conf.pOptions, data, test, rowId);
343 		}
344 		else {
345 		 //see if it's a actionReply or actionNotify filter and output readonly
346 			if (actionId == ZmFilterRule.A_NOTIFY && data) {
347 				var email = data.a;
348 				var content = AjxUtil.isArray(data.content) ?  data.content[0]._content : "";
349 				var maxBodySize = data.maxBodySize;
350 				var subject = data.su;
351 
352 				html[i++] = "<td><table>";
353 				html[i++] = "<tr><td>" + ZmMsg.actionNotifyReadOnlyMsg + "</td></tr>";
354 				html[i++] = "<tr><td>" + ZmMsg.emailLabel + " " + email + " | " + subject + " | " + ZmMsg.maxBodySize + ": " + maxBodySize + "</td><tr>";
355 				html[i++] = "<tr><td>" + ZmMsg.body + ": " + content + "</td></tr></table></td>";
356 			}
357 			else if (actionId == ZmFilterRule.A_REPLY && data) {
358 				var content = AjxUtil.isArray(data.content) ? data.content[0]._content : "";
359 				html[i++] = "<td><table><tr><td>" + ZmMsg.actionReplyReadOnlyMsg + "</td></tr>";
360 				html[i++] = "<tr><td>" + ZmMsg.body + ": " + content + "</td></tr></table></td>";
361 			}
362 			this.setButtonEnabled(DwtDialog.OK_BUTTON, false);
363 		}
364 		html[i++] = "</tr></table></td>";
365 	}
366 	html[i++] = this._getPlusMinusHtml(rowId, isCondition);
367 	html[i++] = "</tr>";
368 
369 	return html.join("");
370 };

372 ZmFilterRuleDialog.prototype._getConditionFromTest =
373 function(test, data) {
374 	var condition;
375 	switch (test) {
376 		case ZmFilterRule.TEST_ADDRESS:
377 			condition = ZmFilterRule.C_ADDRESS_MAP[data.header];
378 			if (!condition) { // shouldn't get here
379 				condition = ZmFilterRule.C_ADDRESS;
380 			}
381 			break;
382 		case ZmFilterRule.TEST_HEADER_EXISTS:	condition = ZmFilterRule.C_HEADER; break;
383 		case ZmFilterRule.TEST_SIZE:			condition = ZmFilterRule.C_SIZE; break;
384 		case ZmFilterRule.TEST_DATE:			condition = ZmFilterRule.C_DATE; break;
385 		case ZmFilterRule.TEST_BODY:			condition = ZmFilterRule.C_BODY; break;
386 		case ZmFilterRule.TEST_ATTACHMENT:		condition = ZmFilterRule.C_ATT; break;
387 		case ZmFilterRule.TEST_MIME_HEADER:		condition = ZmFilterRule.C_MIME_HEADER; break;
388 		case ZmFilterRule.TEST_ADDRBOOK:		condition = ZmFilterRule.C_ADDRBOOK; break;
389 		case ZmFilterRule.TEST_INVITE:			condition = ZmFilterRule.C_INVITE; break;
390 		case ZmFilterRule.TEST_CONVERSATIONS:   condition = ZmFilterRule.C_CONV; break;
391 		case ZmFilterRule.TEST_SOCIAL:          condition = ZmFilterRule.C_SOCIAL; break;
392 		case ZmFilterRule.TEST_FACEBOOK:        condition = ZmFilterRule.C_SOCIAL; break;
393 		case ZmFilterRule.TEST_SOCIALCAST:      condition = ZmFilterRule.C_SOCIAL; break;
394 		case ZmFilterRule.TEST_TWITTER:         condition = ZmFilterRule.C_SOCIAL; break;
395 		case ZmFilterRule.TEST_LINKEDIN:        condition = ZmFilterRule.C_SOCIAL; break;
396 		case ZmFilterRule.TEST_LIST:            condition = ZmFilterRule.C_CONV; break;
397 		case ZmFilterRule.TEST_BULK:            condition = ZmFilterRule.C_CONV; break;
398 		case ZmFilterRule.TEST_ME:              condition = ZmFilterRule.C_ADDRBOOK; break;
399 		case ZmFilterRule.TEST_RANKING:         condition = ZmFilterRule.C_ADDRBOOK; break;
400 		case ZmFilterRule.TEST_IMPORTANCE:      condition = ZmFilterRule.C_CONV; break;
401 		case ZmFilterRule.TEST_FLAGGED:         condition = ZmFilterRule.C_CONV; break;
402 		case ZmFilterRule.TEST_HEADER:
403 			condition = ZmFilterRule.C_HEADER_MAP[data.header];
404 			if (!condition) { // means custom header
405 				condition = ZmFilterRule.C_HEADER;
406 			}  
407 			break;
408 	}
409 
410 	//TODO: find a better way to do this.  Preconditions for opsOptions?
411 	if (condition == ZmFilterRule.C_SOCIAL) {
412 		condition = ZmFilterRule.CONDITIONS[condition];
413 		condition.opsOptions = ZmFilterRule.getSocialFilters();
414 		return condition;
415 	}
416 	return (condition ? ZmFilterRule.CONDITIONS[condition] : null);
417 };

465 /**
466  * Removes a condition or action row from its table. Also cleans up any DWT
467  * objects the row was using.
468  *
469  * @param {String}	rowId			the ID of the row to remove
470  * @param {Boolean}	isCondition		if <code>true</code>, we're removing a condition row

472  * @private
473  */
474 ZmFilterRuleDialog.prototype._removeRow =
475 function(rowId, isCondition) {
476 	var row = Dwt.byId(rowId);
477 	if (!row) { return; }
478 		
479 	var table = Dwt.byId(isCondition ? this._conditionsTableId : this._actionsTableId);
480 	var rows = table.rows;
481 	for (var i = 0; i < rows.length; i++) {
482 		if (rows[i] == row) {
483 			table.deleteRow(i);
484 			break;
485 		}
486 	}
487 	this._removeDwtObjects(rowId);
488 	delete this._inputs[rowId];
489 };

491 /**
492  * Creates an input widget and returns HTML for a table cell that will contain it.
493  * The config for a condition or action is based on its main operator; for conditions
494  * it's called subject ("from", "body", etc), and for actions it's just called the
495  * action ("keep", "fileinto", etc). Each one of those has its own particular inputs.
496  * This method creates one of those inputs.
497  *
498  * @param {Hash|Boolean}	conf		the config for this subject or action; boolean if rendering
499  *										the actual subject or action (means "isCondition")
500  * @param {String}	field		the name of the input field
501  * @param {Array}	options		if the field type is a select, its options
502  * @param {Object}	rowData	the current value of the field, if any
503  * @param {String}	testType	the type of test condition (i.e. headerTest, attachmentTest, bodyTest, etc)
504  * @param {String}	rowId		the ID of the containing row
505  * 
506  * @private
507  */
508 ZmFilterRuleDialog.prototype._createRowComponent =
509 function(conf, field, options, rowData, testType, rowId) {
510 	var tabGroup = this._getCurrentTabScope();
511 
512 	var isCondition, type;
513 	var isMainSelect = AjxUtil.isBoolean(conf);
514 	if (isMainSelect) {
515 		type = ZmFilterRule.TYPE_SELECT;
516 		isCondition = conf;
517 	} else {
518 		type = conf[field];
519 		if (!type) {
520 			return "<td></td>";
521 		}
522 	}
523 
524 	var dataValue = this._getDataValue(isMainSelect, testType, field, rowData);
525 
526 	var id = Dwt.getNextId();
527 	if (type == ZmFilterRule.TYPE_INPUT) {
528 		var inputFieldId = "FilterRuleDialog_INPUTFIELD_" + ZmFilterRuleDialog.CONDITIONS_INDEX++;
529 		var inputId = "FilterRuleDialog_INPUT_" + ZmFilterRuleDialog.CONDITIONS_INDEX++;
530 		var input = new DwtInputField({parent: this, type: DwtInputField.STRING, initialValue: dataValue, size: 20, id: inputFieldId, inputId: inputId});
531 		input.setData(ZmFilterRuleDialog.ROW_ID, rowId);
532 		this._inputs[rowId][field] = {id: id, dwtObj: input};
533 		tabGroup.addMember(input.getTabGroupMember());
534 	}
535 	else if (type == ZmFilterRule.TYPE_SELECT) {
536 		var selectId = "FilterRuleDialog_SELECT_" + ZmFilterRuleDialog.CONDITIONS_INDEX++;
537 		var select = new DwtSelect({parent:this, id: selectId});
538 		select.setData(ZmFilterRuleDialog.ROW_ID, rowId);
539         select.fixedButtonWidth();
540 		this._inputs[rowId][field] = {id: id, dwtObj: select};
541 		if (isMainSelect) {
542 			select.setData(ZmFilterRuleDialog.IS_CONDITION, isCondition);
543 			select.addChangeListener(this._rowChangeLstnr);

1046 	table.deleteRow(index);
1047 	var newIndex = (index >= table.rows.length) ? null : index; // null means add to end
1048 
1049 	var test, data, subjectMod;
1050 	if (isCondition) {
1051 		test = ZmFilterRule.C_TEST_MAP[newValue];
1052 		if (test == ZmFilterRule.TEST_HEADER) {
1053 			subjectMod = ZmFilterRule.C_HEADER_VALUE[newValue];
1054 		}
1055 		else if (test == ZmFilterRule.TEST_ADDRESS) {
1056 			subjectMod = ZmFilterRule.C_ADDRESS_VALUE[newValue];
1057 		}
1058 		data = ZmFilterRule.getConditionData(test, comparator, dataValue, subjectMod);
1059 	} else {
1060 		test = ZmFilterRule.A_VALUE[newValue];
1061 		data = ZmFilterRule.getActionData(test);
1062 	}
1063 
1064 	this._enterTabScope(rowId);
1065 	try {
1066 		var html = this._getRowHtml(data, test, isCondition, rowId);
1067 		if (html) {
1068 			row = Dwt.parseHtmlFragment(html, true);
1069 			if (!row) {
1070 				DBG.println(AjxDebug.DBG1, "Filter rule dialog: no row created!");
1071 				return;
1072 			}
1073 			table.tBodies[0].insertBefore(row, (newIndex != null) ? table.rows[newIndex] : null);
1074 			this._addDwtObjects(row.id);
1075 			this._resetOperations(isCondition);
1076 			tabGroup.removeMember(tabGroup.getTabGroupMemberByName(rowId));
1077 			tabGroup.addMember(this._getCurrentTabScope());
1078 		}
