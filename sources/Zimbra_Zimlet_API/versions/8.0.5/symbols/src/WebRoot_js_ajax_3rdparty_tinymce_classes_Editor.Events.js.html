812 	tinymce.Editor.prototype.bindNativeEvents = function() {
813 		// 'focus', 'blur', 'dblclick', 'beforedeactivate', submit, reset
814 		var self = this, i, settings = self.settings, dom = self.dom, nativeToDispatcherMap;
815 
816 		nativeToDispatcherMap = {
817 			mouseup : 'onMouseUp',
818 			mousedown : 'onMouseDown',
819 			click : 'onClick',
820 			keyup : 'onKeyUp',
821 			keydown : 'onKeyDown',
822 			keypress : 'onKeyPress',
823 			submit : 'onSubmit',
824 			reset : 'onReset',
825 			contextmenu : 'onContextMenu',
826 			dblclick : 'onDblClick',
827 			paste : 'onPaste' // Doesn't work in all browsers yet
828 		};
829 
830 		// Handler that takes a native event and sends it out to a dispatcher like onKeyDown
831 		function eventHandler(evt, args) {
832 			var type = evt.type;
833 
834 			// Don't fire events when it's removed
835 			if (self.removed)
836 				return;
837 
838 			// Sends the native event out to a global dispatcher then to the specific event dispatcher
839 			if (self.onEvent.dispatch(self, evt, args) !== false) {
840 				self[nativeToDispatcherMap[evt.fakeType || evt.type]].dispatch(self, evt, args);
841 			}
842 		};
843 
844 		// Opera doesn't support focus event for contentEditable elements so we need to fake it
845 		function doOperaFocus(e) {
846 			self.focus(true);
847 		};

849 		function nodeChanged(ed, e) {
850 			// Normalize selection for example <b>a</b><i>|a</i> becomes <b>a|</b><i>a</i> except for Ctrl+A since it selects everything
851 			if (e.keyCode != 65 || !tinymce.VK.metaKeyPressed(e)) {
852 				self.selection.normalize();
853 			}
854 
855 			self.nodeChanged();
856 		}
857 
858 		// Add DOM events
859 		each(nativeToDispatcherMap, function(dispatcherName, nativeName) {
860 			var root = settings.content_editable ? self.getBody() : self.getDoc();
861 
862 			switch (nativeName) {
863 				case 'contextmenu':
864 					dom.bind(root, nativeName, eventHandler);
865 					break;
866 
867 				case 'paste':
868 					dom.bind(self.getBody(), nativeName, eventHandler);
869 					break;
870 
871 				case 'submit':
872 				case 'reset':
873 					dom.bind(self.getElement().form || tinymce.DOM.getParent(self.id, 'form'), nativeName, eventHandler);
874 					break;
875 
876 				default:
877 					dom.bind(root, nativeName, eventHandler);
878 			}
879 		});
880 
881 		// Set the editor as active when focused
882 		dom.bind(settings.content_editable ? self.getBody() : (tinymce.isGecko ? self.getDoc() : self.getWin()), 'focus', function(e) {
883 			self.focus(true);
884 		});

886 		if (settings.content_editable && tinymce.isOpera) {
887 			dom.bind(self.getBody(), 'click', doOperaFocus);
888 			dom.bind(self.getBody(), 'keydown', doOperaFocus);

908 			if (e.altKey || e.ctrlKey || e.metaKey) {

910 					var ctrlState = tinymce.isMac ? e.metaKey : e.ctrlKey;

913 						return;

923 					}
