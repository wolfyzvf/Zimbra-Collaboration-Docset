 16 /**
 17  * Creates a new appointment view. The view does not display itself on construction.
 18  * @constructor
 19  * @class
 20  * This class provides a form for creating/editing appointments. It is a tab view with
 21  * five tabs: the appt form, a scheduling page, and three pickers (one each for finding
 22  * attendees, locations, and equipment). The attendee data (people, locations, and
 23  * equipment are all attendees) is maintained here centrally, since it is presented and
 24  * can be modified in each of the five tabs.
 25  *
 26  * @author Parag Shah
 27  *
 28  * @param {DwtShell}	parent			the element that created this view
 29  * @param {String}	className 		class name for this view
 30  * @param {ZmCalendarApp}	calApp			a handle to the owning calendar application
 31  * @param {ZmApptComposeController}	controller		the controller for this view
 32  * 
 33  * @extends		DwtTabView
 34  */
 35 ZmApptComposeView = function(parent, className, calApp, controller) {
 36 
 37 	className = className ? className : "ZmApptComposeView";
 38     var params = {parent:parent, className:className, posStyle:Dwt.ABSOLUTE_STYLE, id:Dwt.getNextId("APPT_COMPOSE_")};
 39 	DwtComposite.call(this, params);
 40 
 41 	this.setScrollStyle(DwtControl.CLIP);
 42 	this._app = calApp;
 43 	this._controller = controller;
 44 	
 45 	// centralized date info
 46 	this._dateInfo = {};
 47 
 48 	// centralized attendee data
 49 	this._attendees = {};
 50 	this._attendees[ZmCalBaseItem.PERSON]	= new AjxVector();	// list of ZmContact
 51 	this._attendees[ZmCalBaseItem.LOCATION]	= new AjxVector();	// list of ZmResource
 52 	this._attendees[ZmCalBaseItem.EQUIPMENT]= new AjxVector();	// list of ZmResource
 53 
 54 	// set of attendee keys (for preventing duplicates)
 55 	this._attendeeKeys = {};
 56 	this._attendeeKeys[ZmCalBaseItem.PERSON]	= {};
 57 	this._attendeeKeys[ZmCalBaseItem.LOCATION]	= {};
 58 	this._attendeeKeys[ZmCalBaseItem.EQUIPMENT]	= {};
 59 
 60 	// Email to type map
 61 	this._attendeeType = {};
 62 
 63 	// for attendees change events
 64 	this._evt = new ZmEvent(ZmEvent.S_CONTACT);
 65 	this._evtMgr = new AjxEventMgr();
 66 	
 67 	this._initialize();
 68 };

102 ZmApptComposeView.prototype.set =
103 function(appt, mode, isDirty) {
104 
105     var isForward = false;
106 
107     //decides whether appt is being edited/forwarded/proposed new time
108     var apptComposeMode = ZmApptComposeView.EDIT;
109 
110 
111     //"mode" should always be set to one of ZmCalItem.MODE_EDIT/ZmCalItem.MODE_EDIT_INSTANCE/ZmCalItem.MODE_EDIT_SERIES/ZmCalItem.MODE_NEW
112     if(ZmCalItem.FORWARD_MAPPING[mode]) {
113         isForward = true;
114         this._forwardMode = mode;
115         mode = ZmCalItem.FORWARD_MAPPING[mode];
116         apptComposeMode = ZmApptComposeView.FORWARD; 
117     } else {
118         this._forwardMode = undefined;        
119     }
120 
121     this._proposeNewTime = (mode == ZmCalItem.MODE_PROPOSE_TIME);
122 
123     if (this._proposeNewTime) {
124         mode = appt.viewMode || ZmCalItem.MODE_EDIT;
125         apptComposeMode = ZmApptComposeView.PROPOSE_TIME;
126     }
127 
128 	this._setData = [appt, mode, isDirty];
129 	this._dateInfo.timezone = appt.getTimezone();
130     this._apptEditView.initialize(appt, mode, isDirty, apptComposeMode);
131     this._apptEditView.show();
132 
133     var editMode = !Boolean(this._forwardMode) && !this._proposeNewTime;
134     this._apptEditView.enableInputs(editMode);
135     this._apptEditView.enableSubjectField(!this._proposeNewTime);
136 
137     var toolbar = this._controller.getToolbar();
138     toolbar.enableAll(true);    
139     toolbar.enable([ZmOperation.ATTACHMENT], editMode);
140 };

250 /**
251  * Updates the set of attendees for this appointment, by adding attendees or by
252  * replacing the current list (with a clone of the one passed in).
253  *
254  * @param attendees	[object]		attendee(s) as string, array, or AjxVector
255  * @param type		[constant]		attendee type (attendee/location/equipment)
256  * @param mode		[constant]*		replace (default) or add
257  * @param index		[int]*			index at which to add attendee
258  * 
259  * @private
260  */
261 ZmApptComposeView.prototype.updateAttendees =
262 function(attendees, type, mode, index) {
263 	attendees = (attendees instanceof AjxVector) ? attendees.getArray() :
264 				(attendees instanceof Array) ? attendees : [attendees];
265 	mode = mode || ZmApptComposeView.MODE_REPLACE;
266 	// Note whether any of the attendees changed.  Needed to decide
267 	// for Locations whether or not to check for conflicts
268 	var changed = false;
269 	var key;
270 	if (mode == ZmApptComposeView.MODE_REPLACE) {
271 		this._attendees[type] = new AjxVector();
272 		var oldKeys = this._attendeeKeys[type];
273 		this._attendeeKeys[type] = {};
274 		for (var i = 0; i < attendees.length; i++) {
275 			var attendee = attendees[i];
276 			this._attendees[type].add(attendee);
277 			key = this._addAttendeeKey(attendee, type);
278 			this._attendeeType[key] = type;
279 			if (key && !oldKeys[key]) {
280 				// New key that was not in the old set
281 				changed = true;
282 			}
283 		}
284 		if ((type == ZmCalBaseItem.LOCATION) && this._locationConflictCallback) {
285 			for (key in oldKeys) {
286 				if (key && !this._attendeeKeys[type][key]) {
287 					// Old location key that is not in the new set
288 					changed = true;
289 					break;
290 				}
291 			}
292 		}
293 	} else if (mode == ZmApptComposeView.MODE_ADD) {
294 		for (var i = 0; i < attendees.length; i++) {
295 			var attendee = attendees[i];
296 			key = this._getAttendeeKey(attendee);
297 			this._attendeeType[key] = type;
298 			if (!this._attendeeKeys[type][key] === true) {
299 				this._attendees[type].add(attendee, index);
300 				this._addAttendeeKey(attendee, type);
301 				changed = true;
302 			}
303 		}
304 	} else if (mode == ZmApptComposeView.MODE_REMOVE) {
305 		for (var i = 0; i < attendees.length; i++) {
306 			var attendee = attendees[i];
307 			key = this._removeAttendeeKey(attendee, type);
308 			delete this._attendeeType[key];
309 			this._attendees[type].remove(attendee);
310 			if (key) {
311 				changed = true;
312 			}
313 		}
314 	}
315 
316     if (changed && (type == ZmCalBaseItem.LOCATION) && this._locationConflictCallback) {
317         this._locationConflictCallback.run(this._attendees[ZmCalBaseItem.LOCATION]);
318     }
319 };

342 ZmApptComposeView.prototype._getAttendeeKey =
343 function(attendee) {
344 	var email = attendee.getLookupEmail() || attendee.getEmail();
345 	var name = attendee.getFullName();
346 	return email ? email : name;
347 };
348 
349 ZmApptComposeView.prototype._addAttendeeKey =
350 function(attendee, type) {
351 	var key = this._getAttendeeKey(attendee);
352 	if (key) {
353 		this._attendeeKeys[type][key] = true;
354 	}
355 	return key;
356 };
357 
358 ZmApptComposeView.prototype._removeAttendeeKey =
359 function(attendee, type) {
360 	var key = this._getAttendeeKey(attendee);
361 	if (key) {
362 		delete this._attendeeKeys[type][key];
363 	}
364 	return key;
365 };

443 ZmApptComposeView.prototype._controlListener =
444 function(ev) {
445 	var newWidth = (ev.oldWidth == ev.newWidth) ? null : ev.newWidth;
446 	var newHeight = (ev.oldHeight == ev.newHeight) ? null : ev.newHeight;
447 
448 	if (!(newWidth || newHeight)) return;
449 
450 	this._apptEditView.resize(newWidth, newHeight);
451 };
