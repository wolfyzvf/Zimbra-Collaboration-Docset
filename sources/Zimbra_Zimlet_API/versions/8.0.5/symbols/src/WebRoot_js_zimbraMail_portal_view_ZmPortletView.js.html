 16 /**
 17  * Creates the portlet view.
 18  * @class
 19  * This class represents the portlet view.
 20  * 
 21  * @param	{Element}		parentEl		the parent element
 22  * @param	{ZmPortlet}		portlet			the portlet
 23  * @param	{String}		className		the class name
 24  * 
 25  * @extends		DwtComposite
 26  */
 27 ZmPortletView = function(parentEl, portlet, className) {
 28     className = className || "ZmPortlet";
 29     DwtComposite.call(this, {parent:DwtShell.getShell(window), className:className, posStyle:DwtControl.STATIC_STYLE});
 30 
 31     // save data
 32     this._portlet = portlet;
 33     this._portlet.view = this;
 34 
 35     this._contentsEl = this.getHtmlElement();
 36     if (parentEl) {
 37         parentEl.portlet = "loaded";
 38         parentEl.innerHTML = "";
 39         parentEl.appendChild(this._contentsEl);
 40     }
 41 
 42     // setup display
 43     this.setIcon(portlet.icon);
 44     this.setTitle(portlet.title);
 45     this.setContentUrl(portlet.actionUrl && portlet.actionUrl.target);
 46 }
 47 ZmPortletView.prototype = new DwtComposite;
 48 ZmPortletView.prototype.constructor = ZmPortletView;
 49 
 50 /**
 51  * Returns a string representation of the object.
 52  * 
 53  * @return		{String}		a string representation of the object
 54  */
 55 ZmPortletView.prototype.toString = function() {
 56     return "ZmPortletView";
 57 };

 63 /**
 64  * Sets the icon.
 65  * 
 66  * @param	{String}	icon		the icon
 67  * 
 68  * @see		AjxImg.setImage
 69  */
 70 ZmPortletView.prototype.setIcon = function(icon) {
 71     if (icon == null || !this._iconEl) return;
 72     AjxImg.setImage(this._iconEl, icon);
 73 };
 74 
 75 /**
 76  * Sets the title.
 77  * 
 78  * @param	{String}		title		the title
 79  */
 80 ZmPortletView.prototype.setTitle = function(title) {
 81     if (title == null || !this._titleEl) return;
 82     this._titleEl.innerHTML = title;
 83 };
 84 
 85 ZmPortletView.prototype.setContent = function(content) {
 86     if (AjxUtil.isString(content)) {
 87         this._contentsEl.innerHTML = content;
 88     }
 89     else if (AjxUtil.ELEMENT_NODE) {
 90         this._contentsEl.innerHTML = "";
 91         this._contentsEl.appendChild(content);
 92     }
 93     else {
 94         this._contentsEl.innerHTML = AjxStringUtil.htmlEncode(String(content));
 95     }
 96 };
 97 
 98 /**
 99  * Sets the content URL as an iframe.
100  * 
101  * @param	{String}	url		the url
102  */
103 ZmPortletView.prototype.setContentUrl = function(url) {
104     if (!url) return;
105 
106     var props = this._portlet.properties;
107     var func = AjxCallback.simpleClosure(ZmPortletView.__replaceProp, null, props);
108     url = url.replace(ZmZimletContext.RE_SCAN_PROP, func);
109     url = this._portlet.zimletCtxt ? this._portlet.zimletCtxt.makeURL({ target: url }, null, props) : url;
110     var html = [
111         "<iframe style='border:none;width:100%;height:100%' ",
112             "src='",url,"'>",
113         "</iframe>"
114     ].join("");
115     this.setContent(html);
116 };
