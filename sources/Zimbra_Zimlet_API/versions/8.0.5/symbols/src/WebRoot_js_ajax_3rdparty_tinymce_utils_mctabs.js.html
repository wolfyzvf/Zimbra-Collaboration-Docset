 25 function MCTabs() {
 26 	this.settings = [];
 27 	this.onChange = tinyMCEPopup.editor.windowManager.createInstance('tinymce.util.Dispatcher');
 28 };
 29 
 30 MCTabs.prototype.init = function(settings) {
 31 	this.settings = settings;
 32 };
 33 
 34 MCTabs.prototype.getParam = function(name, default_value) {
 35 	var value = null;
 36 
 37 	value = (typeof(this.settings[name]) == "undefined") ? default_value : this.settings[name];
 38 
 39 	// Fix bool values
 40 	if (value == "true" || value == "false")
 41 		return (value == "true");
 42 
 43 	return value;
 44 };
 45 
 46 MCTabs.prototype.showTab =function(tab){
 47 	tab.className = 'current';
 48 	tab.setAttribute("aria-selected", true);
 49 	tab.setAttribute("aria-expanded", true);
 50 	tab.tabIndex = 0;
 51 };
 52 
 53 MCTabs.prototype.hideTab =function(tab){
 54 	var t=this;
 55 
 56 	tab.className = '';
 57 	tab.setAttribute("aria-selected", false);
 58 	tab.setAttribute("aria-expanded", false);
 59 	tab.tabIndex = -1;
 60 };
 61 
 62 MCTabs.prototype.showPanel = function(panel) {
 63 	panel.className = 'current'; 
 64 	panel.setAttribute("aria-hidden", false);
 65 };
 66 
 67 MCTabs.prototype.hidePanel = function(panel) {
 68 	panel.className = 'panel';
 69 	panel.setAttribute("aria-hidden", true);
 70 }; 
 71 
 72 MCTabs.prototype.getPanelForTab = function(tabElm) {
 73 	return tinyMCEPopup.dom.getAttrib(tabElm, "aria-controls");
 74 };
 75 
 76 MCTabs.prototype.displayTab = function(tab_id, panel_id, avoid_focus) {
 77 	var panelElm, panelContainerElm, tabElm, tabContainerElm, selectionClass, nodes, i, t = this;
 78 
 79 	tabElm = document.getElementById(tab_id);
 80 
 81 	if (panel_id === undefined) {
 82 		panel_id = t.getPanelForTab(tabElm);
 83 	}
 84 
 85 	panelElm= document.getElementById(panel_id);
 86 	panelContainerElm = panelElm ? panelElm.parentNode : null;
 87 	tabContainerElm = tabElm ? tabElm.parentNode : null;
 88 	selectionClass = t.getParam('selection_class', 'current');
 89 
 90 	if (tabElm && tabContainerElm) {
 91 		nodes = tabContainerElm.childNodes;
 92 
 93 		// Hide all other tabs
 94 		for (i = 0; i < nodes.length; i++) {
 95 			if (nodes[i].nodeName == "LI") {
 96 				t.hideTab(nodes[i]);
 97 			}
 98 		}
 99 
100 		// Show selected tab
101 		t.showTab(tabElm);
102 	}
103 
104 	if (panelElm && panelContainerElm) {
105 		nodes = panelContainerElm.childNodes;
106 
107 		// Hide all other panels
108 		for (i = 0; i < nodes.length; i++) {
109 			if (nodes[i].nodeName == "DIV")
110 				t.hidePanel(nodes[i]);
111 		}
112 
113 		if (!avoid_focus) { 
114 			tabElm.focus();
115 		}
116 
117 		// Show selected panel
118 		t.showPanel(panelElm);
119 	}
120 };

122 MCTabs.prototype.getAnchor = function() {
123 	var pos, url = document.location.href;
124 
125 	if ((pos = url.lastIndexOf('#')) != -1)
126 		return url.substring(pos + 1);
127 
128 	return "";
129 };

132 //Global instance
133 var mcTabs = new MCTabs();
134 
135 tinyMCEPopup.onInit.add(function() {
136 	var tinymce = tinyMCEPopup.getWin().tinymce, dom = tinyMCEPopup.dom, each = tinymce.each;
137 
138 	each(dom.select('div.tabs'), function(tabContainerElm) {
139 		var keyNav;
140 
141 		dom.setAttrib(tabContainerElm, "role", "tablist"); 
142 
143 		var items = tinyMCEPopup.dom.select('li', tabContainerElm);
144 		var action = function(id) {
145 			mcTabs.displayTab(id, mcTabs.getPanelForTab(id));
146 			mcTabs.onChange.dispatch(id);
147 		};
148 
149 		each(items, function(item) {
150 			dom.setAttrib(item, 'role', 'tab');
151 			dom.bind(item, 'click', function(evt) {
152 				action(item.id);
153 			});
154 		});
155 
156 		dom.bind(dom.getRoot(), 'keydown', function(evt) {
157 			if (evt.keyCode === 9 && evt.ctrlKey && !evt.altKey) { // Tab
158 				keyNav.moveFocus(evt.shiftKey ? -1 : 1);
159 				tinymce.dom.Event.cancel(evt);
160 			}
161 		});
162 
163 		each(dom.select('a', tabContainerElm), function(a) {
164 			dom.setAttrib(a, 'tabindex', '-1');
165 		});
166 
167 		keyNav = tinyMCEPopup.editor.windowManager.createInstance('tinymce.ui.KeyboardNavigation', {
168 			root: tabContainerElm,
169 			items: items,
170 			onAction: action,
171 			actOnFocus: true,
172 			enableLeftRight: true,
173 			enableUpDown: true
174 		}, tinyMCEPopup.dom);
175 	});
176 });
