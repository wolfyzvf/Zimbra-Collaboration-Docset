 20 /**
 21  * Creates a dialog containing a listview of items that were (hard) deleted by
 22  * the user.
 23  * @class
 24  * This class represents a dumpster dialog.
 25  *
 26  * @param	{DwtComposite}	parent		the parent
 27  * @param	{String}	className		the class name
 28  *
 29  * @extends		ZmDialog
 30  */
 31 ZmDumpsterDialog = function(parent, className) {
 32 
 33 	var params = {
 34 		parent: parent,
 35 		className: (className || "ZmDumpsterDialog"),
 36 		title: ZmMsg.recoverDeletedItems,
 37 		standardButtons: [DwtDialog.CANCEL_BUTTON]
 38 	};
 39 	ZmDialog.call(this, params);
 40 
 41 	this.getButton(DwtDialog.CANCEL_BUTTON).setText(ZmMsg.close);
 42 
 43 	this._controller = new ZmDumpsterListController(this);
 44 };
 45 
 46 ZmDumpsterDialog.prototype = new ZmDialog;
 47 ZmDumpsterDialog.prototype.constructor = ZmDumpsterDialog;
 48 
 49 ZmDumpsterDialog.prototype.toString =
 50 function() {
 51 	return "ZmDumpsterDialog";
 52 };
 53 
 54 ZmDumpsterDialog.prototype.popup =
 55 function(searchFor, types) {
 56 	this._searchTypes = types ? AjxUtil.toArray(types) : [ZmItem.MSG];
 57 	this._searchFor = searchFor;
 58 	this.runSearchQuery("");
 59 
 60 	ZmDialog.prototype.popup.call(this);
 61 };
 62 
 63 
 64 ZmDumpsterDialog.prototype.runSearchQuery =
 65 function(query) {
 66 	var types = this._searchTypes;
 67 	var searchFor = this._searchFor;
 68 	var params = {
 69 		query: "-in:/Junk " + query, // Users don't want/need to recover deleted spam.
 70 		searchFor: searchFor,
 71 		types: types,
 72 		sortBy: ZmSearch.DATE_DESC,
 73 		noRender: true,
 74 		inDumpster: true,
 75 		skipUpdateSearchToolbar: true, //don't update the main app search toolbar. Otherwise the main app is updated to weird stuff like the mixed view of everything.
 76 		callback: this._controller.show.bind(this._controller, [types])
 77 	};
 78 	this._controller.show(types, null); // Clear the list & set headers
 79 	appCtxt.getSearchController().search(params);
 80 
 81 };
 82

 86 	if (this._inputField) {

 88 	}

 90 	this._controller.cleanup();
 91 };

152 /**
153  * Listview showing deleted items
154  *
155  * @param parent
156  */
157 ZmDumpsterListView = function(parent, controller) {
158 	if (!arguments.length) return;
159 	this._controller = controller;
160 	var params = {
161 		parent: parent,
162 		controller: controller,
163 		pageless: true,
164 		view: this._getViewId(),
165 		headerList: this._getHeaderList(),
166 		type: this._getType(),
167 		parentElement: (parent._htmlElId + "_listview")
168 	};
169 	this._type = this._getType();
170 
171 	ZmListView.call(this, params);
172 
173 	this._listChangeListener = new AjxListener(this, this._changeListener);
174 };

200 ZmDumpsterListView.createView = function(view, parent, controller) {
201 	var app = view.replace(/^dumpster/,"");
202 	switch (app) {
203 		case ZmApp.MAIL:
204 			return new ZmDumpsterMailListView(parent, controller);
205 		case ZmApp.CONTACTS:
206 			return new ZmDumpsterContactListView(parent, controller);
207 		case ZmApp.CALENDAR:
208 			return new ZmDumpsterCalendarListView(parent, controller);
209 		case ZmApp.TASKS:
210 			return new ZmDumpsterTaskListView(parent, controller);
211 		case ZmApp.BRIEFCASE:
212 			return new ZmDumpsterBriefcaseListView(parent, controller);
213 	}
214 };

248 ZmDumpsterMailListView.prototype._getCellContents =
249 function(htmlArr, idx, item, field, colIdx, params) {
250 	var content;
251 	if (field == ZmItem.F_FROM) {
252 		var fromAddr = item.getAddress(AjxEmailAddress.FROM);
253 		if (fromAddr) {
254 			content = "<span style='white-space:nowrap'>";
255 			var name = fromAddr.getName() || fromAddr.getDispName() || fromAddr.getAddress();
256 			content += AjxStringUtil.htmlEncode(name);
257 			content += "</span>";
258 		}
259 	}
260 	else if (field == ZmItem.F_SUBJECT) {
261 		var subj = item.subject || ZmMsg.noSubject;
262 	
263 		content = AjxStringUtil.htmlEncode(subj);
264 	}
265 	if (content) {
266 		htmlArr[idx++] = content;
267 	} else {
268 		idx = ZmListView.prototype._getCellContents.apply(this, arguments);
269 	}
270 	return idx;
271 };

523 ZmDumpsterListController.prototype.show =
524 function(types, results) {
525 
526 	this._appName = ZmItem.APP[types[0]]; // All types should be in the same app
527 	var view = "dumpster" + this._appName;
528 	this.setCurrentViewId(view);
529 	for (var id in this._toolbar) {
530 		this._toolbar[id].setVisible(id == view);
531 	}
532 	for (var id in this._listView) {
533 		this._listView[id].setVisible(id == view);
534 	}
535 	if (results) {
536 		var searchResults = results.getResponse();
537 	
538 		// call base class
539 		ZmListController.prototype.show.call(this, searchResults, view);
540 
541 		this._setup(view);
542 		this._initializeSearchBar();
543 		this._container._updateTabGroup();
544 		this._container._inputField.getInputElement().focus(); //this is the only focus way I could get it to focus on the input element.
545 
546 		var list = searchResults.getResults(searchResults.type);
547 
548 		this.setList(list);
549 		this.setHasMore(searchResults.getAttribute("more"));
550 		this.getCurrentView().set(list);
551 	} else {
552 		this.cleanup();
553 	}
554 };
555 
556 ZmDumpsterListController.prototype.cleanup =
557 function() {
558 	var currentView = this.getCurrentView();
559 	if (currentView) {
560 		currentView.removeAll();
561 	}
562 };

635 ZmDumpsterListController.prototype._getMoveParams =
636 function(dlg) {
637 	var params = ZmListController.prototype._getMoveParams.call(this, dlg);
638 	params.appName = this._appName;
639 	params.overviewId = dlg.getOverviewId(this._appName);
640 	params.treeIds = [ZmApp.ORGANIZER[this._appName]];
641 	params.acceptFolderMatch = true;
642 	params.showDrafts =	true;
643 	var omit = {};
644 	omit[ZmFolder.ID_SPAM] = true;
645 	//bug:60237 remote folders should be excluded from the recovery folder selection
646     var folderTree = appCtxt.getFolderTree();
647 	if (!folderTree) { return params; }
648 	var folders = folderTree.getByType(ZmApp.ORGANIZER[this._appName]);
649 	for (var i = 0; i < folders.length; i++) {
650 		var folder = folders[i];
651         if(folder.link && folder.isRemote()) {
652           omit[folder.id] = true;
653         }
654 	}
655 	params.omit = omit;
656 	return params;
657 };
