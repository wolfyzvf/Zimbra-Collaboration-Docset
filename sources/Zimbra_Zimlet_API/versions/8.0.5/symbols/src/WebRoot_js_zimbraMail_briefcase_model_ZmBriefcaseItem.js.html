 21 /**
 22  * Abstract class
 23  * @class
 24  * This class is a base class for briefcase item classes.
 25  * 
 26  * @param {int}			id			the unique id
 27  * @param {ZmList}		list		a list that contains this item
 28  * @param {Boolean}		noCache		if <code>true</code>, do not cache this item
 29 
 30  * @extends		ZmItem
 31  * 
 32  * @see		ZmBriefcaseBaseItem
 33  */
 34 ZmBriefcaseBaseItem = function(id, list, noCache, type) {
 35 
 36 	if (arguments.length == 0) { return; }
 37 	ZmItem.call(this, type, id, list, noCache);
 38 };

 45 /**
 46  * Gets the path.
 47  * 
 48  * @param	{Boolean}	dontIncludeThisName		if <code>true</code>, do not include this item name in the path
 49  * @return	{String}	the path
 50  */
 51 ZmBriefcaseBaseItem.prototype.getPath =
 52 function(dontIncludeThisName) {
 53 	var briefcase = appCtxt.getById(this.folderId);
 54 	var name = !dontIncludeThisName ? this.name : "";
 55 	return [briefcase.getPath(), "/", name].join("");
 56 };

 58 /**
 59  * Gets the REST URL.
 60  * 
 61  * @param	{Boolean}	dontIncludeThisName		if <code>true</code>, do not include this item name in the path
 62  * @param	{Boolean}	ignoreCustomDocs		if <code>true</code>, ignore custom docs
 63  * @param   {Boolean}   includeVersion			if <code>true</code> include the version if exists (it's latest for the base item)
 64  * @return	{String}	the REST URL
 65  */
 66 ZmBriefcaseBaseItem.prototype.getRestUrl =
 67 function(dontIncludeThisName, ignoreCustomDocs, includeVersion) {
 68 	var url = ZmItem.prototype.getRestUrl.call(this);
 69 	if (dontIncludeThisName) {
 70 		url = url.replace(/[^\/]+$/,"");
 71 	}
 72 	if (includeVersion && this.version){
 73 		url = url + (url.match(/\?/) ? '&' : '?' ) + "ver=" + this.version;
 74 	}
 75 
 76 	return url;
 77 };

129 /**
130  * Gets the icon.
131  * 
132  * @param	{Boolean}	large		if <code>true</code>, return the large icon
133  * @return	{String}	the icon
134  */
135 ZmBriefcaseBaseItem.prototype.getIcon =
136 function(large) {
137 
138 	if (this.isFolder) {
139 		return "Folder";
140 	}
141 
142 	var ct = this.contentType, icon;
143 	if (ct && ct.match(/;/)) {
144 		ct = ct.split(";")[0];
145 	}
146 	var mimeInfo = ct ? ZmMimeTable.getInfo(ct) : null;
147 	if (large) {
148 		icon = mimeInfo ? mimeInfo.imageLarge : "UnknownDoc_48";
149 	} else {
150 		icon = mimeInfo ? mimeInfo.image : "UnknownDoc" ;
151 	}
152 
153 	return icon;
154 };

156 /**
157  * Checks if this item is read only.
158  * 
159  * @return	{Boolean}	<code>true</code> if this item is read only
160  */
161 ZmBriefcaseBaseItem.prototype.isReadOnly =
162 function() {
163 	// if one of the ancestor is readonly then no chances of childs being writable
164 	var isReadOnly = false;
165 	var folder = appCtxt.getById(this.folderId);
166 	var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT);
167 	while (folder && folder.parent && (folder.parent.id != rootId) && !folder.isReadOnly()) {
168 		folder = folder.parent;
169 	}
170 
171 	if (folder && folder.isReadOnly()) {
172 		isReadOnly = true;
173 	}
174 
175 	return isReadOnly;
176 };

207 /**
208  * Creates an item from an attachment.
209  * 
210  * @param	{String}	msgId		the message
211  * @param	{String}	partId		the message part
212  * @param	{String}	name		the item name
213  * @param	{String}	folderId		the folder id
214  */
215 ZmBriefcaseBaseItem.prototype.createFromAttachment =
216 function(msgId, partId, name, folderId, attribs) {
217 
218     attribs = attribs || {};
219 
220     var acctId = appCtxt.getActiveAccount().id;
221 
222     var json = {
223         SaveDocumentRequest: {
224             _jsns: "urn:zimbraMail",
225 			doc: {
226                 m: {
227                     id: msgId,
228                     part: partId
229                 }
230             }
231         }
232     };
233 
234     var doc = json.SaveDocumentRequest.doc;
235     if (attribs.id && attribs.version) {
236         doc.id = attribs.id;
237         doc.ver = attribs.version;
238     }else{
239         doc.l = folderId;
240     }
241     if(attribs.rename){
242         doc.name = attribs.rename;
243     }
244     var params = {
245 		jsonObj: json,
246 		asyncMode: true,
247 		callback: (new AjxCallback(this, this._handleResponseCreateItem, [folderId, attribs.callback])),
248 		errorCallback: (new AjxCallback(this, this._handleErrorCreateItem, [attribs.errorCallback]))
249 	};
250     appCtxt.getAppController().sendRequest(params);
251 };

353 ZmBriefcaseBaseItem.prototype._loadFromDom =
354 function(node) {
355 
356 	this.id = node.id;
357 
358 	if (node.rest)	{ this.restUrl = node.rest; }
359 	if (node.l)		{ this.folderId = node.l; }
360 	if (node.name)	{ this.name = node.name; }
361 	if (node.cr)	{ this.creator = node.cr; }
362 	if (node.d)		{ this.createDate = new Date(Number(node.d)); }
363 	if (node.md)	{ //node.md is seconds since epoch
364         var mdMilliSecs = Number(node.md)*1000;
365         this.modifyDate = new Date(mdMilliSecs);
366     }
367 	if (node.leb)	{ this.modifier = node.leb; }
368 	if (node.s || node.s == 0) //size can be 0
369                     { this.size = Number(node.s); }
370 	if (node.ver)	{ this.version = Number(node.ver) || 0; }
371 	if (node.ct)	{ this.contentType = node.ct.split(";")[0]; }
372 	if (node.tn)	{ this._parseTagNames(node.tn);	}
373     this.locked = false;
374     if (node.loid)    {
375         this.locked = true;
376         this.lockId = node.loid;
377         this.lockUser = node.loe;
378         this.lockTime = new Date(Number(node.lt));
379     }
380 
381     if (node.desc){  this.notes = AjxStringUtil.htmlEncode(node.desc); }
382     this.subject = this.getNotes();
383 
384     this._parseFlags(node.f);
385 
386 };

529 	if (data.l) this.folderId = data.l;
530 	if (data.name) this.name = data.name;
531 	if (data.cr) this.creator = data.cr;
532 	if (data.d) this.createDate = new Date(Number(data.d));
533 	if (data.md)	{ //node.md is seconds since epoch
534 		var mdMilliSecs = Number(data.md)*1000;
535 		this.modifyDate = new Date(mdMilliSecs);
536 	}
537 	if (data.leb) this.modifier = data.leb;
538 	if (data.s) this.size = Number(data.s);
539 	if (data.ver) this.version = Number(data.ver);
540 	if (data.ct) this.contentType = data.ct.split(";")[0];
541     if (data.tn) { this._parseTagNames(data.tn); }

543         this.locked = true;

549         this.locked = false;
550     }
551
