 71 ZmConvListController.prototype.show =
 72 function(searchResults, force) {
 73 	
 74 	if (!force && !this.popShield(null, this.show.bind(this, searchResults, true))) {
 75 		return;
 76 	}
 77 	
 78 	ZmDoublePaneController.prototype.show.call(this, searchResults, searchResults.getResults(ZmItem.CONV));
 79 	if (!appCtxt.isExternalAccount() && !this.isSearchResults && !(searchResults && searchResults.search && searchResults.search.isDefaultToMessageView)) {
 80 		appCtxt.set(ZmSetting.GROUP_MAIL_BY, ZmSetting.GROUP_BY_CONV);
 81 	}

 84 /**
 85  * Handles switching the order of messages within expanded convs.
 86  *
 87  * @param view		[constant]*		the id of the new order
 88  * @param force		[boolean]		if true, always redraw view
 89  */
 90 ZmConvListController.prototype.switchView =
 91 function(view, force) {
 92 
 93 	if (view == ZmSearch.DATE_DESC || view == ZmSearch.DATE_ASC) {
 94 		if (!force && !this.popShield(null, this.switchView.bind(this, view, true))) {
 95 			return;
 96 		}
 97 		if ((appCtxt.get(ZmSetting.CONVERSATION_ORDER) != view) || force) {
 98 			appCtxt.set(ZmSetting.CONVERSATION_ORDER, view);
 99 			if (this._currentViewType == ZmId.VIEW_CONVLIST) {
100 				this._mailListView.redoExpansion();
101 			}
102 			var itemView = this.getItemView();
103 			var conv = itemView && itemView.getItem();
104 			if (conv) {
105 				itemView.set(conv, true);
106 			}
107 		}
108 	} else {
109 		ZmDoublePaneController.prototype.switchView.apply(this, arguments);
110 	}
111 };

122 ZmConvListController.prototype.handleKeyAction =
123 function(actionCode, ev) {
124 
125 	DBG.println(AjxDebug.DBG3, "ZmConvListController.handleKeyAction");
126 	
127 	var mlv = this._mailListView;
128 	
129 	switch (actionCode) {
130 
131 		case ZmKeyMap.EXPAND:
132 		case ZmKeyMap.COLLAPSE:
133 			if (mlv.getSelectionCount() != 1) { return false; }
134 			var item = mlv.getItemFromElement(mlv._kbAnchor);
135 			if (!item) { return false; }
136 			if ((actionCode == ZmKeyMap.EXPAND) != mlv.isExpanded(item)) {
137 				mlv._expandItem(item);
138 			}
139 			break;
140 
141 		case ZmKeyMap.TOGGLE:
142 			if (mlv.getSelectionCount() != 1) { return false; }
143 			var item = mlv.getItemFromElement(mlv._kbAnchor);
144 			if (!item) { return false; }
145 			if (item.type == ZmItem.CONV && !mlv._isExpandable(item)) {
146 				return DwtListView.prototype.handleKeyAction.call(mlv, DwtKeyMap.DBLCLICK);
147 			} else {
148 				mlv._expandItem(item);
149 			}
150 			break;
151 
152 		case ZmKeyMap.EXPAND_ALL:
153 			mlv._expandAll(true);
154 			break;
155 
156 		case ZmKeyMap.COLLAPSE_ALL:
157 			mlv._expandAll(false);
158 			break;
159 
160 		case ZmKeyMap.NEXT_UNREAD_MSG:
161 		case ZmKeyMap.PREV_UNREAD_MSG:
162 			this.lastListAction = actionCode;
163 			var selItem, noBump = false;
164 			if (mlv.getSelectionCount() == 1) {
165 				var sel = mlv.getSelection();
166 				selItem = sel[0];
167 				if (selItem && mlv._isExpandable(selItem)) {
168 					noBump = true;
169 				}
170 			}
171 
172 		case ZmKeyMap.FIRST_UNREAD_MSG:
173 		case ZmKeyMap.LAST_UNREAD_MSG:
174 			var item = (selItem && selItem.type == ZmItem.MSG && noBump) ? selItem :
175 					   this._getUnreadItem(ZmMailListController.ACTION_CODE_WHICH[actionCode], null, noBump);
176 			if (!item) { return; }
177 			if (!mlv.isExpanded(item) && mlv._isExpandable(item)) {
178 				var callback = new AjxCallback(this, this._handleResponseExpand, [actionCode]);
179 				if (item.type == ZmItem.MSG) {
180 					this._expand({conv:appCtxt.getById(item.cid), msg:item, offset:mlv._msgOffset[item.id], callback:callback});
181 				} else {
182 					this._expand({conv:item, callback:callback});
183 				}
184 			} else if (item) {
185 				this._selectItem(mlv, item);
186 			}
187 			break;
188 		
189 		case ZmKeyMap.KEEP_READING:
190 			return this._keepReading(false, ev);
191 			break;
192 
193 		// need to invoke DwtListView method directly since our list view no-ops DBLCLICK
194 		case DwtKeyMap.DBLCLICK:
195 			return DwtListView.prototype.handleKeyAction.apply(mlv, arguments);
196 
197 		// these are for quick reply
198 		case ZmKeyMap.CANCEL:
199 			var itemView = this.getItemView();
200 			if (itemView && itemView._cancelListener) {
201 				itemView._cancelListener();
202 			}
203 			break;
204 		
205 		case ZmKeyMap.SEND:
206 			var itemView = this.getItemView();
207 			if (itemView && itemView._sendListener) {
208 				itemView._sendListener();
209 			}
210 			break;
211 			
212 		default:
213 			return ZmDoublePaneController.prototype.handleKeyAction.apply(this, arguments);
214 	}
215 	return true;
216 };

226 ZmConvListController.prototype._keepReading =
227 function(check, ev) {
228 
229 	if (!this.isReadingPaneOn() || !this._itemViewCurrent()) { return false; }
230 	var mlv = this._mailListView;
231 	if (!mlv || mlv.getSelectionCount() != 1) { return false; }
232 	
233 	var result = false;
234 	var itemView = this.getItemView();
235 	// conv view
236 	if (itemView && itemView.isZmConvView2) {
237 		result = itemView._keepReading(check);
238 		result = result || (check ? !!(this._getUnreadItem(DwtKeyMap.SELECT_NEXT)) :
239 									   this.handleKeyAction(ZmKeyMap.NEXT_UNREAD, ev));
240 	}
241 	// msg view (within an expanded conv)
242 	else if (itemView && itemView.isZmMailMsgView) {
243 		var result = itemView._keepReading(check);
244 		if (!check || !result) {
245 			// go to next unread msg in this expanded conv, otherwise next unread conv
246 			var msg = mlv.getSelection()[0];
247 			var conv = msg && appCtxt.getById(msg.cid);
248 			var msgList = conv && conv.msgs && conv.msgs.getArray();
249 			var msgFound, item;
250 			if (msgList && msgList.length) {
251 				for (var i = 0; i < msgList.length; i++) {
252 					var m = msgList[i];
253 					msgFound = msgFound || (m.id == msg.id);
254 					if (msgFound && m.isUnread) {
255 						item = m;
256 						break;
257 					}
258 				}
259 			}
260 			if (item) {
261 				result = true;
262 				if (!check) {
263 					this._selectItem(mlv, item);
264 				}
265 			}
266 			else {
267 				result = check ? !!(this._getUnreadItem(DwtKeyMap.SELECT_NEXT)) :
268 									this.handleKeyAction(ZmKeyMap.NEXT_UNREAD, ev);
269 			}
270 		}
271 	}
272 	if (!check && result) {
273 		this._checkKeepReading();
274 	}
275 	return result;
276 };

278 /**
279  * Override to handle paging among msgs within an expanded conv.
280  * 
281  * TODO: handle msg paging (current item is expandable msg)
282  * 
283  * @private
284  */
285 ZmConvListController.prototype.pageItemSilently =
286 function(currentItem, forward) {
287 	if (!currentItem) { return; }
288 	if (currentItem.type == ZmItem.CONV) {
289 		ZmMailListController.prototype.pageItemSilently.apply(this, arguments);
290 		return;
291 	}
292 	
293 	var conv = appCtxt.getById(currentItem.cid);
294 	if (!(conv && conv.msgs)) { return; }
295 	var found = false;
296 	var list = conv.msgs.getArray();
297 	for (var i = 0, count = list.length; i < count; i++) {
298 		if (list[i] == currentItem) {
299 			found = true;
300 			break;
301 		}
302 	}
303 	if (!found) { return; }
304 	
305 	var msgIdx = forward ? i + 1 : i - 1;
306 	if (msgIdx >= 0 && msgIdx < list.length) {
307 		var msg = list[msgIdx];
308 		var clv = this._listView[this._currentViewId];
309 		clv.emulateDblClick(msg);
310 	}
311 };

384 /**
385  * Figure out if the given view change is destructive. If so, put up pop shield.
386  * 
387  * @param {string}		viewId		ID of view being hidden
388  * @param {function}	callback	function to call if user agrees to leave
389  * @param {string}		newViewId	ID of view that will be shown
390  */
391 ZmConvListController.prototype.popShield =
392 function(viewId, callback, newViewId) {
393 
394 	var newViewType = newViewId && appCtxt.getViewTypeFromId(newViewId);
395 	var switchingView = (newViewType == ZmId.VIEW_TRAD);
396 	if (this._convView && this._convView.isDirty() && (!newViewType || switchingView)) {
397 		var ps = this._popShield = this._popShield || appCtxt.getYesNoMsgDialog();
398 		ps.reset();
399 		ps.setMessage(ZmMsg.convViewCancel, DwtMessageDialog.WARNING_STYLE);
400 		ps.registerCallback(DwtDialog.YES_BUTTON, this._popShieldYesCallback, this, [switchingView, callback]);
401 		ps.registerCallback(DwtDialog.NO_BUTTON, this._popShieldNoCallback, this, [switchingView, callback]);
402 		ps.popup();
403 		return false;
404 	}
405 	else {
406 		return true;
407 	}
408 };

410 // yes, I want to leave even though I've typed some text
411 ZmConvListController.prototype._popShieldYesCallback =
412 function(switchingView, callback) {
413 	this._convView._replyView.reset();
414 	this._popShield.popdown();
415 	if (switchingView) {
416 		// tell app view mgr it's okay to show TV
417 		appCtxt.getAppViewMgr().showPendingView(true);
418 	}
419 	else if (callback) {
420 		callback();
421 	}
422 };
423 
424 // no, I don't want to leave
425 ZmConvListController.prototype._popShieldNoCallback =
426 function(switchingView, callback) {
427 	this._popShield.popdown();
428 	if (switchingView) {
429 		// attempt to switch to TV was canceled - need to undo changes
430 		this._updateViewMenu(ZmId.VIEW_TRAD);
431 		if (!appCtxt.isExternalAccount() && !this.isSearchResults && !this._currentSearch.isDefaultToMessageView) {
432 			this._app.setGroupMailBy(ZmMailListController.GROUP_BY_SETTING[ZmId.VIEW_TRAD]);
433 		}
434 	}
435 	appCtxt.getKeyboardMgr().grabFocus(this._convView._replyView._input);
436 };

438 ZmConvListController.prototype._listSelectionListener =
439 function(ev) {
440 
441 	var item = ev.item;
442 	if (!item) { return; }
443 	
444 	this._mailListView._selectedMsg = null;
445 	if (ev.field == ZmItem.F_EXPAND && this._mailListView._isExpandable(item)) {
446 		this._toggle(item, false);
447 		return true;
448 	} else {
449 		var handled = ZmDoublePaneController.prototype._listSelectionListener.apply(this, arguments);
450 		if (!handled) {
451 			if (ev.detail == DwtListView.ITEM_DBL_CLICKED) {
452 				if (item.type == ZmItem.CONV && item.numMsgs == 1) {
453 					var msg = item.getFirstHotMsg();
454 					item = msg || item;
455 				}
456 				if (item.type == ZmItem.MSG) {
457 					AjxDispatcher.run("GetMsgController", item && item.nId).show(item, this, null, true);
458 				} else {
459 					AjxDispatcher.run("GetConvController").show(item, this, null, true);
460 				}
461 				return true;
462 			}
463 		}
464 	}
465 	return false;
466 };

474 ZmConvListController.prototype._setSelectedItem =
475 function() {
476 	
477 	var selCnt = this._listView[this._currentViewId].getSelectionCount();
478 	if (selCnt == 1) {
479 		var sel = this._listView[this._currentViewId].getSelection();
480 		var item = (sel && sel.length) ? sel[0] : null;
481 		if (item.type == ZmItem.CONV) {
482 			Dwt.setLoadingTime("ZmConv", new Date());
483 			var convParams = {};
484 			convParams.markRead = this._handleMarkRead(item, true);
485 			if (this.isSearchResults) {
486 				convParams.getMatches = true;
487 			}
488 			else {
489 				convParams.getUnreadOrFirstMsg = true;
490 				// generalize query so if conv is read we get latest msg and not just latest matching msg
491 				var acctId = item.isShared() && ZmOrganizer.parseId(item.id).acctId;
492 				var rootFolderId = acctId ? ['"', acctId, ':1"'].join("") : "1";
493 				var terms = ["underid:" + rootFolderId];
494 				var search = this._currentSearch;
495 				if (search) {
496 					var foldersToExclude = [ZmFolder.ID_TRASH, ZmFolder.ID_SPAM];
497 					for (var i = 0; i < foldersToExclude.length; i++) {
498 						var folderId = foldersToExclude[i];
499 						if (acctId) {
500 							folderId = ['"', acctId, ":", folderId, '"'].join("");
501 						}
502 						// note that Trash and Spam cannot be shared
503 						if (acctId || !search.hasFolderTerm(ZmFolder.QUERY_NAME[folderId])) {
504 							terms.push("underid:" + folderId);
505 						}
506 					}
507 				}
508 				convParams.query = terms.join(" AND NOT ");
509 			}
510 			// if the conv's unread state changed, load it again so we get the correct expanded msg bodies
511 			convParams.forceLoad = item.unreadHasChanged;
512 			item.load(convParams, this._handleResponseSetSelectedItem.bind(this, item));
513 		} else {
514 			ZmDoublePaneController.prototype._setSelectedItem.apply(this, arguments);
515 		}
516 	}
517 };

558 /**
559  * Returns the first matching msg in the conv, if available. No request will
560  * be made to the server if the conv has not been loaded.
561  */
562 ZmConvListController.prototype.getMsg =
563 function(params) {
564 	
565 	// First see if action is being performed on a msg in the conv view in the reading pane
566 	var lv = this._listView[this._currentViewId];
567 	var msg = lv && lv._selectedMsg;
568 	if (msg && DwtMenu.menuShowing()) {
569 		return msg;
570 	}
571 	
572 	var sel = lv.getSelection();
573 	var item = (sel && sel.length) ? sel[0] : null;
574 	if (item) {
575 		if (item.type == ZmItem.CONV) {
576 			return item.getFirstHotMsg(params);
577 		} else if (item.type == ZmItem.MSG) {
578 			return ZmDoublePaneController.prototype.getMsg.apply(this, arguments);
579 		}
580 	}
581 	return null;
582 };

588 function(params, callback) {
589 	params = params || {};
590 	var sel = this._listView[this._currentViewId].getSelection();
591 	var item = (sel && sel.length) ? sel[0] : null;
592 	if (item) {
593 		if (item.type == ZmItem.CONV) {
594 			params.markRead = (params.markRead != null) ? params.markRead : this._handleMarkRead(item, true);
595 			var respCallback = new AjxCallback(this, this._handleResponseGetLoadedMsg, callback);
596 			item.getFirstHotMsg(params, respCallback);
597 		} else if (item.type == ZmItem.MSG) {
598 			ZmDoublePaneController.prototype._getLoadedMsg.apply(this, arguments);
599 		}
600 	} else {
601 		callback.run();
602 	}
603 };

645 /**
646  * Expands the given conv or msg, performing a search to get items if necessary.
647  *
648  * @param params		[hash]			hash of params:
649  *        conv			[ZmConv]		conv to expand
650  *        msg			[ZmMailMsg]		msg to expand (get next page of msgs for conv)
651  *        offset		[int]			index of msg in conv
652  *        getFirstMsg	[boolean]		if true, fetch body of first msg
653  *        callback		[AjxCallback]	callback to run when done
654  */
655 ZmConvListController.prototype._expand =
656 function(params) {
657 
658 	var conv = params.conv;
659 	var offset = params.offset || 0;
660 	var respCallback = new AjxCallback(this, this._handleResponseLoadItem, [params]);
661 	var pageWasCached = false;
662 	if (offset) {
663 		if (this._paginateConv(conv, offset, respCallback)) {
664 			// page was cached, callback won't be run
665 			this._handleResponseLoadItem(params, new ZmCsfeResult(conv.msgs));
666 		}
667 	} else if (!conv._loaded) {
668 		// no msgs have been loaded yet
669 		var getFirstMsg = (params.getFirstMsg === false) ? false : this.isReadingPaneOn();
670 		conv.load({getFirstMsg:getFirstMsg}, respCallback);
671 	} else {
672 		// re-expanding first page of msgs
673 		this._handleResponseLoadItem(params, new ZmCsfeResult(conv.msgs));
674 	}
675 };

677 ZmConvListController.prototype._handleResponseLoadItem =
678 function(params, result) {
679 	if (result) {
680 		this._mailListView._expand(params.conv, params.msg);
681 	}
682 	if (params.callback) {
683 		params.callback.run();
684 	}
685 };

687 /**
688  * Adapted from ZmListController::_paginate
689  */
690 ZmConvListController.prototype._paginateConv =
691 function(conv, offset, callback) {
692 
693 	var list = conv.msgs;
694 	// see if we're out of msgs and the server has more
695 	var limit = appCtxt.get(ZmSetting.CONVERSATION_PAGE_SIZE);
696 	if (offset && list && ((offset + limit > list.size()) && list.hasMore())) {
697 		// figure out how many items we need to fetch
698 		var delta = (offset + limit) - list.size();
699 		var max = delta < limit && delta > 0 ? delta : limit;
700 		if (max < limit) {
701 			offset = ((offset + limit) - max) + 1;
702 		}
703 		var respCallback = new AjxCallback(this, this._handleResponsePaginateConv, [conv, offset, callback]);
704 		var getFirstMsg = this.isReadingPaneOn();
705 		conv.load({offset:offset, limit:limit, getFirstMsg:getFirstMsg}, respCallback);
706 		return false;
707 	} else {
708 		return true;
709 	}
710 };
711 
712 ZmConvListController.prototype._handleResponsePaginateConv =
713 function(conv, offset, callback, result) {
714 
715 	if (!conv.msgs) { return; }
716 
717 	var searchResult = result.getResponse();
718 	conv.msgs.setHasMore(searchResult.getAttribute("more"));
719 	var newList = searchResult.getResults(ZmItem.MSG).getVector();
720 	conv.msgs.cache(offset, newList);
721 	if (callback) {
722 		callback.run(result);
723 	}
724 };

737 //
738 // Since a selection might contain both convs and msgs, we need to split them up and
739 // invoke the action for each type separately.

741 /**
742  * Takes the given list of items (convs and msgs) and splits it into one list of each
743  * type. Since an action applied to a conv is also applied to its msgs, we remove any
744  * msgs whose owning conv is also in the list.
745  */
746 ZmConvListController.prototype._divvyItems =
747 function(items) {
748 	var convs = [], msgs = [];
749 	var convIds = {};
750 	for (var i = 0; i < items.length; i++) {
751 		var item = items[i];
752 		if (item.type == ZmItem.CONV) {
753 			convs.push(item);
754 			convIds[item.id] = true;
755 		} else {
756 			msgs.push(item);
757 		}
758 	}
759 	var msgs1 = [];
760 	for (var i = 0; i < msgs.length; i++) {
761 		if (!convIds[msgs[i].cid]) {
762 			msgs1.push(msgs[i]);
763 		}
764 	}
765 	var lists = {};
766 	lists[ZmItem.MSG] = msgs1;	
767 	lists[ZmItem.CONV] = convs;
768 	
769 	return lists;
770 };

816 /**
817  * Splits the given items into two lists, one of convs and one of msgs, and
818  * applies the given method and args to each.
819  *
820  * @param items		[array]			list of convs and/or msgs
821  * @param method	[string]		name of function to call in parent class
822  * @param args		[array]			additional args to pass to function
823  */
824 ZmConvListController.prototype._applyAction =
825 function(items, method, args) {
826 	args = args ? args : [];
827 	var lists = this._divvyItems(items);
828 	var hasMsgs = false;
829 	if (lists[ZmItem.MSG] && lists[ZmItem.MSG].length) {
830 		args.unshift(lists[ZmItem.MSG]);
831 		ZmDoublePaneController.prototype[method].apply(this, args);
832 		hasMsgs = true;
833 	}
834 	if (lists[ZmItem.CONV] && lists[ZmItem.CONV].length) {
835 		if (hasMsgs) {
836 			args[0] = lists[ZmItem.CONV];
837 		}
838 		else {
839 			args.unshift(lists[ZmItem.CONV]);
840 		}
841 		ZmDoublePaneController.prototype[method].apply(this, args);
842 	}
843 };
