 21 /**
 22  * Creates a toolbar.
 23  * @class
 24  * This class represents a basic toolbar which can add buttons, manage listeners, and
 25  * enable/disabled its buttons.
 26  *
 27  * @author Conrad Damon
 28  *
 29  * @param {Hash}	params		a hash of parameters
 30  * @param	{DwtComposite}	params.parent		the containing widget
 31  * @param	{String}	params.className	the CSS class
 32  * @param	{constant}	params.posStyle		the positioning style
 33  * @param	{String}	params.id			an explicit ID to use for the control's HTML element
 34  * @param	{ZmController}	params.controller	the owning controller
 35  * @param	{String}	params.refElementId	the id of element that contains toolbar
 36  *        
 37  * @extends	DwtToolBar
 38  */
 39 ZmToolBar = function(params) {
 40 	if (arguments.length == 0) return;
 41 
 42 	params.posStyle = params.posStyle || DwtControl.ABSOLUTE_STYLE;
 43 	DwtToolBar.call(this, params);
 44 
 45 	var controller = params.controller || appCtxt.getCurrentController();
 46 	if (controller) {
 47 		this._controller = controller;
 48 		this._keyMap = this._controller.getKeyMapName();
 49 	}
 50 
 51 	this._refElementId = params.refElementId;
 52 	this._buttons = {};
 53 };
 54

 56 ZmToolBar.prototype.constructor = ZmToolBar;

128 /**
129  * Enables or disables the specified buttons.
130  *
131  * @param {Array}	ids		a list of button ids
132  * @param {Boolean}	enabled	if <code>true</code>, enable the buttons
133  */
134 ZmToolBar.prototype.enable =
135 function(ids, enabled) {
136 	ids = (ids instanceof Array) ? ids : [ids];
137 	for (var i = 0; i < ids.length; i++) {
138 		if (this._buttons[ids[i]]) {
139 			this._buttons[ids[i]].setEnabled(enabled);
140 		}
141 	}
142 };
143 
144 ZmToolBar.prototype.setSelected =
145 function(id) {
146     var oldButton = this._selectedId ? this._buttons[this._selectedId] : null;
147     var newButton = id ? this._buttons[id] : null;
148     if (oldButton) {
149         oldButton.setSelected(false);
150         oldButton._noFocus = false;
151     }
152     if (newButton) {
153         newButton.setSelected(true);
154         newButton._noFocus = true;
155 
156         this._selectedId = id;
157     }
158 };

172 /**
173  * Creates a button and adds the button to this toolbar.
174  *
175  * @param {String}	id			the button id
176  * @param {Hash}	params		a hash of parameters:
177  * @param {function}	params.constructor	the constructor for button object (default is {@link DwtToolBarButton})
178  * @param {String}	params.template		the button template
179  * @param {String}	params.text			the button text
180  * @param {String}	params.tooltip		the button tooltip text
181  * @param {String}	params.image		the icon class for the button
182  * @param {String}	params.disImage	the disabled version of icon
183  * @param {Boolean}	params.enabled		if <code>true</code>, button is enabled
184  * @param {String}	params.className	the CSS class name
185  * @param {String}	params.style		the button style
186  * @param {int}	params.index			the position at which to add the button
187  * @param {constant}	params.shortcut		the shortcut id (from {@link ZmKeyMap}) for showing hint
188  * @param {AjxCallback|DwtMenu}	params.menu				the menu creation callback (recommended) or menu
189  * @param {Boolean}	params.menuAbove	if <code>true</code>, popup menu above the button.
190  *
191  * @param {Object}	params.whatToShow		if exists, determines what to show as follows: (for usage, see ZmToolBar.prototype._createButton and DwtButton.prototype.setImage and DwtButton.prototype.setText
192  * @param {Boolean}	params.whatToShow.showImage		if <code>true</code>, display image
193  * @param {Boolean}	params.whatToShow.showText		if <code>true</code>, display text
194  *
195  */
196 ZmToolBar.prototype.createButton =
197 function(id, params) {
198 	var b = this._buttons[id] = this._createButton(params);
199 	if (params.image) {
200 		b.setImage(params.image);
201 	}
202 	if (params.text) {
203 		b.setText(params.text);
204 	}
205 	if (params.tooltip) {
206 		b.setToolTipContent(ZmOperation.getToolTip(id, this._keyMap) || params.tooltip, true);
207 	}
208 	b.setEnabled(params.enabled !== false);
209 	b.setData("_buttonId", id);
210 	if (params.menu) {
211 		b.setMenu(params.menu, false, null, params.menuAbove);
212 	}
213 
214 	return b;
215 };

231 function(params, className) {
232 	var ctor = params.ctor || DwtToolBarButton;
233     var button = new ctor({

235 		style:params.style,
236 		className:className,
237 		index:params.index,
238 		id:params.id,
239 		template: params.template

241 	button.textPrecedence = params.textPrecedence;
242 	button.imagePrecedence = params.imagePrecedence;
243 	button.whatToShow = params.whatToShow;
244 
245 	return button;
246 };

256 /**
257  * Creates an ordered list of which bits of text or images get removed when we need space.
258  * 
259  * @private
260  */
261 ZmToolBar.prototype._createPrecedenceList =
262 function() {
263 	this._precedenceList = [];
264 	for (var id in this._buttons) {
265 		if (ZmOperation.isSep(id)) { continue; }
266 		var b = this._buttons[id];
267 		var tp = b.textPrecedence;
268 		if (tp) {
269 			this._precedenceList.push({id:id, type:"text", precedence:tp});
270 		}
271 		var ip = b.imagePrecedence;
272 		if (ip) {
273 			this._precedenceList.push({id:id, type:"image", precedence:ip});
274 		}
275 	}
276 	this._precedenceList.sort(function(a, b) {
277 		return (a.precedence > b.precedence) ? 1 : (a.precedence < b.precedence) ? -1 : 0;
278 	});
279 };

284 /**
285  * Sets the size. This method is called by the application view manager <code>fitToContainer()</code>,
286  * which happens during initial layout as well as in response to the user changing the browser size.
287  * 
288  * @param	{int}	width	the width (in pixels)
289  * @param	{int}	height	the height (in pixels)
290  */
291 ZmToolBar.prototype.setSize =
292 function(width, height) {
293 	DBG.println("tb", "------ setSize " + width + " x " + height);
294 	var sz = this.getSize();
295 	if (sz && (width != sz.x || height != sz.y)) {
296 		DwtToolBar.prototype.setSize.apply(this, arguments);
297 	}
298 };

300 ZmToolBar.prototype.adjustSize =
301 function() {
302 	if (!this._refElementId || !this._inited) { return; }
303     if (!this._refElement) {
304         this._refElement = document.getElementById(this._refElementId);
305     }
306     var container = this._refElement && this._refElement.parentNode;
307     if (container && container.offsetWidth >= 30) {
308         this._refElement.style.maxWidth = this._refElement.style.width =  (container.offsetWidth - 30);
309         this._refElement.style.overflow = "hidden";
310     }
311 }

328 /**
329  * Adds a button to the element with the given ID. Designed to handle non-ZmToolBar toolbars.
330  * 
331  * @param params	[hash]			hash of params:
332  * 		  parent	[DwtControl]	parent control
333  *        setting	[const]			setting that must be true for this button to be added
334  *        tdId		[string]		ID of TD that is to contain this button
335  *        buttonId	[string]*		ID of the button
336  *        style		[const]*		button style
337  *        type		[string]*		used to differentiate between regular and toolbar buttons
338  *        lbl		[string]*		button text
339  *        icon		[string]*		button icon
340  *        tooltip	[string]*		button tooltip
341  */
342 ZmToolBar.addButton =
343 function(params) {
344 
345 	if (params.setting && !appCtxt.get(params.setting)) { return; }
346 
347 	var button;
348 	var tdId = params.parent._htmlElId + (params.tdId || params.buttonId);
349 	var buttonEl = document.getElementById(tdId);
350 	if (buttonEl) {
351 		var btnParams = {parent:params.parent, style:params.style, id:params.buttonId, template: params.template, className: params.className};
352 		button = (params.type && params.type == "toolbar") ? (new DwtToolBarButton(btnParams)) : (new DwtButton(btnParams));
353 		var hint = Dwt.getAttr(buttonEl, "hint");
354 		ZmToolBar._setButtonStyle(button, hint, params.lbl, params.icon);
355 		if (params.tooltip) {
356 			button.setToolTipContent(params.tooltip, true);
357 		}
358 		button.reparentHtmlElement(tdId);
359 	}
360 
361 	return button;
362 };
363 
364 ZmToolBar._setButtonStyle =
365 function(button, hint, text, image) {
366 	if (hint == "text") {
367 		button.setText(text);
368 	} else if (hint == "icon") {
369 		button.setImage(image);
370 	} else { // add icon and text if no hint (or unsupported hint) provided
371 		button.setText(text);
372 		button.setImage(image);
373 	}
374 };
