 16 /**
 17  * Creates a mime part.
 18  * @class
 19  * This class represents a mime part. Note that the content of the node is
 20  * not copied into this object, for performance reasons. It is typically
 21  * available via the 'bodyParts' list that is populated during node parsing.
 22  * 
 23  * @extends		ZmModel
 24  */
 25 ZmMimePart = function(parent) {
 26 	
 27 	ZmModel.call(this, ZmEvent.S_ATT);
 28 	
 29 	this.parent = parent;
 30 	this.children = new AjxVector();
 31 };
 32 
 33 ZmMimePart.prototype = new ZmModel;
 34 ZmMimePart.prototype.constructor = ZmMimePart;
 35 
 36 ZmMimePart.prototype.isZmMimePart = true;
 37 ZmMimePart.prototype.toString = function() { return "ZmMimePart"; };
 38 
 39 /**
 40  * Returns a ZmMimePart constructed from the given JSON object. If a context
 41  * hash is provided with 'attachments' and 'bodyParts' arrays, and a hash
 42  * 'contentTypes', those will be populated as the node is recursively parsed.
 43  * 
 44  * @param	{object}		node	JSON representation of MIME part
 45  * @param	{hash}			ctxt	optional context
 46  * @return	{ZmMimePart}			a MIME part
 47  */
 48 ZmMimePart.createFromDom =
 49 function(node, ctxt, parent) {
 50 	var mimePart = new ZmMimePart(parent);
 51 	mimePart._loadFromDom(node, ctxt);
 52 	return mimePart;
 53 };

 55 /**
 56  * Returns this part's content.
 57  * 
 58  * @return	{string}	content		the content
 59  */
 60 ZmMimePart.prototype.getContent = 
 61 function() {
 62 	return this.content || (this.node && this.node.content) || "";
 63 };
 64 
 65 /**
 66  * Returns content of the given type, in or below this part.
 67  * 
 68  * @param	{string}	contentType		the content type
 69  * @return	{string}					the content
 70  * 
 71  */
 72 ZmMimePart.prototype.getContentForType = 
 73 function(contentType) {
 74 
 75 	if (this.contentType == contentType) {
 76 		return this.getContent();
 77 	}
 78 	else {
 79 		var children = this.children.getArray();
 80 		if (children.length) {
 81 			for (var i = 0; i < children.length; i++) {
 82 				var content = children[i].getContentForType(contentType);
 83 				if (content) {
 84 					return content;
 85 				}
 86 			}
 87 		}
 88 	}
 89 	return "";
 90 };

152 /**
153  * Returns true if this part should not be considered to be an attachment.
154  * 
155  * @return	{boolean}
156  */
157 ZmMimePart.prototype.isIgnoredPart =
158 function() {
159 	// bug fix #5889 - if parent node was multipart/appledouble,
160 	// ignore all application/applefile attachments - YUCK
161 	if (this.parent && this.parent.contentType == ZmMimeTable.MULTI_APPLE_DBL &&
162 		this.contentType == ZmMimeTable.APP_APPLE_DOUBLE)
163 	{
164 		return true;
165 	}
166 
167 	// bug fix #7271 - dont show renderable body parts as attachments anymore
168 	if (this.isBody && this.getContent() && 
169 		(this.contentType == ZmMimeTable.TEXT_HTML || this.contentType == ZmMimeTable.TEXT_PLAIN))
170 	{
171 		return true;
172 	}
173 
174 	if (this.contentType == ZmMimeTable.MULTI_DIGEST) {
175 		return true;
176 	}
177 
178 	return false;
179 };
180 
181 ZmMimePart.prototype._loadFromDom =
182 function(node, ctxt) {
183 	
184 	this._loadProps(node);
185 	
186 	if (node.content) {
187 		this._loaded = true;
188 	}
189 	
190 	if (ctxt.contentTypes) {
191 		ctxt.contentTypes[node.ct] = true;
192 	}
193 
194 	var isAtt = false;
195 	if (this.contentDisposition == "attachment" || 
196 		this.contentType == ZmMimeTable.MSG_RFC822 || this.contentType == ZmMimeTable.TEXT_CAL ||            
197 		this.fileName || this.contentId || this.contentLocation) {
198 
199 		if (!this.isIgnoredPart()) {
200 			if (ctxt.attachments) {
201 				ctxt.attachments.push(this);
202 			}
203 			isAtt = true;
204 		}
205 	}
206 
207 	if (this.isBody) {
208 		var hasContent = AjxUtil.isSpecified(node.content);
209 		if ((ZmMimeTable.isRenderableImage(this.contentType) || hasContent)) {
210 			if (ctxt.bodyParts) {
211 				if (this.contentType == ZmMimeTable.MULTI_ALT) {
212 					ctxt.bodyParts.push({});
213 				}
214 				else if (ZmMimePart._isPartOfMultipartAlternative(this)) {
215 					var altPart = {};
216 					altPart[this.contentType] = this;
217 					ctxt.bodyParts.push(altPart);
218 				}
219 				else {
220 					ctxt.bodyParts.push(this);
221 				}
222 			}
223 			if (isAtt && ctxt.attachments) {
224 				//To avoid duplication, Remove attachment that was just added as bodypart.
225 				ctxt.attachments.pop();
226 			}
227 		} else if (!isAtt && this.size != 0 && !this.isIgnoredPart()){
228 			if (ctxt.attachments) {
229 				ctxt.attachments.push(this);
230 			}
231 			isAtt = true;
232 		}
233 	}
234 
235 	// bug fix #4616 - dont add attachments part of a rfc822 msg part
236 	if (node.mp && this.contentType != ZmMimeTable.MSG_RFC822) {
237 		for (var i = 0; i < node.mp.length; i++) {
238 			this.children.add(ZmMimePart.createFromDom(node.mp[i], ctxt, this));
239 		}
240 	}
241 };

243 ZmMimePart.prototype._loadProps =
244 function(node) {
245 
246 	this.node				= node;
247 	
248 	// the middle column is for backward compatibility
249 	this.contentType		= this.ct			= node.ct;
250 	this.name									= node.name;
251 	this.part									= node.part;
252 	this.cachekey								= node.cachekey;
253 	this.size				= this.s			= node.s;
254 	this.contentDisposition	= this.cd			= node.cd;
255 	this.contentId			= this.ci			= node.ci;
256 	this.contentLocation	= this.cl			= node.cl;
257 	this.fileName			= this.filename		= node.filename;
258 	this.isTruncated		= this.truncated	= !!(node.truncated);
259 	this.isBody				= this.body			= !!(node.body);
260 };

262 /*
263  * @param {object}		parentNode
264  * @return {true/false}	true if one of the parent in the hierarchy is multipart/alternative otherwise false.
265 */
266 
267 ZmMimePart._isPartOfMultipartAlternative =
268 function(part){
269     if (!part) { return false; }
270     if (part.contentType == ZmMimeTable.MULTI_ALT) { return true; }
271     return ZmMimePart._isPartOfMultipartAlternative(part.parent);
272 };

274 /**
275  * Checks within the given node tree for content within a multipart/alternative part that
276  * we don't have, and then creates and adds a MIME part for it. Assumes that there will be
277  * at most one multipart/alternative.
278  * 
279  * @param {object}		node			
280  * @param {string}		contentType
281  * @param {int}			index
282  * 
283  * @return {ZmMimePart}		the MIME part that was created and added
284  */
285 ZmMimePart.prototype.addAlternativePart =
286 function(node, contentType, index) {
287 
288 	// replace this part if we got new content
289 	if (node.ct == contentType && ZmMimePart._isPartOfMultipartAlternative(this.parent) && node.body &&  !this.getContent()) {
290 		var mimePart = new ZmMimePart(this);
291 		mimePart._loadProps(node);
292 		this.parent.children.replace(index, mimePart);
293 		return mimePart;
294 	}
295 	if (node.mp && node.mp.length) {
296 		for (var i = 0; i < node.mp.length; i++) {
297 			var mimePart = this.children.get(i);
298 			var altPart = mimePart && mimePart.addAlternativePart(node.mp[i], contentType, i);
299 			if (altPart) {
300 				return altPart;
301 			}
302 		}
303 	}
304 };
