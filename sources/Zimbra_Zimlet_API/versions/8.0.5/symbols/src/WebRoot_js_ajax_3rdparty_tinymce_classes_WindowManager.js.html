 25 (function(tinymce) {
 26 	var Dispatcher = tinymce.util.Dispatcher, each = tinymce.each, isIE = tinymce.isIE, isOpera = tinymce.isOpera;
 27 
 28 	/**
 29 	 * This class handles the creation of native windows and dialogs. This class can be extended to provide for example inline dialogs.
 30 	 *
 31 	 * @class tinymce.WindowManager
 32 	 * @example
 33 	 * // Opens a new dialog with the file.htm file and the size 320x240
 34 	 * // It also adds a custom parameter this can be retrieved by using tinyMCEPopup.getWindowArg inside the dialog.
 35 	 * tinyMCE.activeEditor.windowManager.open({
 36 	 *    url : 'file.htm',
 37 	 *    width : 320,
 38 	 *    height : 240
 39 	 * }, {
 40 	 *    custom_param : 1
 41 	 * });
 42 	 * 
 43 	 * // Displays an alert box using the active editors window manager instance
 44 	 * tinyMCE.activeEditor.windowManager.alert('Hello world!');
 45 	 * 
 46 	 * // Displays an confirm box and an alert message will be displayed depending on what you choose in the confirm
 47 	 * tinyMCE.activeEditor.windowManager.confirm("Do you want to do something", function(s) {
 48 	 *    if (s)
 49 	 *       tinyMCE.activeEditor.windowManager.alert("Ok");
 50 	 *    else
 51 	 *       tinyMCE.activeEditor.windowManager.alert("Cancel");
 52 	 * });
 53 	 */
 54 	tinymce.create('tinymce.WindowManager', {
 55 		/**
 56 		 * Constructs a new window manager instance.
 57 		 *
 58 		 * @constructor
 59 		 * @method WindowManager
 60 		 * @param {tinymce.Editor} ed Editor instance that the windows are bound to.
 61 		 */
 62 		WindowManager : function(ed) {
 63 			var t = this;
 64 
 65 			t.editor = ed;
 66 			t.onOpen = new Dispatcher(t);
 67 			t.onClose = new Dispatcher(t);
 68 			t.params = {};
 69 			t.features = {};
 70 		},
 71 
 72 		/**
 73 		 * Opens a new window.
 74 		 *
 75 		 * @method open
 76 		 * @param {Object} s Optional name/value settings collection contains things like width/height/url etc.
 77 		 * @option {String} title Window title. 
 78 		 * @option {String} file URL of the file to open in the window. 
 79 		 * @option {Number} width Width in pixels. 
 80 		 * @option {Number} height Height in pixels. 
 81 		 * @option {Boolean} resizable Specifies whether the popup window is resizable or not. 
 82 		 * @option {Boolean} maximizable Specifies whether the popup window has a "maximize" button and can get maximized or not. 
 83 		 * @option {Boolean} inline Specifies whether to display in-line (set to 1 or true for in-line display; requires inlinepopups plugin). 
 84 		 * @option {String/Boolean} popup_css Optional CSS to use in the popup. Set to false to remove the default one. 
 85 		 * @option {Boolean} translate_i18n Specifies whether translation should occur or not of i18 key strings. Default is true. 
 86 		 * @option {String/bool} close_previous Specifies whether a previously opened popup window is to be closed or not (like when calling the file browser window over the advlink popup). 
 87 		 * @option {String/bool} scrollbars Specifies whether the popup window can have scrollbars if required (i.e. content larger than the popup size specified). 
 88 		 * @param {Object} p Optional parameters/arguments collection can be used by the dialogs to retrive custom parameters.
 89 		 * @option {String} plugin_url url to plugin if opening plugin window that calls tinyMCEPopup.requireLangPack() and needs access to the plugin language js files 
 90 		 */
 91 		open : function(s, p) {
 92 			var t = this, f = '', x, y, mo = t.editor.settings.dialog_type == 'modal', w, sw, sh, vp = tinymce.DOM.getViewPort(), u;
 93 
 94 			// Default some options
 95 			s = s || {};
 96 			p = p || {};
 97 			sw = isOpera ? vp.w : screen.width; // Opera uses windows inside the Opera window
 98 			sh = isOpera ? vp.h : screen.height;
 99 			s.name = s.name || 'mc_' + new Date().getTime();
100 			s.width = parseInt(s.width || 320);
101 			s.height = parseInt(s.height || 240);
102 			s.resizable = true;
103 			s.left = s.left || parseInt(sw / 2.0) - (s.width / 2.0);
104 			s.top = s.top || parseInt(sh / 2.0) - (s.height / 2.0);
105 			p.inline = false;
106 			p.mce_width = s.width;
107 			p.mce_height = s.height;
108 			p.mce_auto_focus = s.auto_focus;
109 
110 			if (mo) {
111 				if (isIE) {
112 					s.center = true;
113 					s.help = false;
114 					s.dialogWidth = s.width + 'px';
115 					s.dialogHeight = s.height + 'px';
116 					s.scroll = s.scrollbars || false;
117 				}
118 			}
119 
120 			// Build features string
121 			each(s, function(v, k) {
122 				if (tinymce.is(v, 'boolean'))
123 					v = v ? 'yes' : 'no';
124 
125 				if (!/^(name|url)$/.test(k)) {
126 					if (isIE && mo)
127 						f += (f ? ';' : '') + k + ':' + v;
128 					else
129 						f += (f ? ',' : '') + k + '=' + v;
130 				}
131 			});
132 
133 			t.features = s;
134 			t.params = p;
135 			t.onOpen.dispatch(t, s, p);
136 
137 			u = s.url || s.file;
138 			u = tinymce._addVer(u);
139 
140 			try {
141 				if (isIE && mo) {
142 					w = 1;
143 					window.showModalDialog(u, window, f);
144 				} else
145 					w = window.open(u, s.name, f);
146 			} catch (ex) {
147 				// Ignore
148 			}
149 
150 			if (!w)
151 				alert(t.editor.getLang('popup_blocked'));
152 		},
153 
154 		/**
155 		 * Closes the specified window. This will also dispatch out a onClose event.
156 		 *
157 		 * @method close
158 		 * @param {Window} w Native window object to close.
159 		 */
160 		close : function(w) {
161 			w.close();
162 			this.onClose.dispatch(this);
163 		},
164 
165 		/**
166 		 * Creates a instance of a class. This method was needed since IE can't create instances
167 		 * of classes from a parent window due to some reference problem. Any arguments passed after the class name
168 		 * will be passed as arguments to the constructor.
169 		 *
170 		 * @method createInstance
171 		 * @param {String} cl Class name to create an instance of.
172 		 * @return {Object} Instance of the specified class.
173 		 * @example
174 		 * var uri = tinyMCEPopup.editor.windowManager.createInstance('tinymce.util.URI', 'http://www.somesite.com');
175 		 * alert(uri.getURI());
176 		 */
177 		createInstance : function(cl, a, b, c, d, e) {
178 			var f = tinymce.resolve(cl);
179 
180 			return new f(a, b, c, d, e);
181 		},
182 
183 		/**
184 		 * Creates a confirm dialog. Please don't use the blocking behavior of this
185 		 * native version use the callback method instead then it can be extended.
186 		 *
187 		 * @method confirm
188 		 * @param {String} t Title for the new confirm dialog.
189 		 * @param {function} cb Callback function to be executed after the user has selected ok or cancel.
190 		 * @param {Object} s Optional scope to execute the callback in.
191 		 * @example
192 		 * // Displays an confirm box and an alert message will be displayed depending on what you choose in the confirm
193 		 * tinyMCE.activeEditor.windowManager.confirm("Do you want to do something", function(s) {
194 		 *    if (s)
195 		 *       tinyMCE.activeEditor.windowManager.alert("Ok");
196 		 *    else
197 		 *       tinyMCE.activeEditor.windowManager.alert("Cancel");
198 		 * });
199 		 */
200 		confirm : function(t, cb, s, w) {
201 			w = w || window;
202 
203 			cb.call(s || this, w.confirm(this._decode(this.editor.getLang(t, t))));
204 		},
205 
206 		/**
207 		 * Creates a alert dialog. Please don't use the blocking behavior of this
208 		 * native version use the callback method instead then it can be extended.
209 		 *
210 		 * @method alert
211 		 * @param {String} t Title for the new alert dialog.
212 		 * @param {function} cb Callback function to be executed after the user has selected ok.
213 		 * @param {Object} s Optional scope to execute the callback in.
214 		 * @example
215 		 * // Displays an alert box using the active editors window manager instance
216 		 * tinyMCE.activeEditor.windowManager.alert('Hello world!');
217 		 */
218 		alert : function(tx, cb, s, w) {
219 			var t = this;
220 
221 			w = w || window;
222 			w.alert(t._decode(t.editor.getLang(tx, tx)));
223 
224 			if (cb)
225 				cb.call(s || t);
226 		},
227 
228 		/**
229 		 * Resizes the specified window or id.
230 		 *
231 		 * @param {Number} dw Delta width.
232 		 * @param {Number} dh Delta height.
233 		 * @param {window/id} win Window if the dialog isn't inline. Id if the dialog is inline.
234 		 */
235 		resizeBy : function(dw, dh, win) {
236 			win.resizeBy(dw, dh);
237 		},
238 
239 		// Internal functions
240 
241 		_decode : function(s) {
242 			return tinymce.DOM.decode(s).replace(/\\n/g, '\n');
243 		}
244 	});
245 }(tinymce));
