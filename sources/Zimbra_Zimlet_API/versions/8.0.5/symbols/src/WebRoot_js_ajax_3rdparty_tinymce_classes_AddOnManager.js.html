 25 (function(tinymce) {
 26 	var Dispatcher = tinymce.util.Dispatcher, each = tinymce.each;
 27 
 28 	/**
 29 	 * This class handles the loading of themes/plugins or other add-ons and their language packs.
 30 	 *
 31 	 * @class tinymce.AddOnManager
 32 	 */
 33 	tinymce.create('tinymce.AddOnManager', {
 34 		AddOnManager : function() {
 35 			var self = this;
 36 
 37 			self.items = [];
 38 			self.urls = {};
 39 			self.lookup = {};
 40 			self.onAdd = new Dispatcher(self);
 41 		},
 42 
 43 		/**
 44 		 * Fires when a item is added.
 45 		 *
 46 		 * @event onAdd
 47 		 */
 48 
 49 		/**
 50 		 * Returns the specified add on by the short name.
 51 		 *
 52 		 * @method get
 53 		 * @param {String} n Add-on to look for.
 54 		 * @return {tinymce.Theme/tinymce.Plugin} Theme or plugin add-on instance or undefined.
 55 		 */
 56 		get : function(n) {
 57 			if (this.lookup[n]) {
 58 				return this.lookup[n].instance;
 59 			} else {
 60 				return undefined;
 61 			}
 62 		},
 63 
 64 		dependencies : function(n) {
 65 			var result;
 66 			if (this.lookup[n]) {
 67 				result = this.lookup[n].dependencies;
 68 			}
 69 			return result || [];
 70 		},
 71 
 72 		/**
 73 		 * Loads a language pack for the specified add-on.
 74 		 *
 75 		 * @method requireLangPack
 76 		 * @param {String} n Short name of the add-on.
 77 		 */
 78 		requireLangPack : function(n) {
 79 			var s = tinymce.settings;
 80 
 81 			if (s && s.language && s.language_load !== false)
 82 				tinymce.ScriptLoader.add(this.urls[n] + '/langs/' + s.language + '.js');
 83 		},
 84 
 85 		/**
 86 		 * Adds a instance of the add-on by it's short name.
 87 		 *
 88 		 * @method add
 89 		 * @param {String} id Short name/id for the add-on.
 90 		 * @param {tinymce.Theme/tinymce.Plugin} o Theme or plugin to add.
 91 		 * @return {tinymce.Theme/tinymce.Plugin} The same theme or plugin instance that got passed in.
 92 		 * @example
 93 		 * // Create a simple plugin
 94 		 * tinymce.create('tinymce.plugins.TestPlugin', {
 95 		 *     TestPlugin : function(ed, url) {
 96 		 *         ed.onClick.add(function(ed, e) {
 97 		 *             ed.windowManager.alert('Hello World!');
 98 		 *         });
 99 		 *     }
100 		 * });
101 		 * 
102 		 * // Register plugin using the add method
103 		 * tinymce.PluginManager.add('test', tinymce.plugins.TestPlugin);
104 		 * 
105 		 * // Initialize TinyMCE
106 		 * tinyMCE.init({
107 		 *    ...
108 		 *    plugins : '-test' // Init the plugin but don't try to load it
109 		 * });
110 		 */
111 		add : function(id, o, dependencies) {
112 			this.items.push(o);
113 			this.lookup[id] = {instance:o, dependencies:dependencies};
114 			this.onAdd.dispatch(this, id, o);
115 
116 			return o;
117 		},
118 		createUrl: function(baseUrl, dep) {
119 			if (typeof dep === "object") {
120 				return dep
121 			} else {
122 				return {prefix: baseUrl.prefix, resource: dep, suffix: baseUrl.suffix};
123 			}
124 		},
125 
126 		/**
127 		 * Add a set of components that will make up the add-on. Using the url of the add-on name as the base url.
128 		 * This should be used in development mode.  A new compressor/javascript munger process will ensure that the 
129 		 * components are put together into the editor_plugin.js file and compressed correctly.
130 		 * @param pluginName {String} name of the plugin to load scripts from (will be used to get the base url for the plugins).
131 		 * @param scripts {Array} Array containing the names of the scripts to load.
132 		 */
133 		addComponents: function(pluginName, scripts) {
134 			var pluginUrl = this.urls[pluginName];
135 			tinymce.each(scripts, function(script){
136 				tinymce.ScriptLoader.add(pluginUrl+"/"+script);	
137 			});
138 		},
139 
140 		/**
141 		 * Loads an add-on from a specific url.
142 		 *
143 		 * @method load
144 		 * @param {String} n Short name of the add-on that gets loaded.
145 		 * @param {String} u URL to the add-on that will get loaded.
146 		 * @param {function} cb Optional callback to execute ones the add-on is loaded.
147 		 * @param {Object} s Optional scope to execute the callback in.
148 		 * @example
149 		 * // Loads a plugin from an external URL
150 		 * tinymce.PluginManager.load('myplugin', '/some/dir/someplugin/editor_plugin.js');
151 		 *
152 		 * // Initialize TinyMCE
153 		 * tinyMCE.init({
154 		 *    ...
155 		 *    plugins : '-myplugin' // Don't try to load it again
156 		 * });
157 		 */
158 		load : function(n, u, cb, s) {
159 			var t = this, url = u;
160 
161 			function loadDependencies() {
162 				var dependencies = t.dependencies(n);
163 				tinymce.each(dependencies, function(dep) {
164 					var newUrl = t.createUrl(u, dep);
165 					t.load(newUrl.resource, newUrl, undefined, undefined);
166 				});
167 				if (cb) {
168 					if (s) {
169 						cb.call(s);
170 					} else {
171 						cb.call(tinymce.ScriptLoader);
172 					}
173 				}
174 			}
175 
176 			if (t.urls[n])
177 				return;
178 			if (typeof u === "object")
179 				url = u.prefix + u.resource + u.suffix;
180 
181 			if (url.indexOf('/') !== 0 && url.indexOf('://') == -1)
182 				url = tinymce.baseURL + '/' + url;
183 
184 			t.urls[n] = url.substring(0, url.lastIndexOf('/'));
185 
186 			if (t.lookup[n]) {
187 				loadDependencies();
188 			} else {
189 				tinymce.ScriptLoader.add(url, loadDependencies, s);
190 			}
191 		}
192 	});
193 
194 	// Create plugin and theme managers
195 	tinymce.PluginManager = new tinymce.AddOnManager();
196 	tinymce.ThemeManager = new tinymce.AddOnManager();
197 }(tinymce));
