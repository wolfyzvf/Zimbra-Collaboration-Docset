 22 /**
 23  * Creates a new search with the given properties.
 24  * @class
 25  * This class represents a search to be performed on the server. It has properties for
 26  * the different search parameters that may be used. It can be used for a regular search,
 27  * or to search within a conversation. The results are returned via a callback.
 28  *
 29  * @param {Hash}		params		a hash of parameters
 30  * @param   {String}	params.query					the query string
 31  * @param	{String}	params.queryHint				the query string that gets appended to the query but not something the user needs to know about
 32  * @param	{AjxVector}	params.types					the item types to search for
 33  * @param	{constant}	params.sortBy					the sort order
 34  * @param	{int}		params.offset					the starting point within result set
 35  * @param	{int}		params.limit					the number of results to return
 36  * @param	{Boolean}	params.getHtml					if <code>true</code>, return HTML part for inlined msg
 37  * @param	{constant}	params.contactSource			where to search for contacts (GAL or personal)
 38  * @param	{Boolean}	params.isGalAutocompleteSearch	if <code>true</code>, autocomplete against GAL
 39  * @param	{constant}	params.galType					the type of GAL autocomplete (account or resource)
 40  * @param	{int}		params.lastId					the ID of last item displayed (for pagination)
 41  * @param	{String}	params.lastSortVal				the value of sort field for above item
 42  * @param	{Boolean}	params.fetch					if <code>true</code>, fetch first hit message
 43  * @param	{int}		params.searchId					the ID of owning search folder (if any)
 44  * @param	{Array}		params.conds					the list of search conditions (<code><SearchCalendarResourcesRequest></code>)
 45  * @param	{Array}		params.attrs					the list of attributes to return (<code><SearchCalendarResourcesRequest></code>)
 46  * @param	{String}	params.field					the field to search within (instead of default)
 47  * @param	{Object}	params.soapInfo					the object with method, namespace, response, and additional attribute fields for creating soap doc
 48  * @param	{Object}	params.response					the canned JSON response (no request will be made)
 49  * @param	{Array}		params.folders					the list of folders for autocomplete
 50  * @param	{Array}		params.allowableTaskStatus		the list of task status types to return (assuming one of the values for "types" is "task")
 51  * @param	{String}	params.accountName				the account name to run this search against
 52  * @param	{Boolean}	params.idsOnly					if <code>true</code>, response returns item IDs only
 53  * @param   {Boolean}   params.inDumpster               if <code>true</code>, search in the dumpster
 54  * @param	{string}	params.origin					indicates what initiated the search
 55  * @param	{boolean}	params.isEmpty					if true, return empty response without sending a request
 56  */
 57 ZmSearch = function(params) {
 58 
 59 	params = params || {};
 60 	for (var p in params) {
 61 		this[p] = params[p];
 62 	}
 63 	this.galType					= this.galType || ZmSearch.GAL_ACCOUNT;
 64 	this.join						= this.join || ZmSearch.JOIN_AND;
 65 
 66 	if (this.query || this.queryHint) {
 67 		// only parse regular searches
 68 		if (!this.isGalSearch && !this.isAutocompleteSearch &&
 69 			!this.isGalAutocompleteSearch && !this.isCalResSearch) {
 70 			
 71 			var pq = this.parsedQuery = new ZmParsedQuery(this.query || this.queryHint);
 72 			this._setProperties();
 73 			var sortTerm = pq.getTerm("sort");
 74 			if (sortTerm) {
 75 				this.sortBy = sortTerm.arg;
 76 			}
 77 		}
 78 	}
 79 
 80 	this.isGalSearch = false;
 81 	this.isCalResSearch = false;
 82 
 83 	if (ZmSearch._mailEnabled == null) {
 84 		ZmSearch._mailEnabled = appCtxt.get(ZmSetting.MAIL_ENABLED);
 85 		if (ZmSearch._mailEnabled) {
 86 			AjxDispatcher.require("MailCore");
 87 		}
 88 	}
 89 
 90 	if (params.checkTypes) {
 91 		var types = AjxUtil.toArray(this.types);
 92 		var enabledTypes = [];
 93 		for (var i = 0; i < types.length; i++) {
 94 			var type = types[i];
 95 			var app = ZmItem.APP[type];
 96 			if (appCtxt.get(ZmApp.SETTING[app])) {
 97 				enabledTypes.push(type);
 98 			}
 99 		}
100 		this.types = AjxVector.fromArray(enabledTypes);
101 	}
102 };

164 /**
165  * Creates a SOAP request that represents this search and sends it to the server.
166  *
167  * @param {Hash}	params		a hash of parameters
168  * @param {AjxCallback}	params.callback		the callback to run when response is received
169  * @param {AjxCallback}	params.errorCallback	the callback to run if there is an exception
170  * @param {ZmBatchCommand}	params.batchCmd		the batch command that contains this request
171  * @param {int}	params.timeout		the timeout value (in seconds)
172  * @param {Boolean}	params.noBusyOverlay	if <code>true</code>, don't use the busy overlay
173  * 
174  * @private
175  */
176 ZmSearch.prototype._executeSoap =
177 function(params) {
178 
179 	this.isGalSearch = (this.contactSource && (this.contactSource == ZmId.SEARCH_GAL));
180 	this.isCalResSearch = (!this.contactSource && this.conds != null);
181     if (appCtxt.isOffline && this.isCalResSearch) {
182         this.isCalResSearch =  appCtxt.isZDOnline();
183     }
184 	if (this.isEmpty) {
185 		this._handleResponseExecute(params.callback);
186 		return null;
187 	}
188 
189 	var soapDoc;
190 	if (!this.response) {
191 		if (this.isGalSearch) {
192 			// DEPRECATED: Use JSON version
193 			soapDoc = AjxSoapDoc.create("SearchGalRequest", "urn:zimbraAccount");
194 			var method = soapDoc.getMethod();
195 			if (this.galType) {
196 				method.setAttribute("type", this.galType);
197 			}
198 			soapDoc.set("name", this.query);
199 			var searchFilterEl = soapDoc.set("searchFilter");
200 			if (this.conds && this.conds.length) {
201 				var condsEl = soapDoc.set("conds", null, searchFilterEl);
202 				this._applySoapCond(this.conds, soapDoc, condsEl);
203 			}
204 		} else if (this.isAutocompleteSearch) {
205 			soapDoc = AjxSoapDoc.create("AutoCompleteRequest", "urn:zimbraMail");
206 			var method = soapDoc.getMethod();
207 			if (this.limit) {
208 				method.setAttribute("limit", this.limit);
209 			}
210 			soapDoc.set("name", this.query);
211 		} else if (this.isGalAutocompleteSearch) {
212 			soapDoc = AjxSoapDoc.create("AutoCompleteGalRequest", "urn:zimbraAccount");
213 			var method = soapDoc.getMethod();
214 			method.setAttribute("limit", this._getLimit());
215 			if (this.galType) {
216 				method.setAttribute("type", this.galType);
217 			}
218 			soapDoc.set("name", this.query);
219 		} else if (this.isCalResSearch) {
220 			soapDoc = AjxSoapDoc.create("SearchCalendarResourcesRequest", "urn:zimbraAccount");
221 			var method = soapDoc.getMethod();
222 			if (this.attrs) {
223 				var attrs = [].concat(this.attrs);
224 				AjxUtil.arrayRemove(attrs, "fullName");
225 				method.setAttribute("attrs", attrs.join(","));
226 			}
227 			var searchFilterEl = soapDoc.set("searchFilter");
228 			if (this.conds && this.conds.length) {
229 				var condsEl = soapDoc.set("conds", null, searchFilterEl);
230 				this._applySoapCond(this.conds, soapDoc, condsEl);
231 			}
232 		} else {
233 			if (this.soapInfo) {
234 				soapDoc = AjxSoapDoc.create(this.soapInfo.method, this.soapInfo.namespace);
235 				// Pass along any extra soap data. (Voice searches use this to pass user identification.)
236 				for (var nodeName in this.soapInfo.additional) {
237 					var node = soapDoc.set(nodeName);
238 					var attrs = this.soapInfo.additional[nodeName];
239 					for (var attr in attrs) {
240 						node.setAttribute(attr, attrs[attr]);
241 					}
242 				}
243 			} else {
244 				soapDoc = AjxSoapDoc.create("SearchRequest", "urn:zimbraMail");
245 			}
246 			var method = this._getStandardMethod(soapDoc);
247 			if (this.types) {
248 				var a = this.types.getArray();
249 				if (a.length) {
250 					var typeStr = [];
251 					for (var i = 0; i < a.length; i++) {
252 						typeStr.push(ZmSearch.TYPE[a[i]]);
253 					}
254 					method.setAttribute("types", typeStr.join(","));
255 					if (this.types.contains(ZmItem.MSG) || this.types.contains(ZmItem.CONV)) {
256 						// special handling for showing participants ("To" instead of "From")
257 						var folder = this.folderId && appCtxt.getById(this.folderId);
258 						method.setAttribute("recip", (folder && folder.isOutbound()) ? "1" : "0");
259 					}
260 					// if we're prefetching the first hit message, also mark it as read
261 					if (this.fetch) {
262 
263 						method.setAttribute("fetch", ( this.fetch == "all" ) ? "all" : "1");
264 						// and set the html flag if we want the html version
265 						if (this.getHtml) {
266 							method.setAttribute("html", "1");
267 						}
268 					}
269 					if (this.markRead) {
270 						method.setAttribute("read", "1");
271 					}
272 				}
273 			}
274 			if (this.inDumpster) {
275 				method.setAttribute("inDumpster", "1");
276 			}
277 		}
278 	}
279 
280 	var soapMethod = this._getStandardMethod(soapDoc);
281 	soapMethod.setAttribute("needExp", 1);
282 
283 	var respCallback = this._handleResponseExecute.bind(this, params.callback);
284 
285 	if (params.batchCmd) {
286 		params.batchCmd.addRequestParams(soapDoc, respCallback);
287 	} else {
288 		return appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:true, callback:respCallback,
289 													   errorCallback:params.errorCallback,
290 													   timeout:params.timeout, noBusyOverlay:params.noBusyOverlay,
291 													   response:this.response});
292 	}
293 };

295 /**
296  * Creates a JSON request that represents this search and sends it to the server.
297  *
298  * @param {Hash}	params		a hash of parameters
299  * @param {AjxCallback}	params.callback		the callback to run when response is received
300  * @param {AjxCallback}	params.errorCallback	the callback to run if there is an exception
301  * @param {ZmBatchCommand}	params.batchCmd		the batch command that contains this request
302  * @param {int}	params.timeout		the timeout value (in seconds)
303  * @param {Boolean}	params.noBusyOverlay	if <code>true</code>, don't use the busy overlay
304  * 
305  * @private
306  */
307 ZmSearch.prototype._executeJson =
308 function(params) {
309 
310 	this.isGalSearch = (this.contactSource && (this.contactSource == ZmId.SEARCH_GAL));
311 	this.isCalResSearch = (!this.contactSource && this.conds != null);
312     if (appCtxt.isOffline && this.isCalResSearch) {
313         this.isCalResSearch = appCtxt.isZDOnline();
314     }
315 	if (this.isEmpty) {
316 		this._handleResponseExecute(params.callback);
317 		return null;
318 	}
319 
320 	var jsonObj, request, soapDoc;
321 	if (!this.response) {
322 		if (this.isGalSearch) {
323 			request = {
324 				_jsns:"urn:zimbraAccount",
325 				needIsOwner: "1",
326 				needIsMember: "directOnly"
327 			};
328 			jsonObj = {SearchGalRequest: request};
329 			if (this.galType) {
330 				request.type = this.galType;
331 			}
332 			request.name = this.query;
333 
334 			// bug #36188 - add offset/limit for paging support
335 			request.offset = this.offset = (this.offset || 0);
336 			request.limit = this._getLimit();
337 
338 			// bug 15878: see same in ZmSearch.prototype._getStandardMethodJson
339 			request.locale = { _content: AjxEnv.DEFAULT_LOCALE };
340 
341 			if (this.lastId) { // add lastSortVal and lastId for cursor-based paging
342 				request.cursor = {id:this.lastId, sortVal:(this.lastSortVal || "")};
343 			}
344 			if (this.sortBy) {
345 				request.sortBy = this.sortBy;
346 			}
347 			if (this.conds && this.conds.length) {
348 				request.searchFilter = {conds:{}};
349 				request.searchFilter.conds = ZmSearch.prototype._applyJsonCond(this.conds, request);
350 			}
351 		} else if (this.isAutocompleteSearch) {
352 			jsonObj = {AutoCompleteRequest:{_jsns:"urn:zimbraMail"}};
353 			request = jsonObj.AutoCompleteRequest;
354 			if (this.limit) {
355 				request.limit = this.limit;
356 			}
357 			request.name = {_content:this.query};
358 		} else if (this.isGalAutocompleteSearch) {
359 			jsonObj = {AutoCompleteGalRequest:{_jsns:"urn:zimbraAccount"}};
360 			request = jsonObj.AutoCompleteGalRequest;
361 			request.limit = this._getLimit();
362 			request.name = this.query;
363 			if (this.galType) {
364 				request.type = this.galType;
365 			}
366 		} else if (this.isCalResSearch) {
367 			jsonObj = {SearchCalendarResourcesRequest:{_jsns:"urn:zimbraAccount"}};
368 			request = jsonObj.SearchCalendarResourcesRequest;
369 			if (this.attrs) {
370 				var attrs = [].concat(this.attrs);
371 				request.attrs = attrs.join(",");
372 			}
373             request.offset = this.offset = (this.offset || 0);
374             request.limit = this._getLimit();
375 			if (this.conds && this.conds.length) {
376 				request.searchFilter = {conds:{}};
377 				request.searchFilter.conds = ZmSearch.prototype._applyJsonCond(this.conds, request);
378 			}
379 		} else {
380 			if (this.soapInfo) {
381 				soapDoc = AjxSoapDoc.create(this.soapInfo.method, this.soapInfo.namespace);
382 				// Pass along any extra soap data. (Voice searches use this to pass user identification.)
383 				for (var nodeName in this.soapInfo.additional) {
384 					var node = soapDoc.set(nodeName);
385 					var attrs = this.soapInfo.additional[nodeName];
386 					for (var attr in attrs) {
387 						node.setAttribute(attr, attrs[attr]);
388 					}
389 				}
390 			} else {
391 				jsonObj = {SearchRequest:{_jsns:"urn:zimbraMail"}};
392 				request = jsonObj.SearchRequest;
393 			}
394 			this._getStandardMethodJson(request);
395 			if (this.types) {
396 				var a = this.types.getArray();
397 				if (a.length) {
398 					var typeStr = [];
399 					for (var i = 0; i < a.length; i++) {
400 						typeStr.push(ZmSearch.TYPE[a[i]]);
401 					}
402 					request.types = typeStr.join(",");
403 
404 					if (this.types.contains(ZmItem.MSG) || this.types.contains(ZmItem.CONV)) {
405 						// special handling for showing participants ("To" instead of "From")
406 						var folder = this.folderId && appCtxt.getById(this.folderId);
407 						request.recip = (folder && folder.isOutbound()) ? "1" : "0";
408 					}
409 
410 					// if we're prefetching the first hit message, also mark it as read
411 					if (this.fetch) {
412                         request.fetch = ( this.fetch == "all" ) ? "all" : 1;
413 						// and set the html flag if we want the html version
414 						if (this.getHtml) {
415 							request.html = 1;
416 						}
417 					}
418 
419 					if (this.markRead) {
420 						request.read = 1;
421 					}
422 
423                     if (this.headers) {
424                         for (var hdr in this.headers) {
425                             if (!request.header) { request.header = []; }
426                             request.header.push({n: this.headers[hdr]});
427                         }
428                     }
429 
430 					if (a.length == 1 && a[0] == ZmItem.TASK && this.allowableTaskStatus) {
431 						request.allowableTaskStatus = this.allowableTaskStatus;
432 					}
433                 }
434             }
435 			if (this.inDumpster) {
436 				request.inDumpster = 1;
437 			}
438         }
439     }
440 
441 	if (request) {
442 		request.needExp = 1;
443 	}
444 
445 
446 	var respCallback = this._handleResponseExecute.bind(this, params.callback);
447 
448 	if (params.batchCmd) {
449 		params.batchCmd.addRequestParams(soapDoc, respCallback);
450 	} else {
451 		var searchParams = {
452 			jsonObj:jsonObj,
453 			soapDoc:soapDoc,
454 			asyncMode:true,
455 			callback:respCallback,
456 			errorCallback:params.errorCallback,
457 			timeout:params.timeout,
458 			noBusyOverlay:params.noBusyOverlay,
459 			response:this.response,
460 			accountName:this.accountName
461 		};
462 		return appCtxt.getAppController().sendRequest(searchParams);
463 	}
464 };

466 ZmSearch.prototype._applySoapCond =
467 function(inConds, soapDoc, condsEl, or) {
468 	if (or || this.join == ZmSearch.JOIN_OR) {
469 		condsEl.setAttribute("or", 1);
470 	}
471 	for (var i = 0; i < inConds.length; i++) {
472 		var c = inConds[i];
473 		if (AjxUtil.isArray(c)) {
474 			var subCondsEl = soapDoc.set("conds", null, condsEl);
475 			this._applySoapCond(c, soapDoc, subCondsEl, true);
476 		} else if (c.attr=="fullName" && c.op=="has") {
477 			var nameEl = soapDoc.set("name", c.value);
478 		} else {
479 			var condEl = soapDoc.set("cond", null, condsEl);
480 			condEl.setAttribute("attr", c.attr);
481 			condEl.setAttribute("op", c.op);
482 			condEl.setAttribute("value", c.value);
483 		}
484 	}
485 };

487 ZmSearch.prototype._applyJsonCond =
488 function(inConds, request, or) {
489 	var outConds = {};
490 	if (or || this.join == ZmSearch.JOIN_OR) {
491 		outConds.or = 1;
492 	}
493 
494 	for (var i = 0; i < inConds.length; i++) {
495 		var c = inConds[i];
496 		if (AjxUtil.isArray(c)) {
497 			if (!outConds.conds)
498 				outConds.conds = [];
499 			outConds.conds.push(this._applyJsonCond(c, request, true));
500 		} else if (c.attr=="fullName" && c.op=="has") {
501 			request.name = {_content: c.value};
502 		} else {
503 			if (!outConds.cond)
504 				outConds.cond = [];
505 			outConds.cond.push({attr:c.attr, op:c.op, value:c.value});
506 		}
507 	}
508 	return outConds;
509 };

511 /**
512  * Converts the response into a {ZmSearchResult} and passes it along.
513  * 
514  * @private
515  */
516 ZmSearch.prototype._handleResponseExecute =
517 function(callback, result) {
518 	
519 	if (result) {
520 		var response = result.getResponse();
521 	
522 		if      (this.isGalSearch)				{ response = response.SearchGalResponse; }
523 		else if (this.isCalResSearch)			{ response = response.SearchCalendarResourcesResponse; }
524 		else if (this.isAutocompleteSearch)		{ response = response.AutoCompleteResponse; }
525 		else if (this.isGalAutocompleteSearch)	{ response = response.AutoCompleteGalResponse; }
526 		else if (this.soapInfo)					{ response = response[this.soapInfo.response]; }
527 		else									{ response = response.SearchResponse; }
528 	}
529 	else {
530 		response = { _jsns: "urn:zimbraMail", more: false };
531 	}
532 	var searchResult = new ZmSearchResult(this);
533 	searchResult.set(response);
534 	result = result || new ZmCsfeResult();
535 	result.set(searchResult);
536 
537 	if (callback) {
538 		callback.run(result);
539 	}
540 };

542 /**
543  * Fetches a conversation from the server.
544  *
545  * @param {Hash}		params				a hash of parameters:
546  * @param {String}		params.cid			the conv ID
547  * @param {AjxCallback}	params.callback		the callback to run with result

1071 ZmParsedQuery.prototype._parse =
1072 function(query) {
1073 
1074 	function getQuotedStr(str, pos, q) {
1075 		var q = q || str.charAt(pos);
1076 		pos++;
1077 		var done = false, ch, quoted = "";
1078 		while (pos < str.length && !done) {
1079 			ch = str.charAt(pos);
1080 			if (ch == q) {
1081 				done = true;
1082 			} else {
1083 				quoted += ch;
1084 				pos++;
1085 			}
1086 		}
1087 
1088 		return done ? {str:quoted, pos:pos + 1} : null;
1089 	}
1090 	
1091 	function skipSpace(str, pos) {
1092 		while (pos < str.length && str.charAt(pos) == " ") {
1093 			pos++;
1094 		}
1095 		return pos;
1096 	}
1097 	
1098 	function fail(reason, query) {
1099 		DBG.println(AjxDebug.DBG1, "ZmParsedQuery failure: " + reason + "; query: [" + query + "]");
1100 		this.parseFailed = reason;
1101 		return tokens;		
1102 	}
1103 
1104 	var len = query.length;
1105 	var tokens = [], ch, lastCh, op, word = "", isEow = false, endOk = true, compound = 0, numParens = 0;
1106 	var pos = skipSpace(query, 0);
1107 	while (pos < len) {
1108 		lastCh = (ch != " ") ? ch : lastCh;
1109 		ch = query.charAt(pos);
1110 		isEow = ZmParsedQuery.IS_EOW[ch];
1111 
1112 		if (ch == ":") {
1113 			if (ZmParsedQuery.IS_OP[word]) {
1114 				op = word;
1115 			} else {
1116 				return fail("unrecognized op '" + word + "'", query);
1117 			}
1118 			word = "";
1119 			pos = skipSpace(query, pos + 1);
1120 			continue;
1121 		}
1122 
1123 		if (isEow) {
1124 			var lcWord = word.toLowerCase();
1125 			var isCondOp = !!ZmParsedQuery.COND_OP[lcWord];
1126 			if (op && word && !(isCondOp && compound > 0)) {
1127 				tokens.push(new ZmSearchToken(op, lcWord));
1128 				if (compound == 0) {
1129 					op = "";
1130 				}
1131 				word = "";
1132 				endOk = true;
1133 			} else if (!op || (op && compound > 0)) {
1134 				if (isCondOp) {
1135 					tokens.push(new ZmSearchToken(lcWord));
1136 					endOk = false;
1137 					if (lcWord == ZmParsedQuery.COND_OR) {
1138 						this.hasOrTerm = true;
1139 					}
1140 				} else if (word) {
1141 					tokens.push(new ZmSearchToken(ZmParsedQuery.OP_CONTENT, word));
1142 				}
1143 				word = "";
1144 			}
1145 		}
1146 
1147 		if (ch == '"') {
1148 			var results = getQuotedStr(query, pos);
1149 			if (results) {
1150 				word = results.str;
1151 				pos = results.pos;
1152 			} else {
1153 				return fail("improper use of quotes", query);
1154 			}
1155 		} else if (ch == ZmParsedQuery.GROUP_OPEN) {
1156 			var done = false;
1157 			if (compound > 0) {
1158 				compound++;
1159 			}
1160 			else if (lastCh == ":") {
1161 				compound = 1;
1162 				// see if parens are being used as secondary quoting mechanism by looking for and/or
1163 				var inside = query.substr(pos, query.indexOf(ZmParsedQuery.GROUP_CLOSE, pos + 1));
1164 				inside = inside && inside.toLowerCase();
1165 				if (inside && (inside.indexOf(" " + ZmParsedQuery.COND_OR + " ") == -1) &&
1166 							  (inside.indexOf(" " + ZmParsedQuery.COND_AND + " ") == -1)) {
1167 					var results = getQuotedStr(query, pos, ZmParsedQuery.GROUP_CLOSE);
1168 					if (results) {
1169 						word = results.str;
1170 						pos = results.pos;
1171 						compound = 0;
1172 					} else {
1173 						return fail("improper use of paren-based quoting", query);
1174 					}
1175 					done = true;
1176 				}
1177 			}
1178 			if (!done) {
1179 				tokens.push(new ZmSearchToken(ch));
1180 				numParens++;
1181 			}
1182 			pos = skipSpace(query, pos + 1);
1183 		} else if (ch == ZmParsedQuery.GROUP_CLOSE) {
1184 			if (compound > 0) {
1185 				compound--;
1186 			}
1187 			if (compound == 0) {
1188 				op = "";
1189 			}
1190 			tokens.push(new ZmSearchToken(ch));
1191 			pos = skipSpace(query, pos + 1);
1192 		} else if (ch == "-" && !word && !op) {
1193 			tokens.push(new ZmSearchToken(ZmParsedQuery.COND_NOT));
1194 			pos = skipSpace(query, pos + 1);
1195 			endOk = false;
1196 		} else {
1197 			if (ch != " ") {
1198 				word += ch;
1199 			}
1200 			pos++;
1201 		}
1202 	}
1203 
1204 	// check for term at end
1205 	if ((pos >= query.length) && op && word) {
1206 		tokens.push(new ZmSearchToken(op, word));
1207 		endOk = true;
1208 	} else if (!op && word) {
1209 		tokens.push(new ZmSearchToken(word));
1210 	}
1211 	
1212 	// remove unnecessary enclosing parens from when a single compound term is expanded, for example when
1213 	// "subject:(foo bar)" is expanded into "(subject:foo subject:bar)"
1214 	if (tokens.length >= 3 && numParens == 1 && tokens[0].op == ZmParsedQuery.GROUP_OPEN &&
1215 			tokens[tokens.length - 1].op == ZmParsedQuery.GROUP_CLOSE) {
1216 		tokens.shift();
1217 		tokens.pop();
1218 	}
1219 
1220 	if (!endOk) {
1221 		return fail("unexpected end of query", query);
1222 	}
1223 	
1224 	return tokens;
1225 };

1450  * and conditional operator (AND, OR, NOT), or a grouping operator (left or right paren).

1454  */

1456

1470 		this.op = op.toLowerCase();

1473 		this.type = ZmParsedQuery.GROUP;
1474 	} else if (op) {
