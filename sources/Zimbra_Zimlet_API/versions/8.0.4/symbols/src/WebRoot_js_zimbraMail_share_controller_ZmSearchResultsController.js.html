 23 /**
 24  * @class
 25  * This controller is used to display the results of a user-initiated search. The results are
 26  * displayed in a tab which has three parts: a search bar at the top that can be used to modify
 27  * the search, a filtering mechanism on the left that can be used to refine the search, and the
 28  * results themselves. The results may be of any type: messages, contacts, etc.
 29  * 
 30  * @param	{DwtShell}		container		the application container
 31  * @param	{ZmApp}			app				the application
 32  * @param	{constant}		type			type of controller
 33  * @param	{string}		sessionId		the session id
 34  * 
 35  * @extends	ZmController
 36  */
 37 ZmSearchResultsController = function(container, app, type, sessionId) {
 38 
 39 	ZmController.apply(this, arguments);
 40 	
 41 	this._initialize();
 42 };
 43 
 44 ZmSearchResultsController.prototype = new ZmController;
 45 ZmSearchResultsController.prototype.constructor = ZmSearchController;
 46 
 47 ZmSearchResultsController.prototype.isZmSearchResultsController = true;
 48 ZmSearchResultsController.prototype.toString = function() { return "ZmSearchResultsController"; };
 49 
 50 ZmSearchResultsController.DEFAULT_TAB_TEXT = ZmMsg.search;
 51 
 52 ZmSearchResultsController.getDefaultViewType =
 53 function(params) {
 54 	return (params && params.appName) ? [ZmId.VIEW_SEARCH_RESULTS, params.appName].join("-") : ZmId.VIEW_SEARCH_RESULTS;
 55 };
 56 ZmSearchResultsController.prototype.getDefaultViewType = ZmSearchResultsController.getDefaultViewType;
 57 
 58 /**
 59  * Displays the given results in a search tab managed by this controller.
 60  * 
 61  * @param {ZmSearchResults}		results		search results
 62  */
 63 ZmSearchResultsController.prototype.show =
 64 function(results, resultsCtlr) {
 65 	var resultsType = results.type;
 66 	results.search.sessionId = this.sessionId;	// in case we reuse this search (eg view switch)
 67 	if (!resultsCtlr) {
 68 		var app = this._resultsApp = appCtxt.getApp(ZmItem.APP[resultsType]) || appCtxt.getCurrentApp();
 69 		app.showSearchResults(results, this._displayResults.bind(this, results.search), this);
 70 	}
 71 	else {
 72 		this._displayResults(results.search, resultsCtlr);
 73 	}
 74 	this._curSearch = results.search;
 75 	this.inactive = true;	// search tabs can always be reused (unless pinned)
 76 };
 77 
 78 // creates the toolbar and filter panel
 79 ZmSearchResultsController.prototype._initialize =
 80 function() {
 81 
 82 	this._toolbar = new ZmSearchResultsToolBar({
 83 				parent:			this._container,
 84 				controller:		this,
 85 				id:				DwtId.makeId(ZmId.SEARCHRESULTS_TOOLBAR, this._currentViewId),
 86 				noMenuButton:	true
 87 			});
 88 	this._toolbar.getButton(ZmSearchToolBar.SEARCH_BUTTON).addSelectionListener(this._searchListener.bind(this));
 89 	var saveButton = this._toolbar.getButton(ZmSearchToolBar.SAVE_BUTTON);
 90 	if (saveButton) {
 91 		saveButton.addSelectionListener(this._saveListener.bind(this));
 92 	}
 93 	this._toolbar.registerEnterCallback(this._searchListener.bind(this));
 94 
 95 	this.isPinned = false;
 96 };
 97 
 98 /**
 99  * Shows the results of the given search in this controller's search tab. The toolbar and filter panel
100  * were created earlier, and the content area (top toolbar and list view) is taken from the controller
101  * that generated the results.
102  * 
103  * @param {ZmSearch}		search			search object
104  * @param {ZmController}	resultsCtlr		passed back from app
105  * @private
106  */
107 ZmSearchResultsController.prototype._displayResults =
108 function(search, resultsCtlr) {
109 	
110 	if (!this._filterPanel) {
111 		this._filterPanel = new ZmSearchResultsFilterPanel({
112 					parent:		this._container,
113 					controller:	this,
114 					id:			DwtId.makeId(ZmId.SEARCHRESULTS_PANEL, this._currentViewId),
115 					resultsApp:	resultsCtlr.getApp().getName()
116 				});
117 	}
118 	this._resultsController = resultsCtlr;
119 	if (appCtxt.getCurrentViewId().indexOf(this._currentViewId) !== -1) {
120 		var elements = {};
121 		elements[ZmAppViewMgr.C_TOOLBAR_TOP] = resultsCtlr.getCurrentToolbar();
122 		elements[ZmAppViewMgr.C_APP_CONTENT] = resultsCtlr.getViewMgr ? resultsCtlr.getViewMgr() : resultsCtlr.getCurrentView();
123 		appCtxt.getAppViewMgr().setViewComponents(this._currentViewId, elements, true);
124 	}
125 	else {
126 		
127 		var elements = {};
128 		elements[ZmAppViewMgr.C_SEARCH_RESULTS_TOOLBAR] = this._toolbar;
129 		elements[ZmAppViewMgr.C_TREE] = this._filterPanel;
130 		elements[ZmAppViewMgr.C_TOOLBAR_TOP] = resultsCtlr.getCurrentToolbar();
131 		elements[ZmAppViewMgr.C_APP_CONTENT] = resultsCtlr.getViewMgr ? resultsCtlr.getViewMgr() : resultsCtlr.getCurrentView();
132 		
133 		this._app.createView({	viewId:		this._currentViewId,
134 								viewType:	this._currentViewType,
135 								elements:	elements,
136 								controller:	this,
137 								hide:		[ ZmAppViewMgr.C_TREE_FOOTER ],
138 								tabParams:	this._getTabParams()});
139 		this._app.pushView(this._currentViewId);
140 		this._filterPanel.reset();
141 		// search tab button menu
142 		var button = appCtxt.getAppChooser().getButton(this.tabId);
143 		var menu = new DwtMenu({ parent: button	});
144 		button.setMenu(menu);
145 		var menuItem;
146 		menuItem = new DwtMenuItem({ parent:menu });
147 		menuItem.setText(ZmMsg.saveCurrentSearch);
148 		menuItem.addSelectionListener(this._saveListener.bind(this));
149 		menuItem = new DwtMenuItem({ parent:menu });
150 		menuItem.setText(ZmMsg.close);
151 		menuItem.addSelectionListener(this._closeListener.bind(this));
152 		menuItem = new DwtMenuItem({ parent:menu, style: DwtMenuItem.SEPARATOR_STYLE });
153 		menuItem = new DwtMenuItem({ parent:menu, style:DwtMenuItem.CHECK_STYLE });
154 		menuItem.setText(ZmMsg.pinned);
155 		menuItem.addSelectionListener(this._pinnedListener.bind(this));
156 	}
157 	
158 	if (search && search.origin == ZmId.SEARCH) {
159 		this._toolbar.setSearch(search);
160 	}
161 	// Tell the user how many results were found
162 	var searchResult = resultsCtlr.getCurrentSearchResults && resultsCtlr.getCurrentSearchResults();
163 	var results = (searchResult && searchResult.getResults()) || resultsCtlr.getList();
164 	var size = results && results.size && results.size();
165 	var plus = (results && results.hasMore && results.hasMore()) ? "+" : "";
166 	var label = size ? AjxMessageFormat.format(ZmMsg.searchResultsLabel, [size, plus]) :
167 					   ZmMsg.searchResultsLabelNone;
168 	this._toolbar.setLabel(label, false);
169     if (resultsCtlr && resultsCtlr.updateTimeIndicator) {
170         resultsCtlr.updateTimeIndicator();
171     }
172 	setTimeout(this._toolbar.focus.bind(this._toolbar), 100);
173 };

179 // returns params for the search tab button
180 ZmSearchResultsController.prototype._getTabParams =
181 function() {
182 	return {
183 		id:				this.tabId,
184 		image:          "CloseGray",
185         hoverImage:     "Close",
186 		text:			ZmSearchResultsController.DEFAULT_TAB_TEXT,
187 		textPrecedence:	90,
188 		tooltip:		ZmSearchResultsController.DEFAULT_TAB_TEXT,
189         style:          DwtLabel.IMAGE_RIGHT
190 	};
191 };

193 // runs a search based on the contents of the input
194 ZmSearchResultsController.prototype._searchListener =
195 function(ev, zimletEvent) {
196 
197 	// add bubble if needed before running search, but don't let "bubble added" callback trigger a search
198 	var toolbar = this._toolbar, element = toolbar && toolbar._searchField.getInputElement();
199 	if (element && toolbar._acList) {
200 		toolbar._settingSearch = true;
201 		toolbar._acList._complete(element);
202 		toolbar._settingSearch = false;
203 	}
204 
205 	var view = appCtxt.getCurrentViewId(); //this view should be the results list view. Somehow it seems to be.
206 	var sortBy = view ? appCtxt.get(ZmSetting.SORTING_PREF, view) : null; // repeat the previous sort order (from same search tab only, which is this case)
207 
208 	// run the search
209 	var query = this._toolbar.getSearchFieldValue();
210 	var params = {
211 		ev:							ev,
212 		zimletEvent:				zimletEvent || "onSearchButtonClick",
213 		query:						query,
214 		isEmpty:					!query,
215 		sessionId:					this.sessionId,
216 		skipUpdateSearchToolbar:	true,
217 		origin:						ZmId.SEARCHRESULTS,
218 		searchFor:					this._curSearch && this._curSearch.searchFor,
219 		sortBy:						sortBy,
220 		errorCallback:				this._errorCallback.bind(this)
221 	};
222 	toolbar.setLabel(ZmMsg.searching);
223 	appCtxt.getSearchController()._toolbarSearch(params);
224 };
225 
226 // Note the error and then eat it - we don't want to show toast or clear out results
227 ZmSearchResultsController.prototype._errorCallback =
228 function(ev) {
229 	this._toolbar.setLabel(ZmMsg.invalidSearch, true);
230 	return true;
231 };
232 
233 // pops up a dialog to save the search
234 ZmSearchResultsController.prototype._saveListener =
235 function(ev) {
236 	appCtxt.getSearchController()._saveButtonListener(ev);
237 };
238 
239 // toggles the pinned state of this tab
240 ZmSearchResultsController.prototype._pinnedListener =
241 function(ev) {
242 	this.isPinned = !this.isPinned;
243 };
244 
245 ZmSearchResultsController.prototype._closeListener =
246 function(ev) {
247 	appCtxt.getAppViewMgr().popView(false, this.getCurrentViewId());
248 };
249 
250 // adds the given term to the search as a bubble
251 ZmSearchResultsController.prototype.addSearchTerm =
252 function(term, skipNotify, addingCond) {
253 	return this._toolbar.addSearchTerm(term, skipNotify, addingCond);
254 };
255 
256 // removes the bubble with the given term
257 ZmSearchResultsController.prototype.removeSearchTerm =
258 function(term, skipNotify) {
259 	this._toolbar.removeSearchTerm(term, skipNotify);
260 };
261 
262 // returns a list of current search terms
263 ZmSearchResultsController.prototype.getSearchTerms =
264 function(term, skipNotify) {
265 	var values = this._toolbar._searchField.getAddresses();
266 	var terms = AjxUtil.map(values, function(member, i) {
267 		return new ZmSearchToken(member);
268 	});
269 	return terms;
270 };
