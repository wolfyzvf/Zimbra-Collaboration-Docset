 16 /**
 17  * Singleton alert class that alerts the user by popping up a message on the desktop.
 18  * @class
 19  * @private
 20  */
 21 ZmDesktopAlert = function() {
 22     if (window.webkitNotifications) {
 23         this.useWebkit = true;
 24     } else if (appCtxt.isOffline && window.platform && (AjxEnv.isWindows || AjxEnv.isMac)) {
 25         this.usePrism = true;
 26     } else {
 27         this.useBrowserPlus = true;
 28     }
 29 };
 30 
 31 ZmDesktopAlert.prototype = new ZmAlert;
 32 ZmDesktopAlert.prototype.constructor = ZmDesktopAlert;
 33 
 34 ZmDesktopAlert.prototype.toString =
 35 function() {
 36 	return "ZmDesktopAlert";
 37 };
 38 
 39 ZmDesktopAlert.getInstance =
 40 function() {
 41 	return ZmDesktopAlert.INSTANCE = ZmDesktopAlert.INSTANCE || new ZmDesktopAlert();
 42 };
 43 
 44 /**
 45  * Returns text to show in a prefs page next to the checkbox to enable this type of alert.
 46  */
 47 ZmDesktopAlert.prototype.getDisplayText =
 48 function() {
 49     if (this.useWebkit) {
 50        return ZmMsg.showPopup;
 51     } else if (this.usePrism) {
 52 		return AjxEnv.isMac ? ZmMsg.showPopupMac : ZmMsg.showPopup;
 53 	} else {
 54 		return ZmMsg.showPopupBrowserPlus;
 55 	}
 56 };
 57 
 58 /**
 59  * Returns any link text to show in a prefs page, for example a link to install browser plus if necessary.
 60  */
 61 ZmDesktopAlert.prototype.getLinkText =
 62 function() {
 63 	return this.useBrowserPlus ? ZmMsg.showPopupBrowserPlusLink : "";
 64 };
 65 
 66 ZmDesktopAlert.prototype.start =
 67 function(title, message, sticky) {
 68     if (this.useWebkit) {
 69         var allowedCallback = this._showWebkitNotification.bind(this, title, message, sticky);
 70         this._checkWebkitPermission(allowedCallback);
 71     } else if (this.usePrism) {
 72 		if (AjxEnv.isMac) {
 73 			try {
 74 				window.platform.showNotification(title, message, "resource://webapp/icons/default/launcher.icns");
 75 			} catch (err) {}
 76 		}
 77 		else if (AjxEnv.isWindows) {
 78 			try {
 79 				window.platform.icon().showNotification(title, message, 5);
 80 			} catch (err) {}
 81 		}
 82 	} else {
 83 		AjxDispatcher.require([ "BrowserPlus" ]);
 84 		var serviceObj = { service: "Notify", version: "2", minversion: "2.0.9" };
 85 		var callback = new AjxCallback(this, this._notityServiceCallback, [title, message]);
 86 		var errorCallback = new AjxCallback(this, this._notityServiceErrorCallback);
 87 		ZmBrowserPlus.getInstance().require(serviceObj, callback, errorCallback);
 88 	}
 89 };
 90 
 91 /* Checks if we have permission to use webkit notifications. If so, or when the user
 92  * grants permission, allowedCallback is called.
 93  */
 94 ZmDesktopAlert.prototype._checkWebkitPermission =
 95 function(allowedCallback) {
 96     var allowed = window.webkitNotifications.checkPermission() == 0;
 97     if (allowed) {
 98         allowedCallback();
 99     } else if (!ZmDesktopAlert.requestedPermission) {
100         ZmDesktopAlert.requestedPermission = true; // Prevents multiple permission requests in one session.
101         window.webkitNotifications.requestPermission(this._checkWebkitPermission.bind(this, allowedCallback));
102     }
103 };
104 
105 ZmDesktopAlert.prototype._showWebkitNotification =
106 function(title, message, sticky) {
107 	sticky = sticky || false;
108     // Icon: I chose to use the favIcon because it's already overridable by skins.
109     // It's a little ugly though.
110     // change for bug#67359: Broken notification image in chrome browser
111     // //var icon = window.favIconUrl;
112 	var icon = skin.hints.notificationBanner;
113     var popup = window.webkitNotifications.createNotification(icon, title, message);
114     popup.show();
115 	popup.onclick = function() {popup.cancel();};
116     if (sticky) {
117         if (!ZmDesktopAlert.notificationArray) {
118             ZmDesktopAlert.notificationArray = [];
119         }
120         ZmDesktopAlert.notificationArray.push(popup);
121     }
122     else {
123         // Close the popup after 5 seconds.
124         setTimeout(popup.cancel.bind(popup), 5000);
125     }
126 };
127 
128 ZmDesktopAlert.prototype._notityServiceCallback =
129 function(title, message, service) {
130 	try {
131 		service.show({ title: title, message: message }, function(){});
132 	} catch (err) {}
133 };
134 
135 ZmDesktopAlert.prototype._notityServiceErrorCallback =
136 function(result) {
137 	DBG.println(AjxDebug.DBG1, "BrowserPlus error: " + (result ? (result.error + " - " + result.verboseError) : result));
138 };

141 /**
142  * Closes desktop notification if any during onbeforeunload event
143  */
144 ZmDesktopAlert.closeNotification =
145 function() {
146     var notificationArray = ZmDesktopAlert.notificationArray,
147         popup;
148 
149     if (notificationArray) {
150         while (popup = notificationArray.pop()) {
151             //notifications may be already closed by the user go for try catch
152             try {
153                 popup.cancel();
154             }
155             catch (e) {
156             }
157         }
158     }
159 };
