 16 /**
 17  * Creates an exception.
 18  * @constructor
 19  * @class
 20  * This is the base class for all exceptions in the Ajax toolkit.
 21  * 
 22  * @author Ross Dargahi
 23  * 
 24  * @param {string} [msg] 		the human readable message
 25  * @param {string|number} [code]	 any error or fault code
 26  * @param {string} [method] 	the name of the method throwing the exception
 27  * @param {string} [detail] 	any additional detail
 28  * 
 29  * @extends		AjxException
 30  */
 31 DwtException = function(msg, code, method, detail) {
 32 	if (arguments.length === 0) {return;}
 33 	AjxException.call(this, msg, code, method, detail);
 34 }
 35 
 36 DwtException.prototype = new AjxException();
 37 DwtException.prototype.constructor = DwtException;
 38 
 39 DwtException.prototype.toString = 
 40 function() {
 41 	return "DwtException";
 42 };
 43
