 21 /**
 22  * Creates the task list controller.
 23  * @class
 24  * This class represents the task list controller.
 25  * 
 26  * @param {DwtControl}					container					the containing shell
 27  * @param {ZmApp}						app							the containing application
 28  * @param {constant}					type						type of controller
 29  * @param {string}						sessionId					the session id
 30  * @param {ZmSearchResultsController}	searchResultsController		containing controller
 31  * 
 32  * @extends		ZmListController
 33  */
 34 ZmTaskListController = function(container, app, type, sessionId, searchResultsController) {
 35 
 36 	ZmListController.apply(this, arguments);
 37 
 38 	if (this.supportsDnD()) {
 39 		this._dragSrc = new DwtDragSource(Dwt.DND_DROP_MOVE);
 40 		this._dragSrc.addDragListener(this._dragListener.bind(this));
 41 	}
 42 
 43 	this._listeners[ZmOperation.EDIT]				= this._editListener.bind(this);
 44 	this._listeners[ZmOperation.PRINT_TASK]			= this._printTaskListener.bind(this);
 45 	this._listeners[ZmOperation.PRINT_TASKFOLDER]	= this._printTaskFolderListener.bind(this);
 46 	this._listeners[ZmOperation.SHOW_ORIG]			= this._showOrigListener.bind(this);
 47 	this._listeners[ZmOperation.MARK_AS_COMPLETED]	= this._markAsCompletedListener.bind(this);
 48     this._listeners[ZmOperation.DELETE]				= this._deleteListener.bind(this);
 49 
 50 	this._listeners[ZmOperation.PRINT]				= null; // override base class to do nothing
 51 
 52     var pref = appCtxt.get(ZmSetting.TASKS_FILTERBY);
 53 	this._currentTaskView = ZmTaskListController.FILTERBY_SETTING_ID[pref];
 54 };

142 ZmTaskListController.prototype.show =
143 function(results, folderId) {
144 
145 	this._folderId = folderId;
146 
147 	this.setList(results.getResults(ZmItem.TASK));
148 
149 	// XXX: WHY?
150 	// find out if we just searched for a shared tasks folder
151 	var folder = appCtxt.getById(folderId);
152 	this._list._isShared = folder ? folder.link : false;
153 	this._list.setHasMore(results.getAttribute("more"));
154 
155 	ZmListController.prototype.show.call(this, results);
156 
157     if (this._taskMultiView) {
158 		var tlv = this._taskMultiView._taskListView;
159 		tlv._saveState({selection:true});
160 		tlv.reset();
161 	}
162     //Generate view Id again as loading the read only view changes the viewId of the TaskListView
163     var viewId = [this.getCurrentViewType(), this.getSessionId()].join(ZmController.SESSION_ID_SEP);
164 
165 	this._setup(viewId);
166 
167 	// reset offset if list view has been created
168 	var lv = this.getListView();
169 	if (lv) { lv.offset = 0; }
170 
171 	var elements = this.getViewElements(viewId, this._taskMultiView);
172 	
173 	this._setView({ view:		viewId,
174 					viewType:	this._currentViewType,
175 					noPush:		this.isSearchResults,
176 					elements:	elements,
177 					isAppView:	true});
178 	if (this.isSearchResults) {
179 		// if we are switching views, make sure app view mgr is up to date on search view's components
180 		appCtxt.getAppViewMgr().setViewComponents(this.searchResultsController.getCurrentViewId(), elements, true);
181 	}
182 
183 	this._setTabGroup(this._tabGroups[viewId]);
184 	this._resetNavToolBarButtons(viewId);
185 
186     // do this last
187 	if (!this._taskTreeController) {
188 		this._taskTreeController = appCtxt.getOverviewController().getTreeController(ZmOrganizer.TASKS);
189 		DBG.timePt("getting tree controller", true);
190 	}
191 
192     this.getListView().setTaskInputVisible(folderId != ZmOrganizer.ID_TRASH);
193 };

210 /**
211  * Switches the view.
212  * 
213  * @param	{DwtComposite}		view		the view
214  */
215 ZmTaskListController.prototype.switchView =
216 function(view) {
217 	if (this._currentTaskView == view) { return; }
218 	this._currentTaskView = view;
219     if (view == ZmSetting.RP_OFF ||	view == ZmSetting.RP_BOTTOM || view == ZmSetting.RP_RIGHT) {
220 		this._taskListView._colHeaderActionMenu = null;
221 		if (view != this._getReadingPanePref()) {
222 			this._setReadingPanePref(view);
223 			this._taskMultiView.setReadingPane();
224 		}
225         // always reset the view menu button icon to reflect the current view
226         var btn = this._toolbar[this._currentViewId].getButton(ZmOperation.VIEW_MENU);
227         btn.setImage(ZmTaskListController.READING_PANE_ICON[view]);
228 	} else {
229         if(view != this._getFilterByPref() && !appCtxt.isExternalAccount()) {
230             this._setFilterByPref(ZmTaskListController.FILTERBY_SETTING[view]);
231         }
232 	}
233 	var sc = appCtxt.getSearchController();
234 	var soapStatus = ZmTaskListController.SOAP_STATUS[view];
235     var currentSearch =  appCtxt.getCurrentSearch();
236     if (currentSearch) sc.redoSearch(currentSearch, false, {allowableTaskStatus:soapStatus});
237 };

268 ZmTaskListController.prototype.handleKeyAction =
269 function(actionCode) {
270 	DBG.println(AjxDebug.DBG3, "ZmTaskListController.handleKeyAction");
271 
272     var lv = this._listView[this._currentViewId];
273     var num = lv.getSelectionCount();
274     var isExternalAccount = appCtxt.isExternalAccount();
275 
276     switch(actionCode) {
277 
278         case ZmKeyMap.MARK_COMPLETE:
279         case ZmKeyMap.MARK_UNCOMPLETE:
280             if (isExternalAccount) { break; }
281             var task = this._listView[this._currentViewId].getSelection()[0];
282             if ((task.isComplete() && actionCode == ZmKeyMap.MARK_UNCOMPLETE) ||
283                     (!task.isComplete() && actionCode == ZmKeyMap.MARK_COMPLETE))
284             {
285                 this._doCheckCompleted(task);
286             }
287             break;
288 
289         case ZmKeyMap.READING_PANE_BOTTOM:
290 		case ZmKeyMap.READING_PANE_RIGHT:
291 		case ZmKeyMap.READING_PANE_OFF:
292 			var menuId = ZmTaskListController.ACTION_CODE_TO_MENU_ID[actionCode];
293 			this._updateViewMenu(menuId);
294             this.switchView(menuId);
295 			break;
296         case ZmKeyMap.MOVE_TO_TRASH:
297             if (isExternalAccount) { break; }
298             if(num) {
299                 var tasks = lv.getSelection();
300                 var nId = ZmOrganizer.normalizeId(tasks[0].folderId);
301                 var isTrash = nId == ZmOrganizer.ID_TRASH;
302                 if(!isTrash){
303                     this._handleCancel(tasks);
304                 }
305             }
306             break;
307 
308         case ZmKeyMap.CANCEL:
309             if (lv && lv.isZmTaskView) {
310                 lv.close();
311                 break;
312             }
313         default:
314             return ZmListController.prototype.handleKeyAction.call(this, actionCode);
315     }
316 
317     return true;
318 };

373 /**
374  * Saves the task.
375  * 
376  * @param		{String}	name		the task name
377  * @param		{AjxCallback}	callback	the save callback
378  * 
379  * @see	ZmTask
380  */
381 ZmTaskListController.prototype.quickSave =
382 function(name, callback, errCallback) {
383 	var folderId = (this._activeSearch && this._activeSearch.search) ? this._activeSearch.search.folderId : null;
384 
385 	var folder = appCtxt.getById(folderId);
386 	if (folder && folder.link) {
387 		folderId = folder.getRemoteId();
388 	}
389 
390 	var task = new ZmTask(this._list, null, folderId);
391 
392 	if (folder && folder.link) {
393 		task.setOrganizer(folder.owner);
394 		task._orig = new ZmTask(this._list);
395 	}
396 
397 	task.setName(name);
398 	task.setViewMode(ZmCalItem.MODE_NEW);
399 	task.location = "";
400 	task.setAllDayEvent(true);
401 
402 	task.save(null, callback, errCallback);
403 };

541 ZmTaskListController.prototype._setupViewMenu =
542 function(view) {
543     var btn = this._toolbar[view].getButton(ZmOperation.VIEW_MENU);
544     var menu = btn.getMenu();
545     if (!menu) {
546 		menu = new ZmPopupMenu(btn);
547 		btn.setMenu(menu);
548 
549 
550         var pref = this._getFilterByPref();
551         for (var i = 0; i < ZmTaskListController.SORT_BY.length; i++) {
552 			var id = ZmTaskListController.SORT_BY[i];
553 			var params = {
554 				image:ZmTaskListController.ICON[id],
555 				text:ZmMsg[ZmTaskListController.MSG_KEY[id]],
556 				style:DwtMenuItem.RADIO_STYLE,
557                 radioGroupId:"TAKS_FILTER_BY"
558 			};
559 			var mi = menu.createMenuItem(id, params);
560 			mi.setData(ZmOperation.MENUITEM_ID, id);

697 function(tasks, mode) {

797 	var actionLogItem = (actionController && actionController.actionPerformed({op: "trash", ids: idList, attrs: {l: ZmOrganizer.ID_TRASH}})) || null;

799 	var summary = ZmList.getActionSummary(ZmMsg.actionTrash, tasks.length, ZmItem.TASK);

802 	if (undoLink && actionController) {
803 		actionController.onPopup();
804 		appCtxt.setStatusMsg({msg: summary+undoLink, transitions: actionController.getStatusTransitions()});
805 	} else {
806 		appCtxt.setStatusMsg(summary);
807 	}

1067 /**
1068  * Gets the checked calendar folder ids.
1069  *
1070  * @param	{Boolean}	localOnly		if <code>true</code>, include local calendars only
1071  * @return	{Array}		an array of folder ids
1072  */
1073 ZmTaskListController.prototype.getTaskFolderIds =
1074 function(localOnly) {
1075     var cc = [];
1076     if(localOnly) {
1077         if(this._taskTreeController) {
1078             if (appCtxt.multiAccounts) {
1079                 var overviews = this._app.getOverviewContainer().getOverviews();
1080                 for (var i in overviews) {
1081                     cc = cc.concat(this._taskTreeController.getTaskFolders(i, false));
1082                 }
1083             } else {
1084                 // bug fix #25512 - avoid race condition
1085                 if (!this._app._overviewPanelContent) {
1086                     this._app.setOverviewPanelContent(true);
1087                 }
1088                 cc = this._taskTreeController.getTaskFolders(this._app.getOverviewId(), false);
1089             }
1090         } else {
1091             this._app._createDeferredFolders(ZmApp.TASKS);
1092             var list = appCtxt.accountList.visibleAccounts;
1093             for (var i = 0; i < list.length; i++) {
1094                 var acct = list[i];
1095                 if (!appCtxt.get(ZmSetting.TASKS_ENABLED, null, acct)) { continue; }
1096 
1097                 var tasks = appCtxt.getFolderTree(acct).getByType(ZmOrganizer.TASKS);
1098                 for (var j = 0; j < tasks.length; j++) {
1099                     if (tasks[j].nId == ZmOrganizer.ID_TRASH) {
1100                         continue;
1101                     }
1102                     cc.push(tasks[j]);
1103                 }
1104             }
1105 
1106         }
1107 
1108         this._taskLocalFolderIds = [];
1109 
1110         for (var i = 0; i < cc.length; i++) {
1111             var cal = cc[i];
1112             if (cal.noSuchFolder) { continue; }
1113 
1114             if (cal.isRemote && !cal.isRemote()) {
1115                 this._taskLocalFolderIds.push(cal.id);
1116             }
1117         }
1118 
1119         // return list of checked calendars
1120         return this._taskLocalFolderIds;
1121     }
