 93 /**
 94  * Formats a size (in bytes) to the largest whole unit. For example,
 95  * AjxUtil.formatSize(302132199) returns "288 MB".
 96  *
 97  * @param size      The size (in bytes) to be formatted.
 98  * @param round     True to round to nearest integer. Default is true.
 99  * @param fractions Number of fractional digits to display, if not rounding.
100  *                  Trailing zeros after the decimal point are trimmed.
101  */
102 AjxUtil.formatSize = 
103 function(size, round, fractions) {
104 	if (round == null) round = true;
105 	if (fractions == null) fractions = 20; // max allowed for toFixed is 20
106 
107 	var formattedUnits = AjxMsg.sizeBytes;
108 	var units = AjxMsg.SIZE_BYTES;
109 	if (size >= 1073741824) {
110 		formattedUnits = AjxMsg.sizeGigaBytes;
111 		units = AjxUtil.SIZE_GIGABYTES;
112 	}
113 	else if (size >= 1048576) {
114 		formattedUnits = AjxMsg.sizeMegaBytes;
115 		units = AjxUtil.SIZE_MEGABYTES;
116 	}
117 	else if (size > 1023) {
118 		formattedUnits = AjxMsg.sizeKiloBytes;
119 		units = AjxUtil.SIZE_KILOBYTES;
120 	}
121 
122 
123 	var formattedSize = AjxUtil.formatSizeForUnits(size, units, round, fractions);
124 	return AjxMessageFormat.format(AjxMsg.formatSizeAndUnits, [formattedSize, formattedUnits]);
125 };

127 /**
128  * Formats a size (in bytes) to a specific unit. Since the unit size is
129  * known, the unit is not shown in the returned string. For example,
130  * AjxUtil.formatSizeForUnit(302132199, AjxUtil.SIZE_MEGABYTES, false, 2) 
131  * returns "288.13".
132  *
133  * @param size      The size (in bytes) to be formatted.
134  * @param units     The unit of measure.
135  * @param round     True to round to nearest integer. Default is true.
136  * @param fractions Number of fractional digits to display, if not rounding.
137  *                  Trailing zeros after the decimal point are trimmed.
138  */
139 AjxUtil.formatSizeForUnits = 
140 function(size, units, round, fractions) {
141 	if (units == null) units = AjxUtil.SIZE_BYTES;
142 	if (round == null) round = true;
143 	if (fractions == null) fractions = 20; // max allowed for toFixed is 20
144 
145 	switch (units) {
146 		case AjxUtil.SIZE_GIGABYTES: { size /= 1073741824; break; }
147 		case AjxUtil.SIZE_MEGABYTES: { size /= 1048576; break; }
148 		case AjxUtil.SIZE_KILOBYTES: { size /= 1024; break; }
149 	}
150 	var dot = I18nMsg.numberSeparatorDecimal !='' ? I18nMsg.numberSeparatorDecimal : '.';
151 	var pattern = I18nMsg.formatNumber.replace(/\..*$/, ""); // Strip off decimal, we'll be adding one anyway
152 	pattern = pattern.replace(/,/, "");       // Remove the ,
153 	if (!round && fractions) {
154 		pattern = pattern += dot;
155 		for (var i = 0; i < fractions; i++) {
156 			pattern += "#";
157 		}
158 	}
159 	return AjxNumberFormat.format(pattern, size);
160 };

162 /**
163  * Performs the opposite of AjxUtil.formatSize in that this function takes a 
164  * formatted size.
165  *
166  * @param units Unit constant: "GB", "MB", "KB", "B". Must be specified 
167  *              unless the formatted size ends with the size marker, in
168  *				which case the size marker in the formattedSize param
169  *				overrides this parameter.
170  */
171 AjxUtil.parseSize = 
172 function(formattedSize, units) {
173 	// NOTE: Take advantage of fact that parseFloat ignores bad chars
174 	//       after numbers
175 	if (typeof formattedSize != _STRING_) {
176 		formattedSize = formattedSize.toString() ;
177 	}
178 	var size = parseFloat(formattedSize.replace(/^\s*/,""));
179 
180 	var marker = /[GMK]?B$/i;
181 	var result = marker.exec(formattedSize);
182 	if (result) {
183 		//alert("units: "+units+", result[0]: '"+result[0]+"'");
184 		units = result[0].toUpperCase();
185 	}
186 	
187 	switch (units) {
188 		case AjxUtil.SIZE_GIGABYTES: size *= 1073741824; break;
189 		case AjxUtil.SIZE_MEGABYTES: size *= 1048576; break; 
190 		case AjxUtil.SIZE_KILOBYTES: size *= 1024; break;
191 	}
192 	
193 	//alert("AjxUtil#parseSize: formattedSize="+formattedSize+", size="+size);
194 	return size;
195 };

207 AjxUtil.onassert = 
208 function(aMessage) {
209 	// Create an exception object and set the message
210 	var myException = new Object();
211 	myException.message = aMessage;
212 	
213 	// Compile a stack trace
214 	var myStack = new Array();
215 	if (AjxEnv.isIE5_5up) {
216 		// On IE, the caller chain is on the arguments stack
217 		var myTrace = arguments.callee.caller;
218 		var i = 0; // stop at 20 since there might be somehow an infinite loop here. Maybe in case of a recursion. 
219 		while (myTrace && i++ < 20) {
220 		    myStack[myStack.length] = myTrace;
221 	    	myTrace = myTrace.caller;
222 		}
223 	} else {
224 		try {
225 			var myTrace = arguments.callee.caller;
226 			while (myTrace) {
227 				myStack[myStack.length] = myTrace;
228 				if (myStack.length > 2) break;
229 				myTrace = myTrace.caller;
230 		    }
231 		} catch (e) {
232 		}
233 	}
234 	myException.stack = myStack;
235 	
236 	// Alert with the message and a description of the stack
237 	var stackString = '';
238 	var MAX_LEN = 170;
239 	for (var i = 1; i < myStack.length; i++) {
240 		if (i > 1) stackString += '\n';
241 		if (i < 11) {
242 			var fs = myStack[i].toString();
243 			if (fs.length > MAX_LEN) {
244 				fs = fs.substr(0,MAX_LEN) + '...';
245 				fs = fs.replace(/\n/g, '');
246 			}
247 			stackString += i + ': ' + fs;
248 		} else {
249 			stackString += '(' + (myStack.length - 11) + ' frames follow)';
250 			break;
251 		}
252 	}
253 	alert('assertion:\n\n' + aMessage + '\n\n---- Call Stack ---\n' + stackString);
254 	
255 	// Now throw the exception
256 	throw myException;
257 };

264 AjxUtil.getInnerText = 
265 function(node) {
266  	if (AjxEnv.isIE)
267  		return node.innerText;

269 	function f(n) {
270 		if (n) {
271 			if (n.nodeType == 3 /* TEXT_NODE */)

273 			if (n.nodeType == 1 /* ELEMENT_NODE */) {

276 				var str = "";

278 					str += f(i);
279 				return str;
280 			}
281 		}
282 		return "";
283 	};
284 	return f(node);

287 /**
288  * This method returns a proxy for the specified object. This is useful when
289  * you want to modify properties of an object and want to keep those values
290  * separate from the values in the original object. You can then iterate
291  * over the proxy's properties and use the <code>hasOwnProperty</code>
292  * method to determine if the property is a new value in the proxy.
293  * <p>
294  * <strong>Note:</strong>
295  * A reference to the original object is stored in the proxy as the "_object_" 
296  * property.
297  *
298  * @param object [object] The object to proxy.
299  * @param level  [number] The number of property levels deep to proxy.
300  *						  Defaults to zero.
301  */
302 AjxUtil.createProxy = 
303 function(object, level) {
304 	var proxy;
305 	var proxyCtor = function(){}; // bug #6517 (Safari doesnt like 'new Function')
306 	proxyCtor.prototype = object;
307 	if (object instanceof Array) {
308 		proxy  = new Array();
309 		var cnt  = object.length;
310 		for(var ix = 0; ix < cnt; ix++) {
311 			proxy[ix] = object[ix];
312 		}
313 	} else {
314 		proxy = new proxyCtor;
315 	}
316 	
317 	if (level) {
318 		for (var prop in object) {
319 			if (typeof object[prop] == "object" && object[prop] !== null)
320 				proxy[prop] = AjxUtil.createProxy(object[prop], level - 1);
321 		}
322 	}	
323 	
324 	proxy._object_ = object;
325 	return proxy;
326 };
327 
328 AjxUtil.unProxy =
329 function(proxy) {
330 	var object = proxy && proxy._object_;
331 	if (object) {
332 		for (var prop in proxy) {
333 			if (proxy.hasOwnProperty(prop) && prop!="_object_") {
334 				object[prop] = proxy[prop];
335 			}
336 		}
337 		return object;
338 	}
339 	return null;
340 }

358 AjxUtil.arrayAsHash =
359 function(array, valueOrFunc) {
360 	array = AjxUtil.toArray(array);
361 	var hash = {};
362 	var func = typeof valueOrFunc == "function" && valueOrFunc;
363 	var value = valueOrFunc || true; 
364 	for (var i = 0; i < array.length; i++) {
365 		var key = array[i];
366 		hash[key] = func ? func(key, hash, i, array) : value;
367 	}
368 	return hash;
369 };

454 /**
455  * Remove duplicates from the given array,
456  * <strong>in-place</strong>. Stable with regards to ordering.
457  *
458  * Please note that this method is O(n^2) if Array is backed by an
459  * array/vector data structure.
460  *
461  * @param array [array]     array to process
462  * @param keyfn [function]  used to extract a comparison key from each
463  *                          list element, default is to compare
464  *                          elements directly. if the comparison key
465  *                          is 'undefined', the element is always
466  *                          retained
467  */
468 AjxUtil.dedup = function(array, keyfn) {
469 	if (!keyfn)
470 		keyfn = function(v) { return v; };
471 
472 	var seen = {};
473 
474 	for (var i = 0; i < array.length; i++) {
475 		var key = keyfn(array[i]);
476 
477 		if (key !== undefined && seen[key]) {
478 			array.splice(i, 1);
479 			i -= 1;
480 		}
481 
482 		seen[key] = true;
483 	}
484 };

505 		}

516 		if (p in object1) {
517 			delete object2[p];
518 		}
519 	}
520 	return AjxUtil.keys(object2);

523 AjxUtil.getFirstElement = function(parent, ename, aname, avalue) {
524     for (var child = parent.firstChild; child; child = child.nextSibling) {
525         if (child.nodeType != AjxUtil.ELEMENT_NODE) continue;
526         if (ename && child.nodeName != ename) continue;
527         if (aname) {
528             var attr = child.getAttributeNode(aname);
529             if (attr.nodeName != aname) continue;
530             if (avalue && attr.nodeValue != avalue) continue;
531         }
532         return child;
533     }
534     return null;
535 };

537 /**
538  * @param params	[hash]		hash of params:
539  *        relative	[boolean]*	if true, return a relative URL
540  *        protocol	[string]*	protocol (trailing : is optional)
541  *        host		[string]*	server hostname
542  *        port		[int]*		server port
543  *        path		[string]*	URL path
544  *        qsReset	[boolean]*	if true, clear current query string
545  *        qsArgs	[hash]*		set of query string names and values
546  */
547 AjxUtil.formatUrl =
548 function(params) {
549 	params = params || {};
550 	var url = [];
551 	var i = 0;
552 	if (!params.relative) {
553 		var proto = params.protocol || location.protocol;
554 		if (proto.indexOf(":") == -1) {
555 			proto = proto + ":";
556 		}
557 		url[i++] = proto;
558 		url[i++] = "//";
559 		url[i++] = params.host || location.hostname;
560 		var port = Number(params.port || location.port);
561 		if (port &&
562 			((proto == ZmSetting.PROTO_HTTP && port != ZmSetting.HTTP_DEFAULT_PORT) ||
563 			 (proto == ZmSetting.PROTO_HTTPS && port != ZmSetting.HTTPS_DEFAULT_PORT))) {
564 			url[i++] =  ":";
565 			url[i++] = port;
566 		}
567 	}
568 	url[i++] = params.path || location.pathname;
569 	var qs = "";
570 	if (params.qsArgs) {
571 		qs = AjxStringUtil.queryStringSet(params.qsArgs, params.qsReset);
572 	} else {
573 		qs = params.qsReset ? "" : location.search;
574 	}
575 	url[i++] = qs;
576 	
577 	return url.join("");
578 };

614 /**
615  * mergesort+dedupe
616 **/
617 AjxUtil.mergeArrays =
618 function(arr1, arr2, orderfunc) {
619 	if(!orderfunc) {
620 		orderfunc = AjxUtil.__mergeArrays_orderfunc;
621 	}
622  	var tmpArr1 = [];
623  	var cnt1 = arr1.length;
624  	for(var i = 0; i < cnt1; i++) {
625  		tmpArr1.push(arr1[i]);
626  	}
627 
628  	var tmpArr2 = [];
629  	var cnt2 = arr2.length;
630  	for(var i = 0; i < cnt2; i++) {
631  		tmpArr2.push(arr2[i]);
632  	} 	
633 	var resArr = [];
634 	while (tmpArr1.length>0 && tmpArr2.length>0) {
635 		if (orderfunc(tmpArr1[0],resArr[resArr.length-1])==0) {
636 			tmpArr1.shift();
637 			continue;
638 		}
639 		
640 		if (orderfunc(tmpArr2[0],resArr[resArr.length-1])==0) {
641 			tmpArr2.shift();
642 			continue;
643 		}		
644 			
645 		if (orderfunc(tmpArr1[0], tmpArr2[0])<0) {
646 			resArr.push(tmpArr1.shift());
647 		} else if (orderfunc(tmpArr1[0],tmpArr2[0])==0) {
648 			resArr.push(tmpArr1.shift());
649 			tmpArr2.shift();
650 		} else {
651 			resArr.push(tmpArr2.shift());
652 		}
653 	}
654 		
655 	while (tmpArr1.length>0) {
656 		if (orderfunc(tmpArr1[0],resArr[resArr.length-1])==0) {
657 			tmpArr1.shift();
658 			continue;
659 		}		
660 		resArr.push(tmpArr1.shift());
661 	}
662 		
663 	while (tmpArr2.length>0) {
664 		if (orderfunc(tmpArr2[0], resArr[resArr.length-1])==0) {
665 			tmpArr2.shift();
666 			continue;
667 		}			
668 		resArr.push(tmpArr2.shift());
669 	}
670 	return resArr;	
671 };

674     if(val1>val2)    return  1;
675     if (val1 < val2) return -1;
676     if(val1 == val2) return  0;

679 AjxUtil.arraySubstract = function (arr1, arr2, orderfunc) {
680 	if(!orderfunc) {
681 		orderfunc = function (val1,val2) {
682 			if(val1>val2)
683 				return 1;
684 			if (val1 < val2)
685 				return -1;
686 			if(val1 == val2)
687 				return 0;
688 		}
689 	}
690  	var tmpArr1 = [];
691  	var cnt1 = arr1.length;
692  	for(var i = 0; i < cnt1; i++) {
693  		tmpArr1.push(arr1[i]);
694  	}
695 
696  	var tmpArr2 = [];
697  	var cnt2 = arr2.length;
698  	for(var i = 0; i < cnt2; i++) {
699  		tmpArr2.push(arr2[i]);
700  	} 	
701  	tmpArr2.sort(orderfunc);
702  	tmpArr1.sort(orderfunc);
703 	var resArr = [];
704 	while(tmpArr1.length > 0 && tmpArr2.length > 0) {
705 		if(orderfunc(tmpArr1[0],tmpArr2[0])==0) {
706 			tmpArr1.shift();
707 			tmpArr2.shift();
708 			continue;
709 		}
710 		
711 		if(orderfunc(tmpArr1[0],tmpArr2[0]) < 0) {
712 			resArr.push(tmpArr1.shift());
713 			continue;
714 		}
715 		
716 		if(orderfunc(tmpArr1[0],tmpArr2[0]) > 0) {
717 			tmpArr2.shift();
718 			continue;
719 		}
720 	}
721 	
722 	while(tmpArr1.length > 0) {
723 		resArr.push(tmpArr1.shift());
724 	}
725 	
726 	return resArr;
727 }

746  * Does a shallow comparison of two arrays.
747  *
748  * @param arr1	[array]
749  * @param arr2	[array]
750  */
751 AjxUtil.arrayCompare =
752 function(arr1, arr2) {
753 	if ((!arr1 || !arr2) && (arr1 != arr2)) {
754 		return false;
755 	}
756 	if (arr1.length != arr2.length) {
757 		return false;
758 	}
759 	for (var i = 0; i < arr1.length; i++) {
760 		if (arr1[i] != arr2[i]) {
761 			return false;
762 		}
763 	}
764 	return true;

768  * Does a shallow comparison of two hashes.
769  *
770  * @param hash1	[hash]
771  * @param hash2	[hash]
772  */
773 AjxUtil.hashCompare =
774 function(hash1, hash2) {
775 
776 	var keys1 = AjxUtil.getHashKeys(hash1);
777 	var keys2 = AjxUtil.getHashKeys(hash2);
778 	if (!AjxUtil.arrayCompare(keys1, keys2)) {
779 		return false;
780 	}
781 	for (var i = 0, len = keys1.length; i < len; i++) {
782 		var key = keys1[i];
783 		if (hash1[key] != hash2[key]) {
784 			return false;
785 		}
786 	}
787 	return true;

871 /**
872  *  Convert non-ASCII characters to valid HTML UNICODE entities 
873  * @param {string}
874  * 
875 */
876 AjxUtil.convertToEntities = function (source){
877 	var result = '', temp, length = 0, i = 0;
878     
879     if (!source || !(length = source.length)) return source;
880     
881 	for(i; i < length; ++i){
882 		if(source.charCodeAt(i) > 127){
883 			temp = source.charCodeAt(i).toString(10);
884 			while(temp.length < 4){
885 				temp = '0' + temp;
886 			}
887 			result += '&#' + temp + ';';
888 		} else {
889 			result += source.charAt(i);
890 		}
891 	}
892 	return result;
893 };
