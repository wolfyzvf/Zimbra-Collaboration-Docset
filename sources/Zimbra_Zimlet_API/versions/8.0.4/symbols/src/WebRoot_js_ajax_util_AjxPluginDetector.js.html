 29 AjxPluginDetector.canDetectPlugins =
 30 function() {
 31 	return AjxEnv.isIE || (navigator.plugins && navigator.plugins.length > 0);
 32 };

 34 AjxPluginDetector.detectFlash =
 35 function() {
 36 	if(AjxEnv.isIE) {
 37 		return AjxPluginDetector.detectActiveXControl('ShockwaveFlash.ShockwaveFlash.1');
 38 	} else {
 39 		return AjxPluginDetector.detectPlugin('Shockwave','Flash'); 
 40 	}
 41 };

 43 AjxPluginDetector.detectPDFReader =
 44 function(){
 45     if(AjxEnv.isIE){
 46         return  ( AjxPluginDetector.detectActiveXControl('PDF.PdfCtrl.5')
 47                 || AjxPluginDetector.detectActiveXControl('AcroExch.Document') );
 48     }else{
 49         var hasPDFReader = false;
 50         if(AjxEnv.isChrome){
 51             hasPDFReader = AjxPluginDetector.detectPlugin('Chrome PDF Viewer');
 52         }else if(AjxEnv.isFirefox){
 53             hasPDFReader = AjxPluginDetector.detectPlugin('Firefox PDF Plugin for Mac OS X');
 54         }
 55         if(!hasPDFReader){
 56             hasPDFReader = AjxPluginDetector.detectPlugin('Adobe Acrobat');
 57         }
 58         return hasPDFReader;
 59     }
 60 };

 80 // If quicktime is installed, returns the version as an array: [major, minor, build]
 81 AjxPluginDetector.getQuickTimeVersion =
 82 function() {
 83 	if(AjxEnv.isIE) {
 84 		var object = new ActiveXObject("QuickTimeCheckObject.QuickTimeCheck.1");
 85 		DBG.println(AjxDebug.DBG1, "AjxPluginDetector: Quicktime is " + object.IsQuickTimeAvailable(0) ? "available" : "not available");
 86 		if (object.IsQuickTimeAvailable(0)) {
 87 			try {
 88 				var version = Number(object.QuickTimeVersion).toString(16);
 89 				var result = [];
 90 				for(var i = 0; i < 3; i++) {
 91 					result[i] = Number(version.charAt(i));
 92 				}
 93 				return result;
 94 			} catch(e) {
 95 				DBG.println(AjxDebug.DBG1, "AjxPluginDetector: Error while checking QuickTimeVersion: " + e);
 96 			}
 97 		}
 98 		return null;
 99 	} else {
100 		var match = AjxPluginDetector.matchPluginName(/QuickTime Plug-in (\d+)\.?(\d+)?\.?(\d+)?/);
101 		if (match) {
102 			DBG.println("AjxPluginDetector: able to find match for QuickTime plugin with version: " + match);
103 			var result = [];
104 			for(var i = 0; i < 3; i++) {
105 				result[i] = Number(match[i + 1] || 0);
106 			}
107 			return result;
108 		} else {
109 			DBG.println("AjxPluginDetector: unable to find match for QuickTime plugin with version");
110 			return null;
111 		}
112 	}
113 };

115 /**
116  * This code is part of JQuery's Flash plugin.
117  * http://jquery.lukelutman.com/plugins/flash/
118  *
119  * @return Flash plugin version
120  */
121 AjxPluginDetector.getFlashVersion =
122 function() {
123     var flashVersion = "0,0,0";
124     // ie
125     try {
126         try {
127             // avoid fp6 minor version lookup issues
128             // see: http://blog.deconcept.com/2006/01/11/getvariable-setvariable-crash-internet-explorer-flash-6/
129             var axo = new ActiveXObject('ShockwaveFlash.ShockwaveFlash.6');
130             try {
131                 axo.AllowScriptAccess = 'always';
132             }
133             catch(e) {
134                 return '6,0,0';
135             }
136         } catch(e) {
137             }
138         flashVersion = new ActiveXObject('ShockwaveFlash.ShockwaveFlash').GetVariable('$version').replace(/\D+/g, ',').match(/^,?(.+),?$/)[1];
139         // other browsers
140     } catch(e) {
141         try {
142             if (navigator.mimeTypes["application/x-shockwave-flash"].enabledPlugin) {
143                 flashVersion = (navigator.plugins["Shockwave Flash 2.0"] || navigator.plugins["Shockwave Flash"]).description.replace(/\D+/g, ",").match(/^,?(.+),?$/)[1];
144             }
145         } catch(e) {
146         }
147     }
148 	return flashVersion;
149 };

171 AjxPluginDetector.detectPlugin =
172 function() {
173 	DBG.println(AjxDebug.DBG1, "-----------------------<br>AjxPluginDetector: Looking for plugin: [" + AjxPluginDetector._argumentsToString(AjxPluginDetector.detectPlugin.arguments) + "]");
174 	var names = AjxPluginDetector.detectPlugin.arguments;
175 	var allPlugins = navigator.plugins;
176 	var pluginsArrayLength = allPlugins.length;
177 	for (var pluginsArrayCounter=0; pluginsArrayCounter < pluginsArrayLength; pluginsArrayCounter++ ) {
178 	    // loop through all desired names and check each against the current plugin name
179 	    var numFound = 0;
180 	    for(var namesCounter=0; namesCounter < names.length; namesCounter++) {
181 			// if desired plugin name is found in either plugin name or description
182 			if (allPlugins[pluginsArrayCounter]) {
183 				if( (allPlugins[pluginsArrayCounter].name.indexOf(names[namesCounter]) >= 0)) {
184 					// this name was found
185 					DBG.println(AjxDebug.DBG1, "AjxPluginDetector: found name match '" + allPlugins[pluginsArrayCounter].name + "'");
186 					numFound++;
187 				} else if (allPlugins[pluginsArrayCounter].description.indexOf(names[namesCounter]) >= 0) {
188 					// this name was found
189 					DBG.println(AjxDebug.DBG1, "AjxPluginDetector: found description match '" + allPlugins[pluginsArrayCounter].description + "'");
190 					numFound++;
191 				}
192 			}
193 	    }
194 	    // now that we have checked all the required names against this one plugin,
195 	    // if the number we found matches the total number provided then we were successful
196 	    if(numFound == names.length) {
197 			DBG.println(AjxDebug.DBG1, "AjxPluginDetector: Found plugin!<br>-----------------------");
198 			return true;
199 	    } else if (numFound) {
200 			DBG.println(AjxDebug.DBG1, "AjxPluginDetector: Found partial plugin match, numFound=" + numFound);
201 		}
202 	}
203 	DBG.println(AjxDebug.DBG1, "AjxPluginDetector: Failed to find plugin.<br>-----------------------");
204 	return false;
205 };

207 AjxPluginDetector.matchPluginName =
208 function(regExp) {
209 	var allPlugins = navigator.plugins;
210 	var pluginsArrayLength = allPlugins.length;
211 	for (var pluginsArrayCounter=0; pluginsArrayCounter < pluginsArrayLength; pluginsArrayCounter++ ) {
212 		var match = allPlugins[pluginsArrayCounter].name.match(regExp);
213 		if (match) {
214 			return match;
215 		}
216 	}
217 	return null;
218 };

220 AjxPluginDetector.detectActiveXControl =
221 function(progId) {
222 	try {
223 		new ActiveXObject(progId);
224 		DBG.println(AjxDebug.DBG1, "AjxPluginDetector: found ActiveXObject '" + progId + "'");
225 		return true;
226 	} catch (e) {
227 		DBG.println(AjxDebug.DBG1, "AjxPluginDetector: unable to find ActiveXObject '" + progId + "'");
228 		return false;
229 	}
230 };

234 	try {

236 		return object.IsQuickTimeAvailable(0);
237 	} catch (e) {
238 		return false;
239 	}

242 // Util method to log arguments, which to my surprise are not actually an array.
243 AjxPluginDetector._argumentsToString =
244 function(args) {
245 	var array = [];
246 	for (var i = 0, count = args.length; i < count; i++) {
247 		array[i] = args[i];
248 	}
249 	return array.join(',')
250 };
