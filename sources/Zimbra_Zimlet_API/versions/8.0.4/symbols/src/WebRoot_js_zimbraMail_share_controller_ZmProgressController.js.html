 16 /**
 17  * @overview
 18  * This file defines the progress controller.
 19  *
 20  * it gets a list of folders to work on, uses the search controller to get all the messages in the folders, in chunks. Callbacks the work passed to it to perform the work
 21  * on the message id chunks.
 22  *
 23  */
 24 
 25 /**
 26  *
 27  * @author Eran Yarkon
 28  *
 29  * @param {DwtControl}		container	the containing shell
 30  * @param {ZmApp}		app		the containing application
 31  * 
 32  * @extends		ZmController
 33  */
 34 ZmProgressController = function(container, app) {
 35 	if (arguments.length == 0) { return; }
 36 	ZmController.call(this, container, app);
 37     this._totalNumMsgs = 0; //for determining if run in background is available
 38 };
 39 
 40 ZmProgressController.prototype = new ZmController;
 41 ZmProgressController.prototype.constructor = ZmProgressController;

 52 	return "ZmProgressController";

 56 ZmProgressController.prototype._getProgressDialog =
 57 function() {
 58 	if (!this._progressDialog ) {
 59 		var dialog = this._progressDialog = new DwtMessageDialog({parent:this._shell, buttons:[DwtDialog.YES_BUTTON, DwtDialog.CANCEL_BUTTON], id: Dwt.getNextId("ZmProgressControllerDialog_")});
 60 		dialog.registerCallback(DwtDialog.CANCEL_BUTTON, new AjxCallback(this, this._cancelAction));
 61 		dialog.registerCallback(DwtDialog.YES_BUTTON, new AjxCallback(this, this._runInBackgroundAction));
 62         dialog.getButton(DwtDialog.YES_BUTTON).setText(ZmMsg.runInBackground);
 63     }
 64     this._progressDialog.getButton(DwtDialog.YES_BUTTON).setVisible(this._totalNumMsgs <= appCtxt.get(ZmSetting.FILTER_BATCH_SIZE));
 65 	return this._progressDialog;
 66 };
 67 
 68 ZmProgressController.prototype._getFinishedDialog =
 69 function() {
 70 	if (!ZmProgressController._finishedDialog) {
 71 		var dialog = ZmProgressController._finishedDialog = appCtxt.getMsgDialog();
 72 		dialog.reset();
 73 	}
 74 
 75 	return ZmProgressController._finishedDialog;
 76 };

 78 /**
 79  * start a progres on a folder list and work definition
 80  * @param folderList - list of folders to work on the messages of, in chunks.
 81  * @param work - implements an unwritten interface. See ZmFilterRulesController.ZmFilterWork for the first example of an implementation
 82  */
 83 ZmProgressController.prototype.start =
 84 function(folderList, work) {
 85 	this._currentWork = work;
 86 	this._currentRun = new ZmProgressRun(folderList);
 87     this._totalNumMsgs = this.getNumMsgs(folderList);
 88 	this._nextChunk();
 89 };
 90 
 91 
 92 /**
 93  * next chunk of work. Get the next chunk of ids from the search
 94  */
 95 ZmProgressController.prototype._nextChunk =
 96 function() {
 97 	var run = this._currentRun;
 98 	var work = this._currentWork;
 99 	if (run._runInBackground) {
100 		//don't get Ids
101 		this._handleRunInBackground(this._getFolderQuery(run._folderList));	
102 	}
103 	else {
104 		var searchParams = {
105 			query:		this._getFolderQuery(run._folderList),
106 			types:		ZmItem.MSG,
107 			limit:		ZmProgressController.CHUNK_SIZE,
108 			idsOnly:	true,
109 			noBusyOverlay: true
110 		};
111 	
112 		if (run._lastItem) {
113 			//this is not the first chunk - supply the last id and sort val to the search.
114 			searchParams.lastId = run._lastItem.id;
115 			searchParams.lastSortVal = run._lastItem.sf;
116 			AjxDebug.println(AjxDebug.PROGRESS, "***** progress search: " + searchParams.query + " --- " + [run._lastItem.id, run._lastItem.sf].join("/"));
117 		}
118 	
119 		var search = new ZmSearch(searchParams);
120 		var respCallback = new AjxCallback(this, this._handleSearchResults);
121 		appCtxt.getSearchController().redoSearch(search, true, null, respCallback);
122 	}
123 };
124 
125 ZmProgressController.prototype._handleRunInBackground = 
126 function(query) {
127 	var run = this._currentRun;
128 	if (run._cancelled) {
129 		return;
130 	}
131 	run._finished = true; //running all at once
132 	var afterWorkCallback = new AjxCallback(this, this._afterChunk);
133 	this._currentWork.doWork(null, query, afterWorkCallback); //callback the work to do it's job on the message ids	
134 };
135 
136 /**
137  * process the returned message ids and
138  * @param result
139  */
140 ZmProgressController.prototype._handleSearchResults =
141 function(result) {
142 	var run = this._currentRun;
143 	if (run._cancelled) {
144 		return;
145 	}
146 
147 	var response = result.getResponse();
148 	var items = response.getResults();
149 
150 	AjxDebug.println(AjxDebug.PROGRESS, "progress search results: " + items.length);
151 	if (!items.length) {
152 		AjxDebug.println(AjxDebug.PROGRESS, "progress with empty search results!");
153 		return;
154 	}
155 
156 	run._lastItem = items[items.length - 1];
157 	run._totalMessagesProcessed += items.length;
158 	var hasMore = response.getAttribute("more");
159 	run._finished = !hasMore;
160 
161 	items = this._getIds(items);
162 
163 	var afterWorkCallback = new AjxCallback(this, this._afterChunk);
164 	this._currentWork.doWork(items, null, afterWorkCallback); //callback the work to do it's job on the message ids
165 
166 };
167 
168 /**
169  * returns here after the work is done on the chunk
170  */
171 ZmProgressController.prototype._afterChunk =
172 function() {
173 	var work = this._currentWork;
174 	var run = this._currentRun;
175 	if (run._cancelled) {
176 		return;
177 	}
178 
179 	var progDialog = this._getProgressDialog();
180 	if (run._finished) {
181 		//search is over, show summary messsage
182 		if (progDialog.isPoppedUp()) {
183 			progDialog.popdown();
184 		}
185 		var messagesProcessed = run._runInBackground ? false : run._totalMessagesProcessed;
186 		var finishedMessage = work.getFinishedMessage(messagesProcessed);
187 		var finishDialog = this._getFinishedDialog();
188 		finishDialog.setMessage(finishedMessage, DwtMessageDialog.INFO_STYLE, work.getFinishedTitle());
189 		finishDialog.popup();
190 		return;
191 	}
192 
193 	if (!run._runInBackground) {
194 		var workMessage = work.getProgressMessage(run._totalMessagesProcessed);
195 		progDialog.setMessage(workMessage, DwtMessageDialog.INFO_STYLE, work.getProgressTitle());
196 		if (!progDialog.isPoppedUp()) {
197 			progDialog.popup();
198 		}
199 	}
200 
201 	this._nextChunk();
202 };

205 /**
206  * extract just the ids from the item objects. (they include also the search value)
207  * @param items
208  */
209 ZmProgressController.prototype._getIds =
210 function(items) {
211 	var ids = [];
212 	if (!items.length) { //not sure if this could happen but I've seen it elsewhere.
213 		items = [items];
214 	}
215 
216 	for (var i = 0; i < items.length; i++) {
217 		ids.push(items[i].id);
218 	}
219 	return ids;
220 };
221 
222 ZmProgressController.prototype._getFolderQuery =
223 function(folderList) {
224 	if (!(folderList instanceof Array)) {
225 		folderList = [folderList];
226 	}
227 	var query = [];
228 	for (var j = 0; j < folderList.length; j++) {
229 		query.push(folderList[j].createQuery());
230 	}
231 	return query.join(" OR ");
232 
233 };

235 /**
236  * Determine total number of messages filters are being applied to.
237  * @param folderList {ZmOrganizer[]} array of folders
238  * @return {int} number of messages
239  */
240 ZmProgressController.prototype.getNumMsgs = 
241 function(folderList) {
242     var numMsgs = 0;
243     if (!(folderList instanceof Array)) {
244         folderList = [folderList];
245     }
246 
247     for (var j = 0; j < folderList.length; j++) {
248         numMsgs += folderList[j].numTotal;
249     }
250     return numMsgs;
251 };
252 
253 ZmProgressController.prototype._cancelAction =
254 function() {
255 	this._currentRun._cancelled = true;
256 	var dialog = this._getProgressDialog();
257 	if (dialog && dialog.isPoppedUp()) {
258 		dialog.popdown();
259 	}
260 };
261 
262 ZmProgressController.prototype._runInBackgroundAction = 
263 function() {
264 	this._currentRun._runInBackground = true;
265 	var dialog = this._getProgressDialog();
266 	if (dialog && dialog.isPoppedUp()) {
267 		dialog.popdown();
268 	}
269 	AjxDebug.println(AjxDebug.PROGRESS, "set to run in background");
270 };

272 /**
273  * internal class to keep track of progress, with last item processed, total messages processed, and the folder list we work on
274  * @param folderList
275  */
276 ZmProgressRun = function(folderList) {
277 	this._lastItem = null;
278 	this._totalMessagesProcessed = 0;
279 	this._folderList = folderList;
280 };
281 
282 ZmProgressRun.CHUNK_SIZE = 100;
