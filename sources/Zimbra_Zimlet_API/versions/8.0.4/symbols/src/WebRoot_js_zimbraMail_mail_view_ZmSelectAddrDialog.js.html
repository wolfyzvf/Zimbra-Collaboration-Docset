 20 /**
 21  * Creates a dialog to allow the user to select a recipient address to search on.
 22  * @class
 23  * Provides the user a way to search on a recipient when they click on the To:
 24  * column in Sent or Drafts. That is in place of actual sorting. See bug 6830.
 25  *
 26  * @author Conrad Damon
 27  *
 28  * @param {DwtComposite}	parent				the parent widget (the shell)
 29  *
 30  * @extends		DwtDialog
 31  */
 32 ZmSelectAddrDialog = function(parent) {
 33 
 34 	if (arguments.length == 0) { return; }
 35 
 36 	DwtDialog.call(this, {parent:parent, title:ZmMsg.findEmailsSentFolderTitle, className:"ZmSelectAddrDialog"});
 37     var okButton = this.getButton(DwtDialog.OK_BUTTON);
 38     if (okButton)
 39 	    okButton.setText(ZmMsg.find);
 40 
 41     // create auto-completer
 42     if (appCtxt.get(ZmSetting.CONTACTS_ENABLED) || appCtxt.get(ZmSetting.GAL_ENABLED)) {
 43         var params = {
 44             dataClass:		appCtxt.getAutocompleter(),
 45             matchValue:		ZmAutocomplete.AC_VALUE_EMAIL,
 46 			contextId:		this.toString()
 47         };
 48         this._acAddrSelectList = new ZmAutocompleteListView(params);
 49     }
 50 
 51 	this.setContent(this._contentHtml());
 52     this._createInputView();
 53     this._createControls();
 54 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._okButtonListener));
 55 };
 56 
 57 ZmSelectAddrDialog.prototype = new DwtDialog;
 58 ZmSelectAddrDialog.prototype.constructor = ZmQuickAddDialog;
 59 
 60 ZmSelectAddrDialog.prototype.isZmSelectAddrDialog = true;
 61 ZmSelectAddrDialog.prototype.toString = function() { return "ZmSelectAddrDialog"; };
 62 
 63 ZmSelectAddrDialog.prototype._contentHtml =
 64 function() {
 65     var html = [];
 66 	var idx = 0;
 67 	html[idx++] = "<table cellpadding=0 cellspacing=0 border=0>";
 68 	html[idx++] = AjxTemplate.expand("share.Dialogs#ZmSelectAddrDialog", {id:this._htmlElId});
 69 	html[idx++] = "</table>";
 70 	return html.join("");
 71 };
 72 
 73 ZmSelectAddrDialog.prototype._createControls =
 74 function() {
 75  if (this._htmlElId) {
 76      var cboxId = this._htmlElId + "_cboxFind";
 77      this._findEmailsCheckbox = new DwtCheckbox({parent: this, id: cboxId, checked: true});
 78      this._findEmailsCheckbox.replaceElement(document.getElementById(cboxId));
 79      this._findEmailsCheckbox.setText(ZmMsg.findEmailsToAll);
 80  }
 81 };
 82 
 83 ZmSelectAddrDialog.prototype._createInputView =
 84 function(){
 85     this._recipientInput = new DwtInputField({parent: this});
 86 	this._recipientInput.setData(Dwt.KEY_OBJECT, this);
 87 	this._recipientInput.setRequired(true);
 88 	Dwt.associateElementWithObject(this._recipientInput.getInputElement(), this);
 89     this._recipientInput.setValidatorFunction(null, DwtInputField.validateEmail);
 90 
 91     var inputEl = this._recipientInput.getInputElement();
 92 	if (this._acAddrSelectList) {
 93 		this._acAddrSelectList.handle(inputEl);
 94 	}
 95 
 96     var id = this._htmlElId + "_addrListTd";
 97 	var td = document.getElementById(id);
 98 	if (td) {
 99 		this._recipientInput.reparentHtmlElement(td);
100 	}
101 };

104 /**
105  * Populates the input field before showing the dialog.
106  *
107  * @param {array}		addrs			list of AjxEmailAddress to show user
108  * @param {string}		folderId		ID of outbound folder
109  */
110 ZmSelectAddrDialog.prototype.popup =
111 function(addrs, folderId) {
112 
113 	this._addrs = [];
114     var idx = folderId.indexOf(":");
115     this._folderId = (idx != -1) ? (folderId.substr(idx + 1)) : folderId;
116 	var used = {}, id = 1;
117 	for (var i = 0, len = addrs.length; i < len; i++) {
118 		var addr = addrs[i];
119 		var email = addr.getAddress();
120 		// remove duplicates
121 		if (!used[email]) {
122 			addr.id = id++;
123             addr.setAddress(AjxEmailAddress.split(email)[0]); //just show email portion
124 			this._addrs.push(addr);
125 			used[email] = true;
126 		}
127 	}
128 	this._addrs.sort(AjxEmailAddress.sortCompareByAddress);
129     var inputValue = "";
130     for (var i=0; i<this._addrs.length; i++){
131         inputValue += this._addrs[i].getAddress() + "; ";
132     }
133 	this._recipientInput.setValue(inputValue);
134     var size = this.getSize();
135 	Dwt.setSize(this._recipientInput.getInputElement(), size.x - 87);
136     DwtDialog.prototype.popup.call(this);
137 
138     if (addrs.length == 0)
139         this._recipientInput.focus();
140 };

142 /**
143  * @private
144  */
145 ZmSelectAddrDialog.prototype._okButtonListener =
146 function() {
147 
148 
149     var error;
150     if (this._recipientInput.isValid() == null) {
151 	    error = this._recipientInput.getValue() ? AjxMsg.invalidEmailAddr : AjxMsg.valueIsRequired;
152 	}
153 
154     if (error) {
155         return this._showError(error);
156     }
157 
158     var emails = this._recipientInput.getValue();
159 
160 	if (emails) {
161         emails = AjxStringUtil.trim(emails);
162         if (emails.substring(emails.length -1) == ";")
163             emails = emails.substring(0, emails.length-1); //remove trailing ";"
164         var query = [];
165         var addr = emails.split(";");
166 		var folder = ZmFolder.QUERY_NAME[this._folderId];
167 	    for (var i = 0; i < addr.length; i++) {
168             if (AjxEmailAddress.isValid(addr[i]))
169 		        query[i] = ["tocc:(", addr[i], ")"].join("");
170             else{
171                 return this._showError(AjxMessageFormat.format(AjxMsg.invalidEmailAddrValue, addr[i]));
172             }
173 
174         }
175         var operator =  this._findEmailsCheckbox && this._findEmailsCheckbox.isSelected() ? " AND " : " OR ";
176 		appCtxt.getSearchController().search({query:"in:" + folder + " AND (" + query.join(operator) + ")"});
177 	}
178 	this.popdown();
179 };
180 
181 /**
182  * @private
183  */
184 ZmSelectAddrDialog.prototype._showError =
185 function(errorMessage) {
186 
187     var dialog = appCtxt.getErrorDialog();
188     dialog.setMessage(errorMessage);
189     dialog.popup(null, true);
190 };

192 /**
193  * Creates the address list view.
194  * @class
195  * This class represents the address list view.
196  *
197  * @param	{DwtComposite}	parent		the parent widget
198  *
199  * @extends		DwtListView
200  * 
201  * @private
202  */
203 ZmAddrListView = function(parent) {
204 
205 	var headerList = this._getHeaderList();
206 	DwtListView.call(this, {parent:parent, headerList:headerList, view:"ALV"});
207 };
208 
209 ZmAddrListView.COL_NAME	= "na";
210 ZmAddrListView.COL_ADDR	= "ad";
211 
212 ZmAddrListView.prototype = new DwtListView;
213 ZmAddrListView.prototype.constructor = ZmAddrListView;
214 
215 ZmAddrListView.prototype.toString =
216 function() {
217 	return "ZmAddrListView";
218 };
219 
220 ZmAddrListView.prototype._getHeaderList =
221 function() {
222 	return [
223 		(new DwtListHeaderItem({field:ZmAddrListView.COL_NAME, text:ZmMsg.name, width:ZmMsg.COLUMN_WIDTH_NAME_SD})),
224 		(new DwtListHeaderItem({field:ZmAddrListView.COL_ADDR, text:ZmMsg.address}))
225 	];
226 };
227 
228 ZmAddrListView.prototype._getCellContents =
229 function(html, idx, item, field, colIdx, params) {
230 	if (field == ZmAddrListView.COL_NAME) {
231 		html[idx++] = AjxStringUtil.htmlEncode(item.getName());
232 	} else if (field == ZmAddrListView.COL_ADDR) {
233 		html[idx++] = AjxStringUtil.htmlEncode(item.getAddress());
234 	}
235 
236 	return idx;
237 };
