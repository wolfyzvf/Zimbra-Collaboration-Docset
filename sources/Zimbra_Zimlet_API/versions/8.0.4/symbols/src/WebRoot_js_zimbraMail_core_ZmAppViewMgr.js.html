 21 /**
 22  * Creates a layout manager from the given components.
 23  * @class
 24  * This class performs view and layout management. It expects there to be an HTML "skin" with
 25  * containers for various components. A container is an empty DIV with a known ID, so that we
 26  * can use it to place the corresponding component's content. A component is a widget; it is
 27  * the widget's HTML element that is positioned and sized based on the container's location and
 28  * size. The containers are part of the flow (they are positioned relatively), so their location
 29  * and size should be adjusted when necessary by the browser. The components are not children of
 30  * their containers within the DOM tree; they are children of the shell, and are positioned
 31  * absolutely. There appears to be a performance gain in keeping our HTML elements closer to the
 32  * top of the DOM tree, possibly because events do not propagate as far.
 33  * 
 34  * A handful of components are positioned statically. Those are generally the ones that appear
 35  * in the top row: banner, user info, etc. The positioning style is set through skin hints.
 36  * <br/>
 37  * <br/>
 38  * The following containers/components are supported:
 39  *
 40  * <ul>
 41  *  <li>banner: displays logo</li>
 42  *  <li>user info: user name</li>
 43  *  <li>quota: quota bar</li>
 44  *  <li>search bar: a text input and a few buttons</li>
 45  *  <li>search results toolbar: search tab only</li>
 46  *  <li>app chooser: a toolbar with buttons for changing apps</li>
 47  *  <li>new button: button for creating something new (email, contact, etc)</li>
 48  *  <li>tree: area on left that usually shows overview (folders, calendars, etc)</li>
 49  *  <li>tree footer: optionally displays mini-calendar</li>
 50  *  <li>top toolbar: a view-specific toolbar</li>
 51  *  <li>app content: used to present data to the user</li>
 52  *  <li>sash: a thin moveable vertical bar for resizing tree width</li>
 53  *  <li>taskbar: mainly for IM app</li>
 54  * </ul>
 55  *
 56  * <br/>
 57  * <br/>
 58  * In general, the app view manager responds to changes in the skin by having each of the
 59  * affected components adapt to its container's new location and/or size. That means that
 60  * we are dependent on the browser to relocate and resize the containers within the skin
 61  * properly.
 62  * <br/>
 63  * <br/>
 64  * The top and bottom toolbars and the app content are treated somewhat differently: they
 65  * come under the purview of "app view management". In general, an application represents a
 66  * view with a toolbar and a content area (which is often a list view). App view management
 67  * allows these views to be pushed and popped as if they were in a stack. That way, the views
 68  * only need be constructed once each.
 69  * <br/>
 70  * <br/>
 71  * The app view components are hidden and shown using two methods: z-index and relocation. 
 72  * Since every component hangs off the shell, it must have a z-index of at least Z_VIEW
 73  * (300) to be visible. It can be hidden by setting its z-index to Z_HIDDEN (100). Since
 74  * both IE and Firefox have display bugs related to the use of z-index, we use relocation as
 75  * well: a hidden component is positioned way off the screen. (In IE, SELECT fields don't
 76  * obey z-index, and in Firefox, the cursor bleeds through.) Note: the above was true in 2005,
 77  * and we haven't rewritten the app view manager substantially since then. Some day we may just
 78  * append the elements to their parent containers within the DOM, but until then we'll do
 79  * absolute positioning.
 80  * <br/>
 81  * <br/>
 82  * A view can open in a tab (in the row of app buttons) rather than replacing the current view. Those
 83  * are handled in essentially the same way (view push and pop), but they also manage the app button.
 84  * We currently manage only a single view in a tab.
 85  *
 86  * @author Conrad Damon
 87  * 
 88  * @param {DwtShell}		shell			the outermost containing element
 89  * @param {ZmController}	controller		the app controller
 90  * @param {Boolean}			isNewWindow		if <code>true</code>, we are a child window of the main app
 91  * @param {Boolean}			hasSkin			if <code>true</code>, the app has provided containing HTML
 92  */
 93 ZmAppViewMgr = function(shell, controller, isNewWindow, hasSkin) {
 94 
 95 	ZmAppViewMgr._setContainerIds();
 96 
 97 	this._shell = shell;
 98 	this._controller = controller;
 99 	this._isNewWindow = isNewWindow;
100 	this._hasSkin = hasSkin;
101 
102 	this._shellSz = this._shell.getSize();
103 	this._shell.addControlListener(this._shellControlListener.bind(this));
104 	this._sashSupported = (window.skin && typeof window.skin.setTreeWidth == "function");
105 
106 	// history support
107 	if (appCtxt.get(ZmSetting.HISTORY_SUPPORT_ENABLED) && !isNewWindow && !AjxEnv.isPrism) {
108 		this._historyMgr = appCtxt.getHistoryMgr();
109 		this._historyMgr.addListener(this._historyChangeListener.bind(this));
110 	}
111 	this._hashViewId			= {};		// matches numeric hash to its view
112 	this._nextHashIndex			= 0;		// index for adding to browser history stack
113 	this._curHashIndex			= 0;		// index of current location in browser history stack
114 	this._noHistory				= false;	// flag to prevent history ops as result of programmatic push/pop view
115 	this._ignoreHistoryChange	= false;	// don't push/pop view as result of history.back() or history.forward()
116 
117 	this._lastViewId	= null;	// ID of previously visible view
118 	this._currentViewId	= null;	// ID of currently visible view
119 	this._hidden		= [];	// stack of views that aren't visible
120 	this._toRemove		= [];	// views to remove from hidden on next view push
121 
122 	this._view		= {};	// information about each view (components, controller, callbacks, app, etc)
123 	this._component	= {};	// component data (container, bounds, current control)
124 	this._app		= {};	// app info (current view)
125 	
126 	// reduce need for null checks
127 	this._emptyView = {component:{}, callback:{}, hide:{}};
128 	
129 	// Hashes keyed by tab ID
130 	this._viewByTabId = {};	// view for the given tab
131 	
132 	// view pre-emption
133 	this._pushCallback = this.pushView.bind(this);
134 	this._popCallback = this.popView.bind(this);
135 	
136 	// placeholder view
137 	this._createLoadingView();
138 };

222 /**
223  * Returns the requested component (widget) for the given view. The search is done
224  * in the following order:
225  * 		1. A component particular to that view
226  * 		2. A component associated with the view's app
227  * 		3. A global component	
228  * 
229  * @param {constant}	cid			component ID
230  * @param {constant}	viewId		view ID
231  */
232 ZmAppViewMgr.prototype.getViewComponent =
233 function(cid, viewId) {
234 	var view = this._view[viewId || this._currentViewId] || this._emptyView;
235 	var app = view.app || appCtxt.getCurrentAppName();
236 	var appView = this._view[app];
237 	var globalView = this._view[ZmAppViewMgr.GLOBAL];
238 	return ((view && view.component[cid]) ||
239 			(appView && appView.component[cid]) ||
240 			(globalView && globalView.component[cid]));
241 };
242 ZmAppViewMgr.prototype.getCurrentViewComponent = ZmAppViewMgr.prototype.getViewComponent;

244 // Returns the view based on the ID, handling global and app views
245 ZmAppViewMgr.prototype._getView =
246 function(viewId, app) {
247 	var view;
248 	if (viewId == ZmAppViewMgr.GLOBAL) {
249 		view = this._view[viewId] || this.createView({ viewId:viewId }); 
250 	}
251 	else if (viewId == ZmAppViewMgr.APP) {
252 		viewId = app || appCtxt.getCurrentAppName();
253 		view = this._view[viewId] || this.createView({ viewId:viewId }); 
254 	}
255 	else {
256 		view = this._view[viewId || this._currentViewId] || this.createView({ viewId:viewId }); 
257 	}
258 	return view;
259 };

261 /**
262  * Registers the given components with the app view manager, and optionally displays them.
263  *
264  * @param	{constant}	viewId		the view id
265  * @param	{hash}		components	a hash of component IDs and matching objects
266  * @param	{boolean}	show		if <code>true</code>, show the components
267  * @param	{constant}	app			name of app (for view ZmAppViewMgr.APP)
268  */
269 ZmAppViewMgr.prototype.setViewComponents =
270 function(viewId, components, show, app) {
271 
272 	DBG.println("avm", "-------------- SET components: " + AjxUtil.keys(components));
273 		
274 	// set up to add component to the appropriate map: global, app, or local
275 	var view = this._getView(viewId, app);
276 	if (!view) { return; }
277 
278 	var list = [];
279 	for (var cid in components) {
280 		var comp = components[cid];
281 		if (!comp) { continue; }
282 		if (this.isHidden(cid, viewId)) { continue; }
283 		
284 		var doShow = show && !this.isHidden(cid, this._currentViewId);
285 		if (doShow) {
286 			// if we're replacing a visible component, hide the old one
287 			var oldComp = this._component[cid] && this._component[cid].control;
288 			if (oldComp && (oldComp != comp)) {
289 				this.showComponent(cid, false, oldComp);
290 			}
291 		}
292 		
293 		view.component[cid] = comp;
294 		
295 		if (this._hasSkin) {
296 			this.getContainer(cid, comp);
297 			list.push(cid);
298 		}
299 
300 		this.displayComponent(cid, doShow);
301 
302 		// TODO: move this code
303 		if (cid == ZmAppViewMgr.C_SASH) {
304 			if (this._sashSupported){
305 				comp.registerCallback(this._appTreeSashCallback, this);
306 				if (appCtxt.get(ZmSetting.FOLDER_TREE_SASH_WIDTH)) {
307 					var newWidth =  appCtxt.get(ZmSetting.FOLDER_TREE_SASH_WIDTH);
308 					var oldWidth = skin.getTreeWidth();
309 					this._appTreeSashCallback(newWidth - oldWidth);
310 				}
311 			}
312 			comp.setCursor("default");
313 		}
314 	}
315 	if (show) {
316 		this._fitToContainer(list);
317 	}
318 };

321 /**
322  * Returns true if the given component should be hidden. Checks local, app, and then
323  * global levels. At any level, the presence of a component trumps whether it is supposed
324  * to be hidden.
325  * 
326  * @param {constant}	cid			component ID
327  * @param {constant}	viewId		view ID
328  */
329 ZmAppViewMgr.prototype.isHidden =
330 function(cid, viewId) {
331 
332 	var view = this._view[viewId || this._currentViewId] || this._emptyView;
333 	var app = view.app || appCtxt.getCurrentAppName();
334 	var appView = this._view[app];
335 	var globalView = this._view[ZmAppViewMgr.GLOBAL];
336 	
337 	if		(view && view.component[cid])				{ return false; }	// view has comp
338 	else if (view && view.hide[cid])					{ return true; }	// view says hide
339 	else if (appView && appView.component[cid])			{ return false; }	// app has comp
340 	else if (appView && appView.hide[cid])				{ return true; }	// app says hide
341 	else if (globalView && globalView.component[cid])	{ return false; }	// global comp
342 	else												{ return globalView && globalView.hide[cid]; }	// global hide
343 };

345 /**
346  * Sets whether the given components should be hidden. That setting can appear at any
347  * of three levels: global, app, or local.
348  * 
349  * @param	{constant}	viewId		the view id
350  * @param	{array}		cidList		list of component IDs
351  * @param	{boolean}	hide		if <code>true</code>, hide the components
352  * @param	{constant}	app			name of app (for view ZmAppViewMgr.APP)
353  */
354 ZmAppViewMgr.prototype.setHiddenComponents =
355 function(viewId, cidList, hide, app) {
356 
357 	cidList = AjxUtil.toArray(cidList);
358 
359 	// set up to add component to the appropriate map: global, app, or local
360 	var view = this._getView(viewId, app);
361 	if (!view) { return; }
362 
363 	for (var i = 0; i < cidList.length; i++) {
364 		view.hide[cidList[i]] = hide;
365 	}
366 };

384 /**
385  * Shows or hides the given component. It may still need to be positioned.
386  * 
387  * @param {constant}	cid		the component ID
388  * @param {boolean}		show	if true, show the component; otherwise hide it
389  * @param {DwtControl}	comp	component (optional)
390  */
391 ZmAppViewMgr.prototype.showComponent =
392 function(cid, show, comp) {
393 	
394 	comp = comp || this.getViewComponent(cid);
395 	
396 	if (comp) {
397 		DBG.println("avm", (show ? "SHOW " : "HIDE ") + cid + " / " + comp.toString() + " / " + comp._htmlElId);
398 		if (show) {
399 			comp.zShow(true);
400 		}
401 		else {
402 			if (comp.getPosition() == Dwt.ABSOLUTE_STYLE) {
403 				comp.setLocation(Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
404 			}
405 			comp.zShow(false);
406 		}
407 	}
408 };

411  * Handles several tasks needed to make sure a component is actually visible.

422 	if (doFit) {
423 		this._fitToContainer(cid);
424 	}

427 /**
428  * Returns the requested container.
429  * 
430  * @param cid
431  * @param comp
432  */
433 ZmAppViewMgr.prototype.getContainer =
434 function(cid, comp) {
435 
436 	var component = this._component[cid] = this._component[cid] || {};
437 	
438 	if (!component.container) {
439 		var contId = ZmAppViewMgr.CONT_ID_KEY[cid];
440 		var contEl = document.getElementById(contId);
441 		if (!contEl) {
442 			// skin may want to omit certain containers
443 			DBG.println(AjxDebug.DBG2, "Skin container '" + contId + "' not found.");
444 			return null;
445 		}
446 		component.container = contEl;
447 		if (comp) {
448 			contEl.innerHTML = "";
449 
450 			// if the container has bounds, fit the component to it now to prevent resize flash
451 			var bounds = this._getContainerBounds(cid);
452 			var toolbarExists = Boolean(this.getViewComponent(ZmAppViewMgr.C_TOOLBAR_TOP));
453 			if (bounds) {
454 				DBG.println("avm", "SET BOUNDS " + cid + ": " + [bounds.x, bounds.y, bounds.width, bounds.height].join("/"));
455 				comp.setBounds(bounds.x, bounds.y, bounds.width, bounds.height, toolbarExists);
456 			}
457 		}
458 	}
459 	
460 	return component.container;
461 };

539 ZmAppViewMgr.prototype.getViewsByType =

541 	var list = [];
542 	for (var viewId in this._view) {
543 		var view = this._view[viewId];
544 		if (view.type == type && (!visible || view.visible)) {
545 			list.push(view);

547 	}
548 	return list;

551 /**
552  * Registers a set of elements comprising an app view.
553  *

555  * @param	{string}		params.viewId		the view ID
556  * @param	{string}		params.viewType		the view type
557  * @param	{String}		params.appName		the name of the owning app
558  * @param	{Hash}			params.elements		a hash of elements
559  * @param	{ZmController}	params.controller	controller responsible for this view
560  * @param	{Hash}			params.callbacks 	a hash of functions to call before/after this view is shown/hidden
561  * @param	{Boolean}		params.isAppView 	if <code>true</code>, this view is an app-level view
562  * @param	{Boolean}		params.isTransient	if <code>true</code>, this view does not go on the hidden stack
563  * @param	{Hash}			params.tabParams	the tab button params; view is opened in app tab instead of being stacked
564  * @param	{Hash}			params.hide			components that aren't displayed in this view

566 ZmAppViewMgr.prototype.createView =

578 		controller:		params.controller,
579 		callback:		params.callbacks || {},

582 		isTransient:	params.isTransient,
583 		isFullScreen:	params.isFullScreen,
584 		hide:			AjxUtil.arrayAsHash(params.hide || [])
585 	};
586 
587 	if (params.appName && !this._app[params.appName]) {
588 		this._app[params.appName] = {};
589 	}
590 
591 	if (!this._isNewWindow && params.tabParams) {
592 		view.tabParams	= params.tabParams;
593 		view.isTabView = true;
594 		this._viewByTabId[params.tabParams.id] = viewId;
595 	}
596 
597 	return view;

600 /**
601  * Makes the given view visible, pushing the previously visible one to the top of the
602  * hidden stack.
603  *
604  * @param {int}		viewId		the ID of the app view to push
605  * @param {Boolean}	force		if <code>true</code>, do not run callbacks
606  *
607  * @returns	{Boolean}	<code>true</code> if the view was pushed (is now visible)
608  */
609 ZmAppViewMgr.prototype.pushView =
610 function(viewId, force) {
611 
612 	if (!viewId) { return false; }
613 	DBG.println("avm", "------- PUSH view: " + viewId);
614 	
615 	viewId = this._viewByTabId[viewId] || viewId;
616 	var view = this._view[viewId] || this._emptyView;
617 	
618 	var isPendingView = (viewId == ZmAppViewMgr.PENDING_VIEW);
619 	if (!isPendingView && !view) {
620 		// view has not been created, bail
621 		return false;
622 	}
623 
624 	if (isPendingView) {
625 		viewId = this._pendingView;
626 	}
627 	DBG.println(AjxDebug.DBG1, "pushView: " + viewId);
628 
629 	var viewController = view.controller;
630 
631 	// if same view, no need to hide previous view or check for callbacks
632 	//also no need to make the view visible, it already is.
633 	if (viewId == this._currentViewId) {
634 		// make sure the new content has focus
635 		if (viewController) {
636 			viewController._restoreFocus();
637 		}
638 		return true;
639 	}
640 
641 	DBG.println(AjxDebug.DBG2, "hidden (before): " + this._hidden);
642 
643 	if (view.isTabView) {
644 		var tp = view.tabParams;
645 		var handled = tp && tp.tabCallback && tp.tabCallback.run(this._currentViewId, viewId);
646 		if (tp && !handled) {
647 			var ac = appCtxt.getAppChooser();
648 			var button = ac.getButton(tp.id);
649 			if (!button) {
650 				button = ac.addButton(tp.id, tp);
651 				button.setHoverImage("Close");
652 			}
653 		}
654 	}
655 
656 	if (isPendingView) {
657 		DBG.println(AjxDebug.DBG1, "push of pending view: " + this._pendingView);
658 		force = true;
659 	}
660 
661 	var curView = this._view[this._currentViewId] || this._emptyView;
662 	if (!this._hideView(this._currentViewId, force || curView.isTabView, false, viewId)) {
663 		this._pendingAction = this._pushCallback;
664 		this._pendingView = viewId;
665 		return false;
666 	}
667 	this.setViewComponents(viewId, view.component);
668 
669 	var curViewController = curView.controller;
670 	var isTransient = curView.isTransient || (curViewController && curViewController.isTransient(this._currentViewId, viewId));
671 	if (this._currentViewId && (this._currentViewId != viewId) && !isTransient) {
672 		this._hidden.push(this._currentViewId);
673 	}
674 
675 	this._removeFromHidden(viewId);
676 	var temp = this._lastViewId;
677 	this._lastViewId = this._currentViewId;
678 	this._currentViewId = viewId;
679 	DBG.println(AjxDebug.DBG2, "app view mgr: current view is now " + this._currentViewId);
680 
681 	if (!this._showView(viewId, force, (viewId != this._currentViewId))) {
682 		this._currentViewId = this._lastViewId;
683 		this._lastViewId = temp;
684 		this._pendingAction = this._pushCallback;
685 		this._pendingView = viewId;
686 		return false;
687 	}
688 	DBG.println(AjxDebug.DBG2, "hidden (after): " + this._hidden);
689 
690 	// a view is being pushed - add it to browser history stack unless we're
691 	// calling this function as a result of browser Back or Forward
692 	if (this._noHistory) {
693 		DBG.println(AjxDebug.DBG2, "noHistory: push " + viewId);
694 		this._noHistory = false;
695 	} else {
696 		if (viewId != ZmId.VIEW_LOADING) {
697 			this._nextHashIndex++;
698 			this._curHashIndex = this._nextHashIndex;
699 			this._hashViewId[this._curHashIndex] = viewId;
700 			DBG.println(AjxDebug.DBG2, "adding to browser history: " + this._curHashIndex + "(" + viewId + ")");
701 			if (this._historyMgr) {
702 				this._historyMgr.add(this._curHashIndex);
703 			}
704 		}
705 	}
706 
707 	this._layout(this._currentViewId);
708 
709 	if (viewController && viewController.setCurrentViewId) {
710 		viewController.setCurrentViewId(viewId);
711 	}
712 	if (view.isAppView) {
713 		this.setAppView(view.app, viewId);
714 	}
715 	
716 	if (this._toRemove.length) {
717 		for (var i = 0; i < this._toRemove.length; i++) {
718 			this._removeFromHidden(this._toRemove[i]);
719 		}
720 		this._toRemove = [];
721 	}
722 
723 	return true;
724 };

726 /**
727  * Hides the currently visible view, and makes the view on top of the hidden stack visible.
728  *
729  * @param	{Boolean}	force	if <code>true</code>, do not run callbacks (which check if popping is OK)
730  * @param	{int}	viewId	the view ID. Only pop if this is current view
731  * @returns	{Boolean}		<code>true</code> if the view was popped
732  */
733 ZmAppViewMgr.prototype.popView =
734 function(force, viewId, skipHistory) {
735 
736 	DBG.println("avm", "------- POP view: " + viewId);
737 	
738 	viewId = this._viewByTabId[viewId] || viewId;
739 	var view = this._view[viewId] || this._emptyView;
740 
741 	if (!this._currentViewId) {
742 		DBG.println(AjxDebug.DBG1, "ERROR: no view to pop");
743 		return false;
744 	}
745 
746 	var isPendingView = (force == ZmAppViewMgr.PENDING_VIEW);
747 	if (isPendingView) {
748 		viewId = force;
749 		force = true;
750 	}
751 
752 	// a tab view is the only type of non-current view we can pop; if it is not the
753 	// current view, push it first so that callbacks etc work as expected
754 	if (viewId && !isPendingView && (this._currentViewId != viewId)) {
755 		if (view.isTabView && (this._currentViewId != viewId)) {
756 			this.pushView(viewId);
757 		}
758 		else {
759 			return false;
760 		}
761 	}
762 
763 	// handle cases where there are no views in the hidden stack (entry via deep link)
764 	var noHide = false, noShow = false;
765 	var goToApp = null;
766 	var curView = this._view[this._currentViewId] || this._emptyView;
767 	if (!this._hidden.length && !this._isNewWindow) {
768 		noHide = !curView.isTabView;
769 		noShow = true;
770 		goToApp = appCtxt.getCurrentAppName() || appCtxt.startApp;
771 	}
772 
773 	DBG.println(AjxDebug.DBG1, "popView: " + this._currentViewId);
774 	DBG.println(AjxDebug.DBG2, "hidden (before): " + this._hidden);
775 	if (!this._hideView(this._currentViewId, force, noHide)) {
776 		this._pendingAction = this._popCallback;
777 		this._pendingView = null;
778 		return false;
779 	}
780 
781 	this._deactivateView(this._currentViewId);
782 
783 	if (curView.isTabView) {
784 		appCtxt.getAppChooser().removeButton(curView.tabParams.id);
785 	}
786 	
787 	if (noShow) {
788 		if (goToApp) {
789 			this._controller.activateApp(ZmApp.MAIL);
790 		}
791 		return !noHide;
792 	}
793 
794 	this._lastViewId = this._currentViewId;
795 	this._currentViewId = this._hidden.pop();
796 
797 	// close this window if no more views exist and it's a child window
798 	if (!this._currentViewId && this._isNewWindow) {
799 		window.close();
800 		return false;
801 	}
802 
803 	DBG.println(AjxDebug.DBG2, "app view mgr: current view is now " + this._currentViewId);
804 	if (!this._showView(this._currentViewId, this._popCallback, null, force, true)) {
805 		DBG.println(AjxDebug.DBG1, "ERROR: pop with no view to show");
806 		return false;
807 	}
808 	this._removeFromHidden(this._currentViewId);
809 	DBG.println(AjxDebug.DBG2, "hidden (after): " + this._hidden);
810 	DBG.println(AjxDebug.DBG2, "hidden (" + this._hidden.length + " after pop): " + this._hidden);
811 
812 	// Move one back in the browser history stack so that we stay in sync, unless
813 	// we're calling this function as a result of browser Back
814 	if (this._historyMgr && !skipHistory) {
815 		if (this._noHistory) {
816 			DBG.println(AjxDebug.DBG2, "noHistory (pop)");
817 			this._noHistory = false;
818 		} else {
819 			this._ignoreHistoryChange = true;
820 			history.back();
821 		}
822 	}
823 
824 	this._layout(this._currentViewId);
825 
826 	return true;
827 };

886 /**
887 * Shows the view that was waiting for return from a popped view's callback. Typically, the
888 * popped view's callback will have put up some sort of dialog, and this function would be
889 * called by a listener on a dialog button.
890 *
891 * @param {Boolean}	show		if <code>true</code>, show the pending view
892 */
893 ZmAppViewMgr.prototype.showPendingView =
894 function(show) {
895 	if (show && this._pendingAction) {
896 		this._pendingAction.run(ZmAppViewMgr.PENDING_VIEW);
897 	}
898 
899 	// If a pop shield has been dismissed and we're not going to show the
900 	// pending view, and we got here via press of browser Back/Forward button,
901 	// then undo that button press so that the browser history is correct.
902 	if (!show) {
903 		if (this._browserAction == ZmAppViewMgr.BROWSER_BACK) {
904 			this._ignoreHistoryChange = true;
905 			history.forward();
906 		} else if (this._browserAction == ZmAppViewMgr.BROWSER_FORWARD) {
907 			this._ignoreHistoryChange = true;
908 			history.back();
909 		}
910 		this._browserAction = null;
911 	}
912 	this._pendingAction = this._pendingView = null;
913 };

993  * Locates and sizes the given list of components to fit within their containers.
994  * 
995  * @private
996  */
997 ZmAppViewMgr.prototype._fitToContainer =
998 function(cidList, isIeTimerHack) {
999 	
1000 	var cidList = AjxUtil.toArray(cidList);
1001 
1002 	for (var i = 0; i < cidList.length; i++) {
1003 		var cid = cidList[i];
1004 		if (!isIeTimerHack && AjxEnv.isIE && (cid == ZmAppViewMgr.C_TASKBAR)) {
1005 			// Hack for bug 36924: ie bar is in the middle of the screen when resizing ie.
1006 			if (!this._ieHackAction) {
1007 				this._ieHackAction = new AjxTimedAction(this, this._fitToContainer, [[ZmAppViewMgr.C_TASKBAR], true]);
1008 			}
1009 			AjxTimedAction.scheduleAction(this._ieHackAction, 1);
1010 		}
1011 
1012 		DBG.println(AjxDebug.DBG3, "fitting to container: " + cid);
1013 		var cont = this.getContainer(cid);
1014 		if (cont) {
1015 			var comp = this.getViewComponent(cid);
1016 			if (comp && (comp.getZIndex() != Dwt.Z_HIDDEN)) {
1017 				var position = this._getComponentPosition(cid);
1018 				var isStatic = (position == Dwt.STATIC_STYLE);
1019 				
1020 				// reset position if skin overrides default of absolute
1021 				var compEl = comp.getHtmlElement();
1022 				if (position) {
1023 					compEl.style.position = position;
1024 				}
1025 
1026 				var component = this._component[cid];
1027 				if (isStatic) {
1028 					if (compEl.parentNode != cont) {
1029 						DBG.println("avm", "APPEND " + cid);
1030 						cont.appendChild(compEl);
1031 					}
1032 					if (comp.adjustSize) {
1033 						comp.adjustSize();
1034 					}
1035 				} else {
1036 					var contBds = Dwt.getBounds(cont);
1037 					// take insets (border + padding) into account
1038 					var insets = Dwt.getInsets(cont);
1039 					Dwt.insetBounds(contBds, insets);
1040 					
1041 					// save bounds
1042 					component.bounds = contBds;
1043 					var toolbarExists = Boolean(this._component[ZmAppViewMgr.C_TOOLBAR_TOP].control);
1044 					DBG.println("avm", "FIT " + cid + ": " + [contBds.x, contBds.y, contBds.width, contBds.height].join("/"));
1045 					comp.setBounds(contBds.x, contBds.y, contBds.width, contBds.height, toolbarExists);
1046 				}
1047 				component.control = comp;
1048 			}
1049 		}
1050 	}
1051 
1052 	if (window.DBG && DBG.getDebugLevel() >= AjxDebug.DBG2) {
1053 		this._debugShowMetrics(cidList);
1054 	}
1055 };

1108 /**
1109  * Tries to hide the given view. First checks to see if the view has a callback
1110  * for when it is hidden. The callback must return true for the view to be hidden.
1111  * 
1112  * @private
1113  */
1114 ZmAppViewMgr.prototype._hideView =
1115 function(viewId, force, noHide, newViewId) {
1116 
1117 	if (!viewId) { return true; }
1118 
1119 	var view = this._view[viewId] || this._emptyView;
1120 	var okToContinue = true;
1121 	var callback = view.callback[ZmAppViewMgr.CB_PRE_HIDE];
1122 	if (callback) {
1123 		DBG.println(AjxDebug.DBG2, "hiding " + viewId);
1124 		okToContinue = callback.run(viewId, force, newViewId);
1125 	}
1126 	if (okToContinue) {
1127 		if (!noHide) {
1128 			this._setViewVisible(viewId, false);
1129 		}
1130         if (appCtxt.get(ZmSetting.USE_KEYBOARD_SHORTCUTS)) {
1131 		    appCtxt.getKeyboardMgr().clearKeySeq();
1132         }

1327 /**
1328  * Handles browser Back/Forward. We compare the new index to our current one
1329  * to see if the user has gone back or forward. If back, we pop the view,
1330  * otherwise we push the appropriate view.
1331  * 
1332  * @param {AjxEvent}	ev	the history event
1333  * 
1334  * @private
1335  */
1336 ZmAppViewMgr.prototype._historyChangeListener =
1337 function(ev) {
1338 	if (appCtxt.inStartup) { return; }
1339 	if (!(ev && ev.data)) { return; }
1340 	if (this._ignoreHistoryChange) {
1341 		this._ignoreHistoryChange = false;
1342 		return;
1343 	}
1344 
1345 	var dlg = DwtBaseDialog.getActiveDialog();
1346 	if (dlg && dlg.isPoppedUp()) {
1347 		dlg.popdown();
1348 	}
1349 
1350 	var hashIndex = parseInt(ev.data);
1351 	this._noHistory = true;
1352 	var viewId = this._hashViewId[hashIndex];
1353 	if (hashIndex == (this._curHashIndex - 1)) {
1354 		// Back button has been pressed
1355 		this._browserAction = ZmAppViewMgr.BROWSER_BACK;
1356 		this.popView();
1357 	} else if (hashIndex == (this._curHashIndex + 1)) {
1358 		// Forward button has been pressed
1359 		this._browserAction = ZmAppViewMgr.BROWSER_FORWARD;
1360 		this.pushView(viewId);
1361 	} else {
1362 		// Not sure where we are - just push the correct view
1363 		this._browserAction = null;
1364 		this.pushView(viewId);
1365 	}
1366 	this._curHashIndex = hashIndex;
1367 
1368 	DBG.println(AjxDebug.DBG2, "History change to " + hashIndex + ", new view: " + viewId);
1369 };

1371 /**
1372  * Handles app/tree movement. If you move the sash beyond the max or min width, pins to the respective width.
1373  * 
1374  * @private
1375  */
1376 ZmAppViewMgr.prototype._appTreeSashCallback =
1377 function(delta) {
1378 	if (!window.skin) { return; }
1379 
1380 	// ask skin for width of tree, rather than hard-coding name of tree div here
1381 	var currentWidth = skin.getTreeWidth();
1382 	if (currentWidth === null) { return 0; }
1383 
1384 	DBG.println(AjxDebug.DBG3, "************ sash callback **************");
1385 	DBG.println(AjxDebug.DBG3, "delta = " + delta);
1386 	DBG.println(AjxDebug.DBG3, "shell width = " + this._shellSz.x);
1387 	DBG.println(AjxDebug.DBG3, "current width = " + currentWidth);
1388 
1389 	// MOW: get the min/max sizes from the skin.hints
1390 	if (!this.treeMinSize) {
1391 		this.treeMinSize = window.skin.hints.tree.minWidth || 150;
1392 		this.treeMaxSize = window.skin.hints.tree.maxWidth || 300;
1393 	}
1394 
1395 	// pin the resize to the minimum and maximum allowable
1396 	if (currentWidth + delta > this.treeMaxSize) {
1397 		delta = Math.max(0, this.treeMaxSize - currentWidth);
1398 	}
1399 	if (currentWidth + delta < this.treeMinSize) {
1400 		delta = Math.min(0, this.treeMinSize - currentWidth);
1401 	}
1402 
1403 	// tell skin to resize the tree to keep the separation of tree/skin clean
1404 	var newTreeWidth = currentWidth + delta;
1405 
1406 	skin.setTreeWidth(newTreeWidth);
1407 
1408 	// call fitAll() on timeout, so we dont get into a problem w/ sash movement code
1409 	var me = this;
1410 	setTimeout(function(){me.fitAll()},0);
1411 	return delta;
1412 };
