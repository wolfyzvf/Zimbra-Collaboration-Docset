 57 ZmApptController.prototype._createToolBar =
 58 function() {
 59 
 60 	var buttons = [ ZmOperation.SEND_INVITE, ZmOperation.SAVE, ZmOperation.CANCEL, ZmOperation.SEP,
 61                     ZmOperation.TAG_MENU
 62                     ];
 63     var secondaryButtons = [ZmOperation.EDIT, ZmOperation.DUPLICATE_APPT, ZmOperation.SEP,
 64                             ZmOperation.REPLY, ZmOperation.REPLY_ALL, ZmOperation.FORWARD_APPT, ZmOperation.PROPOSE_NEW_TIME, ZmOperation.DELETE, ZmOperation.SEP,
 65                             ZmOperation.SHOW_ORIG
 66                             ];
 67     if (appCtxt.get(ZmSetting.PRINT_ENABLED)) {
 68 		buttons.push(ZmOperation.PRINT);
 69 	}
 70 
 71 	this._toolbar = new ZmButtonToolBar({parent:this._container, buttons:buttons, context:this._currentViewId, controller:this, secondaryButtons:secondaryButtons});
 72 	this._toolbar.addSelectionListener(ZmOperation.SAVE, this._saveListener.bind(this));
 73 	this._toolbar.addSelectionListener(ZmOperation.CANCEL, this._cancelListener.bind(this));
 74 	this._toolbar.addSelectionListener(ZmOperation.REPLY, this._replyListener.bind(this));
 75 	this._toolbar.addSelectionListener(ZmOperation.REPLY_ALL, this._replyAllListener.bind(this));
 76 	this._toolbar.addSelectionListener(ZmOperation.FORWARD_APPT, this._forwardListener.bind(this));
 77 	this._toolbar.addSelectionListener(ZmOperation.EDIT, this._editListener.bind(this));
 78 	this._toolbar.addSelectionListener(ZmOperation.PROPOSE_NEW_TIME, this._proposeTimeListener.bind(this));
 79 	this._toolbar.addSelectionListener(ZmOperation.DELETE, this._deleteListener.bind(this));
 80 	this._toolbar.addSelectionListener(ZmOperation.DUPLICATE_APPT, this._duplicateApptListener.bind(this));
 81 	this._toolbar.addSelectionListener(ZmOperation.SHOW_ORIG, this._showOrigListener.bind(this));
 82 
 83 	if (appCtxt.get(ZmSetting.PRINT_ENABLED)) {
 84 		this._toolbar.addSelectionListener(ZmOperation.PRINT, this._printListener.bind(this));
 85 	}
 86 
 87     var sendButton = this._toolbar.getButton(ZmOperation.SEND_INVITE);
 88     sendButton.setVisible(false);
 89 
 90 
 91     var tagButton = this._toolbar.getButton(ZmOperation.TAG_MENU);
 92 	if (tagButton) {
 93 		tagButton.noMenuBar = true;
 94 		this._setupTagMenu(this._toolbar);
 95 	}
 96 	// change default button style to toggle for spell check button
 97 	var spellCheckButton = this._toolbar.getButton(ZmOperation.SPELL_CHECK);
 98 	if (spellCheckButton) {
 99 		spellCheckButton.setAlign(DwtLabel.IMAGE_LEFT | DwtButton.TOGGLE_STYLE);
100 	}
101 
102 };

104 ZmApptController.prototype._initToolbar =
105 function(mode) {
106     ZmCalItemComposeController.prototype._initToolbar.call(this, mode);
107     var saveButton = this._toolbar.getButton(ZmOperation.SAVE);
108     saveButton.setEnabled(false);
109 
110     var editButton = this._toolbar.getButton(ZmOperation.EDIT),
111         forwardApptButton,
112         deleteButton;
113 
114     if (editButton) {
115         if (mode === ZmCalItem.MODE_EDIT_SINGLE_INSTANCE) {
116             if (editButton.getText() !== ZmMsg.editInstance) {
117                 editButton.setText(ZmMsg.editInstance);
118                 forwardApptButton = this._toolbar.getButton(ZmOperation.FORWARD_APPT);
119                 deleteButton = this._toolbar.getButton(ZmOperation.DELETE);
120                 forwardApptButton && forwardApptButton.setText(ZmMsg.forwardInstance);
121                 deleteButton && deleteButton.setText(ZmMsg.deleteApptInstance);
122             }
123         }
124         else if (mode === ZmCalItem.MODE_EDIT_SERIES) {
125             if (editButton.getText() !== ZmMsg.editSeries) {
126                 editButton.setText(ZmMsg.editSeries);
127                 forwardApptButton = this._toolbar.getButton(ZmOperation.FORWARD_APPT);
128                 deleteButton = this._toolbar.getButton(ZmOperation.DELETE);
129                 forwardApptButton && forwardApptButton.setText(ZmMsg.forwardSeries);
130                 deleteButton && deleteButton.setText(ZmMsg.deleteApptSeries);
131             }
132         }
133         else {
134             if (editButton.getText() !== ZmMsg.edit) {
135                 editButton.setText(ZmMsg.edit);
136                 forwardApptButton = this._toolbar.getButton(ZmOperation.FORWARD_APPT);
137                 deleteButton = this._toolbar.getButton(ZmOperation.DELETE);
138                 forwardApptButton && forwardApptButton.setText(ZmMsg.forward);
139                 deleteButton && deleteButton.setText(ZmMsg.del);
140             }
141         }
142     }
143 
144     // bug 68451: disabling edit options for trashed appointments
145     var calItem = this.getCalItem();
146     var calendar = calItem && calItem.getFolder();
147     var isTrash = calendar && calendar.nId==ZmOrganizer.ID_TRASH;
148     var isReadOnly = calendar && calendar.isReadOnly();
149 
150     if(isTrash){
151         this._disableEditForTrashedItems();
152     }
153 
154     if (isReadOnly) {
155         this._disableActionsForReadOnlyAppt();
156     }
157     if (appCtxt.isExternalAccount()) {
158         this._disableActionsForExternalAccount();
159     }
160 };

217 ZmApptController.prototype._deleteListener =
218 function(ev) {
219 	var op = this.getMode();
220     var calItem = this.getCalItem();
221     if(!calItem) {
222         return;
223     }
224 
225     if (calItem.isRecurring()) {
226         var mode = (op == ZmCalItem.MODE_EDIT_SINGLE_INSTANCE)
227             ? ZmCalItem.MODE_DELETE_INSTANCE
228             : ZmCalItem.MODE_DELETE_SERIES;
229         this._app.getCalController()._promptDeleteAppt(calItem, mode);
230     }
231     else {
232         this._app.getCalController()._deleteAppointment(calItem);
233     }
234 };

269 ZmApptController.prototype._saveListener =
270 function(ev) {
271     if(!this.isDirty() || !this.getOpValue()) {
272         return;
273     }
274 	var op = (ev && ev.item instanceof DwtMenuItem)
275 		? ev.item.getData(ZmOperation.KEY_ID) : null;
276 
277 
278     var calItem = this.getCalItem();
279     if(!calItem) {
280         return;
281     }
282     if (calItem.isRecurring() && !op) {
283         var mode = this.getMode();
284         op = (mode == ZmCalItem.MODE_EDIT_SINGLE_INSTANCE) ? ZmOperation.VIEW_APPT_INSTANCE : ZmOperation.VIEW_APPT_SERIES;
285     }
286 
287     var saveCallback = new AjxCallback(this, this._handleSaveResponse);
288     var calViewCtrl = this._app.getCalController();
289     var respCallback = new AjxCallback(calViewCtrl, calViewCtrl._handleResponseHandleApptRespondAction, [calItem, this.getOpValue(), op, saveCallback]);
290 	calItem.getDetails(null, respCallback, this._errorCallback);
291 
292     //this._app.getCalController()._replyAppointment(calItem, true);
293 };

296 function(ev) {

301 	calViewCtrl.duplicateAppt(appt, {isException: isException});

326 ZmApptController.prototype._forwardListener =
327 function(ev) {
328 	var op = this.getMode();
329     var calItem = this.getCalItem();
330     if(!calItem) {
331         return;
332     }
333 
334     var mode = ZmCalItem.MODE_FORWARD;
335     if (calItem.isRecurring()) {
336 		mode = (op == ZmCalItem.MODE_EDIT_SINGLE_INSTANCE)
337 			? ZmCalItem.MODE_FORWARD_SINGLE_INSTANCE
338 			: ZmCalItem.MODE_FORWARD_SERIES;
339 	}
340 
341     this._app.getCalController()._forwardAppointment(calItem, mode);
342 };

344 ZmApptController.prototype._printListener =
345 function() {
346 	var calItem = this.getCalItem();
347     if(!calItem) {
348         return;
349     }
350 	var url = ["/h/printappointments?id=", calItem.invId, "&tz=", AjxTimezone.getServerId(AjxTimezone.DEFAULT)]; //bug:53493
351     if (appCtxt.isOffline) {
352         url.push("&zd=true", "&acct=", this._composeView.getApptEditView().getCalendarAccount().name);
353     }
354 	window.open(appContextPath + url.join(""), "_blank");
355 };

365 ZmApptController.prototype._setupTagMenu =
366 function(parent) {
367 	if (!parent) return;
368 	var tagMenu = parent.getTagMenu();
369 	if (tagMenu) {
370 		tagMenu.addSelectionListener(new AjxListener(this, this._tagListener));
371 	}
372 	if (parent instanceof ZmButtonToolBar) {
373 		var tagButton = parent.getOp(ZmOperation.TAG_MENU);
374 		if (tagButton) {
375 			tagButton.addDropDownSelectionListener(new AjxListener(this, this._tagButtonListener));
376 		}
377 	}

380 ZmApptController.prototype._proposeTimeListener =
381 function(ev) {
382 	var calItem = this.getCalItem();
383     if(!calItem) {
384         return;
385     }
386     //Pass mode edit to open the appt in edit mode. The mode 'propose new time' will be added later.
387 	var mode = ZmCalItem.MODE_EDIT;
388 	if (calItem.isRecurring()) {
389 		mode = this.getMode();
390 	}
391 	var appt = calItem;
392 	var clone = ZmAppt.quickClone(appt);
393 	clone.setProposeTimeMode(true);
394 	clone.getDetails(mode, new AjxCallback(this, this._proposeTimeContinue, [clone, mode]));
395 };

400 	AjxDispatcher.run("GetApptComposeController").proposeNewTime(appt);
401 };

403 ZmApptController.prototype._doTag =
404 function(items, tag, doTag) {
405 
406 	var list = this._getTaggableItems(items);
407 
408 	if (doTag) {
409 		if (list.length > 0 && list.length == items.length) {
410 			// there are items to tag, and all are taggable
411 			ZmBaseController.prototype._doTag.call(this, list, tag, doTag);
412 		} else {
413 			var msg;
414 			var dlg = appCtxt.getMsgDialog();
415 			if (list.length > 0 && list.length < items.length) {
416 				// there are taggable and nontaggable items
417 				var listener = new AjxListener(this, this._handleDoTag, [dlg, list, tag, doTag]);
418 				dlg.setButtonListener(DwtDialog.OK_BUTTON, listener);
419 				msg = ZmMsg.tagReadonly;
420 			} else if (list.length == 0) {
421 				// no taggable items
422 				msg = ZmMsg.nothingToTag;
423 			}
424 			dlg.setMessage(msg);
425 			dlg.popup();
426 		}
427 	} else if (list.length > 0) {
428 		ZmBaseController.prototype._doTag.call(this, list, tag, doTag);
429 	}
430 };

497 ZmApptController.prototype.saveCalItem =
498 function(attId) {
499     var done = true;
500     if (this.isDirty()) {
501         var calItem = this.getCalItem();
502         if(calItem) {
503             var saveCallback = new AjxCallback(this, this._handleSaveResponse);
504             var calViewCtrl = this._app.getCalController();
505             var respCallback =
506                 new AjxCallback(calViewCtrl, calViewCtrl._handleResponseHandleApptRespondAction,
507                     [calItem, this.getOpValue(), null, saveCallback]);
508             calItem.getDetails(null, respCallback, this._errorCallback);
509             done = false;
510         }
511     }
512     if(done && this.isCloseAction()) {
513         this._closeView();
514     }
515 
516 };
