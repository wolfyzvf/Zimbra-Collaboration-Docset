 16 /**
 17  * Creates an iframe.
 18  * @class
 19  * This class represents a simple event proxy. Creates an IFRAME, inserts the given html into it and
 20  * forwards any events to the parent widget, translating mouse coordinates in
 21  * between.
 22  *
 23  * @param {hash}	params		a hash of parameters
 24  * @param	{DwtComposite}		params.parent		the parent 
 25  * @param {string}	params.html 	the HTML code to be inserted in the IFRAME.  There will be
 26  *   slight modifications to it (i.e. the margins and paddings of the HTML
 27  *   element will be set to 0, also any margins for BODY).
 28  * @param {boolean}		[params.noscroll=false] 	if <code>true</code>, do not show the scroll bars
 29  * @param {constant}	params.posStyle		the position style (see {@link DwtControl})
 30  * @param {AjxCallback}	params.processHtmlCallback		the callback that will be called
 31  *   immediately after the HTML code was inserted. A reference to the document object will be passed
 32  * @param {boolean}	params.useKbMgmt	if <code>true</code>, participate in keyboard management
 33  * 
 34  * @author Mihai Bazon
 35  * 
 36  * @extends		DwtControl
 37  */
 38 DwtIframe = function(params) {
 39 	params.posStyle = params.posStyle || DwtControl.STATIC_STYLE;
 40 	params.className = params.className || "DwtIframe";
 41 	DwtControl.call(this, params);
 42 	this._styles = params.styles;
 43 	this._noscroll = params.noscroll;
 44 	this._iframeID = params.id ? DwtId.getIframeId(params.id) : Dwt.getNextId();
 45 	this._onLoadHandler = params.onload;	
 46 	this._processHtmlCallback = params.processHtmlCallback;
 47 	this._hidden = params.hidden;
 48 	if (!this._createFrame(params.html)) {
 49 		this.initFailed = true;
 50 		return;	// this object is still returned
 51 	}
 52 	
 53 	if (params.useKbMgmt) {
 54 		var iframe = this.getIframe();
 55 		var idoc = Dwt.getIframeDoc(iframe);
 56 		var doc = AjxEnv.isIE ? idoc : iframe.contentWindow;
 57 		Dwt.setHandler(doc, DwtEvent.ONKEYDOWN, DwtKeyboardMgr.__keyDownHdlr);
 58 		Dwt.setHandler(doc, DwtEvent.ONKEYUP, DwtKeyboardMgr.__keyUpHdlr);
 59 		Dwt.setHandler(doc, DwtEvent.ONKEYPRESS, DwtKeyboardMgr.__keyPressHdlr);
 60 	}
 61 };

 63 DwtIframe.prototype = new DwtControl;
 64 DwtIframe.prototype.constructor = DwtIframe;

 69 /**
 70  * Gets the iframe.
 71  * 
 72  * @return	{Element}	the iframe
 73  */
 74 DwtIframe.prototype.getIframe = function() {
 75 	return document.getElementById(this._iframeID);
 76 };

 87 /// Forwards events to the parent widget
 88 DwtIframe.prototype._rawEventHandler = function(ev) {
 89 	var iframe = this.getIframe();
 90 	var win = iframe.contentWindow;
 91 	if (AjxEnv.isIE) {
 92 		ev = win.event;
 93 	}
 94 
 95 	var dw;
 96 	// This probably sucks.
 97 	if (/mouse|context|click|select/i.test(ev.type)) {
 98 		dw = new DwtMouseEvent(true);
 99 	}
100 	else {
101 		dw = new DwtUiEvent(true);
102 	}
103 	dw.setFromDhtmlEvent(ev);
104 
105 	// Notify since the manager doesn't know about events in the iframe's document
106 	if (ev.type == "mousedown" || ev.type == "mousewheel") {
107 		DwtOutsideMouseEventMgr.forwardEvent(ev);
108 	}
109 
110 	// HACK! who would have know.. :-(
111 	// perhaps we need a proper mapping
112 	var type = dw.type.toLowerCase();
113 	if (!/^on/.test(type)) {
114 		type = "on" + type;
115 	}
116 	// translate event coordinates
117 	var pos = this.getLocation();
118 
119 	// What I can tell for sure is that we don't want the code below for IE
120 	// and we want it for Gecko, but I can't be sure of other browsers..
121 	// Let's assume they follow Gecko.  Seems mostly a trial and error
122 	// process :(
123 	if (!AjxEnv.isIE) {
124 		var doc = win.document;
125 		var sl = doc.documentElement.scrollLeft || ( doc.body ? doc.body.scrollLeft : 0); 
126 		var st = doc.documentElement.scrollTop || ( doc.body ? doc.body.scrollTop : 0 );
127 		pos.x -= sl;
128 		pos.y -= st;
129 	}
130 
131 	dw.docX += pos.x;
132 	dw.docY += pos.y;
133 	dw.elementX += pos.x;
134 	dw.elementY += pos.y;
135 
136 //   	window.status = dw.type + " doc(" + dw.docX + ", " + dw.docY + ") " +
137 //   		" element(" + dw.elementX + ", " + dw.elementY + ") " +
138 //  		" stopPropagation: " + dw._stopPropagation + ", " +
139 //  		" returnValue: " + dw._returnValue;
140 
141 	var capture = DwtMouseEventCapture.getCaptureObj();
142 	capture = capture && dw.button != DwtMouseEvent.RIGHT; // ignore capture if it's right-click
143 	if (AjxEnv.isIE || !capture) {
144 		// go for Dwt events
145 		DwtEventManager.notifyListeners(type, dw);
146 		this.parent.notifyListeners(type, dw);
147 	} else {
148 		// Satisfy object that holds the mouse capture.
149 
150 		// the following is DOM2, not supported by IE
151 		var fake = document.createEvent("MouseEvents");
152 		fake.initMouseEvent(ev.type,
153 				    true, // can bubble
154 				    true, // cancellable
155 				    document.defaultView, // the view
156 				    0, // event detail ("click count")
157 				    ev.screenX, // screen X
158 				    ev.screenY, // screen Y
159 				    dw.docX, // clientX, but translated to page
160 				    dw.docY, // clientY, translated
161 				    ev.ctrlKey, // key status...
162 				    ev.altKey,
163 				    ev.shiftKey,
164 				    ev.metaKey,
165 				    ev.button,
166 				    ev.relatedTarget);
167 		document.body.dispatchEvent(fake);
168 		// capture[DwtIframe._captureEvents[dw.type]](fake);
169 	}
170 
171 	dw.setToDhtmlEvent(ev);
172 	return dw._returnValue;
173 };
174 
175 // map event names to the handler name in a DwtMouseEventCapture object
176 // DwtIframe._captureEvents = { mousedown : "_mouseDownHdlr",
177 // 			     mousemove : "_mouseMoveHdlr",
178 // 			     mouseout  : "_mouseOutHdlr",
179 // 			     mouseover : "_mouseOverHdlr",
180 // 			     mouseup   : "_mouseUpHdlr" };

183 			     DwtEvent.ONCLICK,
184 			     DwtEvent.ONCONTEXTMENU,
185 			     DwtEvent.ONDBLCLICK,
186 			     DwtEvent.ONFOCUS,
187 			     DwtEvent.ONKEYDOWN,
188 			     DwtEvent.ONKEYPRESS,
189 			     DwtEvent.ONKEYUP,
190 			     DwtEvent.ONMOUSEDOWN,
191 			     DwtEvent.ONMOUSEENTER,
192 			     DwtEvent.ONMOUSELEAVE,
193 			     DwtEvent.ONMOUSEMOVE,
194 			     DwtEvent.ONMOUSEOUT,
195 			     DwtEvent.ONMOUSEOVER,
196 			     DwtEvent.ONMOUSEUP,
197 			     DwtEvent.ONSELECTSTART ];

199 DwtIframe.prototype._createFrame = function(html) {
200 	var myId = this.getHTMLElId();
201 
202 	// this is an inner function so that we can access the object (self).
203 	// it shouldn't create a memory leak since it doesn't directly "see"
204 	// the iframe variable (it's protected below)
205 	function rawHandlerProxy(ev) {
206 		var myElement = document.getElementById(myId);
207 		var self = DwtControl.findControl(myElement);		
208 		return self._rawEventHandler(ev);
209 	};
210 
211 	// closure: protect the reference to the iframe node here.
212 	return (function() {
213 		var iframe, tmp = [], i = 0, idoc;
214 		var myElement = document.getElementById(myId);
215 		var self = DwtControl.findControl(myElement);
216 
217 		if (!self) {
218 			return false;
219 		}
220 		
221 		tmp[i++] = "<iframe";
222 		if (self._noscroll) {
223 			tmp[i++] = " scrolling='no'";
224 		}
225 		if (self._hidden) {
226 			tmp[i++] = " style='visibility:hidden'";
227 		}
228 		tmp[i++] = " frameborder='0' width='100%' id='";
229 		tmp[i++] = self._iframeID;
230 		tmp[i++] = "' name='"+ self._iframeID + "'";
231 		if(self._onLoadHandler){
232 			tmp[i++] = " onload='" + self._onLoadHandler + "'";
233 		}
234 		tmp[i++] = " src='javascript:\"\";' ></iframe>";
235 		self.setContent(tmp.join(''));
236 
237 		// Bug 7523: @import url() lines will make Gecko report
238 		// document.body is undefined until "onload" (unacceptable) so
239 		// we drop these lines now.
240 		html = html.replace(/(<style[^>]*>)[\s\t\u00A0]*((.|\n)*?)[\s\t\u00A0]*<\x2fstyle>/mgi,
241 				    function(s, p1, p2) {
242 					    return p1 + p2.replace(/@import.*?(;|[\s\t\u00A0]*$)/gi, "") + "</style>";
243 				    });
244         html = html.replace(AjxStringUtil.IMG_SRC_CID_REGEX, '<img pnsrc="cid:');
245 		iframe = self.getIframe();
246 		idoc = Dwt.getIframeDoc(iframe);
247 		idoc.open();
248 		if (self._styles) {
249 			idoc.write([ "<style type='text/css'>", self._styles, "</style>" ].join(""));
250 		}
251 		idoc.write(html);
252 		idoc.close();
253 		// if we're not giving a break, we can safely do any postprocessing
254 		// here.  I.e. if we want to drop backgroundImage-s, it's safe to do it
255 		// here because the browser won't have a chance to load them.
256 		if (self._processHtmlCallback) {
257 			self._processHtmlCallback.run(idoc);
258 		}
259 
260 		// if we have margins, the translated coordinates won't be OK.
261 		// it's best to remove them.  THE way to have some spacing is
262 		// to set padding on the body element.
263 		tmp = idoc.documentElement.style;
264 		tmp.margin = tmp.padding = "0";
265 		if (idoc.body) {
266 			idoc.body.style.margin = "0";
267 		}
268 
269 		// assign event handlers
270 		tmp = DwtIframe._forwardEvents;
271 		if (!AjxEnv.isIE) {
272 			idoc = iframe.contentWindow;
273 		}
274 		for (i = tmp.length; --i >= 0;) {
275 			idoc[tmp[i]] = rawHandlerProxy;
276 		}
277 
278 		// catch browser context menus
279 		// idoc[DwtEvent.ONCONTEXTMENU] = DwtShell._preventDefaultPrt;
280 		
281 		return true;
282 	})();
283 };

285 DwtIframe.prototype._resetEventHandlers = function() {
286 	var self = this;
287 
288 	// this is an inner function so that we can access the object (self).
289 	// it shouldn't create a memory leak since it doesn't directly "see"
290 	// the iframe variable (it's protected below)
291 	function rawHandlerProxy(ev) { return self._rawEventHandler(ev); };
292 
293 	// closure: protect the reference to the iframe node here.
294 	(function() {
295 		var iframe, tmp = [], i = 0, idoc;
296 		iframe = self.getIframe();
297 		idoc = Dwt.getIframeDoc(iframe);
298 
299 		// assign event handlers
300 		tmp = DwtIframe._forwardEvents;
301 		if (!AjxEnv.isIE)
302 			idoc = iframe.contentWindow;
303 			
304 			
305 		for (i = tmp.length; --i >= 0;){
306 			idoc[tmp[i]] = rawHandlerProxy;
307 		}
308 
309 		// catch browser context menus
310 		// idoc[DwtEvent.ONCONTEXTMENU] = DwtShell._preventDefaultPrt;
311 	})();
312 };

314 DwtIframe.prototype.setSrc =
315 function(src){
316 
317     src = src || 'javascript:\"\";'
318     var iframe = this.getIframe();
319     iframe.src = src;
320 };

322 DwtIframe.prototype.setIframeContent =

324     var iDoc = this.getDocument();
