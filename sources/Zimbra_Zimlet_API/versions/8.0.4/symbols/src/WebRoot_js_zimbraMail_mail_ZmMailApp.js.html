 21 /**
 22  * Creates and initializes the mail application.
 23  * @constructor
 24  * @class
 25  * The mail app manages and displays mail messages. Messages may be grouped
 26  * into conversations. New messages are created through a composer.
 27  *
 28  * @param	{DwtControl}	container		the container
 29  * @param	{ZmController}	parentController	the parent window controller (set by the child window)
 30  * 
 31  * @author Conrad Damon
 32  * 
 33  * @extends		ZmApp
 34  */
 35 ZmMailApp = function(container, parentController) {
 36 	ZmApp.call(this, ZmApp.MAIL, container, parentController);
 37 
 38 	this._dataSourceCollection	= {};
 39 	this._identityCollection	= {};
 40 	this._signatureCollection	= {};
 41 
 42 	this.numEntries				= 0; // offline, initial sync
 43 	this.globalMailCount		= 0; // offline, new mail count
 44 
 45 	this._throttleStats = [];
 46 	this._addSettingsChangeListeners();
 47     AjxCore.addOnloadListener(this._checkVacationReplyEnabled.bind(this));
 48 };

1007 /**
1008  * Normalize the notifications that occur when a virtual conv gets promoted to a real conv.
1009  * For example, a virtual conv with ID -676 and one msg (ID 676) receives a second msg (ID 677)
1010  * and becomes a real conv with an ID of 678. The following notifications will arrive:
1011  *
1012  *		deleted:	-676
1013  *		created:	c {id:678, n:2}
1014  *					m {id:677, cid:678}
1015  *		modified:	m {id:676, cid:678}
1016  *
1017  * Essentially, we want to handle this as:
1018  *
1019  * 		created:	m {id:677, cid:678}
1020  *		modified:	c {id:-676, _newId: 678}
1021  * 					m {id:676, cid:678}
1022  *
1023  * @private
1024  */
1025 ZmMailApp.prototype.preNotify =
1026 function(notify) {
1027 
1028 	if (!(notify.deleted && notify.created && notify.modified))	{ return notify; }
1029 
1030 	// first, see if we are deleting any virtual convs (which have negative IDs)
1031 	var virtConvDeleted = false;
1032 	var deletedIds = notify.deleted.id && notify.deleted.id.split(",");
1033 	var virtConv = {};
1034 	var newDeletedIds = [];
1035 	if (deletedIds && deletedIds.length) {
1036 		for (var i = 0; i < deletedIds.length; i++) {
1037 			var id = deletedIds[i];
1038 			var nId = ZmOrganizer.normalizeId(id);
1039 			if (nId < 0) {
1040 				virtConv[nId] = true;
1041 				virtConvDeleted = true;
1042 			} else {
1043 				newDeletedIds.push(id);
1044 			}
1045 		}
1046 	}
1047 	if (!virtConvDeleted) {
1048 		return notify;
1049 	}
1050 
1051 	// look for creates of convs that mean a virtual conv got promoted
1052 	var gotNewConv = false;
1053 	var createdMsgs = {};
1054 	var createdConvs = {};
1055 	for (var name in notify.created) {
1056 		var list = notify.created[name];
1057 		if (list && list.length) {
1058 			for (var i = 0; i < list.length; i++) {
1059 				var create = list[i];
1060 				var id = create.id;
1061 				var extra = (name == "m") ? "|cid=" + create.cid + "|l=" + create.l : "|n=" + create.n;
1062 				AjxDebug.println(AjxDebug.NOTIFY, name + ": id=" + id + "|su='" + create.su + "'|f=" + create.f + "|d=" + create.d + extra);
1063 				if (name == "m") {
1064 					createdMsgs[id] = create;
1065 				} else if (name == "c" && (create.n > 1)) {
1066 					// this is *probably* a create for a real conv from a virtual conv
1067 					createdConvs[id] = create;
1068 					gotNewConv = true;
1069 				}
1070 			}
1071 		}
1072 	}
1073 	if (!gotNewConv) {
1074 		return notify;
1075 	}
1076 
1077 	// last thing to confirm virt conv promotion is msg changing cid
1078 	var msgMoved = false;
1079 	var newToOldCid = {};
1080 	var movedMsgs = {};
1081 	var list = notify.modified.m;
1082 	if (list && list.length) {
1083 		for (var i = 0; i < list.length; i++) {
1084 			var mod = list[i];
1085 			var id = mod.id;
1086 			var nId = ZmOrganizer.normalizeId(id);
1087 			var virtCid = nId * -1;
1088 			if (virtConv[virtCid] && createdConvs[mod.cid]) {
1089 				msgMoved = true;
1090 				movedMsgs[id] = mod;
1091 				newToOldCid[mod.cid] = appCtxt.multiAccounts ? ZmOrganizer.getSystemId(virtCid) : virtCid;
1092 				createdConvs[mod.cid]._wasVirtConv = true;
1093 				createdConvs[mod.cid].m = [{id:id}];

1364 /**
1365  * Handles the creates for the given type of mail item.
1366  *
1367  * @param {Array}		creates		a list of JSON create nodes
1368  * @param {constant}	type		the mail item type
1369  * @param {Hash}		items		a hash of created mail items
1370  * @param {ZmMailList}	currList	the list currently being displayed to user
1371  * @param {constant}	sortBy		the sort order
1372  * @param {Hash}		convs		the convs, so we can update folders from msgs
1373  * @param {boolean}		last		if true, okay to mark creates as handled
1374  *
1375  * @return	{Hash}	a hash with booleans gotItem and gotAlertMessage
1376  * 
1377  * @private
1378  */
1379 ZmMailApp.prototype._checkType =
1380 function(creates, type, items, currList, sortBy, convs, last) {
1381 
1382 	var result = { gotMail:false, hasMore:false };
1383 	var nodeName = ZmList.NODE[type];
1384 	var list = creates[nodeName];
1385 	if (!(list && list.length)) { return result; }
1386 
1387 	var throttle;
1388 	if (appCtxt.isOffline) {
1389 		throttle = (appCtxt.get(ZmSetting.OFFLINE_SHOW_ALL_MAILBOXES))
1390 			? appCtxt.accountList.isInitialSyncing()
1391 			: appCtxt.getActiveAccount().isOfflineInitialSync();
1392 	}
1393 	if (throttle) {
1394 		if (!this._maxEntries) {
1395 			var mlv = this.getMailListController().getCurrentView().getMailListView();
1396 			this._maxEntries = mlv && mlv.calculateMaxEntries();
1397 		}
1398 		if (this.numEntries > this._maxEntries) {
1399 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailApp: too many creates: num=" + this.numEntries + ", max=" + this._maxEntries);
1400 			return result;
1401 		}
1402 	}
1403 
1404 	var INTERVAL_LENGTH = 10 * 1000; //10 seconds
1405 	var INTERVAL_THRESHOLD = 20; //throttle more than 20 messages.
1406 	for (var i = 0; i < list.length; i++) {
1407 		var create = list[i];
1408 
1409 		// generic throttling mechanism. Do it per folder. reset every 10 seconds. If more than 40 creates arrive in this interval, stop handling them. 
1410 		// This is used to throttle external accounts syncs but also good in general to prevent the client from hanging in case of a huge burst of updates.
1411 		var folder = create.l || "conv"; //I bundle all conv creates together (since they don't provide folder) to make it simple. There are NOT a lot of conv creates at this stage. we mostly create them in ZmMailList.prototype.notifyCreate from messages.
1412 		var now = new Date();
1413 		var data = this._throttleStats[folder];
1414 		if (!data || now.getTime() - data.intervalStart.getTime() > INTERVAL_LENGTH) {
1415 			data = this._throttleStats[folder] = {
1416 				intervalStart: now,
1417 				count: 0
1418 			}
1419 		}
1420 		data.count++;
1421 		if (data.count > INTERVAL_THRESHOLD) {
1422 			if (data.count == INTERVAL_THRESHOLD + 1) {
1423 				DBG.println(AjxDebug.DBG1, "folder " + folder + " starting to throttle at  " + now);
1424 			}
1425 			result.hasMore = true;
1426 			continue;
1427 		}
1428 
1429 
1430 		AjxDebug.println(AjxDebug.NOTIFY, "ZmMailApp: process create notification:");
1431 		var extra = (type == ZmItem.MSG) ? "|cid=" + create.cid + "|l=" + create.l : "|n=" + create.n;
1432 		AjxDebug.println(AjxDebug.NOTIFY, type + ": id=" + create.id + "|su='" + create.su + "'|f=" + create.f + "|d=" + create.d + extra);
1433 		if (create._handled) {
1434 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailApp: create already handled " + create.id);
1435 			continue;
1436 		}
1437 		if (last) {
1438 			create._handled = true;
1439 		}
1440 
1441 		// new conv does not affect a list of msgs
1442 		if (currList.type == ZmItem.MSG && type == ZmItem.CONV) {
1443 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailApp: msg list ignoring conv create");
1444 			continue;
1445 		}
1446 
1447 		// perform stricter checking if we're in offline mode
1448 		if (appCtxt.isOffline) {
1449 			if ((ZmList.ITEM_TYPE[nodeName] != currList.type) && (currList.type != ZmItem.CONV)) {
1450 				AjxDebug.println(AjxDebug.NOTIFY, "ZmMailApp: type mismatch: " + ZmList.ITEM_TYPE[nodeName] + " / " + currList.type);
1451 				continue;
1452 			}
1453 		}
1454 
1455 		// throttle influx of CREATE notifications during offline initial sync
1456 		if (throttle && this.numEntries > this._maxEntries) {
1457 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailApp: throttling");
1458 			result.hasMore = true;
1459 			break;
1460 		}
1461 
1462 		DBG.println(AjxDebug.DBG1, "ZmMailApp: handling CREATE for node: " + nodeName);
1463 
1464 		var item = appCtxt.getById(create.id);
1465 		if (!item) {
1466 			AjxDebug.println(AjxDebug.NOTIFY, "ZmMailApp: create " + type + " object " + create.id);
1467 			var itemClass = eval(ZmList.ITEM_CLASS[type]);
1468 			item = itemClass.createFromDom(create, {});
1469 		}
1470 		else if (item.type == ZmItem.MSG) {
1471 			// bug 47589: make sure conv knows its folders
1472 			var conv = appCtxt.getById(item.cid);
1473 			if (conv) {
1474 				conv.folders[item.folderId] = true;
1475 			}
1476 		}
1477 		items[item.id] = item;
1478 		result.gotMail = true;
1479 	}
1480 	return result;
1481 };

1803 ZmMailApp.prototype._parseComposeUrl =
1804 function(urlQueryStr){
1805 
1806 	urlQueryStr = urlQueryStr || '';
1807 	urlQueryStr.replace(/^mailto:/i, "");
1808 	
1809 	var match = urlQueryStr.match(/\bto=([^&]+)/i);
1810 	var to = match ? AjxStringUtil.urlComponentDecode(match[1].replace(/\+/g, " ")) : null;
1811 	to = to && AjxEmailAddress.isValid(to) ? AjxStringUtil.urlComponentDecode(to) : AjxStringUtil.htmlEncode(to);
1812 	
1813 	match = urlQueryStr.match(/\bsubject=([^&]+)/i);
1814 	var subject = match ? (AjxStringUtil.urlComponentDecode(match[1]).replace(/\+/g, " ")) : null;
1815 
1816 	match = urlQueryStr.match(/\bcc=([^&]+)/i);
1817 	var cc = match ? AjxStringUtil.urlComponentDecode(match[1].replace(/\+/g, " ")) : null;
1818 	cc = cc && AjxEmailAddress.isValid(cc) ? cc : AjxStringUtil.htmlEncode(cc);
1819 	
1820 	match = urlQueryStr.match(/\bbcc=([^&]+)/i);
1821 	var bcc = match ? AjxStringUtil.urlComponentDecode(match[1].replace(/\+/g, " ")) : null;
1822 	bcc = bcc && AjxEmailAddress.isValid(bcc) ? bcc : AjxStringUtil.htmlEncode(bcc);
1823 	
1824 	match = urlQueryStr.match(/\bbody=([^&]+)/i);
1825 	var body = match ? (AjxStringUtil.urlComponentDecode(match[1]).replace(/\+/g, " ")) : null;
1826 
1827 	return {
1828 		to: to,
1829 		subject: AjxStringUtil.htmlEncode(subject),
1830 		cc: cc,
1831 		bcc: bcc,
1832 		body: AjxStringUtil.htmlEncode(body)
1833 	};
