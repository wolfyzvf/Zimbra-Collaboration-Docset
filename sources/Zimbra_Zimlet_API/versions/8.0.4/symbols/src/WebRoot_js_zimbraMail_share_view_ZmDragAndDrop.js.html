 24 ZmDragAndDrop = function(parent) {
 25     this._view = parent;
 26     this._controller = parent._controller;
 27     this._element = parent.getHtmlElement();
 28     this._initialize();
 29 };
 30 
 31 ZmDragAndDrop.prototype.constructor = ZmDragAndDrop;
 32 
 33 /**
 34 * @return	{boolean}	true if drag and drop is supported
 35 */
 36 ZmDragAndDrop.isSupported = function() {
 37 
 38     //Refer https://github.com/Modernizr/Modernizr/issues/57#issuecomment-4187079 Drag and Drop support
 39     var div = document.createElement('div'),
 40         dragSupport = (('draggable' in div) || ('ondragstart' in div && 'ondrop' in div)),
 41         isSupported = dragSupport && !!window.FileReader;
 42 
 43     if (AjxEnv.isSafari4up && dragSupport) {
 44         isSupported = true;
 45     }
 46 
 47     ZmDragAndDrop.isSupported = function() {
 48         return isSupported;
 49     };
 50 
 51     if (isSupported) {
 52         ZmDragAndDrop.ATTACHMENT_SIZE_LIMIT = appCtxt.get(ZmSetting.ATTACHMENT_SIZE_LIMIT);
 53         ZmDragAndDrop.ATTACHMENT_URL = appCtxt.get(ZmSetting.CSFE_ATTACHMENT_UPLOAD_URI)+"?fmt=extended,raw";
 54     }
 55 
 56     return ZmDragAndDrop.isSupported();
 57 };
 58 
 59 /**
 60  * @return	{boolean} 	true if attachment size exceeded and shows the warning dialog
 61  */
 62 ZmDragAndDrop.isAttachmentSizeExceeded = function(files, showDialog) {
 63     var j,
 64         size,
 65         filesLength,
 66         file;
 67 
 68     if (!files) {
 69         return false;
 70     }
 71 
 72     for (j = 0 , size = 0, filesLength = files.length; j < filesLength; j++) {
 73         file = files[j];
 74         if (file) {
 75             size += file.size || file.fileSize; /*Safari*/
 76             //Showing Error dialog if the attachment size is exceeded
 77             if ((-1 /* means unlimited */ != ZmDragAndDrop.ATTACHMENT_SIZE_LIMIT) &&
 78                 (size > ZmDragAndDrop.ATTACHMENT_SIZE_LIMIT)) {
 79                 if (showDialog) {
 80                     var msgDlg = appCtxt.getMsgDialog();
 81                     var errorMsg = AjxMessageFormat.format(ZmMsg.attachmentSizeError, AjxUtil.formatSize(ZmDragAndDrop.ATTACHMENT_SIZE_LIMIT));
 82                     msgDlg.setMessage(errorMsg, DwtMessageDialog.WARNING_STYLE);
 83                     msgDlg.popup();
 84                 }
 85                 return true;
 86             }
 87         }
 88     }
 89     return false;
 90 };

 92 ZmDragAndDrop.prototype._initialize = function () {
 93 	if (!ZmDragAndDrop.isSupported() && this._element && this._element.id) {
 94 		var tooltip = document.getElementById(this._element.id + '_zdnd_tooltip');
 95 		if (tooltip) {
 96 			tooltip.style.display = "none";
 97 			tooltip.innerHTML = "";
 98 		}
 99 	}
100     if (!this._view || !this._controller || !this._element || !ZmDragAndDrop.isSupported()) {
101         return;
102     }
103     this._addHandlers(this._element);
104     this._dndTooltipEl = document.getElementById(this._element.id + '_zdnd_tooltip');
105     this._setToolTip();
106 };
107 
108 ZmDragAndDrop.prototype._addHandlers = function(el) {
109     Dwt.setHandler(el,"ondragover",this._onDragOver.bind(this));
110     Dwt.setHandler(el,"ondrop", this._onDrop.bind(this));
111 };
112 
113 ZmDragAndDrop.prototype._setToolTip = function(){
114     if (!this._dndTooltipEl) {
115         return;
116     }
117     if (this._view._attachCount > 0 || this._dndFilesLength > 0){
118         this._dndTooltipEl.style.display = "none";
119         this._dndTooltipEl.innerHTML = "";
120     } else {
121         this._dndTooltipEl.innerHTML = ZmMsg.dndTooltip;
122         this._dndTooltipEl.style.display = "block";
123     }
124 };
125 
126 ZmDragAndDrop.prototype._onDragOver = function(ev) {
127     ZmDragAndDrop._stopEvent(ev);
128 };
129 
130 ZmDragAndDrop.prototype._onDrop = function(ev, isEditorDND) {
131     var dt,
132         files,
133         file,
134         j,
135         filesLength;
136 
137     if (!ev || this._view._disableAttachments === true ) {
138         return;
139     }
140 
141     dt = ev.dataTransfer;
142     if (!dt) {
143         return;
144     }
145 
146     files = dt.files;
147     if (!files) {
148         return;
149     }
150 
151     if (files.length) {
152         ZmDragAndDrop._stopEvent(ev);
153     }
154 
155     if (ZmDragAndDrop.isAttachmentSizeExceeded(files, true)) {
156         return;
157     }
158 
159     this._uploadedAttachment = [];
160     this._dndFilesLength = filesLength = files.length;
161 
162     for (j = 0; j < filesLength; j++) {
163         file = files[j];
164         if (file) {
165             if (j === 0) {
166                 this._dndTooltipEl.innerHTML = "<img src='/img/animated/ImgSpinner.gif' width='16' height='16' border='0' style='float:left;'/> <div style='display:inline;'>" + ZmMsg.attachingFiles + "</div>";
167                 this._dndTooltipEl.style.display = "block";
168             }
169             this._view._initProgressSpan(file.name || file.fileName);
170             this._uploadFiles(file, isEditorDND);
171         }
172         else {
173             this._dndFilesLength--;
174         }
175     }
176 };
177 
178 ZmDragAndDrop.prototype._uploadFiles = function(file, isEditorDND) {
179     var req = new XMLHttpRequest();
180     var fileName = file.name || file.fileName;
181     req.open("POST", ZmDragAndDrop.ATTACHMENT_URL, true);
182     req.setRequestHeader("Cache-Control", "no-cache");
183     req.setRequestHeader("X-Requested-With", "XMLHttpRequest");
184     req.setRequestHeader("Content-Type",  (file.type || "application/octet-stream") + ";");
185     req.setRequestHeader("Content-Disposition", 'attachment; filename="'+ AjxUtil.convertToEntities(fileName) + '"');
186     var upload = req.upload;
187     if (upload && upload.addEventListener) {
188         upload.addEventListener("progress", this._view._uploadFileProgress.bind(this._view), false);
189     }
190     req.onreadystatechange = this._handleResponse.bind(this, req, isEditorDND);
191     req.send(file);
192 };
193 
194 ZmDragAndDrop.prototype._handleResponse = function(req, isEditorDND) {
195     if(req.readyState === 4 && req.status === 200) {
196         this._dndFilesLength--;
197         var resp = eval("["+req.responseText+"]");
198         if (resp && resp[0] === 200 && resp.length === 3) {
199             if (isEditorDND) {
200                 this._handleEditorDND(resp[2]);
201             }
202             else {
203                 this._handleNormalDND(resp[2]);
204             }
205         }
206         else {
207             this._handleErrorResponse(resp[0] || resp);
208         }
209         //Clean up the properties
210         if (this._dndFilesLength === 0) {
211             delete this._dndFilesLength;
212             delete this._uploadedAttachment;
213             this._setToolTip();
214         }
215     }
216 };
217 
218 ZmDragAndDrop.prototype._handleNormalDND = function(resp) {
219     if (resp[0].aid) {
220         this._uploadedAttachment.push(resp[0].aid);
221     }
222     if (this._dndFilesLength === 0 && this._uploadedAttachment.length > 0) {
223         this._controller.saveDraft(ZmComposeController.DRAFT_TYPE_AUTO, this._uploadedAttachment.join(","));
224     }
225 };
226 
227 ZmDragAndDrop.prototype._handleEditorDND = function(resp) {
228     this._uploadedAttachment.push(resp[0]);
229     if (this._dndFilesLength === 0 && this._uploadedAttachment.length > 0) {
230         this._uploadedAttachment.clipboardPaste = true;
231         this._controller.saveDraft(ZmComposeController.DRAFT_TYPE_AUTO, this._uploadedAttachment);
232     }
233 };
234 
235 ZmDragAndDrop.prototype._handleErrorResponse = function(respCode) {
236     var warngDlg = appCtxt.getMsgDialog();
237     if (respCode === 413) {
238         warngDlg.setMessage(ZmMsg.errorAttachmentTooBig, DwtMessageDialog.CRITICAL_STYLE);
239     } else {
240         warngDlg.setMessage(AjxMessageFormat.format(ZmMsg.errorAttachment, (respCode || AjxPost.SC_NO_CONTENT)), DwtMessageDialog.CRITICAL_STYLE);
241     }
242     warngDlg.popup();
243 };
244 
245 ZmDragAndDrop._stopEvent = function(ev) {
246     if (ZmDragAndDrop.containFiles(ev)) {
247         if (ev.preventDefault) {
248             ev.preventDefault();
249         }
250         if (ev.stopPropagation) {
251             ev.stopPropagation();
252         }
253     }
254 };
255 
256 ZmDragAndDrop.containFiles =
257 function(ev, type) {
258     if (ev && ev.dataTransfer) {
259         var typesArray = ev.dataTransfer.types;
260         if (typesArray) {
261             type = type || "Files";
262             for (var i = 0, length = typesArray.length; i < length; i++) {
263                 if (typesArray[i] === type) {
264                     return true;
265                 }
266             }
267         }
268     }
269     return false;
270 };
