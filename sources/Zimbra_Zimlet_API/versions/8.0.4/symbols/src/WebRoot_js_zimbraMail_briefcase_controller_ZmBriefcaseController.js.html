110 /**
111  * The list view as a whole is the drop target, since it's the lowest-level widget. Still, we
112  * need to find out which item got dropped onto, so we get that from the original UI event
113  * (a mouseup). The header is within the list view, but not an item, so it's not a valid drop
114  * target. One drawback of having the list view be the drop target is that we can't exercise
115  * fine-grained control on what's a valid drop target. If you enter via an item and then drag to
116  * the header, it will appear to be valid.
117  * 
118  * @protected
119  */
120 ZmBriefcaseController.prototype._dropListener =
121 function(ev) {
122 	var view = this._listView[this._currentViewId];
123 	var div = view.getTargetItemDiv(ev.uiEvent);
124 	var item = view.getItemFromElement(div);
125 	if(!item || !( item.isRevision || item.isFolder) ) {
126 		ZmListController.prototype._dropListener.call(this,ev);
127 	} else {
128 		ev.doIt = false;
129 	}
130 }

229 ZmBriefcaseController.prototype._resetOperations =
230 function(parent, num) {
231 	if (!parent) { return; }
232 
233 	// call base class
234 	ZmListController.prototype._resetOperations.call(this, parent, num);
235 
236 	var items = this._listView[this._currentViewId].getSelection();
237 	var isFolderSelected=false, noOfFolders = 0, isRevisionSelected=false, isBriefcaseItemSelected=false, isMixedSelected=false;
238     var isWebDocSelected= false, hasLocked = false, allLocked = true, sameLockOwner=true;
239     var hasHighestRevisionSelected = false, hasOldRevisionSelected = false;
240 	if (items) {
241 		for (var i = 0; i < items.length; i++) {
242 			var item = items[i];
243 			if (item.isFolder) {
244 				isFolderSelected = true;
245 				noOfFolders++;
246 			}else if(item.isRevision){
247                 isRevisionSelected = true;
248                 if(item.parent.version == item.version){
249                     hasHighestRevisionSelected = true;
250                 }
251 				else {
252 					hasOldRevisionSelected = true;
253 				}
254             }else{
255                 isBriefcaseItemSelected = true;
256             }
257 
258             isWebDocSelected = isWebDocSelected || ( !item.isFolder && item.isWebDoc() );
259 
260             allLocked = allLocked && item.locked;
261 
262             hasLocked = hasLocked || item.locked;
263 
264             sameLockOwner = sameLockOwner && (item.locked && item.lockUser == appCtxt.getActiveAccount().name);
265 		}
266 	}
267 
268     isMixedSelected = isFolderSelected ? (isBriefcaseItemSelected || isRevisionSelected) :  (isBriefcaseItemSelected && isRevisionSelected);
269 
270     var briefcase = appCtxt.getById(this._folderId);
271     if(!(briefcase instanceof ZmBriefcase)){
272         briefcase = null;
273     }
274     var isTrash = (briefcase && briefcase.nId == ZmOrganizer.ID_TRASH);
275     var isShared = ((briefcase && briefcase.nId != ZmOrganizer.ID_TRASH && briefcase.isShared()));
276 	var isReadOnly = briefcase ? briefcase.isReadOnly() : false;
277 	var isMultiFolder = (noOfFolders > 1);
278 	var isItemSelected = (num>0);
279 	var isZimbraAccount = appCtxt.getActiveAccount().isZimbraAccount;
280 	var isMailEnabled = appCtxt.get(ZmSetting.MAIL_ENABLED);
281     var isAdmin = briefcase && briefcase.isAdmin(); 
282 
283     var item = items[0];
284     //bug 65351
285     // treat the latest revision selection as if it was a file selection.
286     // isOldRevision is true if the item is a revision but not the latest.
287     var isOldRevision = hasOldRevisionSelected ? true : item && item.revision && !hasHighestRevisionSelected;
288 	
289 	parent.enable([ZmOperation.SEND_FILE, ZmOperation.SEND_FILE_AS_ATT], (isZimbraAccount && isMailEnabled && isItemSelected && !isMultiFolder && !isFolderSelected));
290 	parent.enable(ZmOperation.TAG_MENU, (!isReadOnly && isItemSelected && !isFolderSelected && !isOldRevision));
291 	parent.enable([ZmOperation.NEW_FILE, ZmOperation.VIEW_MENU], true);
292 	parent.enable([ZmOperation.NEW_SPREADSHEET, ZmOperation.NEW_PRESENTATION, ZmOperation.NEW_DOC], true);
293 	parent.enable([ZmOperation.MOVE, ZmOperation.MOVE_MENU], ( isItemSelected &&  !isReadOnly && !isShared && !isOldRevision));
294     parent.enable(ZmOperation.NEW_FILE, !(isTrash || isReadOnly));
295     parent.enable(ZmOperation.NEW_BRIEFCASE_WIN, (isItemSelected && !isFolderSelected && num==1));
296 
297     var firstItem = items && items[0];
298     var isWebDoc = firstItem && !firstItem.isFolder && firstItem.isWebDoc();
299     var isLocked = firstItem && !firstItem.isFolder && firstItem.locked;
300     var isLockOwner = isLocked && (item.lockUser == appCtxt.getActiveAccount().name);
301 
302 
303     //Rename Operation
304     parent.enable(ZmOperation.RENAME_FILE, ( num ==1 && !isFolderSelected && !isReadOnly && !isOldRevision && (isLocked ? isLockOwner : true) ));
305 
306     //Download - Files
307     parent.enable(ZmOperation.SAVE_FILE, num >0 && (!isFolderSelected || isBriefcaseItemSelected));
308 
309     // Edit
310     parent.enable(ZmOperation.OPEN_FILE, (num == 1 && isWebDoc));
311     parent.enable(ZmOperation.EDIT_FILE, !isReadOnly && (  !isLocked || isLockOwner ) && isWebDoc && !isOldRevision && num == 1);
312 
313     //Delete Operation
314     parent.enable(ZmOperation.DELETE, (!isReadOnly && isItemSelected && (hasHighestRevisionSelected ? !hasOldRevisionSelected : true) && !isMixedSelected && (isLocked ? isLockOwner : true)));
315 
316     if(parent &&  parent instanceof ZmActionMenu){
317 
318         //Open - webDocs
319         parent.getOp(ZmOperation.OPEN_FILE) && parent.getOp(ZmOperation.OPEN_FILE).setVisible(isItemSelected && !isMultiFolder && isWebDoc);
320 
321 	}
322 	//Case 1: Multiple Admins
323 	//Case 2: Stale Lock ( Handle exception )
324 
325 	//Checkin
326 	var op = parent.getOp(ZmOperation.CHECKIN);
327 	if (op) {
328 		var checkinEnabled = !isReadOnly && isLockOwner && !isWebDoc && !isOldRevision;
329 		op.setVisible(checkinEnabled);
330 		parent.enable(ZmOperation.CHECKIN, checkinEnabled && num == 1);
331 	}
332 
333 	//Checkout
334 	op = parent.getOp(ZmOperation.CHECKOUT);
335 	if (op) {
336 		var checkoutEnabled = !isReadOnly && !hasLocked && !isRevisionSelected && !isFolderSelected;
337 		op.setVisible(!isOldRevision && !isLocked);
338 		parent.enable(ZmOperation.CHECKOUT, checkoutEnabled);
339 	}
340 
341 	//Discard Checkout
342 	op = parent.getOp(ZmOperation.DISCARD_CHECKOUT);
343 	if (op) {
344 		var discardCheckoutEnabled = sameLockOwner && !isRevisionSelected;
345 		op.setVisible(discardCheckoutEnabled);
346 		parent.enable(ZmOperation.DISCARD_CHECKOUT, discardCheckoutEnabled && (isAdmin || sameLockOwner || !isShared));
347 	}
348 
349 	//Versioning
350 	op = parent.getOp(ZmOperation.RESTORE_VERSION);
351 	if (op) {
352 		var versionEnabled = (!isReadOnly && num == 1 && isOldRevision);
353 		var isHightestVersion = item && item.isRevision && ( item.parent.version == item.version );
354 		op.setVisible(isOldRevision);
355 		parent.enable(ZmOperation.RESTORE_VERSION, versionEnabled && !isHightestVersion);
356 	}
357 
358     var isDocOpEnabled = !(isTrash || isReadOnly);
359     if (appCtxt.get(ZmSetting.DOCS_ENABLED)) {
360         parent.enable(ZmOperation.NEW_DOC, isDocOpEnabled);
361     }
362     if (appCtxt.get(ZmSetting.SPREADSHEET_ENABLED)) {
363         parent.enable(ZmOperation.NEW_SPREADSHEET, isDocOpEnabled);
364     }
365     if (appCtxt.get(ZmSetting.SLIDES_ENABLED)) {
366         parent.enable(ZmOperation.NEW_PRESENTATION, isDocOpEnabled);
367         parent.enable(ZmOperation.CREATE_SLIDE_SHOW, isDocOpEnabled && isItemSelected);
368     }
369 
370     if(appCtxt.isExternalAccount()) {
371         parent.enable ([ZmOperation.SEND_FILE,
372                         ZmOperation.SEND_FILE_AS_ATT,
373                         ZmOperation.RENAME_FILE,
374                         ZmOperation.MOVE,
375                         ZmOperation.MOVE_MENU,
376                         ZmOperation.NEW_FILE,
377                         ZmOperation.TAG_MENU,
378                         ZmOperation.EDIT_FILE,
379                         ZmOperation.OPEN_FILE,
380                         ZmOperation.CHECKIN,
381                         ZmOperation.CHECKOUT,
382                         ZmOperation.DISCARD_CHECKOUT,
383                         ZmOperation.RESTORE_VERSION,
384                         ZmOperation.NEW_BRIEFCASE_WIN,
385                         ZmOperation.DELETE
386                         ], false);
387         parent.setItemVisible(ZmOperation.TAG_MENU, false);
388     }
389 };

396 ZmBriefcaseController.prototype._doDelete =
397 function(items) {
398 
399 	if (!items) {
400 		items = this._listView[this._currentViewId].getSelection();
401 	}
402     var item = (items instanceof Array) ? items[0] : items;
403     if(!item) return;
404 
405 	var message = ( items.length > 1 ) ?  ( item.isRevision  ? ZmMsg.confirmPermanentDeleteItemList : ZmMsg.confirmDeleteItemList ) : null;
406 	if (!message) {
407         var delMsgFormatter = new AjxMessageFormat( (this._folderId == String(ZmOrganizer.ID_TRASH) || (item.isRevision && item.parent.version != item.version) ) ? ZmMsg.confirmPermanentDeleteItem : ZmMsg.confirmDeleteItem );
408 		message = delMsgFormatter.format(AjxStringUtil.htmlEncode(item.name));
409 	}
410 
411     var dialog = appCtxt.getConfirmationDialog();
412 	dialog.popup(message, new AjxCallback(this, this._doDelete2, [items]));
413 };

415 ZmBriefcaseController.prototype._doDelete2 =
416 function(items) {
417 
418     var item = (items instanceof Array) ? items[0] : items;
419 
420     if(item.isRevision && item.parent.version != item.version){
421         var view = this._parentView[this._currentViewId];
422         view.deleteVersions(items);
423     }else if(item.isFolder){
424         //Bug fix # 80600 force the BatchCommand to use JSON, mimicking the way right click delete behaves
425         var delBatchCmd = new ZmBatchCommand(true, null, true), folder;
426         for(var i=0; i< items.length; i++){
427             folder = items[i].folder;
428             if(folder.isHardDelete()){
429                 delBatchCmd.add(new AjxCallback(folder, folder._delete, [delBatchCmd]));
430             }else{
431                 var trashFolder = appCtxt.getById(ZmFolder.ID_TRASH);
432                 delBatchCmd.add(new AjxCallback(folder, folder.move, [trashFolder, false, null, delBatchCmd]));
433             }
434         }
435         delBatchCmd.run();
436     }else{
437 		for (var i=0; i<items.length; i++) {
438 			if (items[i].isRevision) {
439 				items[i] = items[i].parent;
440 			}
441 		}
442         ZmListController.prototype._doDelete.call(this, items, null, null, true);
443     }
444 };

502 /**
503  * Shows the search results.
504  * 
505  * @param	{Object}	results		the search results
506  */
507 ZmBriefcaseController.prototype.show =
508 function(results) {
509 
510 	this._folderId = results && results.search && results.search.folderId;
511 	this.setList(results.getResults(ZmItem.BRIEFCASE_ITEM));
512 	this._list.setHasMore(results.getAttribute("more"));

516 	this._setup(this._currentViewId);
517 
518 	// start fresh with search results
519 	var lv = this._listView[this._currentViewId];
520 	lv.offset = 0;

522 
523 	var elements = this.getViewElements(this._currentViewId, this._parentView[this._currentViewId]);

525 	this._setView({	view:		this._currentViewId,
526 					viewType:	this._currentViewType,
527 					noPush:		this.isSearchResults,
528 					elements:	elements,
529 					isAppView:	true});
530 	if (this.isSearchResults) {
531 		// if we are switching views, make sure app view mgr is up to date on search view's components
532 		appCtxt.getAppViewMgr().setViewComponents(this.searchResultsController.getCurrentViewId(), elements, true);
533 	}
534 	this._resetNavToolBarButtons();
535 };

537 /**
538  * Change how briefcase items are displayed.
539  * 
540  * @param {constant}	view			the view to show
541  * @param {Boolean}	force			if <code>true</code>, render view even if it's the current view
542  */
543 ZmBriefcaseController.prototype.switchView =
544 function(view, force) {
545 
546 	var viewChanged = (force || view != this._currentViewId);
547 
548 	if (viewChanged) {
