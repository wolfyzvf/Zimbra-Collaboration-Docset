 20 /**
 21  * Creates a revoke share dialog.
 22  * @class
 23  * This class represents a revoke share dialog.
 24  * 
 25  * @param	{DwtComposite}	parent		the parent
 26  * @param	{String}	className		the class name
 27  *  
 28  * @extends		DwtDialog
 29  */
 30 ZmRevokeShareDialog = function(parent, className) {
 31 	className = className || "ZmRevokeShareDialog";
 32 	var title = ZmMsg.revokeShare;
 33 	var buttons = [ DwtDialog.YES_BUTTON, DwtDialog.NO_BUTTON ];
 34 	DwtDialog.call(this, {parent:parent, className:className, title:title, standardButtons:buttons});
 35 	this.setButtonListener(DwtDialog.YES_BUTTON, new AjxListener(this, this._handleYesButton));
 36 	
 37 	var view = this._createView();
 38 	this.setView(view);
 39 
 40 	// create formatters
 41 	this._formatter = new AjxMessageFormat(ZmMsg.revokeShareConfirm);
 42 };
 43 
 44 ZmRevokeShareDialog.prototype = new DwtDialog;
 45 ZmRevokeShareDialog.prototype.constructor = ZmRevokeShareDialog;
 46 
 47 // Public methods
 48 
 49 ZmRevokeShareDialog.prototype.toString =
 50 function() {
 51 	return "ZmRevokeShareDialog";
 52 };
 53 
 54 /**
 55  * Pops-up the dialog.
 56  * 
 57  * @param	{ZmShare}	share		the share
 58  */
 59 ZmRevokeShareDialog.prototype.popup =
 60 function(share) {
 61 	this._share = share;
 62 
 63 	var isPubShare = share.isPublic();
 64 	var isGuestShare = share.isGuest();
 65 	var isAllShare = share.grantee && (share.grantee.type == ZmShare.TYPE_ALL);
 66 
 67 	var params = isPubShare ? ZmMsg.shareWithPublic : isGuestShare ? share.grantee.id : isAllShare ? ZmMsg.shareWithAll :
 68 					(share.grantee.name || ZmMsg.userUnknown);
 69 	this._confirmMsgEl.innerHTML = this._formatter.format(params);
 70 
 71 	this._reply.setReplyType(ZmShareReply.STANDARD);
 72 	this._reply.setReplyNote("");
 73 	this._reply.setVisible(!isPubShare && !isAllShare);
 74 
 75     if (isGuestShare) {
 76         this._reply.setReplyOptions(ZmShareReply.EXTERNAL_USER_OPTIONS);
 77     }
 78     else {
 79         this._reply.setReplyOptions(ZmShareReply.DEFAULT_OPTIONS);
 80     }
 81 
 82 	DwtDialog.prototype.popup.call(this);
 83 	this.setButtonEnabled(DwtDialog.YES_BUTTON, true);
 84 };
 85 
 86 // Protected methods
 87 
 88 ZmRevokeShareDialog.prototype._handleYesButton =
 89 function() {
 90 	var callback = new AjxCallback(this, this._yesButtonCallback);
 91 	this._share.revoke(callback);
 92 };
 93 
 94 ZmRevokeShareDialog.prototype._yesButtonCallback =
 95 function() {
 96 	var share = this._share;
 97 	var replyType = this._reply.getReplyType();
 98 	var sendMail = !(share.isAll() || share.isPublic() || share.invalid); 
 99 	if (replyType != ZmShareReply.NONE && sendMail) {
100 		// initialize rest of share information
101 		share.grantee.email = share.grantee.name || share.grantee.id;
102 		share.grantor.id = appCtxt.get(ZmSetting.USERID);
103 		share.grantor.email = appCtxt.get(ZmSetting.USERNAME);
104 		share.grantor.name = appCtxt.get(ZmSetting.DISPLAY_NAME) || share.grantor.email;
105 		share.link.id = share.object.id;
106 		share.link.name = share.object.name;
107 		share.link.view = ZmOrganizer.getViewName(share.object.type);
108 
109 		share.notes = (replyType == ZmShareReply.QUICK) ? this._reply.getReplyNote() : "";
110 	
111 		share.sendMessage(ZmShare.DELETE);
112 	}
113 
114 	this.popdown();
115 };
116 
117 ZmRevokeShareDialog.prototype._createView =
118 function() {
119 	this._confirmMsgEl = document.createElement("DIV");
120 	this._confirmMsgEl.style.fontWeight = "bold";
121 	this._confirmMsgEl.style.marginBottom = "0.25em";
122 	
123 	var view = new DwtComposite(this);
124 	this._reply = new ZmShareReply(view);
125 	
126 	var element = view.getHtmlElement();
127 	element.appendChild(this._confirmMsgEl);
128 	element.appendChild(this._reply.getHtmlElement());
129 
130 	return view;
131 };
132 
133 ZmRevokeShareDialog.prototype._getSeparatorTemplate =
134 function() {
135 	return "";
136 };
