 16 /**
 17  * Creates a left pane view for suggesting locations
 18  * @constructor
 19  * @class
 20  * This class displays suggested free locations for a new appointment
 21  *
 22  *  @author Vince Bellows
 23  *
 24  * @param apptEditView		[ZmApptQuickAddDialog or
 25  *                           ZmApptComposeView]         View containing the suggestions
 26  * @param container  		[DOM Element]	            The dialog's content Element
 27  * @param controller		[ZmApptComposeController]	The appt compose controller
 28  * @param closeCallback 	[Callback Function]			Function to invoke upon close
 29  */
 30 ZmApptAssistantView = function(parent, controller, apptView, closeCallback) {
 31     if (arguments.length == 0) { return; }
 32 
 33 	DwtComposite.call(this, {parent: parent, posStyle: DwtControl.RELATIVE_STYLE, className: "ZmScheduleAssistantView"});
 34 
 35 	this._controller    = controller;
 36 	this._apptView      = apptView;
 37     this._prefDialog    = appCtxt.getSuggestionPreferenceDialog();
 38     this._closeCallback = closeCallback;
 39 
 40     // For bug 68531
 41     //var app = appCtxt.getApp(ZmApp.CALENDAR);
 42     //this._fbCache       = app.getFreeBusyCache();
 43     this._fbCache       = apptView.getFreeBusyCache();
 44 
 45 	this._rendered      = false;
 46     this.type           = ZmCalBaseItem.LOCATION;
 47     this._resources     = [];
 48 
 49     this._enabled       = false;
 50 
 51     this.numRecurrence  = this.getLocationConflictNumRecurrence();
 52 
 53     this.initialize();
 54 };

184 ZmApptAssistantView.prototype.searchCalendarResources =
185 function(callback, sortBy) {
186 	var currAcct = this._apptView.getCalendarAccount();
187 	var value = (this.type == ZmCalBaseItem.LOCATION) ? "Location" : "Equipment";
188 
189     var conds = [{attr: "zimbraCalResType", op: "eq", value: value}];
190     if(this._prefDialog) {
191         for (var i = 0; i < ZmTimeSuggestionPrefDialog.PREF_FIELDS.length; i++) {
192             var sf = ZmTimeSuggestionPrefDialog.PREF_FIELDS[i];
193 
194             if(!ZmTimeSuggestionPrefDialog.isSearchCondition(sf)) continue;
195 
196             value = AjxStringUtil.trim(this._prefDialog.getPreference(sf));
197 
198             if (value.length) {
199                 var attr = ZmTimeSuggestionPrefDialog.SF_ATTR[sf];
200                 var op = ZmTimeSuggestionPrefDialog.SF_OP[sf] ? ZmTimeSuggestionPrefDialog.SF_OP[sf] : "has";
201                 conds.push({attr: attr, op: op, value: value});
202             }
203         }
204     }
205     
206 	var params = {
207 		sortBy: sortBy,
208 		offset: 0,
209 		limit: ZmContactsApp.SEARCHFOR_MAX,
210 		conds: conds,
211 		attrs: ZmApptAssistantView.ATTRS[this.type],
212 		accountName: appCtxt.isOffline ? currAcct.name : null
213 	};
214 	var search = new ZmSearch(params);
215 	search.execute({callback: new AjxCallback(this, this._handleResponseSearchCalendarResources, callback)});
216 };

218 ZmApptAssistantView.prototype._handleResponseSearchCalendarResources =
219 function(callback, result) {
220 	var resp = result.getResponse();
221 	var items = resp.getResults(ZmItem.RESOURCE).getVector();
222     	if (items)
223     		this._resources = (items instanceof AjxVector) ? items.getArray() : (items instanceof Array) ? items : [items];
224     if(callback) callback.run();
225 };

232 ZmApptAssistantView.prototype._copyResourcesToParams =
233 function(params, emails) {
234     var list = this._resources;
235 	for (var i = list.length; --i >= 0;) {
236 		var item = list[i];
237 		var email = item.getEmail();
238 
239 		// bug: 30824 - Don't list all addresses/aliases of a resource in
240 		// GetFreeBusyRequest.  One should suffice.
241 		if (email instanceof Array) {
242 			email = email[0];
243 		}
244 		emails.push(email);
245 
246         params.items.push(email);
247         params.itemIndex[email] = params.items.length-1;
248 	}
249 }

251 ZmApptAssistantView.prototype.suggestLocations =
252 function(params) {
253     var emails = [];
254     this._copyResourcesToParams(params, emails);
255     this._duration  = this._apptView.getDurationInfo();
256     params.emails   = emails;
257     params.duration = this._duration.duration;
258     params.locationInfo = this.computeLocationAvailability(this._duration, params);
259     this.renderSuggestions(params);
260 };

262 // For a single given time slot, determine the available rooms
263 ZmApptAssistantView.prototype.computeLocationAvailability =
264 function(durationInfo, params) {
265 
266     var locationInfo = {
267             startTime: durationInfo.startTime,
268             endTime:   durationInfo.endTime,
269             locations: new AjxVector()
270         };
271 
272     var list = this._resources;
273     for (var i = list.length; --i >= 0;) {
274         var email = list[i].getEmail();
275 
276         if (email instanceof Array) {
277             email = email[0];
278         }
279 
280         var excludeTimeSlots = this._apptView.getFreeBusyExcludeInfo(email);
281 
282         // Adjust start and end time by 1 msec, to avoid fencepost problems
283         sched = this._fbCache.getFreeBusySlot(durationInfo.startTime+1,
284             durationInfo.endTime-1, email, excludeTimeSlots);
285         isFree = true;
286         if(sched.b) isFree = isFree && ZmApptAssistantView.isBooked(sched.b, durationInfo.startTime, durationInfo.endTime);
287         if(sched.t) isFree = isFree && ZmApptAssistantView.isBooked(sched.t, durationInfo.startTime, durationInfo.endTime);
288         if(sched.u) isFree = isFree && ZmApptAssistantView.isBooked(sched.u, durationInfo.startTime, durationInfo.endTime);
289 
290         //collect all the item indexes of the locations available at this slot
291         if(isFree) {
292             var displayInfo = this._createLocationDisplayInfo(email);
293             locationInfo.locations.add(displayInfo);
294         }
295     }
296     locationInfo.locations.sort(this._compareItems.bind(this));
297     return locationInfo;
298 };

301 ZmApptAssistantView.isBooked =
302 function(slots, startTime, endTime) {
303     for (var i = 0; i < slots.length; i++) {
304         var startConflict = startTime >= slots[i].s && startTime < slots[i].e;
305         var endConflict = endTime > slots[i].s && endTime <= slots[i].e;
306         var inlineSlotConflict = slots[i].s >= startTime && slots[i].e <= endTime;
307         if(startConflict || endConflict || inlineSlotConflict) {
308             return false;
309         }
310     };
311     return true;
312 };

314 ZmApptAssistantView.prototype._createLocationDisplayInfo =
315 function (email) {
316     var info = { email: email };
317     info.locationObj = this.getLocationByEmail(email);
318     info.name = email;
319     info.description = '';
320     if(info.locationObj) {
321         info.name = info.locationObj._fileAs;
322         info.description = info.locationObj.getAttr(ZmResource.F_locationName) ||
323                            info.locationObj.getAttr(ZmResource.F_name);
324         if (info.description == info.name) {
325             info.description = '';
326         }
327         info.contactMail = info.locationObj.getAttr(ZmResource.F_contactMail)
328         info.capacity    = info.locationObj.getAttr(ZmResource.F_capacity)
329     }
330     return info;
331 }

333 ZmApptAssistantView.prototype._sortLocation = function(list) {
334 	if (list) {
335 		list.sort(this._compareItems.bind(this));
336 	}
337 };
338 ZmApptAssistantView.prototype._compareItems = function(item1, item2) {
339 	var aVal = item1.name.toLowerCase();
340 	var bVal = item2.name.toLowerCase();
341 
342 	if (aVal < bVal) {
343         return -1;
344     } else if (aVal > bVal)	{
345         return 1; }
346 	else {
347         return 0;
348     }
349 
350 };

356 ZmApptAssistantView.prototype.getLocationByEmail =
357 function(item) {
358 	var locations = this._resources;
359 	for (var i = 0; i < locations.length; i++) {
360 		var value = locations[i].getEmail();
361 
362         if(value instanceof Array) {
363             for(var j = 0; j < value.length; j++) {
364                 if(item == value[j]) return locations[i];
365             }
366         }
367 		if (item == value) {
368 			return locations[i];
369 		}
370 	}
371 	return null;
