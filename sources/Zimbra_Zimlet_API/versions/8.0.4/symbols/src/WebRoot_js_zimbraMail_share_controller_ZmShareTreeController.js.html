 16 ZmShareTreeController = function() {
 17     ZmTreeController.call(this, ZmOrganizer.SHARE);
 18 };
 19 ZmShareTreeController.prototype = new ZmTreeController;
 20 ZmShareTreeController.prototype.constructor = ZmShareTreeController;
 21 
 22 ZmShareTreeController.prototype.toString = function() {
 23     return "ZmShareTreeController";
 24 };

 30 ZmShareTreeController.prototype.getDataTree = function(account) {
 31     var tree = new ZmFolderTree(ZmOrganizer.SHARE);
 32     var obj = { id: ZmOrganizer.ID_ROOT, name: ZmMsg.sharedFoldersHeader };
 33     tree.root = ZmFolderTree.createFolder(ZmOrganizer.SHARE, null, obj, tree);
 34     return tree;
 35 };

 37 ZmShareTreeController.prototype._createTreeView = function(params) {
