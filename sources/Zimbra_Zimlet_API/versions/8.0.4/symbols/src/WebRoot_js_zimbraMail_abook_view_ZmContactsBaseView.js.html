 21 /**
 22  * Creates the base view.
 23  * @class
 24  * This class represents the base view.
 25  * 
 26  * @param	{Hash}	params		a hash of parameters
 27  * 
 28  * @extends		ZmListView
 29  */
 30 ZmContactsBaseView = function(params) {
 31 
 32 	if (arguments.length == 0) { return; }
 33 
 34 	params.posStyle = params.posStyle || Dwt.ABSOLUTE_STYLE;
 35 	params.type = ZmItem.CONTACT;
 36 	params.pageless = true;
 37 	ZmListView.call(this, params);
 38 
 39 	this._handleEventType[ZmItem.GROUP] = true;
 40 };

 55 /**
 56  * Sets the list.
 57  * 
 58  * @param	{ZmContactList}		list		the list
 59  * @param	{String}	sortField		the sort field
 60  * @param	{String}	folderId		the folder id
 61  */
 62 ZmContactsBaseView.prototype.set =
 63 function(list, sortField, folderId) {
 64 
 65 	if (this._itemsToAdd) {
 66 		this.addItems(this._itemsToAdd);
 67 		this._itemsToAdd = null;
 68 	} else {
 69 		var subList;
 70 		if (list instanceof ZmContactList) {
 71 			// compute the sublist based on the folderId if applicable
 72 			list.addChangeListener(this._listChangeListener);
 73 			// for accounts where gal paging is not supported, show *all* results
 74 			subList = (list.isGal && !list.isGalPagingSupported)
 75 				? list.getVector().clone()
 76 				: list.getSubList(this.offset, this.getLimit(this.offset), folderId);
 77 		} else {
 78 			subList = list;
 79 		}
 80 		this._folderId = folderId;
 81 		DwtListView.prototype.set.call(this, subList, sortField);
 82 	}
 83 	this._setRowHeight();
 84 	this._rendered = true;
 85 
 86 	// check in case there are more items but no scrollbar
 87 	if (this._isPageless) {
 88 		this._checkItemCount();
 89 	}
 90 };

121 /**
122  * overriding in order to prevent distribution lists from being draggable.
123  */
124 ZmContactsBaseView.prototype._getDragProxy =
125 function(dragOp) {
126 	var dndSelection = this.getDnDSelection();
127 	dndSelection = AjxUtil.toArray(dndSelection); // it might be array or just a scalar ZmContact. So this will always get array.
128 	for (var i = 0; i < dndSelection.length; i++) { //if any is a DL - don't allow the drag. There could be a mix in search results.
129 		if (dndSelection[i].isDistributionList()) { //don't allow dragging a DL
130 			return null;
131 		}
132 	}
133 	return ZmListView.prototype._getDragProxy.call(this, dragOp);
134 };

136 /**
137  * @private
138  */
139 ZmContactsBaseView.prototype._changeListener =
140 function(ev) {
141 	var folderId = this._controller.getFolderId();
142 
143 	// if we dont have a folder, then assume user did a search of contacts
144 	if (folderId != null || ev.event != ZmEvent.E_MOVE) {
145 		ZmListView.prototype._changeListener.call(this, ev);
146 
147 		if (ev.event == ZmEvent.E_MODIFY) {
148 			this._modifyContact(ev);
149 			var contact = ev.item || ev._details.items[0];
150 			if (contact instanceof ZmContact) {
151 				this.setSelection(contact, false, true);
152 			}
153 		} else if (ev.event == ZmEvent.E_CREATE) {
154 			var newContact = ev._details.items[0];
155 			var newFolder = appCtxt.getById(newContact.folderId);
156 			var newFolderId = newFolder && (appCtxt.getActiveAccount().isMain ? newFolder.nId : newFolder.id);
157 			var visible = ev.getDetail("visible");
158 
159 			// only add this new contact to the listview if this is a simple
160 			// folder search and it belongs!
161 			if (folderId && newFolder && folderId == newFolderId && visible) {
162 				var index = ev.getDetail("sortIndex");
163 				var alphaBar = this.parent ? this.parent.getAlphabetBar() : null;
164 				var inAlphaBar = alphaBar ? alphaBar.isItemInAlphabetLetter(newContact) : true;
165 				if (index != null && inAlphaBar) {
166 					this.addItem(newContact, index);
167 				}
168 
169 				// always select newly added contact if its been added to the
170 				// current page of contacts
171 				if (inAlphaBar) {
172 					this.setSelection(newContact, false, true);
173 				}
174 			}
175 		} else if (ev.event == ZmEvent.E_DELETE) {
176 			// bug fix #19308 - do house-keeping on controller's list so
177 			// replenishment works as it should
178 			var list = this._controller.getList();
179 			if (list) {
180 				list.remove(ev.item);
181 			}
182 		}
183 	}
184 };

186 ZmContactsBaseView.prototype.setSelection =
187 function(item, skipNotify, setPending) {
188 	if (!item) { return; }
189 
190 	var el = this._getElFromItem(item);
191 	if (el) {
192 		ZmListView.prototype.setSelection.call(this, item, skipNotify);
193 		this._pendingSelection = null;
194 	} else if (setPending) {
195 		this._pendingSelection = {item: item, skipNotify: skipNotify};
196 	}
197 };
198 
199 ZmContactsBaseView.prototype.addItems =
200 function(itemArray) {
201 	ZmListView.prototype.addItems.call(this, itemArray);
202 	if (this._pendingSelection && AjxUtil.indexOf(itemArray, this._pendingSelection.item)!=-1) {
203 		this.setSelection(this._pendingSelection.item, this._pendingSelection.skipNotify);
204 	}
205 }

208 /**
209  * @private
210  */
211 ZmContactsBaseView.prototype._modifyContact =
212 function(ev) {
213 	// if fileAs changed, resort the internal list
214 	// XXX: this is somewhat inefficient. We should just remove this contact and reinsert
215 	if (ev.getDetail("fileAsChanged")) {
216 		var list = this.getList();
217 		if (list) {
218 			list.sort(ZmContact.compareByFileAs);
219 		}
220 	}
221 };

223 /**
224  * @private
225  */
226 ZmContactsBaseView.prototype._setNextSelection =
227 function() {
228 	// set the next appropriate selected item
229 	if (this.firstSelIndex < 0) {
230 		this.firstSelIndex = 0;
231 	}
232 
233 	// get first valid item to select
234 	var item;
235 	if (this._list) {
236 		item = this._list.get(this.firstSelIndex);
237 
238 		// only get the first non-trash contact to select if we're not in Trash
239 		if (this._controller.getFolderId() == ZmFolder.ID_TRASH) {
240 			if (!item) {
241 				item = this._list.get(0);
242 			}
243 		} else if (item == null || (item && item.folderId == ZmFolder.ID_TRASH)) {
244 			item = null;
245 			var list = this._list.getArray();
246 
247 			if (this.firstSelIndex > 0 && this.firstSelIndex == list.length) {
248 				item = list[list.length-1];
249 			} else {
250 				for (var i=0; i < list.length; i++) {
251 					if (list[i].folderId != ZmFolder.ID_TRASH) {
252 						item = list[i];
253 						break;
254 					}
255 				}
256 			}
257 
258 			// reset first sel index
259 			if (item) {
260 				var div = document.getElementById(this._getItemId(item));
261 				if (div) {
262 					var data = this._data[div.id];
263 					this.firstSelIndex = this._list ? this._list.indexOf(data.item) : -1;
264 				}
265 			}
266 		}
267 	}
268 
269 	this.setSelection(item);
270 };

331 /**
332  * Resets the bar.
333  * 
334  * @param	{Object}	useCell		the cell or <code>null</code>
335  * @return	{Boolean}				Whether the cell was changed (false if it was already set to useCell)
336  */
337 ZmContactAlphabetBar.prototype.reset =
338 function(useCell) {
339 	var cell = useCell || this._all;
340 	if (cell != this._current) {
341 		this.setSelected(this._current, false);
342 		this._current = cell;
343 		this._currentLetter = useCell && useCell != this._all ? useCell.innerHTML : null;
344 		this.setSelected(cell, true);
345 		return true;
346 	}
347 	return false;
348 };

397 /**
398  * Sets the cell as selected and performs a new search based on the selection.
399  * 
400  * @param	{Object}	cell		the cell
401  * @param	{String}	letter		the letter to begin the search with
402  * @param	{String}	endLetter	the letter to end the search with
403  */
404 ZmContactAlphabetBar.alphabetClicked =
405 function(cell, letter, endLetter) {
406 	// get reference to alphabet bar - ugh
407 	var clc = AjxDispatcher.run("GetContactListController");
408 	var alphabetBar = clc && clc.getCurrentView() && clc.getCurrentView().getAlphabetBar();
409 	if (alphabetBar && alphabetBar.enabled()) {
410 		if (alphabetBar.reset(cell)) {
411             letter = letter && String(letter).substr(0,1);
412             endLetter = endLetter && String(endLetter).substr(0,1);
413 			clc.searchAlphabet(letter, endLetter);
414         }
415 	}
416 };

418 /**
419  * determine if contact belongs in the current alphabet bar.  Used when creating a new contact and not doing a reload --
420  * such as new contact group from action menu.
421  * @param item  {ZmContact}
422  * @return {boolean} true/false if item belongs in alphabet selection
423  */
424 ZmContactAlphabetBar.prototype.isItemInAlphabetLetter =
425 function(item) {
426     var inCurrentBar = false;
427 	if (item) {
428 	  if (ZmMsg.alphabet && ZmMsg.alphabet.length > 0) {
429 		  var all = ZmMsg.alphabet.split(",")[0]; //get "All" for locale
430 	  }
431 	  var fileAs = item.getFileAs();
432 	  var currentLetter = this.getCurrentLetter();
433 	  if (!currentLetter || currentLetter.toLowerCase() == all) {
434 		  inCurrentBar = true; //All is selected
435 	  }
436 	  else if (currentLetter && fileAs) {
437 		var itemLetter = String(fileAs).substr(0,1).toLowerCase();
438 		var cellLetter = currentLetter.substr(0,1).toLowerCase();
439 		if (itemLetter == cellLetter) {
440 			inCurrentBar = true;
441 		}
442 		else if(AjxStringUtil.isDigit(cellLetter) && AjxStringUtil.isDigit(itemLetter)) {
443 			//handles "123" in alphabet bar
444 			inCurrentBar = true;
445 		}
446 		else if (currentLetter.toLowerCase() == "a-z" && itemLetter.match("[a-z]")) {
447 			//handle A-Z cases for certain locales
448 			inCurrentBar = true;
449 		}
450 	  }
451   }
452   return inCurrentBar;
453 };

455 /**
456  * @private
457  */
458 ZmContactAlphabetBar.prototype._createHtml =
459 function() {
460 	this._alphabetBarId = this._htmlElId + "_alphabet";
461 	var alphabet = ZmMsg.alphabet.split(",");
462 
463 	var subs = {
464 		id: 			this._htmlElId,
465 		alphabet: 		alphabet,
466 		numLetters: 	alphabet.length,
467 		sortVals:		ZmContactAlphabetBar._parseSortVal(ZmMsg.alphabetSortValue),
468 		endSortVals:	ZmContactAlphabetBar._parseSortVal(ZmMsg.alphabetEndSortValue)
469 	};
470 
471 	this.getHtmlElement().innerHTML = AjxTemplate.expand("abook.Contacts#ZmAlphabetBar", subs);
472 };
473 
474 ZmContactAlphabetBar._parseSortVal =
475 function(sortVal) {
476 	if (!sortVal) {
477 		return {};
478 	}
479 	var sortMap = {};
480 	var values = sortVal.split(",");
481 	if (values && values.length) {
482 		for (var i = 0; i < values.length; i++) {
483 			var parts = values[i].split(":");
484 			sortMap[parts[0]] = parts[1];
485 		}
486 	}
487 	return sortMap;
488 };
