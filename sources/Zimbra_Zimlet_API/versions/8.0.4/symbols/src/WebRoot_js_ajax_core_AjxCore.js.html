 21 AjxCore = function() {}

 23 AjxCore._objectIds = [null];
 24 
 25 /**
 26  * @deprecated
 27  * 
 28  * The association is now made by DwtControl using the element's ID, so a separate
 29  * unique ID is not needed.
 30  */
 31 AjxCore.assignId = 
 32 function(anObject) {
 33 	var myId = AjxCore._objectIds.length;
 34 	AjxCore._objectIds[myId]= anObject;
 35 	return myId;
 36 };
 37 
 38 /**
 39  * @deprecated
 40  */
 41 AjxCore.unassignId = 
 42 function(anId) {
 43 	AjxCore._objectIds[anId]= null;
 44 };
 45 
 46 /**
 47  * @deprecated
 48  */
 49 AjxCore.objectWithId = 
 50 function(anId) {
 51 	return AjxCore._objectIds[anId];
 52 };

 54 /**
 55  * Adds a listener to an element, for the given event name.
 56  */
 57 AjxCore.addListener = 
 58 function(eventSource, eventName, action) {
 59 	eventSource = AjxCore._getEventSource(eventSource);
 60 	var listenerStruct = AjxCore._getListenerStruct(eventSource, eventName, true);
 61 	listenerStruct.list[listenerStruct.list.length] = action;
 62 };
 63 
 64 /**
 65  * sets a one time event handler for the given eventName.
 66  */
 67 AjxCore.setEventHandler = 
 68 function(eventSource, eventName, action) {
 69 	eventSource = AjxCore._getEventSource(eventSource);
 70 	var listenerStruct = AjxCore._getListenerStruct(eventSource, eventName, true);
 71 	listenerStruct.single = action;
 72 };
 73 
 74 /**
 75  * removes a listener for a given event
 76  */
 77 AjxCore.removeListener = 
 78 function(eventSource, eventName, action) {
 79 	eventSource = AjxCore._getEventSource(eventSource);
 80 	var listenerStruct = AjxCore._getListenerStruct(eventSource, eventName);
 81 
 82 	if (listenerStruct) {
 83 		var listenerList = listenerStruct.list;
 84 		var len = listenerList.length;
 85 		for (var i = 0; i < len; i++) {
 86 			if (listenerList[i] == action) {
 87 				listenerList[i] = null;
 88 			}
 89 		}
 90 	}
 91 };
 92 
 93 /**
 94  * Removes all listeners for a given eventName, and source
 95  */
 96 AjxCore.removeAllListeners = 
 97 function(eventSource, eventName) {
 98 	eventSource = AjxCore._getEventSource(eventSource);
 99 	var listenerStruct = AjxCore._getListenerStruct(eventSource, eventName);
100 
101 	if (listenerStruct) {
102 		var listenerList = listenerStruct.list;
103 		var len = listenerList.length;
104 		for (var i = 0; i < len; i++) {
105 			listenerList[i] = null;
106 		}
107 	}
108 	AjxCore.unassignId(listenerStruct.id);
109 };

111 /**
112  * notifies listeners of the event. This only needs to be called if
113  * the event is not a standard DOM event. Those types of event callbacks
114  * will be triggered by their event handlers
115  */
116 AjxCore.notifyListeners = 
117 function(eventSource, eventName, arg1) {
118 	eventSource = AjxCore._getEventSource(eventSource);
119 	var listenerStruct = AjxCore._getListenerStruct(eventSource, eventName);
120 	if (listenerStruct) {
121 		eventSource[eventName](arg1);
122 	}
123 };
124 
125 AjxCore._getEventSource = 
126 function(eventSource) {
127 	if (typeof(eventSource) == 'string') {
128 		eventSource = document.getElementById(eventSource);
129 	}
130 	return eventSource;
131 };
132 
133 AjxCore.getListenerStruct = 
134 function (eventSource, eventName) {
135 	return AjxCore._getListenerStruct(eventSource, eventName);
136 };
137 
138 /**
139  * gets the existing struct for the eventSource, or creates a new one.
140  */
141 AjxCore._getListenerStruct = 
142 function(eventSource, eventName, create) {
143 	var listenerStruct = null;
144 	if (eventSource[eventName]) {
145 		var id = eventSource[eventName]._lsListenerStructId;
146 		listenerStruct = AjxCore.objectWithId(id);
147 	} else if (create) {
148 		listenerStruct = AjxCore._setupListener(eventSource, eventName);
149 	}
150 
151 	return listenerStruct;
152 };
153     
154 /**
155  * Creates a listener struct
156  */
157 AjxCore._setupListener = 
158 function(eventSource, eventName, id) {
159 	var listenerStruct = new Object();
160 	listenerStruct.list = new Array();
161 	listenerStruct.single = null;
162 	var id = listenerStruct.id = AjxCore.assignId(listenerStruct);
163 	var handler = AjxCore._createListenerClosure(id);
164 	eventSource[eventName] = handler;
165 	eventSource[eventName]._lsListenerStructId = id;
166 
167 	return listenerStruct;
168 };
169 
170 AjxCore._createListenerClosure = 
171 function(id) {
172 	var closure = function(arg1) {
173 		var listenerStruct = AjxCore.objectWithId(id);
174 		var listenerList = listenerStruct.list;
175 		var len = listenerList.length;
176 		for (var i = 0; i < len; i++) {
177 			var callback = listenerList[i];
178 			if (callback) {
179 				if (typeof(callback) == 'string') {
180 					eval(callback);
181 				} else {
182 					// handle AjxListener callbacks as well as simple functions
183 					if (callback.handleEvent) {
184 						callback.handleEvent(arg1, this);
185 					} else {
186 						callback(arg1, this);
187 					}
188 				}
189 			}
190 		}
191         if (listenerStruct.single) {
192 			var callback = listenerStruct.single;
193 			if (typeof(callback) == 'string') {
194 				eval(callback);
195 			} else {
196 				return callback.handleEvent
197 					? callback.handleEvent(arg1, this)
198 					: callback(arg1, this);
199 			}
200 		}
201 	}
202 	return closure;
203 };

205 /**
206  * Convenience method for adding onload listeners
207  */
208 AjxCore.addOnloadListener = 
209 function(action) {
210 	if (window.onload && (!window.onload._lsListenerStructId)) {
211 		var priorListener = window.onload;
212 		window.onload = null;
213 		AjxCore.addListener(window, "onload", priorListener);
214 	}
215 
216 	AjxCore.addListener(window, "onload", action);
217 };
218

225 		var priorListener = window.onunload;
226 		window.onunload = null;
227 		AjxCore.addListener(window, "onunload", priorListener);
228 	}
229 
230 	AjxCore.addListener(window, "onunload", action);
231 };
