 75 ZmMailListGroup.prototype.getSectionHeader =
 76 function(headerTitle) {
 77     var header = new ZmMailListSectionHeader(this, {headerTitle : headerTitle});
 78     this._sectionHeaders.push(header);
 79     return header.getHeaderHtml();
 80 };

 82 ZmMailListGroup.prototype.resetSectionHeaders =
 83 function() {
 84     for (var i = 0; i < this._sectionHeaders.length; i++) {
 85         var el = this._sectionHeaders[i].getHtmlElement();
 86         el.parentNode.removeChild(el);
 87     }
 88 
 89     this._sectionHeaders = [];
 90 };

134 /**
135  * Return Group object based on groupId
136  * @param {String} groupId
137  * @return {ZmMailListGroup} group object
138  */
139 ZmMailListGroup.getGroup =
140 function(groupId) {
141 
142     switch (groupId){
143         case ZmId.GROUPBY_DATE:
144             return new ZmMailListDateGroup();
145 
146         case ZmId.GROUPBY_SIZE:
147             return new ZmMailListSizeGroup();
148 
149         case ZmId.GROUPBY_PRIORITY:
150             return new ZmMailListPriorityGroup();
151 
152         case ZmId.GROUPBY_FROM:
153             return new ZmMailListFromGroup();
154 
155         default:
156             return null;
157     }
158 
159 };
160 
161 /**
162  * Return the header field based on groupId
163  * @param {String} groupId
164  * @param {boolean} isMultiColumn
165  */
166 ZmMailListGroup.getHeaderField =
167 function(groupId, isMultiColumn) {
168 
169     if (isMultiColumn == false) {
170         return ZmId.FLD_SORTED_BY;
171     }
172 
173     switch (groupId) {
174       case ZmId.GROUPBY_SIZE:
175         return ZmId.FLD_SIZE;
176 
177       case ZmId.GROUPBY_FROM:
178         return ZmId.FLD_FROM;
179 
180       case ZmId.GROUPBY_DATE:
181         return ZmId.FLD_DATE;
182 
183       default:
184         return null;
185     }
186 
187 };
188 
189 /**
190  * Return the group Id based on the sort field
191  * @param {String} sortField
192  */
193 ZmMailListGroup.getGroupIdFromSortField =
194 function(sortField) {
195     switch (sortField) {
196         case ZmId.FLD_FROM:
197             return ZmId.GROUPBY_FROM;
198 
199         case ZmId.FLD_SIZE:
200             return ZmId.GROUPBY_SIZE;
201 
202         case ZmId.FLD_DATE:
203             return ZmId.GROUPBY_DATE;
204 
205         default:
206             return ZmId.GROUPBY_NONE;
207     }
