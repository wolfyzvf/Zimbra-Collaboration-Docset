 23 /**
 24  * Creates the account list.
 25  * @class
 26  * This class is used to store and manage a list of accounts for a mailbox.
 27  *
 28  * @author Parag Shah
 29  */
 30 ZmAccountList = function() {
 31 	this._accounts = {};
 32 	this._count = 0;
 33 	this.visibleAccounts = [];
 34 	this.mainAccount = null;
 35 	this.activeAccount = null;
 36 	this.defaultAccount = null; // the first non-main account.
 37 
 38 	this._evtMgr = new AjxEventMgr();
 39 };

 61 /**
 62  * Gets the number of visible accounts for this mailbox.
 63  *
 64  * @param	{Boolean}	includeInvisible	if <code>true</code>, include the number of invisible accounts for this mailbox

 72 /**
 73  * Adds the account.
 74  * 
 75  * @param	{ZmAccount}	account		the account
 76  */
 77 ZmAccountList.prototype.add =
 78 function(account) {
 79 	this._accounts[account.id] = account;
 80 	this._count++;
 81 
 82 	if (account.visible || account.id == ZmAccountList.DEFAULT_ID) {
 83 		this.visibleAccounts.push(account);
 84 	}
 85 
 86 	if (account.id == ZmAccountList.DEFAULT_ID) {
 87 		this.mainAccount = account;
 88 	}
 89 };

144 /**
145  * Gets the cumulative item count of all accounts for the given folder ID.
146  *
147  * @param {String}	folderId		the folder id
148  * @param {Boolean}	checkUnread		if <code>true</code>, checks the unread count instead of item count
149  * @return	{int}	the item count
150  */
151 ZmAccountList.prototype.getItemCount =
152 function(folderId, checkUnread) {
153 	var count = 0;
154 	for (var i = 0; i < this.visibleAccounts.length; i++) {
155 		var acct = this.visibleAccounts[i];
156 		if (acct.isMain) { continue; } // local account should never have drafts
157 
158 		var fid = ZmOrganizer.getSystemId(folderId, acct);
159 		var folder = appCtxt.getById(fid);
160 		if (folder) {
161 			count += (checkUnread ? folder.numUnread : folder.numTotal);
162 		}
163 	}
164 
165 	return count;
166 };

168 /**
169  * Generates a query.
170  * 
171  * @param {String}	folderId		the folder id
172  * @param	{Array}	types		the types
173  * @return	{String}	the query
174  */
175 ZmAccountList.prototype.generateQuery =
176 function(folderId, types) {
177 	// XXX: for now, let's just search for *one* type at a time
178 	var type = types && types.get(0);
179 	var query = [];
180 	var list = this.visibleAccounts;
181 	var fid = folderId || ZmOrganizer.ID_ROOT;
182 	var syntax = folderId ? "inid" : "underid";
183 	for (var i = 0; i < list.length; i++) {
184 		var acct = list[i];
185 
186 		// dont add any apps not supported by this account
187 		if ((type && !acct.isAppEnabled(ZmItem.APP[type])) || acct.isMain) { continue; }
188 
189 		var part = [syntax, ':"', ZmOrganizer.getSystemId(fid, acct, true), '"'];
190 		query.push(part.join(""));
191 	}
192     if(fid == ZmOrganizer.ID_ROOT) {
193         query.push([syntax, ':"', appCtxt.accountList.mainAccount.id, ':', fid, '"'].join(""));
194     }
195 	DBG.println(AjxDebug.DBG2, "query = " + query.join(" OR "));
196 	return (query.join(" OR "));
197 };

199 /**
200  * Loads each visible account serially by requesting the following requests from
201  * the server in a batch request:
202  * 
203  * <ul>
204  * <li><code><GetInfoRequest></code></li>
205  * <li><code><GetTafReqyuest></code></li>
206  * <li><code><GetFolderRequest></code></li>
207  * </ul>
208  * 
209  * @param {AjxCallback}	callback		the callback to trigger once all accounts have been loaded
210  */
211 ZmAccountList.prototype.loadAccounts =
212 function(callback) {
213 	var list = (new Array()).concat(this.visibleAccounts);
214 	this._loadAccount(list, callback);
215 };
216 
217 /**
218  * @private
219  */
220 ZmAccountList.prototype._loadAccount =
221 function(accounts, callback) {
222 	var acct = accounts.shift();
223 	if (acct) {
224 		var respCallback = new AjxCallback(this, this._loadAccount, [accounts, callback]);
225 		acct.load(respCallback);
226 	} else {
227 		// do any post account load initialization
228 		ZmOrganizer.HIDE_EMPTY[ZmOrganizer.TAG] = true;
229 		ZmOrganizer.HIDE_EMPTY[ZmOrganizer.SEARCH] = true;
230 
231 		// enable compose based on whether at least one account supports smtp
232 		for (var i = 0; i < this.visibleAccounts.length; i++) {
233 			if (appCtxt.get(ZmSetting.OFFLINE_SMTP_ENABLED, null, this.visibleAccounts[i])) {
234 				appCtxt.set(ZmSetting.OFFLINE_COMPOSE_ENABLED, true, null, null, true);
235 				break;
236 			}
237 		}
238 
239 		if (callback) {
240 			callback.run();
241 		}
242 	}
243 };

245 /**
246  * Sets the given account as the active one, which will then be used when fetching
247  * any account-specific data such as settings or folder tree.
248  *
249  * @param {ZmZimbraAccount}	account		the account to make active
250  * @param {Boolean}	skipNotify		if <code>true</code>, skip notify
251  */
252 ZmAccountList.prototype.setActiveAccount =
253 function(account, skipNotify) {
254 	this.activeAccount = account;
255 
256 	this._evt = this._evt || new ZmEvent();
257 	this._evt.account = account;
258 
259 	if (!skipNotify) {
260 		this._evtMgr.notifyListeners("ACCOUNT", this._evt);
261 	}
262 };

275 /**
276  * Checks if any of the non-main, visible accounts is currently doing an initial sync.
277  * 
278  * @return	{Boolean}	<code>true</code> if any of the non-main accounts are doing initial sync
279  */
280 ZmAccountList.prototype.isInitialSyncing =
281 function() {
282 	for (var i = 0; i < this.visibleAccounts.length; i++) {
283 		var acct = this.visibleAccounts[i];
284 		if (acct.isMain) { continue; }
285 
286 		if (acct.isOfflineInitialSync()) {
287 			return true;
288 		}
289 	}
290 
291 	return false;
292 };

294 /**
295  * Returns true if any of the visible accounts have the given status
296  *
297  * @param 	{String}		status 		Status to check for
298  */
299 ZmAccountList.prototype.isSyncStatus =
300 function(status) {
301 	for (var i = 0; i < this.visibleAccounts.length; i++) {
302 		var acct = this.visibleAccounts[i];
303 		if (acct.isMain) { continue; }
304 
305 		if (acct.status == status) {
306 			return true;
307 		}
308 	}
309 
310 	return false;
311 };

313 /**
314  * Checks if there is at least one of the given account types in the
315  * account list. Note: if the given account type is ZCS, the local parent
316  * account is NOT included when searching the account list.
317  *
318  * @param {String}	type	the type of account to check
319  * @return	{Boolean}	<code>true</code> if the account exists
320  */
321 ZmAccountList.prototype.accountTypeExists =
322 function(type) {
323 	for (var i = 0; i < this.visibleAccounts.length; i++) {
324 		var acct = this.visibleAccounts[i];
325 		if (type == ZmAccount.TYPE_ZIMBRA && acct.isMain) { continue; }
326 		if (acct.type == type) { return true; }
327 	}
328 
329 	return false;
330 };

343 /**
344  * @private
345  */
346 ZmAccountList.prototype._sendSync =
347 function(accounts, callback) {
348 	var acct = accounts.shift();
349 	if (acct) {
350 		if (!acct.isMain) { // skip the main account
351 			acct.sync();
352 		}
353 		AjxTimedAction.scheduleAction(new AjxTimedAction(this, this._sendSync, [accounts, callback]), 500);
354 	} else {
355 		if (callback) {
356 			callback.run();
357 		}
358 	}
359 };

361 /**
362  * Creates the main account and all its children. In the normal case, the "main"
363  * account is the only account, and represents the user who logged in. If family
364  * mailbox is enabled, that account is a parent account with dominion over child
365  * accounts. If offline, the main account is the "local" account.
366  *
367  * @param {ZmSettings}	settings	the settings for the main account
368  * @param {Object}	obj		the JSON obj containing meta info about the main account and its children
369  */
370 ZmAccountList.prototype.createAccounts =
371 function(settings, obj) {
372 	// first, replace the dummy main account with real information
373 	var account = appCtxt.accountList.mainAccount;
374 	account.id = obj.id;
375 	account.name = obj.name;
376 	account.isMain = true;
377 	account.isZimbraAccount = true;
378 	account.loaded = true;
379 	account.visible = true;
380 	account.settings = settings;
381 	account.type = ZmAccount.TYPE_ZIMBRA;
382 	account.icon = "AccountZimbra";
383 	account.active = true; // always set active for main/parent account
384 
385 	this._accounts[account.id] = account;
386 	delete this._accounts[ZmAccountList.DEFAULT_ID];
387 
388 	this.setActiveAccount(account);
389 
390 	if (appCtxt.isOffline) {
391 		account.displayName = ZmMsg.localFolders;
392 	}
393 
394 	// second, create all child accounts if applicable
395 	var childAccounts = obj.childAccounts && obj.childAccounts.childAccount;
396 	if (childAccounts) {
397 		for (var i = 0; i < childAccounts.length; i++) {
398 			this.add(ZmZimbraAccount.createFromDom(childAccounts[i]));
399 		}
400 
401 		// set global vars per number of child accounts
402 		appCtxt.multiAccounts = this.size() > 1;
403 		appCtxt.isFamilyMbox = appCtxt.multiAccounts && !appCtxt.isOffline;
404 
405 		this.defaultAccount = appCtxt.isFamilyMbox ? this.mainAccount : this.visibleAccounts[1];
406 	}
407 };

436 /**
437  * Gets the tool tip for the folder.
438  * 
439  * @param	{String}	folderId	the folder id
440  * @return	{String}	the tool tip
441  */
442 ZmAccountList.prototype.getTooltipForVirtualFolder =
443 function(folderId) {
444 	var numTotal = 0;
445 	var sizeTotal = 0;
446 
447 	for (var i = 0; i < this.visibleAccounts.length; i++) {
448 		var acct = this.visibleAccounts[i];
449 		var fid = ZmOrganizer.getSystemId(folderId, acct);
450 		var folder = appCtxt.getById(fid);
451 		if (folder) {
452 			numTotal += folder.numTotal;
453 			sizeTotal += folder.sizeTotal;
454 		}
455 	}
456 
457 	var subs = {
458 		itemText: ZmMsg.messages,
459 		numTotal: numTotal,
460 		sizeTotal: sizeTotal
461 	};
462 
463 	return AjxTemplate.expand("share.App#FolderTooltip", subs);
464 };
