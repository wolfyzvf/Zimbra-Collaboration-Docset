 22 /**
 23  * Constructor
 24  * @class
 25  * This class is responsible for providing unique, predictable IDs for HTML elements.
 26  * That way, code outside the client can easily locate particular elements.
 27  * <br/>
 28  * <br/>
 29  * Not every element that has an associated JS object will have a known ID. Those are
 30  * allocated only for elements it would be useful to locate: major components of the UI,
 31  * toolbars, buttons, views, menus, some menu items, some selects, and miscellaneous other
 32  * components.
 33  * <br/>
 34  * <br/>
 35  * Unless noted otherwise, a getElementById() on any of the non-skin IDs will return a DIV. One exception
 36  * is input fields. The ID is given to the DwtInputField's actual INPUT, rather than to the
 37  * DIV that contains it. Most other exceptions are table-related: TABLE, TR, and TD.
 38  * <br/>
 39  * <br/>
 40  * There is a simple naming scheme for the IDs themselves. Each ID starts with a "z" followed
 41  * by one to a few letters that indicate the type of object (widget) represented by the element:
 42  * 
 43  * <ul>
 44  * <li><b>z</b> a component that is not a special-purpose widget listed below</li>
 45  * <li><b>ztb</b> 		a toolbar (see {@link ZmId.WIDGET})</li>
 46  * <li><b>zb</b> 		a button (see {@link ZmId.WIDGET_TOOLBAR})</li>
 47  * <li><b>zi</b> 		an input field (see {@link ZmId.WIDGET_INPUT})</li>
 48  * <li><b>zm</b> 		a menu (see {@link ZmId.WIDGET_MENU})</li>
 49  * <li><b>zmi</b> 		a menu item (see {@link ZmId.WIDGET_MENU_ITEM})</li>
 50  * <li><b>zs</b> 		a select (see {@link ZmId.WIDGET_SELECT})</li>
 51  * <li><b>zov</b> 		an overview (see {@link ZmId.WIDGET_OVERVIEW})</li>
 52  * <li><b>zt</b> 		a tree (see {@link ZmId.WIDGET_TREE})</li>
 53  * <li><b>zti</b> 		a tree item (see {@link ZmId.WIDGET_TREE_ITEM})</li>
 54  * <li><b>ztih</b> 	a tree item header (see {@link ZmId.WIDGET_TREE_ITEM_HDR})</li>
 55  *</ul>
 56  *
 57  * The skin defines its own container IDs starting with "skin_", which we provide constants for here.
 58  * 
 59  * @author Conrad Damon
 60  */
 61 ZmId = function() {};

282 /**
283  * Generates a tree item ID based on the underlying organizer and the overview ID (since the same
284  * organizer may be represented as tree items in more than one overview). Some sample IDs:
285  * 
286  * <ul>
287  * <li><b>zti|Mail|2</b> Inbox</li>
288  * <li><b>zti|Mail|172</b>			user-created item in mail overview</li>

296  * @param {String}	overviewId	the unique ID for overview
297  * @param {ZmOrganizer}	organizerId	the ID of the data object backing tree item
298  * @param {constant}	type			the organizer type (for headers only)
299  * @return	{String}	the id
300  */
301 ZmId.getTreeItemId =
302 function(overviewId, organizerId, type) {
303 	if (!organizerId && !type) { return; }
304 	if (type) {
305 		return DwtId.makeId(ZmId.WIDGET_TREE_ITEM_HDR, overviewId, type);
306 	} else {
307 		return DwtId.makeId(ZmId.WIDGET_TREE_ITEM, overviewId, organizerId);
308 	}
309 };

311 /**
312  * Generates an ID for a view that fills the content area, or for a component of that
313  * view. A context should be provided if the view is not a singleton (for example,
314  * message view may appear within several double-pane views). The component name
315  * is not joined with the "|" character in order to preserve backward compatibility.
316  * 
317  * <p>
318  * Examples: <code>zv|COMPOSE zv|TV zv|TV|MSG zv|TV|MSG_hdrTable</code>
319  * </p>
320  * 
321  * @param {constant}	viewId		the view identifier (see <code>ZmId.VIEW_</code> constants)
322  * @param {constant}	component	the component identifier (see <code>ZmId.MV_</code> constants)
323  * @param {constant}	context		the ID of owning view
324  * @return	{String}	the id
325  */
326 ZmId.getViewId =
327 function(viewId, component, context) {
328 	var id = DwtId.makeId(ZmId.WIDGET_VIEW, context, viewId);
329 	return component ? [id, component].join("") : id;
330 };

556 /*
557  * ------------
558  * Message view
559  * ------------
560  * 
561  * A message view displays an email message. There are several different instances of message views, which
562  * makes it necessary to include a context (owning view) to be able to identify each one of them.
563  * 
564  * The function to use is:
565  * 
566  * 		ZmId.getViewId(ZmId.VIEW_MSG, component, context)
567  * 
568  * Since message views are not singletons, a context is always necessary. Omit the component only when getting
569  * an ID for a message view itself.
570  * 
571  * To get the ID for a message view, pass the constant for the message view as well as the context, which can be
572  * ZmId.VIEW_CONVLIST, ZmId.VIEW_CONV, ZmId.VIEW_MSG, or ZmId.VIEW_TRAD:
573  * 
574  * 		ZmId.getViewId(ZmId.VIEW_MSG, null, ZmId.VIEW_TRAD)
575  * 
576  * There are also many components within a message view which are useful to retrieve. To get the ID for a
577  * message view component, pass the component ID (ZmId.MV_*):
578  * 
579  * 		ZmId.getViewId(ZmId.VIEW_MSG, ZmId.MV_HDR_TABLE_TOP_ROW, ZmId.VIEW_TRAD)

585  * itself, pass that ID as the context for the IFRAME:
586  * 
587  * 		var iframeId = DwtId.getIframeId(bodyId);
588  * 
589  * For buttons within msg view, pass the context and operation as usual, and add the identifier for
590  * message view (which distinguishes its buttons from, say, those on the VIEW_TRAD toolbar).
591  * 
592  * 		ZmId.getButtonId(ZmId.VIEW_MSG, ZmId.OP_CLOSE, ZmId.VIEW_CONV)
