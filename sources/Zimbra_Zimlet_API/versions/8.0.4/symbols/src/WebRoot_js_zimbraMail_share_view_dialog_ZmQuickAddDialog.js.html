 20 /**
 21  * Creates a generic quick add dialog (which basically mean it has different 
 22  * than regular dialogs). Dialogs always hang off the main shell since their stacking order
 23  * is managed through z-index. See "dwt.Widgets#DwtSemiModalDialog" template.
 24  * @class
 25  * This class represents a modal dialog which has at least a title and the 
 26  * standard buttons (OK/Cancel) and widgets (i.e. buttons, etc) as necessary.
 27  * 
 28  * @author Parag Shah
 29  * 
 30  * @param {DwtComposite}	parent				the parent widget (the shell)
 31  * @param {String}	title				a title for the dialog
 32  * @param {Array}	standardButtons		a list of standard button IDs (default is [{@link DwtDialog.OK_BUTTON}, {@link DwtDialog.CANCEL_BUTTON}])
 33  * @param {Array}	extraButtons 		any extra buttons to be added in addition to the standard ones
 34  * @param {Object}	loc				where to popup (optional)
 35  * 
 36  * @extends		DwtDialog
 37  */
 38 ZmQuickAddDialog = function(parent, title, standardButtons, extraButtons, loc) {
 39 	if (arguments.length == 0) return;
 40 
 41 	DwtDialog.call(this, {parent:parent, title:title, standardButtons:standardButtons,
 42 						  extraButtons:extraButtons, loc:loc});
 43 };
 44 
 45 ZmQuickAddDialog.prototype = new DwtDialog;
 46 ZmQuickAddDialog.prototype.constructor = ZmQuickAddDialog;
 47 
 48 ZmQuickAddDialog.prototype.toString =
 49 function() {
 50 	return "ZmQuickAddDialog";
 51 };
 52 
 53 //
 54 // Data
 55 //
 56 
 57 ZmQuickAddDialog.prototype.TEMPLATE = "dwt.Widgets#DwtSemiModalDialog";

 63 /**
 64  * Adds a selection listener.
 65  * 
 66  * @param	{String}		buttonId		the button id
 67  * @param	{AjxListener}	listener		the listener
 68  */
 69 ZmQuickAddDialog.prototype.addSelectionListener = 
 70 function(buttonId, listener) {
 71 	this._button[buttonId].addSelectionListener(listener);
 72 };
