 22 /**
 23  * @class
 24  * This class represents a calendar item.
 25  *
 26  * @param	{constant}	type		the item type
 27  * @param	{ZmList}	list		the list
 28  * @param	{int}	id				the task id
 29  * @param	{String}	folderId	the folder id
 30  *
 31  * @extends ZmCalBaseItem
 32  */
 33 ZmCalItem = function(type, list, id, folderId) {
 34 	if (arguments.length == 0) { return; }
 35 
 36 	ZmCalBaseItem.call(this, type, list, id, folderId);
 37 
 38 	this.notesTopPart = null; // ZmMimePart containing children w/ message parts
 39 	this.attachments = null;
 40 	this.viewMode = ZmCalItem.MODE_NEW;
 41 	this._recurrence = new ZmRecurrence(this);
 42 	this._noBusyOverlay = null;
 43     this._sendNotificationMail = true;
 44     this.identity = null;
 45     this.isProposeTimeMode = false;
 46     this.isForwardMode = false;
 47 	this.alarmActions = new AjxVector();
 48 	this.alarmActions.add(ZmCalItem.ALARM_DISPLAY);
 49 	this._useAbsoluteReminder = false;
 50     this._ignoreVersion=false; //to ignore revision related attributes(ms & rev) during version conflict
 51 };

519 /**
520  * Gets the notes part. This method will walk the notesParts array looking for
521  * the first part that matches given content type.
522  * 
523  * @param	{constant}	contentType		the content type (see {@link ZmMimeTable.TEXT_PLAIN})	
524  * @return	{String}	the content
525  * 
526  * @see	ZmMimeTable
527  */
528 ZmCalItem.prototype.getNotesPart =
529 function(contentType) {
530 	if (this.notesTopPart) {
531 		var ct = contentType || ZmMimeTable.TEXT_PLAIN;
532 		var content = this.notesTopPart.getContentForType(ct);
533 
534 		// if requested content type not found, try the other
535 		if (!content) {
536 			if (ct == ZmMimeTable.TEXT_PLAIN) {
537 				var div = document.createElement("div");
538 				content = this.notesTopPart.getContentForType(ZmMimeTable.TEXT_HTML);
539 				div.innerHTML = content || "";
540 				var text = AjxStringUtil.convertHtml2Text(div);
541 				return text.substring(1); // above func prepends \n due to div
542 			} else if (ct == ZmMimeTable.TEXT_HTML) {
543 				content = AjxStringUtil.convertToHtml(this.notesTopPart.getContentForType(ZmMimeTable.TEXT_PLAIN));
544 			}
545 		}
546 		return content;
547 	} else {
548 		return this.fragment;
549 	}
550 };

659 /**
660  * Checks if the item is overlapping.
661  * 
662  * @param	{ZmCalItem}	other		the other item to check
663  * @param	{Boolean}	checkFolder	<code>true</code> to check the folder id
664  * @return	{Boolean}	<code>true</code> if the items overlap; <code>false</code> if the items do not overlap or the item folder ids do not match
665  */
666 ZmCalItem.prototype.isOverlapping =
667 function(other, checkFolder) {
668 	if (checkFolder && this.folderId != other.folderId) { return false; }
669 
670 	var tst = this.getStartTime();
671 	var tet = this.getEndTime();
672 	var ost = other.getStartTime();
673 	var oet = other.getEndTime();
674 
675 	return (tst < oet) && (tet > ost);
676 };

2585 		status.errorMessage = (invalid && invalid.length)
2586 			? AjxMessageFormat.format(ZmMsg.apptSendErrorPartial, AjxStringUtil.htmlEncode(invalid.join(", ")))
2587 			: ZmMsg.apptSendErrorAbort;
2588 	} else if(ex.code == ZmCsfeException.MAIL_MESSAGE_TOO_BIG) {
2589         status.errorMessage = (this.type == ZmItem.TASK) ? ZmMsg.taskSaveErrorToobig : ZmMsg.apptSaveErrorToobig;
2590     } else if (ex.code == ZmCsfeException.MAIL_INVITE_OUT_OF_DATE) {
2591         if(!this.isVersionIgnored()){
2592             this.setIgnoreVersion(true);
2593             status.continueSave = true;
2594         }
2595         else{
2596             status.errorMessage = ZmMsg.inviteOutOfDate;
2597             this.setIgnoreVersion(false);
2598         }
2599     } else if (ex.code == ZmCsfeException.MAIL_NO_SUCH_CALITEM) {
2600         status.errorMessage = ex.getErrorMsg([ex.getData("itemId")]);
2601     } else if (ex.code == ZmCsfeException.MAIL_QUOTA_EXCEEDED) {
2602     		if(this.type == ZmItem.APPT){
2603                 status.errorMessage=ZmMsg.errorQuotaExceededAppt;
2604             } else if(this.type == ZmItem.TASK){
2605                 status.errorMessage=ZmMsg.errorQuotaExceededTask;
2606             }
2607     }
2608 
2609     return status;
