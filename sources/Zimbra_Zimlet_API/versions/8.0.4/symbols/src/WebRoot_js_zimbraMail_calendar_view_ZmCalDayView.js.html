110 ZmCalDayView.prototype._updateDays =
111 function() {
112     // When used from the ZmInviteMsgView, the day view requires a unique id - but
113     // underlying views count on the id being the standard ZmId.VIEW_CAL_DAY.  Since
114     // the use from ZmInviteMsgView is read-only, _updateDays is the only superclass
115     // function that we have to fool by using the standard id.
116     var viewId = this.view;
117     this.view = ZmId.VIEW_CAL_DAY;
118     ZmCalColView.prototype._updateDays.call(this);
119     this.view = viewId;
120 }

319 ZmCalDayTabView.prototype._layout =
320 function(refreshApptLayout) {
321 	DBG.println(AjxDebug.DBG2, "ZmCalColView in layout!");
322 	this._updateDays();
323 
324 	var numCols = this._columns.length,
325         sz = this.getSize(),
326         width = sz.x,
327         height = sz.y,
328         hoursWidth = ZmCalColView._HOURS_DIV_WIDTH-1,
329         bodyX = hoursWidth + this._daySepWidth,
330         bodyY,
331         unionX = ZmCalColView._HOURS_DIV_WIDTH,
332         needHorzScroll,
333         scrollFudge,
334         allDayHeadingDiv = document.getElementById(this._allDayHeadingDivId),
335         allDayHeadingDivHeight = Dwt.getSize(allDayHeadingDiv).y,
336         numRows = this._allDayApptsRowLayouts ? (this._allDayApptsRowLayouts.length) : 1,
337         percentageHeight,
338         nearestNoOfRows,
339         allDayScrollHeight,
340         unionSepX;
341 
342 	if (width == 0 || height == 0) { return; }
343 
344     height -= 25;
345 	this._needFirstLayout = false;
346 	bodyX += this._daySepWidth + ZmCalDayTabView._TAB_SEP_WIDTH;
347 
348 	// compute height for hours/grid
349 	this._bodyDivWidth = width - bodyX - ZmCalDayTabView._TAB_SEP_WIDTH;
350 
351 	// size appts divs
352 	this._apptBodyDivHeight = ZmCalColView._DAY_HEIGHT + 1; // extra for midnight to show up
353 	this._apptBodyDivWidth = Math.max(this._bodyDivWidth, this._calcMinBodyWidth(this._bodyDivWidth, numCols));
354 	needHorzScroll = this._apptBodyDivWidth > this._bodyDivWidth;
355 
356 	this._horizontalScrollbar(needHorzScroll);
357 
358 	if (needHorzScroll) this._apptBodyDivWidth -= 18;
359 	scrollFudge = needHorzScroll ? 20 : 0; // need all day to be a little wider then grid
360 
361     if(!this._toggleBtn) {
362         this._setBounds(this._toggleBtnContainerId, 0, Dwt.DEFAULT, hoursWidth+ZmCalDayTabView._UNION_DIV_WIDTH+this._daySepWidth, Dwt.DEFAULT);
363         this._toggleBtn = new DwtButton({parent:this, parentElement: this._toggleBtnContainerId, className: "ZButton ZPicker ZCalToggleBtn"});
364         this._toggleBtn.setText(ZmMsg.calTabsMerge);
365         this._toggleBtn.addListener(DwtEvent.ONCLICK, new AjxListener(this, this._toggleView));
366     }
367 
368 	// column headings
369 	Dwt.setBounds(allDayHeadingDiv, ZmCalDayTabView._UNION_DIV_WIDTH+hoursWidth+1, 0, this._apptBodyDivWidth + scrollFudge, Dwt.DEFAULT);
370 	// div for all day appts
371 	if (this._allDayApptsList && this._allDayApptsList.length > 0) {
372         numRows++;
373     }
374 	this._allDayFullDivHeight = (ZmCalColView._ALL_DAY_APPT_HEIGHT+ZmCalDayTabView._ALL_DAY_APPT_HEIGHT_PAD) * numRows + ZmCalDayTabView._ALL_DAY_APPT_HEIGHT_PAD;
375 
376 	percentageHeight = (this._allDayFullDivHeight/height)*100;
377 	this._allDayDivHeight = this._allDayFullDivHeight;
378 
379 	// if height overflows more than 50% of full height set its height
380 	// to nearest no of rows which occupies less than 50% of total height
381 	if (percentageHeight > 50) {
382 		nearestNoOfRows = Math.floor((0.50*height-ZmCalDayTabView._ALL_DAY_APPT_HEIGHT_PAD)/(ZmCalColView._ALL_DAY_APPT_HEIGHT+ZmCalDayTabView._ALL_DAY_APPT_HEIGHT_PAD));
383 		this._allDayDivHeight = (ZmCalColView._ALL_DAY_APPT_HEIGHT+ZmCalDayTabView._ALL_DAY_APPT_HEIGHT_PAD) * nearestNoOfRows + ZmCalDayTabView._ALL_DAY_APPT_HEIGHT_PAD;
384 	}
385 
386 	this._setBounds(this._allDayApptScrollDivId, bodyX, allDayHeadingDivHeight+ZmCalDayTabView._TAB_BORDER_MARGIN, this._bodyDivWidth, this._allDayDivHeight+ZmCalDayTabView._TAB_BORDER_MARGIN);
387 	this._setBounds(this._allDayDivId, 0, 0, this._apptBodyDivWidth + scrollFudge, this._allDayFullDivHeight+ZmCalDayTabView._TAB_BORDER_MARGIN);
388 
389 	this._allDayVerticalScrollbar(this._allDayDivHeight != this._allDayFullDivHeight);
390 
391 	// div under year
392 	this._setBounds(this._yearAllDayDivId, 0, ZmCalDayTabView._TAB_BORDER_MARGIN, hoursWidth + ZmCalDayTabView._UNION_DIV_WIDTH + this._daySepWidth-1, this._allDayDivHeight);
393     //this._setBounds(this._yearHeadingDivId, 0, this._daySepWidth-1, hoursWidth + ZmCalDayTabView._UNION_DIV_WIDTH + this._daySepWidth, ZmCalColView._ALL_DAY_APPT_HEIGHT+ZmCalDayTabView._TAB_BORDER_MARGIN+1);
394 	// all day scroll
395 	allDayScrollHeight =  this._allDayDivHeight;
396 	this._setBounds(this._allDayScrollDivId, bodyX, 0, this._bodyDivWidth, allDayScrollHeight);
397 
398 	// horiz separator between all day appts and grid
399 	this._setBounds(this._allDaySepDivId, 0, (this._hideAllDayAppt ? ZmCalColView._DAY_HEADING_HEIGHT : allDayScrollHeight)+2, width, ZmCalColView._ALL_DAY_SEP_HEIGHT);
400 
401 	bodyY =  (this._hideAllDayAppt ? ZmCalColView._DAY_HEADING_HEIGHT : allDayScrollHeight) + ZmCalColView._ALL_DAY_SEP_HEIGHT +  (AjxEnv.isIE ? 0 : 2);
402 
403 	this._bodyDivHeight = height - bodyY;
404 
405 	// hours
406 	this._setBounds(this._hoursScrollDivId, 0, bodyY, hoursWidth, this._bodyDivHeight);
407 
408 	// vert sep between hours and grid
409 	this._setBounds(this._leftApptSepDivId, hoursWidth, bodyY-ZmCalDayTabView._TAB_BORDER_WIDTH, this._daySepWidth, ZmCalColView._DAY_HEIGHT);
410 
411 	// div for scrolling grid
412 	this._setBounds(this._bodyDivId, bodyX, bodyY, this._bodyDivWidth, this._bodyDivHeight);
413 
414 	this._setBounds(this._apptBodyDivId, 0, -1, this._apptBodyDivWidth, this._apptBodyDivHeight);
415 
416     // sep in all day area
417     unionSepX = unionX + ZmCalDayTabView._UNION_DIV_WIDTH;
418     this._setBounds(this._unionHeadingSepDivId, unionSepX, ZmCalDayTabView._TAB_BORDER_MARGIN, this._daySepWidth-1, allDayScrollHeight+1);
419 
420     // div for scrolling union
421     this._setBounds(this._unionGridScrollDivId, unionX, bodyY, ZmCalDayTabView._UNION_DIV_WIDTH, this._bodyDivHeight);
422     this._setBounds(this._unionGridDivId, 0, -1, ZmCalDayTabView._UNION_DIV_WIDTH, this._apptBodyDivHeight+ZmCalColView._HOUR_HEIGHT);
423 
424     // sep in grid area
425     this._setBounds(this._unionGridSepDivId, unionSepX, bodyY-ZmCalDayTabView._TAB_BORDER_MARGIN, this._daySepWidth, this._apptBodyDivHeight);
426 
427     this._bodyX = bodyX;
428     this.layoutWorkingHours(this.workingHours);
429 	this._layoutAllDayAppts();
430 
431     this._apptBodyDivOffset   = Dwt.toWindow(document.getElementById(this._apptBodyDivId), 0, 0, null, true);
432     this._apptAllDayDivOffset = Dwt.toWindow(document.getElementById(this._allDayDivId), 0, 0, null, true);
433 
434 
435 	this._layoutAppts();
436     this._layoutUnionData();
437

690 function(appt) {

695 	if (!this._mergedView)

788 /*
789  * compute appt layout for appts that aren't all day
790  */
791 ZmCalDayTabView.prototype._computeApptLayout =
792 function() {
793 //	DBG.println("_computeApptLayout");
794 //	DBG.timePt("_computeApptLayout: start", true);
795 	var layouts = this._layouts = new Array();
796 	var layoutsDayMap = [];
797 	var layoutsAllDay = [];
798 	var list = this.getList();
799 	if (!list) return;
800 
801 	var size = list.size();
802 	if (size == 0) { return; }
803 
804 	var overlap = null;
805 	var overlappingCol = null;
806 
807 	for (var i=0; i < size; i++) {
808 		var ao = list.get(i);
809 
810 		if (ao.isAllDayEvent()) {
811 			continue;
812 		}
813 
814 		var newLayout = { appt: ao, col: 0, maxcol: -1};
815 
816 		overlap = null;
817 		overlappingCol = null;
818 
819 		var asd = ao.startDate;
820 		var aed = ao.endDate;
821 
822 		var asdDate = asd.getDate();
823 		var aedDate = aed.getDate();
824 
825 		var checkAllLayouts = (asdDate != aedDate);
826 		var layoutCheck = [];
827 
828 		// if a appt starts n end in same day, it should be compared only with
829 		// other appts on same day and with those which span multiple days
830 		if (checkAllLayouts) {
831 			layoutCheck.push(layouts);
832 		} else {
833 			layoutCheck.push(layoutsAllDay);
834 			if (layoutsDayMap[asdDate]!=null) {
835 				layoutCheck.push(layoutsDayMap[asdDate]);
836 			}
837 		}
838 
839 		// look for overlapping appts
840 		for (var k = 0; k < layoutCheck.length; k++) {
841 			for (var j=0; j < layoutCheck[k].length; j++) {
842 				var layout = layoutCheck[k][j];
843 				if (ao.isOverlapping(layout.appt, !this._mergedView)) {
844 					if (overlap == null) {
845 						overlap = [];
846 						overlappingCol = [];
847 					}
848 					overlap.push(layout);
849 					overlappingCol[layout.col] = true;
850 					// while we overlap, update our col
851 					while (overlappingCol[newLayout.col]) {
852 						newLayout.col++;
853 					}
854 				}
855 			}
856 		}
857 
858 		// figure out who is on our right
859 		if (overlap != null) {
860 			for (var c in overlap) {
861 				var l = overlap[c];
862 				if (newLayout.col < l.col) {
863 					if (!newLayout.right) newLayout.right = [l];
864 					else newLayout.right.push(l);
865 				} else {
866 					if (!l.right) l.right = [newLayout];
867 					else l.right.push(newLayout);
868 				}
869 			}
870 		}
871 		layouts.push(newLayout);
872 		if (asdDate == aedDate) {
873 			if(!layoutsDayMap[asdDate]) {
874 				layoutsDayMap[asdDate] = [];
875 			}
876 			layoutsDayMap[asdDate].push(newLayout);
877 		} else {
878 			layoutsAllDay.push(newLayout);
879 		}
880 	}
881 
882 	// compute maxcols
883 	for (var i=0; i < layouts.length; i++) {
884 		this._computeMaxCols(layouts[i], -1);
885 		this._layoutMap[this._getItemId(layouts[i].appt)]  = layouts[i];
886 //		DBG.timePt("_computeApptLayout: computeMaxCol "+i, false);
887 	}
888 
889 	delete layoutsAllDay;
890 	delete layoutsDayMap;
891 	delete layoutCheck;
892 	//DBG.timePt("_computeApptLayout: end", false);
893 };
