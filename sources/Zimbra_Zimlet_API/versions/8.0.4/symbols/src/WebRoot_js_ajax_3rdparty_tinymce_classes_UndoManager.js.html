 27 (function(tinymce) {
 28 	var Dispatcher = tinymce.util.Dispatcher;
 29 
 30 	/**
 31 	 * This class handles the undo/redo history levels for the editor. Since the build in undo/redo has major drawbacks a custom one was needed.
 32 	 *
 33 	 * @class tinymce.UndoManager
 34 	 */
 35 	tinymce.UndoManager = function(editor) {
 36 		var self, index = 0, data = [], beforeBookmark, onAdd, onUndo, onRedo;
 37 
 38 		function getContent() {
 39 			// Remove whitespace before/after and remove pure bogus nodes
 40 			return tinymce.trim(editor.getContent({format : 'raw', no_events : 1}).replace(/<span[^>]+data-mce-bogus[^>]+>[\u200B\uFEFF]+<\/span>/g, ''));
 41 		};
 42 
 43 		function addNonTypingUndoLevel() {
 44 			self.typing = false;
 45 			self.add();
 46 		};
 47 
 48 		// Create event instances
 49 		onBeforeAdd = new Dispatcher(self);
 50 		onAdd       = new Dispatcher(self);
 51 		onUndo      = new Dispatcher(self);
 52 		onRedo      = new Dispatcher(self);
 53 
 54 		// Pass though onAdd event from UndoManager to Editor as onChange
 55 		onAdd.add(function(undoman, level) {
 56 			if (undoman.hasUndo())
 57 				return editor.onChange.dispatch(editor, level, undoman);
 58 		});
 59 
 60 		// Pass though onUndo event from UndoManager to Editor
 61 		onUndo.add(function(undoman, level) {
 62 			return editor.onUndo.dispatch(editor, level, undoman);
 63 		});
 64 
 65 		// Pass though onRedo event from UndoManager to Editor
 66 		onRedo.add(function(undoman, level) {
 67 			return editor.onRedo.dispatch(editor, level, undoman);
 68 		});
 69 
 70 		// Add initial undo level when the editor is initialized
 71 		editor.onInit.add(function() {
 72 			self.add();
 73 		});
 74 
 75 		// Get position before an execCommand is processed
 76 		editor.onBeforeExecCommand.add(function(ed, cmd, ui, val, args) {
 77 			if (cmd != 'Undo' && cmd != 'Redo' && cmd != 'mceRepaint' && (!args || !args.skip_undo)) {
 78 				self.beforeChange();
 79 			}
 80 		});
 81 
 82 		// Add undo level after an execCommand call was made
 83 		editor.onExecCommand.add(function(ed, cmd, ui, val, args) {
 84 			if (cmd != 'Undo' && cmd != 'Redo' && cmd != 'mceRepaint' && (!args || !args.skip_undo)) {
 85 				self.add();
 86 			}
 87 		});
 88 
 89 		// Add undo level on save contents, drag end and blur/focusout
 90 		editor.onSaveContent.add(addNonTypingUndoLevel);
 91 		editor.dom.bind(editor.dom.getRoot(), 'dragend', addNonTypingUndoLevel);
 92 		editor.dom.bind(editor.getDoc(), tinymce.isGecko ? 'blur' : 'focusout', function(e) {
 93 			if (!editor.removed && self.typing) {
 94 				addNonTypingUndoLevel();
 95 			}
 96 		});
 97 
 98 		editor.onKeyUp.add(function(editor, e) {
 99 			var keyCode = e.keyCode;
100 
101 			if ((keyCode >= 33 && keyCode <= 36) || (keyCode >= 37 && keyCode <= 40) || keyCode == 45 || keyCode == 13 || e.ctrlKey) {
102 				addNonTypingUndoLevel();
103 			}
104 		});
105 
106 		editor.onKeyDown.add(function(editor, e) {
107 			var keyCode = e.keyCode;
108 
109 			// Is caracter positon keys left,right,up,down,home,end,pgdown,pgup,enter
110 			if ((keyCode >= 33 && keyCode <= 36) || (keyCode >= 37 && keyCode <= 40) || keyCode == 45) {
111 				if (self.typing) {
112 					addNonTypingUndoLevel();
113 				}
114 
115 				return;
116 			}
117 
118 			// If key isn't shift,ctrl,alt,capslock,metakey
119 			if ((keyCode < 16 || keyCode > 20) && keyCode != 224 && keyCode != 91 && !self.typing) {
120 				self.beforeChange();
121 				self.typing = true;
122 				self.add();
123 			}
124 		});
125 
126 		editor.onMouseDown.add(function(editor, e) {
127 			if (self.typing) {
128 				addNonTypingUndoLevel();
129 			}
130 		});
131 
132 		// Add keyboard shortcuts for undo/redo keys
133 		editor.addShortcut('ctrl+z', 'undo_desc', 'Undo');
134 		editor.addShortcut('ctrl+y', 'redo_desc', 'Redo');
135 
136 		self = {
137 			// Explose for debugging reasons
138 			data : data,
139 
140 			/**
141 			 * State if the user is currently typing or not. This will add a typing operation into one undo
142 			 * level instead of one new level for each keystroke.
143 			 *
144 			 * @field {Boolean} typing
145 			 */
146 			typing : false,
147 			
148 			/**
149 			 * This event will fire before a new undo level is added to the undo manager
150 			 * 
151 			 * @event onBeforeAdd
152 			 * @param {tinymce.UndoManager} sender UndoManager instance that is going to add the new level
153 			 * @param {Object} level The new level object containing a bookmark and contents
154 			 */
155 			onBeforeAdd: onBeforeAdd,
156 
157 			/**
158 			 * This event will fire each time a new undo level is added to the undo manager.
159 			 *
160 			 * @event onAdd
161 			 * @param {tinymce.UndoManager} sender UndoManager instance that got the new level.
162 			 * @param {Object} level The new level object containing a bookmark and contents.
163 			 */
164 			onAdd : onAdd,
165 
166 			/**
167 			 * This event will fire when the user make an undo of a change.
168 			 *
169 			 * @event onUndo
170 			 * @param {tinymce.UndoManager} sender UndoManager instance that got the new level.
171 			 * @param {Object} level The old level object containing a bookmark and contents.
172 			 */
173 			onUndo : onUndo,
174 
175 			/**
176 			 * This event will fire when the user make an redo of a change.
177 			 *
178 			 * @event onRedo
179 			 * @param {tinymce.UndoManager} sender UndoManager instance that got the new level.
180 			 * @param {Object} level The old level object containing a bookmark and contents.
181 			 */
182 			onRedo : onRedo,
183 
184 			/**
185 			 * Stores away a bookmark to be used when performing an undo action so that the selection is before
186 			 * the change has been made.
187 			 *
188 			 * @method beforeChange
189 			 */
190 			beforeChange : function() {
191 				beforeBookmark = editor.selection.getBookmark(2, true);
192 			},
193 
194 			/**
195 			 * Adds a new undo level/snapshot to the undo list.
196 			 *
197 			 * @method add
198 			 * @param {Object} l Optional undo level object to add.
199 			 * @return {Object} Undo level that got added or null it a level wasn't needed.
200 			 */
201 			add : function(level) {
202 				var i, settings = editor.settings, lastLevel;
203 
204 				level = level || {};
205 				level.content = getContent();
206 				
207 				self.onBeforeAdd.dispatch(self, level);
208 
209 				// Add undo level if needed
210 				lastLevel = data[index];
211 				if (lastLevel && lastLevel.content == level.content)
212 					return null;
213 
214 				// Set before bookmark on previous level
215 				if (data[index])
216 					data[index].beforeBookmark = beforeBookmark;
217 
218 				// Time to compress
219 				if (settings.custom_undo_redo_levels) {
220 					if (data.length > settings.custom_undo_redo_levels) {
221 						for (i = 0; i < data.length - 1; i++)
222 							data[i] = data[i + 1];
223 
224 						data.length--;
225 						index = data.length;
226 					}
227 				}
228 
229 				// Get a non intrusive normalized bookmark
230 				level.bookmark = editor.selection.getBookmark(2, true);
231 
232 				// Crop array if needed
233 				if (index < data.length - 1)
234 					data.length = index + 1;
235 
236 				data.push(level);
237 				index = data.length - 1;
238 
239 				self.onAdd.dispatch(self, level);
240 				editor.isNotDirty = 0;
241 
242 				return level;
243 			},
244 
245 			/**
246 			 * Undoes the last action.
247 			 *
248 			 * @method undo
249 			 * @return {Object} Undo level or null if no undo was performed.
250 			 */
251 			undo : function() {
252 				var level, i;
253 
254 				if (self.typing) {
255 					self.add();
256 					self.typing = false;
257 				}
258 
259 				if (index > 0) {
260 					level = data[--index];
261 
262 					editor.setContent(level.content, {format : 'raw'});
263 					editor.selection.moveToBookmark(level.beforeBookmark);
264 
265 					self.onUndo.dispatch(self, level);
266 				}
267 
268 				return level;
269 			},
270 
271 			/**
272 			 * Redoes the last action.
273 			 *
274 			 * @method redo
275 			 * @return {Object} Redo level or null if no redo was performed.
276 			 */
277 			redo : function() {
278 				var level;
279 
280 				if (index < data.length - 1) {
281 					level = data[++index];
282 
283 					editor.setContent(level.content, {format : 'raw'});
284 					editor.selection.moveToBookmark(level.bookmark);
285 
286 					self.onRedo.dispatch(self, level);
287 				}
288 
289 				return level;
290 			},
291 
292 			/**
293 			 * Removes all undo levels.
294 			 *
295 			 * @method clear
296 			 */
297 			clear : function() {
298 				data = [];
299 				index = 0;
300 				self.typing = false;
301 			},
302 
303 			/**
304 			 * Returns true/false if the undo manager has any undo levels.
305 			 *
306 			 * @method hasUndo
307 			 * @return {Boolean} true/false if the undo manager has any undo levels.
308 			 */
309 			hasUndo : function() {
310 				return index > 0 || this.typing;
311 			},
312 
313 			/**
314 			 * Returns true/false if the undo manager has any redo levels.
315 			 *
316 			 * @method hasRedo
317 			 * @return {Boolean} true/false if the undo manager has any redo levels.
318 			 */
319 			hasRedo : function() {
320 				return index < data.length - 1 && !this.typing;
321 			}
322 		};
323 
324 		return self;
325 	};
326 })(tinymce);
