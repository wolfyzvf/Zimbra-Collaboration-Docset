 16 /**
 17  * Creates a new reminder controller to manage the reminder dialog and status area.
 18  * @class
 19  *
 20  * This controller uses the following timed actions:
 21  * <ol>
 22  * <li>one for refreshing our "cache" of upcoming appts to notify on</li>
 23  * <li>one for when to next popup the reminder dialog. 
 24  *    by default, next appt start time minus lead time pref (i..e, 5 minutes before).
 25  *    but, also could be controlled by snooze prefs.</li>
 26  * </ol>
 27  * 
 28  * @param	{ZmCalViewController}		calController		the controller
 29  * 
 30  */
 31 ZmReminderController = function(calController, apptType) {
 32 	this._calController = calController;
 33     this._apptType = apptType;
 34 	this._apptState = {};	// keyed on appt.getUniqueId(true)
 35     this._cacheMap = {};    
 36 	this._cachedAppts = new AjxVector(); // set of appts in cache from refresh
 37 	this._activeAppts = new AjxVector(); // set of appts we are actively reminding on
 38 	this._oldAppts = new AjxVector(); // set of appts which are olde and needs silent dismiss    
 39 	this._housekeepingTimedAction = new AjxTimedAction(this, this._housekeepingAction);
 40 	this._refreshTimedAction = new AjxTimedAction(this, this.refresh);
 41 };
 42 
 43 ZmReminderController.prototype.constructor = ZmReminderController;
 44 
 45 /**
 46  * Defines the "active" reminder state.
 47  */
 48 ZmReminderController._STATE_ACTIVE = 1; // appt was in reminder, never dismissed
 49 /**
 50  * Defines the "dismissed" reminder state.
 51  */
 52 ZmReminderController._STATE_DISMISSED = 2; // appt was in reminder, and was dismissed
 53 /**
 54  * Defines the "snoozed" reminder state.
 55  */
 56 ZmReminderController._STATE_SNOOZED = 3; // appt was in reminder, and was snoozed
 57 
 58 ZmReminderController._CACHE_RANGE = 24; // range of appts to grab 24 hours (-1, +23)
 59 ZmReminderController._CACHE_REFRESH = 16; // when to grab another range

 62 function() {
 63 	return "ZmReminderController";

 66 /**
 67  * called when: (1) app first loads, (2) on refresh blocks, (3) after appt cache is cleared. Our
 68  * _apptState info will keep us from popping up the same appt again if we aren't supposed to
 69  * (at least for the duration of the app)
 70  * 
 71  * @private
 72  */
 73 ZmReminderController.prototype.refresh =
 74 function() {
 75 	this._searchTimeRange = this.getSearchTimeRange();
 76     AjxDebug.println(AjxDebug.REMINDER, "reminder search time range: " + this._searchTimeRange.start + " to " + this._searchTimeRange.end);
 77 
 78 	var params = this.getRefreshParams();
 79 	this._calController.getApptSummaries(params);
 80 	
 81 	// cancel outstanding refresh, since we are doing one now, and re-schedule a new one
 82 	if (this._refreshActionId) {
 83 		AjxTimedAction.cancelAction(this._refreshActionId);
 84 	}
 85     AjxDebug.println(AjxDebug.REMINDER, "reminder refresh");
 86 	this._refreshActionId = AjxTimedAction.scheduleAction(this._refreshTimedAction, (AjxDateUtil.MSEC_PER_HOUR * ZmReminderController._CACHE_REFRESH));
 87 };

 89 /**
 90  * Gets the search time range.
 91  * 
 92  * @return	{Hash}	a hash of parameters
 93  */
 94 ZmReminderController.prototype.getSearchTimeRange =
 95 function() {
 96 	var endOfDay = new Date();
 97 	endOfDay.setHours(23,59,59,999);
 98 
 99 	//grab a week's appt backwards
100 	var end = new Date(endOfDay.getTime());
101 	endOfDay.setDate(endOfDay.getDate()-7);
102 
103 	var start = endOfDay;
104 	start.setHours(0,0,0, 0);
105 
106 	return { start: start.getTime(), end: end.getTime() };
107 };

109 ZmReminderController.prototype.getRefreshParams =
110 function() {
111 	
112 	var timeRange = this.getSearchTimeRange();
113 	return {
114 		start: timeRange.start,
115 		end: timeRange.end,
116 		fanoutAllDay: false,
117 		folderIds: this._apptType ==
118             "appt" ? this._calController.getReminderCalendarFolderIds() :
119                      this._calController.getCheckedCalendarFolderIds(true),
120 		callback: (new AjxCallback(this, this._refreshCallback)),
121 		includeReminders: true
122 	};
123 };

141 /**
142  * called after we get upcoming appts from server. Save list,
143  * and call housekeeping.
144  *
145  * @private
146  */
147 ZmReminderController.prototype._refreshCallback =
148 function(list) {
149 	if (this._refreshDelay > 0) {
150 		AjxTimedAction.scheduleAction(new AjxTimedAction(this, this._refreshCallback, [list]), this._refreshDelay);
151 		this._refreshDelay = 0;
152 		return;
153 	}
154 
155 	if (list instanceof ZmCsfeException) {
156 		this._calController._handleError(list, new AjxCallback(this, this._maintErrorHandler));
157 		return;
158 	}
159 
160 	var newList = new AjxVector();
161 	this._cacheMap = {};
162 
163 	// filter recurring appt instances, the alarmData is common for all the instances
164 	var size = list.size();
165 	for (var i = 0; i < size; i++) {
166 		var appt = list.get(i);
167 		var id = appt.id;
168         var hasAlarm = appt.recurring ? appt.isAlarmInstance() : appt.hasAlarmData();
169 		if (hasAlarm) {
170             var alarmData = appt.getAlarmData();
171             alarmData = (alarmData && alarmData.length > 0) ? alarmData[0] : {};
172             AjxDebug.println(AjxDebug.REMINDER, appt.name + " :: " + appt.startDate + " :: " + appt.endDate + " :: " + appt.recurring + " :: " + appt.isException + " :: " + alarmData.nextAlarm + " :: " + alarmData.alarmInstStart);
173 			if (!this._cacheMap[id]) {
174 				this._cacheMap[id] = true;
175 				newList.add(appt);
176 			}
177 		}
178 	}
179 
180 	this._cachedAppts = newList.clone();
181 	this._cachedAppts.sort(ZmCalBaseItem.compareByTimeAndDuration);
182 	this._activeAppts.removeAll();
183 
184 	// cancel outstanding timed action and update now...
185 	this._cancelHousekeepingAction();
186 	this._housekeepingAction();
187 };

189 ZmReminderController.prototype.updateCache =
190 function(list) {
191 	if (!list) { return; }
192 
193 	if (!this._cachedAppts) {
194 		this._cachedAppts = new AjxVector();
195 	}
196 
197     AjxDebug.println(AjxDebug.REMINDER, "updating reminder cache...");
198 	var srchRange = this.getSearchTimeRange();
199 	var count = 0;
200 
201 	// filter recurring appt instances, the alarmData is common for all the instances
202 	var size = list.size();
203 	for (var i = 0; i < size; i++) {
204 		var appt = list.get(i);
205 		var id = appt.id;
206 		if(appt.hasAlarmData() && !this._cacheMap[id] && appt.isStartInRange(srchRange.start, srchRange.end)) {
207 			this._cacheMap[id] = true;
208 			this._cachedAppts.add(appt);
209 			count++;
210 		}
211 	}
212 
213     AjxDebug.println(AjxDebug.REMINDER, "new appts added to reminder cache :" + count);
214 };

221 /**
222  * go through list to see if we should add any cachedAppts to activeAppts and
223  * popup the dialog or not.
224  * 
225  * @private
226  */
227 ZmReminderController.prototype._housekeepingAction =
228 function() {
229     AjxDebug.println(AjxDebug.REMINDER, "reminder house keeping action...");
230 	var rd = this.getReminderDialog();
231 	if (ZmCsfeCommand.noAuth) {
232         AjxDebug.println(AjxDebug.REMINDER, "reminder check: no auth token, bailing");
233 		if (rd && rd.isPoppedUp()) {
234 			rd.popdown();
235 		}
236 		return;
237 	}
238 
239 	if (this._searchTimeRange) {
240 		var newTimeRange = this.getSearchTimeRange();
241 		var diff = newTimeRange.end - this._searchTimeRange.end;
242 		if (diff > AjxDateUtil.MSEC_PER_HOUR) {
243             AjxDebug.println(AjxDebug.REMINDER, "time elapsed - refreshing reminder cache");
244 			this._searchTimeRange = null;
245 			this.refresh();
246 			return;
247 		}
248 	}
249 
250 	var cachedSize = this._cachedAppts.size();
251 	var activeSize = this._activeAppts.size();
252 	if (cachedSize == 0 && activeSize == 0) {
253         AjxDebug.println(AjxDebug.REMINDER, "no appts - empty cached and active list");
254 		this._housekeepingActionId = AjxTimedAction.scheduleAction(this._housekeepingTimedAction, 60*1000);
255 		return;
256 	}
257 
258 	var numNotify = 0;
259 	var toRemove = [];
260 
261 	for (var i=0; i < cachedSize; i++) {
262 		var appt = this._cachedAppts.get(i);
263 
264 		if (!appt || appt.ptst == ZmCalBaseItem.PSTATUS_DECLINED) {
265 			toRemove.push(appt);
266 		} else if (appt.isAlarmInRange()) {
267 			var uid = appt.getUniqueId(true);
268 			var state = this._apptState[uid];
269 			var addToActiveList = false;
270 			if (state == ZmReminderController._STATE_DISMISSED) {
271 				// just remove themn
272 			} else if (state == ZmReminderController._STATE_ACTIVE) {
273 				addToActiveList = true;
274 			} else {
275 				// we need to notify on this one
276 				numNotify++;
277 				addToActiveList = true;
278 				this._apptState[uid] = ZmReminderController._STATE_ACTIVE;
279 			}
280 
281 			if (addToActiveList) {
282 				toRemove.push(appt);
283 				if (!appCtxt.get(ZmSetting.CAL_SHOW_PAST_DUE_REMINDERS) && appt.isAlarmOld()) {
284 					numNotify--;
285 					this._oldAppts.add(appt);
286 				} else {
287 					this._activeAppts.add(appt);
288 				}
289 			}
290 		}
291 	}
292 
293 	// remove any appts in cachedAppts that are no longer supposed to be in there	
294 	// need to do this here so we don't screw up iteration above
295 	for (var i = 0; i < toRemove.length; i++) {
296 		this._cachedAppts.remove(toRemove[i]);
297 	}
298 
299 	// if we have any to notify on, do it
300 	if (numNotify || rd.isPoppedUp()) {
301 		if (this._activeAppts.size() == 0 && rd.isPoppedUp()) {
302             AjxDebug.println(AjxDebug.REMINDER, "popping down reminder dialog");
303             rd.popdown();
304 		} else {
305             AjxDebug.println(AjxDebug.REMINDER, "initializing reminder dialog");
306 			rd.initialize(this._activeAppts);
307 			if (!rd.isPoppedUp()) rd.popup();
308 		}
309 	}
310 
311     AjxDebug.println(AjxDebug.REMINDER, "no of appts active:" + this._activeAppts.size() + ", no of appts cached:" + cachedSize);
312 
313 	if (this._oldAppts.size() > 0) {
314 		this.dismissAppt(this._oldAppts, new AjxCallback(this, this._silentDismissCallback));
315 	}
316 
317 	// need to schedule housekeeping callback, ideally right before next _cachedAppt start time - lead,
318 	// for now just check once a minute...
319 	this._housekeepingActionId = AjxTimedAction.scheduleAction(this._housekeepingTimedAction, 60*1000);
320 };

322 ZmReminderController.prototype._silentDismissCallback =
323 function(list) {
324 	var size = list.size();
325 	for (var i = 0; i < size; i++) {
326 		var appt = list.get(i);
327 		if (appt && appt.hasAlarmData()) {
328 			if(appt.isAlarmInRange()) {
329 				this._activeAppts.add(appt);
330 			}
331 		}
332 	}
333 	this._oldAppts.removeAll();
334 
335 	// cancel outstanding timed action and update now...
336 	this._cancelHousekeepingAction();
337 	this._housekeepingAction();
338 };

340 /**
341  * Dismisses an appointment. This method is called when
342  * an appointment (individually or as part of "dismiss all") is removed from reminders.
343  * 
344  * @param	{AjxVector|Array}	list	a list of {@link ZmAppt} objects
345  * @param	{AjxCallback}		callback		a callback
346  */
347 ZmReminderController.prototype.dismissAppt =
348 function(list, callback) {
349 	var appt;
350 	if (!(list instanceof AjxVector)) {
351 		list = AjxVector.fromArray((list instanceof Array)? list: [list]);
352 	}
353 
354 	for (var i=0; i<list.size(); i++) {
355 		var appt = list.get(i);
356 		this._apptState[appt.getUniqueId(true)] = ZmReminderController._STATE_DISMISSED;
357 		this._activeAppts.remove(appt);
358 	}
359 
360 	this.dismissApptRequest(list, callback);
361 };

363 /**
364  * Snoozes the appointments.
365  * 
366  * @param	{AjxVector}	list	a list of {@link ZmAppt} objects
367  * @return	{Array}	an array of snoozed apt ids
368  */
369 ZmReminderController.prototype.snoozeAppt =
370 function(list) {
371 	var snoozedIds = [];
372 	var appt;
373 	var uid;
374 	for (var i = 0; i < list.size(); i++) {
375 		appt = list.get(i);
376 		uid = appt.getUniqueId(true);
377 		this._apptState[uid] = ZmReminderController._STATE_SNOOZED;
378 		snoozedIds.push(uid);
379 		this._activeAppts.remove(appt);
380 		this._cachedAppts.add(appt);
381 	}
382 	return snoozedIds;
383 };

385 ZmReminderController.prototype.dismissApptRequest = 
386 function(list, callback) {
387 	var soapDoc = AjxSoapDoc.create("DismissCalendarItemAlarmRequest", "urn:zimbraMail");
388 
389 	var dismissedAt = (new Date()).getTime();
390 	for (var i = 0; i < list.size(); i++) {
391 		var appt = list.get(i);
392 		var apptNode = soapDoc.set("appt");
393 		apptNode.setAttribute("id", appt.id);
394 		apptNode.setAttribute("dismissedAt", dismissedAt);
395 	}
396 
397 	// for multi-account, always set account name to be the local account
398 	var acct = appCtxt.multiAccounts && appCtxt.accountList.mainAccount;
399 	var params = {
400 		soapDoc: soapDoc,
401 		asyncMode: true,
402 		accountName: (acct ? acct.name : null),
403 		callback: (new AjxCallback(this, this._handleDismissAppt, [list, callback])),
404 		errorCallback: (new AjxCallback(this, this._handleErrorDismissAppt, [list, callback]))
405 	};
406 	appCtxt.getAppController().sendRequest(params);
407 	return true;
408 };

416 ZmReminderController.prototype._handleDismissAppt =
417 function(list, callback, result) {
418 	if (result.isException()) { return; }
419 
420 	var response = result.getResponse();
421 	var dismissResponse = response.DismissCalendarItemAlarmResponse;
422 	var appts = dismissResponse ? dismissResponse.appt : null;
423 	if (!appts) { return; }
424 
425     this._updateApptAlarmData(list, appts);
426 
427 	if (callback) {
428 		callback.run(list);
429 	}
430 };

432 ZmReminderController.prototype._handleErrorDismissAppt =
433 function(list, callback, response) {
434 };

437 ZmReminderController.prototype._updateApptAlarmData =
438 function(apptList, responseAppts) {
439     var updateData = {};
440     for (var i = 0; i < responseAppts.length; i++) {
441         var appt = responseAppts[i];
442         if (appt && appt.calItemId) {
443             updateData[appt.calItemId] = appt.alarmData ? appt.alarmData : {};
444         }
445     }
446 
447     var size = apptList.size();
448     for (var i = 0; i < size; i++) {
449         var appt = apptList.get(i);
450         if (appt) {
451             if (updateData[appt.id]) {
452                 appt.alarmData = (updateData[appt.id] != {}) ? updateData[appt.id] : null;
453             }
454         }
455     }
456 }

472 ZmReminderController.prototype._snoozeApptAction =
473 function(apptList, snoozeMinutes, beforeAppt) {
474     var chosenSnoozeMilliseconds = snoozeMinutes*60*1000;
475     var added = false;
476 
477     var soapDoc = AjxSoapDoc.create("SnoozeCalendarItemAlarmRequest", "urn:zimbraMail");
478     if (beforeAppt) {
479         // Using a before time, relative to the start of each appointment
480         if (!this._beforeProcessor) {
481             this._beforeProcessor = new ZmSnoozeBeforeProcessor(this._apptType);
482         }
483         added = this._beforeProcessor.execute(apptList, chosenSnoozeMilliseconds, soapDoc);
484     } else {
485         // using a fixed untilTime for all appts
486         added = apptList.size() > 0;
487         var untilTime = (new Date()).getTime() + chosenSnoozeMilliseconds;
488         for (var i = 0; i < apptList.size(); i++) {
489             var appt = apptList.get(i);
490             var actionNode = soapDoc.set(this._apptType);
491             actionNode.setAttribute("id", appt.id);
492             actionNode.setAttribute("until", untilTime);
493         }
494     }
495 
496 
497     if (added) {
498         var respCallback = this._handleResponseSnoozeAction.bind(this, apptList, snoozeMinutes);
499         var errorCallback = this._handleErrorResponseSnoozeAction.bind(this);
500         var acct = appCtxt.multiAccounts && appCtxt.accountList.mainAccount;
501         appCtxt.getAppController().sendRequest({
502             soapDoc:       soapDoc,
503             asyncMode:     true,
504             accountName:   acct,
505             callback:      respCallback,
506             errorCallback: errorCallback
507         });
508     }
509 };

512 ZmReminderController.prototype._handleResponseSnoozeAction =
513 function(apptList, snoozeMinutes, result) {
514     if (result.isException()) { return; }
515 
516 	var response = result.getResponse();
517 	var snoozeResponse = response.SnoozeCalendarItemAlarmResponse;
518 	var appts = snoozeResponse ? snoozeResponse[this._apptType] : null;
519 	if (!appts) { return; }
520 
521     this._updateApptAlarmData(apptList, appts);
522 
523     if (snoozeMinutes == 1) {
524 	    // cancel outstanding timed action and update now...
525 	    this._cancelHousekeepingAction();
526 	    this._housekeepingAction();
527     }
528 };
