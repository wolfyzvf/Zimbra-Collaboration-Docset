 21 /**
 22  * Creates the tree
 23  * @class
 24  * This class represents a tree.
 25  * 
 26  * @param	{constant}	type		the type
 27  * @extends	ZmModel
 28  */
 29 ZmTree = function(type) {
 30 
 31 	if (arguments.length == 0) { return; }
 32 	ZmModel.call(this, type);
 33 
 34 	this.type = type;
 35 	this.root = null;
 36 };
 37 
 38 ZmTree.prototype = new ZmModel;
 39 ZmTree.prototype.constructor = ZmTree;

 83 /**
 84  * Gets the item type by name.
 85  *
 86  * @param	{String}	name		the name
 87  * @return	{String}    type of folder
 88  */
 89 //Bug:47848: new method that returns type of the item given its name
 90 ZmTree.prototype.getFolderTypeByName =
 91 function(name){
 92 
 93     //As folder names are case-insensitive
 94     var formattedName = name.toLowerCase();
 95 
 96     //Iterate through folders of loaded apps
 97     var folderList = appCtxt.getTree(ZmOrganizer.FOLDER).asList();
 98     var type;
 99     var i;
100     for(i=0 ; i < folderList.length ; i ++){
101         var currentName = folderList[i].name;
102         if(formattedName == currentName.toLowerCase()){
103             return folderList[i].type;
104         }
105     }
106 
107     // check for _deferredFolders in the apps that have not been loaded
108     var apps = ZmApp.APPS;
109 
110     for(i=0 ; i<apps.length; i++){
111        var currentApp = appCtxt.getApp(apps[i]);
112        var deferredFolders = currentApp && currentApp._deferredFolders;
113        if(!deferredFolders){
114            continue;
115        }
116        var j;
117        for(j=0 ; j < deferredFolders.length ; j++){
118            var currentFolder = deferredFolders[j];
119            var currentName = currentFolder.obj && currentFolder.obj.name;
120             if(formattedName == currentName.toLowerCase()){
121                 return currentFolder.type;
122             }
123        }
124 
125     }
126     // if still not found return type as "Folder"
127     type = ZmMsg.folder;
128     return type;
129 }

171 /**
172  * Gets the unread hash.
173  * 
174  * @param	{Hash}	unread		the unread hash
175  * @return	{Hash} the unread tree as a hash
176  */
177 ZmTree.prototype.getUnreadHash =
178 function(unread) {
179 	if (!unread) {
180 		unread = {};
181 	}
182 	return this.root ? this._getUnreadHash(this.root, unread) : unread;
183 };
184 
185 /**
186  * @private
187  */
188 ZmTree.prototype._addToList =
189 function(organizer, list, options) {
190 	var incRemote = options && options.includeRemote;
191 	var remoteOnly = options && options.remoteOnly;
192 	var isRemote = organizer.isRemote();
193 	if ((!isRemote && !remoteOnly) || (isRemote && (remoteOnly || incRemote))) {
194 		list.push(organizer);
195 	}
196 	var children = organizer.children.getArray();
197     for (var i = 0; i < children.length; i++) {
198         this._addToList(children[i], list, options);
199     }
200 	return list;
201 };
202 
203 /**
204  * @private
205  */
206 ZmTree.prototype._asString =
207 function(organizer, str) {
208 	if (organizer.id) {
209 		str = str + organizer.id;
210 	}
211 	var children = organizer.children.clone().getArray();
212 	if (children.length) {
213 		children.sort(function(a,b){return a.id - b.id;});
214 		str = str + "[";
215 		for (var i = 0; i < children.length; i++) {
216 			if (children[i].id == ZmFolder.ID_TAGS) { // Tags "folder" added when view is set
217 				continue;
218 			}
219 			if (i > 0) {
220 				str = str + ",";
221 			}
222 			str = this._asString(children[i], str);
223 		}
224 		str = str + "]";
225 	}
226 	return str;
227 };
228 
229 /**
230  * @private
231  */
232 ZmTree.prototype._getUnreadHash =
233 function(organizer, unread) {
234 	unread[organizer.id] = organizer.numUnread;
235 	var children = organizer.children.getArray();
236 	for (var i = 0; i < children.length; i++) {
237 		this._getUnreadHash(children[i], unread);
238 	}
239 
240 	return unread;
241 };
