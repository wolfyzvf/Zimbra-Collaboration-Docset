 16 /**
 17  * Creates an empty filter rule. Conditions and actions will need to be added.
 18  * @constructor
 19  * @class
 20  * ZmFilterRule represents a filter rule. A rule includes one or more conditions
 21  * and one or more actions.
 22  *
 23  * @author Conrad Damon
 24  *
 25  * @param {String}	name			the rule name
 26  * @param {Boolean}	active		if <code>true</code>, if the rule is enabled
 27  * @param {Object}	filterActions	the filter action data as raw json object
 28  * @param {Object}	filterTests	the filter conditions data as raw json object
 29  * 
 30  */
 31 ZmFilterRule = function(name, active, filterActions, filterTests) {
 32 	/**
 33      * The name of the filter rule.
 34      * @type String
 35      */
 36 	this.name = name;
 37 	/**
 38 	 * The filter rule actions.
 39 	 * @type	Object
 40 	 */
 41 	this.actions = filterActions || {};
 42 	/**
 43 	 * The filter rules conditions.
 44 	 * @type	Object
 45 	 */
 46 	this.conditions = filterTests || {};
 47 	this.active = (active !== false);
 48 	if (!filterTests) {
 49 		this.setGroupOp();
 50 	}
 51 
 52 	this.id = ZmFilterRule._nextId++;
 53 };

354 /**
355  * This defines a hash of conditions. Each condition is a hash of parameters. The key of the hash
356  * is also known as the condition "subject". It is the field of an email message that 
357  * the condition is tested against.
358  * 
359  * <p>
360  * The condition parameters are:
361  * <ul>
362  * <li><b>subjectMod</b>	Type of input widget for the subjectModifier, which is a specifier or 
363  *				modifier for the subject (such as which address to look at)</li>
364  * <li><b>smOptions</b>		List of possible values for the subjectModifier ({@link ZmFilterRule.TYPE_SELECT})</li>
365  * <li><b>ops</b>			Type of input widget for choosing the comparator</li>
366  * <li><b>opsOptions</b>	List of possible comparators for this subject ({@link ZmFilterRule.TYPE_SELECT} type)</li>
367  * <li><b>value</b>			Type of input widget for specifying the value</li>
368  * <li><b>vOptions</b>		List of possible values ({@link ZmFilterRule.TYPE_SELECT} type)</li>
369  * <li><b>valueMod</b>		Type of input widget for the valueModifier, which is a specifier or 
370  *				modifier for the value (such as units for size)</li>
371  * <li><b>vmOptions</b>		List of possible values for the valueModifier ({@link ZmFilterRule.TYPE_SELECT} type)</li>
372  * </ul>
373  */

707 	if (socialFilters && socialFilters.length) {
708 		for (var i=0; i<socialFilters.length; i++) {

710 				ops.push(ZmFilterRule.OP_SOCIAL_FACEBOOK)

716 				ops.push(ZmFilterRule.OP_SOCIAL_LINKEDIN);
717 			}
718 			else if (socialFilters[i].toLowerCase() == ZmFilterRule.C_SOCIALCAST.toLowerCase()) {
719 				ops.push(ZmFilterRule.OP_SOCIAL_SOCIALCAST);
720 			}
721 		}
722 	}
723 	return ops;

752 ZmFilterRule.prototype.addCondition =
753 function(testType, comparator, value, subjectMod, caseSensitive) {
754 	//In dialog some tests are options under other tests
755 	if (testType == ZmFilterRule.TEST_SOCIAL && comparator) {
756 		testType = ZmFilterRule.OP_SOCIAL_MAP[comparator];
757 	}
758 	else if (testType == ZmFilterRule.TEST_ADDRBOOK && (comparator == ZmFilterRule.OP_IS_ME || comparator == ZmFilterRule.OP_NOT_ME)) {
759 		testType = ZmFilterRule.TEST_ME;
760 		value = null;
761 		comparator = comparator == ZmFilterRule.OP_IS_ME ? ZmFilterRule.OP_IS : ZmFilterRule.OP_NOT_IS;
762 	}
763 	else if (testType == ZmFilterRule.TEST_ADDRBOOK && value == "ranking") {
764 		testType = ZmFilterRule.TEST_RANKING;
765 		value = null;
766 	}
767 	else if (testType == ZmFilterRule.TEST_CONVERSATIONS && value == ZmFilterRule.C_BULK) {
768 		testType = ZmFilterRule.TEST_BULK;
769 		value = null;
770 	}
771 	else if (testType == ZmFilterRule.TEST_CONVERSATIONS && value == ZmFilterRule.C_LIST) {
772 		testType = ZmFilterRule.TEST_LIST;
773 		value = null;
774 	}
775 	else if (testType == ZmFilterRule.TEST_CONVERSATIONS && (value == ZmFilterRule.IMPORTANCE_HIGH || value == ZmFilterRule.IMPORTANCE_NORMAL || value == ZmFilterRule.IMPORTANCE_LOW))  {
776 		testType = ZmFilterRule.TEST_IMPORTANCE;
777 	}
778 	else if (testType == ZmFilterRule.TEST_CONVERSATIONS && (value == ZmFilterRule.READ || value == ZmFilterRule.PRIORITY || value == ZmFilterRule.FLAGGED)) {
779 		testType = ZmFilterRule.TEST_FLAGGED;
780 	}
781 	
782 	if (!this.conditions[testType]) {
783 		this.conditions[testType] = [];
784 	}
785 	
786 	var cdata = ZmFilterRule.getConditionData(testType, comparator, value, subjectMod, caseSensitive);
787 	this.conditions[testType].push(cdata);
788 	return cdata;
789 };

845 // Static methods
846 
847 ZmFilterRule.getConditionData =
848 function(testType, comparator, value, subjectMod, caseSensitive) {
849 	var conditionData = {};
850 
851 	// add subject modifier
852 	if (subjectMod &&
853 		(testType == ZmFilterRule.TEST_HEADER ||
854 		 testType == ZmFilterRule.TEST_HEADER_EXISTS ||
855 		 testType == ZmFilterRule.TEST_ADDRBOOK ||
856 		 testType == ZmFilterRule.TEST_MIME_HEADER ||
857 		 testType == ZmFilterRule.TEST_ADDRESS ||
858 		 testType == ZmFilterRule.TEST_ME ||
859 		 testType == ZmFilterRule.TEST_RANKING))
860 	{
861 		conditionData.header = subjectMod;
862 	}
863 	
864 	var part = "all"; //default
865 	if (testType == ZmFilterRule.TEST_ADDRESS && value && typeof value == "string") {
866 		var valueArr = value.split(";"); 
867 		if (valueArr && valueArr.length > 1) {
868 			value = valueArr[0];
869 			part = valueArr[1];
870 		}
871 	}
872 	
873 	// normalize negative operator and add comparator
874 	var negativeOp;
875 	switch (comparator) {
876 		case ZmFilterRule.OP_NOT_IS:		negativeOp = ZmFilterRule.OP_IS; break;
877 		case ZmFilterRule.OP_NOT_CONTAINS:	negativeOp = ZmFilterRule.OP_CONTAINS; break;
878 		case ZmFilterRule.OP_NOT_MATCHES:	negativeOp = ZmFilterRule.OP_MATCHES; break;
879 		case ZmFilterRule.OP_NOT_EXISTS:	negativeOp = ZmFilterRule.OP_EXISTS; break;
880 		case ZmFilterRule.OP_NOT_UNDER:		negativeOp = ZmFilterRule.OP_UNDER; break;
881 		case ZmFilterRule.OP_NOT_OVER:		negativeOp = ZmFilterRule.OP_OVER; break;
882 		case ZmFilterRule.OP_NOT_BEFORE:	negativeOp = ZmFilterRule.OP_BEFORE; break;
883 		case ZmFilterRule.OP_NOT_AFTER:		negativeOp = ZmFilterRule.OP_AFTER; break;
884 		case ZmFilterRule.OP_NOT_IN:		negativeOp = ZmFilterRule.OP_IN; break;
885         case ZmFilterRule.OP_NOT_REPLIED:   negativeOp = ZmFilterRule.OP_IS_REPLIED; break;
886         case ZmFilterRule.OP_NOT_REQUESTED: negativeOp = ZmFilterRule.OP_IS_REQUESTED; break;
887         case ZmFilterRule.OP_NOT_READRECEIPT: negativeOp = ZmFilterRule.OP_CONTAINS; break;
888 		case ZmFilterRule.OP_NOT_CONV:      negativeOp = true; break;
889 	}
890 	if (negativeOp) {
891 		conditionData.negative = "1";
892 	}
893 	
894 	if (value == ZmFilterRule.IS_NOT_SOCIAL && 
895 		(testType == ZmFilterRule.TEST_FACEBOOK  ||
896 		 testType == ZmFilterRule.TEST_SOCIALCAST || 
897 		 testType == ZmFilterRule.TEST_TWITTER ||
898 		 testType == ZmFilterRule.TEST_LINKEDIN)) {
899 		conditionData.negative = "1";
900 		value = null;
901 	}
902 
903 	var compType = ZmFilterRule.COMP_TEST_MAP[testType];
904 	if (compType) {
905 		conditionData[compType] = ZmFilterRule.OP_VALUE[negativeOp || comparator];
906 	}
907 
908 	// add data value
909 	if (value) {
910 		switch (testType) {
911 			case ZmFilterRule.TEST_ADDRBOOK:	conditionData.type = value; break;
912 			case ZmFilterRule.TEST_SIZE:		conditionData.s = value; break;
913 			case ZmFilterRule.TEST_DATE:		conditionData.d = value; break;
914 			case ZmFilterRule.TEST_CONVERSATIONS: conditionData.where = value; break;
915 			case ZmFilterRule.TEST_IMPORTANCE:  conditionData.imp = value; break;
916 			case ZmFilterRule.TEST_ADDRESS:     conditionData.part = part;
917 												conditionData.value= value;
918 												break;
919 			case ZmFilterRule.TEST_FLAGGED:     conditionData.flagName = value; break;
920 			default:							conditionData.value = value; break;
921 		}
922 	}
923 
924 	if (testType == ZmFilterRule.TEST_INVITE) {
925 	    conditionData.method = [{_content:ZmFilterRule.OP_VALUE[negativeOp || comparator]}];
926 	}
927 	if (caseSensitive != null) {
928 		conditionData.caseSensitive = caseSensitive;
929 	}
930 
931 	return conditionData;
932 };

959     if (!ZmFilterRule.__preConditionsInitialized) {
960         ZmFilterRule.__preConditionsInitialized = true;
961         ZmFilterRule._setPreconditions();
962     }

965 	if (!pre) { return true; }
966

972 	}
973 	return true;

991 		case ZmFilterRule.OP_UNDER:		negativeOp = ZmFilterRule.OP_NOT_UNDER; break;
992 		case ZmFilterRule.OP_OVER:		negativeOp = ZmFilterRule.OP_NOT_OVER; break;
993 		case ZmFilterRule.OP_BEFORE:	negativeOp = ZmFilterRule.OP_NOT_BEFORE; break;
994 		case ZmFilterRule.OP_AFTER:		negativeOp = ZmFilterRule.OP_NOT_AFTER; break;
995 		case ZmFilterRule.OP_IN:		negativeOp = ZmFilterRule.OP_NOT_IN; break;
996         case ZmFilterRule.OP_IS_REPLIED:   negativeOp = ZmFilterRule.OP_NOT_REPLIED; break;
997         case ZmFilterRule.OP_IS_REQUESTED: negativeOp = ZmFilterRule.OP_NOT_REQUESTED; break;
998         case ZmFilterRule.OP_IS_READRECEIPT: negativeOp = ZmFilterRule.OP_NOT_CONTAINS; break;
999 	    case ZmFilterRule.OP_CONV_IS:  negativeOp = ZmFilterRule.OP_NOT_CONV; break;
1000 	    case ZmFilterRule.OP_IS_ME:     negativeOp = ZmFilterRule.OP_NOT_ME; break;
