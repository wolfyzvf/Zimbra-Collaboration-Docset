 16 /**
 17  * Minimal wrapper around XHR, with no dependencies.
 18  * 
 19  * @author Andy Clark
 20  * 
 21  * @private
 22  */
 23 AjxLoader = function() {}
 24 
 25 //
 26 // Data
 27 //
 28 
 29 AjxLoader.__createXHR;
 30 
 31 if (window.XMLHttpRequest) {
 32     AjxLoader.__createXHR = function() { return new XMLHttpRequest(); };
 33 }
 34 else if (window.ActiveXObject) {
 35     (function(){
 36         var vers = ["MSXML2.XMLHTTP.4.0", "MSXML2.XMLHTTP.3.0", "MSXML2.XMLHTTP", "Microsoft.XMLHTTP"];
 37         for (var i = 0; i < vers.length; i++) {
 38             try {
 39                 new ActiveXObject(vers[i]);
 40                 AjxLoader.__createXHR = function() { return new ActiveXObject(vers[i]); };
 41                 break;
 42             }
 43             catch (e) {
 44                 // ignore
 45             }
 46         }
 47     })();
 48 }

 54 /**
 55  * This function uses XHR to load and return the contents at an arbitrary URL.
 56  * <p>
 57  * It can be called with either a URL string or a parameters object.
 58  *
 59  * @param	{hash}		urlOrParams			a hash of parameters
 60  * @param {string}	urlOrParams.url       the URL to load
 61  * @param {string}	[urlOrParams.method]    the load method (e.g. "GET").
 62  *                                  If this parameter is not specified, then
 63  *                                  the value is determined by whether content
 64  *                                  has been specified: "POST" if specified,
 65  *                                  "GET" otherwise.
 66  * @param {hash}	[urlOrParams.headers]  the map of request headers to set.
 67  * @param {boolean}	[urlOrParams.async]     determines whether the request
 68  *                                  is asynchronous or synchronous. If this
 69  *                                  parameter is not specified, then the value
 70  *                                  is determined by whether a callback has
 71  *                                  been specified: async if a callback is
 72  *                                  specified, sync if no callback.
 73  * @param {string}	[urlOrParams.content]   the content to POST to URL. If not specified, the request method is GET.
 74  * @param {string}	[urlOrParams.userName]  the username of the request.
 75  * @param {string}	[urlOrParams.password]  the password of the request.
 76  * @param {AjxCallback}	[urlOrParams.callback]  the callback to run at end of load.
 77  */
 78 AjxLoader.load = function(urlOrParams) {
 79     var params = urlOrParams;
 80     if (typeof urlOrParams == "string") {
 81         params = { url: urlOrParams };
 82     }
 83 
 84     var req = AjxLoader.__createXHR();
 85     var func = Boolean(params.callback) ? function() { AjxLoader._response(req, params.callback); } : null;
 86     var method = params.method || (params.content != null ? "POST" : "GET");
 87 	
 88 	if (func) {
 89 	    req.onreadystatechange = func;
 90 	}
 91     var async = params.async != null ? params.async : Boolean(func);
 92     req.open(method, params.url, async, params.userName, params.password);
 93     for (var name in params.headers) {
 94         req.setRequestHeader(name, params.headers[name]);
 95     }
 96     req.send(params.content || "");
 97 
 98     return req;
 99 };
100 
101 AjxLoader._response = function(req, callback) {
102     if (req.readyState == 4) {
103         callback.run(req);
104     }
105 };
