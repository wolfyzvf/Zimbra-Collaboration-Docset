 20 /**
 21  * Creates an empty tree view.
 22  * @class
 23  * This class displays data in a tree structure.
 24  *
 25  * @author Conrad Damon
 26  * 
 27  * @param {Hash}	params				the hash of parameters
 28  * @param {DwtControl}	params.parent				the tree's parent widget
 29  * @param {constant}	params.type				the organizer type
 30  * @param {String}	params.className				the CSS class
 31  * @param {constant}	params.posStyle				the positioning style
 32  * @param {constant}	params.overviewId			theoverview ID
 33  * @param {String}	params.headerClass			the CSS class for header item
 34  * @param {DwtDragSource}	params.dragSrc				the drag source
 35  * @param {DwtDropTarget}	params.dropTgt				the drop target
 36  * @param {constant}	params.treeStyle				tree style (see {@link DwtTree})
 37  * @param {Boolean}	params.isCheckedByDefault	sets the default state of "checked" tree style
 38  * @param {Hash}	params.allowedTypes			a hash of org types this tree may display
 39  * @param {Hash}	params.allowedSubTypes		a hash of org types this tree may display below top level
 40  *
 41  * @extends		DwtTree
 42  */
 43 ZmTreeView = function(params) {
 44 
 45 	if (arguments.length == 0) { return; }
 46 
 47 	DwtTree.call(this, {
 48 		parent: params.parent,
 49 		parentElement: params.parentElement,
 50 		style: params.treeStyle,
 51 		isCheckedByDefault: params.isCheckedByDefault,
 52 		className: (params.className || "OverviewTree"),
 53 		posStyle: params.posStyle,
 54 		id: params.id
 55 	});
 56 
 57 	this._headerClass = params.headerClass || "overviewHeader";
 58 	this.overviewId = params.overviewId;
 59 	this.type = params.type;
 60 	this.allowedTypes = params.allowedTypes;
 61 	this.allowedSubTypes = params.allowedSubTypes;
 62 
 63 	this._overview = appCtxt.getOverviewController().getOverview(this.overviewId);
 64 	
 65 	this._dragSrc = params.dragSrc;
 66 	this._dropTgt = params.dropTgt;
 67 
 68 	this._dataTree = null;
 69 	this._treeItemHash = {};	// map organizer to its corresponding tree item by ID
 70 };
 71

 73 ZmTreeView.KEY_ID	= "_treeId_";

 86 /**
 87  * Finds the correct position for an organizer within a node, given
 88  * a sort function.
 89  *
 90  * @param {DwtTreeItem}	node			the node under which organizer is to be added
 91  * @param {ZmOrganizer}	organizer		the organizer
 92  * @param {function}	sortFunction	the function for comparing two organizers
 93  * @return	{int}	the index
 94  */
 95 ZmTreeView.getSortIndex =
 96 function(node, organizer, sortFunction) {
 97 	if (!sortFunction) return null;
 98 	var cnt = node.getItemCount();
 99 	var children = node.getItems();
100 	for (var i = 0; i < children.length; i++) {
101 		if (children[i]._isSeparator) continue;
102 		var child = children[i].getData(Dwt.KEY_OBJECT);
103 		if (!child) continue;
104 		var test = sortFunction(organizer, child);
105 		if (test == -1) {
106 			return i;
107 		}
108 	}
109 	return i;
110 };

140  * @param	{Boolean}	params.collapsed		if <code>true</code>, initially leave the root collapsed

143 ZmTreeView.prototype.set =
144 function(params) {
145 	this._showUnread = params.showUnread;
146 	this._dataTree = params.dataTree;
147 	this._optButton = params.optButton;
148 
149 	this.clearItems();
150 
151 	// create header item
152 	var root = this._dataTree.root;
153 	var isMultiAcctSubHeader = (appCtxt.multiAccounts && (this.type == ZmOrganizer.SEARCH || this.type == ZmOrganizer.TAG));
154 	var imageInfo = this._getHeaderTreeItemImage();
155 	var ti = this._headerItem = new DwtHeaderTreeItem({
156 		parent:				this,
157 		className:			isMultiAcctSubHeader ? "DwtTreeItem" : this._headerClass,
158 		imageInfo:			imageInfo,
159 		id:					ZmId.getTreeItemId(this.overviewId, null, this.type),
160 		optButton:			params.optButton,
161 		dndScrollCallback:	this._overview && this._overview._dndScrollCallback,
162 		dndScrollId:		this._overview && this._overview._scrollableContainerId,
163 		selectable:			(appCtxt.multiAccounts && this.type != ZmOrganizer.SEARCH && this.type != ZmOrganizer.TAG)
164 	});
165 	ti._isHeader = true;
166 	var name = ZmMsg[ZmOrganizer.LABEL[this.type]];
167 	if (name) {
168 		ti.setText(name);
169 	}
170 	ti.setData(Dwt.KEY_ID, root.id);
171 	ti.setData(Dwt.KEY_OBJECT, root);
172 	ti.setData(ZmTreeView.KEY_ID, this.overviewId);
173 	ti.setData(ZmTreeView.KEY_TYPE, this.type);
174 	if (this._dropTgt) {
175 		ti.setDropTarget(this._dropTgt);
176 	}
177 	this._treeItemHash[root.id] = ti;
178 	ti.getHtmlElement().style.overflow = "hidden";
179 	// render the root item's children (ie everything else)
180 	params.treeNode = ti;
181 	params.organizer = root;
182 	this._render(params);
183 	ti.setExpanded(!params.collapsed, null, true);
184 
185 	if (!appCtxt.multiAccounts) {
186 		this.addSeparator();
187 	}
188 
189     //determine if call is from dialog/picker rather than nav tree
190     var isApp = this._overview && this._overview.isAppOverview;
191     var isZimbraAccount = true;
192     var acct;
193     if(appCtxt.multiAccounts && params.account){
194         acct = params.account;
195         isZimbraAccount = acct.isZimbraAccount && !acct.isMain;
196     }
197 
198 
199     // TODO: Find a better way to indicate which trees show the share link
200     var addShareLink =
201         appCtxt.get(ZmSetting.SHARING_ENABLED, null, acct)  && isApp  && isZimbraAccount &&
202         (
203             this.type == ZmOrganizer.FOLDER   ||
204             this.type == ZmOrganizer.ADDRBOOK ||
205             this.type == ZmOrganizer.CALENDAR ||
206             this.type == ZmOrganizer.TASKS    ||
207             this.type == ZmOrganizer.BRIEFCASE
208         )
209     ;
210     if (addShareLink) {
211         //bug 56122: If folderes are deferred for creation, the sort order is messed up.
212         var parentNode = this._treeItemHash[root.id];
213         parentNode._realizeDeferredChildren();
214 
215         var item = new DwtTreeItem({
216             parent: parentNode,
217             deferred:false // NOTE: Needed so we can grab link element
218         });
219         item.setImage("Blank_16");
220         item.enableSelection(false);
221         item.showCheckBox(false);
222 
223         var id = item.getHTMLElId();
224         item.setText(AjxTemplate.expand(this.SHARE_LINK_TEMPLATE, id));
225         var linkEl = document.getElementById(id+"_addshare_link");
226         linkEl.onclick = AjxCallback.simpleClosure(this._handleAddShareLink, this, (acct? acct.id : null));
227         this._addShareLink = item;
228     }
229 
230 	if (appCtxt.getSkinHint("noOverviewHeaders") ||
231 		this._hideHeaderTreeItem())
232 	{
233 		ti.setVisible(false, true);
234 	}
235 };

258 /**
259  * Gets the currently selected organizer(s). If tree view is checkbox style,
260  * return value is an {Array} otherwise, a single {DwtTreeItem} object is returned.
261  * 
262  * @return	{Array|DwtTreeItem}		the selected item(s)
263  */
264 ZmTreeView.prototype.getSelected =
265 function() {
266 	if (this.isCheckedStyle) {
267 		var selected = [];
268 		// bug #44805 - iterate thru the entire tree item hash in case there are
269 		// more than one header items in the tree view (e.g. Imap accounts)
270 		for (var i in this._treeItemHash) {
271 			var ti = this._treeItemHash[i];
272 			if (ti && ti.getChecked()) {
273 				selected.push(ti.getData(Dwt.KEY_OBJECT));
274 			}
275 		}
276 		return selected;
277 	} else {
278 		return (this.getSelectionCount() != 1)
279 			? null : this.getSelection()[0].getData(Dwt.KEY_OBJECT);
280 	}
281 };

300 /**
301  * Draws the children of the given node.
302  *
303  * @param params		[hash]			hash of params:
304  *        treeNode		[DwtTreeItem]	current node
305  *        organizer		[ZmOrganizer]	its organizer
306  *        omit			[Object]*		hash of system folder IDs to ignore	
307  *        include		[object]*		hash of system folder IDs to include
308  *        showOrphans	[boolean]*		if true, show parent chain of any
309  * 										folder of this type, as well as the folder
310  *        searchTypes	[hash]*			types of saved searches to show
311  *        noTooltips	[boolean]*		if true, don't show tooltips for tree items
312  *        startPos		[int]*			start rendering this far into list of children
313  * 
314  * TODO: Add logic to support display of folders that are not normally allowed in
315  * 		this tree, but that have children (orphans) of an allowed type
316  * TODO: Only sort folders we're showing (requires two passes).
317  * 
318  * @private
319  */
320 ZmTreeView.prototype._render =
321 function(params) {
322 	var org = params.organizer;
323 	var children = org.children.getArray();
324 	if (org.isDataSource(ZmAccount.TYPE_IMAP)) {
325 		children.sort(ZmImapAccount.sortCompare);
326 	} else if (ZmTreeView.COMPARE_FUNC[this.type]) {
327 		if (appCtxt.isOffline && this.type == ZmOrganizer.SEARCH) {
328 			var local = [];
329 			for (var j = 0; j < children.length; j++) {
330 				var child = children[j];
331 				if (child && child.type == ZmOrganizer.SEARCH && !child.isOfflineGlobalSearch) {
332 					local.push(child);
333 				}
334 			}
335 			children = local;
336 		}
337 		// IE loses type info on the children array - the props are there and it can be iterated,
338 		// but a function call like sort() blows up. So create an array local to child win.
339 		if (appCtxt.isChildWindow && AjxEnv.isIE) {
340 			var children1 = [];
341 			for (var i = 0, len = children.length; i < len; i++) {
342 				children1.push(children[i]);
343 			}
344 			children = children1;
345 		}
346 		children.sort(eval(ZmTreeView.COMPARE_FUNC[this.type]));
347 	}
348 	DBG.println(AjxDebug.DBG3, "Render: " + org.name + ": " + children.length);
349 	var addSep = true;
350 	var numItems = 0;
351 	var len = children.length;
352     if (params.startPos === undefined && params.lastRenderedFolder ){
353         for (var i = 0, len = children.length; i < len; i++) {
354             if (params.lastRenderedFolder == children[i] ){
355                params.startPos = i + 1; // Next to lastRenderedFolder
356                break;
357             }
358         }
359         DBG.println(AjxDebug.DBG1, "load remaining folders: " + params.startPos);
360     }
361 	for (var i = params.startPos || 0; i < len; i++) {
362 		var child = children[i];
363 		if (!child || (params.omit && params.omit[child.nId])) { continue; }
364 		if (!(params.include && params.include[child.nId])) {
365 			var allowed = ((org.nId == ZmOrganizer.ID_ROOT) && this.allowedTypes[child.type]) ||
366 						  ((org.nId != ZmOrganizer.ID_ROOT) && this.allowedSubTypes[child.type]);
367 			if (!allowed) {
368 				if (params.omitParents) continue;
369 				var proxy = AjxUtil.createProxy(params);
370 				proxy.treeNode = null;
371 				proxy.organizer = child;
372 				this._render(proxy);
373 				continue;
374 			}
375 			if (this._allowedTypes && !this._allowedTypes[child.type]) {
376 				if (params.omitParents) continue;
377 				var proxy = AjxUtil.createProxy(params);
378 				proxy.treeNode = null;
379 				proxy.organizer = child;
380 				this._render(proxy);
381 				continue;
382 			}
383 		}
384 
385 		if (child.numTotal == 0 && (child.nId == ZmFolder.ID_SYNC_FAILURES)) {
386 			continue;
387 		}
388 
389 		var parentNode = params.treeNode;
390 		var account = appCtxt.multiAccounts && child.getAccount();
391 
392 		// bug: 43067 - reparent calendars for caldav-based accounts
393 		if (account && account.isCalDavBased() &&
394 			child.parent.nId == ZmOrganizer.ID_CALENDAR)
395 		{
396 			parentNode = parentNode.parent;
397 		}
398 
399 		// if there's a large number of folders to display, make user click on special placeholder
400 		// to display remainder; we then display them MAX_ITEMS at a time
401 		if (numItems >= ZmTreeView.MAX_ITEMS) {
402 			if (params.startPos) {
403 				// render next chunk
404 				params.startPos = i;
405 				params.len = (params.startPos + ZmTreeView.MAX_ITEMS >= len) ? len : 0;	// hint that we're done
406 				this._showRemainingFolders(params);
407 				return;
408 			} else if (numItems >= ZmTreeView.MAX_ITEMS * 2) {
409 				// add placeholder tree item "Show remaining folders"
410 				var orgs = ZmMsg[ZmOrganizer.LABEL[this.type]].toLowerCase();
411 				var name = AjxMessageFormat.format(ZmMsg.showRemainingFolders, orgs);
412 				child = new ZmFolder({id:ZmFolder.ID_LOAD_FOLDERS, name:name, parent:org});
413 				child._tooltip = AjxMessageFormat.format(ZmMsg.showRemainingFoldersTooltip, [(children.length - i), orgs]);
414 				var ti = this._addNew(parentNode, child);
415 				ti.enableSelection(true);
416 				if (this.isCheckedStyle) {
417 					ti.showCheckBox(false);
418 				}
419                 params.lastRenderedFolder  = children[i-1];
420 				params.showRemainingFoldersNode = ti;
421 				child._showFoldersCallback = new AjxCallback(this, this._showRemainingFolders, [params]);
422 				return;
423 			}
424 		}
425 
426 		// NOTE: Separates public and shared folders
427 		if ((org.nId == ZmOrganizer.ID_ROOT) && child.link && addSep) {
428 			params.treeNode.addSeparator();
429 			addSep = false;
430 		}
431 		this._addNew(parentNode, child, null, params.noTooltips, params.omit);
432 		numItems++;
433 	}
434 };

436 /**
437  * Adds a tree item node for the given organizer to the tree, and then adds its children.
438  *
439  * @param parentNode	[DwtTreeItem]	node under which to add the new one
440  * @param organizer		[ZmOrganizer]	organizer for the new node
441  * @param index			[int]*			position at which to add the new node
442  * @param noTooltips	[boolean]*		if true, don't show tooltips for tree items
443  * @param omit			[Object]*		hash of system folder IDs to ignore
444  * 
445  * @private
446  */
447 ZmTreeView.prototype._addNew =
448 function(parentNode, organizer, index, noTooltips, omit) {
449 	var ti;
450 	// check if we're adding a datasource folder
451 	var dsColl = (organizer.type == ZmOrganizer.FOLDER) && appCtxt.getDataSourceCollection();
452 	var dss = dsColl && dsColl.getByFolderId(organizer.nId);
453 	var ds = (dss && dss.length > 0) ? dss[0] : null;
454 
455 	if (ds && ds.type == ZmAccount.TYPE_IMAP) {
456 		var cname = appCtxt.isFamilyMbox ? null : this._headerClass;
457 		var icon =  "Folder";
458 		ti = new DwtTreeItem({parent:this, text:organizer.getName(), className:cname, imageInfo:icon, selectable: false});
459 	} else {
460 		// create parent chain
461 		if (!parentNode) {
462 			var stack = [];
463 			var parentOrganizer = organizer.parent;
464 			if (parentOrganizer) {
465 				while ((parentNode = this.getTreeItemById(parentOrganizer.id)) == null) {
466 					stack.push(parentOrganizer);
467 					parentOrganizer = parentOrganizer.parent;
468 				}
469 			}
470 			while (parentOrganizer = stack.pop()) {
471 				parentNode = this.getTreeItemById(parentOrganizer.parent.id);
472 				parentNode = new DwtTreeItem({
473 					parent:					parentNode,
474 					text:					parentOrganizer.getName(),
475 					imageInfo:				parentOrganizer.getIconWithColor(),
476 					forceNotifySelection:	true,
477 					dndScrollCallback:		this._overview && this._overview._dndScrollCallback,
478 					dndScrollId:			this._overview && this._overview._scrollableContainerId,
479 					id:						ZmId.getTreeItemId(this.overviewId, parentOrganizer.id)
480 				});
481 				parentNode.setData(Dwt.KEY_ID, parentOrganizer.id);
482 				parentNode.setData(Dwt.KEY_OBJECT, parentOrganizer);
483 				parentNode.setData(ZmTreeView.KEY_ID, this.overviewId);
484 				parentNode.setData(ZmTreeView.KEY_TYPE, parentOrganizer.type);
485 				this._treeItemHash[parentOrganizer.id] = parentNode;
486 			}
487 		}
488 		if (this._addShareLink && this._addShareLink.parent == parentNode) {
489 			var addShareIndex = parentNode.getChildIndex(this._addShareLink);
490 			if (addShareIndex >= 0 && (!index || index > addShareIndex))
491 				index = addShareIndex; // Bug 52053: We must make sure nothing has a higher index than that of this._addShareLink
492 		}
493 		var params = {
494 			parent:				parentNode,
495 			index:				index,
496 			text:				organizer.getName(this._showUnread),
497 			dndScrollCallback:	this._overview && this._overview._dndScrollCallback,
498 			dndScrollId:		this._overview && this._overview._scrollableContainerId,
499 			imageInfo:			organizer.getIconWithColor(),
500 			id:					ZmId.getTreeItemId(this.overviewId, organizer.id)
501 		};
502 		// now add item
503 		ti = new DwtTreeItem(params);
504 	}
505 
506 	if (appCtxt.multiAccounts &&
507 		(organizer.type == ZmOrganizer.SEARCH ||
508 		 organizer.type == ZmOrganizer.TAG))
509 	{
510 		ti.addClassName("DwtTreeItemChildDiv");
511 	}
512 
513 	ti.setDndText(organizer.getName());
514 	ti.setData(Dwt.KEY_ID, organizer.id);
515 	ti.setData(Dwt.KEY_OBJECT, organizer);
516 	ti.setData(ZmTreeView.KEY_ID, this.overviewId);
517 	ti.setData(ZmTreeView.KEY_TYPE, organizer.type);
518 	if (!noTooltips) {
519 		var tooltip = organizer.getToolTip();
520 		if (tooltip) {
521 			ti.setToolTipContent(tooltip);
522 		}
523 	}
524 	if (this._dragSrc) {
525 		ti.setDragSource(this._dragSrc);
526 	}
527 	if (this._dropTgt) {
528 		ti.setDropTarget(this._dropTgt);
529 	}
530 	this._treeItemHash[organizer.id] = ti;
531 
532 	if (ZmTreeView.ADD_SEP[organizer.nId]) {
533 		parentNode.addSeparator();
534 	}
535 
536 	// recursively add children
537 	if (organizer.children && organizer.children.size()) {
538 		this._render({treeNode:ti, organizer:organizer, omit:omit});
539 	}
540 
541 	if (ds && ds.type == ZmAccount.TYPE_IMAP) {
542 		ti.setExpanded(!appCtxt.get(ZmSetting.COLLAPSE_IMAP_TREES));
543 	}
544 
545 	return ti;
546 };

549 /**
550  * Gets the data (an organizer) from the tree item nearest the one
551  * associated with the given ID.
552  *
553  * @param {int}	id	an organizer ID
554  * @return	{Object}	the data or <code>null</code> for none
555  */
556 ZmTreeView.prototype.getNextData =
557 function(id) {
558 	var treeItem = this.getTreeItemById(id);
559 	if(!treeItem || !treeItem.parent) { return null; }
560 
561 	while (treeItem && treeItem.parent) {
562 		var parentN = treeItem.parent;
563 		if (!(parentN instanceof DwtTreeItem)) {
564 			return null;
565 		}
566 		var treeItems = parentN.getItems();
567 		var result = null;
568 		if (treeItems && treeItems.length > 1) {
569 			for(var i = 0; i < treeItems.length; i++) { 
570 				var tmp = treeItems[i];
571 				if (tmp == treeItem) {
572 					var nextData = this.findNext(treeItem, treeItems, i);
573 					if (nextData) { return nextData; }
574 					var prevData = this.findPrev(treeItem, treeItems, i);
575 					if (prevData) {	return prevData; }
576 				}
577 			}
578 		}
579 		treeItem = treeItem.parent;
580 	}
581 	return null;
582 };

584 ZmTreeView.prototype.findNext =
585 function(treeItem, treeItems, i) {
586 	for (var j = i + 1; j < treeItems.length; j++) {
587 		var next = treeItems[j];
588 		if (next && next.getData) {
589 			return next.getData(Dwt.KEY_OBJECT);
590 		}
591 	}
592 	return null;
593 };
594 
595 ZmTreeView.prototype.findPrev =
596 function(treeItem, treeItems, i) {
597 	for (var j = i - 1; j >= 0; j--) {
598 		var prev = treeItems[j];
599 		if (prev && prev.getData) {
600 			return prev.getData(Dwt.KEY_OBJECT);
601 		}
602 	}
603 	return null;
604 };

677 ZmTreeView.prototype._handleAddShare = function() {
678     var dialog = appCtxt.getShareSearchDialog();
679     var shares = dialog.getShares();
680     dialog.popdown();
681     if (shares.length == 0) return;
682 
683     AjxDispatcher.require("Share");
684     var requests = [];
685     for (var i = 0; i < shares.length; i++) {
686         var share = shares[i];
687         var name = share.folderPath.substr(1).replace(/\//g," ");
688         var ownerName = (share.normalizedOwnerName.indexOf('@') >1) ? share.normalizedOwnerName.substr(0, share.normalizedOwnerName.indexOf('@')) : share.normalizedOwnerName;
689         requests.push({
690             _jsns: "urn:zimbraMail",
691             link: {
692                 l: ZmOrganizer.ID_ROOT,
693                 name: ZmShare.getDefaultMountpointName(ownerName, name),
694                 view: share.view,
695                 zid: share.ownerId,
696                 rid: share.folderId
697             }
698         });
699         var params = {
700             jsonObj: {
701                 BatchRequest: {
702                     _jsns: "urn:zimbra",
703                     CreateMountpointRequest: requests
704                 }
705             },
706             asyncMode: true,
707             callback: new AjxCallback(this, this._handleAddShareResponse),
708             errorCallback: new AjxCallback(this, this._handleAddShareError)
709         };
710         appCtxt.getAppController().sendRequest(params);
711     }
712 };
