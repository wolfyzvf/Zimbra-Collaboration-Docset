 18  * @constructor

 36 	if (arguments.length == 0) return;

 80 ZmCalItemView.prototype.set =
 81 function(calItem, prevView, mode) {
 82 	if (this._calItem == calItem) { return; }
 83 
 84 	// So that Close button knows which view to go to
 85     // condition introduced to avoid irrelevant view being persisted as previous view
 86 	var viewMgr = this._controller._viewMgr;
 87 	this._prevView = prevView || (viewMgr && (calItem.folderId != ZmFolder.ID_TRASH) ?
 88 	                              viewMgr.getCurrentViewName() : this._prevView);
 89 
 90 	this.reset();
 91 	this._calItem = calItem;
 92 	this._mode = mode;
 93 	this._renderCalItem(calItem, true);
 94 };

109 ZmCalItemView.prototype._renderCalItem =
110 function(calItem, renderButtons) {
111 	this._lazyCreateObjectManager();
112 
113 	var subs = this._getSubs(calItem);
114 	var closeBtnCellId = this._htmlElId + "_closeBtnCell";
115 	var editBtnCellId = this._htmlElId + "_editBtnCell";
116 	this._hdrTableId = this._htmlElId + "_hdrTable";
117 
118     var calendar = calItem.getFolder();
119     var isReadOnly = calendar.isReadOnly();
120     subs.allowEdit = !isReadOnly && (appCtxt.get(ZmSetting.CAL_APPT_ALLOW_ATTENDEE_EDIT) || calItem.isOrg);
121 
122 	var el = this.getHtmlElement();
123 	el.innerHTML = AjxTemplate.expand("calendar.Appointment#ReadOnlyView", subs);
124 
125     if (renderButtons) {
126         // add the close button
127         this._closeButton = new DwtButton({parent:this, className:"DwtToolbarButton"});
128         this._closeButton.setImage("Close");
129         this._closeButton.setText(ZmMsg.close);
130         this._closeButton.addSelectionListener(new AjxListener(this, this.close));
131         this._closeButton.reparentHtmlElement(closeBtnCellId);
132 
133         if (document.getElementById(editBtnCellId)) {
134             // add the save button for reminders and  move select
135             this._editButton = new DwtButton({parent:this, className:"DwtToolbarButton"});
136             this._editButton.setImage("Edit");
137             this._editButton.setText(ZmMsg.edit);
138             this._editButton.addSelectionListener(new AjxListener(this, this.edit));
139             var calendar = calItem && appCtxt.getById(calItem.folderId);
140             var isTrash = calendar && calendar.id == ZmOrganizer.ID_TRASH;
141             this._editButton.setEnabled(!isTrash);
142             this._editButton.reparentHtmlElement(editBtnCellId);
143         }
144     }
145 
146 	// content/body
147 	var hasHtmlPart = (calItem.notesTopPart && calItem.notesTopPart.getContentType() == ZmMimeTable.MULTI_ALT);
148 	var mode = (hasHtmlPart && appCtxt.get(ZmSetting.VIEW_AS_HTML))
149 		? ZmMimeTable.TEXT_HTML : ZmMimeTable.TEXT_PLAIN;
150 
151 	var bodyPart = calItem.getNotesPart(mode);
152 	if (bodyPart) {
153 		this._msg = this._msg || this._calItem._currentlyLoaded;
154         if (mode === ZmMimeTable.TEXT_PLAIN) {
155             bodyPart = AjxStringUtil.convertToHtml(bodyPart);
156         }
157 		this._makeIframeProxy({container: el, html:bodyPart, isTextMsg:(mode == ZmMimeTable.TEXT_PLAIN)});
158 	}
159 };

176 // returns true if given dates are w/in a single day
177 ZmCalItemView.prototype._isOneDayAppt =
178 function(sd, ed) {
179 	var start = new Date(sd.getTime());
180 	var end = new Date(ed.getTime());
181 
182 	start.setHours(0, 0, 0, 0);
183 	end.setHours(0, 0, 0, 0);
184 
185 	return start.valueOf() == end.valueOf();
186 };

188 ZmCalItemView._getAttachString =
189 function(calItem) {
190 	var str = [];
191 	var j = 0;
192 
193 	var attachList = calItem.getAttachments();
194 	if (attachList) {
195 		for (var i = 0; i < attachList.length; i++) {
196 			str[j++] = ZmApptViewHelper.getAttachListHtml(calItem, attachList[i]);
197 		}
198 	}
199 
200 	return str.join("");
201 };

246 ZmApptView.prototype.edit =
247 function(ev) {
248 	var item = this._calItem;
249 
250     if(!item.isOrg && !(this._editWarningDialog && this._editWarningDialog.isPoppedUp())){
251         var msgDialog = this._editWarningDialog = appCtxt.getMsgDialog();
252         msgDialog.setMessage(ZmMsg.attendeeEditWarning, DwtMessageDialog.WARNING_STYLE);
253         msgDialog.popup();
254         msgDialog.registerCallback(DwtDialog.OK_BUTTON, this.edit, this);
255         return;
256     }else if(this._editWarningDialog){
257         this._editWarningDialog.popdown();
258     }
259     
260 	var mode = ZmCalItem.MODE_EDIT;
261 	if (item.isRecurring()) {
262 		mode = this._mode || ZmCalItem.MODE_EDIT_SINGLE_INSTANCE;
263 	}
264 	item.setViewMode(mode);
265 	var app = this._controller._app;
266 	app.getApptComposeController().show(item, mode);
267 };

275 ZmApptView.prototype._renderCalItem =
276 function(calItem) {
277 	this._lazyCreateObjectManager();
278 
279 	var subs = this._getSubs(calItem);
280 	this._hdrTableId = this._htmlElId + "_hdrTable";
281 
282 
283     var calendar = calItem.getFolder();
284     var isReadOnly = calendar.isReadOnly() || calendar.isInTrash();
285     subs.allowEdit = !isReadOnly && (appCtxt.get(ZmSetting.CAL_APPT_ALLOW_ATTENDEE_EDIT) || calItem.isOrg);
286 
287 	var el = this.getHtmlElement();
288 	el.innerHTML = AjxTemplate.expand("calendar.Appointment#ReadOnlyView", subs);
289 
290 
291     var selParams = {parent: this};
292     var statusSelect = new DwtSelect(selParams);
293 
294     var ptst = {};
295     ptst[ZmCalBaseItem.PSTATUS_NEEDS_ACTION] = ZmMsg.ptstMsgNeedsAction;
296     ptst[ZmCalBaseItem.PSTATUS_ACCEPT] = ZmMsg.ptstMsgAccepted;
297     ptst[ZmCalBaseItem.PSTATUS_TENTATIVE] = ZmMsg.ptstMsgTentative;
298     ptst[ZmCalBaseItem.PSTATUS_DECLINED] = ZmMsg.ptstMsgDeclined;
299 
300     this._ptst = ptst;
301     //var statusMsgs = {};
302     var calItemPtst = calItem.ptst || ZmCalBaseItem.PSTATUS_ACCEPT;
303 
304     var data = null;
305     for (var stat in ptst) {
306         //stat = ptst[index];
307         if (stat === ZmCalBaseItem.PSTATUS_NEEDS_ACTION && calItemPtst !== ZmCalBaseItem.PSTATUS_NEEDS_ACTION) { continue; }
308         data = new DwtSelectOptionData(stat, ZmCalItem.getLabelForParticipationStatus(stat), false, null, ZmCalItem.getParticipationStatusIcon(stat));
309         statusSelect.addOption(data);
310         if (stat == calItemPtst){
311             statusSelect.setSelectedValue(stat);
312         }
313     }
314     if (isReadOnly) { statusSelect.setEnabled(false); }
315 
316     this._statusSelect = statusSelect;
317     this._origPtst = calItemPtst;
318     statusSelect.reparentHtmlElement(this._htmlElId + "_responseActionSelectCell");
319     statusSelect.addChangeListener(new AjxListener(this, this._statusSelectListener));
320 
321     this._statusMsgEl = document.getElementById(this._htmlElId + "_responseActionMsgCell");
322     this._statusMsgEl.innerHTML = ptst[calItemPtst];
323 
324 	// content/body
325 	var hasHtmlPart = (calItem.notesTopPart && calItem.notesTopPart.getContentType() == ZmMimeTable.MULTI_ALT);
326 	var mode = (hasHtmlPart && appCtxt.get(ZmSetting.VIEW_AS_HTML))
327 		? ZmMimeTable.TEXT_HTML : ZmMimeTable.TEXT_PLAIN;
328 
329 	var bodyPart = calItem.getNotesPart(mode);
330 	if (bodyPart) {
331 		this._msg = this._msg || this._calItem._currentlyLoaded;
332         if (mode === ZmMimeTable.TEXT_PLAIN) {
333             bodyPart = AjxStringUtil.convertToHtml(bodyPart);
334         }
335 		this._makeIframeProxy({container: el, html:bodyPart, isTextMsg:(mode == ZmMimeTable.TEXT_PLAIN)});
336 	}
337 };

339 ZmApptView.prototype._getSubs =
340 function(calItem) {
341 	var subject = calItem.getName();
342 	var location = calItem.getAttendeesText(ZmCalBaseItem.LOCATION, true) || calItem.location;
343 	var equipment = calItem.getAttendeesText(ZmCalBaseItem.EQUIPMENT, true);
344 	var isException = calItem._orig.isException;
345 	var dateStr = this._getTimeString(calItem);
346 	//var attendees = calItem.getAttendeesText(ZmCalBaseItem.PERSON);
347     var isAttendees = false;
348     var reqAttendees = calItem.getAttendeesTextByRole(ZmCalBaseItem.PERSON, ZmCalItem.ROLE_REQUIRED, true, this._objectManager, this._htmlElId);
349     var optAttendees = calItem.getAttendeesTextByRole(ZmCalBaseItem.PERSON, ZmCalItem.ROLE_OPTIONAL, true, this._objectManager, this._htmlElId);
350     if(reqAttendees || optAttendees) isAttendees = true;
351 	var org, obo;
352 	var recurStr = calItem.isRecurring() ? calItem.getRecurBlurb() : null;
353 	var attachStr = ZmCalItemView._getAttachString(calItem);
354 
355 	if (isAttendees) {
356 		var organizer = org = calItem.getOrganizer();
357 		var sender = calItem.message.getAddress(AjxEmailAddress.SENDER);
358 		var from = calItem.message.getAddress(AjxEmailAddress.FROM);
359 		var address = sender || from;
360         if(!org && address)	org = address.toString();
361 		if (sender && organizer)
362 			obo = from ? from.toString() : organizer;
363 	}
364 
365 	if (this._objectManager) {
366 		this._objectManager.setHandlerAttr(ZmObjectManager.DATE,
367 											ZmObjectManager.ATTR_CURRENT_DATE,
368 											calItem.startDate);
369 
370 		subject = this._objectManager.findObjects(subject, true);
371 		location = this._objectManager.findObjects(location, true);
372 		equipment = this._objectManager.findObjects(equipment, true);
373 		dateStr = this._objectManager.findObjects(dateStr, true);
374 		if (org) org = this._objectManager.findObjects(org, true, ZmObjectManager.EMAIL);
375 		if (obo) obo = this._objectManager.findObjects(obo, true, ZmObjectManager.EMAIL);
376 	}
377 
378 	return {
379 		id: this._htmlElId,
380 		subject: subject,
381 		location: location,
382 		equipment: equipment,
383 		isException: isException,
384 		dateStr: dateStr,
385         isAttendees: isAttendees,
386         reqAttendees: reqAttendees,
387 		optAttendees: optAttendees,
388 		org: org,
389 		obo: obo,
390 		recurStr: recurStr,
391 		attachStr: attachStr,
392 		folder: appCtxt.getTree(ZmOrganizer.CALENDAR).getById(calItem.folderId),
393 		//folders: String(calItem.id).match(/:/) ? [] : this._controller.getCalendars(),
394 		folderLabel: ZmMsg.calendar,
395 		reminderLabel: ZmMsg.reminder,
396 		alarm: calItem.alarm,
397 		isAppt: true,
398         _infoBarId:this._infoBarId
399 	};
400 };

402 ZmApptView.prototype._getTimeString =
403 function(calItem) {
404 	var sd = calItem._orig.startDate;
405 	var ed = calItem._orig.endDate;
406     var tz = AjxMsg[AjxTimezone.DEFAULT] || AjxTimezone.getServerId(AjxTimezone.DEFAULT)
407 
408 	if (calItem.isRecurring() && this._mode == ZmCalItem.MODE_EDIT_SERIES) {
409 		sd = calItem.startDate;
410 		ed = calItem.endDate;
411         var seriesTZ = calItem.getTimezone();
412 
413         //convert to client timezone if appt's timezone differs
414         if(seriesTZ != AjxTimezone.getServerId(AjxTimezone.DEFAULT)) {
415             var offset1 = AjxTimezone.getOffset(AjxTimezone.DEFAULT, sd);
416 		    var offset2 = AjxTimezone.getOffset(AjxTimezone.getClientId(seriesTZ), sd);
417             sd.setTime(sd.getTime() + (offset1 - offset2)*60*1000);
418             ed.setTime(ed.getTime() + (offset1 - offset2)*60*1000);
419             calItem.setTimezone(AjxTimezone.getServerId(AjxTimezone.DEFAULT));
420         }
421 	}
422 
423 	var isAllDay = calItem.isAllDayEvent();
424 	var isMultiDay = calItem.isMultiDay();
425 	if (isAllDay && isMultiDay) {
426 		var endDate = new Date(ed.getTime());
427 		ed.setDate(endDate.getDate()-1);
428 	}
429 
430 	var pattern = isAllDay ?
431 				  (isMultiDay ? ZmMsg.apptTimeAllDayMulti   : ZmMsg.apptTimeAllDay) :
432 				  (isMultiDay ? ZmMsg.apptTimeInstanceMulti : ZmMsg.apptTimeInstance);
433 	var params = [sd, ed, tz];
434 
435 	return AjxMessageFormat.format(pattern, params);
436 };
