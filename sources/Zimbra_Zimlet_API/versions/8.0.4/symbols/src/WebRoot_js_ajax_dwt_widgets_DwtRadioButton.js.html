 16 /**
 17  * Creates a radio button.
 18  * @constructor
 19  * @class
 20  * This class implements a radio button.
 21  * 
 22  * @param {hash}	params	a hash of parameters
 23  * @param  {DwtComposite}     params.parent	the parent widget
 24  * @param  {constant}     params.style 	the text style. May be one of: {@link DwtCheckbox.TEXT_LEFT} or
 25  * 									{@link DwtCheckbox.TEXT_RIGHT} arithimatically or'd (|) with one of:
 26  * 									{@link DwtCheckbox.ALIGN_LEFT}, {@link DwtCheckbox.ALIGN_CENTER}, or
 27  * 									{@link DwtCheckbox.ALIGN_LEFT}.
 28  * 									The first determines were in the checkbox the text will appear
 29  * 									(if set), the second determine how the content of the text will be
 30  * 									aligned. The default value for this parameter is: 
 31  * 									{@link DwtCheckbox.TEXT_LEFT} | {@link DwtCheckbox.ALIGN_CENTER}
 32  * @param  {string}     params.name		the input control name (required for IE)
 33  * @param  {string}     params.value     the input control value.
 34  * @param  {boolean}     params.checked	the input control checked status (required for IE)
 35  * @param  {string}     params.className	the CSS class
 36  * @param  {constant}     params.posStyle	the positioning style (see {@link DwtControl})
 37  * @param  {string}     params.id		an explicit ID to use for the control's HTML element
 38  * @param  {number}     params.index 	the index at which to add this control among parent's children
 39  * 
 40  * @extends	DwtCheckbox
 41  */
 42 DwtRadioButton = function(params) {
 43 	if (arguments.length == 0) { return; }
 44 	params = Dwt.getParams(arguments, DwtRadioButton.PARAMS);
 45 	params.className = params.className || "DwtRadioButton";
 46 	DwtCheckbox.call(this, params);
 47 }
 48 
 49 DwtRadioButton.PARAMS = DwtCheckbox.PARAMS;
 50 
 51 DwtRadioButton.prototype = new DwtCheckbox;
 52 DwtRadioButton.prototype.constructor = DwtRadioButton;
 53 
 54 DwtRadioButton.prototype.isDwtRadioButton = true;
 55 DwtRadioButton.prototype.isInputControl = true;
 56 DwtRadioButton.prototype.toString = function() { return "DwtRadioButton"; };
