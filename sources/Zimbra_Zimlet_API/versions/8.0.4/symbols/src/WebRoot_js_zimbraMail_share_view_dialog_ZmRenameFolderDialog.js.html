 20 /**
 21  * Creates a rename folder dialog.
 22  * @class
 23  * This class represents a rename folder dialog.
 24  * 
 25  * @param	{DwtComposite}	parent		the parent
 26  * @param	{String}	className		the class name
 27  *  
 28  * @extends		ZmDialog
 29  */
 30 ZmRenameFolderDialog = function(parent, className) {
 31 
 32 	ZmDialog.call(this, {parent:parent, className:className, title:ZmMsg.renameFolder, id:"RenameFolderDialog"});
 33 
 34 	this._setNameField(this._nameFieldId);
 35 };
 36 
 37 ZmRenameFolderDialog.prototype = new ZmDialog;
 38 ZmRenameFolderDialog.prototype.constructor = ZmRenameFolderDialog;
 39 
 40 ZmRenameFolderDialog.prototype.toString = 
 41 function() {
 42 	return "ZmRenameFolderDialog";
 43 };
 44 
 45 /**
 46  * Pops-up the dialog.
 47  * 
 48  * @param	{ZmFolder}		folder		the folder
 49  * @param	{Object}		[source]	(not used)
 50  */
 51 ZmRenameFolderDialog.prototype.popup =
 52 function(folder, source) {
 53 	ZmDialog.prototype.popup.call(this);
 54 	var title = (folder.type == ZmOrganizer.SEARCH) ? ZmMsg.renameSearch : ZmMsg.renameFolder;
 55 	this.setTitle(title + ': ' + folder.getName(false, ZmOrganizer.MAX_DISPLAY_NAME_LENGTH));
 56 	this._nameField.value = folder.getName(false, null, true);
 57 	this._folder = folder;
 58 };
 59 
 60 ZmRenameFolderDialog.prototype._contentHtml = 
 61 function() {
 62 	this._nameFieldId = this._htmlElId + "_name";
 63 	var subs = {id:this._htmlElId, newLabel:ZmMsg.newName};
 64 	return AjxTemplate.expand("share.Dialogs#ZmRenameDialog", subs);
 65 };
 66 
 67 ZmRenameFolderDialog.prototype._okButtonListener =
 68 function(ev) {
 69 	var results = this._getFolderData();
 70 	if (results) {
 71 		DwtDialog.prototype._buttonListener.call(this, ev, results);
 72 	}
 73 };
 74 
 75 ZmRenameFolderDialog.prototype._getFolderData =
 76 function() {
 77 	// check name for presence and validity
 78 	var name = AjxStringUtil.trim(this._nameField.value);
 79 	var msg = ZmFolder.checkName(name, this._folder.parent);
 80 
 81 	// make sure another folder with this name doesn't already exist at this level
 82 	if (!msg) {
 83 		var folder = this._folder.parent.getByName(name);
 84         var folderType = appCtxt.getFolderTree(appCtxt.getActiveAccount()).getFolderTypeByName(name);
 85         if (folder && (folder.id != this._folder.id)) {
 86 			msg = AjxMessageFormat.format(ZmMsg.errorAlreadyExists, [name,ZmMsg[folderType.toLowerCase()]]);
 87 		}
 88 	}
 89 
 90 	return (msg ? this._showError(msg) : [this._folder, name]);
 91 };
 92 
 93 ZmRenameFolderDialog.prototype._enterListener =
 94 function(ev) {
 95 	var results = this._getFolderData();
 96 	if (results) {
 97 		this._runEnterCallback(results);
 98 	}
 99 };
