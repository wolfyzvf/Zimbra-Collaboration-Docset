 16 /**
 17  * Creates a SOAP exception.
 18  * @class
 19  * 
 20  * 
 21  * @param {string} 		[msg]		the human readable message
 22  * @param {constant}		code		the exception code
 23  * @param {string} 		[method] 	the name of the method throwing the exception
 24  * @param {string} 		[detail]		any additional detail
 25  * 
 26  * @extends		AjxException
 27  * 
 28  * @private
 29  */
 30 AjxSoapException = function(msg, code, method, detail) {
 31 	AjxException.call(this, msg, code, method, detail);
 32 }
 33 
 34 AjxSoapException.prototype.toString = 
 35 function() {
 36 	return "AjxSoapException";
 37 }
 38 
 39 AjxSoapException.prototype = new AjxException;
 40 AjxSoapException.prototype.constructor = AjxSoapException;
 41 
 42 /**
 43  * Defines an "internal error" exception.
 44  */
 45 AjxSoapException.INTERNAL_ERROR 	= "INTERNAL_ERROR";
 46 /**
 47  * Defines an "invalid PDU" exception.
 48  */
 49 AjxSoapException.INVALID_PDU 		= "INVALID_PDU";
 50 /**
 51  * Defines an "element exists" exception.
 52  */
 53 AjxSoapException.ELEMENT_EXISTS 	= "ELEMENT_EXISTS";
