 16 ZmCalListView = function(parent, posStyle, controller, dropTgt) {
 17 	if (arguments.length == 0) { return; }
 18 
 19 	var params = {
 20 		parent: parent,
 21 		posStyle: posStyle,
 22 		controller: controller,
 23 		dropTgt: dropTgt,
 24 		view: ZmId.VIEW_CAL_LIST,
 25 		headerList: this._getHeaderList(parent),
 26 		pageless: true
 27 	};
 28 	ZmApptListView.call(this, params);
 29 
 30 	this._dateSearchBar = this._createSearchBar(parent);
 31 
 32 	this._needsRefresh = true;
 33 	this._timeRangeStart = 0;
 34 	this._timeRangeEnd = 0;
 35 	this._title = "";
 36 };

 95 ZmCalListView.prototype.setDate =
 96 function(date, duration, roll) {
 97 	this._date = new Date(date.getTime());
 98 
 99 	var d = new Date(date.getTime());
100 	d.setHours(0, 0, 0, 0);
101 	this._timeRangeStart = d.getTime();
102 	this._timeRangeEnd = this._timeRangeStart + ZmCalListView.DEFAULT_CALENDAR_PERIOD;
103 
104 	this._updateTitle();
105 	this._segmentedDates = [];
106 
107 	// update widgets
108 	var startDate = new Date(this._timeRangeStart);
109 	var endDate = new Date(this._timeRangeEnd);
110 	this._startDateField.value = AjxDateUtil.simpleComputeDateStr(startDate);
111 	this._endDateField.value = AjxDateUtil.simpleComputeDateStr(endDate);
112 
113 	this._updateDateRange(startDate, endDate);
114 
115 	// Notify any listeners
116 	if (this.isListenerRegistered(DwtEvent.DATE_RANGE)) {
117 		if (!this._dateRangeEvent) {
118 			this._dateRangeEvent = new DwtDateRangeEvent(true);
119 		}
120 		this._dateRangeEvent.item = this;
121 		this._dateRangeEvent.start = new Date(this._timeRangeStart);
122 		this._dateRangeEvent.end = new Date(this._timeRangeEnd);
123 		this.notifyListeners(DwtEvent.DATE_RANGE, this._dateRangeEvent);
124 	}
125 };

213 ZmCalListView.prototype._mouseOverAction =
214 function(ev, div) {
215 	DwtListView.prototype._mouseOverAction.call(this, ev, div);
216 	var id = ev.target.id || div.id;
217 	if (!id) { return true; }
218 
219 	// check if we're hovering over a column header
220 	var data = this._data[div.id];
221 	var type = data.type;
222 	if (type && type == DwtListView.TYPE_HEADER_ITEM) {
223 		var itemIdx = data.index;
224 		var field = this._headerList[itemIdx]._field;
225         // Bug: 76489 - Added <span> as workaround to show tooltip as HTML
226         // The ideal fix should add a method in DwtControl to remove the tooltip
227 		this.setToolTipContent('<span>'+this._getHeaderToolTip(field, itemIdx)+'</span>');
228 	} else {
229 		var item = this.getItemFromElement(div);
230 		if (item) {
231 			var match = this._parseId(id);
232 			if (!match) { return; }
233 			this.setToolTipContent(this._getToolTip({field:match.field, item:item, ev:ev, div:div, match:match}));
234 			if (match.field != ZmItem.F_SELECTION && match.field != ZmItem.F_TAG && item.getToolTip) {
235 				// load attendee status if necessary
236 				if (item.otherAttendees && (item.ptstHashMap == null)) {
237 					var clone = ZmAppt.quickClone(item);
238 					var uid = this._currentMouseOverApptId = clone.getUniqueId();
239 					var callback = new AjxCallback(null, ZmApptViewHelper.refreshApptTooltip, [clone, this]);
240 					AjxTimedAction.scheduleAction(new AjxTimedAction(this, this.getApptDetails, [clone, callback, uid]), 2000);
241 				}
242 			}
243 		}
244 	}
245 	return true;
246 };

258 ZmCalListView.prototype._createSearchBar = function(parent) {
259     var id = this._htmlElId;
260 
261     var searchBar = new DwtComposite({parent:parent, className:"ZmCalListViewSearchBar", posStyle:DwtControl.ABSOLUTE_STYLE});
262     searchBar.getHtmlElement().innerHTML = AjxTemplate.expand("calendar.Calendar#ListViewSearchBar",id);
263 
264     var controls = new DwtMessageComposite(searchBar);
265     var message = ZmMsg.showApptsFromThrough;
266     var callback = new AjxCallback(this, this._createSearchBarComponent);
267     var hintsCallback = null;
268     controls.setFormat(message, callback, hintsCallback);
269     controls.reparentHtmlElement(document.getElementById(id+"_searchBarControls"));
270 
271     this._dateRangeField = document.getElementById(id+"_searchBarDate");
272 
273     return searchBar;
274 };

276 ZmCalListView.prototype._createSearchBarComponent = function(searchBar, segment, i) {
277     var isStart = segment.getIndex() == 0;
278     var id = this._htmlElId;
279     var prefix = isStart ? "_start" : "_end";
280 
281     var component = new DwtComposite({parent:searchBar});
282     var templateId = "calendar.Calendar#ListViewSearchBarInput";
283     component.getHtmlElement().innerHTML = AjxTemplate.expand(templateId, id+prefix);
284 
285     var inputId = [id,prefix,"DateInput"].join("");
286     var inputEl = document.getElementById(inputId);
287     inputEl.onchange = AjxCallback.simpleClosure(this._onDatesChange, this, [isStart]);
288 
289     var dateButtonListener = new AjxListener(this, this._dateButtonListener);
290     var dateCalSelectionListener = new AjxListener(this, this._dateCalSelectionListener);
291     var buttonId = [id,prefix,"MiniCal"].join("");
292     var button = ZmCalendarApp.createMiniCalButton(searchBar, buttonId, dateButtonListener, dateCalSelectionListener);
293 
294     if (isStart) {
295         this._startDateField = inputEl;
296         this._startDateButton = button;
297     }
298     else {
299         this._endDateField = inputEl;
300         this._endDateButton = button;
301     }
302 
303     return component;
304 };

306 /**
307  * Event listener triggered when user clicks on the down arrow button to bring
308  * up the date picker.
309  *
310  * @param ev		[Event]		Browser event
311  * @private
312  */
313 ZmCalListView.prototype._dateButtonListener =
314 function(ev) {
315 	var calDate = ev.item == this._startDateButton
316 		? AjxDateUtil.simpleParseDateStr(this._startDateField.value)
317 		: AjxDateUtil.simpleParseDateStr(this._endDateField.value);
318 
319 	// if date was input by user and its foobar, reset to today's date
320 	if (isNaN(calDate)) {
321 		calDate = new Date();
322 		var field = ev.item == this._startDateButton
323 			? this._startDateField : this._endDateField;
324 		field.value = AjxDateUtil.simpleComputeDateStr(calDate);
325 	}
326 
327 	// always reset the date to current field's date
328 	var menu = ev.item.getMenu();
329 	var cal = menu.getItem(0);
330 	cal.setDate(calDate, true);
331 	ev.item.popup();
332 
333     if(AjxEnv.isIE) {
334         //DwtMenu adds padding of 6px each side
335         //IE has to add 12px to width and height to adjust the calendar
336         var menuSize = menu.getSize();
337         menu.setSize(menuSize.x+12, menuSize.y+12);
338         menu.getHtmlElement().style.width = "180px";
339     }
340 };

361 /**
362  * Called when user selects a new date from the date-picker. Normalizes the
363  * start/end dates if user chose start date to be after end date or vice versa.
364  * Also updates the UI with the new date ranges and initiates SearchRequest.
365  *
366  * @param isStartDate
367  * @private
368  */
369 ZmCalListView.prototype._handleDateChange =
370 function(isStartDate) {
371 	var start = AjxDateUtil.simpleParseDateStr(this._startDateField.value);
372 	var end = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
373 
374 	var startTime = start.getTime();
375 	var endTime = end.getTime() + AjxDateUtil.MSEC_PER_DAY;
376 
377 	// normalize dates
378 	if (isStartDate && startTime >= endTime) {
379 		endTime = startTime + AjxDateUtil.MSEC_PER_DAY;
380 		end = new Date(endTime);
381 		this._endDateField.value = AjxDateUtil.simpleComputeDateStr(end);
382 	}
383 	else if (endTime <= startTime) {
384 		startTime = end.getTime() - AjxDateUtil.MSEC_PER_DAY;
385 		start = new Date(startTime);
386 		this._startDateField.value = AjxDateUtil.simpleComputeDateStr(start);
387 	}
388 
389 	this._timeRangeStart = startTime;
390 	this._timeRangeEnd = endTime;
391 
392 	this._updateDateRange(start, end);
393 	this._updateTitle();
394 
395 	this._segmentedDates = [];
396 
397 	this._segmentDates(startTime, endTime);
398 	this.set((new AjxVector()), null, true); // clear the current list
399 	this._search();
400 };

402 /**
403  * Chunks the date range into intervals per the default search period. We do
404  * this to avoid taxing the server with a large date range.
405  *
406  * @param startTime		[String]	start time in ms
407  * @param endTime		[String]	end time in ms
408  * @private
409  */
410 ZmCalListView.prototype._segmentDates =
411 function(startTime, endTime) {
412 	var startPeriod = startTime;
413 	var endPeriod = startTime + ZmCalListView.DEFAULT_SEARCH_PERIOD;
414 
415 	// reset back to end time if we're search less than next block (e.g. two weeks)
416 	if (endPeriod > endTime) {
417 		endPeriod = endTime;
418 	}
419 
420 	do {
421 		this._segmentedDates.push({startTime: startPeriod, endTime: endPeriod});
422 
423 		startPeriod += ZmCalListView.DEFAULT_SEARCH_PERIOD;
424 
425 		var newEndPeriod = endPeriod + ZmCalListView.DEFAULT_SEARCH_PERIOD;
426 		endPeriod = (newEndPeriod > endTime) ? endTime : newEndPeriod;
427 	}
428 	while (startPeriod < endTime);
429 };

431 /**
432  * Makes a SearchRequest for the first chunk of appointments
433  *
434  * @private
435  */
436 ZmCalListView.prototype._search =
437 function() {
438 	var dates = this._segmentedDates.shift();
439 
440 	var params = {
441 		start: dates.startTime,
442 		end: dates.endTime,
443 		folderIds: this._controller.getCheckedCalendarFolderIds(),
444 		callback: (new AjxCallback(this, this._handleSearchResponse)),
445 		noBusyOverlay: true
446 	};
447 
448 	this._controller.apptCache.getApptSummaries(params);
449 };

451 /**
452  * Appends the SearchResponse results to the listview. Attempts to request the
453  * next chunk of appointments if the user's scrollbar isn't shown.
454  *
455  * @param list		[AjxVector]		list returned by ZmApptCache
456  * @private
457  */
458 ZmCalListView.prototype._handleSearchResponse =
459 function(list) {
460 
461 	this.addItems(list.getArray());
462     Dwt.setTitle(this.getTitle());
463 	// if we have more days to fetch, search again for the next set
464 	if (this._segmentedDates.length > 0 && this._getItemsNeeded(true) > 0) {
465 		this._search();
466 	}
467 };

480 /**
481  * Adds the items.
482  * The function is overridden to not to show the "No results found" if anything is present in the list.
483  *
484  * @param	{array}		itemArray		an array of items
485  */
486 ZmCalListView.prototype.addItems =
487 function(itemArray) {
488 	if (AjxUtil.isArray(itemArray)) {
489 		if (!this._list) {
490 			this._list = new AjxVector();
491 		}
492 
493 		// clear the "no results" message before adding!
494 		if (this._list.size() == 0) {
495 			this._resetList();
496 		}
497 
498 		// Prune the appts before passing to the underlying ListView
499 		var showDeclined = appCtxt.get(ZmSetting.CAL_SHOW_DECLINED_MEETINGS);
500 		var filterV = new AjxVector();
501 		for (var i = 0; i < itemArray.length; i++) {
502             var appt = itemArray[i];
503             if (showDeclined || (appt.ptst != ZmCalBaseItem.PSTATUS_DECLINED)) {
504                 filterV.add(appt);
505             }
506 		}
507 
508 		this._renderList(filterV, this._list.size() != 0, true);
509 		this._list.addList(filterV.getArray());
510         this._resetColWidth();
511         //Does not make sense but required to make the scrollbar appear
512         var size = this.getSize();
513         this._listDiv.style.height = (size.y - DwtListView.HEADERITEM_HEIGHT)+"px";
514 	}
515 };
