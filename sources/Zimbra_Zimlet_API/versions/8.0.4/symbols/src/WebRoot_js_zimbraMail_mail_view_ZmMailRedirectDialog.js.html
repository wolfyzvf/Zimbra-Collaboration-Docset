 20 /**
 21  * Creates a Mail Redirect dialog.
 22  * @class
 23  * This class represents a Mail Redirect dialog.
 24  *
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  *
 28  * @extends		DwtDialog
 29  */
 30 ZmMailRedirectDialog = function(parent, className) {
 31 	className = className || "ZmFolderPropsDialog";
 32 
 33     DwtDialog.call(this, {parent:parent, className:className, title:ZmMsg.mailRedirect, id:"RedirectDialog"});
 34 
 35 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, this._handleCancelButton.bind(this));
 36 
 37 	var recipParams = {};
 38 	recipParams.enableContainerInputs		= this.enableInputs.bind(this);
 39 	recipParams.contactPopdownListener		= this.contactPopdownListener.bind(this);
 40 	recipParams.contextId					= this.toString();
 41     this._recipients = new ZmRecipients(recipParams);
 42 
 43     this._useAcAddrBubbles = appCtxt.get(ZmSetting.USE_ADDR_BUBBLES);
 44     this._fieldNames = [AjxEmailAddress.TO];
 45     var data = { id : this._htmlElId, acAddrBubbles : this._useAcAddrBubbles};
 46 
 47     for (var i = 0; i < this._fieldNames.length; i++) {
 48         var typeStr = AjxEmailAddress.TYPE_STRING[this._fieldNames[i]];
 49         var ids =  this._recipients.createRecipientIds(this._htmlElId, typeStr)
 50         data[typeStr + "RowId"]    = ids.row;
 51         data[typeStr + "PickerId"] = ids.picker;
 52         data[typeStr + "InputId"]  = ids.control;
 53         data[typeStr + "CellId"]   = ids.cell;
 54     }
 55 
 56     var html = AjxTemplate.expand("mail.Message#RedirectDialog", data);
 57     this.setContent(html);
 58 
 59 
 60     this._recipients.createRecipientHtml(this, this._htmlElId, this._htmlElId, this._fieldNames);
 61     this._tabGroup.addMember(this._recipients.getField(AjxEmailAddress.TO));
 62 };
 63 
 64 ZmMailRedirectDialog.prototype = new DwtDialog;
 65 ZmMailRedirectDialog.prototype.constructor = ZmMailRedirectDialog;
 66 
 67 ZmMailRedirectDialog.prototype.isZmMailRedirectDialog = true;
 68 ZmMailRedirectDialog.prototype.toString = function() { return "ZmMailRedirectDialog"; };
 69 
 70 
 71 
 72 ZmMailRedirectDialog.prototype.getAddrs =
 73 function() {
 74 	return this._recipients.collectAddrs();
 75 };
 76 
 77 
 78 /**
 79  * Pops-up the properties dialog.
 80  *
 81  * @param	{ZmOrganizer}	organizer		the organizer
 82  */
 83 ZmMailRedirectDialog.prototype.popup =
 84 function(mail) {
 85     this._recipients.setup();
 86 
 87 	DwtDialog.prototype.popup.call(this);
 88 };
 89 
 90 ZmMailRedirectDialog.prototype._resetTabFocus =
 91 function(){
 92 	this._tabGroup.setFocusMember(this._recipients.getField(AjxEmailAddress.TO), true);
 93 };
 94 
 95 
 96 ZmMailRedirectDialog.prototype.popdown =
 97 function() {
 98     this._recipients.reset();
 99 	DwtDialog.prototype.popdown.call(this);
100 };
