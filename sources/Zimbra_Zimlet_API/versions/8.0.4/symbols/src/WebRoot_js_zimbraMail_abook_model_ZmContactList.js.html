 22 /**
 23  * Create a new, empty contact list.
 24  * @class
 25  * This class represents a list of contacts. In general, the list is the result of a
 26  * search. It may be the result of a <code><GetContactsRequest></code>, which returns all of the user's
 27  * local contacts. That list is considered to be canonical.
 28  * <p>
 29  * Loading of all local contacts has been optimized by delaying the creation of {@link ZmContact} objects until
 30  * they are needed. That has a big impact on IE, and not much on Firefox. Loading a subset
 31  * of attributes did not have much impact on load time, probably because a large majority
 32  * of contacts contain only those minimal fields.</p>
 33  *
 34  * @author Conrad Damon
 35  *
 36  * @param {ZmSearch}	search	the search that generated this list
 37  * @param {Boolean}	isGal		if <code>true</code>, this is a list of GAL contacts
 38  * @param {constant}	type		the item type
 39  * 
 40  * @extends		ZmList
 41  */
 42 ZmContactList = function(search, isGal, type) {
 43 
 44 	if (arguments.length == 0) { return; }
 45 	type = type || ZmItem.CONTACT;
 46 	ZmList.call(this, type, search);
 47 
 48 	this.isGal = (isGal === true);
 49 	this.isCanonical = false;
 50 	this.isLoaded = false;
 51 
 52 	this._app = appCtxt.getApp(ZmApp.CONTACTS);
 53 	if (!this._app) { 
 54 		this._emailToContact = this._imAddressToContact = this._phoneToContact = {};
 55 		return;
 56 	}
 57 	this._emailToContact = this._app._byEmail;
 58 	this._imAddressToContact = this._app._byIM;
 59 	this._phoneToContact = this._app._byPhone;
 60 
 61 	this._alwaysUpdateHashes = true; // Should we update the phone & IM fast-lookup hashes even when account features don't require it? (bug #60411)
 62 };

 89 ZmContactList.prototype.addLoadedCallback =
 90 function(callback) {
 91 	if (this.isLoaded) {
 92 		callback.run();
 93 		return;
 94 	}
 95 	if (!this._loadedCallbacks) {
 96 		this._loadedCallbacks = [];
 97 	}
 98 	this._loadedCallbacks.push(callback);
 99 };

101 /**
102  * @private
103  */
104 ZmContactList.prototype._finishLoading =
105 function() {
106 	DBG.timePt("done loading " + this.size() + " contacts");
107 	this.isLoaded = true;
108 	if (this._loadedCallbacks) {
109 		var callback;
110 		while (callback = this._loadedCallbacks.shift()) {
111 			callback.run();
112 		}
113 	}

116 /**
117  * Retrieves the contacts from the back end, and parses the response. The list is then sorted.
118  * This method is used only by the canonical list of contacts, in order to load their content.
119  * <p>
120  * Loading a minimal set of attributes did not result in a significant performance gain.
121  * </p>
122  * 
123  * @private
124  */
125 ZmContactList.prototype.load =
126 function(callback, errorCallback, accountName) {
127 	// only the canonical list gets loaded
128 	this.isCanonical = true;
129 	var respCallback = new AjxCallback(this, this._handleResponseLoad, [callback]);
130 	DBG.timePt("requesting contact list", true);
131     if(appCtxt.isExternalAccount()) {
132         //Do not make a call in case of external user
133         //The rest url constructed wont exist in case of external user
134         if (callback) {
135 		    callback.run();
136 	    }
137         return;
138     }
139 	var args = ZmContactList.URL_ARGS;
140 
141 	// bug 74609: suppress overzealous caching by IE
142 	if (AjxEnv.isIE) {
143 		args = AjxUtil.hashCopy(args);
144 		args.sid = ZmCsfeCommand.getSessionId();
145 	}
146 
147 	var params = {asyncMode:true, noBusyOverlay:true, callback:respCallback, errorCallback:errorCallback};
148 	params.restUri = AjxUtil.formatUrl({
149 		path:["/home/", (accountName || appCtxt.getUsername()),
150 	          ZmContactList.URL].join(""),
151 	    qsArgs: args, qsReset:true
152 	});
153 	DBG.println(AjxDebug.DBG1, "loading contacts from " + params.restUri);
154 	appCtxt.getAppController().sendRequest(params);
155 
156 	ZmContactList.addDlFolder();
157 	
158 };

160 /**
161  * @private
162  */
163 ZmContactList.prototype._handleResponseLoad =
164 function(callback, result) {
165 	DBG.timePt("got contact list");
166 	var text = result.getResponse();
167 	var derefList = [];
168 	if (text) {
169 		var contacts = text.split(ZmContactList.CONTACT_SPLIT_CHAR);
170 		var derefBatchCmd = new ZmBatchCommand(true, null, true);
171 		for (var i = 0, len = contacts.length; i < len; i++) {
172 			var fields = contacts[i].split(ZmContactList.FIELD_SPLIT_CHAR);
173 			var contact = {}, attrs = {};
174 			var groupMembers = [];
175 			var foundDeref = false;
176 			for (var j = 0, len1 = fields.length; j < len1; j += 2) {
177 				if (ZmContactList.IS_CONTACT_FIELD[fields[j]]) {
178 					contact[fields[j]] = fields[j + 1];
179 				} else {
180 					var value = fields[j+1];
181 					switch (fields[j]) {
182 						case ZmContact.F_memberC:
183 							groupMembers.push({type: ZmContact.GROUP_CONTACT_REF, value: value});
184 							foundDeref = true; //load shared contacts
185 							break;
186 						case ZmContact.F_memberG:
187 							groupMembers.push({type: ZmContact.GROUP_GAL_REF, value: value});
188 							foundDeref = true;
189 							break;
190 						case ZmContact.F_memberI:
191 							groupMembers.push({type: ZmContact.GROUP_INLINE_REF, value: value});
192 							foundDeref = true;
193 							break;
194 						default:
195 							attrs[fields[j]] = value;
196 					}
197 				}
198 			}
199 			attrs[ZmContact.F_groups] = groupMembers;
200 			if (foundDeref) {
201 				//batch group members for deref loading
202 				var dummy = new ZmContact(contact["id"], this);
203 				derefBatchCmd.add(new AjxCallback(dummy, dummy.load, [null, null, derefBatchCmd, true]));
204 			}
205 			contact._attrs = attrs;
206 			this._addContact(contact);
207 		}
208 		derefBatchCmd.run();
209 	}
210 
211 	this._finishLoading();
212 
213 	if (callback) {
214 		callback.run();
215 	}
216 };

261 ZmContactList.prototype.add = 
262 function(item, index) {
263 	if (!item.id || !this._idHash[item.id]) {
264 		this._vector.add(item, index);
265 		if (item.id) {
266 			this._idHash[item.id] = item;
267 		}
268 		this._updateHashes(item, true);
269 	}
270 };

272 ZmContactList.prototype.cache = 
273 function(offset, newList) {
274 	var getId = function(){
275 		return this.id;
276 	}
277 	var exists = function(obj) {
278 		return this._vector.containsLike(obj, getId);
279 	}
280 	var unique = newList.sub(exists, this);
281 
282 	this.getVector().merge(offset, unique);
283 	// reparent each item within new list, and add it to ID hash
284 	var list = unique.getArray();
285 	for (var i = 0; i < list.length; i++) {
286 		var item = list[i];
287 		item.list = this;
288 		if (item.id) {
289 			this._idHash[item.id] = item;
290 		}
291 	}
292 };

294 /**
295  * @private
296  */
297 ZmContactList.prototype._addContact =
298 function(contact) {
299 
300 	// note that we don't create a ZmContact here (optimization)
301 	contact.list = this;
302 	this._updateHashes(contact, true);
303 	var fn = [], fl = [];
304 	if (contact._attrs[ZmContact.F_firstName])	{ fn.push(contact._attrs[ZmContact.F_firstName]); }
305 	if (contact._attrs[ZmContact.F_middleName])	{ fn.push(contact._attrs[ZmContact.F_middleName]); }
306 	if (contact._attrs[ZmContact.F_lastName])	{ fn.push(contact._attrs[ZmContact.F_lastName]); }
307 	if (fn.length) {
308 		contact._attrs[ZmContact.X_fullName] = fn.join(" ");
309 	}
310 	if (contact._attrs[ZmContact.F_firstName])	{ fl.push(contact._attrs[ZmContact.F_firstName]); }
311 	if (contact._attrs[ZmContact.F_lastName])	{ fl.push(contact._attrs[ZmContact.F_lastName]); }
312 	contact._attrs[ZmContact.X_firstLast] = fl.join(" ");
313 
314 	this.add(contact);
315 };

317 /**
318  * Converts an anonymous contact object (contained by the JS returned by load request)
319  * into a ZmContact, and updates the containing list if it is the canonical one.
320  *
321  * @param {Object}	contact		a contact
322  * @param {int}	idx		the index of contact in canonical list
323  * 
324  * @private
325  */
326 ZmContactList.prototype._realizeContact =
327 function(contact, idx) {
328 
329 	if (contact instanceof ZmContact) { return contact; }
330 	if (contact && contact.type == ZmItem.CONTACT) { return contact; }	// instanceof often fails in new window
331 
332 	var args = {list:this};
333 	var obj = eval(ZmList.ITEM_CLASS[this.type]);
334 	var realContact = obj && obj.createFromDom(contact, args);
335 
336 	if (this.isCanonical) {
337 		var a = this.getArray();
338 		idx = idx || this.getIndexById(contact.id);
339 		a[idx] = realContact;
340 		this._updateHashes(realContact, true);
341 		this._idHash[contact.id] = realContact;
342 	}
343 
344 	return realContact;
345 };

347 /**
348  * Finds the array index for the contact with the given ID.
349  *
350  * @param {int}	id		the contact ID
351  * @return	{int}	the index
352  * @private
353  */
354 ZmContactList.prototype.getIndexById =
355 function(id) {
356 	var a = this.getArray();
357 	for (var i = 0; i < a.length; i++) {
358 		if (a[i].id == id) {
359 			return i;
360 		}
361 	}
362 	return null;

365 /**
366  * Override in order to make sure the contacts have been realized. We don't
367  * call realizeContact() since this is not the canonical list.
368  *
369  * @param {int}	offset		the starting index
370  * @param {int}	limit		the size of sublist
371  * @return	{AjxVector}	a vector of {@link ZmContact} objects
372  */
373 ZmContactList.prototype.getSubList =
374 function(offset, limit, folderId) {
375 	if (folderId && this.isCanonical) {
376 		// only collect those contacts that belong to the given folderId if provided
377 		var newlist = [];
378 		var sublist = this.getArray();
379 		var offsetCount = 0;
380 		this.setHasMore(false);
381 
382 		for (var i = 0; i < sublist.length; i++) {
383 			sublist[i] = this._realizeContact(sublist[i], i);
384 			var folder = appCtxt.getById(sublist[i].folderId);
385 			if (folder && folder.nId == ZmOrganizer.normalizeId(folderId)) {
386 				if (offsetCount >= offset) {
387 					if (newlist.length == limit) {
388 						this.setHasMore(true);
389 						break;
390 					}
391 					newlist.push(sublist[i]);
392 				}
393 				offsetCount++;
394 			}
395 		}
396 
397 		return AjxVector.fromArray(newlist);
398 	} else {
399 		var vec = ZmList.prototype.getSubList.call(this, offset, limit);
400 		if (vec) {
401 			var a = vec.getArray();
402 			for (var i = 0; i < a.length; i++) {
403 				a[i] = this._realizeContact(a[i], offset + i);
404 			}
405 		}
406 
407 		return vec;
408 	}
409 };

411 /**
412  * Override in order to make sure the contact has been realized. Canonical list only.

414  * @param {int}	id		the contact ID
415  * @return	{ZmContact}	the contact or <code>null</code> if not found

418 function(id) {
419 	if (!id || !this.isCanonical) return null;
420 
421 	var contact = this._idHash[id];
422 	return contact ? this._realizeContact(contact) : null;

425 /**
426  * Gets the contact with the given address, if any (canonical list only).
427  *
428  * @param {String}	address	an email address
429  * @return	{ZmContact}	the contact or <code>null</code> if not found
430  */
431 ZmContactList.prototype.getContactByEmail =
432 function(address) {
433 	if (!address || !this.isCanonical) return null;
434 
435 	var contact = this._emailToContact[address.toLowerCase()];
436 	if (contact) {
437 		contact = this._realizeContact(contact);
438 		contact._lookupEmail = address;	// so caller knows which address matched
439 		return contact;
440 	} else {
441 		return null;
442 	}
443 };

458  * Gets information about the contact with the given phone number, if any (canonical list only).
459  *
460  * @param {String}	phone	the phone number
461  * @return	{Hash}	an object with <code>contact</code> = the contact & <code>field</code> = the field with the matching phone number

472 	} else {
473 		return null;
474 	}

477 /**
478  * Moves a list of items to the given folder.
479  * <p>
480  * This method calls the base class for normal "moves" UNLESS we're dealing w/
481  * shared items (or folder) in which case we must send a CREATE request for the
482  * given folder to the server followed by a hard delete of the shared contact.
483  * </p>
484  *
485  * @param {Hash}	params		a hash of parameters
486  * @param	{Array}       params.items			a list of items to move
487  * @param	{ZmFolder}	params.folder		the destination folder
488  * @param	{Hash}	       params.attrs		the additional attrs for SOAP command
489  * @param	{Boolean}	params.outOfTrash	if <code>true</code>, we are moving contacts out of trash
490  */
491 ZmContactList.prototype.moveItems =
492 function(params) {
493 
494 	params = Dwt.getParams(arguments, ["items", "folder", "attrs", "outOfTrash"]);
495 	params.items = AjxUtil.toArray(params.items);
496 
497 	var moveBatchCmd = new ZmBatchCommand(true, null, true);
498 	var loadBatchCmd = new ZmBatchCommand(true, null, true);
499 	var softMove = [];
500 
501 	// if the folder we're moving contacts to is a shared folder, then dont bother
502 	// checking whether each item is shared or not
503 	if (params.items[0] && params.items[0] instanceof ZmItem) {
504 		for (var i = 0; i < params.items.length; i++) {
505 			var contact = params.items[i];
506 
507 			if (contact.isReadOnly()) { continue; }
508 
509 			softMove.push(contact);
510 		}
511 	} else {
512 		softMove = params.items;
513 	}
514 
515 	// for "soft" moves, handle moving out of Trash differently
516 	if (softMove.length > 0) {
517 		var params1 = AjxUtil.hashCopy(params);
518 		params1.attrs = params.attrs || {};
519 		var toFolder = params.folder;
520 		params1.attrs.l = toFolder.isRemote() ? toFolder.getRemoteId() : toFolder.id;
521 		params1.action = "move";
522         params1.accountName = appCtxt.multiAccounts && appCtxt.accountList.mainAccount.name;
523         if (params1.folder.id == ZmFolder.ID_TRASH) {
524             params1.actionText = ZmMsg.actionTrash;
525             // bug: 47389 avoid moving to local account's Trash folder.
526             params1.accountName = appCtxt.multiAccounts && params.items[0].getAccount().name;
527         } else {
528             params1.actionText = ZmMsg.actionMove;
529             params1.actionArg = toFolder.getName(false, false, true);
530         }
531 		params1.callback = params.outOfTrash && new AjxCallback(this, this._handleResponseMoveItems, params);
532 
533 		this._itemAction(params1);
534 	}
535 };

537 /**
538  * @private
539  */
540 ZmContactList.prototype._handleResponseMoveBatchCmd =
541 function(result) {
542 	var resp = result.getResponse().BatchResponse.ContactActionResponse;
543 	// XXX: b/c the server does not return notifications for actions done on
544 	//      shares, we manually notify - TEMP UNTIL WE GET BETTER SERVER SUPPORT
545 	var ids = resp[0].action.id.split(",");
546 	for (var i = 0; i < ids.length; i++) {
547 		var contact = appCtxt.cacheGet(ids[i]);
548 		if (contact && contact.isShared()) {
549 			contact.notifyDelete();
550 			appCtxt.cacheRemove(ids[i]);
551 		}
552 	}
553 };

589 /**
590  * Deletes contacts after checking that this is not a GAL list.
591  *
592  * @param {Hash}	params		a hash of parameters
593  * @param	{Array}	       params.items			the list of items to delete
594  * @param	{Boolean}	params.hardDelete	if <code>true</code>, force physical removal of items
595  * @param	{Object}	params.attrs			the additional attrs for SOAP command
596  */
597 ZmContactList.prototype.deleteItems =
598 function(params) {
599 	if (this.isGal) {
600 		if (ZmContactList.deleteGalItemsAllowed(params.items)) {
601 			this._deleteDls(params.items);
602 			return;
603 		}
604 		DBG.println(AjxDebug.DBG1, "Cannot delete GAL contacts that are not DLs");
605 		return;
606 	}
607 	ZmList.prototype.deleteItems.call(this, params);
608 };

610 ZmContactList.deleteGalItemsAllowed =
611 function(items) {
612 	var deleteDomainsAllowed = appCtxt.createDistListAllowedDomainsMap;
613 	if (items.length == 0) {
614 		return false; //need a special case since we don't want to enable the "delete" button for 0 items.
615 	}
616 	for (var i = 0; i < items.length; i++) {
617 		var contact = items[i];
618 		var email = contact.getEmail();
619 		var domain = email.split("@")[1];
620 		var isDL = contact && contact.isDistributionList();
621 		//see bug 71368 and also bug 79672 - the !contact.dlInfo is in case somehow dlInfo is missing - so unfortunately if that happens (can't repro) - let's not allow to delete since we do not know if it's an owner
622 		if (!isDL || !deleteDomainsAllowed[domain] || !contact.dlInfo || !contact.dlInfo.isOwner) {
623 			return false;
624 		}
625 	}
626 	return true;
627 };

629 ZmContactList.prototype._deleteDls =
630 function(items, confirmDelete) {
631 
632 	if (!confirmDelete) {
633 		var callback = this._deleteDls.bind(this, items, true);
634 		this._popupDeleteWarningDialog(callback, false, items.length);
635 		return;
636 	}
637 
638 	var reqs = [];
639 	for (var i = 0; i < items.length; i++) {

645 					 _content: contact.getEmail()
646 				},
647 				action: {

651 	}
652 	var jsonObj = {
653 		BatchRequest: {
654 			_jsns: "urn:zimbra",
655 			DistributionListActionRequest: reqs
656 		}

687 function(node) {
688 	var obj = eval(ZmList.ITEM_CLASS[this.type]);
689 	if (obj) {
690 		var item = obj.createFromDom(node, {list:this});
691 		var index = this._sortIndex(item);
692 		// only add if it sorts into this list
693 		var listSize = this.size();
694 		var visible = false;
695 		if (index < listSize || listSize == 0 || (index==listSize && !this._hasMore)) {
696 			this.add(item, index);
697 			this.createLocal(item);
698 			visible = true;
699 		}
700 		this._notify(ZmEvent.E_CREATE, {items: [item], sortIndex: index, visible: visible});

735 /**
736  * Handle modified contact.
737  * 
738  * @private
739  */
740 ZmContactList.prototype.modifyLocal =
741 function(item, details) {
742 	if (details) {
743 		// notify item's list
744 		details.items = [item];
745 		this._notify(ZmEvent.E_MODIFY, details);
746 	}
747 
748 	var contact = details.contact;
749 	if (this.isCanonical || contact.attr[ZmContact.F_email] != details.oldAttr[ZmContact.F_email]) {
750 		// Remove traces of old contact - NOTE: we pass in null for the ID on
751 		// PURPOSE to avoid overwriting the existing cached contact
752 		var oldContact = new ZmContact(null, this);
753 		oldContact.id = details.contact.id;
754 		oldContact.attr = details.oldAttr;
755 		this._updateHashes(oldContact, false);
756 
757 		// add new contact to hashes
758 		this._updateHashes(contact, true);
759 	}
760 
761 	// place in correct position in list
762 	if (details.fileAsChanged) {
763 		this.remove(contact);
764 		var index = this._sortIndex(contact);
765 		var listSize = this.size();
766 		if (index < listSize || listSize == 0 || (index == listSize && !this._hasMore)) {
767 			this.add(contact, index);
768 		}
769 	}
770 
771 	// reset addrbook property
772 	if (contact.addrbook && (contact.addrbook.id != contact.folderId)) {
773 		contact.addrbook = appCtxt.getById(contact.folderId);
774 	}
775 };

787 /**
788  * @private
789  */
790 ZmContactList.prototype._updateHashes =
791 function(contact, doAdd) {
792 
793 	this._app.updateCache(contact, doAdd);
794 
795 	// Update email hash.
796 	for (var index = 0; index < ZmContact.EMAIL_FIELDS.length; index++) {
797 		var field = ZmContact.EMAIL_FIELDS[index];
798 		for (var i = 1; true; i++) {
799 			var aname = ZmContact.getAttributeName(field, i);
800 			var avalue = ZmContact.getAttr(contact, aname);
801 			if (!avalue) break;
802 			if (doAdd) {
803 				this._emailToContact[avalue.toLowerCase()] = contact;
804 			} else {
805 				delete this._emailToContact[avalue.toLowerCase()];
806 			}
807 		}
808 	}
809 
810 	// Update phone hash.
811 	if (appCtxt.get(ZmSetting.VOICE_ENABLED) || this._alwaysUpdateHashes) {
812 		for (var index = 0; index < ZmContact.PHONE_FIELDS.length; index++) {
813 			var field = ZmContact.PHONE_FIELDS[index];
814 			for (var i = 1; true; i++) {
815 				var aname = ZmContact.getAttributeName(field, i);
816 				var avalue = ZmContact.getAttr(contact, aname);
817 				if (!avalue) break;
818 				var digits = this._getPhoneDigits(avalue);
819 				if (digits) {
820 					if (doAdd) {
821 						this._phoneToContact[avalue] = {contact: contact, field: aname};
822 					} else {
823 						delete this._phoneToContact[avalue];
824 					}
825 				}
826 			}
827 		}
828 	}
829 
830 	// Update IM hash.
831 	if (appCtxt.get(ZmSetting.IM_ENABLED) || this._alwaysUpdateHashes) {
832 		for (var index = 0; index < ZmContact.IM_FIELDS.length; index++) {
833 			var field = ZmContact.IM_FIELDS[index];
834 			for (var i = 1; true; i++) {
835 				var aname = ZmContact.getAttributeName(field, i);
836 				var avalue = ZmContact.getAttr(contact, aname);
837 				if (!avalue) break;
838 				if (doAdd) {
839 					this._imAddressToContact[avalue.toLowerCase()] = contact;
840 				} else {
841 					delete this._imAddressToContact[avalue.toLowerCase()];
842 				}
843 			}
844 		}
845 	}
846 };

858 /**
859  * Returns the position at which the given contact should be inserted in this list.
860  * 
861  * @private
862  */
863 ZmContactList.prototype._sortIndex =
864 function(contact) {
865 	var a = this._vector.getArray();
866 	for (var i = 0; i < a.length; i++) {
867 		if (ZmContact.compareByFileAs(a[i], contact) > 0) {
868 			return i;
869 		}
870 	}
871 	return a.length;
872 };
