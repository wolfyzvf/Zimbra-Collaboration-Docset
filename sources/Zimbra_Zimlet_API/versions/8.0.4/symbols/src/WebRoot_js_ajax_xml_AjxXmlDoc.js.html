 35 /**
 36  * <strong>Note:</strong>
 37  * Anybody that uses these regular expressions MUST reset the <code>lastIndex</code>
 38  * property to zero or else the results are not guaranteed to be correct. You should
 39  * use {@link AjxXmlDoc.replaceInvalidChars} instead.
 40  * 
 41  * @private
 42  */
 43 AjxXmlDoc.INVALID_CHARS_RE = /[\u0000-\u0008\u000B-\u000C\u000E-\u001F\uD800-\uDFFF\uFFFE-\uFFFF]/g;
 44 AjxXmlDoc.REC_AVOID_CHARS_RE = /[\u007F-\u0084\u0086-\u009F\uFDD0-\uFDDF]/g;

 53 /**
 54  * Creates an XML doc.
 55  * 
 56  * @return	{AjxXmlDoc}	the XML doc
 57  */
 58 AjxXmlDoc.create =
 59 function() {
 60 	var xmlDoc = new AjxXmlDoc();
 61 	var newDoc = null;
 62 	if (AjxEnv.isIE) {
 63 		newDoc = new ActiveXObject(AjxXmlDoc._msxmlVers);
 64 		newDoc.async = true; // Force Async loading
 65 		if (AjxXmlDoc._msxmlVers == "MSXML2.DOMDocument.4.0") {
 66 			newDoc.setProperty("SelectionLanguage", "XPath");
 67 			newDoc.setProperty("SelectionNamespaces", "xmlns:zimbra='urn:zimbra' xmlns:mail='urn:zimbraMail' xmlns:account='urn:zimbraAccount'");
 68 		}
 69 	} else if (document.implementation && document.implementation.createDocument) {
 70 		newDoc = document.implementation.createDocument("", "", null);
 71 	} else {
 72 		throw new AjxException("Unable to create new Doc", AjxException.INTERNAL_ERROR, "AjxXmlDoc.create");
 73 	}
 74 	xmlDoc._doc = newDoc;
 75 	return xmlDoc;
 76 }

 91 /**
 92  * Creates an XML doc from an XML string.
 93  * 
 94  * @param	{string}		xml		the XML string
 95  * @return	{AjxXmlDoc}		the XML doc
 96  */
 97 AjxXmlDoc.createFromXml =
 98 function(xml) {
 99 	var xmlDoc = AjxXmlDoc.create();
100 	xmlDoc.loadFromString(xml);
101 	return xmlDoc;
102 }

104 /**
105  * Replaces invalid characters in the given string.
106  * 
107  * @param	{string}	s	the string
108  * @return	{string}	the resulting string
109  */
110 AjxXmlDoc.replaceInvalidChars = function(s) {
111 	AjxXmlDoc.INVALID_CHARS_RE.lastIndex = 0;
112 	return s.replace(AjxXmlDoc.INVALID_CHARS_RE, "?");
113 };

115 AjxXmlDoc.getXml =
116 function(node) {
117 	var ser = new XMLSerializer();
118 	return AjxXmlDoc.replaceInvalidChars(ser.serializeToString(node));
119 }

131 AjxXmlDoc.prototype.loadFromString =
132 function(str) {
133 	var doc = this._doc;
134 	doc.loadXML(str);
135 	if (AjxEnv.isIE) {
136 		if (doc.parseError.errorCode != 0)
137 			throw new AjxException(doc.parseError.reason, AjxException.INVALID_PARAM, "AjxXmlDoc.loadFromString");
138 	}
139 }

141 AjxXmlDoc.prototype.loadFromUrl =
142 function(url) {
143 	if(AjxEnv.isChrome || AjxEnv.isSafari) {
144 		var xmlhttp = new window.XMLHttpRequest();
145 		xmlhttp.open("GET", url, false);
146 		xmlhttp.send(null);
147 		var xmlDoc = xmlhttp.responseXML;
148 		this._doc = xmlDoc;
149 	} else {
150 		this._doc.load(url);
151 	}
152 }

154 /**
155  * This function tries to create a JavaScript representation of the DOM. In some cases,
156  * it is easier to work with JS objects rather than do DOM lookups.
157  *
158  * <p>
159  * Rules:
160  * <ol>
161  * <li>The top-level tag gets lost; only it's content is seen important.</li>
162  * <li>Each node will be represented as a JS object.  It's textual content
163  *      will be saved in node.__msh_content (returned by <code>toString()</code>).</li>
164  * <li>Attributes get discarded.</li>
165  * <li>Each subnode will map to a property with its tagName in the parent
166  *      node <code>parent[subnode.tagName] == subnode</code></li>
167  * <li>If multiple nodes with the same tagName have the same parent node, then
168  *      <code>parent[tagName]</code> will be an array containing the objects, rather than a
169  *      single object.</li>
170  * </ol>
171  * 
172  * So what this function allows us to do is for instance this, starting with this XML doc:
173  *
174  * <pre>
175  * <error>
176  *   <code>404</code>
177  *   <name>Not Found</name>
178  *   <description>Page wasn't found on this server.</description>
179  * </error>
180  * </pre>
181  * 
182  * <pre>
183  * var obj = AjxXmlDoc.createFromXml(XML).toJSObject();
184  * alert(obj.code + " " + obj.name + " " + obj.description);
185  * </pre>
186  * 
187  * Here's an array example:
188  * <pre>
189  * <return>
190  *   <item>
191  *     <name>John Doe</name>
192  *     <email>foo@bar.com</email>
193  *   </item>
194  *   <item>
195  *     <name>Johnny Bravo</name>
196  *     <email>bravo@cartoonnetwork.com</email>
197  *   </item>
198  * </return>
199  * </pre>
200  * 
201  * <pre>
202  * var obj = AjxXmlDoc.createFromXml(XML).toJSObject();
203  * for (var i = 0; i < obj.item.length; ++i) {
204  *   alert(obj.item[i].name + " / " + obj.item[i].email);
205  * }
206  * </pre>
207  *
208  * Note that if there's only one <item> tag, then obj.item will be an object
209  * rather than an array.  And if there is no <item> tag, then obj.item will be
210  * undefined.  These are cases that the calling application must take care of.
211  */
212 AjxXmlDoc.prototype.toJSObject = 
213 function(dropns, lowercase, withAttrs) {
214 	_node = function() { this.__msh_content = ''; };
215 	_node.prototype.toString = function() { return this.__msh_content; };
216 	rec = function(i, o) {
217 		var tags = {}, t, n;
218 		for (i = i.firstChild; i; i = i.nextSibling) {
219 			if (i.nodeType == 1) {
220 				t = i.tagName;
221 				if (dropns)      t = t.replace(/^.*?:/, "");
222 				if (lowercase)   t = t.toLowerCase();
223 				n = new _node();
224 				if (tags[t]) {
225 					if (tags[t] == 1) {
226 						o[t] = [ o[t] ];
227 						tags[t] = 2;
228 					}
229 					o[t].push(n);
230 				} else {
231 					o[t] = n;
232 					tags[t] = 1;
233 				}
234 				//do attributes
235 				if(withAttrs) {
236 					if(i.attributes && i.attributes.length) {
237 						for(var ix = 0;ix<i.attributes.length;ix++) {
238 							attr = i.attributes[ix];
239 							n[attr.name] = AjxUtil.isNumeric(attr.value) ? attr.value : String(attr.value);
240 						}
241 					}
242 				}
243 				rec(i, n);
244 			} else if (i.nodeType == 3)
245 				o.__msh_content += i.nodeValue;
246 		}
247 	};
248 	var o = new _node();
249 	rec(this._doc.documentElement, o);
250 	return o;
251 };

253 AjxXmlDoc.prototype.getElementsByTagNameNS = 
254 function(ns, tag) {
255 	var doc = this.getDoc();
256 	return AjxEnv.isIE
257 		? doc.getElementsByTagName(ns + ":" + tag)
258 		: doc.getElementsByTagNameNS(ns, tag);
259 };

272 AjxXmlDoc._init =
273 function() {
274 	if (AjxEnv.isIE) {
275 		var msxmlVers = ["MSXML4.DOMDocument", "MSXML3.DOMDocument", "MSXML2.DOMDocument.4.0",
276 				 "MSXML2.DOMDocument.3.0", "MSXML2.DOMDocument", "MSXML.DOMDocument",
277 				 "Microsoft.XmlDom"];
278 		for (var i = 0; i < msxmlVers.length; i++) {
279 			try {
280 				new ActiveXObject(msxmlVers[i]);
281 				AjxXmlDoc._msxmlVers = msxmlVers[i];
282 				break;
283 			} catch (ex) {
284 			}
285 		}
286 		if (!AjxXmlDoc._msxmlVers) {
287 			throw new AjxException("MSXML not installed", AjxException.INTERNAL_ERROR, "AjxXmlDoc._init");
288 		}
289 	} else if (AjxEnv.isNav || AjxEnv.isOpera || AjxEnv.isSafari) {
290 		// add loadXML to Document's API
291 		Document.prototype.loadXML = function(str) {
292 			var domParser = new DOMParser();
293 			var domObj = domParser.parseFromString(str, "text/xml");
294 			// remove old child nodes since we recycle DOMParser and append new
295 			while (this.hasChildNodes()) {
296 				this.removeChild(this.lastChild);
297 			}
298 			var len = domObj.childNodes.length;
299             for (var i = 0; i < len; i++) {
300                 var importedNode = this.importNode(domObj.childNodes[i], true);
301                 this.appendChild(importedNode);
302             }
303 		}
304 		
305 		if (AjxEnv.isNav) {
306 			_NodeGetXml = function() {
307 				var ser = new XMLSerializer();
308 				return ser.serializeToString(this);
309 			}
310 			Node.prototype.__defineGetter__("xml", _NodeGetXml);
311 		}
312 	}
313 	
314 	AjxXmlDoc._inited = true;
315 };

317 AjxXmlDoc.prototype.set =
318 function(name, value, element) {
319    var p = this._doc.createElement(name);
320       if (value != null) {
321          var cdata = this._doc.createTextNode("");
322          p.appendChild(cdata);
323          cdata.nodeValue = value;
324       }
325       if (element == null) {
326          this.root.appendChild(p);
327       } else {
328          element.appendChild(p);
329       }
330    return p;
331 };

333 AjxXmlDoc.prototype.getDocXml =
334 function() {
335    if (AjxEnv.isSafari)
336       return AjxXmlDoc.getXml(this.getDoc());
337    else
338       return AjxXmlDoc.replaceInvalidChars(this.getDoc().xml);
339 };

341 /**
342  * Creates an XML document with a root element.
343  * 
344  * @param	{string}	rootName	the root name
345  * @return	{AjxXmlDoc}	the XML document
346  */
347 AjxXmlDoc.createRoot =
348 function(rootName) {
349    var xmldoc = AjxXmlDoc.create();
350    var d = xmldoc.getDoc();
351    xmldoc.root = d.createElement(rootName);
352 
353    d.appendChild(xmldoc.root);
354    return xmldoc;
355 };
356 
357 /**
358  * Creates an XML document with the element.
359  * 
360  * @param	{string}	name	the element name
361  * @param	{string}	value	the element value
362  * @return	{AjxXmlDoc}	the XML document
363  */
364 AjxXmlDoc.createElement =
365 function(name, value) {
366 	
367    var xmldoc = AjxXmlDoc.create();
368    var d = xmldoc.getDoc();
369    xmldoc.root = d.createElement(name);
370    if (value != null) {
371    		//xmldoc.root.nodeValue = value;
372    	 	var cdata = d.createTextNode("");
373         xmldoc.root.appendChild(cdata);
374         cdata.nodeValue = value;
375    }
376    
377    d.appendChild(xmldoc.root);
378    return xmldoc;
379    
380 };

383 AjxXmlDoc.prototype.appendChild =
384 function(xmldoc){
385    //Security Exception WRONG_DOCUMENT_ERR thrown when we append nodes created of diff. documents
386    //Chrome/Safari does not like it.
387    if(this._doc != xmldoc._doc && ( AjxEnv.isChrome || AjxEnv.isSafari )){
388         this.root.appendChild(this.getDoc().importNode(xmldoc.root, true));
389    }else{
390         this.root.appendChild(xmldoc.root);
391    }
392 };
