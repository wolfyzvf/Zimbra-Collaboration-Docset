 22 /**
 23  * Default constructor for helper class.
 24  * @class
 25  * Miscellaneous contacts-related utility functions. So far, mostly things that
 26  * {@link ZmContactPicker} and {@link ZmGroupView} both need to perform a contacts search and
 27  * display results in a list view.
 28  *
 29  * @author Conrad Damon
 30  */
 31 ZmContactsHelper = function() {};

 33 /**
 34  * Performs a contact search (in either personal contacts or in the GAL) and populates
 35  * the source list view with the results.
 36  *
 37  * @param	{Hash}	params		a hash of parameters
 38  * @param {Object}	params.obj			the object that is doing the search
 39  * @param {String}	params.query			the query string to search on
 40  * @param {String}	params.queryHint		the query hint (i.e. searching shared folders)
 41  * @param {Boolean}	params.ascending		if <code>true</code>, sort in ascending order
 42  * @param {int}	params.lastId		the ID of last item displayed (for pagination)
 43  * @param {String}	params.lastSortVal	the value of sort field for above item
 44  * @param {AjxCallback}	params.respCallback	the callback to call once response comes back from server
 45  * @param {AjxCallback}	params.errorCallback	the callback to call if error returned from server
 46  * @param {String}	params.accountName	the account to make search request on behalf of
 47  * @param {Array}   params.conds	the conds to restrict the search by (array of {attr:"", op:"", value:""} hashes)
 48  */
 49 ZmContactsHelper.search =
 50 function(params) {
 51 	var o = params.obj;
 52 	if (o._searchButton) {
 53 		o._searchButton.setEnabled(false);
 54 	}
 55 
 56 	params.sortBy = params.ascending ? ZmSearch.NAME_ASC : ZmSearch.NAME_DESC;
 57 	params.types = AjxVector.fromArray([ZmItem.CONTACT]);
 58 	params.offset = params.offset || 0;
 59 	params.limit = ZmContactsApp.SEARCHFOR_MAX;
 60 	params.contactSource = o._contactSource;
 61 	params.field = "contact";
 62 
 63 	var search = new ZmSearch(params);
 64 	search.execute({callback:params.respCallback, errorCallback:params.errorCallback});
 65 };

 67 /**
 68  * Take the contacts and create a list of their email addresses (a contact may have more than one)
 69  * 
 70  * @private
 71  */
 72 ZmContactsHelper._processSearchResponse = 
 73 function(resp, includeContactsWithNoEmail) {
 74 	var vec = resp.getResults(ZmItem.CONTACT);
 75 
 76 	// Take the contacts and create a list of their email addresses (a contact may have more than one)
 77 	var list = [];
 78 	var a = vec.getArray();
 79 	for (var i = 0; i < a.length; i++) {
 80 		var contact = a[i];
 81 		if (contact.isGroup() && !contact.isDL) {
 82 			var members = contact.getGroupMembers().good.toString(AjxEmailAddress.SEPARATOR);
 83 			ZmContactsHelper._addContactToList(list, contact, members, true);
 84 		} else {
 85 			var emails = contact.isGal ? [contact.getEmail()] : contact.getEmails();
 86 			for (var j = 0; j < emails.length; j++) {
 87 				ZmContactsHelper._addContactToList(list, contact, emails[j]);
 88 			}
 89 			if (includeContactsWithNoEmail && emails.length == 0) {
 90 				ZmContactsHelper._addContactToList(list, contact, null);
 91 			}
 92 		}
 93 	}
 94 	
 95 	return list;
 96 };
 97 
 98 /**
 99  * @private
100  */
101 ZmContactsHelper._addContactToList = 
102 function(list, contact, addr, isGroup) {
103 
104 	var email = ZmContactsHelper._wrapContact(contact, addr, isGroup);  
105 	list.push(email);
106 };

108 /**
109  * wrapps the contact inside a AjxEmailAddress object, and adds a couple extra fields to the AjxEmailAddress instance (value, contact, icon [which I'm not sure is used])
110  *
111  * @param contact
112  * @param addr {String} optional.
113  * @param isGroup
114  */
115 ZmContactsHelper._wrapContact =
116 function(contact, addr, isGroup) {
117 
118 	addr = addr || contact.getEmail();
119 	var fileAs = contact.getFileAs();
120 	var name = (fileAs != addr) ? fileAs : "";  //todo ??? this is weird.
121 	var	type = contact.isGal ? ZmContact.GROUP_GAL_REF : ZmContact.GROUP_CONTACT_REF;
122 	var	value = contact.isGal ? (contact.ref || contact.id) : contact.id;  //defaulting to contact.id in the gal case since from GetContactsResponse the ref is not returned and we can end up with it cached without the ref. Probably need to fix that.
123 	var displayName = contact.getFullNameForDisplay();
124 
125 	var email = new AjxEmailAddress(addr, type, name, displayName, isGroup);
126 
127 	email.value = value;
128 	email.id = Dwt.getNextId();
129 	email.__contact = contact;
130 	email.icon = contact.getIcon();
131 	if (contact.isDL) {
132 		email.isGroup = true;
133 		email.canExpand = contact.canExpand;
134 		var ac = window.parentAppCtxt || window.appCtxt;
135 		ac.setIsExpandableDL(addr, email.canExpand);
136 	}
137 	return email;
138 };

140 /**
141  * wrapps the inline address (there's no real ZmContact object) inside AjxEmailAddress and adds the value attribute to it.
142  * this is so we treat real contacts and inline contacts consistently throughout the rest of the code.
143  *
144  * @param value  {String} - the inline email address and/or name (e.g. "john doe <john@doe.com>" or "john@doe.com")
145  */
146 ZmContactsHelper._wrapInlineContact =
147 function(value) {
148 	var email = AjxEmailAddress.parse(value); //from legacy data at least (not sure about new), the format might be something like "Inigo Montoya <inigo@theprincessbride.com>" so we have to parse.
149 	if (!email) {
150 		//this can happen when creating inline in contact group edit, and the user did not suply email address in the inline value
151 		email = new AjxEmailAddress(value, null, value);
152 	}
153 	email.type = ZmContact.GROUP_INLINE_REF;
154 	email.value = value;
155 	email.id = Dwt.getNextId();
156 	return email;
157 };

160 /**
161  * The items are AjxEmailAddress objects
162  * 
163  * @private
164  */
165 ZmContactsHelper._getEmailField =
166 function(html, idx, item, field, colIdx) {
167 	if (field == ZmItem.F_TYPE) {
168 		html[idx++] = AjxImg.getImageHtml(item.icon);
169 	} else if (field == ZmItem.F_NAME) {
170 		html[idx++] = "<nobr>";
171 		html[idx++] = AjxStringUtil.htmlEncode(item.name || ZmMsg.noName);
172 		html[idx++] = "</nobr>";
173 	} else if (field == ZmItem.F_EMAIL) {
174 		html[idx++] = AjxStringUtil.htmlEncode(item.address);
175 	} else if (field == ZmItem.F_DEPARTMENT) {
176 		if (item.__contact) {
177 			html[idx++] = AjxStringUtil.htmlEncode(ZmContact.getAttr(item.__contact, ZmContact.F_department));
178 		}
179 	}
180 	return idx;
181 };
