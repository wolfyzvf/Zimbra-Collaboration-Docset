 78 /**
 79  * Sets the auth token.
 80  * 
 81  * @param	{String}	authToken		the auth token
 82  * @param	{int}		lifetimeMs		the token lifetime in milliseconds
 83  * @param	{String}	sessionId		the session id
 84  * @param	{Boolean}	secure		<code>true</code> for secure
 85  * 
 86  */
 87 ZmCsfeCommand.setAuthToken =
 88 function(authToken, lifetimeMs, sessionId, secure) {
 89 	ZmCsfeCommand._curAuthToken = authToken;
 90 	if (lifetimeMs != null) {
 91 		var exp = null;
 92 		if(lifetimeMs > 0) {
 93 			exp = new Date();
 94 			var lifetime = parseInt(lifetimeMs);
 95 			exp.setTime(exp.getTime() + lifetime);
 96 		}
 97 		AjxCookie.setCookie(document, ZmCsfeCommand._COOKIE_NAME, authToken, exp, "/", null, secure);
 98 	} else {
 99 		AjxCookie.deleteCookie(document, ZmCsfeCommand._COOKIE_NAME, "/");
100 	}
101 	if (sessionId) {
102 		ZmCsfeCommand.setSessionId(sessionId);
103 	}
104 };

125 /**
126  * Sets the session id and, if the session id is new, designates the previous
127  * session id as stale.
128  * 
129  * @param	{String}	sessionId		the session id
130  * 
131  */
132 ZmCsfeCommand.setSessionId =
133 function(sessionId) {
134     var sid = ZmCsfeCommand.extractSessionId(sessionId);
135     if (sid) {
136         if (sid && !ZmCsfeCommand._staleSession[sid]) {
137             if (sid != ZmCsfeCommand._sessionId) {
138                 if (ZmCsfeCommand._sessionId) {
139                     // Mark the old session as stale...
140                     ZmCsfeCommand._staleSession[ZmCsfeCommand._sessionId] = true;
141                 }
142                 // ...before accepting the new session.
143                 ZmCsfeCommand._sessionId = sid;
144             }
145         }
146     }
147 };

154 /**
155  * Isolates the parsing of the various forms of session types that we
156  * might have to handle.
157  *
158  * @param {mixed} session Any valid session object: string, number, object,
159  * or array.
160  * @return {Number|Null} If the input contained a valid session object, the
161  * session number will be returned. If the input is not valid, null will
162  * be returned.
163  */
164 ZmCsfeCommand.extractSessionId =
165 function(session) {
166     var id;
167 
168     if (session instanceof Array) {
169         // Array form
170 	    session = session[0].id;
171     }
172     else if (session && session.id) {
173         // Object form
174         session = session.id;
175     }
176 
177     // We either have extracted the id or were given some primitive form.
178     // Whatever we have at this point, attempt conversion and clean up response.
179     id = parseInt(session, 10);
180     // Normalize response
181     if (isNaN(id)) {
182         id = null;
183     }
184 
185 	return id;
186 };

188 /**
189  * Converts a fault to an exception.
190  * 
191  * @param	{Hash}	fault		the fault
192  * @param	{Hash}	params		a hash of parameters
193  * @return	{ZmCsfeException}	the exception
194  */
195 ZmCsfeCommand.faultToEx =
196 function(fault, params) {
197 	var newParams = {
198 		msg: AjxStringUtil.getAsString(fault.Reason.Text),
199 		code: AjxStringUtil.getAsString(fault.Detail.Error.Code),
200 		method: (params ? params.methodNameStr : null),
201 		detail: AjxStringUtil.getAsString(fault.Code.Value),
202 		data: fault.Detail.Error.a,
203 		trace: (fault.Detail.Error.Trace || "")
204 	};
205 
206 	var request;
207 	if (params) {
208 		if (params.soapDoc) {
209 			// note that we don't pretty-print XML if we get a soapDoc
210 			newParams.request = params.soapDoc.getXml();
211 		} else if (params.jsonRequestObj) {
212 			if (params.jsonRequestObj && params.jsonRequestObj.Header && params.jsonRequestObj.Header.context) {
213 				params.jsonRequestObj.Header.context.authToken = "(removed)";
214 			}
215 			newParams.request = AjxStringUtil.prettyPrint(params.jsonRequestObj, true);
216 		}
217 	}
218 
219 	return new ZmCsfeException(newParams);
220 };

222 /**
223  * Gets the method name of the given request or response.
224  *
225  * @param {AjxSoapDoc|Object}	request	the request
226  * @return	{String}			the method name or "[unknown]"
227  */
228 ZmCsfeCommand.getMethodName =
229 function(request) {
230 
231 	// SOAP request
232 	var methodName = (request && request._methodEl && request._methodEl.tagName)
233 		? request._methodEl.tagName : null;
234 
235 	if (!methodName) {
236 		for (var prop in request) {
237 			if (/Request|Response$/.test(prop)) {
238 				methodName = prop;
239 				break;
240 			}
241 		}
242 	}
243 	return (methodName || "[unknown]");
244 };

246 /**
247  * Sends a SOAP request to the server and processes the response. The request can be in the form
248  * of a SOAP document, or a JSON object.
249  *
250  * @param	{Hash}			params				a hash of parameters:
251  * @param	{AjxSoapDoc}	soapDoc				the SOAP document that represents the request
252  * @param	{Object}		jsonObj				the JSON object that represents the request (alternative to soapDoc)
253  * @param	{Boolean}		noAuthToken			if <code>true</code>, the check for an auth token is skipped
254  * @param	{Boolean}		authToken			authToken to use instead of the local one
255  * @param	{String}		serverUri			the URI to send the request to
256  * @param	{String}		targetServer		the host that services the request
257  * @param	{Boolean}		useXml				if <code>true</code>, an XML response is requested
258  * @param	{Boolean}		noSession			if <code>true</code>, no session info is included
259  * @param	{String}		changeToken			the current change token
260  * @param	{int}			highestNotifySeen 	the sequence # of the highest notification we have processed
261  * @param	{Boolean}		asyncMode			if <code>true</code>, request sent asynchronously
262  * @param	{AjxCallback}	callback			the callback to run when response is received (async mode)
263  * @param	{Boolean}		logRequest			if <code>true</code>, SOAP command name is appended to server URL
264  * @param	{String}		accountId			the ID of account to execute on behalf of
265  * @param	{String}		accountName			the name of account to execute on behalf of
266  * @param	{Boolean}		skipAuthCheck		if <code>true</code> to skip auth check (i.e. do not check if auth token has changed)
267  * @param	{constant}		resend				the reason for resending request
268  * @param	{boolean}		useStringify1		use JSON.stringify1 (gets around IE child win issue with Array)
269  * @param	{boolean}		emptyResponseOkay	if true, empty or no response from server is not an erro
270  */
271 ZmCsfeCommand.prototype.invoke =
272 function(params) {
273 	this.cancelled = false;
274 	if (!(params && (params.soapDoc || params.jsonObj))) { return; }
275 
276 	var requestStr = ZmCsfeCommand.getRequestStr(params);
277 
278 	var rpcCallback;
279 	try {
280 		var uri = (params.serverUri || ZmCsfeCommand.serverUri) + params.methodNameStr;
281 		this._st = new Date();
282 		
283 		var requestHeaders = {"Content-Type": "application/soap+xml; charset=utf-8"};
284 		if (AjxEnv.isIE6 && (location.protocol == "https:")) { //bug 22829
285 			requestHeaders["Connection"] = "Close";
286 		}
287 			
288 		if (params.asyncMode) {
289 			//DBG.println(AjxDebug.DBG1, "set callback for asynchronous response");
290 			rpcCallback = new AjxCallback(this, this._runCallback, [params]);
291 			this._rpcId = AjxRpc.invoke(requestStr, uri, requestHeaders, rpcCallback);
292 		} else {
293 			//DBG.println(AjxDebug.DBG1, "parse response synchronously");
294 			var response = AjxRpc.invoke(requestStr, uri, requestHeaders);
295 			return (!params.returnXml) ? (this._getResponseData(response, params)) : response;
296 		}
297 	} catch (ex) {
298 		this._handleException(ex, params, rpcCallback);
299 	}
300 };

302 /**
303  * Sends a REST request to the server via GET and returns the response.
304  *
305  * @param {Hash}	params			a hash of parameters
306  * @param	{String}       params.restUri			the REST URI to send the request to
307  * @param	{Boolean}       params.asyncMode			if <code>true</code> request sent asynchronously
308  * @param	{AjxCallback}	params.callback			the callback to run when response is received (async mode)
309  */
310 ZmCsfeCommand.prototype.invokeRest =
311 function(params) {
312 
313 	if (!(params && params.restUri)) { return; }
314 
315 	var rpcCallback;
316 	try {
317 		this._st = new Date();
318 		if (params.asyncMode) {
319 			rpcCallback = new AjxCallback(this, this._runCallback, [params]);
320 			this._rpcId = AjxRpc.invoke(null, params.restUri, null, rpcCallback, true);
321 		} else {
322 			var response = AjxRpc.invoke(null, params.restUri, null, null, true);
323 			return response.text;
324 		}
325 	} catch (ex) {
326 		this._handleException(ex, params, rpcCallback);
327 	}
328 };

330 /**
331  * Cancels this request (which must be async).
332  * 
333  */
334 ZmCsfeCommand.prototype.cancel =
335 function() {
336 	DBG.println("req", "CSFE cancel: " + this._rpcId);
337 	if (!this._rpcId) { return; }
338 	this.cancelled = true;
339 	var req = AjxRpc.getRpcRequestById(this._rpcId);
340 	if (req) {
341 		req.cancel();
342 		if (AjxEnv.isFirefox3_5up) {
343 			AjxRpc.removeRpcCtxt(req);
344 		}
345 	}
346 };

359 /**
360  * @private
361  */
362 ZmCsfeCommand._getJsonRequestStr =
363 function(params) {
364 
365 	var obj = {Header:{}, Body:params.jsonObj};
366 
367 	var context = obj.Header.context = {_jsns:"urn:zimbra"};
368 	var ua_name = ["ZimbraWebClient - ", AjxEnv.browser, " (", AjxEnv.platform, ")"].join("");
369 	context.userAgent = {name:ua_name};
370 	if (ZmCsfeCommand.clientVersion) {
371 		context.userAgent.version = ZmCsfeCommand.clientVersion;
372 	}
373 	if (params.noSession) {
374 		context.nosession = {};
375 	} else {
376 		var sessionId = ZmCsfeCommand.getSessionId();
377 		if (sessionId) {
378 			context.session = {_content:sessionId, id:sessionId};
379 		} else {
380 			context.session = {};
381 		}
382 	}
383 	if (params.targetServer) {
384 		context.targetServer = {_content:params.targetServer};
385 	}
386 	if (params.highestNotifySeen) {
387 		context.notify = {seq:params.highestNotifySeen};
388 	}
389 	if (params.changeToken) {
390 		context.change = {token:params.changeToken, type:"new"};
391 	}
392 
393 	// if we're not checking auth token, we don't want token/acct mismatch	
394 	if (!params.skipAuthCheck) {
395 		if (params.accountId) {
396 			context.account = {_content:params.accountId, by:"id"}
397 		} else if (params.accountName) {
398 			context.account = {_content:params.accountName, by:"name"}
399 		}
400 	}
401 	
402 	// Tell server what kind of response we want
403 	if (params.useXml) {
404 		context.format = {type:"xml"};
405 	}
406 
407 	params.methodNameStr = ZmCsfeCommand.getMethodName(params.jsonObj);
408 
409 	// Get auth token from cookie if required
410 	if (!params.noAuthToken) {
411 		var authToken = params.authToken || ZmCsfeCommand.getAuthToken();
412 		if (!authToken) {
413 			throw new ZmCsfeException("AuthToken required", ZmCsfeException.NO_AUTH_TOKEN, params.methodNameStr);
414 		}
415 		if (ZmCsfeCommand._curAuthToken && !params.skipAuthCheck && 
416 			(params.resend != ZmCsfeCommand.REAUTH) && (authToken != ZmCsfeCommand._curAuthToken)) {
417 			throw new ZmCsfeException("AuthToken has changed", ZmCsfeException.AUTH_TOKEN_CHANGED, params.methodNameStr);
418 		}
419 		context.authToken = ZmCsfeCommand._curAuthToken = authToken;
420 	}
421 	else if (ZmCsfeCommand.noAuth) {
422 		throw new ZmCsfeException("Auth required", ZmCsfeException.NO_AUTH_TOKEN, params.methodNameStr);
423 	}
424 
425 	AjxDebug.logSoapMessage(params);
426 	DBG.dumpObj(AjxDebug.DBG1, obj);
427 
428 	params.jsonRequestObj = obj;
429 	
430 	var requestStr = (params.useStringify1 ?
431 	                  JSON.stringify1(obj) : JSON.stringify(obj));
432 
433 	// bug 74240: escape non-ASCII characters to prevent the browser from
434 	// combining decomposed characters in paths
435 	return AjxStringUtil.jsEncode(requestStr)
436 };

438 /**
439  * @private
440  */
441 ZmCsfeCommand._getSoapRequestStr =
442 function(params) {
443 
444 	var soapDoc = params.soapDoc;
445 
446 	if (!params.resend) {
447 
448 		// Add the SOAP header and context
449 		var hdr = soapDoc.createHeaderElement();
450 		var context = soapDoc.set("context", null, hdr, "urn:zimbra");
451 	
452 		var ua = soapDoc.set("userAgent", null, context);
453 		var name = ["ZimbraWebClient - ", AjxEnv.browser, " (", AjxEnv.platform, ")"].join("");
454 		ua.setAttribute("name", name);
455 		if (ZmCsfeCommand.clientVersion) {
456 			ua.setAttribute("version", ZmCsfeCommand.clientVersion);
457 		}
458 	
459 		if (params.noSession) {
460 			soapDoc.set("nosession", null, context);
461 		} else {
462 			var sessionId = ZmCsfeCommand.getSessionId();
463 			var si = soapDoc.set("session", null, context);
464 			if (sessionId) {
465 				si.setAttribute("id", sessionId);
466 			}
467 		}
468 		if (params.targetServer) {
469 			soapDoc.set("targetServer", params.targetServer, context);
470 		}
471 		if (params.highestNotifySeen) {
472 		  	var notify = soapDoc.set("notify", null, context);
473 		  	notify.setAttribute("seq", params.highestNotifySeen);
474 		}
475 		if (params.changeToken) {
476 			var ct = soapDoc.set("change", null, context);
477 			ct.setAttribute("token", params.changeToken);
478 			ct.setAttribute("type", "new");
479 		}
480 	
481 		// if we're not checking auth token, we don't want token/acct mismatch	
482 		if (!params.skipAuthCheck) {
483 			if (params.accountId) {
484 				var acc = soapDoc.set("account", params.accountId, context);
485 				acc.setAttribute("by", "id");
486 			} else if (params.accountName) {
487 				var acc = soapDoc.set("account", params.accountName, context);
488 				acc.setAttribute("by", "name");
489 			}
490 		}
491 	
492 		if (params.skipExpiredToken) {
493 			var tokenControl = soapDoc.set("authTokenControl", null, context);
494 			tokenControl.setAttribute("voidOnExpired", "1");
495 		}	
496 		// Tell server what kind of response we want
497 		if (!params.useXml) {
498 			var js = soapDoc.set("format", null, context);
499 			js.setAttribute("type", "js");
500 		}
501 	}
502 
503 	params.methodNameStr = ZmCsfeCommand.getMethodName(soapDoc);
504 
505 	// Get auth token from cookie if required
506 	if (!params.noAuthToken) {
507 		var authToken = params.authToken || ZmCsfeCommand.getAuthToken();
508 		if (!authToken) {
509 			throw new ZmCsfeException("AuthToken required", ZmCsfeException.NO_AUTH_TOKEN, params.methodNameStr);
510 		}
511 		if (ZmCsfeCommand._curAuthToken && !params.skipAuthCheck && 
512 			(params.resend != ZmCsfeCommand.REAUTH) && (authToken != ZmCsfeCommand._curAuthToken)) {
513 			throw new ZmCsfeException("AuthToken has changed", ZmCsfeException.AUTH_TOKEN_CHANGED, params.methodNameStr);
514 		}
515 		ZmCsfeCommand._curAuthToken = authToken;
516 		if (params.resend == ZmCsfeCommand.REAUTH) {
517 			// replace old auth token with current one
518 			var nodes = soapDoc.getDoc().getElementsByTagName("authToken");
519 			if (nodes && nodes.length == 1) {
520 				DBG.println(AjxDebug.DBG1, "Re-auth: replacing auth token");
521 				nodes[0].firstChild.data = authToken;
522 			} else {
523 				// can't find auth token, just add it to context element
524 				nodes = soapDoc.getDoc().getElementsByTagName("context");
525 				if (nodes && nodes.length == 1) {
526 					DBG.println(AjxDebug.DBG1, "Re-auth: re-adding auth token");
527 					soapDoc.set("authToken", authToken, nodes[0]);
528 				} else {
529 					DBG.println(AjxDebug.DBG1, "Re-auth: could not find context!");
530 				}
531 			}
532 		} else if (!params.resend){
533 			soapDoc.set("authToken", authToken, context);
534 		}
535 	}
536 	else if (ZmCsfeCommand.noAuth && !params.ignoreAuthToken) {
537 		throw new ZmCsfeException("Auth required", ZmCsfeException.NO_AUTH_TOKEN, params.methodNameStr);
538 	}
539 
540 	AjxDebug.logSoapMessage(params);
541 	DBG.printXML(AjxDebug.DBG1, soapDoc.getXml());
542 
543 	return soapDoc.getXml();
544 };

546 /**
547  * Runs the callback that was passed to invoke() for an async command.
548  *
549  * @param {AjxCallback}	callback	the callback to run with response data
550  * @param {Hash}	params	a hash of parameters (see method invoke())
551  * 
552  * @private
553  */
554 ZmCsfeCommand.prototype._runCallback =
555 function(params, result) {
556 	if (!result) { return; }
557 	if (this.cancelled && params.skipCallbackIfCancelled) {	return; }
558 
559 	var response;
560 	if (result instanceof ZmCsfeResult) {
561 		response = result; // we already got an exception and packaged it
562 	} else {
563 		response = this._getResponseData(result, params);
564 	}
565 	this._en = new Date();
566 
567 	if (params.callback && response) {
568 		params.callback.run(response);
569 	} else if (!params.emptyResponseOkay) {
570 		DBG.println(AjxDebug.DBG1, "ZmCsfeCommand.prototype._runCallback: Missing callback!");
571 	}
572 };

574 /**
575  * Takes the response to an RPC request and returns a JS object with the response data.
576  *
577  * @param {Object}	response	the RPC response with properties "text" and "xml"
578  * @param {Hash}	params	a hash of parameters (see method invoke())
579  */
580 ZmCsfeCommand.prototype._getResponseData =
581 function(response, params) {
582 	this._en = new Date();
583 	DBG.println(AjxDebug.DBG1, "ROUND TRIP TIME: " + (this._en.getTime() - this._st.getTime()));
584 
585 	var result = new ZmCsfeResult();
586 	var xmlResponse = false;
587 	var restResponse = Boolean(params.restUri);
588 	var respDoc = null;
589 
590 	// check for un-parseable HTML error response from server
591 	if (!response.success && !response.xml && (/<html/i.test(response.text))) {
592 		// bad XML or JS response that had no fault
593 		var ex = new ZmCsfeException(null, ZmCsfeException.CSFE_SVC_ERROR, params.methodNameStr, "HTTP response status " + response.status);
594 		if (params.asyncMode) {
595 			result.set(ex, true);
596 			return result;
597 		} else {
598 			throw ex;
599 		}
600 	}
601 
602 	if (typeof(response.text) == "string" && response.text.indexOf("{") == 0) {
603 		respDoc = response.text;
604 	} else if (!restResponse) {
605 		// an XML response if we requested one, or a fault
606 		try {
607 			xmlResponse = true;
608 			if (!(response.text || (response.xml && (typeof response.xml) == "string"))) {
609 				if (params.emptyResponseOkay) {
610 					return null;
611 				}
612 				else {
613 					// If we can't reach the server, req returns immediately with an empty response rather than waiting and timing out
614 					throw new ZmCsfeException(null, ZmCsfeException.EMPTY_RESPONSE, params.methodNameStr);
615 				}
616 			}
617 			// responseXML is empty under IE
618 			respDoc = (AjxEnv.isIE || response.xml == null) ? AjxSoapDoc.createFromXml(response.text) :
619 															  AjxSoapDoc.createFromDom(response.xml);
620 		} catch (ex) {
621 			DBG.dumpObj(AjxDebug.DBG1, ex);
622 			if (params.asyncMode) {
623 				result.set(ex, true);
624 				return result;
625 			} else {
626 				throw ex;
627 			}
628 		}
629 		if (!respDoc) {
630 			var ex = new ZmCsfeException(null, ZmCsfeException.SOAP_ERROR, params.methodNameStr, "Bad XML response doc");
631 			DBG.dumpObj(AjxDebug.DBG1, ex);
632 			if (params.asyncMode) {
633 				result.set(ex, true);
634 				return result;
635 			} else {
636 				throw ex;
637 			}
638 		}
639 	}
640 
641 	var obj = restResponse ? response.text : {};
642 
643 	if (xmlResponse) {
644 		DBG.printXML(AjxDebug.DBG1, respDoc.getXml());
645 		obj = respDoc._xmlDoc.toJSObject(true, false, true);
646 	} else if (!restResponse) {
647 		try {
648 			obj = JSON.parse(respDoc);
649 		} catch (ex) {
650 			if (ex.name == "SyntaxError") {
651 				ex = new ZmCsfeException(null, ZmCsfeException.BAD_JSON_RESPONSE, params.methodNameStr, respDoc);
652 				AjxDebug.println(AjxDebug.BAD_JSON, "bad json. respDoc=" + respDoc);
653 			}
654 			DBG.dumpObj(AjxDebug.DBG1, ex);
655 			if (params.asyncMode) {
656 				result.set(ex, true);
657 				return result;
658 			} else {
659 				throw ex;
660 			}
661 		}
662 
663 	}
664 
665 	params.methodNameStr = ZmCsfeCommand.getMethodName(obj.Body);
666 	AjxDebug.logSoapMessage(params);
667 	DBG.dumpObj(AjxDebug.DBG1, obj, -1);
668 
669 	var fault = obj && obj.Body && obj.Body.Fault;
670 	if (fault) {
671 		// JS response with fault
672 		if (AjxUtil.isString(fault) && fault.indexOf("<")==0) { // We got an xml string
673 			fault = AjxXmlDoc.createFromXml(fault).toJSObject(true, false, true);
674 		}
675 		var ex = ZmCsfeCommand.faultToEx(fault, params);
676 		if (params.asyncMode) {
677 			result.set(ex, true, obj.Header);
678 			return result;
679 		} else {
680 			throw ex;
681 		}
682 	} else if (!response.success) {
683 		// bad XML or JS response that had no fault
684 		var ex = new ZmCsfeException(null, ZmCsfeException.CSFE_SVC_ERROR, params.methodNameStr, "HTTP response status " + response.status);
685 		if (params.asyncMode) {
686 			result.set(ex, true);
687 			return result;
688 		} else {
689 			throw ex;
690 		}
691 	} else {
692 		// good response
693 		if (params.asyncMode) {
694 			result.set(obj);
695 		}
696 	}
697 
698 	// check for new session ID
699 	var session = obj.Header && obj.Header.context && obj.Header.context.session;
700     ZmCsfeCommand.setSessionId(session);
701 
702 	return params.asyncMode ? result : obj;
703 };

705 /**
706  * @private
707  */
708 ZmCsfeCommand.prototype._handleException =
709 function(ex, params, callback) {
710 	if (!(ex && (ex instanceof ZmCsfeException || ex instanceof AjxSoapException || ex instanceof AjxException))) {
711 		var newEx = new ZmCsfeException();
712 		newEx.method = params.methodNameStr || params.restUri;
713 		newEx.detail = ex ? ex.toString() : "undefined exception";
714 		newEx.code = ZmCsfeException.UNKNOWN_ERROR;
715 		newEx.msg = "Unknown Error";
716 		ex = newEx;
717 	}
718 	if (params.asyncMode) {
719 		callback.run(new ZmCsfeResult(ex, true));
720 	} else {
721 		throw ex;
722 	}
723 };
