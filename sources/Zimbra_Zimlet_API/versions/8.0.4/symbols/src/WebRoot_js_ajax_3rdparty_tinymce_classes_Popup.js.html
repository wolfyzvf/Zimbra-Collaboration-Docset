 27 // Some global instances
 28 var tinymce = null, tinyMCEPopup, tinyMCE;
 29 
 30 /**
 31  * TinyMCE popup/dialog helper class. This gives you easy access to the
 32  * parent editor instance and a bunch of other things. It's higly recommended
 33  * that you load this script into your dialogs.
 34  *
 35  * @static
 36  * @class tinyMCEPopup
 37  */
 38 tinyMCEPopup = {
 39 	/**
 40 	 * Initializes the popup this will be called automatically.
 41 	 *
 42 	 * @method init
 43 	 */
 44 	init : function() {
 45 		var t = this, w, ti;
 46 
 47 		// Find window & API
 48 		w = t.getWin();
 49 		tinymce = w.tinymce;
 50 		tinyMCE = w.tinyMCE;
 51 		t.editor = tinymce.EditorManager.activeEditor;
 52 		t.params = t.editor.windowManager.params;
 53 		t.features = t.editor.windowManager.features;
 54 
 55 		// Setup local DOM
 56 		t.dom = t.editor.windowManager.createInstance('tinymce.dom.DOMUtils', document, {ownEvents: true, proxy: tinyMCEPopup._eventProxy});
 57 		t.dom.bind(window, 'ready', t._onDOMLoaded, t);
 58 
 59 		// Enables you to skip loading the default css
 60 		if (t.features.popup_css !== false)
 61 			t.dom.loadCSS(t.features.popup_css || t.editor.settings.popup_css);
 62 
 63 		// Setup on init listeners
 64 		t.listeners = [];
 65 
 66 		/**
 67 		 * Fires when the popup is initialized.
 68 		 *
 69 		 * @event onInit
 70 		 * @param {tinymce.Editor} editor Editor instance.
 71 		 * @example
 72 		 * // Alerts the selected contents when the dialog is loaded
 73 		 * tinyMCEPopup.onInit.add(function(ed) {
 74 		 *     alert(ed.selection.getContent());
 75 		 * });
 76 		 * 
 77 		 * // Executes the init method on page load in some object using the SomeObject scope
 78 		 * tinyMCEPopup.onInit.add(SomeObject.init, SomeObject);
 79 		 */
 80 		t.onInit = {
 81 			add : function(f, s) {
 82 				t.listeners.push({func : f, scope : s});
 83 			}
 84 		};
 85 
 86 		t.isWindow = !t.getWindowArg('mce_inline');
 87 		t.id = t.getWindowArg('mce_window_id');
 88 		t.editor.windowManager.onOpen.dispatch(t.editor.windowManager, window);
 89 	},
 90 
 91 	/**
 92 	 * Returns the reference to the parent window that opened the dialog.
 93 	 *
 94 	 * @method getWin
 95 	 * @return {Window} Reference to the parent window that opened the dialog.
 96 	 */
 97 	getWin : function() {
 98 		// Added frameElement check to fix bug: #2817583
 99 		return (!window.frameElement && window.dialogArguments) || opener || parent || top;
100 	},
101 
102 	/**
103 	 * Returns a window argument/parameter by name.
104 	 *
105 	 * @method getWindowArg
106 	 * @param {String} n Name of the window argument to retrive.
107 	 * @param {String} dv Optional default value to return.
108 	 * @return {String} Argument value or default value if it wasn't found.
109 	 */
110 	getWindowArg : function(n, dv) {
111 		var v = this.params[n];
112 
113 		return tinymce.is(v) ? v : dv;
114 	},
115 
116 	/**
117 	 * Returns a editor parameter/config option value.
118 	 *
119 	 * @method getParam
120 	 * @param {String} n Name of the editor config option to retrive.
121 	 * @param {String} dv Optional default value to return.
122 	 * @return {String} Parameter value or default value if it wasn't found.
123 	 */
124 	getParam : function(n, dv) {
125 		return this.editor.getParam(n, dv);
126 	},
127 
128 	/**
129 	 * Returns a language item by key.
130 	 *
131 	 * @method getLang
132 	 * @param {String} n Language item like mydialog.something.
133 	 * @param {String} dv Optional default value to return.
134 	 * @return {String} Language value for the item like "my string" or the default value if it wasn't found.
135 	 */
136 	getLang : function(n, dv) {
137 		return this.editor.getLang(n, dv);
138 	},
139 
140 	/**
141 	 * Executed a command on editor that opened the dialog/popup.
142 	 *
143 	 * @method execCommand
144 	 * @param {String} cmd Command to execute.
145 	 * @param {Boolean} ui Optional boolean value if the UI for the command should be presented or not.
146 	 * @param {Object} val Optional value to pass with the comman like an URL.
147 	 * @param {Object} a Optional arguments object.
148 	 */
149 	execCommand : function(cmd, ui, val, a) {
150 		a = a || {};
151 		a.skip_focus = 1;
152 
153 		this.restoreSelection();
154 		return this.editor.execCommand(cmd, ui, val, a);
155 	},
156 
157 	/**
158 	 * Resizes the dialog to the inner size of the window. This is needed since various browsers
159 	 * have different border sizes on windows.
160 	 *
161 	 * @method resizeToInnerSize
162 	 */
163 	resizeToInnerSize : function() {
164 		var t = this;
165 
166 		// Detach it to workaround a Chrome specific bug
167 		// https://sourceforge.net/tracker/?func=detail&atid=635682&aid=2926339&group_id=103281
168 		setTimeout(function() {
169 			var vp = t.dom.getViewPort(window);
170 
171 			t.editor.windowManager.resizeBy(
172 				t.getWindowArg('mce_width') - vp.w,
173 				t.getWindowArg('mce_height') - vp.h,
174 				t.id || window
175 			);
176 		}, 10);
177 	},
178 
179 	/**
180 	 * Will executed the specified string when the page has been loaded. This function
181 	 * was added for compatibility with the 2.x branch.
182 	 *
183 	 * @method executeOnLoad
184 	 * @param {String} s String to evalutate on init.
185 	 */
186 	executeOnLoad : function(s) {
187 		this.onInit.add(function() {
188 			eval(s);
189 		});
190 	},
191 
192 	/**
193 	 * Stores the current editor selection for later restoration. This can be useful since some browsers
194 	 * looses it's selection if a control element is selected/focused inside the dialogs.
195 	 *
196 	 * @method storeSelection
197 	 */
198 	storeSelection : function() {
199 		this.editor.windowManager.bookmark = tinyMCEPopup.editor.selection.getBookmark(1);
200 	},
201 
202 	/**
203 	 * Restores any stored selection. This can be useful since some browsers
204 	 * looses it's selection if a control element is selected/focused inside the dialogs.
205 	 *
206 	 * @method restoreSelection
207 	 */
208 	restoreSelection : function() {
209 		var t = tinyMCEPopup;
210 
211 		if (!t.isWindow && tinymce.isIE)
212 			t.editor.selection.moveToBookmark(t.editor.windowManager.bookmark);
213 	},
214 
215 	/**
216 	 * Loads a specific dialog language pack. If you pass in plugin_url as a arugment
217 	 * when you open the window it will load the <plugin url>/langs/<code>_dlg.js lang pack file.
218 	 *
219 	 * @method requireLangPack
220 	 */
221 	requireLangPack : function() {
222 		var t = this, u = t.getWindowArg('plugin_url') || t.getWindowArg('theme_url');
223 
224 		if (u && t.editor.settings.language && t.features.translate_i18n !== false && t.editor.settings.language_load !== false) {
225 			u += '/langs/' + t.editor.settings.language + '_dlg.js';
226 
227 			if (!tinymce.ScriptLoader.isDone(u)) {
228 				document.write('<script type="text/javascript" src="' + tinymce._addVer(u) + '"></script>');
229 				tinymce.ScriptLoader.markDone(u);
230 			}
231 		}
232 	},
233 
234 	/**
235 	 * Executes a color picker on the specified element id. When the user
236 	 * then selects a color it will be set as the value of the specified element.
237 	 *
238 	 * @method pickColor
239 	 * @param {DOMEvent} e DOM event object.
240 	 * @param {string} element_id Element id to be filled with the color value from the picker.
241 	 */
242 	pickColor : function(e, element_id) {
243 		this.execCommand('mceColorPicker', true, {
244 			color : document.getElementById(element_id).value,
245 			func : function(c) {
246 				document.getElementById(element_id).value = c;
247 
248 				try {
249 					document.getElementById(element_id).onchange();
250 				} catch (ex) {
251 					// Try fire event, ignore errors
252 				}
253 			}
254 		});
255 	},
256 
257 	/**
258 	 * Opens a filebrowser/imagebrowser this will set the output value from
259 	 * the browser as a value on the specified element.
260 	 *
261 	 * @method openBrowser
262 	 * @param {string} element_id Id of the element to set value in.
263 	 * @param {string} type Type of browser to open image/file/flash.
264 	 * @param {string} option Option name to get the file_broswer_callback function name from.
265 	 */
266 	openBrowser : function(element_id, type, option) {
267 		tinyMCEPopup.restoreSelection();
268 		this.editor.execCallback('file_browser_callback', element_id, document.getElementById(element_id).value, type, window);
269 	},
270 
271 	/**
272 	 * Creates a confirm dialog. Please don't use the blocking behavior of this
273 	 * native version use the callback method instead then it can be extended.
274 	 *
275 	 * @method confirm
276 	 * @param {String} t Title for the new confirm dialog.
277 	 * @param {function} cb Callback function to be executed after the user has selected ok or cancel.
278 	 * @param {Object} s Optional scope to execute the callback in.
279 	 */
280 	confirm : function(t, cb, s) {
281 		this.editor.windowManager.confirm(t, cb, s, window);
282 	},
283 
284 	/**
285 	 * Creates a alert dialog. Please don't use the blocking behavior of this
286 	 * native version use the callback method instead then it can be extended.
287 	 *
288 	 * @method alert
289 	 * @param {String} t Title for the new alert dialog.
290 	 * @param {function} cb Callback function to be executed after the user has selected ok.
291 	 * @param {Object} s Optional scope to execute the callback in.
292 	 */
293 	alert : function(tx, cb, s) {
294 		this.editor.windowManager.alert(tx, cb, s, window);
295 	},
296 
297 	/**
298 	 * Closes the current window.
299 	 *
300 	 * @method close
301 	 */
302 	close : function() {
303 		var t = this;
304 
305 		// To avoid domain relaxing issue in Opera
306 		function close() {
307 			t.editor.windowManager.close(window);
308 			tinymce = tinyMCE = t.editor = t.params = t.dom = t.dom.doc = null; // Cleanup
309 		};
310 
311 		if (tinymce.isOpera)
312 			t.getWin().setTimeout(close, 0);
313 		else
314 			close();
315 	},
316

318 
319 	_restoreSelection : function() {
320 		var e = window.event.srcElement;
321 
322 		if (e.nodeName == 'INPUT' && (e.type == 'submit' || e.type == 'button'))
323 			tinyMCEPopup.restoreSelection();
324 	},

334 	_onDOMLoaded : function() {
335 		var t = tinyMCEPopup, ti = document.title, bm, h, nv;
336 
337 		// Translate page
338 		if (t.features.translate_i18n !== false) {
339 			h = document.body.innerHTML;
340 
341 			// Replace a=x with a="x" in IE
342 			if (tinymce.isIE)
343 				h = h.replace(/ (value|title|alt)=([^"][^\s>]+)/gi, ' $1="$2"')
344 
345 			document.dir = t.editor.getParam('directionality','');
346 
347 			if ((nv = t.editor.translate(h)) && nv != h)
348 				document.body.innerHTML = nv;
349 
350 			if ((nv = t.editor.translate(ti)) && nv != ti)
351 				document.title = ti = nv;
352 		}
353 
354 		if (!t.editor.getParam('browser_preferred_colors', false) || !t.isWindow)
355 			t.dom.addClass(document.body, 'forceColors');
356 
357 		document.body.style.display = '';
358 
359 		// Restore selection in IE when focus is placed on a non textarea or input element of the type text
360 		if (tinymce.isIE) {
361 			document.attachEvent('onmouseup', tinyMCEPopup._restoreSelection);
362 
363 			// Add base target element for it since it would fail with modal dialogs
364 			t.dom.add(t.dom.select('head')[0], 'base', {target : '_self'});
365 		}
366 
367 		t.restoreSelection();
368 		t.resizeToInnerSize();
369 
370 		// Set inline title
371 		if (!t.isWindow)
372 			t.editor.windowManager.setTitle(window, ti);
373 		else
374 			window.focus();
375 
376 		if (!tinymce.isIE && !t.isWindow) {
377 			t.dom.bind(document, 'focus', function() {
378 				t.editor.windowManager.focus(t.id);
379 			});
380 		}
381 
382 		// Patch for accessibility
383 		tinymce.each(t.dom.select('select'), function(e) {
384 			e.onkeydown = tinyMCEPopup._accessHandler;
385 		});
386 
387 		// Call onInit
388 		// Init must be called before focus so the selection won't get lost by the focus call
389 		tinymce.each(t.listeners, function(o) {
390 			o.func.call(o.scope, t.editor);
391 		});
392 
393 		// Move focus to window
394 		if (t.getWindowArg('mce_auto_focus', true)) {
395 			window.focus();
396 
397 			// Focus element with mceFocus class
398 			tinymce.each(document.forms, function(f) {
399 				tinymce.each(f.elements, function(e) {
400 					if (t.dom.hasClass(e, 'mceFocus') && !e.disabled) {
401 						e.focus();
402 						return false; // Break loop
403 					}
404 				});
405 			});
406 		}
407 
408 		document.onkeyup = tinyMCEPopup._closeWinKeyHandler;
409 	},
410 
411 	_accessHandler : function(e) {
412 		e = e || window.event;
413 
414 		if (e.keyCode == 13 || e.keyCode == 32) {
415 			var elm = e.target || e.srcElement;
416 
417 			if (elm.onchange)
418 				elm.onchange();
419 
420 			return tinymce.dom.Event.cancel(e);
421 		}
422 	},
423 
424 	_closeWinKeyHandler : function(e) {
425 		e = e || window.event;
426 
427 		if (e.keyCode == 27)
428 			tinyMCEPopup.close();
429 	},
430 
431 	_eventProxy: function(id) {
432 		return function(evt) {
433 			tinyMCEPopup.dom.events.callNativeHandler(id, evt);
434 		};
435 	}
436 };
