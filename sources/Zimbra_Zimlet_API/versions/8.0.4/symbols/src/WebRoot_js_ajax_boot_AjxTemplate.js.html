 24 AjxTemplate = function() {};

 37 /**
 38  * Sets the base path.
 39  * 
 40  * @param	{string}	basePath		the base path
 41  */
 42 AjxTemplate.setBasePath = function(basePath) {
 43     AjxTemplate._basePath = basePath;
 44 };
 45 /**
 46  * Sets the extension.
 47  * 
 48  * @param	{string}	extension		the extension
 49  */
 50 AjxTemplate.setExtension = function(extension) {
 51     AjxTemplate._extension = extension;
 52 };
 53 
 54 AjxTemplate.register = function(name, func, params, authoritative) {
 55     if (!authoritative && AjxTemplate._templates[name] &&
 56         AjxTemplate._templates[name].authoritative) {
 57         return;
 58     }
 59     AjxTemplate._templates[name] = {
 60         name: name, func: func, params: params || {}, authoritative: authoritative 
 61     };
 62 };
 63 
 64 AjxTemplate.getTemplate = function(name) {
 65     var template = AjxTemplate._templates[name];
 66     return template && template.func;
 67 };
 68 
 69 AjxTemplate.getParams = function(name) {
 70     var template = AjxTemplate._templates[name];
 71     return template && template.params;
 72 };

 74 /**
 75  * Expands the template.
 76  * 
 77  * @param	{string}		name		the template name
 78  * @param	{array}			[data]		the template date
 79  * @param	{array}			[buffer]	the buffer to use for template content
 80  * @return	{string}	the template content		
 81  */
 82 AjxTemplate.expand = function(name, data, buffer) {
 83 	// allow template text to come from document
 84 	if (!AjxTemplate._templates[name] && AjxTemplate.compile) {
 85 		var el = document.getElementById(name);
 86 		if (el) {
 87 			// NOTE: In all major browsers (IE, FF, Saf) the value property
 88 			//       of the textarea will be the literal text of the content.
 89 			//       Using the innerHTML will escape the HTML content which
 90 			//       is not desirable.
 91 			var isTextArea = el.nodeName.toUpperCase() == "TEXTAREA";
 92 			AjxTemplate.compile(name, true, true, isTextArea ? el.value : el.innerHTML);
 93 		}
 94 	}
 95 
 96     var pkg = AjxTemplate.__name2Package(name);
 97     var id = name.replace(/^[^#]*#?/, "");
 98     if (id) {
 99         name = [pkg, id].join("#");
100     }
101 
102     AjxTemplate.require(pkg);
103 
104     var hasBuffer = Boolean(buffer);
105     buffer = buffer || [];
106     var func = AjxTemplate.getTemplate(name);
107     if (func) {
108         try {
109             AjxTemplate._stack.push(pkg);
110             var params = AjxTemplate.getParams(name);
111             func(name, params, data, buffer);
112 	    }
113         catch (e) {
114 	    	buffer.push(this.__formatError(name, e));
115 	    }
116         finally {
117             AjxTemplate._stack.pop();
118         }
119     } else {
120     	buffer.push(this.__formatError(name, "template not found"));
121     }
122 
123     return hasBuffer ? buffer.length : buffer.join("");
124 };
125 
126 /**
127  * Force load of template.
128  * 
129  * @return <code>true</code> if the template is defined
130  * 
131  * @private
132  */
133 AjxTemplate.require = function(name) {
134 	AjxPackage.require({
135 		name: AjxTemplate.__name2Package(name),
136 		basePath: AjxTemplate._basePath,
137 		extension: AjxTemplate._extension
138 	});
139 	return AjxTemplate.getTemplate(name) != null;
140 };
141 
142 // set innerHTML of a DOM element with the results of a template expansion
143 // TODO: have some sort of actual error reporting
144 AjxTemplate.setContent = function(element, name, data) {
145 	if (typeof element == "string") {
146 		element = document.getElementById(element);
147 	}
148 	if (element == null) return;
149 	var html = AjxTemplate.expand(name, data);
150 	element.innerHTML = html;
151 };
152 
153 AjxTemplate.__name2Package = function(name) {
154 	var pkg = name.replace(/#.*$/, "");
155 	if (name.match(/^#/) && AjxTemplate._stack.length > 0) {
156 	    pkg = AjxTemplate._stack[AjxTemplate._stack.length - 1];
157 	}
158 	return pkg;
159 };

161 // temporary API for handling logic errors in templates
162 //	may change to more robust solution later
163 AjxTemplate.__formatError = function(templateName, error) {
164 	return "Error in template '" + templateName + "': " + error;	
165 };
