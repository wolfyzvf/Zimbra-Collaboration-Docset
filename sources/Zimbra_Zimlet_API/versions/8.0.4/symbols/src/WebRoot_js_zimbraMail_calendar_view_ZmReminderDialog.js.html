 21 ZmReminderDialog = function(parent, reminderController, calController, apptType) {
 22 
 23 	// init custom buttons
 24     this._apptType = apptType;
 25     var selectInputId  = "ZmReminderDialog_reminderSelectInput_" + apptType;
 26     var selectButtonId = "ZmReminderDialog_reminderSelectBtn_"   + apptType;
 27     var html = [];
 28     var i = 0;
 29     html[i++] = "<td valign='middle' class='ZmReminderField'>";
 30     html[i++] = ZmMsg.snoozeAll;
 31     html[i++] = "</td><td valign='middle'>";
 32     // Use a table within a dialog button td to cause the selector input field and drop
 33     // down button to align properly and butt up against one another.
 34     html[i++] = "<table cellspacing=0 cellpadding=0><tr></tr><td valign='middle' id='";
 35     html[i++] = selectInputId;
 36     html[i++] = "'></td><td valign='middle' id='";
 37     html[i++] = selectButtonId;
 38     html[i++] = "'></td></tr></table>";
 39     html[i++] = "</td><td valign='middle' class='ZmSnoozeButton' id=\"{0}\"></td>";
 40 
 41 	var snoozeButton = new DwtDialog_ButtonDescriptor(ZmReminderDialog.SNOOZE_BUTTON, ZmMsg.snooze, DwtDialog.ALIGN_LEFT, null, html.join(""));
 42 	var dismissAllButton = new DwtDialog_ButtonDescriptor(ZmReminderDialog.DISMISS_ALL_BUTTON, ZmMsg.dismissAll, DwtDialog.ALIGN_RIGHT);
 43 
 44 	// call base class
 45 	DwtDialog.call(this, {id:"ZmReminderDialog_" + apptType, parent:parent, standardButtons:DwtDialog.NO_BUTTONS, extraButtons:[snoozeButton, dismissAllButton]});
 46 
 47 	this._reminderController = reminderController;
 48 	this._calController = calController;
 49 
 50     this.setContent(this._contentHtml(selectInputId, selectButtonId));
 51     if(this._calController instanceof ZmTaskMgr) {
 52         this.setTitle(ZmMsg.taskReminders);
 53     } else {
 54         this.setTitle(ZmMsg.apptReminders);
 55     }
 56     this.registerCallback(ZmReminderDialog.SNOOZE_BUTTON, this._handleSnoozeButton, this);
 57 	this.registerCallback(ZmReminderDialog.DISMISS_ALL_BUTTON, this._handleDismissAllButton, this);
 58 };

 77 ZmReminderDialog.prototype.popup =
 78 function() {
 79 	DwtDialog.prototype.popup.call(this);
 80 	this._cancelSnooze();
 81 
 82     if (appCtxt.get(ZmSetting.CAL_REMINDER_NOTIFY_BROWSER)) {
 83         AjxPackage.require("Alert");
 84         ZmBrowserAlert.getInstance().start(ZmMsg.reminders);
 85     }
 86 
 87     if (appCtxt.get(ZmSetting.CAL_REMINDER_NOTIFY_SOUNDS)) {
 88         AjxPackage.require("Alert");
 89         ZmSoundAlert.getInstance().start();
 90     }
 91 
 92     if (appCtxt.get(ZmSetting.CAL_REMINDER_NOTIFY_TOASTER)) {
 93         AjxPackage.require("Alert");
 94         var winText = [];
 95         var appts = this._list.getArray();
 96         // only show, at most, five appointment reminders
 97         for (var i = 0; i < appts.length && i < 5; i++) {
 98             var appt = appts[i];
 99             var startDelta = this._computeDelta(appt);
100             var delta = startDelta ? this._formatDeltaString(startDelta) : "";
101             var text = [appt.getName(), ", ", this._getDurationText(appt), "\n(", delta, ")"].join("");
102             if (AjxEnv.isMac) {
103                 ZmDesktopAlert.getInstance().start(ZmMsg.reminders, text, true);
104             } else if (AjxEnv.isWindows) {
105                 winText.push(text);
106             }
107         }
108 
109         if (AjxEnv.isWindows && winText.length > 0) {
110             if (appts.length > 5) {
111                 winText.push(ZmMsg.andMore);
112             }
113             ZmDesktopAlert.getInstance().start(ZmMsg.reminders, winText.join("\n"), true);
114         }
115     }
116 
117     var snoozeSelectMenuListener = this._snoozeSelectMenuListener.bind(this);
118     this._createSnoozeMenu(this._snoozeSelectButton, snoozeSelectMenuListener, this._list);
119 
120 };

122 ZmReminderDialog.prototype.initialize =
123 function(list) {
124 	this._list = new AjxVector();
125 	this._apptData = {};
126 
127 	var html = [];
128 	var idx = 0;
129 	var size = list.size();
130 
131     AjxDebug.println(AjxDebug.REMINDER, "---Reminders [" + (new Date().getTime())+ "]---");
132 
133 	html[idx++] = "<table cellpadding=0 cellspacing=0 border=0 width=100%>";
134 	for (var i = 0; i < size; i++) {
135 		var appt = list.get(i);
136         if (appt.isShared() && appt.isReadOnly()) { continue; }
137         this._list.add(appt);
138 		var uid = appt.getUniqueId(true);
139 		var data = this._apptData[uid] = {appt:appt};
140 		idx = this._addAppt(html, idx, appt, data, (i > 0));
141 	}
142 	html[idx++] = "</table>";
143 
144 	// cleanup before using
145 	if (this._dismissButtons) {
146 		for (var id in this._dismissButtons) {
147 			this._dismissButtons[id].dispose();
148 		}
149 	}
150 	if (this._openButtons) {
151 		for (var id in this._openButtons) {
152 			this._openButtons[id].dispose();
153 		}
154 	}
155 	this._dismissButtons = {};
156 	this._openButtons = {};
157 
158 	var dismissListener = new AjxListener(this, this._dismissButtonListener);
159 	var openListener = new AjxListener(this, this._openButtonListener);
160 	var div = document.getElementById(this._listId);
161 	div.innerHTML = html.join("");
162 
163 	for (var i = 0; i < size; i++) {
164 		var appt = list.get(i);
165 		var uid = appt.getUniqueId(true);
166         var id = appt.id;
167 		var data = this._apptData[uid];
168 
169         if (!data) { continue; }
170 
171         var alarmData = appt.getAlarmData();
172         alarmData = (alarmData && alarmData.length > 0) ? alarmData[0] : {};
173         //bug: 60692 - Add troubleshooting code for late reminders
174         AjxDebug.println(AjxDebug.REMINDER, appt.getReminderName() + " : " + (alarmData.nextAlarm || " NA ") + " / " + (alarmData.alarmInstStart || " NA "));
175 
176 		// dismiss button
177 		var dismissBtn = this._dismissButtons[data.dismissBtnId] = new DwtButton({id:"dismissBtn_" + id, parent:this, className:"DwtToolbarButton", parentElement:data.dismissBtnId});
178 		dismissBtn.setText(ZmMsg.dismiss);
179 		dismissBtn.addSelectionListener(dismissListener);
180 		dismissBtn.apptUid = uid;
181 
182 		// open button
183 		var openBtn = this._openButtons[data.openBtnId] = new DwtButton({id:"openBtn_" + id, parent:this, className:"DwtToolbarButton", parentElement:data.openBtnId});
184 		openBtn.setText(ZmMsg.viewAppointment);
185 		openBtn.addSelectionListener(openListener);
186 		openBtn.apptUid = uid;
187 
188 		this._updateDelta(data);
189 	}
190 
191     var snoozeSelectMenuListener = this._snoozeSelectMenuListener.bind(this);
192     this._createSnoozeMenu(this._snoozeSelectButton, snoozeSelectMenuListener, this._list);
193 };

233 ZmReminderDialog.DEFAULT_SNOOZE = -5;
234 ZmReminderDialog.SNOOZE_MINUTES =
235 // Snooze period in minutes (negative is 'minutes before appt', zero is separator:
236     [-30, -15, -5, -1, 0,
237        1, 5, 10, 15, 30, 45, 60, 120, 240, 480,  1440, 2880,  4320,   5760, 10080, 20160];
238 //                          1hr  2hr  4hr  8hr  1day  2days  3days  4days  1week  2weeks
239 
240 // Snooze period in msec (Entries must match SNOOZE_MINUTES)
241 ZmReminderDialog.SNOOZE_MSEC =
242     [  -30*60*1000,   -15*60*1000,  -5*60*1000,    -1*60*1000,            0,
243          1*60*1000,     5*60*1000,  10*60*1000,    15*60*1000,   30*60*1000,    45*60*1000,   60*60*1000,
244        120*60*1000,   240*60*1000, 480*60*1000,  1440*60*1000, 2880*60*1000,  4320*60*1000, 5760*60*1000,
245      10080*60*1000, 20160*60*1000];
246 
247 // Minutes per:                   minute hour  day   week   endMarker
248 ZmReminderDialog.SCALE_MINUTES = [   1,   60, 1440, 10080,   1000000];

250 ZmReminderDialog.prototype._createSnoozeMenu =
251 function(snoozeSelectButton, menuSelectionListener, apptList) {
252     // create menu for button
253     var snoozeMenu = new DwtMenu({parent:snoozeSelectButton, style:DwtMenu.DROPDOWN_STYLE});
254     snoozeMenu.setSize("150");
255     snoozeSelectButton.setMenu(snoozeMenu, true);
256 
257     var appts = apptList.getArray();
258     var minStartDelta =  1;
259 
260     if (this._apptType == "task") {
261         // Tasks are simpler: No 'before' times allowed, and all fixed times are allowed
262         minStartDelta = 1;
263     } else {
264         for (var i = 0; i < appts.length; i++) {
265             var appt = appts[i];
266             var startDelta = this._computeDelta(appt);
267             // Only limit the snooze menu with appts that have not already started/completed
268             if (startDelta < 0) {
269                 // Limit the list of 'before' times by the appt that will appear the soonest in the future
270                 if ((startDelta > minStartDelta) || (minStartDelta >= 0)) {
271                     minStartDelta = startDelta;
272                 }
273                 // Get the largest snooze period that can be used by any appt (without snoozing past its start)
274                 startDelta = -startDelta;
275             }
276         }
277     }
278 
279     var snoozeFormatter = [];
280     var snoozeFormatterBefore = new AjxMessageFormat(ZmMsg.apptRemindNMinutesBefore); // Before Appt Formatter
281     snoozeFormatter[0] = new AjxMessageFormat(ZmMsg.reminderSnoozeMinutes);       // Minute Formatter
282     snoozeFormatter[1] = new AjxMessageFormat(ZmMsg.reminderSnoozeHours);         // Hour   Formatter
283     snoozeFormatter[2] = new AjxMessageFormat(ZmMsg.reminderSnoozeDays);          // Day    Formatter
284     snoozeFormatter[3] = new AjxMessageFormat(ZmMsg.reminderSnoozeWeeks);         // Week   Formatter
285     var iFormatter = 0;
286     var formatter = null;
287     var snoozeDisplayValue = 0;
288     var scale = 1;
289     var defaultSet = false;
290     var firstMenuItem = null;
291     var addSeparator = false;
292     var anyAdded = false;
293     for (var i = 0; i < ZmReminderDialog.SNOOZE_MINUTES.length; i++) {
294         if (ZmReminderDialog.SNOOZE_MSEC[i] >= minStartDelta) {
295             // Found a snooze period to display
296             snoozeDisplayValue = ZmReminderDialog.SNOOZE_MINUTES[i];
297             if (snoozeDisplayValue == 0) {
298                 // Set up to add a separator if any 'before' time were added; do the
299                 // actual add if any fixed times are added
300                 addSeparator = anyAdded;
301             }
302             else {
303                 if (addSeparator) {
304                     new DwtMenuItem({parent:snoozeMenu, style:DwtMenuItem.SEPARATOR_STYLE});
305                     addSeparator = false;
306                 }
307                 anyAdded = true;
308                 if (snoozeDisplayValue <= 0) {
309                     snoozeDisplayValue = -snoozeDisplayValue;
310                     formatter = snoozeFormatterBefore;
311                     scale = 1;
312                 }else {
313                     if (snoozeDisplayValue >= ZmReminderDialog.SCALE_MINUTES[iFormatter+1]) {
314                         iFormatter++;
315                     }
316                     scale = ZmReminderDialog.SCALE_MINUTES[iFormatter];
317                     formatter = snoozeFormatter[iFormatter];
318                 }
319                 var label = formatter.format(snoozeDisplayValue/scale);
320                 var mi = new DwtMenuItem({parent: snoozeMenu, style: DwtMenuItem.NO_STYLE});
321                 mi.setText(label);
322                 mi.setData("value", snoozeDisplayValue);
323                 if(menuSelectionListener) mi.addSelectionListener(menuSelectionListener);
324 
325                 if (!firstMenuItem) {
326                     // Set the first item as the default
327                     firstMenuItem = mi;
328                     mi.setChecked(true);
329                     this._snoozeSelectInput.setValue(label);
330                     defaultSet = true;
331                 }
332             }
333         }
334     }
335     if (firstMenuItem == null) {
336         // No valid selections
337         this._snoozeSelectInput.setValue("");
338         this._snoozeSelectInput.setEnabled(false, true);
339         snoozeMenu.setEnabled(false, true);
340         snoozeSelectButton.setEnabled(false);
341         this.setButtonEnabled(ZmReminderDialog.SNOOZE_BUTTON, false);
342         this.unregisterCallback(ZmReminderDialog.SNOOZE_BUTTON);
343     } else {
344         this._snoozeSelectInput.setEnabled(true, true);
345         snoozeMenu.setEnabled(true, true);
346         snoozeSelectButton.setEnabled(true, true);
347         this.setButtonEnabled(ZmReminderDialog.SNOOZE_BUTTON, true);
348         this.registerCallback(ZmReminderDialog.SNOOZE_BUTTON, this._handleSnoozeButton, this);
349     }
350 
351 };

367 ZmReminderDialog.prototype._updateDelta =
368 function(data) {
369 	var td = document.getElementById(data.deltaId);
370 	if (td) {
371 		var startDelta = this._computeDelta(data.appt);
372 
373 		if (startDelta >= 0) 							td.className = 'ZmReminderOverdue';
374 		else if (startDelta > ZmReminderDialog.SOON)	td.className = 'ZmReminderSoon';
375 		else											td.className = 'ZmReminderFuture';
376 
377 		td.innerHTML = startDelta ? this._formatDeltaString(startDelta) : "";
378 	}

419 ZmReminderDialog.prototype._openButtonListener =
420 function(ev) {
421 
422     appCtxt.getAppController().setStatusMsg(ZmMsg.allRemindersAreSnoozed, ZmStatusView.LEVEL_INFO);
423 
424     this._handleSnoozeButton();
425 
426 	var obj = DwtControl.getTargetControl(ev);
427 	var data = this._apptData[obj.apptUid];
428 	var appt = data ? data.appt : null;
429     var type = appt.type ? appt.type : ZmItem.APPT;
430 	if (appt && type == ZmItem.APPT) {
431 		AjxDispatcher.require(["CalendarCore", "Calendar"]);
432 
433 		var cc = AjxDispatcher.run("GetCalController");
434 
435 		// the give appt object is a ZmCalBaseItem. We need a ZmAppt
436 		var newAppt = new ZmAppt();
437 		for (var i in appt) {
438 			if (!AjxUtil.isFunction(appt[i])) {
439 				newAppt[i] = appt[i];
440 			}
441 		}
442         var mode = newAppt.isRecurring() ? ZmCalItem.MODE_EDIT_SINGLE_INSTANCE : null;
443 		var callback = new AjxCallback(cc, cc._showAppointmentDetails, newAppt);
444 		newAppt.getDetails(mode, callback, null, null, true);
445 	} else if(appt && type == ZmItem.TASK) {
446         AjxDispatcher.require(["TasksCore", "Tasks"]);
447 
448 		var tlc = AjxDispatcher.run("GetTaskListController");
449 
450 		// the give appt object is a ZmCalBaseItem. We need a ZmAppt
451 		var newTask = new ZmTask();
452 		for (var i in appt) {
453 			if (!AjxUtil.isFunction(appt[i])) {
454 				newTask[i] = appt[i];
455 			}
456 		}
457 		var callback = new AjxCallback(tlc, tlc._editTask, newTask);
458 		newTask.getDetails(null, callback, null, null, true);
459     }

462 ZmReminderDialog.prototype._dismissButtonListener =
463 function(ev) {
464 	var obj = DwtControl.getTargetControl(ev);
465 	var data = this._apptData[obj.apptUid];
466 	if (!data) { return; }
467 
468 	this._reminderController.dismissAppt(data.appt);
469 
470 	// cleanup HTML
471 	var dismissBtn = this._dismissButtons[data.dismissBtnId];
472 	if (dismissBtn) {
473 		dismissBtn.dispose();
474 		delete this._dismissButtons[data.dismissBtnId];
475 	}
476 	var openBtn = this._openButtons[data.openBtnId];
477 	if (openBtn) {
478 		openBtn.dispose();
479 		delete this._openButtons[data.openBtnId];
480 	}
481 	var row = document.getElementById(data.rowId);
482 	if (row) {
483 		var nextRow = row.nextSibling;
484 		if (nextRow && nextRow.getAttribute("name") == "rdsep") {
485 			nextRow.parentNode.removeChild(nextRow);
486 		}
487 		row.parentNode.removeChild(row);
488 	}
489 
490 	delete this._apptData[obj.apptUid];
491 	this._list.remove(data.appt);
492 
493 	if (this._list.size() == 0) {
494 		this.popdown();
495 	}
496 };

499 ZmReminderDialog.prototype._handleSnoozeButton =
500 function() {
501     // check if all fields are populated w/ valid values
502     var errorMsg = [];
503     var snoozeInfo = null;
504     var beforeAppt = false;
505     var snoozeString = this._snoozeSelectInput.getValue();
506     if (!snoozeString) {
507          errorMsg.push(ZmMsg.reminderSnoozeClickNoDuration);
508     } else {
509         snoozeInfo = ZmCalendarApp.parseReminderString(snoozeString);
510         if (snoozeInfo.reminderValue === "" ) {
511             // Returned when no number was specified in the snooze input field
512             errorMsg.push(ZmMsg.reminderSnoozeClickNoNumber);
513         }  else {
514             // Test if the unit is a known one (default behaviour for parseReminderString
515             // returns unknowns as hours)
516             var valid = this._testSnoozeString(snoozeString);
517             if (!valid) {
518                  errorMsg.push(ZmMsg.reminderSnoozeClickUnknownUnit);
519             } else {
520                 // Detect 'before'
521                 //Fix for Bug: 80651 - Check for snooze before object
522                 beforeAppt = snoozeInfo.before;
523             }
524         }
525     }
526     if (errorMsg.length > 0) {
527         var msg = errorMsg.join("<br>");
528         var dialog = appCtxt.getMsgDialog();
529         dialog.reset();
530         dialog.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
531         dialog.popup();
532     }  else {
533         this.popdown();
534         var snoozeMinutes = ZmCalendarApp.convertReminderUnits(snoozeInfo.reminderValue,
535                                                                snoozeInfo.reminderUnits);
536 	    this._reminderController.snoozeAppt(this._list);
537         this._reminderController._snoozeApptAction(this._list, snoozeMinutes, beforeAppt)
538     }
539 
540 };

598 ZmReminderDialog.prototype._getDurationText =
599 function(appt) {
600 	var isMultiDay = appt.isMultiDay();
601 	var start = appt._alarmInstStart ? new Date(appt._alarmInstStart) : appt.startDate ? appt.startDate : null;
602 	// bug: 28598 - alarm for recurring appt might still point to old alarm time
603 	// cannot take endTime directly
604 	var endTime = appt._alarmInstStart ? (start.getTime() + appt.getDuration()) : appt.getEndTime();
605 	var end = new Date(endTime);
606 
607     //for task
608     if(appt.type == ZmItem.TASK && !start && !endTime) { return null; }
609 
610 	if (appt.isAllDayEvent()) {
611 		end = appt.type != ZmItem.TASK ? new Date(endTime - (isMultiDay ? 2 * AjxDateUtil.MSEC_PER_HOUR : 0)) : end;
612 		var pattern = isMultiDay ? ZmMsg.apptTimeAllDayMulti : ZmMsg.apptTimeAllDay;
613 		return start ? AjxMessageFormat.format(pattern, [start, end]) : AjxMessageFormat.format(pattern, [end]); //for task
614 	}
615 	var pattern = isMultiDay ? ZmMsg.apptTimeInstanceMulti : ZmMsg.apptTimeInstance;
616 	return AjxMessageFormat.format(pattern, [start, end, ""]);
617 };

626 ZmReminderDialog.prototype._formatDeltaString =
627 function(deltaMSec) {
628 	var prefix = deltaMSec < 0 ? "In" : "OverdueBy";
629 	deltaMSec = Math.abs(deltaMSec);
630 
631 	// calculate parts
632 	var years =  Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 365));
633 	if (years != 0)
634 		deltaMSec -= years * AjxDateUtil.MSEC_PER_DAY * 365;
635 	var months = Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 30.42));
636 	if (months > 0)
637 		deltaMSec -= Math.floor(months * AjxDateUtil.MSEC_PER_DAY * 30.42);
638 	var days = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_DAY);
639 	if (days > 0)
640 		deltaMSec -= days * AjxDateUtil.MSEC_PER_DAY;
641 	var hours = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_HOUR);
642 	if (hours > 0)
643 		deltaMSec -= hours * AjxDateUtil.MSEC_PER_HOUR;
644 	var mins = Math.floor(deltaMSec / 60000);
645 	if (mins > 0)
646 		deltaMSec -= mins * 60000;
647 	var secs = Math.floor(deltaMSec / 1000);
648 	if (secs > 30 && mins < 59) mins++;
649 
650 	var secs = 0;
651 
652 	// determine message
653 	var amount;
654 	if (years > 0) {
655 		amount = "Years";
656 		if (years <= 3 && months > 0) {
657 			amount = "YearsMonths";
658 		}
659 	} else if (months > 0) {
660 		amount = "Months";
661 		if (months <= 3 && days > 0) {
662 			amount = "MonthsDays";
663 		}
664 	} else if (days > 0) {
665 		amount = "Days";
666 		if (days <= 2 && hours > 0) {
667 			amount = "DaysHours";
668 		}
669 	} else if (hours > 0) {
670 		amount = "Hours";
671 		if (hours < 5 && mins > 0) {
672 			amount = "HoursMinutes";
673 		}
674 	} else {
675 		amount = "Minutes";
676 	}
677 
678 	// format message
679 	var key = ["reminder",prefix,amount].join("");
680 	var args = [deltaMSec, years, months, days, hours, mins, secs];
681 	return AjxMessageFormat.format(ZmMsg[key], args);
682 };
