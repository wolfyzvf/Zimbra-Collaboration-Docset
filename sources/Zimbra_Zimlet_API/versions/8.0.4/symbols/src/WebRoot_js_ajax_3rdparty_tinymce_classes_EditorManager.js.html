118 		/**
119 		 * Initializes a set of editors. This method will create a bunch of editors based in the input.
120 		 *
121 		 * @method init
122 		 * @param {Object} s Settings object to be passed to each editor instance.
123 		 * @example
124 		 * // Initializes a editor using the longer method
125 		 * tinymce.EditorManager.init({
126 		 *    some_settings : 'some value'
127 		 * });
128 		 * 
129 		 * // Initializes a editor instance using the shorter version
130 		 * tinyMCE.init({
131 		 *    some_settings : 'some value'
132 		 * });
133 		 */
134 		init : function(s) {
135 			var t = this, pl, sl = tinymce.ScriptLoader, e, el = [], ed;
136 
137 			function createId(elm) {
138 				var id = elm.id;
139 	
140 				// Use element id, or unique name or generate a unique id
141 				if (!id) {
142 					id = elm.name;
143 	
144 					if (id && !DOM.get(id)) {
145 						id = elm.name;
146 					} else {
147 						// Generate unique name
148 						id = DOM.uniqueId();
149 					}
150 
151 					elm.setAttribute('id', id);
152 				}
153 
154 				return id;
155 			};
156 
157 			function execCallback(se, n, s) {
158 				var f = se[n];
159 
160 				if (!f)
161 					return;
162 
163 				if (tinymce.is(f, 'string')) {
164 					s = f.replace(/\.\w+$/, '');
165 					s = s ? tinymce.resolve(s) : 0;
166 					f = tinymce.resolve(f);
167 				}
168 
169 				return f.apply(s || this, Array.prototype.slice.call(arguments, 2));
170 			};
171 
172 			function hasClass(n, c) {
173 				return c.constructor === RegExp ? c.test(n.className) : DOM.hasClass(n, c);
174 			};
175 
176 			t.settings = s;
177 
178 			// Legacy call
179 			Event.bind(window, 'ready', function() {
180 				var l, co;
181 
182 				execCallback(s, 'onpageload');
183 
184 				switch (s.mode) {
185 					case "exact":
186 						l = s.elements || '';
187 
188 						if(l.length > 0) {
189 							each(explode(l), function(v) {
190 								if (DOM.get(v)) {
191 									ed = new tinymce.Editor(v, s);
192 									el.push(ed);
193 									ed.render(1);
194 								} else {
195 									each(document.forms, function(f) {
196 										each(f.elements, function(e) {
197 											if (e.name === v) {
198 												v = 'mce_editor_' + instanceCounter++;
199 												DOM.setAttrib(e, 'id', v);
200 
201 												ed = new tinymce.Editor(v, s);
202 												el.push(ed);
203 												ed.render(1);
204 											}
205 										});
206 									});
207 								}
208 							});
209 						}
210 						break;
211 
212 					case "textareas":
213 					case "specific_textareas":
214 						each(DOM.select('textarea'), function(elm) {
215 							if (s.editor_deselector && hasClass(elm, s.editor_deselector))
216 								return;
217 
218 							if (!s.editor_selector || hasClass(elm, s.editor_selector)) {
219 								ed = new tinymce.Editor(createId(elm), s);
220 								el.push(ed);
221 								ed.render(1);
222 							}
223 						});
224 						break;
225 					
226 					default:
227 						if (s.types) {
228 							// Process type specific selector
229 							each(s.types, function(type) {
230 								each(DOM.select(type.selector), function(elm) {
231 									var editor = new tinymce.Editor(createId(elm), tinymce.extend({}, s, type));
232 									el.push(editor);
233 									editor.render(1);
234 								});
235 							});
236 						} else if (s.selector) {
237 							// Process global selector
238 							each(DOM.select(s.selector), function(elm) {
239 								var editor = new tinymce.Editor(createId(elm), s);
240 								el.push(editor);
241 								editor.render(1);
242 							});
243 						}
244 				}
245 
246 				// Call onInit when all editors are initialized
247 				if (s.oninit) {
248 					l = co = 0;
249 
250 					each(el, function(ed) {
251 						co++;
252 
253 						if (!ed.initialized) {
254 							// Wait for it
255 							ed.onInit.add(function() {
256 								l++;
257 
258 								// All done
259 								if (l == co)
260 									execCallback(s, 'oninit');
261 							});
262 						} else
263 							l++;
264 
265 						// All done
266 						if (l == co)
267 							execCallback(s, 'oninit');					
268 					});
269 				}
270 			});
271 		},

310 		/**
311 		 * Adds an editor instance to the editor collection. This will also set it as the active editor.
312 		 *
313 		 * @method add
314 		 * @param {tinymce.Editor} editor Editor instance to add to the collection.
315 		 * @return {tinymce.Editor} The same instance that got passed in.
316 		 */
317 		add : function(editor) {
318 			var self = this, editors = self.editors;
319 
320 			// Add named and index editor instance
321 			editors[editor.id] = editor;
322 			editors.push(editor);
323 
324 			self._setActive(editor);
325 			self.onAddEditor.dispatch(self, editor);
326 
327 			// #ifdef jquery
328 
329 			// Patch the tinymce.Editor instance with jQuery adapter logic
330 			if (tinymce.adapter)
331 				tinymce.adapter.patchEditor(editor);
332 
333 			// #endif
334 
335 			return editor;
336 		},

338 		/**
339 		 * Removes a editor instance from the collection.
340 		 *
341 		 * @method remove
342 		 * @param {tinymce.Editor} e Editor instance to remove.
343 		 * @return {tinymce.Editor} The editor that got passed in will be return if it was found otherwise null.
344 		 */
345 		remove : function(editor) {
346 			var t = this, i, editors = t.editors;
347 
348 			// Not in the collection
349 			if (!editors[editor.id])
350 				return null;
351 
352 			delete editors[editor.id];
353 
354 			for (i = 0; i < editors.length; i++) {
355 				if (editors[i] == editor) {
356 					editors.splice(i, 1);
357 					break;
358 				}
359 			}
360 
361 			// Select another editor since the active one was removed
362 			if (t.activeEditor == editor)
363 				t._setActive(editors[0]);
364 
365 			editor.destroy();
366 			t.onRemoveEditor.dispatch(t, editor);
367 
368 			return editor;
369 		},

371 		/**
372 		 * Executes a specific command on the currently active editor.
373 		 *
374 		 * @method execCommand
375 		 * @param {String} c Command to perform for example Bold.
376 		 * @param {Boolean} u Optional boolean state if a UI should be presented for the command or not.
377 		 * @param {String} v Optional value parameter like for example an URL to a link.
378 		 * @return {Boolean} true/false if the command was executed or not.
379 		 */
380 		execCommand : function(c, u, v) {
381 			var t = this, ed = t.get(v), w;
382 
383 			function clr() {
384 				ed.destroy();
385 				w.detachEvent('onunload', clr);
386 				w = w.tinyMCE = w.tinymce = null; // IE leak
387 			};
388 
389 			// Manager commands
390 			switch (c) {
391 				case "mceFocus":
392 					ed.focus();
393 					return true;
394 
395 				case "mceAddEditor":
396 				case "mceAddControl":
397 					if (!t.get(v))
398 						new tinymce.Editor(v, t.settings).render();
399 
400 					return true;
401 
402 				case "mceAddFrameControl":
403 					w = v.window;
404 
405 					// Add tinyMCE global instance and tinymce namespace to specified window
406 					w.tinyMCE = tinyMCE;
407 					w.tinymce = tinymce;
408 
409 					tinymce.DOM.doc = w.document;
410 					tinymce.DOM.win = w;
411 
412 					ed = new tinymce.Editor(v.element_id, v);
413 					ed.render();
414 
415 					// Fix IE memory leaks
416 					if (tinymce.isIE) {
417 						w.attachEvent('onunload', clr);
418 					}
419 
420 					v.page_window = null;
421 
422 					return true;
423 
424 				case "mceRemoveEditor":
425 				case "mceRemoveControl":
426 					if (ed)
427 						ed.remove();
428 
429 					return true;
430 
431 				case 'mceToggleEditor':
432 					if (!ed) {
433 						t.execCommand('mceAddControl', 0, v);
434 						return true;
435 					}
436 
437 					if (ed.isHidden())
438 						ed.show();
439 					else
440 						ed.hide();
441 
442 					return true;
443 			}
444 
445 			// Run command on active editor
446 			if (t.activeEditor)
447 				return t.activeEditor.execCommand(c, u, v);
448 
449 			return false;
450 		},

453 		 * Executes a command on a specific editor by id. This method was added for compatibility with the 2.x branch.
454 		 *
455 		 * @deprecated Use the execCommand method of a editor instance instead.

457 		 * @param {String} id Editor id to perform the command on.

465

468 
469 			return false;
470 		},

486 		/**
487 		 * Adds a language pack, this gets called by the loaded language files like en.js.
488 		 *
489 		 * @method addI18n
490 		 * @param {String} p Prefix for the language items. For example en.myplugin
491 		 * @param {Object} o Name/Value collection with items to add to the language group.
492 		 */
493 		addI18n : function(p, o) {
494 			var lo, i18n = this.i18n;
495 
496 			if (!tinymce.is(p, 'string')) {
497 				each(p, function(o, lc) {
498 					each(o, function(o, g) {
499 						each(o, function(o, k) {
500 							if (g === 'common')
501 								i18n[lc + '.' + k] = o;
502 							else
503 								i18n[lc + '.' + g + '.' + k] = o;
504 						});
505 					});
506 				});
507 			} else {
508 				each(o, function(o, k) {
509 					i18n[p + '.' + k] = o;
510 				});
511 			}
512 		},
