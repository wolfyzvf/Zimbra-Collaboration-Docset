 22 /**
 23  * Creates a controller to run ZimbraMail. Do not call directly, instead use the run()
 24  * factory method.
 25  * @constructor
 26  * @class
 27  * This class is the "ubercontroller", as it manages all the apps as well as bootstrapping
 28  * the ZimbraMail application.
 29  *
 30  * @param {Hash}	params	a hash of parameters
 31  * @param {constant}    params.app		the starting app
 32  * @param  {Element}	params.userShell	the top-level skin container
 33  *        
 34  * @extends	ZmController
 35  */
 36 ZmZimbraMail = function(params) {
 37 
 38     if (arguments.length == 0) { return; }
 39 
 40 	ZmController.call(this, null);
 41 
 42     appCtxt.setAppController(this);
 43 
 44 	// ALWAYS set back reference into our world (also used by unload handler)
 45 	window._zimbraMail = this;
 46 
 47     // app event handling
 48     this._evt = new ZmAppEvent();
 49     this._evtMgr = new AjxEventMgr();
 50     // copy over any statically registered listeners
 51     for (var type in ZmZimbraMail._listeners) {
 52         var list = ZmZimbraMail._listeners[type];
 53         if (list && list.length) {
 54             for (var i = 0; i < list.length; i++) {
 55                 this._evtMgr.addListener(type, list[i]);
 56             }
 57         }
 58     }
 59 
 60     // all subsequent calls to register static app listeners go to instance
 61     ZmZimbraMail.addListener = AjxCallback.simpleClosure(this.addListener, this);
 62     ZmZimbraMail.addAppListener = AjxCallback.simpleClosure(this.addAppListener, this);
 63 
 64     // Create generic operations
 65     ZmOperation.initialize();
 66 
 67     // settings
 68     this._createSettings(params);
 69     this._createEnabledApps();
 70     this._initializeSettings(params);
 71 	this._postInitializeSettings();
 72 
 73 	//update body class to reflect user selected font
 74 	document.body.className = "user_font_" + appCtxt.get(ZmSetting.FONT_NAME);
 75 
 76     // set internal state
 77 	this._shell = appCtxt.getShell();
 78     this._userShell = params.userShell;
 79 
 80     this._requestMgr = new ZmRequestMgr(this); // NOTE: requires settings to be initialized
 81 
 82 	this._upsellView = {};
 83 	this._activeApp = null;
 84 	this._sessionTimer = new AjxTimedAction(null, ZmZimbraMail.executeSessionTimer);
 85 	this._sessionTimerId = -1;
 86 	this._pollActionId = null;	// AjaxTimedAction ID of timer counting down to next poll time
 87 	this._pollRequest = null;	// HTTP request of poll we've sent to server
 88 	this._pollInstantNotifications = false; // if TRUE, we're in "instant notification" mode
 89 	this.statusView = null;
 90 	ZmZimbraMail._exitTimer = new AjxTimedAction(null, ZmZimbraMail.exitSession);
 91 	ZmZimbraMail._exitTimerId = -1;
 92 	ZmZimbraMail.stayOnPagePrompt = false;
 93 	ZmZimbraMail.STAYONPAGE_INTERVAL = 2;  //in minutes
 94     // setup history support
 95     if (appCtxt.get(ZmSetting.HISTORY_SUPPORT_ENABLED) && !AjxEnv.isSafari) {
 96         window.historyMgr = appCtxt.getHistoryMgr();
 97     }
 98 
 99     // create app view manager
100     this._appViewMgr = new ZmAppViewMgr(this._shell, this, false, true);
101 	var hidden = [ ZmAppViewMgr.C_SEARCH_RESULTS_TOOLBAR, ZmAppViewMgr.C_TASKBAR ];
102 	if (!appCtxt.get(ZmSetting.CAL_ALWAYS_SHOW_MINI_CAL)) {
103 		hidden.push(ZmAppViewMgr.C_TREE_FOOTER);
104 	}
105 	this._appViewMgr.setHiddenComponents(ZmAppViewMgr.GLOBAL, hidden, true);
106 
107     // register handlers
108 	AjxDispatcher.setPackageLoadFunction("Zimlet", new AjxCallback(this, this._postLoadZimlet));
109 
110 	AjxDispatcher.setPreLoadFunction(new AjxCallback(this, function() {
111 		this._appViewMgr.pushView(ZmId.VIEW_LOADING);
112 	}));
113 	AjxDispatcher.setPostLoadFunction(new AjxCallback(this, function() {
114 		if (!AjxUtil.arrayContains(this._appViewMgr._toRemove, ZmId.VIEW_LOADING)) {
115 			this._appViewMgr._toRemove.push(ZmId.VIEW_LOADING);
116 		}
117 	}));
118 
119 	for (var i in ZmApp.QS_ARG) {
120 		ZmApp.QS_ARG_R[ZmApp.QS_ARG[i]] = i;
121 	}
122 
123 	this._shell.addGlobalSelectionListener(new AjxListener(this, this._globalSelectionListener));
124 
125     /// go!
126     this.startup(params);
127 };

164 /**
165  * Sets up ZimbraMail, and then starts it by calling its constructor. It is assumed that the
166  * CSFE is on the same host.
167  *
168  * @param {Hash}	params			a hash of parameters
169  * @param {constant}      params.app				te starting app
170  * @param {Boolean}      params.offlineMode		if <code>true</code>, this is the offline client
171  * @param {Boolean}      params.devMode			if <code>true</code>, we are in development environment
172  * @param {Hash}      params.settings			the server prefs/attrs
173  * @param {constant}      params.protocolMode	the protocal mode (http, https or mixed)
174  * @param {Boolean}      params.noSplashScreen	if <code>true</code>, do not show splash screen during startup
175  */
176 ZmZimbraMail.run =
177 function(params) {
178 
179 	if (params.noSplashScreen) {
180 		ZmZimbraMail.killSplash();
181 	}
182 
183 	// Create the global app context
184 	window.appCtxt = new ZmAppCtxt();
185 	appCtxt.rememberMe = false;
186 
187 	// Handle offline mode
188 	if (params.offlineMode) {
189 		DBG.println(AjxDebug.DBG1, "OFFLINE MODE");
190 		appCtxt.isOffline = true;
191 	}
192 
193 	// Create the shell
194 	var userShell = params.userShell = window.document.getElementById(ZmId.SKIN_SHELL);
195 	if (!userShell) {
196 		alert("Could not get user shell - skin file did not load properly");
197 	}
198 	var shell = new DwtShell({userShell:userShell, docBodyScrollable:false, id:ZmId.SHELL});
199 	appCtxt.setShell(shell);
200 
201 	// Go!
202 	new ZmZimbraMail(params);
203 };

250 /**
251  * Returns sort order using a and b as keys into given hash.
252  *
253  * @param {Hash}	hash		a hash with sort values
254  * @param {String}	a			a key into hash
255  * @param {String}	b			a key into hash
256  * @return	{int}	0 if the items are the same; 1 if "a" is before "b"; -1 if "b" is before "a"
257  */
258 ZmZimbraMail.hashSortCompare =
259 function(hash, a, b) {
260 	var appA = a ? Number(hash[a]) : 0;
261 	var appB = b ? Number(hash[b]) : 0;
262 	if (appA > appB) { return 1; }
263 	if (appA < appB) { return -1; }
264 	return 0;
265 };

387 ZmZimbraMail.prototype._getDelegatedSenderEmails =
388 function(sendRights, sendRight) {
389     var emails = [];
390         if (sendRights && sendRights.length){
391             for (var i=0;i<sendRights.length; i++){
392                 var target = sendRights[i].target;
393                 var right =  sendRights[i].right;
394                 if (right == sendRight || right == (sendRight + "DistList")){
395                     for (var j=0;j < target.length; j++){
396                         var emailList = target[j].email;
397                         for (var k=0; k < emailList.length; k++){
398                             var email = emailList[k].addr;
399                             emails.push(email);
400                         }
401                     }
402                 }
403 
404             }
405         }
406     return emails;
407 };

427 ZmZimbraMail.prototype._createSettings = function(params) {
428     // We've received canned SOAP responses for GetInfoRequest and SearchRequest from the
429     // launch JSP, wrapped in a BatchRequest. Jiggle them so that they look like real
430     // responses, and pass them along.
431     if (params.batchInfoResponse) {
432         var batchResponse = params.batchInfoResponse.Body.BatchResponse;
433 
434         // always assume there's a get info response
435 		var infoResponse = batchResponse.GetInfoResponse[0];
436 		if(!infoResponse) {
437 			infoResponse ={}
438 		}
439 		//store per-domain settings in infoResponse obj so we can access it like other settings
440 		infoResponse.domainSettings = params.settings;
441         params.getInfoResponse = {
442             Header: params.batchInfoResponse.Header,
443             Body: { GetInfoResponse: infoResponse}
444         };
445         var session = AjxUtil.get(params.getInfoResponse, "Header", "context", "session");
446         if (session) {
447             ZmCsfeCommand.setSessionId(session);
448         }
449         DBG.println(AjxDebug.DBG1, ["<b>RESPONSE (from JSP tag)</b>"].join(""), "GetInfoResponse");
450         DBG.dumpObj(AjxDebug.DBG1, params.getInfoResponse, -1);
451 
452         // we may have an initial search response
453         if (batchResponse.SearchResponse) {
454             params.searchResponse = {
455                 Body: { SearchResponse: batchResponse.SearchResponse[0] }
456             };
457             DBG.println(AjxDebug.DBG1, ["<b>RESPONSE (from JSP tag)</b>"].join(""), "SearchResponse");
458             DBG.dumpObj(AjxDebug.DBG1, params.searchResponse, -1);
459         }
460     }
461 
462     // create settings
463     var settings = new ZmSettings()
464     appCtxt.setSettings(settings);
465 
466     // We have to pre-initialize the settings in order to create
467     // the enabled apps correctly.
468     settings.setUserSettings({info:params.getInfoResponse.Body.GetInfoResponse, preInit:true});
469 };

471 ZmZimbraMail.prototype._initializeSettings = function(params) {
472     var info = params.getInfoResponse.Body.GetInfoResponse;
473 
474     var settings = appCtxt.getSettings();
475     // NOTE: Skip notify to avoid callbacks which reference objects that aren't set, yet
476     settings.setUserSettings(info, null, true, true, true);
477     settings.userSettingsLoaded = true;
478 
479     // settings structure and defaults
480     var branch = appCtxt.get(ZmSetting.BRANCH);
481     if (window.DBG && !DBG.isDisabled()) {
482         DBG.setTitle("Debug (" + branch + ")");
483     }
484 
485     // setting overrides
486     if (params.settings) {
487         for (var name in params.settings) {
488             var id = settings.getSettingByName(name);
489             if (id) {
490                 settings.getSetting(id).setValue(params.settings[name]);
491             }
492         }
493     }
494 
495     // reset polling interval for offline
496     if (appCtxt.isOffline) {
497         appCtxt.set(ZmSetting.POLLING_INTERVAL, 60, null, null, true);
498     }
499 
500     // Handle dev mode
501     if (params.devMode == "1") {
502         DBG.println(AjxDebug.DBG1, "DEV MODE");
503         appCtxt.set(ZmSetting.DEV, true);
504     }
505 
506     // Handle protocol mode - standardize on trailing :
507     if (params.protocolMode) {
508         var proto = (params.protocolMode.indexOf(":") == -1) ? params.protocolMode + ":" : params.protocolMode;
509         appCtxt.set(ZmSetting.PROTOCOL_MODE, proto);
510     }
511     if (params.httpPort) {
512         appCtxt.set(ZmSetting.HTTP_PORT, params.httpPort);
513     }
514     if (params.httpsPort) {
515         appCtxt.set(ZmSetting.HTTPS_PORT, params.httpsPort);
516     }
517 
518     // hide spam if not enabled
519     if (!appCtxt.get(ZmSetting.SPAM_ENABLED)) {
520         ZmFolder.HIDE_ID[ZmFolder.ID_SPAM] = true;
521     }
522 };

1610 /**
1611  * Make sure the polling loop is running.  Basic flow:
1612  *
1613  *       1) kickPolling():
1614  *             - cancel any existing timers
1615  *             - set a timer for _pollInterval time
1616  *             - call execPoll() when the timer goes off
1617  *
1618  *       2) execPoll():
1619  *             - make the NoOp request, if we're in "instant notifications"
1620  *               mode, this request will hang on the server until there is more data,
1621  *               otherwise it will return immediately.  Call into a handle() func below
1622  *
1623  *       3) handleDoPollXXXX():
1624  *             - call back to kickPolling() above
1625  *
1626  * resetBackoff = TRUE e.g. if we've just received a successful
1627  * response from the server, or if the user just changed our
1628  * polling settings and we want to start in fast mode
1629  * 
1630  * @private
1631  */
1632 ZmZimbraMail.prototype._kickPolling =
1633 function(resetBackoff) {
1634 	DBG.println(AjxDebug.DBG2, [
1635 		"ZmZimbraMail._kickPolling ",
1636 		this._pollInterval, ", ",
1637 		this._pollActionId, ", ",
1638 		this._pollRequest ? "request_pending" : "no_request_pending"
1639 	].join(""));
1640 
1641 	// reset the polling timeout
1642 	if (this._pollActionId) {
1643 		AjxTimedAction.cancelAction(this._pollActionId);
1644 		this._pollActionId = null;
1645 	}
1646 
1647 	if (resetBackoff && this._pollInstantNotifications) {
1648 		// we *were* backed off -- reset the delay back to 1s fastness
1649 		var interval = appCtxt.get(ZmSetting.INSTANT_NOTIFY_INTERVAL);
1650 		if (this._pollInterval > interval) {
1651 			this._pollInterval = interval;
1652 		}
1653 	}
1654 
1655 	if (this._pollInterval && !this._pollRequest) {
1656 		try {
1657 			this._pollActionId = AjxTimedAction.scheduleAction(new AjxTimedAction(this, this._execPoll), this._pollInterval);
1658 		} catch (ex) {
1659 			this._pollActionId = null;
1660 			DBG.println(AjxDebug.DBG1, "Caught exception in ZmZimbraMail._kickPolling.  Polling chain broken!");
1661 		}
1662 	}
1663 };

1665 /**
1666  * We've finished waiting, do the actual poll itself
1667  *
1668  * @private
1669  */
1670 ZmZimbraMail.prototype._execPoll =
1671 function() {
1672 	this._cancelInstantNotify();
1673 
1674 	// It'd be more efficient to make these instance variables, but for some
1675 	// reason that breaks polling in IE.
1676 	var soapDoc = AjxSoapDoc.create("NoOpRequest", "urn:zimbraMail");
1677 	try {
1678         if (this._pollInstantNotifications) {
1679             var method = soapDoc.getMethod();
1680 			var sessionId = ZmCsfeCommand.getSessionId();
1681 			if (sessionId) {
1682             	method.setAttribute("wait", 1);
1683             	method.setAttribute("limitToOneBlocked", 1);
1684 			}
1685         }
1686 		var params = {
1687 			soapDoc: soapDoc,
1688 			asyncMode: true,
1689 			callback: new AjxCallback(this, this._handleResponseDoPoll),
1690 			errorCallback: new AjxCallback(this, this._handleErrorDoPoll),
1691 			noBusyOverlay: true,
1692 			timeout: appCtxt.get(ZmSetting.INSTANT_NOTIFY_TIMEOUT),
1693 			accountName: appCtxt.isOffline && appCtxt.accountList.mainAccount.name
1694 		};
1695 		this._pollRequest = this.sendRequest(params);
1696 
1697 		// bug #42664 - handle case where sync-status-changes fall between 2 client requests
1698 		if (appCtxt.isOffline &&
1699 			!appCtxt.accountList.isInitialSyncing() &&
1700 			appCtxt.accountList.isSyncStatus(ZmZimbraAccount.STATUS_RUNNING))
1701 		{
1702 			this.sendNoOp();
1703 		}
1704 	} catch (ex) {
1705 		this._handleErrorDoPoll(ex); // oops!
1706 	}
1707 };

1709 /**
1710  * @private
1711  */
1712 ZmZimbraMail.prototype._handleErrorDoPoll =
1713 function(ex) {
1714 	if (this._pollRequest) {
1715 		// reset the polling timeout
1716 		if (this._pollActionId) {
1717 			AjxTimedAction.cancelAction(this._pollActionId);
1718 			this._pollActionId = null;
1719 		}
1720 		this._requestMgr.cancelRequest(this._pollRequest);
1721 		this._pollRequest = null;
1722 	}
1723 
1724 	if (this._pollInstantNotifications) {
1725 		// very simple-minded exponential backoff
1726 		this._pollInterval *= 2;
1727 		if (this._pollInterval > (1000 * 60 * 2)) {
1728 			this._pollInterval = 1000 * 60 * 2;
1729 		}
1730 	}
1731 
1732 	var isAuthEx = (ex.code == ZmCsfeException.SVC_AUTH_EXPIRED ||
1733 					ex.code == ZmCsfeException.SVC_AUTH_REQUIRED ||
1734 					ex.code == ZmCsfeException.NO_AUTH_TOKEN);
1735 
1736 	// restart poll timer if we didn't get an auth exception
1737 	if (!isAuthEx) {
1738 		this._kickPolling(false);
1739 	}
1740 
1741 	return !isAuthEx;
1742 };

1758 	}
