 20 /**
 21  * Creates the account alert.
 22  * @class
 23  * This class represents an alert that highlights and flashes an account accordion item.
 24  *
 25  * @param {ZmAccount}		account		the account
 26  * 
 27  * @extends		ZmAlert
 28  */
 29 ZmAccountAlert = function(account) {
 30 	ZmAlert.call(this);
 31 	this.account = account;
 32 	this._alertApps = {};
 33 	appCtxt.accountList.addActiveAcountListener(new AjxListener(this, this._accountListener));
 34 };
 35 
 36 ZmAccountAlert.prototype = new ZmAlert;
 37 ZmAccountAlert.prototype.constructor = ZmAccountAlert;
 38 
 39 /**
 40  * Returns a string representation of the object.
 41  * 
 42  * @return		{String}		a string representation of the object
 43  */
 44 ZmAccountAlert.prototype.toString =
 45 function() {
 46 	return "ZmAccountAlert";
 47 };
 48 
 49 /**
 50  * Gets the alert by account. If the alert does not exist for the specified account, a new 
 51  * alert is created
 52  * 
 53  * @param	{ZmAccount}	account		the account
 54  * @return	{ZmAccountAlert}		the alert
 55  */
 56 ZmAccountAlert.get =
 57 function(account) {
 58 	ZmAccountAlert.INSTANCES = ZmAccountAlert.INSTANCES || {};
 59 	if (!ZmAccountAlert.INSTANCES[account.id]) {
 60 		ZmAccountAlert.INSTANCES[account.id] = new ZmAccountAlert(account);
 61 	}
 62 	return ZmAccountAlert.INSTANCES[account.id];
 63 };
 64 
 65 /**
 66  * Starts the alert.
 67  * 
 68  * @param		{ZmApp}		app		the application
 69  */
 70 ZmAccountAlert.prototype.start =
 71 function(app) {
 72 	if (this.account != appCtxt.getActiveAccount()) {
 73 		this._started = true;
 74 		if (app) {
 75 			this._alertApps[app.getName()] = app;
 76 		}
 77 	}
 78 };
 79 
 80 /**
 81  * Stops the alert.
 82  * 
 83  */
 84 ZmAccountAlert.prototype.stop =
 85 function() {
 86 	this._started = false;
 87 };
 88 
 89 ZmAccountAlert.prototype._accountListener =
 90 function(evt) {
 91 	if (evt.account == this.account) {
 92 		this.stop();
 93 		for (var appName in this._alertApps) {
 94 			this._alertApps[appName].startAlert();
 95 		}
 96 		this._alertApps = {};
 97 	}
 98 };
