 21 /**
 22  * Creates and initializes the contacts application.
 23  * @class
 24  * The contacts app manages the creation and display of contacts, which are grouped
 25  * into address books.
 26  * 
 27  * @param	{DwtControl}	container		the container
 28  * @param	{ZmController}	parentController	the parent controller
 29  * 
 30  * @author Conrad Damon
 31  * 
 32  * @extends		ZmApp
 33  */
 34 ZmContactsApp = function(container, parentController) {
 35 
 36 	ZmApp.call(this, ZmApp.CONTACTS, container, parentController);
 37 
 38 	this.contactsLoaded = {};
 39 	this._contactList = {};		// canonical list by acct ID
 40 	this._initialized = false;
 41 
 42 	// contact lookup caches
 43 	this._byEmail	= {};
 44 	this._byIM		= {};
 45 	this._byPhone	= {};
 46 
 47 	// cache fetched distribution lists
 48 	this._dlCache	= {};
 49 };

366 /**
367  * Checks for the creation of an address book or a mount point to one. Regular
368  * contact creates are handed to the canonical list.
369  * 
370  * @param {Hash}	creates	a hash of create notifications
371  * 
372  * @private
373  */
374 ZmContactsApp.prototype.createNotify =
375 function(creates, force) {
376 	if (!creates["folder"] && !creates["cn"] && !creates["link"]) { return; }
377 	if (!force && !this._noDefer && this._deferNotifications("create", creates)) { return; }
378 
379 	for (var name in creates) {
380 		var list = creates[name];
381 		if (list && list.length) {
382 			for (var i = 0; i < list.length; i++) {
383 				var create = list[i];
384 				if (appCtxt.cacheGet(create.id)) { continue; }
385 
386 				if (name == "folder") {
387 					this._handleCreateFolder(create, ZmOrganizer.ADDRBOOK);
388 				} else if (name == "link") {
389 					this._handleCreateLink(create, ZmOrganizer.ADDRBOOK);
390 				} else if (name == "cn") {
391 					var clc = AjxDispatcher.run("GetContactListController");
392 					if (clc._folderId == ZmFolder.ID_DLS) {
393 						//the simplest solution I could think of to the messy problem that the clcList in this case is GAL and thus
394 						//the contact becomes GAL (in memory) even though it's not on the server. Then it's cached and when going to the contacts it would get an exeption when clicked
395 						//if the user is viewing the DLs folder, they will see the new contact they created anyway when clicking on the "contacts" folder (or whatever other folder they created it in)
396 						continue;
397 					}
398 					var clcList = (clc && clc.getFolderId()) ? clc.getList() : new ZmContactList(null);
399 					if (appCtxt.multiAccounts && clcList.search && clcList.search.folderId != create.l) {
400 						continue;
401 					}
402 					clcList.notifyCreate(create);
403 					var context = window.parentAppCtxt || window.appCtxt;
404 					context.clearAutocompleteCache(ZmAutocomplete.AC_TYPE_CONTACT);
405 					create._handled = true;
406 				}
407 			}
408 		}
409 	}
410 };

412 ZmContactsApp.prototype.modifyNotify =
413 function(modifies, force) {
414 	if (!modifies["cn"]) { return; }
415 	if (!force && !this._noDefer && this._deferNotifications("modify", modifies)) { return; }
416 
417 	this._batchNotify(modifies["cn"]);
418 };

431 /**
432  * @private
433  */
434 ZmContactsApp.prototype.handleOp =
435 function(op) {
436 	switch (op) {
437 		case ZmOperation.NEW_CONTACT:
438 		case ZmOperation.NEW_DISTRIBUTION_LIST:
439 		case ZmOperation.NEW_GROUP: {
440 			var type = (op == ZmOperation.NEW_CONTACT) ? null : ZmItem.GROUP;
441 			var loadCallback = new AjxCallback(this, this._handleLoadNewItem, [type, op == ZmOperation.NEW_DISTRIBUTION_LIST]);
442 			AjxDispatcher.require(["ContactsCore", "Contacts"], false, loadCallback, null, true);
443 			break;
444 		}
445 		case ZmOperation.NEW_ADDRBOOK: {
446 			var loadCallback = new AjxCallback(this, this._handleLoadNewAddrBook);
447 			AjxDispatcher.require(["ContactsCore", "Contacts"], false, loadCallback, null, true);
448 			break;
449 		}
450 	}
451 };

631 /**
632  * Gets the contact with the given address, if any. If it's not in our cache
633  * and we are given a callback, we do a search. If a search is performed then any
634  * addresses in the Address Lookup Group are also searched for.
635  *
636  * @param {String}	address			an email address
637  * @param {AjxCallback}	callback	the callback to run
638  * @return	{ZmContact}	the contact
639  * 
640  * @see		#setAddrLookupGroup
641  */
642 ZmContactsApp.prototype.getContactByEmail =
643 function(address, callback) {
644 	if (!address) { return null; }
645 	var addr = address.toLowerCase();
646 	var contact = this._byEmail[addr];
647 
648 	// if we have a failed search for this address, or have loaded all contacts,
649 	// don't bother doing a search
650 	if (!contact && this._notFound(addr)) {
651 		this._removeAddrFromLookupGroup(addr);
652 		if (callback) { callback.run(null); }
653 		return null;
654 	}
655 
656 	// found a cached contact, return it
657 	if (contact) {
658 		this._removeAddrFromLookupGroup(addr);
659 		contact = this._realizeContact(contact);
660 		contact._lookupEmail = address;	// so caller knows which address matched
661 		if (callback) { callback.run(contact); }
662 		return contact;
663 	}
664 
665 	// search for contact
666 	if (callback) {
667 		var search = null,
668 			isGroupSearch = false,
669 			lookupAddrs = [];
670 		if (this._addrLookupHash && this._addrLookupHash[addr]) {
671 			if (this._addrLookupList) {
672 				for (var i = 0; i < this._addrLookupList.length; i++) {
673 					lookupAddrs.push(this._addrLookupList[i]);
674 				}
675 				search = this._getSearchForAddresses(this._addrLookupList);
676 				isGroupSearch = true;
677 				this._addrLookupList = null;
678 			}
679 			this._addrLookupHash[addr].push(callback);
680 		} else {
681 			search = this._getSearchForAddresses([address]);
682 		}
683 
684 		if (search) {
685 			var respCallback = new AjxCallback(this, this._handleResponseSearch, [isGroupSearch ? lookupAddrs : addr, isGroupSearch, callback]);
686 			search.execute({callback:respCallback, noBusyOverlay:true});
687 		}
688 	}
689 };
690 
691 /**
692  * @private
693  */
694 ZmContactsApp.prototype._handleResponseSearch =
695 function(addr, isGroupSearch, callback, result) {
696 	var resp = result.getResponse();
697 	var contactList = resp && resp.getResults(ZmItem.CONTACT);
698 	if (isGroupSearch) {
699 		var list = contactList.getArray();
700 		for (var i = 0; i < list.length; i++) {
701 			this._updateLookupCache(list[i]);
702 		}
703 		for (var i = 0; i < addr.length; i++) {
704 			var a = addr[i];
705 			if (!this._byEmail[a]) {
706 				this._updateLookupCache(null, a); // Make sure there's a null entry in the map for the address.	
707 			}
708 			var callbacks = this._addrLookupHash[a];
709 			if (callbacks && callbacks.length) {
710 				for (var j = 0; j < callbacks.length; j++) {
711 					callbacks[j].run(this._byEmail[a]);
712 				}
713 			}
714 			this._removeAddrFromLookupGroup(a);
715 		}
716 	} else {
717 		var contact = contactList ? contactList.get(0) : null;	// return null if not found
718 		this._updateLookupCache(contact, addr);
719 		this._byEmail[addr] = contact;
720 		callback.run(contact);
721 	}
722 };

724 /**
725  * Gets the contacts with the given addresses, if any. If there are addresses not in our cache
726  * and we are given a callback, we do a search. Unlike {@link #getContactByEmail}, this method does not
727  * use or modify the Address Lookup Group.
728  *
729  * @param {Array}	addresses	an array of {@link AjxEmailAddress} objects
730  * @param {AjxCallback}	callback	the callback to run
731  * @return	{Array}	an array of [{@link AjxEmailAddress}, {@link ZmContact}] pairs.
732  * 
733  * @see		#setAddrLookupGroup
734  */
735 ZmContactsApp.prototype.getContactsByEmails =
736 function(addresses, callback) {
737 	// Go through the addresses, separating known ones from unknown.
738 	var resultArray = [],
739 		searchAddresses = null,
740 		searchAddressStrings = null;
741 	for (var i = 0, count = addresses.length; i < count; i++) {
742 		var address = addresses[i];
743 		var contact = this.getContactByEmail(address.getAddress());
744 		if (contact || contact === null) {
745 			resultArray.push({ address: address, contact: contact });
746 		} else {
747 			searchAddresses = searchAddresses || [];
748 			searchAddressStrings = searchAddressStrings || [];
749 			searchAddresses.push(address);
750 			searchAddressStrings.push(address.getAddress());
751 		}
752 	}
753 
754 	// See if we can exit without performing a search.
755 	if (!callback) {
756 		return resultArray;
757 	}
758 	if (!searchAddresses) {
759 		callback.run(resultArray);
760 		return resultArray;
761 	}
762 
763 	// Perform the search.
764 	var search = this._getSearchForAddresses(searchAddressStrings);
765 	var respCallback = new AjxCallback(this, this._handleResponseSearchByEmails, [searchAddresses, resultArray, callback]);
766 	search.execute({callback:respCallback});
767 };
768 
769 /**
770  * @private
771  */
772 ZmContactsApp.prototype._handleResponseSearchByEmails =
773 function(addresses, resultArray, callback, result) {
774 	// get contact list
775 	var resp = result.getResponse();
776 	var list = resp && resp.getResults(ZmItem.CONTACT);
777 	if (!list) callback.run(resultArray);
778 
779 	// get contact emails
780 	for (var index = 0, count = list.size(); index < count; index++) {
781 		var contact = list.get(index);
782 		for (var i = 1; true; i++) {
783 			var aname = ZmContact.getAttributeName(ZmContact.F_email, i);
784 			var avalue = contact.getAttr(aname);
785 			if (!avalue) break;
786 			this._byEmail[avalue] = contact;
787 		}
788 	}
789 
790 	// Fill in the results.
791 	for (var i = 0, count = addresses.length; i < count; i++) {
792 		var address = addresses[i];
793 		var contact = this.getContactByEmail(address.getAddress());
794 		resultArray.push({ address: address, contact: contact });
795 	}
796 	callback.run(resultArray);
797 };

799 /**
800  * @private
801  */
802 ZmContactsApp.prototype._getSearchForAddresses =
803 function(addrs) {
804 	var buffer;
805 	if (addrs.length == 1) {
806 		buffer = ["to:", addrs[0], " not #type:group"];
807 	} else {
808 		buffer = ["("];
809 		for (var i = 0, count = addrs.length; i < count; i++) {
810 			if (i > 0) {
811 				buffer.push(" OR ");
812 			}
813 			buffer.push("to:");
814 			buffer.push(addrs[i]);
815 		}
816 		buffer.push(") not #type:group");
817 	}
818 	var params = {
819 		query: buffer.join(""),
820 		limit: addrs.length * 2,
821 		types: AjxVector.fromArray([ZmItem.CONTACT])
822 	};
823 	return new ZmSearch(params);
824 };

831 /**
832  * Sets up a list of email addresses to use to find their contacts with a single search. The addresses passed
833  * in can either be raw email addresses (strings), or {@link AjxEmailAddress} objects. A list of the addresses is kept
834  * so that it can later be used to create a single search query. Each address will also keep track of the
835  * callbacks that will need to be run with its search result (it's a list of callbacks since the same address
836  * may be used in more than one context).
837  * <p>
838  * One example of this group approach is in rendering a message header, where each email address in the header
839  * is rendered based on whether it maps to a contact. The group approach lets us do a single search rather than
840  * several.
841  * </p>
842  *
843  * @param {Array}	addrs		a list of email addresses to look up
844  */
845 ZmContactsApp.prototype.setAddrLookupGroup =
846 function(addrs) {
847 	this._addrLookupList = [];
848 	this._addrLookupHash = {};
849 	if (addrs && addrs.length) {
850 		for (var i = 0; i < addrs.length; i++) {
851 			if (addrs[i]) {
852 				var addr = addrs[i].address || addrs[i];
853 				addr = (addr && AjxUtil.isString(addr)) ? addr.toLowerCase() : null;
854 				if (addr && !this._addrLookupHash[addr]) {
855 					this._addrLookupList.push(addr);
856 					this._addrLookupHash[addr] = [];
857 				}
858 			}
859 		}
860 	}
861 };

863 /**
864  * @private
865  */
866 ZmContactsApp.prototype._removeAddrFromLookupGroup =
867 function(addr) {
868 	if (!(this._addrLookupList && this._addrLookupList.length)) { return; }
869 	AjxUtil.arrayRemove(this._addrLookupList, addr);
870 	delete this._addrLookupHash[addr];
871 };

947 /**
948  * @private
949  */
950 ZmContactsApp.prototype._updateHash =
951 function(contact, doAdd, fields, hash, includeField, isNumeric) {
952 
953 	for (var index = 0; index < fields.length; index++) {
954 		var field = fields[index];
955 		for (var i = 1; true; i++) {
956 			var aname = ZmContact.getAttributeName(field, i);
957 			var avalue = ZmContact.getAttr(contact, aname);
958 			if (!avalue) break;
959 			avalue = isNumeric ? avalue.replace(/[^\d]/g, '') : avalue.toLowerCase();
960 			if (doAdd) {
961 				hash[avalue] = includeField ? {contact:contact, field:aname} : contact;
962 			} else {
963 				delete hash[avalue];
964 			}
965 		}
966 	}
967 };

1001 /**
1002  * Gets a {@link ZmContactList} with all of the user's local contacts. If that's a
1003  * large number, performance may be slow.
1004  * 
1005  * @param {AjxCallback}	callback			the callback to trigger after contact list loaded
1006  * @param {AjxCallback}	errorCallback		the callback to trigger in the event of an error
1007  * @param {ZmZimbraAccount}	account		the account to fetch contacts for
1008  * @return	{ZmContactList}	the contact list
1009  */
1010 ZmContactsApp.prototype.getContactList =
1011 function(callback, errorCallback, account) {
1012 	var acctId = (account && account.id) || appCtxt.getActiveAccount().id;
1013 	if (!this._contactList[acctId]) {
1014 		try {
1015 			// check if a parent controller exists and ask it for the contact list
1016 			if (this._parentController) {
1017 				this._contactList[acctId] = this._parentController.getApp(ZmApp.CONTACTS).getContactList();
1018 			} else {
1019 				this._contactList[acctId] = new ZmContactList(null);
1020 				var respCallback = new AjxCallback(this, this._handleResponseGetContactList, [callback]);
1021 				var accountName = (account && account.getEmail());
1022 				this._contactList[acctId].load(respCallback, errorCallback, accountName);
1023 			}
1024 			return this._contactList[acctId];
1025 		} catch (ex) {
1026 			this._contactList[acctId] = null;
1027 			throw ex;
1028 		}
1029 	} else {
1030 		if (callback && callback.isAjxCallback) {
1031 			callback.run(this._contactList[acctId]);
1032 		}
1033 		return this._contactList[acctId];
1034 	}
1035 };
