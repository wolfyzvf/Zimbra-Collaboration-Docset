 23 /**
 24  * Creates a folder tree controller.
 25  * @class
 26  * This class controls a tree display of folders.
 27  *
 28  * @param {Constant}	type			the type of organizer we are displaying/controlling ({@link ZmOrganizer.FOLDER} or {@link ZmOrganizer.SEARCH})
 29  * @param {DwtDropTarget}		dropTgt		the drop target for this type
 30  * 
 31  * @extends		ZmTreeController
 32  */
 33 ZmFolderTreeController = function(type, dropTgt) {
 34 
 35 	if (!arguments.length) { return; }
 36 
 37 	ZmTreeController.call(this, (type || ZmOrganizer.FOLDER));
 38 
 39 	this._listeners[ZmOperation.NEW_FOLDER]				= this._newListener.bind(this);
 40 	this._listeners[ZmOperation.PRIORITY_FILTER]		= this._priorityFilterListener.bind(this);
 41 	this._listeners[ZmOperation.RENAME_FOLDER]			= this._renameListener.bind(this);
 42 	this._listeners[ZmOperation.SHARE_FOLDER]			= this._shareFolderListener.bind(this);
 43 	this._listeners[ZmOperation.EMPTY_FOLDER]			= this._emptyListener.bind(this);
 44 	this._listeners[ZmOperation.RECOVER_DELETED_ITEMS]	= this._recoverListener.bind(this);
 45 	this._listeners[ZmOperation.SYNC_OFFLINE_FOLDER]	= this._syncOfflineFolderListener.bind(this);
 46 };

 56 /**
 57  * Shows the folder tree with certain folders hidden.
 58  * 
 59  * @param	{Hash}	params		a hash of parameters
 60  * @param	{Array}	params.omit		an array of folder ids to omit
 61  * @param	{ZmAccount}	params.account		the account
 62  */
 63 ZmFolderTreeController.prototype.show =
 64 function(params) {
 65 	var omit = params.omit || {};
 66 	for (var id in ZmFolder.HIDE_ID) {
 67 		omit[id] = true;
 68 	}
 69 	var dataTree = this.type != ZmOrganizer.VOICE && this.getDataTree(params.account);
 70 	if (dataTree) {
 71 	    for (var name in ZmFolder.HIDE_NAME) {
 72 			var folder = dataTree.getByName(name);
 73 			if (folder) {
 74 				omit[folder.id] = true;
 75 			}
 76 		}
 77 	}
 78 	params.omit = omit;
 79 	return ZmTreeController.prototype.show.call(this, params);
 80 };

 82 /**
 83 * Resets and enables/disables operations based on context.
 84 *
 85 * @param {DwtControl}	parent		the widget that contains the operations
 86 * @param {int}			id			the id of the currently selected/activated organizer
 87 */
 88 ZmFolderTreeController.prototype.resetOperations =
 89 function(parent, type, id) {
 90 
 91 	var emptyText = ZmMsg.emptyFolder; //ZmMsg.empty + (ZmFolder.MSG_KEY[id]?" "+ZmFolder.MSG_KEY[id] : "");
 92 	var folder = appCtxt.getById(id);
 93 	var hasContent = ((folder.numTotal > 0) || (folder.children && (folder.children.size() > 0)));
 94 
 95     // disable empty folder option for inbox, sent and drafts: bug 66656
 96     var isEmptyFolderAllowed = true;
 97     var y = folder.rid;
 98     if(y == ZmFolder.ID_INBOX || y == ZmFolder.ID_SENT || y == ZmFolder.ID_DRAFTS){
 99         isEmptyFolderAllowed = false;
100     }
101 
102 	// user folder or Folders header
103 	var nId = ZmOrganizer.normalizeId(id, this.type);
104 	if (nId == ZmOrganizer.ID_ROOT || ((!folder.isSystem()) /*&& !folder.isSyncIssuesFolder()*/)) {
105 		var isShareVisible = (!folder.link || folder.isAdmin());
106         if (appCtxt.isOffline) {
107             isShareVisible = !folder.getAccount().isMain && folder.getAccount().isZimbraAccount;
108         }
109 		parent.enableAll(true);
110 		parent.enable(ZmOperation.SYNC, folder.isFeed()/* || folder.hasFeeds()*/);
111 		parent.enable(ZmOperation.SYNC_ALL, folder.isFeed() || folder.hasFeeds());
112 		parent.enable(ZmOperation.SHARE_FOLDER, isShareVisible);
113 		parent.enable(ZmOperation.EMPTY_FOLDER, ((hasContent || folder.link) && isEmptyFolderAllowed && !appCtxt.isExternalAccount()));	// numTotal is not set for shared folders
114 		parent.enable(ZmOperation.RENAME_FOLDER, !(folder.isDataSource() || appCtxt.isExternalAccount()));		// dont allow datasource'd folder to be renamed via overview
115 		parent.enable(ZmOperation.NEW_FOLDER, !(folder.disallowSubFolder || appCtxt.isExternalAccount()));
116 
117 		if (folder.isRemote() && folder.isReadOnly()) {
118 			parent.enable([ZmOperation.NEW_FOLDER, ZmOperation.MARK_ALL_READ, ZmOperation.EMPTY_FOLDER], false);
119 		}
120         if (appCtxt.isExternalAccount()) {
121 			parent.enable([ZmOperation.DELETE_WITHOUT_SHORTCUT, ZmOperation.MOVE], false);
122 		}
123 	}
124 	// system folder
125 	else {
126 		parent.enableAll(false);
127 		// can't create folders under Drafts or Junk
128 		if (!folder.disallowSubFolder &&
129 			(nId == ZmFolder.ID_INBOX ||
130 			 nId == ZmFolder.ID_SENT  ||
131 			 nId == ZmFolder.ID_TRASH))
132 		{
133 			parent.enable(ZmOperation.NEW_FOLDER, true);
134 		}
135 		// "Empty" for Chats, Junk and Trash
136 		if (nId == ZmFolder.ID_SPAM  ||
137 			nId == ZmFolder.ID_TRASH ||
138 			nId == ZmFolder.ID_CHATS)
139 		{
140 			if (nId == ZmFolder.ID_SPAM) {
141 				emptyText = ZmMsg.emptyJunk;
142 			} else if (nId == ZmFolder.ID_TRASH) {
143 				 emptyText = ZmMsg.emptyTrash;
144 			}
145 			parent.enable(ZmOperation.EMPTY_FOLDER, hasContent);
146 		}
147 		// only allow Inbox and Sent system folders to be share-able for now
148 		if (!folder.link && (nId == ZmFolder.ID_INBOX || nId == ZmFolder.ID_SENT || nId == ZmFolder.ID_DRAFTS)) {
149 			parent.enable([ZmOperation.SHARE_FOLDER, ZmOperation.EDIT_PROPS], true);
150 		}
151         if (appCtxt.multiAccounts) {
152             var isShareVisible = !folder.getAccount().isMain && folder.getAccount().isZimbraAccount;
153             if(nId == ZmFolder.ID_SPAM || nId == ZmFolder.ID_TRASH) {
154                 isShareVisible = false;
155             }
156             parent.enable([ZmOperation.SHARE_FOLDER, ZmOperation.EDIT_PROPS], isShareVisible);
157         }
158 		// bug fix #30435 - enable empty folder for sync failures folder
159 		if (appCtxt.isOffline && nId == ZmOrganizer.ID_SYNC_FAILURES && hasContent) {
160 			parent.enable(ZmOperation.EMPTY_FOLDER, true);
161 		}
162 	}
163 
164 	parent.enable(ZmOperation.EXPAND_ALL, (folder.size() > 0));
165 	if (nId != ZmOrganizer.ID_ROOT && !folder.isReadOnly()) {
166 		// always enable for shared folders since we dont get this info from server
167 		parent.enable(ZmOperation.MARK_ALL_READ, (folder.numUnread > 0 || folder.link));
168 	}
169 
170 	var op = parent.getOp(ZmOperation.EMPTY_FOLDER);
171 	if (op) {
172 		op.setText(emptyText);
173 	}
174 
175     var isTrash = (nId == ZmOrganizer.ID_TRASH);
176 	// are there any external accounts associated to this folder?
177 	var button = parent.getOp(ZmOperation.SYNC);
178 	if (button) {
179 		var syncAllButton = parent.getOp(ZmOperation.SYNC_ALL);
180 		var hasFeeds = folder.hasFeeds();
181 		if (folder.isFeed()) {
182 			button.setEnabled(true);
183 			button.setVisible(true);
184 			button.setText(ZmMsg.checkFeed);
185 			if (syncAllButton) {
186 				syncAllButton.setEnabled(true);
187 				syncAllButton.setVisible(true);
188 				syncAllButton.setText(ZmMsg.checkAllFeed);
189 			}
190 		}
191 		else if (hasFeeds && !isTrash) {
192 			if (syncAllButton){
193 				syncAllButton.setEnabled(true);
194 				syncAllButton.setVisible(true);
195 				syncAllButton.setText(ZmMsg.checkAllFeed);
196 			}
197 		}
198 		else {
199 			var isEnabled = appCtxt.get(ZmSetting.POP_ACCOUNTS_ENABLED) || appCtxt.get(ZmSetting.IMAP_ACCOUNTS_ENABLED);
200 			if (!appCtxt.isOffline && isEnabled) {
201 				var dsCollection = AjxDispatcher.run("GetDataSourceCollection");
202 				var dataSources = dsCollection.getItemsFor(ZmOrganizer.normalizeId(folder.id));
203 				if (dataSources.length > 0) {
204 					button.setText(ZmMsg.checkExternalMail);
205 					button.setEnabled(true);
206 					button.setVisible(true);
207 				} else {
208 					button.setVisible(false);
209 				}
210 			}
211 			else {
212 				button.setVisible(false);
213 			}
214 
215 			if ((!hasFeeds || isTrash) && syncAllButton) {
216 				syncAllButton.setVisible(false);
217 			}
218 		}
219 	}
220 
221 	button = parent.getOp(ZmOperation.SYNC_OFFLINE_FOLDER);
222 	if (button) {
223 		if (!folder.isOfflineSyncable) {
224 			button.setVisible(false);
225 		} else {
226 			button.setVisible(true);
227 			button.setEnabled(true);
228 			var text = (folder.isOfflineSyncing)
229 				? ZmMsg.syncOfflineFolderOff : ZmMsg.syncOfflineFolderOn;
230 			button.setText(text);
231 		}
232 	}
233 	var priorityInboxEnabled = appCtxt.get(ZmSetting.PRIORITY_INBOX_ENABLED);
234 	var priorityInboxOp = parent.getOp(ZmOperation.PRIORITY_FILTER);
235 	if (priorityInboxOp) {
236 		priorityInboxOp.setVisible(priorityInboxEnabled);
237 		priorityInboxOp.setEnabled(priorityInboxEnabled);
238 	}
239 	this._enableRecoverDeleted(parent, isTrash);
240 
241 
242 	// we always enable sharing in case we're in multi-mbox mode
243 	this._resetButtonPerSetting(parent, ZmOperation.SHARE_FOLDER, appCtxt.get(ZmSetting.SHARING_ENABLED));
244 };

323 /**
324  * Called when a left click occurs (by the tree view listener). The folder that
325  * was clicked may be a search, since those can appear in the folder tree. The
326  * appropriate search will be performed.
327  *
328  * @param {ZmOrganizer}		folder		the folder or search that was clicked
329  * 
330  * @private
331  */
332 ZmFolderTreeController.prototype._itemClicked =
333 function(folder) {
334 	// bug 41196 - turn off new mail notifier if inactive account folder clicked
335 	if (appCtxt.isOffline) {
336 		var acct = folder.getAccount();
337 		if (acct && acct.inNewMailMode) {
338 			acct.inNewMailMode = false;
339 			var allContainers = appCtxt.getOverviewController()._overviewContainer;
340 			for (var i in allContainers) {
341 				allContainers[i].updateAccountInfo(acct, true, true);
342 			}
343 		}
344 	}
345 
346 	if (folder.type == ZmOrganizer.SEARCH) {
347 		// if the clicked item is a search (within the folder tree), hand
348 		// it off to the search tree controller
349 		var stc = this._opc.getTreeController(ZmOrganizer.SEARCH);
350 		stc._itemClicked(folder);
351 	} else if (folder.id == ZmFolder.ID_ATTACHMENTS) {
352 		var attController = AjxDispatcher.run("GetAttachmentsController");
353 		attController.show();
354 	}
355     else {
356 		var searchFor = ZmId.SEARCH_MAIL;
357 		if (folder.isInTrash()) {
358 			var app = appCtxt.getCurrentAppName();
359 			// if other apps add Trash to their folder tree, set appropriate type here:
360 			if (app == ZmApp.CONTACTS) {
361 				searchFor = ZmItem.CONTACT;
362 			}
363 		}
364 		var sc = appCtxt.getSearchController();
365 		var acct = folder.getAccount();
366 
367 		var sortBy = appCtxt.get(ZmSetting.SORTING_PREF, folder.nId);
368 		if (!sortBy) {
369 			sortBy = (sc.currentSearch && folder.nId == sc.currentSearch.folderId) ? null : ZmSearch.DATE_DESC;
370 		}
371 		else {
372 			//user may have saved folder with From search then switched views; don't allow From sort in conversation mode
373 			var groupMode = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
374 			if (groupMode == ZmItem.CONV && (sortBy == ZmSearch.NAME_ASC || sortBy == ZmSearch.NAME_DESC)) {
375 				sortBy = appCtxt.get(ZmSetting.SORTING_PREF, appCtxt.getCurrentViewId());  //default to view preference
376 				if (!sortBy) {
377 					sortBy = ZmSearch.DATE_DESC; //default
378 				}
379 				appCtxt.set(ZmSetting.SORTING_PREF, sortBy, folder.nId);
380 			}
381 		}
382 		var params = {
383 			query: folder.createQuery(),
384 			searchFor: searchFor,
385 			getHtml: (folder.nId == ZmFolder.ID_DRAFTS) || appCtxt.get(ZmSetting.VIEW_AS_HTML),
386 			types: ((folder.nId == ZmOrganizer.ID_SYNC_FAILURES) ? [ZmItem.MSG] : null), // for Sync Failures folder, always show in traditional view
387 			sortBy: sortBy,
388 			accountName: (acct && acct.name)
389 		};
390 
391 		sc.resetSearchAllAccounts();
392 
393 		if (appCtxt.multiAccounts) {
394 			// make sure we have permissions for this folder (in case an "external"
395 			// server was down during account load)
396 			if (folder.link && folder.perm == null) {
397 				var folderTree = appCtxt.getFolderTree(acct);
398 				if (folderTree) {
399 					var callback = new AjxCallback(this, this._getPermissionsResponse, [params]);
400 					folderTree.getPermissions({callback:callback, folderIds:[folder.id]});
401 				}
402 				return;
403 			}
404 
405 			if (appCtxt.isOffline && acct.hasNotSynced() && !acct.__syncAsked) {
406 				acct.__syncAsked = true;
407 
408 				var dialog = appCtxt.getYesNoMsgDialog();
409 				dialog.registerCallback(DwtDialog.YES_BUTTON, this._syncAccount, this, [dialog, acct]);
410 				dialog.setMessage(ZmMsg.neverSyncedAsk, DwtMessageDialog.INFO_STYLE);
411 				dialog.popup();
412 			}
413 		}
414 
415 		sc.search(params);
416 	}
417 };

496 /**
497  * Deletes a folder. If the folder is in Trash, it is hard-deleted. Otherwise, it
498  * is moved to Trash (soft-delete). If the folder is Trash or Junk, it is emptied.
499  * A warning dialog will be shown before the Junk folder is emptied.
500  *
501  * @param {DwtUiEvent}	ev		the UI event
502  * 
503  * @private
504  */
505 ZmFolderTreeController.prototype._deleteListener =
506 function(ev) {
507 	var organizer = this._getActionedOrganizer(ev);
508 
509 	// bug fix #35405 - accounts with disallowSubFolder flag set cannot be moved to Trash
510 	var trashFolder = appCtxt.isOffline ? this.getDataTree().getById(ZmFolder.ID_TRASH) : null;
511 	if (trashFolder && trashFolder.disallowSubFolder && organizer.numTotal > 0) {
512 		var d = appCtxt.getMsgDialog();
513 		d.setMessage(ZmMsg.errorCannotDeleteFolder);
514 		d.popup();
515 		return;
516 	}
517 
518 	if (organizer.nId == ZmFolder.ID_SPAM || organizer.isInTrash() || (trashFolder && trashFolder.disallowSubFolder)) {
519 		this._pendingActionData = organizer;
520 		var ds = this._deleteShield = appCtxt.getOkCancelMsgDialog();
521 		ds.reset();
522 		ds.registerCallback(DwtDialog.OK_BUTTON, this._deleteShieldYesCallback, this, organizer);
523 		ds.registerCallback(DwtDialog.CANCEL_BUTTON, this._clearDialog, this, this._deleteShield);
524 		var confirm;
525 		if (organizer.type == ZmOrganizer.SEARCH) {
526 			confirm = ZmMsg.confirmDeleteSavedSearch;
527 		} else if (organizer.disallowSubFolder || organizer.isMountpoint) {
528 			confirm = ZmMsg.confirmDeleteFolder;
529 		} else if (organizer.nId == ZmFolder.ID_TRASH) {
530 			confirm = ZmMsg.confirmEmptyTrashFolder;
531 		} else {
532 			confirm = ZmMsg.confirmEmptyFolder;
533 		}
534 		var msg = AjxMessageFormat.format(confirm, organizer.getName());
535 		ds.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
536 		ds.popup();
537 	} else {
538 		this._doMove(organizer, appCtxt.getById(ZmFolder.ID_TRASH));
539 	}
540 };

587 /**
588  * Don't allow dragging of system folders.
589  *
590  * @param {DwtDragEvent}	ev		the drag event
591  * 
592  * @private
593  */
594 ZmFolderTreeController.prototype._dragListener =
595 function(ev) {
596 	if (ev.action == DwtDragEvent.DRAG_START) {
597 		var folder = ev.srcControl.getData(Dwt.KEY_OBJECT);
598 		ev.srcData = {data:folder, controller:this};
599 		if (!(folder instanceof ZmFolder) || folder.isSystem() /*|| folder.isSyncIssuesFolder()*/) {
600 			ev.operation = Dwt.DND_DROP_NONE;
601 		}
602 	}
603 };

605 /**
606  * Handles the potential drop of something onto a folder. When something is dragged over
607  * a folder, returns true if a drop would be allowed. When something is actually dropped,
608  * performs the move. If items are being dropped, the source data is not the items
609  * themselves, but an object with the items (data) and their controller, so they can be
610  * moved appropriately.
611  *
612  * @param {DwtDropEvent}	ev		the drop event
613  * 
614  * @private
615  */
616 ZmFolderTreeController.prototype._dropListener =
617 function(ev) {
618 
619 	var dropFolder = ev.targetControl.getData(Dwt.KEY_OBJECT);
620 	var data = ev.srcData.data;
621 	var isShiftKey = (ev.shiftKey || ev.uiEvent.shiftKey);
622 
623 	if (ev.action == DwtDropEvent.DRAG_ENTER) {
624 		if (!data) {
625 			ev.doIt = false;
626 			return;
627 		}
628 		var type = ev.targetControl.getData(ZmTreeView.KEY_TYPE);
629 		if (data instanceof ZmFolder) {
630 			ev.doIt = dropFolder.mayContain(data, type) && !dropFolder.disallowSubFolder;
631 		} else if (data instanceof ZmTag) {
632 			ev.doIt = false; // tags cannot be moved
633 		} else {
634 			if (this._dropTgt.isValidTarget(data)) {
635 				ev.doIt = dropFolder.mayContain(data, type);
636 
637 				var action;
638 				var actionData = AjxUtil.toArray(data);
639 
640 				// walk thru the array and find out what action is allowed
641 				for (var i = 0; i < actionData.length; i++) {
642 					if (actionData[i] instanceof ZmItem) {
643 						action |= actionData[i].getDefaultDndAction(isShiftKey);
644 					}
645 				}
646 
647 				var plusDiv = document.getElementById(DwtId.DND_PLUS_ID);
648 				if (action && plusDiv) {
649 					// TODO - what if action is ZmItem.DND_ACTION_BOTH ??
650 					var isCopy = ((action & ZmItem.DND_ACTION_COPY) != 0);
651 					Dwt.setVisibility(plusDiv, isCopy);
652 				}
653 			} else {
654 				ev.doIt = false;
655 			}
656 		}
657 	} else if (ev.action == DwtDropEvent.DRAG_DROP) {
658 		if (data instanceof ZmFolder) {
659 			this._doMove(data, dropFolder);
660 		} else {
661 			var ctlr = ev.srcData.controller;
662 			var items = (data instanceof Array) ? data : [data];
663 			if (appCtxt.multiAccounts && !isShiftKey && !dropFolder.getAccount().isMain &&
664 				this._isMovingAcrossAccount(items, dropFolder))
665 			{
666 				var dialog = appCtxt.getYesNoMsgDialog();
667 				dialog.registerCallback(DwtDialog.YES_BUTTON, this._continueMovingAcrossAccount, this, [dialog, ctlr, items, dropFolder]);
668 				dialog.setMessage(ZmMsg.moveAcrossAccountWarning, DwtMessageDialog.WARNING_STYLE);
669 				dialog.popup();
670 			}
671 			else {
672 				ctlr._doMove(items, dropFolder, null, isShiftKey);
673 			}
674 		}
675 	}
676 };
677 
678 ZmFolderTreeController.prototype._isMovingAcrossAccount =
679 function(items, dropFolder) {
680 	for (var i = 0; i < items.length; i++) {
681 		var item = items[i];
682 		var itemAcct = item.getAccount();
683 		if (itemAcct && itemAcct != dropFolder.getAccount()) {
684 			return true;
685 		}
686 	}
687 	return false;
688 };
689 
690 ZmFolderTreeController.prototype._continueMovingAcrossAccount =
691 function(dialog, ctlr, items, dropFolder) {
692 	dialog.popdown();
693 	ctlr._doMove(items, dropFolder);
694 };
