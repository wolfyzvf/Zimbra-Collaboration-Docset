 23 /**
 24  * Creates an empty folder tree.
 25  * @class
 26  * This class represents a tree of folders. It may be typed, in which case
 27  * the folders are all of that type, or untyped.
 28  * 
 29  * @author Conrad Damon
 30  * 
 31  * @param {constant}	type		the organizer type
 32  * 
 33  * @extends	ZmTree
 34  */
 35 ZmFolderTree = function(type) {
 36 	ZmTree.call(this, type);
 37 };

 73 /**
 74  * Generic function for creating a folder. Handles any organizer type that comes
 75  * in the folder list.
 76  * 
 77  * @param {ZmFolder}	parent		the parent folder
 78  * @param {Object}	obj			the JSON with folder data
 79  * @param {ZmFolderTree}	tree			the containing tree
 80  * @param {String}	elementType		the type of containing JSON element
 81  * @param {Array}	path			the list of path elements
 82  * @param {ZmZimbraAccount}	account		the account this folder belongs to
 83  */
 84 ZmFolderTree.createFromJs =
 85 function(parent, obj, tree, elementType, path, account) {
 86 	if (!(obj && obj.id)) { return; }
 87 
 88 	var folder;
 89 	if (elementType == "search") {
 90 		var types;
 91 		if (obj.types) {
 92 			var t = obj.types.split(",");
 93 			types = [];
 94             var mailEnabled = appCtxt.get(ZmSetting.MAIL_ENABLED);
 95 			for (var i = 0; i < t.length; i++) {
 96                 var type = ZmSearch.TYPE_MAP[t[i]];
 97                 if (!type || (!mailEnabled && (type == ZmItem.CONV || type == ZmItem.MSG))) {
 98                     continue;
 99                 }
100 				types.push(type);
101 			}
102             if (types.length == 0) {
103                 return null;
104             }
105 		}
106 		DBG.println(AjxDebug.DBG2, "Creating SEARCH with id " + obj.id + " and name " + obj.name);
107 		var params = {
108 			id: obj.id,
109 			name: obj.name,
110 			parent: parent,
111 			tree: tree,
112 			numUnread: obj.u,
113 			query: obj.query,
114 			types: types,
115 			sortBy: obj.sortBy,
116 			account: account,
117 			color: obj.color,
118 			rgb: obj.rgb
119 		};
120 		folder = new ZmSearchFolder(params);
121 		ZmFolderTree._fillInFolder(folder, obj, path);
122 		ZmFolderTree._traverse(folder, obj, tree, (path || []), elementType, account);
123 	} else {
124 		var type = obj.view
125 			? (ZmOrganizer.TYPE[obj.view])
126 			: (parent ? parent.type : ZmOrganizer.FOLDER);
127 
128 		if (!type) {
129 			DBG.println(AjxDebug.DBG1, "No known type for view " + obj.view);
130 			return;
131 		}
132 		// let's avoid deferring folders for offline since multi-account folder deferring is hairy
133 		var hasGrants = (obj.acl && obj.acl.grant && obj.acl.grant.length > 0);
134 		if (appCtxt.inStartup && ZmOrganizer.DEFERRABLE[type] && !appCtxt.isOffline) {
135 			var app = appCtxt.getApp(ZmOrganizer.APP[type]);
136 			var defParams = {
137 				type:			type,
138 				parent:			parent,
139 				obj:			obj,
140 				tree:			tree,
141 				path:			path,
142 				elementType:	elementType,
143 				account:		account
144 			};
145 			app.addDeferredFolder(defParams);
146 		} else {
147 			var pkg = ZmOrganizer.ORG_PACKAGE[type];
148 			if (pkg) {
149 				AjxDispatcher.require(pkg);
150 			}
151 			folder = ZmFolderTree.createFolder(type, parent, obj, tree, path, elementType, account);
152             if (appCtxt.isExternalAccount() && folder.isSystem() && folder.id != ZmOrganizer.ID_ROOT) { return; }
153 			ZmFolderTree._traverse(folder, obj, tree, (path || []), elementType, account);
154 		}
155 	}
156 
157 	return folder;
158 };

169 /**
170  * @private
171  */
172 ZmFolderTree._traverse =
173 function(folder, obj, tree, path, elementType, account) {
174 
175 	var isRoot = (folder.nId == ZmOrganizer.ID_ROOT);
176 	if (obj.folder && obj.folder.length) {
177 		if (!isRoot) {
178 			path.push(obj.name);
179 		}
180 		for (var i = 0; i < obj.folder.length; i++) {
181 			var folderObj = obj.folder[i];
182 			var childFolder = ZmFolderTree.createFromJs(folder, folderObj, tree, (elementType || "folder"), path, account);
183 			if (folder && childFolder) {
184 				folder.children.add(childFolder);
185 			}
186 		}
187 		if (!isRoot) {
188 			path.pop();
189 		}
190 	}
191 	
192 	if (obj.search && obj.search.length) {
193 		if (!isRoot) {
194 			path.push(obj.name);
195 		}
196 		for (var i = 0; i < obj.search.length; i++) {
197 			var searchObj = obj.search[i];
198 			var childSearch = ZmFolderTree.createFromJs(folder, searchObj, tree, "search", path, account);
199 			if (childSearch) {
200 				folder.children.add(childSearch);
201 			}
202 		}
203 		if (!isRoot) {
204 			path.pop();
205 		}
206 	}
207 
208 	if (obj.link && obj.link.length) {
209 		for (var i = 0; i < obj.link.length; i++) {
210 			var link = obj.link[i];
211 			var childFolder = ZmFolderTree.createFromJs(folder, link, tree, "link", path, account);
212 			if (childFolder) {
213 				folder.children.add(childFolder);
214 			}
215 		}
216 	}
217 };

219 /**
220  * Creates the folder.
221  * 
222  * @param {String}	type		the folder type
223  * @param {ZmFolder}	parent		the parent folder
224  * @param {Object}	obj			the JSON with folder data
225  * @param {ZmFolderTree}	tree			the containing tree
226  * @param {Array}	path			the list of path elements
227  * @param {String}	elementType		the type of containing JSON element
228  * @param {ZmZimbraAccount}	account		the account this folder belongs to
229  */
230 ZmFolderTree.createFolder =
231 function(type, parent, obj, tree, path, elementType, account) {
232 	var orgClass = eval(ZmOrganizer.ORG_CLASS[type]);
233 	if (!orgClass) { return null; }
234 
235 	DBG.println(AjxDebug.DBG2, "Creating " + type + " with id " + obj.id + " and name " + obj.name);
236 
237 	var params = {
238 		id: 		obj.id,
239 		name: 		obj.name,
240 		parent: 	parent,
241 		tree: 		tree,
242 		color: 		obj.color,
243 		rgb:		obj.rgb,
244 		owner: 		obj.owner,
245 		oname: 		obj.oname,
246 		zid: 		obj.zid,
247 		rid: 		obj.rid,
248 		restUrl: 	obj.rest,
249 		url: 		obj.url,
250 		numUnread: 	obj.u,
251 		numTotal: 	obj.n,
252 		sizeTotal: 	obj.s,
253 		perm: 		obj.perm,
254 		link: 		elementType == "link",
255 		broken: 	obj.broken,
256 		account:	account,
257         retentionPolicy: obj.retentionPolicy
258 	};
259 
260 	var folder = new orgClass(params);
261 	ZmFolderTree._fillInFolder(folder, obj, path);
262 	ZmFolderTree.IS_PARSED[type] = true;
263 
264 	return folder;
265 };

267 /**
268  * @private
269  */
270 ZmFolderTree._fillInFolder =
271 function(folder, obj, path) {
272 	if (path && path.length) {
273 		folder.path = path.join("/");
274 	}
275 
276 	if (obj.f && folder._parseFlags) {
277 		folder._parseFlags(obj.f);
278 	}
279 
280 	folder._setSharesFromJs(obj);
281 };

306 /**
307  * Handles a missing link by marking its organizer as not there, redrawing it in
308  * any tree views, and asking to delete it.
309  *
310  * @param {int}	organizerType		the type of organizer (constants defined in {@link ZmOrganizer})
311  * @param {String}	zid			the zid of the missing folder
312  * @param {String}	rid			the rid of the missing folder
313  * @return	{Boolean}	<code>true</code> if the error is handled
314  */
315 ZmFolderTree.prototype.handleNoSuchFolderError =
316 function(organizerType, zid, rid) {
317 	var items = this.getByType(organizerType);
318 
319 	var treeView;
320 	var handled = false;
321 	if (items) {
322 		for (var i = 0; i < items.length; i++) {
323 			if ((items[i].zid == zid) && (items[i].rid == rid)) {
324 				// Mark that the item is not there any more.
325 				items[i].noSuchFolder = true;
326 
327 				// Change its appearance in the tree.
328 				if (!treeView) {
329 					var overviewId = appCtxt.getAppController().getOverviewId();
330 					treeView = appCtxt.getOverviewController().getTreeView(overviewId, organizerType);
331 				}
332 				var node = treeView.getTreeItemById(items[i].id);
333 				node.setText(items[i].getName(true));
334 
335 				// Ask if it should be deleted now.
336 				this.handleDeleteNoSuchFolder(items[i]);
337 				handled = true;
338 			}
339 		}
340 	}
341 	return handled;
342 };

344 /**
345  * Handles no such folder. The user will be notified that a linked organizer generated a "no such folder",
346  * error, giving the user a chance to delete the folder.
347  *
348  * @param {ZmOrganizer}	organizer	the organizer
349  */
350 ZmFolderTree.prototype.handleDeleteNoSuchFolder =
351 function(organizer) {
352 	var ds = appCtxt.getYesNoMsgDialog();
353 	ds.reset();
354 	ds.registerCallback(DwtDialog.YES_BUTTON, this._deleteOrganizerYesCallback, this, [organizer, ds]);
355 	ds.registerCallback(DwtDialog.NO_BUTTON, appCtxt.getAppController()._clearDialog, this, ds);
356 	var msg = AjxMessageFormat.format(ZmMsg.confirmDeleteMissingFolder, AjxStringUtil.htmlEncode(organizer.getName(false, 0, true)));
357 	ds.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
358 	ds.popup();
359 };

373 /**
374  * Issues a <code><BatchRequest></code> of <code><GetFolderRequest></code>s for existing
375  * mountpoints that do not have permissions set.
376  *
377  * @param	{Hash}	params	a hash of parameters
378  * @param {int}	params.type			the {@link ZmItem} type constant
379  * @param {AjxCallback}	params.callback			the callback to trigger after fetching permissions
380  * @param {Boolean}	params.skipNotify		<code>true</code> to skip notify after fetching permissions
381  * @param {Array}	params.folderIds			the list of folder Id's to fetch permissions for
382  * @param {Boolean}	params.noBusyOverlay		<code>true</code> to not block the UI while fetching permissions
383  * @param {String}	params.accountName		the account to issue request under
384  */
385 ZmFolderTree.prototype.getPermissions =
386 function(params) {
387 	var needPerms = params.folderIds || this._getItemsWithoutPerms(params.type);
388 
389 	// build batch request to get all permissions at once
390 	if (needPerms.length > 0) {
391 		var soapDoc = AjxSoapDoc.create("BatchRequest", "urn:zimbra");
392 		soapDoc.setMethodAttribute("onerror", "continue");
393 
394 		var doc = soapDoc.getDoc();
395 		for (var j = 0; j < needPerms.length; j++) {
396 			var folderRequest = soapDoc.set("GetFolderRequest", null, null, "urn:zimbraMail");
397 			var folderNode = doc.createElement("folder");
398 			folderNode.setAttribute("l", needPerms[j]);
399 			folderRequest.appendChild(folderNode);
400 		}
401 
402 		var respCallback = new AjxCallback(this, this._handleResponseGetShares, [params.callback, params.skipNotify]);
403 		appCtxt.getRequestMgr().sendRequest({
404 			soapDoc: soapDoc, 
405 			asyncMode: true,
406 			callback: respCallback,
407 			noBusyOverlay: params.noBusyOverlay,
408 			accountName: params.accountName
409 		});
410 	} else {
411 		if (params.callback) {
412 			params.callback.run();
413 		}
414 	}
415 };
416 
417 /**
418  * @private
419  */
420 ZmFolderTree.prototype._getItemsWithoutPerms =
421 function(type) {
422 	var needPerms = [];
423 	var orgs = type ? [type] : [ZmOrganizer.FOLDER, ZmOrganizer.CALENDAR, ZmOrganizer.TASKS, ZmOrganizer.BRIEFCASE, ZmOrganizer.ADDRBOOK];
424 
425 	for (var j = 0; j < orgs.length; j++) {
426 		var org = orgs[j];
427 		if (!ZmFolderTree.IS_PARSED[org]) { continue; }
428 
429 		var items = this.getByType(org);
430 
431 		for (var i = 0; i < items.length; i++) {
432 			if (items[i].link && items[i].shares == null) {
433 				needPerms.push(items[i].id);
434 			}
435 		}
436 	}
437 
438 	return needPerms;
439 };

441 /**
442  * @private
443  */
444 ZmFolderTree.prototype._handleResponseGetShares =
445 function(callback, skipNotify, result) {
446 	var batchResp = result.getResponse().BatchResponse;
447 	this._handleErrorGetShares(batchResp);
448 
449 	var resp = batchResp.GetFolderResponse;
450 	if (resp) {
451 		for (var i = 0; i < resp.length; i++) {
452 			var link = resp[i].link ? resp[i].link[0] : null;
453 			if (link) {
454 				var mtpt = appCtxt.getById(link.id);
455 				if (mtpt) {
456 					// update the mtpt perms with the updated link perms
457 					mtpt.perm = link.perm;
458                     if (link.n) mtpt.numTotal=link.n;
459                     if (link.u) mtpt.numUnread=link.u;
460 					mtpt._setSharesFromJs(link);
461 				}
462 
463 				if (link.folder && link.folder.length > 0) {
464 					var parent = appCtxt.getById(link.id);
465 					if (parent) {
466 						// TODO: only goes one level deep - should we recurse?
467 						for (var j = 0; j < link.folder.length; j++) {
468 							if (appCtxt.getById(link.folder[j].id)) { continue; }
469 							parent.notifyCreate(link.folder[j], "link", skipNotify);
470 						}
471 					}
472 				}
473 			}
474 		}
475 	}
476 
477 	if (callback) {
478 		callback.run();
479 	}
480 };

482 /**
483  * Handles errors that come back from the GetShares batch request.
484  *
485  * @param {Array}	organizerTypes	the types of organizer (constants defined in {@link ZmOrganizer})
486  * @param {Object}	batchResp			the response
487  *
488  */
489 ZmFolderTree.prototype._handleErrorGetShares =
490 function(batchResp) {
491 	var faults = batchResp.Fault;
492 	if (faults) {
493 		var rids = [];
494 		var zids = [];
495 		for (var i = 0, length = faults.length; i < length; i++) {
496 			var ex = ZmCsfeCommand.faultToEx(faults[i]);
497 			if (ex.code == ZmCsfeException.MAIL_NO_SUCH_FOLDER) {
498 				var itemId = ex.data.itemId[0];
499 				var index = itemId.lastIndexOf(':');
500 				zids.push(itemId.substring(0, index));
501 				rids.push(itemId.substring(index + 1, itemId.length));
502 			}
503 		}
504 		if (zids.length) {
505 			this._markNoSuchFolder(zids, rids);
506 		}
507 	}
508 };

510 /**
511  * Handles missing links by marking the organizers as not there
512  *
513  * @param {Array}	zids		the zids of the missing folders
514  * @param {Array}	rids		the rids of the missing folders. rids and zids must have the same length
515  *
516  */
517 ZmFolderTree.prototype._markNoSuchFolder =
518 function(zids, rids) {
519 	var treeData = appCtxt.getFolderTree();
520 	var items = treeData && treeData.root
521 		? treeData.root.children.getArray()
522 		: null;
523 
524 	for (var i = 0; i < items.length; i++) {
525 		for (var j = 0; j < rids.length; j++) {
526 			if ((items[i].zid == zids[j]) && (items[i].rid == rids[j])) {
527 				items[i].noSuchFolder = true;
528 			}
529 		}
530 	}
531 };

533 /**
534  * @private
535  */
536 ZmFolderTree.prototype._sortFolder =
537 function(folder) {
538 	var children = folder.children;
539 	if (children && children.length) {
540 		children.sort(ZmFolder.sortCompare);
541 		for (var i = 0; i < children.length; i++)
542 			this._sortFolder(children[i]);
543 	}
