 16 /**
 17  * @class
 18  * This static class serves as a central location for registering and calling
 19  * public API methods. For example, the registration of a public API method for
 20  * pulling up the compose form might look something like this:
 21  * 
 22  * <pre>
 23  * AjxDispatcher.register("Compose", "Mail", new AjxCallback(this, this.doAction));
 24  * </pre>
 25  * 
 26  * and a client call of it like this:
 27  * 
 28  * <pre>
 29  * AjxDispatcher.run("Compose", {action:ZmOperation.NEW_MESSAGE, inNewWindow:false});
 30  * </pre>
 31  * 
 32  * Registration will most likely need to happen in a constructor, when 'this' is
 33  * available, since you'll most likely want the function call to happen in that
 34  * object's context.
 35  * <p>
 36  * A package can also register a callback to be run once the package has loaded. One use case for
 37  * that is to register newly-defined classes as drop targets. There is a wrapper method around
 38  * AjxPackage.require() that will call that method and then run the post-load callback (if any):
 39  *
 40  * <pre>
 41  * AjxDispatcher.require("Calendar");
 42  * </pre>
 43  * 
 44  * @author Conrad Damon
 45  */
 46 AjxDispatcher = function() {}
 47 
 48 // Table of API names, packages, and associated function calls
 49 AjxDispatcher._registry = {};
 50 
 51 // Table of package names and callbacks to run after loading (optional)
 52 AjxDispatcher._package = {};
 53 
 54 AjxDispatcher._preLoad	= [];
 55 AjxDispatcher._postLoad	= [];
 56 AjxDispatcher._loadFunctionsEnabled	= false;
 57 AjxDispatcher._timedAction = null;
 58 
 59 /**
 60  * Adds a function to be called after the given package has been loaded.
 61  * 
 62  * @param {string}	pkg		the name of package
 63  * @param {AjxCallback}	callback	the callback to run after package has loaded
 64  */
 65 AjxDispatcher.addPackageLoadFunction =
 66 function(pkg, callback) {
 67 	var pkgData = AjxDispatcher._getPackageData(pkg);
 68 	if (!pkgData._loaded && !AjxPackage.isDefined(pkg)) {
 69 		pkgData.callback.push(callback);
 70 	} else {
 71 		AjxTimedAction.scheduleAction(new AjxTimedAction(callback, callback.run), 0);
 72 	}
 73 };

 75 /**
 76  * Adds a function to be called while a package is being loaded. A typical use
 77  * is to display a "Loading..." screen.
 78  * 
 79  * @param {AjxCallback}	callback	the callback to run after package has loaded
 80  */
 81 AjxDispatcher.addPreLoadFunction =
 82 function(callback) {
 83 	AjxDispatcher._preLoad.push(callback);
 84 };

 86 /**

 91  */
 92 AjxDispatcher.addPostLoadFunction =

 94 	AjxDispatcher._postLoad.push(callback);
 95 };

122 /**
123  * Checks if the given package has been loaded.
124  * 
125  * @param	{string}	pkg		the name of package
126  * @return	{boolean}	<code>true</code> if the package is loaded
127  */
128 AjxDispatcher.loaded =
129 function(pkg) {
130     var pkgData = AjxDispatcher._getPackageData(pkg);
131 	return (pkgData && pkgData._loaded) || AjxPackage.isDefined(pkg);
132 };
133 
134 /**
135  * Programmatically sets whether the given packages has been loaded. Use with care!
136  * 
137  * @param {string}	pkg		the name of package
138  * @return	{boolean}	if <code>true</code>, the package is loaded
139  */
140 AjxDispatcher.setLoaded =
141 function(pkg, loaded) {
142 	var pkgData = AjxDispatcher._getPackageData(pkg);
143 	pkgData._loaded = loaded;
144 	if (loaded) {
145 		var callbacks = pkgData.callback || [];
146 		for (var i = 0; i < callbacks.length; i++) {
147 			callbacks[i].run();
148 		}
149 	}
150 };

152 /**
153  * Registers an API method so that it may be called.
154  * 
155  * @param {string}	method	the name of the API method
156  * @param {string}	pkg		the name of required package(s)
157  * @param {AjxCallback}	callback	the callback to run for this API call
158  */
159 AjxDispatcher.registerMethod =
160 function(method, pkg, callback) {
161 	AjxDispatcher._registry[method] = {pkg:pkg, callback:callback};
162 };

164 /**
165  * Calls the given API method with the given arguments. It can be passed any
166  * number of arguments (provided after the API name), and they will be passed
167  * to the function that gets called.
168  * 
169  * PS: You are in a maze of twisty callbacks, all alike.
170  * 
171  * @param {string}	method		the name of the API method
172  * @param {boolean}	async			if <code>true</code>, load package asynchronously
173  * @param {AjxCallback}	callback		the callback to run with results (must be present
174  * 										if there are pre- or post-load functions)
175  * @param {boolean}		preLoadOk		if <code>true</code>, okay to run registered pre-load function
176  * 
177  * @private
178  */
179 AjxDispatcher.run =
180 function(params /*, arg1 ... argN */) {
181 	if (!params) { return; }
182 	var method, noLoad, async, callback, preLoadOk;
183 	if (typeof(params) == "string") {
184 		method = params;
185         async = false;
186 		preLoadOk = false;
187 	} else {
188 		method = params.method;
189 		noLoad = params.noLoad;
190 		callback = params.callback;
191         async = params.async != null ? params.async : Boolean(callback);
192 		preLoadOk = params.preLoadOk != null ? params.preLoadOk : (callback != null);
193 	}
194 	var item = AjxDispatcher._registry[method];
195 	if (!item) {
196 		// method hasn't been registered
197 		AjxPackage.__log("API method '" + method + "' not found");
198 		return;
199 	}
200 	AjxPackage.__log("Run method: " + method);
201 	var pkg = item.pkg;
202 	var args = [];
203 	for (var i = 1; i < arguments.length; ++i) {
204 		args.push(arguments[i]);
205 	}
206 	if (callback) {
207 		args.push(callback);
208 	}
209 	
210 	return AjxDispatcher.require(pkg, async, item.callback, args, preLoadOk);
211 };

213 /**
214  * Loads the given package, and runs its requested post-load callback. Clients should
215  * be careful not to mix async and sync calls for the same package, in order to avoid
216  * race conditions.
217  * 
218  * @param {string}	pkg				the name of the API method
219  * @param {boolean}	async				if <code>true</code>, load package asynchronously
220  * @param {AjxCallback}	callback			the callback to run after pkg load
221  * @param {array}	args				the args to pass to callback
222  * @param {boolean}	preLoadOk			if <code>true</code>, okay to run registered pre-load function
223  */
224 AjxDispatcher.require =
225 function(pkg, async, callback, args, preLoadOk) {
226 	if (!pkg) { return; }
227 	
228 	if (typeof(pkg) == "string") {
229 		pkg = [pkg];
230 	}
231 	var unloaded = [];
232 	for (var i = 0; i < pkg.length; i++) {
233 		var p = pkg[i];
234 		if (!AjxDispatcher._getPackageData(p)._loaded) {
235 			unloaded.push(p);
236 		}
237 	}
238 	if (unloaded.length == 0) {
239 		return AjxDispatcher._postLoadCallback(pkg, false, callback, args);
240 	} else {
241 		// need callback in order to run pre-load function
242 		var preLoad = AjxDispatcher._preLoad;
243 		if (preLoadOk && AjxDispatcher._loadFunctionsEnabled && preLoad.length) {
244 			AjxPackage.__log("pre-load function");
245 			AjxDispatcher._timedAction = new AjxCallback(null, AjxDispatcher._continueRequire, [unloaded, async, callback, args]);
246 			for (var i = 0; i < preLoad.length; i++) {
247 				preLoad[i].run();
248 			}
249 			window.setTimeout('AjxDispatcher._timedAction.run()', 0);
250 		} else {
251 			return AjxDispatcher._continueRequire(unloaded, async, callback, args);
252 		}
253 	}
254 };
255 
256 AjxDispatcher._continueRequire =
257 function(pkg, async, callback, args) {
258 	var pkgString = pkg.join(", ");
259 	AjxPackage.__log("------------------------------------- Loading package: " + pkgString);
260 	if (window.console && window.console.log) { console.log("------------------------------------- Loading package: " + pkgString); }
261 	if (async && callback) {
262 		var postLoadCallback = new AjxCallback(null, AjxDispatcher._postLoadCallback, [pkg, true, callback, args]);
263 		AjxPackage.require({name:pkg, callback:postLoadCallback});
264 	} else {
265 		var _st = new Date();
266 		for (var i = 0; i < pkg.length; i++) {
267 			AjxPackage.require(pkg[i]);
268 		}
269 		var _en = new Date();
270 		var t = _en.getTime() - _st.getTime();
271 		AjxPackage.__log("LOAD TIME for " + pkgString + ": " + t);
272 
273 		return AjxDispatcher._postLoadCallback(pkg, true, callback, args);
274 	}
275 };

277 AjxDispatcher._postLoadCallback =
278 function(pkg, pkgWasLoaded, callback, args) {
279     for (var i = 0; i < pkg.length; i++) {
280         AjxDispatcher._getPackageData(pkg[i])._loaded = true;
281     }
282     for (var i = 0; i < pkg.length; i++) {
283 		var pkgData = AjxDispatcher._getPackageData(pkg[i]);
284 		if (pkgWasLoaded && pkgData.callback.length && !pkgData.callbackDone) {
285 			pkgData.callbackDone = true;
286 			AjxPackage.__log("Running post-load package function for " + pkg[i]);
287 			var callbacks = pkgData.callback;
288 			for (var j = 0; j < callbacks.length; j++) {
289 				callbacks[j].run();
290 			}
291 			pkgData.callback.length = 0;
292 		}
293 	}
294 	if (pkgWasLoaded) {
295 		var postLoad = AjxDispatcher._postLoad;
296 		if (AjxDispatcher._loadFunctionsEnabled && postLoad.length) {
297 			for (var i = 0; i < postLoad.length; i++) {
298 				postLoad[i].run();
299 			}
300 		}
301 	}
302 	
303 	if (callback) {
304  		return callback.isAjxCallback ? callback.run1(args) : callback(args);
305 	}
306 };
307 
308 AjxDispatcher._getPackageData = function(pkg) {
309 	if (!AjxDispatcher._package[pkg]) {
310 		AjxDispatcher._package[pkg] = { callback: [] };
311 	}
312 	return AjxDispatcher._package[pkg];
313 };
