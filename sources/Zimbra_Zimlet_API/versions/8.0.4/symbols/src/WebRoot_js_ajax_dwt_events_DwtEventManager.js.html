 17 /**
 18  * static class that wraps around AjxEventManager
 19  * 
 20  * @private
 21  */
 22 DwtEventManager = function() {
 23 };
 24 
 25 DwtEventManager._instance = new AjxEventMgr();
 26 
 27 DwtEventManager._domEventToDwtMap = {
 28 	'ondblclick': DwtEvent.ONDBLCLICK,
 29 	'onmousedown': DwtEvent.ONMOUSEDOWN ,
 30 	'onmouseup': DwtEvent.ONMOUSEUP,
 31 	'onmousemove': DwtEvent.ONMOUSEMOVE,
 32 	'onmouseout': DwtEvent.ONMOUSEOUT,
 33 	'onmouseover': DwtEvent.ONMOUSEOVER,
 34 	'onselectstart': DwtEvent.ONSELECTSTART,
 35 	'onchange': DwtEvent.ONCHANGE
 36 };
 37 
 38 DwtEventManager.addListener = 
 39 function(eventType, listener) {
 40 	DwtEventManager._instance.addListener(eventType, listener);
 41 };
 42 
 43 DwtEventManager.notifyListeners = 
 44 function(eventType, event) {
 45 	DwtEventManager._instance.notifyListeners(eventType, event);
 46 };
 47 
 48 DwtEventManager.removeListener = 
 49 function(eventType, listener) {
 50 	DwtEventManager._instance.removeListener(eventType, listener);
 51 };
