 32 ZmMailPrefsPage.prototype.showMe =
 33 function() {
 34 	ZmPreferencesPage.prototype.showMe.call(this);
 35     if(appCtxt.isOffline){
 36         if(this._initializedAcctId != appCtxt.getActiveAccount().id) {
 37             this._initialized = false;
 38             this._initializedAcctId = appCtxt.getActiveAccount().id;
 39         }
 40     }
 41 	if (!this._initialized) {
 42 		this._initialized = true;
 43 		if (this._blackListControl && this._whiteListControl) {
 44 			var soapDoc = AjxSoapDoc.create("GetWhiteBlackListRequest", "urn:zimbraAccount");
 45 			var callback = new AjxCallback(this, this._handleResponseLoadWhiteBlackList);
 46 			appCtxt.getRequestMgr().sendRequest({soapDoc:soapDoc, asyncMode:true, callback:callback});
 47 		}
 48 	}
 49 };

 51 ZmMailPrefsPage.prototype.reset =
 52 function(useDefaults) {
 53     ZmPreferencesPage.prototype.reset.apply(this, arguments);
 54 
 55     var noDuration = true;
 56     if (this._startDateVal) {
 57         noDuration = (this._startDateVal.value == null || this._startDateVal.value == "");
 58         this._initStartEndDisplayFields();
 59     }
 60 
 61 
 62     var cbox = this.getFormObject(ZmSetting.VACATION_MSG_ENABLED);
 63     if (cbox) {
 64         this._handleEnableVacationMsg(cbox, noDuration);
 65         // HandleEnableVacationMsg will alter other (non-persisted) settings - update
 66         // their 'original' values so the section will not be thought dirty upon exit
 67         this._updateOriginalValue(ZmSetting.VACATION_DURATION_ENABLED);
 68         this._updateOriginalValue(ZmSetting.VACATION_CALENDAR_ENABLED);
 69     }
 70     this._initialAllDayFlag   = this._allDayCheckbox ? this._allDayCheckbox.isSelected() : true;
 71     this._updateOriginalValue(ZmSetting.VACATION_DURATION_ALL_DAY);
 72 
 73 
 74 	this._setPopDownloadSinceControls();
 75 
 76 	if (this._blackListControl && this._whiteListControl) {
 77 		this._blackListControl.reset();
 78 		this._whiteListControl.reset();
 79 	}
 80 };

 82 ZmMailPrefsPage.prototype.isDirty =
 83 function() {
 84 	var isDirty = ZmPreferencesPage.prototype.isDirty.call(this);
 85 	return (!isDirty) ? this.isWhiteBlackListDirty() : isDirty;
 86 };

 88 ZmMailPrefsPage.prototype.isWhiteBlackListDirty =
 89 function() {
 90 	if (this._blackListControl && this._whiteListControl) {
 91 		return this._blackListControl.isDirty() ||
 92 			   this._whiteListControl.isDirty();
 93 	}
 94 	return false;
 95 };

 97 ZmMailPrefsPage.prototype.addCommand =
 98 function(batchCmd) {
 99 	if (this.isWhiteBlackListDirty()) {
100 		var soapDoc = AjxSoapDoc.create("ModifyWhiteBlackListRequest", "urn:zimbraAccount");
101 		this._blackListControl.setSoapContent(soapDoc, "blackList");
102 		this._whiteListControl.setSoapContent(soapDoc, "whiteList");
103 
104 		var respCallback = new AjxCallback(this, this._handleResponseModifyWhiteBlackList);
105 		batchCmd.addNewRequestParams(soapDoc, respCallback);
106 	}
107 };

109 ZmMailPrefsPage.prototype._handleResponseModifyWhiteBlackList =
110 function(result) {
111 	this._blackListControl.saveLocal();
112 	this._whiteListControl.saveLocal();
113 };

115 ZmMailPrefsPage.prototype._setPopDownloadSinceControls =
116 function() {
117 	var popDownloadSinceValue = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE_VALUE);
118     var value = appCtxt.get(ZmSetting.POP_DOWNLOAD_SINCE);
119 	if (popDownloadSinceValue && value) {
120 		var date = AjxDateFormat.parse("yyyyMMddHHmmss'Z'", value);
121 		date.setMinutes(date.getMinutes() - date.getTimezoneOffset());
122 
123 		popDownloadSinceValue.setText(AjxMessageFormat.format(ZmMsg.externalAccessPopCurrentValue, date));
124         popDownloadSinceValue.setVisible(true);
125 	}  else if( popDownloadSinceValue ) {
126         popDownloadSinceValue.setVisible(false);
127     }
128 
129 	var popDownloadSince = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE);
130 	if (popDownloadSince) {
131 		popDownloadSince.setSelectedValue(appCtxt.get(ZmSetting.POP_DOWNLOAD_SINCE));
132 	}

135 ZmMailPrefsPage.prototype._createControls =
136 function() {
137     AjxDispatcher.require(["CalendarCore"]);
138 	ZmPreferencesPage.prototype._createControls.apply(this, arguments);
139 
140 	this._sId = this._htmlElId + "_startMiniCal";
141 	this._eId = this._htmlElId + "_endMiniCal";
142 
143 
144     this._startDateField = Dwt.byId(this._htmlElId + "_VACATION_FROM1");
145 	this._endDateField = Dwt.byId(this._htmlElId + "_VACATION_UNTIL1");
146 
147 	if (this._startDateField && this._endDateField) {
148 		this._startDateVal = Dwt.byId(this._htmlElId + "_VACATION_FROM");
149 		this._endDateVal = Dwt.byId(this._htmlElId + "_VACATION_UNTIL");
150         if(this._startDateVal.value.length < 15){
151             this._startDateVal.value = appCtxt.get(ZmSetting.VACATION_FROM);
152         }
153         if(this._endDateVal.value.length < 15){
154             this._endDateVal.value = appCtxt.get(ZmSetting.VACATION_UNTIL);            
155         }
156 
157 		this._formatter = new AjxDateFormat("yyyyMMddHHmmss'Z'");
158 
159 	    var timeSelectListener = new AjxListener(this, this._timeChangeListener);
160 	    this._startTimeSelect = new ZmTimeInput(this, ZmTimeInput.START);
161 	    this._startTimeSelect.reparentHtmlElement(this._htmlElId + "_VACATION_FROM_TIME");
162 	    this._endTimeSelect = new ZmTimeInput(this, ZmTimeInput.END);
163 	    this._endTimeSelect.reparentHtmlElement(this._htmlElId + "_VACATION_UNTIL_TIME");
164         this._startTimeSelect.addChangeListener(timeSelectListener);
165         this._endTimeSelect.addChangeListener(timeSelectListener);
166 
167         var noDuration = (this._startDateVal.value == null || this._startDateVal.value == "");
168         this._initStartEndDisplayFields();
169 
170 		var dateButtonListener = new AjxListener(this, this._dateButtonListener);
171 		var dateCalSelectionListener = new AjxListener(this, this._dateCalSelectionListener);
172 		var dateFieldListener = AjxCallback.simpleClosure(this._dateFieldListener, this);
173 
174 		this._startDateButton = ZmCalendarApp.createMiniCalButton(this, this._sId, dateButtonListener, dateCalSelectionListener);
175 		this._endDateButton = ZmCalendarApp.createMiniCalButton(this, this._eId, dateButtonListener, dateCalSelectionListener);
176 
177 		Dwt.setHandler(this._startDateField, DwtEvent.ONBLUR, dateFieldListener);
178 		Dwt.setHandler(this._endDateField, DwtEvent.ONBLUR, dateFieldListener);
179 
180 		this._durationCheckbox = this.getFormObject(ZmSetting.VACATION_DURATION_ENABLED);
181         this._extMsgCheckbox = this.getFormObject(ZmSetting.VACATION_EXTERNAL_MSG_ENABLED);
182 		if (this._extMsgCheckbox._textEl) {
183 			this._extMsgCheckbox._textEl.style.paddingRight = "3px";
184 		}
185 
186         this._allDayCheckbox = this.getFormObject(ZmSetting.VACATION_DURATION_ALL_DAY);
187         // Base initial _allDayCheckbox.checked on whether the start is Midnight and end
188         // is 23:59:59 (which fortunately cannot be directly specified by the user).
189         // Do this because we do not have any linkage to the OOO appt.
190         var prefStartDate = this._getPrefDate(ZmSetting.VACATION_FROM);
191         var prefEndDate   = this._getPrefDate(ZmSetting.VACATION_UNTIL);
192         if ((prefStartDate != null) && (prefEndDate != null)) {
193             var startDate = new Date(prefStartDate);
194             startDate.setHours(0, 0, 0, 0);  //we set this just to compare - took me a while to figure out
195             var endDate = new Date(prefEndDate);
196             endDate.setHours(23, 59, 59, 0); //we set this just to compare.
197             this._initialAllDayFlag = ((prefStartDate.getTime() == startDate.getTime()) &&
198                                        (prefEndDate.getTime()   == endDate.getTime()));
199         } else {
200             this._initialAllDayFlag = true;
201         }
202         this._allDayCheckbox.setSelected(this._initialAllDayFlag);
203         this._updateOriginalValue(ZmSetting.VACATION_DURATION_ALL_DAY);
204 	}
205 
206 
207 
208 	var cbox = this.getFormObject(ZmSetting.VACATION_MSG_ENABLED);
209 	if (cbox) {
210 		this._handleEnableVacationMsg(cbox, noDuration);
211         // HandleEnableVacationMsg will alter other (non-persisted) settings - update
212         // their 'orginal' values so the section will not be thought dirty upon exit
213         this._updateOriginalValue(ZmSetting.VACATION_DURATION_ENABLED);
214         this._updateOriginalValue(ZmSetting.VACATION_CALENDAR_ENABLED);
215 	}
216 
217 	// enable downloadSince appropriately based on presence of downloadSinceEnabled
218 	var downloadSinceCbox = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE_ENABLED);
219 	if (downloadSinceCbox) {
220 		var downloadSince = this.getFormObject(ZmSetting.POP_DOWNLOAD_SINCE);
221 		if (downloadSince) {
222 			var enabled = downloadSince.getValue() != "";
223 			downloadSinceCbox.setSelected(enabled);
224 			downloadSince.setEnabled(enabled);
225 		}
226 	}
227 
228 	// Following code makes child nodes as siblings to separate the event-handling 
229 	// between labels and input
230 
231 	var input = Dwt.byId(DwtId.makeId(ZmId.WIDGET_INPUT, ZmId.OP_MARK_READ));
232 	if (input) {
233 		var inputParent = input.parentNode;
234 		var newParent = inputParent && inputParent.parentNode;
235 
236 		if (newParent){
237 		    var txtNode = input.nextSibling;
238 		    inputParent.removeChild(input);
239 		    newParent.appendChild(input);
240 
241 		    var lbl = inputParent.cloneNode(false);
242 		    lbl.innerHTML = txtNode.data;
243 		    lbl.id = lbl.id + "_end";
244 		    inputParent.removeChild(txtNode);
245 		    newParent.appendChild(lbl);
246 		}
247 
248 		// If pref's value is number of seconds, populate the input
249 		var value = appCtxt.get(ZmSetting.MARK_MSG_READ);
250 		if (value > 0) {
251 		    input.value = value;
252 		}
253 	}
254 
255 	this._setPopDownloadSinceControls();
256 };

259 ZmMailPrefsPage.prototype._getPrefDate =
260 function(settingName) {
261     var prefDate = null;
262     var prefDateText = appCtxt.get(settingName);
263     if (prefDateText && (prefDateText != "")) {
264         prefDate = this._formatter.parse(AjxDateUtil.dateGMT2Local(prefDateText));
265     }
266     return prefDate;
267 }

269 ZmMailPrefsPage.prototype._initStartEndDisplayFields =
270 function() {
271     var startDate = new Date();
272     startDate.setHours(0,0,0,0);
273     this._initDateTimeDisplayField(this._startDateVal, this._startTimeSelect,
274         this._startDateField, startDate);
275 
276     var endDate = new Date(startDate);
277     // Defaulting to 11:59 PM. Ignored for all day
278     endDate.setHours(23,59,0,0);
279     this._initDateTimeDisplayField(this._endDateVal, this._endTimeSelect,
280         this._endDateField, endDate);
281 }

283 ZmMailPrefsPage.prototype._initDateTimeDisplayField =
284 function(dateVal, timeSelect, dateField, date) {
285     var dateValue = (dateVal.value != null && dateVal.value != "")
286         ? (this._formatter.parse(dateVal.value)) : date;
287     timeSelect.set(dateValue);
288     dateValue = timeSelect.getValue(dateValue);
289     dateVal.value = this._formatter.format(dateValue);
290     dateField.value = AjxDateUtil.simpleComputeDateStr(dateValue);
291 }

303 ZmMailPrefsPage.prototype._dateButtonListener =
304 function(ev) {
305 	var calDate = ev.item == this._startDateButton
306 		? this._fixAndGetValidDateFromField(this._startDateField)
307 		: this._fixAndGetValidDateFromField(this._endDateField);
308 
309 	var menu = ev.item.getMenu();
310 	var cal = menu.getItem(0);
311 	cal.setDate(calDate, true);

315 ZmMailPrefsPage.prototype._fixAndGetValidDateFromField =
316 function(field) {
317 	var d = AjxDateUtil.simpleParseDateStr(field.value);
318 	if (!d || isNaN(d)) {
319 		d = new Date();
320 		field.value = AjxDateUtil.simpleComputeDateStr(d);
321 	}
322 	return d;
323 };

325 ZmMailPrefsPage.prototype._dateCalSelectionListener =
326 function(ev) {
327 	var parentButton = ev.item.parent.parent;
328 
329 	var newDate = AjxDateUtil.simpleComputeDateStr(ev.detail);
330 
331 	if (parentButton == this._startDateButton) {
332 		this._startDateField.value = newDate;
333 	} else {
334 		if (ev.detail < new Date()) { return; }
335 		this._endDateField.value = newDate;
336 	}
337 
338 	var sd = this._fixAndGetValidDateFromField(this._startDateField);
339 	var ed = this._fixAndGetValidDateFromField(this._endDateField);
340     if(this._startTimeSelect && this._endTimeSelect){
341         sd = this._startTimeSelect.getValue(sd);
342         ed = this._endTimeSelect.getValue(ed);
343     }
344 
345 	
346 	this._fixDates(sd, ed, parentButton == this._endDateButton);
347 
348 	if (this._durationCheckbox.isSelected()) {
349 		this._startDateVal.value = this._formatter.format(sd);
350         this._endDateVal.value = this._formatter.format(ed);
351 	}
352 };

354 ZmMailPrefsPage.prototype._dateFieldListener =
355 function(ev) {
356 	var sd = this._fixAndGetValidDateFromField(this._startDateField);
357 	var ed = this._fixAndGetValidDateFromField(this._endDateField);
358     if(this._startTimeSelect && this._endTimeSelect){
359         sd = this._startTimeSelect.getValue(sd);
360         ed = this._endTimeSelect.getValue(ed);
361     }
362     this._fixDates(sd, ed, DwtUiEvent.getTarget(ev) == this._endDateField);
363     this._startDateVal.value = this._formatter.format(AjxDateUtil.simpleParseDateStr(this._startDateField.value));
364     this._endDateVal.value = this._formatter.format(AjxDateUtil.simpleParseDateStr(this._endDateField.value));
365 };

367 ZmMailPrefsPage.prototype._timeChangeListener =
368 function(ev) {
369    var stDate = AjxDateUtil.simpleParseDateStr(this._startDateField.value);
370    var endDate = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
371    stDate = this._startTimeSelect.getValue(stDate);
372    endDate = this._endTimeSelect.getValue(endDate);
373    this._startDateVal.value = this._formatter.format(stDate);
374    this._endDateVal.value = this._formatter.format(endDate);
375 };
376 
377 /* Fixes the field values so that end date always is later than or equal to start date
378  * @param startDate	{Date}	The value of the start date field or calendar selection
379  * @param endDate	{Date}	The value of the end date field or calendar selection
380  * @param modifyStart {boolean}	Whether to modify the start date or end date when dates overlap. true for start date, false for end date
381 */
382 
383 ZmMailPrefsPage.prototype._fixDates =
384 function(startDate, endDate, modifyStart) {
385 	if (startDate > endDate) {
386 		// Mismatch; start date is after end date
387 		if (modifyStart) {
388 			// Set them to be equal
389 			this._startDateField.value = AjxDateUtil.simpleComputeDateStr(endDate);
390 		} else {
391 			// Put endDate a bit into the future
392 			this._endDateField.value = AjxDateUtil.simpleComputeDateStr(AjxDateUtil.getDateForNextDay(startDate,AjxDateUtil.FRIDAY));
393 		}
394 
395 	}
396 };

454 ZmMailPrefsPage.prototype._handleEnableVacationMsg =
455 function(cbox, noDuration, id, evt) {
456 	var textarea = this.getFormObject(ZmSetting.VACATION_MSG);
457     var extTextarea = this.getFormObject(ZmSetting.VACATION_EXTERNAL_MSG);
458     var externalTypeSelect = this.getFormObject(ZmSetting.VACATION_EXTERNAL_TYPE);
459 	if (textarea) {
460         if (id == ZmSetting.VACATION_DURATION_ALL_DAY) {
461             this._enableDateTimeControls(true);
462         } else if (id == ZmSetting.VACATION_DURATION_ENABLED) {
463             this._allDayCheckbox.setEnabled(cbox.isSelected());
464             this._enableDateTimeControls(cbox.isSelected());
465             var calCheckBox = this.getFormObject(ZmSetting.VACATION_CALENDAR_ENABLED);
466             calCheckBox.setEnabled(cbox.isSelected());
467             var calendarType = this.getFormObject(ZmSetting.VACATION_CALENDAR_TYPE);
468             calendarType.setEnabled(calCheckBox.isSelected() && cbox.isSelected());
469 		}else if(id == ZmSetting.VACATION_EXTERNAL_MSG_ENABLED){
470             externalTypeSelect.setEnabled(cbox.isSelected());
471             extTextarea.setEnabled(cbox.isSelected());
472         }else if(id == ZmSetting.VACATION_CALENDAR_ENABLED){
473             var calendarType = this.getFormObject(ZmSetting.VACATION_CALENDAR_TYPE);
474             calendarType.setEnabled(cbox.isSelected());
475          }else {
476             // MESSAGE_ENABLED, main On/Off switch
477 			var enabled = cbox.getSelectedValue()=="true";
478 			textarea.setEnabled(enabled);
479 
480             this._durationCheckbox.setEnabled(enabled);
481             var val = this._startDateVal.value && enabled ? true : false;
482             this._durationCheckbox.setSelected((val && !noDuration));
483 
484             var calCheckBox = this.getFormObject(ZmSetting.VACATION_CALENDAR_ENABLED);
485             calCheckBox.setEnabled((this._durationCheckbox.isSelected() || appCtxt.get(ZmSetting.VACATION_DURATION_ENABLED)) && enabled);
486             calCheckBox.setSelected(enabled && (appCtxt.get(ZmSetting.VACATION_CALENDAR_APPT_ID) != "-1"));
487 
488             var calendarType = this.getFormObject(ZmSetting.VACATION_CALENDAR_TYPE);
489             calendarType.setEnabled(calCheckBox.isSelected() && this._durationCheckbox.isSelected() && enabled);
490 
491             this._extMsgCheckbox.setEnabled(enabled);
492             var externalEnabled = this._extMsgCheckbox.isSelected() && enabled;
493 
494             externalTypeSelect.setEnabled(externalEnabled);
495             extTextarea.setEnabled(externalEnabled);
496 
497             this._allDayCheckbox.setEnabled(enabled && this._durationCheckbox.isSelected());
498             this._enableDateTimeControls(enabled && this._durationCheckbox.isSelected());
499 		}
500 	}
501 };

541 ZmMailPrefsPage.prototype._preSave =
542 function(callback) {
543     if (this._startDateField && this._endDateField) {
544         var stDate = AjxDateUtil.simpleParseDateStr(this._startDateField.value);
545         var endDate = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
546 
547         var allDay = this._allDayCheckbox.isSelected();
548         if (!allDay) {
549             // Add in time fields if not all-day
550             stDate = this._startTimeSelect.getValue(stDate);
551             endDate = this._endTimeSelect.getValue(endDate);
552         }  else  {
553             // For the prefs, need to set the all day end time
554             endDate.setHours(23, 59, 59, 0);
555         }
556         if (this._startDateField.disabled) {
557             this._startDateVal.value = "";
558         } else {

562              this._endDateVal.value = "";
563         } else {
564             this._endDateVal.value = this._formatter.format(endDate);
565         }
566 
567         this._oldStartDate = appCtxt.get(ZmSetting.VACATION_FROM);
568         this._oldEndDate   = appCtxt.get(ZmSetting.VACATION_UNTIL);
569     }
570 
571     if (callback) {
572         callback.run();
573     }

581 ZmMailPrefsPage.prototype._postSave =
582 function(changed) {
583     var form = this.getFormObject(ZmSetting.POLLING_INTERVAL);
584     if (form && form.getSelectedOption() && form.getSelectedOption().getDisplayValue() == ZmMsg.pollInstant && appCtxt.get(ZmSetting.INSTANT_NOTIFY)
585             && !appCtxt.getAppController().getInstantNotify()){
586         //turn on instant notify if not already on
587         appCtxt.getAppController().setInstantNotify(true);
588     } else {
589         //turn instant notify off if it's on
590         if (appCtxt.getAppController().getInstantNotify()) {
591             appCtxt.getAppController().setInstantNotify(false);
592 		}
593     }
594 
595 	var vacationChangePrefs = [
596 		ZmSetting.VACATION_MSG_ENABLED,
597 		ZmSetting.VACATION_FROM,
598 		ZmSetting.VACATION_UNTIL
599 	];
600 
601 	var modified = false;
602 	for (var i = 0; i < vacationChangePrefs.length; i++) {
603 		if (changed[vacationChangePrefs[i]]) {
604 			modified = true;
605 			break;
606 		}
607 	}
608 	if (modified) {
609         var soapDoc = AjxSoapDoc.create("ModifyPrefsRequest", "urn:zimbraAccount");
610         var node = soapDoc.set("pref", "TRUE");
611         node.setAttribute("name", "zimbraPrefOutOfOfficeStatusAlertOnLogin");
612         appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:true});
613     }
614 
615 	//if old end date was greater than today then fetch appt id from metadata and delete the old appointment
616 	var now = new Date();
617 	if (appCtxt.get(ZmSetting.VACATION_CALENDAR_APPT_ID) != "-1" && this._formatter && this._oldEndDate && 
618 		this._formatter.parse(AjxDateUtil.dateGMT2Local(this._oldEndDate)) > now)
619 	{
620 		ZmAppt.loadById(appCtxt.get(ZmSetting.VACATION_CALENDAR_APPT_ID),new AjxCallback(this, this._oooDeleteApptCallback));
621 	}
622     if (this._durationCheckbox && this._durationCheckbox.isSelected()) {
623         //Create calendar appointments for this out of office request.
624 	    var calCheckBox = this.getFormObject(ZmSetting.VACATION_CALENDAR_ENABLED);
625 	    if (calCheckBox && calCheckBox.isSelected()) {
626             var stDate  = this._getPrefDate(ZmSetting.VACATION_FROM);
627             var endDate = this._getPrefDate(ZmSetting.VACATION_UNTIL);
628 		    var allDay = this._allDayCheckbox.isSelected();
629             if (stDate != null && endDate != null) {
630                 if (allDay) {
631                     // Strip the time of day information - calendar view
632                     // creates all-day appt with just day info
633                     endDate.setHours(0,0,0,0);
634                 }
635 	            var calController = appCtxt.getApp(ZmApp.CALENDAR).getCalController();
636 	            calController.createAppointmentFromOOOPref(stDate,endDate, allDay, new AjxCallback(this, this._oooApptCallback));
637             }
638         }
639     }
640 };

838 ZmWhiteBlackList.prototype._addListener =
839 function() {
840 	if (this._listView.size() >= this._max) {
841 		var dialog = appCtxt.getMsgDialog();
842 		dialog.setMessage(ZmMsg.errorWhiteBlackListExceeded);
843 		dialog.popup();
844 		return;
845 	}
846 
847 	var val,
848         items = AjxStringUtil.trim(this._inputEl.getValue(), true);
849 	if (items.length) {
850         items = AjxStringUtil.split(items, [',', ';', ' ']);
851         for(var i=0; i<items.length; i++) {
852             val = items[i];
853             if(val) {
854                 this._addEmail(AjxStringUtil.htmlEncode(val));
855                 if (!this._add[val]) {
856                     this._add[val] = true;
857                 }
858             }
859         }
860 		this._inputEl.setValue("", true);
861 		this._inputEl.blur();
862 		this._inputEl.focus();
863 
864 		this.updateNumUsed();
865 	}
866 };

868 ZmWhiteBlackList.prototype._removeListener =
869 function() {
870 	var items = this._listView.getSelection();
871 	for (var i = 0; i < items.length; i++) {
872 		var item = items[i];
873 		this._listView.removeItem(item, true);
874 		var addr = item.toString();
875 		if (this._add[addr]) {
876 			delete this._add[addr];
877 		} else {
878 			this._remove[addr] = true;
879 		}
880 	}
881 
882 	this.updateNumUsed();
883 };
