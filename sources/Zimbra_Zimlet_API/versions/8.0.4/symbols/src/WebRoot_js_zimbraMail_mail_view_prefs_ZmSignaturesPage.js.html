 61 ZmSignaturesPage.prototype._rehashByName =
 62 function() {
 63 	this._byName = {};
 64 	for (var id in this._signatures) {
 65 		var signature = this._signatures[id];
 66         if(!signature._new){         // avoid messing with existing signatures with same names
 67 		    this._byName[signature.name] = signature;
 68         }
 69 	}
 70 };

 72 ZmSignaturesPage.prototype.getNewSignatures =
 73 function(onlyValid) {
 74 	var list = [];
 75 	this._rehashByName();
 76 	for (var id in this._signatures) {
 77 		var signature = this._signatures[id];
 78 		if (signature._new && !this._isAutoAddedAndBlank(signature) && !(onlyValid && this._isInvalidSig(signature, true))) {
 79 			list.push(signature);
 80 		}
 81 	}
 82 	return list;
 83 };

 90 ZmSignaturesPage.prototype.getModifiedSignatures =
 91 function() {
 92 
 93 	var array = [];
 94 	for (var id in this._signatures) {
 95 		var signature = this._signatures[id];
 96 		if (signature._new) { continue; }
 97 
 98 		if (this._hasChanged(signature)) {
 99 			array.push(signature);
100 		}
101 	}
102 	return array;
103 };

107 // returns a hash representing the current usage, based on form selects
108 ZmSignaturesPage.prototype._getUsage =
109 function(newOnly) {
110 
111 	var usage = {};
112 	var foundOne;
113 	for (var identityId in this._sigSelect) {
114 		usage[identityId] = {};
115 		for (var j = 0; j < ZmSignaturesPage.SIG_FIELDS.length; j++) {
116 			var field = ZmSignaturesPage.SIG_FIELDS[j];
117 			var select = this._sigSelect[identityId] && this._sigSelect[identityId][field];
118 			if (select) {
119 				var sigId = select.getValue();
120 				if (newOnly && this._newSigId[sigId]) {
121 					return true;
122 				}
123 				else {
124 					usage[identityId][field] = sigId;
125 				}
126 			}
127 		}
128 	}
129 	return newOnly ? false : usage;
130 };

132 // returns a hash representing the current usage, based on identity data

134 function() {
135 
136 	var usage = {};
137 	var collection = appCtxt.getIdentityCollection();
138 	var identities = collection && collection.getIdentities();
139 	for (var i = 0, len = identities.length; i < len; i++) {

144 			usage[identity.id][field] = identity.getField(field) || "";
145 		}

150 /**
151  * Returns a list of usage changes. Each item in the list details the identity,
152  * which type of signature, and the ID of the new signature.
153  */
154 ZmSignaturesPage.prototype.getChangedUsage =
155 function() {
156 
157 	var list = [];
158 	var usage = this._getUsage();
159 	for (var identityId in usage) {
160 		var u1 = this._origUsage[identityId];
161 		var u2 = usage[identityId];
162         if (u1 && u2){
163 		    for (var j = 0; j < ZmSignaturesPage.SIG_FIELDS.length; j++) {
164 			    var field = ZmSignaturesPage.SIG_FIELDS[j];
165 			    var savedSigId = (u1[field]) || ((field == ZmIdentity.REPLY_SIGNATURE) ? ZmIdentity.SIG_ID_NONE : u1[field]);
166 			    var curSigId = this._newSigId[u2[field]] || u2[field];
167 			    if (savedSigId != curSigId) {
168 				    list.push({identity:identityId, sig:field, value:curSigId});
169 		        }
170 		    }
171         }
172 	}
173 	return list;
174 };

190 ZmSignaturesPage.prototype.isDirty =
191 function() {
192 
193 	this._updateSignature();
194 
195 	var printSigs = function(sig) {
196 		if (AjxUtil.isArray(sig)) {
197 			return AjxUtil.map(sig, printSigs).join("\n");
198 		}
199 		return [sig.name, " (", ((sig._orig && sig._orig.value != sig.value) ? (sig._orig.value+" changed to ") : ""), sig.value, ")"].join("");
200 	}
201 
202 	var printUsages = function(usage) {
203 		if (AjxUtil.isArray(usage)) {
204 			return AjxUtil.map(usage, printUsages).join("\n");
205 		}
206 		return ["identityId: ", usage.identity, ", type: ", usage.sig, ", signatureId: ", usage.value].join("");
207 	}
208 
209 	if (this.getNewSignatures(false).length > 0) {
210 		AjxDebug.println(AjxDebug.PREFS, "Dirty preferences:\nNew signatures:\n" + printSigs(this.getNewSignatures(false)));
211 		return true;
212 	}
213 	if (this.getDeletedSignatures().length > 0) {
214 		AjxDebug.println(AjxDebug.PREFS, "Dirty preferences:\nDeleted signatures:\n" + printSigs(this.getDeletedSignatures()));
215 		return true;
216 	}
217 	if (this.getModifiedSignatures().length > 0) {
218 		AjxDebug.println(AjxDebug.PREFS, "Dirty preferences:\nModified signatures:\n" + printSigs(this.getModifiedSignatures()));
219 		return true;
220 	}
221 	if (this.getChangedUsage().length > 0) {
222 		AjxDebug.println(AjxDebug.PREFS, "Dirty preferences:\nSignature usage changed:\n" + printUsages(this.getChangedUsage()));
223 		return true;
224 	}
225 };

227 ZmSignaturesPage.prototype.validate =
228 function() {
229 	this._updateSignature();
230 	this._rehashByName();
231 
232 	for (var id in this._signatures) {
233 		var error = this._isInvalidSig(this._signatures[id]);
234 		if (error) {
235 			this._errorMsg = error;
236 			return false;
237 		}
238 	}
239 	return true;
240 };

242 // The 'strict' parameter will make the function return true if a signature is not
243 // saveable, even if it can be safely ignored. Without it, the function returns an error
244 // only if there's bad user input.
245 ZmSignaturesPage.prototype._isInvalidSig =
246 function(signature, strict) {
247 
248 	var hasName = AjxStringUtil._NON_WHITESPACE.test(signature.name);
249 	var hasContact = Boolean(signature.contactId);
250 	var hasValue = AjxStringUtil._NON_WHITESPACE.test(signature.getValue()) || hasContact;
251 	var isNameDefault = (this._newNamesHash[signature.name] != null);
252 	if (!hasName && !hasValue) {
253 		this._deleteSignature(signature);
254 		if (strict) {
255 			return true;
256 		}
257 	} else if (!hasName || (!hasValue && !isNameDefault)) {
258 		return !hasName ? ZmMsg.signatureNameMissingRequired : ZmMsg.signatureValueMissingRequired;
259 	} else if (strict && isNameDefault && !hasValue) {
260 		return true;
261 	}
262 	if (hasName && this._byName[signature.name]) {
263         // If its a new signature with a in-use name or a existing signature whose name the user is trying to edit to a existing name value
264 		if(signature._new || (this._byName[signature.name].id != signature.id))
265         return AjxMessageFormat.format(ZmMsg.signatureNameDuplicate, AjxStringUtil.htmlEncode(signature.name));
266 	}
267 	var sigValue = signature.value;
268 	var maxLength = appCtxt.get(ZmSetting.SIGNATURE_MAX_LENGTH);
269 	if (maxLength > 0 && sigValue.length > maxLength) {
270 		return AjxMessageFormat.format((signature.contentType == ZmMimeTable.TEXT_HTML)
271 			? ZmMsg.errorHtmlSignatureTooLong
272 			: ZmMsg.errorSignatureTooLong, maxLength);
273 	}
274 
275 	return false;
276 };

283 ZmSignaturesPage.prototype.addCommand =
284 function(batchCommand) {
285 
286 	// delete signatures
287 	var deletedSigs = this.getDeletedSignatures();
288 	for (var i = 0; i < deletedSigs.length; i++) {
289 		var signature = deletedSigs[i];
290 		var callback = new AjxCallback(this, this._handleDeleteResponse, [signature]);
291 		signature.doDelete(callback, null, batchCommand);
292 	}
293 
294 	// modify signatures
295 	var modifiedSigs = this.getModifiedSignatures();
296 	for (var i = 0; i < modifiedSigs.length; i++) {
297 		var signature = modifiedSigs[i];
298 		var comps = this._signatures[signature._htmlElId];
299 		var callback = new AjxCallback(this, this._handleModifyResponse, [signature]);
300 		var errorCallback = new AjxCallback(this, this._handleModifyError, [signature]);
301 		signature.save(callback, errorCallback, batchCommand);
302 	}
303 
304 	// add signatures
305 	var newSigs = this.getNewSignatures(true);
306 	for (var i = 0; i < newSigs.length; i++) {
307 		var signature = newSigs[i];
308 		signature._id = signature.id; // Clearing existing dummy id
309 		signature.id = null;
310 		var callback = new AjxCallback(this, this._handleNewResponse, [signature]);
311 		signature.create(callback, null, batchCommand);
312 	}
313 
314 	// signature usage
315 	var sigChanges = this.getChangedUsage();
316 	if (sigChanges.length) {
317 		var collection = appCtxt.getIdentityCollection();
318 		if (collection) {
319 			for (var i = 0; i < sigChanges.length; i++) {
320 				var usage = sigChanges[i];
321 				var identity = collection.getById(usage.identity);
322 				// don't save usage of new signature just yet
323 				if (identity && !this._isTempId[usage.value]) {
324 					identity.setField(usage.sig, usage.value);
325 					identity.save(null, null, batchCommand);
326 				}
327 			}
328 		}
329 	}
330 };

519 ZmSignaturesPage.prototype._addUsageSelect =
520 function(row, identity, signatures, sigType) {
521 
522 	var select = this._sigSelect[identity.id][sigType] = new DwtSelect(this);
523 	var curSigId = identity.getField(sigType);
524 	var noSigId = (sigType == ZmIdentity.REPLY_SIGNATURE) ? ZmIdentity.SIG_ID_NONE : "";
525 	this._addUsageSelectOption(select, {name:ZmMsg.noSignature, id:noSigId}, sigType, identity);
526 	if (signatures) {
527 		for (var i = 0, len = signatures.length; i < len; i++) {
528 			this._addUsageSelectOption(select, signatures[i], sigType, identity);
529 		}
530 	}
531 	var cell = row.insertCell(-1);
532 	select.reparentHtmlElement(cell);
533 };

535 // if we're adding to a reply select and the identity has no value for reply sig,
536 // default it to the regular sig
537 ZmSignaturesPage.prototype._addUsageSelectOption =
538 function(select, signature, sigType, identity) {
539 
540 	var curSigId = identity.getField(sigType);
541 	if (!curSigId && sigType == ZmIdentity.REPLY_SIGNATURE) {
542 		curSigId = identity.getField(ZmIdentity.SIGNATURE) || ZmIdentity.SIG_ID_NONE;
543 	}
544 	DBG.println(AjxDebug.DBG3, "Adding " + sigType + " option for " + identity.name + ": " + signature.name + " / " + signature.id + " (" + (curSigId == signature.id) + ")");
545 	select.addOption(signature.name, (curSigId == signature.id), signature.id);
546 };

548 // handles addition, removal, or rename of a signature within the form
549 ZmSignaturesPage.prototype._updateUsageSelects =
550 function(signature, action) {
551 
552 	if (!this._initialized) { return; }
553 
554 	for (var id in this._sigSelect) {
555 		for (var sigType in this._sigSelect[id]) {
556 			var select = this._sigSelect[id][sigType];
557 			if (select) {
558 				var hasOption = Boolean(select.getOptionWithValue(signature.id));
559 				var collection = appCtxt.getIdentityCollection();
560 				var identity = collection && collection.getById(id);
561 				if (action == ZmEvent.E_CREATE && !hasOption && identity) {
562 					this._addUsageSelectOption(select, signature, sigType, identity);
563 				}
564 				else if (action == ZmEvent.E_DELETE && hasOption && identity) {
565 					select.removeOptionWithValue(signature.id);
566 					var curSigId = identity.getField(sigType);
567 					if (curSigId == signature.id) {
568 						var noSigId = (sigType == ZmIdentity.REPLY_SIGNATURE) ? ZmIdentity.SIG_ID_NONE : "";
569 						select.setSelectedValue(noSigId);
570 					}
571 				}
572 				else if (action == ZmEvent.E_MODIFY && hasOption) {
573 					select.rename(signature.id, signature.name);
574 				}
575 			}
576 		}
577 	}
578 };

580 ZmSignaturesPage.prototype._identityChangeListener =
581 function(ev) {
582 
583 	var identity = ev.getDetail("item");
584 	if (!identity) { return; }
585 	
586 	var collection = appCtxt.getIdentityCollection();
587 	var id = identity.id;
588 	var signatures = appCtxt.getSignatureCollection().getSignatures(true);
589 	var table = document.getElementById(this._htmlElId + "_SIG_TABLE");
590 	if (ev.event == ZmEvent.E_CREATE) {
591 		var index = collection.getSortIndex(identity);
592 		this._addUsageSelects(identity, table, signatures, index);
593 		for (var i = 0; i < ZmSignaturesPage.SIG_FIELDS.length; i++) {
594 			var field = ZmSignaturesPage.SIG_FIELDS[i];
595 			var select = this._sigSelect[id] && this._sigSelect[id][field];
596 			if (select) {
597 				this._origUsage[id] = this._origUsage[id] || {};
598 				this._origUsage[id][field] = select.getValue();
599 			}
600 		}
601 	}
602 	else if (ev.event == ZmEvent.E_DELETE) {
603 		var row = document.getElementById(id + "_row");
604 		if (row) {
605 			table.deleteRow(row.rowIndex);
606 		}
607 		delete this._origUsage[id];
608 	}
609 	else if (ev.event == ZmEvent.E_MODIFY) {
610 		var row = document.getElementById(id + "_row");
611 		if (row) {
612 			var index = collection.getSortIndex(identity);
