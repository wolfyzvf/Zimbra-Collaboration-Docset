 21 /**
 22  * Creates the zimlet
 23  * @class
 24  * This class represents a zimlet.
 25  * 
 26  * @param	{String}	id		the id
 27  * @param	{String}	name	the name
 28  * @param	{Object}	parent	the parent
 29  * @param	{ZmTree}	tree	the tree
 30  * @param	{String}	color	the color
 31  * @extends		ZmOrganizer
 32  */
 33 ZmZimlet = function(id, name, parent, tree, color) {
 34 	ZmOrganizer.call(this, {type: ZmOrganizer.ZIMLET, id: id, name: name, parent: parent, tree: tree});
 35 };

 70 // Static methods
 71 /**
 72  * @private
 73  */
 74 ZmZimlet.createFromJs =
 75 function(parent, obj, tree, link) {
 76 	if (!obj && obj.length < 1) {return null;}
 77 
 78 	// create zimlet root
 79 	var zimletRoot = new ZmZimlet(ZmZimlet.ID_ZIMLET_ROOT, ZmMsg.zimlets, parent, tree, null, null);
 80 	if (obj && obj.length) {
 81 		var id = ZmZimlet.ID_ZIMLET;
 82 		for (var i = 0; i < obj.length; i++) {
 83 			var lbl = obj[i].processMessage(obj[i].zimletPanelItem.label);
 84 			// bug fix #23860 - unique-ify zimlet ID's so they dont conflict!
 85 			var zimletId = (++id) + "_z";
 86 			var childZimlet = new ZmZimlet(zimletId, lbl, zimletRoot, tree, null, null);
 87 			zimletRoot.children.add(childZimlet);
 88 			// WARNING: it's a bit unorthodox to do this linkage
 89 			// here, but we really do need these objects know about
 90 			// each other.
 91 			childZimlet._zimletContext = obj[i];
 92 			childZimlet._zimletContext._id = zimletId;
 93 			childZimlet._toolTip = obj[i].zimletPanelItem.toolTipText;
 94 			obj[i]._organizer = childZimlet;
 95 		}
 96 	}
 97 	return zimletRoot;
 98 };

100 /**
101  * Compares and sorts the zimlets by name (case-insensitive).
102  * 
103  * @param	{ZmZimlet}	zimletA		the zimlet
104  * @param	{ZmZimlet}	zimletB		the zimlet
105  * @return	{int}	0 if the zimlets match; 1 if "a" is before "b"; -1 if "b" is before "a"
106  */
107 ZmZimlet.sortCompare =
108 function(zimletA, zimletB) {
109 	var check = ZmOrganizer.checkSortArgs(zimletA, zimletB);
110 	if (!check) {return check;}
111 
112 	// sort by name
113 	var zimletAName = zimletA.name.toLowerCase();
114 	var zimletBName = zimletB.name.toLowerCase();
115 	if (zimletAName < zimletBName) {return -1;}
116 	if (zimletAName > zimletBName) {return 1;}
117 	return 0;
118 };

132 // Public methods
133 /**
134  * Resets the names.
135  * 
136  */
137 ZmZimlet.prototype.resetNames =
138 function() {
139 	var oldName = this.name;
140 	var oldToolTip = this._toolTip;
141 	if(this._zimletContext && this._toolTip) {
142 		this._toolTip = this._zimletContext.processMessage(this._toolTip);
143 	}
144 	if(this._zimletContext && this.name) {
145 		this.name = this._zimletContext.processMessage(this.name);
146 	}
147 	// Update only if there was a change
148 	if((oldName != this.name) || (oldToolTip != this._toolTip)) {
149 		var fields = {};
150 		fields[ZmOrganizer.F_NAME] = true;
151 		var details = {};
152 		details.fields = fields;
153 		this._notify(ZmEvent.E_MODIFY, details);
154 	}
155 };
