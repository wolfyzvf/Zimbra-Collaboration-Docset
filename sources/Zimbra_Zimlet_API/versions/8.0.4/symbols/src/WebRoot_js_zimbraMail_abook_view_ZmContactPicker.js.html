 22 /**
 23  * Creates a dialog that lets the user select addresses from a contact list.
 24  * @constructor
 25  * @class
 26  * This class creates and manages a dialog that lets the user select addresses
 27  * from a contact list. Two lists are maintained, one with contacts to select
 28  * from, and one that contains the selected addresses. Between them are buttons
 29  * to shuffle addresses back and forth between the two lists.
 30  *
 31  * @author Conrad Damon
 32  * 
 33  * @param {Array}	buttonInfo		the transfer button IDs and labels
 34  * 
 35  * @extends		DwtDialog
 36  */
 37 ZmContactPicker = function(buttonInfo) {
 38 
 39 	DwtDialog.call(this, {parent:appCtxt.getShell(), title:ZmMsg.selectAddresses, id: "ZmContactPicker"});
 40 
 41 	this._buttonInfo = buttonInfo;
 42 	this._initialized = false;
 43 	this._emailListOffset = 0; //client side paginating over email list. Offset of current page of email addresses. Quite different than _lastServerOffset if contacts have 0 or more than 1 email addresses.
 44 	this._serverContactOffset = 0; //server side paginating over contact list. Offset of last contact block we got from the server (each contact could have 0, 1, or more emails so we have to keep track of this separate from the view offset.
 45 	this._ascending = true; //asending or descending search. Keep it stored for pagination to do the right sort.
 46 	this._defaultQuery = ".";
 47 	this._emailList = new AjxVector();
 48 	this._detailedSearch = appCtxt.get(ZmSetting.DETAILED_CONTACT_SEARCH_ENABLED);
 49 	this._searchCleared = {};
 50 	this._ignoreSetDragBoundries = true;
 51 
 52 	this._searchErrorCallback = new AjxCallback(this, this._handleErrorSearch);
 53 };

 91 /**
 92 * Displays the contact picker dialog. The source list is populated with
 93 * contacts, and the target list is populated with any addresses that are
 94 * passed in. The address button that was used to popup the dialog is set
 95 * as the active button.
 96 *
 97 * @param {String}	buttonId	the button ID of the button that called us
 98 * @param {Hash}	addrs		a hash of 3 vectors (one for each type of address)
 99 * @param {String}	str		initial search string
100 */
101 ZmContactPicker.prototype.popup =
102 function(buttonId, addrs, str, account) {
103 
104 	if (!this._initialized) {
105 		this._initialize(account);
106 		this._initialized = true;
107 	}
108 	else if (appCtxt.multiAccounts && this._account != account) {
109 		this._account = account;
110 		this._resetSelectDiv();
111 	}
112 	this._emailListOffset = 0;
113 
114 	var searchFor = this._searchInSelect ? this._searchInSelect.getValue() : ZmContactsApp.SEARCHFOR_CONTACTS;
115 
116 	// reset column sorting preference
117 	this._chooser.sourceListView.setSortByAsc(ZmItem.F_NAME, true);
118 
119 	// reset button states
120 	this._chooser.reset();
121 	if (buttonId) {
122 		this._chooser._setActiveButton(buttonId);
123 	}
124 
125 	// populate target list if addrs were passed in
126 	if (addrs) {
127 		for (var id in addrs) {
128 			this._chooser.addItems(addrs[id], DwtChooserListView.TARGET, true, id);
129 		}
130 	}
131 
132 	for (var fieldId in this._searchField) {
133 		var field = this._searchField[fieldId];
134 		field.disabled = false;
135 		if (str) {
136 			field.className = "";
137 			field.value = AjxUtil.isObject(str) ? (str[fieldId] || "") : str;
138 			this._searchCleared[fieldId] = true;
139 		} else {
140 			field.className = "searchFieldHint";
141 			field.value = ZmContactPicker.HINT[fieldId];
142 			this._searchCleared[fieldId] = false;
143 		}
144 	}
145 	var focusField = this._searchField[ZmContactPicker.SEARCH_BASIC] || this._searchField[ZmContactPicker.SEARCH_NAME];
146 	focusField.focus();
147 
148 	// reset paging buttons
149 	this._prevButton.setEnabled(false);
150 	this._nextButton.setEnabled(false);
151 
152 	this.search(null, true, true);
153 
154 	DwtDialog.prototype.popup.call(this);
155     if ((this.getLocation().x < 0 ||  this.getLocation().y < 0) ){
156                 // parent window size is smaller than Dialog size
157                 this.setLocation(0,30);
158                 var size = Dwt.getWindowSize();
159                 var currentSize = this.getSize();
160                 var dragElement = document.getElementById(this._dragHandleId);
161                 DwtDraggable.setDragBoundaries(dragElement, 100 - currentSize.x, size.x - 100, 0, size.y - 100);
162     }
163 };

191 /**
192  * Performs a search.
193  * 
194  * @private
195  */
196 ZmContactPicker.prototype.search =
197 function(colItem, ascending, firstTime, lastId, lastSortVal, offset) {
198 	if (offset == undefined) {
199 		//this could be a call from DwtChooserListView.prototype._sortColumn, which means we have to reset the result and both server and client pagination.
200 		//In any case the results should be reset or are already reset so doesn't hurt to reset.
201 		this._resetResults();
202 	}
203 
204 	if (ascending === null || ascending === undefined) {
205 		ascending = this._ascending;
206 	}
207 	else {
208 		this._ascending = ascending;
209 	}
210 	
211 	var query;
212 	var queryHint = [];
213 	var conds = [];
214 	if (this._detailedSearch) {
215 		var nameQuery = this.getSearchFieldValue(ZmContactPicker.SEARCH_NAME);
216 		var emailQuery = this.getSearchFieldValue(ZmContactPicker.SEARCH_EMAIL);
217 		var deptQuery = this.getSearchFieldValue(ZmContactPicker.SEARCH_DEPT);
218 		var phoneticQuery = this.getSearchFieldValue(ZmContactPicker.SEARCH_PHONETIC);
219 		var isGal = this._searchInSelect && (this._searchInSelect.getValue() == ZmContactsApp.SEARCHFOR_GAL);
220 		query = nameQuery || "";
221 		if (emailQuery) {
222 			if (isGal) {
223 				conds.push([{attr:ZmContact.F_email, op:"has", value: emailQuery},
224 				{attr:ZmContact.F_email2, op:"has", value: emailQuery},
225 				{attr:ZmContact.F_email3, op:"has", value: emailQuery},
226 				{attr:ZmContact.F_email4, op:"has", value: emailQuery},
227 				{attr:ZmContact.F_email5, op:"has", value: emailQuery},
228 				{attr:ZmContact.F_email6, op:"has", value: emailQuery},
229 				{attr:ZmContact.F_email7, op:"has", value: emailQuery},
230 				{attr:ZmContact.F_email8, op:"has", value: emailQuery},
231 				{attr:ZmContact.F_email9, op:"has", value: emailQuery},
232 				{attr:ZmContact.F_email10, op:"has", value: emailQuery},
233 				{attr:ZmContact.F_email11, op:"has", value: emailQuery},
234 				{attr:ZmContact.F_email12, op:"has", value: emailQuery},
235 				{attr:ZmContact.F_email13, op:"has", value: emailQuery},
236 				{attr:ZmContact.F_email14, op:"has", value: emailQuery},
237 				{attr:ZmContact.F_email15, op:"has", value: emailQuery},
238 				{attr:ZmContact.F_email16, op:"has", value: emailQuery}
239 				]);
240 			} else {
241 				queryHint.push("to:"+emailQuery+"*");
242 			}
243 		}
244 		if (deptQuery && isGal) {
245 			conds.push({attr:ZmContact.F_department, op:"has", value: deptQuery});
246 		}
247 		if (phoneticQuery && !isGal) {
248 			var condArr = [];
249 			var phoneticQueryPieces = phoneticQuery.split(/\s+/);
250 			for (var i=0; i<phoneticQueryPieces.length; i++) {
251 				condArr.push("#"+ZmContact.F_phoneticFirstName + ":" + phoneticQueryPieces[i]);
252 				condArr.push("#"+ZmContact.F_phoneticLastName + ":" + phoneticQueryPieces[i]);
253 			}
254 			queryHint.push(condArr.join(" || "));
255 		}
256 	} else {
257 		query = this.getSearchFieldValue(ZmContactPicker.SEARCH_BASIC);
258 	}
259 	
260 
261 	if (this._searchInSelect) {
262 		var searchFor = this._searchInSelect.getValue();
263 		this._contactSource = (searchFor == ZmContactsApp.SEARCHFOR_CONTACTS || searchFor == ZmContactsApp.SEARCHFOR_PAS)
264 			? ZmItem.CONTACT
265 			: ZmId.SEARCH_GAL;
266 
267 		if (searchFor == ZmContactsApp.SEARCHFOR_PAS) {
268 			queryHint.push(ZmSearchController.generateQueryForShares([ZmId.ITEM_CONTACT]) || "is:local");
269 		} else if (searchFor == ZmContactsApp.SEARCHFOR_CONTACTS) {
270 			queryHint.push("is:local");
271 		} else if (searchFor == ZmContactsApp.SEARCHFOR_GAL) {
272             ascending = true;
273         }
274 	} else {
275 		this._contactSource = appCtxt.get(ZmSetting.CONTACTS_ENABLED, null, this._account)
276 			? ZmItem.CONTACT
277 			: ZmId.SEARCH_GAL;
278 
279 		if (this._contactSource == ZmItem.CONTACT) {
280 			queryHint.push("is:local");
281 		}
282 	}
283     
284     if (!query.length && this._contactSource == ZmId.SEARCH_GAL) {
285 		query = this._defaultQuery;
286 	}
287 
288     if (this._contactSource == ZmItem.CONTACT && query != "") {
289         query = query.replace(/\"/g, '\\"');
290         query = "\"" + query + "\"";
291     }
292 
293 	if (this._searchIcon) { //does not exist in ZmGroupView case
294 		this._searchIcon.className = "DwtWait16Icon";
295 	}
296 
297 	// XXX: line below doesn't have intended effect (turn off column sorting for GAL search)
298 	if (this._chooser) { //_chooser not defined in ZmGroupView but we also do not support sorting there anyway
299 		this._chooser.sourceListView.sortingEnabled = (this._contactSource == ZmItem.CONTACT);
300 	}
301 
302 	var params = {
303 		obj:			this,
304 		ascending:		ascending,
305 		query:			query,
306 		queryHint:		queryHint.join(" "),
307 		conds:			conds,
308 		offset:			offset || 0,
309 		lastId:			lastId,
310 		lastSortVal:	lastSortVal,
311 		respCallback:	(new AjxCallback(this, this._handleResponseSearch, [firstTime])),
312 		errorCallback:	this._searchErrorCallback,
313 		accountName:	(this._account && this._account.name),
314 		expandDL:		true
315 	};
316 	ZmContactsHelper.search(params);
317 };

319 /**
320  * @private

322 ZmContactPicker.prototype._contentHtml =
323 function(account) {
324 	var showSelect;
325 	if (appCtxt.multiAccounts) {
326 		var list = appCtxt.accountList.visibleAccounts;
327 		for (var i = 0; i < list.length; i++) {
328 			var account = list[i];
329 			if (appCtxt.get(ZmSetting.CONTACTS_ENABLED, null, account) &&
330 				(appCtxt.get(ZmSetting.GAL_ENABLED, null, account) ||
331 				 appCtxt.get(ZmSetting.SHARING_ENABLED, null, account)))
332 			{
333 				showSelect = true;

335 			}
336 		}

339 					  (appCtxt.get(ZmSetting.GAL_ENABLED) ||
340 					   appCtxt.get(ZmSetting.SHARING_ENABLED)));
341 	}
342 
343 	var subs = {
344 		id: this._htmlElId,
345 		showSelect: showSelect,
346 		detailed: this._detailedSearch
347 	};
348 
349 	return (AjxTemplate.expand("abook.Contacts#ZmContactPicker", subs));

352 /**
353  * @private
354  */
355 ZmContactPicker.prototype._resetSelectDiv =
356 function() {
357     this._searchInSelect.clearOptions();
358 
359     if (appCtxt.multiAccounts) {
360         var accts = appCtxt.accountList.visibleAccounts;
361         var org = ZmOrganizer.ITEM_ORGANIZER;
362         org = ZmOrganizer.ITEM_ORGANIZER[ZmItem.CONTACT];
363 
364         for (var i = 0; i < accts.length; i++) {
365             this._searchInSelect.addOption(accts[i].displayName, false, accts[i].id);
366             var folderTree = appCtxt.getFolderTree(accts[i]);
367             var data = [];
368             data = data.concat(folderTree.getByType(org));
369             for (var j = 0; j < data.length; j++) {
370                 var addrsbk = data[j];
371                 if(addrsbk.noSuchFolder) { continue; }
372                 this._searchInSelect.addOption(addrsbk.getName(), false, addrsbk.id, "ImgContact");
373             }
374             if(accts[i].isZimbraAccount && !accts[i].isMain) {
375                 if (appCtxt.get(ZmSetting.CONTACTS_ENABLED, null, this._account)) {
376                     if (appCtxt.get(ZmSetting.SHARING_ENABLED, null, this._account))
377                         this._searchInSelect.addOption(ZmMsg.searchPersonalSharedContacts, false, ZmContactsApp.SEARCHFOR_PAS, "ImgContact");
378                 }
379 
380                 if (appCtxt.get(ZmSetting.GAL_ENABLED, null, this._account)) {
381                     this._searchInSelect.addOption(ZmMsg.GAL, true, ZmContactsApp.SEARCHFOR_GAL, "ImgContact");
382                 }
383 
384                 if (!appCtxt.get(ZmSetting.INITIALLY_SEARCH_GAL, null, this._account) ||
385                         !appCtxt.get(ZmSetting.GAL_ENABLED, null, this._account))
386                 {
387                     this._searchInSelect.setSelectedValue(ZmContactsApp.SEARCHFOR_CONTACTS);
388                 }
389             }
390         }
391 
392         for (var k = 0; k < accts.length; k++) {
393             this._searchInSelect.enableOption(accts[k].id, false);
394         }
395     } else {
396 
397         if (appCtxt.get(ZmSetting.CONTACTS_ENABLED, null, this._account)) {
398             this._searchInSelect.addOption(ZmMsg.contacts, false, ZmContactsApp.SEARCHFOR_CONTACTS);
399 
400             if (appCtxt.get(ZmSetting.SHARING_ENABLED, null, this._account))
401                 this._searchInSelect.addOption(ZmMsg.searchPersonalSharedContacts, false, ZmContactsApp.SEARCHFOR_PAS);
402         }
403 
404         if (appCtxt.get(ZmSetting.GAL_ENABLED, null, this._account)) {
405             this._searchInSelect.addOption(ZmMsg.GAL, true, ZmContactsApp.SEARCHFOR_GAL);
406         }
407 
408         if (!appCtxt.get(ZmSetting.INITIALLY_SEARCH_GAL, null, this._account) ||
409                 !appCtxt.get(ZmSetting.GAL_ENABLED, null, this._account))
410         {
411             this._searchInSelect.setSelectedValue(ZmContactsApp.SEARCHFOR_CONTACTS);
412         }
413 
414     }
415 };

417 ZmContactPicker.prototype.clearSearch =
418 function(el) {
419 	if (el) {
420 		for (var fieldId in this._searchField) {
421 			if (el == this._searchField[fieldId]) {
422 				if (!this._searchCleared[fieldId]) {
423 					el.className = el.value = "";
424 					this._searchCleared[fieldId] = true;
425 				}
426 				break;
427 			}
428 		}
429 	}
430 };

432 ZmContactPicker.prototype.getSearchFieldValue =
433 function(fieldId) {
434 	if (!fieldId && !this._detailedSearch) {
435 		fieldId = ZmContactPicker.SEARCH_BASIC;
436 	}
437 	var field = this._searchField[fieldId];
438 	return (this._searchCleared[fieldId] && field) ? AjxStringUtil.trim(field.value) : "";
439 };

441 /**
442  * called only when ZmContactPicker is first created. Sets up initial layout.
443  * 
444  * @private
445  */
446 ZmContactPicker.prototype._initialize =
447 function(account) {
448 
449 	// create static content and append to dialog parent
450 	this.setContent(this._contentHtml(account));
451 
452 	this._searchIcon = document.getElementById(this._htmlElId + "_searchIcon");
453 
454 	// add search button
455 	this._searchButton = new DwtButton({parent:this, parentElement:(this._htmlElId+"_searchButton")});
456 	this._searchButton.setText(ZmMsg.search);
457 	this._searchButton.addSelectionListener(new AjxListener(this, this._searchButtonListener));
458 
459 	// add select menu
460 	var selectCellId = this._htmlElId + "_listSelect";
461 	var selectCell = document.getElementById(selectCellId);
462 	if (selectCell) {
463 		this._searchInSelect = new DwtSelect({parent:this, parentElement:selectCellId, id: Dwt.getNextId("ZmContactPickerSelect_")});
464 		this._resetSelectDiv();
465 		this._searchInSelect.addChangeListener(new AjxListener(this, this._searchTypeListener));
466 	} else {
467 		this.setSize("600");
468 	}
469 
470 	// add chooser
471 	this._chooser = new ZmContactChooser({parent:this, buttonInfo:this._buttonInfo});
472 	this._chooser.reparentHtmlElement(this._htmlElId + "_chooser");
473 	this._chooser.resize(this.getSize().x-25, ZmContactPicker.CHOOSER_HEIGHT);
474 
475 	// add paging buttons
476 	var pageListener = new AjxListener(this, this._pageListener);
477 	this._prevButton = new DwtButton({parent:this, parentElement:(this._htmlElId+"_pageLeft")});
478 	this._prevButton.setText(ZmMsg.previous);
479 	this._prevButton.setImage("LeftArrow");
480 	this._prevButton.addSelectionListener(pageListener);
481 
482 	this._nextButton = new DwtButton({parent:this, style:DwtLabel.IMAGE_RIGHT, parentElement:(this._htmlElId+"_pageRight")});
483 	this._nextButton.setText(ZmMsg.next);
484 	this._nextButton.setImage("RightArrow");
485 	this._nextButton.addSelectionListener(pageListener);
486 
487 	var pageContainer = document.getElementById(this._htmlElId + "_paging");
488 	if (pageContainer) {
489 		Dwt.setSize(pageContainer, this._chooser.sourceListView.getSize().x);
490 	}
491 
492 	// init listeners
493 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._okButtonListener));
494 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._cancelButtonListener));
495 
496 	var fieldMap = {};
497 	var rowMap = {};
498 	this.mapFields(fieldMap, rowMap);
499 
500 	this._searchField = {};
501 	for (var fieldId in fieldMap) {
502 		var field = Dwt.byId(fieldMap[fieldId]);
503 		if (field) {
504 			this._searchField[fieldId] = field;
505 			Dwt.setHandler(field, DwtEvent.ONKEYUP, ZmContactPicker._keyPressHdlr);
506 			Dwt.setHandler(field, DwtEvent.ONCLICK, ZmContactPicker._onclickHdlr);
507 		}
508 	}
509 
510 	this._searchRow = {};
511 	for (var rowId in rowMap) {
512 		row = Dwt.byId(rowMap[rowId]);
513 		if (row) {
514 			this._searchRow[rowId] = row;
515 		}
516 	}
517 	this._updateSearchRows(this._searchInSelect && this._searchInSelect.getValue() || ZmContactsApp.SEARCHFOR_CONTACTS);
518 	this._keyPressCallback = new AjxCallback(this, this._searchButtonListener);
519 };

521 ZmContactPicker.prototype.mapFields =
522 function(fieldMap, rowMap) {
523 	if (this._detailedSearch) {
524 		fieldMap[ZmContactPicker.SEARCH_NAME] = this._htmlElId + "_searchNameField";
525 		fieldMap[ZmContactPicker.SEARCH_EMAIL] = this._htmlElId + "_searchEmailField";
526 		fieldMap[ZmContactPicker.SEARCH_DEPT] = this._htmlElId + "_searchDepartmentField";
527 		fieldMap[ZmContactPicker.SEARCH_PHONETIC] = this._htmlElId + "_searchPhoneticField";
528 		rowMap[ZmContactPicker.SEARCH_NAME] = this._htmlElId + "_searchNameRow";
529 		rowMap[ZmContactPicker.SEARCH_PHONETIC] = this._htmlElId + "_searchPhoneticRow";
530 		rowMap[ZmContactPicker.SEARCH_EMAIL] = this._htmlElId + "_searchEmailRow";
531 		rowMap[ZmContactPicker.SEARCH_DEPT] = this._htmlElId + "_searchDepartmentRow";
532 	}
533 	else {
534 		fieldMap[ZmContactPicker.SEARCH_BASIC] = this._htmlElId + "_searchField";
535 		rowMap[ZmContactPicker.SEARCH_BASIC] = this._htmlElId + "_searchRow";
536 	}
537 };

550 /**
551  * @private
552  */
553 ZmContactPicker.prototype._handleResponseSearch =
554 function(firstTime, result) {
555 	var resp = result.getResponse();
556 	var serverHasMore = resp.getAttribute("more");
557 	var serverPaginationSupported = resp.getAttribute("paginationSupported") !== false; //if it's not specified (such as the case of SearchResponse, i.e. not Gal) it IS supported.
558 	this._serverHasMoreAndPaginationSupported = serverHasMore && serverPaginationSupported;
559 	var offset = resp.getAttribute("offset");
560 	this._serverContactOffset = offset || 0;
561 	var info = resp.getAttribute("info");
562 	var expanded = info && info[0].wildcard[0].expanded == "0";
563 
564 	//the check for firstTime is so when the picker is popped up we probably don't want to overwhelm them with a warning message. So only show it if the user plays with the picker, using the drop-down or the search box.
565 	if (!firstTime && !serverPaginationSupported && (serverHasMore || expanded)) { //no idea what the expanded case is
566 		var d = appCtxt.getMsgDialog();
567 		d.setMessage(ZmMsg.errorSearchNotExpanded);
568 		d.popup();
569 		if (expanded) { return; }
570 	}
571 
572 	// this method will expand the list depending on the number of email
573 	// addresses per contact.
574 	var emailArray = ZmContactsHelper._processSearchResponse(resp, this._includeContactsWithNoEmail); //this._includeContactsWithNoEmail - true in the ZmGroupView case 
575 	var emailList = AjxVector.fromArray(emailArray);
576 
577 	if (serverPaginationSupported) {
578 		this._emailList.addList(emailArray); //this internally calls concat. we do not need "merge" here because we use the _serverContactOffset as a marker of where to search next, never searching a block we already did.
579 	}
580 	else {
581 		this._emailList = emailList;
582 	}
583 
584 	this._showResults();
585 
586 };

588 ZmContactPicker.prototype._showResults =
589 function() {
590 
591 	var list = this.getSubList();
592 
593 	// special case 1 - search forward another server block, to fill up a page. Could search several times.
594 	if (list.size() < ZmContactsApp.SEARCHFOR_MAX && this._serverHasMoreAndPaginationSupported) {
595 		this.search(null, null, null, null, null, this._serverContactOffset + ZmContactsApp.SEARCHFOR_MAX); //search another page
596 		return;
597 	}
598 
599 	if (this._searchIcon) { //does not exist in ZmGroupView case
600 		this._searchIcon.className = "ImgSearch";
601 	}
602 	this._searchButton.setEnabled(true);
603 
604 	// special case 2 - no results, and no more to search (that was covered in special case 1) - so display the "no results" text.
605 	if (list.size() == 0 && this._emailListOffset == 0) {
606 		this._setResultsInView(list); //empty the list
607 		this._nextButton.setEnabled(false);
608 		this._prevButton.setEnabled(false);
609 		this._setNoResultsHtml();
610 		return;
611 	}
612 
613 	// special case 3 - If the AB ends with a long list of contacts w/o addresses,
614 	// we may get an empty list.  If that's the case, roll back the offset
615 	// not 100% sure this case could still happen after all my changes but it was there in the code, so I keep it just in case.
616 	if (list.size() == 0) {
617 		this._emailListOffset -= ZmContactsApp.SEARCHFOR_MAX;
618 		this._emailListOffset  = Math.max(0, this._emailListOffset);
619 	}
620 
621 	var more = this._serverHasMoreAndPaginationSupported  //we can get more from the server
622 				|| (this._emailListOffset + ZmContactsApp.SEARCHFOR_MAX) < this._emailList.size(); //or we have more on the client we didn't yet show
623 	this._prevButton.setEnabled(this._emailListOffset > 0);
624 	this._nextButton.setEnabled(more);
625 
626 	this._resetSearchColHeaders(); // bug #2269 - enable/disable sort column per type of search
627 	this._setResultsInView(list);
628 
629 };

659 			this._tabGroup.removeMember(field);
660 	}

696 function() {
