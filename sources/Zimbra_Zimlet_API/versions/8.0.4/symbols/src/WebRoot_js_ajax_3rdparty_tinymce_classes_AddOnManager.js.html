 27 (function(tinymce) {
 28 	var Dispatcher = tinymce.util.Dispatcher, each = tinymce.each;
 29 
 30 	/**
 31 	 * This class handles the loading of themes/plugins or other add-ons and their language packs.
 32 	 *
 33 	 * @class tinymce.AddOnManager
 34 	 */
 35 	tinymce.create('tinymce.AddOnManager', {
 36 		AddOnManager : function() {
 37 			var self = this;
 38 
 39 			self.items = [];
 40 			self.urls = {};
 41 			self.lookup = {};
 42 			self.onAdd = new Dispatcher(self);
 43 		},
 44 
 45 		/**
 46 		 * Fires when a item is added.
 47 		 *
 48 		 * @event onAdd
 49 		 */
 50 
 51 		/**
 52 		 * Returns the specified add on by the short name.
 53 		 *
 54 		 * @method get
 55 		 * @param {String} n Add-on to look for.
 56 		 * @return {tinymce.Theme/tinymce.Plugin} Theme or plugin add-on instance or undefined.
 57 		 */
 58 		get : function(n) {
 59 			if (this.lookup[n]) {
 60 				return this.lookup[n].instance;
 61 			} else {
 62 				return undefined;
 63 			}
 64 		},
 65 
 66 		dependencies : function(n) {
 67 			var result;
 68 			if (this.lookup[n]) {
 69 				result = this.lookup[n].dependencies;
 70 			}
 71 			return result || [];
 72 		},
 73 
 74 		/**
 75 		 * Loads a language pack for the specified add-on.
 76 		 *
 77 		 * @method requireLangPack
 78 		 * @param {String} n Short name of the add-on.
 79 		 */
 80 		requireLangPack : function(n) {
 81 			var s = tinymce.settings;
 82 
 83 			if (s && s.language && s.language_load !== false)
 84 				tinymce.ScriptLoader.add(this.urls[n] + '/langs/' + s.language + '.js');
 85 		},
 86 
 87 		/**
 88 		 * Adds a instance of the add-on by it's short name.
 89 		 *
 90 		 * @method add
 91 		 * @param {String} id Short name/id for the add-on.
 92 		 * @param {tinymce.Theme/tinymce.Plugin} o Theme or plugin to add.
 93 		 * @return {tinymce.Theme/tinymce.Plugin} The same theme or plugin instance that got passed in.
 94 		 * @example
 95 		 * // Create a simple plugin
 96 		 * tinymce.create('tinymce.plugins.TestPlugin', {
 97 		 *     TestPlugin : function(ed, url) {
 98 		 *         ed.onClick.add(function(ed, e) {
 99 		 *             ed.windowManager.alert('Hello World!');
100 		 *         });
101 		 *     }
102 		 * });
103 		 * 
104 		 * // Register plugin using the add method
105 		 * tinymce.PluginManager.add('test', tinymce.plugins.TestPlugin);
106 		 * 
107 		 * // Initialize TinyMCE
108 		 * tinyMCE.init({
109 		 *    ...
110 		 *    plugins : '-test' // Init the plugin but don't try to load it
111 		 * });
112 		 */
113 		add : function(id, o, dependencies) {
114 			this.items.push(o);
115 			this.lookup[id] = {instance:o, dependencies:dependencies};
116 			this.onAdd.dispatch(this, id, o);
117 
118 			return o;
119 		},
120 		createUrl: function(baseUrl, dep) {
121 			if (typeof dep === "object") {
122 				return dep
123 			} else {
124 				return {prefix: baseUrl.prefix, resource: dep, suffix: baseUrl.suffix};
125 			}
126 		},
127 
128 		/**
129 		 * Add a set of components that will make up the add-on. Using the url of the add-on name as the base url.
130 		 * This should be used in development mode.  A new compressor/javascript munger process will ensure that the 
131 		 * components are put together into the editor_plugin.js file and compressed correctly.
132 		 * @param pluginName {String} name of the plugin to load scripts from (will be used to get the base url for the plugins).
133 		 * @param scripts {Array} Array containing the names of the scripts to load.
134 		 */
135 		addComponents: function(pluginName, scripts) {
136 			var pluginUrl = this.urls[pluginName];
137 			tinymce.each(scripts, function(script){
138 				tinymce.ScriptLoader.add(pluginUrl+"/"+script);	
139 			});
140 		},
141 
142 		/**
143 		 * Loads an add-on from a specific url.
144 		 *
145 		 * @method load
146 		 * @param {String} n Short name of the add-on that gets loaded.
147 		 * @param {String} u URL to the add-on that will get loaded.
148 		 * @param {function} cb Optional callback to execute ones the add-on is loaded.
149 		 * @param {Object} s Optional scope to execute the callback in.
150 		 * @example
151 		 * // Loads a plugin from an external URL
152 		 * tinymce.PluginManager.load('myplugin', '/some/dir/someplugin/editor_plugin.js');
153 		 *
154 		 * // Initialize TinyMCE
155 		 * tinyMCE.init({
156 		 *    ...
157 		 *    plugins : '-myplugin' // Don't try to load it again
158 		 * });
159 		 */
160 		load : function(n, u, cb, s) {
161 			var t = this, url = u;
162 
163 			function loadDependencies() {
164 				var dependencies = t.dependencies(n);
165 				tinymce.each(dependencies, function(dep) {
166 					var newUrl = t.createUrl(u, dep);
167 					t.load(newUrl.resource, newUrl, undefined, undefined);
168 				});
169 				if (cb) {
170 					if (s) {
171 						cb.call(s);
172 					} else {
173 						cb.call(tinymce.ScriptLoader);
174 					}
175 				}
176 			}
177 
178 			if (t.urls[n])
179 				return;
180 			if (typeof u === "object")
181 				url = u.prefix + u.resource + u.suffix;
182 
183 			if (url.indexOf('/') !== 0 && url.indexOf('://') == -1)
184 				url = tinymce.baseURL + '/' + url;
185 
186 			t.urls[n] = url.substring(0, url.lastIndexOf('/'));
187 
188 			if (t.lookup[n]) {
189 				loadDependencies();
190 			} else {
191 				tinymce.ScriptLoader.add(url, loadDependencies, s);
192 			}
193 		}
194 	});
195 
196 	// Create plugin and theme managers
197 	tinymce.PluginManager = new tinymce.AddOnManager();
198 	tinymce.ThemeManager = new tinymce.AddOnManager();
199 }(tinymce));
