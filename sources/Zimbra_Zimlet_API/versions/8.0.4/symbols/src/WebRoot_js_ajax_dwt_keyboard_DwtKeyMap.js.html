 17 /**
 18  * Creates and loads a key map.
 19  * @constructor
 20  * @class
 21  * This class provides the basic keyboard mappings for {@link Dwt} components. The
 22  * key bindings are taken from the class AjxKeys, which is populated from a
 23  * properties file. The identifiers used in the properties file must match
 24  * those used here.
 25  * 
 26  * @author Ross Dargahi
 27  * 
 28  * @param	{boolean}		subclassInit		if <code>true</code>, the sub-class will initialize
 29  * 
 30  */
 31 DwtKeyMap = function(subclassInit) {
 32 	if (subclassInit) {	return };
 33 
 34 	this._map			= {};
 35 	this._args			= {};
 36 	this._checkedMap	= {};	// cache results of _checkMap()
 37 
 38 	this._load(this._map, AjxKeys);
 39 
 40 	DwtKeyMap.MOD_ORDER[DwtKeyMap.ALT]		= 1;
 41 	DwtKeyMap.MOD_ORDER[DwtKeyMap.CTRL]		= 2;
 42 	DwtKeyMap.MOD_ORDER[DwtKeyMap.META]		= 3;
 43 	DwtKeyMap.MOD_ORDER[DwtKeyMap.SHIFT]	= 4;
 44 };

287 /**
288  * Converts a properties representation of shortcuts into a hash. The
289  * properties version is actually a reverse map of what we want, so we
290  * have to swap keys and values. Handles platform-specific shortcuts,
291  * and inheritance. The properties version is made available via a
292  * servlet.
293  * 
294  * @param {hash}	map			the hash to populate with shortcuts
295  * @param {hash}	keys			the properties version of shortcuts
296  * 
297  * @private
298  */
299 DwtKeyMap.prototype._load =
300 function(map, keys) {
301 
302 	// preprocess for platform-specific bindings
303 	var curPlatform = AjxEnv.platform.toLowerCase();
304 	for (var propName in keys) {
305 		var parts = propName.split(".");
306 		var last = parts[parts.length - 1];
307 		if (last == "win" || last == "mac" || last == "linux") {
308 			if (last == curPlatform) {
309 				var baseKey = parts.slice(0, parts.length - 1).join(".");
310 				keys[baseKey] = keys[propName];
311 			}
312 			keys[propName] = null;
313 		}
314 	}
315 	
316 	for (var propName in keys) {
317 		var propValue = AjxStringUtil.trim(keys[propName]);
318 		if (!propValue || (typeof keys[propName] != "string")) { continue; }
319 		var parts = propName.split(".");
320 		var field = parts[parts.length - 1];
321 		var isMap = (parts.length == 2);
322 		var action = isMap ? null : parts[1];
323 		if (parts[0] == "keys") {
324 			this._processKeyDef(action, field, propValue);
325 			continue;
326 		}
327 		if (field != DwtKeyMap.INHERIT && field != "keycode") { continue; }
328 		var mapName = parts[0];
329 		if ((this._checkedMap[mapName] === false) ||
330 			(!this._checkedMap[mapName] && !this._checkMap(mapName))) { continue; }
331 		if (!map[mapName]) {
332 			map[mapName] = {};
333 		}
334 		if (!this._checkAction(mapName, action)) { continue; }
335 		var keySequences = propValue.split(/\s*;\s*/);
336 		for (var i = 0; i < keySequences.length; i++) {
337 			var ks = this._canonicalize(keySequences[i]);
338 			if (field == DwtKeyMap.INHERIT) {
339 				var parents = ks.split(/\s*,\s*/);
340 				var parents1 = [];
341 				for (var p = 0; p < parents.length; p++) {
342 					parents1[p] = parents[p];
343 				}
344 				map[mapName][parts[1]] = parents1.join(",");
345 			} else if (field == "keycode") {
346 				map[mapName][ks] = action;
347 			}
348 		}
349 	}
350 };

381 /**
382  * Sets up constants for a modifier key as described in a properties file.
383  * 
384  * @param {string}	key		the ctrl, alt, shift, or meta
385  * @param {string}	field	the display or keycode
386  * @param {string|number}	value	the property value
387  * 
388  * @private
389  */
390 DwtKeyMap.prototype._processKeyDef = 
391 function(key, field, value) {
392 	if (!key || !field || !value) { return; }
393 	if (field == "display") {
394 		DwtKeyMap[key.toUpperCase()] = value;
395 	} else if (field == "keycode") {
396 		DwtKeyMap.IS_MODIFIER[value] = true;
397 	}
398 };

400 /**
401  * Ensures a predictable order for the modifiers in a key sequence:
402  * <pre>
403  * Alt Ctrl Meta Shift
404  * </pre>
405  * 
406  * Example: "Shift+Ctrl+U" will be transformed into "Ctrl+Shift+U"
407  * 
408  * @param {String}	ks	the key sequence
409  * 
410  * @private
411  */
412 DwtKeyMap.prototype._canonicalize =
413 function(ks) {
414 	var keys = ks.split(DwtKeyMap.SEP);
415 	var result = [];
416 	for (var i = 0; i < keys.length; i++) {
417 		var key = keys[i];
418 		var parts = key.split(DwtKeyMap.JOIN);
419 		if (parts.length > 2) {
420 			var mods = parts.slice(0, parts.length - 1);
421 			mods.sort(function(a, b) {
422 				var sortA = DwtKeyMap.MOD_ORDER[a] || 0;
423 				var sortB = DwtKeyMap.MOD_ORDER[b] || 0;
424 				return Number(sortA - sortB);
425 			});
426 			mods.push(parts[parts.length - 1]);
427 			result.push(mods.join(DwtKeyMap.JOIN));
428 		} else {
429 			result.push(key);
430 		}
431 	}
432 	return result.join(",");
433 };
