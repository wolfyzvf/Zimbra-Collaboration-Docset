 34 ZmApptList.prototype.loadFromSummaryJs =
 35 function(appts) {
 36 	if (!appts) { return; }
 37 
 38 	for (var i = 0; i < appts.length; i++) {
 39 		var apptNode = appts[i];
 40 		var instances = apptNode ? apptNode.inst : null;
 41 		if (instances) {
 42 			var args = {list:this};
 43 			for (var j = 0; j < instances.length; j++) {
 44 				var appt = ZmAppt.createFromDom(apptNode, args, instances[j]);
 45 				if (appt) this.add(appt);
 46 			}
 47 		}
 48 	}
 49 }

 61 /**
 62  * Merges all the sorted vectors in the specified array into a single sorted vector.
 63  * 
 64  * @param	{AjxVector}	vecArray		the array
 65  * @return	{AjxVector}	the resulting array
 66  */
 67 ZmApptList.mergeVectors = 
 68 function(vecArray) {
 69 	var result = new AjxVector();	
 70 	if(!vecArray) {  return result; }
 71 
 72 	// clone the single array case!
 73 	if (vecArray.length == 1) return vecArray[0].clone();
 74 	for (var i=0; i < vecArray.length; i++) result.addList(vecArray[i]);
 75 	ZmApptList.sortVector(result);
 76 	return result;
 77 };

 79 ZmApptList.toVector =
 80 function(apptList, startTime, endTime, fanoutAllDay, includeReminders) {
 81 	var result = new AjxVector();
 82 	var list = apptList.getVector();
 83 	var size = list.size();
 84 	for (var i = 0; i < size; i++) {
 85 		var ao = list.get(i);
 86 		var folder = ao.getFolder();
 87 		if (ao.isInRange(startTime, endTime) || (ao.isAlarmInRange() && includeReminders)) {
 88 			if (ao.isAllDayEvent() && !fanoutAllDay) {
 89 				result.add(ZmAppt.quickClone(ao));
 90 			} else {
 91 				ZmApptList._fanout(ao, result, startTime, endTime, fanoutAllDay, includeReminders);
 92 			}
 93 		}
 94 	}
 95 	ZmApptList.sortVector(result);
 96 	return result;
 97 };

 99 /**
100  * fanout multi-day appoints into multiple single day appts. This has nothing to do with recurrence...
101  * TODO: should be more efficient by not fanning out appts in if part of the while if they are not in the range.
102  * 
103  * @private
104  */
105 ZmApptList._fanout =
106 function(orig, result, startTime, endTime, fanoutAllDay, includeReminders) {
107 	var appt = ZmAppt.quickClone(orig);
108 	var fanoutNum = 0;
109 
110 	// HACK: Avoid "strange" appt durations that occur at transition
111 	//       days for timezones w/ DST. For example, going from DST to
112 	//       STD, the duration for a single day is 25 hours; while the
113 	//       transition from STD to DST, the duration is 23 hours. So
114 	//       we advance 12 hours (just to be safe) and then subtract
115 	//       off the extra hours.
116 	var origEndTime = orig.getEndTime();
117 	if (appt.isAllDayEvent()) {
118 		var origEndDate = new Date(origEndTime);
119 		origEndDate.setHours(0, 0, 0, 0);
120 
121 		appt.setEndDate(origEndDate);
122 		origEndTime = origEndDate.getTime();
123 	}
124 
125     //while (appt.isInRange(startTime,endTime) || (appt.isAlarmInRange() && includeReminders)) {
126     while (orig.isInRange(startTime,endTime) || (appt.isAlarmInRange() && includeReminders)) {
127 		if (appt.isMultiDay()) {
128 			var apptStartTime = appt.getStartTime();
129 			// bug 12205: If someone mistypes "2007" as "200", we get into
130 			//            a seemingly never-ending loop trying to fanout
131 			//            every day even *before* the startTime of the view.
132 			var nextDay = new Date(apptStartTime);
133 			nextDay.setDate(nextDay.getDate()+1);
134 			if (origEndTime < nextDay.getTime()) {
135 				nextDay = new Date(origEndTime);
136 			}
137 			nextDay.setHours(0,0,0,0);
138             if(AjxDateUtil.isDayShifted(nextDay)) {
139                 AjxDateUtil.rollToNextDay(nextDay);
140             }
141 
142             var slice = ZmAppt.quickClone(appt);
143             slice._fanoutFirst = (fanoutNum == 0);
144             slice._orig = orig;
145             slice.setEndDate(nextDay);
146             slice._fanoutLast = (slice.getEndTime() == origEndTime);
147             slice._fanoutNum = fanoutNum;
148             slice.uniqStartTime = slice.getStartTime();					// need to construct uniq id later
149             result.add(slice);
150 
151 			fanoutNum++;
152 			appt.setStartDate(nextDay);
153 			if (appt.getStartTime() >= appt.getEndTime())
154 				break;
155 		} else {
156 			if (orig.isInRange(startTime,endTime)  || (appt.isAlarmInRange() && includeReminders) ) {
157 				appt._fanoutFirst = fanoutNum == 0;
158 				appt._fanoutLast = appt.getEndTime() == origEndTime;
159 				if (!appt._fanoutFirst)
160 					appt._orig = orig;
161 				appt._fanoutNum = fanoutNum;
162 				appt.uniqStartTime = appt.getStartTime();						// need to construct uniq id later
163 				result.add(appt);
164 			}
165 			break;
166 		}
167 	}
168 };

170 /**
171  * Gets a new appointment list containing only appointment in the given range.
172  * 
173  * @param	{Date}	startTime		the start time
174  * @param	{Date}	endTime		the end time
175  * @author {ZmApptList}	the new list
176  */
177 ZmApptList.prototype.getSubset =
178 function(startTime, endTime) {
179 	var result  = new ZmApptList();
180 	var list = this.getVector();
181 	var size = list.size();
182 	for (var i=0; i < size; i++) {
183 		var ao = list.get(i);
184 		if (ao.isInRange(startTime, endTime)) {
185 			result.add(ao);
186 		}
187 	}
188 	return result;
189 };

192 /**
193  * Moves a list of items to the given folder.
194  * <p>
195  * Search results are treated as though they're in a temporary folder, so that they behave as
196  * they would if they were in any other folder such as Inbox. When items that are part of search
197  * results are moved, they will disappear from the view, even though they may still satisfy the
198  * search.
199  * </p>
200  *
201  * @param	{Hash}			params					a hash of parameters
202  * @param	{Array}			params.items			a list of items to move
203  * @param	{ZmFolder}		params.folder			the destination folder
204  * @param	{Hash}			params.attrs			the additional attrs for SOAP command
205  * @param	{AjxCallback}	params.callback			the callback to run after each sub-request
206  * @param	{closure}		params.finalCallback	the callback to run after all items have been processed
207  * @param	{int}			params.count			the starting count for number of items processed
208  * @param	{boolean}		params.noUndo			true if the action is not undoable (e.g. performed as an undo)
209  * @param	{String}		params.actionText		optional text to display in the confirmation toast instead of the default summary. May be set explicitly to null to disable the confirmation toast entirely
210  */
211 ZmApptList.prototype.moveItems =
212 function(params) {
213 	params = Dwt.getParams(arguments, ["items", "folder", "attrs", "callback", "errorCallback" ,"finalCallback", "noUndo", "actionText"]);
214 
215 	var params1 = AjxUtil.hashCopy(params);
216 	params1.items = AjxUtil.toArray(params.items);
217 	params1.attrs = params.attrs || {};
218 	if (params1.folder.id == ZmFolder.ID_TRASH) {
219 		params1.actionText = (params.actionText !== null) ? (params.actionText || ZmMsg.actionTrash) : null;
220 		params1.action = "trash";
221         //This code snippet differs from the ZmList.moveItems
222         var currentView = appCtxt.getCurrentView();
223         if(currentView) {
224             var viewController = currentView.getController();
225             if(viewController) {
226                 //Since it is a drag and drop, only one item can be dragged - so get the first element from array
227                 return viewController._deleteAppointment(params1.items[0]);
228             }
229         }
230 	} else {
231 		params1.actionText = (params.actionText !== null) ? (params.actionText || ZmMsg.actionMove) : null;
232 		params1.actionArg = params.folder.getName(false, false, true);
233 		params1.action = "move";
234 		params1.attrs.l = params.folder.id;
235 	}
236 
237     if (appCtxt.multiAccounts) {
238 		// Reset accountName for multi-account to be the respective account if we're
239 		// moving a draft out of Trash.
240 		// OR,
241 		// check if we're moving to or from a shared folder, in which case, always send
242 		// request on-behalf-of the account the item originally belongs to.
243         var folderId = params.items[0].getFolderId();
244         var fromFolder = appCtxt.getById(folderId);
245 		if ((params.items[0].isDraft && params.folder.id == ZmFolder.ID_DRAFTS) ||
246 			(params.folder.isRemote()) || (fromFolder.isRemote()))
247 		{
248 			params1.accountName = params.items[0].getAccount().name;
249 		}
250 	}
251 
252     //Error Callback
253     params1.errorCallback = params.errorCallback;
254 
255 	this._itemAction(params1);
256 };
