 16 /**
 17  * Creates a filter rules object.
 18  * @constructor
 19  * @class
 20  * This class represents a set of filter rules. The rules are maintained in a {@link AjxVector}
 21  * and have an order. Each rule is a {@link ZmFilterRule}. Filter rules can be added and
 22  * edited via a {@link ZmFilterRuleDialog}.
 23  *
 24  * @author Conrad Damon
 25  *
 26  * @param {String}	accountName		the name of the account this set of filter rules belongs to
 27  * 
 28  * @extends		ZmModel
 29  */
 30 ZmFilterRules = function(accountName, outgoing) {
 31 
 32 	ZmModel.call(this, ZmEvent.S_FILTER);
 33 
 34 	this._vector = new AjxVector();
 35 	this._ruleIdHash = {};
 36 	this._ruleNameHash = {};
 37 	this._initialized = false;
 38 	this._accountName = accountName;
 39 	this._outgoing = outgoing;
 40 };
 41 
 42 ZmFilterRules.prototype = new ZmModel;
 43 ZmFilterRules.prototype.constructor = ZmFilterRules;
 44 
 45 ZmFilterRules.prototype.toString =
 46 function() {
 47 	return "ZmFilterRules";
 48 };
 49 
 50 /**
 51  * Adds a rule to the list.
 52  *
 53  * @param {ZmFilterRule}	rule			the rule to be added
 54  * @param {ZmFilterRule}	referenceRule	the rule after which to add the new rule
 55  * @param {AjxCallback}	callback		the callback
 56  */
 57 ZmFilterRules.prototype.addRule = 
 58 function(rule, referenceRule, callback) {
 59 	DBG.println(AjxDebug.DBG3, "FILTER RULES: add rule '" + rule.name + "'");
 60 	var index = referenceRule ? this._vector.indexOf(referenceRule) : 0;
 61 	this._insertRule(rule, index);
 62 	this._saveRules(index, true, callback);
 63 };
 64 
 65 /**
 66  * Removes a rule from the list.
 67  *
 68  * @param {ZmFilterRule}	rule			the rule to be removed
 69  */
 70 ZmFilterRules.prototype.removeRule = 
 71 function(rule) {
 72 	if (!rule) { return; }
 73 	DBG.println(AjxDebug.DBG3, "FILTER RULES: remove rule '" + rule.name + "'");
 74 	var index = this.getIndexOfRule(rule);
 75 	this._vector.removeAt(index);
 76 	delete this._ruleIdHash[rule.id];
 77 	delete this._ruleNameHash[rule.name];
 78 	this._deleteMode = true;
 79     this._saveRules(index, true);
 80 };
 81 
 82 /**
 83  * Moves a rule up in the list. If the rule is the first in the list, it isn't moved.
 84  *
 85  * @param {ZmFilterRule}	rule			the rule to be moved
 86  */
 87 ZmFilterRules.prototype.moveUp = 
 88 function(rule) {
 89 	if (!rule) { return; }
 90 	DBG.println(AjxDebug.DBG3, "FILTER RULES: move up rule '" + rule.name + "'");
 91 	var index = this.getIndexOfRule(rule);
 92 	if (index == 0) { return; }
 93 
 94 	var prevRule = this._vector.removeAt(index - 1);
 95 	this._insertRule(prevRule, index);
 96 	this._saveRules(index - 1, true);
 97 };
 98 
 99 /**
100  * Moves a rule down in the list. If the rule is the last in the list, it isn't moved.
101  *
102  * @param {ZmFilterRule}	rule			the rule to be moved
103  */
104 ZmFilterRules.prototype.moveDown = 
105 function(rule) {
106 	if (!rule) { return; }
107 	DBG.println(AjxDebug.DBG3, "FILTER RULES: move down rule '" + rule.name + "'");
108 	var index = this.getIndexOfRule(rule);
109 	if (index >= (this._vector.size() - 1)) { return; }
110 	
111 	var nextRule = this._vector.removeAt(index + 1);
112 	this._insertRule(nextRule, index);
113 	this._saveRules(index + 1, true);
114 };
115 
116 /**
117  * Moves a rule to the bottom of the list.  If the rule is the last in the list, it isn't moved.
118  * @param rule  {ZmFilterRule}  rule    the rule to be moved
119  * @param skipSave  {boolean}   true to not save
120  */
121 ZmFilterRules.prototype.moveToBottom = 
122 function(rule, skipSave) {
123 	if (!rule) { return; }
124 	var index = this.getIndexOfRule(rule);
125 	if (index >= (this._vector.size() - 1)) { return; }
126 	
127 	while (index < this._vector.size() -1) {
128 		var nextRule = this._vector.removeAt(index+1);
129 		this._insertRule(nextRule, index);
130 		index++;
131 	}
132 	if (!skipSave) {
133 		this._saveRules(index, true);
134 	}
135 };

139  *
140  * @param {ZmFilterRule}	rule			the rule to mark active/inactive
141  * @param {Boolean}	active			if <code>true</code>, the rule is marked active

143 ZmFilterRules.prototype.setActive =
144 function(rule, active) {
145 	if (!rule) { return; }
146 	DBG.println(AjxDebug.DBG3, "FILTER RULES: set active rule '" + rule.name + "', " + active);
147 	rule.active = active;
148 	this._saveRules(null, false);
149 };

222 /**
223  * Loads the rules from the server.
224  *
225  * @param {Boolean}	force			if <code>true</code>, get rules from server
226  * @param {AjxCallback}	callback		the callback
227  */
228 ZmFilterRules.prototype.loadRules = 
229 function(force, callback) {
230 	// return cache?
231 	if (this._initialized && !force) {
232 		if (callback) {
233 			callback.run(new ZmCsfeResult(this._vector));
234 			return;
235 		}
236 		return this._vector;
237 	}
238 
239 	// fetch from server:
240 	DBG.println(AjxDebug.DBG3, "FILTER RULES: load rules");
241 	var params = {
242 		soapDoc: AjxSoapDoc.create(this._outgoing ? "GetOutgoingFilterRulesRequest" : "GetFilterRulesRequest", "urn:zimbraMail"),
243 		asyncMode: true,
244 		callback: (new AjxCallback(this, this._handleResponseLoadRules, [callback])),
245 		accountName:this._accountName
246 	};
247 	appCtxt.getAppController().sendRequest(params);
248 };
249 
250 ZmFilterRules.prototype._handleResponseLoadRules =
251 function(callback, result) {
252 	this._vector.removeAll();
253 	this._ruleIdHash = {};
254 	this._ruleNameHash = {};
255 
256 	var r = result.getResponse();
257 	var resp = this._outgoing ? r.GetOutgoingFilterRulesResponse : r.GetFilterRulesResponse;
258 	var children = resp.filterRules[0].filterRule;
259 	if (children) {
260 		for (var i = 0; i < children.length; i++) {
261 			var ruleNode = children[i];
262 			var rule = new ZmFilterRule(ruleNode.name, ruleNode.active, ruleNode.filterActions[0], ruleNode.filterTests[0]);
263 			this._insertRule(rule);
264 		}
265 	}
266 
267 	this._initialized = true;
268 
269 	if (callback) {
270 		result.set(this._vector);
271 		callback.run(result);
272 	} else {
273 		return this._vector;
274 	}
275 };

291 /**
292  * Saves the rules to the server.
293  *
294  * @param {int}	index			the index of rule to select in list after save
295  * @param {Boolean}	notify			if <code>true</code>, notify listeners of change event
296  * @param {AjxCallback}	callback		the callback
297  * 
298  * @private
299  */
300 ZmFilterRules.prototype._saveRules = 
301 function(index, notify, callback) {
302 	var requestKey = this._outgoing ? "ModifyOutgoingFilterRulesRequest" : "ModifyFilterRulesRequest";
303 	var jsonObj = {};
304 	jsonObj[requestKey] = {_jsns:"urn:zimbraMail"};
305 
306 	var request = jsonObj[requestKey];
307 
308 	var rules = this._vector.getArray();
309 	if (rules.length > 0) {
310 		request.filterRules = [{filterRule:[]}];
311 		var filterRuleObj = request.filterRules[0].filterRule;
312 
313 		for (var i = 0; i < rules.length; i++) {
314 			var r = rules[i];
315 			var ruleObj = {
316 				active: r.active,
317 				name: r.name,
318 				filterActions: [],
319 				filterTests: []
320 			};
321 			ruleObj.filterActions.push(r.actions);
322 			ruleObj.filterTests.push(r.conditions);
323 			filterRuleObj.push(ruleObj);
324 		}
325 	} else {
326 		request.filterRules = {};
327 	}
328 
329 	var params = {
330 		jsonObj: jsonObj,
331 		asyncMode: true,
332 		callback: (new AjxCallback(this, this._handleResponseSaveRules, [index, notify, callback])),
333 		errorCallback: (new AjxCallback(this, this._handleErrorSaveRules)),
334 		accountName: this._accountName
335 	};
336 	appCtxt.getAppController().sendRequest(params);
337 };
338 
339 ZmFilterRules.prototype._handleResponseSaveRules =
340 function(index, notify, callback, result) {
341 	if (notify) {
342 		this._notify(ZmEvent.E_MODIFY, {index: index});
343 	}
344 
345     if(this._deleteMode){
346         appCtxt.setStatusMsg(ZmMsg.filtersDeleted);
347         this._deleteMode = false;
348     }
349 
350     else if(appCtxt.getFilterRuleDialog().isEditMode()){
351 	    appCtxt.setStatusMsg(ZmMsg.filtersEdited);
352     }
353     else{
354         appCtxt.setStatusMsg(ZmMsg.filtersSaved);
355     }
356 
357 	if (callback) {
358 		callback.run(result);
359 	}
360 };

362 /**
363  * The save failed. Show an error dialog.
364  * @param {AjxException}	ex		the exception
365  * 
366  * @private
367  */
368 ZmFilterRules.prototype._handleErrorSaveRules =
369 function(ex) {
370 	if (ex.code == ZmCsfeException.SVC_PARSE_ERROR ||
371 		ex.code == ZmCsfeException.SVC_INVALID_REQUEST)
372 	{
373 		var msgDialog = appCtxt.getMsgDialog();
374 		msgDialog.setMessage([ZmMsg.filterError, " ", AjxStringUtil.htmlEncode(ex.msg)].join(""), DwtMessageDialog.CRITICAL_STYLE);
375 		msgDialog.popup();
376         //only reload rules if the filter rule dialog is not popped up or if a new rule is being added
377         //get index for refreshing list view
378         if (!appCtxt.getFilterRuleDialog() || !appCtxt.getFilterRuleDialog().isPoppedUp() || !appCtxt.getFilterRuleDialog().isEditMode()) {
379             var filterRulesController = ZmPreferencesApp.getFilterRulesController(this._outgoing);
380             var sel = filterRulesController.getListView() ? filterRulesController.getListView().getSelection()[0] : null;
381             var index = sel ? this.getIndexOfRule(sel) - 1: null;   //new filter is inserted into vector, subtract 1 to get the selected index
382 		    var respCallback = new AjxCallback(this, this._handleResponseHandleErrorSaveRules, [index]);
383 		    this.loadRules(true, respCallback);
384         }
385 		return true;
386 	}
387 	return false;
388 };
389 
390 // XXX: the caller should probably be the one doing this
391 ZmFilterRules.prototype._handleResponseHandleErrorSaveRules =
392 function(index) {
393 	var prefController = AjxDispatcher.run("GetPrefController");
394 	var prefsView = prefController.getPrefsView();
395 	var section = ZmPref.getPrefSectionWithPref(ZmSetting.FILTERS);
396 	if (section && prefsView && prefsView.getView(section.id)) {
397         var filterRulesController = ZmPreferencesApp.getFilterRulesController(this._outgoing);
398 		filterRulesController.resetListView(index);
399 	}
400 };
