 17  * @overview
 18  * This file contains the exception class.
 19  */
 20 /**
 21  * Creates an exception.
 22  * @class
 23  * This class represents an exception returned by the server as a response, generally as a fault. The fault
 24  * data is converted to properties of the exception.
 25  *
 26  * @param {Hash}	params	a hash of parameters
 27  * @param {String}      params.msg		the explanation (Fault.Reason.Text)
 28  * @param {String}      params.code		the error code (Fault.Detail.Error.Code)
 29  * @param {String}      params.method	the request name
 30  * @param {String}      params.detail	the Fault.Code.Value
 31  * @param {Object}      [params.data]		an optional structured fault data (Fault.Detail.Error.a)
 32  * @param {String}      params.trace		the trace info (Fault.Detail.Error.Trace)
 33  * @param {String}       params.request	the SOAP or JSON that represents the request
 34  * 
 35  * @extends		AjxException
 36  */
 37 ZmCsfeException = function(params) {
 38 
 39 	params = Dwt.getParams(arguments, ZmCsfeException.PARAMS);
 40 
 41 	AjxException.call(this, params.msg, params.code, params.method, params.detail);
 42 	
 43 	if (params.data) {
 44 		this.data = {};
 45 		for (var i = 0; i < params.data.length; i++) {
 46 			var item = params.data[i];
 47 			var key = item.n;
 48 			if (!this.data[key]) {
 49 				this.data[key] = [];
 50 			}
 51 			this.data[key].push(item._content);
 52 		}
 53 	}
 54 	
 55 	this.trace = params.trace;
 56 	this.request = params.request;
 57 };
 58 
 59 ZmCsfeException.PARAMS = ["msg", "code", "method", "detail", "data", "trace"];
 60 
 61 ZmCsfeException.prototype = new AjxException;
 62 ZmCsfeException.prototype.constructor = ZmCsfeException;
 63 ZmCsfeException.prototype.isZmCsfeException = true;
 64

 66  * Returns a string representation of the object.

 70 ZmCsfeException.prototype.toString =

 72 	return "ZmCsfeException";
 73 };

 87 /**
 88  * Gets the error messages.
 89  * 
 90  * @param	{String}	code	the code
 91  * @param	{Array}	args		the message format args
 92  * 
 93  * @return	{String}	the message
 94  */
 95 ZmCsfeException.getErrorMsg =
 96 function(code, args) {
 97 	var msg = ZMsg[code];
 98 	if (!msg) {
 99 		ZmCsfeException._unknownFormat = ZmCsfeException._unknownFormat || new AjxMessageFormat(ZMsg.unknownError);
100 		return ZmCsfeException._unknownFormat.format(code);
101 	}
102 	this.msg = this.msg || msg;
103 	return args ? AjxMessageFormat.format(msg, args) : msg;
104 };
105 
106 //
107 // Public methods
108 //
109 
110 /**
111  * Gets the error message.
112  * 
113  * @param	{Array}	args		the message format args
114  * @return	{String}	the message
115  */
116 ZmCsfeException.prototype.getErrorMsg =
117 function(args) {
118 	return ZmCsfeException.getErrorMsg(this.code, args);
119 };
120 
121 /**
122  * Gets the data.
123  * 
124  * @param	{Object}	key		the key
125  * 
126  * @return	{Object}	the data
127  */
128 ZmCsfeException.prototype.getData =
129 function(key) {
130 	return this.data && this.data[key];
131 };
