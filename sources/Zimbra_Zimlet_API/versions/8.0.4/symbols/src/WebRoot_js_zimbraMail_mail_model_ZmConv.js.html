 16 /**
 17  * Creates a conversation.
 18  * @constructor
 19  * @class
 20  * This class represents a conversation, which is a collection of mail messages
 21  * which have the same subject.
 22  *
 23  * @param {int}	id		a unique ID
 24  * @param {ZmMailList}		list		a list that contains this conversation
 25  * 
 26  * @extends		ZmMailItem
 27  */
 28 ZmConv = function(id, list) {
 29 
 30 	ZmMailItem.call(this, ZmItem.CONV, id, list);
 31 	
 32 	// conversations are always sorted by date desc initially
 33 	this._sortBy = ZmSearch.DATE_DESC;
 34 	this._listChangeListener = new AjxListener(this, this._msgListChangeListener);
 35 	this.folders = {};
 36 };

 39 ZmConv.prototype.constructor = ZmConv;

 42 ZmConv.prototype.toString = function() { return "ZmConv"; };

 74 /**
 75  * Ensures that the requested range of msgs is loaded, getting them from the server if needed.
 76  * Because the list of msgs returned by the server contains info about which msgs matched the
 77  * search, we need to be careful about caching those msgs within the conv. This load function
 78  * should be used when in a search context, for example when expanding a conv that is the result
 79  * of a search.
 80  *
 81  * @param {Hash}		params						a hash of parameters:
 82  * @param {String}		params.query				the query used to retrieve this conv
 83  * @param {constant}	params.sortBy				the sort constraint
 84  * @param {int}			params.offset				the position of first msg to return
 85  * @param {int}			params.limit				the number of msgs to return
 86  * @param {Boolean}		params.getHtml				if <code>true</code>, return HTML part for inlined msg
 87  * @param {Boolean}		params.getFirstMsg			if <code>true</code>, fetch the first matching msg in the conv
 88  * @param {Boolean}		params.getMatches			if <code>true</code>, fetch all matching msgs in the conv 
 89  * @param {Boolean}		params.getAllMsgs			if <code>true</code>, fetch all msgs in the conv 
 90  * @param {Boolean}		params.getUnreadMsgs		if <code>true</code>, fetch all unread msgs in the conv
 91  * @param {Boolean}		params.getUnreadOrFirstMsg	if <code>true</code>, fetch unread, or first if none are unread
 92  * @param {Boolean}		params.markRead				if <code>true</code>, mark that msg read
 93  * @param {boolean}		params.needExp				if not <code>false</code>, have server check if addresses are DLs
 94  * @param {AjxCallback}	callback					the callback to run with results
 95  */
 96 ZmConv.prototype.load =
 97 function(params, callback) {
 98 
 99 	params = params || {};
100 	var ctlr = appCtxt.getCurrentController();
101 	var query = params.query;
102 	if (!query) {
103 		query = (ctlr && ctlr.getSearchString) 
104 			? ctlr.getSearchString()
105 			: appCtxt.get(ZmSetting.INITIAL_SEARCH);
106 	}
107 	var queryHint = params.queryHint;
108 	if (!queryHint) {
109 		queryHint = (ctlr && ctlr.getSearchStringHint)
110 			? ctlr.getSearchStringHint() : "";
111 	}
112 	var sortBy = params.sortBy || ZmSearch.DATE_DESC;
113 	var offset = params.offset || 0;
114 	var limit = params.limit || appCtxt.get(ZmSetting.CONVERSATION_PAGE_SIZE);
115 
116 	var doSearch = true;
117 	if (this._loaded && this.msgs && this.msgs.size() && !params.forceLoad) {
118 		var size = this.msgs.size();
119 		if (this._sortBy != sortBy || this._query != query || (size != this.numMsgs && !offset)) {
120 			this.msgs.clear();
121 		} else if (!this.msgs.hasMore() || offset + limit <= size) {
122 			doSearch = false;	// we can use cached msg list
123 		}
124 	}
125 	if (!doSearch) {
126 		if (callback) {
127 			callback.run(this._createResult());
128 		}
129 	} else {
130 		this._sortBy = sortBy;
131 		this._query = query;
132 		this._offset = offset;
133 		this._limit = limit;
134 
135 		var searchParams = {
136 			query: query,
137 			queryHint: queryHint,
138 			types: (AjxVector.fromArray([ZmItem.MSG])),
139 			sortBy: sortBy,
140 			offset: offset,
141 			limit: limit,
142 			getHtml: (params.getHtml || this.isDraft || appCtxt.get(ZmSetting.VIEW_AS_HTML)),
143 			accountName: (appCtxt.multiAccounts && this.getAccount().name)
144 		};
145 		var search = this.search = new ZmSearch(searchParams);
146 
147 		var fetchId = (params.getFirstMsg && "1") || (params.getMatches && "hits") || (params.getAllMsgs && "all") ||
148 					  (params.getUnreadMsgs && "u") || (params.getUnreadOrFirstMsg && "u1") || "0";
149 		var convParams = {
150 			cid:		this.id,
151 			callback:	(new AjxCallback(this, this._handleResponseLoad, [params, callback])),
152 			fetchId:	fetchId,
153 			markRead:	params.markRead,
154 			noTruncate:	params.noTruncate,
155 			needExp:	Boolean(fetchId)
156 		};
157 		search.getConv(convParams);
158 	}
159 };
160 
161 ZmConv.prototype._handleResponseLoad =
162 function(params, callback, result) {
163 	var results = result.getResponse();
164 	if (!params.offset) {
165 		this.msgs = results.getResults(ZmItem.MSG);
166 		this.msgs.convId = this.id;
167 		this.msgs.addChangeListener(this._listChangeListener);
168 		this.msgs.setHasMore(results.getAttribute("more"));
169 		this._loaded = true;
170 	}
171 	if (callback) {
172 		callback.run(result);
173 	}
174 };

176 /**
177  * This method supports ZmZimletBase::getMsgsForConv. It loads all of this conv's
178  * messages, including their content. Note that it is not search-based, and uses
179  * GetConvRequest rather than SearchConvRequest.
180  * 
181  * @param {Hash}			params				a hash of parameters
182  * @param {Boolean}			params.fetchAll		if <code>true</code>, fetch content of all msgs
183  * @param {AjxCallback}		callback			the callback
184  * @param {ZmBatchCommand}	batchCmd			the batch cmd that contains this request
185  */
186 ZmConv.prototype.loadMsgs =
187 function(params, callback, batchCmd) {
188 
189 	params = params || {};
190 	var jsonObj = {GetConvRequest:{_jsns:"urn:zimbraMail"}};
191 	var request = jsonObj.GetConvRequest;
192 	var c = request.c = {
193 		id:		this.id,
194 		html:	(params.getHtml || this.isDraft || appCtxt.get(ZmSetting.VIEW_AS_HTML))
195 	}
196 	if (params.fetchAll) {
197 		c.fetch = "all";
198 	}
199 	ZmMailMsg.addRequestHeaders(c);
200 
201 	// never pass "undefined" as arg to a callback!
202 	var respCallback = this._handleResponseLoadMsgs.bind(this, callback || null);
203 	if (batchCmd) {
204 		batchCmd.addRequestParams(jsonObj, respCallback);
205 	} else {
206 		appCtxt.getAppController().sendRequest({jsonObj:jsonObj, asyncMode:true, callback:respCallback});
207 	}
208 };
209 
210 ZmConv.prototype._handleResponseLoadMsgs =
211 function(callback, result) {
212 	var resp = result.getResponse().GetConvResponse.c[0];
213 	this.msgIds = [];
214 
215 	if (!this.msgs) {
216 		// create new msg list
217 		this.msgs = new ZmMailList(ZmItem.MSG, this.search);
218 		this.msgs.convId = this.id;
219 		this.msgs.addChangeListener(this._listChangeListener);
220 	}
221 	else {
222 		//don't recreate if it already exists, so we don't lose listeners.. (see ZmConvView2.prototype.set)
223 		this.msgs.removeAllItems();
224 	}
225 	this.msgs.setHasMore(false);
226 	this._loaded = true;
227 
228 	var len = resp.m.length;
229 	//going from last to first since GetConvRequest returns the msgs in order of creation (older first) but we keep things newer first.
230 	for (var i = len - 1; i >= 0; i--) {
231 		var msgNode = resp.m[i];
232 		this.msgIds.push(msgNode.id);
233 		msgNode.su = resp.su;
234 		// construct ZmMailMsg's so they get cached
235 		var msg = ZmMailMsg.createFromDom(msgNode, {list: this.msgs});
236 		this.msgs.add(msg);
237 	}
238 
239 	if (callback) { callback.run(result); }
240 };

242 /**
243  * Adds the message at the given index.
244  *
245  * @param	{ZmMailMsg}		msg		the message to add
246  * @param	{int}			index	where to add it
247  */
248 ZmConv.prototype.addMsg =
249 function(msg, index) {
250 
251 	if (!this.msgs) {
252 		this.msgs = new ZmMailList(ZmItem.MSG, this.search);
253 		this.msgs.convId = this.id;
254 		this.msgs.addChangeListener(this._listChangeListener);
255 		this.msgs.setHasMore(false);
256 	}
257 	this.msgs.add(msg, index);
258 	this.msgIds = [];
259 	var a = this.msgs.getArray();
260 	for (var i = 0, len = a.length; i < len; i++) {
261 		this.msgIds.push(a[i].id);
262 	}
263 };

327 /**
328  * Checks if the conversation is read only. Returns false if it cannot be determined.
329  * 
330  * @return	{Boolean}	<code>true</code> if the conversation is read only
331  */
332 ZmConv.prototype.isReadOnly =
333 function() {
334 	
335 	if (this._loaded && this.msgs && this.msgs.size()) {
336 		// conv has been loaded, check each msg
337 		var msgs = this.msgs.getArray();
338 		for (var i = 0; i < msgs.length; i++) {
339 			if (msgs[i].isReadOnly()) {
340 				return true;
341 			}
342 		}
343 	}
344 	else {
345 		// conv has not been loaded, see if it's constrained to a folder
346 		var folderId = this.getFolderId();
347 		var folder = folderId && appCtxt.getById(folderId);
348 		return !!(folder && folder.isReadOnly());
349 	}
350 	return false;
351 };

353 /**
354  * Checks if this conversation has a message that matches the given search.
355  * If the search is not present or not matchable, the provided default value is
356  * returned.
357  *
358  * @param {ZmSearch}	search			the search to match against
359  * @param {Object}	defaultValue		the value to return if search is not matchable
360  * @return	{Boolean|Object}	<code>true</code> if this conversation has the message
361  */
362 ZmConv.prototype.hasMatchingMsg =
363 function(search, defaultValue) {
364 	if (search && search.isMatchable() && this.msgs) {
365 		var msgs = this.msgs.getArray();
366 		for (var i = 0; i < msgs.length; i++) {
367 			var msg = msgs[i];
368 			if (search.matches(msg) && !msg.ignoreJunkTrash() && this.folders[msg.folderId]) {
369 				return true;
370 			}
371 		}
372 	} else {
373 		return defaultValue;
374 	}
375 	return false;
376 };

378 ZmConv.prototype.ignoreJunkTrash =
379 function() {
380 	return Boolean((this.numMsgs == 1) && this.folders &&
381 				   ((this.folders[ZmFolder.ID_SPAM] && !appCtxt.get(ZmSetting.SEARCH_INCLUDES_SPAM)) ||
382 			 	    (this.folders[ZmFolder.ID_TRASH] && !appCtxt.get(ZmSetting.SEARCH_INCLUDES_TRASH))));
383 };

385 ZmConv.prototype.getAccount =
386 function() {
387     // pull out the account from the fully-qualified ID
388 	if (!this.account) {
389         var folderId = this.getFolderId();
390         var folder = folderId && appCtxt.getById(folderId);
391         // make sure current folder is not remote folder
392         // in that case getting account from parseID will fail if
393         // the shared account is also configured in ZD
394         if (!(folder && folder._isRemote)) {
395             this.account = ZmOrganizer.parseId(this.id).account;
396         }
397     }
398 
399     // fallback on the active account if account not found from parsed ID (most
400     // likely means this is a conv inside a shared folder of the active acct)
401     if (!this.account) {
402         this.account = appCtxt.getActiveAccount();
403     }
404     return this.account;
405 
406 };

408 /**
409 * Handles a modification notification.
410 * TODO: Bundle MODIFY notifs (should bubble up to parent handlers as well)
411 *
412 * @param obj		item with the changed attributes/content
413 * 
414 * @private
415 */
416 ZmConv.prototype.notifyModify =
417 function(obj, batchMode) {
418 	var fields = {};
419 	// a conv's ID can change if it's a virtual conv becoming real; 'this' will be
420 	// the old conv; if we can, we switch to using the new conv, which will be more
421 	// up to date; the new conv will be available if it was received via search results
422 	if (obj._newId != null) {
423 		var conv = appCtxt.getById(obj._newId) || this;
424 		conv._oldId = this.id;
425 		conv.id = obj._newId;
426 		appCtxt.cacheSet(conv._oldId);
427 		appCtxt.cacheSet(conv.id, conv);
428 		conv.msgs = conv.msgs || this.msgs;
429 		if (conv.msgs) {
430 			conv.msgs.convId = conv.id;
431 			var a = conv.msgs.getArray();
432 			for (var i = 0; i < a.length; i++) {
433 				a[i].cid = conv.id;
434 			}
435 		}
436 		conv.folders = AjxUtil.hashCopy(this.folders);
437 		if (conv.list && conv._oldId && conv.list._idHash[conv._oldId]) {
438 			delete conv.list._idHash[conv._oldId];
439 			conv.list._idHash[conv.id] = conv;
440 		}
441 		fields[ZmItem.F_ID] = true;
442 		conv._notify(ZmEvent.E_MODIFY, {fields : fields});
443 	}
444 	if (obj.n != null) {
445 		this.numMsgs = obj.n;
446 		fields[ZmItem.F_SIZE] = true;
447 		this._notify(ZmEvent.E_MODIFY, {fields : fields});
448 	}
449 
450 	return ZmMailItem.prototype.notifyModify.apply(this, arguments);
451 };

453 /**
454  * Checks if any of the msgs within this conversation has the given value for
455  * the given flag. If the conv hasn't been loaded, looks at the conv-level flag.
456  *
457  * @param {constant}	flag		the flag (see <code>ZmItem.FLAG_</code> constants)
458  * @param {Boolean}	value		the test value
459  * @return	{Boolean}	<code>true</code> if the flag exists
460  */
461 ZmConv.prototype.hasFlag =
462 function(flag, value) {
463 	if (!this.msgs) {
464 		return (this[ZmItem.FLAG_PROP[flag]] == value);
465 	}
466 	var msgs = this.msgs.getArray();
467 	for (var j = 0; j < msgs.length; j++) {
468 		var msg = msgs[j];
469 		if (msg[ZmItem.FLAG_PROP[flag]] == value) {
470 			return true;
471 		}
472 	}
473 	return false;
474 };

476 /**
477  * Checks to see if a change in the value of a msg flag changes the value of the conv's flag. That will happen
478  * for the first msg to get an off flag turned on, or when the last msg to have an on flag turns it off.
479  */
480 ZmConv.prototype._checkFlags = 
481 function(flags) {
482 
483 	var convOn = {};
484 	var msgsOn = {};
485 	for (var i = 0; i < flags.length; i++) {
486 		var flag = flags[i];
487 		if (!(flag == ZmItem.FLAG_FLAGGED || flag == ZmItem.FLAG_UNREAD || flag == ZmItem.FLAG_MUTE || flag == ZmItem.FLAG_ATTACH || flag == ZmItem.FLAG_PRIORITY)) { continue; }
488 		convOn[flag] = this[ZmItem.FLAG_PROP[flag]];
489 		msgsOn[flag] = this.hasFlag(flag, true);
490 	}			
491 	var doNotify = false;
492 	var flags = [];
493 	for (var flag in convOn) {
494 		if (convOn[flag] != msgsOn[flag]) {
495 			this[ZmItem.FLAG_PROP[flag]] = msgsOn[flag];
496 			flags.push(flag);
497 			doNotify = true;
498 		}
499 	}
500 
501 	if (doNotify) {
502 		this._notify(ZmEvent.E_FLAGS, {flags: flags});
503 	}
504 };

506 /**
507  * Figure out if any tags have been added or removed by comparing what we have now with what
508  * our messages have.
509  * 
510  * @private
511  */
512 ZmConv.prototype._checkTags = 
513 function() {
514 	var newTags = {};
515 	var allTags = {};
516 	
517 	for (var tagId in this.tagHash) {
518 		allTags[tagId] = true;
519 	}
520 
521 	var msgs = this.msgs.getArray();
522 	if (!(msgs && msgs.length)) { return; }
523 	for (var i = 0; i < msgs.length; i++) {
524 		for (var tagId in msgs[i].tagHash) {
525 			newTags[tagId] = true;
526 			allTags[tagId] = true;
527 		}
528 	}
529 
530 	var notify = false;
531 	for (var tagId in allTags) {
532 		if (!this.tagHash[tagId] && newTags[tagId]) {
533 			if (this.tagLocal(tagId, true)) {
534 				notify = true;
535 			}
536 		} else if (this.tagHash[tagId] && !newTags[tagId]) {
537 			if (this.tagLocal(tagId, false)) {
538 				notify = true;
539 			}
540 		}
541 	}
542 
543 	if (notify) {
544 		this._notify(ZmEvent.E_TAGS);
545 	}
546 };

573 /**
574  * Gets the first matching msg of this conv, loading the conv msg list if necessary. If the
575  * msg itself hasn't been loaded we also load the conv. The conv load is a SearchConvRequest
576  * which fetches the content of the first matching msg and returns it via a callback. If no
577  * callback is provided, the conv will not be loaded - if it already has a msg list, the msg
578  * will come from there; otherwise, a skeletal msg with an ID is returned. Note that a conv
579  * always has at least one msg.
580  * 
581  * @param {Hash}	params	a hash of parameters
582  * @param {String}      params.query				the query used to retrieve this conv
583  * @param {constant}      params.sortBy			the sort constraint
584  * @param {int}	      params.offset			the position of first msg to return
585  * @param {int}	params.limit				the number of msgs to return
586  * @param {AjxCallback}	callback			the callback to run with results
587  * 
588  * @return	{ZmMailMsg}	the message
589  */
590 ZmConv.prototype.getFirstHotMsg =
591 function(params, callback) {
592 	
593 	var msg;
594 	params = params || {};
595 
596 	if (this.msgs && this.msgs.size()) {
597 		msg = this.msgs.getFirstHit(params.offset, params.limit);
598 	}
599 
600 	if (callback) {
601 		if (msg && msg._loaded && !params.forceLoad) {
602 			callback.run(msg);
603 		} else {
604 			var respCallback = new AjxCallback(this, this._handleResponseGetFirstHotMsg, [params, callback]);
605 			params.getFirstMsg = true;
606 			this.load(params, respCallback);
607 		}
608 	} else {
609 		// do our best to return a "realized" message by checking cache
610 		if (!msg && this.msgIds && this.msgIds.length) {
611 			var id = this.msgIds[0];
612 			msg = appCtxt.getById(id) || new ZmMailMsg(id);
613 		}
614 		return msg;
615 	}
616 };
617 
618 ZmConv.prototype._handleResponseGetFirstHotMsg =
619 function(params, callback) {
620 	var msg = this.msgs.getFirstHit(params.offset, params.limit);
621 	// should have a loaded msg
622 	if (msg && msg._loaded) {
623 		if (callback) {
624 			callback.run(msg);
625 		}
626 	} else {
627 		// desperate measures - get msg content from server
628 		if (!msg && this.msgIds && this.msgIds.length) {
629 			msg = new ZmMailMsg(this.msgIds[0]);
630 		}
631 		var respCallback = new AjxCallback(this, this._handleResponseLoadMsg, [msg, callback]);
632 		msg.load({getHtml:params.getHtml, callback:respCallback});
633 	}
634 };

643 ZmConv.prototype._loadFromDom =
644 function(convNode) {
645 	this.numMsgs = convNode.n;
646 	this.date = convNode.d;
647 	this._parseFlags(convNode.f);
648 	this._parseTagNames(convNode.tn);
649 	if (convNode.e) {
650 		for (var i = 0; i < convNode.e.length; i++) {
651 			this._parseParticipantNode(convNode.e[i]);
652 		}
653 	}
654 	this.participantsElided = convNode.elided;
655 	this.subject = convNode.su;
656 	this.fragment = convNode.fr;
657 	this.sf = convNode.sf;
658 
659 	// note that the list of msg IDs in a search result is partial - only msgs that matched are included
660 	if (convNode.m) {
661 		this.msgIds = [];
662 		for (var i = 0, count = convNode.m.length; i < count; i++) {
663 			var msgNode = convNode.m[i];
664 			this.msgIds.push(msgNode.id);
665 			this.folders[msgNode.l] = true;
666 		}
667 		if (count == 1) {
668 			var msgNode = convNode.m[0];
669 
670 			// bug 49067 - SearchConvResponse does not return the folder ID w/in
671 			// the msgNode as fully qualified so reset if this 1-msg conv was
672 			// returned by a simple folder search
673 			var searchFolderId = this.list && this.list.search && this.list.search.folderId;
674 			if (searchFolderId) {
675 				this.folderId = searchFolderId;
676 				this.folders[searchFolderId] = true;
677 			} else if (msgNode.l) {
678 				this.folderId = msgNode.l;
679 				this.folders[msgNode.l] = true;
680 			}
681 			else {
682 				AjxDebug.println(AjxDebug.NOTIFY, "no folder added for conv");
683 			}
684 			if (msgNode.s) {
685 				this.size = msgNode.s;
686 			}
687 
688 			if (msgNode.autoSendTime) {
689 				var timestamp = parseInt(msgNode.autoSendTime);
690 				if (timestamp) {
691 					this.setAutoSendTime(new Date(timestamp));
692 				}
693 			}
694 		}
695 	}
696 
697 	// Grab the metadata, keyed off the section name
698 	if (convNode.meta) {
699 		this.meta = {};
700 		for (var i = 0; i < convNode.meta.length; i++) {
701 			var section = convNode.meta[i].section;
702 			this.meta[section] = {};
703 			this.meta[section]._attrs = {};
704 			for (a in convNode.meta[i]._attrs) {
705 				this.meta[section]._attrs[a] = convNode.meta[i]._attrs[a];
706 			}
707 		}
708 	}
709 };

711 ZmConv.prototype._loadFromMsg =
712 function(msg) {
713 	this.date = msg.date;
714 	this.isFlagged = msg.isFlagged;
715 	this.isUnread = msg.isUnread;
716 	for (var i = 0; i < msg.tags.length; i++) {
717 		this.tagLocal(msg.tags[i], true);
718 	}
719 	var a = msg.participants ? msg.participants.getArray() : null;
720 	this.participants = new AjxVector();
721 	if (a && a.length) {
722 		for (var i = 0; i < a.length; i++) {
723 			var p = a[i];
724 			if ((msg.isDraft && p.type == AjxEmailAddress.TO) ||
725 				(!msg.isDraft && p.type == AjxEmailAddress.FROM)) {
726 				this.participants.add(p);
727 			}
728 		}
729 	}
730 	this.subject = msg.subject;
731 	this.fragment = msg.fragment;
732 	this.sf = msg.sf;
733 	this.msgIds = [msg.id];
734 	//add a flag to redraw this conversation when additional information is available
735 	this.redrawConvRow = true;
736 };

738 ZmConv.prototype._msgListChangeListener =
739 function(ev) {
740 	if (ev.type != ZmEvent.S_MSG) {	return; }
741 	if (ev.event == ZmEvent.E_TAGS || ev.event == ZmEvent.E_REMOVE_ALL) {
742 		this._checkTags();
743 	} else if (ev.event == ZmEvent.E_FLAGS) {
744 		this._checkFlags(ev.getDetail("flags"));
745 	} else if (ev.event == ZmEvent.E_DELETE || ev.event == ZmEvent.E_MOVE) {
746 		// a msg was moved or deleted, see if this conv's row should remain
747 		if (this.list && this.list.search && !this.hasMatchingMsg(this.list.search, true)) {
748             this.moveLocal(ev.item && ev.item.folderId);
749 			this._notify(ev.event);
750 		}
751 	}
752 };

768 // Updates the conversation fragment based on the newest message in the conversation, optionally ignoring an array of messages
769 ZmConv.prototype.updateFragment =
770 function(ignore) {
771 	var best;
772 	var size = this.msgs && this.msgs.size();
773 	if (size) {
774 		for (var j = size - 1; j >= 0; j--) {
775 			var candidate = this.msgs.get(j);
776 			if (ignore && AjxUtil.indexOf(ignore, candidate) != -1) { continue; }
777 			if (candidate.fragment && (!best || candidate.date > best.date)) {
778 				best = candidate;
779 			}
780 		}
781 	}
782 	if (best) {
783 		this.fragment = best.fragment;
784 	}
785 };

798 	var list = [];

803 		}
804 	}
805 	return AjxVector.fromArray(list);
