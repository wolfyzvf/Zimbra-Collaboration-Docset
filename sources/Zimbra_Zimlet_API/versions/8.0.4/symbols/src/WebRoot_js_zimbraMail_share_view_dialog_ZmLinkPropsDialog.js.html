 20 /**
 21  * Creates a link properties dialog.
 22  * @class
 23  * This class represents a dialog used to set the properties of an HTML link.
 24  * 
 25  * @param	{DwtControl}	shell		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtDialog
 29  */
 30 ZmLinkPropsDialog = function(shell, className) {
 31 	className = className || "ZmLinkPropsDialog";
 32 	DwtDialog.call(this, {parent:shell, className:className, title:ZmMsg.linkProperties});
 33 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 34 
 35 //	this._cache = AjxDispatcher.run("GetNotebookCache");
 36 
 37 	// set view
 38 	this.setView(this._createView());
 39 };

 50 /**
 51  * Pops-up the dialog.
 52  * 
 53  * @param	{Hash}	linkInfo		a hash of link information
 54  * @param	{String}	linkInfo.url	the link URL
 55  * @param	{String}	linkInfo.text	the link text
 56  * @param	{AjxCallback}	callback		the callback
 57  */
 58 ZmLinkPropsDialog.prototype.popup =
 59 function(linkInfo, callback) {
 60 	this._linkInfo = linkInfo || {};
 61 	this._callback = callback;
 62 
 63 	var isUrlLink = this._linkInfo.url;
 64 	if (appCtxt.get(ZmSetting.NOTEBOOK_ENABLED)) {
 65 		var root = appCtxt.getById(ZmOrganizer.ID_ROOT);
 66 		var children = root.children.getArray();
 67 
 68 		this._notebookSelect.clearOptions();
 69 		this.__addNotebookChildren(children, "");
 70 		this.notebookChangeListener();
 71 		
 72 		this._pageRadioEl.checked = !isUrlLink;
 73 		this._urlRadioEl.checked = isUrlLink;
 74 
 75 		this._pageInput.setValue("");
 76 	}
 77 
 78 	this._urlInput.setValue(this._linkInfo.url || "http://www.");
 79 	this._titleInput.setValue(this._linkInfo.text || "");
 80 
 81 	ZmLinkPropsDialog._setRequiredFields(this, !isUrlLink);
 82 
 83 	DwtDialog.prototype.popup.call(this);
 84 	ZmLinkPropsDialog._enableFieldsOnEdit(this);
 85 };

 87 ZmLinkPropsDialog.prototype.popdown =
 88 function() {
 89 	if (this._acPageList) {
 90 		//this._acPageList.reset();
 91 		this._acPageList.show(false);
 92 	}
 93 	DwtDialog.prototype.popdown.call(this);
 94 };

103 ZmLinkPropsDialog.prototype.autocompleteMatch =
104 function(s) {
105 	return [];
106 	
107 	var notebookId = this._notebookSelect.getValue();
108 	var pages = this._cache.getPagesInFolder(notebookId);
109 
110 	var m = [];
111 	var lows = s.toLowerCase();
112 	for (var p in pages) {
113 		var lowp = p.toLowerCase();
114 		var index = lowp.indexOf(lows);
115 		if (index != -1) {
116 			var page = pages[p];
117 			var object = {
118 				text: [
119 					p.substr(0,index),
120 					"<B>",p.substr(index, s.length),"</B>",
121 					p.substr(index + s.length)
122 				].join(""),
123 				name: p,
124 				data: page
125 			};
126 			m.push(object);
127 		}
128 	}
129 	return m;
130 };

145 ZmLinkPropsDialog._handleEdit =
146 function(event) {
147 	var target = DwtUiEvent.getTarget(event);
148 	var dialog = Dwt.getObjectFromElement(target);
149 	ZmLinkPropsDialog._enableFieldsOnEdit(dialog);
150 	return true;
151 };
152 
153 ZmLinkPropsDialog._handleKeyUp = function(event){
154 	if (DwtInputField._keyUpHdlr(event)) {
155 		var target = DwtUiEvent.getTarget(event);
156 		var inputField = Dwt.getObjectFromElement(target);
157 		var dialog = inputField.parent.parent;
158 		return ZmLinkPropsDialog._enableFieldsOnEdit(dialog);
159 	}
160 	return false;
161 };
162 
163 ZmLinkPropsDialog._handleLinkTo =
164 function(event) {
165 	var target = DwtUiEvent.getTarget(event);
166 	var dialog = Dwt.getObjectFromElement(target);
167 	var isPage = target == dialog._pageRadioEl;
168 	ZmLinkPropsDialog._setRequiredFields(dialog, isPage);
169 	ZmLinkPropsDialog._enableFieldsOnEdit(dialog);
170 	return true;
171 };
172 
173 ZmLinkPropsDialog._handleUrlTest = function(event) {
174 	var target = DwtUiEvent.getTarget(event);
175 	var dialog = Dwt.getObjectFromElement(target);
176 
177 	var winurl = dialog._urlInput.getValue();
178 	if (!winurl) { return; }
179 	var winname = "_new";
180 	var winfeatures = [
181 		"width=",(window.outerWidth || 640),",",
182 		"height=",(window.outerHeight || 480),",",
183 		"location,menubar,",
184 		"resizable,scrollbars,status,toolbar"
185 	].join("");
186 
187 	var win = open(winurl, winname, winfeatures);
188 };
189 
190 ZmLinkPropsDialog._enableFieldsOnEdit =
191 function(dialog) {
192 	var enabled = false;
193 	if (dialog._pageRadioEl && dialog._pageRadioEl.checked) {
194 		enabled = true;
195 	}
196 	else {
197 		enabled = dialog._urlInput.getValue().replace(/^\s+(.*)\s+$/,"$1") != "";
198 	}
199 	dialog.setButtonEnabled(DwtDialog.OK_BUTTON, enabled);
200 };
201 
202 ZmLinkPropsDialog._setRequiredFields =
203 function(dialog, isPage) {
204 	if (dialog._pageInput) {
205 		dialog._pageInput.setRequired(isPage);
206 	}
207 	dialog._urlInput.setRequired(!isPage);
208 };

212 ZmLinkPropsDialog.prototype._handleOkButton =
213 function(event) {
214 	this.popdown();
215 	if (this._callback) {
216 		var title = this._titleInput.getValue().replace(/^\s+(.*)\s+$/,"$1");
217 		var link;
218 		if (this._pageRadioEl && this._pageRadioEl.checked) {
219 			var notebookId = this._notebookSelect.getValue();
220 			var notebook = appCtxt.getById(notebookId);
221 			var value = AjxStringUtil.trim(this._pageInput.getValue());
222 			link = [
223 				"[[",
224 					"/",notebook.getSearchPath(),"/",value,
225 					(title ? "|" : ""),title,
226 				"]]"
227 			].join("");
228 		}
229 		else {
230 			var doc = this._linkInfo.document || document;
231 			link = doc.createElement("A");
232 			link.href = AjxStringUtil.trim(this._urlInput.getValue());
233 			link.innerHTML = AjxStringUtil.htmlEncode(title || link.href);
234 		}
235 
236 		this._callback.run(link, this._linkInfo.target);
237 	}
238 };

303 		this._notebookSelect = new DwtSelect({parent:view});
304 		var notebookChange = new AjxListener(this,this.notebookChangeListener);

320

324

333

344 		table.border = 0;
345 		table.cellSpacing = 10;

348 		var cell = row.insertCell(-1);
349 		cell.appendChild(this._urlInput.getHtmlElement());
350 		var cell = row.insertCell(-1);
351 		cell.innerHTML = [
352 			"<span ",
353 				"id='",urlTestId,"' ",
354 				"onmouseover='this.style.cursor=\"pointer\"' ",
355 				"onmouseout='this.style.cursor=\"default\"' ",

389 		//Dwt.setHandler(pageInputEl, DwtEvent.ONKEYUP, ZmLinkPropsDialog._handleKeyUp);

391 		tabs.push(Dwt.byId(typePageId), this._notebookSelect, this._pageSelect, this._pageInput, Dwt.byId(typeUrlId), this._urlInput);

394 	tabs.push(this._titleInput);
395 
396 	// create properties
397 	var props = new DwtPropertySheet(view);
398 	if (!appCtxt.get(ZmSetting.NOTEBOOK_ENABLED)) {
399 		props.addProperty(linkUrlLabel, this._urlInput);
400 	}
401 	props.addProperty(ZmMsg.linkTitleLabel, this._titleInput);
402 
403 	for (var i = 0; i < tabs.length; i++) {
404 		this._tabGroup.addMember(tabs[i], i);
405 	}
406 
407 	return view;
408 };

412 ZmLinkPropsDialog.prototype.__addNotebookChildren =
413 function(children, depth) {
414 	if (!children) return;
415 
416 	for (var i = 0; i < children.length; i++) {
417 		var child = children[i];
418 		//this._notebookSelect.addOption(depth+child.name, false, child.id);
419 		if(child instanceof ZmNotebook ){
420 		this._notebookSelect.addOption(child.getSearchPath(), false, child.id);
421 		var grandChildren = child.children.getArray();
422 		this.__addNotebookChildren(grandChildren, depth+" ");
423 		}
424 	}
425 };

427 ZmLinkPropsDialog.prototype.notebookChangeListener =
428 function(s) {
429 	var notebookId = this._notebookSelect.getValue();
430 	this._pageSelect.clearOptions();
431 	this._pageSelect.setText("Loading...");
432 	var pages = this._cache.getPagesInFolder(notebookId);
433 	var isEmpty = true;
434 	for (var p in pages) {
435 		if(p != "_Index"){
436 			this._pageSelect.addOption(p, false, p);
437 			isEmpty = false;
438 		}
439 	}	
440 	if(isEmpty){
441 		this._pageSelect.setText(ZmMsg.notAvailable);
442 		this._pageSelect.disable();
443 	}else{
444 		this._pageSelect.enable();
445 	}
446 	this.pageChangeListener();
447 };

449 ZmLinkPropsDialog.prototype.pageChangeListener =
450 function(s) {
451 	var pageName = this._pageSelect.getValue();
452 	this._pageInput.setValue(pageName);
453 	ZmLinkPropsDialog._enableFieldsOnEdit(this);
454 };
