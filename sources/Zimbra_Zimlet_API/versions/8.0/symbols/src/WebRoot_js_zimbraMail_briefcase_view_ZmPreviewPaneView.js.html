 16 ZmPreviewPaneView = function(parent, controller, dropTgt) {
 17 
 18 	if (arguments.length == 0) { return; }
 19 
 20     var params = {};
 21     params.className = params.className || "ZmPreviewPaneView";
 22     params.parent = parent;
 23     params.controller = controller;
 24     params.posStyle = Dwt.ABSOLUTE_STYLE;
 25     DwtComposite.call(this, params);
 26 
 27 	this._controller = controller;
 28 
 29 	this._vertMsgSash = new DwtSash({parent:this, style:DwtSash.HORIZONTAL_STYLE, className:"AppSash-horiz",
 30 									 threshold:ZmPreviewPaneView.SASH_THRESHOLD, posStyle:Dwt.ABSOLUTE_STYLE});
 31 	this._vertMsgSash.registerCallback(this._sashCallback, this);
 32 
 33 	this._horizMsgSash = new DwtSash({parent:this, style:DwtSash.VERTICAL_STYLE, className:"AppSash-vert",
 34 									  threshold:ZmPreviewPaneView.SASH_THRESHOLD, posStyle:Dwt.ABSOLUTE_STYLE});
 35 	this._horizMsgSash.registerCallback(this._sashCallback, this);
 36 
 37 	this._previewView = new ZmPreviewView({parent:this, posStyle:DwtControl.ABSOLUTE_STYLE, controller: this._controller});
 38 	
 39     this._detailListView = new ZmDetailListView(this, this._controller, this._controller._dropTgt );
 40     this._detailListView.addSelectionListener(new AjxListener(this, this._listSelectionListener));
 41     this._listSelectionShortcutDelayAction = new AjxTimedAction(this, this._listSelectionTimedAction);
 42     this._delayedSelectionItem = null;
 43 	this.setReadingPane();
 44 };

 80 ZmPreviewPaneView.prototype.setReadingPane =
 81 function() {
 82 
 83 	var tlv = this._detailListView, tv = this._previewView;
 84 	var readingPaneEnabled = this._controller.isReadingPaneOn();
 85 	if (!readingPaneEnabled) {
 86 		tv.setVisible(false);
 87 		this._vertMsgSash.setVisible(false);
 88 		this._horizMsgSash.setVisible(false);
 89 	} else {		
 90 		tv.setVisible(true);
 91 		var readingPaneOnRight = this._controller.isReadingPaneOnRight();
 92 		var newSash = readingPaneOnRight ? this._vertMsgSash : this._horizMsgSash;
 93 		var oldSash = readingPaneOnRight ? this._horizMsgSash : this._vertMsgSash;
 94 		oldSash.setVisible(false);
 95 		newSash.setVisible(true);
 96 	}
 97 
 98 	tlv.reRenderListView();
 99 
100 
101 	tv.noTab = !readingPaneEnabled || AjxEnv.isIE;
102 	var sz = this.getSize();
103 	this._resetSize(sz.x, sz.y, true);
104 };

174 ZmPreviewPaneView.prototype._resetSize =
175 function(newWidth, newHeight, force) {
176 
177 
178 	if (newWidth <= 0 || newHeight <= 0) { return; }
179 	if (!force && newWidth == this._lastResetWidth && newHeight == this._lastResetHeight) { return; }
180 
181 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
182 
183 	if (this.isPreviewPaneVisible()) {
184 		var sash = this.getSash();
185 		var sashSize = sash.getSize();
186 		var sashThickness = readingPaneOnRight ? sashSize.x : sashSize.y;
187 		if (readingPaneOnRight) {
188 			var listViewWidth = this._vertSashX || (Number(ZmMsg.LISTVIEW_WIDTH)) || Math.floor(newWidth / 2.5);
189 			this._detailListView.resetSize(listViewWidth, newHeight);
190 			sash.setLocation(listViewWidth, 0);
191 			this._previewView.setBounds(listViewWidth + sashThickness, 0,
192 									newWidth - (listViewWidth + sashThickness), newHeight);
193 		} else {
194 			var listViewHeight = this._horizSashY || (Math.floor(newHeight / 2) - DwtListView.HEADERITEM_HEIGHT);
195 			this._detailListView.resetSize(newWidth, listViewHeight);
196 			sash.setLocation(0, listViewHeight);
197 			this._previewView.setBounds(0, listViewHeight + sashThickness, newWidth,
198 									newHeight - (listViewHeight + sashThickness));
199 		}
200 	} else {
201 		this._detailListView.resetSize(newWidth, newHeight);
202 	}
203 	this._detailListView._resetColWidth();
204 
205 	this._lastResetWidth = newWidth;
206 	this._lastResetHeight = newHeight;
207 };

209 ZmPreviewPaneView.prototype._sashCallback =
210 function(delta) {
211 
212 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
213 	if (delta > 0) {
214 		if (readingPaneOnRight) {
215 			// moving sash right
216 			var minMsgViewWidth = 300;
217 			var currentMsgWidth = this._previewView.getSize(true).x;
218 			delta = Math.max(0, Math.min(delta, currentMsgWidth - minMsgViewWidth));
219 			var newListWidth = ((AjxEnv.isIE) ? this._vertMsgSash.getLocation().x : this._detailListView.getSize(true).x) + delta;
220 			if (delta > 0) {
221 				this._detailListView.resetSize(newListWidth, Dwt.DEFAULT);
222 				this._previewView.setBounds(this._previewView.getLocation().x + delta, Dwt.DEFAULT,
223 										currentMsgWidth - delta, Dwt.DEFAULT);
224 				
225 			} else {
226 				delta = 0;
227 			}
228 			
229 			
230 			
231 		} else {
232 			// moving sash down
233 			var newMsgViewHeight = this._previewView.getSize().y - delta;
234 			var minMsgViewHeight = 150;
235 			if (newMsgViewHeight > minMsgViewHeight) {
236 				this._detailListView.resetSize(Dwt.DEFAULT, this._detailListView.getSize(true).y + delta);
237 				this._previewView.setBounds(Dwt.DEFAULT, this._previewView.getLocation().y + delta,
238 							Dwt.DEFAULT, newMsgViewHeight);
239 			} else {
240 				delta = 0;
241 			}
242 		}
243 	} else {
244 		var absDelta = Math.abs(delta);
245 
246 		if (readingPaneOnRight) {
247 			// moving sash left
248 			var currentWidth = this._vertMsgSash.getLocation().x;
249 			absDelta = Math.max(0, Math.min(absDelta, currentWidth - 300));
250 
251 			if (absDelta > 0) {
252 				delta = -absDelta;
253 				this._detailListView.resetSize(currentWidth - absDelta, Dwt.DEFAULT);
254 				this._previewView.setBounds(this._previewView.getLocation().x - absDelta, Dwt.DEFAULT,
255 						this._previewView.getSize(true).x + absDelta, Dwt.DEFAULT);
256 			} else {
257 				delta = 0;
258 			}
259 		} else {
260 			// moving sash up
261 			if (!this._minMLVHeight) {
262 				var list = this._detailListView.getList();
263 				if (list && list.size()) {
264 					var item = list.get(0);
265 					var div = document.getElementById(this._detailListView._getItemId(item));
266 					this._minMLVHeight = DwtListView.HEADERITEM_HEIGHT + (Dwt.getSize(div).y * 2);
267 				} else {
268 					this._minMLVHeight = DwtListView.HEADERITEM_HEIGHT;
269 				}
270 			}
271 
272 			if (this.getSash().getLocation().y - absDelta > this._minMLVHeight) {
273 				// moving sash up
274 				this._detailListView.resetSize(Dwt.DEFAULT, this._detailListView.getSize(true).y - absDelta);
275 				this._previewView.setBounds(Dwt.DEFAULT, this._previewView.getLocation().y - absDelta,
276 						Dwt.DEFAULT, this._previewView.getSize(true).y + absDelta);
277 			} else {
278 				delta = 0;
279 			}
280 		}
281 	}
282 
283 	if (delta) {
284 		this._detailListView._resetColWidth();
285 		if (readingPaneOnRight) {
286 			this._vertSashX = this._vertMsgSash.getLocation().x;
287 		} else {
288 			this._horizSashY = this._horizMsgSash.getLocation().y;
289 		}
290 	}
291 
292 	return delta;
293 };

350 ZmPreviewPaneView.prototype._listSelectionListener =
351 function(ev){
352     var item = ev.item, handled = false;
353     if(!item) {
354     	return;
355     }
356 
357     var cs = appCtxt.isOffline && appCtxt.getCurrentSearch();
358     if (cs) {
359         appCtxt.accountList.setActiveAccount(item.getAccount());
360     }
361     var noChange = ev && ev._details && ev._details.oldFolderId == item.folderId;
362     if ((ev.event == ZmEvent.E_MOVE && noChange) || ev.event == ZmEvent.E_DELETE)
363         return;
364 
365     if(ev.field == ZmItem.F_EXPAND && this._detailListView._isExpandable(item)){
366         this._detailListView.expandItem(item);   
367     } else if(this._controller.isReadingPaneOn() && item){
368     	if (ev.kbNavEvent) {
369     		if (this._listSelectionShortcutDelayActionId) {
370     			AjxTimedAction.cancelAction(this._listSelectionShortcutDelayActionId); 
371     		}
372     		this._delayedSelectionItem = item;
373     		this._listSelectionShortcutDelayActionId = AjxTimedAction.scheduleAction(this._listSelectionShortcutDelayAction,
374     				ZmPreviewPaneView.LIST_SELECTION_SHORTCUT_DELAY)
375     	} else {
376     		this._previewView.set(item);
377     	}
378     }
379 };

397 ZmPreviewPaneView.prototype._handleVersions =
398 function(item, result){
399     result =  result.getResponse();
400     result = result.ListDocumentRevisionsResponse.doc;
401 
402     var revisions = this._getRevisionItems(item, result);
403     this._detailListView.expand(item, revisions);
404 };

406 ZmPreviewPaneView.prototype._getRevisionItems =
407 function(item, revisions){
408     var revisionItems = [];
409     for(var i=0; i<revisions.length; i++){
410         var rev = revisions[i];
411         var rItem = new ZmRevisionItem(this._getRevisionId(rev), item);
412         rItem.set(rev);
413         revisionItems.push(rItem);        
414     }
415     return AjxVector.fromArray(revisionItems);
416 };

421 };

432 ZmPreviewPaneView.prototype.restoreVersion =
433 function(verItem){
434     if(verItem.isRevision){
435         var item =  verItem.parent;
436         if(item && item.version != verItem.revision ){
437             item.restoreVersion(verItem.version, new AjxCallback(this, this.refreshItem, item));
438         }
439     }
440 };

442 ZmPreviewPaneView.prototype.deleteVersions =
443 function(items){
444     items = items || this._detailListView.getSelection();
445     if(!items || items.length == 0) return;
446 
447     var delVerBatchCmd = new ZmBatchCommand(true, null, true);
448     for(var i=0; i<items.length; i++){
449         delVerBatchCmd.add(new AjxCallback(this, this.deleteVersion, [items[i], delVerBatchCmd]));
450     }
451     delVerBatchCmd.run();
452 };
453 
454 ZmPreviewPaneView.prototype.deleteVersion =
455 function(verItem, batchCmd){
456     if(verItem.isRevision){
457         var item =  verItem.parent;
458         if(item && item.version != verItem.revision ){
459             item.deleteVersion(verItem.version, new AjxCallback(this, this.refreshItem, item), batchCmd);
460         }
461     }
462 };

466     this._detailListView.collapse(item, true);    
467     this._expand(item);

568 ZmPreviewView.prototype._handleError =
569 function(item, errorCode, error){
570 
571     this.enablePreview(true);
572 
573     if(item){
574 
575         var restUrl = item.getRestUrl();
576         restUrl = AjxStringUtil.fixCrossDomainReference(restUrl);
577 
578         //Try to generate, otherwise fallback
579         if(ZmMimeTable.isRenderable(item.contentType)){
580             this._iframePreview.setSrc(restUrl);
581         }else if(ZmMimeTable.isMultiMedia(item.contentType)){
582             html = [
583                 "<div style='height:100%;width:100%;text-align:center;vertical-align:middle;padding-top:30px;'>",
584                 "<embed src='",restUrl,"'/>",
585                 "</div>"
586             ].join('');
587             this._iframePreview.setIframeContent(html);
588         }else{
589             //Show Download Link
590             var downloadLink = restUrl+ "?disp=a";
591             var html = [
592                 "<div style='height:100%;width:100%;text-align:center;vertical-align:middle;padding-top:30px;font-family: \'Helvetica Neue\',Helvetica,Arial,\'Liberation Sans\',sans-serif;'>",
593                     AjxMessageFormat.format(ZmMsg.previewDownloadLink, downloadLink),
594                 "</div>"
595             ].join('');
596             this._iframePreview.setIframeContent(html);
597         }
598         
599     }    
600 };

603 function(url){
604 
605     if(!window._zmPreviewView)
606         window._zmPreviewView = this;

608     url = url + ( url.match(/\?/) ? '&' : '?' ) + "callback=ZmPreviewView._errorCallback";

613 ZmPreviewView.prototype.set =
614 function(item){
615     if(!item){
616         this.enablePreview(false);
617         return;
618     }
619 
620     this._oldItem = this._previewItem;
621     this._previewItem = item;
622     this.enablePreview(true);
623 
624     this._previewContent = false;
625 
626     if(item.isFolder){
627         this._setFolder(item);
628         return;
629     }
630 
631 
632     this._setHeader(item);
633 
634     var restUrl = item.getRestUrl();
635     restUrl = AjxStringUtil.fixCrossDomainReference(restUrl);
636 
637     if(ZmMimeTable.isWebDoc(item.contentType)){
638         restUrl = restUrl + ( restUrl.match(/\?/) ? '&' : '?' ) + "viewonly=1";
639     }else{
640 
641         this._setupLoading();
642 
643         //Send everything trough ConvertD
644         restUrl = this._setupErrorCallback(restUrl);
645         restUrl += ( restUrl.match(/\?/) ? '&' : '?' ) + "fmt=native&view=html";
646     }
647 
648     this._iframePreview.setSrc(restUrl);
649 	Dwt.setLoadedTime("ZmBriefcaseItem"); //iframe src set but item may not be downloaded by browser
650 };

660     }catch(ex){
661         //At times the previous item is not loaded or in the process of loading, causes iframe.body to be null.
662         DBG.println("ZmPreviewView#_setupLoading");
663         DBG.println("New Item:"+ this._previewItem.name );
664 		DBG.println("  "+ex);
665     }

668 ZmPreviewView.prototype._resetIframeHeight =
669 function(){
670     var iframe = this._iframePreview.getIframe();
671     var doc = this._iframePreview.getDocument();
672     var origHeight = AjxEnv.isIE ? doc.body.scrollHeight : 0;
673     var h = Math.max(doc.body.scrollHeight, doc.documentElement.scrollHeight, origHeight);
674     iframe.style.height = h + "px";
675     if (AjxEnv.isWebKitBased) {
676         // bug: 39434, WebKit specific
677         // After the iframe ht is set there is change is body.scrollHeight, weird.
678         // So reset ht to make the entire body visible.
679         var newHt = Math.max(doc.body.scrollHeight, doc.documentElement.scrollHeight);
680         if (newHt > h) {
681             iframe.style.height = newHt + "px";
682         }
683     }
684 };

741 ZmPreviewView.prototype._setHeader =
742 function(item){
743 
744     //Name
745     this._headerName.innerHTML = AjxStringUtil.htmlEncode(item.name);
746 
747     //Image icon
748     var contentType = item.contentType;
749     if(contentType && contentType.match(/;/)) {
750         contentType = contentType.split(";")[0];
751     }
752     var mimeInfo = contentType ? ZmMimeTable.getInfo(contentType) : null;
753     var icon = "Img" + ( mimeInfo ? mimeInfo.imageLarge : "UnknownDoc_48");
754     this._headerImage.className = icon;
755 
756     //Modified & Created
757     var dateFormatter = AjxDateFormat.getDateTimeInstance(AjxDateFormat.LONG, AjxDateFormat.SHORT);
758     if (this._headerModified && item.modifyDate) {
759         this._headerModified.innerHTML = dateFormatter.format(item.modifyDate);
760 	}
761     if(this._headerModifier)
762         this._headerModifier.innerHTML = item.modifier;
763     if (this._headerCreated && item.createDate) {
764         this._headerCreated.innerHTML = dateFormatter.format(item.createDate);
765 	}
766     if(this._headerCreator)
767         this._headerCreator.innerHTML = item.creator;
768 
769     if(this._lockStatus)
770         this._lockStatus.innerHTML = AjxImg.getImageHtml(item.locked ? "PadLock" : "Blank_16");
771 
772     if(this._headerLockTime){
773         if(item.locked){
774             dateFormatter = AjxDateFormat.getDateInstance();
775             this._headerLockTime.innerHTML = dateFormatter.format(item.lockTime);
776         }else{
777             this._headerLockTime.innerHTML = ""
778         }        
779     }
780 
781     if(this._headerLockUser){
782         this._headerLockUser.innerHTML = item.locked ? item.lockUser : "";
783     }
784 
785     this.setNotes(item);
786 };
