 23 /**
 24  * Creates the application.
 25  * @class
 26  * This object represents a Zimbra Application. This class is a base class for application classes.
 27  * "App" is a useful abstraction for a set of related functionality, such as mail,
 28  * address book, or calendar. Looked at another way, an app is a collection of one or more controllers.
 29  * 
 30  * @param	{String}	name		the application name
 31  * @param	{DwtControl}	container	the control that contains components
 32  * @param	{ZmController}	parentController	the parent window controller (set by the child window)
 33  *
 34  */
 35 ZmApp = function(name, container, parentController) {
 36 
 37 	if (arguments.length == 0) return;
 38 	
 39 	this._name = name;
 40 	this._container = container;
 41 	this._parentController = parentController;
 42 	this._active = false;
 43 	this.currentSearch = null;
 44     this._defaultFolderId = null;   //reqd in case of external user
 45 
 46 	this._deferredFolders = [];
 47 	this._deferredFolderHash = {};
 48 	this._deferredNotifications = [];
 49 
 50 	this._sessionController		= {};
 51 	this._nextSessionId			= {};
 52 	this._curSessionId			= {};
 53 
 54 	ZmApp.DROP_TARGETS[name] = {};
 55 
 56 	this._defineAPI();
 57 	if (!parentController) {
 58 		this._registerSettings();
 59 	}
 60 	this._registerOperations();
 61 	this._registerItems();
 62 	this._registerOrganizers();
 63 	if (!parentController) {
 64 		this._setupSearchToolbar();
 65 	}
 66 	this._registerApp();
 67 
 68 };

139 /**
140  * Registers and stores information about an app. Note: Setting a value that evaluates to
141  * false (such as 0 or an empty string) will not do anything.
142  *
143  * @param {constant}	app				the app ID
144  * @param {Hash}	params			a hash of parameters
145  * @param params.mainPkg			[string]	main package that contains the app
146  * @param params.nameKey			[string]	msg key for app name
147  * @param params.icon				[string]	name of app icon class
148  * @param params.textPrecedence		[int]		order for removing button text
149  * @param params.imagePrecedence	[int]		order for removing button image
150  * @param params.chooserTooltipKey	[string]	msg key for app tooltip
151  * @param params.viewTooltipKey		[string]	msg key for app view menu tooltip
152  * @param params.defaultSearch		[constant]	type of item to search for in the app
153  * @param params.organizer			[constant]	main organizer for this app
154  * @param params.overviewTrees		[array]		list of tree IDs to show in overview
155  * @param params.hideZimlets		[boolean]	if true, hide Zimlet tree in overview
156  * @param params.searchTypes		[array]		list of types of saved searches to show in overview
157  * @param params.gotoActionCode		[constant]	key action for jumping to this app
158  * @param params.newActionCode		[constant]	default "new" action code
159  * @param params.actionCodes		[hash]		keyboard actions mapped to operations
160  * @param params.newItemOps			[hash]		IDs of operations that create a new item, and their text keys
161  * @param params.newOrgOps			[hash]		IDs of operations that create a new organizer, and their text keys
162  * @param params.qsViews			[array]		list of views to handle in query string
163  * @param params.chooserSort		[int]		controls order of apps in app chooser toolbar
164  * @param params.defaultSort		[int]		controls order in which app is chosen as default start app
165  * @param params.trashViewOp		[constant]	menu choice for "Show Only ..." in Trash view
166  * @param params.upsellUrl			[string]	URL for content of upsell
167  * @param params.searchResultsTab	[string]	if true, show search results in a tab
168  *
169  * @private
170  */
171 ZmApp.registerApp =
172 function(app, params) {
173 
174 	// TODO: why the ifs? this should only be called once per app
175 	if (params.mainPkg)				{ ZmApp.MAIN_PKG[app]			= params.mainPkg; }
176 	if (params.nameKey)				{ ZmApp.NAME[app]				= params.nameKey; }
177 	if (params.icon)				{ ZmApp.ICON[app]				= params.icon; }
178 	if (params.textPrecedence)		{ ZmApp.TEXT_PRECEDENCE[app]	= params.textPrecedence; }
179 	if (params.imagePrecedence)		{ ZmApp.IMAGE_PRECEDENCE[app]	= params.imagePrecedence; }
180 	if (params.chooserTooltipKey)	{ ZmApp.CHOOSER_TOOLTIP[app]	= params.chooserTooltipKey; }
181 	if (params.viewTooltipKey)		{ ZmApp.VIEW_TOOLTIP[app]		= params.viewTooltipKey; }
182 	if (params.defaultSearch)		{ ZmApp.DEFAULT_SEARCH[app]		= params.defaultSearch; }
183 	if (params.organizer)			{ ZmApp.ORGANIZER[app]			= params.organizer; }
184 	if (params.overviewTrees)		{ ZmApp.OVERVIEW_TREES[app]		= params.overviewTrees; }
185 	if (params.hideZimlets)			{ ZmApp.HIDE_ZIMLETS[app]		= params.hideZimlets; }
186 	if (params.searchTypes) 		{ ZmApp.SEARCH_TYPES[app]		= params.searchTypes; }
187 	if (params.gotoActionCode)		{ ZmApp.GOTO_ACTION_CODE[app]	= params.gotoActionCode; }
188 	if (params.newActionCode)		{ ZmApp.NEW_ACTION_CODE[app]	= params.newActionCode; }
189 	if (params.qsViews)				{ ZmApp.QS_VIEWS[app]			= params.qsViews; }
190 	if (params.chooserSort)			{ ZmApp.CHOOSER_SORT[app]		= params.chooserSort; }
191 	if (params.defaultSort)			{ ZmApp.DEFAULT_SORT[app]		= params.defaultSort; }
192 	if (params.trashViewOp)			{ ZmApp.TRASH_VIEW_OP[app]		= params.trashViewOp; }
193 	if (params.upsellUrl)			{ ZmApp.UPSELL_URL[app]			= params.upsellUrl; }
194 	//if (params.quickCommandType)	{ ZmApp.QUICK_COMMAND_TYPE[app]	= params.quickCommandType; }
195 	if (params.searchResultsTab)	{ ZmApp.SEARCH_RESULTS_TAB[app]	= params.searchResultsTab; }
196 
197 	if (params.searchTypes) {
198 		ZmApp.SEARCH_TYPES_R[app] = {};
199 		for (var i = 0; i < params.searchTypes.length; i++) {
200 			ZmApp.SEARCH_TYPES_R[app][params.searchTypes[i]] = true;
201 		}
202 	}
203 
204 	if (params.gotoActionCode) {
205 		ZmApp.GOTO_ACTION_CODE_R[params.gotoActionCode] = app;
206 	}
207 
208 	if (params.actionCodes) {
209 		for (var ac in params.actionCodes) {
210 			if (!ac) { continue; }
211 			ZmApp.ACTION_CODES_R[ac] = app;
212 			ZmApp.ACTION_CODES[ac] = params.actionCodes[ac];
213 		}
214 	}
215 
216     var appEnabled = appCtxt.get(ZmApp.SETTING[app]);
217 	if (params.newItemOps && appEnabled) {
218 		for (var op in params.newItemOps) {
219 			if (!op) { continue; }
220 			ZmApp.OPS_R[op] = app;
221 			ZmOperation.NEW_ITEM_OPS.push(op);
222 			ZmOperation.NEW_ITEM_KEY[op] = params.newItemOps[op];
223 		}
224 	}
225 	if (params.newOrgOps && appEnabled) {
226 		for (var op in params.newOrgOps) {
227 			if (!op) { continue; }
228 			ZmApp.OPS_R[op] = app;
229 			ZmOperation.NEW_ORG_OPS.push(op);
230 			ZmOperation.NEW_ORG_KEY[op] = params.newOrgOps[op];
231 		}
232 	}
233 
234 	if (params.qsViews) {
235 		for (var i = 0; i < params.qsViews.length; i++) {
236 			ZmApp.QS_VIEWS[params.qsViews[i]] = app;
237 		}
238 	}
239 
240     /* if (params.quickCommandType) {
241         ZmQuickCommand.itemTypes.push(params.quickCommandType);
242     } */
243 };

246 /**
247  * Runs the given function for all known (e.g. part of ZmApp.CLASS)
248  * app classes, passing args.
249  * NOTE: This runs class functions only, not instance (prototype) functions.
250  * @static
251  * @param funcName {String} The name of the function we will run on each
252  * application.
253  * @param mixed {mixed} 0 to n additional arguments are passed to funcName
254  * via apply.
255  */
256 ZmApp.runAppFunction =
257 function(funcName) {
258     var args;
259 
260     for (var appName in ZmApp.CLASS) {
261         var app = window[ZmApp.CLASS[appName]];
262         var func = app && app[funcName];
263         if (func && (typeof(func) == "function")) {
264             args = args || Array.prototype.slice.call(arguments, 1);
265             func.apply(app, args);
266         }
267     }
268 };

428 /**
429  * Gets the remote folder ids.
430  *
431  * @param	{Object}	account		the account
432  * @return	{Array}		an array of {String} ids
433  */
434 ZmApp.prototype.getRemoteFolderIds =
435 function(account) {
436 	// XXX: optimize by caching this list? It would have to be cleared anytime
437 	// folder structure changes
438 	var list = [];
439 	if (appCtxt.getOverviewController(true)) {
440 		var type = ZmApp.ORGANIZER[this.getName()];
441 
442 		// first, make sure there aren't any deferred folders that need to be created
443 		if (this._deferredFolders.length) {
444 			this._createDeferredFolders(type);
445 		}
446 
447 		var tree = appCtxt.getFolderTree(account);
448 		var folders = tree ? tree.getByType(type) : [];
449 		for (var i = 0; i < folders.length; i++) {
450 			var folder = folders[i];
451 			if (folder.isRemote()) {
452 				list.push(folder.id);
453 			}
454 		}
455 	}
456 	return list;
457 };

537 /**
538  * Resets the current overview, preserving expansion.
539  *
540  * @param {String}		overviewId	the id of overview to reset
541  */
542 ZmApp.prototype.resetOverview =
543 function(overviewId) {
544 	var overview = overviewId ? appCtxt.getOverviewController().getOverview(overviewId) : this.getOverview();
545 	if (overview) {
546 		var expIds = [];
547 		var treeIds = overview.getTreeViews(), len = treeIds.length;
548 		for (var i = 0; i < len; i++) {
549 			var treeId = treeIds[i];
550 			var treeView = overview.getTreeView(treeId);
551 			var items = treeView.getTreeItemList();
552 			var len1 = items.length;
553 			for (var j = 0; j < len1; j++) {
554 				var treeItem = items[j];
555 				if (treeItem._expanded) {
556 					expIds.push(treeItem._htmlElId);
557 				}
558 			}
559 		}
560 		overview.clear();
561 		overview.set(this._getOverviewTrees());
562 		len = expIds.length;
563 		for (var i = 0; i < len; i++) {
564 			var treeItem = DwtControl.fromElementId(expIds[i]);
565 			if (treeItem && !treeItem._expanded) {
566 				treeItem.setExpanded(true);
567 			}
568 		}
569 	}
570 };

658 /**
659  * Returns a controller of the given type and class. If no sessionId is provided, then
660  * the controller's session ID will be an incremental number. If a sessionId is given,
661  * then a check is made for an existing controller with that session ID. If none is
662  * found, one is created and given that session ID.
663  * 
664  * @param	{hash}							params						hash of params:
665  * @param	{string}						controllerClass				string name of controller class
666  * @param	{string}						sessionId					unique identifier for this controller
667  * @param 	{ZmSearchResultsController}		searchResultsController		containing controller
668  */
669 ZmApp.prototype.getSessionController =
670 function(params) {
671 	
672 	var type;
673 	try {
674 		type = eval(params.controllerClass).getDefaultViewType(params);
675 	}
676 	catch (ex) {
677 		throw new AjxException("Session controller " + params.controllerClass + " must implement getDefaultViewType()");
678 	}
679 	
680 	// track controllers of this type
681 	if (!this._sessionController[type]) {
682 		this._sessionController[type] = {};
683 		this._nextSessionId[type] = 1;
684 	}
685 
686 	// check if we've already created a session controller with the given ID
687 	var sessionId = params.sessionId;
688 	if (sessionId && this._sessionController[type][sessionId]) {
689 		return this._sessionController[type][sessionId];
690 	}
691 
692 	// re-use an inactive controller if possible
693 	var controller;
694 	if (!sessionId) {
695 		var controllers = this._sessionController[type];
696 		for (var id in controllers) {
697 			if (controllers[id].inactive && !controllers[id].isPinned && !controllers[id].isHidden) {
698 				controller = controllers[id];
699 				break;
700 			}
701 		}
702 	}
703 
704 	sessionId = (controller && controller.getSessionId()) || sessionId || String(this._nextSessionId[type]++);
705 
706 	if (!controller) {
707 		var ctlrClass = eval(params.controllerClass);
708 		controller = this._sessionController[type][sessionId] =
709 			new ctlrClass(this._container, this, type, sessionId, params.searchResultsController);
710 	}
711 	this._curSessionId[type] = sessionId;
712 	controller.inactive = false;
713 
714 	return controller;
715 };

973 /**
974  * Notify change listeners with a list of notifications, rather than a single
975  * item, so that they can optimize. For example, a list view can wait to
976  * fix its alternation of dark and light rows until after all the moved ones
977  * have been taken out, rather than after the removal of each row.
978  *
979  * @param mods	{Array}		list of notification objects
980  */
981 ZmApp.prototype._batchNotify =
982 function(mods) {
983 
984 	if (!(mods && mods.length >= ZmApp.BATCH_NOTIF_LIMIT)) { return; }
985 
986 	var notifs = {}, item, gotOne = false;
987 	for (var i = 0, len = mods.length; i < len; i++) {
988 		var mod = mods[i];
989 		item = appCtxt.cacheGet(mod.id);
990 		if (item) {
991 			var ev = item.notifyModify(mod, true);
992 			if (ev) {
993 				if (!notifs[ev]) {
994 					notifs[ev] = [];
995 				}
996 				mod.item = item;
997 				notifs[ev].push(mod);
998 				gotOne = true;
999 			}
1000 		}
1001 	}
1002 
1003 	if (!gotOne || !item) { return; }
1004 
1005 	var list = item.list;
1006 	if (!list) { return; }
1007 	list._evt.batchMode = true;
1008 	list._evt.item = item;	// placeholder - change listeners like it to be there
1009 	list._evt.items = null;
1010 	for (var ev in notifs) {
1011 		var details = {notifs:notifs[ev]};
1012 		list._notify(ev, details);
1013 	}
1014 };
