 16 /**
 17  * @overview
 18  * 
 19  * This file defines an application event.
 20  *
 21  */
 22 
 23 /**
 24  * Creates an empty application event.
 25  * @class
 26  * This class represents an event related to a change of state for an individual
 27  * application or for ZCS as a whole.
 28  * 
 29  * @param {Object}	the application to which this event applies; if <code>null</code>, the event applies to ZCS
 30  * 
 31  * @extends		ZmEvent
 32  */
 33 ZmAppEvent = function(app) {
 34 	ZmEvent.call(this);
 35 };
 36 
 37 ZmAppEvent.prototype = new ZmEvent;
 38 ZmAppEvent.prototype.constructor = ZmAppEvent;
 39 
 40 /**
 41  * Event used to notify listeners before startup (i.e. before the first
 42  * app is activated). This is a bit of a misnomer because this event occurs
 43  * after the apps are initialized but before the first app is shown. This
 44  * allows code to be executed after the apps have registered settings
 45  * but before the app actually acts on those settings.
 46  *
 47  * @see ZmAppEvent.POST_STARTUP
 48  */
 49 ZmAppEvent.PRE_STARTUP	= "PRESTARTUP";
 50 
 51 /**
 52  * Defines the event used to notify listeners post-startup.
 53  */
 54 ZmAppEvent.POST_STARTUP	= "POSTSTARTUP";
 55 /**
 56  * Defines the event used to notify listeners pre-launch.
 57  */
 58 ZmAppEvent.PRE_LAUNCH	= "PRELAUNCH";
 59 /**
 60  * Defines the event used to notify listeners post-launch.
 61  */
 62 ZmAppEvent.POST_LAUNCH	= "POSTLAUNCH";
 63 /**
 64  * Defines the event used to notify listeners post-render.
 65  */
 66 ZmAppEvent.POST_RENDER	= "POSTRENDER";
 67 
 68 ZmAppEvent.ACTIVATE	= "ACTIVATE";
 69 
 70 // Triggered after processing of an async response finishes
 71 ZmAppEvent.RESPONSE = "RESPONSE";
 72 
 73 /**
 74  * Returns a string representation of the object.
 75  * 
 76  * @return		{String}		a string representation of the object
 77  */
 78 ZmAppEvent.prototype.toString =
 79 function() {
 80 	return "ZmAppEvent";
 81 };
