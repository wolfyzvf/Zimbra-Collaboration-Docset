 20 /**
 21  * Creates a new search dialog.
 22  * @class
 23  * This class represents a new search dialog.
 24  * 
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		ZmNewOrganizerDialog
 29  */
 30 ZmNewSearchDialog = function(parent, className) {
 31 
 32 	ZmNewOrganizerDialog.call(this, parent, className, ZmMsg.saveSearch, ZmOrganizer.SEARCH);
 33 
 34 };
 35 
 36 ZmNewSearchDialog.prototype = new ZmNewOrganizerDialog;
 37 ZmNewSearchDialog.prototype.constructor = ZmNewSearchDialog;
 38 
 39 ZmNewSearchDialog.prototype.toString = 
 40 function() {
 41 	return "ZmNewSearchDialog";
 42 };
 43 
 44 //overriden properties
 45 
 46 ZmNewSearchDialog.prototype._folderLocationLabel = ZmMsg.newSearchParent;
 47 
 48 
 49 /**
 50  * Pops-up the dialog.
 51  * 
 52  * @param	{Hash}	params		a hash of parameters
 53  * @param	{String}	params.search		the search
 54  * @param	{Boolean}	params.showOverview	if <code>true</code>, make the overview visible
 55  * 
 56  */
 57 ZmNewSearchDialog.prototype.popup =
 58 function(params) {
 59 	ZmNewOrganizerDialog.prototype.popup.call(this, params);
 60 
 61 	var account = appCtxt.multiAccounts ? appCtxt.getActiveAccount() : null;
 62 
 63 	var ov = this._getOverviewOrOverviewContainer();
 64 	
 65 	this._folderTreeView = ov.getTreeView(ZmOrganizer.FOLDER);
 66 	this._searchTreeView = ov.getTreeView(ZmOrganizer.SEARCH);
 67 	this._search = params.search;
 68 	this._searchTreeView.setSelected(appCtxt.getFolderTree(account).root, true);
 69 	this._isGlobalSearch = appCtxt.multiAccounts && appCtxt.getSearchController().searchAllAccounts;
 70 
 71 	if (appCtxt.multiAccounts) {
 72 		this._searchTreeView.setVisible(true);
 73 		this._makeOverviewVisible(this._curOverviewId);
 74 	}
 75 
 76 	var overviewDiv = document.getElementById(this._folderTreeCellId);
 77 	if (overviewDiv) {
 78 		Dwt.setVisible(overviewDiv, (params.showOverview && !this._isGlobalSearch));
 79 	}
 80 
 81 };
 82 
 83 
 84 ZmNewSearchDialog.prototype._getFolderData =
 85 function() {
 86 
 87 	var ret = ZmNewOrganizerDialog.prototype._getFolderData.call(this);
 88 	if (!ret) {
 89 		return;
 90 	}
 91 
 92 	ret.isGlobal = this._isGlobalSearch;
 93 	ret.search = this._search;
 94 
 95 	return ret;
 96 };
 97

104     ZmNewOrganizerDialog.prototype._setupFolderControl.call(this);
105 	this._treeIds = [ZmOrganizer.FOLDER, ZmOrganizer.SEARCH];
106 };
