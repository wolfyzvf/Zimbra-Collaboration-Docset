 58 /**
 59  * Converts the given text to mixed-case. The input text is one or more words
 60  * separated by spaces. The output is a single word in mixed (or camel) case.
 61  * 
 62  * @param {string}	text		text to convert
 63  * @param {string|RegEx}	sep		text separator (defaults to any space)
 64  * @param {boolean}	camel		if <code>true</code>, first character of result is lower-case
 65  * @return	{string}	the resulting string
 66  */
 67 AjxStringUtil.toMixed =
 68 function(text, sep, camel) {
 69 	if (!text || (typeof text != "string")) { return ""; }
 70 	sep = sep || /\s+/;
 71 	var wds = text.split(sep);
 72 	var newText = [];
 73 	newText.push(camel ? wds[0].toLowerCase() : wds[0].substring(0, 1).toUpperCase() + wds[0].substring(1).toLowerCase());
 74 	for (var i = 1; i < wds.length; i++) {
 75 		newText.push(wds[i].substring(0, 1).toUpperCase() + wds[i].substring(1).toLowerCase());
 76 	}
 77 	return newText.join("");
 78 };

 93 /**
 94  * Removes white space from the beginning and end of a string, optionally compressing internal white space. By default, white
 95  * space is defined as a sequence of  Unicode whitespace characters (\s in regexes). Optionally, the user can define what
 96  * white space is by passing it as an argument.
 97  *
 98  * <p>TODO: add left/right options</p>
 99  *
100  * @param {string}	str      	the string to trim
101  * @param {boolean}	compress 	whether to compress internal white space to one space
102  * @param {string}	space    	a string that represents a user definition of white space
103  * @return	{string}	a trimmed string
104  */
105 AjxStringUtil.trim =
106 function(str, compress, space) {
107 
108 	if (!str) {return "";}
109 
110 	var trim_re = AjxStringUtil.TRIM_RE;
111 
112 	var compress_re = AjxStringUtil.COMPRESS_RE;
113 	if (space) {
114 		trim_re = new RegExp("^" + space + "+|" + space + "+$", "g");
115 		compress_re = new RegExp(space + "+", "g");
116 	} else {
117 		space = " ";
118 	}
119 	str = str.replace(trim_re, '');
120 	if (compress) {
121 		str = str.replace(compress_re, space);
122 	}
123 
124 	return str;
125 };

164 /**
165 * Splits a string, ignoring delimiters that are in quotes or parentheses. Comma
166 * is the default split character, but the user can pass in a string of multiple
167 * delimiters. It can handle nested parentheses, but not nested quotes.
168 *
169 * <p>TODO: handle escaped quotes</p>
170 *
171 * @param {string} str	the string to split
172 * @param {string}	[dels]	an optional string of delimiter characters
173 * @return	{array}	an array of strings
174 */
175 AjxStringUtil.split =
176 function(str, dels) {
177 
178 	if (!str) {return [];}
179 	var i = 0;
180 	dels = dels ? dels : ',';
181 	var isDel = new Object();
182 	if (typeof dels == 'string') {
183 		isDel[dels] = 1;
184 	} else {
185 		for (i = 0; i < dels.length; i++) {
186 			isDel[dels[i]] = 1;
187 		}
188 	}
189 
190 	var q = false;
191 	var p = 0;
192 	var start = 0;
193 	var chunk;
194 	var chunks = [];
195 	var j = 0;
196 	for (i = 0; i < str.length; i++) {
197 		var c = str.charAt(i);
198 		if (c == '"') {
199 			q = !q;
200 		} else if (c == '(') {
201 			p++;
202 		} else if (c == ')') {
203 			p--;
204 		} else if (isDel[c]) {
205 			if (!q && !p) {
206 				chunk = str.substring(start, i);
207 				chunks[j++] = chunk;
208 				start = i + 1;
209 			}
210 		}
211 	}
212 	chunk = str.substring(start, str.length);
213 	chunks[j++] = chunk;
214 
215 	return chunks;
216 };

239 /**
240  * Wraps text to the given length and optionally quotes it. The level of quoting in the
241  * source text is preserved based on the prefixes. Special lines such as email headers
242  * always start a new line.
243  *
244  * @param {hash}	params	a hash of parameters
245  * @param {string}      params.text 				the text to be wrapped
246  * @param {number}      [params.len=80]				the desired line length of the wrapped text, defaults to 80
247  * @param {string}      [params.pre]				an optional string to prepend to each line (useful for quoting)
248  * @param {string}      [params.before]				text to prepend to final result
249  * @param {string}      [params.after]				text to append to final result
250  * @param {boolean}		[params.preserveReturns]	if true, don't combine small lines
251  *
252  * @return	{string}	the wrapped/quoted text
253  */
254 AjxStringUtil.wordWrap =
255 function(params) {
256 
257 	if (!(params && params.text)) { return ""; }
258 
259 	var text = params.text;
260 	var before = params.before || "", after = params.after || "";
261 
262 	// For HTML, just surround the content with the before and after, which is
263 	// typically a block-level element that puts a border on the left
264 	if (params.htmlMode) {
265 		return [before, text, after].join("");
266 	}
267 
268 	var len = params.len || 80;
269 	var pre = params.pre || "";
270 	var eol = "\n";
271 
272 	text = AjxStringUtil.trim(text, false, "[\t ]");
273 	text = text.replace(/\r\n/g, eol);
274 	var lines = text.split(eol);
275 	var words = [];
276 
277 	// Divides lines into words. Each word is part of a hash that also has
278 	// the word's prefix, whether it's a paragraph break, and whether it's
279 	// special (cannot be wrapped into a previous line)
280 	for (var l = 0, llen = lines.length; l < llen; l++) {
281 		var line = lines[l];
282 		// get this line's prefix
283 		var m = line.match(/^([\s>\|]+)/);
284 		var prefix = m ? m[1] : "";
285 		if (prefix) {
286 			line = line.substr(prefix.length);
287 		}
288 		var wds = AjxStringUtil.splitKeepLeadingWhitespace(line);
289 		if (wds && wds[0] && wds[0].length) {
290 			var isSpecial = AjxStringUtil.MSG_SEP_RE.test(line) || AjxStringUtil.COLON_RE.test(line) ||
291 							AjxStringUtil.HDR_RE.test(line) || AjxStringUtil.SIG_RE.test(line);
292 			for (var w = 0, wlen = wds.length; w < wlen; w++) {
293 				var lastWord = params.preserveReturns && (w == wlen - 1);
294 				words.push({w:wds[w], p:prefix, special:(isSpecial && w == 0), lastWord:lastWord});
295 			}
296 		} else {
297 			words.push({para:true, p:prefix});	// paragraph marker
298 		}
299 	}
300 
301 	// Take the array of words and put them back together. We break for a new line
302 	// when we hit the max line length, change prefixes, or hit a special word.
303 	var max = params.len || 72;
304 	var addPrefix = params.pre || "";
305 	var apl = addPrefix.length;
306 	var result = "", curLen = 0, wds = [], curP = null;
307 	for (var i = 0, len = words.length; i < len; i++) {
308 		var word = words[i];
309 		var w = word.w, p = word.p;
310 		var sp = wds.length ? 1 : 0;
311 		var pl = (curP === null) ? 0 : curP.length;
312 		if (word.para) {
313 			// paragraph break - output what we have, add a blank line
314 			if (wds.length) {
315 				result += addPrefix + (curP || "") + wds.join("").replace(/^ +/,"") + eol;
316 			}
317 			result += addPrefix + p + eol;
318 			wds = [];
319 			curLen = 0;
320 			curP = null;
321 		} else if ((apl + pl + curLen + sp + w.length <= max) && (p == curP || curP === null) && !word.special) {
322 			// still room left on the current line, add the word
323 			wds.push(w);
324 			curLen += w.length;
325 			curP = p;
326 			if (word.lastWord && words[i + 1]) {
327 				words[i + 1].special = true;
328 			}
329 		} else {
330 			// output what we have and start a new line
331 			if (wds.length) {
332 				result += addPrefix + (curP || "") + wds.join("").replace(/^ +/,"") + eol;
333 			}
334 			wds = [w];
335 			curLen = w.length;
336 			curP = p;
337 			if (word.lastWord && words[i + 1]) {
338 				words[i + 1].special = true;
339 			}
340 		}
341 	}
342 
343 	// handle last line
344 	if (wds.length) {
345 		result += addPrefix + curP + wds.join("").replace(/^ /,"") + eol;
346 	}
347 
348 	return [before, result, after].join("");
349 };

351 /**
352  * Quotes text with the given quote character. For HTML, surrounds the text with the
353  * given strings. Does no wrapping.
354  *
355  * @param {hash}	params	a hash of parameters
356  * @param {string}      params.text 				the text to be wrapped
357  * @param {string}      [params.pre]				prefix for quoting
358  * @param {string}      [params.before]				text to prepend to final result
359  * @param {string}      [params.after]				text to append to final result
360  *
361  * @return	{string}	the quoted text
362  */
363 AjxStringUtil.quoteText =
364 function(params) {
365 
366 	if (!(params && params.text)) { return ""; }
367 
368 	var text = params.text;
369 	var before = params.before || "", after = params.after || "";
370 
371 	// For HTML, just surround the content with the before and after, which is
372 	// typically a block-level element that puts a border on the left
373 	if (params.htmlMode || !params.pre) {
374 		return [before, text, after].join("");
375 	}
376 
377 	var len = params.len || 80;
378 	var pre = params.pre || "";
379 	var eol = "\n";
380 
381 	text = AjxStringUtil.trim(text);
382 	text = text.replace(/\n\r/g, eol);
383 	var lines = text.split(eol);
384 	var result = [];
385 
386 	for (var l = 0, llen = lines.length; l < llen; l++) {
387 		var line = AjxStringUtil.trim(lines[l]);
388 		result.push(pre + line + eol);
389 	}
390 
391 	return before + result.join("") + after;
392 };

627 /**
628  * Converts the string to HTML.
629  * 
630  * @param	{string}	str		the string
631  * @return	{string}	the resulting string
632  */
633 AjxStringUtil.convertToHtml =
634 function(str, quotePrefix) {
635 	if (!str) {return "";}
636 
637 	if (quotePrefix) {
638 		// Convert a section of lines prefixed with > or |
639 		// to a section encapsuled in <blockquote> tags
640 		var prefix_re = /^(>|\|\s+)/;
641 		var lines = str.split(/\r?\n/);

657 					level--;
658 				}
659 			}
660 			lines[i] = line;
661 		}
662 		while (level > 0) {

664 			level--;
665 		}

681 	}
682

685 		.replace(/  /mg, "  ")

822 /**
823  * Forces a string to wrap at "pixelWidth" using "className" on hidden 'AjxStringUtil._calcDIV'.
824  * Returns "origString" with "<br>" tags inserted to force wrapping.
825  * Breaks string on embedded space characters, EOL ("/n") and "<br>" tags when possible.
826  *
827  * @returns		"origString" with "<br>" tags inserted to force wrapping.
828  * 
829  * @private
830  */
831 AjxStringUtil.wrap =
832 function(origString, pixelWidth, className) {
833 	var calcDIV = AjxStringUtil.calcDIV();
834 	if (arguments.length == 3) calcDIV.className = className;
835 
836 	var newString = "";
837 	var newLine = "";
838 	var textRows = origString.split("/n");
839 	for (var trCount = 0; trCount < textRows.length; trCount++) {
840 		if (trCount != 0) {
841 			newString += newLine + "<br>";
842 			newLine = "";
843 		}
844 		htmlRows = textRows[trCount].split("<br>");
845 		for (var hrCount=0; hrCount<htmlRows.length; hrCount++) {
846 			if (hrCount != 0) {
847 				newString += newLine + "<br>";
848 				newLine = "";
849 			}
850 			words = htmlRows[hrCount].split(" ");
851 			var wCount=0;
852 			while (wCount<words.length) {
853 				calcDIV.innerHTML = newLine + " " + words[wCount];
854 				var newLinePixels = calcDIV.offsetWidth;
855 				if (newLinePixels > pixelWidth) {
856 					// whole "words[wCount]" won't fit on current "newLine" - insert line break, avoid incrementing "wCount"
857 					calcDIV.innerHTML = words[wCount];
858 					newLinePixels = newLinePixels - calcDIV.offsetWidth;
859 					if ( (newLinePixels >= pixelWidth) || (calcDIV.offsetWidth <= pixelWidth) ) {
860 						// either a) excess caused by <space> character or b) will fit completely on next line
861 						// so just break without incrementing "wCount" and append next time
862 						newString += newLine + "<br>";
863 						newLine = "";
864 					}
865 					else { // must break "words[wCount]"
866 						var keepLooping = true;
867 						var atPos = 0;
868 						while (keepLooping) {
869 							atPos++;
870 							calcDIV.innerHTML = newLine + " " + words[wCount].substring(0,atPos);
871 							keepLooping = (calcDIV.offsetWidth <= pixelWidth);
872 						}
873 						atPos--;
874 						newString += newLine + words[wCount].substring(0,atPos) + "<br>";
875 						words[wCount] = words[wCount].substr(atPos);
876 						newLine = "";
877 					}
878 				} else { // doesn't exceed pixelWidth, append to "newLine" and increment "wCount"
879 					newLine += " " + words[wCount];
880 					wCount++;
881 				}
882 			}
883 		}
884 	}
885 	newString += newLine;
886 	return newString;
887 };

1294 AjxStringUtil._prettyPrint =
1295 function(obj, recurse, showFuncs, omit) {
1296 
1297 	var indentLevel = 0;
1298 	var showBraces = false;
1299 	var stopRecursion = false;
1300 	if (arguments.length > 4) {
1301 		indentLevel = arguments[4];
1302 		showBraces = arguments[5];
1303 		stopRecursion = arguments[6];
1304 	}
1305 
1306 	if (AjxUtil.isObject(obj)) {
1307 		var objStr = obj.toString ? obj.toString() : "";
1308 		if (omit && objStr && omit[objStr]) {
1309 			return "[" + objStr + "]";
1310 		}
1311 		if (AjxStringUtil._visited.contains(obj)) {
1312 			return "[visited object]";
1313 		} else {
1314 			AjxStringUtil._visited.add(obj);
1315 		}
1316 	}
1317 
1318 	var indent = AjxStringUtil.repeat(" ", indentLevel);
1319 	var text = "";
1320 
1321 	if (obj === undefined) {
1322 		text += "[undefined]";
1323 	} else if (obj === null) {
1324 		text += "[null]";
1325 	} else if (AjxUtil.isBoolean(obj)) {
1326 		text += obj ? "true" : "false";
1327 	} else if (AjxUtil.isString(obj)) {
1328 		text += '"' + AjxStringUtil._escapeForHTML(obj) + '"';
1329 	} else if (AjxUtil.isNumber(obj)) {
1330 		text += obj;
1331 	} else if (AjxUtil.isObject(obj)) {
1332 		var isArray = AjxUtil.isArray(obj) || AjxUtil.isArray1(obj);

1340                     keys.push(i);
1341                 }
1342 			}
1343

1368 					}
1369 				}
1370 
1371 				if (i > 0) {

1373 				}

1377                     keyString = "// [" + key + "]:\n" + indent;
1378                 } else {

1380                 }

1382 					text += keyString + "[" + key + "]";
1383 				} else if (value != null) {

1385 				} else {

1387 				}
1388 			}
1389 			if (i > 0) {

1394 			}
1395 		}

1756 // Matches a line of text against some regexes to see if has structural meaning within a mail msg.
1757 AjxStringUtil._getLineType =
1758 function(testLine) {
1759 
1760 	var type = AjxStringUtil.ORIG_UNKNOWN;
1761 	
1762 	// see if the line matches any known delimiters or quote patterns
1763 	for (var j = 0; j < AjxStringUtil.MSG_REGEXES.length; j++) {
1764 		var msgTest = AjxStringUtil.MSG_REGEXES[j];
1765 		var regex = msgTest.regex;
1766 		if (regex.test(testLine.toLowerCase())) {
1767 			// line that starts and ends with | is considered ASCII art (eg a table) rather than quoted
1768 			if (msgTest.type == AjxStringUtil.ORIG_QUOTED && /^\s*\|.*\|\s*$/.test(testLine)) {
1769 				continue;
1770 			}
1771 			type = msgTest.type;
1772 			break;	// first match wins
1773 		}
