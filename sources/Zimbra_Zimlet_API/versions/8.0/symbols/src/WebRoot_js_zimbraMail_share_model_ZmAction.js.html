 20 /**
 21  * @class
 22  * Represents an undoable action (e.g. move an item)
 23  * This class is a generic superclass that does very little on its own; the real work is being done in subclasses

177 ZmItemMoveAction.prototype._doMove = function(callback, errorCallback, folderId) {
178 	this._item.list.moveItems({
179 		items:			[this._item],
180 		folder:			appCtxt.getById(folderId),
181 		noUndo:			true,
182 		finalCallback:	this._handleDoMove.bind(this, this._item.folderId, folderId),
183 		actionText:		ZmItemMoveAction.UNDO_MSG[this._op],
184 
185 		fromFolderId: this._toFolderId
186 	});
187 };
188 
189 ZmItemMoveAction.prototype._handleDoMove = function(oldFolderId, newFolderId, params) {
190 	var lists = [];
191 	for (var id in params.idHash) {
192 		var item = params.idHash[id];
193 		if (item instanceof ZmConv)
194 			item.folderId = newFolderId;
195 		var list = item && item.list;
196 		if (AjxUtil.indexOf(lists, list)==-1)
197 			lists.push(list);
198 	}
199 	for (var i=0; i<lists.length; i++) {
200 		lists[i]._notify(ZmEvent.E_MOVE, {oldFolderId:oldFolderId});
201 	}
202 	ZmListController.handleProgress({state:ZmListController.PROGRESS_DIALOG_CLOSE});
203 	ZmBaseController.showSummary(params.actionSummary);
204 };
205 
206 ZmItemMoveAction.prototype.undo = function(callback, errorCallback) {
207 	this._doMove(callback, errorCallback, this._fromFolderId);
208 };
209 
210 ZmItemMoveAction.prototype.redo = function(callback, errorCallback) {
211 	this._doMove(callback, errorCallback, this._toFolderId);
212 };

214 ZmItemMoveAction.multipleUndo = function(actions, redo, fromFolderId) {
215 	var masterAction = actions && actions.length && actions[0];
216 	var sortingTable = {};
217 	for (var i=0; i<actions.length; i++) {
218 		var action = actions[i];
219 		if (action instanceof ZmItemMoveAction) {
220 			var from = action.getFromFolderId();
221 			var to = action.getToFolderId();
222 			var item = action.getItem();
223 			var type = (item && item.list && item.list.type) || 0;
224 			if (!sortingTable[from]) sortingTable[from] = {};
225 			if (!sortingTable[from][to]) sortingTable[from][to] = {};
226 			if (!sortingTable[from][to][type]) sortingTable[from][to][type] = [];
227 			sortingTable[from][to][type].push(action);
228 		}
229 	}
230 	for (var from in sortingTable) {
231 		for (var to in sortingTable[from]) {
232 			for (var type in sortingTable[from][to]) {
233 				var subset = sortingTable[from][to][type];
234 				var items = [];
235 				var list = null;
236 				var hasMasterAction = false;
237 				var commonop;
238 				for (var i=0; i<subset.length; i++) {
239 					var action = subset[i];
240 					if (action == masterAction)
241 						hasMasterAction = true;
242 					var item = action.getItem();
243 					items.push(item);
244 					if (!list && item.list)
245 						list = item.list;
246 					var op = action.getOp && action.getOp();
247 					if (!commonop)
248 						commonop = op;
249 					else if (commonop != op)
250 						commonop = "move";
251 				}
252 				if (list) {
253 					list.moveItems({
254 						items: items,
255 						folder: appCtxt.getById(redo ? to : from),
256 						noUndo: true,
257 						fromFolderId: fromFolderId,
258 						actionText: hasMasterAction ? (commonop && ZmItemMoveAction.UNDO_MSG[commonop]) : null
259 					});
260 				}
261 			}
262 		}
263 	}
264 };

266 ZmItemMoveAction.multipleRedo = function(actions) {
267 	ZmItemMoveAction.multipleUndo(actions, true);

320 ZmOrganizerMoveAction.multipleUndo = function(actions, redo) {
321 	for (var i=0; i<actions.length; i++) {
322 		var action = actions[i];
323 		if (action instanceof ZmOrganizerMoveAction) {
324 			action._doMove(null, null, redo ? this._toFolderId : this._fromFolderId);
325 		}
326 	}

333 /**
334  * @class
335  * Represents a collection of undoable actions that will be performed as a whole
336  * 
337  * @extends		ZmAction

339  */

341 ZmCompositeAction = function(toFolderId) {
342 	ZmAction.call(this);
343 	this._actions = {};
344 	this._toFolderId = toFolderId;
345 };
346 
347 ZmCompositeAction.prototype = new ZmAction;
348 ZmCompositeAction.prototype.constructor = ZmCompositeAction;
349 ZmCompositeAction.prototype.type = ZmAction.ACTION_ZMCOMPOSITEACTION;
350 
351 ZmCompositeAction.prototype.toString = function() {
352 	return "ZmCompositeAction";
353 };
354 
355 /**
356  * Add an action the the collection
357  *
358  * @param	{ZmAction}	action	An action to add
359  */
360 ZmCompositeAction.prototype.addAction = function(action) {
361 	if (action && action!=this && action instanceof ZmAction) {
362 		var type = action.type;
363 		if (!this._actions[type])
364 			this._actions[type] = [];
365 		this._actions[type].push(action);
366 	}
367 };
368 
369 ZmCompositeAction.prototype.getActions = function(type) {
370 	return this._actions[type] || [];
371 };
372 
373 ZmCompositeAction.prototype.hasActions = function(type) {
374 	return this._actions[type] && this._actions[type].length>0;
375 };
376 
377 ZmCompositeAction.prototype.undo = function(callback, errorCallback) {
378 
379 	if (this.hasActions(ZmAction.ACTION_ZMITEMMOVEACTION)) {
380 		ZmItemMoveAction.multipleUndo(this.getActions(ZmAction.ACTION_ZMITEMMOVEACTION), null, this._toFolderId);
381 	}
382 
383 	if (this.hasActions(ZmAction.ACTION_ZMORGANIZERMOVEACTION)) {
384 		ZmOrganizerMoveAction.multipleUndo(this.getActions(ZmAction.ACTION_ZMORGANIZERMOVEACTION));
385 	}
386 
387 	if (this.hasActions(ZmAction.ACTION_ZMCOMPOSITEACTION) || this.hasActions(ZmAction.ACTION_ZMITEMACTION)) {
388 		var actions = this.getActions(ZmAction.ACTION_ZMCOMPOSITEACTION).concat(this.getActions(ZmAction.ACTION_ZMITEMACTION));
389 		for (var i=0; i<actions.length; i++) {
390 			actions[i].undo();
391 		}
392 	}
393 };
394 
395 ZmCompositeAction.prototype.redo = function(callback, errorCallback) {
396 
397 	if (this.hasActions(ZmAction.ACTION_ZMITEMMOVEACTION)) {
398 		ZmItemMoveAction.multipleRedo(this.getActions(ZmAction.ACTION_ZMITEMMOVEACTION));
399 	}
400 
401 	if (this.hasActions(ZmAction.ACTION_ZMORGANIZERMOVEACTION)) {
402 		ZmOrganizerMoveAction.multipleRedo(this.getActions(ZmAction.ACTION_ZMORGANIZERMOVEACTION));
403 	}
404 
405 	if (this.hasActions(ZmAction.ACTION_ZMCOMPOSITEACTION) || this.hasActions(ZmAction.ACTION_ZMITEMACTION) || this.hasActions(ZmAction.ACTION_ZMORGANIZERACTION)) {
406 		var actions = this.getActions(ZmAction.ACTION_ZMCOMPOSITEACTION).concat(this.getActions(ZmAction.ACTION_ZMITEMACTION)).concat(this.getActions(ZmAction.ACTION_ZMORGANIZERACTION));
407 		for (var i=0; i<actions.length; i++) {
408 			actions[i].redo();
409 		}
410 	}
411 };
