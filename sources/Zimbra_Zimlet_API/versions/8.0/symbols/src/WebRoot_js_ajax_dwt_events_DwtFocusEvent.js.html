 20 DwtFocusEvent = function(init) {
 21 	if (arguments.length == 0) return;
 22 	DwtEvent.call(this, true);
 23 	this.reset();
 24 }
 25 DwtFocusEvent.prototype = new DwtEvent;
 26 DwtFocusEvent.prototype.constructor = DwtFocusEvent;
 27 
 28 DwtFocusEvent.FOCUS = 1;
 29 DwtFocusEvent.BLUR = 2;
 30 
 31 DwtFocusEvent.prototype.toString = 
 32 function() {
 33 	return "DwtFocusEvent";
 34 }
 35 
 36 DwtFocusEvent.prototype.reset = 
 37 function() {
 38 	this.dwtObj = null;
 39 	this.state = DwtFocusEvent.FOCUS;
 40 }
