 16 /**
 17  * Creates the briefcase icon view.
 18  * @class
 19  * This class represents the briefcase icon view.
 20  * 
 21  * @param	{Hash}	params		a hash of parameters
 22  * 
 23  * @extends		ZmBriefcaseBaseView
 24  */
 25 ZmBriefcaseIconView = function(params) {
 26 	ZmBriefcaseBaseView.call(this, params);
 27 	this.getHtmlElement().style.backgroundColor = "white";
 28 }
 29 
 30 ZmBriefcaseIconView.prototype = new ZmBriefcaseBaseView;
 31 ZmBriefcaseIconView.prototype.constructor = ZmBriefcaseIconView;
 32 
 33 /**
 34  * Returns a string representation of the object.
 35  * 
 36  * @return		{String}		a string representation of the object
 37  */
 38 ZmBriefcaseIconView.prototype.toString =
 39 function() {
 40 	return "ZmBriefcaseIconView";
 41 };
 42 
 43 // Data
 44 ZmBriefcaseIconView.prototype._createItemHtml =
 45 function(item, params) {
 46 	
 47 	var name = item.name;
 48 	var contentType = item.contentType;
 49 	
 50 	if(contentType && contentType.match(/;/)) {
 51 			contentType = contentType.split(";")[0];
 52 	}
 53 	var mimeInfo = contentType ? ZmMimeTable.getInfo(contentType) : null;
 54 	icon = "Img" + ( mimeInfo ? mimeInfo.imageLarge : "UnknownDoc_48");
 55 
 56 	if(item.isFolder) {
 57 		icon = "ImgBriefcase_48";
 58 	}
 59 	
 60 	if(name.length>14){
 61 		name = name.substring(0,14)+"...";
 62 	}
 63 	
 64 	var div = document.createElement("div");
 65 	div.className = "ZmBriefcaseItemSmall";
 66 	
 67 	var htmlArr = [];
 68 	var idx = 0;
 69 
 70 	var icon = null;
 71 	if (!icon) {
 72 		var contentType = item.contentType;
 73 		if(contentType && contentType.match(/;/)) {
 74 			contentType = contentType.split(";")[0];
 75 		}
 76 		var mimeInfo = contentType ? ZmMimeTable.getInfo(contentType) : null;
 77 		icon = mimeInfo ? mimeInfo.image : "UnknownDoc" ;
 78 		if(item.isFolder){
 79 			icon = "Folder";
 80 		}
 81 	}
 82 	
 83 	htmlArr[idx++] = "<table><tr>";
 84     if (appCtxt.get(ZmSetting.SHOW_SELECTION_CHECKBOX)) {
 85         htmlArr[idx++] = "<td>";
 86         idx = this._getImageHtml(htmlArr, idx, "CheckboxUnchecked", this._getFieldId(item, ZmItem.F_SELECTION));
 87         htmlArr[idx++] = "</td>";
 88     }
 89     htmlArr[idx++] = "<td><div class='Img";
 90 	htmlArr[idx++] = icon;
 91 	htmlArr[idx++] = "'></div></td><td nowrap>";
 92 	htmlArr[idx++] = AjxStringUtil.htmlEncode(item.name);
 93 	htmlArr[idx++] = "</td><tr></table>";
 94 	
 95 	if (params && params.isDragProxy) {
 96 		Dwt.setPosition(div, Dwt.ABSOLUTE_STYLE);
 97 	}
 98 	div.innerHTML = htmlArr.join("");
 99 	
100 	this.associateItemWithElement(item, div);
101 	return div;
102 };
103

105 function(list, sortField, doNotIncludeFolders){
106 
107     doNotIncludeFolders = true;
108 
109     ZmBriefcaseBaseView.prototype.set.call(this, list, sortField, doNotIncludeFolders);
110 
111 };
