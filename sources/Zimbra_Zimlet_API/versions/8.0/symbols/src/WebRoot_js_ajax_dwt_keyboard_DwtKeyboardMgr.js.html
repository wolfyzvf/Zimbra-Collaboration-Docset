 16 /**
 17  * Creates an empty keyboard manager. Tab groups must be added for it to do
 18  * anything.
 19  * @constructor
 20  * @class
 21  * This class is responsible for managing keyboard events. That includes dispatching
 22  * keyboard events, as well as managing focus and tab groups. It is at the heart of the
 23  * Dwt keyboard navigation framework.
 24  * <p>
 25  * {@link DwtKeyboardMgr} intercepts key strokes and translates
 26  * them into actions which it then dispatches to the component with focus. If the key
 27  * stroke is a TAB (or Shift-TAB), then focus is moved based on the current tab group.
 28  * </p><p>
 29  * A {@link DwtShell} instantiates its own <i>DwtKeyboardMgr</i> at construction.
 30  * The keyboard manager may then be retrieved via the shell's <code>getKeyboardMgr()</code>
 31  * function. Once a handle to the shell's keyboard manager is retrieved, then the user is free
 32  * to add tab groups and register keymaps and handlers with the keyboard manager.
 33  * </p><p>
 34  * Focus is managed among a stack of tab groups. The TAB button will move the focus within the
 35  * current tab group. When a non-TAB is received, we first check if the control can handle it.
 36  * In general, control key events simulate something the user could do with the mouse, and change
 37  * the state/appearance of the control. For example, ENTER on a DwtButton simulates a button
 38  * press. If the control does not handle the key event, the event is handed to the application,
 39  * which handles it based on its current state. The application key event handler is in a sense
 40  * global, since it does not matter which control received the event.
 41  * </p><p>
 42  * At any given time there is a default handler, which is responsible for determining what
 43  * action is associated with a particular key sequence, and then taking it. A handler should support
 44  * the following methods:
 45  * 
 46  * <ul>
 47  * <li><i>getKeyMapName()</i> -- returns the name of the map that defines shortcuts for this handler</li>
 48  * <li><i>handleKeyAction()</i> -- performs the action associated with a shortcut</li>
 49  * <li><i>handleKeyEvent()</i>	-- override; handler solely responsible for handling event</li>
 50  * </ul>
 51  * </p>
 52  *
 53  * @author Ross Dargahi
 54  *
 55  * @param	{DwtShell}	shell		the shell
 56  * @see DwtShell
 57  * @see DwtTabGroup
 58  * @see DwtKeyMap
 59  * @see DwtKeyMapMgr
 60  * 
 61  * @private
 62  */
 63 DwtKeyboardMgr = function(shell) {
 64 	DwtKeyboardMgr.__shell = shell;
 65 	this.__tabGrpStack = [];
 66 	this.__defaultHandlerStack = [];
 67 	this.__tabGroupChangeListenerObj = new AjxListener(this, this.__tabGrpChangeListener);
 68 	this.__kbEventStatus = DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
 69 	this.__keyTimeout = 750;
 70 	this.__currTabGroup = null;
 71 	this.__currDefaultHandler = null;
 72 	this._evtMgr = new AjxEventMgr();
 73 };

 87 /**
 88  * Checks if the event may be a shortcut from within an input (text input or
 89  * textarea). Since printable characters are echoed, the shortcut must be non-printable:
 90  * 
 91  * <ul>
 92  * <li>Alt or Ctrl or Meta plus another key</li>
 93  * <li>Esc</li>
 94  * </ul>
 95  * 
 96  * @param {DwtKeyEvent}	ev	the key event
 97  * @return	{boolean}	<code>true</code> if the event may be a shortcut
 98  */
 99 DwtKeyboardMgr.isPossibleInputShortcut =
100 function(ev) {
101 	var target = DwtUiEvent.getTarget(ev);
102     return (!DwtKeyMap.IS_MODIFIER[ev.keyCode] && (ev.keyCode == 27 || DwtKeyMapMgr.hasModifier(ev)) ||
103 			(target && target.nodeName.toUpperCase() == "INPUT" && (ev.keyCode == 13 || ev.keyCode == 3)));
104 };

106 /**
107  * Pushes the tab group onto the stack and makes it the active tab group.
108  * 
109  * @param 	{DwtTabGroup}	tabGroup	the tab group to push onto the stack
110  * 
111  * @see		#popTabGroup
112  */
113 DwtKeyboardMgr.prototype.pushTabGroup =
114 function(tabGroup) {
115 //	DBG.println("kbnav", "PUSH tab group " + tabGroup.__name);
116 	if (!this.__keyboardHandlingInited || !tabGroup) { return; }
117 		
118 	this.__tabGrpStack.push(tabGroup);
119 	this.__currTabGroup = tabGroup;
120 	var focusMember = tabGroup.getFocusMember();
121 	if (!focusMember) {
122 		focusMember = tabGroup.resetFocusMember(true);
123 	}
124 	if (!focusMember) {
125 //		DBG.println("kbnav", "DwtKeyboardMgr.pushTabGroup: tab group " + tabGroup.__name + " has no members!");
126 		return;
127 	}
128 	tabGroup.addFocusChangeListener(this.__tabGroupChangeListenerObj);
129 	this.grabFocus(focusMember);
130 };

132 /**
133  * Pops the current tab group off the top of the tab group stack. The previous 
134  * tab group (if there is one) then becomes the current tab group.
135  * 
136  * @param {DwtTabGroup} [tabGroup]		the tab group to pop. If supplied, then the tab group
137  * 		stack is searched for the tab group and it is removed. If <code>null</code>, then the
138  * 		top tab group is popped.
139  * 
140  * @return {DwtTabGroup}	the popped tab group or <code>null</code> if there is one or less tab groups
141  */
142 DwtKeyboardMgr.prototype.popTabGroup =
143 function(tabGroup) {
144 	if (!this.__keyboardHandlingInited) { return; }
145 	if (!tabGroup) return;
146 //	DBG.println("kbnav", "POP tab group " + tabGroup.__name);
147 	
148 	// we never want an empty stack
149 	if (this.__tabGrpStack.length <= 1) {
150 		return null;
151 	}
152 	
153 	/* If we are popping a tab group that is not on the top of the stack then 
154 	 * we need to find it and remove it. */
155 	if (tabGroup && this.__tabGrpStack[this.__tabGrpStack.length - 1] != tabGroup) {
156 		var a = this.__tabGrpStack;
157 		var len = a.length;
158 		for (var i = len - 1; i >= 0; i--) {
159 			if (tabGroup == a[i]) {
160 				a[i].dump();
161 				break;
162 			}
163 		}
164 		
165 		/* If there is no match in the stack for tabGroup, then simply return null,
166 		 * else if the match is not the top item on the stack, then remove it from 
167 		 * the stack. Else we are dealing with the topmost item on the stack so handle it 
168 		 * as a simple pop. */
169 		if (i < 0) { // No match
170 			return null;
171 		} else if (i != len - 1) { // item is not on top
172 			// Remove tabGroup
173 			a.splice(i, 1);
174 			return tabGroup;
175 		}
176 	} 
177 
178 	var tabGroup = this.__tabGrpStack.pop();
179 	tabGroup.removeFocusChangeListener(this.__tabGroupChangeListenerObj);
180 	
181 	var currTg = null;
182 	if (this.__tabGrpStack.length > 0) {
183 		currTg = this.__tabGrpStack[this.__tabGrpStack.length - 1];
184 		var focusMember = currTg.getFocusMember();
185 		if (!focusMember) {
186 			focusMember = currTg.resetFocusMember(true);
187 		}
188 		if (focusMember) {
189 			this.grabFocus(focusMember);
190 		}
191 	}
192 	this.__currTabGroup = currTg;
193 
194 	return tabGroup;
195 };

197 /**
198  * Replaces the current tab group with the given tab group.
199  * 
200  * @param {DwtTabGroup} tabGroup 	the tab group to use
201  * @return {DwtTabGroup}	the old tab group
202  */
203 DwtKeyboardMgr.prototype.setTabGroup =
204 function(tabGroup) {
205 	if (!this.__enabled || !this.__keyboardHandlingInited) { return; }
206 	
207 	var otg = this.popTabGroup();
208 	this.pushTabGroup(tabGroup);
209 	return otg;
210 };

223 function(handler) {
224 	if (!this.__enabled || !this.__keyboardHandlingInited || !handler) { return; }

226 		
227 	this.__defaultHandlerStack.push(handler);

231 DwtKeyboardMgr.prototype.popDefaultHandler =
232 function() {
233 //	DBG.println("kbnav", "POP default handler");
234 	// we never want an empty stack
235 	if (!this.__keyboardHandlingInited || (this.__defaultHandlerStack.length <= 1)) { return; }
236 
237 //	DBG.println("kbnav", "Default handler stack length: " + this.__defaultHandlerStack.length);
238 	var handler = this.__defaultHandlerStack.pop();
239 	this.__currDefaultHandler = this.__defaultHandlerStack[this.__defaultHandlerStack.length - 1];
240 //	DBG.println("kbnav", "Default handler is now: " + this.__currDefaultHandler);
241 
242 	return handler;
243 };

245 /**
246  * Sets the focus to the given object.
247  * 
248  * @param {HTMLInputElement|DwtControl|string} focusObj		the object to which to set focus, or its ID
249  */ 
250 DwtKeyboardMgr.prototype.grabFocus =
251 function(focusObj) {
252 	if (!this.__enabled) { return; }
253 	if (!this.__keyboardHandlingInited) {
254 		return;
255 	}
256 	if (typeof focusObj == "string") {
257 		focusObj = document.getElementById(focusObj);
258 	}
259 	if (!focusObj) return;
260 
261 	/* We may not be using tab groups, so be prepared for that case */
262 	if (this.__currTabGroup) {
263 		this.__currTabGroup.setFocusMember(focusObj, false, true);
264 	}
265 		
266 	this.__doGrabFocus(focusObj);
267 };

365 /**
366  * Enables/disables keyboard nav.
367  * 
368  * @param 	{boolean}	enabled		if <code>true</code>, enable keyboard nav
369  */
370 DwtKeyboardMgr.prototype.enable =
371 function(enabled) {
372 	DBG.println(AjxDebug.DBG2, "keyboard nav enabled: " + enabled);
373 	this.__enabled = enabled;
374 	if (enabled){
375 		this.__checkStatus();	// make sure we're initialized
376 		Dwt.setHandler(document, DwtEvent.ONKEYDOWN, DwtKeyboardMgr.__keyDownHdlr);
377 		Dwt.setHandler(document, DwtEvent.ONKEYUP, DwtKeyboardMgr.__keyUpHdlr);
378 		Dwt.setHandler(document, DwtEvent.ONKEYPRESS, DwtKeyboardMgr.__keyPressHdlr);
379 	} else {
380 		Dwt.clearHandler(document, DwtEvent.ONKEYDOWN);
381 		Dwt.clearHandler(document, DwtEvent.ONKEYUP);
382 		Dwt.clearHandler(document, DwtEvent.ONKEYPRESS);
383 	}
384 };

452 /** 
453  * Sets keyboard focus to the given object.
454  *
455  * @private
456  */
457 DwtKeyboardMgr.prototype.__doGrabFocus =
458 function(focusObj) {
459 
460 	if (!focusObj) { return; }
461 	
462 	var dwtInputCtrl = focusObj.isInputControl;
463 	
464 //	DBG.println("kbnav", "DwtKeyboardMgr._doGrabFocus: " + focusObj);
465 	DBG.println("focus", "DwtKeyboardMgr._doGrabFocus: " + focusObj);
466 	if (dwtInputCtrl || !(focusObj instanceof DwtControl)) {
467 		DBG.println("focus", "DwtKeyboardMgr._doGrabFocus: input or non-control");
468 		// dealing with an input field
469 		if (this.__focusObj instanceof DwtControl && !this.__dwtInputCtrl) {
470 			// ctrl -> input
471 			this.__oldFocusObj = this.__focusObj;
472 		}
473 		this.__focusObj = focusObj;
474 		this.__dwtInputCtrl = dwtInputCtrl;
475 		var el = dwtInputCtrl ? focusObj.getInputElement() : focusObj;
476 		// IE throws JS error if you try to focus a disabled or invisible input
477 		if ((!AjxEnv.isIE && focusObj.focus) ||
478 			(AjxEnv.isIE && focusObj.focus && !el.disabled && Dwt.getVisible(el))) {
479 			// ignore exception - IE sometimes still throws error, don't know why
480 			try {
481 				focusObj.focus();
482 			} catch(ex) {}
483 		}
484 	} else {
485 		DBG.println("focus", "DwtKeyboardMgr._doGrabFocus: control");
486 		/* If the current focus of obj and the one grabbing focus are both DwtControls
487 		 * then we need to simulate a blur on the control losing focus */
488 		if (this.__dwtCtrlHasFocus && (this.__focusObj instanceof DwtControl)) {
489 			// ctrl -> ctrl: blur old ctrl
490 			DwtKeyboardMgr.__onBlurHdlr();
491 			this.__dwtCtrlHasFocus = true;	// reset
492 		}
493 			
494 		this.__focusObj = focusObj;
495 		this.__dwtInputCtrl = false;
496 		
497 		/* If a DwtControl already has focus, then we need to manually call
498 		 * DwtKeyboardMgr.__onFocusHdlr to simulate focus since calling the focus()
499 		 * method on the input field does nothing. */
500 		if (this.__dwtCtrlHasFocus) {
501 			// ctrl -> ctrl: tell newly focused ctrl it got focus
502 			DwtKeyboardMgr.__onFocusHdlr();
503 		} else {
504 			DwtKeyboardMgr.__onFocusHdlr();
505 			// input -> ctrl: set browser focus to keyboard input field
506 			this._kbFocusField.focus();
507 		}
508 	}
509 };

529 /**
530  * Blur handler for keyboard focus input field.
531  * 
532  * @private
533  */
534 DwtKeyboardMgr.__onBlurHdlr =
535 function(ev) {
536 	DBG.println("focus", "DwtKeyboardMgr.__onBlurHdlr");
537 
538 	var kbMgr = DwtKeyboardMgr.__shell.getKeyboardMgr();
539 
540 	// Got to play the trick with HTML elements which get focus before blur is
541 	// called on the old focus object. (see _grabFocus)
542 	var focusObj = kbMgr.__oldFocusObj || kbMgr.__focusObj;
543 //	DBG.println("kbnav", "DwtKeyboardMgr: ONBLUR - " + focusObj);

549 	 * the address bar or search field in FF. The bug is that depending on where
550 	 * you had focus, you could get a visual artifact (temporary). However, the code
551 	 * broke the fact that when focus leaves the browser window, then returns, then
552 	 * if a DwtControl had focus it will not get the appropriate highlight.
553 	 */	
554 //	kbMgr.__oldFocusObj = kbMgr.__focusObj = null;
555 	kbMgr.__oldFocusObj = null;
556 	kbMgr.__dwtCtrlHasFocus = false;	
557 };

560 /**
561  * @private
562  */
563 DwtKeyboardMgr.__keyUpHdlr =
564 function(ev) {
565 	ev = DwtUiEvent.getEvent(ev);
566 //	DBG.println("kbnav", "keyup: " + ev.keyCode);
567 
568 	var kbMgr = DwtKeyboardMgr.__shell.getKeyboardMgr();
569 	if (kbMgr._evtMgr.notifyListeners(DwtEvent.ONKEYUP, ev) === false) {
570 		return false;
571 	}
572 
573 	// clear saved Gecko key
574 	if (AjxEnv.isMac && AjxEnv.isGeckoBased && ev.keyCode == 0) {
575 		return DwtKeyboardMgr.__keyDownHdlr(ev);
576 	} else {
577 		return DwtKeyboardMgr.__handleKeyEvent(ev);
578 	}
579 };

581 /**
582  * @private
583  */
584 DwtKeyboardMgr.__keyPressHdlr =
585 function(ev) {
586 	ev = DwtUiEvent.getEvent(ev);
587 //	DBG.println("kbnav", "keypress: " + (ev.keyCode || ev.charCode));
588 
589 	var kbMgr = DwtKeyboardMgr.__shell.getKeyboardMgr();
590 	if (kbMgr._evtMgr.notifyListeners(DwtEvent.ONKEYPRESS, ev) === false) {
591 		return false;
592 	}
593 
594 	DwtKeyEvent.geckoCheck(ev);
595 
596 	return DwtKeyboardMgr.__handleKeyEvent(ev);
597 };

599 /**
600  * @private
601  */
602 DwtKeyboardMgr.__handleKeyEvent =
603 function(ev) {
604 
605 	if (DwtKeyboardMgr.__shell._blockInput) { return false; }
606 
607 	ev = DwtUiEvent.getEvent(ev, this);
608 //	DBG.println("kbnav", [ev.type, ev.keyCode, ev.charCode, ev.which].join(" / "));
609 	var kbMgr = DwtKeyboardMgr.__shell.getKeyboardMgr();
610 	var kev = DwtShell.keyEvent;
611 	kev.setFromDhtmlEvent(ev);
612 
613 	if (kbMgr.__kbEventStatus != DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED) {
614 		return kbMgr.__processKeyEvent(ev, kev, false);
615 	}
616 };

618 /*
619  * There are a number of focus cases that we must handle because of the way
620  * tabbing works in the browser, and because of user actions:
621  * 
622  * Case 1
623  * ------
624  * User tabs in from address bar. w/FF we get no indication of this. Focus will
625  * go to the next input field as seen by the browser
626  * 
627  * Solution: Make our hidden input field the first tab item. On focus get the last
628  * element that had focus from the tab group and set focus to that
629  * 
630  * Case 2
631  * ------
632  * User clicks in an input that is part of the tab group hierarchy. 
633  * 
634  * Solution: When the user types into such a field we will detect the key event and
635  * set the tab groups current focus member appropriately
636  * 
637  * Case 3
638  * ------
639  * User clicks in an input that is not part of the tab group hierarchy
640  * 
641  * Solution: Not much we can do here except ignore events. This is really not a good
642  * thing as all visible elements should be part of the tabbing hierarchy
643  * 
644  * Case 4
645  * ------
646  * User clicks in an external input field (e.g. the browser address bar)
647  * 
648  * Solution: we actually don't have to do anything here as case 1 should adress
649  * the situation when the user clicks/tabs back into elements we control
650  */
651 
652 /**
653  * This method does a focus check. If a DwtControl does not have focus, then we
654  * are dealing with an input element. If this element is not the current focus object
655  * then try and set it to the current focus object (case 2). If the object is not in the
656  * tab group hierarchy return false indicating that we should leave all events
657  * related to it alone (case 3).
658  * 
659  * @private
660  */
661 DwtKeyboardMgr.__syncFocus =
662 function(kbMgr, obj) {
663 //	DBG.println("kbnav", "sync focus ----- obj: " + obj + ", __focusObj: " + kbMgr.__focusObj + ", __dwtCtrlHasFocus: "
664 //			+ kbMgr.__dwtCtrlHasFocus + ", __dwtInputCtrl: " + kbMgr.__dwtInputCtrl + ", __oldFocusObj: " + kbMgr.__oldFocusObj);
665 	/* (BUG 8588) If obj is not the hidden input field (it's some other input field), and if
666 	 * a control has focus, then we have a focus mismatch and we need to blur the
667 	 * control that thinks it has focus. That can happen due to the way focus
668 	 * can be set in input fields. */ 
669 	if ((obj != kbMgr._kbFocusField) && kbMgr.__dwtCtrlHasFocus) {
670 //		DBG.println("kbnav", "FOCUS MISMATCH! focus obj: " + kbMgr.__focusObj + ", obj: " + obj);
671 //		DwtKeyboardMgr.__onBlurHdlr();
672 		kbMgr._kbFocusField.focus();
673 	}
674 	
675 //	DBG.println("kbnav", "DwtKeyboardMgr.__syncFocus: focus obj: " + kbMgr.__focusObj + " - obj: " + obj);
676 	if (!kbMgr.__dwtCtrlHasFocus) {
677 		// DwtInputField DwtHtmlEditor
678 		if ((obj != kbMgr.__focusObj) && !kbMgr.__dwtInputCtrl) {
679 //			DBG.println("kbnav", "Focus out of sync, resetting");
680 			if (kbMgr.__currTabGroup && kbMgr.__currTabGroup.setFocusMember(obj)) {
681 				kbMgr.__focusObj = obj;
682 				kbMgr.__oldFocusObj = null;
683 			} else {
684 				return false;
685 			}
686 		}
687 	}
688 	return true;
689 };

691 /**
692  * @private
693  */
694 DwtKeyboardMgr.__keyDownHdlr =
695 function(ev) {
696 
697 	try {
698 
699 	ev = DwtUiEvent.getEvent(ev, this);
700 	var kbMgr = DwtKeyboardMgr.__shell.getKeyboardMgr();
701 	ev.focusObj = null;
702 	if (kbMgr._evtMgr.notifyListeners(DwtEvent.ONKEYDOWN, ev) === false) {
703 		return false;
704 	};
705 
706 	if (DwtKeyboardMgr.__shell._blockInput) { return false; }
707 //	DBG.println("kbnav", [ev.type, ev.keyCode, ev.charCode, ev.which].join(" / "));
708 	if (kbMgr && !kbMgr.isEnabled()) { return true; }  // Allow key events to propagate when keyboard manager is disabled (to avoid taking over browser shortcuts). Bugzilla #45469.
709 	if (!kbMgr || !kbMgr.__checkStatus()) { return false; }
710 	var kev = DwtShell.keyEvent;
711 	kev.setFromDhtmlEvent(ev);
712 	var keyCode = DwtKeyEvent.getCharCode(ev);
713 //	DBG.println("kbnav", "keydown: " + keyCode + " -------- " + ev.target);
714 
715 	// Popdown any tooltip
716 	DwtKeyboardMgr.__shell.getToolTip().popdown();
717 	
718 	// Sync up focus if needed
719 	var focusInTGMember = DwtKeyboardMgr.__syncFocus(kbMgr, kev.target);
720 	
721 	if (!focusInTGMember) {
722 //		DBG.println("kbnav", "Object is not in tab hierarchy");
723 	}
724 			
725 	/* The first thing we care about is the tab key since we want to manage
726 	 * focus based on the tab groups. 
727 	 * 
728 	 * If the tab hit happens in the currently
729 	 * focused obj, the go to the next/prev element in the tab group. 
730 	 * 
731 	 * If the tab happens in an element that is in the tab group hierarchy, but that 
732 	 * element is not the currently focus element in the tab hierarchy (e.g. the user
733 	 * clicked in it and we didnt detect it) then sync the tab group's current focus 
734 	 * element and handle the tab
735 	 * 
736 	 * If the tab happens in an object not under the tab group hierarchy, then set
737 	 * focus to the current focus object in the tab hierarchy i.e. grab back control
738 	 */
739 	 if (keyCode == DwtKeyMapMgr.TAB_KEYCODE) {
740 	 	if (kbMgr.__currTabGroup && !DwtKeyMapMgr.hasModifier(kev)) {
741 		 	// If a menu is popped up then don't act on the Tab
742 		 	if (!DwtMenu.menuShowing()) {
743 //			 	DBG.println("kbnav", "Tab");
744 			 	// If the tab hit is in an element or if the current tab group has
745 			 	// a focus member
746 				if (focusInTGMember || kbMgr.__currTabGroup.getFocusMember()) {
747 				 	if (!kev.shiftKey) {
748 //				 		kbMgr.__currTabGroup.dump("kbnav");
749 				 		kbMgr.__currTabGroup.getNextFocusMember(true);
750 				 	} else {
751 				 		kbMgr.__currTabGroup.getPrevFocusMember(true);
752 				 	}
753 			 	} else {
754 //			 		DBG.println("kbnav", "DwtKeyboardMgr.__keyDownHdlr: no current focus member, resetting to first in tab group");
755 			 		// If there is no current focus member, then reset
756 			 		kbMgr.__currTabGroup.resetFocusMember(true);
757 			 	}
758 		 	}
759 		 	return kbMgr.__processKeyEvent(ev, kev, false, DwtKeyboardMgr.__KEYSEQ_HANDLED);
760 	 	} else {
761 	 		// No tab groups registered, or Alt or Ctrl was down. Let the browser handle it.
762 		 	return kbMgr.__processKeyEvent(ev, kev, true, DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED);
763 	 	}
764 	 } else if (kbMgr.__currTabGroup && !focusInTGMember && AjxEnv.isGecko && kev.target instanceof HTMLHtmlElement) {
765 	 	/* With FF we focus get set to the <html> element when tabbing in
766 	 	 * from the address or search fields. What we want to do is capture
767 	 	 * this here and reset the focus to the first element in the tabgroup
768 	 	 * 
769 	 	 * TODO Verify this trick is needed/works with IE/Safari
770 	 	 */
771 		kbMgr.__currTabGroup.resetFocusMember(true);
772 	 }
773 	 
774 	// If the focus object is a DwtControl, then clear the keyboard focus field
775 	if (kbMgr.__dwtCtrlHasFocus) {
776 		kbMgr._kbFocusField.value = "";
777 	}
778 	 
779 	// Filter out modifier keys. If we're in an input field, filter out legitimate input.
780 	// (A shortcut from an input field must use a modifier key.)
781 	if (DwtKeyMap.IS_MODIFIER[keyCode] || (!kbMgr.__dwtCtrlHasFocus && (kbMgr.__killKeySeqTimedActionId == -1) &&
782 		DwtKeyMapMgr.isInputElement(kev.target) && !DwtKeyboardMgr.isPossibleInputShortcut(kev))) {
783 
784 	 	return kbMgr.__processKeyEvent(ev, kev, true, DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED);
785 	}
786 	 
787 	/* Cancel any pending time action to kill the keysequence */
788 	if (kbMgr.__killKeySeqTimedActionId != -1) {
789 		AjxTimedAction.cancelAction(kbMgr.__killKeySeqTimedActionId);
790 		kbMgr.__killKeySeqTimedActionId = -1;
791 	}
792 		
793  	var parts = [];
794 	if (kev.altKey) 	{ parts.push(DwtKeyMap.ALT); }
795 	if (kev.ctrlKey) 	{ parts.push(DwtKeyMap.CTRL); }
796  	if (kev.metaKey) 	{ parts.push(DwtKeyMap.META); }
797 	if (kev.shiftKey) 	{ parts.push(DwtKeyMap.SHIFT); }
798 	parts.push(keyCode);
799 	kbMgr.__keySequence[kbMgr.__keySequence.length] = parts.join(DwtKeyMap.JOIN);
800 
801 //	DBG.println("kbnav", "KEYCODE: " + keyCode + " - KEY SEQ: " + kbMgr.__keySequence.join(""));
802 	
803 	var handled = DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
804 
805 	// First see if the control that currently has focus can handle the key event
806 	var obj = ev.focusObj || kbMgr.__focusObj;
807 	DBG.println("focus", "DwtKeyboardMgr::__keyDownHdlr - focus object: " + obj);
808 	if (obj && (obj.handleKeyAction) && (kbMgr.__dwtCtrlHasFocus || kbMgr.__dwtInputCtrl || (obj.hasFocus && obj.hasFocus()))) {
809 //		DBG.println("kbnav", obj + " has focus: " + obj.hasFocus());
810 		handled = kbMgr.__dispatchKeyEvent(obj, kev);
811 		while ((handled == DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED) && obj.parent && obj.parent.getKeyMapName) {
812 			obj = obj.parent;
813 			handled = kbMgr.__dispatchKeyEvent(obj, kev);
814 		}
815 	}
816 
817 	// If the currently focused control didn't handle the event, hand it to the default key
818 	// event handler
819 	if ((handled == DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED) && kbMgr.__currDefaultHandler &&
820 		!(kbMgr.__currTabGroup && kbMgr.__currTabGroup.isDefaultHandlingBlocked())) {
821 		handled = kbMgr.__dispatchKeyEvent(kbMgr.__currDefaultHandler, kev);
822 	}
823 
824 	// see if we should let browser handle the event as well; note that we need to set the 'handled' var rather than
825 	// just the 'propagate' one below, since the keyboard mgr isn't built for both it and the browser to handle the event.
826 	if (kev.forcePropagate) {
827 		handled = DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
828 		kev.forcePropagate = false;
829 	}
830 	
831 	kbMgr.__kbEventStatus = handled;
832 	var propagate = (handled == DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED);
833 
834 	if (handled != DwtKeyboardMgr.__KEYSEQ_PENDING) {
835 		kbMgr.clearKeySeq();
836 	}
837 
838 	return kbMgr.__processKeyEvent(ev, kev, propagate);
839 
840 	} catch (ex) {
841 		AjxException.reportScriptError(ex);
842 	}
843 };

845 /**
846  * Handles event dispatching
847  * 
848  * @private
849  */
850 DwtKeyboardMgr.prototype.__dispatchKeyEvent = 
851 function(hdlr, ev, forceActionCode) {
852 
853 	if (hdlr && hdlr.handleKeyEvent) {
854 		var handled = hdlr.handleKeyEvent(ev);
855 		return handled ? DwtKeyboardMgr.__KEYSEQ_HANDLED : DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
856 	}
857 	var mapName = (hdlr && hdlr.getKeyMapName) ? hdlr.getKeyMapName() : null;
858 	if (!mapName) {
859 		return DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
860 	}
861 //	DBG.println("kbnav", "DwtKeyboardMgr.__dispatchKeyEvent: handler " + hdlr.toString() + " handling " + this.__keySequence + " for map: " + mapName);
862 	var actionCode = this.__keyMapMgr.getActionCode(this.__keySequence, mapName, forceActionCode);
863 	if (actionCode == DwtKeyMapMgr.NOT_A_TERMINAL) {
864 //		DBG.println("kbnav", "scheduling action to kill key sequence");
865 		/* setup a timed action to redispatch/kill the key sequence in the event
866 		 * the user does not press another key in the allotted time */
867 		this.__hdlr = hdlr;
868 		this.__mapName = mapName;
869 		this.__ev = ev;
870 		this.__killKeySeqTimedActionId = AjxTimedAction.scheduleAction(this.__killKeySeqTimedAction, this.__keyTimeout);
871 		return DwtKeyboardMgr.__KEYSEQ_PENDING;	
872 	} else if (actionCode != null) {
873 		/* It is possible that the component may not handle a valid action
874 		 * particulary actions defined in the default map */
875 //		DBG.println("kbnav", "DwtKeyboardMgr.__dispatchKeyEvent: handling action: " + actionCode);
876 		if (!hdlr.handleKeyAction) {
877 			return DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
878 		}
879 		var result = hdlr.handleKeyAction(actionCode, ev);
880 		return result ? DwtKeyboardMgr.__KEYSEQ_HANDLED : DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
881 	} else {	
882 //		DBG.println("kbnav", "DwtKeyboardMgr.__dispatchKeyEvent: no action code for " + this.__keySequence);
883 		return DwtKeyboardMgr.__KEYSEQ_NOT_HANDLED;
884 	}
885 };
