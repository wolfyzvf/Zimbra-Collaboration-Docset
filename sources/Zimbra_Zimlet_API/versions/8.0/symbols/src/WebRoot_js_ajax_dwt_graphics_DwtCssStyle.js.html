 97 DwtCssStyle.getProperty = 
 98 function(htmlElement, cssPropName) {
 99 	var result;
100 	if (htmlElement.ownerDocument == null) {
101 		// IE5.5 does not support ownerDocument
102 		for(var parent = htmlElement.parentNode; parent.parentNode != null; parent = parent.parentNode);
103 		var doc = parent;
104 	} else {
105 		var doc = htmlElement.ownerDocument;
106 	}
107 
108 	if (doc.defaultView && doc.defaultView.getComputedStyle) {
109 		var cssDecl = doc.defaultView.getComputedStyle(htmlElement, "");
110 		if (cssDecl && cssDecl.length > 0) { //on Chrome/Safari it returns cssDecl with length 0 for some elements for some reason. (a wild guess could be invisible items, as it happens with invite toolbar when it's invisible) So in that case fall back on the other ways.
111 			return cssDecl.getPropertyValue(cssPropName);
112 		}
113 	}
114 	
115 	// Convert CSS -> DOM name for IE etc
116 	var tokens = cssPropName.split("-");
117 	// Shift one word off the array and capitalize the rest
118 	var propName = tokens.shift() + AjxUtil.map(tokens, AjxStringUtil.capitalize).join("");
119 
120 	if (htmlElement.currentStyle) {
121 		return htmlElement.currentStyle[propName];
122 	} else if (htmlElement.style) {
123 		return htmlElement.style[propName];
124 	}
125 };

127 DwtCssStyle.getComputedStyleObject = 
128 function(htmlElement) {
129 	if (htmlElement.ownerDocument == null) {
130 		// IE5.5 does not suppoert ownerDocument
131 		for(var parent = htmlElement.parentNode; parent.parentNode != null; parent = parent.parentNode);
132 		var doc = parent;
133 	} else {
134 		var doc = htmlElement.ownerDocument;
135 	}
136 	
137 	if (doc.defaultView) {
138 		var style = doc.defaultView.getComputedStyle(htmlElement, null);
139 		if (!style && htmlElement.style) {
140 // TODO: destructive ?
141 			htmlElement.style.display = "";
142 			style = doc.defaultView.getComputedStyle(htmlElement, null);
143 		}
144 		return style || {};
145 	} else if (htmlElement.currentStyle)
146 		return htmlElement.currentStyle;
147 	else if (htmlElement.style)
148 		return htmlElement.style;
149 };

151 DwtCssStyle.removeProperty = function(el, prop) {
152 	if (prop instanceof Array) {
153 		for (var i = prop.length; --i >= 0;)
154 			DwtCssStyle.removeProperty(el, prop[i]);
155 	} else {
156 		if (AjxEnv.isIE) {
157 			el.style.removeAttribute(prop, true);
158 		} else {
159 			prop = prop.replace(/([A-Z])/g, "-$1");
160 			el.style.removeProperty(prop);
161 		}
162 	}
163 };

165 /**
166  * Adds a rule to a stylesheet.
167  * 
168  * @param {StyleSheet}	stylesheet		a CSS stylesheet
169  * @param {string}		selector		rule selector
170  * @param {string}		declaration		styles
171  * @param {string}		index			insertion index (optional)
172  * 
173  * @return	index at which rule was inserted (for later removal)
174  */
175 DwtCssStyle.addRule =
176 function(stylesheet, selector, declaration, index) {
177 	if (stylesheet.addRule) {	// IE
178 		stylesheet.addRule(selector, declaration, index);
179 		return (index == null) ? (stylesheet.rules.length - 1) : index;
180 	}
181 	else {
182 		stylesheet.insertRule(selector + "{" + declaration + "}", index);
183 		return (index == null) ? (stylesheet.cssRules.length - 1) : index;
184 	}
185 };
186 
187 /**
188  * Removes the rule at the given index.
189  * 
190  * @param {StyleSheet}	stylesheet		a CSS stylesheet
191  * @param {string}		index			insertion index (optional)
192  */
193 DwtCssStyle.removeRule =
194 function(stylesheet, index) {
195 	if (stylesheet.removeRule) {	// IE
196 		stylesheet.removeRule(index);
197 	}
198 	else {
199 		stylesheet.deleteRule(index);
200 	}
201 };
