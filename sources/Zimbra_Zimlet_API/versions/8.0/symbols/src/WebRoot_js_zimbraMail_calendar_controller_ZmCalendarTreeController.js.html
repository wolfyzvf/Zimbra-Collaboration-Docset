 16 /**
 17  * Creates a calendar tree controller.
 18  * @constructor
 19  * @class
 20  * This class manages the calendar tree controller.
 21  *
 22  * @author Parag Shah
 23  *
 24  * @extends		ZmTreeController
 25  */
 26 ZmCalendarTreeController = function() {
 27 
 28 	ZmTreeController.call(this, ZmOrganizer.CALENDAR);
 29 
 30 	this._listeners[ZmOperation.NEW_CALENDAR]			= this._newListener.bind(this);
 31 	this._listeners[ZmOperation.ADD_EXTERNAL_CALENDAR]	= this._addExternalCalendarListener.bind(this);
 32 	this._listeners[ZmOperation.CHECK_ALL]				= this._checkAllListener.bind(this);
 33 	this._listeners[ZmOperation.CLEAR_ALL]				= this._clearAllListener.bind(this);
 34 	this._listeners[ZmOperation.DETACH_WIN]				= this._detachListener.bind(this);
 35 	this._listeners[ZmOperation.SHARE_CALENDAR]			= this._shareCalListener.bind(this);
 36     this._listeners[ZmOperation.MOVE]					= this._moveListener.bind(this);
 37 	this._listeners[ZmOperation.RECOVER_DELETED_ITEMS]	= this._recoverListener.bind(this);
 38 
 39 	this._eventMgrs = {};
 40 };

 50 function(ev) {
 51 
 52     ZmTreeController.prototype._treeListener.call(this, ev);
 53 
 54 	if(ev.detail == DwtTree.ITEM_EXPANDED){
 55         var calItem = ev.item;
 56         var calendar = calItem.getData(Dwt.KEY_OBJECT);

 60 	}

 85 /**
 86  * Gets all calendars.
 87  * 
 88  * @param	{String}	overviewId		the overview id
 89  * @param   {boolean}   includeTrash    True to include trash, if checked.
 90  * @return	{Array}		an array of {@link ZmCalendar} objects
 91  */
 92 ZmCalendarTreeController.prototype.getCalendars =
 93 function(overviewId, includeTrash) {
 94 	var calendars = [];
 95 	var items = this._getItems(overviewId);
 96 	for (var i = 0; i < items.length; i++) {
 97 		var item = items[i];
 98 		if (item._isSeparator) { continue; }
 99 	    var calendar = item.getData(Dwt.KEY_OBJECT);
100         if (calendar) {
101             if (calendar.id == ZmOrganizer.ID_TRASH && !includeTrash) continue;
102 			calendars.push(calendar);
103         }
104 	}
105 
106 	return calendars;
107 };

109 /**
110  * Gets the owned calendars.
111  * 
112  * @param	{String}	overviewId		the overview id
113  * @param	{String}	owner		the owner
114  * @return	{Array}		an array of {@link ZmCalendar} objects
115  */
116 ZmCalendarTreeController.prototype.getOwnedCalendars =
117 function(overviewId, owner) {
118 	var calendars = [];
119 	var items = this._getItems(overviewId);
120 	for (var i = 0; i < items.length; i++) {
121 		var item = items[i];
122 		if (!item || item._isSeparator) { continue; }
123 		var calendar = item.getData(Dwt.KEY_OBJECT);
124 		if (calendar && calendar.getOwner() == owner) {
125 			calendars.push(calendar);
126 		}
127 	}
128 
129 	return calendars;
130 };

132 ZmCalendarTreeController.prototype.addSelectionListener =
133 function(overviewId, listener) {
134 	// Each overview gets its own event manager
135 	if (!this._eventMgrs[overviewId]) {
136 		this._eventMgrs[overviewId] = new AjxEventMgr;
137 		// Each event manager has its own selection event to avoid multi-threaded
138 		// collisions
139 		this._eventMgrs[overviewId]._selEv = new DwtSelectionEvent(true);
140 	}
141 	this._eventMgrs[overviewId].addListener(DwtEvent.SELECTION, listener);
142 };

147 		this._eventMgrs[overviewId].removeListener(DwtEvent.SELECTION, listener);
148 	}

153 ZmCalendarTreeController.prototype.resetOperations = 
154 function(actionMenu, type, id) {
155 	if (actionMenu) {
156 		var calendar = appCtxt.getById(id);
157 		var nId;
158 		if (calendar) {
159 			nId = calendar.nId;
160             var isShareVisible = (!calendar.link || calendar.isAdmin()) && nId != ZmFolder.ID_TRASH;
161             if (appCtxt.isOffline) {
162                 var acct = calendar.getAccount();
163                 isShareVisible = !acct.isMain && acct.isZimbraAccount;
164             }
165 			actionMenu.enable(ZmOperation.SHARE_CALENDAR, isShareVisible);
166 			actionMenu.enable(ZmOperation.SYNC, calendar.isFeed());
167 		} else {
168 			nId = ZmOrganizer.normalizeId(id);
169 		}
170 		var isTrash = (nId == ZmFolder.ID_TRASH);
171 		actionMenu.enable(ZmOperation.DELETE_WITHOUT_SHORTCUT, (nId != ZmOrganizer.ID_CALENDAR && nId != ZmOrganizer.ID_TRASH));
172 		this.setVisibleIfExists(actionMenu, ZmOperation.EMPTY_FOLDER, nId == ZmFolder.ID_TRASH);
173 		var hasContent = ((calendar.numTotal > 0) || (calendar.children && (calendar.children.size() > 0)));
174 		actionMenu.enable(ZmOperation.EMPTY_FOLDER,hasContent);
175 
176         var moveItem = actionMenu.getItemById(ZmOperation.KEY_ID,ZmOperation.MOVE);
177 
178 		var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT);
179 		if (id == rootId) {
180 			var items = this._getItems(this._actionedOverviewId);
181 			var foundChecked = false;
182 			var foundUnchecked = false;
183 			for (var i = 0; i < items.length; i++) {
184 				var item = items[i];
185 				if (item._isSeparator) continue;
186 				item.getChecked() ? foundChecked = true : foundUnchecked = true;
187 			}
188 			actionMenu.enable(ZmOperation.CHECK_ALL, foundUnchecked);
189 			actionMenu.enable(ZmOperation.CLEAR_ALL, foundChecked);
190 		}
191 
192 		this._enableRecoverDeleted(actionMenu, isTrash);
193 
194 		// we always enable sharing in case we're in multi-mbox mode
195 		this._resetButtonPerSetting(actionMenu, ZmOperation.SHARE_CALENDAR, appCtxt.get(ZmSetting.SHARING_ENABLED));
196 		this._resetButtonPerSetting(actionMenu, ZmOperation.FREE_BUSY_LINK, appCtxt.getActiveAccount().isZimbraAccount);
197 
198         var fbLinkMenuItem = actionMenu.getMenuItem(ZmOperation.FREE_BUSY_LINK);
199         if(fbLinkMenuItem){
200             //setting up free busy link submenu
201             this._fbLinkSubMenu = (this._fbLinkSubMenu) ? this._fbLinkSubMenu : this._getFreeBusySubMenu(actionMenu);
202 
203             fbLinkMenuItem.setMenu(this._fbLinkSubMenu);
204         }
205 
206 
207 	}
208 };

309 ZmCalendarTreeController.prototype._getActionMenu =
310 function(ev) {
311 	var organizer = ev.item.getData(Dwt.KEY_OBJECT);
312 	if (organizer.type != this.type &&
313         organizer.nId != ZmOrganizer.ID_TRASH) {
314         return null;
315     }
316 	var menu = ZmTreeController.prototype._getActionMenu.apply(this, arguments);
317     var isTrash = organizer.nId == ZmOrganizer.ID_TRASH;
318     //bug 67531: "Move" Option should be disabled for the default calendar
319     var isCalendar = organizer.nId == ZmOrganizer.ID_CALENDAR;
320     menu.enableAll(!isTrash);
321     menu.enable(ZmOperation.MOVE, !isCalendar && !isTrash);
322     menu.enable(ZmOperation.EMPTY_FOLDER, isTrash);
323     var menuItem = menu.getMenuItem(ZmOperation.EMPTY_FOLDER);
324     if (menuItem) {
325         menuItem.setText(isTrash ? ZmMsg.emptyTrash : ZmMsg.emptyFolder);
326     }
327     return menu;
328 };

330 // Method that is run when a tree item is left-clicked
331 ZmCalendarTreeController.prototype._itemClicked =
332 function(organizer) {
333 	if (organizer.type != ZmOrganizer.CALENDAR) {
334         if (organizer._showFoldersCallback) {
335             organizer._showFoldersCallback.run();
336             return;
337         }
338 
339         if (organizer.id == ZmOrganizer.ID_TRASH) return;
340 
341 		var appId = ZmOrganizer.APP[organizer.type];
342 		var app = appId && appCtxt.getApp(appId);
343 		if (app) {
344 			var callback = new AjxCallback(this, this._postActivateApp, [organizer, app]);
345 			appCtxt.getAppController().activateApp(appId, null, callback);
346 		}
347 		else {
348 			appCtxt.setStatusMsg({
349 				msg:	AjxMessageFormat.format(ZmMsg.appUnknown, [appId]),
350 				level:	ZmStatusView.LEVEL_WARNING
351 			});
352 		}
353 	}
354 };

367 // Handles a drop event
368 ZmCalendarTreeController.prototype._dropListener =
369 function(ev) {
370 	var data = ev.srcData.data;
371     var dropFolder = ev.targetControl.getData(Dwt.KEY_OBJECT);
372 
373 	var appts = (!(data instanceof Array)) ? [data] : data;
374 	var isShiftKey = (ev.shiftKey || ev.uiEvent.shiftKey);
375 
376 	if (ev.action == DwtDropEvent.DRAG_ENTER) {
377 
378         var type = ev.targetControl.getData(ZmTreeView.KEY_TYPE);
379 
380         if(data instanceof ZmCalendar){
381              ev.doIt = dropFolder.mayContain(data, type) && !data.isSystem();
382         }
383 		else if (!(appts[0] instanceof ZmAppt)) {
384 			ev.doIt = false;
385 		}
386 		else if (this._dropTgt.isValidTarget(data)) {
387 			var type = ev.targetControl.getData(ZmTreeView.KEY_TYPE);
388 			ev.doIt = dropFolder.mayContain(data, type);
389 
390 			var action;
391 			// walk thru the array and find out what action is allowed
392 			for (var i = 0; i < appts.length; i++) {
393 				if (appts[i] instanceof ZmItem) {
394 					action |= appts[i].getDefaultDndAction(isShiftKey);
395 				}
396 			}
397 
398 			if (((action & ZmItem.DND_ACTION_COPY) != 0)) {
399 				var plusDiv = (appts.length == 1)
400 					? ev.dndProxy.firstChild.nextSibling
401 					: ev.dndProxy.firstChild.nextSibling.nextSibling;
402 
403 				Dwt.setVisibility(plusDiv, true);
404 			}
405 		}
406 	}
407 	else if (ev.action == DwtDropEvent.DRAG_DROP) {
408 		var ctlr = ev.srcData.controller;
409 		var cc = AjxDispatcher.run("GetCalController");
410         if (!isShiftKey && cc.isMovingBetwAccounts(appts, dropFolder.id)) {
411             var dlg = appCtxt.getMsgDialog();
412             dlg.setMessage(ZmMsg.orgChange, DwtMessageDialog.WARNING_STYLE);
413             dlg.popup();
414 		} else {
415             if(data instanceof ZmCalendar){
416                 // Root node's type is folder, but it's labelled 'Calendars'.  Pass the proper
417                 // name down to the status message.
418                 var folderName = (dropFolder.nId == ZmFolder.ID_ROOT) ? ZmMsg.calendars : null;
419                 this._doMove(data, dropFolder, folderName);
420             } else{
421                 ctlr._doMove(appts, dropFolder, null, isShiftKey);
422             }
423 		}
424 	}
425 };

478 ZmCalendarTreeController.prototype._changeListener =
479 function(ev, treeView, overviewId) {
480 	ZmTreeController.prototype._changeListener.call(this, ev, treeView, overviewId);
481 
482 	if (ev.type != this.type || ev.handled) { return; }
483 
484 	var fields = ev.getDetail("fields") || {};
485 	if (ev.event == ZmEvent.E_CREATE ||
486 		ev.event == ZmEvent.E_DELETE ||
487 		(ev.event == ZmEvent.E_MODIFY && fields[ZmOrganizer.F_FLAGS]))
488 	{
489 		var aCtxt = appCtxt.isChildWindow ? parentAppCtxt : appCtxt;
490 		var controller = aCtxt.getApp(ZmApp.CALENDAR).getCalController();
491 		controller._updateCheckedCalendars();
492 
493 		// if calendar is deleted, notify will initiate the refresh action
494 		if (ev.event != ZmEvent.E_DELETE) {
495             var calIds = controller.getCheckedCalendarFolderIds();
496             AjxDebug.println(AjxDebug.CALENDAR, "tree change listener refreshing calendar event '" + ev.event + "' with checked folder ids " + calIds.join(","));
497 			controller._refreshAction(true);
498 			ev.handled = true;
499 		}
500     }
501 };

579 		this._eventMgrs[overviewId].isListenerRegistered(type))

589 ZmCalendarTreeController.prototype._getItems =
590 function(overviewId) {
591 	var treeView = this.getTreeView(overviewId);
592 	if (treeView) {
593 		var account = appCtxt.multiAccounts ? treeView._overview.account : null;
594 		if (!appCtxt.get(ZmSetting.CALENDAR_ENABLED, null, account)) { return []; }
595 
596 		var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT, account);
597 		var root = treeView.getTreeItemById(rootId);
598 		if (root) {
599 			var totalItems = [];
600 			this._getSubItems(root, totalItems);
601 			return totalItems;
602 		}
603 	}
604 	return [];
605 };

607 ZmCalendarTreeController.prototype._getSubItems =
608 function(root, totalItems) {
609 	if (!root || (root && root._isSeparator)) { return; }
610 
611 	var items = root.getItems();
612     //items is an array
613 	for (var i = 0; i < items.length; i++) {
614 		var item = items[i];
615 		if (item && !item._isSeparator) {
616 			totalItems.push(item);

622 ZmCalendarTreeController.prototype._setAllChecked =
623 function(ev, checked) {
624 	var overviewId = this._actionedOverviewId;
625 	var items = this._getItems(overviewId);
626 	var checkedItems = [];
627 	var item, organizer;
628 	for (var i = 0;  i < items.length; i++) {
629 		item = items[i];
630 		if (item._isSeparator) { continue; }
631 		organizer = item.getData(Dwt.KEY_OBJECT);
632 		if (!organizer || organizer.type != ZmOrganizer.CALENDAR) { continue; }
633 		item.setChecked(checked);
634 		checkedItems.push(item);
635 	}
636 
637 	// notify listeners of selection
638 	if (checkedItems.length && this._eventMgrs[overviewId]) {
639 		this._notifyListeners(overviewId, DwtEvent.SELECTION, checkedItems, DwtTree.ITEM_CHECKED, ev, this._eventMgrs[overviewId]._selEv);
640 	}
641 };

749 ZmCalendarTreeController.CONN_TYPE_CLEARTEXT = "cleartext";
750 ZmCalendarTreeController.CONN_TYPE_SSL = "ssl";
751 ZmCalendarTreeController.SSL_PORT = "443";

753 ZmCalendarTreeController.DATA_SOURCE_ATTR = "p:/calendar/dav/_USERNAME_/user";

755 ZmCalendarTreeController.prototype.createDataSource =
756 function(organizer, errorCallback) {
757     var calDav = this._extCalData && this._extCalData.calDav ? this._extCalData.calDav : null;
758     if(!calDav) { return; }
759 
760     var url,
761         port,
762         urlPort,
763         hostUrl,
764         jsonObj,
765         connType = ZmCalendarTreeController.CONN_TYPE_CLEARTEXT,
766         dsa = ZmCalendarTreeController.DATA_SOURCE_ATTR;
767 
768     hostUrl = calDav.hostUrl;
769     if(hostUrl.indexOf(":") === -1) {
770         url = hostUrl;
771         port = ZmCalendarTreeController.SSL_PORT;
772     }
773     else {
774         urlPort = hostUrl.split(":");
775         url = urlPort[0];
776         port = urlPort[1];
777     }
778 
779     if(port == ZmCalendarTreeController.SSL_PORT) {
780         connType = ZmCalendarTreeController.CONN_TYPE_SSL;
781     }
782 
783     if(calDav.hostUrl.indexOf(ZmMsg.sharedCalCalDAVServerGoogle) === -1) { // Not google url
784         dsa = ZmCalendarTreeController.DATA_SOURCE_ATTR_YAHOO;
785     }
786 
787     jsonObj = {
788         CreateDataSourceRequest : {
789             _jsns : "urn:zimbraMail",
790             caldav : {
791                 name : organizer.name,
792                 pollingInterval : ZmCalendarTreeController.POLLING_INTERVAL,
793                 isEnabled : "1",
794                 l : organizer.nId,
795                 host : url,
796                 port : port,
797                 connectionType : connType,
798                 username : calDav.userName,
799                 password : calDav.password,
800                 a : {
801                     n : "zimbraDataSourceAttribute",
802                     _content : dsa
803                 }
804             }
805         }
806     };
807 
808     this._extCalData = null;
809     delete this._extCalData;
810     var accountName = (appCtxt.multiAccounts ? appCtxt.accountList.mainAccount.name : null);
811 
812     var params = {
813             jsonObj: jsonObj,
814             asyncMode: true,
815             sensitive: true,
816             noAuthToken: true,
817             callback: new AjxCallback(this, this.createDataSourceCallback),
818             errorCallback: new AjxCallback(this, this.createDataSourceErrorCallback),
819             accountName: accountName
820         };
821     appCtxt.getAppController().sendRequest(params);
822 };
