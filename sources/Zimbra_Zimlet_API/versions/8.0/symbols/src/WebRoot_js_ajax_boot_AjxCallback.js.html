 16 /**
 17  * Creates a callback which consists of at least a function reference, and possibly also
 18  * an object to call it from.
 19  * @constructor
 20  * @class
 21  * This class represents a callback function which can be called standalone, or from a
 22  * given object. What the callback takes as arguments and what it returns are left to the
 23  * client.
 24  *
 25  * @author Conrad Damon
 26  * 
 27  * @param {object}	obj		the object to call the function from
 28  * @param {function}	func	the callback function
 29  * @param {array}	args   the default arguments
 30  * 
 31  */
 32 AjxCallback = function(obj, func, args) {
 33 	if (arguments.length == 0) return;
 34 
 35     if (typeof arguments[0] == "function") {
 36         this.obj = null;
 37         this.func = arguments[0];
 38         this.args = arguments[1];
 39     }
 40     else {
 41         this.obj = obj;
 42         this.func = func;
 43         this.args = args;
 44     }
 45 };
 46 
 47 AjxCallback.prototype.isAjxCallback = true;
 48 AjxCallback.prototype.toString = function() { return "AjxCallback"; }
 49 
 50 AjxCallback.NOP = new AjxCallback(function(){});
 51 
 52 /**
 53  * Runs the callback function, from within the object if there is one. The
 54  * called function passed arguments are the concatenation of the argument
 55  * array passed to this object's constructor and the argument array passed
 56  * to the <code>run</code> method. Whatever the called function returns is
 57  * returned to the caller.
 58  *
 59  * @param {array}	[arg1..argN]	the first argument which will be appended to the argument
 60  *				array passed to this object's constructor. Any number of
 61  *				arguments may be passed to the <code>run</code> method.
 62  */
 63 AjxCallback.prototype.run =
 64 function(/* arg1 ... argN */) {
 65 	// combine original args with new ones
 66 	var args = [];
 67 
 68 	// sometimes we want to pass a null or false argument, so simply
 69 	// checking for if (this.args) won't do.
 70 	if (typeof this.args != "undefined") {
 71 		if (this.args!==null && this.args instanceof Array) { // IE mysteriously screws up the instanceof test if this.args is null
 72 			// NOTE: We must NOT use this.args directly if this method's
 73 			//       params are gonna be pushed onto the array because it
 74 			//       will change the original args!
 75 			args = arguments.length > 0 ? args.concat(this.args) : this.args;
 76 		} else {
 77 			args.push(this.args);
 78 		}
 79 	}
 80 
 81 	for (var i = 0; i < arguments.length; ++i) {
 82 		args.push(arguments[i]);
 83 	}
 84 
 85 	// invoke function
 86 	if (this.func) {
 87 		return this.func.apply(this.obj || window, args);
 88 	}
 89 };

 91 /**
 92  * This version of {@link AjxCallback.run} is here for {@link AjxDispatcher}, because it has a <code>run()</code>
 93  * method in which it marshals arguments into an array. That leads to a problem
 94  * in which the arguments are marshalled twice, so that by the time AjxDispatcher
 95  * calls <code>callback.run(args)</code>, the args have already been collected into an array.
 96  * Then when the function is invoked, it gets passed an actual array instead of the
 97  * intended arg list. Calling <code>callback.run.apply(callback, args)</code> works on Firefox,
 98  * but IE throws the error "Object expected", so we do this instead.
 99  *
100  * @param	{array}	argList	 an array of arguments and treats them as an argument list, instead of as a single argument
101  * 
102  * @private
103  */
104 AjxCallback.prototype.run1 =
105 function(argList) {
106 	// combine original args with new ones
107 	var args = [];
108 
109 	// sometimes we want to pass a null or false argument, so simply
110 	// checking for if (this.args) won't do.
111 	if (typeof this.args != "undefined") {
112 		if (this.args!==null && this.args instanceof Array) {
113 			// NOTE: We must NOT use this.args directly if this method's
114 			//       params are gonna be pushed onto the array because it
115 			//       will change the original args!
116 			args = arguments.length > 0 ? args.concat(this.args) : this.args;
117 		} else {
118 			args.push(this.args);
119 		}
120 	}
121 
122 	if (argList && argList.length) {
123 		for (var i = 0; i < argList.length; ++i) {
124 			args.push(argList[i]);
125 		}
126 	}
127 
128 	// invoke function
129 	if (this.func) {
130 		return this.func.apply(this.obj || window, args);
131 	}
132 };

134 /**
135  * This method returns a plain function that will call your supplied "func" in the context
136  * of "obj" and pass to it, in this order, any additional arguments that you
137  * pass to <code>simpleClosure</code> and the arguments that were passed to it at the call
138  * time.
139  *
140  * <p>
141  * An example should do:
142  *
143  * <pre>
144  *   div.onclick = AjxCallback.simpleClosure(this.handler, this, "some data");
145  *   ...
146  *   this.handler = function(data, event) {
147  *      // event will be passed for DOM2 compliant browsers
148  *      // and data is "some data"
149  *   };
150  * </pre>
151  * 
152  * @param	{function}	func		the function
153  * @param	{object}	obj			the object to call the function from
154  * @param	{array}	[arg1...argN]		any number of arguments
155  */
156 AjxCallback.simpleClosure = function(func, obj) {
157 	var args = [];
158 	for (var i = 2; i < arguments.length; ++i)
159 		args.push(arguments[i]);
160 	return function() {
161 		var args2 = [];
162 		for (var i = 0; i < arguments.length; ++i)
163 			args2.push(arguments[i]);
164 		return func.apply(obj || this, args.concat(args2));
165 	};
166 };

168 AjxCallback.returnFalse = function() { return false; };

170 AjxCallback.isNull = function(x) { return x == null; };
