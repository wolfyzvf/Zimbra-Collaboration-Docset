 16 /**
 17  * @overview
 18  * This file contains the search application class.
 19  */
 20 
 21 /**
 22  * Creates and initializes the search application.
 23  * @constructor
 24  * @class
 25  * The search app manages user-initiated searches.
 26  *
 27  * @param	{DwtControl}	container		the container
 28  * @param	{ZmController}	parentController	the parent window controller (set by the child window)
 29  * 
 30  * @author Conrad Damon
 31  * 
 32  * @extends		ZmApp
 33  */
 34 ZmSearchApp = function(container, parentController) {
 35 
 36 	ZmApp.call(this, ZmApp.SEARCH, container, parentController);
 37 };
 38 
 39 ZmSearchApp.prototype = new ZmApp;
 40 ZmSearchApp.prototype.constructor = ZmSearchApp;
 41 
 42 ZmSearchApp.prototype.isZmSearchApp = true;
 43 ZmSearchApp.prototype.toString = function() {	return "ZmSearchApp"; };
 44 
 45 ZmApp.SEARCH					= ZmId.APP_SEARCH;
 46 ZmApp.CLASS[ZmApp.SEARCH]		= "ZmSearchApp";
 47 ZmApp.SETTING[ZmApp.SEARCH]		= ZmSetting.SEARCH_ENABLED;
 48 
 49 ZmSearchApp.prototype.getSearchResultsController =
 50 function(sessionId, appName) {
 51 	return this.getSessionController({
 52 				controllerClass:	"ZmSearchResultsController",
 53 				sessionId:			sessionId,
 54 				appName:			appName
 55 			});
 56 };
 57 
 58 // override so we don't try to set overview panel content
 59 ZmSearchApp.prototype.activate =
 60 function(active) {
 61 	this._active = active;
 62 };
