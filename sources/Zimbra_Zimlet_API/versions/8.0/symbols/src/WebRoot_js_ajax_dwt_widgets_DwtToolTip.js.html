 16 /**
 17  * Singleton tooltip class.
 18  */
 19 DwtToolTip = function(shell, className, dialog) {
 20 
 21 	if (arguments.length == 0) { return; }
 22 
 23 	this.shell = shell;
 24 	this._dialog = dialog;
 25 	this._poppedUp = false;
 26 	this._div = document.createElement("div");
 27 	this._div.className = className || "DwtToolTip";
 28 	this._div.style.position = DwtControl.ABSOLUTE_STYLE;
 29 	this.shell.getHtmlElement().appendChild(this._div);
 30 	Dwt.setZIndex(this._div, Dwt.Z_HIDDEN);
 31 	Dwt.setLocation(this._div, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
 32 
 33     // create html
 34     // NOTE: This id is ok because there's only ever one instance of a tooltip
 35     var templateId = "dwt.Widgets#" + this._borderStyle;
 36     this._div.innerHTML = AjxTemplate.expand(templateId, "tooltip");
 37 
 38     var params = AjxTemplate.getParams(templateId);
 39     this._offsetX = (params.width != null) ? Number(params.width) : DwtToolTip.POPUP_OFFSET_X;
 40     this._offsetY = (params.height != null) ? Number(params.height) : DwtToolTip.POPUP_OFFSET_Y;
 41 
 42     // save reference to content div
 43     this._contentDiv = document.getElementById("tooltipContents");
 44 
 45     Dwt.setHandler(this._div, DwtEvent.ONMOUSEOVER, AjxCallback.simpleClosure(this._mouseOverListener, this));
 46 };
 47 
 48 DwtToolTip.prototype.isDwtToolTip = true;
 49 DwtToolTip.prototype.toString = function() { return "DwtToolTip"; };

 71 DwtToolTip.prototype.getContent =
 72 function() {
 73     return this._div.innerHTML;
 74 };

 76 DwtToolTip.prototype.setContent =
 77 function(content, setInnerHTML) {
 78 	this._content = content;
 79 	if(setInnerHTML) {
 80         this._contentDiv.innerHTML = this._content;
 81     }
 82 };

 84 /**
 85  * Shows the tooltip. By default, its position will be relative to the location of the
 86  * cursor when the mouseover event happened. Alternatively, the control that generated
 87  * the tooltip can be passed in, and the tooltip will be positioned relative to it. If
 88  * the control is a large composite control (eg a DwtListView), the hover event can be
 89  * passed so that the actual target of the event can be found.
 90  * 
 91  * @param {number}			x					X-coordinate of cursor
 92  * @param {number}			y					Y-coordinate of cursor
 93  * @param {boolean}			skipInnerHTML		if true, do not copy content to DOM
 94  * @param {boolean}			popdownOnMouseOver	if true, hide tooltip on mouseover
 95  * @param {DwtControl}		obj					control that tooltip is for (optional)
 96  * @param {DwtHoverEvent}	hoverEv				hover event (optional)
 97  */
 98 DwtToolTip.prototype.popup = 
 99 function(x, y, skipInnerHTML, popdownOnMouseOver, obj, hoverEv) {
100     if (this._popupAction) {
101         AjxTimedAction.cancelAction(this._popupAction);
102         this._popupAction = null;
103     }
104 	// popdownOnMouseOver may be true to pop down the tooltip if the mouse hovers over the tooltip. Optionally,
105 	// it can be an AjxCallback that will be called after popping the tooltip down.
106     this._popdownOnMouseOver = popdownOnMouseOver;
107     if (this._content != null) {
108 		if(!skipInnerHTML) {
109             this._contentDiv.innerHTML = this._content;
110         }
111 
112 		this._popupAction = new AjxTimedAction(this, this._positionElement, [x, y, obj, hoverEv]);
113 		AjxTimedAction.scheduleAction(this._popupAction, 5);
114 	}
115 };

117 /*
118 * setSticky allows making the tooltip not to popdown. 
119 * IMPORTANT: Tooltip is singleton inside Zimbra i.e. only one instance of tooltip is reused by all objects. 
120 * So, it is very important for the code setting tooltip to sticky to have some mechanism to close the tooltip by itself. 
121 * Like have a close-button inside tooltip and when clicked, should set the setSticky(false) and then close the tooltip.
122 *
123 * If setSticky(true) is called, _poppedUp is set to false, which is essentially pretending the tooltip is not
124 * up. In that case, a call to popdown will not close the tooltip. And that means tooltip will stay up even if some other
125 * code path calls popdown on the singleton tooltip.
126 *
127 */
128 DwtToolTip.prototype.setSticky = 
129 function(bool) {
130 	this._poppedUp = !bool;
131 };
132 
133 DwtToolTip.prototype.popdown = 
134 function() {
135     this._popdownOnMouseOver = false;
136     if (this._popupAction) {
137         AjxTimedAction.cancelAction(this._popupAction);
138         this._popupAction = null;
139     }
140 	if (this._content != null && this._poppedUp) {
141 		Dwt.setLocation(this._div, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
142 		this._poppedUp = false;
143 	}
144 };

150 // Positions the tooltip relative to the base element based on vertical and horizontal offsets.
151 DwtToolTip.prototype._positionElement = 
152 function(startX, startY, obj, hoverEv) {
153 	
154     this._popupAction = null;
155 	
156 	var wdSize = DwtShell.getShell(window).getSize();
157 	var wdWidth = wdSize.x, wdHeight = wdSize.y;
158 
159 	var tooltipX, tooltipY, baseLoc;
160 	var baseEl = obj && obj.getTooltipBase(hoverEv);
161 	if (baseEl) {
162 		baseLoc = Dwt.toWindow(baseEl);
163 		var baseSz = Dwt.getSize(baseEl);
164 		tooltipX = baseLoc.x + this._offsetX;
165 		tooltipY = baseLoc.y + baseSz.y + this._offsetY;
166 	}
167 	else {
168 		tooltipX = startX + this._offsetX;
169 		tooltipY = startY + this._offsetY;
170 	}
171 
172 	var popupSize = Dwt.getSize(this._div);
173 	var popupWidth = popupSize.x, popupHeight = popupSize.y;
174 
175 	// check for sufficient room to the right
176 	if (tooltipX + popupWidth > wdWidth - DwtToolTip.WINDOW_GUTTER) {
177 		tooltipX = wdWidth - DwtToolTip.WINDOW_GUTTER - popupWidth;
178 	}
179 	// check for sufficient room below
180 	if (tooltipY + popupHeight > wdHeight - DwtToolTip.WINDOW_GUTTER) {
181 		tooltipY = (baseLoc ? baseLoc.y : tooltipY) - this._offsetY - popupHeight;
182 	}
183 
184 	Dwt.setLocation(this._div, tooltipX, tooltipY);
185 	var zIndex = this._dialog ? this._dialog.getZIndex() + Dwt._Z_INC : Dwt.Z_TOOLTIP;
186 	Dwt.setZIndex(this._div, zIndex);
187     this._poppedUp = true;
188 };
189 
190 DwtToolTip.prototype._mouseOverListener = 
191 function(ev) {
192     if (this._popdownOnMouseOver && this._poppedUp) {
193         var callback = (this._popdownOnMouseOver.isAjxCallback) ? this._popdownOnMouseOver : null;
194         this.popdown();
195         if (callback) {
196             callback.run();
197 		}
198     }
199 };
