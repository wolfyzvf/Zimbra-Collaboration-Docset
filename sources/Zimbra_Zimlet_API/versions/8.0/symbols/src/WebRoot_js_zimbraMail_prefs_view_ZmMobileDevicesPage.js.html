 16 /**
 17  * Creates the mobile devices page.
 18  * @class
 19  * This class represents the mobile devices page.
 20  * 
 21  * @param {DwtControl}	parent			the containing widget
 22  * @param {Object}	section			the page
 23  * @param {ZmPrefController}	controller		the prefs controller
 24  * 
 25  * @extends	ZmPreferencesPage
 26  * 
 27  * @private
 28  */
 29 ZmMobileDevicesPage = function(parent, section, controller) {
 30 	ZmPreferencesPage.apply(this, arguments);
 31 
 32 	this._deviceController = controller.getMobileDevicesController();
 33 };
 34 
 35 ZmMobileDevicesPage.prototype = new ZmPreferencesPage;
 36 ZmMobileDevicesPage.prototype.constructor = ZmMobileDevicesPage;
 37 
 38 ZmMobileDevicesPage.prototype.toString =
 39 function () {
 40     return "ZmMobileDevicesPage";
 41 };
 42 
 43 
 44 // ZmPreferencesPage methods
 45 
 46 /**
 47  * @private
 48  */
 49 ZmMobileDevicesPage.prototype.showMe =
 50 function() {
 51 	ZmPreferencesPage.prototype.showMe.apply(this, arguments);
 52 
 53 	if (!this._rendered) {
 54 		var params = {
 55 			parent:this,
 56 			buttons:this._deviceController.getToolbarButtons(),
 57 			posStyle:Dwt.STATIC_STYLE,
 58 			context:ZmId.VIEW_MOBILE_DEVICES,
 59 			parentElement:(this._htmlElId+"_deviceToolbar")
 60 		};
 61 		this._toolbar = new ZmButtonToolBar(params);
 62 
 63 		params = {
 64 			parent:this,
 65 			parentElement:(this._htmlElId+"_deviceList")
 66 		};
 67 		this.listView = new ZmMobileDeviceListView(params);
 68 		this.listView.setMultiSelect(false);
 69 
 70 		this._deviceController.initialize(this._toolbar, this.listView);
 71 		this._rendered = true;
 72 	}
 73 
 74 	this._deviceController.loadDeviceInfo();
 75 };
 76 
 77 ZmMobileDevicesPage.prototype.reset =
 78 function(useDefaults) {
 79 	ZmPreferencesPage.prototype.reset.apply(this, arguments);
 80 
 81 	if (this._controller.getTabView().getActiveView() == this) {
 82 		this._deviceController.loadDeviceInfo();
 83 	}
 84 };
 85 
 86 ZmMobileDevicesPage.prototype.hasResetButton =
 87 function() {
 88 	return false;
 89 };

 91 /**
 92  * Creates a mobile device list.
 93  * @class
 94  * A list view that displays user mobile devices. The data is in the form of a
 95  * list of {@link ZmMobileDevice} objects.
 96  *
 97  * @param {Hash}	params	a hash of parameters
 98  * 
 99  * 
100  * @extends		DwtListView
101  * 
102  * @private
103  */
104 ZmMobileDeviceListView = function(params) {
105 
106 	params.headerList = this._getHeaderList();
107 	DwtListView.call(this, params);
108 };
109 
110 ZmMobileDeviceListView.prototype = new DwtListView;
111 ZmMobileDeviceListView.prototype.constructor = ZmMobileDeviceListView;
112 
113 
114 // Consts
115 
116 ZmMobileDeviceListView.F_DEVICE			= "de";
117 ZmMobileDeviceListView.F_STATUS			= "st";
118 ZmMobileDeviceListView.F_ID				= "id";
119 ZmMobileDeviceListView.F_PROTOCOL		= "pr";
120 ZmMobileDeviceListView.F_PROVISIONABLE	= "pv";
121 
122 
123 // Public methods
124 
125 ZmMobileDeviceListView.prototype.toString =
126 function() {
127 	return "ZmMobileDeviceListView";
128 };
129 
130 ZmMobileDeviceListView.prototype._getHeaderList =
131 function() {
132 
133 	var headerList = [];
134 	headerList.push(new DwtListHeaderItem({field:ZmMobileDeviceListView.F_DEVICE, text:ZmMsg.mobileDevice}));
135 	headerList.push(new DwtListHeaderItem({field:ZmMobileDeviceListView.F_ID, text:ZmMsg.mobileDeviceId, width:ZmMsg.COLUMN_WIDTH_ID_MDL}));
136 	headerList.push(new DwtListHeaderItem({field:ZmMobileDeviceListView.F_STATUS, text:ZmMsg.status, width:ZmMsg.COLUMN_WIDTH_STATUS_MDL}));
137 	headerList.push(new DwtListHeaderItem({field:ZmMobileDeviceListView.F_PROTOCOL, text:ZmMsg.mobileProtocolVersion, width:ZmMsg.COLUMN_WIDTH_PROTOCOL_MDL}));
138 	headerList.push(new DwtListHeaderItem({field:ZmMobileDeviceListView.F_PROVISIONABLE, text:ZmMsg.mobileProvisionable, width:ZmMsg.COLUMN_WIDTH_PROVISIONABLE_MDL}));
139 
140 	return headerList;
141 };
142 
143 ZmMobileDeviceListView.prototype._getCellContents =
144 function(html, idx, item, field, colIdx, params) {
145 
146 	if (field == ZmMobileDeviceListView.F_DEVICE) {
147 		html[idx++] = "<nobr>";
148 		html[idx++] = item.type;
149 		if (item.ua) {
150 			html[idx++] = " (";
151 			html[idx++] = item.ua;
152 			html[idx++] = ")";
153 		}
154 		html[idx++] = "</nobr>";
155 	} else if (field == ZmMobileDeviceListView.F_STATUS) {
156 		html[idx++] = item.getStatusString();
157 	} else if (field == ZmMobileDeviceListView.F_ID) {
158 		html[idx++] = item.id;
159 	} else if (field == ZmMobileDeviceListView.F_PROTOCOL) {
160 		html[idx++] = item.protocol;
161 	} else if (field == ZmMobileDeviceListView.F_PROVISIONABLE) {
162 		html[idx++] = item.provisionable ? AjxMsg.yes : AjxMsg.no;
163 	}
164 
165 	return idx;
166 };
