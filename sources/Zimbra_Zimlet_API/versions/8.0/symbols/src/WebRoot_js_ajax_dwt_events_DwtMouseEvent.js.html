 17 /**
 18  * 
 19  * @private
 20  */
 21 DwtMouseEvent = function() {
 22 	DwtUiEvent.call(this, true);
 23 	this.reset(true);
 24 };
 25 
 26 DwtMouseEvent.prototype = new DwtUiEvent;
 27 DwtMouseEvent.prototype.constructor = DwtMouseEvent;
 28 
 29 DwtMouseEvent.prototype.toString = 
 30 function() {
 31 	return "DwtMouseEvent";
 32 };
 33 
 34 DwtMouseEvent.NONE		= 0;
 35 DwtMouseEvent.LEFT 		= 1;
 36 DwtMouseEvent.MIDDLE	= 2;
 37 DwtMouseEvent.RIGHT		= 3;
 38 
 39 DwtMouseEvent.prototype.reset =
 40 function(dontCallParent) {
 41 	if (!dontCallParent) {
 42 		DwtUiEvent.prototype.reset.call(this);
 43 	}
 44 	this.button = 0;
 45 };
 46 
 47 DwtMouseEvent.prototype.setFromDhtmlEvent =
 48 function(ev, obj) {
 49 	ev = DwtUiEvent.prototype.setFromDhtmlEvent.apply(this, arguments);
 50 	if (!ev) { return; }
 51 
 52 	if (ev.which) { // Mozilla or Safari3
 53 		switch (ev.which) {
 54 			case 1:  this.button = DwtMouseEvent.LEFT; break;
 55 			case 2:  this.button = DwtMouseEvent.MIDDLE; break;
 56 			case 3:  this.button = DwtMouseEvent.RIGHT; break;
 57 			default: this.button = DwtMouseEvent.NONE;
 58 		}
 59 	} else if (ev.button) { // IE
 60 		if ((ev.button & 1) != 0) {
 61 			this.button = DwtMouseEvent.LEFT;
 62 		} else if ((ev.button & 2) != 0) {
 63 			this.button = DwtMouseEvent.RIGHT;
 64 		} else if ((ev.button & 4) != 0) {
 65 			this.button = DwtMouseEvent.MIDDLE;
 66 		} else {
 67 			this.button = DwtMouseEvent.NONE;
 68 		}
 69 	}
 70 
 71 	if (AjxEnv.isMac && this.button) {
 72 		// Mac only comes with one button, but can take a USB multibutton mouse. Single-button will translate
 73 		// CTRL-LEFT into RIGHT, but leave ctrlKey set to true. Convert that into vanilla RIGHT click. That
 74 		// means we can't distinguish a CTRL-RIGHT, but oh well.
 75 		if (this.ctrlKey && (this.button == DwtMouseEvent.LEFT || this.button == DwtMouseEvent.RIGHT)) {
 76 			this.button = DwtMouseEvent.RIGHT;
 77 			this.ctrlKey = false;
 78 		}
 79 		// allow alt-key to be used for ctrl-select
 80 		if (this.altKey) {
 81 			this.ctrlKey = true;
 82 			this.altKey = false;
 83 		}
 84 	}
 85 };
