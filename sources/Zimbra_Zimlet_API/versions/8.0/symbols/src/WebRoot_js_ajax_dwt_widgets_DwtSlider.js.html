 16 /**
 17  * Creates a slider.
 18  * @constructor
 19  * @class
 20  * This class represents a slider.
 21  *
 22  * @param {DwtControl}	parent    the parent widget
 23  * @param {DwtSlider.HORIZONTAL|DwtSlider.VERTICAL}	orientation		the orientation of the slider
 24  * @param {string}	[className] 	the CSS class. If not provided defaults to "DwtHorizontalSlider" or "DwtVerticalSlider"
 25  * @param {constant}	[posStyle=DwtControl.STATIC_STYLE] the positioning style (see {@link DwtControl})
 26  * 
 27  * @extends		DwtControl
 28  */
 29 DwtSlider = function(parent, orientation, className, posStyle) {
 30     if (arguments.length == 0) return;
 31     this._orientation = orientation || DwtSlider.HORIZONTAL;
 32     className = className || (this._orientation == DwtSlider.HORIZONTAL ? "DwtHorizontalSlider" : "DwtVerticalSlider");
 33     DwtControl.call(this, {parent:parent, className:className, posStyle:posStyle});
 34 
 35     this._size = 0;
 36     this._buttonSize = 0;
 37 
 38     this._value = 0;
 39     this._minimum = 0;
 40     this._maximum = 100;
 41     
 42     this._isDragging = false;
 43 
 44 	DwtDragTracker.init(this, null, 0, 0, this._dragListener, this);
 45 
 46     this._createHtml();
 47 };
 48 
 49 DwtSlider.prototype = new DwtControl;
 50 DwtSlider.prototype.constructor = DwtSlider;
 51 
 52 /**
 53  * Defines the "horizontal" orientation.
 54  */
 55 DwtSlider.HORIZONTAL = 1;
 56 /**
 57  * Defines the "vertical" orientation.
 58  */
 59 DwtSlider.VERTICAL = 2;
 60 
 61 DwtSlider.prototype.toString =
 62 function() {
 63     return "DwtSlider";
 64 };

 66 /**
 67  * Sets the value of the slider, moving the position button accordingly.
 68  *
 69  * @param {number}		value		the value
 70  * @param {boolean}	notify			if <code>true</code>, to notify change listeners of the new value
 71  */
 72 DwtSlider.prototype.setValue =
 73 function(value, notify) {
 74 	// Adjust the value into the valid range.
 75 	value = Math.max(this._minimum, value);
 76 	value = Math.min(this._maximum, value);
 77 	this._value = value;
 78 
 79 	// Move the button.
 80 	var location = this._valueToLocation(value);
 81     var property = this._orientation == DwtSlider.HORIZONTAL ? "left" : "top";
 82     var element = this._getButtonElement();
 83     element.style[property] = location;
 84     
 85 	// Send notification.
 86 	if (notify) {
 87 		if (!this._changeEvent) {
 88 			this._changeEvent = new DwtEvent(true);
 89 			this._changeEvent.dwtObj = this;
 90 		}
 91 	    this.notifyListeners(DwtEvent.ONCHANGE, this._changeEvent);
 92 	}
 93 };

 98  * @return	{number}	the value

100 DwtSlider.prototype.getValue =

102 	return this._value;
103 };

105 /**
106  * Sets the range and value of the slider.
107  *
108  * @param {number}	minimum	the minimum allowed value
109  * @param {number}	maximum	the maximum allowed value
110  * @param {number}	value		the value
111  * @param {boolean}	notify	if <code>true</code>, notify change listeners of the new value
112  */
113 DwtSlider.prototype.setRange =
114 function(minimum, maximum, newValue, notify) {
115 	if (minimum >= maximum) {
116 		throw new DwtException("Invalid slider range: [" + minimum + ", " + maximum + "]");
117 	};
118 
119 	this._minimum = minimum;
120 	this._maximum = maximum;
121 	if (typeof newValue == "undefined") {
122 		newValue = minimum;
123 	}
124 	this.setValue(newValue, notify);
125 };

167 DwtSlider.prototype._setLocation =
168 function(location, notify) {
169 	var value = this._locationToValue(location);
170 	this.setValue(value, notify);
171 };
172 
173 DwtSlider.prototype._getLocation =
174 function() {
175 	return this._valueToLocation(this._value);
176 };
177 
178 DwtSlider.prototype._valueToLocation =
179 function(value) {
180 	if (this._orientation == DwtSlider.HORIZONTAL) {
181 	    return (value - this._minimum) / (this._maximum - this._minimum) * (this._size - this._buttonSize);
182 	} else {
183 	    return this._size - this._buttonSize - (value - this._minimum) / (this._maximum - this._minimum) * (this._size - this._buttonSize);
184 	}
185 };
186 
187 DwtSlider.prototype._locationToValue =
188 function(location) {
189 	if (this._orientation == DwtSlider.HORIZONTAL) {
190 	    return location / (this._size - this._buttonSize) * (this._maximum - this._minimum) + this._minimum;
191 	} else {
192 	    return (this._size - this._buttonSize - location) / (this._size - this._buttonSize) * (this._maximum - this._minimum) + this._minimum;
193 	}
194 };
195 
196 DwtSlider.prototype._calculateSizes =
197 function() {
198 	var property = this._orientation == DwtSlider.HORIZONTAL ? "x" : "y";
199 	this._buttonSize = Dwt.getSize(this._getButtonElement())[property];
200 	this._size = Dwt.getSize(this.getHtmlElement())[property];
201 	if (this._buttonSize >= this._size) {
202 		throw new DwtException("Invalid slider sizes");
203 	}
204 };
205 
206 DwtSlider.prototype._getButtonElement =
207 function() {
208 	return document.getElementById(this._htmlElId + "_button");
209 };
210 
211 DwtSlider.prototype._createHtml =
212 function() {
213     var element = this.getHtmlElement();
214     var args = { id:this._htmlElId };
215     var template = this._orientation == DwtSlider.HORIZONTAL ? 
216     	"dwt.Widgets#DwtHorizontalSlider" : 
217     	"dwt.Widgets#DwtVerticalSlider";
218     element.innerHTML = AjxTemplate.expand(template, args);
219     this._calculateSizes();
220 };
221 
222 DwtSlider.prototype._dragListener =
223 function(obj, a, b) {
224 	var elementProperty = this._orientation == DwtSlider.HORIZONTAL ? "x" : "y";
225 	var eventProperty = this._orientation == DwtSlider.HORIZONTAL ? "docX" : "docY";
226 	if (obj.state == DwtDragTracker.STATE_START) {
227 		// If clicked outside of button, move button immediately.
228 		var windowLocation = Dwt.toWindow(this.getHtmlElement(), 0, 0);
229 		var clickLocation = obj.mouseEv[eventProperty] - windowLocation[elementProperty];
230 		var buttonLocation = this._getLocation();
231 		if (clickLocation < buttonLocation || clickLocation > (buttonLocation + this._buttonSize)) {
232 			this._setLocation(clickLocation - this._buttonSize / 2, true);
233 		}
234 
235 		// Save the original position in the tracker's user data.
236 		obj.userData = { location: this._getLocation(), value: this._value };
237 		this._isDragging = true;
238 		this._moved = false;
239 	} else {
240 		if (obj.state == DwtDragTracker.STATE_END) {
241 			this._isDragging = false;
242 		} else if (obj.state == DwtDragTracker.STATE_DRAGGING) {
243 			this._moved = true;
244 		}
245 		if (this._moved) {
246 			var location = obj.userData.location + obj.delta[elementProperty];
247 			this._setLocation(location, true);
248 		}
249 	}
250 };
