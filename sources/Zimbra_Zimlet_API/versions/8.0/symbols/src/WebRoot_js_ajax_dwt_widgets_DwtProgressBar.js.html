 17 /**
 18  * Creates a progress bar.
 19  * @constructor
 20  * @class
 21  * This class represents a progress bar.
 22  * 
 23  * @param {DwtComposite}	parent    the parent container
 24  * @param {string}	className the CSS class name
 25  * @param {constant}	posStyle  the position style (see {@link DwtControl})
 26  * 
 27  * @author Greg Solovyev
 28  * 
 29  * @extends		DwtComposite
 30  */
 31 DwtProgressBar = function(parent, className, posStyle) {
 32 	if (arguments.length == 0) return;
 33 	posStyle = posStyle || DwtControl.STATIC_STYLE;
 34 	DwtComposite.call(this, {parent:parent, posStyle:posStyle});
 35 	this._maxValue = 100;
 36 	this._value = 0;
 37 	this._quotabarDiv = null;
 38 	this._quotausedDiv = null;	
 39 
 40 	this._progressBgColor = null;// "#66cc33";	//	MOW: removing this so the color can be skinned
 41 												// 		set the color in the class "quotaused"
 42 	this._progressCssClass = "quotaused";
 43 	
 44 	this._wholeBgColor = null;
 45 	this._wholeCssClass = "quotabar";	
 46 	this._createHTML();
 47 }
 48 
 49 DwtProgressBar.prototype = new DwtComposite;
 50 DwtProgressBar.prototype.constructor = DwtProgressBar;

117 /**
118  * Sets the progress value.
119  * 
120  * @param	{number}		val		the value
121  */
122 DwtProgressBar.prototype.setValue = 
123 function(val) {
124 	this._value = parseInt(val);
125 	var percent;
126 
127 	if(this._value == this._maxValue)
128 		percent = 100;
129 	else 
130 		percent = Math.min(Math.round((this._value / this._maxValue) * 100), 100);	
131 
132 	if(isNaN(percent))
133 		percent = "0";
134 			
135 	if(!this._quotabarDiv) {
136 		this._quotabarDiv = document.createElement("div")
137 		if(this._wholeCssClass)
138 			this._quotabarDiv.className = this._wholeCssClass;
139 
140 		if(this._wholeBgColor)
141 			this._quotabarDiv.backgroundColor = this._wholeBgColor;
142 		
143 		this._cell.appendChild(this._quotabarDiv);
144 	}
145 	if(!this._quotausedDiv) {
146 		this._quotausedDiv = document.createElement("div")
147 		if(this._progressCssClass)
148 			this._quotausedDiv.className = this._progressCssClass;
149 			
150 		if(this._progressBgColor)
151 			this._quotausedDiv.style.backgroundColor = this._progressBgColor;
152 			
153 		this._quotabarDiv.appendChild(this._quotausedDiv);			
154 	}	
155 
156 	this._quotausedDiv.style.width = percent + "%";
157 }

159 /**
160  * Sets the value by percentage.
161  * 
162  * @param	{string}		percent		the value as a percentage (for example: "10%")
163  */
164 DwtProgressBar.prototype.setValueByPercent =
165 function (percent){
166 	this.setMaxValue(100);
167 	this.setValue (percent.replace(/\%/gi, ""));
168 }

200 /**
201  * Sets the label.
202  * 
203  * @param	{string}		text		the label
204  * @param	{boolean}		isRightAlign	if <code>true</code>, if the label is right aligned
205  */
206 DwtProgressBar.prototype.setLabel =
207 function( text, isRightAlign) {
208 	var labelNode = document.createTextNode(text);
209 	var position = isRightAlign ? -1 : 0;
210 	var labelCell = this._row.insertCell(position) ;
211 	labelCell.appendChild (labelNode);
212 }

218 DwtProgressBar.prototype._createHTML = 
219 function() {
220 	this._table = document.createElement("table");
221 	this._table.border = this._table.cellpadding = this._table.cellspacing = 0;	
222 
223 	this._row = this._table.insertRow(-1);
224 
225 	//if(AjxEnv.isLinux)
226 		//this._row.style.lineHeight = 13;
227 	
228 	this._cell = this._row.insertCell(-1);
229 	
230 	this.getHtmlElement().appendChild(this._table);
231 }
