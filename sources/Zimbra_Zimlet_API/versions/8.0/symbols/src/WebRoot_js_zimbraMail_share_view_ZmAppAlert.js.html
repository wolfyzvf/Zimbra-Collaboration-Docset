 17 /**
 18  * This class represents an alert that highlights and flashes an application tab.
 19  *
 20  * @param {ZmApp}		app 		the application
 21  * @class
 22  * @private
 23  */
 24 ZmAppAlert = function(app) {
 25 	this.app = app;
 26 };
 27 
 28 ZmAppAlert.prototype.isZmAppAlert = true;
 29 ZmAppAlert.prototype.toString = function() { return "ZmAppAlert"; };
 30 
 31 /**
 32  * Starts the alert.
 33  */
 34 ZmAppAlert.prototype.start =
 35 function() {
 36 	var appButton = this._getAppButton();
 37 	if (!appButton) { return; }
 38 	
 39     if (!appButton.isSelected) {
 40 		appButton.showAlert(true);
 41         //add a stop alert listener
 42         if (!this._stopAlertListenerObj) {
 43            this._stopAlertListenerObj = new AjxListener(this, this.stop);
 44            appButton.addSelectionListener(this._stopAlertListenerObj);
 45         }
 46     }
 47 };
 48 
 49 /**
 50  * Stops the alert.
 51  */
 52 ZmAppAlert.prototype.stop =
 53 function() {
 54 	var appButton = this._getAppButton();
 55 	if (!appButton) { return; }
 56     if (appButton.isSelected) {
 57         appButton.showAlert(false);
 58     }
 59 };
 60 
 61 ZmAppAlert.prototype._getAppButton =
 62 function() {
 63 	return appCtxt.getAppController().getAppChooserButton(this.app.getName());
 64 };
