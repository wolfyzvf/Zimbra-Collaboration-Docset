 21 /**
 22  * Creates the task folder.
 23  * @class
 24  * This class represents a task folder.
 25  * 
 26  * @author Parag Shah
 27  *
 28  * @param	{Hash}	params		a hash of parameters
 29  * @param {int}	params.id			the numeric ID
 30  * @param {String}	params.name		the name
 31  * @param {ZmOrganizer}	params.parent		the parent organizer
 32  * @param {ZmTree}	params.tree		the tree model that contains this organizer
 33  * @param {String}	params.color	the color
 34  * @param {String}	params.url		the URL for this organizer's feed
 35  * @param {String}	params.owner	the owner
 36  * @param {String}	params.zid		the Zimbra id of owner, if remote share
 37  * @param {String}	params.rid		the remote id of organizer, if remote share
 38  * @param {String}	params.restUrl	[the REST URL of this organizer.
 39  * 
 40  * @extends		ZmFolder
 41  */
 42 ZmTaskFolder = function(params) {
 43 	params.type = ZmOrganizer.TASKS;
 44 	ZmFolder.call(this, params);
 45 }

 62 ZmTaskFolder.prototype.getIcon =
 63 function() {
 64 	if (this.id == ZmFolder.ID_ROOT)	{ return null; }
 65 	if (this.link)						{ return "SharedTaskList"; }
 66 	return "TaskList";
 67 };

 75 /**
 76  * Sets the free/busy.
 77  * 
 78  * @param	{AjxCallback}		callback		the callback
 79  * @param	{AjxCallback}		errorCallback		the error callback
 80  * @param	{Boolean}		exclude		checks to exclose free busy
 81  */
 82 ZmTaskFolder.prototype.setFreeBusy =
 83 function(exclude, callback, errorCallback) {
 84 	if (this.excludeFreeBusy == exclude) return;
 85 	// NOTE: Don't need to store the value since the response will
 86 	//       report that the object was modified.
 87 	this._organizerAction({action: "fb", attrs: {excludeFreeBusy: exclude ? "1" : "0"}, callback: callback, errorCallback: errorCallback});
 88 };

 90 ZmTaskFolder.prototype.mayContain =
 91 function(what) {
 92 	if (!what) return true;
 93 
 94 	var invalid = false;
 95 
 96 	if (this.id == ZmFolder.ID_ROOT) {
 97 		// cannot drag anything onto root folder
 98 		invalid = true;
 99 	} else if (this.link) {
100 		// cannot drop anything onto a read-only task folder
101 		invalid = this.isReadOnly();
102 	}
103 
104 	if (!invalid) {
105 		// An item or an array of items is being moved
106 		var items = (what instanceof Array) ? what : [what];
107 		var item = items[0];
108 
109 		if (item.type != ZmItem.TASK) {
110 			// only tasks are valid for task folders
111 			invalid = true;
112 		} else {
113 			// can't move items to folder they're already in; we're okay if
114 			// we have one item from another folder
115 			if (!invalid && item.folderId) {
116 				invalid = true;
117 				for (var i = 0; i < items.length; i++) {
118 					var item = items[i];
119 					var folder = appCtxt.getById(item.folderId);
120 					if (folder != this) {
121 						invalid = false;
122 						break;
123 					}
124 				}
125 			}
126 		}
127 	}
128 
129 	return !invalid;
130 };

135 ZmTaskFolder.prototype.notifyCreate =
136 function(obj) {
137 	var t = ZmFolderTree.createFromJs(this, obj, this.tree);
138 	var i = ZmOrganizer.getSortIndex(t, ZmTaskFolder.sortCompare);
139 	this.children.add(t, i);
140 	t._notify(ZmEvent.E_CREATE);
141 };
142 
143 ZmTaskFolder.prototype.notifyModify =
144 function(obj) {
145 	ZmFolder.prototype.notifyModify.call(this, obj);
146 
147 	if (obj.f != null && !obj._isRemote) {
148 		this._parseFlags(obj.f);
149 		// TODO: Should a F_EXCLUDE_FB property be added to ZmOrganizer?
150 		//       It doesn't make sense to require the base class to know about
151 		//       all the possible fields in sub-classes. So I'm just using the
152 		//       modified property name as the key.
153 		var fields = {};
154 		fields["excludeFreeBusy"] = true;
155 		this._notify(ZmEvent.E_MODIFY, {fields: fields});
156 	}
157 };

172 /**
173  * Comparison function for folders. Intended for use on a list of user folders
174  * through a call to <code>Array.sort()</code>.
175  * 
176  * @param	{ZmTaskFolder}	calA		item A
177  * @param	{ZmTaskFolder}	calB		item B
178  * 
179  * @return	{int}	0 if the folders match
180  */
181 ZmTaskFolder.sortCompare =
182 function(calA, calB) {
183 	var check = ZmOrganizer.checkSortArgs(calA, calB);
184 	if (check != null) return check;
185 
186 	// links appear after personal calendars
187 	if (calA.link != calB.link) {
188 		return calA.link ? 1 : -1;
189 	}
190 
191 	// sort by calendar name
192 	var calAName = calA.name.toLowerCase();
193 	var calBName = calB.name.toLowerCase();
194 	if (calAName < calBName) return -1;
195 	if (calAName > calBName) return 1;
196 	return 0;
197 };
