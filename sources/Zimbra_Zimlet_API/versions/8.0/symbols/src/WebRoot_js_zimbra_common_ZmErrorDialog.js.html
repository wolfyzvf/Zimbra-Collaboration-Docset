 22 /**
 23  * Creates an error dialog.
 24  * @class
 25  * Creates an error dialog which will have a "Send Error Report" button.
 26  * A normal {@link DwtMessageDialog} with a "Send Error Report" button that will post user info to the 
 27  * server when clicked.
 28  * 
 29  * @param	{Object}	parent		the parent
 30  * @param	{Hash}		msgs		a hash of messages
 31  * @param	{String}	msgs.showDetails		the show details message
 32  * @param	{String}	msgs.hideDetails		the hide details message
 33  * 
 34  * @extends DwtMessageDialog
 35  */
 36 ZmErrorDialog = function(parent, msgs) {
 37 
 38 	// go ahead and cache the navigator and subject info now (since it should never change)		
 39 	this._strNav = this._getNavigatorInfo();
 40 	this._subjPfx = this._getSubjectPrefix();
 41 
 42 	var reportButton = new DwtDialog_ButtonDescriptor(ZmErrorDialog.REPORT_BUTTON, msgs.report, DwtDialog.ALIGN_LEFT);
 43 	var detailButton = new DwtDialog_ButtonDescriptor(ZmErrorDialog.DETAIL_BUTTON, msgs.showDetails, DwtDialog.ALIGN_LEFT);
 44 	DwtMessageDialog.call(this, {parent:parent, extraButtons:[reportButton, detailButton], id:"ErrorDialog"});
 45 
 46 	this.registerCallback(ZmErrorDialog.REPORT_BUTTON, this._reportCallback, this);
 47 	this.registerCallback(ZmErrorDialog.DETAIL_BUTTON, this.showDetail, this);
 48 	
 49 	this._showDetailsMsg = msgs.showDetails;
 50 	this._hideDetailsMsg = msgs.hideDetails;
 51 
 52 	this._setAllowSelection();
 53 };
 54 
 55 ZmErrorDialog.prototype = new DwtMessageDialog;
 56 ZmErrorDialog.prototype.constructor = ZmErrorDialog;
 57

 87 /**
 88  * Resets the dialog.
 89  * 
 90  */
 91 ZmErrorDialog.prototype.reset =
 92 function() {
 93 	this.setDetailString();
 94 	DwtMessageDialog.prototype.reset.call(this);
 95 };
 96 
 97 /**
 98 * Sets the text to display when the "Show Details" button is pressed.
 99 *
100 * @param {String}	text	the detail text
101 */
102 ZmErrorDialog.prototype.setDetailString = 
103 function(text) {
104 	if (!(this._button[ZmErrorDialog.DETAIL_BUTTON])) { return; }
105 
106 	this._button[ZmErrorDialog.DETAIL_BUTTON].setVisible(text != null);
107 	this._detailStr = text;
108 };
109 
110 /**
111  * Sets the message style (info/warning/critical) and content.
112  *
113  * @param {String}	msgStr		the message text
114  * @param {String}	detailStr	the detail text
115  * @param {constant}	style		the style (see {@link DwtMessageDialog} <code>_STYLE</code> constants)
116  * @param {String}	title		the dialog box title
117  */
118 ZmErrorDialog.prototype.setMessage =
119 function(msgStr, detailStr, style, title) {
120 	this._msgStr = msgStr;
121 	this.setDetailString(detailStr);
122 	this._msgStyle = style;
123 	this._msgTitle = title;
124 
125 	// clear the 'detailsVisible' flag and reset the title of the 'showDetails' button
126 	this._detailsVisible = false;
127 	this._button[ZmErrorDialog.DETAIL_BUTTON].setText(this._showDetailsMsg);
128 	
129 	// Set the content, enveloped
130 	this._updateContent();
131 };
132 
133 /**
134  * Sets/updates the content
135  */
136 ZmErrorDialog.prototype._updateContent = 
137 function() {
138 	var data = {
139 		message: this._msgStr,
140 		detail: this._detailStr,
141 		showDetails: this._detailsVisible
142 	};
143 	var html = AjxTemplate.expand("zimbra.Widgets#ZmErrorDialogContent", data);
144 	this.setSize(Dwt.CLEAR, this._detailsVisible ? "300" : Dwt.CLEAR);
145 	DwtMessageDialog.prototype.setMessage.call(this, html, this._msgStyle, this._msgTitle);
146 };
147 
148 /**
149  * Pops-up the error dialog.
150  * 
151  * @param {Object}	loc				the desired location
152  * @param {Boolean}	hideReportButton	if <code>true</code>, do not show "Send Error Report" button
153  * 
154  */
155 ZmErrorDialog.prototype.popup =
156 function(loc, hideReportButton) {
157 	if (hideReportButton) {
158 		this.setButtonVisible(ZmErrorDialog.REPORT_BUTTON, false);
159 	}
160 	DwtMessageDialog.prototype.popup.call(this, loc);
161 };
162 
163 /**
164  * Pops-down the dialog.
165  * 
166  */
167 ZmErrorDialog.prototype.popdown =
168 function() {
169 	DwtMessageDialog.prototype.popdown.call(this);
170 
171 	// reset dialog
172 	this.setSize(Dwt.CLEAR, Dwt.CLEAR);
173 	this.setButtonVisible(ZmErrorDialog.REPORT_BUTTON, true);
174 };

179 /**
180  * @private
181  */
182 ZmErrorDialog.prototype._getNavigatorInfo =
183 function() {
184 	var strNav = [];
185 	var idx = 0;
186 
187 	// Add the url
188 	strNav[idx++] = "\n\n";
189 	strNav[idx++] = "href: ";
190 	strNav[idx++] = location.href;
191 	strNav[idx++] = "\n";
192 
193 	for (var i in navigator) {
194 		// Skip functions
195 		if(typeof navigator[i] == "function") {continue;}
196 		if(typeof navigator[i] == "unknown") {continue;}	// IE7
197 		if(AjxEnv.isIE && i === "mimeTypes") {continue;}
198 		strNav[idx++] = i + ": " + navigator[i] + "\n";
199 	}
200 	return strNav.join("");
201 };
202 
203 /**
204  * @private
205  */
206 ZmErrorDialog.prototype._getSubjectPrefix = 
207 function() {
208 	var strSubj = [];
209 	var idx = 0;
210 
211 	strSubj[idx++] = "ER: ";
212 
213 	if (AjxEnv.isIE) 				strSubj[idx++] = "IE ";
214 	else if (AjxEnv.isFirefox)		strSubj[idx++] = "FF ";
215 	else if (AjxEnv.isMozilla)		strSubj[idx++] = "MOZ ";
216 	else if (AjxEnv.isSafari)		strSubj[idx++] = "SAF ";
217 	else if (AjxEnv.isOpera)		strSubj[idx++] = "OPE ";
218 	else							strSubj[idx++] = "UKN ";
219 
220 	if (AjxEnv.isWindows)			strSubj[idx++] = "WIN ";
221 	else if (AjxEnv.isLinux)		strSubj[idx++] = "LNX ";
222 	else if (AjxEnv.isMac)			strSubj[idx++] = "MAC ";
223 	else							strSubj[idx++] = "UNK ";
224 
225 	strSubj[idx++] = appCtxt.get(ZmSetting.CLIENT_VERSION) + " ";
226 	return strSubj.join("");
227 };
228 
229 /**
230  * @private
231  */
232 ZmErrorDialog.prototype._getUserPrefs = 
233 function() {
234 	var currSearch = appCtxt.getCurrentSearch();
235 	var strPrefs = [];
236 	var idx = 0;
237 
238 	// Add username and current search
239 	strPrefs[idx++] = "\n\n";
240 	strPrefs[idx++] = "username: ";
241 	strPrefs[idx++] = appCtxt.get(ZmSetting.USERNAME);
242 	strPrefs[idx++] = "\n";
243 	if (currSearch) {
244 		strPrefs[idx++] = "currentSearch: ";
245 		strPrefs[idx++] = currSearch.query;
246 		strPrefs[idx++] = "\n";
247 	}
248 	for (var i in ZmSetting.INIT) {
249 		if (ZmSetting.INIT[i][0]) {
250 			strPrefs[idx++] = ZmSetting.INIT[i][0];
251 			strPrefs[idx++] = ": ";
252 			strPrefs[idx++] = ("" + ZmSetting.INIT[i][3]);
253 			strPrefs[idx++] = "\n";
254 		}
255 	}
256 	return strPrefs.join("");
257 };

261 /**
262  * @private
263  */
264 ZmErrorDialog.prototype._reportCallback =
265 function() {
266 	this._iframe = document.createElement("iframe");
267 	this._iframe.style.width = this._iframe.style.height = 0;
268 	this._iframe.style.visibility = "hidden";
269 
270 	var contentDiv = this._getContentDiv();
271 	contentDiv.appendChild(this._iframe);
272 
273 	var strPrefs = this._getUserPrefs();
274 	var formId = Dwt.getNextId();
275 
276 	// generate html form for submission via POST
277 	var html = [];
278 	var idx = 0;
279 	var subject = this._subjPfx + this._detailStr.substring(0,40);
280 	var scheme = (location.protocol == 'https:') ? "https:" : "http:";
281 	html[idx++] = "<html><head></head><body><form id='";
282 	html[idx++] = formId;
283 	html[idx++] = "' method='POST' action='";
284 	html[idx++] = scheme;
285 	html[idx++] = appCtxt.get(ZmSetting.ERROR_REPORT_URL) || ZmErrorDialog.DEFAULT_REPORT_URL;
286 	html[idx++] = "'>";
287 	html[idx++] = "<textarea name='details'>";
288 	html[idx++] = this._detailStr;
289 	html[idx++] = "version - ";
290 	html[idx++] = appCtxt.get(ZmSetting.CLIENT_VERSION);
291 	html[idx++] = "\n";
292 	html[idx++] = "release - ";
293 	html[idx++] = appCtxt.get(ZmSetting.CLIENT_RELEASE);
294 	html[idx++] = "\n";
295 	html[idx++] = "date - ";
296 	html[idx++] = appCtxt.get(ZmSetting.CLIENT_DATETIME);
297 	html[idx++] = "</textarea><textarea name='navigator'>";
298 	html[idx++] = this._strNav;
299 	html[idx++] = "</textarea><textarea name='prefs'>";
300 	html[idx++] = strPrefs;
301 	html[idx++] = "</textarea><textarea name='subject'>";
302 	html[idx++] = subject;
303 	html[idx++] = "</textarea></form></body></html>";
304 
305 	var idoc = Dwt.getIframeDoc(this._iframe);
306 	idoc.open();
307 	idoc.write(html.join(""));
308 	idoc.close();
309 
310 	// submit the form!
311 	var form = idoc.getElementById(formId);
312 	if (form) {
313 		form.submit();
314 		appCtxt.setStatusMsg(ZmMsg.errorReportSent);
315 	}
316 
317 	this.popdown();
318 };

320 /**
321  * Displays the detail text
322  */
323 ZmErrorDialog.prototype.showDetail = 
324 function() {
325 	this._detailsVisible = !this._detailsVisible;
326 	this._updateContent();
327 	this._button[ZmErrorDialog.DETAIL_BUTTON].setText(this._detailsVisible ? this._hideDetailsMsg : this._showDetailsMsg);
328 };
