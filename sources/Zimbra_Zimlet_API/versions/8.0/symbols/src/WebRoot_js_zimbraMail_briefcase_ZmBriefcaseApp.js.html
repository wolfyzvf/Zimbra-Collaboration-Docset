214 /**
215  * Checks for the creation of a briefcase or a mount point to one, or of an item
216  *
217  * @param {Hash}	creates		a hash of create notifications
218  * 
219  * @private
220  */
221 ZmBriefcaseApp.prototype.createNotify =
222 function(creates, force) {
223 
224 	if (!creates["folder"] && !creates["doc"] && !creates["link"]) { return; }
225 	if (!force && !this._noDefer && this._deferNotifications("create", creates)) { return; }
226 
227 	for (var name in creates) {
228 		var clist = creates[name];
229 		for (var i = 0; (clist != null) && i < clist.length; i++) {
230 			var create = clist[i];
231 			if (appCtxt.cacheGet(create.id)) { continue; }
232 
233 			if (name == "folder") {
234 				this._handleCreateFolder(create, ZmOrganizer.BRIEFCASE);
235 			} else if (name == "link") {
236 				this._handleCreateLink(create, ZmOrganizer.BRIEFCASE);
237 			} else if (name == "doc") {
238 				var bc = AjxDispatcher.run("GetBriefcaseController");
239                 bc.handleCreateNotify(create);
240 			}
241 		}
242 	}
243 };
244 
245 ZmBriefcaseApp.prototype.modifyNotify =
246 function(modifies, force) {
247     if (!modifies["doc"]) { return; }
248 	var bc = AjxDispatcher.run("GetBriefcaseController");
249     bc.handleModifyNotify(modifies);
250 };

252 ZmBriefcaseApp.prototype.handleOp =
253 function(op) {
254 
255 	switch (op) {
256 		case ZmOperation.NEW_FILE: {
257 			var loadCallback = new AjxCallback(this, this._handleNewItem);
258 			AjxDispatcher.require(["BriefcaseCore", "Briefcase"], false, loadCallback, null, true);
259 			break;
260 		}
261 		case ZmOperation.NEW_BRIEFCASE: {
262 			var loadCallback = new AjxCallback(this, this._handleLoadNewBriefcase);
263 			AjxDispatcher.require(["BriefcaseCore", "Briefcase"], false, loadCallback, null, true);
264 			break;
265 		}
266 		case ZmOperation.NEW_PRESENTATION: {
267 			var loadCallback = new AjxCallback(this, this.newDoc, [ZmMimeTable.APP_ZIMBRA_SLIDES]);
268 			AjxDispatcher.require(["BriefcaseCore", "Briefcase"], true, loadCallback, null);
269 			break;
270 		}
271 
272 		 case ZmOperation.NEW_SPREADSHEET: {
273 			 var newDocCallback = new AjxCallback(this, this.newDoc, [ZmMimeTable.APP_ZIMBRA_SPREADSHEET]);
274 			 AjxDispatcher.require(["BriefcaseCore", "Briefcase"], true, newDocCallback, null);
275 			 break;
276 		}
277 
278 		 case ZmOperation.NEW_DOC: {
279 			 var newDocCallback = new AjxCallback(this, this.newDoc, [ZmMimeTable.APP_ZIMBRA_DOC]);
280 			 AjxDispatcher.require(["BriefcaseCore", "Briefcase"], true, newDocCallback, null);
281 			 break;
282 		}
283 	}
284 };

286 /**
287  * Creates a new document.
288  * 
289  * @param	{String}	contentType		the content type
290  * @param	{String}	new				the document name
291  * @param	{String}	winName			the name of the popup doc window
292  */
293 ZmBriefcaseApp.prototype.newDoc =
294 function(contentType, name, winName) {
295 	var overviewController = appCtxt.getOverviewController();
296 	var treeController = overviewController.getTreeController(ZmOrganizer.BRIEFCASE);
297 	var folderId = ZmOrganizer.ID_BRIEFCASE;
298 	if (treeController) {
299 		var treeView = treeController.getTreeView(this.getOverviewId());
300 		var briefcase = treeView ? treeView.getSelected() : null;
301 		folderId = briefcase ? briefcase.id : ZmOrganizer.ID_BRIEFCASE;
302 	}
303 
304     if (AjxDispatcher.run("GetBriefcaseController").chkFolderPermission(folderId)) {
305         var url = this.getEditURLForContentType(contentType) + "?" + (name ?"name=" + name + "&" : "") + "l="+folderId + "&skin=" + appCurrentSkin + "&localeId=" + AjxEnv.DEFAULT_LOCALE;
306         if (window.appCoverageMode)
307             url = url + "&coverage=1";
308         var winname = winName || name || (new Date()).getTime().toString();
309         window.open(url, winname); //bug:44324 removed new launching window
310     }
311 };

343 /**
344  * Gets the edit URL.
345  * 
346  * @param	{String}	contentType		the content type
347  * 
348  * @return	{String}	the URL
349  */
350 ZmBriefcaseApp.prototype.getEditURLForContentType =
351 function(contentType) {
352 	AjxDispatcher.require("Startup1_1");
353 	var editPage = "Slides.jsp";
354 	switch(contentType) {
355 		case ZmMimeTable.APP_ZIMBRA_SLIDES:			editPage = "Slides.jsp"; break;
356 		case ZmMimeTable.APP_ZIMBRA_SPREADSHEET:	editPage = "SpreadsheetDoc.jsp"; break;

359 	};
360 	return (appContextPath + "/public/" + editPage);
361 };

422 /**
423  * Performs a search.
424  * 
425  * @param {Hash}	params			a hash of parameters
426  * @param {String}	params.folderId			the ID of briefcase folder to search in
427  * @param {String}	[params.query]				the query to send (overrides folderId)
428  * @param {AjxCallback}	[params.callback]			the callback
429  * @param {String}	[params.accountName]		the account name
430  * @param {Boolean}	[params.noRender]			if <code>true</code>, do not display results
431  */
432 ZmBriefcaseApp.prototype.search =
433 function(params) {
434 
435 	params = params || {};
436     var folderId = params.folderId || (appCtxt.isExternalAccount() ? this.getDefaultFolderId() : ZmOrganizer.ID_BRIEFCASE);
437 	var folder = appCtxt.getById(folderId);
438 
439 	var searchParams = {
440 		query:			params.query || folder.createQuery(),
441 		types:			[ZmItem.BRIEFCASE_ITEM],
442 		limit:			this.getLimit(),
443 		searchFor:		ZmId.ITEM_BRIEFCASE,
444 		callback:		params.callback,
445 		accountName:	params.accountName,
446 		noRender:		params.noRender
447 	};
448 	var sc = appCtxt.getSearchController();
449 	sc.searchAllAccounts = false;
450 	sc.search(searchParams);
451 };
