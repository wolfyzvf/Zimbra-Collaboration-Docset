 16 /**
 17  * Creates a new, empty filter rules controller.
 18  * @class
 19  * This class represents the filter rules controller. This controller manages
 20  * the filter rules page, which has a button toolbar and a list view of the rules.
 21  *
 22  * @author Conrad Damon
 23  *
 24  * @param {DwtShell}		container		the shell
 25  * @param {ZmPreferencesApp}	prefsApp		the preferences application
 26  * 
 27  * @extends		ZmController
 28  */
 29 ZmFilterRulesController = function(container, prefsApp, prefsView, parent, outgoing) {
 30 
 31 	ZmController.call(this, container, prefsApp);
 32 
 33 	this._prefsView = prefsView;
 34 	this._parent = parent;
 35 
 36 	this._filterRulesView = new ZmFilterRulesView(this._prefsView, this);
 37 
 38 	this._outgoing = Boolean(outgoing);
 39 
 40 	this._buttonListeners = {};
 41 	this._buttonListeners[ZmOperation.ADD_FILTER_RULE] = new AjxListener(this, this._addListener);
 42 	this._buttonListeners[ZmOperation.EDIT_FILTER_RULE] = new AjxListener(this, this._editListener);
 43 	this._buttonListeners[ZmOperation.REMOVE_FILTER_RULE] = new AjxListener(this, this._removeListener);
 44 	this._buttonListeners[ZmOperation.RUN_FILTER_RULE] = new AjxListener(this, this._runListener);
 45 	this._progressController = new ZmProgressController(container, prefsApp);
 46 };

 71 /**
 72  * Initializes the controller.
 73  * 
 74  * @param	{ZmToolBar}	toolbar		the toolbar
 75  * @param	{ZmListView}	listView		active list view
 76  * @param   {ZmListView}    listView        not active list view
 77  */
 78 ZmFilterRulesController.prototype.initialize =
 79 function(toolbar, listView, notActiveListView) {
 80 	// always reset the the rules to make sure we get the right one for the *active* account
 81 	this._rules = AjxDispatcher.run(this._outgoing ? "GetOutgoingFilterRules" : "GetFilterRules");
 82 
 83 	if (toolbar) {
 84 		var buttons = this.getToolbarButtons();
 85 		for (var i = 0; i < buttons.length; i++) {
 86 			var id = buttons[i];
 87 			if (this._buttonListeners[id]) {
 88 				toolbar.addSelectionListener(id, this._buttonListeners[id]);
 89 			}
 90 		}
 91 		this._resetOperations(toolbar, 0);
 92 	}
 93 
 94 	if (notActiveListView) {
 95 		this._notActiveListView = notActiveListView;
 96 		notActiveListView.addSelectionListener(new AjxListener(this, this._listSelectionListener));
 97 		notActiveListView.addActionListener(new AjxListener(this, this._listActionListener));
 98 		this.resetListView(0);
 99 	}
100 	
101 	if (listView) {
102 		this._listView = listView;
103 		listView.addSelectionListener(new AjxListener(this, this._listSelectionListener));
104 		listView.addActionListener(new AjxListener(this, this._listActionListener));
105 		this.resetListView(0);
106 	}
107 	
108 };

143 ZmFilterRulesController.prototype._handleResponseSetListView =
144 function(selectedIndex, result) {
145 	this._listView.set(result.getResponse().clone());
146 	this._notActiveListView.set(result.getResponse().clone());
147 	var rule = this._rules.getRuleByIndex(selectedIndex || 0);
148 	if (rule && rule.active) {
149 		this._listView.setSelection(rule);
150 	}
151 	else if (rule) {
152 		this._notActiveListView.setSelection(rule);
153 	}
154 };

268 ZmFilterRulesController.prototype.handleBeforeFilterChange =
269 function(okCallback, cancelCallback) {
270 	if (this._outgoing && (appCtxt.getSettings().getSetting(ZmSetting.SAVE_TO_SENT).getValue()===false || ZmPref.getFormValue(ZmSetting.SAVE_TO_SENT)===false)) {
271 		var dialog = appCtxt.getConfirmationDialog();
272 		if (!this._saveToSentMessage) {
273 			var html = [];
274 			var i = 0;
275 			html[i++] = "<table cellspacing=0 cellpadding=0 border=0><tr><td valign='top'>";
276 			html[i++] = AjxImg.getImageHtml("Warning_32");
277 			html[i++] = "</td><td class='DwtMsgArea'>";
278 			html[i++] = ZmMsg.filterOutgoingNoSaveToSentWarning;
279 			html[i++] = "</td></tr></table>";
280 			this._saveToSentMessage = html.join("");
281 		}
282 		var handleSaveToSentYesListener = new AjxListener(this, this._handleSaveToSentYes, [okCallback]);
283 		var handleSaveToSentNoListener = new AjxListener(this, this._handleSaveToSentNo, [okCallback]);
284 		
285 		dialog.popup(this._saveToSentMessage, handleSaveToSentYesListener, handleSaveToSentNoListener, cancelCallback);
286 		dialog.setTitle(AjxMsg.warningMsg);
287 	} else {
288 		if (okCallback)
289 			okCallback.run();
290 	}
291 };

293 ZmFilterRulesController.prototype._handleSaveToSentYes =
294 function(callback) {
295 	var settings = appCtxt.getSettings();
296 	var setting = settings.getSetting(ZmSetting.SAVE_TO_SENT);
297 	ZmPref.setFormValue(ZmSetting.SAVE_TO_SENT, true);
298 	if (!setting.getValue()) {
299 		setting.setValue(true);
300 		settings.save([setting], callback);
301 	} else {
302 		if (callback)
303 			callback.run();
304 	}
305 };

339 */
340 ZmFilterRulesController.prototype._removeListener =
341 function(ev) {
342 	var listView = this.getListView();
343 	if (!listView) { return; }
344 	var sel = listView.getSelection();

346 	//bug:16053 changed getYesNoCancelMsgDialog to getYesNoMsgDialog
347 	var ds = this._deleteShield = appCtxt.getYesNoMsgDialog();
348 	ds.reset();
349 	ds.registerCallback(DwtDialog.NO_BUTTON, this._clearDialog, this, this._deleteShield);
350 	ds.registerCallback(DwtDialog.YES_BUTTON, this._deleteShieldYesCallback, this, rule);
351 	var msg = AjxMessageFormat.format(ZmMsg.askDeleteFilter, AjxStringUtil.htmlEncode(rule.name));
352 	ds.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
353 	ds.popup();

356 ZmFilterRulesController.prototype._runListener =
357 function(ev) {
358 	// !!! do *NOT* get choose folder dialog from appCtxt since this one has checkboxes!
359 	if (!this._chooseFolderDialog) {
360 		AjxDispatcher.require("Extras");
361 		this._chooseFolderDialog = new ZmChooseFolderDialog(appCtxt.getShell());
362 	}
363 	this._chooseFolderDialog.reset();
364 	this._chooseFolderDialog.registerCallback(DwtDialog.OK_BUTTON, this._runFilterOkCallback, this, this._chooseFolderDialog);
365 
366 	// bug 42725: always omit shared folders
367 	var omit = {};
368 	var tree = appCtxt.getTree(ZmOrganizer.FOLDER);
369 	var children = tree.root.children.getArray();
370 	for (var i = 0; i < children.length; i++) {
371 		var child = children[i];
372 		if (child.type == ZmOrganizer.FOLDER && child.isRemote()) {
373 			omit[child.id] = true;
374 		}
375 	}
376 
377 	var params = {
378 		treeIds:		[ZmOrganizer.FOLDER],
379 		title:			ZmMsg.chooseFolder,
380 		overviewId:		this.toString() + (this._outgoing ? "_outgoing":"_incoming"),
381 		description:	ZmMsg.chooseFolderToFilter,
382 		skipReadOnly:	true,
383 		hideNewButton:	true,
384 		treeStyle:		DwtTree.CHECKEDITEM_STYLE,
385 		appName:		ZmApp.MAIL,
386 		omit:			omit
387 	};
388 	this._chooseFolderDialog.popup(params);
389 
390 	var foundForwardAction;
391 	var listView = this.getListView();
392 	var sel = listView && listView.getSelection();
393 	for (var i = 0; i < sel.length; i++) {
394 		if (sel[i].actions[ZmFilterRule.A_NAME_FORWARD]) {
395 			foundForwardAction = true;
396 			break;
397 		}
398 	}
399 
400 	if (foundForwardAction) {
401 		var dialog = appCtxt.getMsgDialog();
402 		dialog.setMessage(ZmMsg.filterForwardActionWarning);
403 		dialog.popup();
404 	}
405 };

475 /**
476 * Resets the toolbar button states, depending on which rule is selected.
477 * The list view enforces single selection only. If the first rule is selected,
478 * "Move Up" is disabled. Same for last rule and "Move Down". They're both
479 * disabled if there aren't at least two rules.
480 *
481 * @param parent		[ZmButtonToolBar]	the toolbar
482 * @param numSel		[int]				number of rules selected (0 or 1)
483 * @param sel		[Array]				list of selected rules
484 */
485 ZmFilterRulesController.prototype._resetOperations =
486 function(parent, numSel, sel) {
487 	var numRules = this._rules.getNumberOfRules();
488 	if (numSel == 1) {
489 		parent.enableAll(true);
490 	} else {
491 		parent.enableAll(false);
492 		parent.enable(ZmOperation.ADD_FILTER_RULE, true);
493 		if (numSel > 1) {
494 			parent.enable(ZmOperation.RUN_FILTER_RULE, true);
495 		}
496 	}
497 
498 	if (numRules == 0) {
499 		parent.enable(ZmOperation.EDIT_FILTER_RULE, false);
500 		parent.enable(ZmOperation.REMOVE_FILTER_RULE, false);
501 	}
502 };

572 /**
573  * do the work. (in this case apply filters). Either msgIds or query should be set but not both.
574  * @param msgIds {String} chunk of message ids to do the work on.
575  * @param query {String} query to run filter against
576  * @param callback
577  */
578 ZmFilterWork.prototype.doWork =
579 function(msgIds, query, callback) {
580 	var filterSel = this._filterSel;
581 	var soapDoc = AjxSoapDoc.create(this._outgoing ? "ApplyOutgoingFilterRulesRequest" : "ApplyFilterRulesRequest", "urn:zimbraMail");
582 	var filterRules = soapDoc.set("filterRules", null);
583 	for (var i = 0; i < filterSel.length; i++) {
584 		var rule = soapDoc.set("filterRule", null, filterRules);
585 		rule.setAttribute("name", filterSel[i].name);
586 	}
587 	var noBusyOverlay = false;
588 	if (msgIds) {
589 		var m = soapDoc.set("m");
590 		m.setAttribute("ids", msgIds.join(","));
591 	}
592 	else {
593 		soapDoc.set("query", query);
594 		noBusyOverlay = true;
595 	}
596 
597 	var params = {
598 		soapDoc: soapDoc,
599 		asyncMode: true,
600 		noBusyOverlay: noBusyOverlay,
601 		callback: (new AjxCallback(this, this._handleRunFilter, [callback]))
602 	};
603 	appCtxt.getAppController().sendRequest(params);
604 };
605 
606 /**
607  * private method - gets the result of the filter request, and keeps track of total messages affected.
608  * @param callback
609  * @param result
610  */
611 ZmFilterWork.prototype._handleRunFilter =
612 function(callback, result) {
613 	var r = result.getResponse();
614 	var resp = this._outgoing ? r.ApplyOutgoingFilterRulesResponse : r.ApplyFilterRulesResponse;
615 	var num = (resp && resp.m && resp.m.length)
616 		? (resp.m[0].ids.split(",").length) : 0;
617 	this._totalNumMessagesAffected += num;
618 	callback.run();
619 };
