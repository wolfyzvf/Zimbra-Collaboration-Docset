 22 /**
 23  * @class
 24  * This is a base class for dialogs. Given content, this class will take care of 
 25  * showing and hiding the dialog, as well as dragging it.
 26  * <p>
 27  * Dialogs always hang off the main shell since their stacking order is managed through z-index.
 28  *
 29  * @author Ross Dargahi
 30  * @author Conrad Damon
 31  * 
 32  * @param {hash}	params		a hash of parameters
 33  * @param	{DwtComposite}	params.parent	the parent widget (the shell)
 34  * @param	{string}	params.className		the CSS class
 35  * @param	{string}	params.title		the title
 36  * @param	{number}	[params.zIndex=Dwt.Z_DIALOG]		the z-index to set for this dialog when it is visible
 37  * @param	{DwtBaseDialog.MODAL|DwtBaseDialog.MODELESS}	[params.mode=DwtBaseDialog.MODAL] 		the modality of the dialog
 38  * @param	{DwtPoint}	params.loc			the location at which to popup the dialog. Defaults to being centered within its parent
 39  * @param	{DwtControl}	params.view 		the the control whose element is to be re-parented
 40  * @param	{string}	params.dragHandleId 		the the ID of element used as drag handle
 41  * 
 42  * @extends	DwtComposite
 43  */
 44 DwtBaseDialog = function(params) {
 45 	if (arguments.length == 0) { return; }
 46 
 47 	params = Dwt.getParams(arguments, DwtBaseDialog.PARAMS);
 48 	var parent = params.parent;
 49 	if (!(parent instanceof DwtShell)) {
 50 		throw new DwtException("DwtBaseDialog parent must be a DwtShell", DwtException.INVALIDPARENT, "DwtDialog");
 51 	}
 52 	params.className = params.className || "DwtBaseDialog";
 53 	params.posStyle = DwtControl.ABSOLUTE_STYLE;
 54 	this._title = params.title || "";
 55 
 56 	DwtComposite.call(this, params);
 57 
 58 	this._shell = parent;
 59 	this._zIndex = params.zIndex || Dwt.Z_DIALOG;
 60 	this._mode = params.mode || DwtBaseDialog.MODAL;
 61 	
 62 	this._loc = new DwtPoint();
 63 	if (params.loc) {
 64 		this._loc.x = params.loc.x;
 65 		this._loc.y = params.loc.y
 66 	} else {
 67 		this._loc.x = this._loc.y = Dwt.LOC_NOWHERE;
 68 	}
 69 	
 70 	// Default dialog tab group. Note that we disable application handling of
 71 	// keyboard shortcuts, since we don't want the view underneath reacting to
 72 	// keystrokes in the dialog.
 73 	this._tabGroup = new DwtTabGroup(this.toString(), true);
 74 
 75     this._dragHandleId = params.dragHandleId || this._htmlElId + "_handle";
 76 	this._createHtml();
 77     this._initializeDragging(this._dragHandleId);
 78 
 79 	if (params.view) {
 80 		this.setView(params.view);
 81     }
 82 
 83 	// reset tab index
 84     this.setZIndex(Dwt.Z_HIDDEN); // not displayed until popup() called
 85 	this._position(DwtBaseDialog.__nowhereLoc);
 86 }

189 /**
190  * Pops-up the dialog, makes the dialog visible in places. Everything under the dialog will
191  * become veiled if we are modal. Note: popping up a dialog will block
192  * keyboard actions from being delivered to the global key action handler (if one
193  * is registered).
194  *
195  * @param {DwtPoint}		loc		the desired location
196  */
197 DwtBaseDialog.prototype.popup =
198 function(loc) {
199 	if (this._poppedUp) { return; }
200 
201 	this.cleanup(true);
202 	var thisZ = this._zIndex;
203 
204 	// if we're modal, setup the veil effect, and track which dialogs are open
205 	if (this._mode == DwtBaseDialog.MODAL) {
206 		thisZ = this._setModalEffect(thisZ);
207 	}
208 
209 	this._shell._veilOverlay.activeDialogs.push(this);
210 	
211 	// use whichever has a value, local has precedence
212 	if (loc) {
213 		this._loc.x = loc.x;
214 		this._loc.y = loc.y;
215 	}
216 	this._position(loc);
217 
218     //reset TAB focus before popup of dialog.
219     //method be over-written to focus a different member.
220     this._resetTabFocus();
221 
222 	this.setZIndex(thisZ);
223 	this._poppedUp = true;
224 
225 	// Push our tab group
226 	var kbMgr = this._shell.getKeyboardMgr();
227 	kbMgr.pushTabGroup(this._tabGroup);
228 	kbMgr.pushDefaultHandler(this);
229 
230 	this.notifyListeners(DwtEvent.POPUP, this);
231 };

239 };

241 DwtBaseDialog.prototype.focus = 
242 function () {
243 	// if someone is listening for the focus to happen, give control to them,
244 	// otherwise focus on this dialog.
245 	if (this.isListenerRegistered(DwtEvent.ONFOCUS)) {
246 		this.notifyListeners(DwtEvent.ONFOCUS);
247 	} else if (this._focusElementId){
248 		var focEl = document.getElementById(this._focusElementId);
249 		if (focEl) {
250 			focEl.focus();
251 		}
252 	}
253 };

265 /**
266  * Pops-down and hides the dialog.
267  * 
268  */
269 DwtBaseDialog.prototype.popdown =
270 function() {
271 
272 	if (this._poppedUp) {
273 		this._poppedUp = false;
274 		this.cleanup(false);
275 	
276 		//var myZIndex = this.getZIndex();
277 	    var myZIndex = this._zIndex;
278 		this.setZIndex(Dwt.Z_HIDDEN);
279 		//TODO we should not create an object everytime we popdown a dialog (ditto w/popup)
280 		this._position(DwtBaseDialog.__nowhereLoc);
281 		if (this._mode == DwtBaseDialog.MODAL) {
282 			this._undoModality(myZIndex);
283 		} else {
284 			this._shell._veilOverlay.activeDialogs.pop();
285 		}
286 		//this.removeKeyListeners();
287 		
288 		// Pop our tab group
289 		var kbMgr = this._shell.getKeyboardMgr();
290 		kbMgr.popTabGroup(this._tabGroup);
291 		kbMgr.popDefaultHandler();
292 		
293 		this.notifyListeners(DwtEvent.POPDOWN, this);
294 	}
295 };

297 /**
298  * Sets the content of the dialog to a new view. Essentially re-parents
299  * the supplied control's HTML element to the dialogs HTML element
300  * 
301  * @param {DwtControl} newView		the control whose element is to be re-parented
302  */
303 DwtBaseDialog.prototype.setView =
304 function(newView) {
305 	this.reset();
306 	if (newView) {
307 		this._getContentDiv().appendChild(newView.getHtmlElement());
308 	}
309 };

313  * to add any additional behavior, but should still call up into this method.

316 DwtBaseDialog.prototype.reset =
317 function() {
318 	this._loc.x = this._loc.y = Dwt.LOC_NOWHERE;
319 }

321 /**
322 * Cleans up the dialog so it can be used again later.
323 * 
324 * @param	{boolean}		bPoppedUp		if <code>true</code>, the dialog is popped-up; <code>false</code> otherwise
325 */
326 DwtBaseDialog.prototype.cleanup =
327 function(bPoppedUp) {
328 	//TODO handle different types of input fields e.g. checkboxes etc
329 	var inputFields = this._getInputFields();
330 	
331 	if (inputFields) {
332 		var len = inputFields.length;
333 		for (var i = 0; i < len; i++) {
334 			inputFields[i].disabled = !bPoppedUp;
335 			if (bPoppedUp)
336 				inputFields[i].value = "";
337 		}
338 	}
339 }

384 /**
385  * Gets the active dialog.
386  * 
387  * @return	{DwtBaseDialog}		the active dialog
388  */
389 DwtBaseDialog.getActiveDialog = 
390 function() {
391 	var dialog = null;
392 	var shellObj = DwtShell.getShell(window);
393 	if (shellObj) {
394 		var len = shellObj._veilOverlay.activeDialogs.length;
395 		if (len > 0) {
396 			dialog = shellObj._veilOverlay.activeDialogs[len - 1];
397 		}
398 	}
399 	return dialog;
400 };

406 /**
407  * @private
408  */
409 DwtBaseDialog.prototype._initializeDragging =
410 function(dragHandleId) {
411 	var dragHandle = document.getElementById(dragHandleId);
412 	if (dragHandle) {
413 		var control = DwtControl.fromElementId(window._dwtShellId);
414 		if (control) {
415 			var p = Dwt.getSize(control.getHtmlElement());
416 			var dragObj = document.getElementById(this._htmlElId);
417 			var size = this.getSize();
418 			var dragEndCb = new AjxCallback(this, this._dragEnd);
419 			var dragCb = new AjxCallback(this, this._duringDrag);
420 			var dragStartCb = new AjxCallback(this, this._dragStart);
421 
422 			DwtDraggable.init(dragHandle, dragObj, 0,
423 							  document.body.offsetWidth - 10, 0, document.body.offsetHeight - 10, dragStartCb, dragCb, dragEndCb);
424 		}
425 	}
426 };

444 /**
445  * @private
446  */
447 DwtBaseDialog.prototype._createHtmlFromTemplate = function(templateId, data) {
448     // set default params
449     data.dragId = this._dragHandleId;
450     data.title = this._title;
451     data.icon = "";
452     data.closeIcon1 = "";
453     data.closeIcon2 = "";
454     data.controlsTemplateId = this.CONTROLS_TEMPLATE;
455 
456     // expand template
457     DwtComposite.prototype._createHtmlFromTemplate.call(this, templateId, data);
458 
459     // remember elements
460     this._titleBarEl = document.getElementById(data.id+"_titlebar");
461     this._titleEl = document.getElementById(data.id+"_title");
462     this._contentEl = document.getElementById(data.id+"_content");
463 
464     // NOTE: This is for backwards compatibility. There are just
465     //       too many sub-classes of dialog that expect to return
466     //       the dialog contents via the _getContentHtml method.
467     this.setContent(this._getContentHtml());
468 };

470 /**
471  * @private
472  */
473 DwtBaseDialog.prototype._setModalEffect =
474 function() {
475 	// place veil under this dialog
476 	var dialogZ = this._shell._veilOverlay.dialogZ;
477 	var currentDialogZ = null;
478 	var thisZ, veilZ;
479 	if (dialogZ.length)
480 		currentDialogZ = dialogZ[dialogZ.length - 1];
481 	if (currentDialogZ) {
482 		thisZ = currentDialogZ + 2;
483 		veilZ = currentDialogZ + 1;
484 	} else {
485 		thisZ = this._zIndex;
486 		veilZ = Dwt.Z_VEIL;
487 	}
488 	this._shell._veilOverlay.veilZ.push(veilZ);
489 	this._shell._veilOverlay.dialogZ.push(thisZ);
490 	Dwt.setZIndex(this._shell._veilOverlay, veilZ);
491 	return thisZ;
492 };
493 
494 /**
495  * @private
496  */
497 DwtBaseDialog.prototype._undoModality =
498 function (myZIndex) {
499 	var veilZ = this._shell._veilOverlay.veilZ;
500 	veilZ.pop();
501 	var newVeilZ = veilZ[veilZ.length - 1];
502 	Dwt.setZIndex(this._shell._veilOverlay, newVeilZ);
503 	this._shell._veilOverlay.dialogZ.pop();
504 	this._shell._veilOverlay.activeDialogs.pop();
505 	if (this._shell._veilOverlay.activeDialogs.length > 0 ) {
506 		this._shell._veilOverlay.activeDialogs[0].focus();
507 	}
508 };
