 16 /**
 17  * Creates a dialog allowing user to choose between an Instance or Series for an appointment.
 18  * @constructor
 19  * @class
 20  *
 21  * @author Parag Shah
 22  * @param {Hash}	params			a hash of parameters
 23  * @param {DwtComposite}      params.parent			a parent widget (the shell)
 24  * @param {String}	params.title			a title of dialog
 25  * @param {String}	params.confirmMsg 	a dialog confirmation message 
 26  * @param {String}	params.choiceLabel1	a label value for choice 1
 27  * @param {String}	params.choiceLabel2	a label value for choice 2
 28  * 
 29  * @extends	DwtDialog
 30  */
 31 ZmApptDeleteNotifyDialog = function(params) {
 32 
 33     params = Dwt.getParams(arguments, ZmApptDeleteNotifyDialog.PARAMS);
 34     var buttons = [ DwtDialog.YES_BUTTON, DwtDialog.NO_BUTTON, DwtDialog.CANCEL_BUTTON ];
 35     DwtDialog.call(this, {parent: params.parent, standardButtons:buttons, id:"CONFIRM_DELETE_APPT_DIALOG"});
 36 
 37     this._choiceLabel1 = params.choiceLabel1;
 38     this._choiceLabel2 = params.choiceLabel2;
 39     this._confirmMsg   = params.confirmMsg;
 40     this._choice2WarningMsg = params.choice2WarningMsg;
 41 
 42 	this.setTitle(params.title || AjxMsg.confirmTitle);
 43 	this.setContent(this._setHtml());
 44 	this._cacheFields();
 45     this.registerCallback(DwtDialog.YES_BUTTON, new AjxCallback(this, this._handleYesButton));
 46     this.registerCallback(DwtDialog.NO_BUTTON, new AjxCallback(this, this._handleNoButton));
 47 };
 48 
 49 ZmApptDeleteNotifyDialog.PARAMS = ["parent", "title", "confirmMsg", "choiceLabel1", "choiceLabel2"];
 50 
 51 ZmApptDeleteNotifyDialog.prototype = new DwtDialog;
 52 ZmApptDeleteNotifyDialog.prototype.constructor = ZmApptDeleteNotifyDialog;
 53

 61 /**
 62  * Initializes the dialog.
 63  *

 65  * @param	{String}	attId		the id
 66  */
 67 ZmApptDeleteNotifyDialog.prototype.initialize =
 68 function(appt, attId) {
 69 	this._appt = appt;
 70 	this._attId = attId;

 86 /**
 87  * Checks if the default option is checked.

104 ZmApptDeleteNotifyDialog.prototype._setHtml =
105 function() {
106     this._confirmMessageDivId = Dwt.getNextId();
107 	this._defaultRadioId	= Dwt.getNextId();
108 	this._notifyChoiceName	= Dwt.getNextId();
109 
110 	var html = new Array();
111 	var i = 0;
112 
113 	html[i++] = "<div style='width:300px' id='";
114     html[i++] = this._confirmMessageDivId;
115     html[i++] = "'>";
116 	html[i++] = this._confirmMsg;
117 	html[i++] = "<br>";
118 	html[i++] = "</div><p>";
119 	html[i++] = "<table align=center border=0 width='300px'>";
120 	html[i++] = "<tr><td width=1%><input checked value='1' type='radio' id='";
121 	html[i++] = this._defaultRadioId;
122 	html[i++] = "' name='";
123 	html[i++] = this._notifyChoiceName;
124 	html[i++] = "'></td><td>";
125 	html[i++] = "<label for='" + this._defaultRadioId + "'>";
126 	html[i++] = this._choiceLabel1;
127 	html[i++] = "</label>";
128 	html[i++] = "</td></tr>";
129 	html[i++] = "<tr><td width=1% valign=top><input value='2' type='radio' id='";
130 	html[i++] = this._defaultRadioId + this._notifyChoiceName;
131 	html[i++] = "' name='";
132 	html[i++] = this._notifyChoiceName;
133 	html[i++] = "'></td><td>";
134 	html[i++] = "<label for='" + this._defaultRadioId + this._notifyChoiceName + "'>"
135 	html[i++] = this._choiceLabel2;
136 	html[i++] = "</label>";
137     if (this._choice2WarningMsg) {
138         html[i++] = "<br><span style='font-style:italic'>" + this._choice2WarningMsg + "</span>";
139     }
140 	html[i++] = "</td></tr>";
141 	html[i++] = "</table>";
142 
143 	return html.join("");
144 };
145 
146 ZmApptDeleteNotifyDialog.prototype._cacheFields =
147 function() {
148 	this._defaultRadio = document.getElementById(this._defaultRadioId); 		delete this._defaultRadioId;	
149 };
150 
151 ZmApptDeleteNotifyDialog.prototype.popup = 
152 function(yesCallback, noCallback) {
153 
154 	this._yesCallback = yesCallback;
155 	this._noCallback = noCallback;
156 
157 	this.setButtonVisible(DwtDialog.CANCEL_BUTTON, Boolean(noCallback));
158 
159 	DwtDialog.prototype.popup.call(this);
160 };
161 
162 ZmApptDeleteNotifyDialog.prototype._handleYesButton =
163 function(ev) {
164 	if (this._yesCallback) this._yesCallback.run(ev);
165 	this.popdown();
166 };
167 
168 ZmApptDeleteNotifyDialog.prototype._handleNoButton =
169 function(ev) {
170 	if (this._noCallback) this._noCallback.run(ev);
171 	this.popdown();
172 };
