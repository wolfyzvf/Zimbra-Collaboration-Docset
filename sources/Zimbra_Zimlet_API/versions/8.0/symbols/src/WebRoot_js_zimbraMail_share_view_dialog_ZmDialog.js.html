 22 /**
 23  * Creates a dialog.
 24  * @class
 25  * This class is a base class for miscellaneous organizer-related dialogs. An instance
 26  * of this class can be re-used to show different overviews.
 27  *
 28  * @author Conrad Damon
 29  *
 30  * @param {Hash}	params		a hash of parameters
 31  * @param	{DwtControl}	params.parent		the parent widget
 32  * @param	{DwtMsgDialog}	params.msgDialog		the message dialog
 33  * @param	{String}	params.className		the CSS class name
 34  * @param	{String}	params.title		the dialog title
 35  * @param	{Array|constant}	params.standardButtons		an array of standard buttons to include. Defaults to {@link DwtDialog.OK_BUTTON} and {@link DwtDialog.CANCEL_BUTTON}.
 36  * @param	{Array}	params.extraButtons		a list of {@link DwtDialog_ButtonDescriptor} objects describing custom buttons to add to the dialog
 37  * @param	{DwtControl}	params.view				the dialog contents
 38  * 
 39  * @extends	DwtDialog
 40  */
 41 ZmDialog = function(params) {
 42 	if (arguments.length == 0) { return; }
 43 
 44 	DwtDialog.call(this, params);
 45 
 46 	if (!params.view) {
 47 		this.setContent(this._contentHtml());
 48 	}
 49 
 50 	if (this._button[DwtDialog.OK_BUTTON]) {
 51 		this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._okButtonListener));
 52 	}
 53 
 54 	this._overview = {};
 55 	this._opc = appCtxt.getOverviewController();
 56 
 57 	this._baseTabGroupSize = this._tabGroup.size();
 58 };
 59 
 60 ZmDialog.prototype = new DwtDialog;
 61 ZmDialog.prototype.constructor = ZmDialog;
 62 
 63 ZmDialog.prototype.isZmDialog = true;
 64 ZmDialog.prototype.toString = function() { return "ZmDialog"; };
 65 
 66 /**
 67  * @private
 68  */
 69 ZmDialog.prototype._contentHtml =
 70 function() {
 71 	return "";
 72 };
 73 
 74 /**
 75  * Sets the view for this dialog.
 76  * 
 77  * @param	{DwtComposite}		newView		the view
 78  * @param	{Boolean}		noReset		if <code>true</code>, do not reset the dialog; <code>false</code> otherwise
 79  */
 80 ZmDialog.prototype.setView =
 81 function(newView, noReset) {
 82 	this.reset();
 83 	if (newView) {
 84         var contentDiv = this._getContentDiv();
 85         var el = newView.getHtmlElement();
 86 		contentDiv.parentNode.replaceChild(el, contentDiv);
 87 		this._contentDiv = el;
 88 	}
 89 };
 90 
 91 ZmDialog.prototype.popup =
 92 function() {
 93 	// Bug 38281: for multiuse dialogs, we need to re-add the discretionary
 94 	// tab stops to the base list (the dialog buttons)
 95 	this._tabGroup.__members._array.splice(this._baseTabGroupSize);
 96 	this._tabGroup.addMember(this._getTabGroupMembers());
 97 
 98 	DwtDialog.prototype.popup.call(this);
 99 	if (this._focusElement) {
100 		appCtxt.getKeyboardMgr().grabFocus(this._focusElement);	
101 	}
102 };
103 
104 ZmDialog.prototype.reset =
105 function() {
106 	if (this._nameField) {
107 		this._nameField.value = "";
108 	}
109 	DwtDialog.prototype.reset.call(this);
110 };

125 	this._nameField = this._focusElement = document.getElementById(fieldId);

131 /**
132  * Displays the given list of tree views in an overview, creating it if
133  * necessary, and appends the overview to an element in the dialog. Since
134  * dialogs may be reused, it is possible that it will display different
135  * overviews. That is handled by making sure that only the current overview is
136  * visible.
137  * 
138  * @param params		[hash]				hash of params:
139  *        treeIds		[array]				list of tree views to show
140  *        omit			[hash]				IDs of organizers to exclude
141  *        fieldId		[string]			DOM ID of element that contains overview
142  *        overviewId	[string]*			ID for the overview
143  *        noRootSelect	[boolean]*			if true, don't make root tree item(s) selectable
144  * @param forceSingle	[boolean]*			if true, don't make multi-account overviews
145  * @private
146  */
147 ZmDialog.prototype._setOverview =
148 function(params, forceSingle) {
149 
150 	// when in multi-account mode, hide the old overview since we don't know
151 	// whether we want to show an overview container or just a single-account overview.
152 	if (appCtxt.multiAccounts) {
153 		var oldOverview = this._opc.getOverviewContainer(this._curOverviewId) ||
154 						  this._opc.getOverview(this._curOverviewId);
155 
156 		if (oldOverview) {
157 			oldOverview.setVisible(false);
158 		}
159 	}
160 
161 	// multi-account uses overview container
162 	if (appCtxt.multiAccounts && !forceSingle) {
163 		// use overviewId as the containerId; container will assign overviewId's
164 		var containerId = this._curOverviewId = params.overviewId;
165 		var ovContainer = this._opc.getOverviewContainer(containerId);
166 		if (!ovContainer) {
167 			var overviewParams = {
168 				overviewClass:	"dialogOverviewContainer",
169 				headerClass:	"DwtTreeItem",
170 				noTooltips:		true,
171 				treeStyle:		params.treeStyle,
172 				treeIds:		params.treeIds,
173 				overviewTrees:	params.overviewTrees,
174 				omit:			params.omit,
175 				omitPerAcct:	params.omitPerAcct,
176 				selectable:		params.selectable
177 			};
178 			var containerParams = {
179 				appName: params.appName,
180 				containerId: containerId
181 			};
182 			ovContainer = this._opc.createOverviewContainer(containerParams, overviewParams);
183 			ovContainer.setSize(Dwt.DEFAULT, "200");
184 			document.getElementById(params.fieldId).appendChild(ovContainer.getHtmlElement());
185 		}
186 
187 		// make overview container visible
188 		ovContainer.setVisible(true);
189 
190 		return ovContainer;
191 	}
192 
193 	// single-account overview handling
194 	var overviewId = this._curOverviewId = params.overviewId;
195 	var overview = this._opc.getOverview(overviewId);
196 	if (!overview) {
197 		var ovParams = {
198 			overviewId:		overviewId,
199 			overviewClass:	params.overviewClass || "dialogOverview",
200 			headerClass:	"DwtTreeItem",
201 			noTooltips:		true,
202 			treeStyle:		params.treeStyle,
203 			treeIds:		params.treeIds,
204 			account:		((appCtxt.multiAccounts && params.forceSingle) ? appCtxt.getActiveAccount() : (params.account || appCtxt.getActiveAccount())),
205 			skipImplicit: 	true
206 		};
207 		overview = this._overview[overviewId] = this._opc.createOverview(ovParams);
208 		this._renderOverview(overview, params.treeIds, params.omit, params.noRootSelect);
209 		document.getElementById(params.fieldId).appendChild(overview.getHtmlElement());
210 	}
211 
212 	this._makeOverviewVisible(overviewId);
213 
214 	return overview;
215 };
216 
217 /**
218  * @private
219  */
220 ZmDialog.prototype._makeOverviewVisible =
221 function(overviewId) {
222 	for (var id in this._overview) {
223 		this._overview[id].setVisible(id == overviewId);
224 	}
225 };

227 /**
228  * Renders the tree views in the overview, and makes the header items
229  * selectable (since they can generally be targets of whatever action the dialog
230  * is facilitating).
231  * 
232  * @param overview		[ZmOverview]		the overview
233  * @param treeIds		[array]				list of tree views to show
234  * @param omit			[hash]*				IDs of organizers to exclude
235  * @param noRootSelect	[boolean]*			if true, don't make root tree item(s) selectable
236  * @private
237  */
238 ZmDialog.prototype._renderOverview =
239 function(overview, treeIds, omit, noRootSelect) {
240 	overview.set(treeIds, omit);
241 	if (!noRootSelect) {
242 		for (var i = 0; i < treeIds.length; i++) {
243 			var treeView = overview.getTreeView(treeIds[i]);
244 			var hi = treeView && treeView.getHeaderItem();
245 			if (hi) hi.enableSelection(true);
246 		}
247 	}
248 };

252  */
253 ZmDialog.prototype._getOverview =
254 function() {
255 	return this._overview[this._curOverviewId];

266 /**
267  * @private
268  */
269 ZmDialog.prototype._showError =
270 function(msg, loc) {
271 	var nLoc = loc || (new DwtPoint(this.getLocation().x + 50, this.getLocation().y + 100));
272 	var msgDialog = appCtxt.getMsgDialog();
273 
274 	msgDialog.reset();
275     msgDialog.setMessage(AjxStringUtil.htmlEncode(msg), DwtMessageDialog.CRITICAL_STYLE);
276 	msgDialog.popup(nLoc);
277 };
