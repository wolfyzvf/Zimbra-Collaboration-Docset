 21 /**
 22  * Creates an item.
 23  * @class
 24  * An item is a piece of data that may contain user content. Most items are taggable. Currently,
 25  * the following things are items: conversation, message, attachment, appointment, and contact.
 26  * <br/>
 27  * <br/>
 28  * An item typically appears in the context of a containing list. Its event handling
 29  * is generally handled by the list so we avoid having the same listeners on each item. If we
 30  * create a context where an item stands alone outside a list context, then the item will have
 31  * its own listeners and do its own notification handling.
 32  *
 33  * @author Conrad Damon
 34  * 
 35  * @param {constant}	type		type of object (conv, msg, etc)
 36  * @param {int}			id			the unique id
 37  * @param {ZmList}		list		a list that contains this item
 38  * @param {Boolean}		noCache		if <code>true</code>, do not cache this item
 39  * 
 40  * @extends		ZmModel
 41  */
 42 ZmItem = function(type, id, list, noCache) {
 43 
 44 	if (arguments.length == 0) { return; }
 45 	ZmModel.call(this, type);
 46 
 47 	this.type = type;
 48 	this.id = id;
 49 	this.list = list;
 50 	this._list = {};
 51 
 52 	this.tags = [];
 53 	this.tagHash = {};
 54 	this.folderId = 0;
 55 
 56 	// make sure the cached item knows which lists it is in, even if those other lists
 57 	// have separate instances of this item - propagate view IDs from currently cached item
 58 	var curItem = appCtxt.getById(id);
 59 	if (curItem) {
 60 		this._list = AjxUtil.hashCopy(curItem._list);
 61 	}
 62 	if (list) {
 63 		this._list[list.id] = true;
 64 	}
 65 	
 66 	if (id && !noCache) {
 67 		appCtxt.cacheSet(id, this);
 68 	}
 69 };

211 /**
212  * Registers an item and stores information about the given item type.
213  *
214  * @param {constant}	item		the item type
215  * @param	{Hash}	params			a hash of parameters
216  * @param {constant}	params.app			the app that handles this item type
217  * @param {String}		params.nameKey		the message key for item name
218  * @param {String}		params.countKey 	the message key for plural of item name
219  * @param {String}		params.icon			the name of item icon class
220  * @param {String}		params.soapCmd		the SOAP command for acting on this item
221  * @param {String}		params.itemClass	the name of class that represents this item
222  * @param {String}		params.node			the SOAP response node for this item
223  * @param {constant}	params.organizer	the associated organizer
224  * @param {String}		params.searchType	the associated type in SearchRequest
225  * @param {function}	params.resultsList	the function that returns a {@link ZmList} for holding search results of this type
226  */
227 ZmItem.registerItem =
228 function(item, params) {
229 	if (params.app)				{ ZmItem.APP[item]					= params.app; }
230 	if (params.nameKey)			{ ZmItem.MSG_KEY[item]				= params.nameKey; }
231 	if (params.countKey)	    { ZmItem.COUNT_KEY[item]		    = params.countKey; }
232 	if (params.icon)			{ ZmItem.ICON[item]					= params.icon; }
233 	if (params.soapCmd)			{ ZmItem.SOAP_CMD[item]				= params.soapCmd; }
234 	if (params.itemClass)		{ ZmList.ITEM_CLASS[item]			= params.itemClass; }
235 	if (params.node)			{ ZmList.NODE[item]					= params.node; }
236 	if (params.organizer)		{ ZmOrganizer.ITEM_ORGANIZER[item]	= params.organizer; }
237 	if (params.searchType)		{ ZmSearch.TYPE[item]				= params.searchType; }
238 	if (params.resultsList)		{ ZmItem.RESULTS_LIST[item]			= params.resultsList; }
239 
240 	if (params.node) {
241 		ZmList.ITEM_TYPE[params.node] = item;
242 	}
243 
244 	if (params.dropTargets) {
245 		if (!ZmApp.DROP_TARGETS[params.app]) {
246 			ZmApp.DROP_TARGETS[params.app] = {};
247 		}
248 		ZmApp.DROP_TARGETS[params.app][item] = params.dropTargets;
249 	}
250 };

253 * Gets an item id by taking a normalized id (or an item id) and returning the item id.
254 * 
255 * @param	{String}	id		the normalized id
256 * @return	{String}	the item id
257 */
258 ZmItem.getItemId =
259 function(id) {
260 	if (!id) {
261 		return id;
262 	}
263 	if (!ZmItem.SHORT_ID_RE) {
264 		var shell = DwtShell.getShell(window);

267 	return id.replace(ZmItem.SHORT_ID_RE, '');

297 ZmItem.prototype.getAccount =
298 function() {
299 	if (!this.account) {
300 		var account;
301 
302 		if (this.folderId) {
303 			var ac = window.parentAppCtxt || window.appCtxt;
304 			var folder = ac.getById(this.folderId);
305 			account = folder && folder.getAccount();
306 		}
307 
308 		if (!account) {
309 			var parsed = ZmOrganizer.parseId(this.id);
310 			account = parsed && parsed.account;
311 		}
312 		this.account = account;
313 	}
314 	return this.account;

517 // Notification handling
518 
519 // For delete and modify notifications, we first apply the notification to this item. Then we
520 // see if the item is a member of any other lists. If so, we have those other copies of this
521 // item handle the notification as well. Each will notify through the list that created it.
522 
523 ZmItem.prototype.notifyDelete =
524 function() {
525 	this._notifyDelete();
526 	for (var listId in this._list) {
527 		var list = appCtxt.getById(listId);
528 		if (!list || (this.list && listId == this.list.id)) { continue; }
529 		var ctlr = list.controller;
530 		if (!ctlr || ctlr.inactive || (ctlr.getList().id != listId)) { continue; }
531 		var doppleganger = list.getById(this.id);
532 		if (doppleganger) {
533 			doppleganger._notifyDelete();
534 		}
535 	}
536 };
537 
538 ZmItem.prototype._notifyDelete =
539 function() {
540 	this.deleteLocal();
541 	if (this.list) {
542 		this.list.deleteLocal([this]);
543 	}
544 	this._notify(ZmEvent.E_DELETE);
545 };
546 
547 ZmItem.prototype.notifyModify =
548 function(obj, batchMode) {
549 	this._notifyModify(obj, batchMode);
550 	for (var listId in this._list) {
551 		var list = listId ? appCtxt.getById(listId) : null;
552 		if (!list || (this.list && (listId == this.list.id))) { continue; }
553 		var ctlr = list.controller;
554 		if (!ctlr || ctlr.inactive || (ctlr.getList().id != listId)) { continue; }
555 		var doppleganger = list.getById(this.id);
556 		if (doppleganger) {
557 			doppleganger._notifyModify(obj, batchMode);
558 		}
559 	}
560 };

562 /**
563  * Handles a modification notification.
564  *
565  * @param {Object}	obj			the item with the changed attributes/content
566  * @param {boolean}	batchMode	if true, return event type and don't notify
567  */
568 ZmItem.prototype._notifyModify =
569 function(obj, batchMode) {
570 	// empty string is meaningful here, it means no tags
571 	if (obj.tn != null) {
572 		this._parseTagNames(obj.tn);
573 		this._notify(ZmEvent.E_TAGS);
574 	}
575 	// empty string is meaningful here, it means no flags
576 	if (obj.f != null) {
577 		var flags = this._getFlags();
578 		var origFlags = {};
579 		for (var i = 0; i < flags.length; i++) {
580 			origFlags[flags[i]] = this[ZmItem.FLAG_PROP[flags[i]]];
581 		}
582 		this._parseFlags(obj.f);
583 		var changedFlags = [];
584 		for (var i = 0; i < flags.length; i++) {
585 			var on = this[ZmItem.FLAG_PROP[flags[i]]];
586 			if (origFlags[flags[i]] != on) {
587 				changedFlags.push(flags[i]);
588 			}
589 		}
590 		if (changedFlags.length) {
591 			this._notify(ZmEvent.E_FLAGS, {flags: changedFlags});
592 		}
593 	}
594 	if (obj.l != null && obj.l != this.folderId) {
595 		var details = {oldFolderId:this.folderId};
596 		this.moveLocal(obj.l);
597 		if (this.list) {
598 			this.list.moveLocal([this], obj.l);
599 		}
600 		if (batchMode) {
601 			delete obj.l;			// folder has been handled
602 			return ZmEvent.E_MOVE;
603 		} else {
604 			this._notify(ZmEvent.E_MOVE, details);
605 		}
606 	}
607 };

622 /**
623  * Sets the given flag change to this item. Both the flags string and the
624  * flag properties are affected.
625  *
626  * @param {constant}	flag	the flag that changed
627  * @param {Boolean}	on	<code>true</code> if the flag is now set
628  *
629  * @return	{String}		the new flags string
630  */
631 ZmItem.prototype.setFlag =
632 function(flag, on) {
633 	this.flagLocal(flag, on);
634 	var flags = this.flags || "";
635 	if (on && flags.indexOf(flag) == -1) {
636 		flags = flags + flag;
637 	} else if (!on && flags.indexOf(flag) != -1) {
638 		flags = flags.replace(flag, "");
639 	}
640 	this.flags = flags;
641 
642 	return flags;
643 };

645 /**
646  * Adds or removes the given tag for this item.
647  *
648  * @param {Object}		tag		tag name
649  * @param {Boolean}		doTag		<code>true</code> if tag is being added; <code>false</code> if it is being removed
650  * @return	{Boolean}	<code>true</code> to notify
651  */
652 ZmItem.prototype.tagLocal =
653 function(tag, doTag) {
654 	var bNotify = false;
655 	if (doTag) {
656 		if (!this.tagHash[tag]) {
657 			bNotify = true;
658 			this.tags.push(tag);
659 			this.tagHash[tag] = true;
660 		}
661 	} else {
662 		for (var i = 0; i < this.tags.length; i++) {
663 			if (this.tags[i] == tag) {
664 				this.tags.splice(i, 1);
665 				delete this.tagHash[tag];
666 				bNotify = true;
667 				break;
668 			}
669 		}
670 	}
671 	
672 	return bNotify;
673 };

743 /**
744  * Takes a comma-separated list of tag IDs and applies the tags to this item.
745  * 
746  * @private
747  */
748 ZmItem.prototype._parseTags =
749 function(str) {	
750 	this.tags = [];
751 	this.tagHash = {};
752 	if (str && str.length) {
753 		var tags = str.split(",");
754 		for (var i = 0; i < tags.length; i++) {
755 			var tagId = Number(tags[i]);
756 			if (tagId >= ZmOrganizer.FIRST_USER_ID[ZmOrganizer.TAG])
757 				this.tagLocal(tagId, true);
758 		}
759 	}
760 };
761 
762 /**
763  * Takes a comma-separated list of tag names and applies the tags to this item.
764  *
765  * @private
766  */
767 ZmItem.prototype._parseTagNames =
768 function(str) {
769 	this.tags = [];
770 	this.tagHash = {};
771 	if (!str || !str.length) {
772 		return;
773 	}
774 	
775 	// server escapes comma with backslash
776 	str = str.replace(/\\,/g, "\u001D");
777 	var tags = str.split(",");
778 	
779 	for (var i = 0; i < tags.length; i++) {
780 		var tagName = tags[i].replace("\u001D", ",");
781 		this.tagLocal(tagName, true);
782 	}
783 };

785 /**
786  * Takes a string of flag chars and applies them to this item.
787  * 
788  * @private
789  */
790 ZmItem.prototype._parseFlags =
791 function(str) {
792 	this.flags = str;
793 	for (var i = 0; i < ZmItem.ALL_FLAGS.length; i++) {
794 		var flag = ZmItem.ALL_FLAGS[i];
795 		var on = (str && (str.indexOf(flag) != -1)) ? true : false;
796 		this.flagLocal(flag, on);
797 	}
798 };

874 ZmItem.prototype.getSortedTags =
875 function() {
876 	var numTags = this.tags && this.tags.length;
877 	if (numTags) {
878 		var tagList = appCtxt.getAccountTagList(this);
879 		var ta = [];
880 		for (var i = 0; i < numTags; i++) {
881 			var tag = tagList.getByNameOrRemote(this.tags[i]);
882 			//tag could be missing if this was called when deleting a whole tag (not just untagging one message). So this makes sure we don't have a null item.
883 			if (!tag) {
884 				continue;
885 			}
886 			ta.push(tag);
887 		}
888 		ta.sort(ZmTag.sortCompare);
889 		return ta;
890 	}
891 	return null;
892 };
