 16 /**
 17  * Creates a rectangle.
 18  * @constructor
 19  * @class
 20  * This class represents a rectangle. A point has an x-coordinate, y-coordinate, height and width.
 21  * 
 22  * @author Ross Dargahi
 23  * 
 24  * @param {number} x 	the x coordinate
 25  * @param {number} y 	the y coordinate
 26  * @param {number} width 	the width
 27  * @param {number} height 	the height
 28  */
 29 DwtRectangle = function(x, y, width, height) {
 30 
 31 	/**
 32 	 * The x-coordinate.
 33 	 * @type	number
 34 	 */
 35 	this.x = x;
 36 	/**
 37 	 * The y-coordinate.
 38 	 * @type	number
 39 	 */
 40 	this.y = y;
 41 	/**
 42 	 * The width.
 43 	 * @type	number
 44 	 */
 45 	this.width = width;
 46 	/**
 47 	 * The height.
 48 	 * @type	number
 49 	 */
 50 	this.height = height;
 51 }
 52 
 53 /**
 54  * Returns a string representation of the object.
 55  * 
 56  * @return		{string}		a string representation of the object
 57  */
 58 DwtRectangle.prototype.toString = 
 59 function() {
 60 	return "DwtRectangle";
 61 }
 62 
 63 /**
 64  * Sets the values of the rectangle.
 65  * 
 66  * @param {number} x 	the x coordinate
 67  * @param {number} y 	the y coordinate
 68  * @param {number} width 	the width
 69  * @param {number} height 	the height
 70  */
 71  DwtRectangle.prototype.set =
 72  function(x, y, width, height) {
 73  	this.x = x;
 74  	this.y = y;
 75  }
