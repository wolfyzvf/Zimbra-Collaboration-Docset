 16 ZmPriorityMessageFilterDialog = function() {
 17 
 18 	DwtDialog.call(this, {parent:appCtxt.getShell(), className:"ZmPriorityMessageFilterDialog", title:ZmMsg.activityStream});
 19 
 20 	// set content
 21 	this.setContent(this._contentHtml());
 22 	this._initialize();
 23 	var okButton = this.getButton(DwtDialog.OK_BUTTON);
 24 	okButton.setText(ZmMsg.save);
 25 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._okButtonListener));
 26 	this._rules = AjxDispatcher.run("GetFilterRules");
 27 };

 38 ZmPriorityMessageFilterDialog.prototype._initialize = 
 39 function() {
 40 	var streamListener = this._onMoveMsgIntoStream.bind(this);
 41 	var advancedListener = this._onAdvancedControls.bind(this);
 42 	var params = {};
 43 	params.parent = this;
 44 	params.template = "prefs.Pages#PriorityMessageFilterPrompt";
 45 	params.id = "PriorityInboxDialog";
 46 	params.form = {
 47 		items: [
 48 			{ id: "MOVE_MSG_STREAM", type: "DwtCheckbox", label: ZmMsg.enableActivityStream, checked: false, onclick: streamListener},
 49 			{ id: "NOT_TO_ME", type: "DwtCheckbox", label: ZmMsg.moveNotToMe, checked: false},

 51 			{ id: "NOT_IN_ADDR", type: "DwtCheckbox", label: ZmMsg.moveNotInAddrBook, checked: false},
 52 			{ id: "DL_SUBSCRIBED", type: "DwtCheckbox", label: ZmMsg.moveMessagesFromDL, checked: true},
 53 			{ id: "MASS_MARKETING", type: "DwtCheckbox", label: ZmMsg.massMarketingMessages, checked: true}

 56 	this._priorityMessageForm = new DwtForm(params);
 57 	this._priorityMessageForm.setScrollStyle(DwtControl.CLIP);
 58 	var div = document.getElementById("PRIORITYMESSAGE_PROMPT_FORM");
 59 	this._priorityMessageForm.appendElement(div);
 60 	
 61 	this._moveMsgIntoStream = this._priorityMessageForm.getControl("MOVE_MSG_STREAM");
 62 	this._notToMe = this._priorityMessageForm.getControl("NOT_TO_ME");
 63 	this._selectField = this._priorityMessageForm.getControl("SELECT_FIELD");
 64 	this._selectField.fixedButtonWidth();
 65 	this._notInMyAddrBk = this._priorityMessageForm.getControl("NOT_IN_ADDR");
 66 	this._dlSubscribedTo = this._priorityMessageForm.getControl("DL_SUBSCRIBED");
 67 	this._massMarketing = this._priorityMessageForm.getControl("MASS_MARKETING");
 68 	
 69 	this._streamHash = {};
 70 	this._streamHash[ZmFilterRule.TEST_BULK] = {control: this._massMarketing, negative: false};
 71 	this._streamHash[ZmFilterRule.TEST_LIST] = {control: this._dlSubscribedTo, negative: false};
 72 	this._streamHash[ZmFilterRule.TEST_ADDRBOOK] = {control: this._notInMyAddrBk, negative: true, headerValue: "from"};
 73 	this._streamHash[ZmFilterRule.TEST_ME] = {control: this._notToMe, negative: true, headerValue: "to"};
 74 	
 75     this._advancedControls = new DwtText({parent:this,className:"FakeAnchor"});
 76     this._advancedControls.setText(ZmMsg.advancedControls);
 77     this._advancedControls.getHtmlElement().onclick = advancedListener;
 78     this._advancedControls.replaceElement(document.getElementById("PriorityInboxAdvancedControls"));

105 ZmPriorityMessageFilterDialog.prototype._onAdvancedControls = 
106 function(controlId) {
107 	var filterRuleDialog = appCtxt.getFilterRuleDialog();
108 	var isPriority = false;
109 	var rule = this._activityStreamRule;	
110 	
111 	if (rule) {
112 		filterRuleDialog.popup(rule, true);		
113 	}
114 	else {
115 		//create rule with default conditions
116 		var ruleName = isPriority ? ZmMsg.markAsPriorityRule : ZmMsg.activityStreamsRule;
117 		var rule = new ZmFilterRule(ruleName, true, {}, {});
118         rule.addAction(ZmFilterRule.A_FOLDER, ZmMsg.activityStreamsRule);
119         for (var id in this._streamHash) {
120             if (id == ZmFilterRule.TEST_ME) {
121 				var meTestValue = this._selectField.getValue() == ZmMsg.to ? ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_TO] : ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_TO_CC];
122                 rule.addCondition(id, ZmFilterRule.OP_NOT_IS, null, meTestValue);	
123             }
124             else if (id == ZmFilterRule.TEST_ADDRBOOK) {
125                 rule.addCondition(id, ZmFilterRule.OP_NOT_IN ,"contacts", this._streamHash[id].headerValue); //Address in From not in Contacts	
126             }
127             else {
128                 rule.addCondition(id);
129             }
130         }
131 		rule.setGroupOp(ZmFilterRule.GROUP_ANY);		
132 		filterRuleDialog.popup(rule, true);
133 	}
134 };

136 ZmPriorityMessageFilterDialog.prototype._setStreamSelections = 
137 function() {
138 	if (this._activityStreamRule) {
139 		if (this._activityStreamRule.active) {
140 			this._moveMsgIntoStream.setEnabled(true);
141 			this._moveMsgIntoStream.setSelected(true);
142 		}
143 		else {
144 			this._moveMsgIntoStream.setSelected(false);
145 		}
146 		var conditions = this._activityStreamRule.conditions;
147 		//initialize checkboxes before loading them
148 		this._massMarketing.setSelected(false);
149 		this._dlSubscribedTo.setSelected(false);
150 		this._notInMyAddrBk.setSelected(false);
151 		this._notToMe.setSelected(false);
152 
153 		for (var c in conditions) {
154 			var length = AjxUtil.isArray(conditions[c]) ? conditions[c].length : -1;
155 			for (var i=0; i<length; i++) {
156 				var isNegative = AjxUtil.isArray(conditions[c]) && conditions[c][i].negative ? (conditions[c][i].negative == "1") : false;
157 				if (this._streamHash[c]) {
158 					if (isNegative && (c == ZmFilterRule.TEST_ADDRBOOK || c == ZmFilterRule.TEST_ME)) {
159 						var header = AjxUtil.isArray(conditions[c]) && conditions[c][i].header;
160 						if (c == ZmFilterRule.TEST_ADDRBOOK) {
161 							value = ZmFilterRule.C_FROM;
162 						}
163 						else if (c == ZmFilterRule.TEST_ME) {
164 							if (header &&  header.toUpperCase() == ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_TO].toUpperCase()) {
165 								value = header;
166 								this._selectField.setSelected(0);
167 							}
168 							else if (header &&  header.toUpperCase() == ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_TO_CC].toUpperCase()) {
169 								value = header;
170 								this._selectField.setSelected(1);
171 							}
172 						}
173 						//var value = (c == ZmFilterRule.TEST_ADDRBOOK) ? ZmFilterRule.C_FROM : ZmFilterRule.C_TO;
174 						if (header && header.toLowerCase() == value.toLowerCase()) {
175 							this._streamHash[c].control.setSelected(true);
176 							this._streamHash[c].control.setEnabled(true);
177 						}
178 					}
179 					else if (!isNegative && !(c == ZmFilterRule.TEST_ADDRBOOK || c == ZmFilterRule.TEST_ME)) {
180 						this._streamHash[c].control.setSelected(true);
181 						this._streamHash[c].control.setEnabled(true);
182 					}
183 				} 
184 			}
185 		}
186 	}
187 	else {
188 		this._moveMsgIntoStream.setSelected(false);
189 	}
190 	this._onMoveMsgIntoStream();	
191 };

193 ZmPriorityMessageFilterDialog.prototype._okButtonListener = 
194 function() {
195 	//build filter
196 	var foundCondition = false;
197 	var needSave = false; 
198 	var condition = {};
199 	var activityRule = this._rules.getRuleByName(ZmMsg.activityStreamsRule);
200 	
201 	//handle activity streams
202 	foundCondition = false;
203 	if (this._moveMsgIntoStream.isSelected()) {
204 		var streamRule = new ZmFilterRule(ZmMsg.activityStreamsRule, true, {}, {});
205 		streamRule.addAction(ZmFilterRule.A_FOLDER, ZmMsg.activityStreamsRule); 
206 		streamRule.setGroupOp(ZmFilterRule.GROUP_ANY);
207 		
208 		for (var id in this._streamHash) {
209 			var control = this._streamHash[id].control;
210 			var negative = this._streamHash[id].negative;
211 			var headerValue = this._streamHash[id].headerValue;
212 			if (control.isSelected()) {
213 				if (id == ZmFilterRule.TEST_ME) {
214 					var meTestValue = this._selectField.getValue() == ZmMsg.to ? ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_TO] : ZmFilterRule.C_ADDRESS_VALUE[ZmFilterRule.C_TO_CC];
215 					streamRule.addCondition(id, ZmFilterRule.OP_NOT_IS, null, meTestValue);	
216 				}
217 				else if (id == ZmFilterRule.TEST_ADDRBOOK) {
218 					streamRule.addCondition(id, ZmFilterRule.OP_NOT_IN ,"contacts", headerValue); //Address in From not in Contacts	
219 				}
220 				else {
221 					streamRule.addCondition(id);
222 				}
223 				foundCondition = true;
224 			}
225 			else if (activityRule) {
226 				if (id == ZmFilterRule.TEST_ME && this._activityStreamRule.conditions[ZmFilterRule.TEST_ME]) {
227 					//if we uncheck the me filter we need to know which headerValue we are removing ("to" or "to,cc")
228 					activityRule = this._removeCondition(activityRule, id, negative, this._activityStreamRule.conditions[ZmFilterRule.TEST_ME][0].headerValue);	
229 				}
230 				else {
231 					activityRule = this._removeCondition(activityRule, id, negative, headerValue);
232 				}
233 			}
234 		}
235 		
236 		if (foundCondition && activityRule) {
237 			for (var id in streamRule.conditions) {
238 				activityRule.conditions[id] = streamRule.conditions[id];
239 			}
240 	
241 			for (var id in streamRule.actions) {
242 				activityRule.actions[id] = streamRule.actions[id];
243 			}
244 	
245 			activityRule.active = true;
246 			needSave = true;
247 		}
248 		else if(foundCondition) {
249 			this._rules.insertRule(streamRule); //insert last
250 			needSave = true;
251 		}
252 		else if (activityRule) {
253 			return this._handleConditionsError(ZmMsg.ruleNoConditonActivityFilter);	
254 		}
255 	}
256 	else if (activityRule) {
257 		//set existing rule to be non-active
258 		activityRule.active = false;
259 		needSave = true;
260 	}
261 	
262 	if (needSave) {
263 		this._rules.saveRules(null, true);
264 		this._createActivityStreamsFolder();
265 	}
266 	
267 	this.popdown();
268 			
269 };

285 /**
286  * checks condition and value to determine if it should be removed; comparators are not checked
287  * @param rule
288  * @param condition
289  * @param isNegative
290  * @param headerValue
291  */
292 ZmPriorityMessageFilterDialog.prototype._removeCondition = 
293 function(rule, condition, isNegative, headerValue) {
294 	var c = rule.conditions[condition];
295 	if (c) {
296 		for (var i=0; i<c.length; i++) {
297 			var negativeCheck = isNegative ? c[i].negative == "1" : !c[i].negative;
298 			var headerCheck = headerValue ? c[i].header == headerValue : true;
299 			if (condition == ZmFilterRule.TEST_CONVERSATIONS) {
300 				headerCheck = headerValue ? c[i].where == headerValue : true;
301 			}
302 			if (negativeCheck && headerCheck) {				
303 				c.splice(i, 1);
304 				rule.conditions[condition] = c;
305 			}
306 		} 			
307 	}
308 	return rule;
309 };

318 ZmPriorityMessageFilterDialog.prototype._createActivityStreamsFolder =
319 function() {
320 	var jsonObj = {CreateFolderRequest:{_jsns:"urn:zimbraMail"}};
321 	var folder = jsonObj.CreateFolderRequest.folder = {l: ZmOrganizer.ID_ROOT, name: ZmMsg.activityStreamFolder, fie: 1};
322 	return appCtxt.getAppController().sendRequest({
323 		jsonObj: jsonObj,
324 		asyncMode: true,
325 		callback:  new AjxCallback(this, this._handleActivityStreamsFolderCreate)
326 	});
327 };
328 
329 ZmPriorityMessageFilterDialog.prototype._handleActivityStreamsFolderCreate = 
330 function(result) {
331 	var resp = result && result._data && result._data.CreateFolderResponse;
332 	if (resp) {
333 		appCtxt.set(ZmSetting.MAIL_ACTIVITYSTREAM_FOLDER, resp.folder[0].id);
334 	}
335 };
