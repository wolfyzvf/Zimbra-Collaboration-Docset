 20 DwtKeyEvent = function() {
 21 	DwtUiEvent.call(this, true);
 22 	this.reset(true);
 23 }

 54 DwtKeyEvent.prototype = new DwtUiEvent;
 55 DwtKeyEvent.prototype.constructor = DwtKeyEvent;
 56 
 57 DwtKeyEvent.prototype.toString =
 58 function() {
 59 	return "DwtKeyEvent";
 60 }
 61 
 62 DwtKeyEvent.isKeyEvent =
 63 function(ev) {
 64 	return (ev.type.search(/^key/i) != -1);
 65 }
 66 
 67 DwtKeyEvent.isKeyPressEvent =
 68 function(ev) {
 69 	return (AjxEnv.isIE && ev.type == "keydown") || (ev.type == "keypress");
 70 }
 71 
 72 DwtKeyEvent.prototype.reset =
 73 function(dontCallParent) {
 74 	if (!dontCallParent)
 75 		DwtUiEvent.prototype.reset.call(this);
 76 	this.keyCode = 0;
 77 	this.charCode = 0;
 78 }
 79 
 80 DwtKeyEvent.prototype.isCommand =
 81 function(ev) {
 82 	return AjxEnv.isMac && this.metaKey || this.ctrlKey;
 83 }
 84 
 85 DwtKeyEvent.prototype.setFromDhtmlEvent =
 86 function(ev, obj) {
 87 	ev = DwtUiEvent.prototype.setFromDhtmlEvent.apply(this, arguments);
 88 	if (!ev) { return; }
 89 	this.charCode = ev.charCode || ev.keyCode;
 90 	this.keyCode = ev.keyCode;
 91 }

 93 /**
 94  * Simple function to return key code from a key event. The code is in keyCode for keydown/keyup.
 95  * Gecko puts it in charCode for keypress.
 96  */
 97 DwtKeyEvent.getCharCode =
 98 function(ev) {
 99 	ev = DwtUiEvent.getEvent(ev);
100 	var key = AjxEnv.isSafari ? ev.keyCode : (ev.charCode || ev.keyCode);
101 	if (key == 0 && AjxEnv.isMac && AjxEnv.isGeckoBased && ev.type == "keyup" && DwtKeyEvent._geckoCode) {
102 		// if Mac Gecko, return keyCode saved from keypress event
103 		key = DwtKeyEvent._geckoCode;
104 		DwtKeyEvent._geckoCode = null;
105 	}
106 	return key;
107 }
108 
109 DwtKeyEvent.copy =
110 function(dest, src) {
111 	DwtUiEvent.copy(dest, src);
112 	dest.charCode = src.charCode;
113 	dest.keyCode = src.keyCode;
114 }

116 /**
117  * Workaround for the bug where Mac Gecko returns a keycode of 0 for many shifted chars for
118  * keydown and keyup. Since it returns a char code for keypress, we save it so that the
119  * ensuing keyup can pick it up.
120  *
121  * FF2 returns keycode 0 for: ~ ! @ # $ % ^ & * ( ) - _ + | : < > ? Alt-anything
122  * FF3 returns keycode 0 for: ~ _ | : < > ?
123  *
124  * FF2 returns incorrect keycode for Ctrl plus any of: 1 2 3 4 5 6 7 8 9 0 ; ' , . /
125  *
126  * https://bugzilla.mozilla.org/show_bug.cgi?id=448434
127  *
128  * @param ev
129  */
130 DwtKeyEvent.geckoCheck =
131 function(ev) {
132 
133 	ev = DwtUiEvent.getEvent(ev);
134 	if (ev.type == "keypress") {
135 		DwtKeyEvent._geckoCode = null;
136 		if (AjxEnv.isMac && AjxEnv.isGeckoBased) {
137 			var ch = String.fromCharCode(ev.charCode);
138 			DwtKeyEvent._geckoCode = DwtKeyEvent.MAC_FF_CODE[ch];
139 		}
140 	}
141 };
