 16 /**

 18  * @constructor
 19  * @class
 20  * This class displays free/busy information for an appointment's attendees. An
 21  * attendee may be a person, a location, or equipment.

 28  * @param dateInfo			[object]					hash of date info

191 ZmFreeBusySchedulerView.prototype.cleanup =
192 function() {
193 	if (!this._rendered) return;
194 
195     if(this._timedActionId)  {
196         AjxTimedAction.cancelAction(this._timedActionId);
197         this._timedActionId = null;
198     }

203 	}
204 	this._activeInputIdx = null;
205 
206 	// cleanup all attendees row
207 	var allAttCells = this._allAttendeesSlot._coloredCells;
208 	while (allAttCells.length > 0) {
209 		allAttCells[0].className = ZmFreeBusySchedulerView.FREE_CLASS;
210 		allAttCells.shift();
211 	}
212 
213 	for (var i in this._emailToIdx) {
214 		delete this._emailToIdx[i];
215 	}
216 
217 	this._curValStartDate = "";
218 	this._curValEndDate = "";
219 
220 	this._resetAttendeeCount();
221 
222 	// reset autocomplete lists
223 	if (this._acContactsList) {
224 		this._acContactsList.reset();
225 		this._acContactsList.show(false);
226 	}
227 	if (this._acEquipmentList) {

233     this._emptyRowIndex = null;

307 // Add the attendee, then create a new empty slot since we've now filled one.
308 ZmFreeBusySchedulerView.prototype._autocompleteCallback =
309 function(text, el, match) {
310     if(match && match.fullAddress) {
311         el.value = match.fullAddress;
312     }
313 	if (match && match.item) {
314 		if (match.item.isGroup && match.item.isGroup()) {
315 			var members = match.item.getGroupMembers().good.getArray();
316 			for (var i = 0; i < members.length; i++) {
317 				el.value = members[i].address;
318 
319                 if(el._acHandlerInProgress) { return; }
320                 el._acHandlerInProgress = true;
321 				var index = this._handleAttendeeField(el);
322                 this._editView.showConflicts();
323                 el._acHandlerInProgress = false;
324 
325 				if (index && ((i+1) < members.length)) {
326 					el = this._schedTable[index].inputObj.getInputElement();
327 				}
328 			}
329 		} else {
330             if(el._acHandlerInProgress) { return; }
331             el._acHandlerInProgress = true;
332 			this._handleAttendeeField(el, match.item);
333             this._editView.showConflicts();
334             el._acHandlerInProgress = false;
335 		}
336 	}
337 };

339 // Enter listener. If the user types a return when no autocomplete list is showing,
340 // then go ahead and add a new empty slot.
341 ZmFreeBusySchedulerView.prototype._autocompleteKeyUpCallback =
342 function(ev, aclv, result) {
343 	var key = DwtKeyEvent.getCharCode(ev);
344 	if ((key == 3 || key == 13) && !aclv.getVisible()) {
345 		var el = DwtUiEvent.getTargetWithProp(ev, "id");
346         if(el._acHandlerInProgress) { return; }
347         el._acHandlerInProgress = true;
348         this._handleAttendeeField(el);
349         this._editView.showConflicts();
350         el._acHandlerInProgress = false;
351 	}
352 };

391 /**
392  * Adds a new, empty slot with a select for the attendee type, an input field,
393  * and cells for free/busy info.
394  *
395  * @param isAllAttendees	[boolean]*	if true, this is the "All Attendees" row
396  * @param organizer			[string]*	organizer
397  * @param drawBorder		[boolean]*	if true, draw borders to indicate appt time
398  * @param index				[int]*		index at which to add the row
399  * @param updateTabGroup	[boolean]*	if true, add this row to the tab group
400  * @param setFocus			[boolean]*	if true, set focus to this row's input field
401  */
402 ZmFreeBusySchedulerView.prototype._addAttendeeRow =
403 function(isAllAttendees, organizer, drawBorder, index, updateTabGroup, setFocus) {
404 	index = index || this._attendeesTable.rows.length;
405 
406 	// store some meta data about this table row
407 	var sched = {};
408 	var dwtId = Dwt.getNextId();	// container for input
409 	sched.dwtNameId		= dwtId + "_NAME_";			// TD that contains name
410 	sched.dwtTableId	= dwtId + "_TABLE_";		// TABLE with free/busy cells
411 	sched.dwtSelectId	= dwtId + "_SELECT_";		// TD that contains select menu
412 	sched.dwtInputId	= dwtId + "_INPUT_";		// input field
413 	sched.idx = index;
414 	sched._coloredCells = [];
415 	this._schedTable[index] = sched;
416 
417 	this._dateBorder = this._getBordersFromDateInfo();
418 
419 	var data = {
420 		id: dwtId,
421 		sched: sched,
422 		isAllAttendees: isAllAttendees,
423 		organizer: organizer,
424 		cellCount: ZmFreeBusySchedulerView.FREEBUSY_NUM_CELLS,
425         isComposeMode: this.isComposeMode,
426         dateBorder: this._dateBorder
427 	};
428 
429 	var tr = this._attendeesTable.insertRow(index);
430 	var td = tr.insertCell(-1);
431     if(isAllAttendees) {
432         td.className = "ZmSchedulerAllTd";
433     }
434 	td.innerHTML = AjxTemplate.expand("calendar.Appointment#AttendeeName", data);
435 
436 	var td = tr.insertCell(-1);
437 	td.innerHTML = AjxTemplate.expand("calendar.Appointment#AttendeeFreeBusy", data);
438     td.style.padding = "0";
439 
440 	for (var k = 0; k < data.cellCount; k++) {
441 		var id = sched.dwtTableId + "_" + k;
442 		var fbDiv = document.getElementById(id);
443 		if (fbDiv) {
444 			fbDiv._freeBusyCellIndex = k;
445 			fbDiv._schedTableIdx = index;
446 			fbDiv._schedViewPageId = this._svpId;
447 		}
448 	}
449 
450 	// create DwtInputField and DwtSelect for the attendee slots, add handlers
451 	if (!isAllAttendees && !organizer) {
452 		// add DwtSelect
453 		var button;
454 		var btnId = sched.dwtSelectId;
455 		var btnDiv = document.getElementById(btnId);
456 		if (this.isComposeMode && btnDiv) {
457             button  = new DwtButton({parent: this, parentElement: btnId, className: 'ZAttRole'});
458             button.setText("");
459             button.setImage("AttendeesRequired");
460             button.setMenu(new AjxListener(this, this._getAttendeeRoleMenu, [index]));
461             sched.btnObj = button;
462 		}
463 		// add DwtInputField
464 		var nameDiv = document.getElementById(sched.dwtNameId);
465 		if (nameDiv) {
466 			var dwtInputField = new DwtInputField({parent: this, type: DwtInputField.STRING, maxLen: 256});
467 			dwtInputField.setDisplay(Dwt.DISPLAY_INLINE);
468 			var inputEl = dwtInputField.getInputElement();
469             Dwt.setSize(inputEl, Dwt.DEFAULT, "22px")
470 			inputEl.className = "ZmSchedulerInput";
471 			inputEl.id = sched.dwtInputId;
472             inputEl.style.border = "0px";
473 			sched.attType = inputEl._attType = ZmCalBaseItem.PERSON;
474 			sched.inputObj = dwtInputField;
475 			if (button) {
476 				button.dwtInputField = dwtInputField;
477 			}
478 			dwtInputField.reparentHtmlElement(sched.dwtNameId);
479 		}
480 
481 		sched.ptstObj = document.getElementById(sched.dwtNameId+"_ptst");
482 
483         Dwt.setVisible(sched.ptstObj, this.isComposeMode ? this._editView.getRsvp() : true);
484 
485 		// set handlers
486 		var attendeeInput = document.getElementById(sched.dwtInputId);
487 		if (attendeeInput) {
488 			this._activeInputIdx = index;
489 			// handle focus moving to/from an enabled input
490 			Dwt.setHandler(attendeeInput, DwtEvent.ONFOCUS, ZmFreeBusySchedulerView._onFocus);
491 			Dwt.setHandler(attendeeInput, DwtEvent.ONBLUR, ZmFreeBusySchedulerView._onBlur);
492 			attendeeInput._schedViewPageId = this._svpId;
493 			attendeeInput._schedTableIdx = index;
494 		}
495 	}
496 
497 	if (drawBorder) {
498 		this._updateBorders(sched, isAllAttendees);
499 	}
500     
501 	if (setFocus && sched.inputObj) {
502 		this._kbMgr.grabFocus(sched.inputObj);
503 	}
504 	return index;
505 };

598 
599 	var idx = inputEl._schedTableIdx;
600 	if (idx != this._activeInputIdx) return;
601 
602 	var sched = this._schedTable[idx];
603 	if (!sched) return;
604 	var input = sched.inputObj;
605 	if (!input) return;
606 
607 	var value = input.getValue();
608 	if (value) {
609 		value = AjxStringUtil.trim(value.replace(/[;,]$/, ""));	// trim separator, white space
610 	}
611 	var curAttendee = sched.attendee;
612 	var type = sched.attType;
613 
614 	if (value) {
615 		if (curAttendee) {
616 			// user edited slot with an attendee in it
617             var lookupEmail = this.getEmail(curAttendee);
618             var emailTextShortForm = ZmApptViewHelper.getAttendeesText(curAttendee, type, true);
619             //parse the email id to separate the name and email address

622 			if (emailAddr == lookupEmail || emailAddr == emailTextShortForm) {
623 				return;

631 
632

636                         //if duplicate - do nothing

638                     }
639 					this._emailToIdx[email[i]] = idx;
640 				}
641 			} else {
642                 if(this._isDuplicate(email)) {
643                     //if duplicate - do nothing
644                     return;
645                 }
646 				this._emailToIdx[email] = idx;
647 			}
648 
649 			// go get this attendee's free/busy info if we haven't already
650 			if (sched.uid != email) {
651 				this._getFreeBusyInfo(this._getStartTime(), email);
652 			}
653             var attendeeType = sched.btnObj ? sched.btnObj.getData(ZmFreeBusySchedulerView._VALUE) : null;

688 		this._resetRow(sched, false, type);
689         // bug:43660 removing row (splicing array) causes index mismatch.

1481  * half-hour border covers the corresponding time block, whereas an end time
1482  * does not. For example, an appt with a start time of 5:00 causes the 5:00 -
1483  * 5:30 block to be marked. The end time of 5:30 does not cause the 5:30 - 6:00
1484  * block to be marked.
1485  *
1486  * @param time		[Date or int]		time
1487  * @param isEnd		[boolean]*			if true, this is an appt end time
1488  * @param adjust	[boolean]*			Specify whether the time should be

1490  * 										not specified, assumed to be true.

1493 function(time, isEnd, adjust) {
1494     var hourmin,
1495         seconds;
1496     adjust = adjust != null ? adjust : true;
1497     if(adjust) {
1498         var dayStartTime = this._getStartTime();
1499         var indexTime = (time instanceof Date) ? time.getTime() : time;
1500         hourmin = (indexTime - dayStartTime)/60000; //60000 = 1000(msec) * 60 (sec) - hence, dividing by 60000 means calculating the minutes and
1501         seconds = (indexTime - dayStartTime)%60000; //mod by 60000 means calculating the seconds remaining
1502     }
1503     else {
1504         var d = (time instanceof Date) ? time : new Date(time);
1505         hourmin = d.getHours() * 60 + d.getMinutes();
1506         seconds = d.getSeconds();
1507     }
1508     var idx = Math.floor(hourmin / 60) * 2;
1509 	var minutes = hourmin % 60;
1510 	if (minutes >= 30) {
1511 		idx++;
1512 	}
1513 	// end times don't mark blocks on half-hour boundary
1514 	if (isEnd && (minutes == 0 || minutes == 30)) {
1515 		// block even if it exceeds 1 second
1516 		//var s = d.getSeconds();
1517 		if (seconds == 0) {
1518 			idx--;
1519 		}
1520 	}
