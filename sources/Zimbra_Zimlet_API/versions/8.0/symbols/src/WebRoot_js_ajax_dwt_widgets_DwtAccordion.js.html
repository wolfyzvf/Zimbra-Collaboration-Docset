 15 /**
 16  * Creates an empty accordion widget.
 17  * @constructor
 18  * @class
 19  * This class implements an accordion widget, which is a stack of expandable
 20  * accordion headers. Clicking on an accordion header's button expands it in
 21  * place.
 22  *
 23  * @author Parag Shah
 24  *
 25  * @param {DwtControl}	parent	the parent widget
 26  * @param {string}	className	the CSS class
 27  * @param {Dwt.STATIC_STYLE|Dwt.ABSOLUTE_STYLE|Dwt.RELATIVE_STYLE}	posStyle	the positioning style
 28  * 
 29  * @extends		DwtComposite
 30  */
 31 DwtAccordion = function(parent, className, posStyle) {
 32 
 33 	if (arguments.length == 0) return;
 34 	DwtComposite.call(this, {parent:parent, className:className, posStyle:(posStyle || Dwt.ABSOLUTE_STYLE)});
 35 
 36 	this._initialize(className);
 37 };
 38 
 39 DwtAccordion.prototype = new DwtComposite;
 40 DwtAccordion.prototype.constructor = DwtAccordion;

 55 /**
 56  * Adds an item to the accordion, in the form of a table row.
 57  *
 58  * @param {hash}	params		a hash of parameters
 59  * @param {string}      params.title			the text for accordion header
 60  * @param {hash}      params.data			the item data
 61  * @param {string}      params.icon			the icon
 62  * @param {boolean}      params.hideHeader	if <code>true</code>, do not show header (ideal when there's only one visible header item)
 63  */
 64 DwtAccordion.prototype.addAccordionItem =
 65 function(params) {
 66 
 67 	if (!this.isListenerRegistered(DwtEvent.CONTROL)) {
 68 		this.addControlListener(new AjxListener(this, this._controlListener));
 69 	}
 70 
 71 	var itemNum = this.__ITEMCOUNT++;
 72 	var item = new DwtAccordionItem(itemNum, params.title, params.data, this);
 73 	var subs = {
 74 		id: this._htmlElId,
 75 		itemNum: itemNum,
 76 		title: params.title,
 77 		icon: params.icon
 78 	};
 79 
 80 	// append new accordion item
 81 	var row = this._table.insertRow(-1);
 82 	var cell = row.insertCell(-1);
 83 	cell.id = this._htmlElId + "_cell_" + itemNum;
 84 	cell.className = "ZAccordionCell";
 85 	cell.innerHTML = AjxTemplate.expand("dwt.Widgets#ZAccordionItem", subs);
 86 
 87 	// add onclick event handler to header DIV
 88 	var headerDiv = document.getElementById(this._htmlElId + "_header_" + itemNum);
 89 	if (params.hideHeader) {
 90 		Dwt.setVisible(headerDiv, false);
 91 	} else {
 92 		headerDiv.onclick = AjxCallback.simpleClosure(this._handleOnClickHeader, this, item);
 93 		headerDiv.oncontextmenu = AjxCallback.simpleClosure(this._handleOnRightClickHeader, this, item);
 94 		headerDiv.onmouseover = AjxCallback.simpleClosure(this._handleOnMouseoverHeader, this, item);
 95 		headerDiv.onmouseout = AjxCallback.simpleClosure(this._handleOnMouseoutHeader, this, item);
 96 	}
 97 	this._items.push(item);
 98 
 99 	return item;
100 };

112 /**
113  * Gets the accordion item with the given ID.
114  *
115  * @param {number}	id	the accordion item ID
116  * @return	{DwtAccordionItem}		the item or <code>null</code> if not found
117  */
118 DwtAccordion.prototype.getItem =
119 function(id) {
120 	for (var i = 0; i < this._items.length; i++) {
121 		if (this._items[i].id == id) {
122 			return this._items[i];
123 		}
124 	}
125 	return null;
126 };

165 /**
166  * Allows the accordion items to be clickable or not. If disabled, the label of
167  * each accordion item will be grayed out.
168  *
169  * @param {boolean}	enabled			if <code>true</code>, enabled.
170  */
171 DwtAccordion.prototype.setEnabled =
172 function(enabled) {
173 	if (enabled == this._enabled) { return; }
174 
175 	this._enabled = enabled;
176 
177 	for (var i in this._items) {
178 		var item = this._items[i];
179 		if (this._currentItemId != item.id) {
180 			item._setEnabled(enabled);
181 		}
182 	}
183 };

185 /**
186  * Resizes the accordion. This override applies accordion size changes to accordion items as well.
187  *
188  * @param {number}	width		the new width for accordion
189  * @param {number}	height		the new height for accordion
190  */
191 DwtAccordion.prototype.resize =
192 function(width, height) {
193 	if (width) {
194 		// if width changed, resize all header items
195 		for (var i = 0; i < this._items.length; i++) {
196 			var id = this._items[i].id;
197 			var title = document.getElementById(this._htmlElId + "_title_" + id);
198 			var fudge = 30;
199 
200 			var iconCell = document.getElementById(this._htmlElId + "_icon_" + id);
201 			if (iconCell && iconCell.className && iconCell.className != "") {
202 				fudge += 16; // the default width of an icon
203 			}
204 			Dwt.setSize(title, width - fudge);
205 
206 		}
207 	}
208 
209 	var newHeight;
210 	if (height) {
211 		var hdr = document.getElementById(this._htmlElId + "_header_" + this._currentItemId);
212 		if (hdr) {
213 			var hdrHeightSum = Dwt.getSize(hdr).y * this._getVisibleHeaderCount();
214 			newHeight = (height - hdrHeightSum); // force min. height of 100px?
215 		}
216 	}
217 
218 	var body = document.getElementById(this._htmlElId + "_body_" + this._currentItemId);
219 	if (body) {
220 		Dwt.setSize(body, width, newHeight);
221 		if (body.firstChild) {
222 			Dwt.setSize(body.firstChild, width, newHeight);
223 		}
224 	}
225 };

227 DwtAccordion.prototype.setBounds =
228 function(x, y, width, height) {
229 	DwtComposite.prototype.setBounds.call(this, x, y, width, height);
230 
231 	this.resize(width, height);

244 /**
245  * Expands the accordion item with the given ID by making its body visible. The bodies of
246  * other items are hidden.
247  *
248  * @param {number}	id	the accordion item ID
249  * @param {boolean}	notify	if <code>true</code>, selection listeners are to be notified
250  */
251 DwtAccordion.prototype.expandItem =
252 function(id, notify) {
253 	var selectedItem;
254 	for (var i = 0; i < this._items.length; i++) {
255 		var itemId = this._items[i].id;
256 		var header = document.getElementById(this._htmlElId + "_header_" + itemId);
257 		var body = document.getElementById(this._htmlElId + "_body_" + itemId);
258 		var cell = document.getElementById(this._htmlElId + "_cell_" + itemId);
259 		var status = document.getElementById(this._htmlElId + "_status_" + itemId);
260 
261 		if (id == itemId) {
262 			Dwt.setVisible(body, true);
263 			header.className = "ZAccordionHeader ZWidget ZSelected";
264 			status.className = "ImgAccordionOpened";
265 			cell.style.height = "100%";
266 			this._currentItemId = id;
267 			selectedItem = this._items[i];
268 		} else {
269 			Dwt.setVisible(body, false);
270 			header.className = "ZAccordionHeader ZWidget";
271 			status.className = "ImgAccordionClosed";
272 			cell.style.height = "0px";
273 		}
274 	}
275 
276 	if (selectedItem && notify && this.isListenerRegistered(DwtEvent.SELECTION)) {
277 		this.notifySelectionListeners(selectedItem);
278 	}
279 };
280 
281 DwtAccordion.prototype.notifySelectionListeners =
282 function(selectedItem) {
283 	var selEv = DwtShell.selectionEvent;
284 	selEv.item = this;
285 	selEv.detail = selectedItem;
286 	this.notifyListeners(DwtEvent.SELECTION, selEv);
287 };

289 /**
290  * Attaches the HTML content of the given control to the accordion item with
291  * the given ID.
292  *
293  * @param {number}	id				the accordion item ID
294  * @param {DwtControl}	contentObject		the control that contains this item's content
295  */
296 DwtAccordion.prototype.setItemContent =
297 function(id, contentObject) {
298 	var aiBody = this.getBody(id);
299 	if (aiBody) {
300 		this._items[id].control = contentObject;
301 		contentObject.reparentHtmlElement(aiBody);
302 		var size = contentObject.getSize();
303 		this.resize(size.x, size.y);
304 	}
305 };

370 /**
371  * Creates the HTML skeleton for the accordion.
372  * 
373  * @private
374  */
375 DwtAccordion.prototype._initialize =
376 function() {
377 	this._items = [];
378 	this.__ITEMCOUNT = 0;
379 
380 	this.getHtmlElement().innerHTML = AjxTemplate.expand("dwt.Widgets#ZAccordion", {id: this._htmlElId});
381 	this._table = document.getElementById(this._htmlElId + "_accordion_table");
382 
383 	this._setMouseEventHdlrs();

386 /**
387  * Returns the number of accordion items which have visible headers.
388  * 
389  * @private
390  */
391 DwtAccordion.prototype._getVisibleHeaderCount =
392 function() {
393 	var count = 0;
394 	for (var i = 0; i < this._items.length; i++) {
395 		var hdr = document.getElementById(this._htmlElId + "_header_" + this._items[i].id);
396 		if (hdr && Dwt.getVisible(hdr)) {
397 			count++;
398 		}
399 	}
400 	return count;
401 };

406 /**
407  * When a header button is clicked, the item is expanded. Also, any listeners
408  * are notified.
409  *
410  * @param {DwtAccordionItem}	item		the accordion item whose header was clicked
411  * @param {DwtUiEvent}	ev		the click event
412  * 
413  * @private
414  */
415 DwtAccordion.prototype._handleOnClickHeader =
416 function(item, ev) {
417 	if (!this._enabled) { return; }
418 
419 	this.expandItem(item.id, true);
420 };
421 
422 /**
423  * When a header button is right-clicked, any listeners will be notified so a
424  * context menu can be shown, for example.
425  *
426  * @param {DwtAccordionItem}	item		the accordion item whose header was clicked
427  * @param {DwtUiEvent}	ev		the click event
428  * 
429  * @private
430  */
431 DwtAccordion.prototype._handleOnRightClickHeader =
432 function(item, ev) {
433 	ev = ev || window.event;
434 
435 	if (this.isListenerRegistered(DwtEvent.ONCONTEXTMENU)) {
436 		var selEv = DwtShell.selectionEvent;
437 		DwtUiEvent.copy(selEv, ev);
438 		selEv.item = this;
439 		selEv.detail = item;
440 		this.notifyListeners(DwtEvent.ONCONTEXTMENU, selEv);
441 	}
442 };
443 
444 DwtAccordion.prototype._handleOnMouseoverHeader =
445 function(item, ev) {
446 	ev = ev || window.event;
447 
448 	if (this.isListenerRegistered(DwtEvent.ONMOUSEOVER)) {
449 		var selEv = DwtShell.selectionEvent;
450 		DwtUiEvent.copy(selEv, ev);
451 		selEv.item = this;
452 		selEv.detail = item;
453 		this.notifyListeners(DwtEvent.ONMOUSEOVER, selEv);
454 	}
455 };
456 
457 DwtAccordion.prototype._handleOnMouseoutHeader =
458 function(ev) {
459 	this.setToolTipContent(null);
460 };

462 /**
463  * Handles a resize event.
464  *
465  * @param {DwtEvent}	ev	the control event
466  * 
467  * @private
468  */
469 DwtAccordion.prototype._controlListener =
470 function(ev) {
471 	if (this.getScrollStyle() != Dwt.CLIP) { return; }
472 
473 	var newWidth = (ev.oldWidth != ev.newWidth) ? ev.newWidth : null;
474 	var newHeight = (ev.oldHeight != ev.newHeight) ? ev.newHeight : null;
475 
476 	if ((!newWidth && !newHeight) || ev.newWidth < 0 || ev.newHeight < 0) { return;	}
477 
478 	this.resize(newWidth, newHeight);
479 };

544 /**
545  * @private
546  */
547 DwtAccordionItem.prototype._setEnabled =
548 function(enabled) {
549 	var titleCell = document.getElementById(this.accordion._htmlElId + "_title_" + this.id);
550 	if (titleCell) {
551 		if (enabled) {
552 			Dwt.delClass(titleCell, "ZDisabled");
553 		} else {
554 			Dwt.addClass(titleCell, "ZDisabled");
555 		}
556 	}
557 
558 	var status = document.getElementById(this.accordion._htmlElId + "_status_" + this.id);
559 	if (status) {
560 		if (enabled) {
561 			Dwt.delClass(status, "ZDisabledImage ZDisabled");
562 		} else {
563 			Dwt.addClass(status, "ZDisabledImage ZDisabled");
564 		}
565 	}
