 16 ZmTaskMultiView = function(params) {
 17 
 18 	if (arguments.length == 0) { return; }
 19 
 20     params.className = params.className || "ZmTaskMultiView";
 21     params.mode = ZmId.VIEW_TASKMULTI;
 22 
 23 	var view = params.controller.getCurrentViewId();
 24 	params.id = ZmId.getViewId(view);
 25 	DwtComposite.call(this, params);
 26 
 27 	this._controller = params.controller;
 28 
 29     this._taskListView = this._createTaskListView(params);
 30 
 31 	this._vertMsgSash = new DwtSash({parent:this, style:DwtSash.HORIZONTAL_STYLE, className:"AppSash-horiz",
 32 									 threshold:ZmTaskMultiView.SASH_THRESHOLD, posStyle:Dwt.ABSOLUTE_STYLE});
 33 	this._vertMsgSash.registerCallback(this._sashCallback, this);
 34 
 35 	this._horizMsgSash = new DwtSash({parent:this, style:DwtSash.VERTICAL_STYLE, className:"AppSash-vert",
 36 									  threshold:ZmTaskMultiView.SASH_THRESHOLD, posStyle:Dwt.ABSOLUTE_STYLE});
 37 	this._horizMsgSash.registerCallback(this._sashCallback, this);
 38 
 39 	this._taskView = new ZmTaskView(this, DwtControl.ABSOLUTE_STYLE, this._controller);
 40 
 41 	this.setReadingPane();
 42 }

 69 /**
 70  * Displays the reading pane, based on the current settings.
 71  */
 72 ZmTaskMultiView.prototype.setReadingPane =
 73 function() {
 74 
 75 	var tlv = this._taskListView, tv = this._taskView;
 76 	var readingPaneEnabled = this._controller.isReadingPaneOn();
 77     var readingPaneOnRight = this._controller.isReadingPaneOnRight();
 78 
 79     if (!readingPaneEnabled) {
 80 		tv.setVisible(false);
 81 		this._vertMsgSash.setVisible(false);
 82 		this._horizMsgSash.setVisible(false);
 83 	} else {
 84 		if (!tv.getVisible()) {
 85 			if (tlv.getSelectionCount() == 1) {
 86 				this._controller._setSelectedItem();
 87 			} else {
 88 				tv.reset();
 89 			}
 90 		}
 91 		tv.setVisible(true);
 92 		var newSash = readingPaneOnRight ? this._vertMsgSash : this._horizMsgSash;
 93 		var oldSash = readingPaneOnRight ? this._horizMsgSash : this._vertMsgSash;
 94 		oldSash.setVisible(false);
 95 		newSash.setVisible(true);
 96 	}
 97 
 98 	tlv.reRenderListView();
 99     if(readingPaneOnRight) {
100         tlv.setSortByAsc(ZmItem.F_SORTED_BY, true);
101     } else {
102         tlv.setSortByAsc(ZmItem.F_DATE, true);
103     }
104 
105     tv.noTab = !readingPaneEnabled || AjxEnv.isIE;
106 	var sz = this.getSize();
107 	this._resetSize(sz.x, sz.y, true);
108 };

170 ZmTaskMultiView.prototype._resetSize =
171 function(newWidth, newHeight, force) {
172 
173 
174 	if (newWidth <= 0 || newHeight <= 0) { return; }
175 	if (!force && newWidth == this._lastResetWidth && newHeight == this._lastResetHeight) { return; }
176 
177 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
178 
179 	if (this.isTaskViewVisible()) {
180 		var sash = this.getSash();
181 		var sashSize = sash.getSize();
182 		var sashThickness = readingPaneOnRight ? sashSize.x : sashSize.y;
183 		if (readingPaneOnRight) {
184 			var listViewWidth = this._vertSashX || (Number(ZmMsg.LISTVIEW_WIDTH)) || Math.floor(newWidth / 2.5);
185 			this._taskListView.resetSize(listViewWidth, newHeight);
186 			sash.setLocation(listViewWidth, 0);
187 			this._taskView.setBounds(listViewWidth + sashThickness, 0,
188 									newWidth - (listViewWidth + sashThickness), newHeight);
189 		} else {
190 			var listViewHeight = this._horizSashY || (Math.floor(newHeight / 2) - DwtListView.HEADERITEM_HEIGHT);
191 			this._taskListView.resetSize(newWidth, listViewHeight);
192 			sash.setLocation(0, listViewHeight);
193 			this._taskView.setBounds(0, listViewHeight + sashThickness, newWidth,
194 									newHeight - (listViewHeight + sashThickness));
195 		}
196 	} else {
197 		this._taskListView.resetSize(newWidth, newHeight);
198 	}
199 	this._taskListView._resetColWidth();
200 
201 	this._lastResetWidth = newWidth;
202 	this._lastResetHeight = newHeight;
203 };

205 ZmTaskMultiView.prototype._sashCallback =
206 function(delta) {
207 
208 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
209 
210 	if (delta > 0) {
211 		if (readingPaneOnRight) {
212 			// moving sash right
213 			var minMsgViewWidth = this._taskView.getMinWidth();
214 			var currentMsgWidth = this._taskView.getSize().x;
215 			delta = Math.max(0, Math.min(delta, currentMsgWidth - minMsgViewWidth));
216 			var newListWidth = ((AjxEnv.isIE) ? this._vertMsgSash.getLocation().x : this._taskListView.getSize().x) + delta;
217 
218 			if (delta > 0) {
219 				this._taskListView.resetSize(newListWidth, Dwt.DEFAULT);
220 				this._taskView.setBounds(this._taskView.getLocation().x + delta, Dwt.DEFAULT,
221 										currentMsgWidth - delta, Dwt.DEFAULT);
222 			} else {
223 				delta = 0;
224 			}
225 		} else {
226 			// moving sash down
227 			var newMsgViewHeight = this._taskView.getSize().y - delta;
228 			var minMsgViewHeight = this._taskView.getMinHeight();
229 			if (newMsgViewHeight > minMsgViewHeight) {
230 				this._taskListView.resetSize(Dwt.DEFAULT, this._taskListView.getSize().y + delta);
231 				this._taskView.setBounds(Dwt.DEFAULT, this._taskView.getLocation().y + delta,
232 										Dwt.DEFAULT, newMsgViewHeight);
233 			} else {
234 				delta = 0;
235 			}
236 		}
237 	} else {
238 		var absDelta = Math.abs(delta);
239 
240 		if (readingPaneOnRight) {
241 			// moving sash left
242 			if (!this._minMLVWidth) {
243 				var firstHdr = this._taskListView._headerList[0];
244 				var hdrWidth = firstHdr._width;
245 				if (hdrWidth == "auto") {
246 					var header = Dwt.byId(firstHdr._id);
247 					hdrWidth = header && Dwt.getSize(header).x;
248 				}
249 				this._minMLVWidth = hdrWidth;
250 			}
251 
252 			var currentWidth = ((AjxEnv.isIE) ? this._vertMsgSash.getLocation().x : this._taskListView.getSize().x);
253 			absDelta = Math.max(0, Math.min(absDelta, currentWidth - this._minMLVWidth));
254 
255 			if (absDelta > 0) {
256 				delta = -absDelta;
257 				this._taskListView.resetSize(currentWidth - absDelta, Dwt.DEFAULT);
258 				this._taskView.setBounds(this._taskView.getLocation().x - absDelta, Dwt.DEFAULT,
259 										this._taskView.getSize().x + absDelta, Dwt.DEFAULT);
260 			} else {
261 				delta = 0;
262 			}
263 		} else {
264 			// moving sash up
265 			if (!this._minMLVHeight) {
266 				var list = this._taskListView.getList();
267 				if (list && list.size()) {
268 					var item = list.get(0);
269 					var div = document.getElementById(this._taskListView._getItemId(item));
270 					this._minMLVHeight = DwtListView.HEADERITEM_HEIGHT + (Dwt.getSize(div).y * 2);
271 				} else {
272 					this._minMLVHeight = DwtListView.HEADERITEM_HEIGHT;
273 				}
274 			}
275 
276 			if (this.getSash().getLocation().y - absDelta > this._minMLVHeight) {
277 				// moving sash up
278 				this._taskListView.resetSize(Dwt.DEFAULT, this._taskListView.getSize().y - absDelta);
279 				this._taskView.setBounds(Dwt.DEFAULT, this._taskView.getLocation().y - absDelta,
280 										Dwt.DEFAULT, this._taskView.getSize().y + absDelta);
281 			} else {
282 				delta = 0;
283 			}
284 		}
285 	}
286 
287 	if (delta) {
288 		this._taskListView._resetColWidth();
289 		if (readingPaneOnRight) {
290 			this._vertSashX = this._vertMsgSash.getLocation().x;
291 		} else {
292 			this._horizSashY = this._horizMsgSash.getLocation().y;
293 		}
294 	}
295 
296 	return delta;
297 };

299 ZmTaskMultiView.prototype._selectFirstItem =
300 function() {
301 	var list = this._taskListView.getList();
302 	var selectedItem = list ? list.get(0) : null;
303 	if (selectedItem) {
304 		this._taskListView.setSelection(selectedItem, false);
305 	}
306 };

311 	return readingPaneOnRight ? this._vertMsgSash : this._horizMsgSash;
