 20 /**
 21  * Creates a folder notification dialog.
 22  * @class
 23  * This class represents a folder notification dialog.
 24  * 
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtDialog
 29  */
 30 ZmFolderNotifyDialog = function(parent, className) {
 31 
 32     className = className || "ZmFolderNotifyDialog";
 33 
 34 	var extraButtons = [ new DwtDialog_ButtonDescriptor(ZmFolderPropsDialog.ADD_SHARE_BUTTON, ZmMsg.addShare, DwtDialog.ALIGN_LEFT)];
 35 
 36 	DwtDialog.call(this, {parent:parent, className:className, title:ZmMsg.folderNotify, extraButtons:extraButtons});
 37     this.getButton(DwtDialog.OK_BUTTON).setText(ZmMsg.notify);
 38 
 39     this.registerCallback(ZmFolderPropsDialog.ADD_SHARE_BUTTON, this._handleAddShareButton, this);
 40 
 41 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 42 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._handleCancelButton));
 43 
 44 	this._folderChangeListener = new AjxListener(this, this._handleFolderChange);
 45 	
 46     this.setView(this._createView());
 47 };
 48 
 49 ZmFolderNotifyDialog.prototype = new DwtDialog;
 50 ZmFolderNotifyDialog.prototype.constructor = ZmFolderNotifyDialog;
 51

 53 
 54 ZmFolderNotifyDialog.ADD_SHARE_BUTTON = ++DwtDialog.LAST_BUTTON;
 55 
 56 ZmFolderNotifyDialog.SHARES_HEIGHT = "9em";
 57 
 58 // Public methods
 59 
 60 ZmFolderNotifyDialog.prototype.toString =
 61 function() {
 62 	return "ZmFolderNotifyDialog";
 63 };
 64 
 65 /**
 66  * Pops-up the notification dialog.
 67  * 
 68  * @param	{ZmOrganizer}	organizer		the organizer
 69  */
 70 ZmFolderNotifyDialog.prototype.popup =
 71 function(organizer) {
 72 
 73     this._organizer = organizer;
 74 	organizer.addChangeListener(this._folderChangeListener);
 75 
 76     this._populateShares(organizer);
 77 
 78     this._reply.setReplyType(ZmShareReply.STANDARD);
 79     this._reply.setReplyNote("");
 80 
 81     DwtDialog.prototype.popup.call(this);
 82 };
 83 
 84 ZmFolderNotifyDialog.prototype.popdown =
 85 function() {
 86 	this._organizer.removeChangeListener(this._folderChangeListener);
 87 	this._organizer = null;
 88 	DwtDialog.prototype.popdown.call(this);
 89 };

 93 ZmFolderNotifyDialog.prototype._handleAddShareButton =
 94 function(event) {
 95 	var sharePropsDialog = appCtxt.getSharePropsDialog();
 96 	sharePropsDialog.popup(ZmSharePropsDialog.NEW, this._organizer, null);
 97 };
 98 
 99 ZmFolderNotifyDialog.prototype._handleOkButton =
100 function(event) {
101 
102     var replyType = this._reply.getReplyType();
103     var notes = replyType == ZmShareReply.QUICK ? this._reply.getReplyNote() : "";
104     var shares = this._organizer.shares;
105 
106     for (var i = 0; i < shares.length; i++) {
107         var share = shares[i];
108         var email = share.grantee.email;
109         if (!email) {
110             // last resort: check if grantee name is a valid email address
111             if (AjxEmailAddress.isValid(share.grantee.name))
112                 email = share.grantee.name;
113         }
114 
115         if (!email) { continue; }
116 
117         var addrs = new AjxVector();
118         var addr = new AjxEmailAddress(email, AjxEmailAddress.TO);
119         addrs.add(addr);
120 
121         var tmpShare = new ZmShare({object:share.object});
122 
123         tmpShare.grantee.id = share.grantee.id;
124         tmpShare.grantee.email = email;
125         tmpShare.grantee.name = share.grantee.name;
126 
127 			// REVISIT: What if you have delegated access???
128         if(tmpShare.object.isRemote()) {
129             tmpShare.grantor.id = tmpShare.object.zid;
130             tmpShare.grantor.email = tmpShare.object.owner;
131             tmpShare.grantor.name = tmpShare.grantor.email;
132             tmpShare.link.id = tmpShare.object.rid;
133         }else {
134             tmpShare.grantor.id = appCtxt.get(ZmSetting.USERID);
135             tmpShare.grantor.email = appCtxt.get(ZmSetting.USERNAME);
136             tmpShare.grantor.name = appCtxt.get(ZmSetting.DISPLAY_NAME) || tmpShare.grantor.email;
137             tmpShare.link.id = tmpShare.object.id;
138         }
139 
140         tmpShare.link.perm = share.link.perm;
141         tmpShare.link.name = tmpShare.object.name;
142         tmpShare.link.view = ZmOrganizer.getViewName(tmpShare.object.type);
143         tmpShare.link.inh = this._inheritEl ? this._inheritEl.checked : true;
144 
145         tmpShare.notes = notes;
146 
147         tmpShare.sendMessage(ZmShare.NOTIFY, addrs);
148     }
149 
150     this.popdown();
151 };

153 ZmFolderNotifyDialog.prototype._handleCancelButton =
154 function(event) {
155 	this.popdown();
156 };
157 
158 ZmFolderNotifyDialog.prototype._handleFolderChange =
159 function(event) {
160     this._populateShares(this._organizer);
161 };
162 
163 ZmFolderNotifyDialog.prototype._populateShares =
164 function(organizer) {
165     
166     this._sharesGroup.setContent("");
167 
168 	var link = organizer.link;
169 	var shares = organizer.shares;
170 	var visible = ((!link || organizer.isAdmin()) && shares && shares.length > 0);
171 	if (visible) {
172 		AjxDispatcher.require("Share");
173 		var table = document.createElement("TABLE");
174 		table.border = 0;
175 		table.cellSpacing = 0;
176 		table.cellPadding = 3;
177 		for (var i = 0; i < shares.length; i++) {
178 			var share = shares[i];
179 			var row = table.insertRow(-1);
180 
181 			var nameEl = row.insertCell(-1);
182 			nameEl.style.paddingRight = "15px";
183 			var nameText = share.grantee.name || ZmMsg.userUnknown;
184 			if (share.isAll()) nameText = ZmMsg.shareWithAll;
185 			else if (share.isPublic()) nameText = ZmMsg.shareWithPublic;
186 			nameEl.innerHTML = AjxStringUtil.htmlEncode(nameText);
187 
188 			var roleEl = row.insertCell(-1);
189 			roleEl.style.paddingRight = "15px";
190 			roleEl.innerHTML = ZmShare.getRoleName(share.link.perm);
191 		}
192 		this._sharesGroup.setElement(table);
193 
194 		var width = Dwt.DEFAULT;
195 		var height = shares.length > 5 ? ZmFolderNotifyDialog.SHARES_HEIGHT : Dwt.CLEAR;
196 
197 		var insetElement = this._sharesGroup.getInsetHtmlElement();
198 		Dwt.setScrollStyle(insetElement, Dwt.SCROLL);
199 		Dwt.setSize(insetElement, width, height);
200         this.getButton(DwtDialog.OK_BUTTON).setEnabled(true);
201     }else{
202         this._sharesGroup.setContent("<center>"+ZmMsg.noShareDetailsFound+"</center>");
203         this.getButton(DwtDialog.OK_BUTTON).setEnabled(false);
204     }
205 };
206 
207 ZmFolderNotifyDialog.prototype._createView =
208 function() {
209 
210     var view = new DwtComposite(this);
211 
212     // add message group
213 	this._reply = new ZmShareReply(view, null, [ZmShareReply.STANDARD, ZmShareReply.QUICK]);
214 
215     this._messageGroup = new DwtGrouper(view);
216 	this._messageGroup.setLabel(ZmMsg.message);
217 	this._messageGroup.setView(this._reply);
218     view.getHtmlElement().appendChild(this._messageGroup.getHtmlElement());
219 
220     // setup shares group
221     this._sharesGroup = new DwtGrouper(view);
222     this._sharesGroup.setLabel(ZmMsg.folderSharing);
223     this._sharesGroup.setVisible(true);
224     this._sharesGroup.setScrollStyle(Dwt.SCROLL);    
225     view.getHtmlElement().appendChild(this._sharesGroup.getHtmlElement());
226 
227     return view;
228 };
