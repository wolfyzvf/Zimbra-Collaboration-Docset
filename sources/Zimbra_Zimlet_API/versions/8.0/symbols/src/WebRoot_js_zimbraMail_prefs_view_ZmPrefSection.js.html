 16 /**
 17  * Creates an empty preferences section of the given type.
 18  * @constructor
 19  * @class
 20  * This class represents a single page of preferences available by selecting one of the
 21  * preference tabs. During construction, skeleton HTML is created. The preferences
 22  * aren't added until the page becomes visible.
 23  *
 24  * @param {DwtControl}	parent			the containing widget
 25  * @param {constant}	section			the preferences page
 26  * @param {ZmController}	controller		the prefs controller
 27  * 
 28  * @extends		DwtTabViewPage
 29  * 
 30  * @private
 31  */
 32 ZmPrefSection = function(parent, section, controller) {
 33 	DwtTabViewPage.call(this, parent, "ZmPreferencesPage");
 34 
 35 	this._section = section;
 36 	this._controller = controller;
 37 	this._title = [ZmMsg.zimbraTitle, controller.getApp().getDisplayName(), section.title].join(": ");
 38 
 39 	this._rendered = false; // for DwtTabViewPage
 40 	this.hasRendered = false;
 41 };
 42 
 43 ZmPrefSection.prototype = new DwtTabViewPage;
 44 ZmPrefSection.prototype.constructor = ZmPrefSection;
 45 
 46 ZmPrefSection.prototype.toString =
 47 function () {
 48     return "ZmPrefSection";
 49 };
 50 
 51 //
 52 // Public methods
 53 //
 54 
 55 // DwtTabViewPage methods
 56 
 57 ZmPrefSection.prototype.showMe =
 58 function() {
 59 	Dwt.setTitle(this._title);
 60     if (this.hasRendered) { return; }
 61 
 62     // expand section template
 63     var templateId = this._section.templateId;
 64     var data = { id: this._htmlElId };
 65     data.isEnabled = AjxCallback.simpleClosure(this._isEnabled, this);
 66     data.expandField = AjxCallback.simpleClosure(this._expandField, this, data);
 67 
 68     this._contentEl.innerHTML = AjxTemplate.expand(templateId, data);
 69 
 70     // create controls
 71     var details = ZmPreferencesApp[this._section.id];
 72     var prefs = details && details.prefs;
 73     if (prefs) {
 74         for (var i = 0; i < prefs.length; i++) {
 75             var settingId = prefs[i];
 76         }
 77     }
 78 };
 79 
 80 /**
 81  * Gets the title.
 82  * 
 83  * @return	{String}		the title
 84  */
 85 ZmPrefSection.prototype.getTitle =
 86 function() {
 87 	return this._title;
 88 };
 89 
 90 //
 91 // Protected methods
 92 //
 93 
 94 ZmPrefSection.prototype._isEnabled =
 95 function(prefId) {
 96     return this._controller.checkPreCondition(ZmPref.SETUP[prefId]);
 97 };
 98 
 99 ZmPrefSection.prototype._expandField =
100 function(data, prefId) {
101     var templateId = this._section.templateId.replace(/#.*$/, "#"+prefId);
102     return AjxTemplate.expand(templateId, data);
103 };
