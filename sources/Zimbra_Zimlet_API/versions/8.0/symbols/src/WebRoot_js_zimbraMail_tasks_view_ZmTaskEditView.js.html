113 ZmTaskEditView.prototype._populateForEdit =
114 function(calItem, mode) {
115 	ZmCalItemEditView.prototype._populateForEdit.call(this, calItem, mode);
116 
117 	if (calItem.startDate) {
118 		var sd = new Date(calItem.startDate.getTime());
119 		this._startDateField.value = AjxDateUtil.simpleComputeDateStr(sd);
120 	}
121 	if (calItem.endDate) {
122 		var ed = new Date(calItem.endDate.getTime());
123 		this._endDateField.value = AjxDateUtil.simpleComputeDateStr(ed);
124 	}
125 
126 
127     var rd = new Date(calItem.remindDate.getTime());
128 
129     if (calItem.alarm) {
130         if (calItem.remindDate && calItem._reminderAbs) {
131             this._remindTimeSelect.set(calItem.remindDate);
132         }
133     } else {
134         var now = AjxDateUtil.roundTimeMins(new Date(), 30);
135         this._remindTimeSelect.set(now);        
136     }
137 
138     if (this._hasReminderSupport) {
139         this._remindDateField.value = AjxDateUtil.simpleComputeDateStr(rd);
140         this._reminderCheckbox.setSelected(calItem.alarm);
141         this._setRemindersEnabled(calItem.alarm);
142 
143         if (calItem.alarmActions.contains(ZmCalItem.ALARM_EMAIL)) {
144             this._reminderEmailCheckbox.setSelected(true);
145         }
146         if (calItem.alarmActions.contains(ZmCalItem.ALARM_DEVICE_EMAIL)) {
147             this._reminderDeviceEmailCheckbox.setSelected(true);
148         }
149 
150         this._setEmailReminderControls();
151     }
152 
153 	this._location.setValue(calItem.getLocation());
154 	this._setPriority(calItem.priority);
155 	this._statusSelect.setSelectedValue(calItem.status);
156     this._pCompleteSelectInput.setValue(this.formatPercentComplete(calItem.pComplete));
157     if (!this._notesHtmlEditor.getContent() && calItem.message){
158         this._notesHtmlEditor.setContent(calItem.message.getInviteDescriptionContentValue(ZmMimeTable.TEXT_PLAIN) || "");
159     }
160 };

162 ZmTaskEditView.prototype._populateForSave =
163 function(calItem) {
164 
165 	ZmCalItemEditView.prototype._populateForSave.call(this, calItem);
166 
167 	calItem.location = this._location.getValue();
168 	// TODO - normalize
169 	var startDate = AjxDateUtil.simpleParseDateStr(this._startDateField.value);
170 	var endDate = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
171 
172 	if (startDate) {
173 		calItem.setStartDate(startDate, true);
174 	} else {
175 		calItem.startDate = null;	// explicitly null out in case item has old data
176 	}
177 
178 	if (endDate) {
179 		calItem.setEndDate(endDate, true);
180 	} else {
181 		calItem.endDate = null;		// explicitly null out in case item has old data
182 	}
183 
184     //set reminder
185     var reminders = [
186         { control: this._reminderEmailCheckbox,       action: ZmCalItem.ALARM_EMAIL        },
187         { control: this._reminderDeviceEmailCheckbox, action: ZmCalItem.ALARM_DEVICE_EMAIL }
188     ];
189     if (this._hasReminderSupport && this._reminderCheckbox.isSelected()) {
190         var remindDate = AjxDateUtil.simpleParseDateStr(this._remindDateField.value);
191         calItem.alarm = true;
192         calItem.remindDate = remindDate;
193         remindDate = this._remindTimeSelect.getValue(remindDate);
194         var remindFmtStr = AjxDateUtil.getServerDateTime(remindDate,true);
195         calItem.setTaskReminder(remindFmtStr);
196         for (var i = 0; i < reminders.length; i++) {
197             var reminder = reminders[i];
198             if (reminder.control && reminder.control.isSelected()) {
199                 calItem.addReminderAction(reminder.action);
200             }
201             else {
202                 calItem.removeReminderAction(reminder.action);
203             }
204         }
205     } else {
206        calItem.alarm = false;
207        calItem.remindDate = new Date();
208        calItem.setTaskReminder(null);
209     }
210     
211 	calItem.setAllDayEvent(true);
212     calItem.pComplete = this.getpCompleteInputValue();
213 	calItem.priority = this._getPriority();
214 	calItem.status = this._statusSelect.getValue();
215 
216     //bug:51913 disable alarm when stats is completed
217     if(this.getpCompleteInputValue() == 100 && this._statusSelect.getValue() == ZmCalendarApp.STATUS_COMP) {
218        calItem.alarm = false;
219        calItem.remindDate = new Date();
220        calItem.setTaskReminder(null);
221        for (var i = 0; i < reminders.length; i++) {
222            var reminder = reminders[i];
223            if (reminder.control && reminder.control.isSelected()) {
224                calItem.removeReminderAction(reminder.action);
225            }
226        }
227     }
228 
229 //	XXX: uncomment when supported
230 //	this._getRecurrence(calItem);	// set any recurrence rules LAST
231 
232 	return calItem;
233 };

235 ZmTaskEditView.prototype.isValid =
236 function() {
237 	var errorMsg;
238 	var subj = AjxStringUtil.trim(this._subjectField.getValue());
239 
240 	if (subj && subj.length) {
241 		var startDate = AjxStringUtil.trim(this._startDateField.value);
242         var endDate =   AjxStringUtil.trim(this._endDateField.value);
243 		if (startDate.length > 0 && (!ZmTimeSelect.validStartEnd(this._startDateField, this._endDateField))) {
244             if(endDate.length <= 0) {
245                 errorMsg = ZmMsg.errorEmptyTaskDueDate;
246             } else {
247                 errorMsg = ZmMsg.errorInvalidDates;
248             }
249 		}
250 		var remindTime =  ZmTimeSelect.parse(this._remindTimeSelect.getInputField().getValue());
251 		if (!remindTime) {
252 			errorMsg = AjxMsg.invalidTimeString;
253 		}
254         if(Math.round(this.getpCompleteInputValue()) > 100) {
255            errorMsg = ZmMsg.errorInvalidPercentage;
256         }
257     } else {
258 		errorMsg = ZmMsg.errorMissingSubject;
259 	}
260 
261 	if (errorMsg) {
262 		throw errorMsg;
263 	}
264 
265 	return true;
266 };

366 ZmTaskEditView.prototype.formatPercentComplete = function(pValue) {
367 
368    var formatter = new AjxMessageFormat(AjxMsg.percentageString);
369    if(AjxUtil.isString(pValue) && pValue.indexOf("%") != -1) {
370        return formatter.format(Math.round(pValue.replace(/[%]/g,"")));
371    } else {
372        return formatter.format(Math.round(pValue));
373    }

456 ZmTaskEditView.prototype._remindDateBtnListener =
457 function(ev) {
458 	var calDate = ev.item == this._remindDateButton
459 		? AjxDateUtil.simpleParseDateStr(this._remindDateField.value)
460 		: null;
461 
462 	// if date was input by user and its foobar, reset to today's date
463 	if (calDate == null || isNaN(calDate)) {
464 		calDate = new Date();
465 		var field = ev.item == this._remindDateButton ? this._remindDateField : null;
466 
467         var calEndDate = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
468         if (calEndDate != null || isNaN(calEndDate)) {
469             calDate = calEndDate;
470         }
471         
472         field.value = AjxDateUtil.simpleComputeDateStr(calDate);
473 	}
474 
475 	// always reset the date to current field's date
476 	var menu = ev.item.getMenu();
477 	var cal = menu.getItem(0);
478 	cal.setDate(calDate, true);
479 	ev.item.popup();
480 };

482 ZmTaskEditView.prototype._checkReminderDate =
483 function(){

491 ZmTaskEditView.prototype._dateCalSelectionListener = function(ev) {
492 
493     ZmCalItemEditView.prototype._dateCalSelectionListener.call(this,ev);
494 
495     var parentButton = ev.item.parent.parent;
496 	var newDate = AjxDateUtil.simpleComputeDateStr(ev.detail);
497 
498     var ed = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
499     var sd = AjxDateUtil.simpleParseDateStr(this._startDateField.value);
500     var rd = AjxDateUtil.simpleParseDateStr(this._remindDateField.value);
501 
502 	// change the start/end date if they mismatch
503 	if (parentButton == this._endDateButton) {
504 		if(rd && (rd.valueOf() > ev.detail.valueOf())) {
505             this._remindDateField.value = newDate;
506         }
507 		//this._endDateField.value = newDate;
508 	} else if(parentButton == this._remindDateButton) {
509         if (ed && (ed.valueOf() < ev.detail.valueOf())) {
510 			this._endDateField.value = newDate;
511         }
512         if (ed == null && sd && (sd.valueOf() < ev.detail.valueOf())) {
513 			this._startDateField.value = newDate;
514         }
515 		this._remindDateField.value = newDate;
516 	} else if(parentButton == this._startDateButton) {
517         if (ed == null && rd && (rd.valueOf() > ev.detail.valueOf())) {
518               this._remindDateField.value = newDate;
519         }
520     }
521     this._checkReminderDate();
522 };

553 // Returns a string representing the form content
554 ZmTaskEditView.prototype._formValue =
555 function(excludeAttendees) {
556 	var vals = [];
557 
558 	vals.push(this._subjectField.getValue());
559 	vals.push(this._location.getValue());
560 	vals.push(this._getPriority());
561 	vals.push(this._folderSelect.getValue());
562 	vals.push(this.getpCompleteInputValue());
563 	vals.push(this._statusSelect.getValue());
564 	var startDate = AjxDateUtil.simpleParseDateStr(this._startDateField.value);
565 	if (startDate) vals.push(AjxDateUtil.getServerDateTime(startDate));
566 	var endDate = AjxDateUtil.simpleParseDateStr(this._endDateField.value);
567 	if (endDate) vals.push(AjxDateUtil.getServerDateTime(endDate));
568 
569     if (this._hasReminderSupport) {
570         var hasReminder = this._reminderCheckbox.isSelected();
571         vals.push(hasReminder);
572         if (hasReminder) {
573             var remindDate = AjxDateUtil.simpleParseDateStr(this._remindDateField.value);
574             remindDate = this._remindTimeSelect.getValue(remindDate);
575             if(remindDate) {
576                 vals.push(
577                     AjxDateUtil.getServerDateTime(remindDate)
578                 );
579             }
580             vals.push(this._reminderEmailCheckbox.isSelected());
581             vals.push(this._reminderDeviceEmailCheckbox.isSelected());
582         }
583     }
584 
585 	//vals.push(this._repeatSelect.getValue());
586 	vals.push(this._notesHtmlEditor.getContent());
587 
588 	var str = vals.join("|");
589 	str = str.replace(/\|+/, "|");
590 	return str;
591 };

629 ZmTaskEditView.prototype._handleCompleteOnBlur =
630 function(inputEl) {
631 	var pCompleteString = inputEl.value;
632     if (!pCompleteString) {
633 		inputEl.value = this.formatPercentComplete(0);
634 		return;
635 	}
636     var newVal = this.getpCompleteInputValue();
637     if (newVal == 100) {
638         this._statusSelect.setSelectedValue(ZmCalendarApp.STATUS_COMP);
639         this._unSelectRemindersCheckbox();   //bug:51913 disable alarm when stats is completed
640     } else if (newVal == 0) {
641         this._statusSelect.setSelectedValue(ZmCalendarApp.STATUS_NEED);
642     } else if ((newVal > 0 || newVal < 100) && (this._statusSelect.getValue() != ZmCalendarApp.STATUS_COMP || this._statusSelect.getValue() != ZmCalendarApp.STATUS_NEED))
643     {
644         this._statusSelect.setSelectedValue(ZmCalendarApp.STATUS_INPR);
645     }
646     inputEl.value = this.formatPercentComplete(pCompleteString);
647 };

674 ZmTaskEditView.prototype._selectListener =
675 function(ev) {
676 	var newVal = ev._args.newValue;
677 	var oldVal = ev._args.oldValue;
678 
679 	if (newVal == oldVal) { return; }
680 
681 	var selObj = ev._args.selectObj;
682 
683 	if (selObj == this._statusSelect) {
684 		if (newVal == ZmCalendarApp.STATUS_COMP) {
685 			this._pCompleteSelectInput.setValue(this.formatPercentComplete(100));
686             this._unSelectRemindersCheckbox();    //bug:51913 disable alarm when stats is completed
687 		} else if (newVal == ZmCalendarApp.STATUS_NEED) {
688 			this._pCompleteSelectInput.setValue(this.formatPercentComplete(0));
689 		} else if (newVal == ZmCalendarApp.STATUS_INPR) {
690 			if (this.getpCompleteInputValue() == "100") {
691 				this._pCompleteSelectInput.setValue(this.formatPercentComplete(0));
692 			}
693 		}
694 	} else {
695 		if (newVal == 100) {
696 			this._statusSelect.setSelectedValue(ZmCalendarApp.STATUS_COMP);
697             this._unSelectRemindersCheckbox();
698 		} else if (newVal == 0) {
699 			this._statusSelect.setSelectedValue(ZmCalendarApp.STATUS_NEED);
700 		} else if ((oldVal == 0 || oldVal == 100) &&
701 			 		(newVal > 0 || newVal < 100) &&
702 					(this._statusSelect.getValue() == ZmCalendarApp.STATUS_COMP ||
703 					 this._statusSelect.getValue() == ZmCalendarApp.STATUS_NEED))
704 		{
705 			this._statusSelect.setSelectedValue(ZmCalendarApp.STATUS_INPR);
706 		}
707 	}
708 };
