 16 /**
 17  * Singleton wrapper of BrowserPlus.
 18  * @class
 19  * @private
 20  */
 21 ZmBrowserPlus = function() {
 22 };
 23 
 24 ZmBrowserPlus.prototype.toString =
 25 function() {
 26 	return "ZmBrowserPlus";
 27 };
 28 
 29 ZmBrowserPlus.getInstance =
 30 function() {
 31 	return ZmBrowserPlus.INSTANCE = ZmBrowserPlus.INSTANCE || new ZmBrowserPlus();
 32 };
 33 
 34 /**
 35  * Loads a browser plus service.
 36  * @param serviceObj    [Object]		Description of service object.
 37  * @param callback 		[AjxCallback] 	Called when service is available.
 38  * @param errorCallback [AjxCallback] 	Called if service does not load.
 39  */
 40 ZmBrowserPlus.prototype.require =
 41 function(serviceObj, callback, errorCallback) {
 42 	if (!errorCallback) {
 43 		this._defaultErrorCallbackObj = this._defaultErrorCallbackObj || new AjxCallback(this, this._defaultErrorCallback);
 44 		errorCallback = this._defaultErrorCallbackObj; 
 45 	}
 46 
 47 	if (YAHOO.bp.isInitialized()) {
 48 		this._require(serviceObj, callback, errorCallback);
 49 	} else {
 50 		var self = this;
 51 		function onInit(result) {
 52 			self._onInit(result, serviceObj, callback, errorCallback);
 53 		}
 54 		YAHOO.bp.init({ locale: appCtxt.get(ZmSetting.LOCALE_NAME) }, onInit);
 55 	}
 56 };
 57 
 58 ZmBrowserPlus.prototype._onInit =
 59 function(result, serviceObj, callback, errorCallback) {
 60 	if (result.success) {
 61 		this._require(serviceObj, callback, errorCallback);
 62 		var requireArgs = {
 63 			services: [serviceObj],
 64 			progressCallback: function() {}
 65 		};
 66 		var self = this;
 67 		function requireCallback(result) {
 68 			self._onRequire(result, serviceObj, callback, errorCallback);
 69 		}
 70 		YAHOO.bp.require(requireArgs, requireCallback);
 71 	} else if (errorCallback) {
 72 		errorCallback.run(result);
 73 	}
 74 };
 75 
 76 ZmBrowserPlus.prototype._require =
 77 function(serviceObj, callback, errorCallback) {
 78 	var requireArgs = {
 79 		services: [serviceObj],
 80 		progressCallback: function() {}
 81 	};
 82 	var self = this;
 83 	function requireCallback(result) {
 84 		self._onRequire(result, serviceObj, callback, errorCallback);
 85 	}
 86 	YAHOO.bp.require(requireArgs, requireCallback);
 87 };
 88 
 89 ZmBrowserPlus.prototype._onRequire =
 90 function(result, serviceObj, callback, errorCallback) {
 91 	if (result.success) {
 92 		callback.run(YAHOO.bp[serviceObj.service]);
 93 	} else if (errorCallback) {
 94 		errorCallback.run(result);
 95 	}
 96 };
 97 
 98 ZmBrowserPlus.prototype._defaultErrorCallback =
 99 function(result) {
100 	DBG.println(AjxDebug.DBG1, "BrowserPlus error: " + (result ? (result.error + " - " + result.verboseError) : result));
101 	switch (result.error) {
102 		case "bpPlugin.platformDisabled": // BrowserPlus is disabled.
103 			appCtxt.setStatusMsg({ msg: ZmMsg.browserPlusDisabled, level: ZmStatusView.LEVEL_WARNING });
104 			break;
105 		case "core.permissionError": // User denied permission to use the service. 
106 			break;
107 		case "bp.notInstalled": // BrowserPlus isn't installed.
108 			var dialog = appCtxt.getMsgDialog();
109 			dialog.setMessage(ZmMsg.browserPlusMissing, DwtMessageDialog.CRITICAL_STYLE);
110 			dialog.popup();
111 			break;
112 	}
113 };
