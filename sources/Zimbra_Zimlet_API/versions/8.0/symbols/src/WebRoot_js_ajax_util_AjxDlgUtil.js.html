 21 AjxDlgUtil = {
 22 
 23 	cache : {},
 24 
 25 	getDialogLayout : function(name, url, msg) {
 26 		var time = new Date().getTime();
 27 
 28 		var txt, cache = AjxDlgUtil.cache;
 29 		if (cache[name]) {
 30 			txt = cache[name];
 31 		} else {
 32 			// WARNING: synchronous request!
 33 			// Also we don't treat errors at this point >-) so you better
 34 			// know what you're doing.
 35 			var res = AjxRpc.invoke(null, url + "?v=" + time, null, null, true, 5000);
 36 			cache[name] = txt = res.text;
 37 		}
 38 
 39 		var ids = {};
 40 
 41 		// get rid of the comments
 42 		txt = txt.replace(/<!--.*?-->/, "");
 43 
 44 		// replace $msg and $id fields
 45 		txt = txt.replace(/\$([a-zA-Z0-9_.]+)/g, function(str, p1) {
 46 			if (/^([^.]+)\.(.*)$/.test(p1)) {
 47 				var prefix = RegExp.$1;
 48 				var name = RegExp.$2;
 49 				switch (prefix) {
 50 				    case "id":
 51 					var id = ids[name];
 52 					if (!id)
 53 						id = ids[name] = Dwt.getNextId();
 54 					return id;
 55 				    case "msg":
 56 					return msg[name];
 57 				}
 58 			}
 59 			return str;
 60 		});
 61 
 62 		return { ids: ids, html: txt };
 63 	}
 64 
 65 };
