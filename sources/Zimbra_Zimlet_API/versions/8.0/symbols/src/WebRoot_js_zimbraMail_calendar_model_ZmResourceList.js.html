 16 /**
 17  * Create a new, empty calendar resources list.
 18  * @constructor
 19  * @class
 20  * This class represents a list of calendar resources. A calendar resource can be a
 21  * location or a piece of equipment. All calendar resource records are stored in the GAL.
 22  *
 23  * @author Conrad Damon
 24  *
 25  * @param {constant}	resType	the type of resources (location or equipment)
 26  * @param {ZmSearch}	search	the search that generated this list
 27  * 
 28  * @extends		ZmContactList
 29  * 
 30  * @see		ZmCalBaseItem
 31  */
 32 ZmResourceList = function(resType, search) {
 33 	ZmContactList.call(this, search, true, ZmItem.RESOURCE);
 34 
 35 	this.resType = resType;
 36 	
 37 	this._nameToResource = {};
 38 	this._emailToResource = {};
 39 	this._app = appCtxt.getApp(ZmApp.CALENDAR);
 40 };
 41 
 42 ZmResourceList.ATTRS =
 43 	[ZmResource.F_name, ZmResource.F_mail, ZmResource.F_type, ZmResource.F_locationName,
 44 	 ZmResource.F_capacity, ZmResource.F_contactMail, ZmResource.F_description];
 45 
 46 ZmResourceList.prototype = new ZmContactList;
 47 ZmResourceList.prototype.constructor = ZmResourceList;
 48 
 49 ZmResourceList.prototype.toString =
 50 function() {
 51 	return "ZmResourceList";
 52 };
 53 
 54 /**
 55  * Loads the list.
 56  * 
 57  * @param	{ZmBatchCommand}	batchCmd		the batch command
 58  */
 59 ZmResourceList.prototype.load =
 60 function(batchCmd) {
 61 	var conds = [];
 62 	var value = (this.resType == ZmCalBaseItem.LOCATION) ? ZmResource.ATTR_LOCATION : ZmResource.ATTR_EQUIPMENT;
 63 	conds.push({attr: ZmResource.F_type, op: "eq", value: value});
 64 	var params = {conds: conds, join: ZmSearch.JOIN_OR, attrs: ZmResourceList.ATTRS};
 65     if(batchCmd) {
 66         var search = new ZmSearch(params);        
 67 	    search.execute({callback: new AjxCallback(this, this._handleResponseLoad), batchCmd: batchCmd});
 68     }else{
 69         if (appCtxt.isOffline) {
 70             if (appCtxt.isZDOnline()) {
 71                 this.searchCalResources(params);
 72             }
 73         } else {
 74             this.searchCalResources(params);
 75         }
 76     }
 77 };
 78 
 79 /**
 80  * Searches the calendar resources.
 81  * 
 82  * @param	{Hash}	params		a hash of parameters
 83  */
 84 ZmResourceList.prototype.searchCalResources =
 85 function(params) {
 86     var soapDoc = AjxSoapDoc.create("SearchCalendarResourcesRequest", "urn:zimbraAccount");
 87     var method = soapDoc.getMethod();
 88     if (params.attrs) {
 89         AjxUtil.arrayRemove(params.attrs, "fullName");
 90         method.setAttribute("attrs", params.attrs.join(","));
 91     }
 92     var searchFilterEl = soapDoc.set("searchFilter");
 93     if (params.conds && params.conds.length) {
 94         var condsEl = soapDoc.set("conds", null, searchFilterEl);
 95         if (params.join == ZmSearch.JOIN_OR) {
 96             condsEl.setAttribute("or", 1);
 97         }
 98         for (var i = 0; i < params.conds.length; i++) {
 99             var cond = params.conds[i];
100             if (cond.attr=="fullName" && cond.op=="has") {
101                 var nameEl = soapDoc.set("name", cond.value);
102             } else {
103                 var condEl = soapDoc.set("cond", null, condsEl);
104                 condEl.setAttribute("attr", cond.attr);
105                 condEl.setAttribute("op", cond.op);
106                 condEl.setAttribute("value", cond.value);
107             }
108         }
109     }
110 
111     var response = appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:false,
112         timeout:params.timeout, noBusyOverlay:params.noBusyOverlay});
113     var result = new ZmCsfeResult(response, false);
114 
115     var search = new ZmSearch(params);
116     search.isCalResSearch = true;
117 
118     var searchResult = new ZmSearchResult(search);
119     searchResult.set(response.SearchCalendarResourcesResponse);
120     result.set(searchResult);
121     
122     this._handleResponseLoad(result);
123 };
124 
125 ZmResourceList.prototype._handleResponseLoad = 
126 function(result) {
127 	var resp = result.getResponse();
128 	this._vector = resp.getResults(ZmItem.RESOURCE).getVector();
129 	var a = this._vector.getArray();
130 	for (var i = 0; i < a.length; i++) {
131 		var resource = a[i];
132 		this._updateHashes(resource);
133 		this._idHash[resource.id] = resource;
134 	}
135 	//bug:16436 this._loaded changed to this.isLoaded 
136 	this.isLoaded = true;
137 	this._galAutocompleteEnabled = false;
138 };
139 
140 ZmResourceList.prototype._updateHashes =
141 function(resource) {
142 	this._app.updateResourceCache(resource);
143 	var name = resource.getFullName();
144 	if (name) {
145 		this._nameToResource[name.toLowerCase()] = resource;
146 	}
147 	var email = resource.getEmail();
148 	if (email) {
149 		this._emailToResource[email.toLowerCase()] = resource;
150 	}
151 };

159 /**
160  * Gets the resource with the given name, if any. Canonical list only.
161  * Since names aren't guaranteed to be unique, this returns the last resource
162  * with the given name.
163  *
164  * @param {String}	name	the resource name
165  * @return	{ZmResource}	the resource or <code>null</code> if not found
166  */
167 ZmResourceList.prototype.getResourceByName = 
168 function(name) {
169 	if (!name || !this.isCanonical) return null;
170 
171 	return this._nameToResource[name.toLowerCase()];
172 };
173 
174 /**
175  *Gets the resource with the given address, if any. Canonical list only.
176  *
177  * @param {String}	address	an email address
178  * @return	{ZmResource}	the resource or <code>null</code> if not found
179  */
180 ZmResourceList.prototype.getResourceByEmail = 
181 function(address) {
182 	if (!address || !this.isCanonical) return null;
183 
184 	return this._emailToResource[address.toLowerCase()];
185 };
