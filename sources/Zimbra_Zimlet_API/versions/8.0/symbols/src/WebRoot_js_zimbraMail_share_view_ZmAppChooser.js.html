 23 /**
 24  * @class
 25  * This class represents a zimbra application chooser. The chooser is the "tab application" toolbar shown
 26  * in the Zimbra Web Client. The toolbar buttons are represented as "tabs".
 27  * 
 28  * @param	{Hash}	params		a hash of parameters
 29  * 
 30  * @extends	ZmToolBar
 31  */
 32 ZmAppChooser = function(params) {
 33 
 34 	params.className = params.className || "ZmAppChooser";
 35 	params.width = appCtxt.getSkinHint("appChooser", "fullWidth") ? "100%" : null;
 36 
 37 	ZmToolBar.call(this, params);
 38 
 39     Dwt.setLocation(this.getHtmlElement(), Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
 40 
 41 	this.setScrollStyle(Dwt.CLIP);
 42 
 43 	this._buttonListener = new AjxListener(this, this._handleButton);
 44     this._initOverFlowTabs();
 45 	var buttons = params.buttons;
 46 	for (var i = 0; i < buttons.length; i++) {
 47 		var id = buttons[i];
 48 		if (id == ZmAppChooser.SPACER) {
 49 			this.addSpacer(ZmAppChooser.SPACER_HEIGHT);
 50 		} else {
 51 			this._createButton(id);
 52 		}
 53 	}
 54 
 55 	this._createPrecedenceList();
 56 	this._inited = true;
 57 };
 58 
 59 ZmAppChooser.prototype = new ZmToolBar;
 60 ZmAppChooser.prototype.constructor = ZmAppChooser;
 61 
 62 ZmAppChooser.prototype._initOverFlowTabs =
 63 function(){
 64     this._leftOverflow = document.getElementById("moreTabsLeftContainer");
 65     this._rightOverflow = document.getElementById("moreTabsRightContainer");
 66     if (this._leftOverflow) {
 67 		this._leftOverflow.onclick = this._showLeftTab.bind(this);
 68 	}
 69 	if (this._rightOverFlow) {
 70     	this._rightOverflow.onclick = this._showRightTab.bind(this);
 71 	}
 72     this._leftBtnIndex = -1;
 73     this._deletedButtons = [];
 74 };
 75 
 76 ZmAppChooser.prototype._showLeftTab =
 77 function(){
 78     var items = this.getItems();
 79     if(this._leftBtnIndex > -1){
 80         items[this._leftBtnIndex].setVisible(true);
 81         this._leftBtnIndex--;
 82     }
 83     this._checkArrowVisibility();
 84 };
 85 
 86 ZmAppChooser.prototype._showRightTab =
 87 function(){
 88     var items = this.getItems();
 89     this._leftBtnIndex++;
 90     items[this._leftBtnIndex].setVisible(false);
 91     this._checkArrowVisibility();
 92 };
 93 
 94 ZmAppChooser.prototype._showTab =
 95 function(id, ev){
 96     var button = this._buttons[id];
 97     this._moreTabsBtn.getMenu().popdown();
 98     if (!button) return;
 99     if (!button.getVisible()){ // Left side
100         var found = false;
101         for (var index in this._buttons){
102             if (!found && this._buttons[index].getHTMLElId() == button.getHTMLElId())
103               found = true;
104             else if (this._buttons[index].getVisible())
105                 break;
106             if (found){
107                 this._buttons[index].setVisible(true)
108                 this._leftBtnIndex--;
109             }
110         }
111     }else { // Right side
112         while(this._isTabOverflow(button.getHtmlElement())){
113             this._showRightTab();
114         }
115     }
116     this._checkArrowVisibility();
117     appCtxt.getAppController()._appButtonListener(ev);
118 
119 
120 };

122 ZmAppChooser.prototype._attachMoreTabMenuItems =
123 function(menu){
124 
125     for (var deletedIndex=0; deletedIndex < this._deletedButtons.length; deletedIndex++){
126         var mi = menu.getItemById("_menuItemId", this._deletedButtons[deletedIndex] + "_menu");
127         if (mi) {
128             menu.removeChild(mi);
129             mi.dispose();
130         }
131     }
132 
133     this._deletedButtons = [];
134     for(var index in this._buttons){
135         var item = menu.getItemById("_menuItemId", index + "_menu");
136         if (item){
137             if (item.getText() != this._buttons[index].getText()){
138                 item.setText(this._buttons[index].getText());
139             }
140         } else {
141             var mi = new DwtMenuItem({parent:menu, style:DwtMenuItem.CASCADE_STYLE, id: index + "_menu"});
142             mi.setData("_menuItemId", index + "_menu" );
143             mi.setData(Dwt.KEY_ID, index);
144             mi.addSelectionListener(this._showTab.bind(this, index));
145             mi.setText(this._buttons[index].getText());
146         }
147     }
148 
149     if(menu.getHtmlElement().style.width == "0px"){
150         this._moreTabsBtn.popup();
151     }
152 };
153 
154 ZmAppChooser.prototype._showOverflowTabsMenu =
155 function(){
156     var menu = new DwtMenu({parent:this._moreTabsBtn});
157     menu.addPopupListener(new AjxListener(this, this._attachMoreTabMenuItems,[menu]));
158     return menu;
159 };

162 ZmAppChooser.prototype._checkArrowVisibility =
163 function(){
164     var items = this.getItems();
165     if (this._leftBtnIndex < 0)
166         this._setArrowVisibility(this._leftOverflow, "none");
167     else
168         this._setArrowVisibility(this._leftOverflow, "");
169 
170     if (!this._isTabOverflow(items[items.length -1].getHtmlElement())){
171         this._setArrowVisibility(this._rightOverflow, "none");
172     }else{
173         this._setArrowVisibility(this._rightOverflow, "");
174     }
175     this._adjustWidth();
176 };
177 
178 
179 
180 ZmAppChooser.prototype._setArrowVisibility =
181 function(element, option){
182 		if (!element) return;
183     element.style.display = option|| "";
184     var display = ((this._leftOverflow && this._leftOverflow.style.display == "none") && 
185 					(this._rightOverflow && this._rightOverflow.style.display == "none")) ? "none" : "";
186 	var moreTabsMenu = document.getElementById("moreTabsMenu");
187 	if (moreTabsMenu) {
188     	moreTabsMenu.style.display = display;
189 	}
190     if (display != "none" && !this._moreTabsBtn && moreTabsMenu ){
191         var containerEl = moreTabsMenu;
192         var button = new DwtToolBarButton({parent:DwtShell.getShell(window), id: "moreTabsMenuBtn", style:"background:none no-repeat scroll 0 0 transparent; border: none"});
193         button.setToolTipContent(ZmMsg.more, true);
194         button.setText("");
195         button.reparentHtmlElement(moreTabsMenu);
196         button.setMenu(new AjxListener(this, this._showOverflowTabsMenu));
197         this._moreTabsBtn =  button;
198     }
199 };

252 ZmAppChooser.prototype._checkTabOverflowAdd =
253 function(button) {
254     var display = "none";
255     if (this._isTabOverflow(button)){
256         display = "";
257     }
258     this._setArrowVisibility(this._rightOverflow, display);
259     this._adjustWidth();
260 };
261 
262 ZmAppChooser.prototype._isTabOverflow =
263 function(tab){
264         var tabPos = tab.offsetLeft + tab.clientWidth + 30;
265         if (!this._refElement){
266             this._refElement = document.getElementById(this._refElementId);
267         }
268         var container = this._refElement && this._refElement.parentNode;
269 
270         if (!container) return false;
271         var offsetWidth = container.offsetWidth;
272         return (offsetWidth < tabPos);
273 };
274 
275 ZmAppChooser.prototype._adjustWidth =
276 function(){
277     var container = this._refElement && this._refElement.parentNode;
278 	if (container && container.offsetWidth >= 30) {
279     	this._refElement.style.maxWidth = this._refElement.style.width =  (container.offsetWidth - 30);
280     	this._refElement.style.overflow = "hidden";
281 	}
282 
283 };

285 ZmAppChooser.prototype._checkTabOverflowDelete =
286 function(index){
287     var items = this.getItems();
288     if(this._isTabOverflow(items[items.length - 1])){
289       return;
290     }
291     this._showLeftTab();
292 }

295 /**
296  * Adds a button to the toolbar.
297  * 
298  * @param	{String}	id		the button id
299  * @param	{Hash}		params		a hash of parameters
300  * @param	{String}	params.text			the text
301  * @param	{String}	params.image		the image
302  * @param	{int}	params.index		the index
303  * @param	{String}	params.tooltip		the tool top
304  * @param	{String}	params.textPrecedence	the image precedence
305  * @param	{String}	params.imagePrecedence		the image precedence
306  * 
307  * @return	{ZmAppButton}		the newly created button
308  */
309 ZmAppChooser.prototype.addButton =
310 function(id, params) {
311 
312 	var buttonParams = {parent:this, id:ZmId.getButtonId(ZmId.APP, id), text:params.text,
313 						image:params.image, index:params.index};
314     buttonParams.style = params.style ? params.style : DwtLabel.IMAGE_LEFT;
315     var button = new ZmAppButton(buttonParams);
316 	button.setToolTipContent(params.tooltip, true);
317 	button.textPrecedence = params.textPrecedence;
318 	button.imagePrecedence = params.imagePrecedence;
319 	button.setData(Dwt.KEY_ID, id);
320 	button.addSelectionListener(this._buttonListener);
321 	this._buttons[id] = button;
322     this._checkTabOverflowAdd(button.getHtmlElement());
323 	return button;
324 };
325 
326 /**
327  * Removes a button.
328  * 
329  * @param	{String}	id		the id of the button to remove
330  */
331 ZmAppChooser.prototype.removeButton =
332 function(id) {
333 	var button = this._buttons[id];
334 	if (button) {
335         var index = this.__getButtonIndex(id);
336 		button.dispose();
337 		this._buttons[id] = null;
338 		delete this._buttons[id];
339 		if (index != null) {
340 			this._removeItem(this._items[index]);
341 		}
342         if (this._moreTabsBtn &&
343             this._moreTabsBtn.getMenu() &&
344             this._moreTabsBtn.getMenu().getItemCount() > 0){
345             this._deletedButtons.push(id);
346         }
347         this._checkTabOverflowDelete(index);
348 	}
349 };

351 /**
352  * Replaces a button.
353  * 
354  * @param	{String}	oldId		the old button id
355  * @param	{String}	newId		the new button id
356  * @param	{Hash}		params		a hash of parameters
357  * @param	{String}	params.text		the text
358  * @param	{String}	params.image	the image
359  * @param	{int}	params.index		the index
360  * @param	{String}	params.tooltip			the tool tip
361  * @param	{String}	params.textPrecedence	the text display precedence
362  * @param	{String}	params.imagePrecedence	the image display precedence
363  * 
364  * @return	{ZmAppButton}		the newly created button
365  */
366 ZmAppChooser.prototype.replaceButton =
367 function(oldId, newId, params) {
368 	if (!this._buttons[oldId]) { return null; }
369 	params.index = this.__getButtonIndex(oldId);
370 	this.removeButton(oldId);
371 	return this.addButton(newId, params);
372 };
373 
374 ZmAppChooser.prototype.getButton =
375 function(id) {
376 	return this._buttons[id];
377 };

379 /**
380  * Sets the specified button as selected.
381  * 
382  * @param	{String}	id		the button id
383  */
384 ZmAppChooser.prototype.setSelected =
385 function(id) {
386 	var oldBtn = this._buttons[this._selectedId];
387 	if (this._selectedId && oldBtn) {
388         this.__markPrevNext(this._selectedId, false);
389 		oldBtn.setSelected(false);
390 		oldBtn._noFocus = false;
391     }
392 
393 	var newBtn = this._buttons[id];
394 	if (newBtn) {
395 		newBtn.setSelected(true);
396 
397 		if (newBtn._toggleText != null && newBtn._toggleText != "") {
398 			// hide text for previously selected button first
399 			if (oldBtn) {
400 				oldBtn._toggleText = (oldBtn._toggleText != null && oldBtn._toggleText != "")
401 					? oldBtn._toggleText : oldBtn.getText();
402 				oldBtn.setText("");
403 			}
404 
405 			// reset original text for  newly selected button
406 			newBtn.setText(newBtn._toggleText);
407 			newBtn._toggleText = null;
408 		}
409 
410 		newBtn._noFocus = true;
411 	}
412 
413 	this._selectedId = id;
414 };

420 function(id) {
421 	this.addButton(id, {text:ZmMsg[ZmApp.NAME[id]] || ZmApp.NAME[id], image:ZmApp.ICON[id], tooltip:ZmMsg[ZmApp.CHOOSER_TOOLTIP[id]],
422 						textPrecedence:ZmApp.TEXT_PRECEDENCE[id], imagePrecedence:ZmApp.IMAGE_PRECEDENCE[id]});
423 };
