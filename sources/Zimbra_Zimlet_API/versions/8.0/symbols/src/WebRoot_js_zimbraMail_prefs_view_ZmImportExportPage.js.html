 16 /**
 17  * Creates the import/export page.
 18  * @class
 19  * This class represents the import/export page.
 20  * 
 21  * @param {DwtControl}	parent			the containing widget
 22  * @param {Object}	section			the page
 23  * @param {ZmPrefController}	controller		the prefs controller
 24  * 
 25  * @extends	ZmPreferencesPage
 26  * 
 27  * @private
 28  */
 29 ZmImportExportPage = function(parent, section, controller) {
 30 	ZmPreferencesPage.apply(this, arguments);
 31 };
 32 
 33 ZmImportExportPage.prototype = new ZmPreferencesPage;
 34 ZmImportExportPage.prototype.constructor = ZmImportExportPage;
 35 
 36 ZmImportExportPage.prototype.toString =
 37 function () {
 38     return "ZmImportExportPage";
 39 };
 40 
 41 //
 42 // ZmPreferencesPage methods
 43 //
 44 
 45 ZmImportExportPage.prototype.reset =
 46 function(useDefaults) {
 47 	ZmPreferencesPage.prototype.reset.apply(this, arguments);
 48 	var button = this.getFormObject("IMPORT_BUTTON");
 49 	if (button) {
 50 		button.setEnabled(true);
 51 	}
 52 };
 53 
 54 ZmImportExportPage.prototype.hasResetButton =
 55 function() {
 56 	return false;
 57 };
 58 
 59 //
 60 // Protected methods
 61 //
 62 
 63 ZmImportExportPage.prototype._setupCustom = function(id, setup, value) {
 64 	if (id == "EXPORT_FOLDER") {
 65 		var view = new ZmExportView({parent:this});
 66 		this.setFormObject(id, view);
 67 		return view;
 68 	}
 69 	if (id == "EXPORT_BUTTON") {
 70 		var button = new DwtButton({parent:this, id: id});
 71 		button.setText(setup.displayName);
 72 		button.addSelectionListener(new AjxListener(this, this._handleExportButton));
 73 		this.setFormObject(id, button);
 74 		return button;
 75 	}
 76 	if (id == "IMPORT_FOLDER") {
 77 		var view = new ZmImportView({parent:this});
 78 		this.setFormObject(id, view);
 79 		return view;
 80 	}
 81 	if (id == "IMPORT_BUTTON") {
 82 		var button = new DwtButton({parent:this, id: id});
 83 		button.setText(setup.displayName);
 84 		button.addSelectionListener(new AjxListener(this, this._handleImportButton));
 85 		this.setFormObject(id, button);
 86 		return button;
 87 	}
 88 	return ZmPreferencesPage.prototype._setupCustom.apply(this, arguments);
 89 };
 90 
 91 // handlers
 92 
 93 ZmImportExportPage.prototype._handleImportButton = function() {
 94 	var button = this.getFormObject("IMPORT_BUTTON");
 95 	if (button) {
 96 		button.setEnabled(false);
 97 	}
 98 
 99 	// get import params
100 	var importView = this.getFormObject("IMPORT_FOLDER");
101 	var params = {};
102     params = importView && importView.getParams();
103 	params.callback = params.errorCallback = new AjxCallback(this, this._handleImportComplete);
104 
105 	// import
106 	var controller = appCtxt.getImportExportController();
107 	if (controller.importData(params)) {
108 		var params = {
109 			msg:	ZmMsg.importStarted,
110 			level:	ZmStatusView.LEVEL_INFO
111 		};
112 		appCtxt.setStatusMsg(params);
113 	}
114 	else if (button) {
115 		button.setEnabled(true);
116 	}
117 };
118 
119 ZmImportExportPage.prototype._handleExportButton = function() {
120 	// get export params
121 	var exportView = this.getFormObject("EXPORT_FOLDER");
122 	var params = exportView.getParams();
123 
124 	// export
125 	var controller = appCtxt.getImportExportController();
126 	controller.exportData(params);
127 };
128 
129 ZmImportExportPage.prototype._handleImportComplete = function() {
130 	var button = this.getFormObject("IMPORT_BUTTON");
131 	if (button) {
132 		button.setEnabled(true);
133 	}
134 };
