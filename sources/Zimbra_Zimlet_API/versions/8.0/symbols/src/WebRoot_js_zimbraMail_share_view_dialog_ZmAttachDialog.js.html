 20 /**
 21  * Creates an attachment dialog.
 22  * @class
 23  * This class represents an attachment dialog.
 24  * 
 25  * @param	{DwtControl}	shell		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtDialog
 29  */
 30 ZmAttachDialog = function(shell, className) {
 31 
 32 	className = className || "ZmAttachDialog";
 33 	DwtDialog.call(this, {parent:shell, className:className, title:ZmMsg.attachFile});
 34 
 35 	// Initialize
 36 	this._createBaseHtml();
 37 
 38 	// Ok and Cancel Actions
 39 	this._defaultCancelCallback = new AjxCallback(this, this._defaultCancelListener);
 40 	this._cancelListener = null;
 41 
 42 	this._defaultOkCallback = new AjxCallback(this, this._defaultOkListener);
 43 	this._okListener = null;
 44 
 45 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, function() {
 46 		this._cancelButtonListener();
 47 	}));
 48 
 49 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, function() {
 50 		this._okButtonListener();
 51 	}));
 52 
 53 
 54 	var okButton = this.getButton(DwtDialog.OK_BUTTON);
 55 	okButton.setText(ZmMsg.attach);
 56 
 57 };

231 /**
232  * Uploads the attachments.
233  * 
234  * @param	{AjxCallback}		callback		the callback
235  * @param	{Object}			uploadForm		the upload form
236  */
237 ZmAttachDialog.prototype.upload =
238 function(callback, uploadForm) {
239 	if (!callback) {
240 		callback = false;
241 	}
242 	this.setButtonEnabled(DwtDialog.OK_BUTTON, false);
243 	this.setButtonEnabled(DwtDialog.CANCEL_BUTTON, true);
244 	this.setFooter(ZmMsg.attachingFiles);
245 	this._cancelUpload = false;
246 	this._processUpload(callback, uploadForm);
247 };
248 
249 ZmAttachDialog.prototype._processUpload =
250 function(callback, uploadForm) {
251 	var ajxCallback = new AjxCallback(this, this._uploadDoneCallback, [callback]);
252 	var um = appCtxt.getUploadManager();
253 	window._uploadManager = um;
254 
255 	try {
256 		um.execute(ajxCallback, uploadForm);
257 	} catch (ex) {
258 		ajxCallback.run();
259 	}
260 };
261 
262 ZmAttachDialog.prototype._uploadDoneCallback =
263 function(callback, status, attId) {
264 	if (this._cancelUpload) { return; }
265 
266 	this.setButtonEnabled(DwtDialog.CANCEL_BUTTON, true);
267 
268 	if (status == AjxPost.SC_OK) {
269 		this.setFooter(ZmMsg.attachingFilesDone);
270 		if (callback) {
271 			callback.run(status, attId);
272 		}
273 	} else if (status == AjxPost.SC_UNAUTHORIZED) {
274 		// auth failed during att upload - let user relogin, continue with compose action
275 		var ex = new AjxException("401 response during attachment upload", ZmCsfeException.SVC_AUTH_EXPIRED);
276 		appCtxt.getAppController()._handleException(ex, {continueCallback:callback});
277 	} else {
278 		// bug fix #2131 - handle errors during attachment upload.
279 		var msg = AjxMessageFormat.format(ZmMsg.errorAttachment, (status || AjxPost.SC_NO_CONTENT));
280 
281 		switch (status) {
282 			// add other error codes/message here as necessary
283 			case AjxPost.SC_REQUEST_ENTITY_TOO_LARGE:	msg += " " + ZmMsg.errorAttachmentTooBig + "<br><br>"; break;
284 			default:									msg += " "; break;
285 		}
286 		var dialog = appCtxt.getMsgDialog();
287 		dialog.setMessage(msg, DwtMessageDialog.CRITICAL_STYLE);
288 		dialog.popup();
289 
290 		this.setFooter(ZmMsg.attachingFilesError);
291 	}
292 
293 	this.setButtonEnabled(DwtDialog.OK_BUTTON, true);
294 };

452 // Attachments
453 ZmMyComputerTabViewPage.prototype._addAttachmentField =
454 function() {
455 	if (this._attachCount >= ZmMyComputerTabViewPage.MAX_NO_ATTACHMENTS) { return; }
456 
457 	this._attachCount++;
458 
459 	var row = this.attachmentTable.insertRow(-1);
460 	var cell = row.insertCell(-1);
461 	var fieldId = Dwt.getNextId();
462 
463 	var subs = {
464 		id: fieldId,
465 		uploadName: ZmMyComputerTabViewPage.UPLOAD_FIELD_NAME
466 	};
467 	cell.innerHTML = AjxTemplate.expand("share.Dialogs#ZmAttachDialog-MyComputerTab-AddAttachment", subs);
468 
469 	var removeEl = document.getElementById(fieldId+"_remove");   
470 	removeEl.onclick = AjxCallback.simpleClosure(this._removeAttachmentField, this, row);
471 
472     var inputId = fieldId+"_input";
473 	if (this._focusElId == -1) {
474 		this._focusElId = inputId;
475 	}    
476     var inputEl = document.getElementById(inputId);
477     var sizeEl = document.getElementById(fieldId+"_size");
478 
479     //HTML5
480     if(ZmAttachDialog.supportsHTML5){
481         Dwt.setHandler(inputEl, "onchange", AjxCallback.simpleClosure(this._handleFileSize, this, inputEl, sizeEl));
482     }
483 
484 	// trap key presses in IE for input field so we can ignore ENTER key (bug 961)
485 	if (AjxEnv.isIE) {
486 		inputEl.onkeydown = AjxCallback.simpleClosure(this._handleKeys, this);
487 	}
488 };

490 ZmMyComputerTabViewPage.prototype._handleFileSize =
491 function(inputEl, sizeEl){
492 
493     var files = inputEl.files;
494     if(!files) return;
495 
496     var sizeStr = [], className, totalSize =0;
497     for(var i=0; i<files.length;i++){
498         var file = files[i];
499         var size = file.size || file.fileSize /*Safari*/;
500         if(size > appCtxt.get(ZmSetting.ATTACHMENT_SIZE_LIMIT))
501             className = "RedC";
502         totalSize += size;
503     }
504 
505     if(sizeEl) {
506         sizeEl.innerHTML = "  ("+AjxUtil.formatSize(totalSize, true)+")";
507         if(className)
508             Dwt.addClass(sizeEl, "RedC");
509         else
510             Dwt.delClass(sizeEl, "RedC");
511     }
512 };

536 ZmMyComputerTabViewPage.prototype.gotAttachments =
537 function() {
538 	var atts = document.getElementsByName(ZmMyComputerTabViewPage.UPLOAD_FIELD_NAME);
539 
540 	for (var i = 0; i < atts.length; i++)
541 		if (atts[i].value.length) {

543 		}

547 ZmMyComputerTabViewPage.prototype.resetAttachments =
548 function() {
549 	// CleanUp
550 	this._cleanTable(this.attachmentTable);
551 	this._attachCount = 0;
552 	if (ZmMyComputerTabViewPage.SHOW_NO_ATTACHMENTS > ZmMyComputerTabViewPage.MAX_NO_ATTACHMENTS) {
553 		ZmMyComputerTabViewPage.SHOW_NO_ATTACHMENTS = ZmMyComputerTabViewPage.MAX_NO_ATTACHMENTS;
554 	}
555 
556 	// Re-initialize UI
557 	this._focusElId = -1;
558 	var row = this.attachmentTable.insertRow(-1);
559 	var cell = row.insertCell(-1);
560 	cell.appendChild(document.createElement("br"));
561 	cell.appendChild(document.createElement("br"));
562 
563 	for (var i = 0; i < ZmMyComputerTabViewPage.SHOW_NO_ATTACHMENTS; i++) {
564 		this._addAttachmentField();
565 	}
566 	delete i;
567 };

590 ZmMyComputerTabViewPage.prototype._validateFileSize =
591 function(){
592 
593     var atts = document.getElementsByName(ZmMyComputerTabViewPage.UPLOAD_FIELD_NAME);
594     var file, size;
595 	for (var i = 0; i < atts.length; i++){
596         file = atts[i].files;
597         if(!file || file.length == 0) continue;
598         for(var j=0; j<file.length;j++){
599             var f = file[j];
600             size = f.size || f.fileSize /*Safari*/;
601             if(size > appCtxt.get(ZmSetting.ATTACHMENT_SIZE_LIMIT)){
602                 return false;
603             }
604         }
605     }
606 	return true;
607 };

609 ZmMyComputerTabViewPage.prototype.validate =
610 function(){
611     var status, errorMsg;
612     if(ZmAttachDialog.supportsHTML5){
613         status = this._validateFileSize();
614         errorMsg = AjxMessageFormat.format(ZmMsg.attachmentSizeError, AjxUtil.formatSize(appCtxt.get(ZmSetting.ATTACHMENT_SIZE_LIMIT)));
615     }else{
616         status = true;
617     }
618 
619     return {status: status, error:errorMsg};
620 };
