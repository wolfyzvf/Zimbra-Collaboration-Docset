1 /**
  2  * Date group divides messages into the following sections:
  3  * Today
  4  * Yesterday
  5  * Day of week -- not today or yesterday, but still within this week
  6  * Last Week
  7  * Two Weeks Ago
  8  * Three Weeks Ago
  9  * Earlier this Month
 10  * Last Month
 11  * Older
 12  *
 13  */
 14 ZmMailListDateGroup = function(){
 15     this.id = ZmId.GROUPBY_DATE;
 16 	this.field = ZmItem.F_DATE;
 17 	this._weekStartDay = appCtxt.get(ZmSetting.CAL_FIRST_DAY_OF_WEEK) || 0;
 18 	var dayOfWeek = this._getToday().getDay();
 19 	this._keys = this._sortKeys(dayOfWeek, false);
 20     ZmMailListGroup.call(this);
 21 };

 74 /**
 75  *  returns HTML string for all sections.
 76  *  @param {boolean} sortAsc    true/false if sort ascending
 77  *  @return {String} HTML for all sections including section header
 78  * @param sortAsc
 79  */
 80 ZmMailListDateGroup.prototype.getAllSections =
 81 function(sortAsc) {
 82     var htmlArr = [];
 83 	var dayOfWeek = this._getToday().getDay();
 84     var keys = sortAsc ? this._sortKeys(dayOfWeek, sortAsc) : this._keys; //keys have already been sorted desc
 85     for (var i=0; i<keys.length; i++) {
 86         if (this._section[keys[i]].length > 0) {
 87             htmlArr.push(this.getSectionHeader(ZmMailListDateGroup.SECTION_TITLE[keys[i]]));
 88             htmlArr.push(this._section[keys[i]].join(""));
 89         }
 90         else if (this._showEmptySectionHeader) {
 91             htmlArr.push(this.getSectionHeader(ZmMailListDateGroup.SECTION_TITLE[keys[i]]));
 92         }
 93     }
 94     return htmlArr.join("");
 95 };

 97 /**
 98  * Adds item to section
 99  * @param {ZmMailMsg} msg   mail message
100  * @param {String} item  HTML to add to section
101  * @return {String} section returns section if successfully added, else returns null
102  */
103 ZmMailListDateGroup.prototype.addMsgToSection =
104 function(msg, item) {
105    for (var i=0; i<this._keys.length; i++) {
106        if (this.isMsgInSection(this._keys[i], msg)) {
107         this._section[this._keys[i]].push(item);
108         return this._keys[i];
109        }
110    }
111 };

113 /**
114  * Determines if message is in group
115  * @param {String} section ID of section
116  * @param {ZmMailMsg} msg
117  * @return {boolean} true/false
118  */
119 ZmMailListDateGroup.prototype.isMsgInSection =
120 function(section, msg) {
121 
122    switch(section){
123        case ZmMailListDateGroup.TODAY:
124          return this._isToday(msg);
125 
126        case ZmMailListDateGroup.YESTERDAY:
127         return this._isYesterday(msg);
128 
129        case ZmMailListDateGroup.MONDAY:
130         return this._isDayOfWeek(msg, AjxDateUtil.MONDAY);
131 
132        case ZmMailListDateGroup.TUESDAY:
133         return this._isDayOfWeek(msg, AjxDateUtil.TUESDAY);
134 
135        case ZmMailListDateGroup.WEDNESDAY:
136         return this._isDayOfWeek(msg, AjxDateUtil.WEDNESDAY);
137 
138        case ZmMailListDateGroup.THURSDAY:
139         return this._isDayOfWeek(msg, AjxDateUtil.THURSDAY);
140 
141        case ZmMailListDateGroup.FRIDAY:
142         return this._isDayOfWeek(msg, AjxDateUtil.FRIDAY);
143 
144        case ZmMailListDateGroup.SATURDAY:
145         return this._isDayOfWeek(msg, AjxDateUtil.SATURDAY);
146 
147        case ZmMailListDateGroup.SUNDAY:
148         return this._isDayOfWeek(msg, AjxDateUtil.SUNDAY);
149 
150        case ZmMailListDateGroup.LAST_WEEK:
151         return this._isWeeksAgo(msg, ZmMailListDateGroup.LAST_WEEK, ZmMailListDateGroup.YESTERDAY);
152 
153        case ZmMailListDateGroup.TWO_WEEKS_AGO:
154         return this._isWeeksAgo(msg, ZmMailListDateGroup.TWO_WEEKS_AGO, ZmMailListDateGroup.LAST_WEEK);
155 
156        case ZmMailListDateGroup.THREE_WEEKS_AGO:
157         return this._isWeeksAgo(msg, ZmMailListDateGroup.THREE_WEEKS_AGO, ZmMailListDateGroup.TWO_WEEKS_AGO);
158 
159        case ZmMailListDateGroup.EARLIER_THIS_MONTH:
160         return this._isEarlierThisMonth(msg);
161 
162        case ZmMailListDateGroup.LAST_MONTH:
163         return this._isLastMonth(msg);
164 
165        case ZmMailListDateGroup.OLDER:
166         return this._isOlder(msg);
167 
168        default:
169         return false;
170    }
171 };

206 /**
207  * determines if mail message was received today
208  * @param msg {ZmMailMsg} mail msg
209  * @return {boolean}
210  */
211 ZmMailListDateGroup.prototype._isToday =
212 function(msg){
213     if(msg){
214         var today = this._getToday();
215         var d = this._getDateFromMsg(msg, true);
216         if (d) {
217             return today.getTime() == d.getTime();
218         }
219     }
220     return false;
221 };

223 /**
224  * determines if msg was received yesterday
225  * @param msg {ZmMailMsg} mail msg
226  * @return {boolean}
227  */
228 ZmMailListDateGroup.prototype._isYesterday =
229 function(msg) {
230     if (msg) {
231         var today = this._getToday();
232         var yesterday = new Date();
233         yesterday.setTime(today.getTime() - AjxDateUtil.MSEC_PER_DAY);
234         var d = this._getDateFromMsg(msg, true);
235         if (d) {
236             return yesterday.getTime() == d.getTime();
237         }
238     }
239 
240     return false;
241 };

243 /**
244  * message is this week, but not today or yesterday
245  * @param msg {ZmMailMsg} mail msg
246  * @param dayOfWeek {integer} the day of the week to check against (e.g. Monday)
247  * @return {boolean}
248  */
249 ZmMailListDateGroup.prototype._isDayOfWeek =
250 function(msg, dayOfWeek) {
251     if (msg) {
252         var today = this._getToday();
253         var thisWeek = AjxDateUtil.getWeekNumber(today);
254         var thisYear = today.getYear();
255 
256         var d = this._getDateFromMsg(msg, true);
257         if (d) {
258             return d.getDay() == dayOfWeek && AjxDateUtil.getWeekNumber(d) == thisWeek &&
259                    !this._isYesterday(msg) && !this._isToday(msg) && thisYear == d.getYear();
260         }
261     }
262     return false;
263 };

265 /**
266  * Determines if msg is from X number of weeks ago.
267  * @param {ZmMailMsg} msg the mail message to evaluate
268  * @param {String} minGroup Group oldest in date  (e.g. three weeks ago)
269  * @param {String} maxGroup Group newest in date  (e.g. two weeks ago)
270  */
271 ZmMailListDateGroup.prototype._isWeeksAgo =
272 function(msg, minGroup, maxGroup) {
273 	if (msg) {
274 		var today = this._getToday();
275 		var max = today.getTime() - ZmMailListDateGroup.TIME[maxGroup];
276 		var min = today.getTime() - ZmMailListDateGroup.TIME[minGroup];
277 		var d = this._getDateFromMsg(msg, true);
278 		if (d) {
279 			return d.getTime() >= min && d.getTime() < max;
280 		}
281 	}
282 
283 	return false;
284 };

286 /**
287  * message is earlier this month and also more than 3 weeks ago
288  * @param msg {ZmMailMsg} mail msg
289  * @return {boolbean}
290  */
291 ZmMailListDateGroup.prototype._isEarlierThisMonth =
292 function(msg) {
293     if (msg) {
294         var today = this._getToday();
295         var threeWeeksAgo = today.getTime() - ZmMailListDateGroup.TIME[ZmMailListDateGroup.THREE_WEEKS_AGO];
296         var thisMonth = today.getMonth();
297         var thisYear = today.getYear();
298         var d = this._getDateFromMsg(msg, true);
299         if (d) {
300             return d.getTime() < threeWeeksAgo && (d.getYear() == thisYear && d.getMonth() == thisMonth);
301         }
302     }
303     return false;
304 };
305 
306 /**
307  * message is last month and also more than 3 weeks ago from today
308  * @param msg {ZmMailMsg} mail message
309  * @return {boolean}
310  */
311 ZmMailListDateGroup.prototype._isLastMonth =
312 function(msg) {
313     if (msg) {
314         var today = this._getToday();
315 	    var threeWeeksAgo = today.getTime() - ZmMailListDateGroup.TIME[ZmMailListDateGroup.THREE_WEEKS_AGO];
316         var thisMonth = today.getMonth();
317         var thisYear = today.getYear();
318         var lastMonth = this._calculateLastMonth(thisMonth);
319         var d = this._getDateFromMsg(msg, true);
320         if (d) {
321             if(d.getMonth() != thisMonth) {
322                 if (d.getYear() == thisYear) {
323                     return d.getMonth() == lastMonth && d.getTime() < threeWeeksAgo;
324                 }
325 				else if (d.getYear() == (thisYear -1) && thisMonth == 0 && lastMonth == 11) {
326 					//handle the january, december case
327 					return d.getTime() < threeWeeksAgo;	
328 				}
329             }
330         }
331 
332     }
333     return false;
334 };

336 /**
337  * message is more than a month old
338  * @param msg {ZmMailMsg} mail msg
339  * @return {boolean}
340  */
341 ZmMailListDateGroup.prototype._isOlder =
342 function(msg) {
343     if (msg) {
344         var today = this._getToday();
345         var threeWeeksAgo = today.getTime() - ZmMailListDateGroup.TIME[ZmMailListDateGroup.THREE_WEEKS_AGO];
346         var d = this._getDateFromMsg(msg, true);
347         if (d) {
348             return d.getTime() < threeWeeksAgo && !this._isEarlierThisMonth(msg) && !this._isLastMonth(msg);
349         }
350     }
351     return false;
352 };

354 ZmMailListDateGroup.prototype._getDateFromMsg =
355 function(msg, resetHours) {
356     if (msg) {
357         var d = msg.sentDate ? new Date(msg.sentDate) : new Date(msg.date);
358         if (d && resetHours) {
359             d.setHours(0, 0, 0, 0);
360         }
361         return d;
362     }
363     return null;
364 };

366 /**
367  * Sorts sections (e.g. Today, Yesterday, Days, Last Week, etc) by ASC or DESC order.  dayOfWeek is used to sort the week days in ASC/DESC order
368  * @param dayOfWeek {integer} day value of today
369  * @param sortAsc  {boolean} true if sort ascending
370  * @return keys {array} sorted keys
371  */
372 ZmMailListDateGroup.prototype._sortKeys =
373 function(dayOfWeek, sortAsc) {
374     var keys = [];
375 	var sortedDays = this._sortThisWeek(dayOfWeek);
376 	sortedDays = sortedDays.slice(2); //account for today & yesterday
377 	keys = [ZmMailListDateGroup.TODAY, ZmMailListDateGroup.YESTERDAY];
378 	keys = keys.concat(sortedDays);
379 	keys = keys.concat([ZmMailListDateGroup.LAST_WEEK, ZmMailListDateGroup.TWO_WEEKS_AGO, ZmMailListDateGroup.THREE_WEEKS_AGO,
380 						ZmMailListDateGroup.EARLIER_THIS_MONTH, ZmMailListDateGroup.LAST_MONTH, ZmMailListDateGroup.OLDER]);
381     if (sortAsc) {
382         keys.reverse();
383     }
384     return keys;
385 };

387 ZmMailListDateGroup.prototype._calculateLastMonth =
388 function(month) {
389     var lastMonth = month -1;
390     if (lastMonth == -1){
391         lastMonth = 11;
392     }
393     return lastMonth;
394 };

412 /**
413  * Sort days for this week.  If today is Monday & preferences is start week with Sunday, result will by [Monday, Sunday]
414  * @param firstDay {integer}  day value of today
415  * @return sorteDays {array} array of sorted days
416  */
417 ZmMailListDateGroup.prototype._sortThisWeek =
418 function(firstDay) {
419 	var sortedDays = [];
420 	var count = 0;
421 	var foundStart = false;
422 	while (count < 7 && !foundStart) {
423 		if (firstDay == this._weekStartDay) {
424 			foundStart = true;
425 		}
426 		sortedDays[count] = ZmMailListDateGroup.WEEKDAYS[firstDay];
427 		firstDay--;
428 		if (firstDay < 0) {
429 			firstDay = 6;
430 		}
431 		count++;
432 	}
433 
434 	return sortedDays;
435 };
