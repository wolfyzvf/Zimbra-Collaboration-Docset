 16 /**
 17  * @overview
 18  * 
 19  * This file accesses the set of system retention policies
 20  *
 21  */
 22 
 23 ZmSystemRetentionPolicy = function() {
 24 };
 25 
 26 ZmSystemRetentionPolicy.prototype.constructor = ZmSystemRetentionPolicy;
 27 
 28 ZmSystemRetentionPolicy.prototype.toString =
 29 function() {
 30 	return "ZmSystemRetentionPolicy";
 31 };
 32 
 33 ZmSystemRetentionPolicy.prototype.getKeepPolicies =
 34 function() {
 35 	return this._keepPolicies;
 36 };
 37 
 38 ZmSystemRetentionPolicy.prototype.getPurgePolicies =
 39 function() {
 40 	return this._purgePolicies;
 41 };
 42 
 43 
 44 // Read the system retention policies from the server
 45 ZmSystemRetentionPolicy.prototype.getPolicies =
 46 function(callback, batchCmd) {
 47     this._keepPolicies  = new Array();
 48     this._purgePolicies = new Array();
 49 
 50     var jsonObj = {GetSystemRetentionPolicyRequest:{_jsns:"urn:zimbraMail"}};
 51     var request = jsonObj.GetSystemRetentionPolicyRequest;
 52     var respCallback = this._handleResponseGetPolicies.bind(this, callback);
 53     if (batchCmd) {
 54         batchCmd.addRequestParams(jsonObj, respCallback);
 55     } else {
 56         appCtxt.getRequestMgr().sendRequest({jsonObj:jsonObj, asyncMode:true, callback:respCallback});
 57     }
 58 };
 59 
 60 
 61 
 62 ZmSystemRetentionPolicy.prototype._handleResponseGetPolicies =
 63 function(callback, result) {
 64 	var resp = result.getResponse().GetSystemRetentionPolicyResponse;
 65     if (resp.retentionPolicy) {
 66         if (resp.retentionPolicy[0].keep && resp.retentionPolicy[0].keep[0] &&
 67             resp.retentionPolicy[0].keep[0].policy) {
 68             this._keepPolicies = resp.retentionPolicy[0].keep[0].policy;
 69         }
 70         if (resp.retentionPolicy[0].purge && resp.retentionPolicy[0].purge[0] &&
 71             resp.retentionPolicy[0].purge[0].policy) {
 72             this._purgePolicies = resp.retentionPolicy[0].purge[0].policy;
 73         }
 74     }
 75 	if (callback) {
 76 		callback.run(this._keepPolicies, this._purgePolicies);
 77 	}
 78 };
