 18  * This file contains the new task folder dialog.

 22 /**
 23  * Creates the new task folder dialog.
 24  * @class
 25  * This class represents the new task folder dialog.
 26  * 
 27  * @param	{DwtControl}	parent		the parent
 28  * @param	{String}	className		the class name
 29  * 
 30  * @extends		ZmNewOrganizerDialog
 31  */
 32 ZmNewTaskFolderDialog = function(parent, className) {
 33 	ZmNewOrganizerDialog.call(this, parent, className, ZmMsg.createNewTaskFolder, ZmOrganizer.TASKS);
 34 };
 35 
 36 ZmNewTaskFolderDialog.prototype = new ZmNewOrganizerDialog;
 37 ZmNewTaskFolderDialog.prototype.constructor = ZmNewTaskFolderDialog;
 38 
 39 
 40 // Public methods
 41 
 42 /**
 43  * Returns a string representation of the object.
 44  * 
 45  * @return		{String}		a string representation of the object
 46  */
 47 ZmNewTaskFolderDialog.prototype.toString =
 48 function() {
 49 	return "ZmNewTaskFolderDialog";
 50 };
 51 
 52 
 53 // Protected methods
 54 
 55 // overload since we always want to init the color to grey
 56 ZmNewTaskFolderDialog.prototype._initColorSelect =
 57 function() {
 58 	var option = this._colorSelect.getOptionWithValue(ZmOrganizer.C_ORANGE);
 59 	this._colorSelect.setSelectedOption(option);
 60 };
 61 
 62 ZmNewTaskFolderDialog.prototype._getRemoteLabel =
 63 function() {
 64 	return ZmMsg.addRemoteTasks;
 65 };
 66 
 67 // overload so we dont show this
 68 ZmNewTaskFolderDialog.prototype._createFolderContentHtml =
 69 function(html, idx) {
 70 	return idx;
 71 };
