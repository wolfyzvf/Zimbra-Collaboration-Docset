198  *

200  * @param {constant}	action			the new message, reply, forward, or an invite action
201  * @param {Boolean}		inNewWindow		if <code>true</code>, we are in detached window
202  * @param {ZmMailMsg}	msg				the original message (reply/forward), or address (new message)
203  * @param {String}		toOverride 		the initial value for To: field
204  * @param {String}		ccOverride		Cc: addresses (optional)
205  * @param {String}		subjOverride 	the initial value for Subject: field
206  * @param {String}		extraBodyText	the canned text to prepend to body (invites)
207  * @param {AjxCallback}	callback		the callback to run after view has been set
208  * @param {String}		accountName		the on-behalf-of From address
209  * @param {String}		accountName		on-behalf-of From address
210  * @param {boolean}		hideView		if true, don't show compose view
211  */
212 ZmComposeController.prototype.doAction =
213 function(params) {
214 
215 	AjxDebug.println(AjxDebug.REPLY, "-------------- " + params.action + " ----------------------------------");
216 	var ac = window.parentAppCtxt || window.appCtxt;
217 	if (params.action == ZmOperation.REPLY || params.action == ZmOperation.REPLY_ALL) {
218 		var str = ["Browser: " + AjxEnv.browser,
219 				   "Offline: " + Boolean(ac.isOffline),
220 				   "Multi accts: " + ac.multiAccounts,
221 				   "New Window: " + params.inNewWindow,
222 				   "Reading pane location: " + ac.get(ZmSetting.READING_PANE_LOCATION)].join(" / ");
223 		AjxDebug.println(AjxDebug.REPLY, str);
224 	}
225 	
226 	// in zdesktop, it's possible there are no accounts that support smtp
227 	if (ac.isOffline && !ac.get(ZmSetting.OFFLINE_COMPOSE_ENABLED)) {
228 		var d = ac.getMsgDialog();
229 		d.setMessage(ZmMsg.composeDisabled, DwtMessageDialog.CRITICAL_STYLE);
230 		d.popup();
231 		return;
232 	}
233 
234 	this._msgSent = false;
235 	if (params.inNewWindow) {
236         var msgId = params.msg ? params.msg.nId : (this._msg ? this._msg.nId : Dwt.getNextId());
237 		var newWinObj = ac.getNewWindow(false, ZmComposeController.NEW_WINDOW_WIDTH, ZmComposeController.NEW_WINDOW_HEIGHT, ZmId.VIEW_COMPOSE + "_" + msgId);
238 
239 		// this is how child window knows what to do once loading:
240 		newWinObj.command = "compose";
241 		newWinObj.params = params;
242         if (newWinObj.win) {
243             newWinObj.win.focus();
244         }
245 	} else {
246 		this._setView(params);
247 		this._listController = params.listController;
248 	}
249 };

324 ZmComposeController.prototype.popShield =
325 function() {
326 	var dirty = this._composeView.isDirty();
327 	if (!dirty && (this._draftType != ZmComposeController.DRAFT_TYPE_AUTO)) {
328 		return true;
329 	}
330 
331 	var ps = this._popShield = appCtxt.getYesNoCancelMsgDialog();
332 	if (this._draftType == ZmComposeController.DRAFT_TYPE_AUTO) {
333 		// Messsage has been saved, but never explicitly by the user.
334 		// Ask if he wants to keep the autosaved draft.
335 		ps.reset();
336 		ps.setMessage(ZmMsg.askSaveAutosavedDraft, DwtMessageDialog.WARNING_STYLE);
337 		if (dirty) {
338 			ps.registerCallback(DwtDialog.YES_BUTTON, this._popShieldYesCallback, this);
339 		} else {
340 			ps.registerCallback(DwtDialog.YES_BUTTON, this._popShieldNoCallback, this);
341 		}
342 		ps.registerCallback(DwtDialog.NO_BUTTON, this._popShieldDiscardCallback, this);
343 		ps.registerCallback(DwtDialog.CANCEL_BUTTON, this._popShieldDismissCallback, this);
344 	} else if (this._canSaveDraft()) {
345 		ps.reset();
346 		ps.setMessage(ZmMsg.askSaveDraft, DwtMessageDialog.WARNING_STYLE);
347 		ps.registerCallback(DwtDialog.YES_BUTTON, this._popShieldYesCallback, this);
348 		ps.registerCallback(DwtDialog.NO_BUTTON, this._popShieldNoCallback, this);
349 		ps.registerCallback(DwtDialog.CANCEL_BUTTON, this._popShieldDismissCallback, this);
350 	} else {
351 		ps = this._popShield = appCtxt.getYesNoMsgDialog();
352 		ps.setMessage(ZmMsg.askLeaveCompose, DwtMessageDialog.WARNING_STYLE);
353 		ps.registerCallback(DwtDialog.YES_BUTTON, this._popShieldYesCallback, this);
354 		ps.registerCallback(DwtDialog.NO_BUTTON, this._popShieldDismissCallback, this);
355 	}
356 	ps.addPopdownListener(this._dialogPopdownListener);
357 	ps.popup(this._composeView._getDialogXY());
358 
359 	return false;
360 };

501 /**
502  * Sends the message represented by the content of the compose view.
503  * 
504  * @private
505  */
506 ZmComposeController.prototype._sendMsg =
507 function(attId, docIds, draftType, callback, contactId) {
508 
509 	var isTimed = Boolean(this._sendTime);
510 	draftType = draftType || (isTimed ? ZmComposeController.DRAFT_TYPE_DELAYSEND : ZmComposeController.DRAFT_TYPE_NONE);
511 	var isDraft = draftType != ZmComposeController.DRAFT_TYPE_NONE;
512 	var isAutoSave = draftType == ZmComposeController.DRAFT_TYPE_AUTO;
513 	// bug fix #38408 - briefcase attachments need to be set *before* calling
514 	// getMsg() but we cannot do that without having a ZmMailMsg to store it in.
515 	// File this one under WTF.
516 	var tempMsg;
517 	if (docIds) {
518 		tempMsg = new ZmMailMsg();
519 		this._composeView.setDocAttachments(tempMsg, docIds);
520 	}
521 	var msg = this._composeView.getMsg(attId, isDraft, tempMsg, isTimed, contactId);
522 
523 	if (!msg) {
524 		return;
525 	}
526 
527 	if (this._autoSaveTimer) {
528         //If this._autoSaveTimer._timer is null then this._autoSaveTimer.kill(); is already called.
529         //Due to browsers cleartimeout taking some time, we are again checking this._autoSaveTimer._timer to prevent unnecessary autosave call
530         //Bug:74148
531         if (isAutoSave && isDraft && !this._autoSaveTimer._timer) {
532             return;
533         }
534 		//kill the timer, no save is attempted while message is pending
535         this._autoSaveTimer.kill();
536 	}
537 
538 	var origMsg = msg._origMsg;
539 	var isCancel = (msg.inviteMode == ZmOperation.REPLY_CANCEL);
540 	var isModify = (msg.inviteMode == ZmOperation.REPLY_MODIFY);
541 
542 	if (isCancel || isModify) {
543 		var appt = origMsg._appt;
544 		var respCallback = this._handleResponseCancelOrModifyAppt.bind(this);
545 		if (isCancel) {
546 			appt.cancel(origMsg._mode, msg, respCallback);
547 		} else {
548 			appt.save();
549 		}
550 		return;
551 	}
552 
553 	var ac = window.parentAppCtxt || window.appCtxt;
554 	var acctName = appCtxt.multiAccounts
555 		? this._composeView.getFromAccount().name : this._accountName;
556 
557 	if (isDraft) {
558 		if (appCtxt.multiAccounts) {
559 			// for offline, save drafts based on account owner of From: dropdown
560 			acctName = ac.accountList.getAccount(msg.fromSelectValue.accountId).name;
561 		} else {
562 			acctName = ac.getActiveAccount().name;
563 		}
564 		if (msg._origMsg && msg._origMsg.isDraft) {
565 			// if shared folder, make sure we save the draft under the owner account name
566 			var folder = msg.folderId ? ac.getById(msg.folderId) : null;
567 			if (folder && folder.isRemote()) {
568 				acctName = folder.getOwner();
569 			}
570 		}
571 	} else {
572 		// if shared folder, make sure we send the email on-behalf-of
573 		var folder = msg.folderId ? ac.getById(msg.folderId) : null;
574 		if (folder && folder.isRemote() && this._composeView.sendMsgOboIsOK()) {
575 			acctName = folder.getOwner();            
576 		}
577 	}
578 
579 	if (origMsg) {
580 		origMsg.sendAsMe = !this._composeView.sendMsgOboIsOK();
581 	}
582 
583 	// If this message had been saved from draft and it has a sender (meaning
584 	// it's a reply from someone else's account) then get the account name from
585 	// the from field.
586 	if (!acctName && !isDraft && origMsg && origMsg.isDraft) {
587 		if (this._composeView.sendMsgOboIsOK()) {
588 			if (origMsg._addrs[ZmMailMsg.HDR_FROM] &&
589 				origMsg._addrs[ZmMailMsg.HDR_SENDER] &&
590 				origMsg._addrs[ZmMailMsg.HDR_SENDER].size())
591 			{
592 				acctName = origMsg._addrs[ZmMailMsg.HDR_FROM].get(0).address;
593 			}
594 		} else {
595 			origMsg.sendAsMe = true; // hack.
596 		}
597 	}
598 
599 	// check for read receipt
600 	var requestReadReceipt = !this.isHidden && this.isRequestReadReceipt();
601 
602 	var respCallback = this._handleResponseSendMsg.bind(this, draftType, msg, callback);
603 	var errorCallback = this._handleErrorSendMsg.bind(this, draftType, msg);
604 	msg.send(isDraft, respCallback, errorCallback, acctName, null, requestReadReceipt, null, this._sendTime, isAutoSave);
605 	this._resetDelayTime();
606 };

643 ZmComposeController.prototype._handleErrorSendMsg =
644 function(draftType, msg, ex) {
645     var retVal = false;
646 	if (!this.isHidden) {
647 		this.resetToolbarOperations();
648 		this._composeView.enableInputs(true);
649 	}
650 
651     appCtxt.notifyZimlets("onSendMsgFailure", [this, ex, msg]);//notify Zimlets on failure
652     if (ex && ex.code) {
653 	
654         var msg = null;
655         var showMsg = false;
656         if (ex.code == ZmCsfeException.MAIL_SEND_ABORTED_ADDRESS_FAILURE) {
657             var invalid = ex.getData ? ex.getData(ZmCsfeException.MAIL_SEND_ADDRESS_FAILURE_INVALID) : null;
658             var invalidMsg = (invalid && invalid.length)
659                 ? AjxMessageFormat.format(ZmMsg.sendErrorInvalidAddresses, invalid.join(", ")) : null;
660             msg = ZmMsg.sendErrorAbort + "<br/>" +  AjxStringUtil.htmlEncode(invalidMsg);
661             this.popupErrorDialog(msg, ex, true, true, false, true);
662             retVal = true;
663         } else if (ex.code == ZmCsfeException.MAIL_SEND_PARTIAL_ADDRESS_FAILURE) {
664             var invalid = ex.getData ? ex.getData(ZmCsfeException.MAIL_SEND_ADDRESS_FAILURE_INVALID) : null;
665             msg = (invalid && invalid.length)
666                 ? AjxMessageFormat.format(ZmMsg.sendErrorPartial, AjxStringUtil.htmlEncode(invalid.join(", ")))
667                 : ZmMsg.sendErrorAbort;
668             showMsg = true;
669         } else if (ex.code == AjxException.CANCELED) {
670             if (draftType == ZmComposeController.DRAFT_TYPE_AUTO) {
671                 if (!this._autoSaveInterval) {
672                     // Request was cancelled due to a ZmRequestMgr send timeout.
673                     // The server can either be hung or this particular message is taking
674                     // too long to process. Backoff the send interval - restored to
675                     // default upon first successful save
676                     this._autoSaveInterval = appCtxt.get(ZmSetting.AUTO_SAVE_DRAFT_INTERVAL);
677                 }
678                 if (this._autoSaveInterval) {
679                     // Cap the save attempt interval at 5 minutes
680                     this._autoSaveInterval *= 2;
681                     if (this._autoSaveInterval > 300) {
682                         this._autoSaveInterval = 300;
683                     }
684                 }
685             }
686             msg = ZmMsg.cancelSendMsgWarning;
687             this._composeView.setBackupForm();
688             retVal = true;
689         } else if (ex.code == ZmCsfeException.MAIL_QUOTA_EXCEEDED) {
690 		    msg = ZmMsg.errorQuotaExceeded;
691 	    }
692 
693         if (this._uploadingProgress){
694 		    this._composeView._resetUpload(true);
695 		    msg = ZmMsg.attachingFilesError + (ex.msg || "");
696             showMsg = true;
697         }
698         if (msg && showMsg) {
699             var msgDialog = appCtxt.getMsgDialog();
700             msgDialog.setMessage(msg, DwtMessageDialog.CRITICAL_STYLE);
701             msgDialog.popup();
702             retVal = true;
703         }
704     }
705 
706     // Assume the user stays on the compose view, so we need the timer.
707     // (it was canceled when send was called)
708     this._initAutoSave();
709     return retVal;
710 };
