 22 /**
 23  * Creates a search tree controller.
 24  * @class
 25  * This class controls a tree display of saved searches.
 26  *
 27  * @author Conrad Damon
 28  * 
 29  * @extends		ZmFolderTreeController
 30  */
 31 ZmSearchTreeController = function() {
 32 
 33 	ZmFolderTreeController.call(this, ZmOrganizer.SEARCH);
 34 
 35 	this._listeners[ZmOperation.RENAME_SEARCH] = this._renameListener.bind(this);
 36 };
 37 
 38 ZmSearchTreeController.prototype = new ZmFolderTreeController;
 39 ZmSearchTreeController.prototype.constructor = ZmSearchTreeController;
 40 
 41 ZmSearchTreeController.prototype.isZmSearchTreeController = true;
 42 ZmSearchTreeController.prototype.toString = function() { return "ZmSearchTreeController"; };
 43 
 44 ZmSearchTreeController.APP_JOIN_CHAR = "-";
 45 
 46 // Public methods
 47 
 48 /**
 49  * Shows the tree of this type.
 50  *
 51  * @param	{Hash}	params		a hash of parameters
 52  * @param	{String}	params.overviewId		the overview ID
 53  * @param	{Boolean}	params.showUnread		if <code>true</code>, unread counts will be shown
 54  * @param	{Array}	params.omit				a hash of organizer IDs to ignore
 55  * @param	{Boolean}	params.forceCreate	if <code>true</code>, tree view will be created
 56  * @param	{ZmZimbraAccount}	params.account	the account to show tree for (if not currently active account)
 57  * 
 58  */
 59 ZmSearchTreeController.prototype.show =
 60 function(params) {
 61 	var id = params.overviewId;
 62 	if (!this._treeView[id] || params.forceCreate) {
 63 		this._treeView[id] = this._setup(id);
 64 	}
 65     var dataTree = this.getDataTree(params.account);
 66     if (dataTree) {
 67 		params.dataTree = dataTree;
 68 		params.searchTypes = {};
 69 		params.omit = params.omit || {};
 70 		params.omit[ZmFolder.ID_TRASH] = true;
 71 		params.omitParents = true;
 72 		var setting = ZmOrganizer.OPEN_SETTING[this.type];
 73 		params.collapsed = !(!setting || (appCtxt.get(setting, null, params.account) !== false));
 74 		var overview = this._opc.getOverview(id);
 75 		if (overview && overview.showNewButtons)
 76 			this._setupOptButton(params);
 77 		this._treeView[id].set(params);
 78 		this._checkTreeView(id);
 79 	}
 80 	
 81 	return this._treeView[id];
 82 };

 94 /**
 95 * Resets and enables/disables operations based on context.
 96 *
 97 * @param {ZmControl}	parent		the widget that contains the operations
 98 * @param {constant}	type		the type
 99 * @param {String}	id			the currently selected/activated organizer
100 */
101 ZmSearchTreeController.prototype.resetOperations =
102 function(parent, type, id) {
103 	parent.enableAll(true);
104 	var search = appCtxt.getById(id);
105 	parent.enable(ZmOperation.EXPAND_ALL, (search.size() > 0));
106 };

156 /**
157  * Called when a left click occurs (by the tree view listener). The saved
158  * search will be run.
159  *
160  * @param {ZmSearchFolder}		searchFolder		the search that was clicked
161  * 
162  * @private
163  */
164 ZmSearchTreeController.prototype._itemClicked =
165 function(searchFolder) {
166 	if (searchFolder._showFoldersCallback) {
167 		searchFolder._showFoldersCallback.run();
168 		return;
169 	}
170 
171 	appCtxt.getSearchController().redoSearch(searchFolder.search, false, {getHtml: appCtxt.get(ZmSetting.VIEW_AS_HTML)});
172 };

174 /**
175  * @private
176  */
177 ZmSearchTreeController.prototype._getMoveParams =
178 function(dlg) {
179 	var params = ZmTreeController.prototype._getMoveParams.apply(this, arguments);
180 	params.overviewId = dlg.getOverviewId(this.type);
181 	params.treeIds = [ZmOrganizer.FOLDER, ZmOrganizer.SEARCH];
182 	return params;
183 };

197 /**
198  * Shows or hides the tree view. It is hidden only if there are no saved
199  * searches that belong to the owning app, and we have been told to hide empty
200  * tree views of this type.
201  * 
202  * @param {constant}	overviewId		the overview ID
203  * 
204  * @private
205  */
206 ZmSearchTreeController.prototype._checkTreeView =
207 function(overviewId) {
208 	var treeView = this._treeView[overviewId];
209 	if (!overviewId || !treeView) { return; }
210 
211 	var account = this._opc.getOverview(overviewId).account;
212 	var rootId = (appCtxt.multiAccounts && !account.isMain)
213 		? (ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT, account))
214 		: ZmOrganizer.ID_ROOT;
215 	var hide = ZmOrganizer.HIDE_EMPTY[this.type] && !treeView.getTreeItemById(rootId).getItemCount();
216 	this._treeView[overviewId].setVisible(!hide);
217 };
