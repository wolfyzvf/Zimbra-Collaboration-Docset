 16 /**
 17  * Creates a border less button.
 18  * @constructor
 19  * @class
 20  * This class represents a button without a border.
 21  *
 22  * @param {hash}	params		a hash of parameters
 23  * @param {DwtComposite}      params.parent		the parent widget
 24  * @param {constant}      params.style			the button style (see {@link DwtButton})
 25  * @param {string}      params.className		the CSS class
 26  * @param {constant}      params.posStyle		the positioning style (see {@link Dwt})
 27  * @param {DwtButton.ACTION_MOUSEUP|DwtButton.ACTION_MOUSEDOWN}      params.actionTiming	if {@link DwtButton.ACTION_MOUSEUP}, then the button is triggered
 28  *											on mouseup events, else if {@link DwtButton.ACTION_MOUSEDOWN},
 29  * 											then the button is triggered on mousedown events
 30  * @param {string}      params.id			the ID to use for the control's HTML element
 31  * @param {number}      params.index 		the index at which to add this control among parent's children
 32  * 
 33  * @extends		DwtButton
 34  */
 35 DwtBorderlessButton = function(params) {
 36 	if (arguments.length == 0) { return; }
 37 	params = Dwt.getParams(arguments, DwtBorderlessButton.PARAMS);
 38 
 39 	DwtButton.call(this, params);
 40 }
 41 
 42 DwtBorderlessButton.PARAMS = ["parent", "style", "className", "posStyle", "actionTiming", "id", "index"];
 43 
 44 DwtBorderlessButton.prototype = new DwtButton;
 45 DwtBorderlessButton.prototype.constructor = DwtBorderlessButton;
 46 
 47 DwtBorderlessButton.prototype.toString =
 48 function() {
 49 	return "DwtBorderlessButton";
 50 }
