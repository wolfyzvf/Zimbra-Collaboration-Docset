 22 /**
 23  * Creates a new task controller to manage task creation/editing.
 24  * @class
 25  * This class manages task creation/editing.
 26  *
 27  * @author Parag Shah
 28  *
 29  * @param {DwtShell}	container	the containing shell
 30  * @param {ZmApp}		app			the containing app
 31  * @param {constant}	type		controller type
 32  * @param {string}		sessionId	the session id
 33  * 
 34  * @extends		ZmCalItemComposeController
 35  */
 36 ZmTaskController = function(container, app, type, sessionId) {
 37 	if (arguments.length == 0) { return; }
 38 	ZmCalItemComposeController.apply(this, arguments);
 39 };
 40 
 41 ZmTaskController.prototype = new ZmCalItemComposeController;
 42 ZmTaskController.prototype.constructor = ZmTaskController;

 44 ZmTaskController.prototype.isZmTaskController = true;
 45 ZmTaskController.prototype.toString = function() { return "ZmTaskController"; };

 58 ZmTaskController.prototype.saveCalItem =
 59 function(attId) {
 60 	var calItem = this._composeView.getCalItem(attId);
 61 	if (calItem) {
 62 		this._saveCalItemFoRealz(calItem, attId);
 63 		return true;
 64 	}
 65 	return false;
 66 };
 67 
 68 ZmTaskController.prototype.isCloseAction =
 69 function() {
 70     return this._action == ZmCalItemComposeController.SAVE;
 71 };
 72 
 73 ZmTaskController.prototype._handleResponseSave =
 74 function(calItem, result) {
 75 	ZmCalItemComposeController.prototype._handleResponseSave.call(this, calItem);
 76 	if(this._action == ZmCalItemComposeController.SAVE) {
 77 		this.closeView();	
 78 	}
 79 	// XXX: null out message so we re-fetch task next time its opened
 80 	// To optimize, we should save the modified contents into cache'd task item
 81 	if (calItem && calItem._orig)
 82 		calItem._orig.message = null;
 83 
 84     //Cache the item for further processing
 85     calItem.cache();
 86     //need to set rev,ms for next soap request
 87     calItem.setFromSavedResponse(result);
 88     
 89 	appCtxt.setStatusMsg(ZmMsg.taskSaved);
 90     if(calItem.alarm == true || calItem.isAlarmModified()) {
 91         this._app.getReminderController().refresh();
 92     }    
 93 };

101 function() {

103 };
104 
105 ZmTaskController.prototype._setComposeTabGroup =
106 function(setFocus) {
107 	var tg = this._createTabGroup();
108 	var rootTg = appCtxt.getRootTabGroup();
109 	tg.newParent(rootTg);
110 	tg.addMember(this._toolbar);
111 	this._composeView._addTabGroupMembers(tg);
112 
113 	var focusItem = this._composeView || this._composeView._getDefaultFocusItem() || tg.getFirstMember(true);
114 	var ta = new AjxTimedAction(this, this._setFocus, [focusItem, !setFocus]);
115 	AjxTimedAction.scheduleAction(ta, 10);
116 };

123 /**
124  * Checks if the tasks is moving from local to remote folder (or vice versa).
125  * 
126  * @param	{ZmTask}	task			the task
127  * @param	{String}	newFolderId		the folder id
128  * @return	{Boolean}	<code>true</code> if moving from local to remote folder
129  */
130 ZmTaskController.prototype.isMovingBetwAccounts =
131 function(task, newFolderId) {
132 	var isMovingBetw = false;
133 	if (task._orig) {
134 		var origFolder = task._orig.getFolder();
135 		var newFolder = appCtxt.getById(newFolderId);
136 		if (origFolder && newFolder) {
137 			if ((origFolder.id != newFolderId) &&
138 				((origFolder.link && !newFolder.link) || (!origFolder.link && newFolder.link)))
139 			{
140 				isMovingBetw = true;
141 			}
142 		}
143 	}
144 	return isMovingBetw;
145 };

157 ZmTaskController.prototype._printListener =
158 function() {
159 	var url = ["/h/printtasks?id=", this._composeView._calItem.invId];
160     
161     if (appCtxt.isOffline) {
162         var acctName = this._composeView._calItem.getAccount().name;
163         url.push("&acct=", acctName);
164     }
165 	window.open([appContextPath, url.join(""), "&tz=", AjxTimezone.getServerId(AjxTimezone.DEFAULT)].join(""), "_blank");
166 };
