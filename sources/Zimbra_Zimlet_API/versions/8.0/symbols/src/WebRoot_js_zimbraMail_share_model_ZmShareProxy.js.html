 16 ZmShareProxy = function(params) {
 17     params.type = ZmOrganizer.SHARE;
 18     ZmOrganizer.call(this, params);
 19 };
 20 ZmShareProxy.prototype = new ZmOrganizer;
 21 ZmShareProxy.prototype.constructor = ZmShareProxy;
 22 
 23 ZmShareProxy.prototype.toString = function() {
 24     return "ZmShareProxy";
 25 };
 26 
 27 // Constants
 28 
 29 ZmShareProxy.ID_LOADING = -1;
 30 ZmShareProxy.ID_NONE_FOUND = -2;
 31 ZmShareProxy.ID_WARNING = -2;
 32 ZmShareProxy.ID_ERROR = -3;
 33 
 34 // Data
 35 
 36 ZmShareProxy.prototype.TOOLTIP_TEMPLATE = "share.Widgets#ZmShareProxyToolTip";
 37 
 38 // ZmOrganizer methods
 39 
 40 ZmShareProxy.prototype.getIcon = function() {
 41     // icons for loading states
 42     var m = String(this.id).match(/^(-\d)(?::(.*))?$/);
 43     switch (Number(m && m[1])) { // NOTE: case is === !!!
 44         case ZmShareProxy.ID_LOADING: return "Spinner";
 45         case ZmShareProxy.ID_NONE_FOUND: return "Warning";
 46         case ZmShareProxy.ID_ERROR: return "Critical";
 47     }
 48 
 49     // icon for share owner
 50     if (!this.shareInfo) return "SharedMailFolder";
 51 
 52     // icon based on view type
 53     var type = ZmOrganizer.TYPE[this.shareInfo.view];
 54     var orgPackage = ZmOrganizer.ORG_PACKAGE[type];
 55     if (orgPackage) AjxDispatcher.require(orgPackage);
 56     var orgClass = window[ZmOrganizer.ORG_CLASS[type]];
 57     return orgClass ? orgClass.prototype.getIcon.call(this) : "Folder";
 58 };
 59 
 60 ZmShareProxy.prototype.getToolTip = function(force) {
 61     if (!this.shareInfo) return null;
 62     if (force || !this._tooltip) {
 63         this._tooltip = AjxTemplate.expand(this.TOOLTIP_TEMPLATE, this.shareInfo);
 64     }
 65     return this._tooltip;
 66 };
