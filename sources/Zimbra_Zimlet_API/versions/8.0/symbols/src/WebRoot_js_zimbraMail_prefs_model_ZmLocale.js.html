 16 /**
 17  * Default constructor.
 18  * @class
 19  * This class represents a locale.
 20  * 
 21  * @param	{String}	id		the id
 22  * @param	{String}	name	the name
 23  * @param	{String}	image	the image
 24  * @param	{String}	localName	the name in user's locale
 25  *
 26  * @see		ZmLocale.create
 27  */
 28 ZmLocale = function(id, name, image, localName) {
 29 	this.id = id;
 30 	this.name = name;
 31 	this.localName = localName;
 32 	this._image = image;
 33 };
 34 
 35 //List of RTL supporting languages
 36 ZmLocale.RTLLANGUAGES = {
 37     ar:"Arabic",
 38     iw:"Hebrew"
 39 };
 40 
 41 ZmLocale.localeMap = {};
 42 ZmLocale.languageMap = {};
 43 
 44 /**
 45  * Creates the locale.
 46  * 
 47  * @param	{String}	id		the locale id (for example, <code>en_US</code>)
 48  * @param	{String}	name	the locale name
 49  * @param	{String}	localName	the name in user's locale
 50  */
 51 ZmLocale.create =
 52 function(id, name, localName) {
 53 	var index = id.indexOf("_");
 54 	var languageId;
 55 	var country = null;
 56 	if (index == -1) {
 57 		languageId = id;
 58 	}
 59 	else {
 60 		languageId = id.substr(0, index);
 61 		country = id.substring(id.length - 2);
 62 	}
 63 
 64 	var languageObj = ZmLocale.languageMap[languageId];
 65 	if (!languageObj) {
 66 		languageObj = new ZmLocale(languageId, name, null, localName);
 67 		ZmLocale.languageMap[languageId] = languageObj;
 68 		ZmLocale.localeMap[id] = languageObj;
 69 	}
 70 	if (country) {
 71 		var localeObj = new ZmLocale(id, name, null, localName);
 72 		languageObj._add(localeObj);
 73 		ZmLocale.localeMap[id] = localeObj;
 74 		return localeObj;
 75 	}
 76 	else {
 77 		languageObj.name = name;
 78 		return languageObj;
 79 	}
 80 };

 82 /**
 83  * Checks if there are more than one selectable locale.
 84  * 
 85  * @return	{Boolean}	<code>true</code> if there are more than one selectable locale
 86  */
 87 ZmLocale.hasChoices =
 88 function() {
 89 	var count = 0;
 90 	for (var id in ZmLocale.localeMap) {
 91 		var locale = ZmLocale.localeMap[id]; 
 92 		if (!locale.locales) {
 93 			count++;
 94 		}
 95 		if (count >= 2) {
 96 			return true;
 97 		}
 98 	}
 99 	return false;
100 };

102 /**
103  * Gets the image.
104  * 
105  * @return	{String}	the image
106  */
107 ZmLocale.prototype.getImage =
108 function() {
109 	return this._image;
110 };
111 
112 /**
113  * Gets the name in both the locale itself, and in the local (user) locale. 
114  *
115  * @return	{String}	the name
116  */
117 ZmLocale.prototype.getNativeAndLocalName =
118 function() {
119 	if (this.name == this.localName) {
120 		/* don't show both if they are the same - it looks extremely funny */
121 		return this.name;
122 	}
123 	return [this.localName, " - ", this.name].join("");
124 };
125 
126 ZmLocale.prototype._add =
127 function(locale) {
128 	(this.locales = this.locales || []).push(locale);
129 };
