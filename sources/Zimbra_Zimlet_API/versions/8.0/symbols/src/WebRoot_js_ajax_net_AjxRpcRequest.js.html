 17 /**
 18  * @constructor
 19  * @class
 20  * This class encapsulates the XML HTTP request, hiding differences between
 21  * browsers. The internal request object depends on the browser. While it is 
 22  * possible to use this class directly, {@link AjxRpc} provides a managed interface
 23  * to this class 
 24  *
 25  * @author Ross Dargahi
 26  * @author Conrad Damon
 27  * 
 28  * @param {string} [id]		the ID to identify this object
 29  * 
 30  * @see AjxRpc
 31  * 
 32  */
 33 AjxRpcRequest = function(id) {
 34 	if (!AjxRpcRequest.__inited) {
 35 		AjxRpcRequest.__init();
 36 	}
 37 
 38 	/**
 39 	 * The id for this object.
 40 	 */
 41 	this.id = id;
 42 	this.__httpReq = AjxRpcRequest.__msxmlVers
 43 		? (new ActiveXObject(AjxRpcRequest.__msxmlVers))
 44 		: (new XMLHttpRequest());
 45 };
 46 
 47 AjxRpcRequest.prototype.isAjxRpcRequest = true;
 48 AjxRpcRequest.prototype.toString = function() { return "AjxRpcRequest"; };
 49 
 50 AjxRpcRequest.TIMEDOUT		= -1000;		// Timed out exception
 51 AjxRpcRequest.HTTP_GET		= "get";		//HTTP GET
 52 AjxRpcRequest.HTTP_POST		= "post";		//HTTP POST
 53 AjxRpcRequest.HTTP_PUT		= "put";		//HTTP PUT
 54 AjxRpcRequest.HTTP_DELETE	= "delete";		//HTTP DELETE
 55 
 56 AjxRpcRequest.__inited		= false;
 57 AjxRpcRequest.__msxmlVers	= null;
 58 
 59 
 60 /**
 61  * Sends this request to the target URL. If there is a callback, the request is
 62  * performed asynchronously.
 63  * 
 64  * @param {string} [requestStr] 	the HTTP request string/document
 65  * @param {string} serverUrl 	the request target 
 66  * @param {array} [requestHeaders] an array of HTTP request headers
 67  * @param {AjxCallback} callback 	the callback for asynchronous requests. This callback 
 68  * 		will be invoked when the requests completes. It will be passed the same
 69  * 		values as when this method is invoked synchronously (see the return values
 70  * 		below) with the exception that if the call times out (see timeout param 
 71  * 		below), then the object passed to the callback will be the same as in the 
 72  * 		error case with the exception that the status will be set to 
 73  * 		{@link AjxRpcRequest.TIMEDOUT}.
 74  * @param {Constant} [method] 		the HTTP method -- GET, POST, PUT, DELETE. if <code>true</code>, use get method for backward compatibility
 75  * @param {number} [timeout] 		the timeout (in milliseconds) after which the request is canceled
 76  * 
 77  * @return {object|hash}	if invoking in asynchronous mode, then it will return the id of the 
 78  * 		underlying {@link AjxRpcRequest} object. Else if invoked synchronously, if
 79  * 		there is no error (i.e. we get a HTTP result code of 200 from the server),
 80  * 		an object with the following attributes is returned
 81  * 		<ul>
 82  * 		<li>text - the string response text</li>
 83  * 		<li>xml - the string response xml</li>
 84  * 		<li>success - boolean set to true</li>
 85  * 		</ul>
 86  * 		If there is an eror, then the following will be returned
 87  * 		<ul>
 88  * 		<li>text - the string response text<li>
 89  * 		<li>xml - the string response xml </li>
 90  * 		<li>success - boolean set to <code>false</code></li>
 91  * 		<li>status - HTTP status</li>
 92  * 		</ul>
 93  * 
 94  * @throws	{AjxException.NETWORK_ERROR}	a network error occurs
 95  * @throws	{AjxException.UNKNOWN_ERROR}	an unknown error occurs
 96  * 
 97  * @see AjxRpc.invoke
 98  */
 99 AjxRpcRequest.prototype.invoke =
100 function(requestStr, serverUrl, requestHeaders, callback, method, timeout) {
101 
102 	var asyncMode = (callback != null);
103 	var m = requestStr && requestStr.match(/.*"(\w+Request)"/);
104 	this.methodName = m ? m[1] : serverUrl || "";	// for debugging
105 
106 	// An exception here will be caught by AjxRpc.invoke
107 	var httpMethod = AjxRpcRequest.HTTP_POST;
108 	if (method) {
109 		httpMethod = method === true ? AjxRpcRequest.HTTP_GET : method;
110 	}
111 	this.__httpReq.open(httpMethod, serverUrl, asyncMode);
112 
113 	if (asyncMode) {
114 		if (timeout) {
115 			var action = new AjxTimedAction(this, this.__handleTimeout, [callback]);
116 			callback._timedActionId = AjxTimedAction.scheduleAction(action, timeout);
117 		}
118 		var tempThis = this;
119 		this.__httpReq.onreadystatechange = function(ev) {
120 			if (window.AjxRpcRequest) {
121 				AjxRpcRequest.__handleResponse(tempThis, callback);
122 			}
123 		}
124 	} else {
125 		// IE appears to run handler even on sync requests, so we need to clear it
126 		this.__httpReq.onreadystatechange = function(ev) {};
127 	}
128 
129 	if (requestHeaders) {
130 		for (var i in requestHeaders) {
131             if (requestHeaders.hasOwnProperty(i)) {
132                 this.__httpReq.setRequestHeader(i, requestHeaders[i]);
133             }
134 		}
135 	}
136 
137 	AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " RPC send: " + this.id);
138 	this.__httpReq.send(requestStr);
139 	if (asyncMode) {
140 		return this.id;
141 	} else {
142 		if (this.__httpReq.status == 200 || this.__httpReq.status == 201) {
143 			return {text:this.__httpReq.responseText, xml:this.__httpReq.responseXML, success:true};
144 		} else {
145 			return {text:this.__httpReq.responseText, xml:this.__httpReq.responseXML, success:false, status:this.__httpReq.status};
146 		}
147 	}
148 };

150 /**
151  * Cancels a pending request.
152  * 
153  */
154 AjxRpcRequest.prototype.cancel =
155 function() {
156 	AjxRpc.freeRpcCtxt(this);
157     if (AjxEnv.isFirefox3_5up) {
158 		// bug 55911
159         this.__httpReq.onreadystatechange = function(){};
160     }
161     this.__httpReq.abort();
162 };

164 /**
165  * Handler that runs when an asynchronous request timesout.
166  *
167  * @param {AjxCallback} callback 	the callback to run after timeout
168  * 
169  * @private
170  */
171 AjxRpcRequest.prototype.__handleTimeout =
172 function(callback) {
173 	this.cancel();
174 	callback.run( {text:null, xml:null, success:false, status:AjxRpcRequest.TIMEDOUT} );
175 };
176 
177 /**
178  * Handler that runs when an asynchronous response has been received. It runs a
179  * callback to initiate the response handling.
180  *
181  * @param {AjxRpcRequest}	req		the request that generated the response
182  * @param {AjxCallback}	callback	the callback to run after response is received
183  * 
184  * @private
185  */
186 AjxRpcRequest.__handleResponse =
187 function(req, callback) {
188 
189 	try {
190 
191 	if (!req || !req.__httpReq) {
192 
193 		req.cancel();
194 
195 		// If IE receives a 500 error, the object reference can be lost
196 		DBG.println(AjxDebug.DBG1, "Async RPC request: Lost request object!!!");
197 		AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " Async RPC request: Lost request object!!!");
198 		callback.run( {text:null, xml:null, success:false, status:500} );
199 		AjxRpc.freeRpcCtxt(req);
200 		return;
201 	}
202 
203 	if (req.__httpReq.readyState == 4) {
204 		if (callback._timedActionId !== null) {
205 			AjxTimedAction.cancelAction(callback._timedActionId);
206 		}
207 
208 		var status = 500;
209 		try {
210 			status = req.__httpReq.status;
211 		} catch (ex) {
212 			// Use default status of 500 above.
213 		}
214 
215 		if (status == 200 || status == 201) {
216 			callback.run( {text:req.__httpReq.responseText, xml:req.__httpReq.responseXML, success:true, reqId:req.id} );
217 		} else {
218 			callback.run( {text:req.__httpReq.responseText, xml:req.__httpReq.responseXML, success:false, status:status, reqId:req.id} );
219 		}
220 
221 		AjxRpc.freeRpcCtxt(req);
222 	}
223 
224 	} catch (ex) {
225 		if (window.AjxException) {
226 			AjxException.reportScriptError(ex);
227 		}
228 	}
229 };

231 /**
232  * @private
233  */
234 AjxRpcRequest.__init =
235 function() {
236 	if (!window.XMLHttpRequest && window.ActiveXObject) {
237 		// search for the latest xmlhttp version on user's machine (IE 6)
238 		var msxmlVers = ["MSXML2.XMLHTTP.4.0", "MSXML2.XMLHTTP.3.0", "MSXML2.XMLHTTP", "Microsoft.XMLHTTP"];
239 		for (var i = 0; i < msxmlVers.length; i++) {
240 			try {
241 				var x = new ActiveXObject(msxmlVers[i]);
242 				AjxRpcRequest.__msxmlVers = msxmlVers[i];
243 				break;
244 			} catch (ex) {
245 				// do nothing
246 			}
247 		}
248 		if (!AjxRpcRequest.__msxmlVers) {
249 			throw new AjxException("MSXML not installed", AjxException.INTERNAL_ERROR, "AjxRpc._init");
250 		}
251 	}
252 	AjxRpcRequest.__inited = true;
253 };
