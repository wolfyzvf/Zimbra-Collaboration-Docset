 16 ZmNewCalendarDialog = function(parent, className) {
 17     if (arguments.length == 0) { return; }
 18 	var title = ZmMsg.createNewCalendar;
 19 	var type = ZmOrganizer.CALENDAR;
 20     var back = new DwtDialog_ButtonDescriptor(ZmNewCalendarDialog.BACK_BUTTON, ZmMsg.back , DwtDialog.ALIGN_LEFT);
 21 	ZmNewOrganizerDialog.call(this, parent, className, title, type, [back]);
 22     this.setButtonListener(ZmNewCalendarDialog.BACK_BUTTON, this._backButtonListener.bind(this));
 23     this.getButton(ZmNewCalendarDialog.BACK_BUTTON).setVisibility(false);
 24 };
 25

 27 ZmNewCalendarDialog.prototype.constructor = ZmNewCalendarDialog;

 94 /** 
 95  * Checks the input for validity and returns the following array of values:
 96  * <ul>
 97  * <li> parentFolder
 98  * <li> name
 99  * <li> color
100  * <li> URL
101  * <li> excludeFB
102  * </ul>
103  * 
104  * @private
105  */
106 ZmNewCalendarDialog.prototype._getFolderData =
107 function() {
108 	var data = ZmNewOrganizerDialog.prototype._getFolderData.call(this);
109 	if (data) {
110 		data.f = this._excludeFbCheckbox.checked ? "b#" : "#";
111         var url =  this._iCalData ? this._iCalData.url : "";
112         if(url) {
113             data.url = url;
114             this._iCalData = null;
115             delete this._iCalData;
116         }
117 	}
118 	return data;
119 };
