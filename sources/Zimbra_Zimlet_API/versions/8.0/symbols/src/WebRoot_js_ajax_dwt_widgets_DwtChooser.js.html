 16 /**
 17  * Creates a control that allows the user to select items from a list, and
 18  * places the selected items in another list.
 19  * @constructor
 20  * @class
 21  * This class creates and manages a control that lets the user
 22  * select items from a list. Two lists are maintained, one with items to select 
 23  * from, and one that contains the selected items. Between them are buttons 
 24  * to shuffle items back and forth between the two lists.
 25  * <p>
 26  * There are two types of buttons: one or more transfer buttons move items from
 27  * the source list to the target list, and the remove button moves items from the
 28  * target list to the source list. The client can specify its transfer buttons.
 29  * If no specification is given, there will be a single transfer button called 
 30  * "Add".</p>
 31  * <p>
 32  * The parent must implement search(columnItem, ascending) if column sorting
 33  * is supported. It should also create a subclass of {@link DwtChooser} which returns
 34  * the appropriate source and target list views, themselves subclasses of
 35  * {@link DwtChooserListView}. Those subclasses must implement _getHeaderList() and
 36  * _createItemHtml(item).</p>
 37  * <p>
 38  * There are two different layout styles, horizontal (with the list views at the
 39  * left and right) and vertical (with the list views at the top and bottom). There
 40  * are two different selection styles, single and multiple, which control how many
 41  * items may appear in the target list view. There are two different transfer modes:
 42  * one where items are copied between lists, and one where they're moved.</p>
 43  *
 44  * @author Conrad Damon
 45  *
 46  * @param	{hash}		params		a hash of parameters
 47  * @param {DwtComposite}	params.parent			the containing widget
 48  * @param {string}	params.className		the CSS class
 49  * @param {string}	params.slvClassName	the CSS class for source list view
 50  * @param {string}	params.tlvClassName	the CSS class for target list view
 51  * @param {array}	params.buttonInfo		the id/label pairs for transfer buttons
 52  * @param {DwtChooser.HORIZ_STYLE|DwtChooser.VERT_STYLE}	params.layoutStyle	the layout style (vertical or horizontal)
 53  * @param {DwtChooser.SINGLE_SELECT|DwtChooser.MULTI_SELECT}	params.selectStyle	the multi-select (default) or single-select
 54  * @param {constant}	params.mode			the items are moved or copied
 55  * @param {boolean}	params.noDuplicates	if <code>true</code>, prevent duplicates in target list
 56  * @param {number}	params.singleHeight	the height of list view for single select style
 57  * @param {number}	params.listSize		the list width (if {@link DwtChooser.HORIZ_STYLE}) or height (if {@link DwtChooser.VERT_STYLE})
 58  * @param {boolean}	params.sourceEmptyOk	if <code>true</code>, do not show "No Results" in source list view
 59  * @param {boolean}	params.allButtons		if <code>true</code>, offer "Add All" and "Remove All" buttons
 60  * @param {boolean}	params.hasTextField	if <code>true</code>, create a text field for user input
 61  * 
 62  * @extends		DwtComposite
 63  */
 64 DwtChooser = function(params) {
 65 
 66 	if (arguments.length == 0) return;
 67 	DwtComposite.call(this, params.parent, params.className);
 68 
 69 	this._slvClassName = params.slvClassName;
 70 	this._tlvClassName = params.tlvClassName;
 71 	this._layoutStyle = params.layoutStyle ? params.layoutStyle : DwtChooser.HORIZ_STYLE;
 72 	this._selectStyle = params.selectStyle ? params.selectStyle : DwtChooser.MULTI_SELECT;
 73 	this._mode = params.listStyle ? params.listStyle : DwtChooser.MODE_MOVE;
 74 	this._noDuplicates = (params.noDuplicates !== false);
 75 	this._singleHeight = params.singleHeight ? params.singleHeight : 45; // 45 = header row + row with icon
 76 	this._listSize = params.listSize;
 77 	this._sourceEmptyOk = params.sourceEmptyOk;
 78 	this._allButtons = params.allButtons;
 79 	this._hasTextField = params.hasTextField;
 80 
 81 	this._handleButtonInfo(params.buttonInfo);
 82 	this._mode = params.mode ? params.mode :
 83 						this._hasMultiButtons ? DwtChooser.MODE_COPY : DwtChooser.MODE_MOVE;
 84 
 85 	this._createHtml();
 86 	this._initialize();
 87 	
 88 	var parentSz = params.parent.getSize();
 89 	var listWidth = params.listWidth || parentSz.x;
 90 	var listHeight = params.listHeight || parentSz.y;
 91 	if (listWidth && listHeight) {
 92 		this.resize(listWidth, listHeight);
 93 	}
 94 
 95 };

135 /**
136  * Sets the given list view with the given list. Defaults to source view.
137  *
138  * @param {AjxVector|Array|Object|Hash}	items			a list of items or hash of lists
139  * @param {DwtChooserListView.SOURCE|DwtChooserListView.TARGET}	view			the view to set
140  * @param {boolean}	clearOtherView	if <code>true</code>, clear out other view
141  */
142 DwtChooser.prototype.setItems =
143 function(items, view, clearOtherView) {
144 	view = view ? view : DwtChooserListView.SOURCE;
145 	this._reset(view);
146 	this.addItems(items, view, true);
147 	this._selectFirst(view);
148 	if (clearOtherView) {
149 		this._reset((view == DwtChooserListView.SOURCE) ? DwtChooserListView.TARGET : DwtChooserListView.SOURCE);
150 	}
151 };

153 /**
154  * Gets a copy of the items in the given list. If that's the target list and 
155  * there are multiple transfer buttons, then a hash with a vector for each one
156  * is returned. Otherwise, a single vector is returned. Defaults to target view.
157  *
158  * @param {DwtChooserListView.SOURCE|DwtChooserListView.TARGET}	view			the view to set
159  * @return	{AjxVector|array|Object|hash}		the item(s)
160  */
161 DwtChooser.prototype.getItems =
162 function(view) {
163 	view = view ? view : DwtChooserListView.TARGET;
164 	if (view == DwtChooserListView.SOURCE) {
165 		return this.sourceListView.getList().clone();
166 	} else {	
167 		if (this._hasMultiButtons) {
168 			var data = {};
169 			for (var i in this._data) {
170 				data[i] = this._data[i].clone();
171 			}
172 			return data;
173 		} else {
174 			return this._data[this._buttonInfo[0].id].clone();
175 		}
176 	}
177 };

179 /**
180  * Adds items to the given list view.
181  *
182  * @param {AjxVector|array|Object|hash}	items			a list of items or hash of lists
183  * @param {DwtChooserListView.SOURCE|DwtChooserListView.TARGET}	view			the view to set
184  * @param {boolean}	skipNotify	if <code>true</code>, do not notify listeners
185  * @param {string}	id			the button ID
186  */
187 DwtChooser.prototype.addItems =
188 function(items, view, skipNotify, id) {
189 	view = view ? view : DwtChooserListView.SOURCE;
190 	var list = (items instanceof AjxVector) ? items.getArray() : (items instanceof Array) ? items : [items];
191 	if (view == DwtChooserListView.SOURCE) {
192 		for (var i = 0; i < list.length; i++) {
193 			this._addToSource(list[i], null, skipNotify);
194 		}
195 	} else {
196 		var data;
197 		if (this._selectStyle == DwtChooser.SINGLE_SELECT) {
198 			this.targetListView._resetList();
199 			list = (list.length > 0) ? [list[0]] : list;
200 		}
201 		for (var i = 0; i < list.length; i++) {
202 			this._addToTarget(list[i], id, skipNotify);
203 //			if (this._selectStyle == DwtChooser.SINGLE_SELECT) {
204 //				return;
205 //			}
206 		}
207 	}
208 	if (view == DwtChooserListView.SOURCE) {
209 		var list = this.sourceListView.getList();
210 		this._sourceSize = list ? list.size() : 0;
211 	}
212 };

214 /**
215  * Removes items from the given list view.
216  *
217  * @param {AjxVector|array|Object|hash}	list			a list of items or hash of lists
218  * @param {DwtChooserListView.SOURCE|DwtChooserListView.TARGET}	view			the view to set
219  * @param {boolean}	skipNotify	if <code>true</code>, do not notify listeners
220  */
221 DwtChooser.prototype.removeItems =
222 function(list, view, skipNotify) {
223 	list = (list instanceof AjxVector) ? list.getArray() : (list instanceof Array) ? list : [list];
224 	for (var i = 0; i < list.length; i++) {
225 		(view == DwtChooserListView.SOURCE) ? this._removeFromSource(list[i], skipNotify) : this._removeFromTarget(list[i], skipNotify);
226 	}
227 };
228 
229 /**
230  * Moves or copies items from the source list to the target list, paying attention
231  * to current mode.
232  *
233  * @param {AjxVector|array|Object|hash}	list			a list of items or hash of lists
234  * @param {string}	id			the ID of the transfer button that was used
235  * @param {boolean}	skipNotify	if <code>true</code>, do not notify listeners
236  */
237 DwtChooser.prototype.transfer =
238 function(list, id, skipNotify) {
239 	id = id ? id : this._activeButtonId;
240 	this._setActiveButton(id);
241 	if (this._mode == DwtChooser.MODE_MOVE) {
242 		if (this._selectStyle == DwtChooser.SINGLE_SELECT) {
243 			var tlist = this.targetListView.getList();
244 			if (tlist && tlist.size()) {
245 				this.remove(tlist, true);
246 			}
247 		}
248 		this.removeItems(list, DwtChooserListView.SOURCE, true);
249 	}
250 	this.addItems(list, DwtChooserListView.TARGET, skipNotify);
251 	this.sourceListView.deselectAll();
252 };
253 
254 /**
255  * Removes items from target list, paying attention to current mode. Also handles button state.
256  *
257  * @param {AjxVector|array|Object|hash}	list			a list of items or hash of lists
258  * @param {boolean}	skipNotify	if <code>true</code>, do not notify listeners
259  */
260 DwtChooser.prototype.remove =
261 function(list, skipNotify) {
262 	list = (list instanceof AjxVector) ? list.getArray() : (list instanceof Array) ? list : [list];
263 	if (this._mode == DwtChooser.MODE_MOVE) {
264 		for (var i = 0; i < list.length; i++) {
265 			var index = this._getInsertionIndex(this.sourceListView, list[i]);
266 			this.sourceListView.addItem(list[i], index, true);
267 		}
268 		this._sourceSize = list ? list.length : 0;
269 	}
270 	this.removeItems(list, DwtChooserListView.TARGET);
271 };

273 /**
274  * Sets the select style to the given style. Performs a resize
275  * in order to adjust the layout, and changes the label on the transfer button if it's
276  * the default one.
277  *
278  * @param {DwtChooser.SINGLE_SELECT|DwtChooser.MULTI_SELECT}	style		the style single or multiple select
279  * @param {boolean}	noResize	if <code>true</code>, do not perform resize
280  */
281 DwtChooser.prototype.setSelectStyle =
282 function(style, noResize) {
283 	if (style == this._selectStyle) return;
284 	
285 	this._selectStyle = style;
286 	if (this._defLabel) {
287 		var button = this._button[this._buttonInfo[0].id];
288 		button.setText((style == DwtChooser.SINGLE_SELECT) ? AjxMsg.select : AjxMsg.add);
289 	}
290 	if (!noResize) {
291 		var curSz = this.getSize();
292 		this.resize(curSz.x, curSz.y);
293 	}
294 	
295 	// "Add All" and "Remove All" buttons only shown if MULTI_SELECT
296 	if (this._allButtons) {
297 		this._addAllButton.setVisible(style == DwtChooser.MULTI_SELECT);
298 		this._removeAllButton.setVisible(style == DwtChooser.MULTI_SELECT);
299 		this._enableButtons();
300 	}
301 
302 	// if we're going from multi to single, preserve only the first target item
303 	if (style == DwtChooser.SINGLE_SELECT) {
304 		var list = this.targetListView.getList();
305 		var a = list ? list.clone().getArray() : null;
306 		if (a && a.length) {
307 			this._reset(DwtChooserListView.TARGET);
308 			this.addItems(a[0], DwtChooserListView.TARGET, true);
309 			this.targetListView.deselectAll();
310 			if (a.length > 1 && this._mode == DwtChooser.MODE_MOVE) {
311 				this.addItems(a.slice(1), DwtChooserListView.SOURCE, true);
312 			}
313 			this._enableButtons();
314 		}
315 	}
316 	
317 	this.sourceListView.multiSelectEnabled = (style == DwtChooser.MULTI_SELECT);
318 	this.targetListView.multiSelectEnabled = (style == DwtChooser.MULTI_SELECT);
319 };

321 /**
322  * Resets one or both list views, and the buttons. Defaults to resetting both list views.
323  *
324  * @param {DwtChooser.SINGLE_SELECT|DwtChooser.MULTI_SELECT}	style		the style single or multiple select
325  */
326 DwtChooser.prototype.reset =
327 function(view) {
328 	this._reset(view);
329 	this._setActiveButton(this._buttonInfo[0].id); // make first button active by default
330 	this._enableButtons();
331 	if (this._hasTextField) {
332 		this._textField.setValue("");
333 	}
334 };

336 /**
337  * Resets one or both list views. Defaults to resetting both list views.
338  *
339  * @param {DwtChooser.SINGLE_SELECT|DwtChooser.MULTI_SELECT}	style		the style single or multiple select
340  * 
341  * @private
342  */
343 DwtChooser.prototype._reset =
344 function(view) {
345 	// clear out source list view and related data
346 	if (!view || view == DwtChooserListView.SOURCE) {
347 		this.sourceListView._resetList();
348 	}
349 
350 	// clear out target list view and related data
351 	if (!view || view == DwtChooserListView.TARGET) {
352 		this.targetListView._resetList();
353 		for (var i in this._data) {
354 			this._data[i].removeAll();
355 		}
356 	}
357 };

419 /**
420  * Creates the HTML framework, with placeholders for elements which are created later.
421  * 
422  * @private
423  */
424 DwtChooser.prototype._createHtml = 
425 function() {
426 
427 	this._sourceListViewDivId	= Dwt.getNextId();
428 	this._targetListViewDivId	= Dwt.getNextId();
429 	this._buttonsDivId			= Dwt.getNextId();
430 	this._removeButtonDivId		= Dwt.getNextId();
431 	if (this._allButtons) {
432 		this._addAllButtonDivId		= Dwt.getNextId();
433 		this._removeAllButtonDivId	= Dwt.getNextId();
434 	}
435 	if (this._hasTextField) {
436 		this._textFieldTdId = Dwt.getNextId();
437 	}
438 
439 	var html = [];
440 	var idx = 0;
441 	
442 	if (this._layoutStyle == DwtChooser.HORIZ_STYLE) {
443 		// start new table for list views
444 		html[idx++] = "<table>";
445 		html[idx++] = "<tr>";
446 
447 		// source list
448 		html[idx++] = "<td id='";
449 		html[idx++] = this._sourceListViewDivId;
450 		html[idx++] = "'></td>";
451 
452 		// transfer buttons
453 		html[idx++] = "<td valign='middle' id='";
454 		html[idx++] = this._buttonsDivId;
455 		html[idx++] = "'>";
456 		if (this._allButtons) {
457 			html[idx++] = "<div id='";
458 			html[idx++] = this._addAllButtonDivId;
459 			html[idx++] = "'></div><br>";
460 		}
461 		for (var i = 0; i < this._buttonInfo.length; i++) {
462 			var id = this._buttonInfo[i].id;
463 			html[idx++] = "<div id='";
464 			html[idx++] = this._buttonDivId[id];
465 			html[idx++] = "'></div><br>";
466 		}
467 		// remove button
468 		html[idx++] = "<br><div id='";
469 		html[idx++] = this._removeButtonDivId;
470 		html[idx++] = "'></div>";
471 		if (this._allButtons) {
472 			html[idx++] = "<br><div id='";
473 			html[idx++] = this._removeAllButtonDivId;
474 			html[idx++] = "'></div><br>";
475 		}
476 		html[idx++] = "</td>";
477 
478 		// target list
479 		html[idx++] = "<td id='";
480 		html[idx++] = this._targetListViewDivId;
481 		html[idx++] = "'></td>";
482 
483 		html[idx++] = "</tr>";
484 		
485 		if (this._hasTextField) {
486 			html[idx++] = "<tr><td>";
487 			html[idx++] = "<table width=100%><tr><td style='white-space:nowrap; width:1%'>";
488 			html[idx++] = AjxMsg.add;
489 			html[idx++] = ":</td><td id='";
490 			html[idx++] = this._textFieldTdId;
491 			html[idx++] = "'></td></tr></table>";
492 			html[idx++] = "</td><td> </td><td> </td></tr>";
493 		}
494 
495 		html[idx++] = "</table>";
496 	} else {
497 		// source list
498 		html[idx++] = "<div id='";
499 		html[idx++] = this._sourceListViewDivId;
500 		html[idx++] = "'></div>";
501 
502 		// transfer buttons
503 		html[idx++] = "<div align='center' id='";
504 		html[idx++] = this._buttonsDivId;
505 		html[idx++] = "'>";
506 		html[idx++] = "<table class='ZPropertySheet' cellspacing='6'><tr>";
507 		if (this._allButtons) {
508 			html[idx++] = "<td id='";
509 			html[idx++] = this._addAllButtonDivId;
510 			html[idx++] = "'></td>";
511 		}
512 		for (var i = 0; i < this._buttonInfo.length; i++) {
513 			var id = this._buttonInfo[i].id;
514 			html[idx++] = "<td id='";
515 			html[idx++] = this._buttonDivId[id];
516 			html[idx++] = "'></td>";
517 		}
518 		// remove button
519 		html[idx++] = "<td id='";
520 		html[idx++] = this._removeButtonDivId;
521 		html[idx++] = "'></td>";
522 		if (this._allButtons) {
523 			html[idx++] = "<td id='";
524 			html[idx++] = this._removeAllButtonDivId;
525 			html[idx++] = "'></td>";
526 		}
527 		html[idx++] = "</tr></table></div>";
528 
529 		// target list
530 		html[idx++] = "<div id='";
531 		html[idx++] = this._targetListViewDivId;
532 		html[idx++] = "'></div>";
533 	}
534 
535 	this.getHtmlElement().innerHTML = html.join("");
536 };

538 /*
539 * Takes button info and sets up various bits of internal data for later use.
540 */
541 DwtChooser.prototype._handleButtonInfo =
542 function(buttonInfo) {
543 
544 	if (!buttonInfo) {
545 		this._defLabel = (this._selectStyle == DwtChooser.SINGLE_SELECT) ? AjxMsg.select : AjxMsg.add;
546 		buttonInfo = [ { label: this._defLabel } ];
547 	}
548 	this._buttonInfo = buttonInfo;
549 
550 	// create IDs for button elements and their containers
551 	this._buttonDivId = {};
552 	this._buttonId = {};
553 	if (this._buttonInfo.length == 1) {
554 		if (!this._buttonInfo[0].id) {
555 			this._buttonInfo[0].id = Dwt.getNextId("DwtChooserButtonInfo_");
556 		}
557 		this._activeButtonId = this._buttonInfo[0].id;
558 	}
559 	for (var i = 0; i < this._buttonInfo.length; i++) {
560 		var id = this._buttonInfo[i].id;
561 		this._buttonDivId[id] = Dwt.getNextId("DwtChooserButtonDiv_");
562 		this._buttonId[id] = Dwt.getNextId("DwtChooserButton_");
563 	}
564 	this._hasMultiButtons = (this._buttonInfo.length > 1);
565 };

567 /**
568  * Creates and places elements into the DOM.
569  * 
570  * @private
571  */
572 DwtChooser.prototype._initialize =
573 function() {
574 
575 	// create and add transfer buttons
576 	var buttonListener = new AjxListener(this, this._transferButtonListener);
577 	this._button = {};
578 	this._buttonIndex = {};
579 	this._data = {};
580 	for (var i = 0; i < this._buttonInfo.length; i++) {
581 		var id = this._buttonInfo[i].id;
582 		this._button[id] = this._setupButton(id, this._buttonId[id], this._buttonDivId[id], this._buttonInfo[i].label);
583 		this._button[id].addSelectionListener(buttonListener);
584 		this._buttonIndex[id] = i;
585 		this._data[id] = new AjxVector();
586 	}
587 
588 	// create and add source list view
589 	this.sourceListView = this._createSourceListView();
590 	this._addListView(this.sourceListView, this._sourceListViewDivId);
591 	this.sourceListView.addSelectionListener(new AjxListener(this, this._sourceListener));
592 
593 	// create and add target list view
594 	this.targetListView = this._createTargetListView();
595 	this._addListView(this.targetListView, this._targetListViewDivId);
596 	this.targetListView.addSelectionListener(new AjxListener(this, this._targetListener));
597

599 	this._removeButtonId = Dwt.getNextId("DwtChooserRemoveButton_");
600 	this._removeButton = this._setupButton(DwtChooser.REMOVE_BTN_ID, this._removeButtonId, this._removeButtonDivId, AjxMsg.remove);
601 	this._removeButton.addSelectionListener(new AjxListener(this, this._removeButtonListener));
602 
603 	if (this._allButtons) {
604 		// create and add "Add All" and "Remove All" buttons
605 		this._addAllButtonId = Dwt.getNextId();
606 		this._addAllButton = this._setupButton(DwtChooser.ADD_ALL_BTN_ID, this._addAllButtonId, this._addAllButtonDivId, AjxMsg.addAll);
607 		this._addAllButton.addSelectionListener(new AjxListener(this, this._addAllButtonListener));
608 		this._removeAllButtonId = Dwt.getNextId();
609 		this._removeAllButton = this._setupButton(DwtChooser.REMOVE_ALL_BTN_ID, this._removeAllButtonId, this._removeAllButtonDivId, AjxMsg.removeAll);
610 		this._removeAllButton.addSelectionListener(new AjxListener(this, this._removeAllButtonListener));
611 		if (this._selectStyle == DwtChooser.SINGLE_SELECT) {
612 			this._addAllButton.setVisible(false);
613 			this._removeAllButton.setVisible(false);
614 		}
615 	}
616 
617 	if (this._hasTextField) {
618 		var params = {parent: this, type: DwtInputField.STRING};
619 		this._textField = new DwtInputField(params);
620 		this._textField.reparentHtmlElement(this._textFieldTdId);
621 		this._textField._chooser = this;
622 		this._textField.setHandler(DwtEvent.ONKEYUP, DwtChooser._onKeyUp);
623 		Dwt.setSize(this._textField.getInputElement(), "100%", Dwt.DEFAULT);
624 	}
625 
626 	if (this._selectStyle == DwtChooser.SINGLE_SELECT) {
627 		this.sourceListView.multiSelectEnabled = false;
628 		this.targetListView.multiSelectEnabled = false;
629 	}
630 };

936  * @private

938 DwtChooser.prototype._addToSource =

941 	if (!item._chooserIndex) {
942 		var list = this.sourceListView.getList();

948 /**
949  * Adds an item to the target list. If there are multiple transfer buttons, it keeps
950  * the items grouped depending on which button was used to move them.
951  *
952  * @param {Object}	item		the item to add
953  * @param {string}	id			the ID of the transfer button that was used
954  * @param {boolean}	skipNotify	if <code>true</code>, don't notify listeners
955  * 
956  * @private
957  */
958 DwtChooser.prototype._addToTarget =
959 function(item, id, skipNotify) {
960 	if (!item) return;
961 	id = id ? id : this._activeButtonId;
962 	if (this._noDuplicates && this._data[id] && this._isDuplicate(item, this._data[id])) {
963 		return;
964 	}
965 
966 	// item is being added to target list with multiple transfer buttons,
967 	// so we need to clone it on second and subsequent transfers
968 	var list = this.targetListView.getList();
969 	if (list && list.contains(item) && item.clone) {
970 		var newItem = item.clone();
971 		newItem.id = Dwt.getNextId();
972 		item = newItem;
973 	}
974 
975 	var idx = null;
976 	if (this._hasMultiButtons) {
977 		// get a list of all the items in order
978 		list = [];
979 		for (var i = 0; i < this._buttonInfo.length; i++) {
980 			list = list.concat(this._data[this._buttonInfo[i].id].getArray());
981 		}
982 		// find the first item with a higher button index
983 		var buttonIdx = this._buttonIndex[id];
984 		for (idx = 0; idx < list.length; idx++) {
985 			var testButtonIdx = this._buttonIndex[list[idx]._buttonId];
986 			if (testButtonIdx > buttonIdx) {
987 				break;
988 			}
989 		}
990 	}
991 
992 	item._buttonId = id;
993 	item.id = item.id || Dwt.getNextId();
994 	this._data[id].add(item);
995 	this.targetListView.addItem(item, idx, skipNotify);
996 };
