 22 /**
 23  * Creates a search controller.
 24  * @class
 25  * This class represents the search controller.
 26  * 
 27  * @param {DwtControl}	container	the top-level container
 28  * @extends	ZmController
 29  */
 30 ZmSearchController = function(container) {
 31 
 32 	ZmController.call(this, container);
 33 
 34 	this._inited = false;
 35 	this._contactSource = ZmItem.CONTACT;
 36 	this._results = null;
 37 
 38 	if (appCtxt.get(ZmSetting.SEARCH_ENABLED)) {
 39 		this._setView();
 40 	}
 41 };

 63 /**
 64  * Performs a search by date.
 65  * 
 66  * @param	{Date}	d		the date or <code>d</code> for now
 67  * @param	{String}	searchFor	the search for string
 68  */
 69 ZmSearchController.prototype.dateSearch =
 70 function(d, searchFor) {
 71 	d = d || new Date();
 72     var formatter = AjxDateFormat.getDateInstance(AjxDateFormat.SHORT);
 73     var date = formatter.format(d);
 74 	var groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
 75 	var query = "date:" + date;
 76 	this.search({
 77 		query:			query,
 78 		types:			[groupBy],
 79 		searchFor:		searchFor,
 80 		origin:			ZmId.SEARCH,
 81 		userInitiated:	true
 82 	});
 83 };

 85 /**
 86  * Performs a search by from address.
 87  * 
 88  * @param	{String}	address		the from address
 89  */
 90 ZmSearchController.prototype.fromSearch =
 91 function(address) {
 92 
 93 	// always search for mail when doing a "from: <address>" search
 94 	var groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
 95 	var terms = AjxUtil.map(AjxUtil.toArray(address), function(addr) {
 96 		return "from:" + ((addr && addr.isAjxEmailAddress) ? addr.getAddress() : addr);
 97 	});
 98 	
 99 	this.search({
100 		query:			terms.join(" OR "),
101 		types:			[groupBy],
102 		origin:			ZmId.SEARCH,
103 		userInitiated:	true
104 	});
105 };

108 /**
109  * Performs a search by to address.
110  *
111  * @param	{String}	address		the to address
112  */
113 ZmSearchController.prototype.toSearch =
114 function(address) {
115 
116 	// always search for mail when doing a "tocc: <address>" search
117 	var groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
118 	var terms = AjxUtil.map(AjxUtil.toArray(address), function(addr) {
119 		return "tocc:" + ((addr && addr.isAjxEmailAddress) ? addr.getAddress() : addr);
120 	});
121 
122 	var params = {
123 		types:			[groupBy],
124 		origin:			ZmId.SEARCH,
125 		userInitiated:	true,
126 		query:			terms.join(" OR ")
127 	}
128     if (this.currentSearch && this.currentSearch.folderId == ZmFolder.ID_SENT) {
129 		if (terms.length > 1) {
130 			params.query = "(" + params.query + ")";
131 		}
132 		params.query = "in:sent AND " + params.query;
133 	}
134     this.search(params);
135 };

237 /**
238  * Performs a search and displays the results.
239  *
240  * @param {Hash}	params		a hash of parameters:
241  * 
242  * @param {String}		query						the search string
243  * @param {constant}	searchFor					the semantic type to search for
244  * @param {Array}		types						the item types to search for
245  * @param {constant}	sortBy						the sort constraint
246  * @param {int}			offset						the starting point in list of matching items
247  * @param {int}			limit						the maximum number of items to return
248  * @param {int}			searchId					the ID of owning search folder (if any)
249  * @param {Boolean}		noRender					if <code>true</code>, results will not be passed to controller
250  * @param {Boolean}		userText					if <code>true</code>, text was typed by user into search box
251  * @param {AjxCallback}	callback					the async callback
252  * @param {AjxCallback}	errorCallback				the async callback to run if there is an exception
253  * @param {Object}		response					the canned JSON response (no request will be made)
254  * @param {boolean}		skipUpdateSearchToolbar     don't update the search toolbar (e.g. from the ZmDumpsterDialog where the search is called from its own search toolbar
255  * @param {string}		origin						indicates what initiated the search
256  * @param {string}		sessionId					session ID of search results tab (if search came from one)
257  * @param {Boolean}		noGal     if true, don't search GAL. This is to override the this._contactSource value in contacts search, specifically for clicking on TAGS. 
258  *
259  */
260 ZmSearchController.prototype.search =
261 function(params) {
262 
263 	// if the search string starts with "$set:" then it is a command to the client
264 	if (params.query && (params.query.indexOf("$set:") == 0 || params.query.indexOf("$cmd:") == 0)) {
265 		appCtxt.getClientCmdHandler().execute((params.query.substr(5)), this);
266 		return;
267 	}
268 
269 	params.searchAllAccounts = this.searchAllAccounts;
270 	var respCallback = this._handleResponseSearch.bind(this, params.callback);
271 	this._doSearch(params, params.noRender, respCallback, params.errorCallback);
272 };

279 	if (callback) {
280 		callback.run(result);
281 	}
282 };

284 /**
285  * Performs the given search. It takes a ZmSearch, rather than constructing one out of the currently selected menu
286  * choices. Aside from re-executing a search, it can be used to perform a canned search.
287  *
288  * @param {ZmSearch}	search		the search object
289  * @param {Boolean}		noRender		if <code>true</code>, results will not be passed to controller
290  * @param {Object}	changes		the hash of changes to make to search
291  * @param {AjxCallback}	callback		the async callback
292  * @param {AjxCallback}	errorCallback	the async callback to run if there is an exception
293  */
294 ZmSearchController.prototype.redoSearch =
295 function(search, noRender, changes, callback, errorCallback) {
296 
297 	var params = {};
298 	params.query		= search.query;
299 	params.queryHint	= search.queryHint;
300 	params.types		= search.types;
301 	params.sortBy		= search.sortBy;
302 	params.offset		= search.offset;
303 	params.limit		= search.limit;
304 	params.fetch		= search.fetch;
305 	params.searchId		= search.searchId;
306 	params.lastSortVal	= search.lastSortVal;
307 	params.endSortVal	= search.endSortVal;
308 	params.lastId		= search.lastId;
309 	params.soapInfo		= search.soapInfo;
310 	params.accountName	= search.accountName;
311 	params.searchFor	= this._searchFor;
312 	params.idsOnly		= search.idsOnly;
313 	params.inDumpster   = search.inDumpster;
314 	params.userInitiated = search.userInitiated;
315 	params.sessionId	= search.sessionId;
316 
317 	if (changes) {
318 		for (var key in changes) {
319 			params[key] = changes[key];
320 		}
321 	}
322 
323 	this._doSearch(params, noRender, callback, errorCallback);
324 };

362 /**
363  * Gets the item types. Assembles a list of item types to return based on a search menu value (which can
364  * be passed in).
365  *
366  * @param {Hash}	params			a hash of arguments for the search
367  * @param {String}	searchFor		general description of what to search for
368  * @return	{AjxVector}		a list of types
369  * 
370  * @see		#search
371  */
372 ZmSearchController.prototype.getTypes =
373 function(params) {
374 	var types = new AjxVector();
375 	var searchFor = params.searchFor || this._searchFor;
376 
377 	var groupBy;
378 	if ((searchFor == ZmId.SEARCH_MAIL) && appCtxt.get(ZmSetting.MAIL_ENABLED))	{
379 		groupBy = appCtxt.getApp(ZmApp.MAIL).getGroupMailBy();
380 	}
381 
382 	if (searchFor == ZmId.SEARCH_MAIL) {
383 		types.add(groupBy);
384 	} else {
385 		types.add(searchFor);
386 	}
387 
388 	return types;
389 };

391 /**
392  * Selects the appropriate item in the overview based on the search. Selection only happens
393  * if the search was a simple search for a folder, tag, or saved search. A check is done to
394  * make sure that item is not already selected, so selection should only occur for a query
395  * manually run by the user.
396  *
397  * @param {ZmSearch}	searchObj		the current search
398  */
399 ZmSearchController.prototype.updateOverview =
400 function(searchObj) {
401 	var search = searchObj || appCtxt.getCurrentSearch();
402 
403 	var id, type;
404 	if (search && (search.isSimple() || search.searchId)) {
405 		if (search.searchId) {
406 			id = this._getNormalizedId(search.searchId);
407 			type = ZmOrganizer.SEARCH;
408 		} else if (search.folderId) {
409 			id = this._getNormalizedId(search.folderId);
410 			var folderTree = appCtxt.getFolderTree();
411 			var folder = folderTree && folderTree.getById(id);
412 			type = folder ? folder.type : ZmOrganizer.FOLDER;
413             if (search.searchFor == ZmItem.TASK) {
414                 type = ZmOrganizer.TASKS;
415             }
416 		} else if (search.tagId) {
417 			id = this._getNormalizedId(search.tagId);
418 			type = ZmOrganizer.TAG;
419 		}
420 		if (type) {
421 			var app = appCtxt.getCurrentApp();
422 			var overview = app && app.getOverview();
423 			if (overview) {
424 				overview.setSelected(id, type);
425 			}
426 		}
427 	}
428 };

430 /**
431  * @private
432  */
433 ZmSearchController.prototype._getSuitableSortBy =
434 function(types) {
435 	var sortBy;
436 
437 	if (types.size() == 1) {
438 		var type = types.get(0);
439 		var viewType;
440 		switch (type) {
441 			case ZmItem.CONV:		viewType = ZmId.VIEW_CONVLIST; break;
442 			case ZmItem.MSG:		viewType = ZmId.VIEW_TRAD; break;
443 			case ZmItem.CONTACT:	viewType = ZmId.VIEW_CONTACT_SIMPLE; break;
444 			case ZmItem.APPT:		viewType = ZmId.VIEW_CAL; break;
445 			case ZmItem.TASK:		viewType = ZmId.VIEW_TASKLIST; break;
446 			case ZmId.SEARCH_GAL:	viewType = ZmId.VIEW_CONTACT_SIMPLE; break;
447 			case ZmItem.BRIEFCASE_ITEM:	viewType = ZmId.VIEW_BRIEFCASE_DETAIL; break;
448 			// more types go here as they are suported...
449 		}
450 
451 		if (viewType) {
452 			sortBy = appCtxt.get(ZmSetting.SORTING_PREF, viewType);
453 		}
454         //bug:1108 & 43789#c19 (changelist 290073) since sort-by-[RCPT|ATTACHMENT|FLAG|PRIORITY] gives exception with querystring.
455         // Avoided [RCPT|ATTACHMENT|FLAG|PRIORITY] sorting with querysting instead used date sorting
456         var queryString = this._searchToolBar.getSearchFieldValue();
457         if(queryString && queryString.length > 0)
458         {
459             if((sortBy == ZmSearch.RCPT_ASC || sortBy == ZmSearch.RCPT_DESC)) {
460                sortBy = (sortBy == ZmSearch.RCPT_ASC) ?  ZmSearch.DATE_ASC : ZmSearch.DATE_DESC;
461             } else if((sortBy == ZmSearch.FLAG_ASC || sortBy == ZmSearch.FLAG_DESC)) {
462                sortBy = (sortBy == ZmSearch.FLAG_ASC) ?  ZmSearch.DATE_ASC : ZmSearch.DATE_DESC;
463             } else if((sortBy == ZmSearch.ATTACH_ASC || sortBy == ZmSearch.ATTACH_DESC)) {
464                sortBy = (sortBy == ZmSearch.ATTACH_ASC) ?  ZmSearch.DATE_ASC : ZmSearch.DATE_DESC;
465             } else if((sortBy == ZmSearch.PRIORITY_ASC || sortBy == ZmSearch.PRIORITY_DESC)) {
466                sortBy = (sortBy == ZmSearch.PRIORITY_ASC) ?  ZmSearch.DATE_ASC : ZmSearch.DATE_DESC;
467             }
468         }
469 	}
470 
471 	return sortBy;
472 };

474 /**
475  * Performs the search.
476  *
477  * @param {Hash}	params		a hash of params for the search
478  * @param {String}	params.searchFor	the search for
479  * @param {String}	params.query	the search query
480  * @param {String}	params.userText	the user text
481  * @param {Array}	params.type		an array of types
482  * @param {boolean} params.forceSearch     Ignores special processing and just executes the search.
483  * @param {Boolean}	noRender		if <code>true</code>, the search results will not be rendered
484  * @param {AjxCallback}	callback		the callback
485  * @param {AjxCallback}	errorCallback	the error callback
486  * @param {boolean} params.skipUpdateSearchToolbar     don't update the search toolbar (e.g. from the ZmDumpsterDialog where the search is called from its own search toolbar
487  * @param {Boolean}		noGal     if true, don't search GAL. This is to override the this._contactSource value in contacts search, specifically for clicking on TAGS.
488  *
489  * @see	#search
490  * 
491  * @private
492  */
493 ZmSearchController.prototype._doSearch =
494 function(params, noRender, callback, errorCallback) {
495 
496 	var searchFor = this._searchFor = params.searchFor || this._searchFor || ZmSearchToolBar.MENU_ITEMS[0];
497 	appCtxt.notifyZimlets("onSearch", [params.query]);
498 
499 	if (!params.skipUpdateSearchToolbar && this._searchToolBar) {
500 		var value = (appCtxt.get(ZmSetting.SHOW_SEARCH_STRING) || params.userText)
501 			? params.query : null;
502 		this._searchToolBar.setSearchFieldValue(value || "");
503 
504 		// bug: 42512 - deselect global inbox if searching via search toolbar
505 		if (appCtxt.multiAccounts && params.userText && this.searchAllAccounts) {
506 			appCtxt.getCurrentApp().getOverviewContainer().deselectAll();
507 		}
508 	}
509 
510 	// get types from search type if not passed in explicitly
511 	var types = params.types;
512 
513 	if (types && !types.isAjxVector) { // change a scalar type to array. (unless it's a vector)
514 		types = AjxUtil.toArray(types);
515 	}
516 
517 	if (!types || (types.length == 0) || (types.isAjxVector && types.size() == 0)) {
518 		types = (params.skipUpdateSearchToolbar && searchFor == ZmId.SEARCH_MAIL && this._prevTypes) || this.getTypes(params);
519 	}
520 
521 	if (types instanceof Array) { // convert array to AjxVector if necessary
522 		types = AjxVector.fromArray(types);
523 	}
524 	if (searchFor == ZmId.SEARCH_MAIL) {
525         this._prevTypes = params.types || this._prevTypes; // Saves previous user action, if user selects view
526 		params = appCtxt.getApp(ZmApp.MAIL).getSearchParams(params);
527 	}
528 
529 	if (searchFor == ZmItem.TASK) {
530 		var tlc = AjxDispatcher.run("GetTaskListController");
531 		params.allowableTaskStatus = tlc && tlc.getAllowableTaskStatus();
532 	}
533 
534 	if (params.searchAllAccounts && !params.queryHint) {
535 		params.queryHint = appCtxt.accountList.generateQuery(null, types);
536 		params.accountName = appCtxt.accountList.mainAccount.name;
537 	}
538 	else if (this._inclSharedItems) {
539 		// a query hint is part of the query that the user does not see
540 		params.queryHint = ZmSearchController.generateQueryForShares(types.getArray());
541 	}
542 
543 	// only set contact source if we are searching for contacts
544 	params.contactSource = !params.noGal && (types.contains(ZmItem.CONTACT) || types.contains(ZmId.SEARCH_GAL))
545 		? this._contactSource : null;
546 	if (params.contactSource == ZmId.SEARCH_GAL) {
547 		params.expandDL = true;
548 	}
549 
550 	// find suitable sort by value if not given one (and if applicable)
551 	params.sortBy = params.sortBy || this._getSuitableSortBy(types);
552 	params.types = types;
553 	var search = new ZmSearch(params);
554 	
555 	// force outbound folder into msg view
556 	if (searchFor == ZmId.SEARCH_MAIL && !params.isViewSwitch && search.folderId) {
557 		var folder = appCtxt.getById(search.folderId);
558 		if (folder && folder.isOutbound()) {
559 			search.types = new AjxVector([ZmItem.MSG]);
560 			search.isOutboundFolder = true;
561 		}
562 	}
563 
564 	var respCallback = this._handleResponseDoSearch.bind(this, search, noRender, callback, params.noUpdateOverview);
565 	if (!errorCallback) {
566 		errorCallback = this._handleErrorDoSearch.bind(this, search);
567 	}
568 
569 	// calendar searching is special so hand it off if necessary
570 	search.calController = null;
571 	if (searchFor == ZmItem.APPT && !params.forceSearch && !params.inDumpster) {
572 		var searchResultsController, sessionId;
573 		if (search.userInitiated && ZmApp.SEARCH_RESULTS_TAB[ZmApp.CALENDAR]) {
574 			searchResultsController = appCtxt.getApp(ZmApp.SEARCH).getSearchResultsController(search.sessionId, ZmApp.CALENDAR);
575 			sessionId = searchResultsController.getCurrentViewId();
576 		}
577 		var controller = AjxDispatcher.run("GetCalController", sessionId, searchResultsController);
578 		if (controller && types.contains(ZmItem.APPT)) {
579 			search.calController = controller;
580 			controller.handleUserSearch(params, respCallback);
581 		} else {
582             search.execute({callback:respCallback, errorCallback:errorCallback});            
583         }
584 	} else {
585 		search.execute({callback:respCallback, errorCallback:errorCallback});
586 	}
587 };

589 /**
590  * Takes the search result and hands it to the appropriate controller for display.
591  *
592  * @param {ZmSearch}	search			contains search info used to run search against server
593  * @param {Boolean}		noRender		<code>true</code> to skip rendering results
594  * @param {AjxCallback}	callback		the callback to run after processing search response
595  * @param {Boolean}	noUpdateOverview	<code>true</code> to skip updating the overview
596  * @param {ZmCsfeResult}	result			the search results
597  */
598 ZmSearchController.prototype._handleResponseDoSearch =
599 function(search, noRender, callback, noUpdateOverview, result) {
600 
601 	DBG.println("s", "SEARCH was user initiated: " + Boolean(search.userInitiated));
602 	var results = result && result.getResponse();
603 	if (!results) { return; }
604 
605 	if (!results.type) {
606 		results.type = search.types.get(0);
607 	}
608 
609 	this.currentSearch = search;
610 	DBG.timePt("execute search", true);
611 
612 	if (!noRender) {
613 		this._showResults(results, search, noUpdateOverview);
614 	}
615 
616 	if (callback) {
617 		callback.run(result);
618 	}
619 };

621 /**
622  * @private
623  */
624 ZmSearchController.prototype._showResults =
625 function(results, search, noUpdateOverview) {
626 
627 	this._results = results = (results && results.isZmSearchResult) ? results : new ZmSearchResult(search);
628 
629 	DBG.timePt("handle search results");
630 
631 	if (appCtxt.get(ZmSetting.SAVED_SEARCHES_ENABLED)) {
632 		var saveBtn = this._searchToolBar && this._searchToolBar.getButton(ZmSearchToolBar.SAVE_BUTTON);
633 		if (saveBtn) {
634 			saveBtn.setEnabled(this._contactSource != ZmId.SEARCH_GAL);
635 		}
636 	}
637 
638 	var app = search.calController ? appCtxt.getApp(ZmApp.CALENDAR) : appCtxt.getApp(ZmItem.APP[results.type]) || appCtxt.getCurrentApp();
639 	var appName = app.getName();
640 	if (search.userInitiated && ZmApp.SEARCH_RESULTS_TAB[appName]) {
641 		var ctlr = (search.calController && search.calController.searchResultsController) ||
642 					appCtxt.getApp(ZmApp.SEARCH).getSearchResultsController(search.sessionId, appName);
643 		DBG.println("sr", "New search results controller: " + ctlr.viewId);
644 		ctlr.show(results, search.calController);
645 		this._searchToolBar.setSearchFieldValue("");
646 	}
647 	else if (app.showSearchResults) {
648 		// show results based on type - may invoke package load
649 		var loadCallback = this._handleLoadShowResults.bind(this, results, search, noUpdateOverview);
650 		app.currentSearch = search;
651 		app.currentQuery = search.query;
652 		app.showSearchResults(results, loadCallback);
653 	}
654 };

668 /**
669  * Handle a few minor errors where we show an empty result set and issue a
670  * status message to indicate why the query failed. Those errors are: no such
671  * folder, no such tag, and bad query. If it's a "no such folder" error caused
672  * by the deletion of a folder backing a mountpoint, we pass it along for
673  * special handling by ZmZimbraMail.
674  * 
675  * @private
676  */
677 ZmSearchController.prototype._handleErrorDoSearch =
678 function(search, ex) {
679 	DBG.println(AjxDebug.DBG1, "Search exception: " + ex.code);
680 
681 	if (ex.code == ZmCsfeException.MAIL_NO_SUCH_TAG ||
682 		ex.code == ZmCsfeException.MAIL_QUERY_PARSE_ERROR ||
683 		ex.code == ZmCsfeException.MAIL_TOO_MANY_TERMS ||
684 		(ex.code == ZmCsfeException.MAIL_NO_SUCH_FOLDER && !(ex.data.itemId && ex.data.itemId.length)))
685 	{
686 		var msg = ex.getErrorMsg();
687 		appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_WARNING);
688 		var results = new ZmSearchResult(search);
689 		results.type = search.types ? search.types.get(0) : null;
690 		this._showResults(results, search);
691 		return true;
692 	}
693 	return false;
694 };

696 /**
697  * Provides a string to add to the query when the search includes shared items.
698  * 
699  * @param types		[array]		list of item types
700  * 
701  * @private
702  */
703 ZmSearchController.generateQueryForShares =
704 function(types, account) {
705 	var ac = window.parentAppCtxt || window.appCtxt;
706 	var list = [];
707 	for (var j = 0; j < types.length; j++) {
708 		var type = types[j];
709 		var app = ac.getApp(ZmItem.APP[type]);
710 		if (app) {
711 			var ids = app.getRemoteFolderIds(account);
712 			for (var i = 0; i < ids.length; i++) {
713 				var id = ids[i];
714 				var idText = AjxUtil.isNumeric(id) ? id : ['"', id, '"'].join("");
715 				list.push("inid:" + idText);
716 			}
717 		}
718 	}
719 
720 	if (list.length > 0) {
721 		list.push("is:local");
722 		return list.join(" OR ");
723 	}
724 
725 	return null;
726 };

738 /**
739  * Runs a search based on the state of the toolbar.
740  * 
741  * @param {Hash}	params		a hash of parameters:
742  * 
743  * @param {Event}		ev							browser event	
744  * @param {string}		zimletEvent					type of notification to send zimlets
745  * @param {string}		query						search string (optional, overrides input field)
746  * @param {string}		origin						indicates what initiated the search
747  * @param {string}		sessionId					session ID of search results tab (if search came from one)
748  * @param {boolean}		skipUpdateSearchToolbar     don't update the search toolbar (e.g. from the ZmDumpsterDialog where the search is called from its own search toolbar
749  * 
750  * @private
751  */
752 ZmSearchController.prototype._toolbarSearch =
753 function(params) {
754 
755 	// find out if the custom search menu item is selected and pass it the event
756 	var result = params.searchFor || this._searchToolBar.getSearchType();
757 	if (result && result.listener) {
758 		result.listener.run(params.ev);
759 	} else {
760 		var queryString = params.query || this._searchToolBar.getSearchFieldValue();
761 		var userText = (queryString.length > 0);
762 		if (queryString) {
763 			this._currentQuery = null;
764 		} else {
765 			queryString = this._currentQuery || "";
766 		}
767 
768 		appCtxt.notifyZimlets(params.zimletEvent, [queryString]);
769 		var searchParams = {
770 			query:						queryString,
771 			userText:					userText,
772 			userInitiated:				true,
773 			getHtml:					appCtxt.get(ZmSetting.VIEW_AS_HTML),
774 			searchFor:					result,
775 			skipUpdateSearchToolbar:	params.skipUpdateSearchToolbar,
776 			origin:						params.origin,
777 			sessionId:					params.sessionId,
778 			errorCallback:				params.errorCallback,
779 			isEmpty:					params.isEmpty || !queryString
780 		};
781 		this.search(searchParams);
782 	}
783 };

802 /**
803  * @private
804  */
805 ZmSearchController.prototype._searchMenuListener =
806 function(ev, id, noFocus) {
807 	var btn = this._searchToolBar.getButton(ZmSearchToolBar.TYPES_BUTTON);
808 	if (!btn) { return; }
809 
810 	var menu = btn.getMenu();
811 	var item = ev ? ev.item : (menu.getItemById(ZmSearchToolBar.MENUITEM_ID, id));
812 
813 	if (!item || (!!(item._style & DwtMenuItem.SEPARATOR_STYLE))) { return; }
814 	id = item.getData(ZmSearchToolBar.MENUITEM_ID);
815 
816 	var selItem = menu.getSelectedItem();
817 	var sharedMI = menu.getItemById(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_SHARED);
818 
819 	this._searchToolBar.setPeopleAutocomplete(id);
820 
821 	// enable shared menu item if not a gal search
822 	if (id == ZmId.SEARCH_GAL) {
823 		this._contactSource = ZmId.SEARCH_GAL;
824 		if (sharedMI) {
825 			sharedMI.setChecked(false, true);
826 			sharedMI.setEnabled(false);
827 		}
828 	} else {
829 		if (sharedMI) {
830 			// we allow user to check "Shared Items" for appointments since it
831 			// is based on whats checked in their tree view
832 			if (id == ZmItem.APPT || id == ZmId.SEARCH_CUSTOM) {
833 				sharedMI.setChecked(false, true);
834 				sharedMI.setEnabled(false);
835 			} else {
836 				sharedMI.setEnabled(true);
837 			}
838 		}
839 		this._contactSource = ZmItem.CONTACT;
840 	}
841 	this._inclSharedItems = sharedMI && sharedMI.getChecked();
842 
843 	// search all accounts? Only applies to multi-account mbox
844 	var allAccountsMI = menu.getItemById(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_ALL_ACCOUNTS);
845 	if (allAccountsMI) {
846 		if (id == ZmItem.APPT) {
847 			this.resetSearchAllAccounts();
848 			allAccountsMI.setEnabled(false);
849 		} else {
850 			allAccountsMI.setEnabled(true);
851 			this.searchAllAccounts = allAccountsMI && allAccountsMI.getChecked();
852 		}
853 	}
854 
855 	if (id == ZmId.SEARCH_SHARED) {
856 		var icon = this.searchAllAccounts
857 			? allAccountsMI.getImage() : selItem.getImage();
858 
859 		if (this._inclSharedItems) {
860 			icon = this._getSharedImage(selItem);
861 		}
862 
863 		btn.setImage(icon);
864 	}
865 	else if (id == ZmId.SEARCH_ALL_ACCOUNTS) {
866 		var icon = (this.searchAllAccounts && !this._inclSharedItems)
867 			? item.getImage()
868 			: (this._inclSharedItems) ? this._getSharedImage(selItem) : selItem.getImage();
869 		btn.setImage(icon);
870 	}
871 	else {
872 		// only set search for if a "real" search-type menu item was clicked
873 		this._searchFor = id;
874 		var icon = item.getImage();
875 
876 		if (this._inclSharedItems) {
877 			icon = this._getSharedImage(selItem);
878 		}
879 		else if (this.searchAllAccounts) {
880 			icon = allAccountsMI.getImage();
881 		}
882 
883 		btn.setImage(icon);
884 	}
885 
886 	// set button tooltip
887 	var tooltip = ZmMsg[ZmSearchToolBar.TT_MSG_KEY[id]];
888 	if (id != ZmId.SEARCH_SHARED) { 
889 		btn.setToolTipContent(tooltip);
890 	}
891 	
892 	if (!noFocus) {
893 		// restore focus to INPUT if user changed type
894 		setTimeout(this._searchToolBar.focus.bind(this._searchToolBar), 10);
895 	}
896 };
