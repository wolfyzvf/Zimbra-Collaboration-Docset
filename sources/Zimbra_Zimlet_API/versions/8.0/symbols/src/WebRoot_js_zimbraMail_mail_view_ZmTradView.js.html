 16 ZmTradView = function(params) {
 17 
 18 	params.className = params.className || "ZmTradView";
 19 	params.mode = ZmId.VIEW_TRAD;
 20 	ZmDoublePaneView.call(this, params);
 21 }
 22 
 23 ZmTradView.prototype = new ZmDoublePaneView;
 24 ZmTradView.prototype.constructor = ZmTradView;
 25 
 26 ZmTradView.prototype.isZmTradView = true;
 27 ZmTradView.prototype.toString = function() { return "ZmTradView"; };
 28 
 29 ZmTradView.prototype._createMailListView =
 30 function(params) {
 31 	return new ZmMailMsgListView(params);
 32 };
 33 
 34 ZmTradView.prototype._createMailItemView =
 35 function(params) {
 36 	params.id = ZmId.getViewId(ZmId.VIEW_MSG, null, params.view);
 37 	return new ZmMailMsgView(params);
 38 };
