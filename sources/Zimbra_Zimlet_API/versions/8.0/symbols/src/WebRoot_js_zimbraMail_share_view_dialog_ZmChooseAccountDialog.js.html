 16 /**
 17  * Creates a dialog with a list of visible accounts to choose from
 18  * @class
 19  * This class represents choose account dialog.
 20  *
 21  * @param	{DwtControl}	parent		the parent, usually the global DwtShell
 22  *
 23  * @extends		ZmDialog
 24  */
 25 ZmChooseAccountDialog = function(parent) {
 26 	ZmDialog.call(this, {parent:parent});
 27 	this._createControls();
 28 };
 29 
 30 ZmChooseAccountDialog.prototype = new ZmDialog;
 31 ZmChooseAccountDialog.prototype.constructor = ZmChooseAccountDialog;
 32 
 33 ZmChooseAccountDialog.prototype.toString =
 34 function() {
 35 	return "ZmChooseAccountDialog";
 36 };
 37 
 38 /**
 39  * Pops up this dialog
 40  *
 41  * @param selectedAccount	{ZmZimbraAccount}	Optional. The account to have initially "selected". Otherwise, the active account is selected.
 42  * @param accountType		{String}			Optional. Only offer accounts of this type. Otherwise, all visible accounts are offered.
 43  * @param chooserMessage	{String}			Optional. The message to prompt user with. A default message is used if none provided.
 44  * @param title				{String}			Optional. Dialog title.
 45  */
 46 ZmChooseAccountDialog.prototype.popup =
 47 function(selectedAccount, accountType, chooserMessage, title) {
 48 	this.setTitle(title || ZmMsg.chooseAccount);
 49 
 50 	this._chooseMessageEl.innerHTML = chooserMessage || ZmMsg.chooseAccount;
 51 
 52 	var activeAcct = selectedAccount || appCtxt.getActiveAccount();
 53 	var accounts = appCtxt.accountList.visibleAccounts;
 54 
 55 	var html = [];
 56 	var idx = 0;
 57 
 58 	html[idx++] = "<table border=0 cellpadding=1 cellspacing=1>";
 59 	for (var i = 0; i < accounts.length; i++) {
 60 		var acct = accounts[i];
 61 		if (appCtxt.isOffline && acct.isMain) { continue; }
 62 		if (accountType && acct.type != accountType) { continue; }
 63 
 64 		var icon = appCtxt.isOffline ? acct.getIcon() : null;
 65 		var inputId = Dwt.getNextId();
 66 
 67 		html[idx++] = "<tr><td><input type='checkbox' name='";
 68 		html[idx++] = this._inputName;
 69 		html[idx++] = "'";
 70 		if (acct == activeAcct) {
 71 			html[idx++] = " checked";
 72 		}
 73 		html[idx++] = " _acctId='";
 74 		html[idx++] = acct.id;
 75 		html[idx++] = "' id='";
 76 		html[idx++] = inputId;
 77 		html[idx++] = "'></td>";
 78 		if (icon) {
 79 			html[idx++] = "<td>";
 80 			html[idx++] = AjxImg.getImageHtml(icon);
 81 			html[idx++] = "</td>";
 82 		}
 83 		html[idx++] = "<td><label for='";
 84 		html[idx++] = inputId;
 85 		html[idx++] = "'>";
 86 		html[idx++] = acct.getDisplayName();
 87 		html[idx++] = "</label></td></tr>";
 88 	}
 89 	html[idx++] = "</table>";
 90 	this._accountSelectEl.innerHTML = html.join("");
 91 
 92 	ZmDialog.prototype.popup.call(this);
 93 };
 94 
 95 ZmChooseAccountDialog.prototype._okButtonListener =
 96 function(ev) {
 97 	var selected = document.getElementsByName(this._inputName);
 98 	var accountIds = [];
 99 	for (var i = 0; i < selected.length; i++) {
100 		if (selected[i].checked) {
101 			accountIds.push(selected[i].getAttribute("_acctId"));
102 		}
103 	}
104 	DwtDialog.prototype._buttonListener.call(this, ev, [accountIds]);
105 };
106 
107 ZmChooseAccountDialog.prototype._enterListener =
108 function(ev) {
109 	this._okButtonListener.call(this, ev);
110 };
111 
112 ZmChooseAccountDialog.prototype._contentHtml =
113 function() {
114 	return AjxTemplate.expand("share.Widgets#ZmChooseAccountDialog", {id:this._htmlElId});
115 };
116 
117 ZmChooseAccountDialog.prototype._createControls =
118 function() {
119 	this._accountSelectEl = document.getElementById(this._htmlElId+"_accountSelectId");
120 	this._chooseMessageEl = document.getElementById(this._htmlElId+"_chooseAccountMsg");
121 	this._inputName = this._htmlElId + "_accountCheckbox";
122 };
