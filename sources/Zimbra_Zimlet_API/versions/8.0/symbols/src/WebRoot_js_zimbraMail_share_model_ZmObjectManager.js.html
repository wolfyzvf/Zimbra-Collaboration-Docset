 22 /**
 23  * Creates an object manager.
 24  * @class
 25  * This class is used to high-light objects within a given view.
 26  * 
 27  * @author Kevin Henrikson
 28  *
 29  * @param {DwtComposite}	view			the view this manager is going to high-light for
 30  * @param {AjxCallback}	selectCallback  the callback triggered when user clicks on high-light object (provide if you want to do something before the clicked on object opens its corresponding view)
 31  * @param {Boolean}	skipHandlers 	<code>true</code> to avoid adding the standard handlers
 32  */
 33 ZmObjectManager = function(view, selectCallback, skipHandlers) {
 34 
 35 	this._selectCallback = selectCallback;
 36 	this._uuid = Dwt.getNextId();
 37 	this._objectIdPrefix = "OBJ_PREFIX_";
 38 	this._objectHandlers = {};
 39 
 40 	// don't include when looking for objects. only used to provide tool tips for images
 41 	if (appCtxt.get(ZmSetting.MAIL_ENABLED) && window["ZmImageAttachmentObjectHandler"]) {
 42 		this._imageAttachmentHandler = new ZmImageAttachmentObjectHandler();
 43 	}
 44 
 45 	// create handlers (see registerHandler below)
 46 	if (!skipHandlers) {
 47         this.initialized = false;
 48         this._addAutoHandlers();
 49 	} else {
 50         this.initialized = true;
 51     }
 52 
 53     this.sortHandlers();
 54 	this.reset();
 55 	this.setView(view);
 56 }

 72 /**
 73  * Registers the handler.
 74  * 
 75  * @param	{Object}	obj		the object
 76  * @param	{constant}	type	the type
 77  * @param	{constant}	priority	the priority
 78  */
 79 ZmObjectManager.registerHandler =
 80 function(obj, type, priority) {
 81 	if (typeof obj == "string") {
 82 		obj = eval(obj);
 83 	}
 84 	var c = ZmObjectManager._autohandlers;
 85 	if (!obj.__registered) {
 86 		var id = c.push(obj);
 87 		var i = id - 1;
 88 		if(type) {
 89 			c[i].useType = type;
 90 		}
 91 		if(priority) {
 92 			c[i].usePrio = priority;
 93 		}
 94 		obj.__registered = true;
 95 	}
 96 };

 98 /**
 99  * @private
100  */
101 ZmObjectManager.unregisterHandler =
102 function(obj) {
103 	if (typeof obj == "string") {
104 		obj = eval(obj);
105 	}
106  	var c = ZmObjectManager._autohandlers, i;
107 	for (i = c.length; --i >= 0;) {
108 		if (c[i] === obj) {
109 			c.splice(i, 1);
110 			break;
111 		}
112 	}
113 };

125 /**
126  * Gets the handlers.
127  * 
128  * @return	{Array}	an array of {@link ZmObjectHandler} objects
129  */
130 ZmObjectManager.prototype.getHandlers =
131 function() {
132 	if (!this.initialized && appCtxt.zimletsPresent()) {
133 		var zimletMgr = appCtxt.getZimletMgr();
134 		if (zimletMgr.isLoaded()) {
135 			this.initialized = true;
136 			var zimlets = zimletMgr.getContentZimlets();
137 			for (var i = 0; i < zimlets.length; i++) {
138 				this.addHandler(zimlets[i], zimlets[i].type, zimlets[i].prio);
139 			}
140 		}
141 	}
142 	return this._objectHandlers;
143 };

145 /**
146  * Adds the handler.
147  * 
148  * @param	{ZmObjectHandler}	h		the handler
149  * @param	{constant}			type	the type
150  * @param	{constant}		priority	the priority
151  */
152 ZmObjectManager.prototype.addHandler =
153 function(h, type, priority) {
154 	type = type || (h.getTypeName() ? h.getTypeName() : "none");
155 	priority = priority ? priority : -1;
156 	h._prio = priority;
157 	//DBG.println(AjxDebug.DBG3, "addHandler " + h + " type: " + type + " prio: " + priority);
158 	var oh = this.getHandlers();
159 	if (!oh[type]) {oh[type] = [];}
160 	oh[type].push(h);
161 };
162 
163 /**
164  * Removes the handler.
165  * 
166  * @param	{ZmObjectHandler}	h		the handler
167  * @param	{constant}			type	the type
168  */
169 ZmObjectManager.prototype.removeHandler =
170 function(h, type) {
171 	type = type || (h.getTypeName() ? h.getTypeName() : "none");
172 	var oh = this.getHandlers();
173 	if (oh[type]) {
174 		for (var i = 0, count = oh[type].length; i < count; i++) {
175 			if (oh[type][i] == h) {
176 				oh[type].splice(i, 1);
177 				break;
178 			}
179 		}
180 	}
181 };

183 /**
184  * Sorts the handlers.
185  * 
186  */
187 ZmObjectManager.prototype.sortHandlers =
188 function() {
189 	this._allObjectHandlers = [];
190     var objectHandlers = this.getHandlers();
191     for (i in objectHandlers) {
192 		// Object handlers grouped by Type
193 		objectHandlers[i].sort(ZmObjectManager.__byPriority);
194 
195 		// Copy each array to a single array of all Object Handlers
196 		for (var k = 0; k < objectHandlers[i].length; k++) {
197 			this._allObjectHandlers.push(objectHandlers[i][k]);
198 		}
199 	}
200 	this._allObjectHandlers.sort(ZmObjectManager.__byPriority);
201 };

203 /**
204  * @private
205  */
206 ZmObjectManager.prototype._addAutoHandlers =
207 function() {
208 	var c = ZmObjectManager._autohandlers, i, obj, prio;
209 	for (i = 0; i < c.length; ++i) {
210 		obj = c[i];
211 		var	zim = obj;
212 		var type = obj.TYPE;
213 		if (appCtxt.zimletsPresent()) {
214 			if (!(obj instanceof ZmZimletBase)) {
215 				zim = new obj();
216 			}
217 		}
218 		if (obj.useType) {
219 			type = obj.useType;
220 		}
221 		if (obj.usePrio) {
222 			prio = obj.usePrio;
223 		}
224 		this.addHandler(zim, type, prio);
225 	}
226 };

288 /**
289  * Finds objects.
290  * 
291  * @param	{String}	content		the content
292  * @param	{Boolean}	htmlEncode	<code>true</code> to HTML encode the content
293  * @param	{constant}	type		the type
294  * @param	{Boolean}	isTextMsg	<code>true</code> if is text msg
295  * @param	{hash}		options		arbitrary options to pass to handler
296  *
297  * @return	{String}	the object
298  */
299 ZmObjectManager.prototype.findObjects =
300 function(content, htmlEncode, type, isTextMsg, options) {
301 	if  (!content) {return "";}
302 	var html = [];
303 	var idx = 0;
304 
305 	var maxIndex = content.length;
306 	var lastIndex = 0;
307 
308     var objectHandlers = this.getHandlers();
309     while (true) {
310 		var lowestResult = null;
311 		var lowestIndex = maxIndex;
312 		var lowestHandler = null;
313 
314 		// if given a type, just go thru the handler defined for that type.
315 		// otherwise, go thru every handler we have. Regardless, ask each handler
316 		// to find us a match >= to lastIndex. Handlers that didn't match last
317 		// time will simply return, handlers that matched last time that we didn't
318 		// use (because we found a closer match) will simply return that match again.
319 		//
320 		// when we are done, we take the handler with the lowest index.
321 		var i;
322 		var handlers;
323 		var chunk;
324 		var result = null;
325 		if (type) {
326 			//DBG.println(AjxDebug.DBG3, "findObjects type [" + type + "]");
327 			handlers = objectHandlers[type];
328 			if (handlers) {
329 				for (i = 0; i < handlers.length; i++) {
330 					//DBG.println(AjxDebug.DBG3, "findObjects by TYPE (" + handlers[i] + ")");
331 					result = handlers[i].findObject(content, lastIndex);
332 					// No match keep trying.
333 					if(!result) {continue;}
334 					// Got a match let's handle it.
335 					if (result.index >= lowestIndex) {break;}
336 					lowestResult = result;
337 					lowestIndex = result.index;
338 					lowestHandler = handlers[i];
339 				}
340 			}
341 			// If it's an email address just handle it and return the result.
342 			if (type == "email" || content instanceof AjxEmailAddress) {
343 				if (lowestHandler) {
344                     content = this._getAjxEmailAddress(content);
345 					this.generateSpan(lowestHandler, html, idx, content, null, options);
346 				} else {
347 					html[idx++] = AjxStringUtil.htmlEncode(content.toString());
348 				}
349 				return html.join("");
350 			}
351 		} else {
352 			for (var j = 0; j < this._allObjectHandlers.length; j++) {
353 				var handler = this._allObjectHandlers[j];
354 				//DBG.println(AjxDebug.DBG3, "findObjects trying (" + handler + ")");
355 				result = handler.findObject(content, lastIndex);
356 				if (result && result.index < lowestIndex) {
357 					lowestResult = result;
358 					lowestIndex = result.index;
359 					lowestHandler = handler;
360 				}
361 			}
362 		}
363 
364 		if (!lowestResult) {
365 			// all done
366 			// do last chunk
367 			chunk = content.substring(lastIndex, maxIndex);
368 			if (htmlEncode) {
369 				html[idx++] = AjxStringUtil.htmlEncode(chunk, !!isTextMsg);
370 			} else {
371 				html[idx++] = chunk;
372 			}
373 			break;
374 		}
375 
376 		//  add anything before the match
377 		if (lowestIndex > lastIndex) {
378 			chunk = content.substring(lastIndex, lowestIndex);
379 			if (htmlEncode) {
380 				html[idx++] = AjxStringUtil.htmlEncode(chunk, !!isTextMsg);
381 			} else {
382 				html[idx++] = chunk;
383 			}
384 		}
385 
386 		// add the match
387 		if(lowestHandler) {
388 			idx = this.generateSpan(lowestHandler, html, idx, lowestResult[0], lowestResult.context);
389 		} else {
390 			html[idx++] = lowestResult[0];
391 		}
392 
393 		// update the index
394 		lastIndex = lowestResult.index + (lowestResult.matchLength || lowestResult[0].length);
395 	}
396 
397 	return html.join("");
398 };

401 /**
402  * Added this customized method for the sake of ZmMailMsgView performance.
403  * 
404  * TODO: Integrate this method to findObjectsInNode()
405  * 
406  * @private
407  */
408 ZmObjectManager.prototype.processObjectsInNode = function(doc, node){
409 
410     var objectManager = this;
411 	doc = doc || node.ownerDocument;
412     var tmpdiv = doc.createElement("div");
413 
414     var recurse = function(node, handlers) {
415 		var tmp, i, val, next;
416 		switch (node.nodeType) {
417 		    case 1:	// ELEMENT_NODE
418 			node.normalize();
419 			tmp = node.tagName.toLowerCase();
420 
421 			if (next == null) {
422 				if (/^(img|a)$/.test(tmp)) {
423                     var isMailTo = (tmp == 'a' && ZmMailMsgView._MAILTO_RE.test(node.href));
424 					if (tmp == "a" && node.target
425 					    && (isMailTo || ZmMailMsgView._URL_RE.test(node.href)))
426 					{
427 						// tricky.
428 						var txt = isMailTo ? node.href :RegExp.$1 ;
429 						tmp = doc.createElement("div");
430 						tmp.innerHTML = objectManager.findObjects(AjxStringUtil.trim(txt));
431 						tmp = tmp.firstChild;
432 						if (tmp.nodeType == 3 /* Node.TEXT_NODE */) {
433 							// probably no objects were found.  A warning would be OK here
434 							// since the regexps guarantee that objects _should_ be found.
435 							return tmp.nextSibling;
436 						}
437 						// here, tmp is an object span, but it
438 						// contains the URL (href) instead of
439 						// the original link text.
440 						node.parentNode.insertBefore(tmp, node); // add it to DOM
441 						tmp.innerHTML = "";
442 						tmp.appendChild(node); // we have the original link now
443 						return tmp.nextSibling;	// move on
444 					}
445 					handlers = false;
446 				}
447 			} else {
448 				// consider processed
449 				node = next;
450 			}
451 
452                         // bug 28264: the only workaround possible seems to be
453                         // to remove textIndent styles that have a negative value:
454                         if (parseFloat(node.style.textIndent) < 0)
455                                 node.style.textIndent = "";
456 
457                         for (i = node.firstChild; i; i = recurse(i, handlers));
458 			return node.nextSibling;
459 
460 		    case 3:	// TEXT_NODE
461 		    case 4:	// CDATA_SECTION_NODE (just in case)
462 			// generate ObjectHandler-s
463 			if (handlers && /[^\s\xA0]/.test(node.data)) try {
464  				var a = null, b = null;
465 
466 				if (!AjxEnv.isIE) {
467 					// this block of code is supposed to free the object handlers from
468 					// dealing with whitespace.  However, IE sometimes crashes here, for
469 					// reasons that weren't possible to determine--hence we avoid this
470 					// step for IE.  (bug #5345)
471 					var results = /^[\s\xA0]+/.exec(node.data);
472 					if (results) {
473 						a = node;
474 						node = node.splitText(results[0].length);
475 					}
476 					results = /[\s\xA0]+$/.exec(node.data);
477 					if (results)
478 						b = node.splitText(node.data.length - results[0].length);
479 				}
480 
481 				tmp = tmpdiv;
482 				var code = objectManager.findObjects(node.data, true, null, false);
483 				var disembowel = false;
484 				if (AjxEnv.isIE) {
485 					// Bug #6481, #4498: innerHTML in IE massacrates whitespace
486 					//            unless it sees a <pre> in the code.
487 					tmp.innerHTML = [ "<pre>", code, "</pre>" ].join("");
488 					disembowel = true;
489 				} else {
490 					tmp.innerHTML = code;
491 				}
492 
493 				if (a)
494 					tmp.insertBefore(a, tmp.firstChild);
495 				if (b)
496 					tmp.appendChild(b);
497 
498 				a = node.parentNode;
499 				if (disembowel)
500 					tmp = tmp.firstChild;
501 				while (tmp.firstChild)
502 					a.insertBefore(tmp.firstChild, node);
503 				tmp = node.nextSibling;
504 				a.removeChild(node);
505 				return tmp;
506 			} catch(ex) {};
507 		}
508 		return node.nextSibling;
509 	};
510 
511     // Parse through the DOM directly and find objects.
512 	if (node && node.childNodes && node.childNodes.length) {
513 		for (var i = 0; i < node.childNodes.length; i++){
514 			recurse(node.childNodes[i], true);
515 		}
516 	}
517 };

519 /**
520  * @private
521  */
522 ZmObjectManager.prototype.findObjectsInNode =
523 function(node, re_discard, re_allow, callbacks) {
524 	var objectManager = this, doc = node.ownerDocument, tmpdiv = doc.createElement("div");
525 
526 	if (!re_discard)
527 		re_discard = /^(script|link|object|iframe|applet)$/i;
528 
529 	// This inner function does the actual work.  BEWARE that it return-s
530 	// in various places, not only at the end.
531 	var recurse = function(node, handlers) {
532 		var tmp, i, val, next;
533 		switch (node.nodeType) {
534 		    case 1:	// ELEMENT_NODE
535 			node.normalize();
536 			tmp = node.tagName.toLowerCase();
537 			if (callbacks && callbacks.foreachElement) {
538 				next = callbacks.foreachElement(node, tmp, re_discard, re_allow);
539 			}
540 			if (next == null) {
541 				if (/^(img|a)$/.test(tmp)) {
542 					if (tmp == "a" && node.target
543 					    && (ZmMailMsgView._URL_RE.test(node.href)
544 						|| ZmMailMsgView._MAILTO_RE.test(node.href)))
545 					{
546 						// tricky.
547 						var txt = RegExp.$1;
548 						tmp = doc.createElement("div");
549 						tmp.innerHTML = objectManager.findObjects(AjxStringUtil.trim(RegExp.$1));
550 						tmp = tmp.firstChild;
551 						if (tmp.nodeType == 3 /* Node.TEXT_NODE */) {
552 							// probably no objects were found.  A warning would be OK here
553 							// since the regexps guarantee that objects _should_ be found.
554 							return tmp.nextSibling;
555 						}
556 						// here, tmp is an object span, but it
557 						// contains the URL (href) instead of
558 						// the original link text.
559 						node.parentNode.insertBefore(tmp, node); // add it to DOM
560 						tmp.innerHTML = "";
561 						tmp.appendChild(node); // we have the original link now
562 						return tmp.nextSibling;	// move on
563 					}
564 					handlers = false;
565 				} else if (re_discard.test(tmp) || (re_allow && !re_allow.test(tmp))) {
566 					tmp = node.nextSibling;
567 					node.parentNode.removeChild(node);
568 					return tmp;
569 				}
570 			} else {
571 				// consider processed
572 				node = next;
573 			}
574 
575 			if (AjxEnv.isIE) {
576 				// strips expression()-s, bwuahahaha!
577 				// granted, they get lost on the server-side anyway, but assuming some get through...
578 				// the line below exterminates them.
579 				node.style.cssText = node.style.cssText;
580 			}
581 
582 			for (i = node.firstChild; i; i = recurse(i, handlers));
583 			return node.nextSibling;
584 
585 		    case 3:	// TEXT_NODE
586 		    case 4:	// CDATA_SECTION_NODE (just in case)
587 			// generate ObjectHandler-s
588 			if (handlers && /[^\s\xA0]/.test(node.data)) try {
589  				var a = null, b = null;
590 
591 				if (!AjxEnv.isIE) {
592 					// this block of code is supposed to free the object handlers from
593 					// dealing with whitespace.  However, IE sometimes crashes here, for
594 					// reasons that weren't possible to determine--hence we avoid this
595 					// step for IE.  (bug #5345)
596 					var results = /^[\s\xA0]+/.exec(node.data);
597 					if (results) {
598 						a = node;
599 						node = node.splitText(results[0].length);
600 					}
601 					results = /[\s\xA0]+$/.exec(node.data);
602 					if (results)
603 						b = node.splitText(node.data.length - results[0].length);
604 				}
605 
606 				tmp = tmpdiv;
607 				var code = objectManager.findObjects(node.data, true, null, false);
608 				var disembowel = false;
609 				if (AjxEnv.isIE) {
610 					// Bug #6481, #4498: innerHTML in IE massacrates whitespace
611 					//            unless it sees a <pre> in the code.
612 					tmp.innerHTML = [ "<pre>", code, "</pre>" ].join("");
613 					disembowel = true;
614 				} else {
615 					tmp.innerHTML = code;
616 				}
617 
618 				if (a)
619 					tmp.insertBefore(a, tmp.firstChild);
620 				if (b)
621 					tmp.appendChild(b);
622 
623 				a = node.parentNode;
624 				if (disembowel)
625 					tmp = tmp.firstChild;
626 				while (tmp.firstChild)
627 					a.insertBefore(tmp.firstChild, node);
628 				tmp = node.nextSibling;
629 				a.removeChild(node);
630 				return tmp;
631 			} catch(ex) {};
632 		}
633 		return node.nextSibling;
634 	};
635 	var df = doc.createDocumentFragment();
636 	while (node.firstChild) {
637 		df.appendChild(node.firstChild); // NODE now out of the displayable DOM
638 		recurse(df.lastChild, true, this);	 // parse tree and findObjects()
639 	}
640 	node.appendChild(df);	// put nodes back in the document
641 };

643 /**
644  * Finds content by going through the content and return the result of the object handler's match call.
645  * 
646  * @param	{String}	content		the content
647  * @param	{constant}	type		the content type
648  * @return	{String}	the object
649  * @private
650  */
651 ZmObjectManager.prototype.findMatch =
652 function(content, type) {
653 	if  (!content) {return "";}
654 
655 	var maxIndex = content.length;
656 	var lastIndex = 0;
657 
658 	var lowestResult = null;
659 	var lowestIndex = maxIndex;
660 	var lowestHandler = null;
661 
662 	// if given a type, just go thru the handler defined for that type.
663 	// otherwise, go thru every handler we have.
664 	// when we are done, we take the handler with the lowest index.
665 	var i;
666 
667 	var result = null;
668 	if (type) {
669 		//DBG.println(AjxDebug.DBG3, "findObjects type [" + type + "]");
670 		var handlers = this.getHandlers()[type];
671 		if (handlers) {
672 			for (i = 0; i < handlers.length; i++) {
673 				//DBG.println(AjxDebug.DBG3, "findObjects by TYPE (" + handlers[i] + ")");
674 				result = handlers[i].findObject(content, lastIndex);
675 				// No match keep trying.
676 				if(!result) {continue;}
677 				// Got a match let's handle it.
678 				if (result.index >= lowestIndex) {break;}
679 				lowestResult = result;
680 				lowestIndex = result.index;
681 				lowestHandler = handlers[i];
682 			}
683 		}
684 	} else {
685 		for (var j = 0; j < this._allObjectHandlers.length; j++) {
686 			var handler = this._allObjectHandlers[j];
687 			//DBG.println(AjxDebug.DBG3, "findObjects trying (" + handler + ")");
688 			result = handler.findObject(content, lastIndex);
689 			if (result && result.index < lowestIndex) {
690 				lowestResult = result;
691 				lowestIndex = result.index;
692 				lowestHandler = handler;
693 			}
694 		}
695 	}
696 	return lowestResult;
697 };

699 /**
700  * Dives recursively into the given DOM node.  Creates ObjectHandlers in text
701  * nodes and cleans the mess in element nodes.  Discards by default "script",
702  * "link", "object", "style", "applet" and "iframe" (most of them shouldn't
703  * even be here since (1) they belong in the <head> and (2) are discarded on
704  * the server-side, but we check, just in case).
705  *
706  * @private
707  */
708 ZmObjectManager.prototype.processHtmlNode =
709 function(node, handlers, discard, ignore) {
710 	var doc = node.ownerDocument;
711 	handlers = handlers != null ? handlers : true;
712 	var discardRe = discard instanceof RegExp ? discard : null;
713 	if (!discardRe) {
714 		discard = discard || [ "script", "link", "object", "style", "applet", "iframe" ];
715 		discard = discard instanceof Array ? discard : [ discard ];
716 		discardRe = new RegExp("^("+discard.join("|")+")$", "i");
717 	}
718 
719 	var ignoreRe = ignore instanceof RegExp ? ignore : null;
720 	if (!ignoreRe && ignore) {
721 		ignore = ignore instanceof Array ? ignore : [ ignore ];
722 		ignoreRe = new RegExp("^("+ignore.join("|")+")$", "i");
723 	}
724 
725 	var tmp, i, val;
726 	switch (node.nodeType) {
727 	    case 1:	// ELEMENT_NODE
728 		node.normalize();
729 		tmp = node.tagName.toLowerCase();
730 		if (/^(img|a)$/.test(tmp)) {
731 			if (tmp == "a"
732 			    && (ZmMailMsgView._URL_RE.test(node.href)
733 				|| ZmMailMsgView._MAILTO_RE.test(node.href)))
734 			{
735 				// tricky.
736 				var txt = RegExp.$1;
737 				tmp = doc.createElement("div");
738 				tmp.innerHTML = this.findObjects(AjxStringUtil.trim(RegExp.$1));
739 				tmp = tmp.firstChild;
740 				if (tmp.nodeType == 3 /* Node.TEXT_NODE */) {
741 					// probably no objects were found.  A warning would be OK here
742 					// since the regexps guarantee that objects _should_ be found.
743 					// DBG.println(AjxDebug.DBG1, "No objects found for potentially valid text!");
744 					return tmp.nextSibling;
745 				}
746 				// here, tmp is an object span, but it
747 				// contains the URL (href) instead of
748 				// the original link text.
749 				node.parentNode.insertBefore(tmp, node); // add it to DOM
750 				tmp.innerHTML = "";
751 				tmp.appendChild(node); // we have the original link now
752 				return tmp.nextSibling;	// move on
753 			}
754 			handlers = false;
755 		}
756 		else if (discardRe.test(tmp)) {
757 			tmp = node.nextSibling;
758 			node.parentNode.removeChild(node);
759 			return tmp;
760 		}
761 		else if (ignoreRe && ignoreRe.test(tmp)) {
762 			tmp = node.nextSibling;
763 			var fragment = doc.createDocumentFragment();
764 			for (var child = node.firstChild; child; child = child.nextSibling) {
765 				fragment.appendChild(child);
766 			}
767 			node.parentNode.replaceChild(fragment, node);
768 			return tmp;
769 		}
770 		else if (tmp == "style") {
771 			return node.nextSibling;
772 		}
773 
774 		if (AjxEnv.isIE) {
775 			// strips expression()-s, bwuahahaha!
776 			// granted, they get lost on the server-side anyway, but assuming some get through...
777 			// the line below exterminates them.
778 			node.style.cssText = node.style.cssText;
779 		}
780 
781 		var child = node.firstChild;
782 		while (child) {
783 			child = this.processHtmlNode(child, handlers, discardRe, ignoreRe);
784 		}
785 		return node.nextSibling;
786 
787 	    case 3:	// TEXT_NODE
788 	    case 4:	// CDATA_SECTION_NODE (just in case)
789 		// generate ObjectHandler-s
790 		if (handlers && /[^\s\xA0]/.test(node.data)) try {
791 			var a = null, b = null;
792 
793 			if (!AjxEnv.isIE) {
794 				// this block of code is supposed to free the object handlers from
795 				// dealing with whitespace.  However, IE sometimes crashes here, for
796 				// reasons that weren't possible to determine--hence we avoid this
797 				// step for IE.  (bug #5345)
798 				if (/^[\s\xA0]+/.test(node.data)) {
799 					a = node;
800 					node = node.splitText(RegExp.lastMatch.length);
801 				}
802 				if (/[\s\xA0]+$/.test(node.data))
803 					b = node.splitText(node.data.length - RegExp.lastMatch.length);
804 			}
805 
806 			tmp = doc.createElement("div");
807 			var code  = this.findObjects(node.data, true, null, false);
808 			var disembowel = false;
809 			if (AjxEnv.isIE) {
810 				// Bug #6481, #4498: innerHTML in IE massacrates whitespace
811 				//            unless it sees a <pre> in the code.
812 				tmp.innerHTML = [ "<pre>", code, "</pre>" ].join("");
813 				disembowel = true;
814 			} else {
815 				tmp.innerHTML = code;
816 			}
817 
818 			if (a)
819 				tmp.insertBefore(a, tmp.firstChild);
820 			if (b)
821 				tmp.appendChild(b);
822 
823 			a = node.parentNode;
824 			if (disembowel)
825 				tmp = tmp.firstChild;
826 			while (tmp.firstChild)
827 				a.insertBefore(tmp.firstChild, node);
828 			tmp = node.nextSibling;
829 			a.removeChild(node);
830 			return tmp;
831 		} catch(ex) {};
832 	}
833 	return node.nextSibling;
834 };

880  * @private
881  */
882 ZmObjectManager.prototype._mouseOverListener =
883 function(ev) {
884 	var span = this._findObjectSpan(ev.target);
885 	if (!span) {return false;}
886 	var object = this._objects[span.id];
887 	if (!object) {return false;}
888 
889 	span.className = object.handler.getHoveredClassName(object.object, object.context, span.id);
890 	if (object.handler.hasToolTipText()) {
891 		var shell = DwtShell.getShell(window);
892 		var manager = shell.getHoverMgr();
893 		if ((!manager.isHovering() || manager.getHoverObject() != object) && !DwtMenu.menuShowing()) {
894 			manager.reset();
895 			manager.setHoverOverDelay(ZmObjectManager._TOOLTIP_DELAY);
896 			manager.setHoverObject(object);
897 			manager.setHoverOverData(object);
898 			manager.setHoverOverListener(this._hoverOverListener);
899 			manager.hoverOver(ev.docX, ev.docY);
900 			ev.hoverStarted = true;
901 		}
902 	}
903 
904 	ev._returnValue = true;
905 	ev._dontCallPreventDefault = true;
906 	return false;
907 };

962 /**
963  * @private
964  */
965 ZmObjectManager.prototype._mouseDownListener =
966 function(ev) {
967 
968 	// "authoritative" means a previous listener doesn't want propagation to get reset
969 	if (!ev._authoritative) {
970 		ev._dontCallPreventDefault = true;
971 		ev._returnValue = true;
972 		ev._stopPropagation = false;
973 	}
974 
975 	var span = this._findObjectSpan(ev.target);
976 	if (!span) {
977 		return true;
978 	}
979 	var object = this._objects[span.id];
980 	if (!object) {
981 		return true;
982 	}
983 
984 	ev._stopPropagation = true;
985 
986 	var shell = DwtShell.getShell(window);
987 	var manager = shell.getHoverMgr();
988 	manager.setHoverOutDelay(0);
989 	manager.setHoverOutData(object);
990 	manager.setHoverOutListener(this._hoverOutListener);
991 	manager.hoverOut();
992 
993 	span.className = object.handler.getActiveClassName(object.object, object.context, span.id);
994 	if (ev.button == DwtMouseEvent.RIGHT) {
995 		// NOTE: we need to know if the current view is a dialog since action
996 		//       menu needs to be a higher z-index
997 		var isDialog = (this._view instanceof DwtDialog);
998 		var menu = object.handler.getActionMenu(object.object, span, object.context, isDialog);
999 		if (menu) {
1000 			menu.popup(0, ev.docX, ev.docY);
1001 			// if we have an action menu, don't let the browser show its context menu too
1002 			ev._dontCallPreventDefault = false;
1003 			ev._returnValue = false;
1004 			ev._stopPropagation = true;
1005 			return true;
1006 		}
1007 	} else if (ev.button == DwtMouseEvent.LEFT) {
1008 		if (this._selectCallback) {
1009 			this._selectCallback.run();
1010 		}
1011 		object.handler.selected(object.object, span, ev, object.context);
1012 		return true;
1013 	}
1014 	return false;
1015 };
