 16 /**
 17  * Creates a left pane view for suggesting time/locations
 18  * @constructor
 19  * @class
 20  * This class displays suggested free time/location for sending invites to attendees

 25  * @param controller		[ZmApptComposeController]	the appt compose controller

 27 ZmScheduleAssistantView = function(parent, controller, apptEditView, closeCallback) {
 28     this._kbMgr = appCtxt.getKeyboardMgr();
 29     this._attendees = [];
 30     this._workingHours = {};
 31     this._fbStat = new AjxVector();
 32     this._fbStatMap = {};
 33     this._schedule = {};
 34 
 35 	ZmApptAssistantView.call(this, parent, controller, apptEditView, closeCallback);
 36 };

 58 ZmScheduleAssistantView.prototype._createMiniCalendar =
 59 function(date) {
 60 	date = date ? date : new Date();
 61 
 62 	var firstDayOfWeek = appCtxt.get(ZmSetting.CAL_FIRST_DAY_OF_WEEK) || 0;
 63 
 64     //todo: need to use server setting to decide the weekno standard
 65     var serverId = AjxTimezone.getServerId(AjxTimezone.DEFAULT);
 66     var useISO8601WeekNo = (serverId && serverId.indexOf("Europe")==0 && serverId != "Europe/London");
 67 
 68 	this._miniCalendar = new ZmMiniCalendar({parent: this, posStyle:DwtControl.RELATIVE_STYLE,
 69 	    firstDayOfWeek: firstDayOfWeek, showWeekNumber: appCtxt.get(ZmSetting.CAL_SHOW_CALENDAR_WEEK),
 70         useISO8601WeekNo: useISO8601WeekNo});
 71     this._miniCalendar.setDate(date);
 72 	this._miniCalendar.setScrollStyle(Dwt.CLIP);
 73 	this._miniCalendar.addSelectionListener(new AjxListener(this, this._miniCalSelectionListener));
 74 	this._miniCalendar.addDateRangeListener(new AjxListener(this, this._miniCalDateRangeListener));
 75 	this._miniCalendar.setMouseOverDayCallback(new AjxCallback(this, this._miniCalMouseOverDayCallback));
 76 	this._miniCalendar.setMouseOutDayCallback(new AjxCallback(this, this._miniCalMouseOutDayCallback));
 77 
 78 	var workingWeek = [];
 79 	for (var i = 0; i < 7; i++) {
 80 		var d = (i + firstDayOfWeek) % 7;
 81 		workingWeek[i] = (d > 0 && d < 6);
 82 	}
 83 	this._miniCalendar.setWorkingWeek(workingWeek);
 84 
 85 	var app = appCtxt.getApp(ZmApp.CALENDAR);
 86 	var show = app._active || appCtxt.get(ZmSetting.CAL_ALWAYS_SHOW_MINI_CAL);
 87 	this._miniCalendar.setSkipNotifyOnPage(show && !app._active);
 88 	if (!app._active) {
 89 		this._miniCalendar.setSelectionMode(DwtCalendar.DAY);
 90 	}
 91 
 92     this._miniCalendar.reparentHtmlElement(this._htmlElId + "_suggest_minical");
 93 };

171 ZmScheduleAssistantView.prototype._getTimeFrame =
172 function() {
173 	var di = {};
174 	ZmApptViewHelper.getDateInfo(this._apptView, di);
175     var startDate = this._date;
176     if (!this._date || !this._suggestTime) {
177         startDate = AjxDateUtil.simpleParseDateStr(di.startDate);
178     }
179     var endDate = new Date(startDate);
180     startDate.setHours(0, 0, 0, 0);
181     endDate.setTime(startDate.getTime() + AjxDateUtil.MSEC_PER_DAY);
182 	return {start:startDate, end:endDate};
183 };

222 ZmScheduleAssistantView.prototype.updateAttendees =
223 function(attendees) {
224 
225     if(attendees instanceof AjxVector) attendees = attendees.getArray();
226 
227     this._attendees = [];
228 
229     this.addOrganizer();
230 
231     var attendee;
232     for (var i = attendees.length; --i >= 0;) {
233             attendee = attendees[i].getEmail();
234             if (attendee instanceof Array) {
235                 attendee = attendee[i][0];
236             }
237             this._attendees.push(attendee);
238     }
239 
240     // *** Separate Suggestions pane, only invoked to show suggestions, so changing
241     //     force refresh to True
242     this.reset(this._date, this._attendees, true);
243 };

245 ZmScheduleAssistantView.prototype.updateAttendee =
246 function(attendee) {
247 
248     var email = (typeof attendee == 'string') ? attendee : attendee.getEmail();
249     if(this._attendees.length == 0) {
250         this.addOrganizer();
251         this._attendees.push(email);
252     }else {
253         var found = false;
254         for (var i = this._attendees.length; --i >= 0;) {
255             if(email == this._attendees[i]) {
256                 found = true;
257                 break;
258             }
259         }
260         if(!found) this._attendees.push(email);
261     }
262 
263     // *** Separate Suggestions pane, only invoked to show suggestions, so changing
264     //     force refresh to True
265     this.reset(this._date, this._attendees, true);
266 };

269 ZmScheduleAssistantView.prototype.reset =
270 function(date, attendees, forceRefresh) {
271     this._date = date || this._miniCalendar.getDate();
272     if(!this._apptView.isSuggestionsNeeded() || !this.isSuggestionsEnabled()) {
273         var isGalEnabled = appCtxt.get(ZmSetting.GROUP_CALENDAR_ENABLED) && appCtxt.get(ZmSetting.GAL_ENABLED);
274         if(this._timeSuggestions && !isGalEnabled) this._timeSuggestions.removeAll();
275         this.clearMiniCal();
276         if(!this.isSuggestionsEnabled()) {
277            if(isGalEnabled) this._timeSuggestions.setShowSuggestionsHTML(this._date);
278         }
279         return;
280     }
281 
282     var newDuration = this._apptView.getDurationInfo().duration;
283     var newKey = this.getFormKey(this._date, attendees);
284     if(newKey != this._key || newDuration != this._duration) {
285         if(this._currentSuggestions){
286             this._currentSuggestions.removeAll();
287             this.clearMiniCal();
288         }
289         if(forceRefresh) this.suggestAction(false, false);
290     }
291 };

315 // This should only be called for time suggestions
316 ZmScheduleAssistantView.prototype._findFreeBusyInfo =
317 function(params) {
318 
319     var currAcct = this._apptView.getCalendarAccount();
320 	// Bug: 48189 Don't send GetFreeBusyRequest for non-ZCS accounts.
321 	if (appCtxt.isOffline && (!currAcct.isZimbraAccount || currAcct.isMain)) {
322         //todo: avoid showing smart scheduler button for non-ZCS accounts - offline client
323         return;
324 	}
325 
326 	var tf = this._timeFrame = this._getTimeFrame();
327     if (params.fbEndTime) {
328         // Override the time frame.  Used for checking location
329         // recurrence collisions
330         tf.end = new Date(params.fbEndTime);
331     }
332 	var emails = [], attendeeEmails = [], email;
333 
334     params.itemIndex = {};
335     params.items = [];
336     params.timeFrame = tf;
337 
338     this._copyResourcesToParams(params, emails);
339 
340     var attendees = this._apptView.getRequiredAttendeeEmails();
341     this._attendees = [];
342 
343 
344     var attendee;
345     for (var i = attendees.length; --i >= 0;) {
346         this._addAttendee(attendees[i], params, emails, attendeeEmails);
347     }
348     params._nonOrganizerAttendeeEmails = attendeeEmails.slice();
349     //include organizer in the scheduler suggestions
350     var organizer = this.getOrganizer();
351     this._addAttendee(organizer.getEmail(), params, emails, attendeeEmails);
352 
353     params.emails = emails;
354     params.attendeeEmails = attendeeEmails;
355 
356     this._key = this.getFormKey(tf.start, this._attendees);
357 
358     if((this._attendees.length == 0) && this._suggestTime) {
359         this._timeSuggestions.setNoAttendeesHtml();
360         return;
361     }
362 
363 	if (this._freeBusyRequest) {
364 		appCtxt.getRequestMgr().cancelRequest(this._freeBusyRequest, null, true);
365 	}
366 
367     var callback;
368     if (params.fbCallback) {
369         // Custom FB processing
370         callback = params.fbCallback;
371     } else {
372         if (this._suggestTime) {
373             callback = new AjxCallback(this, this.getWorkingHours, [params]);
374         } else {
375             callback = new AjxCallback(this, this.suggestLocations, [params]);
376         }
377     }
378 
379     var acct = (appCtxt.multiAccounts) ? this._apptView.getCalendarAccount() : null;
380     var fbParams = {
381                     startTime: tf.start.getTime(),
382                     endTime: tf.end.getTime(),
383                     emails: emails,
384                     callback: callback,
385                     errorCallback: callback,
386                     noBusyOverlay: true,
387                     account: acct
388     };
389 
390     this._freeBusyRequest = this._fbCache.getFreeBusyInfo(fbParams);
391 };

419 ZmScheduleAssistantView.prototype.getWorkingHours =
420 function(params) {
421 
422     //clear fb request info
423     this._freeBusyRequest = null;
424 
425     if (this._workingHoursRequest) {
426         appCtxt.getRequestMgr().cancelRequest(this._workingHoursRequest, null, true);
427     }
428 
429     var onlyIncludeMyWorkingHours     = params.onlyIncludeMyWorkingHours     = this.isOnlyMyWorkingHoursIncluded();
430     var onlyIncludeOthersWorkingHours = params.onlyIncludeOthersWorkingHours = this.isOnlyOthersWorkingHoursIncluded();
431 
432     if(!onlyIncludeMyWorkingHours && !onlyIncludeOthersWorkingHours) {
433          // Non-working hours can be used for the organizer and all attendees
434          this.suggestTimeSlots(params);
435          return;   
436     }
437 
438     var organizer = this.getOrganizer();
439     this._organizerEmail = organizer.getEmail();
440 
441     var emails =  [];
442     if (onlyIncludeOthersWorkingHours) {
443         emails = params._nonOrganizerAttendeeEmails;
444     }
445     if (onlyIncludeMyWorkingHours) {
446         emails = emails.concat([this._organizerEmail]);
447     }
448 
449     var acct = (appCtxt.multiAccounts) ? this._apptView.getCalendarAccount() : null;
450 
451     //optimization: fetch working hrs for a week - wrking hrs pattern repeat everyweek
452     var weekStartDate = new Date(params.timeFrame.start.getTime());
453     var dow = weekStartDate.getDay();
454     weekStartDate.setDate(weekStartDate.getDate()-((dow+7))%7);
455 
456 
457     var whrsParams = {
458         startTime: weekStartDate.getTime(),
459         endTime: weekStartDate.getTime() + 7*AjxDateUtil.MSEC_PER_DAY,
460         emails: emails,
461         callback: new AjxCallback(this, this._handleWorkingHoursResponse, [params]),
462         errorCallback: new AjxCallback(this, this._handleWorkingHoursError, [params]),
463         noBusyOverlay: true,
464         account: acct
465     };
466 
467     this._workingHoursRequest = this._fbCache.getWorkingHours(whrsParams);
468 };

481 ZmScheduleAssistantView.prototype._handleWorkingHoursResponse =

506 ZmScheduleAssistantView.prototype.suggestTimeSlots =
507 function(params) {
508 
509     var startDate = this._timeFrame.start;
510     startDate.setHours(0, 0, 0, 0);
511     var startTime = startDate.getTime();
512 
513     var cDate = new Date();
514 
515     //ignore suggestions that are in past
516     if(startTime == cDate.setHours(0, 0, 0, 0)) {
517         startDate = new Date();
518         startTime = startDate.setHours(startDate.getHours(), ((startDate.getMinutes() >=30) ? 60 : 30), 0, 0);
519     }
520 
521     var endDate = new Date(startTime);
522     endDate.setHours(23, 59, 0, 0);
523     var endTime = endDate.getTime();
524     var durationInfo = this._duration = this._apptView.getDurationInfo();
525 
526     params.duration = durationInfo.duration;
527 
528     this._fbStat = new AjxVector();
529     this._fbStatMap = {};
530     this._totalUsers = this._attendees.length;
531     this._totalLocations =  this._resources.length;
532 
533     while(startTime < endTime) {
534         this.computeAvailability(startTime, startTime + durationInfo.duration, params);
535         startTime += AjxDateUtil.MSEC_PER_HALF_HOUR;
536     }
537 
538     params.locationInfo = this.computeLocationAvailability(durationInfo, params);
539 
540     this._fbStat.sort(ZmScheduleAssistantView._slotComparator);
541     //DBG.dumpObj(this._fbStat);
542     this.renderSuggestions(params);
543 
544     //highlight minicalendar to mark suggested days in month
545     this.highlightMiniCal();
546 };

574 ZmScheduleAssistantView.prototype.computeAvailability =
575 function(startTime, endTime, params) {
576     
577     var dayStartTime = (new Date(startTime)).setHours(0,0,0,0);
578     var dayEndTime = dayStartTime + AjxDateUtil.MSEC_PER_DAY;
579 
580     var key = this.getKey(startTime, endTime);
581     var fbInfo;
582 
583     if(!params.miniCalSuggestions && this._fbStatMap[key]) {
584         fbInfo = this._fbStatMap[key];
585     }else {
586         fbInfo = {
587             startTime: startTime,
588             endTime: endTime,
589             availableUsers: 0,
590             availableLocations: 0,
591             attendees: [],
592             locations: []
593         };
594     }
595 
596     var attendee, sched, isFree;
597     for(var i = this._attendees.length; --i >= 0;) {
598         attendee = this._attendees[i];
599 
600         var excludeTimeSlots = this._apptView.getFreeBusyExcludeInfo(attendee);
601         sched = this._fbCache.getFreeBusySlot(dayStartTime, dayEndTime, attendee, excludeTimeSlots);
602 
603         // Last entry will be the organizer, all others are attendees
604         // Organizer and Attendees have separate checkboxes indicating whether to apply non-working hours to them.
605         var isOrganizer = (i == (this._attendees.length-1));
606         var onlyUseWorkingHours = isOrganizer ?
607             params.onlyIncludeMyWorkingHours :  params.onlyIncludeOthersWorkingHours;
608         isFree = onlyUseWorkingHours ?  this.isWithinWorkingHour(attendee, startTime, endTime) : true;
609 
610         //ignore time slots for non-working hours of this user
611         if(!isFree) continue;
612 
613         if(sched.b) isFree = isFree && ZmApptAssistantView.isBooked(sched.b, startTime, endTime);
614         if(sched.t) isFree = isFree && ZmApptAssistantView.isBooked(sched.t, startTime, endTime);
615         if(sched.u) isFree = isFree && ZmApptAssistantView.isBooked(sched.u, startTime, endTime);
616 
617         //collect all the item indexes of the attendees available at this slot
618         if(isFree) {
619             if(!params.miniCalSuggestions) fbInfo.attendees.push(params.itemIndex[attendee]);
620             fbInfo.availableUsers++;
621         }
622     }
623 
624     if (this.isSuggestRooms()) {
625 
626         var list = this._resources, resource;
627         for (var i = list.length; --i >= 0;) {
628             attendee = list[i];
629             resource = attendee.getEmail();
630 
631             if (resource instanceof Array) {
632                 resource = resource[0];
633             }
634 
635             var excludeTimeSlots = this._apptView.getFreeBusyExcludeInfo(resource);
636             sched = this._fbCache.getFreeBusySlot(dayStartTime, dayEndTime, resource, excludeTimeSlots);
637             isFree = true;
638             if(sched.b) isFree = isFree && ZmApptAssistantView.isBooked(sched.b, startTime, endTime);
639             if(sched.t) isFree = isFree && ZmApptAssistantView.isBooked(sched.t, startTime, endTime);
640             if(sched.u) isFree = isFree && ZmApptAssistantView.isBooked(sched.u, startTime, endTime);
641 
642             //collect all the item indexes of the locations available at this slot
643             if(isFree) {
644                 if(!params.miniCalSuggestions) fbInfo.locations.push(params.itemIndex[resource]);
645                 fbInfo.availableLocations++;
646             }
647         }
648     }
649 
650     //mini calendar suggestions should avoid collecting all computed information in array for optimiziation
651     if(!params.miniCalSuggestions && fbInfo.availableUsers > 0) {
652         var showOnlyGreenSuggestions = params.showOnlyGreenSuggestions;
653         if(!showOnlyGreenSuggestions || (fbInfo.availableUsers == this._totalUsers)) {
654             this._fbStat.add(fbInfo);
655             this._fbStatMap[key] = fbInfo;            
656         }
657     }
658 
659     return fbInfo;
660 };

663 ZmScheduleAssistantView._slotComparator =
664 function(slot1, slot2) {
665 	if(slot1.availableUsers < slot2.availableUsers) {
666         return 1;
667     }else if(slot1.availableUsers > slot2.availableUsers) {
668         return -1;
669     }else {
670         return slot1.startTime < slot2.startTime ? -1 : (slot1.startTime > slot2.startTime ? 1 : 0);
671     }

691 ZmScheduleAssistantView.prototype.isWithinWorkingHour =
692 function(attendee, startTime, endTime) {
693 
694     var dayStartTime = (new Date(startTime)).setHours(0,0,0,0);
695     var dayEndTime = dayStartTime + AjxDateUtil.MSEC_PER_DAY;
696 
697     var workingHours = this._fbCache.getWorkingHrsSlot(dayStartTime, dayEndTime, attendee);
698 
699     //if working hours could not be retrieved consider all time slots for suggestion
700     if(workingHours && workingHours.n) {
701         workingHours = this._fbCache.getWorkingHrsSlot(dayStartTime, dayEndTime, this._organizerEmail);
702         if(workingHours && workingHours.n) return true;
703     }
704 
705     if(!workingHours) return false;
706 
707     var slots = workingHours.f;
708 
709     //working hours are indicated as free slots
710     if(!slots) return false;
711 
712     //convert working hrs relative to the searching time before comparing
713     var slotStartDate, slotEndDate, slotStartTime, slotEndTime;
714     for (var i = 0; i < slots.length; i++) {
715         slotStartDate = new Date(slots[i].s);
716         slotEndDate = new Date(slots[i].e);
717         slotStartTime = (new Date(startTime)).setHours(slotStartDate.getHours(), slotStartDate.getMinutes(), 0, 0);
718         slotEndTime = slotStartTime + (slots[i].e - slots[i].s);
719         if(startTime >= slotStartTime && endTime <= slotEndTime) {
720             return true;
721         }
722     };
723     return false;
724 };

762 ZmScheduleAssistantView.prototype.getMonthFreeBusyInfo =
763 function() {
764     var range = this._miniCalendar.getDateRange();
765     var startDate = range.start;
766     var endDate = range.end;
767 
768     var params = {
769         items: [],
770         itemIndex: {},
771         focus: false,
772         timeFrame: {
773             start: startDate,
774             end: endDate
775         },
776         miniCalSuggestions: true
777     };
778 
779     //avoid suggestions for past date
780     var currentDayTime = (new Date()).setHours(0,0,0,0);
781     if(currentDayTime >= startDate.getTime() && currentDayTime <= endDate.getTime()) {
782         //reset start date if the current date falls within the month date range - to ignore free busy info from the past
783         startDate = params.timeFrame.start = new Date(currentDayTime);
784         if(endDate.getTime() == currentDayTime) {
785             endDate = params.timeFrame.end = new Date(currentDayTime + AjxDateUtil.MSEC_PER_DAY);
786         }
787     }else if(endDate.getTime() < currentDayTime) {
788         //avoid fetching free busy info for dates in the past
789         return;
790     }
791 
792     var list = this._resources;
793     var emails = [], attendeeEmails = [];
794 
795 
796     for (var i = list.length; --i >= 0;) {
797         var item = list[i];
798         var email = item.getEmail();
799         if (email instanceof Array) {
800             email = email[0];
801         }
802         emails.push(email);
803 
804         params.items.push(email);
805 		params.itemIndex[email] = params.items.length -1;
806 
807     }
808 
809     var attendees = this._apptView.getRequiredAttendeeEmails();
810 
811     var attendee;
812     for (var i = attendees.length; --i >= 0;) {
813         attendee = attendees[i];
814         params.items.push(attendee);
815         params.itemIndex[attendee] = params.items.length-1;
816         emails.push(attendee);
817         attendeeEmails.push(attendee);        
818     }
819 
820     params._nonOrganizerAttendeeEmails = attendeeEmails.slice();
821 
822     //include organizer in the scheduler suggestions
823     var organizer = this.getOrganizer();
824     var organizerEmail = organizer.getEmail();
825     params.items.push(organizerEmail);
826     params.itemIndex[organizerEmail] = params.items.length-1;
827     emails.push(organizerEmail);
828     attendeeEmails.push(organizerEmail);
829 
830     params.emails = emails;
831     params.attendeeEmails = attendeeEmails;
832 
833     var callback = new AjxCallback(this, this._handleMonthFreeBusyInfo, [params]);
834     var acct = (appCtxt.multiAccounts)
835             ? this._apptView.getCalendarAccount() : null;
836 
837 
838     var fbParams = {
839         startTime: startDate.getTime(),
840         endTime: endDate.getTime(),
841         emails: emails,
842         callback: callback,
843         errorCallback: callback,
844         noBusyOverlay: true,
845         account: acct
846     };
847 
848     this._monthFreeBusyRequest = this._fbCache.getFreeBusyInfo(fbParams);
849 };

851 ZmScheduleAssistantView.prototype._handleMonthFreeBusyInfo =
852 function(params) {
853 
854     //clear fb request info
855     this._monthFreeBusyRequest = null;
856 
857     if (this._monthWorkingHrsReq) {
858         appCtxt.getRequestMgr().cancelRequest(this._monthWorkingHrsReq, null, true);
859     }
860 
861     var onlyIncludeMyWorkingHours     = this.isOnlyMyWorkingHoursIncluded();
862     var onlyIncludeOthersWorkingHours = this.isOnlyOthersWorkingHoursIncluded();
863 
864     if(!onlyIncludeMyWorkingHours && !onlyIncludeOthersWorkingHours) {
865         this.suggestMonthTimeSlots(params);
866         return;
867     }
868 
869     var organizer = this.getOrganizer();
870     this._organizerEmail = organizer.getEmail();
871 
872     this._workingHoursKey = this.getWorkingHoursKey();
873 
874     var acct = (appCtxt.multiAccounts) ? this._apptView.getCalendarAccount() : null;
875 
876     //optimization: fetch working hrs for a week - wrking hrs pattern repeat everyweek
877     var weekStartDate = new Date(params.timeFrame.start.getTime());
878     var dow = weekStartDate.getDay();
879     weekStartDate.setDate(weekStartDate.getDate()-((dow+7))%7);
880 
881     var emails = onlyIncludeOthersWorkingHours ? params._nonOrganizerAttendeeEmails : null;
882 
883     if (onlyIncludeMyWorkingHours) {
884         emails = emails && emails.concat([this._organizerEmail]);
885     }
886 
887     var whrsParams = {
888         startTime: weekStartDate.getTime(),
889         endTime: weekStartDate.getTime() + 7*AjxDateUtil.MSEC_PER_DAY,
890         emails: emails,
891         callback: new AjxCallback(this, this._handleMonthWorkingHoursResponse, [params]),
892         errorCallback: new AjxCallback(this, this._handleMonthWorkingHoursError, [params]),
893         noBusyOverlay: true,
894         account: acct
895     };
896 
897     this._monthWorkingHrsReq = this._fbCache.getWorkingHours(whrsParams);
898 };

916 ZmScheduleAssistantView.prototype.suggestMonthTimeSlots =
917 function(params) {
918 
919     var startDate = params.timeFrame.start;
920     startDate.setHours(0, 0, 0, 0);
921     var startTime = startDate.getTime();
922     var endTime = params.timeFrame.end.getTime();
923     var duration = this._duration = this._apptView.getDurationInfo().duration;
924 
925     params.duration = duration;
926 
927     this._fbStat = new AjxVector();
928     this._fbStatMap = {};
929     this._totalUsers = this._attendees.length;
930     this._totalLocations =  this._resources.length;
931 
932     params.dates = {};
933     params.colors = {};
934 
935     var key, fbStat, freeSlotFound = false, dayStartTime, dayEndTime;
936 
937     //suggest for entire minicalendar range
938     while(startTime < endTime) {
939 
940         dayStartTime = startTime;
941         dayEndTime = dayStartTime + AjxDateUtil.MSEC_PER_DAY;
942 
943         freeSlotFound = false;
944 
945         while(dayStartTime < dayEndTime) {
946             fbStat = this.computeAvailability(dayStartTime, dayStartTime + duration, params);
947             dayStartTime += AjxDateUtil.MSEC_PER_HALF_HOUR;
948 
949             var roomsAvailable = (fbStat.availableLocations > 0);
950             var includeRooms   = this.isSuggestRooms();
951             if(fbStat && fbStat.availableUsers == this._totalUsers && (!includeRooms || roomsAvailable)) {
952                 this._addColorCode(params, startTime, ZmMiniCalendar.COLOR_GREEN);
953                 freeSlotFound = true;
954                 //found atleast one free slot that can accomodate all attendees and atleast one recources
955                 break;
956             }
957         }
958 
959         if(!freeSlotFound) {                        
960             this._addColorCode(params, startTime, ZmMiniCalendar.COLOR_RED); 
961         }
962 
963         startTime += AjxDateUtil.MSEC_PER_DAY;
964     }
965 
966     this._miniCalendar.setColor(params.dates, true, params.colors);
967 };
