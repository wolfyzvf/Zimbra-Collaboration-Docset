 16 /**
 17  * @class
 18  * This class is the base class of all DWT events.
 19  * 
 20  * @param {boolean} 	__init 	a dummy parameter used for class initialization
 21  * 
 22  * @author Ross Dargahi
 23  * @author Conrad Damon
 24  * 
 25  */
 26 DwtEvent = function(__init) {
 27 	if (arguments.length == 0) return;
 28 	/**
 29 	 * The Dwt object that generated the event
 30 	 * @type DwtControl
 31 	 */
 32 	this.dwtObj = null;
 33 }
 34 
 35 /**
 36  * Returns a string representation of the object.
 37  * 
 38  * @return		{string}		a string representation of the object
 39  */
 40 DwtEvent.prototype.toString = 
 41 function() {
 42 	return "DwtEvent";
 43 }
