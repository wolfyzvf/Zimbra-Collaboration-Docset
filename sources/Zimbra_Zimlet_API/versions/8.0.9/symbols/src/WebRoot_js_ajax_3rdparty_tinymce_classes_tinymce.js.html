281 		 * Performs an iteration of all items in a collection such as an object or array. This method will execure the
282 		 * callback function for each item in the collection, if the callback returns false the iteration will terminate.
283 		 * The callback has the following format: cb(value, key_or_index).
284 		 *
285 		 * @method each
286 		 * @param {Object} o Collection to iterate.
287 		 * @param {function} cb Callback function to execute for each item.
288 		 * @param {Object} s Optional scope to execute the callback in.
289 		 * @example
290 		 * // Iterate an array
291 		 * tinymce.each([1,2,3], function(v, i) {
292 		 *     console.debug("Value: " + v + ", Index: " + i);
293 		 * });
294 		 * 
295 		 * // Iterate an object
296 		 * tinymce.each({a : 1, b : 2, c: 3], function(v, k) {
297 		 *     console.debug("Value: " + v + ", Key: " + k);
298 		 * });
299 		 */
300 		each : function(o, cb, s) {
301 			var n, l;
302 
303 			if (!o)
304 				return 0;
305 
306 			s = s || o;
307 
308 			if (o.length !== undef) {
309 				// Indexed arrays, needed for Safari
310 				for (n=0, l = o.length; n < l; n++) {
311 					if (cb.call(s, o[n], n, o) === false)
312 						return 0;
313 				}
314 			} else {
315 				// Hashtables
316 				for (n in o) {
317 					if (o.hasOwnProperty(n)) {
318 						if (cb.call(s, o[n], n, o) === false)
319 							return 0;
320 					}
321 				}
322 			}
323 
324 			return 1;
325 		},

445 		 * Creates a class, subclass or static singleton.
446 		 * More details on this method can be found in the Wiki.
447 		 *
448 		 * @method create
449 		 * @param {String} s Class name, inheritage and prefix.
450 		 * @param {Object} p Collection of methods to add to the class.
451 		 * @param {Object} root Optional root object defaults to the global window object.
452 		 * @example
453 		 * // Creates a basic class
454 		 * tinymce.create('tinymce.somepackage.SomeClass', {
455 		 *     SomeClass : function() {
456 		 *         // Class constructor
457 		 *     },
458 		 * 
459 		 *     method : function() {
460 		 *         // Some method
461 		 *     }
462 		 * });
463 		 *
464 		 * // Creates a basic subclass class
465 		 * tinymce.create('tinymce.somepackage.SomeSubClass:tinymce.somepackage.SomeClass', {
466 		 *     SomeSubClass: function() {
467 		 *         // Class constructor
468 		 *         this.parent(); // Call parent constructor
469 		 *     },
470 		 * 
471 		 *     method : function() {
472 		 *         // Some method
473 		 *         this.parent(); // Call parent method
474 		 *     },
475 		 * 
476 		 *     'static' : {
477 		 *         staticMethod : function() {
478 		 *             // Static method
479 		 *         }
480 		 *     }
481 		 * });
482 		 *
483 		 * // Creates a singleton/static class
484 		 * tinymce.create('static tinymce.somepackage.SomeSingletonClass', {
485 		 *     method : function() {
486 		 *         // Some method
487 		 *     }
488 		 * });
489 		 */
490 		create : function(s, p, root) {
491 			var t = this, sp, ns, cn, scn, c, de = 0;
492 
493 			// Parse : <prefix> <class>:<super class>
494 			s = /^((static) )?([\w.]+)(:([\w.]+))?/.exec(s);
495 			cn = s[3].match(/(^|\.)(\w+)$/i)[2]; // Class name
496 
497 			// Create namespace for new class
498 			ns = t.createNS(s[3].replace(/\.\w+$/, ''), root);
499 
500 			// Class already exists
501 			if (ns[cn])
502 				return;
503 
504 			// Make pure static class
505 			if (s[2] == 'static') {
506 				ns[cn] = p;
507 
508 				if (this.onCreate)
509 					this.onCreate(s[2], s[3], ns[cn]);
510 
511 				return;
512 			}
513 
514 			// Create default constructor
515 			if (!p[cn]) {
516 				p[cn] = function() {};
517 				de = 1;
518 			}
519 
520 			// Add constructor and methods
521 			ns[cn] = p[cn];
522 			t.extend(ns[cn].prototype, p);
523 
524 			// Extend
525 			if (s[5]) {
526 				sp = t.resolve(s[5]).prototype;
527 				scn = s[5].match(/\.(\w+)$/i)[1]; // Class name
528 
529 				// Extend constructor
530 				c = ns[cn];
531 				if (de) {
532 					// Add passthrough constructor
533 					ns[cn] = function() {
534 						return sp[scn].apply(this, arguments);
535 					};
536 				} else {
537 					// Add inherit constructor
538 					ns[cn] = function() {
539 						this.parent = sp[scn];
540 						return c.apply(this, arguments);
541 					};
542 				}
543 				ns[cn].prototype[cn] = ns[cn];
544 
545 				// Add super methods
546 				t.each(sp, function(f, n) {
547 					ns[cn].prototype[n] = sp[n];
548 				});
549 
550 				// Add overridden methods
551 				t.each(p, function(f, n) {
552 					// Extend methods if needed
553 					if (sp[n]) {
554 						ns[cn].prototype[n] = function() {
555 							this.parent = sp[n];
556 							return f.apply(this, arguments);
557 						};
558 					} else {
559 						if (n != cn)
560 							ns[cn].prototype[n] = f;
561 					}
562 				});
563 			}
564 
565 			// Add static methods
566 			t.each(p['static'], function(f, n) {
567 				ns[cn][n] = f;
568 			});
569 
570 			if (this.onCreate)
571 				this.onCreate(s[2], s[3], ns[cn].prototype);
572 		},

662 		/**
663 		 * Adds an unload handler to the document. This handler will be executed when the document gets unloaded.
664 		 * This method is useful for dealing with browser memory leaks where it might be vital to remove DOM references etc.
665 		 *
666 		 * @method addUnload
667 		 * @param {function} f Function to execute before the document gets unloaded.
668 		 * @param {Object} s Optional scope to execute the function in.
669 		 * @return {function} Returns the specified unload handler function.
670 		 * @example
671 		 * // Fixes a leak with a DOM element that was palces in the someObject
672 		 * tinymce.addUnload(function() {
673 		 *     // Null DOM element to reduce IE memory leak
674 		 *     someObject.someElement = null;
675 		 * });
676 		 */
677 		addUnload : function(f, s) {
678 			var t = this, unload;
679 
680 			unload = function() {
681 				var li = t.unloads, o, n;
682 
683 				if (li) {
684 					// Call unload handlers
685 					for (n in li) {
686 						o = li[n];
687 
688 						if (o && o.func)
689 							o.func.call(o.scope, 1); // Send in one arg to distinct unload and user destroy
690 					}
691 
692 					// Detach unload function
693 					if (win.detachEvent) {
694 						win.detachEvent('onbeforeunload', fakeUnload);
695 						win.detachEvent('onunload', unload);
696 					} else if (win.removeEventListener)
697 						win.removeEventListener('unload', unload, false);
698 
699 					// Destroy references
700 					t.unloads = o = li = w = unload = 0;
701 
702 					// Run garbarge collector on IE
703 					if (win.CollectGarbage)
704 						CollectGarbage();
705 				}

708 			function fakeUnload() {
709 				var d = document;
710 
711 				function stop() {
712 					// Prevent memory leak
713 					d.detachEvent('onstop', stop);
714 
715 					// Call unload handler
716 					if (unload)
717 						unload();
718 
719 					d = 0;
720 				};
721 
722 				// Is there things still loading, then do some magic
723 				if (d.readyState == 'interactive') {
724 					// Fire unload when the currently loading page is stopped
725 					if (d)
726 						d.attachEvent('onstop', stop);
727 
728 					// Remove onstop listener after a while to prevent the unload function
729 					// to execute if the user presses cancel in an onbeforeunload
730 					// confirm dialog and then presses the browser stop button
731 					win.setTimeout(function() {
732 						if (d)
733 							d.detachEvent('onstop', stop);
734 					}, 0);
735 				}
736 			};
737 
738 			f = {func : f, scope : s || this};
739 
740 			if (!t.unloads) {
741 				// Attach unload handler
742 				if (win.attachEvent) {
743 					win.attachEvent('onunload', unload);
744 					win.attachEvent('onbeforeunload', fakeUnload);
745 				} else if (win.addEventListener)
746 					win.addEventListener('unload', unload, false);
747 
748 				// Setup initial unload handler array
749 				t.unloads = [f];
750 			} else
751 				t.unloads.push(f);
752 
753 			return f;
754 		},

758 		 *
759 		 * @method removeUnload
760 		 * @param {function} f Function to remove from unload handler list.
761 		 * @return {function} Removed function name or null if it wasn't found.

763 		removeUnload : function(f) {
764 			var u = this.unloads, r = null;
765 
766 			tinymce.each(u, function(o, i) {

768 					u.splice(i, 1);
769 					r = f;
770 					return false;
771 				}
772 			});
773 
774 			return r;

779 		 *
780 		 * @method explode
781 		 * @param {string} s String to split.
782 		 * @param {string} d Delimiter to split by.

784 		 * // Split a string into an array with a,b,c
785 		 * var arr = tinymce.explode('a, b,   c');

789 				return s;
