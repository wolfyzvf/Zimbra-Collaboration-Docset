 21 /**
 22  * Creates a request manager.
 23  * @class
 24  * This class manages the sending of requests to the server, and handles the
 25  * responses, including refresh blocks and notifications.
 26  *
 27  * @author Conrad Damon
 28  * 
 29  * @param {ZmController}	controller	the main controller
 30  */
 31 ZmRequestMgr = function(controller) {
 32 
 33 	this._controller = controller;
 34 	
 35 	appCtxt.setRequestMgr(this);
 36 
 37 	ZmCsfeCommand.setServerUri(appCtxt.get(ZmSetting.CSFE_SERVER_URI));
 38 	var cv = appCtxt.get(ZmSetting.CLIENT_VERSION);
 39 	ZmCsfeCommand.clientVersion = (!cv || cv.indexOf('@') == 0) ? "dev build" : cv;
 40 	
 41 	this._shell = appCtxt.getShell();
 42 
 43     this._highestNotifySeen = 0;
 44 
 45 	this._cancelActionId = {};
 46 	this._pendingRequests = {};
 47 
 48 	this._useXml = appCtxt.get(ZmSetting.USE_XML);
 49 	this._logRequest = appCtxt.get(ZmSetting.LOG_REQUEST);
 50 	this._stdTimeout = appCtxt.get(ZmSetting.TIMEOUT);
 51 
 52 	this._unreadListener = new AjxListener(this, this._unreadChangeListener);
 53 };
 54 
 55 // request states
 56 ZmRequestMgr._SENT		= 1;
 57 ZmRequestMgr._RESPONSE	= 2;
 58 ZmRequestMgr._CANCEL	= 3;
 59 
 60 // retry settings
 61 ZmRequestMgr.RETRY_MAX			= 2;	// number of times to retry before throwing exception
 62 ZmRequestMgr.RETRY_DELAY		= 5;	// seconds to delay between retries
 63 ZmRequestMgr.RETRY_ON_EXCEPTION = {};	// which exceptions to retry on
 64 ZmRequestMgr.RETRY_ON_EXCEPTION[ZmCsfeException.EMPTY_RESPONSE] = true;
 65 
 66 ZmRequestMgr._nextReqId = 1;
 67 
 68 ZmRequestMgr.OFFLINE_HEAP_DUMP          = "heapdump_upload";
 69 ZmRequestMgr.OFFLINE_MUST_RESYNC        = "resync";
 70 ZmRequestMgr.OFFLINE_FOLDER_MOVE_FAILED = "foldermove_failed";

 72  * Returns a string representation of the object.
 73  * 
 74  * @return		{String}		a string representation of the object
 75  */
 76 ZmRequestMgr.prototype.toString =
 77 function() {
 78 	return "ZmRequestMgr";

 81 /**
 82  * Sends a request to the CSFE and processes the response. Notifications and
 83  * refresh blocks that come in the response header are handled. Also handles
 84  * exceptions by default, though the caller can pass in a special callback to
 85  * run for exceptions. The error callback should return true if it has
 86  * handled the exception, and false if standard exception handling should still
 87  * be performed.
 88  *
 89  * @param {Hash}		params				a hash of parameters
 90  * @param {AjxSoapDoc}	soapDoc				the SOAP document that represents the request
 91  * @param {Object}		jsonObj				the JSON object that represents the request (alternative to soapDoc)
 92  * @param {Boolean}		asyncMode			if <code>true</code>, request will be made asynchronously
 93  * @param {AjxCallback}	callback			the next callback in chain for async request
 94  * @param {AjxCallback}	errorCallback		the callback to run if there is an exception
 95  * @param {AjxCallback}	continueCallback	the callback to run after user re-auths
 96  * @param {int}			timeout				the timeout value (in seconds)
 97  * @param {Boolean}		noBusyOverlay		if <code>true</code>, don't use the busy overlay
 98  * @param {String}		accountName			the name of account to execute on behalf of
 99  * @param {Object}		response			the pre-determined response (no request will be made)
100  * @param {Boolean}		skipAuthCheck		if <code>true</code>, do not check if auth token has changed
101  * @param {constant}	resend				the reason for resending request
102  * @param {Boolean}		sensitive			if <code>true</code>, attempt to use secure conn to protect data
103  * @param {Boolean}		noSession			if <code>true</code>, no session info is included
104  * @param {String}		restUri				the REST URI to send the request to
105  * @param {boolean}		emptyResponseOkay	if true, empty or no response from server is not an erro
106  * @param {boolean}		useChangeToken	    if true, request will try to use change token in header
107  */
108 ZmRequestMgr.prototype.sendRequest =
109 function(params) {
110 
111 	var response = params.response;
112 	if (response) {
113 		if (params.reqId) {
114 			params = this._pendingRequests[params.reqId] || params;
115 			params.response = response;
116 		}
117 		params.asyncMode = true;	// canned response set up async style
118 		return this._handleResponseSendRequest(params, new ZmCsfeResult(response));
119 	}
120 	
121 	var reqId = params.reqId = ("Req_"+ZmRequestMgr._nextReqId++);
122 	DBG.println("req", "assign req ID: " + reqId);
123 	var timeout = params.timeout = (params.timeout != null) ? params.timeout : this._stdTimeout;
124 	if (timeout) {
125 		timeout = timeout * 1000; // convert seconds to ms
126 	}
127 	var asyncCallback = params.asyncMode ? new AjxCallback(this, this._handleResponseSendRequest, [params]) : null;
128 
129 	if (params.sensitive) {
130 		DBG.println(AjxDebug.DBG2, "request contains sensitive data");
131 		// NOTE: If only http mode is available, there's nothing we can
132 		//       do. And if we're already using https mode, then there's
133 		//       nothing we need to do. We only attempt to send the
134 		//       request securely if mixed mode is enabled and the app
135 		//       was loaded using http.
136 		var isHttp = document.location.protocol == ZmSetting.PROTO_HTTP;
137 		var isMixedMode = appCtxt.get(ZmSetting.PROTOCOL_MODE) == ZmSetting.PROTO_MIXED;
138 		if(isHttp && isMixedMode) {
139 			return this._sensitiveRequest(params, reqId);
140 		}
141 	}
142 
143 	var command = new ZmCsfeCommand();
144 	// bug fix #10652, 82704 - dont set change token if accountName is not main account or is not specified
145 	// (since we're executing on someone else's mbox)
146 	var accountName = params.accountName;
147 	if (!accountName) {
148 		var acct = appCtxt.getActiveAccount();
149 		accountName = (acct && acct.id != ZmAccountList.DEFAULT_ID) ? acct.name : null;
150 	}
151 	var changeToken = null;
152 	if (params.useChangeToken && (!accountName || (accountName === appCtxt.accountList.mainAccount.name))) {
153 		changeToken = this._changeToken;
154 	}
155 	var cmdParams, methodName;
156 
157 	if (params.restUri) {
158 		cmdParams =	{	restUri:			params.restUri,
159 						asyncMode:			params.asyncMode,
160 						callback:			asyncCallback
161 					};
162 	} else {
163 		cmdParams = {	jsonObj:			params.jsonObj,
164 						soapDoc:			params.soapDoc,
165 						accountName:		accountName,
166 						useXml:				this._useXml,
167 						changeToken:		changeToken,
168 						asyncMode:			params.asyncMode,
169 						callback:			asyncCallback,
170 						logRequest:			this._logRequest,
171 						highestNotifySeen:	this._highestNotifySeen,
172 						noAuthToken:		true,	// browser will handle auth token cookie
173 						skipAuthCheck:		params.skipAuthCheck,
174 						resend:				params.resend,
175 						noSession:			params.noSession,
176 						useStringify1:		AjxEnv.isIE && params.fromChildWindow,
177 						emptyResponseOkay:	params.emptyResponseOkay
178 					};
179 		methodName = params.methodName = ZmCsfeCommand.getMethodName(cmdParams.jsonObj || cmdParams.soapDoc);
180 	}
181 
182 	appCtxt.currentRequestParams = params;
183 	DBG.println("req", "send request " + reqId + ": " + methodName);
184 	var cancelParams = timeout ? [reqId, params.errorCallback, params.noBusyOverlay] : null;
185 	if (!params.noBusyOverlay) {
186 		var cancelCallback = null;
187 		var showBusyDialog = false;
188 		if (timeout) {
189 			DBG.println("req", "ZmRequestMgr.sendRequest: timeout for " + reqId + " is " + timeout);
190 			cancelCallback = new AjxCallback(this, this.cancelRequest, cancelParams);
191 			this._shell.setBusyDialogText(ZmMsg.askCancel);
192 			showBusyDialog = true;
193 		}
194 		// put up busy overlay to block user input
195 		this._shell.setBusy(true, reqId, showBusyDialog, timeout, cancelCallback);
196 	} else if (timeout) {
197 		var action = new AjxTimedAction(this, this.cancelRequest, cancelParams);
198 		this._cancelActionId[reqId] = AjxTimedAction.scheduleAction(action, timeout);
199 		DBG.println("req", "schedule cancel action for reqId " + reqId + ": " + this._cancelActionId[reqId]);
200 	}
201 
202 	this._pendingRequests[reqId] = command;
203 
204 	try {
205 		DBG.println("req", "invoke req: " + params.reqId);
206 		var response = params.restUri ? command.invokeRest(cmdParams) : command.invoke(cmdParams);
207 		command.state = ZmRequestMgr._SENT;
208 	} catch (ex) {
209 		DBG.println("req", "caught exception on invoke of req: " + params.reqId);
210 		this._handleResponseSendRequest(params, new ZmCsfeResult(ex, true));
211 		return;
212 	}
213 
214 	return (params.asyncMode) ? reqId : (this._handleResponseSendRequest(params, response));
215 };

217 /**
218  * @private
219  * @param {Array}	params.ignoreErrs	list of error codes that can be ignored, when params.errorCallback does not exists.
220  */
221 ZmRequestMgr.prototype._handleResponseSendRequest =
222 function(params, result) {
223 	DBG.println("req", "ZmRequestMgr.handleResponseSendRequest for req: " + params.reqId);
224 	var isCannedResponse = (params.response != null);
225 	if (!isCannedResponse) {
226 		if (!this._pendingRequests[params.reqId]) {
227 			DBG.println("req", "ZmRequestMgr.handleResponseSendRequest no pending request for " + params.reqId);
228 			return;
229 		}
230 		if (this._pendingRequests[params.reqId].state == ZmRequestMgr._CANCEL) {
231 			DBG.println("req", "ZmRequestMgr.handleResponseSendRequest state=CANCEL for " + params.reqId);
232 			return;
233 		}
234 	
235 		this._pendingRequests[params.reqId].state = ZmRequestMgr._RESPONSE;
236 	
237 		if (!params.noBusyOverlay) {
238 			this._shell.setBusy(false, params.reqId); // remove busy overlay
239 		} else if (params.timeout) {
240 		}
241 	}
242 
243 	var response, refreshBlock;
244 	try {
245 		if (params.asyncMode && !params.restUri) {
246 			response = result.getResponse(); // may throw exception
247 		} else {
248 			// for sync responses, manually throw exception if necessary
249 			if (result._isException) {
250 				throw result._data;
251 			} else {
252 				response = result;
253 			}
254 		}
255 		if (response.Header) {
256 			refreshBlock = this._handleHeader(response.Header);
257 		}
258 	} catch (ex) {
259 		DBG.println("req", "Request " + params.reqId + " got an exception");
260 		var ecb = params.errorCallback;
261 		if (ecb) {
262 			var handled = ecb.isAjxCallback ? ecb.run(ex) : ecb(ex);
263 			if (!handled) {
264 				this._handleException(ex, params);
265 			}
266 		} else {
267 			var ignore = function(ignoreErrs, errCode){
268 			/*
269 				Checks errCode exits in ignoreErrs
270 			*/
271 				if (ignoreErrs && (ignoreErrs.length > 0)){
272 					for (var val in ignoreErrs)
273 						if (ignoreErrs[val] == errCode) 
274 							return true;
275 				}
276 				return false;
277 			}(params.ignoreErrs, ex.code)
278 			
279 			if (!ignore)
280 				this._handleException(ex, params);
281 		}
282 		var hdr = result.getHeader();
283 		if (hdr) {
284 			this._handleHeader(hdr);
285 			this._handleNotifications(hdr);
286 		}
287 		this._clearPendingRequest(params.reqId);
288 		return;
289 	}
290 
291 	if (params.asyncMode && !params.restUri) {
292 		result.set(response.Body);
293 	}
294 
295     // if we didn't get an exception, then we should make sure that the
296     // poll timer is running (just in case it got an exception and stopped)
297 	if (!appCtxt.isOffline && !isCannedResponse) {
298 		this._controller._kickPolling(true);
299 	}
300 
301 	var methodName = ZmCsfeCommand.getMethodName(params.jsonObj || params.soapDoc);
302 	if (params.asyncMode && params.callback) {
303 		DBG.println(AjxDebug.DBG1, "------------------------- Running response callback for " + methodName);
304 		var cb = params.callback;
305 		cb.isAjxCallback ? cb.run(result) : cb(result);
306 	}
307 
308 	DBG.println(AjxDebug.DBG1, "------------------------- Processing notifications for " + methodName);
309 	this._handleNotifications(response.Header);
310 
311 	this._clearPendingRequest(params.reqId);
312 
313 	if (refreshBlock && (!appCtxt.isOffline || !appCtxt.multiAccounts) && !params.more) {
314 		this._refreshHandler(refreshBlock);
315 	}
316 	
317 	if (!params.asyncMode) {
318 		return response.Body;
319 	}
320 	
321 	var ctlr = this._controller;
322 	if (ctlr._evtMgr && ctlr._evtMgr.isListenerRegistered(ZmAppEvent.RESPONSE)) {
323 		ctlr._evt.request = methodName;
324 		ctlr.notify(ZmAppEvent.RESPONSE);
325 	}
326 };

328 /**
329  * Cancels the request.
330  * 
331  * @param	{String}	reqId		the request id
332  * @param	{AjxCallback}	errorCallback		the callback
333  * @param	{Boolean}	noBusyOverlay	if <code>true</code>, do not show busy overlay
334  */
335 ZmRequestMgr.prototype.cancelRequest =
336 function(reqId, errorCallback, noBusyOverlay) {
337 	DBG.println("req", "ZmRequestMgr.cancelRequest: " + reqId);
338 	if (!this._pendingRequests[reqId]) { return; }
339 	if (this._pendingRequests[reqId].state == ZmRequestMgr._RESPONSE) { return; }
340 
341 	this._pendingRequests[reqId].state = ZmRequestMgr._CANCEL;
342 	if (!noBusyOverlay) {
343 		this._shell.setBusy(false, reqId);
344 	}
345 	DBG.println("req", "canceling the XHR");
346 	this._pendingRequests[reqId].cancel();
347 	if (errorCallback) {
348 		DBG.println("req", "calling the error callback");
349 		var ex = new AjxException("Request canceled", AjxException.CANCELED, "ZmRequestMgr.prototype.cancelRequest");
350 		errorCallback.isAjxCallback ? errorCallback.run(ex) : errorCallback(ex);
351 	}
352 	this._clearPendingRequest(reqId);
353 };
354 
355 /**
356  * @private
357  */
358 ZmRequestMgr.prototype._clearPendingRequest =
359 function(reqId) {
360 	var request = this._pendingRequests[reqId];
361 	if (request) {
362 		if (request.iframeId) {
363 			var iframe = document.getElementById(request.iframeId);
364 			if (iframe) {
365 				iframe.parentNode.removeChild(iframe);
366 			}
367 		}
368 		delete this._pendingRequests[reqId];
369 	}
370 	var cancelId = this._cancelActionId[reqId];
371 	if (cancelId && cancelId != -1) {
372 		DBG.println("req", "unschedule cancel action for reqId " + reqId + ": " + cancelId);
373 		AjxTimedAction.cancelAction(cancelId);
374 		this._cancelActionId[reqId] = -1;
375 	}
376 };

378 /**
379  * Handles a response's SOAP header, except for notifications. Updates our
380  * change token, and processes a <code><refresh></code> block if there is one (happens
381  * when a new session is created on the server).
382  *
383  * @param {Object}	hdr	a SOAP header
384  * 
385  * @private
386  */
387 ZmRequestMgr.prototype._handleHeader =
388 function(hdr) {
389 
390 	var ctxt = hdr && hdr.context;
391 	if (!ctxt) { return; }
392 
393 	// update change token if we got one
394 	if (ctxt.change) {
395 		this._changeToken = ctxt.change.token;
396 	}
397 
398 	// offline/zdesktop only
399 	if (ctxt.zdsync && ctxt.zdsync.account) {
400 		var acctList = ctxt.zdsync.account;
401 		for (var i = 0; i < acctList.length; i++) {
402             var acct = appCtxt.accountList.getAccount(acctList[i].id);
403 			if (acct) {
404                 //server is sending info to get user's consent on something.
405                 var dialog = acctList[i].dialog;
406                 if(dialog) {
407                     this._handleOfflineInfoDialog(dialog[0], acct)
408                 }
409 				acct.updateState(acctList[i]);
410 			}
411 		}
412 	}
413 
414 	if (ctxt.refresh) {
415 		this._controller.runAppFunction("_clearDeferredFolders");
416 		this._loadTrees(ctxt.refresh);
417 		this._controller.runAppFunction("_createVirtualFolders");
418 		this._highestNotifySeen = 0;
419 	}
420 
421 	return ctxt.refresh;
422 };
423 /**
424  * Handles server's notification to get user's consent on something
425  *
426  * @param {Object}	dlg is json object
427  * @param {Object}	account object
428  *
429  * @private
430  */
431 ZmRequestMgr.prototype._handleOfflineInfoDialog =
432 function(dlg, acct) {
433 
434     if(!dlg.type) {
435         return;
436     }
437     var cont;
438     switch(dlg.type) {
439         case ZmRequestMgr.OFFLINE_HEAP_DUMP: {
440             cont = ZmMsg.offlineHeapDump;
441             break;
442         }
443         case ZmRequestMgr.OFFLINE_MUST_RESYNC: {
444             cont = AjxMessageFormat.format(ZmMsg.offlineMustReSync, acct.name);
445             break;
446         }
447         case ZmRequestMgr.OFFLINE_FOLDER_MOVE_FAILED: {
448             appCtxt.setStatusMsg(ZmMsg.offlineMoveFolderError);
449             cont = "";
450             break;
451         }
452         default: {
453             cont = "";
454             break;
455         }
456     }
457     if(!cont || cont == "") {
458         return;
459     }
460     var dialog = appCtxt.getOkCancelMsgDialog();
461     dialog.setMessage(cont);
462     dialog.registerCallback(DwtDialog.OK_BUTTON, this._handleOfflineDialogAction, this, [dialog, dlg.type, acct.id, true]);
463     dialog.registerCallback(DwtDialog.CANCEL_BUTTON, this._handleOfflineDialogAction, this, [dialog, dlg.type, acct.id, false]);
464     dialog.popup();
465 };
466 /**
467  * Sends DialogActionRequest with user's consent YES/NO
468  * @param {object} dlg is getOkCancelMsgDialog
469  * @param {string} type
470  * @param {string} acctId Account ID
471  * @param {boolean} action
472  */
473 ZmRequestMgr.prototype._handleOfflineDialogAction =
474 function(dlg, type, acctId, action) {
475     var args = {
476 			jsonObj: { DialogActionRequest: { _jsns: "urn:zimbraOffline", type: type, id:acctId, action: action ? "yes" : "no" } },
477             callback: new AjxCallback(this, this._handleOfflineDialogActionResp, dlg),
478 			errorCallback: new AjxCallback(this, this._handleOfflineDialogActionResp, dlg),
479 			asyncMode: true
480 		};
481     this.sendRequest(args);
482 };

484  * callback to hide dialog
485  *
486  * @param dlg

492         dlg.popdown();
493     }

496 /**
497  * For transient network exceptions, retry the request after a small delay.
498  * We will only retry a limited number of times.
499  * 
500  * @param {AjxException}	ex			the exception
501  * @param {Hash}	params		a hash of the original request params
502  */
503 ZmRequestMgr.prototype._handleException =
504 function(ex, params) {
505 	var handled = false;
506 	if (ZmRequestMgr.RETRY_ON_EXCEPTION[ex.code]) {
507 		params.retryCount = params.retryCount || 0;
508 		if (params.retryCount < ZmRequestMgr.RETRY_MAX) {
509 			DBG.println(AjxDebug.DBG1, "RETRY " + ex.method + " due to " + ex.code);
510 			params.resend = ZmCsfeCommand.RETRY;
511 			params.retryCount++;
512 			AjxTimedAction.scheduleAction(new AjxTimedAction(this, 
513 				function() {
514 					this.sendRequest(params);
515 				}), ZmRequestMgr.RETRY_DELAY * 1000);
516 			handled = true;
517 		}
518 	}
519 	
520 	if (!handled) {
521 		this._controller._handleException(ex, params);
522 	}
523 };

525 /**
526  * Handles the <code><notify></code> block of a response's SOAP header.
527  *
528  * @param {Object}	hdr	a SOAP header
529  * 
530  * @private
531  */
532 ZmRequestMgr.prototype._handleNotifications =
533 function(hdr) {
534 
535 	if (hdr && hdr.context && hdr.context.notify) {
536         for (var i = 0; i < hdr.context.notify.length; i++) {
537         	var notify = hdr.context.notify[i];
538         	var seq = notify.seq;
539             // BUG?  What if the array isn't in sequence-order?  Could we miss notifications?
540 			var sid = hdr.context && ZmCsfeCommand.extractSessionId(hdr.context.session);
541             if (notify.seq > this._highestNotifySeen && !(sid && ZmCsfeCommand._staleSession[sid])) {
542                 DBG.println(AjxDebug.DBG1, "Handling notification[" + i + "] seq=" + seq);
543                 this._highestNotifySeen = seq;
544                 this._notifyHandler(notify);
545             } else {
546             	DBG.println(AjxDebug.DBG1, "SKIPPING notification[" + i + "] seq=" + seq + " highestNotifySeen=" + this._highestNotifySeen);
547 	      	}
548     	}
549 	}
550 };

552 /**
553  * A <code><refresh></code> block is returned in a SOAP response any time the session ID has 
554  * changed. It always happens on the first SOAP command (GetInfoRequest).
555  * After that, it happens after a session timeout. 
556  * 
557  * @param {Object}	refresh	the refresh block (JSON)
558  * @private
559  */
560 ZmRequestMgr.prototype._refreshHandler =
561 function(refresh) {
562 
563 	DBG.println(AjxDebug.DBG1, "Handling REFRESH");
564 	AjxDebug.println(AjxDebug.NOTIFY, "REFRESH block received");
565 	if (!appCtxt.inStartup) {
566 		this._controller._execPoll();
567 	}
568 
569 	if (refresh.version) {
570 		if (!this._canceledReload) {
571 			var curVersion = appCtxt.get(ZmSetting.SERVER_VERSION);
572 			if (curVersion != refresh.version) {
573 				appCtxt.set(ZmSetting.SERVER_VERSION, refresh.version);
574 				if (curVersion) {
575 					var dlg = appCtxt.getYesNoMsgDialog();
576 					dlg.reset();
577 					dlg.registerCallback(DwtDialog.YES_BUTTON, this._reloadYesCallback, this, [dlg, curVersion, refresh.version]);
578 					dlg.registerCallback(DwtDialog.NO_BUTTON, this._reloadNoCallback, this, [dlg, refresh]);
579 					var msg = AjxMessageFormat.format(ZmMsg.versionChangeRestart, [curVersion, refresh.version]);
580 					dlg.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
581 					dlg.popup();
582 					return;
583 				}
584 			}
585 		}
586 	}
587 
588 	if (!this._recentlyRefreshed) {
589 		// Run any app-requested refresh routines
590 		this._controller.runAppFunction("refresh", false, refresh);
591 		this._recentlyRefreshed = true;
592 		this._lastSkippedRefresh = null;
593 	} else {
594 		this._lastSkippedRefresh = refresh;
595 	}
596 
597 	if (!this._refreshTimer)
598 		this._refreshTimer = new AjxTimedAction(this, this._refreshTimeout);
599 
600 	AjxTimedAction.scheduleAction(this._refreshTimer, 5000);
601 };

603 ZmRequestMgr.prototype._refreshTimeout =
604 function() {
605 	if (this._lastSkippedRefresh) {
606 		this._controller.runAppFunction("refresh", false, this._lastSkippedRefresh);
607 		this._lastSkippedRefresh = null;
608 	}
609 	this._recentlyRefreshed = false;
610 };

664 /**
665  * To handle notifications, we keep track of all the models in use. A model could
666  * be an item, a list of items, or an organizer tree. Currently we never get an
667  * organizer by itself.
668  * 
669  * @private
670  */
671 ZmRequestMgr.prototype._notifyHandler =
672 function(notify) {
673 	DBG.println(AjxDebug.DBG1, "Handling NOTIFY");
674 	AjxDebug.println(AjxDebug.NOTIFY, "Notification block:");
675 	AjxDebug.dumpObj(AjxDebug.NOTIFY, notify);
676 	this._controller.runAppFunction("preNotify", false, notify);
677 	if (notify.deleted && notify.deleted.id) {
678 		this._handleDeletes(notify.deleted);
679 	}
680 	if (notify.created) {
681 		this._handleCreates(notify.created);
682 	}
683 	if (notify.modified) {
684 		this._handleModifies(notify.modified);
685 	}
686 	this._controller.runAppFunction("postNotify", false, notify);
687 };
688 
689 /**
690  * A delete notification hands us a list of IDs which could be anything. First, we
691  * run any app delete handlers. Any IDs which have been handled by an app will
692  * be nulled out. The generic handling here will be applied to the rest - the item is
693  * retrieved from the item cache and told it has been deleted.
694  *
695  * @param {Object}	deletes	the node containing all 'deleted' notifications
696  * 
697  * @private
698  */
699 ZmRequestMgr.prototype._handleDeletes =
700 function(deletes) {
701 	var ids = deletes.id.split(",");
702 	this._controller.runAppFunction("deleteNotify", false, ids);
703 
704 	for (var i = 0; i < ids.length; i++) {
705 		var id = ids[i];
706 		if (!id) { continue; }
707 		var item = appCtxt.cacheGet(id);
708 		DBG.println(AjxDebug.DBG2, "ZmRequestMgr: handling delete notif for ID " + id);
709 		if (item && item.notifyDelete) {
710 			item.notifyDelete();
711 			appCtxt.cacheRemove(id);
712 			item = null;
713 		}
714 	}
715 };
716 
717 /**
718  * Create notifications hand us full XML nodes. First, we run any app
719  * create handlers, which will mark any create nodes that they handle. Remaining
720  * creates are handled here.
721  * 
722  * @param {Object}	creates	the node containing all 'created' notifications
723  * 
724  * @private
725  */
726 ZmRequestMgr.prototype._handleCreates =
727 function(creates) {
728 	this._controller.runAppFunction("createNotify", false, creates);
729 
730 	for (var name in creates) {
731         if (creates.hasOwnProperty(name)) {
732             var list = creates[name];
733             for (var i = 0; i < list.length; i++) {
734                 var create = list[i];
735                 if (create._handled) { continue; }
736                 // ignore create notif for item we already have (except tags, which can reuse IDs)
737                 if (appCtxt.cacheGet(create.id) && name != "tag") { continue; }
738 
739                 DBG.println(AjxDebug.DBG1, "ZmRequestMgr: handling CREATE for node: " + name);
740                 if (name == "tag") {
741                     var account = appCtxt.multiAccounts && ZmOrganizer.parseId(create.id).account;
742                     var tagTree = appCtxt.getTagTree(account);
743                     if (tagTree) {
744                         tagTree.root.notifyCreate(create);
745                     }
746                 } else if (name == "folder" || name == "search" || name == "link") {
747                     var parentId = create.l;
748                     var parent = appCtxt.getById(parentId);
749                     if (parent && parent.notifyCreate && parent.type != ZmOrganizer.TAG) { // bug #37148
750                         parent.notifyCreate(create, name);
751                     }
752                 }
753             }
754         }
755 	}
756 };
757 
758 /**
759  * First, we run any app modify handlers, which will mark any nodes that
760  * they handle. Remaining modify notifications are handled here.
761  * 
762  * @param {Object}	modifies	the node containing all 'modified' notifications
763  * 
764  * @private
765  */
766 ZmRequestMgr.prototype._handleModifies =
767 function(modifies) {
768 
769 	this._controller.runAppFunction("modifyNotify", false, modifies);
770 
771 	for (var name in modifies) {
772 		if (name == "mbx") {
773 			var mboxes = modifies[name];
774 			for (var i = 0; i < mboxes.length; i++) {
775 				var mbox = mboxes[i];
776 				var acctId = mbox.acct;
777 				var account = acctId && appCtxt.accountList.getAccount(acctId);
778 				var setting = appCtxt.getSettings(account).getSetting(ZmSetting.QUOTA_USED);
779 				setting.notifyModify({_name:name, s:mbox.s, account:account});
780 			}
781 			continue;
782 		}
783 
784 		var list = modifies[name];
785 		for (var i = 0; i < list.length; i++) {
786 			var mod = list[i];
787 			if (mod._handled) { continue; }
788 			DBG.println(AjxDebug.DBG2, "ZmRequestMgr: handling modified notif for ID " + mod.id + ", node type = " + name);
789 			var item = appCtxt.cacheGet(mod.id);
790 
791 			// bug fix #31991 - for contact modifies, check the contact list
792 			// Since we lazily create ZmContact items, it wont be in the global cache.
793 			// TODO: move to contacts app
794 			if (!item && name == "cn" && AjxDispatcher.loaded("ContactsCore")) {
795 				var capp = appCtxt.getApp(ZmApp.CONTACTS);
796 				if (capp.isContactListLoaded()) {
797 					item = capp.getContactList().getById(mod.id);
798 				}
799 			}
800 
801 			if (item && item.notifyModify) {
802 				mod._isRemote = (name == "folder" && item.link);	// remote subfolder
803 				item.notifyModify(mod);
804 			}
805 		}
806 	}
807 };

809 /**
810  * Changes browser title if it's a folder or tag whose unread count has changed.
811  *
812  * @param ev	the event
813  * 
814  * @private
815  */
816 ZmRequestMgr.prototype._unreadChangeListener =
817 function(ev) {
818 	if (ev.event == ZmEvent.E_MODIFY) {
819 		var fields = ev.getDetail("fields");
820 		if (fields && fields[ZmOrganizer.F_UNREAD]) {
821 			var organizers = ev.getDetail("organizers");
822 			var organizer = organizers ? organizers[0] : null;
823 			var id = organizer ? (organizer.isSystem() ? organizer.nId : organizer.id) : null;
824 			var search = appCtxt.getCurrentSearch();
825 			if (search) {
826 				var searchFolder = appCtxt.multiAccounts && appCtxt.getById(search.folderId);
827 				var searchFolderId = (searchFolder && searchFolder.getAccount() == appCtxt.getActiveAccount())
828 					? searchFolder.nId : search.folderId;
829 
830 				if (id && (id == searchFolderId || id == search.tagId)) {
831 					Dwt.setTitle(search.getTitle());
832 				}
833 			}
834 			var mailApp = appCtxt.getApp(ZmApp.MAIL);
835 			if (mailApp) {
836 				mailApp.setNewMailNotice(organizer);
837 			}
838 		}
839 	}
840 };

842 ZmRequestMgr.prototype._sensitiveRequest =
843 function(params, reqId) {
844 	DBG.println(AjxDebug.DBG2, "sending request securely");
845 	// adjust command parameters
846 	// TODO: Because of timing issues, should we not use session info?
847 	// TODO: But for batch commands, some updates would not be seen immediately.
848 	// TODO: To avoid security warning, send response in URL; so limit length
849 	params.noSession = true;
850 
851 	// information
852 	var requestStr = ZmCsfeCommand.getRequestStr(params);
853 	var loc = document.location;
854 	var port = appCtxt.get(ZmSetting.HTTPS_PORT);
855 	if (port && port != ZmSetting.DEFAULT_HTTPS_PORT) {
856 		port = ":"+port;
857 	}
858 
859 	// create iframe
860 	var iframe = document.createElement("IFRAME");
861 	iframe.style.display = "none";
862 	iframe.id = Dwt.getNextId();
863 	document.body.appendChild(iframe);
864 
865 	// set contents
866 	var iframeDoc = Dwt.getIframeDoc(iframe);
867 	iframeDoc.write(
868 		"<form ",
869 			"id=",iframe.id,"-form ",
870 			"target=",iframe.id,"-iframe ",
871 			"method=POST ",
872 			"action='https://",loc.hostname,port,appContextPath,"/public/secureRequest.jsp'",
873 		">",
874 			"<input type=hidden name=reqId value='",reqId,"'>",
875 			"<textarea name=data>",
876 				AjxStringUtil.htmlEncode(requestStr),
877 			"</textarea>",
878 		"</form>",
879 		"<iframe name=",iframe.id,"-iframe></iframe>"
880 	);
881 	iframeDoc.close();
882 
883 	// save the params for the response
884 	params.iframeId = iframe.id;
885 	this._pendingRequests[reqId] = params;
886 
887 	// submit form
888 	var form = iframeDoc.getElementById(iframe.id+"-form");
889 	form.submit();
890 };
