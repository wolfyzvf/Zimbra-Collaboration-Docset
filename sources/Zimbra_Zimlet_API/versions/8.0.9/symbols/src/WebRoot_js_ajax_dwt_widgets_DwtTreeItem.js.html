 17 /**
 18  * Creates a Tree Item.
 19  * @constructor
 20  * @class
 21  * This class implements a tree item widget.
 22  *
 23  * @author Ross Dargahi
 24  * 
 25  * @param {hash}	params				a hash of parameters
 26  * @param {DwtComposite}      params.parent	the parent widget
 27  * @param {number}      params.index 			the index at which to add this control among parent's children
 28  * @param {string}      params.text 					the label text for the tree item
 29  * @param {string}      params.imageInfo			the icon for the left end of the tree item
 30  * @param {string}      params.extraInfo				the icon for the right end of the tree item
 31  * @param {string}      params.expandNodeImage		the icon to use for expanding tree item (instead of default)
 32  * @param {string}      params.collapseNodeImage     the icon to use for collapsing tree item (instead of default)
 33  * @param {string}      params.className				the CSS class
 34  * @param {constant}      params.posStyle				the positioning style (see {@link DwtControl})
 35  * @param {boolean}      params.deferred				if <code>true</code>, postpone initialization until needed.
 36  * @param {boolean}      params.selectable			if <code>true</code>, this item is selectable
 37  * @param {boolean}      params.forceNotifySelection	force notify selection even if checked style
 38  * @param {boolean}      params.forceNotifyAction		force notify action even if checked style
 39  * @param {boolean}      params.singleClickAction		if <code>true</code>, an action is performed in single click
 40  * @param {AjxCallback}      params.dndScrollCallback	the callback triggered when scrolling of a drop area for an object being dragged
 41  * @param {string}      params.dndScrollId			the id
 42  *        
 43  * @extends		DwtComposite		
 44  */
 45 DwtTreeItem = function(params) {
 46 
 47     if (arguments.length == 0) { return; }    
 48 
 49     params = Dwt.getParams(arguments, DwtTreeItem.PARAMS);
 50 	var parent = params.parent;
 51 	if (parent instanceof DwtTree) {
 52 		this._tree = parent;
 53 	} else if (parent instanceof DwtTreeItem) {
 54 		this._tree = parent._tree;
 55 	} else {
 56 		throw new DwtException("DwtTreeItem parent must be a DwtTree or DwtTreeItem", DwtException.INVALIDPARENT, "DwtTreeItem");
 57 	}
 58 
 59 	this._origClassName = params.className || "DwtTreeItem";
 60 	this._textClassName = [this._origClassName, "Text"].join("-");
 61 	this._selectedClassName = [this._origClassName, DwtCssStyle.SELECTED].join("-");
 62 	this._selectedFocusedClassName = [this._origClassName, DwtCssStyle.SELECTED, DwtCssStyle.FOCUSED].join("-");
 63 	this._actionedClassName = [this._origClassName, DwtCssStyle.ACTIONED].join("-");
 64 	this._dragOverClassName = [this._origClassName, DwtCssStyle.DRAG_OVER].join("-");
 65     this._treeItemTextClass = "DwtTreeItem-Text";
 66     this._treeItemExtraImgClass = "DwtTreeItem-ExtraImg";
 67 
 68 	params.deferred = (params.deferred !== false);
 69 	params.className = null;
 70 	DwtComposite.call(this, params);
 71 
 72 	this._imageInfoParam = params.imageInfo;
 73 	this._extraInfo = params.extraInfo;
 74 	this._textParam = params.text;
 75 	this._deferred = params.deferred;
 76 	this._expandNodeImage = params.expandNodeImage || "NodeExpanded";
 77 	this._collapseNodeImage = params.collapseNodeImage || "NodeCollapsed";
 78 	this._itemChecked = false;
 79 	this._initialized = false;
 80 	this._selectionEnabled = Boolean(params.selectable !== false);
 81 	this._forceNotifySelection = Boolean(params.forceNotifySelection);
 82 	this._actionEnabled = true;
 83 	this._forceNotifyAction = Boolean(params.forceNotifyAction);
 84 	this._dndScrollCallback = params.dndScrollCallback;
 85 	this._dndScrollId = params.dndScrollId;
 86     this._contextEnabled = (!(!(parent._optButton)) || parent._contextEnabled) && this._selectionEnabled;
 87 
 88 	if (params.singleClickAction) {
 89 		this._singleClickAction = true;
 90 		this._selectedFocusedClassName = this._selectedClassName = this._textClassName;
 91 		this._hoverClassName = [this._origClassName, DwtCssStyle.HOVER].join("-");
 92 	} else {
 93 		this._hoverClassName = this._textClassName;
 94 	}
 95 
 96 	// if our parent is DwtTree or our parent is initialized and is not deferred
 97 	// type or is expanded, then initialize ourself, else wait
 98 	if (parent instanceof DwtTree || (parent._initialized && (!parent._deferred || parent._expanded)) || !params.deferred) {
 99 		this._initialize(params.index);
100 	} else {
101 		parent._addDeferredChild(this, params.index);
102 		this._index = params.index;
103 	}
104 };

182 /**
183  * Expands or collapses this tree item.
184  *
185  * @param {boolean}	expanded		if <code>true</code>, expands this node; otherwise collapses it
186  * @param {boolean}	recurse		if <code>true</code>, expand children recursively (does not apply to collapsing)
187  * @param	{boolean}	skipNotify		if <code>true</code>, do not notify the listeners
188  */
189 DwtTreeItem.prototype.setExpanded =
190 function(expanded, recurse, skipNotify) {
191 	// Go up the chain, ensuring that parents are expanded/initialized
192 	if (expanded) {
193 		var p = this.parent;
194 		while (p instanceof DwtTreeItem && !p._expanded) {
195 			p.setExpanded(true);
196 			p = p.parent;
197 		}
198 		// Realize any deferred children
199 		this._realizeDeferredChildren();
200 	}
201 		
202 	// If we have children, then allow for expanding/collapsing
203 	if (this.getNumChildren()) {
204 		if (expanded && recurse) {
205 			if (!this._expanded) {
206 				this._expand(expanded, null, skipNotify);
207 			}
208 			var a = this.getChildren();
209 			for (var i = 0; i < a.length; i++) {
210 				if (a[i] instanceof DwtTreeItem) {
211 					a[i].setExpanded(expanded, recurse, skipNotify);
212 				}
213 			}
214 		} else if (this._expanded != expanded) {
215 			this._expand(expanded, null, skipNotify);
216 		}
217 	}
218 };

397 DwtTreeItem.prototype.removeChild =
398 function(child) {
399 	if (child._initialized) {
400 		this._tree._deselect(child);
401 		if (this._childDiv) {
402 			this._childDiv.removeChild(child.getHtmlElement());
403 		}
404 	}
405 	this._children.remove(child);
406 
407 	// if we have no children and we are expanded, then mark us a collapsed.
408 	// Also if there are no deferred children, then make sure we remove the
409 	// expand/collapse icon and replace it with a blank16Icon.
410 	if (this._children.size() == 0) {
411 		if (this._expanded)
412 			this._expanded = false;
413 		
414 		if (this._initialized && this._nodeCell) {
415 			AjxImg.setImage(this._nodeCell, "Blank_16");
416 			var imgEl = AjxImg.getImageElement(this._nodeCell);
417 			if (imgEl)
418 				Dwt.clearHandler(imgEl, DwtEvent.ONMOUSEDOWN);
419 		}
420 	}
421 };

428 DwtTreeItem.prototype.handleKeyAction =
429 function(actionCode, ev) {
430 
431 	switch (actionCode) {
432 		
433 		case DwtKeyMap.ENTER:
434 			this._tree.setEnterSelection(this, true);
435 			break;
436 
437 
438 		case DwtKeyMap.NEXT: {
439 			var ti = this._tree._getNextTreeItem(true);
440 			if (ti) {
441 				ti._tree.setSelection(ti, false, true);
442 			}
443 			break;
444 		}
445 
446 		case DwtKeyMap.PREV: {
447 			var ti = this._tree._getNextTreeItem(false);
448 			if (ti) {
449 				ti._tree.setSelection(ti, false, true);
450 			}
451 			break;
452 		}
453 
454 		case DwtKeyMap.EXPAND: {
455 			if (!this._expanded) {
456 				this.setExpanded(true, false, true);
457 			}
458 			break;
459 		}
460 
461 		case DwtKeyMap.COLLAPSE: {
462 			if (this._expanded) {
463 				this.setExpanded(false, false, true);
464 			}
465 			break;
466 		}
467 
468 		case DwtKeyMap.ACTION: {
469 			var target = this.getHtmlElement();
470 			var p = Dwt.toWindow(target, 0, 0);
471 			var s = this.getSize();
472 			var docX = p.x + s.x / 4;
473 			var docY = p.y + s.y / 2;
474 			this._gotMouseDownRight = true;
475 			this._emulateSingleClick({dwtObj:this, target:target, button:DwtMouseEvent.RIGHT,
476 									  docX:docX, docY:docY, kbNavEvent:true});
477 			break;
478 		}
479 
480 		default:
481 			return false;
482 
483 	}
484 
485 	return true;
486 };

497 DwtTreeItem.prototype._initialize =
498 function(index, realizeDeferred, forceNode) {
499 	this._checkState();
500 	if (AjxEnv.isIE) {
501 		this._setEventHdlrs([DwtEvent.ONMOUSEENTER, DwtEvent.ONMOUSELEAVE]);
502 	}
503 	if (AjxEnv.isSafari) {	// bug fix #25016
504 		this._setEventHdlrs([DwtEvent.ONCONTEXTMENU]);
505 	}
506 	var data = {
507 		id: this._htmlElId,
508 		divClassName: this._origClassName,
509 		isCheckedStyle: this._tree.isCheckedStyle,
510 		textClassName: this._textClassName
511 	};
512 
513 	this._createHtmlFromTemplate(this.TEMPLATE, data);
514 
515 	// add this object's HTML element to the DOM
516 	this.parent._addItem(this, index, realizeDeferred);
517 
518 	// cache DOM objects here
519 	this._itemDiv = document.getElementById(data.id + "_div");
520 	this._nodeCell = document.getElementById(data.id + "_nodeCell");
521 	this._checkBoxCell = document.getElementById(data.id + "_checkboxCell");
522 	this._checkBox = document.getElementById(data.id + "_checkbox");
523 	this._checkedImg = document.getElementById(data.id + "_checkboxImg");
524 	this._imageCell = document.getElementById(data.id + "_imageCell");
525 	this._textCell = document.getElementById(data.id + "_textCell");
526 	this._extraCell = document.getElementById(data.id + "_extraCell");
527 
528     if (!this._contextEnabled){
529         var tableNode = document.getElementById(data.id + "_table");
530         tableNode.style.tableLayout = "auto";
531     }
532 	// If we have deferred children, then make sure we set up accordingly
533 	if (this._nodeCell) {
534 		this._nodeCell.style.width = this._nodeCell.style.height = DwtTreeItem._NODECELL_DIM;
535 		if (this._children.size() > 0 || forceNode) {
536 			AjxImg.setImage(this._nodeCell, this._collapseNodeImage);
537 			this.addNodeIconListeners();
538 		}
539 	}
540 
541 	if (this._extraCell) {
542 		AjxImg.setImage(this._extraCell, (this._extraInfo ||  "Blank_16"));
543 		this._extraCell.className = this._treeItemExtraImgClass;
544 	}
545 
546 	// initialize checkbox
547 	if (this._tree.isCheckedStyle && this._checkBox) {
548 		this._checkBox.onclick = AjxCallback.simpleClosure(this._handleCheckboxOnclick, this);
549 		this.showCheckBox(this._checkBoxVisible);
550 		this.setChecked(this._tree.isCheckedByDefault, true);
551 	}
552 
553 	// initialize icon
554 	if (this._imageCell && this._imageInfoParam) {
555 		AjxImg.setImage(this._imageCell, this._imageInfoParam);
556 		this._imageInfo = this._imageInfoParam;
557 	}
558 
559 	// initialize text
560 	if (this._textCell && this._textParam) {
561 		this._textCell.innerHTML = this._text = this._textParam;
562 	}
563 	this._expanded = this._selected = this._actioned = false;
564 	this._gotMouseDownLeft = this._gotMouseDownRight = false;
565 	this._addMouseListeners();
566 
567 	this._initialized = true;
568 };

590 DwtTreeItem.prototype._addMouseListeners =
591 function() {
592 	var events = [DwtEvent.ONMOUSEDOWN, DwtEvent.ONMOUSEUP, DwtEvent.ONDBLCLICK];
593 	if (AjxEnv.isIE) {
594 		events.push(DwtEvent.ONMOUSEENTER, DwtEvent.ONMOUSELEAVE);
595 	} else {
596 		events.push(DwtEvent.ONMOUSEOVER, DwtEvent.ONMOUSEOUT);
597 	}
598 	if (AjxEnv.isSafari) {
599 		events.push(DwtEvent.ONCONTEXTMENU);
600 	}
601 	for (var i = 0; i < events.length; i++) {
602 		this.addListener(events[i], DwtTreeItem._listeners[events[i]]);
603 	}
604 };

606 DwtTreeItem.prototype._addDeferredChild =
607 function(child, index) {
608 	// If we are initialized, then we need to add a expansion node
609 	if (this._initialized && this._children.size() == 0) {
610 		if (this._nodeCell) {
611 			AjxImg.setImage(this._nodeCell, this._collapseNodeImage);
612 			var imgEl = AjxImg.getImageElement(this._nodeCell);
613 			if (imgEl) {
614 				Dwt.setHandler(imgEl, DwtEvent.ONMOUSEDOWN, DwtTreeItem._nodeIconMouseDownHdlr);
615 				Dwt.setHandler(imgEl, DwtEvent.ONMOUSEUP, DwtTreeItem._nodeIconMouseUpHdlr);
616 			}
617 		}
618 	}
619 	this._children.add(child, index);
620 };

625 DwtTreeItem.prototype._addItem =
626 function(item, index, realizeDeferred) {
627 	if (!this._children.contains(item)) {
628 		this._children.add(item, index);
629 	}
630 
631 	if (this._childDiv == null) {
632 		this._childDiv = document.createElement("div");
633 		this._childDiv.className = (this.parent != this._tree)
634 			? "DwtTreeItemChildDiv" : "DwtTreeItemLevel1ChildDiv";
635 		this.getHtmlElement().appendChild(this._childDiv);
636 		if (!this._expanded) {
637 			this._childDiv.style.display = "none";
638 		}
639 	}
640 
641 	if (realizeDeferred && this._nodeCell) {
642 		if (AjxImg.getImageClass(this._nodeCell) == AjxImg.getClassForImage("Blank_16")) {
643 			AjxImg.setImage(this._nodeCell, this._expanded ? this._expandNodeImage : this._collapseNodeImage);
644 			var imgEl = AjxImg.getImageElement(this._nodeCell);
645 			if (imgEl) {
646 				Dwt.setHandler(imgEl, DwtEvent.ONMOUSEDOWN, DwtTreeItem._nodeIconMouseDownHdlr);
647 			}

758 function(ev) {
759 	var obj = DwtControl.getTargetControl(ev);
760 	var mouseEv = DwtShell.mouseEvent;
761 	mouseEv.setFromDhtmlEvent(ev, obj);

784 DwtTreeItem.prototype._expand =
785 function(expand, ev, skipNotify) {
786 	if (expand !== this._expanded) {
787 		if (!expand) {
788 			this._expanded = false;
789 			this._childDiv.style.display = "none";
790 			if (this._nodeCell) {
791 				AjxImg.setImage(this._nodeCell, this._collapseNodeImage);
792 			}
793 			this._tree._itemCollapsed(this, ev, skipNotify);
794 		} else {
795 			// The first thing we need to do is initialize any deferred children so that they
796 			// actually have content
797 			this._realizeDeferredChildren();
798 			this._expanded = true;
799 			if(this._childDiv && this._childDiv.style)
800 				this._childDiv.style.display = "block";
801 			if (this._nodeCell) {
802 				AjxImg.setImage(this._nodeCell, this._expandNodeImage);
803 			}
804 			this._tree._itemExpanded(this, ev, skipNotify);
805 		}	
806 	}
807 };

809 DwtTreeItem.prototype._realizeDeferredChildren =
810 function() {
811 	var a = this._children.getArray();
812 	for (var i = 0; i < a.length; i++) {
813 		var treeItem = a[i];
814 		if (!treeItem._initialized) {
815 			treeItem._initialize(treeItem._index, true);
816 		} else if (treeItem._isSeparator && !treeItem.div && this._childDiv) {
817 			// Note: separators marked as initialized on construction
818 			var div = treeItem.div = document.createElement("div");
819 			div.className = "vSpace";
820 			this._childDiv.appendChild(div);
821 			treeItem._initialized = true;
822 		}
823 	}
824 	if (this._needsSort) {
825 		if (a.length) {
826 			this._setChildElOrder();
827 		}
828 		delete this.__needsSort;
829 	}
830 };

925

927 		treeItem._gotMouseDownLeft = true;
928 	} else if (ev.button == DwtMouseEvent.RIGHT && (treeItem._actionEnabled || treeItem._forceNotifyAction)) {
929 		treeItem._gotMouseDownRight = true;
930 	}

965 DwtTreeItem._mouseUpListener =
966 function(ev) {
967 	var treeItem = ev.dwtObj;
968 	if (!treeItem) { return false; }
969 	// Ignore any mouse events in the child div i.e. the div which 
970 	// holds all the items children. In the case of IE, no clicks are
971 	// reported when clicking in the padding area (note all children
972 	// are indented using padding-left style); however, mozilla
973 	// reports mouse events that happen in the padding area
974 	if (ev.target == treeItem._childDiv) { return; }
975 
976 	if (ev.button == DwtMouseEvent.LEFT && treeItem._gotMouseDownLeft) {
977 		treeItem._tree._itemClicked(treeItem, ev);
978 	} else if (ev.button == DwtMouseEvent.RIGHT && treeItem._gotMouseDownRight) {
979 		treeItem._tree._itemActioned(treeItem, ev);
980 	}
981 };

983 DwtTreeItem._doubleClickListener =
984 function(ev) {
985 	var treeItem = ev.dwtObj;
986 	if (!treeItem) { return false; }
987 	// See comment in DwtTreeItem.prototype._mouseDownListener
988 	if (ev.target == treeItem._childDiv) { return; }
989 
990 	var obj = DwtControl.getTargetControl(ev);
991 	var mouseEv = DwtShell.mouseEvent;
992 	mouseEv.setFromDhtmlEvent(ev, obj);
993 	if (mouseEv.button == DwtMouseEvent.LEFT || mouseEv.button == DwtMouseEvent.NONE) {	// NONE for IE bug
994 		mouseEv.dwtObj._tree._itemDblClicked(mouseEv.dwtObj, mouseEv);
995 	}
996 };
