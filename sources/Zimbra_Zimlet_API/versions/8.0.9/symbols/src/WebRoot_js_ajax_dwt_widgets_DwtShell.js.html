 16 /**
 17  * Creates a shell.
 18  * @constructor
 19  * @class
 20  * This class represents a shell, the first widget that must be instantiated in a Dwt based 
 21  * application. By default the shell covers the whole browser window, though it may also be 
 22  * instantiated within an HTML element.
 23  * <p>
 24  * {@link DwtShell} should <b>NOT</b> be subclassed.
 25  * </p>
 26  *
 27  * @author Ross Dargahi
 28  * 
 29  * @param	{hash}	params		a hash of parameters
 30  * @param {string}	params.className			the CSS class name
 31  * @param {boolean}	params.docBodyScrollable	if <code>true</code>, then the document body is set to be scrollable
 32  * @param {Element}	params.userShell			an HTML element that will be reparented into an absolutely
 33  *											postioned container in this shell. This is useful in the situation where you have an HTML 
 34  *											template and want to use this in context of Dwt.
 35  * @param {Boolean}	params.useCurtain			if <code>true</code>, a curtain overlay is created to be used between hidden and viewable elements 
 36  *											using z-index (see {@link Dwt}) for various layering constants)
 37  *
 38  * @extends		DwtComposite
 39  */
 40 DwtShell = function(params) {
 41 	if (window._dwtShellId) {
 42 		throw new DwtException("DwtShell already exists for window", DwtException.INVALID_OP, "DwtShell");
 43 	}
 44 
 45 	var className = params.className || "DwtShell";
 46 	DwtComposite.call(this, {className:className});
 47 
 48 	// HACK! This is a hack to make sure that the control methods work 
 49 	// with DwtShell since the parent of DwtShell is null. 
 50 	this.__ctrlInited = true;
 51 
 52 	document.body.style.margin = 0;
 53 	if (!params.docBodyScrollable) {
 54 		if (AjxEnv.isIE) {
 55 			document.body.onscroll = DwtShell.__onBodyScroll;
 56 		}
 57 		document.body.style.overflow = "hidden";
 58 	}
 59 
 60 	document.body.onselect = DwtShell._preventDefaultSelectPrt;
 61 	document.body.onselectstart = DwtShell._preventDefaultSelectPrt;
 62 	document.body.oncontextmenu = DwtShell._preventDefaultPrt;
 63 	window.onresize = DwtShell._resizeHdlr;
 64 
 65 	var htmlElement = document.createElement("div");
 66 	this._htmlElId = window._dwtShellId = htmlElement.id = params.id || Dwt.getNextId();
 67 	DwtControl.ALL_BY_ID[this._htmlElId] = this;
 68 
 69 	htmlElement.className = className;
 70 	htmlElement.style.width = htmlElement.style.height = "100%";
 71 	if (htmlElement.style.overflow) {
 72 		htmlElement.style.overflow = null;
 73 	}
 74 
 75 	// if there is a user shell (body content), move it below this shell
 76 	// into a container that's absolutely positioned
 77 	try {
 78 		if (params.userShell) {
 79 			document.body.removeChild(params.userShell);
 80 		}
 81 	} catch (ex) {}
 82 	document.body.appendChild(htmlElement);
 83 	if (params.userShell) {
 84 		var userShellContainer = new DwtControl({parent:this, posStyle:Dwt.ABSOLUTE_STYLE});
 85 		userShellContainer.getHtmlElement().appendChild(params.userShell);
 86 		userShellContainer.setSize("100%", "100%");
 87 		userShellContainer.zShow(true);
 88 	}
 89 	this.shell = this;
 90 
 91 	// Busy overlay - used when we want to enforce a modal busy state
 92 	this._createBusyOverlay(htmlElement);
 93 
 94 	// Veil overlay - used by DwtDialog to disable underlying app
 95 	this._veilOverlay = document.createElement("div");
 96 	this._veilOverlay.className = (!AjxEnv.isLinux) ? "VeilOverlay" : "VeilOverlay-linux";
 97 	this._veilOverlay.style.position = "absolute";
 98 	this._veilOverlay.style.cursor = AjxEnv.isIE6up ? "not-allowed" : "wait";
 99 	Dwt.setBounds(this._veilOverlay, 0, 0, "100%", "100%");
100 	Dwt.setZIndex(this._veilOverlay, Dwt.Z_HIDDEN);
101 	this._veilOverlay.veilZ = new Array();
102 	this._veilOverlay.veilZ.push(Dwt.Z_HIDDEN);
103 	this._veilOverlay.dialogZ = new Array();
104 	this._veilOverlay.activeDialogs = new Array();
105 	this._veilOverlay.innerHTML = "<table cellspacing=0 cellpadding=0 style='width:100%; height:100%'><tr><td> </td></tr></table>";
106 	htmlElement.appendChild(this._veilOverlay);
107 
108 	// Curtain overlay - used between hidden and viewable elements using z-index
109 	if (params.useCurtain) {
110 		this._curtainOverlay = document.createElement("div");
111 		this._curtainOverlay.className = "CurtainOverlay";
112 		this._curtainOverlay.style.position = "absolute";
113 		Dwt.setBounds(this._curtainOverlay, 0, 0, "100%", "100%")
114 		Dwt.setZIndex(this._curtainOverlay, Dwt.Z_CURTAIN);
115 		this._curtainOverlay.innerHTML = "<table cellspacing=0 cellpadding=0 style='width:100%; height:100%'><tr><td> </td></tr></table>";
116 		htmlElement.appendChild(this._curtainOverlay);
117 	}
118 
119 	this._uiEvent = new DwtUiEvent(true);
120 	this._currWinSize = Dwt.getWindowSize();
121 
122 	// tooltip singleton used by all controls in shell
123 	this._toolTip = new DwtToolTip(this);
124 	this._hoverMgr = new DwtHoverMgr();
125 	
126 	this._keyboardMgr = new DwtKeyboardMgr(this);
127 }

177 /**
178  * Sets the busy overlay. The busy overlay disables input to the application and makes the 
179  * cursor a wait cursor. Optionally a work in progress (WIP) dialog may be requested. Since
180  * multiple calls to this method may be interleaved, it accepts a unique ID to keep them
181  * separate. We also maintain a count of outstanding calls to <code>setBusy(true)</code>. When that count
182  * changes between 0 and 1, the busy overlay is applied or removed.
183  * 
184  * @param {boolean}	busy			if <code>true</code>, set the busy overlay, otherwise hide the busy overlay
185  * @param {number}	id					a unique ID for this instance
186  * @param {boolean}	showBusyDialog 		if <code>true</code>, show the WIP dialog
187  * @param {number}	busyDialogDelay 		the number of ms to delay before popping up the WIP dialog
188  * @param {AjxCallback}	cancelBusyCallback	the callback to run when OK button is pressed in WIP dialog
189  */ 
190 DwtShell.prototype.setBusy =
191 function(busy, id, showBusyDialog, busyDialogDelay, cancelBusyCallback) {
192 	if (busy) {
193 		this._setBusyCount++;
194 	} else if (this._setBusyCount > 0) {
195 		this._setBusyCount--;
196 	}
197 
198     if (!this._setBusy && (this._setBusyCount > 0)) {
199 		// transition from non-busy to busy state
200 		Dwt.setCursor(this._busyOverlay, "wait");
201     	Dwt.setVisible(this._busyOverlay, true);
202     	this._setBusy = this._blockInput = true;
203     	DBG.println(AjxDebug.DBG2, "set busy overlay, id = " + id);
204     } else if (this._setBusy && (this._setBusyCount <= 0)) {
205 		// transition from busy to non-busy state
206 	    Dwt.setCursor(this._busyOverlay, "default");
207 	    Dwt.setVisible(this._busyOverlay, false);
208 	    this._setBusy = this._blockInput = false;
209     	DBG.println(AjxDebug.DBG2, "remove busy overlay, id = " + id);
210 	}
211 	
212 	// handle busy dialog whether we've changed state or not
213 	if (busy && showBusyDialog) {
214 		if (busyDialogDelay && busyDialogDelay > 0) {
215 			this._busyActionId[id] = AjxTimedAction.scheduleAction(this._busyTimedAction, busyDialogDelay);
216 		} else {
217 			this._showBusyDialogAction(id);
218 		}
219 
220 		this._cancelBusyCallback = cancelBusyCallback;
221 		if (this._busyDialog) {
222 			this._busyDialog.setButtonEnabled(DwtShell.CANCEL_BUTTON, (cancelBusyCallback != null));
223 		}
224 	} else {
225     	if (this._busyActionId[id] && (this._busyActionId[id] != -1)) {
226     		AjxTimedAction.cancelAction(this._busyActionId[id]);
227     		this._busyActionId[id] = -1;
228     	}
229    		if (this._busyDialog && this._busyDialog.isPoppedUp) {
230     		this._busyDialog.popdown();
231    		}
232     } 
233 }

339 /**
340  * Adds a focus listener.
341  * 
342  * @param	{AjxListener}	listener		the listener
343  */
344 DwtShell.prototype.addFocusListener =
345 function(listener) {
346 	if (!this._hasFocusHandler) {
347 		var doc = document;
348 		if (typeof doc.onfocusin != "undefined" ) {  // if (IE)
349 			doc.attachEvent("onfocusin", DwtShell.__focusHdlr);
350 		} else {
351 			window.addEventListener("focus", DwtShell.__focusHdlr, false);
352 		}
353 		this._hasFocusHandler = true;
354 	}
355 	this.addListener(DwtEvent.ONFOCUS, listener);
356 };

370 			window.addEventListener("blur", DwtShell.__blurHdlr, false);
371 		}
372 		this._hasBlurHandler = true;
373 	}
374 	this.addListener(DwtEvent.ONBLUR, listener);

440 DwtShell.prototype._getBusyDialog =
441 function(htmlElement) {
442 	if (!this._busyDialog) {
443 		var cancelButton = new DwtDialog_ButtonDescriptor(DwtShell.CANCEL_BUTTON, AjxMsg.cancelRequest, DwtDialog.ALIGN_CENTER);
444 	    this._busyDialog = new DwtDialog({parent:this, className:"DwtShellBusyDialog", title:AjxMsg.workInProgress,
445 	    								  standardButtons:DwtDialog.NO_BUTTONS, extraButtons:[cancelButton], zIndex:Dwt.BUSY + 10});
446 	    this._busyDialog.registerCallback(DwtShell.CANCEL_BUTTON, this._busyCancelButtonListener, this);
447 	    var txtId = Dwt.getNextId();
448 	    var html = [
449 	        "<table class='DialogContent'><tr>",
450 	            "<td><div class='WaitIcon'></div></td><td class='MsgText' id='", txtId, "'> </td>",
451 	        "</tr></table>"].join("");
452 	    
453 	    this._busyDialog.setContent(html);
454 	    this._busyDialogTxt = document.getElementById(txtId);
455 		if (this._busyDialogText) {
456 			this._busyDialogTxt.innerHTML = this._busyDialogText;
457 		}
458 		if (this._busyDialogTitle) {
459 			this._busyDialog.setTitle(this._busyDialogTitle);
460 		}
461 		this._busyDialog.setButtonEnabled(DwtShell.CANCEL_BUTTON, (this._cancelBusyCallback != null));
462 	}
463 	return this._busyDialog;
464 };

480 DwtShell._preventDefaultSelectPrt =
481 function(ev) {
482     var evt = DwtControl.fromElementId(window._dwtShellId)._uiEvent;
483     evt.setFromDhtmlEvent(ev, true);
484 
485 	if (evt.dwtObj && evt.dwtObj instanceof DwtControl && !evt.dwtObj.preventSelection(evt.target)) {
486         evt._stopPropagation = false;
487         evt._returnValue = true;
488     } else {
489         evt._stopPropagation = true;
490         evt._returnValue = false;
491     }
492     evt.setToDhtmlEvent(ev);
493     return !evt._stopPropagation;
494 };
495 
496 DwtShell._preventDefaultPrt =
497 function(ev) {
498 	ev = DwtUiEvent.getEvent(ev);
499 	var target = ev.target ? ev.target : ev.srcElement;
500 	
501     var evt = DwtControl.fromElementId(window._dwtShellId)._uiEvent;
502     evt.setFromDhtmlEvent(ev, true);
503 	//default behavior
504     evt._stopPropagation = true;
505     evt._returnValue = false;
506 	if (evt.dwtObj && evt.dwtObj instanceof DwtControl && !evt.dwtObj.preventContextMenu(evt.target)) {
507         evt._stopPropagation = false;
508         evt._returnValue = true;
509     } else if (target != null && typeof(target) == 'object') {
510      	if ((target.tagName == "A" ||  target.tagName == "a") && target.href) {
511 	        evt._stopPropagation = false;
512     	    evt._returnValue = true;
513     	}
514     } 
515     
516     evt.setToDhtmlEvent(ev);
517     return evt._returnValue;
518 };

521 /* This the resize handler to track when the browser window size changes */
522 DwtShell._resizeHdlr =
523 function(ev) {
524 	var shell = DwtControl.fromElementId(window._dwtShellId);
525 	if (shell.isListenerRegistered(DwtEvent.CONTROL)) {
526 	 	var evt = DwtShell.controlEvent;
527 	 	evt.reset();
528 	 	evt.oldWidth = shell._currWinSize.x;
529 	 	evt.oldHeight = shell._currWinSize.y;
530 	 	shell._currWinSize = Dwt.getWindowSize();
531 	 	evt.newWidth = shell._currWinSize.x;
532 	 	evt.newHeight = shell._currWinSize.y;
533 	 	shell.notifyListeners(DwtEvent.CONTROL, evt);
534 	} else {
535 		shell._currWinSize = Dwt.getWindowSize();
536 	}
537 };

546 DwtShell.__focusHdlr =
547 function() {
548 	var focusEvent = DwtShell.focusEvent;
549 	var self = DwtShell.getShell(window);
550 	focusEvent.dwtObj = self;
551 	focusEvent.state = DwtFocusEvent.FOCUS;
552 	self.notifyListeners(DwtEvent.ONFOCUS, focusEvent);
553 };

555 DwtShell.__blurHdlr =
556 function() {
557 	var focusEvent = DwtShell.focusEvent;
558 	var self = DwtShell.getShell(window);
559 	focusEvent.dwtObj = self;
560 	focusEvent.state = DwtFocusEvent.BLUR;
561 	self.notifyListeners(DwtEvent.ONBLUR, focusEvent);
562 };
