 16 /**
 17  * @overview
 18  * This file contains the result class.
 19  */
 20 
 21 /**
 22  * Creates a CSFE result object.
 23  * @class
 24  * This class represents the result of a CSFE request. The data is either the 
 25  * response that was received, or an exception. If the request resulted in a 
 26  * SOAP fault from the server, there will also be a SOAP header present.
 27  *
 28  * @author Conrad Damon
 29  * 
 30  * @param {Object}	data			the response data
 31  * @param {Boolean}	isException	if <code>true</code>, the data is an exception object
 32  * @param {Object}	header			the SOAP header
 33  * 
 34  */
 35 ZmCsfeResult = function(data, isException, header) {
 36 	this.set(data, isException, header);
 37 };
 38 
 39 ZmCsfeResult.prototype.isZmCsfeResult = true;
 40 ZmCsfeResult.prototype.toString = function() { return "ZmCsfeResult"; };
 41 
 42 /**
 43  * Sets the content of the result.
 44  *
 45  * @param {Object}	data			the response data
 46  * @param {Boolean}	isException	if <code>true</code>, the data is an exception object
 47  * @param {Object}	header			the SOAP header
 48  */
 49 ZmCsfeResult.prototype.set =
 50 function(data, isException, header) {
 51 	this._data = data;
 52 	this._isException = (isException === true);
 53 	this._header = header;
 54 };
 55 
 56 /**
 57  * Gets the response data. If there was an exception, throws the exception.
 58  * 
 59  * @return	{Object}	the data
 60  */
 61 ZmCsfeResult.prototype.getResponse =
 62 function() {
 63 	if (this._isException) {
 64 		throw this._data;
 65 	} else {
 66 		return this._data;
 67 	}
 68 };
 69 
 70 /**
 71  * Gets the exception object, if any.
 72  * 
 73  * @return	{ZmCsfeException}	the exception or <code>null</code> for none
 74  */
 75 ZmCsfeResult.prototype.getException =
 76 function() {
 77 	return this._isException ? this._data : null;
 78 };
 79 
 80 /**
 81  * Checks if this result is an exception.
 82  * 
 83  * @return	{Boolean}	<code>true</code> if an exception
 84  */
 85 ZmCsfeResult.prototype.isException = 
 86 function() {
 87 	return this._isException;
 88 };
 89 
 90 /**
 91  * Gets the SOAP header that came with a SOAP fault.
 92  * 
 93  * @return	{String}	the header
 94  */
 95 ZmCsfeResult.prototype.getHeader =
 96 function() {
 97 	return this._header;
 98 };
