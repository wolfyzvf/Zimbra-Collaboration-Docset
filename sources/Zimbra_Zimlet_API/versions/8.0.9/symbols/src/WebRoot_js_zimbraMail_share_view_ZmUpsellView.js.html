 16 /**
 17  * 
 18  * @extends		DwtControl
 19  * @private
 20  */
 21 ZmUpsellView = function(params) {
 22 	DwtControl.call(this, params);
 23 }
 24 ZmUpsellView.prototype = new DwtControl;
 25 ZmUpsellView.prototype.constructor = ZmUpsellView;
 26 
 27 ZmUpsellView.prototype.toString = function() {
 28 	return "ZmUpsellView";
 29 };
 30 
 31 ZmUpsellView.prototype.setBounds =
 32 function(x, y, width, height, showToolbar) {
 33     var deltaHeight = 0;
 34     if(!showToolbar) {
 35         deltaHeight = this._getToolbarHeight();
 36     }
 37 	DwtControl.prototype.setBounds.call(this, x, y - deltaHeight, width, height + deltaHeight);
 38 	var id = "iframe_" + this.getHTMLElId();
 39 	var iframe = document.getElementById(id);
 40 	if(iframe) {
 41     	iframe.width = width;
 42     	iframe.height = height + deltaHeight;
 43 	}
 44 };
 45 
 46 ZmUpsellView.prototype._getToolbarHeight =
 47 function() {
 48     var topToolbar = appCtxt.getAppViewMgr().getViewComponent(ZmAppViewMgr.C_TOOLBAR_TOP);
 49 	if (topToolbar) {
 50 		var sz = topToolbar.getSize();
 51 		var height = sz.y ? sz.y : topToolbar.getHtmlElement().clientHeight;
 52 		return height;
 53 	}
 54 	return 0;
 55 };
