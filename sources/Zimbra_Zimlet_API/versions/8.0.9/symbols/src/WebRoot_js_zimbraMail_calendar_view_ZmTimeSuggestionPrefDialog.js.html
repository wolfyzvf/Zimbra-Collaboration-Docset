 20 /**
 21  * Creates a time/location suggestion preference  dialog.
 22  * @class
 23  * This class represents a time/location preference dialog.
 24  *
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  *
 28  * @extends		DwtDialog
 29  */
 30 ZmTimeSuggestionPrefDialog = function(parent, className) {
 31 
 32     this._prefs = {};
 33     this._prefFields = {};
 34     this._prefLoaded = false;
 35 
 36 	className = className || "ZmTimeSuggestionPrefDialog";
 37 	DwtDialog.call(this, {parent:parent, className:className, title:ZmMsg.suggestionPreferences});
 38 
 39 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 40 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._handleCancelButton));
 41 
 42 };
 43 
 44 ZmTimeSuggestionPrefDialog.prototype = new DwtDialog;
 45 ZmTimeSuggestionPrefDialog.prototype.constructor = ZmTimeSuggestionPrefDialog;
 46 
 47 // Constants
 48 
 49 ZmTimeSuggestionPrefDialog.META_DATA_KEY = "MD_LOCATION_SEARCH_PREF";
 50 ZmTimeSuggestionPrefDialog.PREF_FIELDS = ["name", "site", "capacity", "building", "desc", "floor",
 51                                           "my_working_hrs_pref", "others_working_hrs_pref",
 52                                           "recurrenceSelect"];
 53 
 54 // corresponding attributes for search command
 55 ZmTimeSuggestionPrefDialog.SF_ATTR = {};
 56 ZmTimeSuggestionPrefDialog.SF_ATTR["name"]		  = "fullName";
 57 ZmTimeSuggestionPrefDialog.SF_ATTR["capacity"]	  = "zimbraCalResCapacity";
 58 ZmTimeSuggestionPrefDialog.SF_ATTR["desc"]        = "description";
 59 ZmTimeSuggestionPrefDialog.SF_ATTR["site"]		  = "zimbraCalResSite";
 60 ZmTimeSuggestionPrefDialog.SF_ATTR["building"]	  = "zimbraCalResBuilding";
 61 ZmTimeSuggestionPrefDialog.SF_ATTR["floor"]		  = "zimbraCalResFloor";
 62 
 63 // search field compares ops - listed here if not substring ("has")
 64 ZmTimeSuggestionPrefDialog.SF_OP = {};
 65 ZmTimeSuggestionPrefDialog.SF_OP["capacity"]	= "ge";
 66 ZmTimeSuggestionPrefDialog.SF_OP["floor"]		= "eq";
 67 
 68 ZmTimeSuggestionPrefDialog.MY_WORKING_HOURS_FIELD = 'my_working_hrs_pref';
 69 ZmTimeSuggestionPrefDialog.OTHERS_WORKING_HOURS_FIELD = 'others_working_hrs_pref';
 70 ZmTimeSuggestionPrefDialog.RECURRENCE = 'recurrenceSelect';
 71 
 72 ZmTimeSuggestionPrefDialog.CHECKBOX_FIELDS = {};
 73 ZmTimeSuggestionPrefDialog.CHECKBOX_FIELDS[ZmTimeSuggestionPrefDialog.MY_WORKING_HOURS_FIELD]      = true;
 74 ZmTimeSuggestionPrefDialog.CHECKBOX_FIELDS[ZmTimeSuggestionPrefDialog.OTHERS_WORKING_HOURS_FIELD]   = true;
 75 
 76 ZmTimeSuggestionPrefDialog.DEFAULT_VAL = {};
 77 ZmTimeSuggestionPrefDialog.DEFAULT_VAL[ZmTimeSuggestionPrefDialog.MY_WORKING_HOURS_FIELD]    = 'true';
 78 ZmTimeSuggestionPrefDialog.DEFAULT_VAL[ZmTimeSuggestionPrefDialog.OTHERS_WORKING_HOURS_FIELD] = 'true';
 79 ZmTimeSuggestionPrefDialog.DEFAULT_NUM_RECURRENCE = 4;
 80 ZmTimeSuggestionPrefDialog.MAX_NUM_RECURRENCE = 10;
 81 ZmTimeSuggestionPrefDialog.DEFAULT_VAL[ZmTimeSuggestionPrefDialog.RECURRENCE] =
 82     ZmTimeSuggestionPrefDialog.DEFAULT_NUM_RECURRENCE.toString();

133 ZmTimeSuggestionPrefDialog.prototype.setContent =
134 function(text) {
135 	var d = this._getContentDiv();
136 	if (d) {
137 		d.innerHTML = text || "";
138 	}
139 
140     this._recurrenceSelect = new DwtSelect({id:this._htmlElId + "_recurrenceSelect",
141                                 parent:this, parentElement:(this._htmlElId + "_recurrence")});
142     for (var i = 1; i <= ZmTimeSuggestionPrefDialog.MAX_NUM_RECURRENCE; i++) {
143         this._recurrenceSelect.addOption(i.toString(), (i == 1), i);
144     }
145 
146 
147     this._dlgId = AjxCore.assignId(this);
148 
149     var element, id;
150     for(var i=0; i<ZmTimeSuggestionPrefDialog.PREF_FIELDS.length; i++) {
151         id = ZmTimeSuggestionPrefDialog.PREF_FIELDS[i];
152         element = document.getElementById(this.getHTMLElId() + "_" + id);
153         if (element) {
154             this._prefFields[id] = element;
155             this._prefs[id] = this.getPreferenceFieldValue(id);
156         }
157     }
158 };

170 ZmTimeSuggestionPrefDialog.prototype.readPrefs =
171 function(text) {
172     var field;
173     for(var id in this._prefFields) {
174         this._prefs[id] = this.getPreferenceFieldValue(id)
175     }
176 };
177 
178 ZmTimeSuggestionPrefDialog.prototype.getPreferenceFieldValue =
179 function(id) {
180     if (id == "recurrenceSelect") {
181         return this._recurrenceSelect.getValue();
182     } else {
183         var field = this._prefFields[id];
184         if(!field) return;
185 
186         if(ZmTimeSuggestionPrefDialog.CHECKBOX_FIELDS[id]){
187             return field.checked ? 'true' : 'false';
188         }else {
189             return field.value;
190         }
191     }
192 };
193 
194 ZmTimeSuggestionPrefDialog.prototype.setPreferenceFieldValue =
195 function(id, value) {
196     if (id == "recurrenceSelect") {
197        this._recurrenceSelect.setSelectedValue(value);
198     } else {
199         var field = this._prefFields[id];
200         if(!field) return;
201 
202         if(ZmTimeSuggestionPrefDialog.CHECKBOX_FIELDS[id]){
203             field.checked = (value == 'true');
204         }else {
205             field.value = value || "";
206         }
207     }
208 };
209 
210 ZmTimeSuggestionPrefDialog.prototype.getSearchPreference =
211 function(account, prefSearchCallback) {
212     var md = new ZmMetaData(account || this._account);
213     var callback = new AjxCallback(this, this.processSearchPreference, [prefSearchCallback]);
214     md.get(ZmTimeSuggestionPrefDialog.META_DATA_KEY, null, callback);
215 };
216 
217 ZmTimeSuggestionPrefDialog.prototype.processSearchPreference =
218 function(prefSearchCallback, metadataResponse) {
219     this._prefs = {};
220 
221     var objPrefs = metadataResponse.getResponse().BatchResponse.GetMailboxMetadataResponse[0].meta[0]._attrs;
222     for (name in objPrefs) {
223         if(name && objPrefs[name]) {
224             this._prefs[name] = objPrefs[name];
225             this.setPreferenceFieldValue(name, this._prefs[name]);
226         }
227     }
228 
229     //set default value for the preferences
230     for(var id in ZmTimeSuggestionPrefDialog.DEFAULT_VAL) {
231         if(!this._prefs[id]) {
232             this.setPreferenceFieldValue(id, ZmTimeSuggestionPrefDialog.DEFAULT_VAL[id]);            
233         }
234     }
235 
236     this._prefLoaded = true;
237     if(prefSearchCallback) prefSearchCallback.run();
238 };
239 
240 ZmTimeSuggestionPrefDialog.prototype.setSearchPreference =
241 function() {
242     var md = new ZmMetaData(this._account);
243     var newPrefs = {};
244     for(var id in this._prefs) {
245         if(this._prefs[id] != "") newPrefs[id] = this._prefs[id];
246     }
247     return md.set(ZmTimeSuggestionPrefDialog.META_DATA_KEY, newPrefs);
248 };
249 
250 ZmTimeSuggestionPrefDialog.isSearchCondition =
251 function(id) {
252     return Boolean(ZmTimeSuggestionPrefDialog.SF_ATTR[id]);
253 };
254 
255 ZmTimeSuggestionPrefDialog.prototype.handleRoomCheckbox =
256 function() {
257     this.enableLocationFields(true);
258 };
259 
260 ZmTimeSuggestionPrefDialog.prototype.enableLocationFields =
261 function(enable) {
262     for(var id in ZmTimeSuggestionPrefDialog.SF_ATTR) {
263         if(!this._prefFields[id]) continue;
264         this._prefFields[id].disabled = !enable;
265     }
266 };
267 
268 ZmTimeSuggestionPrefDialog._handleRoomCheckbox =
269 function(ev) {
270 	var el = DwtUiEvent.getTarget(ev);
271 	var dlg = AjxCore.objectWithId(el._dlgId);
272 	if (!dlg) { return; }
273     dlg.handleRoomCheckbox();
274 };
