 24 	if (arguments.length == 0) return;

 26 
 27 	// ACE?

 29 	this.ACE_DEBUG = false;

 32 		this._ace_componentsLoading = 0;
 33     }
 34 
 35 	DwtHtmlEditor.call(this, {parent:parent, className:"ZmHtmlEditor", posStyle:posStyle,
 36 							  content:content, mode:mode});

 38 	this.addStateChangeListener(new AjxListener(this, this._rteStateChangeListener));

 45 
 46 	this._ignoreWords = {};

160 ZmHtmlEditor.prototype.getContent =
161 function(insertFontStyle, onlyInnerContent ) {
162 	this.discardMisspelledWords();
163 
164 	// NOTE: this code is same as base class except we use insertFontStyle
165 	// (which shouldnt be in base).
166 	var content;
167 	if (this._mode == DwtHtmlEditor.HTML) {
168 		AjxDebug.println(AjxDebug.REPLY, "ZmHtmlEditor::getContent - in HTML mode");
169 		var iframeDoc = this._getIframeDoc();
170 
171 		var html = (iframeDoc && iframeDoc.body && iframeDoc.body.innerHTML) || "";
172 		AjxDebug.println(AjxDebug.REPLY, "content from iframe body: " + AjxStringUtil.htmlEncode(html.substr(0,200)));
173 		this._blankDiv = this._blankDiv || this._getFontStyle("");
174 		this._blankHtml = this._blankHtml || this._embedHtmlContent("", true);
175 		if (this._pendingContent && (!html || html == this._blankDiv || html == this._blankHtml)) {
176 			html = this._pendingContent;
177 			AjxDebug.println(AjxDebug.REPLY, "using pending content: " + AjxStringUtil.htmlEncode(html.substr(0,200)));
178 		}
179 		content = this._embedHtmlContent(html, insertFontStyle, onlyInnerContent);
180 		if (this.ACE_ENABLED) {
181 			content = this._serializeAceObjects(content);
182 		}
183 	} else {
184 		content = document.getElementById(this._textAreaId).value;
185 	}
186 
187 	return content;
188 };

190 ZmHtmlEditor.prototype.checkMisspelledWords =
191 function(callback, onExitCallback, errCallback){
192 	var text = this.getTextVersion();
193 	if (/\S/.test(text)) {
194 		AjxDispatcher.require("Extras");
195 		this._spellChecker = new ZmSpellChecker(this);
196 		this._spellCheck = null;
197 		this._spellCheckSuggestionListenerObj = new AjxListener(this, this._spellCheckSuggestionListener);
198 		if (!this.onExitSpellChecker) {
199 			this.onExitSpellChecker = onExitCallback;
200 		}
201 		var params = {
202 			text: text,
203 			ignore: AjxUtil.keys(this._ignoreWords).join()
204 		};
205 		this._spellChecker.check(params, callback, errCallback);
206 		return true;
207 	}
208 
209 	return false;
210 };

242 ZmHtmlEditor.prototype.discardMisspelledWords =
243 function(keepModeDiv) {
244 	if (!this._spellCheck) { return; }
245 
246 	if (this._mode == DwtHtmlEditor.HTML) {
247 		var doc = this._getIframeDoc();
248 		doc.body.style.display = "none";
249 
250 		var p = null;
251 		var spanIds = this._spellCheck.spanIds;
252 		for (var i in spanIds) {
253 			var span = doc.getElementById(i);
254 			if (!span) continue;
255 
256 			p = span.parentNode;
257 			while (span.firstChild) {
258 				p.insertBefore(span.firstChild, span);
259 			}
260 			p.removeChild(span);
261 		}
262 
263 		if (!AjxEnv.isIE) {
264 			doc.body.normalize(); // IE crashes here.
265 		} else {
266 			doc.body.innerHTML = doc.body.innerHTML; // WTF.
267 		}
268 
269 		// remove the spell check styles
270 		p = doc.getElementById("ZM-SPELLCHECK-STYLE");
271 		if (p) {
272 			p.parentNode.removeChild(p);
273 		}
274 
275 		doc.body.style.display = "";
276 		this._unregisterEditorEventHandler(doc, "contextmenu");
277 
278 	} else if (this._spellCheckDivId != null) {
279 		var div = document.getElementById(this._spellCheckDivId);
280 		var scrollTop = div.scrollTop;
281 		var textArea = document.getElementById(this._textAreaId);
282 		// bug: 41760 - HACK. Convert the nbsps back to spaces since Gecko seems
283 		// to return control characters for HTML entities.
284 		if (AjxEnv.isGeckoBased) {
285 			div.innerHTML = AjxStringUtil.htmlDecode(div.innerHTML, true);
286 		}
287 		textArea.value = AjxUtil.getInnerText(div);
288 
289 		// avoid mem. leaks, hopefully
290 		div.onclick = null;
291 		div.oncontextmenu = null;
292 		div.onmousedown = null;
293 		div.parentNode.removeChild(div);
294 		textArea.style.display = "";
295 		textArea.scrollTop = scrollTop;
296 	}
297 
298 	this._spellCheckDivId = this._spellCheck = null;
299 	window.status = "";
300 
301 	if (!keepModeDiv) {
302 		this._spellCheckHideModeDiv();
303 	}
304 
305 	if (this.onExitSpellChecker) {
306 		this.onExitSpellChecker.run();
307 	}
308 };

328 ZmHtmlEditor.prototype._loadExternalStyle =
329 function(path) {
330 	var doc = this._getIframeDoc();
331 	// check if already loaded
332 	var style = doc.getElementById(path);
333 	if (!style) {
334 		style = doc.createElement("link");
335 		style.id = path;
336 		style.rel = "stylesheet";
337 		style.type = "text/css";
338 		var style_url = appContextPath + path + "?v=" + cacheKillerVersion;
339 		if (AjxEnv.isGeckoBased || AjxEnv.isSafari) {
340 			style_url = document.baseURI.replace(
341 					/^(https?:\x2f\x2f[^\x2f]+).*$/, "$1") + style_url;
342 		}
343 		// servlet caches CSS unless there's a debug param
344 		var debugLevel = DBG && DBG.getDebugLevel();
345 		if (debugLevel) {
346 			style_url = style_url + "&debug=" + debugLevel;
347 		}
348 		style.href = style_url;
349 		var head = doc.getElementsByTagName("head")[0];
350 		if (!head) {
351 			head = doc.createElement("head");
352 			var docEl = doc.documentElement;
353 			if (docEl) {
354 				docEl.insertBefore(head, docEl.firstChild);
355 			}
356 		}
357 		head.appendChild(style);
358 	}

373 ZmHtmlEditor.prototype.highlightMisspelledWords =
374 function(words, keepModeDiv) {
375 	this.discardMisspelledWords(keepModeDiv);
376 
377 	var word, style, doc, body, self = this,
378 		spanIds     = {},
379 		wordIds     = {},
380 		regexp      = [ "([^A-Za-z0-9']|^)(" ],
381 		suggestions = {};
382 
383 	// preparations: initialize some variables that we then save in
384 	// this._spellCheck (the current spell checker context).
385 	for (var i = 0; i < words.length; ++i) {
386 		word = words[i].word;
387 		if (!suggestions[word]) {
388 			i && regexp.push("|");
389 			regexp.push(word);
390 			var a = words[i].suggestions.split(/\s*,\s*/);
391 			if (!a[a.length-1])
392 				a.pop();
393 			suggestions[word] = a;
394 			if (suggestions[word].length > 5)
395 				suggestions[word].length = 5;
396 		}
397 	}
398 	regexp.push(")([^A-Za-z0-9']|$)");
399 	regexp = new RegExp(regexp.join(""), "gm");
400 
401 	function hiliteWords(text, textWhiteSpace) {
402 		text = textWhiteSpace
403 			? AjxStringUtil.convertToHtml(text)
404 			: AjxStringUtil.htmlEncode(text);
405 
406 		var m;
407 
408 		regexp.lastIndex = 0;
409 		while (m = regexp.exec(text)) {
410 			var str = m[0];
411 			var prefix = m[1];
412 			var word = m[2];
413 			var suffix = m[3];
414 
415 			var id = Dwt.getNextId();
416 			spanIds[id] = word;
417 			if (!wordIds[word])
418 				wordIds[word] = [];
419 			wordIds[word].push(id);
420 
421 			var repl = [
422 				prefix,
423 				'<span word="',
424 				word, '" id="', id, '" class="ZM-SPELLCHECK-MISSPELLED">',
425 				word, '</span>',
426 				suffix
427 				].join("");
428 			text = [
429 				text.substr(0, m.index),
430 				repl,
431 				text.substr(m.index + str.length)
432 			].join("");
433 
434 			// All this crap necessary because the suffix
435 			// must be taken into account at the next
436 			// match and JS regexps don't have look-ahead
437 			// constructs (except \b, which sucks).  Oh well.
438 			regexp.lastIndex = m.index + repl.length - suffix.length;
439 		}
440 		return text;
441 	};
442 
443 	var doc;
444 
445 	// having the data, this function will parse the DOM and replace
446 	// occurrences of the misspelled words with <span
447 	// class="ZM-SPELLCHECK-MISSPELLED">word</span>
448 	rec = function(node) {
449 		switch (node.nodeType) {
450 			case 1: /* ELEMENT */
451 				for (var i = node.firstChild; i; i = rec(i));
452 				node = node.nextSibling;
453 				break;
454 			case 3: /* TEXT */
455 				if (!/[^\s\xA0]/.test(node.data)) {
456 					node = node.nextSibling;
457 					break;
458 				}
459 				// for correct handling of whitespace we should
460 				// not mess ourselves with leading/trailing
461 				// whitespace, thus we save it in 2 text nodes.
462 				var a = null, b = null;
463 
464 				var result = /^[\s\xA0]+/.exec(node.data);
465 				if (result) {
466 					// a will contain the leading space
467 					a = node;
468 					node = node.splitText(result[0].length);
469 				}
470 				result = /[\s\xA0]+$/.exec(node.data);
471 				if (result) {
472 					// and b will contain the trailing space
473 					b = node.splitText(node.data.length - result[0].length);
474 				}
475 
476 				var text = hiliteWords(node.data, false);
477 				text = text.replace(/^ +/, " ").replace(/ +$/, " ");
478 				var div = doc.createElement("div");
479 				div.innerHTML = text;
480 
481 				// restore whitespace now
482 				if (a) {
483 					div.insertBefore(a, div.firstChild);
484 				}
485 				if (b) {
486 					div.appendChild(b);
487 				}
488 
489 				var p = node.parentNode;
490 				while (div.firstChild) {
491 					p.insertBefore(div.firstChild, node);
492 				}
493 				div = node.nextSibling;
494 				p.removeChild(node);
495 				node = div;
496 				break;
497 			default :
498 				node = node.nextSibling;
499 		}
500 		return node;
501 	};
502 
503 	if (this._mode == DwtHtmlEditor.HTML) {
504 		// HTML mode; See the "else" branch for the TEXT mode--code differs
505 		// quite a lot.  We should probably implement separate functions as
506 		// this already becomes long.
507 
508 		doc = this._getIframeDoc();
509 		body = doc.body;
510 
511 		// load the spell check styles, if not already there.
512 		this._loadExternalStyle("/css/spellcheck.css");
513 
514 		body.style.display = "none";	// seems to have a good impact on speed,
515 										// since we may modify a lot of the DOM
516 		if (!AjxEnv.isIE) {
517 			body.normalize();
518 		} else {
519 			body.innerHTML = body.innerHTML;
520 		}
521 		rec(body);
522 		if (!AjxEnv.isIE) {
523 			body.normalize();
524 		} else {
525 			body.innerHTML = body.innerHTML;
526 		}
527 		body.style.display = ""; // redisplay the body
528 		this._registerEditorEventHandler(doc, "contextmenu");
529 	}
530 	else { // TEXT mode
531 		var textArea = document.getElementById(this._textAreaId);
532 		var scrollTop = textArea.scrollTop;
533 		var size = Dwt.getSize(textArea);
534 		textArea.style.display = "none";
535 		var div = document.createElement("div");
536 		div.className = "TextSpellChecker";
537 		this._spellCheckDivId = div.id = Dwt.getNextId();
538 		div.style.overflow = "auto";
539 		if (!AjxEnv.isIE) {
540 			// FIXME: we substract borders/padding here.  this sucks.
541 			size.x -= 4;
542 			size.y -= 6;
543 		}
544 		div.style.width = size.x + "px";
545 		div.style.height = size.y + "px";
546 
547 		div.innerHTML = AjxStringUtil.convertToHtml(this.getContent());
548 		doc = document;
549 		rec(div);
550 
551 		textArea.parentNode.insertBefore(div, textArea);
552 		div.scrollTop = scrollTop;
553 		div.oncontextmenu = div.onclick
554 			= function(ev) { self._handleSpellCheckerEvents(ev || window.event); };
555 	}
556 
557 	this._spellCheckShowModeDiv();
558 
559 	// save the spell checker context
560 	this._spellCheck = {
561 		suggestions: suggestions,
562 		spanIds: spanIds,
563 		wordIds: wordIds
564 	};
565 };

567 ZmHtmlEditor.prototype.setSize =
568 function(x, y) {
569 	var div = this._spellCheckDivId && document.getElementById(this._spellCheckDivId);
570 	var main = document.getElementById(this.getBodyFieldId());
571 
572 	// FUDGE: we must substract borders and paddings - yuck.
573 	var delta = this._mode == DwtHtmlEditor.HTML ? 8 : 10;
574 
575 
576 	if (x == Dwt.CLEAR) {
577 		main.style.width = null;
578 		if (div) div.style.width = null;
579 	} else if (x == Dwt.DEFAULT) {
580 		main.style.width = "auto";
581 		if (div) div.style.width = "auto";
582 	} else if (typeof(x) == "number") {
583 		x -= delta + 4;
584 
585 		// bug fix #6786 - normalize width/height if less than zero
586 		if (x < 0) x = 0;
587 
588 		main.style.width = x + 5 + "px";
589 		if (div) {
590 			if (!AjxEnv.isIE) {
591 				x = x > 4 ? (x-4) : x;
592 			}
593 			div.style.width = x + "px";
594 		}
595 	}
596 
597 
598 	if (y == Dwt.CLEAR) {
599 		main.style.height = null;
600 		if (div) div.style.height = null;
601 	} else if (y == Dwt.DEFAULT) {
602 		main.style.height = "auto";
603 		if (div) div.style.height = "auto";
604 	} else if (typeof(y) == "number") {
605 
606 		// subtract spellchecker DIV if applicable
607 		if (this._spellCheckModeDivId) {
608 			var spellCheckDivHeight = document.getElementById(this._spellCheckModeDivId).offsetHeight;
609 			y -= (isNaN(spellCheckDivHeight) ? 0 : spellCheckDivHeight);
610 		}
611 		if (this._mode == DwtHtmlEditor.HTML && this._toolbars.length > 0) {
612 			for (var i = 0; i < this._toolbars.length; i++) {
613 				var toolbar = this._toolbars[i];
614 				y -= toolbar.getHtmlElement().offsetHeight;
615 			}
616 		}
617 
618 		// subtract fudge factor
619 		y -= delta;
620 		if (y < 0) y = 0;
621 
622 		main.style.height = y + "px";
623 		if (div) {
624 			if (!AjxEnv.isIE) {
625 				y = y > 4 ? (y-4) : y;
626 			} else {
627 				y += 2;

840 ZmHtmlEditor.prototype.__onTableOperationsPopup =
841 function(menu) {
842 	this.focus();
843 	var table = this.getNearestElement("table");
844 	var items = menu._tblItems;
845 	for (var i in items) {
846 		items[i].setEnabled(!!table);
847 	}
848 	if (!table) { return; }
849 
850 	menu.setData("table", table);
851 
852 	if (!AjxEnv.isIE) {
853 		// Can we split? (the cell has to be a merged cell)
854 		var td = this.getNearestElement("td");
855 		var splitEnabled = td && ((td.colSpan && td.colSpan > 1) || (td.rowSpan && td.rowSpan > 1));
856 		if (items.splitCells) { //this is only if geckoBased
857 			items.splitCells.setEnabled(splitEnabled);
858 		}
859 
860 		// Can we merge? (multiple cells are selected and none of them is previously merged)
861 		var a = this.getSelectedCells();
862 		var canMerge = true;
863 		var howMany = 0;
864 		try {
865 			for (var i = a.length; --i >= 0;) {
866 				var r = a[i];
867 				for (var j = r.length; --j >= 0;) {
868 					var td = r[j];
869 					++howMany;
870 					if (td.rowSpan > 1 || td.colSpan > 1) {
871 						throw "can't merge";
872 					}
873 				}
874 			}
875 		} catch(ex) {
876 			canMerge = false;
877 		}
878 		if (howMany < 2)
879 			canMerge = false;
880 		if (items.mergeCells) { //this is only if geckoBased
881 			items.mergeCells.setEnabled(canMerge);
882 		}
883 	}

1504 ZmHtmlEditor.prototype._splitBlockquote =
1505 function(blockquote, element) {
1506 	var range, el, offset=null, coffset=null;
1507 	var text1,text2;
1508 
1509 	// Get range, el, text and offset
1510 	if (AjxEnv.isIE) {
1511 		var iFrameDoc = this._getIframeDoc();
1512 		range = iFrameDoc.selection.createRange();
1513 		range.collapse(false);
1514 		el = element;
1515 
1516 		// IE doesn't let us get the offset directly, so we count the number of times we can use moveStart() until we're out of the containing element
1517 		var dummy = "###"+Dwt.getNextId()+"###";
1518 		range.pasteHTML(dummy);
1519 		offset = el.innerHTML.indexOf(dummy);
1520 		el.innerHTML = el.innerHTML.replace(dummy,"");
1521 		text1 = offset ? el.innerHTML.substring(0, offset) : ""; // Extract text before and after breakpoint
1522 		text2 = el.innerHTML.substring(offset);
1523 
1524 	} else {
1525 		range = this._getRange();
1526 		range.collapse(false);
1527 		var type = range.startContainer.nodeType;
1528 		if (type==3 || type==4 || type==8) {
1529 			offset = range.startOffset;
1530 			el = range.startContainer.parentNode;
1531 			text1 = offset ? range.startContainer.textContent.substring(0, offset) : ""; // Extract text before and after breakpoint
1532 			text2 = range.startContainer.textContent.substring(offset);
1533 			var _el = range.startContainer.previousSibling;
1534 			while (_el) {
1535 				text1 = this._getElementHTML(_el) + text1; // offset is relative to local TextNode, but we need to use the entire text of the surrounding element
1536 				_el = _el.previousSibling; // so we extract the text from all siblings and put it together
1537 			}
1538 			_el = range.startContainer.nextSibling;
1539 			while (_el) {
1540 				text2 = text2 + this._getElementHTML(_el); // Same for text after the breakpoint
1541 				_el = _el.nextSibling;
1542 			}
1543 		} else {
1544 			coffset = range.startOffset; // Breakpoint is not inside a TextNode, store the element offset instead
1545 			el = range.startContainer;
1546 		}
1547 	}
1548 
1549 
1550 	var id = el.id = el.id || Dwt.getNextId();
1551 	var blockquote2 = blockquote.cloneNode(true); // Create an orphaned clone of the blockquote. This will be meddled with before getting attached to the DOM tree
1552 	var el1 = el;
1553 	var el2 = Dwt.byId(id, blockquote2); // Can't use document.getElementById on orphaned trees
1554 	el2.removeAttribute("id");
1555 
1556 	if (offset!==null) {
1557 		el1.innerHTML = text1; // Insert the text we extracted earlier
1558 		el2.innerHTML = text2;
1559 	} else if (coffset!==null) {
1560 		this._removeNextSiblings(el1.childNodes[coffset]); // cut away all siblings after breakpoint for el1
1561 		var c2 = el2.childNodes[coffset];
1562 		this._removePreviousSiblings(c2); // and all sibling before breakpoint for el2
1563 		this._removeElement(c2);
1564 	}
1565 
1566 	// Prune off all "later" siblings in the blockquote tree
1567 	while (el1 != blockquote) {
1568 		this._removeNextSiblings(el1);
1569 		el1 = el1.parentNode;
1570 	}
1571 
1572 	// Prune off all "prior" siblings in the blockquote2 tree
1573 	while (el2 != blockquote2) {
1574 		this._removePreviousSiblings(el2);
1575 		el2 = el2.parentNode;
1576 	}
1577 
1578 	// Now we've effectively cut the original blockquote in half, with the second half present in blockquote2
1579 	if (blockquote.nextSibling) {
1580 		blockquote.parentNode.insertBefore(blockquote2, blockquote.nextSibling);
1581 	} else {
1582 		blockquote.parentNode.appendChild(blockquote2);
1583 	}
1584 
1585 	if (AjxEnv.isIE) {
1586 		// Hack to get IE to properly place the cursor between the two blockquotes
1587 		var p = document.createElement("p");
1588 		var span1 = document.createElement("span");
1589 		var span2 = document.createElement("span");
1590 		var br = document.createElement("br");
1591 		span1.appendChild(span2);
1592 		span1.appendChild(br);
1593 		p.appendChild(span1);
1594 		blockquote2.parentNode.insertBefore(p, blockquote2);
1595 		this._setCursor(span2);
1596 		this._setIEFiller(span1); // We need to remove this element when we want to reconnect the blockquotes, so give it something we can find again
1597 		span2.parentNode.removeChild(span2);
1598 	} else {
1599 		if (AjxEnv.isSafari) {
1600 			var t = this;
1601 			setTimeout(function(){
1602 				var p;
1603 				if (blockquote.lastChild && blockquote.lastChild.childElementCount==1 && blockquote.lastChild.firstChild instanceof HTMLBRElement)
1604 					p = blockquote.lastChild;
1605 				else if (blockquote2.firstChild && blockquote2.firstChild.childElementCount==1 && blockquote2.firstChild.firstChild instanceof HTMLBRElement)
1606 					p = blockquote2.firstChild;
1607 				if (p) {
1608 					blockquote.parentNode.insertBefore(p, blockquote2);
1609 					range.setStart(p,0);
1610 					var sel = t._getSelection();
1611 					sel.removeAllRanges();
1612 					sel.addRange(range);
1613 				}
1614 			},5);
1615 		}
1616 		range.setStartAfter(blockquote); // Set the breakpoint between the blockquotes (which is immediately after the first one, duh)
1617 		var sel = this._getSelection();
1618 		sel.removeAllRanges();
1619 		sel.addRange(range);
1620 	}
