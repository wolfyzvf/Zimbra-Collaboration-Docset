 16 /**
 17  * 
 18  * 
 19  * @private
 20  */
 21 DwtSelectionEvent = function(init) {
 22 	if (arguments.length == 0) return;
 23 	DwtUiEvent.call(this, true);
 24 	this.reset(true);
 25 }
 26 
 27 DwtSelectionEvent.prototype = new DwtUiEvent;
 28 DwtSelectionEvent.prototype.constructor = DwtSelectionEvent;
 29 
 30 DwtSelectionEvent.prototype.toString = 
 31 function() {
 32 	return "DwtSelectionEvent";
 33 }
 34 
 35 DwtSelectionEvent.prototype.reset =
 36 function(dontCallParent) {
 37 	if (!dontCallParent)
 38 		DwtUiEvent.prototype.reset.call(this);
 39 	this.button = 0;
 40 	this.detail = null;
 41 	this.item = null;
 42 }
