 20 /**
 21  * Creates a dialog for choosing a tag.
 22  * @class
 23  * This class presents the user with their list of tags in a tree view so that they
 24  * can choose one. There is a text input that can be used to filter the list.
 25  *
 26  * @param	{DwtControl}	parent		the parent
 27  * @param	{String}	className		the class name
 28  * 
 29  * @extends	ZmDialog
 30  */
 31 ZmPickTagDialog = function(parent, className) {
 32 
 33 	var newButton = new DwtDialog_ButtonDescriptor(ZmPickTagDialog.NEW_BUTTON, ZmMsg._new, DwtDialog.ALIGN_LEFT);
 34 	var params = {parent:parent, className:className, title:ZmMsg.pickATag, extraButtons:[newButton], id: "ZmPickTagDialog"};
 35 	ZmDialog.call(this, params);
 36 
 37 	this._createControls();
 38 	this._setNameField(this._inputDivId);
 39 	this._tagTreeChangeListener = new AjxListener(this, this._handleTagTreeChange);
 40 	appCtxt.getTagTree().addChangeListener(this._tagTreeChangeListener);
 41 	this.registerCallback(ZmPickTagDialog.NEW_BUTTON, this._showNewDialog, this);
 42 	this._creatingTag = false;
 43 	this._treeViewListener = new AjxListener(this, this._treeViewSelectionListener);
 44 	this._lastVal = "";
 45 };
 46 
 47 ZmPickTagDialog.prototype = new ZmDialog;
 48 ZmPickTagDialog.prototype.constructor = ZmPickTagDialog;
 49 
 50 ZmPickTagDialog.NEW_BUTTON = DwtDialog.LAST_BUTTON + 1;
 51 
 52 ZmPickTagDialog.prototype.toString = 
 53 function() {
 54 	return "ZmPickTagDialog";
 55 };
 56 
 57 /**
 58  * Pops-up the dialog.
 59  * 
 60  * @param	{Hash}	params		a hash of parameters
 61  * @param	{ZmAccount}	params.account		the account
 62  */
 63 ZmPickTagDialog.prototype.popup = 
 64 function(params) {
 65 
 66 	params = params || {};
 67 	this._account = params.account;
 68 
 69 	if (appCtxt.multiAccounts && params.account) {
 70 		appCtxt.getTagTree().removeChangeListener(this._tagTreeChangeListener);
 71 		appCtxt.getTagTree(params.account).addChangeListener(this._tagTreeChangeListener);
 72 	}
 73 
 74 	// all this is done here instead of in the constructor due to multi-account issues
 75 	var overviewId = (appCtxt.multiAccounts && params.account)
 76 		? ([this.toString(), "-", params.account.name].join("")) : this.toString();
 77 
 78 	var ovParams = {
 79 		overviewId:			overviewId,
 80 		treeIds:			[ZmOrganizer.TAG],
 81 		fieldId:			this._tagTreeDivId,
 82 		account:			params.account
 83 	};
 84 	this._setOverview(ovParams, true);
 85 	this._tagTreeView = this._getOverview().getTreeView(ZmOrganizer.TAG);
 86 	this._tagTreeView.removeSelectionListener(this._treeViewListener);
 87 	this._tagTreeView.addSelectionListener(this._treeViewListener);
 88 	var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT);
 89 	var root = this._tagTreeView.getTreeItemById(rootId);
 90 	if (root) {
 91 		root.enableSelection(false);
 92 	}
 93 
 94 	this._loadTags();	// item list for this account's tree view will be cached after the first time
 95 	this._resetTreeView();
 96 	this._focusElement = this._inputField;
 97 	this._inputField.setValue("");
 98 	var tags = appCtxt.getTagTree(this._account).asList();
 99 	if (tags.length == 1) {
100 		this._tagTreeView.setSelected(tags[0], true, true);
101 	}
102 	ZmDialog.prototype.popup.apply(this, arguments);
103 };
104 
105 ZmPickTagDialog.prototype._contentHtml = 
106 function() {
107 	this._tagTreeDivId = this._htmlElId + "_tagTreeDivId";
108 	this._inputDivId = this._htmlElId + "_inputDivId";
109 
110 	return AjxTemplate.expand("share.Widgets#ZmPickTagDialog", {id:this._htmlElId});
111 };
112 
113 ZmPickTagDialog.prototype._createControls =
114 function() {
115 	this._inputField = new DwtInputField({parent: this});
116 	document.getElementById(this._inputDivId).appendChild(this._inputField.getHtmlElement());
117 	this._inputField.addListener(DwtEvent.ONKEYUP, new AjxListener(this, this._handleKeyUp));
118 };
119 
120 ZmPickTagDialog.prototype._showNewDialog = 
121 function() {
122 	var dialog = appCtxt.getNewTagDialog();
123 	dialog.reset();
124 	dialog.registerCallback(DwtDialog.OK_BUTTON, this._newCallback, this);
125 	dialog.popup(null, this._account);
126 };
127 
128 ZmPickTagDialog.prototype._newCallback = 
129 function(parent, name) {
130 	appCtxt.getNewTagDialog().popdown();
131 	var ttc = this._opc.getTreeController(ZmOrganizer.TAG);
132 	ttc._doCreate(parent, name);
133 	this._creatingTag = true;
134 };
135 
136 ZmPickTagDialog.prototype._loadTags =
137 function() {
138 	this._tags = [];
139 	var items = this._tagTreeView.getTreeItemList();
140 	for (var i = 0, len = items.length; i < len; i++) {
141 		var tag = items[i].getData(Dwt.KEY_OBJECT);
142 		if (tag.id != ZmOrganizer.ID_ROOT) {
143 			this._tags.push({id:tag.id, name:tag.getName(false, null, true, true).toLowerCase()});
144 		}
145 	}
146 };
147 
148 ZmPickTagDialog.prototype._handleTagTreeChange =
149 function(ev) {
150 	// TODO - listener for changing tags
151 	if (ev.event == ZmEvent.E_CREATE && this._creatingTag) {
152 		var tag = ev.getDetail("organizers")[0];
153 		this._tagTreeView.setSelected(tag, true);
154 		this._creatingTag = false;
155 	}
156 	this._loadTags();
157 };
158 
159 ZmPickTagDialog.prototype._okButtonListener = 
160 function(ev) {
161 	DwtDialog.prototype._buttonListener.call(this, ev, [this._tagTreeView.getSelected()]);
162 };
163 
164 ZmPickTagDialog.prototype._handleKeyUp =
165 function(ev) {
166 
167 	var key = DwtKeyEvent.getCharCode(ev);
168 	if (key == 9) {
169 		return;
170 	} else if (key == 40) {
171 		this._overview[this._curOverviewId].focus();
172 		return;
173 	}
174 	
175 	var num = 0, firstMatch;
176 	var value = this._inputField.getValue().toLowerCase();
177 	if (value == this._lastVal) { return; }
178 	for (var i = 0, len = this._tags.length; i < len; i++) {
179 		var tagInfo = this._tags[i];
180 		var ti = this._tagTreeView.getTreeItemById(tagInfo.id);
181 		if (ti) {
182 			var matches = (tagInfo.name.indexOf(value) == 0);
183 			ti.setVisible(matches);
184 			if (matches && !firstMatch) {
185 				firstMatch = tagInfo.id;
186 			}
187 		}
188 	}
189 
190 	if (firstMatch) {
191 		this._tagTreeView.setSelected(appCtxt.getById(firstMatch), true, true);
192 	}
193 	this._lastVal = value;
194 };
195 
196 ZmPickTagDialog.prototype._resetTreeView =
197 function() {
198 	// make all tree items visible (in case there was prior filtering)
199 	for (var i = 0, len = this._tags.length; i < len; i++) {
200 		var ti = this._tagTreeView.getTreeItemById(this._tags[i].id);
201 		if (ti) {
202 			ti.setVisible(true);
203 		}
204 	}
205 
206 	this._tagTreeView.getHeaderItem().setExpanded(true, false, true);
207 };
208 
209 ZmPickTagDialog.prototype._treeViewSelectionListener =
210 function(ev) {
211 
212 	if (ev.detail != DwtTree.ITEM_SELECTED && ev.detail != DwtTree.ITEM_DBL_CLICKED)	{ return; }
213 
214 	var tag = ev.item.getData(Dwt.KEY_OBJECT);
215 	if (tag) {
216 		var value = tag.getName(false, null, true, true);
217 		this._lastVal = value.toLowerCase();
218 		this._inputField.setValue(value);
219 		if (ev.detail == DwtTree.ITEM_DBL_CLICKED || ev.enter) {
220 			this._okButtonListener();
221 		}
222 	}
223 };
224 
225 ZmPickTagDialog.prototype._getTabGroupMembers =
226 function() {
227 	return [this._inputField, this._overview[this._curOverviewId]];
228 };
229 
230 ZmPickTagDialog.prototype._enterListener =
231 function(ev) {
232 	this._okButtonListener.call(this, ev);
233 };
