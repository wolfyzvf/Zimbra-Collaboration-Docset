 17 /**
 18  * Resets the AjxPost object.
 19  * @constructor
 20  * @class
 21  * This singleton class makes an HTTP POST to the server and receives the response, passing returned data
 22  * to a callback. This class is used to upload files from the client browser to the server using the file
 23  * upload feature of POST.
 24  *
 25  * @param	{string}	iframeId		the iframe ID
 26  * 
 27  * @author Conrad Damon
 28  * 
 29  * @private
 30  */
 31 AjxPost = function(iframeId) {
 32 	this._callback = null;
 33 	this._iframeId = iframeId;
 34 }
 35

 63 /**
 64 * Submits the form.
 65 *
 66 * @param callback		function to return to after the HTTP response is received
 67 * @param formId			DOM ID of the form
 68 */
 69 AjxPost.prototype.execute =
 70 function(callback, form, optionalTimeout) {
 71 	// bug fix #7361
 72 	var tags = form.getElementsByTagName("input");
 73 	var inputs = new Array();
 74 	for (var i = 0; i < tags.length; i++) {
 75 		var tag = tags[i];
 76 		if (tag.type == "file") {
 77 			inputs.push(tag);
 78 			continue;
 79 		}
 80 		// clean up form from previous posts
 81 		if (tag.name && tag.name.match(/^filename\d+$/)) {
 82 			tag.parentNode.removeChild(tag);
 83 			i--; // list is live, so stay on same index
 84 			continue;
 85 		}
 86 	}
 87 
 88     this._addHiddenFileNames(inputs);
 89 
 90 	form.target = this._iframeId;
 91 	this._callback = callback;
 92 	var req = new AjxPostRequest(form);
 93 	var failureAction = new AjxTimedAction(this, this._onFailure, [req.id]);
 94 	var timeout = optionalTimeout? optionalTimeout: 5000;
 95 	AjxPost._outStandingRequests[req.id] = req;
 96 	try {
 97 		req.send(failureAction, timeout);
 98 	} catch (ex) {
 99 		if (AjxEnv.isIE) {
100 			if (ex.number == -2147024891) { // 0x80070005: E_ACCESSDENIED (Couldn't open file)
101 				throw new AjxException(ZmMsg.uploadErrorAccessDenied, ex.number);
102 			}
103 		}
104 		throw ex;
105 	}
106 };
107 
108 AjxPost.prototype._addHiddenFileNames =
109 function(inputs){
110     var m = 0;
111     for (var i = 0; i < inputs.length; i++) {
112         var fileInput = inputs[i];
113         if(fileInput.files && fileInput.files.length > 1){
114             var files = fileInput.files, fileStr=[];
115             for(var j=0; j<files.length; j++){
116                var f = files[j];
117                fileStr.push(f.name || f.fileName);
118             }
119             this._addHiddenFileName(inputs[i], fileStr.join('\n'), ++m);
120         }else{
121             this._addHiddenFileName(inputs[i], inputs[i].value, ++m);
122         }
123     }
124 
125 };
126 
127 AjxPost.prototype._addHiddenFileName =
128 function(inputField, fileName, index){
129     var hidden = document.createElement("input");
130     hidden.type = "hidden";
131     hidden.name = "filename" + (index);
132     hidden.value = fileName;
133     inputField.parentNode.insertBefore(hidden, inputField);
134 
135 };

140 AjxPost.prototype._onFailure =
141 function (reqId){
142 	var req = AjxPost._outStandingRequests[reqId];
143 	req.cancel();
144 	delete AjxPost._outStandingRequests[reqId];
145 	if (this._callback) {
146 		this._callback.run([404]);
147 		this._callback = null;
148 	}
149 };

153 /**
154 * Processes the HTTP response from the form post. The server needs to make sure this function is
155 * called and passed the appropriate args. Something like the following should do the trick:
156 * <code>
157 *        out.println("<html><head></head><body onload=\"window.parent._uploadManager.loaded(" + results +");\"></body></html>");
158 * </code>
159 *
160 * @param status		an HTTP status
161 * @param id			the id for any attachments that were uploaded
162 */
163 AjxPost.prototype.loaded =
164 function(status, reqId, id) {
165 	//alert(document.getElementById(this._iframeId).contentWindow.document.documentElement.innerHTML);
166 	var req = AjxPost._outStandingRequests[reqId];
167 	if (req && !req.hasBeenCancelled()) {
168 		req.cancelTimeout();
169 	}
170 	delete AjxPost._outStandingRequests[reqId];
171 	if (this._callback) {
172 		this._callback.run(status, id);
173 		this._callback = null;
174 	}
175 };

177 /**
178  * @class
179  * 
180  * @private
181  */
182 AjxPostRequest = function(form) {
183 	this.id = AjxPost._reqIds++;
184 	this._cancelled = false;
185 	this._form = form;
186 	var inp = form.elements.namedItem("requestId");
187 	if (!inp) {
188 		inp = form.ownerDocument.createElement('input');
189 		inp.type = "hidden";
190 		inp.name = "requestId";
191 	}
192 	inp.value = this.id;
193 	form.appendChild(inp);
194 };
195 
196 AjxPostRequest.prototype.send =
197 function(failureAction, timeout) {
198 	this._form.submit();
199 };
200 
201 AjxPostRequest.prototype.hasBeenCancelled =
202 function() {
203 	return this._cancelled;
204 };
205 
206 AjxPostRequest.prototype.cancelTimeout =
207 function() {
208 	AjxTimedAction.cancelAction(this._timeoutId);
209 };
210 
211 AjxPostRequest.prototype.cancel =
212 function() {
213 	this._cancelled = true;
214 };
