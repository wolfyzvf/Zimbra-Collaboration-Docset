 21 /**
 22 * Creates an empty organizer.
 23 * @class
 24 * This class represents an "organizer", which is something used to classify or contain
 25 * items. So far, that's either a tag or a folder. Tags and folders are represented as
 26 * a tree structure, though tags are flat and have only one level below the root. Folders
 27 * can be nested.
 28 *
 29 * @author Conrad Damon
 30 *
 31 * @param	{Hash}	params		a hash of parameters
 32 * @param {constant}	params.type		the organizer type
 33 * @param {int}		params.id			the numeric ID
 34 * @param {String}	params.name		the name
 35 * @param {ZmOrganizer}	params.parent		the parent organizer
 36 * @param {ZmTree}	params.tree		the tree model that contains this organizer
 37 * @param {constant}	params.color		the color for this organizer
 38 * @param {String}	params.rgb		the color for this organizer, as HTML RGB value
 39 * @param {Boolean}	params.link		<code>true</code> if this organizer is shared
 40 * @param {Boolean}	params.broken		<code>true</code> if this link is broken
 41 * @param {int}	params.numUnread	the number of unread items for this organizer
 42 * @param {int}	params.numTotal	the number of items for this organizer
 43 * @param {Boolean}	params.noTooltip	do not show tooltip 
 44 * @param {int}	params.sizeTotal	the total size of organizer's items
 45 * @param {String}	params.url		the URL for this organizer's feed
 46 * @param {String}	params.owner		the owner for this organizer
 47 * @param {String}	params.oname		the owner's name for this organizer
 48 * @param {String}	params.zid		the Zimbra ID of owner, if remote folder
 49 * @param {String}	params.rid		the remote ID of organizer, if remote folder
 50 * @param {String}	params.restUrl	the REST URL of this organizer.
 51 * @param {String}	params.newOp		the name of operation run by button in overview header
 52 * @param {ZmZimbraAccount}	params.account	the account this organizer belongs to
 53 */
 54 ZmOrganizer = function(params) {
 55 
 56 	if (arguments.length == 0) { return; }
 57 
 58 	this.type = params.type;
 59 	var id = this.id = params.id;
 60 	this.nId = ZmOrganizer.normalizeId(id);
 61 	this.name = ZmFolder.MSG_KEY[this.nId] ? ZmMsg[ZmFolder.MSG_KEY[this.nId]] : params.name;
 62 	this._systemName = this.nId < 256 && params.name;
 63 	this.parent = params.parent;
 64 	this.tree = params.tree;
 65 	this.numUnread = params.numUnread || 0;
 66 	this.numTotal = params.numTotal || 0;
 67 	this.noTooltip = params.noTooltip;
 68 	this.sizeTotal = params.sizeTotal || 0;
 69 	this.url = params.url;
 70 	this.owner = params.owner;
 71 	this.oname = params.oname;
 72 	this.link = params.link || (Boolean(params.zid)) || (this.parent && this.parent.link);
 73 	this.isMountpoint = params.link;
 74 	this.zid = params.zid;
 75 	this.rid = params.rid;
 76 	this.restUrl = params.restUrl;
 77 	this.account = params.account;
 78     this.perm = params.perm;
 79 	this.noSuchFolder = params.broken; // Is this a link to some folder that ain't there.
 80 	this._isAdmin = this._isReadOnly = this._hasPrivateAccess = null;
 81     this.retentionPolicy = params.retentionPolicy;
 82 
 83 	this.color =
 84         params.color ||
 85         (this.parent && this.parent.color) ||
 86         ZmOrganizer.ORG_COLOR[id] ||
 87         ZmOrganizer.ORG_COLOR[this.nId] ||
 88         ZmOrganizer.DEFAULT_COLOR[this.type] ||
 89         ZmOrganizer.C_NONE
 90     ;
 91 	this.isColorCustom = params.rgb != null; //set so we know if the user chose a custom color (to distiguish from basic color or none
 92 	this.rgb =
 93         params.rgb ||
 94         ZmOrganizer.COLOR_VALUES[this.color] ||
 95         ZmOrganizer.COLOR_VALUES[ZmOrganizer.ORG_DEFAULT_COLOR]
 96     ;
 97 
 98 	if (appCtxt.isOffline && !this.account && this.id == this.nId) {
 99 		this.account = appCtxt.accountList.mainAccount;
100 	}
101 
102 	if (id && params.tree) {
103 		appCtxt.cacheSet(id, this);
104 		if (this.link) {
105 			// also store under ID that items use for parent folder ("l" attribute in node)
106 			appCtxt.cacheSet([this.zid, this.rid].join(":"), this);
107 		}
108 	}
109 
110 	this.children = new AjxVector();
111 };

325 ZmOrganizer.registerOrg =
326 function(org, params) {
327 	if (params.nameKey)			{ ZmOrganizer.MSG_KEY[org]				= params.nameKey; }
328 	if (params.app)				{
329 		ZmOrganizer.APP[org] = params.app;
330 		if (!ZmOrganizer.APP2ORGANIZER[params.app]) {
331 			ZmOrganizer.APP2ORGANIZER[params.app] = [];
332 		}
333 		ZmOrganizer.APP2ORGANIZER[params.app].push(org);
334         ZmOrganizer.APP2ORGANIZER_R[org] = params.app;
335 	}
336 	if (params.defaultFolder)	{ ZmOrganizer.DEFAULT_FOLDER[org]		= params.defaultFolder; }
337 	if (params.precondition)	{ ZmOrganizer.PRECONDITION[org]			= params.precondition; }
338 	if (params.soapCmd)			{ ZmOrganizer.SOAP_CMD[org]				= params.soapCmd; }
339 	if (params.firstUserId)		{ ZmOrganizer.FIRST_USER_ID[org]		= params.firstUserId; }
340 	if (params.orgClass)		{ ZmOrganizer.ORG_CLASS[org]			= params.orgClass; }
341 	if (params.orgPackage)		{ ZmOrganizer.ORG_PACKAGE[org]			= params.orgPackage; }
342 	if (params.labelKey)		{ ZmOrganizer.LABEL[org]				= params.labelKey; }
343 	if (params.itemsKey)		{ ZmOrganizer.ITEMS_KEY[org]			= params.itemsKey; }
344 	if (params.hasColor)		{ ZmOrganizer.HAS_COLOR[org]			= params.hasColor; }
345 	if (params.views)			{ ZmOrganizer.VIEWS[org]				= params.views; }
346 	if (params.folderKey)		{ ZmOrganizer.FOLDER_KEY[org]			= params.folderKey; }
347 	if (params.mountKey)		{ ZmOrganizer.MOUNT_KEY[org]			= params.mountKey; }
348 	if (params.deferrable)		{ ZmOrganizer.DEFERRABLE[org]			= params.deferrable; }
349 	if (params.pathInName)		{ ZmOrganizer.PATH_IN_NAME[org]			= params.pathInName; }
350 	if (params.openSetting)		{ ZmOrganizer.OPEN_SETTING[org]			= params.openSetting; }
351 	if (params.newOp)			{ ZmOrganizer.NEW_OP[org]				= params.newOp; }
352 	if (params.displayOrder)	{ ZmOrganizer.DISPLAY_ORDER[org]		= params.displayOrder; }
353 	if (params.hideEmpty)		{ ZmOrganizer.HIDE_EMPTY[org]			= params.hideEmpty; }
354 	ZmOrganizer.SHAREABLE[org]	= !params.disableShare; 
355 
356 	if (!appCtxt.isChildWindow || params.childWindow ) {
357 		if (params.compareFunc)		{ ZmTreeView.COMPARE_FUNC[org]			= params.compareFunc; }
358 		if (params.treeController)	{ ZmOverviewController.CONTROLLER[org]	= params.treeController; }
359 	}
360 
361 	ZmOrganizer.TREE_TYPE[org] = params.treeType || org; // default to own type
362 	ZmOrganizer.CREATE_FUNC[org] = params.createFunc || "ZmOrganizer.create";
363 
364 	if (params.views) {
365 		ZmOrganizer.VIEW_HASH[org] = AjxUtil.arrayAsHash(ZmOrganizer.VIEWS[org]);
366 	}
367 
368 	if (params.hasColor) {
369 		ZmOrganizer.DEFAULT_COLOR[org] = (params.defaultColor != null)
370 			? params.defaultColor
371 			: ZmOrganizer.ORG_DEFAULT_COLOR;
372 	}
373 
374 	if (params.orgColor) {
375 		for (var id in params.orgColor) {
376 			ZmOrganizer.ORG_COLOR[id] = params.orgColor[id];
377 		}
378 	}
379 
380 	if (params.dropTargets) {
381 		if (!ZmApp.DROP_TARGETS[params.app]) {
382 			ZmApp.DROP_TARGETS[params.app] = {};
383 		}
384 		ZmApp.DROP_TARGETS[params.app][org] = params.dropTargets;
385 	}
386 };

404 /**
405  * Creates an organizer via <code><CreateFolderRequest></code>. Attribute pairs can
406  * be passed in and will become attributes of the folder node in the request.
407  * 
408  * @param {Hash}	params	a hash of parameters
409  */
410 ZmOrganizer.create =
411 function(params) {
412 	var jsonObj = {CreateFolderRequest:{_jsns:"urn:zimbraMail"}};
413 	var folder = jsonObj.CreateFolderRequest.folder = {};
414 	var errorCallback = params.errorCallback || new AjxCallback(null, ZmOrganizer._handleErrorCreate, params);
415 	var type = params.type;
416 
417 	// set attributes
418 	params.view = params.view || ZmOrganizer.VIEWS[type] ? ZmOrganizer.VIEWS[type][0] : null;
419 	for (var i in params) {
420 		if (i == "type" || i == "errorCallback" || i == "account") { continue; }
421 
422 		var value = params[i];
423 		if (i == "color") {
424 			value = ZmOrganizer.getColorValue(value, type);
425 		}
426 		if (value) {
427 			folder[i] = value;
428 		}
429 	}
430 	//adding support to asyncMode == false didn't eventually help me, but why not keep it?
431 	var asyncMode = params.asyncMode === undefined ? true : params.asyncMode; //default is true
432 
433 	return appCtxt.getAppController().sendRequest({
434 		jsonObj: jsonObj,
435 		asyncMode: asyncMode,
436 		accountName: (params.account && params.account.name),
437 		callback: params.callback,
438 		callbackAfterNotifications: params.callbackAfterNotifications, 
439 		errorCallback: errorCallback
440 	});
441 };

443 /**
444  * @private
445  */
446 ZmOrganizer._handleErrorCreate =
447 function(params, ex) {
448 	if (!params.url && !params.name) { return false; }
449 	
450 	var msg;
451 	if (params.name && (ex.code == ZmCsfeException.MAIL_ALREADY_EXISTS)) {
452 		var type = appCtxt.getFolderTree(appCtxt.getActiveAccount()).getFolderTypeByName(params.name);
453         msg = AjxMessageFormat.format(ZmMsg.errorAlreadyExists, [params.name, type.toLowerCase()]);
454 	} else if (params.url) {
455 		var errorMsg = (ex.code == ZmCsfeException.SVC_RESOURCE_UNREACHABLE) ? ZmMsg.feedUnreachable : ZmMsg.feedInvalid;
456 		msg = AjxMessageFormat.format(errorMsg, params.url);
457 	}
