 16 /**
 17  * Creates an POP account.
 18  * @class
 19  * This class represents an POP account.
 20  * 
 21  * @param	{String}	id		the id
 22  * 
 23  * @extends		ZmDataSource
 24  */ZmPopAccount = function(id) {
 25 	ZmDataSource.call(this, ZmAccount.TYPE_POP, id);
 26 };
 27 
 28 ZmPopAccount.prototype = new ZmDataSource;
 29 ZmPopAccount.prototype.constructor = ZmPopAccount;
 30 
 31 // Constants
 32 /**
 33  * Defines the "cleartext" port.
 34  * 
 35  * @type	int
 36  */
 37 ZmPopAccount.PORT_CLEAR 	= 110;
 38 /**
 39  * Defines the "ssl" port.
 40  * 
 41  * @type	int
 42  */
 43 ZmPopAccount.PORT_SSL 		= 995;
 44 ZmPopAccount.PORT_DEFAULT	= ZmPopAccount.PORT_CLEAR;
 45 
 46 
 47 // advanced settings
 48 ZmPopAccount.prototype.ELEMENT_NAME = "pop3";
 49 ZmPopAccount.prototype.port = ZmPopAccount.PORT_DEFAULT;
 50 
 51 
 52 // Public methods
 53 
 54 ZmPopAccount.prototype.toString =
 55 function() {
 56 	return "ZmPopAccount";
 57 };
 58 
 59 /**
 60  * Gets the default port.
 61  * 
 62  * @return	{int}		the port
 63  */
 64 ZmPopAccount.prototype.getDefaultPort =
 65 function() {
 66 	return (this.connectionType == ZmDataSource.CONNECT_SSL)
 67 		? ZmPopAccount.PORT_SSL : ZmPopAccount.PORT_DEFAULT;
 68 };
