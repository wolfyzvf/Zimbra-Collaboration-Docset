 29 /**
 30  * Creates the account.
 31  * @class
 32  * This class represents an account.
 33  * 
 34  * @param	{constant}	type	the account type (see <code>ZmAccount.TYPE_</code> constants)
 35  * @param	{String}	id		the account id
 36  * @param	{String}	name	the account name
 37  * @see		ZmAccount
 38  */
 39 ZmAccount = function(type, id, name) {
 40 	if (arguments.length == 0) { return; }
 41 
 42 	this.id = id;
 43 	this.name = name;
 44 	this.type = type || ZmAccount.TYPE_ZIMBRA;
 45 };

105 /**
106  * Gets the name of the specified type.
107  * 
108  * @param	{constant}	type		the type (see <code>ZmAccount.TYPE_</code> constants)
109  * @return	{String}	the name or unknown
110  * 
111  * @see		ZmAccount
112  */
113 ZmAccount.getTypeName =
114 function(type) {
115 	switch (type) {
116 		case ZmAccount.TYPE_AOL:		return ZmMsg.aol;
117 		case ZmAccount.TYPE_GMAIL:		return ZmMsg.gmail;
118 		case ZmAccount.TYPE_IMAP:		return ZmMsg.accountTypeImap;
119 		case ZmAccount.TYPE_LIVE:		return ZmMsg.msLive;
120 		case ZmAccount.TYPE_MSE:		return ZmMsg.msExchange;
121 		case ZmAccount.TYPE_EXCHANGE:	return ZmMsg.msExchange;
122 		case ZmAccount.TYPE_PERSONA:	return ZmMsg.accountTypePersona;
123 		case ZmAccount.TYPE_POP:		return ZmMsg.accountTypePop;
124 		case ZmAccount.TYPE_YMP:		return ZmMsg.yahooMail;
125 		case ZmAccount.TYPE_ZIMBRA:		return ZmMsg.zimbraTitle;
126 	}
127 	return ZmMsg.unknown;
128 };
