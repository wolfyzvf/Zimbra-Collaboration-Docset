116 ZmMailListView.prototype.set =
117 function(list, sortField) {
118 
119 	var s = this._controller._activeSearch && this._controller._activeSearch.search;
120 	this._folderId = s && s.folderId;
121     if (this._folderId) {
122         this._group = this.getGroup(this._folderId);
123     }
124 
125 	var sortBy = s && s.sortBy;
126 	if (sortBy) {
127 		var column;
128 		if (sortBy == ZmSearch.SUBJ_DESC || sortBy == ZmSearch.SUBJ_ASC) {
129 			column = ZmItem.F_SUBJECT;
130 		} else if (sortBy == ZmSearch.DATE_DESC || sortBy == ZmSearch.DATE_ASC) {
131 			column = ZmItem.F_DATE;
132 		} else if (sortBy == ZmSearch.NAME_DESC || sortBy == ZmSearch.NAME_ASC) {
133 			column = ZmItem.F_FROM;
134 		} else if (sortBy == ZmSearch.SIZE_DESC || sortBy == ZmSearch.SIZE_ASC) {
135 			column = ZmItem.F_SIZE;
136 		}
137 		if (column) {
138 			var sortByAsc = (sortBy == ZmSearch.SUBJ_ASC || sortBy == ZmSearch.DATE_ASC || sortBy == ZmSearch.NAME_ASC || sortBy == ZmSearch.SIZE_ASC);
139 			this.setSortByAsc(column, sortByAsc);
140 		}
141 	}
142 
143 
144     ZmListView.prototype.set.apply(this, arguments);
145 
146     this.markDefaultSelection(list);
147 };

167 ZmMailListView.prototype.handleKeyAction =
168 function(actionCode, ev) {
169 
170 	switch (actionCode) {
171 		case DwtKeyMap.SELECT_ALL:

173 			var ctlr = this._controller;
174 			ctlr._resetOperations(ctlr.getCurrentToolbar(), this.getSelectionCount());
175 			return true;

178 		case DwtKeyMap.SELECT_PREV:
179 			// Block widget shortcut for space since we want to handle it as app shortcut.
180 			if (ev.charCode === 32) {
181 				return false;
182 			}
183 			this._controller.lastListAction = actionCode;
184 		
185 		default:
186 			return DwtListView.prototype.handleKeyAction.apply(this, arguments);
187 	}

263 function(field, item) {
264 	if (field != "fr" && field != "su" && field != "st")

275 			//zimlet not available or disabled, set _ignoreProcessingGetMailCellStyle to true

277 			this._ignoreProcessingGetMailCellStyle = true;
278 		}
279 	}

376 ZmMailListView.prototype._getHeaders =
377 function(viewId, headerList) {
378 
379 	this._initHeaders();
380 	var hList = [];
381 
382 	this._defaultCols = headerList.join(ZmListView.COL_JOIN);
383 	var isMultiColumn = !this._controller.isReadingPaneOnRight();
384 	var userHeaders = isMultiColumn && appCtxt.get(ZmSetting.LIST_VIEW_COLUMNS, viewId);
385 	var headers = headerList;
386 	if (userHeaders && isMultiColumn) {
387 		headers = userHeaders.split(ZmListView.COL_JOIN);
388 		//we have to do it regardless of the size of headers and headerList, as items could be added and removed, masking each other as far as length (previous code compared length)
389 		headers = this._normalizeHeaders(headers, headerList);
390 	}
391     // adding account header in _normalizeHeader method
392     // sometimes doesn't work since we check for array length which is bad.
393 
394     // in ZD in case of All-Mailbox search always make sure account header is added to header array
395     if(appCtxt.isOffline && appCtxt.getSearchController().searchAllAccounts && isMultiColumn) {
396         var isAccHdrEnabled = false;
397         for (var k=0; k< headers.length; k++) {
398             if(headers[k] == ZmItem.F_ACCOUNT) {
399                 isAccHdrEnabled = true;
400             }
401         }
402         if(!isAccHdrEnabled) {
403             headers.splice(headers.length - 1, 0, ZmId.FLD_ACCOUNT);
404         }
405 
406     }
407 
408 	for (var i = 0, len = headers.length; i < len; i++) {
409 		var header = headers[i];
410 		var field = header.substr(0, 2);
411 		var hdrParams = this._headerInit[field];
412 		if (!hdrParams) { continue; }
413 		var pre = hdrParams.precondition;
414 		if (!pre || appCtxt.get(pre)) {
415 			hdrParams.field = field;
416 			// multi-account, account header is always initially invisible
417 			// unless user is showing global inbox. Ugh.
418 			if (appCtxt.multiAccounts &&
419 				appCtxt.accountList.size() > 2 &&
420 				appCtxt.get(ZmSetting.OFFLINE_SHOW_ALL_MAILBOXES) &&
421 				header.indexOf(ZmItem.F_ACCOUNT) != -1)
422 			{
423 				hdrParams.visible = true;
424 				this._showingAccountColumn = true;
425 			} else {
426 				var visible = (appCtxt.multiAccounts && header == ZmItem.F_ACCOUNT && !userHeaders)
427 					? false : (header.indexOf("*") == -1);
428                 if (!userHeaders && isMultiColumn) {
429                     //this is the default header
430                     if (typeof hdrParams.visible === "undefined") {
431                         //if the visible header is not set than use the computed value
432                         hdrParams.visible = visible;
433                     }
434                 } else {
435                     hdrParams.visible = visible;
436                 }
437 			}
438 			hList.push(new DwtListHeaderItem(hdrParams));
439 		}
440 	}
441 
442 	return hList;
443 };

445 /**
446  * Cleans up the list of columns in various ways:
447  * 		- Add new fields in penultimate position
448  * 		- Remove duplicate fields
449  * 		- Remove fields that aren't valid for the view
450  *
451  * @param userHeaders	[Array]		user-defined set of column headers
452  * @param headerList	[Array]		default set of column headers
453  */
454 ZmMailListView.prototype._normalizeHeaders =
455 function(userHeaders, headerList) {
456 
457 	// strip duplicates and invalid headers
458 	var allHeaders = AjxUtil.arrayAsHash(headerList);
459 	var headers = [], used = {}, starred = {};
460 	for (var i = 0; i < userHeaders.length; i++) {
461 		var hdr = userHeaders[i];
462 		var idx = hdr.indexOf("*");
463 		if (idx != -1) {
464 			hdr = hdr.substr(0, idx);
465 			starred[hdr] = true;
466 		}
467 		if (allHeaders[hdr] && !used[hdr]) {
468 			headers.push(hdr);
469 			used[hdr] = true;
470 		}
471 	}
472 
473 	// add columns this account doesn't know about
474 	for (var j = 0; j < headerList.length; j++) {
475 		var hdr = headerList[j];
476 		if (!used[hdr]) {
477 			// if account field, add it but initially invisible
478 			if (hdr == ZmId.FLD_ACCOUNT) {
479 				starred[ZmItem.F_ACCOUNT] = true;
480 			}
481 			if (hdr == ZmId.FLD_SELECTION) {
482 				//re-add selection checkbox at the beginning (no idea why the rest is added one before last item, but not gonna change it for now
483 				headers.unshift(hdr); //unshift adds item at the beginning
484 			}
485 			else {
486 				headers.splice(headers.length - 1, 0, hdr);
487 			}
488 		}
489 	}
490 
491 	// rebuild the list, preserve invisibility
492 	var list = AjxUtil.map(headers, function(hdr) {
493 		return starred[hdr] ? hdr + "*" : hdr; });
494 
495 	// save implicit pref with newly added column
496 	appCtxt.set(ZmSetting.LIST_VIEW_COLUMNS, list.join(ZmListView.COL_JOIN), this.view);
497 	return list;
498 };

907 /**
908  * Returns a list of the largest subset of the given participants that will fit within the
909  * given width. The participants are assumed to be ordered oldest to most recent. We return
910  * as many of the most recent as possible.
911  *
912  * @private
913  * @param {array}		participants		list of AjxEmailAddress
914  * @param {ZmMailItem}	item				item that contains the participants
915  * @param {int}			width				available space in pixels
916  * 
917  * @return list of participant objects with 'name' and 'index' fields
918  */
919 ZmMailListView.prototype._fitParticipants =
920 function(participants, item, availWidth) {
921 
922 	availWidth -= 15;	// safety margin
923 
924 	var sepWidth = AjxStringUtil.getWidth(AjxStringUtil.LIST_SEP, item.isUnread);
925 	var ellWidth = AjxStringUtil.getWidth(AjxStringUtil.ELLIPSIS, item.isUnread);
926 
927 	// first see if we can fit everyone with their full names
928 	var list = [];
929 	var pLen = Math.min(20, participants.length);
930 	var width = 0;
931 	for (var i = 0; i < pLen; i++) {
932 		var p = participants[i];
933 		var field = p.name || p.address || p.company || "";
934 		width += AjxStringUtil.getWidth(AjxStringUtil.htmlEncode(field), item.isUnread);
935 		list.push({name:field, index:i});
936 	}
937 	width += (pLen - 1) * sepWidth;
938 	if (width < availWidth) {
939 		return list;
940 	}
941 
942 	// now try with display (first) names; fit as many of the most recent as we can
943 	list = [];
944 	for (var i = 0; i < pLen; i++) {
945 		var p = participants[i];
946 		var field = p.dispName || p.address || p.company || "";
947 		list.push({name:field, index:i});
948 	}
949 	while (list.length) {
950 		var width = 0;
951 		// total the width of the names
952 		for (var i = 0; i < list.length; i++) {
953 			width += AjxStringUtil.getWidth(AjxStringUtil.htmlEncode(list[i].name), item.isUnread);
954 		}
955 		// add the width of the separators
956 		width += (list.length - 1) * sepWidth;
957 		// add the width of the ellipsis if we've dropped anyone
958 		if (list.length < pLen) {
959 			width += ellWidth;
960 		}
961 		if (width < availWidth) {
962 			return list;
963 		} else {
964 			list.shift();
965 		}
966 	}
967 
968 	// not enough room for even one participant, just return the last one
969 	var p = participants[pLen - 1];
970 	var field = p.dispName || p.address || p.company || "";
971 	return [{name:field, index:pLen - 1}];
972 };
