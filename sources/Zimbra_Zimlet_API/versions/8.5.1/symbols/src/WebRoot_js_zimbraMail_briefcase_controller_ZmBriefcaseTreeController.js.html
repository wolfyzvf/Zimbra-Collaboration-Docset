 28 /**
 29  * Creates the briefcase tree controller.
 30  * @class
 31  * This class is a controller for the tree view used by the briefcase application.
 32  *
 33  * @param	{constant}	type		the organizer (see {@link ZmOrganizer.BRIEFCASE})
 34  * 
 35  * @author Parag Shah
 36  * 
 37  * @extends		ZmFolderTreeController
 38  */
 39 ZmBriefcaseTreeController = function(type) {
 40 
 41 	ZmFolderTreeController.call(this, (type || ZmOrganizer.BRIEFCASE));
 42 
 43 	this._listeners[ZmOperation.NEW_BRIEFCASE] = new AjxListener(this, this._newListener);
 44 	this._listeners[ZmOperation.SHARE_BRIEFCASE] = new AjxListener(this, this._shareBriefcaseListener);
 45 
 46 	this._eventMgrs = {};
 47     this._app = appCtxt.getApp(ZmApp.BRIEFCASE);
 48 };
 49 
 50 ZmBriefcaseTreeController.prototype = new ZmFolderTreeController;
 51 ZmBriefcaseTreeController.prototype.constructor = ZmBriefcaseTreeController;

 65 ZmBriefcaseTreeController.prototype.resetOperations =
 66 function(actionMenu, type, id) {
 67 
 68 	var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT);
 69      if (actionMenu && id != rootId) {
 70 		var briefcase = appCtxt.getById(id);
 71 		if (!briefcase) { return; }
 72         var nId = ZmOrganizer.normalizeId(id);
 73 		var isRoot = (nId == rootId);
 74 		var isBriefcase = (nId == ZmOrganizer.getSystemId(ZmOrganizer.ID_BRIEFCASE));
 75 		var isTopLevel = (!isRoot && briefcase.parent.id == rootId);
 76 		var isLink = briefcase.link;
 77 		var isLinkOrRemote = isLink || briefcase.isRemote();
 78         var isTrash = (nId == ZmFolder.ID_TRASH);
 79         var isReadOnly = briefcase ? briefcase.isReadOnly() : false;
 80 
 81         var deleteText = ZmMsg.del;
 82 
 83         actionMenu.getOp(ZmOperation.EMPTY_FOLDER).setVisible(isTrash);
 84 
 85         if (isTrash) {
 86             var hasContent = ((briefcase.numTotal > 0) || (briefcase.children && (briefcase.children.size() > 0)));
 87             actionMenu.enableAll(false);
 88             actionMenu.enable(ZmOperation.EMPTY_FOLDER,hasContent);
 89             actionMenu.getOp(ZmOperation.EMPTY_FOLDER).setText(ZmMsg.emptyTrash);            
 90         } else {
 91             actionMenu.enableAll(true);
 92             var showEditMenu = (!isLinkOrRemote || !isReadOnly || (isLink && isTopLevel) || ZmBriefcaseTreeController.__isAllowed(briefcase.parent, ZmShare.PERM_DELETE));
 93             actionMenu.enable(ZmOperation.DELETE_WITHOUT_SHORTCUT, showEditMenu && !isBriefcase);
 94             actionMenu.enable(ZmOperation.EDIT_PROPS, showEditMenu);
 95 
 96 			var menuItem;
 97             menuItem = actionMenu.getMenuItem(ZmOperation.NEW_BRIEFCASE);
 98             menuItem.setText(ZmMsg.newFolder);
 99             menuItem.setImage("NewFolder");
100             menuItem.setEnabled((!isLinkOrRemote || ZmBriefcaseTreeController.__isAllowed(briefcase, ZmShare.PERM_CREATE_SUBDIR) || briefcase.isAdmin() || ZmShare.getRoleFromPerm(briefcase.perm) == ZmShare.ROLE_MANAGER));
101 
102             if (appCtxt.get(ZmSetting.SHARING_ENABLED)) {
103                 isBriefcase = (!isRoot && briefcase.parent.id == rootId) || type==ZmOrganizer.BRIEFCASE;
104                 menuItem = actionMenu.getMenuItem(ZmOperation.SHARE_BRIEFCASE);
105                 menuItem.setText(ZmMsg.shareFolder);
106                 menuItem.setImage(isBriefcase ? "SharedMailFolder" : "Section");
107                 var isShareVisible = (!isLinkOrRemote || briefcase.isAdmin());
108                 if (appCtxt.isOffline) {
109                     var acct = briefcase.getAccount();
110                     isShareVisible = !acct.isMain && acct.isZimbraAccount;
111                 }
112                 menuItem.setEnabled(isShareVisible);
113             }
114         }
115         var op = actionMenu.getOp(ZmOperation.DELETE_WITHOUT_SHORTCUT);
116         if (op) {
117             op.setText(deleteText);
118         }
119 		this._enableRecoverDeleted(actionMenu, isTrash);
120 
121         // we always enable sharing in case we're in multi-mbox mode
122         this._resetButtonPerSetting(actionMenu, ZmOperation.SHARE_BRIEFCASE, appCtxt.get(ZmSetting.SHARING_ENABLED));
123 
124 	}
125 
126 };

138 ZmBriefcaseTreeController.__isAllowed =
139 function(organizer, perm) {
140 	var allowed = true;
141 	if (organizer.link || organizer.isRemote()) {
142 		allowed = false; // change assumption to not allowed
143 
144 		// REVISIT: bug 10801
145 		var share = organizer.getMainShare();
146 		if (share && !share.isPermRestricted(perm)) {
147 			allowed = share.isPermAllowed(perm);
148 		}
149 	}
150 	return allowed;
151 };

153 // Returns a list of desired header action menu operations
154 ZmBriefcaseTreeController.prototype._getHeaderActionMenuOps =
155 function() {
156     var ops = [];
157     if (!appCtxt.isExternalAccount()) {
158         ops.push(ZmOperation.NEW_BRIEFCASE);
159     }
160     ops.push(ZmOperation.EXPAND_ALL);
161 	if (!appCtxt.isExternalAccount()) {
162 		ops.push(ZmOperation.FIND_SHARES);
163 	}
164 	return ops;
165 };
166 
167 // Returns a list of desired action menu operations
168 ZmBriefcaseTreeController.prototype._getActionMenuOps =
169 function() {
170 	var ops = [ZmOperation.NEW_BRIEFCASE];
171 	if (appCtxt.get(ZmSetting.SHARING_ENABLED)) {
172 		ops.push(ZmOperation.SHARE_BRIEFCASE);
173 	}
174 	ops.push(ZmOperation.DELETE_WITHOUT_SHORTCUT, ZmOperation.EDIT_PROPS, ZmOperation.EMPTY_FOLDER, ZmOperation.RECOVER_DELETED_ITEMS);
175 	return ops;
176 };

195 // Method that is run when a tree item is left-clicked
196 ZmBriefcaseTreeController.prototype._itemClicked =
197 function(folder) {
198 	appCtxt.getApp(ZmApp.BRIEFCASE).search({
199         folderId:folder.id,
200         callback: new AjxCallback(this, this._handleSearchResponse, [folder])
201     });
202 };

206 ZmBriefcaseTreeController.prototype._shareBriefcaseListener =
207 function(ev) {
208 	this._pendingActionData = this._getActionedOrganizer(ev);
209 
210 	var briefcase = this._pendingActionData;
211 	var share = null;
212 
213 	var sharePropsDialog = appCtxt.getSharePropsDialog();
214 	sharePropsDialog.popup(ZmSharePropsDialog.NEW, briefcase, share);
215 };

217 ZmBriefcaseTreeController.prototype._notifyListeners =
218 function(overviewId, type, items, detail, srcEv, destEv) {
219 	if (this._eventMgrs[overviewId] && this._eventMgrs[overviewId].isListenerRegistered(type)) {
220 		if (srcEv) DwtUiEvent.copy(destEv, srcEv);
221 		destEv.items = items;
222 		if (items.length == 1) destEv.item = items[0];
223 		destEv.detail = detail;
224 		this._eventMgrs[overviewId].notifyListeners(type, destEv);
225 	}
226 };

233 ZmBriefcaseTreeController.prototype._getItems =
234 function(overviewId) {
235 	var treeView = this.getTreeView(overviewId);
236 	if (treeView) {
237 		var rootId = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT);
238 		var root = treeView.getTreeItemById(rootId);
239 		if (root) {
240 			return root.getItems();
241 		}
242 	}
243 	return [];  
244 };

246 ZmBriefcaseTreeController.prototype.show =
247 function(params) {
248 	params.include = {};
249 	params.include[ZmFolder.ID_TRASH] = true;
250     params.showUnread = false;
251     var treeView = ZmFolderTreeController.prototype.show.call(this, params);
252 
253     treeView._controller = this;
254     // Finder to BriefcaseTreeView drag and drop
255     this._initDragAndDrop(treeView);
256 
257     return treeView;
258 };

269 ZmBriefcaseTreeController.prototype._handleSearchResponse =
270 function(folder, result) {
271     // bug fix #49568 - Trash is special when in Briefcase app since it
272     // is a FOLDER type in BRIEFCASE tree. So reset selection if clicked
273     if (folder.nId == ZmFolder.ID_TRASH) {
274         this._treeView[this._app.getOverviewId()].setSelected(ZmFolder.ID_TRASH, true);
275     }
276 };
