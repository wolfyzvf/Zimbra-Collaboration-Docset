 27 /**
 28  * Creates the task folder.
 29  * @class
 30  * This class represents a task folder.
 31  * 
 32  * @author Parag Shah
 33  *
 34  * @param	{Hash}	params		a hash of parameters
 35  * @param {int}	params.id			the numeric ID
 36  * @param {String}	params.name		the name
 37  * @param {ZmOrganizer}	params.parent		the parent organizer
 38  * @param {ZmTree}	params.tree		the tree model that contains this organizer
 39  * @param {String}	params.color	the color
 40  * @param {String}	params.url		the URL for this organizer's feed
 41  * @param {String}	params.owner	the owner
 42  * @param {String}	params.zid		the Zimbra id of owner, if remote share
 43  * @param {String}	params.rid		the remote id of organizer, if remote share
 44  * @param {String}	params.restUrl	[the REST URL of this organizer.
 45  * 
 46  * @extends		ZmFolder
 47  */
 48 ZmTaskFolder = function(params) {
 49 	params.type = ZmOrganizer.TASKS;
 50 	ZmFolder.call(this, params);
 51 }

 68 ZmTaskFolder.prototype.getIcon =
 69 function() {
 70 	if (this.id == ZmFolder.ID_ROOT)	{ return null; }
 71 	if (this.link)						{ return "SharedTaskList"; }
 72 	return "TaskList";
 73 };

 81 /**
 82  * Sets the free/busy.
 83  * 
 84  * @param	{AjxCallback}		callback		the callback
 85  * @param	{AjxCallback}		errorCallback		the error callback
 86  * @param	{Boolean}		exclude		checks to exclose free busy
 87  */
 88 ZmTaskFolder.prototype.setFreeBusy =
 89 function(exclude, callback, errorCallback) {
 90 	if (this.excludeFreeBusy == exclude) return;
 91 	// NOTE: Don't need to store the value since the response will
 92 	//       report that the object was modified.
 93 	this._organizerAction({action: "fb", attrs: {excludeFreeBusy: exclude ? "1" : "0"}, callback: callback, errorCallback: errorCallback});
 94 };
 95 
 96 ZmTaskFolder.prototype.mayContain =
 97 function(what) {
 98 	if (!what) return true;
 99 
100 	var invalid = false;
101 
102 	if (this.id == ZmFolder.ID_ROOT) {
103 		// cannot drag anything onto root folder
104 		invalid = true;
105 	} else if (this.link) {
106 		// cannot drop anything onto a read-only task folder
107 		invalid = this.isReadOnly();
108 	}
109 
110 	if (!invalid) {
111 		// An item or an array of items is being moved
112 		var items = (what instanceof Array) ? what : [what];
113 		var item = items[0];
114 
115 		if (item.type != ZmItem.TASK) {
116 			// only tasks are valid for task folders
117 			invalid = true;
118 		} else {
119 			// can't move items to folder they're already in; we're okay if
120 			// we have one item from another folder
121 			if (!invalid && item.folderId) {
122 				invalid = true;
123 				for (var i = 0; i < items.length; i++) {
124 					var item = items[i];
125 					var folder = appCtxt.getById(item.folderId);
126 					if (folder != this) {
127 						invalid = false;
128 						break;
129 					}
130 				}
131 			}
132 		}
133 	}
134 
135 	return !invalid;
136 };

141 ZmTaskFolder.prototype.notifyCreate =
142 function(obj) {
143 	var t = ZmFolderTree.createFromJs(this, obj, this.tree);
144 	var i = ZmOrganizer.getSortIndex(t, ZmTaskFolder.sortCompare);
145 	this.children.add(t, i);
146 	t._notify(ZmEvent.E_CREATE);
147 };
148 
149 ZmTaskFolder.prototype.notifyModify =
150 function(obj) {
151 	ZmFolder.prototype.notifyModify.call(this, obj);
152 
153 	if (obj.f != null && !obj._isRemote) {
154 		this._parseFlags(obj.f);
155 		// TODO: Should a F_EXCLUDE_FB property be added to ZmOrganizer?
156 		//       It doesn't make sense to require the base class to know about
157 		//       all the possible fields in sub-classes. So I'm just using the
158 		//       modified property name as the key.
159 		var fields = {};
160 		fields["excludeFreeBusy"] = true;
161 		this._notify(ZmEvent.E_MODIFY, {fields: fields});
162 	}
163 };

178 /**
179  * Comparison function for folders. Intended for use on a list of user folders
180  * through a call to <code>Array.sort()</code>.
181  * 
182  * @param	{ZmTaskFolder}	calA		item A
183  * @param	{ZmTaskFolder}	calB		item B
184  * 
185  * @return	{int}	0 if the folders match
186  */
187 ZmTaskFolder.sortCompare =
188 function(calA, calB) {
189 	var check = ZmOrganizer.checkSortArgs(calA, calB);
190 	if (check != null) return check;
191 
192 	// links appear after personal calendars
193 	if (calA.link != calB.link) {
194 		return calA.link ? 1 : -1;
195 	}
196 
197 	// sort by calendar name
198 	var calAName = calA.name.toLowerCase();
199 	var calBName = calB.name.toLowerCase();
200 	if (calAName < calBName) return -1;
201 	if (calAName > calBName) return 1;
202 	return 0;
203 };
