 22 /**
 23  * Creates a password update dialog.
 24  * @class
 25  * This class represents a password update dialog.
 26  *
 27  * @param	{DwtComposite}	parent		the parent
 28  * @param	{String}	className		the class name
 29  *
 30  * @extends		ZmDialog
 31  */
 32 ZmPasswordUpdateDialog = function(parent, className) {
 33 
 34 	ZmDialog.call(this, {parent:parent, className:className, title:ZmMsg.changePassword, id:"PasswdChangeDialog"});
 35 	this._setNameField(this._nameFieldId);
 36     this._createControls();
 37 };
 38 
 39 ZmPasswordUpdateDialog.prototype = new ZmDialog;
 40 ZmPasswordUpdateDialog.prototype.constructor = ZmPasswordUpdateDialog;
 41 
 42 ZmPasswordUpdateDialog.prototype.toString =
 43 function() {
 44 	return "ZmPasswordUpdateDialog";
 45 };
 46 
 47 /**
 48  * Pops-up the dialog.
 49  */
 50 ZmPasswordUpdateDialog.prototype.popup =
 51 function(acct) {
 52    	ZmDialog.prototype.popup.call(this);
 53     this.acct = acct;
 54     var desc  = document.getElementById(this._htmlElId + "_desc");
 55     this._toggleOKButton(false);
 56     desc.innerHTML = AjxMessageFormat.format(ZmMsg.offlinePasswordUpdate, this.acct.name);
 57     var acctTd  = document.getElementById(this._htmlElId + "_acct");
 58     acctTd.innerHTML = this.acct.name;
 59 	this._nameField.value = "";
 60 };
 61 
 62 ZmPasswordUpdateDialog.prototype._createControls =
 63 function() {
 64     this.setTitle(ZmMsg.offlineAccountAuth);
 65     this._toggleOKButton(false);
 66     var cancelBtn = this.getButton(DwtDialog.CANCEL_BUTTON);
 67     cancelBtn.setText(ZmMsg.dismiss);
 68     var okBtn = this.getButton(DwtDialog.OK_BUTTON);
 69 	okBtn.setText(ZmMsg.save);
 70     this._nameField._dlgEl = this._htmlElId;
 71     Dwt.setHandler(this._nameField, DwtEvent.ONKEYUP, this._handleKeyUp);
 72 
 73 };
 74 
 75 
 76 ZmPasswordUpdateDialog.prototype._contentHtml =
 77 function() {
 78 	this._nameFieldId = this._htmlElId + "_name";
 79 	var subs = {id:this._htmlElId, labelAcct:ZmMsg.account, labelPasswd:ZmMsg.password};
 80 	return AjxTemplate.expand("share.Dialogs#ZmPasswordUpdateDialog", subs);
 81 };
 82 
 83 ZmPasswordUpdateDialog.prototype._okButtonListener =
 84 function(ev) {
 85     var pwd = AjxStringUtil.trim(this._nameField.value);
 86     if (pwd && pwd.length > 0 ) {
 87         var soapDoc = AjxSoapDoc.create("ChangePasswordRequest", "urn:zimbraOffline");
 88         soapDoc.setMethodAttribute("id", this.acct.id);
 89         soapDoc.set("password", pwd);
 90 
 91         appCtxt.getAppController().sendRequest({
 92             soapDoc:soapDoc,
 93             asyncMode:true,
 94             noBusyOverlay:true,
 95             callback: new AjxCallback(this, this._handlePasswordUpdateResult),
 96             accountName:this.name
 97         });
 98     }
 99 };
100 
101 
102 /**
103  *  Updates password for specified account
104  *
105  */
106 
107 ZmPasswordUpdateDialog.prototype._handlePasswordUpdateResult =
108 function(result) {
109     var resp = result.getResponse();
110     resp = resp.ChangePasswordResponse;
111     if (resp && resp.status == "success") {
112         this.popdown();
113         var msg = AjxMessageFormat.format(ZmMsg.offlinePasswordUpdateSuccess, this.acct.name);
114         appCtxt.setStatusMsg(msg, ZmStatusView.LEVEL_INFO);
115     } else {
116         appCtxt.setStatusMsg(ZmMsg.offlinePasswordUpdateFailure, ZmStatusView.LEVEL_WARNING);
117         this._nameField.value = "";
118         this._toggleOKButton(false);
119     }
120 };
121 
122 ZmPasswordUpdateDialog.prototype._enterListener =
123 function(ev) {
124 	var pwd = AjxStringUtil.trim(this._nameField.value);
125     if (pwd && pwd.length > 0 ) {
126 	    this._okButtonListener();
127 	}
128 };
129 
130 ZmPasswordUpdateDialog.prototype._handleKeyUp =
131 function(ev) {
132 
133     var key = DwtKeyEvent.getCharCode(ev);
134 	if (key == 9) {
135 		return;
136 	}
137     var el = DwtUiEvent.getTarget(ev);
138     var val = el && el.value;
139     var dlgEl  = el && el._dlgEl && DwtControl.ALL_BY_ID[el._dlgEl];
140     dlgEl._toggleOKButton(val.length > 0);
141 };
142 
143 ZmPasswordUpdateDialog.prototype._toggleOKButton =
144 function(enable) {
145     var okBtn = this.getButton(DwtDialog.OK_BUTTON);
146 	okBtn.setEnabled(enable);
147 };
