 22 /**
 23  * @class
 24  * This is a folder choosing widget designed to be displayed as a dropdown hanging off a menu. Other than
 25  * that, it works mostly like ZmChooseFolderDialog. Instead of a New button, it has a "New Folder" menu item
 26  * at the bottom. Items are moved when a menu item is clicked, instead of the OK button in the dialog.
 27  * 
 28  * The implementation mostly relies on calling ZmChooseFolderDialog and ZmDialog methods, since those were
 29  * already written. A cleaner implementation would probably include a base widget that this and the dialog
 30  * would use.
 31  *
 32  * @author Eran Yarkon
 33  *
 34  * @param {hash}		params			a hash of parameters
 35  * @param {DwtComposite}      params.parent			the parent widget
 36  * @param {string}      params.className			the CSS class
 37  * @param {constant}      params.posStyle			the positioning style (see {@link Dwt})
 38  *
 39  * @extends		DwtComposite
 40  */
 41 ZmFolderChooser = function(params) {
 42 	if (arguments.length == 0) { return; }
 43 	params.className = params.className || "ZmFolderChooser";
 44 	DwtComposite.call(this, params);
 45 
 46 	this._overview = {};
 47 	this._opc = appCtxt.getOverviewController();
 48 	this._treeView = {};
 49 	this._folderTreeDivId = this._htmlElId + "_folderTreeDivId";
 50 
 51 	this._uuid = Dwt.getNextId();
 52 
 53 	AjxDispatcher.require("Extras");	// ZmChooseFolderDialog
 54 	this._changeListener = ZmChooseFolderDialog.prototype._folderTreeChangeListener.bind(this);
 55 	this._treeViewListener = this._treeViewSelectionListener.bind(this);
 56 
 57 	var moveMenu = params.parent;
 58 	moveMenu._addItem(this, params.index); //this is what DwtMenuItem does. Allows this item to be in the menu items table - better for layout purposes such as consistent widths
 59 	this._moveOnFolderCreate = true;
 60 	
 61 	if (!params.hideNewButton) {
 62 		//add separator menu item on the move menu (the parent)
 63 		new DwtMenuItem({parent:moveMenu, style:DwtMenuItem.SEPARATOR_STYLE});
 64 	
 65 		// add New button
 66 		var newFolderItem = this._newButton = new DwtMenuItem({parent:moveMenu, id: moveMenu.getHTMLElId() + "|NEWFOLDER"});
 67 		var appName = appCtxt.getCurrentAppName();	
 68 		var defaultApp = ZmApp.MAIL;
 69 		var newTextKey = ZmFolderChooser.NEW_ORG_KEY[appName] || ZmFolderChooser.NEW_ORG_KEY[defaultApp];
 70 		var newImage = ZmFolderChooser.NEW_ORG_ICON[appName] || ZmFolderChooser.NEW_ORG_ICON[defaultApp];
 71 		var newShortcut = ZmFolderChooser.NEW_ORG_SHORTCUT[appName];
 72 		newFolderItem.setText(ZmMsg[newTextKey]);
 73 		newFolderItem.setImage(newImage);
 74 		if (newShortcut) {
 75 			newFolderItem.setShortcut(appCtxt.getShortcutHint(this._keyMap, newShortcut));
 76 		}
 77 		newFolderItem.addSelectionListener(this._showNewDialog.bind(this));
 78 	}
 79 
 80 	this._init();
 81 };

207 ZmFolderChooser.prototype._treeViewSelectionListener =
208 function(ev) {
209 	if (ev.detail != DwtTree.ITEM_SELECTED) {
210 		return;
211 	}
212 	//set in DwtTree.prototype._itemClicked and DwtTree.prototype.setSelection (which is called by DwtTreeItem.prototype.handleKeyAction)
213 	if (!ev.clicked && !ev.enter) {
214 		return;
215 	}
216 
217 	//I kept this logic from ZmChooseFolderDialog.prototype._treeViewSelectionListener. Not sure what it means exactly
218 	if (this._getOverview() instanceof ZmAccountOverviewContainer) {
219 		if (ev.item instanceof DwtHeaderTreeItem) {
220 			return;
221 		}
222 
223 		var oc = this._opc.getOverviewContainer(this._curOverviewId);
224 		var overview = oc.getOverview(ev.item.getData(ZmTreeView.KEY_ID));
225 		oc.deselectAll(overview);
226 	}
227 
228 	var organizer = ev.item && ev.item.getData(Dwt.KEY_OBJECT);
229 	if (organizer.id == ZmFolder.ID_LOAD_FOLDERS) {
230 		return; // user clicked on "Show More Folders", it's not a real selection, it just expanded more folders.
231 	}
232 	var value = organizer ? organizer.getName(null, null, true) : ev.item.getText();
233 	this._lastVal = value.toLowerCase();
234 	this._doSelection();
235 };

237 /**
238  * copied mostly from ZmChooseFolderDialog.prototype._okButtonListener  
239  * @param tgtFolder
240  */
241 ZmFolderChooser.prototype._doSelection =
242 function(tgtFolder) {
243     tgtFolder = tgtFolder || this._getOverview().getSelected();
244     if  (!tgtFolder) {
245         tgtFolder = appCtxt.getById(this._selected);
246     }
247 	var folderList = (tgtFolder && (!(tgtFolder instanceof Array)))
248 		? [tgtFolder] : tgtFolder;
249 
250 	var msg = (!folderList || (folderList && folderList.length == 0))
251 		? ZmMsg.noTargetFolder : null;
252 
253 	//todo - what is that? can you move stuff to multiple targets?  annotation on ZmChooseFolderDialog show it might be for filters on multiple folders. obviously in that case we can't have a drop down. we might have to keep that folder dialog
254 	
255 	// check for valid target
256 	if (!msg && this._data) {
257 		for (var i = 0; i < folderList.length; i++) {
258 			var folder = folderList[i];
259 			if (folder.mayContain && !folder.mayContain(this._data, null, this._acceptFolderMatch)) {
260 				if (this._data instanceof ZmFolder) {
261 					msg = ZmMsg.badTargetFolder;
262 				}
263 				else {
264 					var items = AjxUtil.toArray(this._data);
265 					for (var i = 0; i < items.length; i++) {
266 						var item = items[i];
267 						if (!item) {
268 							continue;
269 						}
270 						if (item.isDraft && (folder.nId != ZmFolder.ID_TRASH && folder.nId != ZmFolder.ID_DRAFTS && folder.rid != ZmFolder.ID_DRAFTS)) {
271 							// can move drafts into Trash or Drafts
272 							msg = ZmMsg.badTargetFolderForDraftItem;
273 							break;
274 						}
275 						else if ((folder.nId == ZmFolder.ID_DRAFTS || folder.rid == ZmFolder.ID_DRAFTS) && !item.isDraft)	{
276 							// only drafts can be moved into Drafts
277 							msg = ZmMsg.badItemForDraftsFolder;
278 							break;
279 						}
280 					}
281 					if (!msg) {
282 						msg = ZmMsg.badTargetFolderItems;
283 					}
284 				}
285 				break;
286 			}
287 		}
288 	}
289 
290 	if (msg) {
291 		ZmDialog.prototype._showError.call(this, msg);
292 		return;
293 	}
294 	if (this._selectionCallback) {
295 		this._selectionCallback(tgtFolder);
296 	}
297 };

327 ZmFolderChooser.prototype._showNewDialog =
328 function() {
329 	var item = this._getOverview().getSelected(true);
330 	var newType = (item && item.type) || this._treeIds[0];
331 	var ftc = this._opc.getTreeController(newType);
332 	var dialog = ftc._getNewDialog();
333 	dialog.reset();
334 	dialog.registerCallback(DwtDialog.OK_BUTTON, ZmChooseFolderDialog.prototype._newCallback, this, [ftc, dialog]);
335 	this.parent.popdown(); //pop it down so it doenst pop down when user clicks on the "new" dialog, confusing them. this is also consistent with the tag menu "new".
336 	dialog.popup();
337 };
