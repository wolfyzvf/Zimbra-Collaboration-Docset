 36 ZmDocletMgr.prototype._uploadSaveDocs2 =
 37 function(files, status, guids, name, content, ct, enableDesc, desc) {
 38     // create document wrappers
 39     var soapDoc = AjxSoapDoc.create("BatchRequest", "urn:zimbra", null);
 40     soapDoc.setMethodAttribute("onerror", "continue");
 41     for (var i = 0; i < files.length; i++) {
 42         var file = files[i];
 43         if (file.done) continue;
 44 
 45         var saveDocNode = soapDoc.set("SaveDocumentRequest", null, null, "urn:zimbraMail");
 46         saveDocNode.setAttribute("requestId", i);
 47 
 48         var docNode = soapDoc.set("doc", null, saveDocNode);
 49         if (file.id) {
 50             docNode.setAttribute("id", file.id);
 51             docNode.setAttribute("ver", file.version);
 52         }
 53         else {
 54             docNode.setAttribute("l", file.folderId);
 55         }
 56 
 57         if(ct){
 58             docNode.setAttribute("ct", ct);
 59         }
 60 
 61         if(file.guid) {
 62             var uploadNode = soapDoc.set("upload", null, docNode);
 63             uploadNode.setAttribute("id", file.guid);
 64         }
 65 
 66         if(name!=null && content!=null) {
 67             var contentNode = soapDoc.set("content", content, docNode);
 68             docNode.setAttribute("name", name);
 69         }
 70 
 71         docNode.setAttribute("descEnabled", enableDesc ? "true" : "false");
 72         if(desc){
 73             docNode.setAttribute("desc", desc);
 74         }
 75     }
 76 
 77     var args = [ files, status, guids ];
 78     var callback = new AjxCallback(this, this._uploadSaveDocsResponse, args);
 79     var params = {
 80         soapDoc:soapDoc,
 81         asyncMode:true,
 82         callback:callback
 83     };
 84 
 85     var parentAppCtxt = window.opener && window.opener.appCtxt;
 86 	if (parentAppCtxt) {
 87         var acct = parentAppCtxt.getActiveAccount();
 88         params.accountName = (acct && acct.id != ZmAccountList.DEFAULT_ID) ? acct.name : null;
 89     }
 90     this.sendRequest(params);
 91 };

110 ZmDocletMgr.prototype._uploadSaveDocsResponse =
111 function(files, status, guids, response) {
112     var resp = response && response._data && response._data.BatchResponse;
113 
114     var folderIds = [];
115 
116     // mark successful uploads
117     if (resp && resp.SaveDocumentResponse) {
118         for (var i = 0; i < resp.SaveDocumentResponse.length; i++) {
119             var saveDocResp = resp.SaveDocumentResponse[i];
120             files[saveDocResp.requestId].done = true;
121             files[saveDocResp.requestId].rest = saveDocResp.doc[0].rest;
122             files[saveDocResp.requestId].ver = saveDocResp.doc[0].ver;
123             files[saveDocResp.requestId].id = saveDocResp.doc[0].id;
124             if(files[saveDocResp.requestId].folderId) {
125                 folderIds.push(files[saveDocResp.requestId].folderId);
126             }
127         }
128     }
129 
130     // check for conflicts
131     var conflicts = null;
132     if (resp && resp.Fault) {
133         var errors = [];
134         conflicts = [];
135         for (var i = 0; i < resp.Fault.length; i++) {
136             var fault = resp.Fault[i];
137             var error = fault.Detail.Error;
138             var code = error.Code;
139             var attrs = error.a;
140             if (code == ZmCsfeException.MAIL_ALREADY_EXISTS ||
141                 code == ZmCsfeException.MODIFY_CONFLICT) {
142                 var file = files[fault.requestId];
143                 for (var p in attrs) {
144                     var attr = attrs[p];
145                     switch (attr.n) {
146                         case "id": { file.id = attr._content; break; }
147                         case "ver": { file.version = attr._content; break; }
148                         case "rest": { file.rest = attr._content; break; }
149 
150                     }
151                 }
152                 conflicts.push(file);
153             }
154             else {
155                 DBG.println("Unknown error occurred: "+code);
156                 errors[fault.requestId] = fault;
157             }
158         }
159         // TODO: What to do about other errors?
160     }
161 
162 	// poke parent window to get notifications
163 	var parentAppCtxt = window.opener && window.opener.appCtxt;
164 	var parentAppCtlr = parentAppCtxt && parentAppCtxt.getAppController();
165 	if (parentAppCtlr && !parentAppCtlr.getInstantNotify()) {
166 		parentAppCtlr.sendNoOp();
167 	}
168 
169     if (this._saveCallback) {
170         //Pass on the conflicts to callback
171         this._saveCallback.run(files, conflicts);
172     }
173 };

196 ZmDocletMgr.prototype.fetchDocumentContent =
197 function(item) {
198     var restURL = item.rest;
199     var urlParts = AjxStringUtil.parseURL(restURL);
200     if(urlParts && urlParts.path) {
201         var result = AjxRpc.invoke("", urlParts.path + "?fmt=native&ver=" + item.version, {}, null, true);
202         var docContent = "";
203         if(result && result.success) {
204             docContent = this._pendingContent = result.text;
205         }
206         return docContent;
207     }
208     return "";
209 };

212 ZmDocletMgr.prototype.getItemInfo =
213 function(params)
214 {
215     var soapDoc = AjxSoapDoc.create("GetItemRequest", "urn:zimbraMail");
216     var folderNode = soapDoc.set("item");
217 
218     if(params.path){
219         folderNode.setAttribute("path", params.path);
220     }else if(params.folderId && params.id){ //bug:19658
221         folderNode.setAttribute("l", params.folderId);
222         folderNode.setAttribute("id", params.id);
223     }else if(params.folderId && params.name){
224         folderNode.setAttribute("l", params.folderId);
225         folderNode.setAttribute("name", params.name);
226     }else if(params.id){
227         folderNode.setAttribute("id", params.id);
228     }
229 
230     var args = [];
231     var asyncMode = (params.callback?true:false);
232 
233     var handleResponse = null;
234     if(asyncMode){
235         handleResponse = new AjxCallback(this, this.handleGetItemResponse,[params]);
236     }
237 
238     var parentAppCtxt = window.opener && window.opener.appCtxt;
239     if (parentAppCtxt && parentAppCtxt.multiAccounts) {
240         var acct = parentAppCtxt.getActiveAccount();
241         params.accountName = acct && acct.name;
242     }
243 
244     var reqParams = {
245         soapDoc: soapDoc,
246         asyncMode: asyncMode,
247         callback: handleResponse,
248         accountName: params.accountName
249     };
250 
251     var response = this.sendRequest(reqParams);
252 
253     if(!asyncMode && response){
254         var item = this.handleGetItemResponse(params,response.GetItemResponse);
255         return item;
256     }
257 
258     return null;
259 };

261 ZmDocletMgr.prototype.handleGetItemResponse =
262 function(params,response)
263 {
264 
265     var path = params.path;
266     var callback = params.callback;
267 
268     var getItemResponse = response;
269     if(response && response._data){
270         getItemResponse = response && response._data && response._data.GetItemResponse;
271     }
272 
273     var docResp = getItemResponse && getItemResponse.doc && getItemResponse.doc[0];
274 
275     var item = null;
276 
277     if(docResp){
278         item = new ZmItem();
279         var data = docResp;
280         if (data.id) item.id = data.id;        
281         if (data.rest) item.rest = data.rest;
282         if (data.l) item.folderId = data.l;
283         if (data.name) item.name = data.name;
284         if (data.cr) item.creator = data.cr;
285         if (data.d) item.createDate = new Date(Number(data.d));
286         if (data.md) item.modifyDate = new Date(Number(data.md));
287         if (data.leb) item.modifier = data.leb;
288         if (data.s) item.size = Number(data.s);
289         if (data.ver) item.version = Number(data.ver);
290         if (data.ct) item.contentType = data.ct.split(";")[0];
291         item.folderId = docResp.l || ZmOrganizer.ID_BRIEFCASE;
292         item.locked = false;
293         if (data.loid)    {
294             item.locked = true;
295             item.lockId = data.loid;
296             item.lockUser = data.loe;
297             item.lockTime = new Date(Number(data.lt));
298         }
299         item.descEnabled = data.descEnabled;
300     }
301 
302     if(callback){
303         callback.run(item);
304     }
305 
306     return item;
307 };

327 ZmDocletMgr.prototype.checkInvalidDocName = function(fileName) {
328 
329     var message;
330     fileName = fileName.replace(/^\s+/,"").replace(/\s+$/,"");
331 
332     if(fileName == ""){
333         message = ZmMsg.emptyDocName;
334     }else if (!ZmOrganizer.VALID_NAME_RE.test(fileName) || ZmAppCtxt.INVALID_NAME_CHARS_RE.test(fileName)) {
335         //Bug fix # 79986 - < > , ? | / \ * : are invalid filenames
336         message = AjxMessageFormat.format(ZmMsg.errorInvalidName, AjxStringUtil.htmlEncode(fileName));
337     } else if ( fileName.length > ZmOrganizer.MAX_NAME_LENGTH){
338         message = AjxMessageFormat.format(ZmMsg.nameTooLong, ZmOrganizer.MAX_NAME_LENGTH);
339     }
340 
341     return message;
342 };

344 ZmDocletMgr.prototype.unlock =
345 function(item, callback, errorCallback, accountName){
346 
347     var json = {
348 		ItemActionRequest: {
349 			_jsns: "urn:zimbraMail",
350 			action: {
351 				id:	item.id,
352 				op:	"unlock"
353 			}
354 		}
355 	};
356 
357 	var params = {
358 		jsonObj:		json,
359 		asyncMode:		Boolean(callback),
360 		callback:		callback,
361 		errorCallback:	errorCallback,
362 		accountName:	accountName
363 	};
364 	return this.sendRequest(params);
365
