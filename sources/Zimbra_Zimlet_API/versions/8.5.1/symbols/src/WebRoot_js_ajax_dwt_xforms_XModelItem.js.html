 35 XModelItemFactory.createItem = function (attributes, parentItem, xmodel) {
 36 	// assign a modelItem to the item
 37 	var type = attributes.type;
 38 	var constructor = this.getItemTypeConstructor(type || _UNTYPED_);
 39 
 40 	var item = new constructor();
 41 	item._setAttributes(attributes);
 42 	if (item.id != null && item.ref == null) item.ref = item.id;
 43 
 44 	// idPath is mostly used for debugging...
 45 	var idPath = this.getIdPath(attributes, parentItem);
 46 	item.__idPath = idPath;
 47 
 48 	item.__xmodel = xmodel;
 49 	item.__parentItem = parentItem;
 50 
 51 //DBG.println("XModelItemFactory.createItem(", attributes.id, ") idPath='", idPath, "' type='", item.type,"'");
 52 	item.initModelItem();
 53 
 54 	// add the item to its model's index
 55 	xmodel.indexItem(item, idPath);
 56 
 57 	return item;
 58 }
 59 
 60 XModelItemFactory.getIdPath = function (attributes, parentItem) {
 61 	if (attributes.path) return attributes.path;
 62 	return this.getFullPath(attributes.id, (parentItem ? parentItem.getIdPath() : ""));
 63 }
 64 
 65 XModelItemFactory.getFullPath = function (itemPath, parentPath) {
 66 	if (itemPath == null) return null;
 67 	if (parentPath == null) parentPath = "";
 68 	
 69 	var path = itemPath;
 70 	if (itemPath == ".") {
 71 		path = parentPath;
 72 
 73 	} else if (itemPath == "..") {
 74 		parentPath = parentPath.split("/");
 75 		path = parentPath.slice(0, parentPath.length - 1).join("/");
 76 
 77 	} else if (parentPath == "") {
 78 		path = itemPath;
 79 
 80 	} else {
 81 		path = parentPath + "/" + itemPath;
 82 	}
 83 	return path;
 84 }

 90 XModelItemFactory.createItemType = function (typeConstant, typeName, constructor, superClassConstructor) {
 91 	if (constructor == null) constructor = new Function();
 92 	if (typeof superClassConstructor == "string") superClassConstructor = this.getItemTypeConstructor(superClassConstructor);
 93 	if (superClassConstructor == null) superClassConstructor = XModelItem;
 94 
 95 	// initialize the constructor
 96 	constructor.prototype = new superClassConstructor();	
 97 
 98 	constructor.prototype.type = typeName;
 99 	constructor.prototype.constructor = constructor;
100 	constructor.prototype.toString = new Function("return '[XModelItem:" + typeName + " path=\"' + this.getIdPath() + '\"]'");
101 	constructor.toString = new Function("return '[Class XModelItem:" + typeName + "]'");
102 	
103 	// put the item type into the typemap
104 	this.registerItemType(typeConstant, typeName, constructor);
105 
106 	// return the prototype
107 	return constructor;
108 }
109 
110 XModelItemFactory.registerItemType = function(typeConstant, typeName, constructor) {
111 	// assign the type constant to the window so everyone else can use it
112 	window[typeConstant] = typeName;
113 	this.typeConstructorMap[typeName] = constructor;	
114 }

117 XModelItemFactory.getItemTypeConstructor = function (typeName) {
118 	var typeConstructor = this.typeConstructorMap[typeName];
119 	if (typeConstructor == null) typeConstructor = this.typeConstructorMap["string"];
120 	return typeConstructor;
121 }

196 XModel.registerErrorMessage("valueIsRequired", AjxMsg.valueIsRequired);
197 XModelItem.prototype.validate = function (value, form, formItem, instance) {
198 
199 	// see if it's required
200 	if (value == null || value === "") {
201 		if (this.getRequired()) {
202 			throw this.getModel().getErrorMessage("valueIsRequired", value);
203 		}
204     }
205     
206 	// next validate the type
207 	//	this will throw an exception if something went wrong
208 	//	also, value may be coerced to a particular type by the validator
209 	else {
210 		value = this.validateType(value);
211 	}
212 	
213 	// if they defined any constraints, 
214 	var constraints = this.getConstraints();
215 	if (constraints == null) return value;
216 
217 	if (! (AjxUtil.isInstance(constraints, Array))) constraints = [constraints];
218 	for (var i = 0; i < constraints.length; i++) {
219 		var constraint = constraints[i];
220 		if (constraint.type == "method") {
221 			// The constraint method should either return a value, or throw an
222 			// exception.
223 			value = constraint.value.call(this, value, form, formItem, instance);
224 		}
225 // 		if (isValid == false) {
226 // 			throw this.getModel().getErrorMessage(constraint.errorMessageId, value);
227 // 		}
228 	}
229 	return value;
230 }

332 XModelItem.prototype.validateString = function(value) {
333 	if (value == null) return;
334 	
335 	if (!AjxUtil.isString(value)) {
336 		throw this.getModel().getErrorMessage("notAString", value);
337 	}
338 
339 	value = this._normalizeAndValidate(value);
340 
341     var length = this.getLength();
342     if (length !== null) {
343         if (value.length !== length) {
344             throw this.getModel().getErrorMessage("stringLenWrong", length);
345         }
346     }
347     else {
348 		var maxLength = this.getMaxLength();
349 		if (maxLength !== null && value.length > maxLength) {
350 			throw this.getModel().getErrorMessage("stringTooLong", maxLength);
351 		}
352 	
353 		var minLength = this.getMinLength();
354 		if (minLength !== null && value.length < minLength) {
355 			throw this.getModel().getErrorMessage("stringTooShort", minLength);
356 		}
357     }
358     
359     return value;
360 }

400 /**
401  * Normalizes value against whiteSpace facet and then validates 
402  * against pattern and enumeration facets.
403  * @private
404  */
405 XModelItem.prototype._normalizeAndValidate = function(value) {
406 
407     var whiteSpace = this.getWhiteSpace();
408     if (whiteSpace !== null) {
409     	if (whiteSpace === "replace" || whiteSpace === "collapse") {
410     		value = value.replace(/[\t\r\n]/g, " ");
411     	}
412     	if (whiteSpace === "collapse") {
413     		value = value.replace(/^\s+/,"").replace(/\s+$/,"").replace(/[ ]+/, " ");
414     	}
415     }
416 	
417     var pattern = this.getPattern();
418     if (pattern != null) {
419     	var matched = false;
420     	for (var i = 0; i < pattern.length; i++) {
421     		if (pattern[i].test(value)) {
422     			matched = true;
423     			break;
424     		}
425     	}
426 		if (!matched) {
427 			throw this.getModel().getErrorMessage("stringMismatch", value);
428 		}    	
429     }
430     
431     var enumeration = this.getEnumeration();
432     if (enumeration !== null) {
433     	var matched = false;
434     	for (var i = 0; i < enumeration.length; i++) {
435     		if (enumeration[i] === value) {
436     			matched = true;
437     			break;
438     		}
439     	}
440     	if (!matched) {
441 			throw this.getModel().getErrorMessage("stringMismatch", value);
442     	}
443     }
444     
445 	return value;
446 }

586 XModelItem.prototype.validateNumber = function(value) {
587 	value = this._normalizeAndValidate(value);
588 
589 	var nvalue = parseFloat(value);
590 
591 	if (isNaN(nvalue) || !AjxUtil.FLOAT_RE.test(value)) {
592 		throw this.getModel().getErrorMessage("notANumber", value);
593 	}
594 
595 	var totalDigits = this.getTotalDigits();
596 	if (this.totalDigits !== null) {
597 		var wholePart = Math.floor(nvalue);
598 		if (wholePart.toString().length > totalDigits) {
599 			throw this.getModel().getErrorMessage("numberTotalExceeded", value, totalDigits);
600 		}
601 	}
602 
603 	var fractionDigits = this.getFractionDigits();
604 	if (this.fractionDigits !== null) {
605 		var fractionPart = String(nvalue - Math.floor(nvalue));
606 		if (fractionPart.indexOf('.') != -1 && fractionPart.replace(/^\d*\./,"").length > fractionDigits) {
607 			throw this.getModel().getErrorMessage("numberFractionExceeded", value, fractionDigits);
608 		}
609 	}
610 
611 	var maxInclusive = this.getMaxInclusive();
612 	if (maxInclusive !== null && nvalue > maxInclusive) {
613 		throw this.getModel().getErrorMessage("numberMoreThanMax", maxInclusive);
614 	}
615 	
616 	var maxExclusive = this.getMaxExclusive();
617 	if (maxExclusive !== null && nvalue >= maxExclusive) {
618 		throw this.getModel().getErrorMessage("numberMoreThanEqualMax", maxExclusive);
619 	}
620 
621 	var minInclusive = this.getMinInclusive();
622 	if (minInclusive !== null && nvalue < minInclusive) {
623 		throw this.getModel().getErrorMessage("numberLessThanMin",  minInclusive);
624 	}
625 	
626 	var minExclusive = this.getMinExclusive();
627 	if (minExclusive !== null && nvalue <= minExclusive) {
628 		throw this.getModel().getErrorMessage("numberLessThanEqualMin", minExclusive);
629 	}
630 
631 	return nvalue;
632 }

653 // methods
654 XModelItem.prototype.validateDate = function(value) {
655 	
656 	if (AjxUtil.isInstance(value, Date)) return value;
657 	if (AjxUtil.isString(value)) {
658 		value = value.toLowerCase();
659 		var date = new Date();
660 
661 		if (value.indexOf("/") > -1) {
662 			var dateStrs = value.split("/");
663 			if (dateStrs.length == 3){
664 				var month = dateStrs[0];
665 				var day = dateStrs[1];
666 				var year = dateStrs[2];							
667 					
668 				if (month.length <= 2 && day.length <= 2 && year.length == 4) {
669 					//remove the preceeding 0 of the date value,
670 					//otherwise parseInt will evaluate it as 0
671 					month = parseInt(XModel.removePreceedingZero(month));
672 					day = parseInt(XModel.removePreceedingZero(day));
673 					year = parseInt(XModel.removePreceedingZero(year));							
674 					
675 					if (!isNaN(month) && !isNaN(day) && !isNaN(year)) {
676 						month -= 1;
677 						date.setFullYear(year, month, day);
678 						date.setHours(0,0,0,0);
679 						return date;

687 						date.setFullYear(year, month, day);
688 						date.setHours(0,0,0,0);
689 						return date; */
690 					}
691 				}
692 			}
693 		} else {
694 			// set to midnight today according to local time
695 			date.setHours(0,0,0,0);
696 			
697 			if (value == AjxMsg.today) {
698 				return date;
699 			} else if (value == AjxMsg.yesterday) {
700 				date.setTime(date.getTime() - this.msecInOneDay);
701 				return date;
702 			} else if (value == AjxMsg.tomorrow) {
703 				date.setTime(date.getTime() + this.msecInOneDay);
704 				return date;
705 			}
706 		}
707 	}
708 	throw this.getModel().getErrorMessage("invalidDateString", value);
709 	return value;

725 XModel.registerErrorMessage("invalidTimeString",		 AjxMsg.invalidTimeString);
726 // time is returned as a number of milliseconds since
727 XModelItem.prototype.validateTime = function (value) {
728 
729 	if (AjxUtil.isNumber(value)) return value;
730 	
731 	if (AjxUtil.isInstance(value, Date)) {
732 		return ((value.getHours() * 360) + (value.getMinutes() * 60) + value.getSeconds()) * 1000;
733 	}
734 	
735 	if (AjxUtil.isString(value)) {
736 		value = value.toLowerCase();
737 		if (value.indexOf(":") > -1) {
738 			value = value.split(":");
739 
740 			var isPM = false;
741 			var lastPiece = value[value.length - 1];
742 			isPM = (lastPiece.indexOf(I18nMsg.periodPm.toLowerCase()) > -1);
743 
744 			var hour = parseInt(value[0]);
745 			var min = parseInt(value[1]);
746 			var sec = (value.length == 3 ? parseInt(value[2]) : 0);
747 			if (!isNaN(hour) && !isNaN(min) && !isNaN(sec)) {
748 				hour -= 1;
749 				if (isPM && hour > 11) hour += 12;
750 				
751 				return ((hour * 360) + (min * 60) + sec) * 1000;
752 			}
753 		}
754 	}
755 	throw this.getModel().getErrorMessage("invalidTimeString", value);
756 }

760 XModelItem.prototype.validateDateTime = function (value) {
761 
762 	if (AjxUtil.isInstance(value, Date)) return value;
763 	if (AjxUtil.isNumber(value)) return value;
764 	if (AjxUtil.isString(value)) {
765 		// try to get the value as a date
766 		//  (this will ignore time fields, and will throw an exeception if we couldn't parse a date)
767 		var date = this.validateDate(value);
768 		
769 		// if it has a time component
770 		if (value.indexOf(":") > -1) {
771 			var time = value.split(" ")[1];
772 			// this will validate the time string and will throw an exception if it doesn't match
773 			time = this.validateTimeString(time);
774 			
775 			date.setTime(date.getTime() + time);
776 		}
777 		return date;
778 	}
779 	// probably should never get here...
780 	throw this.getModel().getErrorMessage("invalidDatetimeString", value);
781 }

935 Enum_XModelItem.prototype.validateType = function (value) {
936 	// if the selection is open, they can enter any value they want
937 	var selectionIsOpen = this.getSelection() == _OPEN_;
938 	if (selectionIsOpen) return value;
939 	
940 	// selection is not open: it must be one of the supplied choices
941 	var choices = this.getChoices();
942 	for (var i = 0; i < choices.length; i++) {
943 		var choice = choices[i];
944 		if (AjxUtil.isInstance(choice, Object)) {
945 			if (choice.value == value) return value;
946 		} else {
947 			if (choice == value) return value;
948 		}
949 	}
950 	
951 	// if we get here, we didn't match any of the choices
952 	throw this.getModel().getErrorMessage("didNotMatchChoice", value);
953 }
