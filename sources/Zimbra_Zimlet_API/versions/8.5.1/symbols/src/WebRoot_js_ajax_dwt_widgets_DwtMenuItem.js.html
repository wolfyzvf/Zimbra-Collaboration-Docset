 23 /**
 24  * Creates a menu item.
 25  * @constructor
 26  * @class
 27  * Menu items can be part of a radio group, or can be checked style menu items.
 28  *
 29  * @author Ross Dargahi
 30  * 
 31  * @param {hash}	params		a hash of parameters
 32  * @param  {DwtComposite}     params.parent		the parent widget
 33  * @param  {constant}     params.style			the menu item style
 34  * @param  {string}     params.radioGroupId 	the radio group that the menu item is part of
 35  * @param  {number}     params.index 			the position in menu
 36  * @param  {string}     params.className		the CSS class
 37  * @param  {constant}     params.posStyle		the positioning style (see {@link DwtControl})
 38  * @param  {string}     params.id			an explicit ID to use for the control's HTML element
 39  * 
 40  * @extends		DwtButton
 41  */
 42 DwtMenuItem = function(params) {
 43 	if (arguments.length == 0) { return; }
 44 	params = Dwt.getParams(arguments, DwtMenuItem.PARAMS);
 45 
 46 	// check parameters
 47 	var parent = params.parent;
 48 	if (!(parent && parent.isDwtMenu)) {
 49 		throw new DwtException("Parent must be a DwtMenu object", DwtException.INVALIDPARENT, "DwtMenuItem");
 50 	}
 51 
 52 	var style = params.style = params.style || DwtMenuItem.NO_STYLE;
 53 	if (parent._style == DwtMenu.BAR_STYLE && style != DwtMenuItem.PUSH_STYLE) {
 54 		throw new DwtException("DwtMenuItemInit: invalid style", DwtException.INVALID_PARAM, "DwtMenuItem");
 55 	}
 56 
 57 	// call super constructor
 58 	style &= ~DwtLabel.IMAGE_RIGHT; // remove image right style
 59 	style |= DwtButton.ALWAYS_FLAT | DwtLabel.IMAGE_LEFT; // set default styles
 60 	params.className = (style & DwtMenuItem.SEPARATOR_STYLE)
 61 		? "ZMenuItemSeparator" : (params.className || "ZMenuItem");
 62 	params.listeners = DwtMenuItem._listeners;
 63 	DwtButton.call(this, params);
 64 
 65 	this.setDropDownImages("Cascade", "Cascade", "Cascade", "Cascade");
 66 	this._radioGroupId = params.radioGroupId;
 67 
 68 	// add this item at the specified index
 69 	if (parent._addItem) {
 70 		parent._addItem(this, params.index);
 71 	}
 72 
 73 	// add listeners if not menu item separator
 74 	if (!(style & DwtMenuItem.SEPARATOR_STYLE)) {
 75 		this.addSelectionListener(this.__handleItemSelect.bind(this));
 76 	}
 77 };

146 /**
147  * Creates the menu item.
148  * 
149  * @param	{hash}	params		a hash of parameters
150  */
151 DwtMenuItem.create =
152 function(params) {
153 	var mi = new DwtMenuItem(params);
154 	if (params.imageInfo) {
155 		mi.setImage(params.imageInfo);
156 	}
157 	if (params.text) {
158 		mi.setText(params.text);
159 	}
160 	mi.setEnabled(params.enabled !== false);
161 	return mi;
162 };

174 /**
175  * Sets the checked flag.
176  *

179  */
180 DwtMenuItem.prototype.setChecked =
181 function(checked, skipNotify) {
182 	this._setChecked(checked, null, skipNotify);
183 	this.parent._checkItemAdded(this);

192 DwtMenuItem.prototype.setText =
193 function(text) {
194 	DwtButton.prototype.setText.call(this, text);
195 	if (this.parent.isPoppedUp()) {
196 		// resize menu if we reset text on the fly
197 		this.parent.render();
198 	}
199 };

247 DwtMenuItem.prototype._setChecked =
248 function(checked, ev, skipNotify) {
249 	var isCheck = this._style & DwtMenuItem.CHECK_STYLE;
250 	var isRadio = this._style & DwtMenuItem.RADIO_STYLE;
251 	if ((isCheck || isRadio) && this._itemChecked != checked) {
252 		this._itemChecked = checked;
253 
254 		if (this._checkEl) {
255 			this._checkEl.innerHTML = "";
256 			var icon = checked ? (isCheck ? "MenuCheck" : "MenuRadio") : "Blank_9";
257 			AjxImg.setImage(this._checkEl, icon);
258 			if (checked) {
259 				// deselect currently selected radio button
260 				if (isRadio) {
261 					this.parent._radioItemSelected(this, skipNotify);
262 				}
263 			}
264 		}
265 	}
266 };

271 DwtMenuItem.prototype._submenuItemAdded =
272 function() {
273 	if (this.isSeparator()) { return; }
274 
275 	if (this._cascCell == null) {
276 		this._cascCell = this._row.insertCell(-1);
277 		this._cascCell.noWrap = true;
278 	}
279 };

284 		this._dropDownEl.innerHTML = "";
285 	}

288 DwtMenuItem.prototype._popupMenu =
289 function(delay, kbGenerated) {
290 	var menu = this.getMenu();
291 	var pp = this.parent.parent;
292 	var pb = this.getBounds();
293 	var ws = menu.shell.getSize();
294 	var s = menu.getSize();
295 	var x;
296 	var y;
297 	var vBorder;
298 	var hBorder;
299 	var ppHtmlElement = pp.getHtmlElement();
300 	if (pp._style == DwtMenu.BAR_STYLE) {
301 		vBorder = (ppHtmlElement.style.borderLeftWidth == "") ? 0 : parseInt(ppHtmlElement.style.borderLeftWidth);
302 		x = pb.x + vBorder;
303 		hBorder = (ppHtmlElement.style.borderTopWidth == "") ? 0 : parseInt(ppHtmlElement.style.borderTopWidth);
304 		hBorder += (ppHtmlElement.style.borderBottomWidth == "") ? 0 : parseInt(ppHtmlElement.style.borderBottonWidth);
305 		y = pb.y + pb.height + hBorder;		
306 		x = ((x + s.x) >= ws.x) ? x - (x + s.x - ws.x): x;
307 	}
308 	else { // Drop Down
309 		vBorder = (ppHtmlElement.style.borderLeftWidth == "") ? 0 : parseInt(ppHtmlElement.style.borderLeftWidth);
310 		vBorder += (ppHtmlElement.style.borderRightWidth == "") ? 0 : parseInt(ppHtmlElement.style.borderRightWidth);
311 		x = pb.x + pb.width + vBorder;
312 		hBorder = (ppHtmlElement.style.borderTopWidth == "") ? 0 : parseInt(ppHtmlElement.style.borderTopWidth);
313 		y = pb.y + hBorder;
314         if (menu.centerOnParentVertically()) {
315             y += pb.height / 2;
316         }
317 		//x = ((x + s.x) >= ws.x) ? pb.x - s.x - vBorder : x;
318 	}
319 	menu.popup(delay, x, y, kbGenerated);
320 };

341 DwtMenuItem.prototype.__handleItemSelect =
342 function(event) {
343 	this.setDisplayState(DwtControl.NORMAL);
344 	if (this.isStyle(DwtMenuItem.CHECK_STYLE)) {
345 		this._setChecked(!this._itemChecked, null, true);
346 		event.detail = this.getChecked() ? DwtMenuItem.CHECKED : DwtMenuItem.UNCHECKED;
347 	}
348 	else if (this.isStyle(DwtMenuItem.RADIO_STYLE)) {
349 		this._setChecked(true, true);
350 		this.parent._radioItemSelected(this, true);
351 		event.detail = this.getChecked() ? DwtMenuItem.CHECKED : DwtMenuItem.UNCHECKED;
352 	}
353 	else if (this.isStyle(DwtMenuItem.PUSH_STYLE)) {
354 		if (this._menu) {
355 			if (this._isMenuPoppedUp()) {
356 				DwtMenu.closeActiveMenu(event);
357 			}
358 			else {
359 				this._popupMenu();
360 			}
361 		}
362 		return;
363 	}
364 	if (!this.isStyle(DwtMenuItem.CASCADE_STYLE)) {
365 		if (this._selectableWithSubmenu || !this._menu || !this._menu.isPoppedUp || !this._menu.isPoppedUp()) {
366 			DwtMenu.closeActiveMenu(event);
367 		}
368 	}
369 };

371 DwtMenuItem._mouseOverListener =
372 function(ev) {
373 	var menuItem = ev.dwtObj;
374 	if (!menuItem) { return false; }
375 	var menu = menuItem.parent;
376 	if (menu._hoveredItem) {
377 		var mouseEv = new DwtMouseEvent();
378 		mouseEv.dwtObj = menu._hoveredItem;
379 		DwtButton._mouseOutListener(mouseEv);
380 	}
381 	if (menuItem.isSeparator()) { return false; }
382 	DwtButton._mouseOverListener(ev, menuItem);
383 	menu._hoveredItem = menuItem;
384 	menu._popdownSubmenus();
385 	if (menuItem._menu && !ev.ersatz) {
386 		menuItem._popupMenu(menuItem._hoverDelay);
387 	}
388 };
389 
390 DwtMenuItem._mouseOutListener =
391 function(ev) {
392 	var menuItem = ev.dwtObj;
393 	var submenu = menuItem && menuItem.getMenu();
394 	if (submenu && submenu.isPoppedUp()) { return; }
395 	DwtButton._mouseOutListener(ev);
396 	if (menuItem) {
397 		menuItem.parent._hoveredItem = null;
398 	}
399 };

401 /*
402  * returns menu item table row element
403  */
404 DwtMenuItem.prototype.getRowElement =
405 function() {
406 	var el = this._textEl ||
407 		this._dropDownEl ||
408 		(this._iconEl && this._iconEl.left) ||
409 		(this._iconEl && this._iconEl.right);
410 	if (el) {
411 		return el.parentNode;
412 	}
413 };
