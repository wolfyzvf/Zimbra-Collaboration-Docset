 28 /**
 29  * Creates an empty contact.
 30  * @class
 31  * This class represents a contact (typically a person) with all its associated versions
 32  * of email address, home and work addresses, phone numbers, etc. Contacts can be filed/sorted
 33  * in different ways, with the default being Last, First. A contact is an item, so
 34  * it has tagging and flagging support, and belongs to a list.
 35  * <p>
 36  * Most of a contact's data is kept in attributes. These include name, phone, etc. Meta-data and
 37  * data common to items are not kept in attributes. These include flags, tags, folder, and
 38  * modified/created dates. Since the attribute data for contacts is loaded only once, a contact
 39  * gets its attribute values from that canonical list.
 40  * </p>
 41  *
 42  * @param {int}	id		the unique ID
 43  * @param {ZmContactList}	list		the list that contains this contact
 44  * @param {constant}	type		the item type
 45  * @param {object}	newDl		true if this is a new DL
 46  *
 47  * @extends		ZmItem
 48  */
 49 ZmContact = function(id, list, type, newDl) {
 50 	if (arguments.length == 0) { return; }
 51 
 52 	type = type || ZmItem.CONTACT;
 53 	ZmItem.call(this, type, id, list);
 54 
 55 	this.attr = {};
 56 	this.isGal = (this.list && this.list.isGal) || newDl;
 57 	if (newDl) {
 58 		this.folderId = ZmFolder.ID_DLS;
 59 		this.dlInfo = {	isMember: false,
 60 						isOwner: true,
 61 						subscriptionPolicy: null,
 62 						unsubscriptionPolicy: null,
 63 						description: "",
 64 						displayName: "",
 65 						notes: "",
 66 						hideInGal: false,
 67 						mailPolicy: null,
 68 						owners: [appCtxt.get(ZmSetting.USERNAME)]
 69 		};
 70 
 71 	}
 72 
 73 	this.participants = new AjxVector(); // XXX: need to populate this guy (see ZmConv)
 74 };

394 /**
395  * Creates a contact from an XML node.
396  *
397  * @param {Object}	node		a "cn" XML node
398  * @param {Hash}	args		args to pass to the constructor
399  * @return	{ZmContact}	the contact
400  */
401 ZmContact.createFromDom =
402 function(node, args) {
403 	// check global cache for this item first
404 	var contact = appCtxt.cacheGet(node.id);
405 
406 	// make sure the revision hasnt changed, otherwise contact is out of date
407 	if (contact == null || (contact && contact.rev != node.rev)) {
408 		contact = new ZmContact(node.id, args.list);
409 		if (args.isGal) {
410 			contact.isGal = args.isGal;
411 		}
412 		contact._loadFromDom(node);
413 		//update the canonical list
414 		appCtxt.getApp(ZmApp.CONTACTS).getContactList().add(contact);
415 	} else {
416 		if (node.m) {
417 			contact.attr[ZmContact.F_groups] = node.m;
418 		}
419 		if (node.ref) {
420 			contact.ref = node.ref;
421 		}
422 		if (node.tn) {
423 			contact._parseTagNames(node.tn);
424 		}
425 		AjxUtil.hashUpdate(contact.attr, node._attrs);	// merge new attrs just in case we don't have them
426 		contact.list = args.list || new ZmContactList(null);
427 	}
428 
429 	return contact;
430 };

432 /**
433  * Compares two contacts based on how they are filed. Intended for use by
434  * sort methods.
435  *
436  * @param {ZmContact}		a		a contact
437  * @param {ZmContact}		b		a contact
438  * @return	{int}	0 if the contacts are the same; 1 if "a" is before "b"; -1 if "b" is before "a"
439  */
440 ZmContact.compareByFileAs =
441 function(a, b) {
442 	var aFileAs = (a instanceof ZmContact) ? a.getFileAs(true) : ZmContact.computeFileAs(a._attrs).toLowerCase();
443 	var bFileAs = (b instanceof ZmContact) ? b.getFileAs(true) : ZmContact.computeFileAs(b._attrs).toLowerCase();
444 
445 	if (!bFileAs || (aFileAs > bFileAs)) return 1;
446 	if (aFileAs < bFileAs) return -1;
447 	return 0;
448 };

450 /**
451  * Figures out the filing string for the contact according to the chosen method.
452  *
453  * @param {ZmContact|Hash}	contact		a contact or a hash of contact attributes
454  */
455 ZmContact.computeFileAs =
456 function(contact) {
457 	var attr = (contact instanceof ZmContact) ? contact.getAttrs() : contact;
458 	if (!attr) return;
459 
460 	if (attr[ZmContact.F_dlDisplayName]) {
461 		//this is only DL case. But since this is sometimes just the attrs,
462 		//I can't always use isDistributionList method.
463 		return attr[ZmContact.F_dlDisplayName];
464 	}
465 
466 	var val = parseInt(attr.fileAs);
467 	var fa;
468 	var idx = 0;
469 
470 	switch (val) {
471 		case ZmContact.FA_LAST_C_FIRST: 										// Last, First
472 		default: {
473 			// if GAL contact, use full name instead (bug fix #4850,4009)
474 			if (contact && contact.isGal) {
475 				if (attr.fullName) { // bug fix #27428 - if fullName is Array, return first
476 					return (attr.fullName instanceof Array) ? attr.fullName[0] : attr.fullName;
477 				}
478 				return ((attr.email instanceof Array) ? attr.email[0] : attr.email);
479 			}
480 			fa = ZmContact.fileAsLastFirst(attr.firstName, attr.lastName, attr.fullName, attr.nickname);
481 		}
482 		break;
483 
484 		case ZmContact.FA_FIRST_LAST: { 										// First Last
485 			fa = ZmContact.fileAsFirstLast(attr.firstName, attr.lastName, attr.fullName, attr.nickname);
486 		}
487 		break;
488 
489 		case ZmContact.FA_COMPANY: {											// Company
490 			if (attr.company) fa = attr.company;
491 		}
492 		break;
493 
494 		case ZmContact.FA_LAST_C_FIRST_COMPANY: {								// Last, First (Company)
495 			var name = ZmContact.fileAsLastFirst(attr.firstName, attr.lastName, attr.fullName, attr.nickname);
496 			fa = ZmContact.fileAsNameCompany(name, attr.company);
497 		}
498 		break;
499 
500 		case ZmContact.FA_FIRST_LAST_COMPANY: {									// First Last (Company)
501 			var name = ZmContact.fileAsFirstLast(attr.firstName, attr.lastName, attr.fullName, attr.nickname);
502 			fa = ZmContact.fileAsNameCompany(name, attr.company);
503 		}
504 		break;
505 
506 		case ZmContact.FA_COMPANY_LAST_C_FIRST: {								// Company (Last, First)
507 			var name = ZmContact.fileAsLastFirst(attr.firstName, attr.lastName);
508 			fa = ZmContact.fileAsCompanyName(name, attr.company);
509 		}
510 		break;
511 
512 		case ZmContact.FA_COMPANY_FIRST_LAST: {									// Company (First Last)
513 			var name = ZmContact.fileAsFirstLast(attr.firstName, attr.lastName);
514 			fa = ZmContact.fileAsCompanyName(name, attr.company);
515 		}
516 		break;
517 
518 		case ZmContact.FA_CUSTOM: {												// custom looks like this: "8:foobar"
519 			return attr.fileAs.substring(2);
520 		}
521 		break;
522 	}
523 	return fa || attr.fullName || "";
524 };

635 /**
636  * Normalizes the numbering of the given attribute names and
637  * returns a new object with the re-numbered attributes. For
638  * example, if the attributes contains a "foo2" but no "foo",
639  * then the "foo2" attribute will be renamed to "foo" in the
640  * returned object.
641  *
642  * @param {Hash}	attrs  a hash of attributes to normalize.
643  * @param {String}	[prefix] if specified, only the the attributes that match the given prefix will be returned
644  * @param {Array}	[ignore] if specified, the attributes that are present in the array will not be normalized
645  * @return	{Hash}	a hash of normalized attributes
646  */
647 ZmContact.getNormalizedAttrs = function(attrs, prefix, ignore) {
648 	var nattrs = {};
649 	if (attrs) {
650 		// normalize attribute numbering
651 		var names = AjxUtil.keys(attrs);
652 		names.sort(ZmContact.__BY_ATTRIBUTE);
653 		var a = {};
654 		for (var i = 0; i < names.length; i++) {
655 			var name = names[i];
656 			// get current count
657 			var nprefix = ZmContact.getPrefix(name);
658 			if (prefix && prefix != nprefix) continue;
659 			if (AjxUtil.isArray(ignore) && AjxUtil.indexOf(ignore, nprefix)!=-1) {
660 				nattrs[name] = attrs[name];
661 			} else {
662 				if (!a[nprefix]) a[nprefix] = 0;
663 				// normalize, if needed
664 				var nname = ZmContact.getAttributeName(nprefix, ++a[nprefix]);
665 				nattrs[nname] = attrs[name];
666 			}
667 		}
668 	}
669 	return nattrs;
670 };

674 	var aa = a.match(ZmContact.__RE_ATTRIBUTE) || [a,a,1];

676 	return aa[1] == bb[1] ? Number(aa[2]) - Number(bb[2]) : aa[1].localeCompare(bb[1]);

1014 	var grantees = right.grantee;
1015 	if (!grantees) {

1019 		var grantee = grantees[i];
1020

1022 
1023 		if (mailPolicy == ZmGroupView.MAIL_POLICY_SPECIFIC) {
1024 			specificMailers.push(grantee.name);
1025 		}
1026 		else if (mailPolicy == ZmGroupView.MAIL_POLICY_ANYONE) {
1027 			break;
1028 		}
1029 		else if (mailPolicy == ZmGroupView.MAIL_POLICY_INTERNAL) {
1030 			break;
1031 		}
1032 		else if (mailPolicy == ZmGroupView.MAIL_POLICY_MEMBERS) {
1033 			if (grantee.name == this.getEmail()) {
1034 				//this means only members of this DL can send.
1035 				break;
1036 			}
1037 			else {
1038 				//must be another DL, and we do allow it, so treat it as regular user.
1039 				specificMailers.push(grantee.name);
1040 				mailPolicy = ZmGroupView.MAIL_POLICY_SPECIFIC;
1041 			}
1042 		}
1043 	}
1044 	mailPolicy = mailPolicy || ZmGroupView.MAIL_POLICY_ANYONE;
