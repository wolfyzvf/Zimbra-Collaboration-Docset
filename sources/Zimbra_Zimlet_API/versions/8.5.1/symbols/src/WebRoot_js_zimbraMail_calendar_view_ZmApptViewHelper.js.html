 74 /**
 75  * Gets an object with the indices of the currently selected time fields.
 76  *
 77  * @param {ZmApptEditView}	tabView		the edit/tab view containing time widgets
 78  * @param {Hash}	dateInfo	a hash of date info to fill in
 79  */
 80 ZmApptViewHelper.getDateInfo =
 81 function(tabView, dateInfo) {
 82 	dateInfo.startDate = tabView._startDateField.value;
 83 	dateInfo.endDate = tabView._endDateField.value;
 84     var tzoneSelect = tabView._tzoneSelect || tabView._tzoneSelectStart;
 85     dateInfo.timezone = tzoneSelect ? tzoneSelect.getValue() : "";
 86     if (tabView._allDayCheckbox && tabView._allDayCheckbox.checked) {
 87 		dateInfo.showTime = false;
 88 
 89         //used by DwtTimeInput - advanced time picker
 90         dateInfo.startTimeStr = dateInfo.endTimeStr = null;
 91 
 92         //used by DwtTimeSelect
 93         dateInfo.startHourIdx = dateInfo.startMinuteIdx = dateInfo.startAmPmIdx =
 94 		dateInfo.endHourIdx = dateInfo.endMinuteIdx = dateInfo.endAmPmIdx = null;
 95 
 96         dateInfo.isAllDay = true;
 97     } else {
 98 		dateInfo.showTime = true;
 99 
100         if(tabView._startTimeSelect instanceof DwtTimeSelect) {
101             dateInfo.startHourIdx = tabView._startTimeSelect.getSelectedHourIdx();
102             dateInfo.startMinuteIdx = tabView._startTimeSelect.getSelectedMinuteIdx();
103             dateInfo.startAmPmIdx = tabView._startTimeSelect.getSelectedAmPmIdx();
104             dateInfo.endHourIdx = tabView._endTimeSelect.getSelectedHourIdx();
105             dateInfo.endMinuteIdx = tabView._endTimeSelect.getSelectedMinuteIdx();
106             dateInfo.endAmPmIdx = tabView._endTimeSelect.getSelectedAmPmIdx();
107         }else {
108             dateInfo.startHourIdx = dateInfo.startMinuteIdx = dateInfo.startAmPmIdx =
109             dateInfo.endHourIdx = dateInfo.endMinuteIdx = dateInfo.endAmPmIdx = null;            
110         }
111 
112         if(tabView._startTimeSelect instanceof DwtTimeInput) {
113             dateInfo.startTimeStr = tabView._startTimeSelect.getTimeString();
114             dateInfo.endTimeStr = tabView._endTimeSelect.getTimeString();
115         }else {
116             dateInfo.startTimeStr = dateInfo.endTimeStr = null;
117         }
118 
119         dateInfo.isAllDay = false;
120 	}

123 ZmApptViewHelper.handleDateChange = 
124 function(startDateField, endDateField, isStartDate, skipCheck, oldStartDate) {
125 	var needsUpdate = false;
126 	var sd = AjxDateUtil.simpleParseDateStr(startDateField.value);
127 	var ed = AjxDateUtil.simpleParseDateStr(endDateField.value);
128 
129 	// if start date changed, reset end date if necessary
130 	if (isStartDate) {
131 		// if date was input by user and it's foobar, reset to today's date
132 		if (!skipCheck) {
133 			if (sd == null || isNaN(sd)) {
134 				sd = new Date();
135 			}
136 			// always reset the field value in case user entered date in wrong format
137 			startDateField.value = AjxDateUtil.simpleComputeDateStr(sd);
138 		}
139 
140 		if (ed.valueOf() < sd.valueOf()) {
141 			endDateField.value = startDateField.value;
142         }else if(oldStartDate != null) {
143             var delta = ed.getTime() - oldStartDate.getTime();
144             var newEndDate = new Date(sd.getTime() + delta);
145             endDateField.value = AjxDateUtil.simpleComputeDateStr(newEndDate);
146         }
147 		needsUpdate = true;
148 	} else {
149 		// if date was input by user and it's foobar, reset to today's date
150 		if (!skipCheck) {
151 			if (ed == null || isNaN(ed)) {
152 				ed = new Date();
153 			}
154 			// always reset the field value in case user entered date in wrong format
155 			endDateField.value = AjxDateUtil.simpleComputeDateStr(ed);
156 		}
157 
158 		// otherwise, reset start date if necessary
159 		if (sd.valueOf() > ed.valueOf()) {
160 			startDateField.value = endDateField.value;
161 			needsUpdate = true;
162 		}
163 	}
164 
165 	return needsUpdate;
166 };

168 ZmApptViewHelper.getApptToolTipText =
169 function(origAppt, controller) {
170     if(origAppt._toolTip) {
171         return origAppt._toolTip;
172     }
173     var appt = ZmAppt.quickClone(origAppt);
174     var organizer = appt.getOrganizer();
175 	var sentBy = appt.getSentBy();
176 	var userName = appCtxt.get(ZmSetting.USERNAME);
177 	if (sentBy || (organizer && organizer != userName)) {
178 		organizer = (appt.message && appt.message.invite && appt.message.invite.getOrganizerName()) || organizer;
179 		if (sentBy) {
180 			var contactsApp = appCtxt.getApp(ZmApp.CONTACTS);
181 			var contact = contactsApp && contactsApp.getContactByEmail(sentBy);
182 			sentBy = (contact && contact.getFullName()) || sentBy;
183 		}
184 	} else {
185 		organizer = null;
186 		sentBy = null;
187 	}
188 
189 	var params = {
190 		appt: appt,
191 		cal: (appt.folderId != ZmOrganizer.ID_CALENDAR && controller) ? controller.getCalendar() : null,
192 		organizer: organizer,
193 		sentBy: sentBy,
194 		when: appt.getDurationText(false, false),
195 		location: appt.getLocation(),
196 		width: "250",
197         hideAttendees: true
198 	};
199 
200 	var toolTip = origAppt._toolTip = AjxTemplate.expand("calendar.Appointment#Tooltip", params);
201     return toolTip;
202 };

205 ZmApptViewHelper.getDayToolTipText =
206 function(date, list, controller, noheader, emptyMsg, isMinical) {
207 	
208 	if(!emptyMsg){
209 		emptyMsg = ZmMsg.noAppts;
210 	}
211 
212 	var html = new AjxBuffer();
213 
214 	var formatter = DwtCalendar.getDateFullFormatter();	
215 	var title = formatter.format(date);
216 	
217 	html.append("<div>");
218 
219 	html.append("<table cellpadding='0' cellspacing='0' border='0'>");
220 	if (!noheader) html.append("<tr><td><div class='calendar_tooltip_month_day_label'>", title, "</div></td></tr>");
221 	html.append("<tr><td>");
222 	html.append("<table cellpadding='1' cellspacing='0' border='0'>");
223 	
224 	var size = list ? list.size() : 0;
225 
226 	var useEmptyMsg = true;
227 	var dateTime = date.getTime();
228 	for (var i = 0; i < size; i++) {
229 		var ao = list.get(i);
230 
231 		if (ao.isAllDayEvent()) {
232             // Multi-day all day appts will be broken up into one sub-appt per day, so only show
233        		// the one that matches the selected date
234             var apptDate = new Date(ao.startDate.getTime());
235             apptDate.setHours(0,0,0,0);
236             if (apptDate.getTime() != dateTime) continue;
237 
238 			useEmptyMsg = false;
239             if(!isMinical && ao.toString() == "ZmAppt") {
240                 html.append("<tr><td><div class=appt>");
241                 html.append(ZmApptViewHelper.getApptToolTipText(ao, controller));
242                 html.append("</div></td></tr>");
243             }
244             else {
245                 //DBG.println("AO    "+ao);
246                 var widthField = AjxEnv.isIE ? "width:500px;" : "min-width:300px;";
247                 html.append("<tr><td><div style='" + widthField + "' class=appt>");
248                 html.append(ZmApptViewHelper._allDayItemHtml(ao, Dwt.getNextId(), controller, true, true));
249                 html.append("</div></td></tr>");
250             }
251 		}
252 	    else {
253 			useEmptyMsg = false;
254 		    if(!isMinical && ao.toString() == "ZmAppt") {
255                 html.append("<tr><td><div class=appt>");
256                 html.append(ZmApptViewHelper.getApptToolTipText(ao, controller));
257                 html.append("</div></td></tr>");
258             }
259             else {
260                 var color = ZmCalendarApp.COLORS[controller.getCalendarColor(ao.folderId)];
261                 var isNew = ao.status == ZmCalBaseItem.PSTATUS_NEEDS_ACTION;
262     
263                 html.append("<tr><td class='calendar_month_day_item'><div class='", color, isNew ? "DarkC" : "C", "'>");
264                 if (isNew) html.append("<b>");
265                 //html.append("• ");
266                 //var dur = ao.getShortStartHour();
267                 var dur = ao.getDurationText(false, false);
268                 html.append(dur);
269                 if (dur != "") {
270                     html.append(" ");
271                 }
272                 html.append(AjxStringUtil.htmlEncode(ao.getName()));
273                 if (isNew) html.append("</b>");
274                 html.append("</div>");
275                 html.append("</td></tr>");
276             }
277 		}
278 	}
279 	if (useEmptyMsg) {
280 		html.append("<tr><td>"+emptyMsg+"</td></tr>");
281 	}
282 	html.append("</table>");
283 	html.append("</td></tr></table>");
284 	html.append("</div>");
285 
286 	return html.toString();
287 };

289 /**
290  * Returns a list of calendars based on certain conditions. Especially useful
291  * for multi-account
292  *
293  * @param folderSelect	[DwtSelect]		DwtSelect object to populate
294  * @param folderRow		[HTMLElement]	Table row element to show/hide
295  * @param calendarOrgs	[Object]		Hash map of calendar ID to calendar owner
296  * @param calItem		[ZmCalItem]		a ZmAppt or ZmTask object
297  */
298 ZmApptViewHelper.populateFolderSelect =
299 function(folderSelect, folderRow, calendarOrgs, calItem) {
300 	// get calendar folders (across all accounts)
301 	var org = ZmOrganizer.ITEM_ORGANIZER[calItem.type];
302 	var data = [];
303 	var folderTree;
304 	var accounts = appCtxt.accountList.visibleAccounts;
305 	for (var i = 0; i < accounts.length; i++) {
306 		var acct = accounts[i];
307 
308 		var appEnabled = ZmApp.SETTING[ZmItem.APP[calItem.type]];
309 		if ((appCtxt.isOffline && acct.isMain) ||
310 			!appCtxt.get(appEnabled, null, acct))
311 		{
312 			continue;
313 		}
314 
315 		folderTree = appCtxt.getFolderTree(acct);
316 		data = data.concat(folderTree.getByType(org));
317 	}
318 
319 	// add the local account last for multi-account
320 	if (appCtxt.isOffline) {
321 		folderTree = appCtxt.getFolderTree(appCtxt.accountList.mainAccount);
322 		data = data.concat(folderTree.getByType(org));
323 	}
324 
325 	folderSelect.clearOptions();
326     
327 	for (var i = 0; i < data.length; i++) {
328 		var cal = data[i];
329 		var acct = cal.getAccount();
330 
331 		if (cal.noSuchFolder || cal.isFeed() || (cal.link && cal.isReadOnly()) || cal.isInTrash()) { continue; }
332 
333 		if (appCtxt.multiAccounts &&
334 			cal.nId == ZmOrganizer.ID_CALENDAR &&
335 			acct.isCalDavBased())
336 		{
337 			continue;
338 		}
339 
340         var id = cal.link ? cal.getRemoteId() : cal.id;
341 		calendarOrgs[id] = cal.owner;
342 
343 		// bug: 28363 - owner attribute is not available for shared sub folders
344 		if (cal.isRemote() && !cal.owner && cal.parent && cal.parent.isRemote()) {
345 			calendarOrgs[id] = cal.parent.getOwner();
346 		}
347 
348 		var selected = ((calItem.folderId == cal.id) || (calItem.folderId == id));
349 		var icon = appCtxt.multiAccounts ? acct.getIcon() : cal.getIconWithColor();
350 		var name = AjxStringUtil.htmlDecode(appCtxt.multiAccounts
351 			? ([cal.getName(), " (", acct.getDisplayName(), ")"].join(""))
352 			: cal.getName());
353 		var option = new DwtSelectOption(id, selected, name, null, null, icon);
354 		folderSelect.addOption(option, selected);
355 	}
356 
357     ZmApptViewHelper.folderSelectResize(folderSelect);
358     //todo: new ui hide folder select if there is only one folder
359 };

361 /**
362  * Takes a string, AjxEmailAddress, or contact/resource and returns
363  * a ZmContact or a ZmResource. If the attendee cannot be found in
364  * contacts, locations, or equipment, a new contact or
365  * resource is created and initialized.
366  *
367  * @param item			[object]		string, AjxEmailAddress, ZmContact, or ZmResource
368  * @param type			[constant]*		attendee type
369  * @param strictText	[boolean]*		if true, new location will not be created from free text
370  * @param strictEmail	[boolean]*		if true, new attendee will not be created from email address
371  */
372 ZmApptViewHelper.getAttendeeFromItem =
373 function(item, type, strictText, strictEmail, checkForAvailability) {
374 
375 	if (!item || !type) return null;
376 
377 	if (type == ZmCalBaseItem.LOCATION && !ZmApptViewHelper._locations) {
378 		if (!appCtxt.get(ZmSetting.GAL_ENABLED)) {
379 			//if GAL is disabled then user does not have permission to load locations.
380 			return null;
381 		}
382 		var locations = ZmApptViewHelper._locations = appCtxt.getApp(ZmApp.CALENDAR).getLocations();
383         if(!locations.isLoaded) {
384             locations.load();
385         }
386 
387 	}
388 	if (type == ZmCalBaseItem.EQUIPMENT && !ZmApptViewHelper._equipment) {
389 		var equipment = ZmApptViewHelper._equipment = appCtxt.getApp(ZmApp.CALENDAR).getEquipment();
390         if(!equipment.isLoaded) {
391             equipment.load();
392         }                
393 	}
394 	
395 	var attendee = null;
396 	if (item.type == ZmItem.CONTACT || item.type == ZmItem.RESOURCE) {
397 		// it's already a contact or resource, return it as is
398 		attendee = item;
399 	} else if (item instanceof AjxEmailAddress) {
400 		var addr = item.getAddress();
401 		// see if we have this contact/resource by checking email address
402 		attendee = ZmApptViewHelper._getAttendeeFromAddr(addr, type);
403 
404 		// Bug 7837: preserve the email address as it was typed
405 		//           instead of using the contact's primary email.
406 		if (attendee && (type === ZmCalBaseItem.PERSON || type === ZmCalBaseItem.GROUP)) {
407 			attendee = AjxUtil.createProxy(attendee);
408 			attendee._inviteAddress = addr;
409 			attendee.getEmail = function() {
410 				return this._inviteAddress || this.constructor.prototype.getEmail.apply(this);
411 			};
412 		}
413 
414 		if (!checkForAvailability && !attendee && !strictEmail) {
415 			// AjxEmailAddress has name and email, init a new contact/resource from those
416 			if (type === ZmCalBaseItem.PERSON) {
417 				attendee = new ZmContact(null, null, ZmItem.CONTACT);
418 			}
419 			else if (type === ZmCalBaseItem.GROUP) {
420 				attendee = new ZmContact(null, null, ZmItem.GROUP);
421 			}
422 			else {
423 				attendee = new ZmResource(type);
424 			}
425 			attendee.initFromEmail(item, true);
426 		}
427 		attendee.canExpand = item.canExpand;
428 		var ac = window.parentAppCtxt || window.appCtxt;
429 		ac.setIsExpandableDL(addr, attendee.canExpand);
430 	} else if (typeof item == "string") {
431 		item = AjxStringUtil.trim(item);	// trim white space
432 		item = item.replace(/;$/, "");		// trim separator
433 		// see if it's an email we can use for lookup
434 	 	var email = AjxEmailAddress.parse(item);
435 	 	if (email) {
436 	 		var addr = email.getAddress();
437 	 		// is it a contact/resource we already know about?
438 			attendee = ZmApptViewHelper._getAttendeeFromAddr(addr, type);
439 			if (!checkForAvailability && !attendee && !strictEmail) {
440 				if (type === ZmCalBaseItem.PERSON || type === ZmCalBaseItem.FORWARD) {
441 					attendee = new ZmContact(null, null, ZmItem.CONTACT);
442 				}
443 				else if (type === ZmCalBaseItem.GROUP) {
444 					attendee = new ZmContact(null, null, ZmItem.GROUP);
445 				}
446 				else if (type === ZmCalBaseItem.LOCATION) {
447 					attendee = new ZmResource(null, ZmApptViewHelper._locations, ZmCalBaseItem.LOCATION);
448 				}
449 				else if (type === ZmCalBaseItem.EQUIPMENT) {
450 					attendee = new ZmResource(null, ZmApptViewHelper._equipment, ZmCalBaseItem.EQUIPMENT);
451 				}
452 				attendee.initFromEmail(email, true);
453 			} else if (attendee && (type === ZmCalBaseItem.PERSON || type === ZmCalBaseItem.GROUP)) {
454 				// remember actual address (in case it's email2 or email3)
455 				attendee._inviteAddress = addr;
456                 attendee.getEmail = function() {
457 				    return this._inviteAddress || this.constructor.prototype.getEmail.apply(this);
458 			    };
459 			}
460 		}
461 	}
462 	return attendee;
463 };

465 ZmApptViewHelper._getAttendeeFromAddr =
466 function(addr, type) {
467 
468 	var attendee = null;
469 	if (type === ZmCalBaseItem.PERSON || type === ZmCalBaseItem.GROUP || type === ZmCalBaseItem.FORWARD) {
470 		var contactsApp = appCtxt.getApp(ZmApp.CONTACTS);
471 		attendee = contactsApp && contactsApp.getContactByEmail(addr);
472 	} else if (type == ZmCalBaseItem.LOCATION) {
473         attendee = ZmApptViewHelper._locations.getResourceByEmail(addr);
474 	} else if (type == ZmCalBaseItem.EQUIPMENT) {
475 		attendee = ZmApptViewHelper._equipment.getResourceByEmail(addr);
476 	}
477 	return attendee;
478 };

511 /**
512 * Creates a string from a list of attendees/locations/resources. If an item
513 * doesn't have a name, its address is used.
514 *
515 * @param list					[array]			list of attendees (ZmContact or ZmResource)
516 * @param type					[constant]		attendee type
517 * @param includeDisplayName		[boolean]*		if true, include location info in parens (ZmResource)
518 * @param includeRole		    [boolean]*		if true, include attendee role
519 */
520 ZmApptViewHelper.getAttendeesString = 
521 function(list, type, includeDisplayName, includeRole) {
522 	if (!(list && list.length)) return "";
523 
524 	var a = [];
525 	for (var i = 0; i < list.length; i++) {
526 		var attendee = list[i];
527 		var text = ZmApptViewHelper.getAttendeesText(attendee, type);
528 		if (includeDisplayName && list.length == 1) {
529 			var displayName = attendee.getAttr(ZmResource.F_locationName);
530 			if (displayName) {
531 				text = [text, " (", displayName, ")"].join("");
532 			}
533 		}
534         if(includeRole) {
535             text += " " + (attendee.getParticipantRole() || ZmCalItem.ROLE_REQUIRED);
536         }
537 		a.push(text);
538 	}
539 
540 	return a.join(ZmAppt.ATTENDEES_SEPARATOR);
541 };

584 /**
585 * Creates a string of attendees by role. this allows to show only count elements, with "..." appended.
586 *
587 * @param list					[array]			list of attendees (ZmContact or ZmResource)
588 * @param type					[constant]		attendee type
589 * @param role      		        [constant]      attendee role
590 * @param count                  [number]        number of attendees to be returned
591 */
592 ZmApptViewHelper.getAttendeesByRole =
593 function(list, type, role, count) {
594     if (!(list && list.length)) return "";
595 
596 	var res = [];
597 
598 	var attendees = ZmApptViewHelper.getAttendeesArrayByRole(list, role);
599 	for (var i = 0; i < attendees.length; i++) {
600 		if (count && i > count) {
601 			res.push(" ...");
602 			break;
603 		}
604 		if (i > 0) {
605 			res.push(ZmAppt.ATTENDEES_SEPARATOR);
606 		}
607 		res.push(attendees[i].getAttendeeText(type));
608 	}
609 	return res.join("");
610 };

771 /**
772  * Gets the attach list as HTML.
773  * 
774  * @param {ZmCalItem}	calItem			calendar item
775  * @param {Object}		attach			a generic Object contain meta info about the attachment
776  * @param {Boolean}		hasCheckbox		<code>true</code> to insert a checkbox prior to the attachment
777  * @return	{String}	the HTML
778  * 
779  * TODO: replace string onclick handlers with funcs
780  */
781 ZmApptViewHelper.getAttachListHtml =
782 function(calItem, attach, hasCheckbox, getLinkIdCallback) {
783 	var msgFetchUrl = appCtxt.get(ZmSetting.CSFE_MSG_FETCHER_URI);
784 
785 	// gather meta data for this attachment
786 	var mimeInfo = ZmMimeTable.getInfo(attach.ct);
787 	var icon = mimeInfo ? mimeInfo.image : "GenericDoc";
788 	var size = attach.s;
789 	var sizeText;
790 	if (size != null) {
791 		if (size < 1024)		sizeText = size + " B";
792 		else if (size < 1024^2)	sizeText = Math.round((size/1024) * 10) / 10 + " KB";
793 		else 					sizeText = Math.round((size / (1024*1024)) * 10) / 10 + " MB";
794 	}
795 
796 	var html = [];
797 	var i = 0;
798 
799 	// start building html for this attachment
800 	html[i++] = "<table border=0 cellpadding=0 cellspacing=0><tr>";
801 	if (hasCheckbox) {
802 		html[i++] = "<td width=1%><input type='checkbox' checked value='";
803 		html[i++] = attach.part;
804 		html[i++] = "' name='";
805 		html[i++] = ZmCalItem.ATTACHMENT_CHECKBOX_NAME;
806 		html[i++] = "'></td>";
807 	}
808 
809 	var hrefRoot = ["href='", msgFetchUrl, "&id=", calItem.invId, "&part=", attach.part].join("");
810 	html[i++] = "<td width=20><a target='_blank' class='AttLink' ";
811 	if (getLinkIdCallback) {
812 		var imageLinkId = getLinkIdCallback(attach.part, ZmCalItem.ATT_LINK_IMAGE);
813 		html[i++] = "id='";
814 		html[i++] = imageLinkId;
815 		html[i++] = "' ";
816 	}
817 	html[i++] = hrefRoot;
818 	html[i++] = "'>";
819 	html[i++] = AjxImg.getImageHtml(icon);
820 
821 	html[i++] = "</a></td><td><a target='_blank' class='AttLink' ";
822 
823 	if (appCtxt.get(ZmSetting.MAIL_ENABLED) && attach.ct == ZmMimeTable.MSG_RFC822) {
824 		html[i++] = " href='javascript:;' onclick='ZmCalItemView.rfc822Callback(";
825 		html[i++] = '"';
826 		html[i++] = calItem.invId;
827 		html[i++] = '"';
828 		html[i++] = ",\"";
829 		html[i++] = attach.part;
830 		html[i++] = "\"); return false;'";
831 	} else {
832 		html[i++] = hrefRoot;
833 		html[i++] = "'";
834 	}
835 	if (getLinkIdCallback) {
836 		var mainLinkId = getLinkIdCallback(attach.part, ZmCalItem.ATT_LINK_MAIN);
837 		html[i++] = " id='";
838 		html[i++] = mainLinkId;
839 		html[i++] = "'";
840 	}
841 	html[i++] = ">";
842 	html[i++] = AjxStringUtil.htmlEncode(attach.filename);
843 	html[i++] = "</a>";
844 
845 	var addHtmlLink = (appCtxt.get(ZmSetting.VIEW_ATTACHMENT_AS_HTML) &&
846 					   attach.body == null && ZmMimeTable.hasHtmlVersion(attach.ct));
847 
848 	if (sizeText || addHtmlLink) {
849 		html[i++] = " (";
850 		if (sizeText) {
851 			html[i++] = sizeText;
852 			html[i++] = ") ";
853 		}
854 		var downloadLinkId = "";
855 		if (getLinkIdCallback) {
856 			downloadLinkId = getLinkIdCallback(attach.part, ZmCalItem.ATT_LINK_DOWNLOAD);
857 		}
858 		if (addHtmlLink) {
859 			html[i++] = "<a style='text-decoration:underline' target='_blank' class='AttLink' ";
860 			if (getLinkIdCallback) {
861 				html[i++] = "id='";
862 				html[i++] = downloadLinkId;
863 				html[i++] = "' ";
864 			}
865 			html[i++] = hrefRoot;
866 			html[i++] = "&view=html'>";
867 			html[i++] = ZmMsg.preview;
868 			html[i++] = "</a> ";
869 		}
870 		if (attach.ct != ZmMimeTable.MSG_RFC822) {
871 			html[i++] = "<a style='text-decoration:underline' class='AttLink' onclick='ZmZimbraMail.unloadHackCallback();' ";
872 			if (getLinkIdCallback) {
873 				html[i++] = " id='";
874 				html[i++] = downloadLinkId;
875 				html[i++] = "' ";
876 			}
877 			html[i++] = hrefRoot;
878 			html[i++] = "&disp=a'>";
879 			html[i++] = ZmMsg.download;
880 			html[i++] = "</a>";
881 		}
882 	}
883 
884 	html[i++] = "</td></tr></table>";
885 
886 	// Provide lookup id and label for offline mode
887 	if (!attach.mid) {
888 		attach.mid = calItem.invId;
889 		attach.label = attach.filename;
890 	}
891 
892 	return html.join("");
893 };
