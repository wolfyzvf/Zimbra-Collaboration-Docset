 28 /**
 29  * This class is a base class for any controller that manages a list of items such as mail messages
 30  * or contacts. It can handle alternative views of the same list.
 31  *
 32  * @author Conrad Damon
 33  *
 34  * @param {DwtControl}					container					the containing shell
 35  * @param {ZmApp}						app							the containing application
 36  * @param {constant}					type						type of controller
 37  * @param {string}						sessionId					the session id
 38  * @param {ZmSearchResultsController}	searchResultsController		containing controller
 39  * 
 40  * @extends		ZmBaseController
 41  */
 42 ZmListController = function(container, app, type, sessionId, searchResultsController) {
 43 
 44 	if (arguments.length == 0) { return; }
 45 	ZmBaseController.apply(this, arguments);
 46 
 47 	// hashes keyed by view type
 48 	this._navToolBar = {};			// ZmNavToolBar
 49 	this._listView = this._view;	// ZmListView (back-compatibility for bug 60073)
 50 
 51 	this._list = null;				// ZmList
 52 	this._activeSearch = null;
 53 	this._newButton = null;
 54 	this._actionMenu = null;		// ZmActionMenu
 55 	this._actionEv = null;
 56 	
 57 	if (this.supportsDnD()) {
 58 		this._dropTgt = new DwtDropTarget("ZmTag");
 59 		this._dropTgt.markAsMultiple();
 60 		this._dropTgt.addDropListener(this._dropListener.bind(this));
 61 	}
 62 
 63 	this._menuPopdownListener = this._menuPopdownActionListener.bind(this);
 64 	
 65 	this._itemCountText = {};
 66 	this._continuation = {count:0, totalItems:0};
 67 };

 93 function(searchResults) {
 94 	
 95 	this._activeSearch = searchResults;

223 /**
224  * Handles the key action.
225  * 
226  * @param	{constant}	actionCode		the action code
227  * @return	{Boolean}	<code>true</code> if the action is handled
228  */
229 ZmListController.prototype.handleKeyAction =
230 function(actionCode, ev) {
231 
232 	DBG.println(AjxDebug.DBG3, "ZmListController.handleKeyAction");
233 	var listView = this._view[this._currentViewId];
234 	var result = false;
235 
236 	switch (actionCode) {
237 
238 		case DwtKeyMap.DBLCLICK:
239 			return listView.handleKeyAction(actionCode);
240 
241 		case ZmKeyMap.SHIFT_DEL:
242 		case ZmKeyMap.DEL:
243 			var tb = this.getCurrentToolbar();
244 			var button = tb && (tb.getButton(ZmOperation.DELETE) || tb.getButton(ZmOperation.DELETE_MENU));
245 			if (button && button.getEnabled()) {
246 				this._doDelete(this.getSelection(), (actionCode == ZmKeyMap.SHIFT_DEL));
247 				result = true;
248 			}
249 			break;
250 
251 		case ZmKeyMap.NEXT_PAGE:
252 			var ntb = this._navToolBar[this._currentViewId];
253 			var button = ntb ? ntb.getButton(ZmOperation.PAGE_FORWARD) : null;
254 			if (button && button.getEnabled()) {
255 				this._paginate(this._currentViewId, true);
256 				result = true;
257 			}
258 			break;
259 
260 		case ZmKeyMap.PREV_PAGE:
261 			var ntb = this._navToolBar[this._currentViewId];
262 			var button = ntb ? ntb.getButton(ZmOperation.PAGE_BACK) : null;
263 			if (button && button.getEnabled()) {
264 				this._paginate(this._currentViewId, false);
265 				result = true;
266 			}
267 			break;
268 
269 		// Esc pops search results tab
270 		case ZmKeyMap.CANCEL:
271 			var ctlr = this.isSearchResults && this.searchResultsController;
272 			if (ctlr) {
273 				ctlr._closeListener();
274 			}
275 			break;
276 
277 		default:
278 			return ZmBaseController.prototype.handleKeyAction.apply(this, arguments);
279 	}
280 	return result;
281 };

316 /**
317  * Initializes action menu: menu items and listeners
318  * 
319  * @private
320  */
321 ZmListController.prototype._initializeActionMenu =
322 function() {
323 
324 	if (this._actionMenu) { return; }
325 
326 	var menuItems = this._getActionMenuOps();
327 	if (!menuItems) { return; }
328 
329 	var menuParams = {parent:this._shell,
330 		menuItems:	menuItems,
331 		context:	this._getMenuContext(),
332 		controller:	this
333 	};
334 	this._actionMenu = new ZmActionMenu(menuParams);
335 	this._addMenuListeners(this._actionMenu);
336 	if (appCtxt.get(ZmSetting.TAGGING_ENABLED)) {
337 		this._setupTagMenu(this._actionMenu);
338 	}

341 /**
342  * @private
343  */
344 ZmListController.prototype._addMenuListeners =
345 function(menu) {
346 
347 	var menuItems = menu.opList;
348 	for (var i = 0; i < menuItems.length; i++) {
349 		var menuItem = menuItems[i];
350 		if (this._listeners[menuItem]) {
351 			menu.addSelectionListener(menuItem, this._listeners[menuItem], 0);
352 		}
353 	}
354 	menu.addPopdownListener(this._menuPopdownListener);
355 };
356 
357 ZmListController.prototype._menuPopdownActionListener =
358 function(ev) {
359 
360 	var view = this.getListView();
361 	if (!this._pendingActionData) {
362 		if (view && view.handleActionPopdown) {
363 			view.handleActionPopdown(ev);
364 		}
365 	}
366 	// Reset back to item count unless there is multiple selection
367 	var selCount = view ? view.getSelectionCount() : -1;
368 	if (selCount <= 1) {
369 		this._setItemCountText();
370 	}
371 };

377 /**
378  * List selection event - handle flagging if a flag icon was clicked, otherwise
379  * reset the toolbar based on how many items are selected.
380  * 
381  * @private
382  */
383 ZmListController.prototype._listSelectionListener =
384 function(ev) {
385 
386 	if (ev.field == ZmItem.F_FLAG) {
387 		this._doFlag([ev.item]);
388 		return true;
389 	} 
390 	else {
391 		var lv = this._listView[this._currentViewId];
392 		if (lv) {
393 			if (appCtxt.get(ZmSetting.SHOW_SELECTION_CHECKBOX) && !ev.ctrlKey) {
394 				if (lv.setSelectionHdrCbox) {
395 					lv.setSelectionHdrCbox(false);
396 				}
397 			}
398 			this._resetOperations(this.getCurrentToolbar(), lv.getSelectionCount());
399 			if (ev.shiftKey) {
400 				this._setItemSelectionCountText();
401 			}
402 			else {
403 				this._setItemCountText();
404 			}
405 		}
406 	}
407 	return false;
408 };

410 /**
411  * List action event - set the dynamic tag menu, and enable operations in the
412  * action menu based on the number of selected items. Note that the menu is not
413  * actually popped up here; that's left up to the subclass, which should
414  * override this function.

416  * @private

455 /**
456  * @private
457  */
458 ZmListController.prototype._dragListener =
459 function(ev) {
460 
461 	if (this.isSearchResults && ev.action == DwtDragEvent.DRAG_START) {
462 		this.searchResultsController.showOverview(true);
463 	}
464 	else if (ev.action == DwtDragEvent.SET_DATA) {
465 		ev.srcData = {data: ev.srcControl.getDnDSelection(), controller: this};
466 	}
467 	else if (this.isSearchResults && (ev.action == DwtDragEvent.DRAG_END || ev.action == DwtDragEvent.DRAG_CANCEL)) {
468 		this.searchResultsController.showOverview(false);
469 	}
470 };

472 /**
473  * The list view as a whole is the drop target, since it's the lowest-level widget. Still, we
474  * need to find out which item got dropped onto, so we get that from the original UI event
475  * (a mouseup). The header is within the list view, but not an item, so it's not a valid drop
476  * target. One drawback of having the list view be the drop target is that we can't exercise
477  * fine-grained control on what's a valid drop target. If you enter via an item and then drag to
478  * the header, it will appear to be valid.
479  * 
480  * @protected
481  */
482 ZmListController.prototype._dropListener =
483 function(ev) {
484 
485 	var view = this._view[this._currentViewId];
486 	var div = view.getTargetItemDiv(ev.uiEvent);
487 	var item = view.getItemFromElement(div);
488 
489 	// only tags can be dropped on us
490 	var data = ev.srcData.data;
491 	if (ev.action == DwtDropEvent.DRAG_ENTER) {
492 		ev.doIt = (item && (item instanceof ZmItem) && !item.isReadOnly() && this._dropTgt.isValidTarget(data));
493         // Bug: 44488 - Don't allow dropping tag of one account to other account's item
494         if (appCtxt.multiAccounts) {
495            var listAcctId = item ? item.getAccount().id : null;
496            var tagAcctId = (data.account && data.account.id) || data[0].account.id;
497            if (listAcctId != tagAcctId) {
498                ev.doIt = false;
499            }
500         }
501 		DBG.println(AjxDebug.DBG3, "DRAG_ENTER: doIt = " + ev.doIt);
502 		if (ev.doIt) {
503 			view.dragSelect(div);
504 		}
505 	} else if (ev.action == DwtDropEvent.DRAG_DROP) {
506 		view.dragDeselect(div);
507 		var items = [item];
508 		var sel = this.getSelection();
509 		if (sel.length) {
510 			var vec = AjxVector.fromArray(sel);
511 			if (vec.contains(item)) {
512 				items = sel;
513 			}
514 		}
515 		this._doTag(items, data, true);
516 	} else if (ev.action == DwtDropEvent.DRAG_LEAVE) {
517 		view.dragDeselect(div);
518 	} else if (ev.action == DwtDropEvent.DRAG_OP_CHANGED) {
519 		// nothing
520 	}
521 };

582 /**
583  * Manages the progress dialog that appears when an action is performed on a large number of items.
584  * The arguments include a state and any arguments relative to that state. The state is one of:
585  * 
586  * 			ZmListController.PROGRESS_DIALOG_INIT
587  *			ZmListController.PROGRESS_DIALOG_UPDATE
588  *			ZmListController.PROGRESS_DIALOG_CLOSE
589  *  
590  * @param {hash}		params		a hash of params:
591  * @param {constant}	state		state of the dialog
592  * @param {AjxCallback}	callback	cancel callback (INIT)
593  * @param {string}		summary		summary text (UPDATE)
594  */
595 ZmListController.handleProgress =
596 function(params) {
597 
598 	var dialog = appCtxt.getCancelMsgDialog();
599 	if (params.state == ZmListController.PROGRESS_DIALOG_INIT) {
600 		dialog.reset();
601 		dialog.registerCallback(DwtDialog.CANCEL_BUTTON, params.callback);
602 		ZmListController.progressDialogReady = true;
603 	}
604 	else if (params.state == ZmListController.PROGRESS_DIALOG_UPDATE && ZmListController.progressDialogReady) {
605 		dialog.setMessage(params.summary, DwtMessageDialog.INFO_STYLE, AjxMessageFormat.format(ZmMsg.inProgress));
606 		if (!dialog.isPoppedUp()) {
607 			dialog.popup();
608 		}
609 	}
610 	else if (params.state == ZmListController.PROGRESS_DIALOG_CLOSE) {
611 		dialog.unregisterCallback(DwtDialog.CANCEL_BUTTON);
612 		dialog.popdown();
613 		ZmListController.progressDialogReady = false;
614 	}
615 };

635 /**
636  * @private
637  */
638 ZmListController.prototype._search =
639 function(view, offset, limit, callback, isCurrent, lastId, lastSortVal) {
640 	var originalSearch = this._activeSearch && this._activeSearch.search;
641 	var params = {
642 		query:			this.getSearchString(),
643 		queryHint:		this.getSearchStringHint(),
644 		types:			originalSearch && originalSearch.types || [], // use types from original search
645 		userInitiated:	originalSearch && originalSearch.userInitiated,
646 		sortBy:			appCtxt.get(ZmSetting.SORTING_PREF, view),
647 		offset:			offset,
648 		limit:			limit,
649 		lastId:			lastId,
650 		lastSortVal:	lastSortVal
651 	};
652 	// add any additional params...
653 	this._getMoreSearchParams(params);
654 
655 	var search = new ZmSearch(params);
656 	if (isCurrent) {
657 		this._currentSearch = search;
658 	}
659 
660 	appCtxt.getSearchController().redoSearch(search, true, null, callback);
661 };

663 /**
664  * Gets next or previous page of items. The set of items may come from the
665  * cached list, or from the server (using the current search as a base).
666  * <p>
667  * The loadIndex is the index'd item w/in the list that needs to be loaded -
668  * initiated only when user is in CV and pages a conversation that has not
669  * been loaded yet.</p>
670  * <p>
671  * Note that this method returns a value even though it may make an
672  * asynchronous SOAP request. That's possible as long as no caller
673  * depends on the results of that request. Currently, the only caller that
674  * looks at the return value acts on it only if no request was made.</p>
675  *
676  * @param {constant}	view		the current view
677  * @param {Boolean}	forward		if <code>true</code>, get next page rather than previous
678  * @param {int}		loadIndex	the index of item to show
679  * @param {int}	limit		the number of items to fetch
680  * 
681  * @private
682  */
683 ZmListController.prototype._paginate =
684 function(view, forward, loadIndex, limit) {
685 
686 	var needMore = false;
687 	var lv = this._view[view];
688 	if (!lv) { return; }
689 	var offset, max;
690 
691     limit = limit || lv.getLimit(offset);
692 
693 	if (lv._isPageless) {
694 		offset = this._list.size();
695 		needMore = true;
696 	} else {
697 		offset = lv.getNewOffset(forward);
698 		needMore = (offset + limit > this._list.size());
699 		this.currentPage = this.currentPage + (forward ? 1 : -1);
700 		this.maxPage = Math.max(this.maxPage, this.currentPage);
701 	}
702 
703 	// see if we're out of items and the server has more
704 	if (needMore && this._list.hasMore()) {
705 		lv.offset = offset; // cache new offset
706 		if (lv._isPageless) {
707 			max = limit;
708 		} else {
709 			// figure out how many items we need to fetch
710 			var delta = (offset + limit) - this._list.size();
711 			max = delta < limit && delta > 0 ? delta : limit;
712 			if (max < limit) {
713 				offset = ((offset + limit) - max) + 1;
714 			}
715 		}
716 
717 		// handle race condition - user has paged quickly and we don't want
718 		// to do second fetch while one is pending
719 		if (this._searchPending) { return false;	}
720 
721 		// figure out if this requires cursor-based paging
722 		var list = lv.getList();
723 		var lastItem = list && list.getLast();
724 		var lastSortVal = (lastItem && lastItem.id) ? lastItem.sf : null;
725 		var lastId = lastSortVal ? lastItem.id : null;
726 
727 		this._setItemCountText(ZmMsg.loading);
728 
729 		// get next page of items from server; note that callback may be overridden
730 		this._searchPending = true;
731 		var respCallback = this._handleResponsePaginate.bind(this, view, false, loadIndex, offset);
732 		this._search(view, offset, max, respCallback, true, lastId, lastSortVal);
733 		return false;
734 	} else if (!lv._isPageless) {
735 		lv.offset = offset; // cache new offset
736 		this._resetOperations(this._toolbar[view], 0);
737 		this._resetNavToolBarButtons(view);
738 		this._setViewContents(view);
739 		this._resetSelection();
740 		return true;
741 	}
742 };

744 /**
745  * Updates the list and the view after a new page of items has been retrieved.
746  *
747  * @param {constant}	view				the current view
748  * @param {Boolean}	saveSelection			if <code>true</code>, maintain current selection
749  * @param {int}	loadIndex				the index of item to show
750  * @param {ZmCsfeResult}	result			the result of SOAP request
751  * @param {Boolean}	ignoreResetSelection	if <code>true</code>, do not reset selection
752  * 
753  * @private
754  */
755 ZmListController.prototype._handleResponsePaginate =
756 function(view, saveSelection, loadIndex, offset, result, ignoreResetSelection) {
757 
758 	var searchResult = result.getResponse();
759 
760 	// update "more" flag
761 	this._list.setHasMore(searchResult.getAttribute("more"));
762 
763 	this._cacheList(searchResult, offset);
764 
765 	var lv = this._view[this._currentViewId];
766 	var num = lv._isPageless ? this.getSelectionCount() : 0;
767 	this._resetOperations(this._toolbar[view], num);
768 
769 	// remember selected index if told to
770 	var selItem = saveSelection ? this.getSelection()[0] : null;
771 	var selectedIdx = selItem ? lv.getItemIndex(selItem) : -1;
772 
773 	var items = searchResult && searchResult.getResults().getArray();
774 	if (lv._isPageless && items && items.length) {
775 		lv._itemsToAdd = items;
776 	} else {
777 		lv._itemsToAdd = null;
778 	}
779 	var wasEmpty = (lv._isPageless && (lv.size() == 0));
780 
781 	this._setViewContents(view);
782 
783 	// add new items to selection if all results selected, in a way that doesn't call deselectAll()
784 	if (lv.allSelected) {
785 		for (var i = 0, len = items.length; i < len; i++) {
786 			lv.selectItem(items[i], true);
787 			lv.setSelectionCbox(items[i], false);
788 		}
789 		lv.setSelectionHdrCbox(true);
790 		DBG.println("scr", "pagination - selected more items: " + items.length);
791 		DBG.println("scr", "items selected: " + this.getSelectionCount());
792 	}
793 	this._resetNavToolBarButtons(view);
794 
795 	// bug fix #5134 - some views may not want to reset the current selection
796 	if (!ignoreResetSelection && !lv._isPageless) {
797 		this._resetSelection(selectedIdx);
798 	} else if (wasEmpty) {
799 		lv._setNextSelection();
800 	}
801 
802 	if (!appCtxt.isChildWindow) {
803 		appCtxt.getAppController().focusContentPane();
804 	}
805 	this._searchPending = false;
806 };

855 /**

857  */
858 ZmListController.prototype._replenishList =
859 function(view, replCount, callback) {
860 
861 	// determine if there are any more items to replenish with
862 	var idxStart = this._view[view].offset + this._view[view].size();
863 	var totalCount = this._list.size();
864 
865 	if (idxStart < totalCount) {

867 		var idxEnd = (idxEnd > totalCount) ? totalCount : (idxStart + replCount);
868 		var list = this._list.getVector().getArray();
869 		var sublist = list.slice(idxStart, idxEnd);
870 		var subVector = AjxVector.fromArray(sublist);
871 		this._view[view].replenish(subVector);
872 		if (callback) {
873 			callback.run();
874 		}
875 	} else {
876 		// replenish from server request
877 		this._getMoreToReplenish(view, replCount, callback);

1191 /**
1192  * Records total items and last item before we do any more searches. Adds a couple
1193  * params to the args for the list action method.
1194  *
1195  * @param {function}	actionMethod		the controller action method
1196  * @param {Array}		args				an arg list for above (except for items arg)
1197  * @param {Hash}		params				the params that will be passed to list action method
1198  * @param {closure}		allDoneCallback		the callback to run after all items processed
1199  * 
1200  * @private
1201  */
1202 ZmListController.prototype._setupContinuation =
1203 function(actionMethod, args, params, allDoneCallback, notIdsOnly) {
1204 
1205 	// need to use AjxCallback here so we can prepend items arg when calling it
1206 	var actionCallback = new AjxCallback(this, actionMethod, args);
1207 	params.finalCallback = this._continueAction.bind(this, {actionCallback:actionCallback, allDoneCallback:allDoneCallback, notIdsOnly: notIdsOnly});
1208 	
1209 	params.count = this._continuation.count;
1210 	params.idsOnly = !notIdsOnly;
1211 
1212 	if (!this._continuation.lastItem) {
1213 		this._continuation.lastItem = params.list.getVector().getLast();
1214 		this._continuation.totalItems = params.list.size();
1215 	}
1216 };

1218 /**
1219  * See if we are performing an action on all items, including ones that match the current search
1220  * but have not yet been retrieved. If so, keep doing searches and performing the action on the
1221  * results, until there are no more results.
1222  *
1223  * The arguments in the action callback should be those after the initial 'items' argument. The
1224  * array of items retrieved by the search is prepended to the callback's argument list before it
1225  * is run.
1226  *
1227  * @param {Hash}		params				a hash of parameters
1228  * @param {AjxCallback}	actionCallback		the callback with action to be performed on search results
1229  * @param {closure} 	allDoneCallback		the callback to run when we're all done
1230  * @param {Hash}		actionParams		the params from <code>ZmList._itemAction</code>, added when this is called
1231  * 
1232  * @private
1233  */
1234 ZmListController.prototype._continueAction =
1235 function(params, actionParams) {
1236 
1237 	var lv = this._view[this._currentViewId];
1238 	var cancelled = actionParams && actionParams.cancelled;
1239 	var contResult = this._continuation.result;
1240 	var hasMore = contResult ? contResult.getAttribute("more") : (this._list ? this._list.hasMore() : false);
1241 	DBG.println("sa", "lv.allSelected: " + lv.allSelected + ", hasMore: " + hasMore);
1242 	if (lv.allSelected && hasMore && !cancelled) {
1243 		var cs = this._currentSearch;
1244 		var limit = ZmListController.CONTINUATION_SEARCH_ITEMS;
1245 		var searchParams = {
1246 			query:		this.getSearchString(),
1247 			queryHint:	this.getSearchStringHint(),
1248 			types:		cs.types,
1249 			sortBy:		cs.sortBy,
1250 			limit:		limit,
1251 			idsOnly:	!params.notIdsOnly
1252 		};
1253 
1254 		var list = contResult ? contResult.getResults() : this._list.getArray();
1255 		var lastItem = this._continuation.lastItem;
1256 		if (!lastItem) {
1257 			lastItem = list && list[list.length - 1];
1258 		}
1259 		if (lastItem) {
1260 			searchParams.lastId = lastItem.id;
1261 			searchParams.lastSortVal = lastItem.sf;
1262 			DBG.println("sa", "***** continuation search: " + searchParams.query + " --- " + [lastItem.id, lastItem.sf].join("/"));
1263 		} else {
1264 			searchParams.offset = limit + (this._continuation.search ? this._continuation.search.offset : 0);
1265 		}
1266 
1267 		this._continuation.count = actionParams.numItems;
1268 		if (!this._continuation.totalItems) {
1269 			this._continuation.totalItems = list.length;
1270 		}
1271 
1272 		this._continuation.search = new ZmSearch(searchParams);
1273 		var respCallback = this._handleResponseContinueAction.bind(this, params.actionCallback);
1274 		appCtxt.getSearchController().redoSearch(this._continuation.search, true, null, respCallback);
1275 	} else {
1276 		DBG.println("sa", "end of continuation");
1277 		if (contResult) {
1278 			if (lv.allSelected) {
1279 				// items beyond page were acted on, give user a total count
1280 				if (actionParams.actionTextKey) {
1281 					var type = contResult.type;
1282 					if (type === ZmId.SEARCH_MAIL) {
1283 						type = this._list.type; //get the specific CONV/MSG type instead of the "searchFor" "MAIL".
1284 					}
1285 					actionParams.actionSummary = ZmList.getActionSummary({
1286 						actionTextKey:  actionParams.actionTextKey,
1287 						numItems:       this._continuation.totalItems,
1288 						type:           type,
1289 						actionArg:      actionParams.actionArg
1290 					});
1291 				}
1292 				lv.deselectAll();
1293 			}
1294 			this._continuation = {count:0, totalItems:0};
1295 		}
1296 		if (params.allDoneCallback) {
1297 			params.allDoneCallback();
1298 		}
1299 
1300 		ZmListController.handleProgress({state:ZmListController.PROGRESS_DIALOG_CLOSE});
1301 		ZmBaseController.showSummary(actionParams.actionSummary, actionParams.actionLogItem, actionParams.closeChildWin);
1302 	}
1303 };
