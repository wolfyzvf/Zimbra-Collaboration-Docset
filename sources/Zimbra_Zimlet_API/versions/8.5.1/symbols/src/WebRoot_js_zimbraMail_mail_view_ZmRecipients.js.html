 22 /**
 23  * @class
 24  * This class provides a central area for managing email recipient fields. It is not a control,
 25  * and does not exist within the widget hierarchy.
 26  * 
 27  * @param {hash}		params						a hash of params:
 28  * @param {function}	resetContainerSizeMethod	callback for when size needs to be adjusted
 29  * @param {function}	enableContainerInputs		callback for enabling/disabling input fields
 30  * @param {function}	reenter						callback to enable design mode
 31  * @param {AjxListener}	contactPopdownListener		listener called when contact picker pops down
 32  * @param {string}		contextId					ID of owner (used for autocomplete list)
 33  */
 34 ZmRecipients = function(params) {
 35 
 36 	this._divId			    = {};
 37 	this._buttonTdId	    = {};
 38 	this._fieldId		    = {};
 39 	this._using			    = {};
 40 	this._button		    = {};
 41 	this._field			    = {};
 42 	this._divEl			    = {};
 43 	this._addrInputField    = {};
 44 
 45     this._resetContainerSize = params.resetContainerSizeMethod;
 46     this._enableContainerInputs = params.enableContainerInputs;
 47     this._reenter = params.reenter;
 48     this._contactPopdownListener = params.contactPopdownListener;
 49 	this._contextId = params.contextId;
 50 
 51     this._bubbleOps = {};
 52     this._bubbleOps[AjxEmailAddress.TO]  = ZmOperation.MOVE_TO_TO;
 53     this._bubbleOps[AjxEmailAddress.CC]  = ZmOperation.MOVE_TO_CC;
 54     this._bubbleOps[AjxEmailAddress.BCC] = ZmOperation.MOVE_TO_BCC;
 55     this._opToField = {};
 56     this._opToField[ZmOperation.MOVE_TO_TO]  = AjxEmailAddress.TO;
 57     this._opToField[ZmOperation.MOVE_TO_CC] = AjxEmailAddress.CC;
 58     this._opToField[ZmOperation.MOVE_TO_BCC] = AjxEmailAddress.BCC;
 59 };

 74 ZmRecipients.prototype.createRecipientIds =
 75 function(htmlElId, typeStr) {
 76     var ids = {};
 77     var components = ["row", "picker", "control", "cell"];
 78     for (var i = 0; i < components.length; i++) {
 79         ids[components[i]] = [htmlElId, typeStr, components[i]].join("_")
 80     }
 81     return ids;
 82 }

 85 ZmRecipients.prototype.createRecipientHtml =
 86 function(parent, viewId, htmlElId, fieldNames) {
 87 
 88     this._fieldNames = fieldNames;
 89 	var contactsEnabled = appCtxt.get(ZmSetting.CONTACTS_ENABLED);
 90 	var galEnabled = appCtxt.get(ZmSetting.GAL_ENABLED);
 91 
 92     	// init autocomplete list
 93     if (contactsEnabled || galEnabled || appCtxt.isOffline) {
 94 		var params = {
 95 			dataClass:		appCtxt.getAutocompleter(),
 96 			matchValue:		ZmAutocomplete.AC_VALUE_FULL,
 97 			keyUpCallback:	this._acKeyupHandler.bind(this),
 98 			contextId:		this._contextId
 99 		};
100 		this._acAddrSelectList = new ZmAutocompleteListView(params);
101 	}
102 
103 	var isPickerEnabled = contactsEnabled || galEnabled || appCtxt.multiAccounts;	
104 	
105 	this._pickerButton = {};
106 
107 	// process compose fields
108 	for (var i = 0; i < fieldNames.length; i++) {
109 		var type = fieldNames[i];
110 		var typeStr = AjxEmailAddress.TYPE_STRING[type];
111 
112 		// save identifiers
113         var ids = this.createRecipientIds(htmlElId, typeStr);
114 		this._divId[type] = ids.row;
115 		this._buttonTdId[type] = ids.picker;
116 		var inputId = this._fieldId[type] = ids.control;
117 
118 		// save field elements
119 		this._divEl[type] = document.getElementById(this._divId[type]);
120 		var aifId;
121 		var aifParams = {
122 			parent:								parent,
123 			autocompleteListView:				this._acAddrSelectList,
124 			bubbleAddedCallback:				this._bubblesChangedCallback.bind(this),
125 			bubbleRemovedCallback:				this._bubblesChangedCallback.bind(this),
126 			bubbleMenuCreatedCallback:			this._bubbleMenuCreated.bind(this),
127 			bubbleMenuResetOperationsCallback:	this._bubbleMenuResetOperations.bind(this),
128 			inputId:							inputId,
129 			type:								type
130 		}
131 		var aif = this._addrInputField[type] = new ZmAddressInputField(aifParams);
132 		aifId = aif._htmlElId;
133 		aif.reparentHtmlElement(ids.cell);
134 
135 		// save field control
136 		this._field[type] = document.getElementById(this._fieldId[type]);
137 		if (this._field[type]) {
138 			this._field[type].addrType = type;
139 		}
140 
141 		// create picker
142 		if (isPickerEnabled) {
143 
144 			// bug 78318 - if GAL enabled but not contacts, we need some things defined to handle GAL search
145 			if (!contactsEnabled) {
146 				appCtxt.getAppController()._createApp(ZmApp.CONTACTS);
147 			}
148 
149 			var pickerId = this._buttonTdId[type];
150 			var pickerEl = document.getElementById(pickerId);
151 			if (pickerEl) {
152 				var buttonId = ZmId.getButtonId(viewId, ZmRecipients.OP[type]);
153 				var button = this._pickerButton[type] = new DwtButton({parent:parent, id:buttonId});
154 				button.setText(pickerEl.innerHTML);
155 				button.replaceElement(pickerEl);
156 
157 				button.addSelectionListener(this.addressButtonListener.bind(this));
158 				button.addrType = type;
159 
160 				// autocomplete-related handlers
161 				if (contactsEnabled || appCtxt.isOffline) {
162 					this._acAddrSelectList.handle(this._field[type], aifId);
163 				} else {
164 					this._setEventHandler(this._fieldId[type], "onKeyUp");
165 				}
166 
167 				this._button[type] = button;
168 			}
169 		}
170 	}
171 };

234 // Adds the given addresses to the form. We need to add each address separately in case it's a DL.
235 ZmRecipients.prototype.addAddresses =
236 function(type, addrVec, used) {
237 
238 	var addrAdded = false;
239 	used = used || {};
240 	var addrList = [];
241 	var addrs = AjxUtil.toArray(addrVec);
242 	if (addrs && addrs.length) {
243 		for (var i = 0, len = addrs.length; i < len; i++) {
244 			var addr = addrs[i];
245 			var email = addr.isAjxEmailAddress ? addr && addr.getAddress() : addr;
246 			if (!email) { continue; }
247 			email = email.toLowerCase();
248 			if (!used[email]) {
249 				this.setAddress(type, addr);	// add the bubble now
250 				used[email] = true;
251 				addrAdded = true;
252 			}
253 		}
254 	}
255 	return addrAdded;
256 };

259 /**
260  * Sets an address field.
261  *
262  * @param type	the address type
263  * @param addr	the address string
264  *
265  * XXX: if addr empty, check if should hide field
266  *
267  * @private
268  */
269 ZmRecipients.prototype.setAddress =
270 function(type, addr) {
271 
272 	addr = addr || "";
273 
274 	var addrStr = addr.isAjxEmailAddress ? addr.toString() : addr;
275 
276 	//show first, so focus works on IE.
277 	if (addrStr.length && !this._using[type]) {
278 		this._using[type] = true;
279 		this._showAddressField(type, true);
280 	}
281 
282 	var addrInput = this._addrInputField[type];
283 	if (!addrStr) {
284 		addrInput.clear();
285 	}
286 	else {
287 		if (addr.isAjxEmailAddress) {
288 			var match = {isDL: addr.isGroup && addr.canExpand, email: addrStr};
289 			addrInput.addBubble({address:addrStr, match:match, skipNotify:true, noFocus:true});
290 		}
291 		else {
292 			this._setAddrFieldValue(type, addrStr);
293 		}
294 	}
295 };

329 // Grab the addresses out of the form. Optionally, they can be returned broken
330 // out into good and bad addresses, with an aggregate list of the bad ones also
331 // returned. If the field is hidden, its contents are ignored.
332 ZmRecipients.prototype.collectAddrs =
333 function() {
334 
335 	var addrs = {};
336 	addrs[ZmRecipients.BAD] = new AjxVector();
337 	for (var i = 0; i < this._fieldNames.length; i++) {
338 		var type = this._fieldNames[i];
339 
340 		if (!this._field[type]) { //this check is in case we don't have all fields set up (might be configurable. Didn't look deeply).
341 			continue;
342 		}
343 
344 		var val = this.getAddrFieldValue(type);
345 		if (val.length == 0) { continue; }
346 		val = val.replace(/[; ,]+$/, "");	// ignore trailing (and possibly extra) separators
347 		var result = AjxEmailAddress.parseEmailString(val, type, false);
348 		if (result.all.size() == 0) { continue; }
349 		addrs.gotAddress = true;
350 		addrs[type] = result;
351 		if (result.bad.size()) {
352 			addrs[ZmRecipients.BAD].addList(result.bad);
353 			if (!addrs.badType) {
354 				addrs.badType = type;
355 			}
356 		}
357 	}
358 	return addrs;
359 };

376 // Address buttons invoke contact picker
377 ZmRecipients.prototype.addressButtonListener =
378 function(ev, addrType) {
379 	if (appCtxt.isWebClientOffline()) return;
380 
381 	var obj = ev ? DwtControl.getTargetControl(ev) : null;
382 	if (this._enableContainerInputs) {
383 		this._enableContainerInputs(false);
384 	}
385 
386 	if (!this._contactPicker) {
387 		AjxDispatcher.require("ContactsCore");
388 		var buttonInfo = [];
389         for (var i = 0; i < this._fieldNames.length; i++) {
390             buttonInfo[i] = { id: this._fieldNames[i],
391                               label : ZmMsg[AjxEmailAddress.TYPE_STRING[this._fieldNames[i]]]};
392         }
393 		this._contactPicker = new ZmContactPicker(buttonInfo);
394 		this._contactPicker.registerCallback(DwtDialog.OK_BUTTON, this._contactPickerOkCallback, this);
395 		this._contactPicker.registerCallback(DwtDialog.CANCEL_BUTTON, this._contactPickerCancelCallback, this);
396 	}
397 
398 	var curType = obj ? obj.addrType : addrType;
399 	var addrList = {};
400 	for (var i = 0; i < this._fieldNames.length; i++) {
401 		var type = this._fieldNames[i];
402 		addrList[type] = this._addrInputField[type].getAddresses(true);
403 	}
404 	if (this._contactPopdownListener) {
405 		this._contactPicker.addPopdownListener(this._contactPopdownListener);
406 	}
407 	var str = (this._field[curType].value && !(addrList[curType] && addrList[curType].length))
408 		? this._field[curType].value : "";
409 
410 	var account;
411 	if (appCtxt.multiAccounts && this._fromSelect) {
412 		var addr = this._fromSelect.getSelectedOption().addr;
413 		account = appCtxt.accountList.getAccountByEmail(addr.address);
414 	}
415 	this._contactPicker.popup(curType, addrList, str, account);
416 };

435 ZmRecipients.prototype._acKeyupHandler =
436 function(ev, acListView, result, element) {
437 	var key = DwtKeyEvent.getCharCode(ev);
438 	// process any printable character or enter/backspace/delete keys
439 	if (result && element && (ev.inputLengthChanged ||
440 		(key == 3 || key == 13 || key == 8 || key == 46 ||
441 		(AjxEnv.isMac && key == 224)))) // bug fix #24670
442 	{
443 		element.value = element.value && element.value.replace(/;([^\s])/g, function(all, group){return "; "+group}) || ""; // Change ";" to "; " if it is not succeeded by a whitespace
444 	}
445 };

457 ZmRecipients.prototype._bubbleMenuCreated =
458 function(addrInput, menu) {
459 
460 	this._bubbleActionMenu = menu;

474 ZmRecipients.prototype._bubbleMenuResetOperations =
475 function(addrInput, menu) {
476 	var sel = addrInput.getSelection();
477     for (var i = 0; i < this._fieldNames.length; i++) {
478         var type = this._fieldNames[i];
479 		var op = this._bubbleOps[type];

484 ZmRecipients.prototype._bubbleMove =
485 function(ev) {
486 
487 	var sourceInput = ZmAddressInputField.menuContext.addrInput;
488 	var op = ev && ev.item && ev.item.getData(ZmOperation.KEY_ID);
489 	var type = this._opToField[op];
490 	var targetInput = this._addrInputField[type];
491 	if (sourceInput && targetInput) {
492 		var sel = sourceInput.getSelection();
493 		if (sel.length) {
494 			for (var i = 0; i < sel.length; i++) {
495 				var bubble = sel[i];
496 				this._showAddressField(type, true);
497 				targetInput.addBubble({bubble:bubble});
498 				sourceInput.removeBubble(bubble.id);
499 			}
500 		}
501 	}
502 };

540 // Transfers addresses from the contact picker to the compose view.
541 ZmRecipients.prototype._contactPickerOkCallback =
542 function(addrs) {
543 
544 	if (this._enableContainerInputs) {
545 		this._enableContainerInputs(true);
546 	}
547 	for (var i = 0; i < this._fieldNames.length; i++) {
548 		var type = this._fieldNames[i];
549 		this.setAddress(type, "");
550         // If there was only one button, the picker will just return the list of selections,
551         // not a list per button type
552         var typeAddrs = (this._fieldNames.length == 1) ? addrs :  addrs[type];
553 		var addrVec = ZmRecipients.expandAddrs(typeAddrs);
554 		this.addAddresses(type, addrVec);
555 	}
556 
557 	// Still need this here since REMOVING stuff with the picker does not call removeBubble in the ZmAddressInputField.
558 	// Also - it's better to do it once than for every bubble in this case. user might add many addresses with the picker
559 	this._bubblesChangedCallback();
560 
561 	if (this._contactPopdownListener) {
562 		this._contactPicker.removePopdownListener(this._contactPopdownListener);
563 	}
564 	this._contactPicker.popdown();
565 	if (this._reenter) {
566 		this._reenter();
567 	}
568 };

570 // Expands any addresses that are groups
571 ZmRecipients.expandAddrs =
572 function(addrs) {
573 	var addrsNew = [];
574 	var addrsArray = (addrs instanceof AjxVector) ? addrs.getArray() : addrs;
575 	if (addrsArray && addrsArray.length) {
576 		for (var i = 0; i < addrsArray.length; i++) {
577 			var addr = addrsArray[i];
578 			if (addr) {
579 				if (addr.isGroup && !(addr.__contact && addr.__contact.isDL)) {
580 					var members = addr.__contact ? addr.__contact.getGroupMembers().good.getArray() :
581 												   AjxEmailAddress.split(addr.address);
582 					addrsNew = addrsNew.concat(members);
583 				}
584 				else {
585 					addrsNew.push(addr);
586 				}
587 			}
588 		}
589 	}
590 	return AjxVector.fromArray(addrsNew);
591 };
