 22 /**
 23  * Creates a calendar.
 24  * @constructor
 25  * @class
 26  *
 27  * @author Andy Clark
 28  *
 29  * @param {Hash}		params			a hash of parameters:
 30  * @param {int}			params.id		the numeric ID
 31  * @param {String}		params.name		the name
 32  * @param {ZmOrganizer}	params.parent	the parent organizer
 33  * @param {ZmTree}		params.tree		the tree model that contains this organizer
 34  * @param {constant}	params.color	the color for this calendar
 35  * @param {String}		params.url		the URL for this organizer's feed
 36  * @param {String}		params.owner	the owner of this calendar
 37  * @param {String}		params.zid		the Zimbra id of owner, if remote share
 38  * @param {String}		params.rid		the remote id of organizer, if remote share
 39  * @param {String}		params.restUrl	the REST URL of this organizer
 40  * 
 41  * @extends		ZmFolder
 42  */
 43 ZmCalendar = function(params) {
 44 	params.type = ZmOrganizer.CALENDAR;
 45 	ZmFolder.call(this, params);
 46 };

 60 /**
 61  * Creates a new calendar. The color and flags will be set later in response
 62  * to the create notification. This function is necessary because calendar
 63  * creation needs custom error handling.
 64  * 
 65  * @param	{Hash}	params		a hash of parameters
 66  */
 67 ZmCalendar.create =
 68 function(params) {
 69 	params.errorCallback = new AjxCallback(null, ZmCalendar._handleErrorCreate, params);
 70 	ZmOrganizer.create(params);
 71 };
 72 
 73 ZmCalendar._handleErrorCreate =
 74 function(params, ex) {
 75 	if (params.url && (ex.code == ZmCsfeException.SVC_PARSE_ERROR)) {
 76 		msg = AjxMessageFormat.format(ZmMsg.calFeedInvalid, params.url);
 77 		ZmOrganizer._showErrorMsg(msg);
 78 		return true;
 79 	}
 80 	return ZmOrganizer._handleErrorCreate(params, ex);
 81 };

 95 /**
 96  * Sets the free/busy.
 97  * 
 98  * @param	{Boolean}	        exclude		    if <code>true</code>, exclude free busy
 99  * @param	{AjxCallback}	    callback		the callback
100  * @param	{AjxCallback}	    errorCallback	the error callback
101  * @param   {ZmBatchCommand}    batchCmd        optional batch command
102  */
103 ZmCalendar.prototype.setFreeBusy = 
104 function(exclude, callback, errorCallback, batchCmd) {
105 	if (this.excludeFreeBusy == exclude) { return; }
106 	// NOTE: Don't need to store the value since the response will
107 	//       report that the object was modified.
108 	this._organizerAction({action: "fb", attrs: {excludeFreeBusy: exclude ? "1" : "0"},
109                            callback: callback, errorCallback: errorCallback, batchCmd: batchCmd});
110 };

112 ZmCalendar.prototype.setChecked = 
113 function(checked, batchCmd) {
114 	if (this.isChecked == checked) { return; }
115 	this.checkAction(checkAction, batchCmd);
116 };
117 
118 ZmCalendar.prototype.checkAction = 
119 function(checked, batchCmd) {
120 	var action = checked ? "check" : "!check";
121 	var checkedCallback = new AjxCallback(this, this.checkedCallback, [checked]);
122 	this._organizerAction({action: action, batchCmd: batchCmd,callback: checkedCallback});
123 };
124 
125 ZmCalendar.prototype.checkedCallback = 
126 function(checked, result) {
127 	var overviewController = appCtxt.getOverviewController();
128 	var treeController = overviewController.getTreeController(this.type);
129 	var overviewId = appCtxt.getCurrentApp().getOverviewId();
130 	var treeView = treeController.getTreeView(overviewId);
131 
132 	if (treeView && this.id && treeView._treeItemHash[this.id]) {
133 		treeView._treeItemHash[this.id].setChecked(checked);
134 	}
135 };

137 /**
138  * Checks if the given object(s) may be placed in this folder.
139  *
140  * For calendars being dragged, the current target cannot:
141  *   - Be the parent of the dragged calendar
142  *   - Be the dragged calendar
143  *   - Be an ancestor of the dragged calendar
144  *   - Contain a calendar with the same name as the dragged calendar
145  *   - Be a shared calendar
146  *
147  * @param {Object}	what		the object(s) to possibly move into this folder (item or organizer)
148  * @return	{Boolean}	<code>true</code> if the object may be placed in this folder
149  */
150 ZmCalendar.prototype.mayContain =
151 function(what) {
152     if (!what) { return true; }
153 
154     var invalid = false;
155     if (what instanceof ZmCalendar) {
156         // Calendar DnD, possibly nesting calendars
157         invalid = ((what.parent == this) ||  (what.id == this.id)  || this.isChildOf(what) ||
158                    (!this.isInTrash() && this.hasChild(what.name)) || this.link);
159     } else {
160         //exclude the deleted folders
161         if(this.noSuchFolder) return invalid;
162 
163 		if (this.nId == ZmOrganizer.ID_ROOT) {
164 			// cannot drag anything onto root folder
165 			invalid = true;
166 		} else if (this.link) {
167 			// cannot drop anything onto a read-only addrbook
168 			invalid = this.isReadOnly();
169 		}
170 
171 		if (!invalid) {
172 			// An item or an array of items is being moved
173 			var items = (what instanceof Array) ? what : [what];
174 			var item = items[0];
175 
176 			// can't move items to folder they're already in; we're okay if
177 			// we have one item from another folder
178             if (item && item.folderId) {
179                 invalid = true;
180                 for (var i = 0; i < items.length; i++) {
181                     var folder = appCtxt.getById(items[i].folderId);
182                     if(items[i].isReadOnly() && folder.owner != this.owner) {
183                         invalid = true;
184                         break;
185                     }
186                     if (item.viewMode == ZmCalItem.MODE_NEW || folder != this) {
187                         invalid = false;
188                         break;
189                     }
190 
191                 }
192             }
193 		}
194 
195 	}
196 
197 	return !invalid;
198 };

203 ZmCalendar.prototype.notifyCreate =
204 function(obj) {
205 	var calendar = ZmFolderTree.createFromJs(this, obj, this.tree);
206 	var index = ZmOrganizer.getSortIndex(calendar, ZmCalendar.sortCompare);
207 	this.children.add(calendar, index);
208 	calendar._notify(ZmEvent.E_CREATE);
209 };

211 ZmCalendar.prototype.notifyModify =
212 function(obj) {
213 	ZmFolder.prototype.notifyModify.call(this, obj);
214 
215 	var doNotify = false;
216 	var fields = {};
217 	if (obj.f != null && !obj._isRemote) {
218 		this._parseFlags(obj.f);
219 		// TODO: Should a F_EXCLUDE_FB property be added to ZmOrganizer?
220 		//       It doesn't make sense to require the base class to know about
221 		//       all the possible fields in sub-classes. So I'm just using the
222 		//       modified property name as the key.
223 		fields["excludeFreeBusy"] = true;
224 		doNotify = true;
225 	}
226 
227 	if (doNotify) {
228 		this._notify(ZmEvent.E_MODIFY, {fields: fields});
229 	}
230 };

232 ZmCalendar.prototype.notifyDelete =
233 function(obj) {
234 
235     if(this.isRemote() && !this._deleteAction){
236         var overviewController = appCtxt.getOverviewController();
237         var treeController = overviewController.getTreeController(this.type);
238         var overviewId = appCtxt.getCurrentApp().getOverviewId();
239         var treeView = treeController && treeController.getTreeView(overviewId);
240         var node = treeView && treeView.getTreeItemById(this.id);
241         this.noSuchFolder = true;
242         if(node) {
243             node.setText(this.getName(true));
244         }
245     }else{
246         ZmFolder.prototype.notifyDelete.call(this, obj);
247     }
248 };

268 ZmCalendar.sortCompare = 
269 function(calA, calB) {
270 	var check = ZmOrganizer.checkSortArgs(calA, calB);
271 	if (check != null) { return check; }
272 
273 	// links appear after personal calendars
274 	if (calA.link != calB.link) {
275 		return calA.link ? 1 : -1;
276 	}
277 
278 	// sort by calendar name
279 	var calAName = calA.name.toLowerCase();
280 	var calBName = calB.name.toLowerCase();
281 	if (calAName < calBName) return -1;
282 	if (calAName > calBName) return 1;
283 	return 0;
284 };

298 ZmCalendar.prototype.getRestUrl =
299 function(acct, noRemote) {
300 
301     if(!appCtxt.multiAccounts){
302         return ZmFolder.prototype.getRestUrl.call(this, noRemote);
303     }
304 
305 	// return REST URL as seen by server
306 	if (this.restUrl) {
307 		return this.restUrl;
308 	}
309 
310 	// if server doesn't tell us what URL to use, do our best to generate
311 	url = this._generateRestUrl(acct);
312 	DBG.println(AjxDebug.DBG3, "NO REST URL FROM SERVER. GENERATED URL: " + url);
313 
314 	return url;
315 };
316 
317 ZmCalendar.prototype._generateRestUrl =
318 function(acct) {
319 	var loc = document.location;
320 	var owner = this.getOwner();
321 	var uname = owner || appCtxt.get(ZmSetting.USERNAME);
322     if (appCtxt.multiAccounts) {
323         uname = appCtxt.get(ZmSetting.USERNAME, null, acct)
324     }
325 	var m = uname.match(/^(.*)@(.*)$/);
326 	var host = loc.host || (m && m[2]);
327 
328 	// REVISIT: What about port? For now assume other host uses same port
329 	if (loc.port && loc.port != 80 && (owner == appCtxt.get(ZmSetting.USERNAME))) {
330 		host = host + ":" + loc.port;
331 	}
332 
333 	return [
334 		loc.protocol, "//", host, "/service/user/", uname, "/",
335 		AjxStringUtil.urlEncode(this.getSearchPath(true))
336 	].join("");
337 };
