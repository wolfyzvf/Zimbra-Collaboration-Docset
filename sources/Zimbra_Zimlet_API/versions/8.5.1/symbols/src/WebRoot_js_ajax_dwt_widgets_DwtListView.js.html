 22 /**
 23  * Creates a list view.
 24  * @constructor
 25  * @class
 26  * A list view presents a list of items as rows with fields (columns).
 27  * 
 28  * @author Parag Shah
 29  * @author Conrad Damon
 30  * 
 31  * @param {hash}	params		a hash of parameters
 32  * @param  {DwtComposite}     parent		the parent widget
 33  * @param {string}	className		the CSS class
 34  * @param {constant}	posStyle		the positioning style (see {@link DwtControl})
 35  * @param  {array}	headerList	a list of IDs for columns
 36  * @param {boolean}	noMaximize	if <code>true</code>, all columns are fixed-width (otherwise, one will
 37  * 											expand to fill available space)
 38  * @param  {constant}     view			the ID of view
 39  * 
 40  * @extends		DwtComposite
 41  */
 42 DwtListView = function(params) {
 43 	if (arguments.length == 0) { return; }
 44 	params = Dwt.getParams(arguments, DwtListView.PARAMS);
 45 	params.className = params.className || "DwtListView";
 46 	DwtComposite.call(this, params);
 47 
 48 	this._view = params.view || Dwt.getNextId();
 49 	if (params.headerList) {
 50 		var htmlElement = this.getHtmlElement();
 51 
 52         var html = new Array(50);
 53         var idx = 0;
 54         var headId = Dwt.getNextId();
 55         var colId = Dwt.getNextId();
 56         html[idx++] = "<table width='100%'><tr><td ";
 57         html[idx++] = "id=" + headId;
 58         html[idx++] = "></td></tr><tr><td ";
 59         html[idx++] = "id=" + colId;
 60         html[idx++] = "></td></tr></table>";
 61         htmlElement.innerHTML = html.join("");
 62 
 63         var headHtml = document.getElementById(headId);
 64         this._listColDiv = document.createElement("div");
 65         this._listColDiv.id = DwtId.getListViewId(this._view, DwtId.LIST_VIEW_HEADERS);
 66         headHtml.appendChild(this._listColDiv);
 67 
 68         var colHtml = document.getElementById(colId);
 69         this._listDiv = this.useListElement() ? document.createElement("ul"):document.createElement("div");
 70         this._listDiv.id = DwtId.getListViewId(this._view, DwtId.LIST_VIEW_ROWS);
 71         this._listDiv.className = "DwtListView-Rows";
 72         colHtml.appendChild(this._listDiv);
 73 
 74 
 75 		// setup vars needed for sorting
 76 		this._bSortAsc = false;
 77 		this._currentColId = null;
 78 		this.sortingEnabled = true;
 79 	} else {
 80 		this._listDiv = document.getElementById(params.id);
 81 		this.setScrollStyle(DwtControl.SCROLL); // auto scroll
 82 	}
 83 		
 84 	this._setMouseEventHdlrs();
 85 	
 86 	this._listenerMouseOver = this._mouseOverListener.bind(this);
 87 	this._listenerMouseOut = this._mouseOutListener.bind(this);
 88 	this._listenerMouseDown = this._mouseDownListener.bind(this);
 89 	this._listenerMouseUp = this._mouseUpListener.bind(this);
 90 	this._listenerMouseMove = this._mouseMoveListener.bind(this);
 91 	this._listenerDoubleClick = this._doubleClickListener.bind(this);
 92 	this.addListener(DwtEvent.ONMOUSEOVER, this._listenerMouseOver);
 93 	this.addListener(DwtEvent.ONMOUSEOUT, this._listenerMouseOut);
 94 	this.addListener(DwtEvent.ONMOUSEDOWN, this._listenerMouseDown);
 95 	this.addListener(DwtEvent.ONMOUSEUP, this._listenerMouseUp);
 96 	this.addListener(DwtEvent.ONMOUSEMOVE, this._listenerMouseMove);
 97 	this.addListener(DwtEvent.ONDBLCLICK, this._listenerDoubleClick);
 98 
 99 	this._evtMgr = new AjxEventMgr();
100 	this._selectedItems = new AjxVector();
101 	this._selAnchor = null; 
102 	this._kbAnchor = null; 
103 	this._selEv = new DwtSelectionEvent(true);
104 	this._actionEv = new DwtListViewActionEvent(true);
105 	this._stateChangeEv = new DwtEvent(true);
106 	this._headerList = params.headerList;
107 	this._noMaximize = params.noMaximize;
108 	if (this._headerList) {
109 		this._parentEl = this._listDiv;
110 	} else {
111 		this._parentEl = this.getHtmlElement();
112 		if (this.useListElement()) {
113 			//insert unordered list element
114 			var ul = document.createElement("ul");
115 			ul.className = "DwtListView-Rows";
116 			this._parentEl.appendChild(ul);
117 			this._parentEl = ul;
118 		}
119 	}
120 	
121 	this._list = null;
122 	this.offset = 0;
123 	this.headerColCreated = false;
124 	this.multiSelectEnabled = true;
125 	this.firstSelIndex = -1;
126 
127 	// the key is the HTML ID of the item's associated DIV; the value is an object
128 	// with information about that row
129 	this._data = {};
130 
131     // item classes
132     this._rowClass = [ this._className, DwtListView.ROW_CLASS ].join("");
133 	var nc = this._normalClass = DwtListView.ROW_CLASS;
134 	this._selectedClass = [nc, DwtCssStyle.SELECTED].join("-");
135 	this._viewedButUnselectedClass = [nc, DwtCssStyle.ALT_SELECTED].join("-");
136 	this._disabledSelectedClass = [this._selectedClass, DwtCssStyle.DISABLED].join("-");
137 	this._kbFocusClass = [nc, DwtCssStyle.FOCUSED].join("-");
138 	this._dndClass = [nc, DwtCssStyle.DRAG_PROXY].join("-");
139 	this._rightClickClass = [this._selectedClass, DwtCssStyle.ACTIONED].join("-");
140 
141     this._styleRe = this._getStyleRegex();
142 };

190 /**
191  * Sets the enabled flag.
192  * 
193  * @param	{boolean}	enabled		if <code>true</code>, enable the list view
194  */
195 DwtListView.prototype.setEnabled =
196 function(enabled) {
197 	DwtComposite.prototype.setEnabled.call(this, enabled);
198 	// always remove listeners to avoid adding listeners multiple times
199 	this.removeListener(DwtEvent.ONMOUSEOVER, this._listenerMouseOver);
200 	this.removeListener(DwtEvent.ONMOUSEOUT, this._listenerMouseOut);
201 	this.removeListener(DwtEvent.ONMOUSEDOWN, this._listenerMouseDown);
202 	this.removeListener(DwtEvent.ONMOUSEUP, this._listenerMouseUp);
203 	this.removeListener(DwtEvent.ONMOUSEMOVE, this._listenerMouseMove);
204 	this.removeListener(DwtEvent.ONDBLCLICK, this._listenerDoubleClick);
205 	// now re-add listeners, if needed
206 	if (enabled) {
207 		this.addListener(DwtEvent.ONMOUSEOVER, this._listenerMouseOver);
208 		this.addListener(DwtEvent.ONMOUSEOUT, this._listenerMouseOut);
209 		this.addListener(DwtEvent.ONMOUSEDOWN, this._listenerMouseDown);
210 		this.addListener(DwtEvent.ONMOUSEUP, this._listenerMouseUp);
211 		this.addListener(DwtEvent.ONMOUSEMOVE, this._listenerMouseMove);
212 		this.addListener(DwtEvent.ONDBLCLICK, this._listenerDoubleClick);
213 	}
214 	// modify selection classes
215 	var selection = this.getSelectedItems();
216 	if (selection) {
217 		var elements = selection.getArray();
218         for (var i = 0; i < elements.length; i++) {
219             Dwt.delClass(elements[i], this._styleRe, enabled ? this._selectedClass : this._disabledSelectedClass);
220 		}
221 	}
222 };

224 DwtListView.prototype.createHeaderHtml =
225 function(defaultColumnSort, isColumnHeaderTableFixed) {
226 	// does this list view have headers or have they already been created?
227 	if (!this._headerList || this.headerColCreated) { return; }
228 
229 	this._headerHash = {};
230 	this._headerIdHash = {};
231 
232 	var idx = 0;
233 	var htmlArr = [];
234 
235 	htmlArr[idx++] = "<table id='";
236 	htmlArr[idx++] = DwtId.getListViewHdrId(DwtId.WIDGET_HDR_TABLE, this._view);
237 	htmlArr[idx++] = "' height=100%";
238 	htmlArr[idx++] = this._noMaximize ? ">" : " width=100%>";
239 	htmlArr[idx++] = "<tr>";
240 
241 	var numCols = this._headerList.length;
242 	for (var i = 0; i < numCols; i++) {
243 		var headerCol = this._headerList[i];
244 		var field = headerCol._field;
245 		headerCol._index = i;
246 		var id = headerCol._id = DwtId.getListViewHdrId(DwtId.WIDGET_HDR, this._view, field);
247 
248 		this._headerHash[field] = headerCol;
249 		this._headerIdHash[id] = headerCol;
250 
251 		if (headerCol._variable) {
252 			this._variableHeaderCol = headerCol;
253 		}
254 
255 		if (headerCol._visible) {
256 			idx = this._createHeader(htmlArr, idx, headerCol, i, numCols, id, defaultColumnSort);
257 		}
258 	}
259 	htmlArr[idx++] = "</tr></table>";
260 
261 	this._listColDiv.innerHTML = htmlArr.join("");
262 	this._listColDiv.className = "DwtListView-ColHeader" + (isColumnHeaderTableFixed ? " FixedColumnHeaderTables" : "");
263 
264 	setTimeout($.proxy(function() {
265 		//run this after the header is fully visible otherwise width will be inaccurate.
266 		//TODO run this as a callback after grid is visible instead of settimeout.
267 		for (var i = 0; i < numCols; i++) {
268 			//find the elements with width auto and create the styles for it.
269 			var headerCol = this._headerList[i];
270 			if (headerCol._cssClass && headerCol._resizeable && headerCol._width === 'auto') {
271 				this._createHeaderCssStyle(headerCol, this._calcRelativeWidth(i));
272 			}
273 		}
274 	}, this), 0);
275 
276 	// for each sortable column, sets its identifier
277 	var numResizeable = 0, resizeableCol;
278 	for (var j = 0; j < this._headerList.length; j++) {
279 		var headerCol = this._headerList[j];
280 		var cell = document.getElementById(headerCol._id);
281 		if (!cell) { continue; }
282 
283 		if (headerCol._sortable && headerCol._field == defaultColumnSort) {
284 			cell.className = "DwtListView-Column DwtListView-ColumnActive";
285 		}
286 
287 		if (headerCol._resizeable) {
288 			// always get the sibling cell to the right
289 			var sashId = DwtId.getListViewHdrId(DwtId.WIDGET_HDR_SASH, this._view, headerCol._field);
290 			var sashCell = document.getElementById(sashId);
291 			if (sashCell) {
292 				this.associateItemWithElement(headerCol, sashCell, DwtListView.TYPE_HEADER_SASH, sashId, {index:j});
293 			}
294 			numResizeable++;
295 			resizeableCol = headerCol;
296 		}
297 		this.associateItemWithElement(headerCol, cell, DwtListView.TYPE_HEADER_ITEM, headerCol._id, {index:j});
298 	}
299 
300 	if (numResizeable == 1) {
301 		resizeableCol._resizeable = false;
302 	}
303 
304 	this.headerColCreated = true;
305 };

307 DwtListView.prototype._createHeader =
308 function(htmlArr, idx, headerCol, i, numCols, id, defaultColumnSort) {
309 
310 	var field = headerCol._field;
311 
312 	htmlArr[idx++] = "<td id='";
313 	htmlArr[idx++] = id;
314 	htmlArr[idx++] = "' class='";
315     var tmpClass = (id == this._currentColId) ? "DwtListView-Column DwtListView-ColumnActive"
316 		: "DwtListView-Column";
317     tmpClass += headerCol._sortable ? "" : " DwtDefaultCursor";
318     htmlArr[idx++] = tmpClass + "'";
319 	if (headerCol._width) {
320 		htmlArr[idx++] = " width=";
321 		htmlArr[idx++] = headerCol._width;
322 		if (headerCol._cssClass && headerCol._resizeable && headerCol._width !== 'auto') {
323 			this._createHeaderCssStyle(headerCol, headerCol._width);
324 		}
325 		if (headerCol._widthUnits) {
326 			htmlArr[idx++] = headerCol._widthUnits;
327 		}
328     }
329 	if (headerCol._tooltip && DwtControl.useBrowserTooltips) {
330 		htmlArr[idx++] = " title='" + headerCol._tooltip + "'";
331 	}
332 	htmlArr[idx++] = ">";
333 	// must add a div to force clipping :(
334 	htmlArr[idx++] = "<div";
335 	var headerColWidth = null;
336 	if (headerCol._width && headerCol._width != "auto") {
337 		//why we need to + 2 here ? It causes the misalign of the list items in IE
338 		if (AjxEnv.isIE) {
339 			headerColWidth = headerCol._width;
340 		} else {
341 			headerColWidth = headerCol._width + 2;
342 		}
343 		if (headerCol._widthUnits) {
344 			headerColWidth += headerCol._widthUnits;
345 		}
346 	}
347 	if (!!headerColWidth) {
348 		htmlArr[idx++] = " style='overflow: hidden; width: ";
349 		htmlArr[idx++] = headerColWidth;
350 		htmlArr[idx++] = "'>";
351 	} else {
352 		htmlArr[idx++] = ">";
353 	}
354 
355 	// add new table for icon/label/sorting arrow
356 	htmlArr[idx++] = "<table width=100%><tr>";
357 	if (headerCol._iconInfo) {
358 		var idText = ["id='", DwtId.getListViewHdrId(DwtId.WIDGET_HDR_ICON, this._view, field), "'"].join("");
359 		htmlArr[idx++] = "<td><center>";
360 		htmlArr[idx++] = AjxImg.getImageHtml(headerCol._iconInfo, null, idText);
361 		htmlArr[idx++] = "</center></td>";
362 	}
363 
364 	if (headerCol._label) {
365 		htmlArr[idx++] = "<td id='";
366 		htmlArr[idx++] = DwtId.getListViewHdrId(DwtId.WIDGET_HDR_LABEL, this._view, field);
367 		htmlArr[idx++] = "' class='DwtListHeaderItem-label'>";
368 		htmlArr[idx++] = headerCol._label;
369 		htmlArr[idx++] = "</td>";
370 	}
371 
372 	if (headerCol._sortable && !headerCol._noSortArrow) {
373 		var arrowIcon = this._bSortAsc ? "ColumnUpArrow" : "ColumnDownArrow";
374 
375 		htmlArr[idx++] = "<td align=right style='padding-right:2px' width='8px' id='";
376 		htmlArr[idx++] = DwtId.getListViewHdrId(DwtId.WIDGET_HDR_ARROW, this._view, field);
377 		htmlArr[idx++] = "'>";
378 		var isDefault = (field == defaultColumnSort);
379 		htmlArr[idx++] = AjxImg.getImageHtml(arrowIcon, isDefault ? null : "visibility:hidden");
380 		htmlArr[idx++] = "</td>";
381 		if (isDefault) {
382 			this._currentColId = id;
383 		}
384 	}
385 
386 	// ALWAYS add "sash" separators
387 	if (i < (numCols - 1)) {
388 		htmlArr[idx++] = "<td width=6>";
389 		htmlArr[idx++] = "<table align=right width=4 height=100% id='";
390 		htmlArr[idx++] = DwtId.getListViewHdrId(DwtId.WIDGET_HDR_SASH, this._view, field);
391 		htmlArr[idx++] = "'><tr>";
392 		htmlArr[idx++] = "<td class='DwtListView-Sash'><div style='width: 1px; height: ";
393 		htmlArr[idx++] = (DwtListView.HEADERITEM_HEIGHT - 2);
394 		htmlArr[idx++] = "px; background-color: #8A8A8A;margin-left:2px'></div></td><td class='DwtListView-Sash'><div style='width: 1px; height: ";
395 		htmlArr[idx++] = (DwtListView.HEADERITEM_HEIGHT - 2);
396 		htmlArr[idx++] = "px;'></div></td></tr></table>";
397 		htmlArr[idx++] = "</td>";
398 	}
399 
400 	htmlArr[idx++] = "</tr></table>";
401 	htmlArr[idx++] = "</div></td>";
402 
403 	return idx;
404 };

406 DwtListView.prototype._createHeaderCssStyle =
407 	function(headerCol, width) {
408 		if (headerCol._cssClass && headerCol._resizeable) {
409 			if (headerCol._cssRuleIndex) {
410 				DwtCssStyle.removeRule(document.styleSheets[0], headerCol._cssRuleIndex);
411 			}
412 			//add a dynamic stylesheet for this header
413 			var selector = '';
414 			if (this.parent && this.parent._className) {
415 				//add the parent class name to selector so that styles for all types of view can co-exist
416 				selector += "." + this.parent._className;
417 			}
418 			selector += " ." + headerCol._cssClass;
419 			var declaration = "width:" + width + ($.isNumeric(width)?"px":"") + ";";
420 			headerCol._cssRuleIndex = DwtCssStyle.addRule(document.styleSheets[0], selector, declaration,headerCol._cssRuleIndex);
421 		}
422 	}
423 /**
424  * Gets the index of the given item.
425  *
426  * @param	{Object}	item		the item
427  * @return	{number}	the index or <code>null</code> if not found
428  */
429 DwtListView.prototype.getItemIndex =
430 function(item) {
431 	var list = this._list;
432 	if (list) {
433 		var len = list.size();
434 		for (var i = 0; i < len; ++i) {
435 			if (list.get(i).id == item.id) {
436 				return i;
437 			}
438 		}
439 	}
440 	return null;
441 };

499 DwtListView.prototype._renderList =
500 function(list, noResultsOk, doAdd) {
501 	if (list instanceof AjxVector && list.size()) {
502 		var now = new Date();
503 		var size = list.size();
504 		var htmlArr = [];
505 		for (var i = 0; i < size; i++) {
506 			var item = list.get(i);
507 			var div = this._createItemHtml(item, {now:now}, !doAdd, i);
508 			if (div) {
509 				if (div instanceof Array) {
510 					for (var j = 0; j < div.length; j++){
511 						this._addRow(div[j]);
512 					}
513 				} else if (div.tagName || doAdd) {
514 					this._addRow(div);
515 				} else {
516 					htmlArr.push(div);
517 				}
518 			}
519 		}
520 		if (htmlArr.length) {
521 			this._parentEl.innerHTML = htmlArr.join("");
522 		}
523 	} else if (!noResultsOk) {
524 		this._setNoResultsHtml();
525 	}
526 };

528 /**
529  * Adds the items.
530  * 
531  * @param	{array}		itemArray		an array of items
532  */
533 DwtListView.prototype.addItems =
534 function(itemArray) {
535 	if (AjxUtil.isArray(itemArray)) {
536 		if (!this._list) {
537 			this._list = new AjxVector();
538 		}
539

541 		if (this._list.size() == 0) {
542 			this._resetList();
543 		}
544 		this._renderList(AjxVector.fromArray(itemArray), null, true);
545 		this._list.addList(itemArray);
546 	}
547 };

549 /**
550  * Adds a row for the given item to the list view.
551  *
552  * @param {Object}	item			the data item
553  * @param {number}	index			the index at which to add item to list and list view
554  * @param {boolean}	skipNotify	if <code>true</code>, do not notify listeners
555  * @param {number}	itemIndex		index at which to add item to list, if different
556  * 									from the one for the list view
557  */
558 DwtListView.prototype.addItem =
559 function(item, index, skipNotify, itemIndex) {
560 	if (!this._list) {

587  * Removes a row for the given item to the list view.
588  *
589  * @param {Object}	item			the data item
590  * @param {boolean}	skipNotify	if <code>true</code>, do not notify listeners
591  * @param {boolean}	skipAlternation		if <code>true</code>, do not fix alternation
592  */
593 DwtListView.prototype.removeItem =
594 function(item, skipNotify, skipAlternation) {
595 
596 	var itemEl = this._getElFromItem(item);
597 	if (!itemEl) { return; }
598 
599 	var altIndex = this._getRowIndex(item);	// get index before we remove row
600 
601 	this._selectedItems.remove(itemEl);

603 		this._rightSelItem = null;
604 	}

615 	if (this._data[id]) {
616 		this._data[id] = null;
617 		delete this._data[id];
618 	}
619 	if (!skipAlternation) {
620 		this._fixAlternation(altIndex);
621 	}
622 
623 	if (!skipNotify && this._evtMgr.isListenerRegistered(DwtEvent.STATE_CHANGE)) {
624 		this._evtMgr.notifyListeners(DwtEvent.STATE_CHANGE, this._stateChangeEv);

1008 		// is different from the old selection.

1022 	// highlighted row.
1023 	if (row.id != this._dragHighlight) {
1024 		this._dragHighlight = row.id;
1025 		this._updateDragSelection(row, true);
1026 	}

1125 DwtListView.prototype.handleKeyAction =

1127 	switch (actionCode) {
1128 		case DwtKeyMap.SELECT:			this._emulateSingleClick({target:this._kbAnchor, button:DwtMouseEvent.LEFT, kbNavEvent:true}); break;
1129 		case DwtKeyMap.SELECT_CURRENT:	this._emulateSingleClick({target:this._kbAnchor, button:DwtMouseEvent.LEFT, ctrlKey:true, kbNavEvent:true}); break;
1130 		case DwtKeyMap.SELECT_NEXT:		this._selectItem(true, false, true); break;
1131 		case DwtKeyMap.SELECT_PREV:		this._selectItem(false, false, true); break;
1132 		case DwtKeyMap.ADD_SELECT_NEXT: this._selectItem(true, true, true); break;
1133 		case DwtKeyMap.ADD_SELECT_PREV: this._selectItem(false, true, true); break;
1134 		case DwtKeyMap.PREV:			this._setKbFocusElement(false); break;
1135 		case DwtKeyMap.NEXT:			this._setKbFocusElement(true); break;
1136 		case DwtKeyMap.DBLCLICK: {
1137 			if (!this._kbAnchor) { break; }
1138 			var anchorSelected = false;
1139 			var a = this.getSelectedItems().getArray();
1140 			for (var i = 0; i < a.length; i++) {
1141 				if (a[i] == this._kbAnchor) {
1142 					anchorSelected = true;
1143 					break;
1144 				}
1145 			}
1146 			if (anchorSelected) {
1147 				this.emulateDblClick(this.getItemFromElement(this._kbAnchor), true);
1148 			} else {
1149 				this._emulateSingleClick({target:this._kbAnchor, button:DwtMouseEvent.LEFT, kbNavEvent:true});
1150 			}
1151 			break;
1152 		}
1153 
1154 		case DwtKeyMap.SELECT_ALL:
1155 			this.selectAll();
1156 			break;
1157 
1158 		case DwtKeyMap.SELECT_FIRST:
1159 		case DwtKeyMap.SELECT_LAST:
1160 			var item = (actionCode == DwtKeyMap.SELECT_FIRST) ? this._getFirstItem() : this._getLastItem();
1161 			if (item) {
1162 				this.setSelection(item);
1163 				this._scrollList(this._kbAnchor);
1164 			}
1165 			break;
1166 
1167 		case DwtKeyMap.ACTION:
1168 			if (this._evtMgr.isListenerRegistered(DwtEvent.ACTION)) {
1169 				var p = Dwt.toWindow(this._kbAnchor, 0, 0);
1170 				var s = Dwt.getSize(this._kbAnchor);
1171 				var docX = p.x + s.x / 4;
1172 				var docY = p.y + s.y / 2;
1173 				this._emulateSingleClick({target:this._kbAnchor, button:DwtMouseEvent.RIGHT, docX:docX, docY:docY, kbNavEvent:true});
1174 			}
1175 			break;
1176 
1177 		case DwtKeyMap.PAGE_UP:
1178 		case DwtKeyMap.PAGE_DOWN:
1179 			this.scrollPage(actionCode == DwtKeyMap.PAGE_UP);
1180 			break;
1181 
1182 		default:

1184 	}

2560  * column unless the change makes the width of the last column less than
2561  * MIN_COLUMN_WIDTH.
2562  *
2563  * @param ev
2564  * 
2565  * @private
2566  */
2567 DwtListView.prototype._handleColSashDrop =
2568 function(ev) {
2569 	if (this._headerSash == null || ev.button == DwtMouseEvent.RIGHT) {	return false; }
2570 		
2571 	// destroy the sash
2572 	var parent = this._headerSash.parentNode;
2573 	if (parent) {
2574 		parent.removeChild(this._headerSash);
2575 	}
2576 	delete this._headerSash;
2577 
2578 	// force all relative widths to be static
2579 	for (var i = 0; i < this._headerList.length; i++) {
2580 		this._headerList[i]._width = this._calcRelativeWidth(i);
2581 	}
2582 
2583 	// find out where the user dropped the sash and update column width
2584 	var headerIdx = this._getItemData(this._clickDiv, "index");
2585 	if (headerIdx == null) { return false; }
2586 
2587 	var delta = ev.docX - this._headerSashX;
2588 
2589 	var fcol = this._headerList[headerIdx];
2590 
2591 	var col1 = fcol;
2592 	var col2;// = this._variableHeaderCol;
2593 	var resized = [];
2594 
2595 	if (delta < 0) {
2596 		if ((col1 == col2) || !col2) {
2597 			col2 = this._getNextResizeableColumnHeader(col1);
2598 		}
2599 		if (!col2) return false;
2600 		//delta =    - Math.min(fcol._width - DwtListView.MIN_COLUMN_WIDTH, -delta);
2601 		delta = Math.max(DwtListView.MIN_COLUMN_WIDTH - fcol._width, delta);
2602 		fcol._width = Math.max(fcol._width + delta, DwtListView.MIN_COLUMN_WIDTH);
2603 		col2._width = Math.max(this._calcRelativeWidth(col2._index) - delta, DwtListView.MIN_COLUMN_WIDTH);
2604 		resized.push(fcol._index, col2._index);
2605 		
2606 	} else if (delta > 0) {
2607 
2608 		var remain = delta;
2609 		while (remain > 0) {
2610 			if ((col1 == col2) || !col2) {
2611 				col2 = this._getNextResizeableColumnHeader(col1, [], false);
2612 			}
2613 			//if (!col2) return false;

2615 				delta -= remain;

2617 			}
2618 			var col2width = this._calcRelativeWidth(col2._index);
2619 			var room = col2width - DwtListView.MIN_COLUMN_WIDTH;
2620

2622 				remain -= room;
2623 				col2width = DwtListView.MIN_COLUMN_WIDTH;

2625 				col2width -= remain;
2626 				remain = 0;
2627 			}
2628 			col2._width = col2width;
2629 			resized.push(col2._index);
2630 			col1 = col2;
2631 		}
2632 	
2633 		fcol._width = Math.max(fcol._width + delta, DwtListView.MIN_COLUMN_WIDTH);
2634 		resized.push(fcol._index);
2635 
2636 	}
