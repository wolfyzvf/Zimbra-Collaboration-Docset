 22 /**
 23  * Creates the social application.
 24  * @class
 25  * This class represents the social application.
 26  *
 27  * @param	{DwtControl}	container		the container
 28  *
 29  * @extends		ZmApp
 30  */
 31 ZmSocialApp = function(container) {
 32 	ZmApp.call(this, ZmApp.SOCIAL, container);
 33 }
 34 
 35 ZmSocialApp.prototype = new ZmApp;
 36 ZmSocialApp.prototype.constructor = ZmSocialApp;
 37 
 38 ZmSocialApp.prototype.isZmSocialApp = true;
 39 ZmSocialApp.prototype.toString = function() {	return "ZmSocialApp"; };
 40 
 41 //
 42 // Constants
 43 //
 44 
 45 ZmApp.SOCIAL                        = ZmId.APP_SOCIAL;
 46 ZmApp.CLASS[ZmApp.SOCIAL]		    = "ZmSocialApp";
 47 ZmApp.SETTING[ZmApp.SOCIAL]		    = ZmSetting.SOCIAL_ENABLED;
 48 ZmApp.UPSELL_SETTING[ZmApp.SOCIAL]	= ZmSetting.SOCIAL_EXTERNAL_URL;    // use presence of URL to determine enabled
 49 ZmApp.LOAD_SORT[ZmApp.SOCIAL]	    = 100;
 50 
 51 ZmSocialApp.prototype._registerApp = function() {
 52 	ZmApp.registerApp(ZmApp.SOCIAL, {
 53 		nameKey:            "communityName",
 54 		icon:               "Globe",
 55 		chooserTooltipKey:  "goToSocial",
 56 		chooserSort:        32,
 57 		defaultSort:        100,
 58 		upsellUrl:			ZmSetting.SOCIAL_EXTERNAL_URL
 59 	});
 60 
 61 	// overwrite community name with value from settings, if any
 62 	var appName = appCtxt.get(ZmSetting.SOCIAL_NAME);
 63 	if (appName) {
 64 		ZmMsg[ZmApp.NAME[this._name]] = appName;
 65 	}
 66 };
 67 
 68 // User has clicked refresh button
 69 ZmSocialApp.prototype.runRefresh = function() {
 70 
 71 	var mainCtlr = appCtxt.getAppController(),
 72 		communityView = mainCtlr._appIframeView[ZmApp.SOCIAL];
 73 
 74 	if (communityView) {
 75 		communityView.runRefresh();
 76 	}
 77 };
