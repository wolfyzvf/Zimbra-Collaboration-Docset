132 /**
133  * Fetches a message from the server.
134  *
135  * @param {Hash}			params					a hash of parameters
136  * @param {ZmZimbraMail}	params.sender			the provides access to sendRequest()
137  * @param {int}				params.msgId			the ID of the msg to be fetched.
138  * @param {int}				params.partId 			the msg part ID (if retrieving attachment part, i.e. rfc/822)
139  * @param {int}				params.ridZ   			the RECURRENCE-ID in Z (UTC) timezone
140  * @param {Boolean}			params.getHtml			if <code>true</code>, try to fetch html from the server
141  * @param {Boolean}			params.markRead			if <code>true</code>, mark msg read
142  * @param {AjxCallback}		params.callback			the async callback
143  * @param {AjxCallback}		params.errorCallback	the async error callback
144  * @param {Boolean}			params.noBusyOverlay	if <code>true</code>, do not put up busy overlay during request
145  * @param {Boolean}			params.noTruncate		if <code>true</code>, do not truncate message body
146  * @param {ZmBatchCommand}	params.batchCmd			if set, request gets added to this batch command
147  * @param {String}			params.accountName		the name of the account to send request on behalf of
148  * @param {boolean}			params.needExp			if not <code>false</code>, have server check if addresses are DLs
149  */
150 ZmMailMsg.fetchMsg =
151 function(params) {
152 	var jsonObj = {GetMsgRequest:{_jsns:"urn:zimbraMail"}};
153 	var request = jsonObj.GetMsgRequest;
154 	var m = request.m = {};
155 	m.id = params.msgId;
156 	if (params.partId) {
157 		m.part = params.partId;
158 	}
159 	if (params.markRead) {
160 		m.read = 1;
161 	}
162 	if (params.getHtml) {
163 		m.html = 1;
164 	}
165 	if (params.needExp !== false) {
166 		m.needExp = 1;
167 	}
168 
169 	if (params.ridZ) {
170 		m.ridZ = params.ridZ;
171 	}
172 
173 	ZmMailMsg.addRequestHeaders(m);
174 
175 	if (!params.noTruncate) {
176 		m.max = appCtxt.get(ZmSetting.MAX_MESSAGE_SIZE) || ZmMailApp.DEFAULT_MAX_MESSAGE_SIZE;
177 	}
178 
179 	if (params.batchCmd) {
180 		params.batchCmd.addRequestParams(jsonObj, params.callback);
181 	} else {
182 		var newParams = {
183 			jsonObj:		jsonObj,
184 			asyncMode:		true,
185             offlineCache:   true,
186 			callback:		ZmMailMsg._handleResponseFetchMsg.bind(null, params.callback),
187 			errorCallback:	params.errorCallback,
188 			noBusyOverlay:	params.noBusyOverlay,
189 			accountName:	params.accountName
190 		};
191         newParams.offlineCallback = ZmMailMsg._handleOfflineResponseFetchMsg.bind(null, m.id, newParams.callback);
192 		params.sender.sendRequest(newParams);
193 	}
194 };

232 /**
233  * Gets a vector of addresses of the given type.
234  *
235  * @param {constant}	type			an email address type
236  * @param {Hash}		used			an array of addresses that have been used. If not <code>null</code>,
237  *										then this method will omit those addresses from the
238  * 										returned vector and will populate used with the additional new addresses
239  * @param {Boolean}		addAsContact	if <code>true</code>, emails should be converted to {@link ZmContact} objects
240  * @param {boolean}		dontUpdateUsed	if true, do not update the hash of used addresses
241  * 
242  * @return	{AjxVector}	a vector of email addresses
243  */
244 ZmMailMsg.prototype.getAddresses =
245 function(type, used, addAsContact, dontUpdateUsed) {
246 	if (!used) {
247 		return this._addrs[type];
248 	} else {
249 		var a = this._addrs[type].getArray();
250 		var addrs = [];
251 		for (var i = 0; i < a.length; i++) {
252 			var addr = a[i];
253 			var email = addr.getAddress();
254 			if (!email) { continue; }
255 			email = email.toLowerCase();
256 			if (!used[email]) {
257 				var contact = addr;
258 				if (addAsContact) {
259 					var cl = AjxDispatcher.run("GetContacts");
260 					contact = cl.getContactByEmail(email);
261 					if (contact == null) {
262 						contact = new ZmContact(null);
263 						contact.initFromEmail(addr);
264 					}
265 				}
266 				addrs.push(contact);
267 			}
268 			if (!dontUpdateUsed) {
269 				used[email] = true;
270 			}
271 		}
272 		return AjxVector.fromArray(addrs);
273 	}
274 };

276 /**
277  * Gets a Reply-To address if there is one, otherwise the From address
278  * unless this message was sent by the user, in which case, it is the To
279  * field (but only in the case of Reply All). A list is returned, since
280  * theoretically From and Reply To can have multiple addresses.
281  * 
282  * @return	{AjxVector}	an array of {@link AjxEmailAddress} objects
283  */
284 ZmMailMsg.prototype.getReplyAddresses =
285 function(mode, aliases, isDefaultIdentity) {
286 
287 	if (!this.isSent) { //ignore reply_to for sent messages.
288 		// reply-to has precedence over everything else
289 		var addrVec = this._addrs[AjxEmailAddress.REPLY_TO];
290 	}
291 	if (!addrVec && this.isInvite() && this.needsRsvp()) {
292 		var invEmail = this.invite.getOrganizerEmail(0);
293 		if (invEmail) {
294 			return AjxVector.fromArray([new AjxEmailAddress(invEmail)]);
295 		}
296 	}
297 
298 	if (!(addrVec && addrVec.size())) {
299 		if (mode == ZmOperation.REPLY_CANCEL || (this.isSent && mode == ZmOperation.REPLY_ALL)) {
300 			addrVec = this.isInvite() ? this._getAttendees() : this.getAddresses(AjxEmailAddress.TO, aliases, false, true);
301 		} else {
302 			addrVec = this.getAddresses(AjxEmailAddress.FROM, aliases, false, true);
303 			if (addrVec.size() == 0) {
304 				addrVec = this.getAddresses(AjxEmailAddress.TO, aliases, false, true);
305 			}
306 		}
307 	}
308 	return addrVec;
309 };

311 ZmMailMsg.prototype._getAttendees =
312 function() {
313 	var attendees = this.invite.components[0].at;
314 	var emails = new AjxVector();
315 	for (var i = 0; i < (attendees ? attendees.length : 0); i++) {
316 		var at = attendees[i];
317 		emails.add(new AjxEmailAddress(at.a, null, null, at.d));
318 	}
319 
320 	return emails;
321 };

366 /**
367  * Gets the header string.
368  * 
369  * @param	{constant}	hdr			the header (see <code>ZmMailMsg.HDR_</code> constants)
370  * @param	{boolean}	htmlMode	if true, format as HTML
371  * @return	{String}	the value
372  */
373 ZmMailMsg.prototype.getHeaderStr =
374 function(hdr, htmlMode) {
375 
376 	var key, value;
377 	if (hdr == ZmMailMsg.HDR_DATE) {
378 		if (this.sentDate) {
379 			var formatter = AjxDateFormat.getDateTimeInstance(AjxDateFormat.FULL, AjxDateFormat.MEDIUM);
380 			value = formatter.format(new Date(this.sentDate));
381 		}
382 	} else if (hdr == ZmMailMsg.HDR_SUBJECT) {
383 		value = this.subject;
384 	} else {
385 		var addrs = this.getAddresses(hdr);
386 		value = addrs.toString(", ", true);
387 	}
388 
389 	var key = ZmMailMsg.HDR_KEY[hdr] + ": ";
390 	if (!value) { return; }
391 	if (htmlMode) {
392 		key = "<b>" + key + "</b>";
393 		value = AjxStringUtil.convertToHtml(value);
394 	}
395 
396 	return key + value;
397 };

710 /**
711  * Gets the full message object from the back end based on the current message ID, and
712  * fills in the message.
713  *
714  * @param {Hash}			params					a hash of parameters:
715  * @param {Boolean}			params.getHtml			if <code>true</code>, try to fetch html from the server
716  * @param {Boolean}			params.markRead			if <code>true</code>, mark msg read
717  * @param {Boolean}			params.forceLoad		if <code>true</code>, get msg from server
718  * @param {AjxCallback}		params.callback			the async callback
719  * @param {AjxCallback}		params.errorCallback	the async error callback
720  * @param {Boolean}			params.noBusyOverlay	if <code>true</code>, do not put up busy overlay during request
721  * @param {Boolean}			params.noTruncate		if <code>true</code>, do not set max limit on size of msg body
722  * @param {ZmBatchCommand}	params.batchCmd			if set, request gets added to this batch command
723  * @param {String}			params.accountName		the name of the account to send request on behalf of
724  * @param {boolean}			params.needExp			if not <code>false</code>, have server check if addresses are DLs
725  */
726 ZmMailMsg.prototype.load =
727 function(params) {
728 	if (this._loading && !params.forceLoad) {
729 		//the only way to not get partial results is to try in some timeout.
730 		//this method will be called again, eventually, the message will be finished loading, and the callback would be called safely.
731 		this._loadingWaitCount = (this._loadingWaitCount || 0) + 1;
732 		if (this._loadingWaitCount > 20) {
733 			//give up after 20 timeouts (about 10 seconds) - maybe request got lost. send another request below.
734 			this._loadingWaitCount = 0;
735 			this._loading = false;
736 		}
737 		else {
738 			setTimeout(this.load.bind(this, params), 500);
739 			return;
740 		}
741 	}
742 	// If we are already loaded, then don't bother loading
743 	if (!this._loaded || params.forceLoad) {
744 		this._loading = true;
745 		var respCallback = this._handleResponseLoad.bind(this, params, params.callback);
746 		params.getHtml = params.getHtml || this.isDraft || appCtxt.get(ZmSetting.VIEW_AS_HTML);
747 		params.sender = appCtxt.getAppController();
748 		params.msgId = this.id;
749 		params.partId = this.partId;
750 		params.callback = respCallback;
751 		var errorCallback = this._handleResponseLoadFail.bind(this, params, params.errorCallback);
752 		params.errorCallback = errorCallback;
753 		ZmMailMsg.fetchMsg(params);
754 	} else {
755 		if (params.callback) {
756 			params.callback.run(new ZmCsfeResult()); // return exceptionless result
757 		}
758 	}
759 };

768 	if (this.participants) {
769 		this.participants.removeAll();
770 	}

786 		this._loadCallback.run(result);

895 /**
896  * Returns the first body part, of the given type if provided. May invoke a
897  * server call if it needs to fetch an alternative part.
898  * 
899  * @param	{string}		contentType		MIME type
900  * @param	{callback}		callback		callback
901  * 
902  * @return	{ZmMimePart}					MIME part
903  */
904 ZmMailMsg.prototype.getBodyPart =
905 function(contentType, callback) {
906 
907 	if (contentType) {
908 		this._lastContentType = contentType;
909 	}
910 
911 	function getPart(ct) {
912 		var bodyParts = this.getBodyParts(ct);
913 		for (var i = 0; i < bodyParts.length; i++) {
914 			var part = bodyParts[i];
915 			// should be a ZmMimePart, but check just in case
916 			part = part.isZmMimePart ? part : part[ct];
917 			if (!ct || (part.contentType === ct)) {
918 				return part;
919 			}
920 		}
921 	}
922 	var bodyPart = getPart.call(this,contentType);
923 	
924 	if (this.isInvite()) {
925 		if (!bodyPart) {
926 			if (contentType === ZmMimeTable.TEXT_HTML) {
927 				//text/html not available so look for text/plain
928 				bodyPart = getPart.call(this,ZmMimeTable.TEXT_PLAIN);
929 			} else if (contentType === ZmMimeTable.TEXT_PLAIN) {
930 				//text/plain not available so look for text/html
931 				bodyPart = getPart.call(this,ZmMimeTable.TEXT_HTML);
932 			}
933 		}
934 		// bug: 46071, handle missing body part/content
935 		if (!bodyPart || (bodyPart && !bodyPart.getContent())) {
936 			bodyPart = this.getInviteDescriptionContent(contentType);
937 		}
938 	}
939 
940 	if (callback) {
941 		if (bodyPart) {
942 			callback.run(bodyPart);
943 		}
944 		// see if we should try to fetch an alternative part
945 		else if (this.hasContentType(ZmMimeTable.MULTI_ALT) &&
946 				((contentType == ZmMimeTable.TEXT_PLAIN && this.hasContentType(ZmMimeTable.TEXT_PLAIN)) ||
947 				 (contentType == ZmMimeTable.TEXT_HTML && this.hasContentType(ZmMimeTable.TEXT_HTML)))) {
948 
949 			ZmMailMsg.fetchMsg({
950 				sender:		appCtxt.getAppController(),
951 				msgId:		this.id,
952 				getHtml:	(contentType == ZmMimeTable.TEXT_HTML),
953 				callback:	this._handleResponseFetchAlternativePart.bind(this, contentType, callback)
954 			});
955 		}
956 		else {
957 			callback.run();
958 		}
959 	}
960 
961 	return bodyPart;
962 };

1028 /**
1029  * Extracts and returns the text content out of a text/calendar part.
1030  * 
1031  * @param	{ZmMimePart}	bodyPart	a text/calendar MIME part
1032  * @return	{string}					text content
1033  */
1034 ZmMailMsg.getTextFromCalendarPart =
1035 function(bodyPart) {
1036 
1037 	// NOTE: IE doesn't match multi-line regex, even when explicitly
1038 	// specifying the "m" attribute.
1039 	var bpContent = bodyPart ? bodyPart.getContent() : "";
1040 	var lines = bpContent.split(/\r\n/);
1041 	var desc = [];
1042 	var content = "";
1043 	for (var i = 0; i < lines.length; i++) {
1044 		var line = lines[i];
1045 		if (line.match(/^DESCRIPTION:/)) {
1046 			desc.push(line.substr(12));
1047 			for (var j = i + 1; j < lines.length; j++) {
1048 				line = lines[j];
1049 				if (line.match(/^\s+/)) {
1050 					desc.push(line.replace(/^\s+/, " "));
1051 					continue;
1052 				}
1053 				break;
1054 			}
1055 			break;
1056 		}
1057         else if (line.match(/^COMMENT:/)) {
1058             //DESCRIPTION is sent as COMMENT in Lotus notes.
1059             desc.push(line.substr(8));
1060             for (var j = i + 1; j < lines.length; j++) {
1061                 line = lines[j];
1062                 if (line.match(/^\s+/)) {
1063                     desc.push(line.replace(/^\s+/, " "));
1064                     continue;
1065                 }
1066                 break;
1067             }
1068             break;
1069         }
1070 	}
1071 	if (desc.length > 0) {
1072 		content = desc.join("");
1073 		content = content.replace(/\\t/g, "\t");
1074 		content = content.replace(/\\n/g, "\n");
1075 		content = content.replace(/\\(.)/g, "$1");
1076 	}
1077 	return content;
