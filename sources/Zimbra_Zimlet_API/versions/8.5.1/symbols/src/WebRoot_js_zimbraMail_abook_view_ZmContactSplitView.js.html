 27 /**
 28  * Creates a contact split view.
 29  * @class
 30  * This class represents the contact split view.
 31  * 
 32  * @param	{Hash}	params		a hash of parameters
 33  * @extends	DwtComposite
 34  */
 35 ZmContactSplitView = function(params) {
 36 	if (arguments.length == 0) { return; }
 37 
 38 	params.className = params.className || "ZmContactSplitView";
 39 	params.posStyle = params.posStyle || Dwt.ABSOLUTE_STYLE;
 40 	params.id = Dwt.getNextId('ZmContactSplitView_');
 41 	DwtComposite.call(this, params);
 42 
 43 	this._controller = params.controller;
 44 	this.setScrollStyle(Dwt.CLIP);
 45 
 46 	this._changeListener = new AjxListener(this, this._contactChangeListener);
 47 
 48 	this._initialize(params.controller, params.dropTgt);
 49 
 50 	var folderTree = appCtxt.getFolderTree();
 51 	if (folderTree) {
 52 		folderTree.addChangeListener(new AjxListener(this, this._addrbookTreeListener));
 53 	}
 54 
 55 	ZmTagsHelper.setupListeners(this);
 56 
 57 };

148 /**
149  * Sets the contact.
150  * 
151  * @param	{ZmContact}	contact		the contact
152  * @param	{Boolean}	isGal		<code>true</code> if is GAL
153  * 
154  */
155 ZmContactSplitView.prototype.setContact =
156 function(contact, isGal) {
157 	if (contact.isDistributionList() || !isGal) {
158 		// Remove and re-add listeners for current contact if exists
159 		if (this._contact) {
160 			this._contact.removeChangeListener(this._changeListener);
161 		}
162 		contact.addChangeListener(this._changeListener);
163 	}
164 
165 	var oldContact = this._contact;
166 	this._contact = this._item = contact;
167 
168 	if (this._contact.isLoaded) {
169 		this._setContact(contact, isGal, oldContact);
170 	} else {
171 		var callback = new AjxCallback(this, this._handleResponseLoad, [isGal, oldContact]);
172 		var errorCallback = new AjxCallback(this, this._handleErrorLoad);
173 		this._contact.load(callback, errorCallback, null, contact.isGroup());
174 	}
175 };

185 ZmContactSplitView.handleDLScroll =
186 function(ev) {
187 
188 	var target = DwtUiEvent.getTarget(ev);
189 	var view = DwtControl.findControl(target);
190 	if (!view) { return; }
191 	var div = view._dlScrollDiv;
192 	if (div.clientHeight == div.scrollHeight) { return; }
193 	var contactDL = appCtxt.getApp(ZmApp.CONTACTS).getDL(view._dlContact.getEmail());
194 	var listSize = view.getDLSize();
195 	if (contactDL && (contactDL.more || (listSize < contactDL.list.length))) {
196 		var params = {scrollDiv:	div,
197 					  rowHeight:	view._rowHeight,
198 					  threshold:	10,
199 					  limit:		ZmContact.DL_PAGE_SIZE,
200 					  listSize:		listSize};
201 		var needed = ZmListView.getRowsNeeded(params);
202 		DBG.println("dl", "scroll, items needed: " + needed);
203 		if (needed) {
204 			DBG.println("dl", "new offset: " + listSize);
205 			var respCallback = new AjxCallback(null, ZmContactSplitView._handleResponseDLScroll, [view]);
206 			view._dlContact.getDLMembers(listSize, null, respCallback);
207 		}
208 	}
209 };

211 ZmContactSplitView._handleResponseDLScroll =
212 function(view, result) {
213 
214 	var list = result.list;
215 	if (!(list && list.length)) { return; }
216 
217 	var html = [];
218 	view._listPart._getImageHtml(html, 0, null);
219 	var subs = {seenone:true, html:html};
220 	var row = document.getElementById(view._dlLastRowId);
221 	var table = row && document.getElementById(view._detailsId);
222 	if (row) {
223 		var rowIndex = row.rowIndex + 1;
224 		for (var i = 0, len = list.length; i < len; i++) {
225 			view._distributionList.list.push(list[i]);
226 			subs.value = view.__findObjects(view._objectManager, list[i], ZmObjectManager.EMAIL);
227 			var rowIdText = "";
228 			var newRow = table.insertRow(rowIndex + i);
229 			if (i == len - 1) {
230 				newRow.id = view._dlLastRowId = Dwt.getNextId();
231 			}
232 			newRow.valign = "top";
233 			newRow.innerHTML = AjxTemplate.expand("abook.Contacts#SplitView_dlmember-expanded", subs);
234 		}
235 //		view._dlScrollDiv.scrollTop = 0;
236 	}
237 	DBG.println("dl", table.rows.length + " rows");
238 };

310 /**
311  * @private
312  */
313 ZmContactSplitView.prototype._initialize =
314 function(controller, dropTgt) {
315 	this.getHtmlElement().innerHTML = AjxTemplate.expand("abook.Contacts#SplitView", {id:this._htmlElId});
316 
317 	// alphabet bar based on *optional* existence in template and msg properties
318 	var alphaDivId = this._htmlElId + "_alphabetbar";
319 	var alphaDiv = document.getElementById(alphaDivId);
320 	if (alphaDiv && ZmMsg.alphabet && ZmMsg.alphabet.length>0) {
321 		this._alphabetBar = new ZmContactAlphabetBar(this);
322 		this._alphabetBar.reparentHtmlElement(alphaDivId);
323 	}
324 
325 	var splitviewCellId = this._htmlElId + "_splitview";
326 	this._splitviewCell = document.getElementById(splitviewCellId);
327 
328 	// create listview based on *required* existence in template
329 	var listviewCellId = this._htmlElId + "_listview";
330 	this._listviewCell = document.getElementById(listviewCellId);
331 	this._listPart = new ZmContactSimpleView({parent:this, controller:controller, dropTgt:dropTgt});
332 	this._listPart.reparentHtmlElement(listviewCellId);
333 
334 	var sashCellId = this._htmlElId + "_sash";
335 	this._sash = new DwtSash(this, DwtSash.HORIZONTAL_STYLE, null, 5, Dwt.ABSOLUTE_STYLE);
336 	this._sash.registerCallback(this._sashCallback, this);
337 	this._sash.replaceElement(sashCellId, false, true);
338 
339 	var contentCellId = this._htmlElId + "_contentCell";
340 	this._contentCell = document.getElementById(contentCellId);
341 
342 	// define well-known Id's
343 	this._iconCellId	= this._htmlElId + "_icon";
344 	this._titleCellId	= this._htmlElId + "_title";
345 	this._tagCellId		= this._htmlElId + "_tags_contact";
346 	this._tagGroupCellId	= this._htmlElId + "_tags_group";
347 	this._headerRowId	= this._htmlElId + "_headerRow";
348 	this._contactBodyId = this._htmlElId + "_body";
349 	this._contentId		= this._htmlElId + "_content";
350 	this._detailsId		= this._htmlElId + "_details";
351 
352 	// contact groups is not child of DwtTabGroup
353 	this._contactGroupView = new ZmContactView({ parent: this, controller: this._controller });
354 	this._contactGroupView.setVisible(false);
355 	this._contactGroupView.reparentHtmlElement(this._contentId);
356 	this._contactGroupView._setMouseEventHdlrs();
357 	this._groupObjectManager = new ZmObjectManager(this._contactGroupView);
358 
359 	// create an empty slate
360 	this._contactView = new ZmContactView({ parent: this, controller: this._controller });
361 	this._contactView.reparentHtmlElement(this._contentId);
362 	this._contactView._setMouseEventHdlrs();
363 	this._objectManager = new ZmObjectManager(this._contactView);
364 	this._contentCell.style.right = "0px";
365 };

408 /**
409  * @private
410  */
411 ZmContactSplitView.prototype._addrbookTreeListener =
412 function(ev, treeView) {
413 	if (!this._contact) { return; }
414 
415 	var fields = ev.getDetail("fields");
416 	if (ev.event == ZmEvent.E_MODIFY && fields && fields[ZmOrganizer.F_COLOR]) {
417 		var organizers = ev.getDetail("organizers");
418 		if (!organizers && ev.source) {
419 			organizers = [ev.source];
420 		}
421 
422 		for (var i = 0; i < organizers.length; i++) {
423 			var organizer = organizers[i];
424 			var folderId = this._contact.isShared()
425 				? appCtxt.getById(this._contact.folderId).id
426 				: this._contact.folderId;
427 
428 			if (organizer.id == folderId) {
429 				this._setTags();
430 			}
431 		}
432 	}
433 };

443 /**
444  * @private
445  */
446 ZmContactSplitView.prototype._setContact =
447 function(contact, isGal, oldContact, expandDL, isBack) {
448 
449 	//first gather the dl info and dl members. Those are async requests so calling back here after
450 	//it is done with isBack set to true.
451 	if (contact.isDistributionList() && !isBack) {
452 		var callbackHere = this._setContact.bind(this, contact, isGal, oldContact, expandDL, true);
453 		contact.gatherExtraDlStuff(callbackHere);
454 		return;
455 	}
456 
457 	var addrBook = contact.getAddressBook();
458 	var color = addrBook ? addrBook.color : ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.ADDRBOOK];
459 	var subs = {
460 		id: this._htmlElId,
461 		contact: contact,
462 		addrbook: addrBook,
463 		contactHdrClass: (ZmOrganizer.COLOR_TEXT[color] + "Bg"),
464 		isInTrash: (addrBook && addrBook.isInTrash())
465 	};
466 
467 	if (contact.isGroup()) {
468 		this._groupObjectManager.reset();
469 
470 		if (addrBook) {
471 			subs.folderIcon = addrBook.getIcon();
472 			subs.folderName = addrBook.getName();
473 		}
474 
475 		if (contact.isDistributionList()) {
476 			var dlInfo = subs.dlInfo = contact.dlInfo;
477 		}
478 		subs.groupMembers = contact.getAllGroupMembers();
479 		subs.findObjects = AjxCallback.simpleClosure(this.__findObjects, this, this._groupObjectManager);
480 
481 		this._resetVisibility(true);
482 
483 		this._contactGroupView.getHtmlElement().innerHTML = AjxTemplate.expand("abook.Contacts#SplitViewGroup", subs);
484 
485 		if (contact.isDistributionList()) {
486 			if (this._subscriptionButton) {
487 				this._subscriptionButton.dispose();
488 			}
489 			this._subscriptionButton = new DwtButton({parent:this, parentElement:(this._htmlElId + "_subscriptionButton")});
490 			this._subscriptionButton.setEnabled(true);
491 			this._subscriptionMsg = document.getElementById(this._htmlElId + "_subscriptionMsg");
492 			this._updateSubscriptionButtonAndMsg(contact);
493 			var subListener = new AjxListener(this, this._subscriptionListener, contact);
494 			this._subscriptionButton.addSelectionListener(subListener);
495 		}
496 
497 		var size = this.getSize();
498 		this._sizeChildren(size.x, size.y);
499 	} else {
500 		subs.view = this;
501 		subs.isGal = isGal;
502 		subs.findObjects = AjxCallback.simpleClosure(this.__findObjects, this, this._objectManager);
503 		subs.attrs = contact.getNormalizedAttrs();
504 		subs.encode = {};
505 		subs.encode.IM = AjxCallback.simpleClosure(this._encodeIM, this);
506 		subs.expandDL = expandDL;
507 
508 		if (contact.isDL && contact.canExpand) {
509 			this._dlContact = contact;
510 			this._dlScrollDiv = this._dlScrollDiv || document.getElementById(this._contentId);
511 			var respCallback = new AjxCallback(this, this._showDL, [subs]);
512 			contact.getDLMembers(0, null, respCallback);
513 			return;
514 		}
515 		this._showContact(subs);
516 	}
517 
518 	this._setTags();
519 	Dwt.setLoadedTime("ZmContactItem");
520 };

547 ZmContactSplitView.prototype._handleSubscriptionResponse =
548 function(contact, subscribe, result) {
549 	var status = result._data.SubscribeDistributionListResponse.status;
550 	var subscribed = status == "subscribed";
551 	var unsubscribed = status == "unsubscribed";
552 	var awaitingApproval = status == "awaiting_approval";
553 	this._subscriptionButton.setEnabled(!awaitingApproval);
554 	if (!awaitingApproval) {
555 		contact.dlInfo.isMember = subscribed;
556 	}
557 	if (subscribed || unsubscribed) {
558 		contact.clearDlInfo();
559 		contact._notify(ZmEvent.E_MODIFY);
560 	}
561 	var msg = subscribed ? ZmMsg.dlSubscribed
562 			: unsubscribed ? ZmMsg.dlUnsubscribed
563 			: awaitingApproval && subscribe ? ZmMsg.dlSubscriptionRequested
564 			: awaitingApproval && !subscribe ? ZmMsg.dlUnsubscriptionRequested
565 			: ""; //should not happen. Keep this as separate case for ease of debug when it does happen somehow.
566 	var dlg = appCtxt.getMsgDialog();
567 	var name = contact.getEmail();
568 	dlg.setMessage(AjxMessageFormat.format(msg, name), DwtMessageDialog.INFO_STYLE);
569 	dlg.popup();
570 
571 };

573 ZmContactSplitView.prototype._updateSubscriptionButtonAndMsg =
574 function(contact) {
575 	var dlInfo = contact.dlInfo;
576 	var policy = dlInfo.isMember ? dlInfo.unsubscriptionPolicy : dlInfo.subscriptionPolicy;
577 	if (policy == ZmContactSplitView.SUBSCRIPTION_POLICY_REJECT) {
578 		this._subscriptionButton.setVisible(false);
579 	}
580 	else {
581 		this._subscriptionButton.setVisible(true);
582 		this._subscriptionButton.setText(dlInfo.isMember ? ZmMsg.dlUnsubscribe: ZmMsg.dlSubscribe);
583 	}
584 	var statusMsg = dlInfo.isOwner && dlInfo.isMember ? ZmMsg.youAreOwnerAndMember
585 			: dlInfo.isOwner ? ZmMsg.youAreOwner
586 			: dlInfo.isMember ? ZmMsg.youAreMember
587 			: "";
588 	if (statusMsg != '') {
589 		statusMsg = "<li>" + statusMsg + "</li>";
590 	}
591 	var actionMsg;
592 	if (!dlInfo.isMember) {
593 		actionMsg =	policy == ZmContactSplitView.SUBSCRIPTION_POLICY_APPROVAL ? ZmMsg.dlSubscriptionRequiresApproval
594 			: policy == ZmContactSplitView.SUBSCRIPTION_POLICY_REJECT ? ZmMsg.dlSubscriptionNotAllowed
595 			: "";
596 	}
597 	else {
598 		actionMsg =	policy == ZmContactSplitView.SUBSCRIPTION_POLICY_APPROVAL ? ZmMsg.dlUnsubscriptionRequiresApproval
599 			: policy == ZmContactSplitView.SUBSCRIPTION_POLICY_REJECT ? ZmMsg.dlUnsubscriptionNotAllowed
600 			: "";
601 
602 	}
603 	if (actionMsg != '') {
604 		actionMsg = "<li>" + actionMsg + "</li>";
605 	}
606 	this._subscriptionMsg.innerHTML = statusMsg + actionMsg;
607 
608 };

623 ZmContactSplitView._showContactList =
624 function(data, names, typeFunc, hideType) {
625 
626 	data.names = names;
627 	var html = [];
628 	for (var i = 0; i < names.length; i++) {
629 		var name = names[i];
630 		data.name = name;
631 		data.type = (typeFunc && typeFunc(data, name)) || ZmMsg["AB_FIELD_" + name];
632 		data.type = hideType ? "" : data.type;
633 		html.push(ZmContactSplitView._showContactListItem(data));
634 	}
635 
636 	return html.join("");
637 };

639 ZmContactSplitView._showContactListItem =
640 function(data) {
641 
642 	var isEmail = (data.objectType == ZmObjectManager.EMAIL);
643 	var i = 0;
644 	var html = [];
645 	while (true) {

1231 		idx = this._getImageHtml(htmlArr, idx, "CheckboxUnchecked", this._getFieldId(contact, ZmItem.F_SELECTION));
1232 	}
1233 
1234 	// icon
1235 	htmlArr[idx++] = AjxImg.getImageHtml(contact.getIcon(folder), null, "id=" + this._getFieldId(contact, "type"),null, null, ["ZmContactIcon"]);
1236 
1237 	// file as
1238 	htmlArr[idx++] = "<div id='" + this._getFieldId(contact, "fileas") + "'>";
1239 	htmlArr[idx++] = AjxStringUtil.htmlEncode(contact.getFileAs() || contact.getFileAsNoName());
1240 	htmlArr[idx++] = "</div>";
1241 	htmlArr[idx++] = "<div class='ZmListFlagsWrapper'>";
1242 
1243 	if (!params.isDragProxy) {
1244 		// if read only, show lock icon in place of the tag column since we dont
1245 		// currently support tags for "read-only" contacts (i.e. shares)
1246 		var isLocked = folder ? folder.link && folder.isReadOnly() : contact.isLocked();
1247 		if (isLocked) {
1248 			htmlArr[idx++] = AjxImg.getImageHtml("ReadOnly");
1249 		} else if (!contact.isReadOnly() && appCtxt.get(ZmSetting.TAGGING_ENABLED)) {
1250 			// otherwise, show tag if there is one
1251 			idx = this._getImageHtml(htmlArr, idx, contact.getTagImageInfo(), this._getFieldId(contact, ZmItem.F_TAG), ["Tag"]);
1252 		}
1253 	}
1254 
1255 	if (appCtxt.get(ZmSetting.IM_ENABLED)) {
1256 		var presence = contact.getImPresence();
1257 		var img = presence ? presence.getIcon() : "Blank_16";
1258 		idx = this._getImageHtml(htmlArr, idx, img, this._getFieldId(contact, ZmItem.F_PRESENCE), ["Presence"]);
1259 	}
1260 
1261 	htmlArr[idx++] = "</div></div></li>";
1262 
1263 	if (div) {
1264 		div.innerHTML = htmlArr.join("");
1265 		return div;
1266 	} else {
1267 		return htmlArr.join("");
1268 	}
1269 };
