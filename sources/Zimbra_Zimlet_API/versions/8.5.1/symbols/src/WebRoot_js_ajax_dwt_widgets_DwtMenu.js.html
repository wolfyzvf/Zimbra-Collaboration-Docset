 22 /**
 23  * Creates a menu.
 24  * @constructor
 25  * @class
 26  * Creates a menu object to menu items can be added. Menus can be created in various styles as
 27  * follows:
 28  * <ul>
 29  * <li>DwtMenu.BAR_STYLE - Traditional menu bar</li>
 30  * <li>DwtMenu.POPUP_STYLE - Popup menu</li>
 31  * <li>DwtMenu.DROPDOWN_STYLE - Used when a menu is a drop down (e.g. parent is a button or another menu item)</li>
 32  * <li>DwtMenu.DROPDOWN_CENTERV_STYLE - like a dropdown, but position to the right, centered vertically on the parent</li>
 33  * <li>DwtMenu.COLOR_PICKER_STYLE - Menu is hosting a single color picker</li>
 34  * <li>DwtMenu.CALENDAR_PICKER_STYLE - Menu is hostng a single calendar</li>
 35  * <li>DwtMenu.GENERIC_WIDGET_STYLE - Menu is hosting a single "DwtInsertTableGrid"</li>
 36  * </ul>
 37   *
 38  * @author Ross Dargahi
 39  * 
 40  * @param {hash}	params		a hash of parameters
 41  * @param       {DwtComposite}	params.parent		the parent widget
 42  * @param {constant}      params.style			the menu style
 43  * @param {string}        params.className		the CSS class
 44  * @param {constant}      params.posStyle		the positioning style (see {@link DwtControl})
 45  * @param {constant}      params.layout			layout to use: DwtMenu.LAYOUT_STACK, DwtMenu.LAYOUT_CASCADE or DwtMenu.LAYOUT_SCROLL. A value of [true] defaults to DwtMenu.LAYOUT_CASCADE and a value of [false] defaults to DwtMenu.LAYOUT_STACK.
 46  * @param {int}		  params.maxRows=0	    	if >0 and layout = LAYOUT_CASCADE or DwtMenu.LAYOUT_SCROLL, define how many rows are allowed before cascading/scrolling
 47  * @param {boolean}		params.congruent		if the parent is a DwtMenuItem, align so that the submenu "merges" with the parent menu
 48  * 
 49  * @extends		DwtComposite
 50  */
 51 
 52 DwtMenu = function(params) {
 53 	this._created = false;
 54 	if (arguments.length == 0) { return; }
 55 	params = Dwt.getParams(arguments, DwtMenu.PARAMS);
 56 
 57 	this._origStyle = params.style;
 58 	var parent = params.parent;
 59 	if (parent) {
 60 		if (parent instanceof DwtMenuItem || parent instanceof DwtButton) {
 61 			if ((params.style == DwtMenu.GENERIC_WIDGET_STYLE) ||
 62                 (params.style == DwtMenu.DROPDOWN_CENTERV_STYLE)) {
 63 				this._style = params.style;
 64 			} else {
 65                 this._style = DwtMenu.DROPDOWN_STYLE;
 66  			}
 67 		} else {
 68 			this._style = params.style || DwtMenu.POPUP_STYLE;
 69 		}
 70 		if (!params.posStyle) {
 71 			params.posStyle = (this._style == DwtMenu.BAR_STYLE) ? DwtControl.STATIC_STYLE : DwtControl.ABSOLUTE_STYLE;
 72 		}
 73 	}
 74 	params.className = params.className || "DwtMenu";
 75 
 76 	this._layoutStyle = params.layout == null || params.layout;
 77 	if (this._layoutStyle === true) {
 78 		this._layoutStyle = DwtMenu.LAYOUT_CASCADE;
 79 	} else if (this._layoutStyle === false) {
 80 		this._layoutStyle = DwtMenu.LAYOUT_STACK;
 81 	}
 82 	this._maxRows = this._layoutStyle && params.maxRows || 0;
 83 	this._congruent = params.congruent;
 84 
 85 	// Hack to force us to hang off of the shell for positioning.
 86 	params.parent = (parent instanceof DwtShell) ? parent : parent.shell;
 87 	DwtComposite.call(this, params);
 88 	this.parent = parent;
 89 
 90 	if (this._isPopupStyle() && (this._layoutStyle == DwtMenu.LAYOUT_STACK)) {
 91 		this.setScrollStyle(DwtControl.SCROLL);
 92 	}
 93 
 94 	if (!parent) { return; }
 95 
 96 	var events = AjxEnv.isIE ? [DwtEvent.ONMOUSEDOWN, DwtEvent.ONMOUSEUP] :
 97 							   [DwtEvent.ONMOUSEDOWN, DwtEvent.ONMOUSEUP, DwtEvent.ONMOUSEOVER, DwtEvent.ONMOUSEOUT];
 98 	this._setEventHdlrs(events);
 99 	this._hasSetMouseEvents = true;
100 	
101 	var htmlElement = this.getHtmlElement();
102 
103 	if (params.posStyle != DwtControl.STATIC_STYLE) {
104 		Dwt.setLocation(htmlElement, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
105 	}
106 
107 	// Don't need to create table for color picker and calendar picker styles
108 	if (this._style != DwtMenu.COLOR_PICKER_STYLE &&
109 		this._style != DwtMenu.CALENDAR_PICKER_STYLE &&
110 		this._style != DwtMenu.GENERIC_WIDGET_STYLE)
111 	{
112 		this._table = document.createElement("table");
113 		this._table.border = this._table.cellPadding = this._table.cellSpacing = 0;
114 		this._table.className = "DwtMenuTable";
115 		this._table.id = Dwt.getNextId();
116 
117 
118 		if (this._layoutStyle == DwtMenu.LAYOUT_SCROLL) {
119 			this._setupScroll();
120 		} else {
121 			htmlElement.appendChild(this._table);
122 		}
123 		this._table.backgroundColor = DwtCssStyle.getProperty(htmlElement, "background-color");
124     }
125 
126 	if (params.style != DwtMenu.BAR_STYLE) {
127 		this.setZIndex(Dwt.Z_HIDDEN);
128  		this._isPoppedUp = false;
129 	} else {
130 		DwtMenu._activeMenuIds.add(htmlElement.id, null, true);
131 		this._isPoppedUp = true;
132  	}
133 	this._popdownAction = new AjxTimedAction(this, this._doPopdown);
134 	this._popdownActionId = -1;
135 	this._popupAction = new AjxTimedAction(this, this._doPopup);
136 	this._popupActionId = -1;
137 
138 	this._outsideListener = new AjxListener(null, DwtMenu._outsideMouseDownListener);
139 
140 	this._menuItemsHaveChecks = false;	
141 	this._menuItemsHaveIcons = false;
142 	this._menuItemsWithSubmenus = 0;
143 	this.__currentItem = null;
144 	this.__preventMenuFocus = false;
145 
146 	// Default menu tab group.
147 	this._tabGroup = new DwtTabGroup(this.toString());
148 	this._tabGroup.addMember(this);
149 	this._created = true;
150 
151     // When items are added, the menu listens to selection events
152     // and will propagate the event to listeners that are registered
153     // on the menu itself.
154     this._itemSelectionListener = new AjxListener(this, this._propagateItemSelection);
155 };

287 /**
288  * Gets the item by id.
289  * 
290  * @param	{string}	key		the id key
291  * @param	{Object}	id		the id value
292  * @return	{DwtMenuItem}	the menu item
293  */
294 DwtMenu.prototype.getItemById =
295 function(key, id) {
296 	var items = this.getItems();
297 	for (var i = 0; i < items.length; i++) {
298 		var itemId = items[i].getData(key);
299 		if (itemId == id) {
300 			items[i].index = i; //needed in some caller
301 			return items[i];
302 		}
303 	}
304 	return null;
305 };

348 DwtMenu.prototype.popup =
349 function(msec, x, y, kbGenerated) {
350 	if (this._style == DwtMenu.BAR_STYLE) return;
351 	
352 	if (this._popdownActionId != -1) {
353 		AjxTimedAction.cancelAction(this._popdownActionId);
354 		this._popdownActionId = -1;
355 	} else {
356 		if (this._isPoppedUp || (this._popupActionId != -1 && msec && msec > 0)) {
357 			return;
358 		} else if (this._popupActionId != -1){
359 			AjxTimedAction.cancelAction(this._popupActionId);
360 			this._popupActionId = -1;
361 		}
362 
363 		if (!msec) {
364 			this._doPopup(x, y, kbGenerated);
365 		} else {
366 			this._popupAction.args = [x, y, kbGenerated];
367 			this._popupActionId = AjxTimedAction.scheduleAction(this._popupAction, msec);
368 		}
369 	}
370 };
371 
372 DwtMenu.prototype.popdown =
373 function(msec, ev) {
374 	if (this._style == DwtMenu.BAR_STYLE) return;
375 
376 	if (this._popupActionId != -1) {
377 		AjxTimedAction.cancelAction(this._popupActionId);	
378 		this._popupActionId = -1;
379 	} else {
380 		if (!this._isPoppedUp || this._popdownActionId != -1)
381 			return;
382 		if (msec == null || msec == 0)
383 			this._doPopdown(ev);
384 		else
385 			this._popdownActionId = AjxTimedAction.scheduleAction(this._popdownAction, msec);
386 	}
387 };

389 DwtMenu.prototype._setupScroll = function() {
390 	var htmlElement = this.getHtmlElement();
391 	this._table.style.position = "relative";
392 			
393 	this._topScroller = document.createElement("div");
394 	this._topScroller.className = "DwtMenuScrollTop";
395 	this._topScroller.id = Dwt.getNextId();
396 	
397 	this._imgDivTop = document.createElement("div");
398 	this._imgDivTop.className = "ImgUpArrowSmall";
399 	this._topScroller.appendChild(this._imgDivTop);
400 	Dwt.setHandler(this._imgDivTop, DwtEvent.ONMOUSEOUT, DwtMenu._stopEvent);
401 	Dwt.setHandler(this._imgDivTop, DwtEvent.ONMOUSEOVER, DwtMenu._stopEvent);
402 	htmlElement.appendChild(this._topScroller);
403 
404 	this._tableContainer = document.createElement("div");
405 	this._tableContainer.appendChild(this._table);
406 	htmlElement.appendChild(this._tableContainer);
407 
408 	this._bottomScroller = document.createElement("div");
409 	this._bottomScroller.className = "DwtMenuScrollBottom";
410 	this._bottomScroller.id = Dwt.getNextId();
411 	
412 	this._imgDivBottom = document.createElement("div");
413 	this._imgDivBottom.className = "ImgDownArrowSmall";
414 	Dwt.setHandler(this._imgDivBottom, DwtEvent.ONMOUSEOUT, DwtMenu._stopEvent);
415 	Dwt.setHandler(this._imgDivBottom, DwtEvent.ONMOUSEOVER, DwtMenu._stopEvent);
416 	this._bottomScroller.appendChild(this._imgDivBottom);
417 	htmlElement.appendChild(this._bottomScroller);
418 
419 	//scroll up
420 	var scrollUpStartListener = AjxCallback.simpleClosure(this._scroll, this, this._table.id, true, false);
421 	var scrollUpStopListener = AjxCallback.simpleClosure(this._scroll, this, this._table.id, false, false);
422 	var mouseOutTopListener = AjxCallback.simpleClosure(this._handleMouseOut, this, this._topScroller.id, this._table.id);
423 	var mouseOutBottomListener = AjxCallback.simpleClosure(this._handleMouseOut, this, this._bottomScroller.id, this._table.id);
424 
425 	Dwt.setHandler(this._topScroller, DwtEvent.ONMOUSEDOWN, scrollUpStartListener);
426 	Dwt.setHandler(this._topScroller, DwtEvent.ONMOUSEUP, scrollUpStopListener);
427 	if (!AjxEnv.isIE) {
428 		Dwt.setHandler(this._topScroller, DwtEvent.ONMOUSEOUT, mouseOutTopListener);
429 	} else {
430 		Dwt.setHandler(this._topScroller, DwtEvent.ONMOUSELEAVE, scrollUpStopListener);
431 	}
432 
433 	//scroll down
434 	var scrollDownStartListener = AjxCallback.simpleClosure(this._scroll, this, this._table.id, true, true);
435 	var scrollDownStopListener = AjxCallback.simpleClosure(this._scroll, this, this._table.id, false, true);
436 
437 	Dwt.setHandler(this._bottomScroller, DwtEvent.ONMOUSEDOWN, scrollDownStartListener);
438 	Dwt.setHandler(this._bottomScroller, DwtEvent.ONMOUSEUP, scrollDownStopListener);
439 	Dwt.setHandler(this._bottomScroller, DwtEvent.ONMOUSEUP, scrollDownStopListener);
440 	if (!AjxEnv.isIE) {
441 		Dwt.setHandler(this._bottomScroller, DwtEvent.ONMOUSEOUT, mouseOutBottomListener);
442 	} else {
443 		Dwt.setHandler(this._bottomScroller, DwtEvent.ONMOUSELEAVE, scrollDownStopListener);
444 	}
445 
446 	var wheelListener = AjxCallback.simpleClosure(this._handleScroll, this, this._table.id);
447 	Dwt.setHandler(htmlElement, DwtEvent.ONMOUSEWHEEL, wheelListener);
448 };

450 DwtMenu.prototype.render =
451 function(x, y) {
452 	var windowSize = this.shell.getSize();
453 	var mySize = this.getSize();
454 	var htmlEl = this.getHtmlElement();
455 
456 	// bug 9583 - can't query border size so just subtract generic padding
457 	windowSize.y -= 10 + (AjxEnv.isIE ? 20 : 0);
458 	windowSize.x -= 28;
459 
460 	var isScroll = this._layoutStyle == DwtMenu.LAYOUT_SCROLL;
461 	var isPopup = this._isPopupStyle();
462 	var isCascade = this._layoutStyle == DwtMenu.LAYOUT_CASCADE;
463 	if (this._table) {
464 		if (isPopup && isCascade) {
465 			var space = windowSize.y;
466 			var newY = null;
467 			var rows = this._table.rows;
468 			var numRows = rows.length;
469 			var maxRows = this._maxRows;
470 			var height = mySize.y;
471 			var requiredSpace = space - 25; // Account for space on top & bottom of menu.
472 			for (var i = numRows - 1; i >= 0; i--) {
473 				height -= Dwt.getSize(rows[i]).y;
474 				if (height < requiredSpace) {
475 					break;
476 				}
477 			}
478 			var count = maxRows ? Math.min(i + 1, maxRows) : (i + 1);
479 			for (var j = count; j < numRows; j++) {
480 				var row = rows[(j - count) % count];
481 				var cell = row.insertCell(-1);
482 				cell.className = "DwtMenuCascadeCell";
483 				var child = rows[j].cells[0].firstChild;
484 				while (child != null) {
485 					cell.appendChild(child);
486 					child = child.nextSibling;
487 				}
488 			}
489 			for (j = rows.length - 1; j >= count; j--) {
490 				this._table.deleteRow(count);
491 			}
492 			var offset = numRows % count;
493 			if (offset > 0) {
494 				for (var j = offset; j < count; j++) {
495 					var row = rows[j];
496 					var cell = row.insertCell(-1);
497 					cell.className = "DwtMenuCascadeCell";
498 					cell.empty = true;
499 					cell.innerHTML = " ";
500 				}
501 			}
502 
503 			mySize = this.getSize();
504 			if (newY) {
505 				y = newY - mySize.y;
506 			}
507 		} else if (isPopup && isScroll) {
508 			var rows = this._table.rows;
509 			var numRows = rows.length;
510 			var maxRows = this._maxRows;
511 			var limRows = maxRows ? Math.min(maxRows, numRows) : numRows;
512 			var availableSpace = windowSize.y - 25; // Account for space on top & bottom of menu.
513 
514 			var height = 20; //for scroll buttons
515 			for (var i = 0; i < limRows; i++) {
516 				var rowSize = Dwt.getSize(rows[i]).y;
517 				if (height + rowSize <= availableSpace)
518 					height += rowSize;
519 				else
520 					break;
521 			}
522 			mySize.y = height;
523 		}
524 	}
525 	var newW = "auto";
526 	var newH = "auto";
527 	if (isPopup && isScroll) {
528 		newH = mySize.y;
529 		if (this._tableContainer)
530 			this._tableContainer.style.height = (newH - 20) +"px";
531 	} else if ((isPopup && isCascade) || y + mySize.y < windowSize.y - 5 ) {
532 		newH = "auto";
533 	} else {
534 		newH = windowSize.y - y - 5;
535 	}
536     if(isScroll) {
537 	    if (this._table) {
538 		    this._table.style.width = mySize.x;
539         }
540         newW = mySize.x;
541     }
542     this.setSize(newW, newH);
543 	// NOTE: This hack is needed for FF/Moz because the containing div
544 	//	   allows the inner table to overflow. When the menu cascades
545 	//	   and the menu items get pushed off of the visible area, the
546 	//	   div's border doesn't surround the menu items. This hack
547 	//	   forces the outer div's width to surround the table.
548 
549 	if ((AjxEnv.isGeckoBased || AjxEnv.isSafari || (this._origStyle == DwtMenu.CALENDAR_PICKER_STYLE)) && this._table && !isScroll) {
550 		htmlEl.style.width = (mySize.x + (isPopup && !isCascade ? 10 : 0)) + "px";
551 	}
552 
553 	// Popup menu type
554 	var newX = x + mySize.x >= windowSize.x ? windowSize.x - mySize.x : x;
555 	if (this.parent instanceof DwtMenuItem) {
556 		Dwt.delClass(htmlEl, "DwtMenu-congruentLeft");
557 		Dwt.delClass(htmlEl, "DwtMenu-congruentRight");
558 
559 		var pbound = this.parent.getBounds();
560 		var pmstyle = DwtCssStyle.getComputedStyleObject(this.parent.parent.getHtmlElement()); // Get the style for the DwtMenu holding the parent DwtMenuItem
561 		var tstyle = DwtCssStyle.getComputedStyleObject(htmlEl); // Get the style for this menu (includes skinning)
562 
563 		//if the cascading extends over the edge of the screen, cascade to the left
564 		if ( ((newX > pbound.x && newX < pbound.x + pbound.width) || (pbound.x > newX && pbound.x < newX + mySize.x)) && pbound.x >= mySize.x) {
565 			var totalWidth = parseInt(tstyle.width);

568 			newX = (parseInt(pmstyle.left) || pbound.x) - (totalWidth || mySize.x);
569 			if (this._congruent) {
570 				var offset;
571 				if (AjxEnv.isIE)
572 					offset = parseInt(tstyle.borderLeftWidth);
573 				else

575 				if (!isNaN(offset)) {
576 					newX += offset;
577 					Dwt.addClass(htmlEl, "DwtMenu-congruentLeft");
578 				}
579 			}
580 
581 		} else { // Cascade to the right
582 			var left = parseInt(pmstyle.left) || (pbound.x - (parseInt(pmstyle.paddingLeft) || 0));
583 			var width = parseInt(pmstyle.width) || pbound.width;
584 			newX = left + width;
585 			if (this._congruent) {
586 				var offset = parseInt(pmstyle.paddingRight) + parseInt(tstyle.paddingLeft) + parseInt(tstyle.borderLeftWidth);
587 				if (!isNaN(offset)) {
588 					newX += offset;
589 					Dwt.addClass(htmlEl, "DwtMenu-congruentRight");
590 				}
591 			}
592 		}
593 	}
594 
595     if (this._style === DwtMenu.DROPDOWN_CENTERV_STYLE) {
596         y -=  mySize.y/2;
597         if (y < 0) {
598             y = 0;
599         }
600     }
601 	var newY = isPopup && y + mySize.y >= windowSize.y ? windowSize.y - mySize.y : y;
602 
603 	if (this.parent instanceof DwtMenuItem && this._congruent) {
604 		var offset = (parseInt(tstyle.paddingTop) || 0) - (parseInt(tstyle.borderTopWidth) || 0);
605 		if (offset>0)
606 			newY -= offset;
607 	}
608 
609 	this.setLocation(newX, newY);

617 DwtMenu.prototype._handleScroll =
618 function(divID, ev) {

654 DwtMenu.prototype._scroll =
655 function(divID, scrolling, direction, ev) {
656 	var div = divID ? document.getElementById(divID) : null;
657 	if (div && scrolling) {
658 		var rows = div.rows;
659 		var step = Dwt.getSize(rows[0]).y || 10;
660 		if (this._direction != direction || !this._scrollTimer) {
661 			this._popdownSubmenus();
662 			this._direction = direction;
663 			if (this._scrollTimer) {
664 				clearInterval(this._scrollTimer);
665 				this._scrollTimer = null;
666 			}
667 	
668 			if (direction) { //scroll down
669 				this._scrollTimer = setInterval(AjxCallback.simpleClosure(this._doScroll, this, div, -step), 100);
670 				this._doScroll(div, -step);
671 			} else { //scroll up
672 				this._scrollTimer = setInterval(AjxCallback.simpleClosure(this._doScroll, this, div, step), 100);
673 				this._doScroll(div, step);
674 			}
675 		}
676 	} else {
677 		if (this._scrollTimer) {
678 			clearInterval(this._scrollTimer);
679 			this._scrollTimer = null;
680 		}
681 	}
682 };

684 DwtMenu.prototype._doScroll =
685 function(div, step) {
686 	if (div && step) {
687 		var old = parseInt(div.style.top) || 0;
688 		var top;
689 		if (step < 0) { // scroll down
690 			var rows = this._table.rows || null;
691 			var height = rows && rows.length && Dwt.getSize(rows[0]).y;
692 			var max = div.scrollHeight - (parseInt(div.parentNode.style.height) || ((this._maxRows || (rows && rows.length)) * height) || 0);
693 			if (Math.abs(old + step) <= max) {
694 				top = old + step;
695 			} else {
696 				top = -max;
697 			}
698 		} else { // scroll up
699 			if ((old + step) < 0) {
700 				top = old + step;
701 			} else {
702 				top = 0;
703 			}
704 		}
705 		Dwt.setLocation(div, Dwt.DEFAULT, top);
706 	}
707 };

715  *

724 DwtMenu.prototype.scrollToIndex = 
725 function(index, justMakeVisible) {
726 	if (this._created && this._layoutStyle == DwtMenu.LAYOUT_SCROLL && index !== null && index >= 0 && this._table) {
727 		var rows = this._table.rows;
728 		if (rows) {
729 			var maxRows = this._maxRows;
730 			var visibleHeight = 0;
731 			var rowHeights = [];
732 			for (var i = 0, numRows = rows.length; i < numRows; i++) {
733 				var h = Dwt.getSize(rows[i]).y;
734 				if (i < maxRows)
735 					visibleHeight += h;
736 				rowHeights.push(h);
737 			}
738 		
739 			var itemHeight = rowHeights[index];
740 			var currentOffset = parseInt(this._table.style.top) || 0;
741 			if (index >= rows.length)
742 				index = rows.length-1;
743 			
744 			var itemOffset = 0;
745 			for (var i=0; i<index && i<rowHeights.length; i++) {
746 				itemOffset += rowHeights[i];
747 			}
748 			var delta = 0;
749 			if (justMakeVisible) {
750 				if (itemOffset < -currentOffset) {
751 					delta = -(itemOffset + currentOffset); // Scroll up, making the item the topmost visible row
752 				} else if (itemOffset + itemHeight > visibleHeight - currentOffset) {
753 					delta = -(itemOffset + currentOffset - visibleHeight + itemHeight); // Scroll down, making the item the lowermost visible row
754 				} // else do not scroll; item is already visible
755 			} else {
756 				delta = -(itemOffset + currentOffset); // Scroll so that the item is the topmost visible row
757 			}
758 			if (delta) {
759 				this._popdownSubmenus();
760 				this._doScroll(this._table, delta);
761 			}
762 		}
763 	}
764 };

766 DwtMenu.prototype.handleKeyAction =
767 function(actionCode, ev) {
768 	// For now don't deal with anything but BAR, POPUP, and DROPDOWN style menus
769 	switch (this._style) {
770 		case DwtMenu.BAR_STYLE:
771 		case DwtMenu.POPUP_STYLE:
772         case DwtMenu.DROPDOWN_STYLE:
773         case DwtMenu.DROPDOWN_CENTERV_STYLE:
774 			break;
775 			
776 		default:
777 			return false;
778 	}
779 
780 	switch (actionCode) {
781 
782 		case DwtKeyMap.PAGE_UP:
783 		case DwtKeyMap.PAGE_DOWN:
784 			var item = this.__currentItem || this._children.get(0);
785 			var index = this.getItemIndex(item);
786 			if (this._maxRows && index!=-1) {
787 				this.setSelectedItem(index + ((actionCode == DwtKeyMap.PAGE_UP) ? -this._maxRows : this._maxRows));
788 			} else {
789 				this.setSelectedItem(actionCode == DwtKeyMap.PAGE_DOWN);
790 			}
791 			break;
792 
793 		case DwtKeyMap.SELECT_NEXT:
794 		case DwtKeyMap.SELECT_PREV: 
795 			this.setSelectedItem(actionCode == DwtKeyMap.SELECT_NEXT);
796 			break;
797 
798 		case DwtKeyMap.SELECT:
799 			if (this.__currentItem) {
800 				this.__currentItem._emulateSingleClick();
801 			}
802 			break;
803 		
804 		case DwtKeyMap.SUBMENU:
805 			if (this.__currentItem && this.__currentItem._menu) {
806 				this.__currentItem._popupMenu(0, true);	
807 			}
808 			break;
809 			
810 		case DwtKeyMap.PARENTMENU:
811 			if (this.parent instanceof DwtMenuItem)
812 				this.popdown(0);
813 			break;
814 			
815 		case DwtKeyMap.CANCEL:
816 			if (this.__currentItem) {
817 				var mev = new DwtMouseEvent();
818 				this._setMouseEvent(mev, {dwtObj:this.__currentItem});
819 				this.notifyListeners(DwtEvent.ONMOUSEOUT, mev);
820 				this.__currentItem = null;
821 			}
822 			this.popdown(0);
823 			break;		
824 			
825 		default:
826 			return false;		
827 	}
828 	
829 	return true;
830 };

885 /**
886  * Programmatically selects a menu item. The item can be specified with an index,
887  * or as the next or previous item based on which item is currently selected. If
888  * the new item is a separator or is disabled, it won't be selected. Instead, the
889  * next suitable item will be used.
890  * 
891  * @param {boolean|number}	which		if <code>true</code>, selects the next menu item
892  * 									if <code>false</code>, selects the previous menu item
893  * 									if <code>DwtMenuItem</code>, select that menu item
894  * 									if <code>int</code>, selects the menu item with that index
895  */
896 DwtMenu.prototype.setSelectedItem =
897 function(which) {
898 	var currItem = this.__currentItem;
899 	if (typeof(which) == "boolean") {
900 		currItem = !currItem
901 			? this._children.get(0)
902 			: which ? this._children.getNext(currItem) : this._children.getPrev(currItem);
903 	} else if (which instanceof DwtMenuItem) {
904 		if (this._children.contains(which))
905 			currItem = which;
906 	} else {
907 		which = Math.max(0, Math.min(this._children.size()-1, which));
908 		currItem = this._children.get(which);
909 	}
910 	// While the current item is not enabled or is a separator, try another
911 	while (currItem) {
912 		if (!currItem.isStyle) { // this is not a DwtMenuItem
913 			currItem.focus();
914 			break;
915 		}
916 		else if (!currItem.isStyle(DwtMenuItem.SEPARATOR_STYLE) && currItem.getEnabled() && currItem.getVisible()) {
917 			break;
918 		}
919 		currItem = (which === false) ? this._children.getPrev(currItem) : this._children.getNext(currItem);
920 	}
921 	if (!currItem) { return; }
922 
923 	// if we have a current item then we need to make sure we simulate a
924 	// mouseout event so that the UI can behave correctly
925 	var mev = new DwtMouseEvent();
926 	if (this.__currentItem) {
927 		this._setMouseEvent(mev, {dwtObj:this.__currentItem});
928 		this.__currentItem.notifyListeners(AjxEnv.isIE ? DwtEvent.ONMOUSELEAVE : DwtEvent.ONMOUSEOUT, mev);
929 	}
930 	this._setMouseEvent(mev, {dwtObj:currItem});
931 	currItem.notifyListeners(AjxEnv.isIE ? DwtEvent.ONMOUSEENTER : DwtEvent.ONMOUSEOVER, mev);	// mouseover selects a menu item
932 	this.__currentItem = currItem;
933 	this.scrollToItem(currItem, true);
934 };

1140 	}

1157 	/* If the popup was keyboard generated, then pick the first enabled child item
1158 	 * we do this by simulating a DwtKeyMap.SELECT_NEXT keyboard action */

1160 	 	this.handleKeyAction(DwtKeyMap.SELECT_NEXT);
1161 	} else if (this.__currentItem) {
1162 		this.setSelectedItem(this.__currentItem);
1163 	}

1178 DwtMenu.prototype._doPopdown =
1179 function(ev) {
1180 	// Notify all sub menus to pop themselves down
1181 	var a = this._children.getArray();
1182 	var s = this._children.size();
1183 	for (var i = 0; i < s; i++) {
1184 		if ((a[i] instanceof DwtMenuItem) && !(a[i].isStyle(DwtMenuItem.SEPARATOR_STYLE))) {
1185 			a[i]._popdownMenu();
1186 		}
1187 	}
1188 	this.setZIndex(Dwt.Z_HIDDEN);
1189 	this.setLocation(Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
1190 	this._ev = ev;
1191 
1192 	this.notifyListeners(DwtEvent.POPDOWN, this);
1193 
1194 	var omem = DwtOutsideMouseEventMgr.INSTANCE;
1195 	omem.stopListening({id:"DwtMenu", obj:this});
1196 
1197 	if (DwtMenu._activeMenu == this) {
1198 		DwtMenu._activeMenu = null;
1199 		DwtMenu._activeMenuUp = false;
1200 	}
1201 	DwtMenu._activeMenuIds.remove(this._htmlElId);
1202 	DwtMenu._activeMenus.remove(this);
1203 	this._popdownActionId = -1;
1204 	this._isPoppedUp = false;
1205 
1206 	if (this._isPopupStyle() && this._table && this._table.rows && this._table.rows.length && this._table.rows[0].cells.length)	{
1207 		var numColumns = this._table.rows[0].cells.length;
1208 		var numRows = this._table.rows.length;
1209 		for (var i = 1; i < numColumns; i++) {
1210 			for (var j = 0; j < numRows; j++) {
1211 				var cell = this._table.rows[j].cells[i];
1212 				if (!cell.empty) {
1213 					var child = cell.firstChild;
1214 					var row = this._table.insertRow(this._table.rows.length);
1215 					var cell = row.insertCell(0);
1216 					while (child != null) {
1217 						cell.appendChild(child);
1218 						child = child.nextSibling;
1219 					}
1220 				}
1221 			}
1222 		}
1223 		for (var j = 0; j < numRows; j++) {
1224 			var row = this._table.rows[j];
1225 			for (var i = row.cells.length - 1; i > 0; i--) {
1226 				row.deleteCell(i);
1227 			}
1228 		}
1229 	}
1230 	
1231 	// set the current item (used in kb nav) to null
1232 	this.__currentItem = null;
1233 
1234 	// Undo highlight if there's a hovered-over item
1235 	if (this._hoveredItem) {
1236 		var ev = new DwtMouseEvent();
1237 		ev.dwtObj = this._hoveredItem;
1238 		DwtButton._mouseOutListener(ev);
1239 	}
1240 	
1241 	// Take our tabgroup out of play
1242 	DwtShell.getShell(window).getKeyboardMgr().popTabGroup(this._tabGroup);	
1243 };

1256 DwtMenu._outsideMouseDownListener =
1257 function(ev) {
1258 
1259 	if (DwtMenu._activeMenuUp) {
1260 		var menu = DwtMenu._activeMenu;
1261 
1262 		// assuming that the active menu is the parent of all other menus
1263 		// that are up, search through the array of child menu dom IDs as
1264 		// well as our own.
1265 		var id = menu._htmlElId;
1266 		var htmlEl = DwtUiEvent.getTarget(ev);
1267 		while (htmlEl != null) {
1268 			if (htmlEl.id && htmlEl.id != "" && 
1269 				(htmlEl.id == id || htmlEl.id == menu._associatedElId ||
1270 				 DwtMenu._activeMenuIds.binarySearch(htmlEl.id) != -1 )) {
1271 				return false;
1272 			}
1273 			htmlEl = htmlEl.parentNode;
1274 		}
1275 
1276 		// If we've gotten here, the mousedown happened outside the active
1277 		// menu, so we hide it.
1278 		menu.popdown(0, ev);
1279 		
1280 		//it should remove all the active menus 
1281 		var cMenu = null ;
1282 		do {
1283 			cMenu = DwtMenu._activeMenus.getLast();
1284 			if (cMenu!= null && cMenu instanceof DwtMenu) cMenu.popdown();
1285 		} while (cMenu != null) ;
1286 	}
1287 };
