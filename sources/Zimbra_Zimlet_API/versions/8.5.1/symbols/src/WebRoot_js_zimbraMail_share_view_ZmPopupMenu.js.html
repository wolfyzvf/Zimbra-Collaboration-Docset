 26 /**
 27  * Creates a popup menu.
 28  * @class
 29  * This class represents a basic popup menu which can add menu items, manage listeners, and
 30  * enable/disabled its menu items.
 31  *
 32  * @author Conrad Damon
 33  *
 34  * @param {DwtComposite}	parent		the containing widget
 35  * @param {string}	className		the CSS class
 36  * @param {string}	id			an explicit ID to use for the control's HTML element
 37  * @param {ZmController}	controller	the owning controller
 38  * 
 39  * @extends		DwtMenu
 40  */
 41 ZmPopupMenu = function(parent, className, id, controller) {
 42 
 43 	if (arguments.length == 0) return;
 44 	var params = Dwt.getParams(arguments, ZmPopupMenu.PARAMS);
 45 	params.className = params.className ? params.className : "ActionMenu";
 46 	params.style = params.style || DwtMenu.POPUP_STYLE;
 47     params.id = params.id || "POPUP_" + Dwt.getNextId();
 48 	DwtMenu.call(this, params);
 49 
 50 	controller = controller || appCtxt.getCurrentController();
 51 	if (controller) {
 52 		this._controller = controller;
 53 		this._keyMap = this._controller.getKeyMapName();
 54 	}
 55 
 56 	this._menuItems = {};
 57 };
 58 
 59 ZmPopupMenu.PARAMS = ["parent", "className", "id", "controller"];
 60 
 61 ZmPopupMenu.prototype = new DwtMenu;
 62 ZmPopupMenu.prototype.constructor = ZmPopupMenu;
 63 
 64 ZmPopupMenu.prototype.toString = 
 65 function() {
 66 	return "ZmPopupMenu";
 67 };
 68 
 69 ZmPopupMenu.MENU_ITEM_ID_KEY = "menuItemId";
 70 
 71 /**
 72  * Adds a section listener.
 73  * 
 74  * @param	{string}		menuItemId		the menu item id
 75  * @param	{AjxListener}	listener		the selection listener
 76  * @param	{number}		index				the index where to insert the listener
 77  */
 78 ZmPopupMenu.prototype.addSelectionListener =
 79 function(menuItemId, listener, index) {
 80 	var menuItem = this._menuItems[menuItemId];
 81 	if (menuItem) {
 82 		menuItem.addSelectionListener(listener, index);
 83 	}
 84 };
 85 
 86 /**
 87  * Removes a section listener.
 88  * 
 89  * @param	{string}		menuItemId		the menu item id
 90  * @param	{AjxListener}	listener		the selection listener
 91  */
 92 ZmPopupMenu.prototype.removeSelectionListener =
 93 function(menuItemId, listener) {
 94 	var menuItem = this._menuItems[menuItemId];
 95 	if (menuItem) {
 96 		menuItem.removeSelectionListener(listener);
 97 	}
 98 };

100 ZmPopupMenu.prototype.popup =
101 function(delay, x, y, kbGenerated) {
102 	delay = delay ? delay : 0;
103 	x = (x != null) ? x : Dwt.DEFAULT;
104 	y = (y != null) ? y : Dwt.DEFAULT;
105 	DwtMenu.prototype.popup.call(this, delay, x, y, kbGenerated);
106 };

108 /**
109  * Enables/disables menu items.
110  *
111  * @param {array}	ids		a list of menu item IDs
112  * @param {boolean}		enabled	if <code>true</code>, enable the menu items
113  */
114 ZmPopupMenu.prototype.enable =
115 function(ids, enabled) {
116 	ids = (ids instanceof Array) ? ids : [ids];
117 	for (var i = 0; i < ids.length; i++) {
118 		if (this._menuItems[ids[i]]) {
119 			this._menuItems[ids[i]].setEnabled(enabled);
120 		}
121 	}
122 };
123 
124 /**
125  * Enables/disables all menu items.
126  *
127  * @param {boolean}		enabled	if <code>true</code>, enable the menu items
128  */
129 ZmPopupMenu.prototype.enableAll =
130 function(enabled) {
131 	for (var i in this._menuItems) {
132 		this._menuItems[i].setEnabled(enabled);
133 	}
134 };

136 /**
137  * Creates a menu item and adds the item to this menu.
138  *
139  * @param {string}	id			the menu item ID
140  * @param {hash}	params		a hash of parameters
141  * @param {string}	params.text		the menu item text
142  * @param {string}	params.image		the icon class for the or menu item
143  * @param {string}	params.disImage	disabled version of icon
144  * @param {boolean}	params.enabled		if <code>true</code>, menu item is enabled
145  * @param {constant}	params.style			the menu item style
146  * @param {string}	params.radioGroupId	the ID of radio group for this menu item
147  * @param {constant}	params.shortcut		the shortcut ID (from {@link ZmKeyMap}) for showing hint
148  * 
149  * @see		DwtMenuItem
150  */
151 ZmPopupMenu.prototype.createMenuItem =
152 function(id, params, htmlElId) {
153 	var mi = this._menuItems[id] = new DwtMenuItem({parent:this, style:params.style, radioGroupId:params.radioGroupId,
154 													id: (htmlElId || params.id || id), index: params.index});
155 	if (params.image) {
156 		mi.setImage(params.image);
157 	}
158 	if (params.text) {
159 		mi.setText(params.text);
160 	}
161 	if (params.shortcut) {
162 		mi.setShortcut(appCtxt.getShortcutHint(this._keyMap, params.shortcut));
163 	}
164 
165 	mi.setEnabled(params.enabled !== false);
166 	mi.setData(ZmPopupMenu.MENU_ITEM_ID_KEY, id);
167 
168     //set context menu tr id
169     var row = mi.getRowElement();
170 	if (row) {
171 		row.setAttribute("id", "POPUP" + "_" + mi.getHTMLElId().toString().replace(/\s/g, ""));
172     }
173 
174 	return mi;
175 };

177 /**
178  * Gets the menu item with the given ID.
179  *
180  * @param {string}	id		an operation ID
181  * @return	{DwtMenuItem}		the menu item
182  */
183 ZmPopupMenu.prototype.getMenuItem =
184 function(id) {
185 	return this._menuItems[id];
186 };
187 
188 /**
189  * sets an item visibility. finds the menu item by id. 
190  *
191  * @param	{String}	id  the operation id
192  * @param	{Boolean}	visible
193  */
194 ZmPopupMenu.prototype.setItemVisible =
195 function(id, visible) {
196 	var item = this.getMenuItem(id);
197 	if (!item) {
198 		return;
199 	}
200 	item.setVisible(visible);
201 };

206  * @return	{array}		an array of {@link DwtMenuItem} objects
207  */
208 ZmPopupMenu.prototype.getMenuItems =
209 function() {
210 	return this._menuItems;

213 /**
214  * Gets the menu search sub-menu (if any).
215  *
216  * @return {DwtMenu}        the menu
217  */
218 ZmPopupMenu.prototype.getSearchMenu =
219 function() {
220     var menuItem = this.getMenuItem(ZmOperation.SEARCH_MENU);
221     if (menuItem) {
222         return menuItem.getMenu();
223     }
224 };

227 function() {
228 	var menuItem = this.getMenuItem(ZmOperation.CONTACTGROUP_MENU);
229 	if (menuItem) {

231 	}
