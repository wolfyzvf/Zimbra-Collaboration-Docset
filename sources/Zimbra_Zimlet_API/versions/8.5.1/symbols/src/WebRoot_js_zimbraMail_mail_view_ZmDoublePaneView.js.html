 22 ZmDoublePaneView = function(params) {
 23 
 24 	if (arguments.length == 0) { return; }
 25 
 26 	var view = this._view = params.view = params.controller.getCurrentViewId();
 27 	params.id = ZmId.getViewId(view);
 28 	DwtComposite.call(this, params);
 29 
 30 	this._controller = params.controller;
 31 	this._initHeader();
 32 
 33 	params.className = null;
 34 	params.id = DwtId.getListViewId(view);
 35 	params.parent = this;
 36 	params.posStyle = Dwt.ABSOLUTE_STYLE;
 37 	this._mailListView = this._createMailListView(params);
 38 
 39 	// create the item view
 40 	params.className = null;
 41 	this._itemView = this._createMailItemView(params);
 42 
 43 	// create a sash for each of the two reading pane locations
 44 	this._vertSash = new DwtSash({parent:this, style:DwtSash.HORIZONTAL_STYLE, className:"AppSash-horiz",
 45 								  threshold:ZmDoublePaneView.SASH_THRESHOLD, posStyle:Dwt.ABSOLUTE_STYLE});
 46 	this._vertSash.registerCallback(this._sashCallback, this);
 47 	this._vertSash.addListener(DwtEvent.ONMOUSEUP, new AjxListener(this, this._sashVertRelease));
 48 	this._horizSash = new DwtSash({parent:this, style:DwtSash.VERTICAL_STYLE, className:"AppSash-vert",
 49 								   threshold:ZmDoublePaneView.SASH_THRESHOLD, posStyle:Dwt.ABSOLUTE_STYLE});
 50 	this._horizSash.registerCallback(this._sashCallback, this);
 51 	this._horizSash.addListener(DwtEvent.ONMOUSEUP, new AjxListener(this, this._sashHorizRelease));
 52 	this.addListener(DwtEvent.CONTROL, new AjxListener(this, this._controlEventListener));
 53 
 54 	this.setReadingPane();
 55 };

 83 /**
 84  * Displays the reading pane, based on the current settings.
 85  */
 86 ZmDoublePaneView.prototype.setReadingPane =
 87 function(noSet) {
 88 
 89 	var mlv = this._mailListView, mv = this._itemView;
 90 	var readingPaneEnabled = this._controller.isReadingPaneOn();
 91 	if (!readingPaneEnabled) {
 92 		mv.setVisible(false);
 93 		this._vertSash.setVisible(false);
 94 		this._horizSash.setVisible(false);
 95 	} else {
 96 		if (!mv.getVisible()) {
 97 			if (mlv.getSelectionCount() == 1) {
 98 				this._controller._setSelectedItem();
 99 			} else {
100 				mv.reset();
101 			}
102 		}
103 		var readingPaneOnRight = this._controller.isReadingPaneOnRight();
104 		mv.setVisible(true, readingPaneOnRight);
105 		var newSash = readingPaneOnRight ? this._vertSash : this._horizSash;
106 		var oldSash = readingPaneOnRight ? this._horizSash : this._vertSash;
107 		oldSash.setVisible(false);
108 		newSash.setVisible(true);
109 	}
110 
111 	mlv.reRenderListView();
112     if (!noSet) {
113 	    mv.setReadingPane();
114     }
115 
116 	mv.noTab = !readingPaneEnabled || AjxEnv.isIE;
117 	var sz = this.getSize();
118 	this._resetSize(sz.x, sz.y, true);
119 };

231 ZmDoublePaneView.prototype._resetSize = 
232 function(newWidth, newHeight, force) {
233 
234 	if (newWidth <= 0 || newHeight <= 0) { return; }
235 	if (!force && newWidth == this._lastResetWidth && newHeight == this._lastResetHeight) { return; }
236 
237 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
238 
239 	if (this.isReadingPaneVisible()) {
240 		var sash = this.getSash();
241 		var sashSize = sash.getSize();
242 		var sashThickness = readingPaneOnRight ? sashSize.x : sashSize.y;
243 		if (readingPaneOnRight) {
244 			var listViewWidth = this.getReadingSashPosition(true) || (Number(ZmMsg.LISTVIEW_WIDTH)) || Math.floor(newWidth / 2.5);
245 			this._mailListView.resetSize(listViewWidth, newHeight);
246 			sash.setLocation(listViewWidth, 0);
247 			this._itemView.setBounds(listViewWidth + sashThickness, 0,
248 									newWidth - (listViewWidth + sashThickness), newHeight);
249 		} else {
250 			var listViewHeight = this.getReadingSashPosition(false) || (Math.floor(newHeight / 2) - DwtListView.HEADERITEM_HEIGHT);
251 			this._mailListView.resetSize(newWidth, listViewHeight);
252 			sash.setLocation(0, listViewHeight);
253 			this._itemView.setBounds(0, listViewHeight + sashThickness, newWidth,
254 									newHeight - (listViewHeight + sashThickness));
255 		}
256 	} else {
257 		this._mailListView.resetSize(newWidth, newHeight);
258 	}
259 	this._mailListView._resetColWidth();
260 
261 	this._lastResetWidth = newWidth;
262 	this._lastResetHeight = newHeight;
263 };

265 ZmDoublePaneView.prototype._sashCallback =
266 function(delta) {
267 
268 	var readingPaneOnRight = this._controller.isReadingPaneOnRight();
269 	var listView = this._mailListView;
270 	var itemView = this._itemView;
271 	//See bug 69593 for the reason for "true"
272 	var itemViewSize = itemView.getSize(true);
273 	var listViewSize = listView.getSize(true);
274 
275 	var newListViewSize;
276 	var newItemViewBounds;
277 
278 	var absDelta = Math.abs(delta);
279 
280 	if (readingPaneOnRight) {
281 		var currentListViewWidth = AjxEnv.isIE ? this._vertSash.getLocation().x : listViewSize.x;
282 		var currentItemViewWidth = itemViewSize.x;
283 		delta = this._getLimitedDelta(delta, currentItemViewWidth, itemView.getMinWidth(), currentListViewWidth, ZmDoublePaneView.MIN_LISTVIEW_WIDTH);
284 		if (!delta) {
285 			return 0;
286 		}
287 		newListViewSize = {width: currentListViewWidth + delta, height: Dwt.DEFAULT};
288 		newItemViewBounds = {
289 			left: itemView.getLocation().x + delta,
290 			top: Dwt.DEFAULT,
291 			width: currentItemViewWidth - delta,
292 			height: Dwt.DEFAULT
293 		};
294 	}
295 	else {
296 		//reading pane on bottom
297 		var currentListViewHeight = AjxEnv.isIE ? this._horizSash.getLocation().y : listViewSize.y;
298 		var currentItemViewHeight = itemViewSize.y;
299 		delta = this._getLimitedDelta(delta, currentItemViewHeight, itemView.getMinHeight(), currentListViewHeight, this._getMinListViewHeight(listView));
300 		if (!delta) {
301 			return 0;
302 		}
303 		newListViewSize = {width: Dwt.DEFAULT, height: currentListViewHeight + delta};
304 		newItemViewBounds = {
305 			left: Dwt.DEFAULT,
306 			top: itemView.getLocation().y + delta,
307 			width: Dwt.DEFAULT,
308 			height: currentItemViewHeight - delta
309 		};
310 	}
311 
312 	listView.resetSize(newListViewSize.width, newListViewSize.height);
313 	itemView.setBounds(newItemViewBounds.left, newItemViewBounds.top, newItemViewBounds.width, newItemViewBounds.height);
314 
315 	listView._resetColWidth();
316 	if (readingPaneOnRight) {
317 		this._vertSashX = this._vertSash.getLocation().x + delta;
318 	}
319 	else {
320 		this._horizSashY = this._horizSash.getLocation().y + delta;
321 	}
322 
323 	return delta;
324 };
325 
326 /**
327  * returns the delta after limiting it based on minimum view dimension (which is either width/height - this code doesn't care)
328  *
329  * @param delta
330  * @param currentItemViewDimension
331  * @param minItemViewDimension
332  * @param currentListViewDimension
333  * @param minListViewDimension
334  * @returns {number}
335  * @private
336  */
337 ZmDoublePaneView.prototype._getLimitedDelta =
338 function(delta, currentItemViewDimension, minItemViewDimension, currentListViewDimension, minListViewDimension) {
339 	if (delta > 0) {
340 		// moving sash right or down
341 		return Math.max(0, Math.min(delta, currentItemViewDimension - minItemViewDimension));
342 	}
343 	// moving sash left or up
344 	var absDelta = Math.abs(delta);
345 	return -Math.max(0, Math.min(absDelta, currentListViewDimension - minListViewDimension));
346 };

348 ZmDoublePaneView.prototype._getMinListViewHeight =
349 function(listView) {
350 	if (this._minListViewHeight) {
351 		return this._minListViewHeight;
352 	}
353 
354 	var list = listView.getList();
355 	if (!list || !list.size()) {
356 		return DwtListView.HEADERITEM_HEIGHT;
357 	}
358 	//only cache it if there's a list, to prevent a subtle bug of setting to just the header height if
359 	//user first views an empty list.
360 	var item = list.get(0);
361 	var div = document.getElementById(listView._getItemId(item));
362 	this._minListViewHeight = DwtListView.HEADERITEM_HEIGHT + Dwt.getSize(div).y * 2;
363 	return this._minListViewHeight;
364 };

386 ZmDoublePaneView.prototype._staleHandler =
387 function() {
388 
389 	var search = this._controller._currentSearch;
390 	if (search) {
391 		search.lastId = search.lastSortVal = null;
392 		search.offset = search.limit = 0;
393 		var params = {isRefresh: true};
394 		var mlv = this._mailListView
395 		if (mlv.getSelectionCount() == 1) {
396 			var sel = mlv.getSelection();
397 			var selItem = sel && sel[0];
398 			var curItem = this.getItem();
399 			if (selItem && curItem && selItem.id == curItem.id) {
400 				params.selectedItem = selItem;
401 			}
402 		}
403 		appCtxt.getSearchController().redoSearch(search, false, params);
404 	}
405 };

412 /**
413  * Returns the sash location (in pixels) based on reading pane preference.
414  * 
415  * @param {boolean}		readingPaneOnRight   true if reading pane is on the right
416  */
417 ZmDoublePaneView.prototype.getReadingSashPosition =
418 function(readingPaneOnRight) {
419 	if (readingPaneOnRight) {
420 		if (!this._vertSashX) {
421 			var value = this._readingPaneSashVertPos || appCtxt.get(ZmSetting.READING_PANE_SASH_VERTICAL);
422 			var percentWidth = value / 100;
423 			var screenWidth = this.getSize().x;
424 			this._vertSashX = Math.round(percentWidth * screenWidth);
425 		}
426 		return this._vertSashX;
427 	}
428 	else {
429 		if (!this._horizSashY) {
430 			var value = this._readingPaneSashHorizPos || appCtxt.get(ZmSetting.READING_PANE_SASH_HORIZONTAL);
431 			var percentHeight = value / 100;
432 			var screenHeight = this.getSize().y;
433 			this._horizSashY = Math.round(percentHeight * screenHeight);
434 		}
435 		return this._horizSashY;
436 	}
437 };

439 /**
440  * Sets the location of sash (in percentage) depending upon reading pane preference.
441  * 
442  * @param {boolean}		readingPaneOnRight	true if reading pane is on the right
443  * @param {int}			value   			location of sash (in pixels)
444  */
445 ZmDoublePaneView.prototype.setReadingSashPosition =
446 function(readingPaneOnRight, value) {
447 	if (readingPaneOnRight) {
448 		var screenWidth = this.getSize().x;
449 		var sashWidthPercent = Math.round((value / screenWidth) * 100);
450 		if (this._controller.isSearchResults) {
451 			this._readingPaneSashVertPos = sashWidthPercent;
452 		}
453 		else {
454 			appCtxt.set(ZmSetting.READING_PANE_SASH_VERTICAL, sashWidthPercent);
455 		}
456 	}
457 	else {
458 		var screenHeight = this.getSize().y;
459 		var sashHeightPercent = Math.round((value/screenHeight) * 100);
460 		if (this._controller.isSearchResults) {
461 			this._readingPaneSashHorizPos = sashHeightPercent;
462 		}
463 		else {
464 			appCtxt.set(ZmSetting.READING_PANE_SASH_HORIZONTAL, sashHeightPercent);
465 		}
466 	}
467 };

479 ZmDoublePaneView.prototype._controlEventListener =
480 function(ev) {
481 	//resize can be called multiple times based on the browser so wait till resizing is complete
482 	if (ev && (ev.newWidth == ev.requestedWidth) && (ev.newHeight == ev.requestedHeight))  {
483 		var readingPaneOnRight = this._controller.isReadingPaneOnRight();
484 		//reset the sash values and resize the pane based on settings
485 		if (readingPaneOnRight) {
486 			this._readingPaneSashVertPos = appCtxt.get(ZmSetting.READING_PANE_SASH_VERTICAL);
487 			delete this._vertSashX;
488 		} else {
489 			this._readingPaneSashHorizPos = appCtxt.get(ZmSetting.READING_PANE_SASH_HORIZONTAL);
490 			delete this._horizSashY;
491 		}
492 		var sz = this.getSize();
493 		this._resetSize(sz.x, sz.y, true);
494 	}
495 };
