 51 ZmBackupPage.prototype.showMe =

 66             this._restoreListView.setSelection(r[0]);
 67         }
 68     }
 69 };

106 ZmBackupPage.prototype._handleBackupAccountsButton =
107 function() {
108     this._backupButton.setEnabled(false);
109 
110     var accts = this.getAccounts()._vector.getArray();
111 
112     var settingsObj = appCtxt.getSettings();
113     var setting = settingsObj.getSetting(ZmSetting.OFFLINE_BACKUP_ACCOUNT_ID);
114 
115     var checked = [];
116     for (var i = 0; i < accts.length; i++) {
117         if (accts[i].active) {
118             checked.push(accts[i].id);
119         }
120     }
121     if(checked.length < 1) {
122         this._backupButton.setEnabled(true);
123         return;
124     }
125 
126     setting.setValue(checked);
127 
128     var soapDoc = AjxSoapDoc.create("BatchRequest", "urn:zimbra", null);
129     soapDoc.setMethodAttribute("onerror", "continue");
130     for (var k=0; k<checked.length; k++) {
131         var requestNode = soapDoc.set("AccountBackupRequest",null,null,"urn:zimbraOffline");
132         requestNode.setAttribute("id", checked[k]);
133     }
134 
135     var params = {
136         soapDoc: soapDoc,
137         asyncMode: true,
138         errorCallback: null,
139         accountName: appCtxt.accountList.mainAccount.name
140     };
141 
142     params.callback = new AjxCallback(this, this._handleBackupStarted);
143     var appController = appCtxt.getAppController();
144     appController.sendRequest(params);
145 };

184 ZmBackupPage.prototype._handleRestoreBackupButton =
185 function() {
186 
187     this._restoreButton.setEnabled(false);
188 
189     var backups = this.getBackups()._vector.getArray();
190     var sel = [];
191     for (var i = 0; i < backups.length; i++) {
192         if (backups[i].active) {
193             var bk = backups[i];
194             sel.push(backups[i]);
195             break;
196         }
197     }
198 
199     if(sel.length < 1) {
200         this._restoreButton.setEnabled(true);
201         return;
202     }
203 
204     var soapDoc = AjxSoapDoc.create("AccountRestoreRequest", "urn:zimbraOffline");
205     var method = soapDoc.getMethod();
206     method.setAttribute("id", sel[0].acct);
207     method.setAttribute("time", sel[0].timestamp);
208     var respCallback = new AjxCallback(this, this._handleAccountRestoreSent, [sel[0].acct]);
209     var params = {
210         soapDoc:soapDoc,
211         callback:respCallback,
212         asyncMode:true
213     };
214     appCtxt.getAppController().sendRequest(params);
215 
216 };

223         appCtxt.setStatusMsg(AjxMessageFormat.format(ZmMsg.offlineBackupRestored, appCtxt.accountList.getAccount(id).name));

227 ZmBackupPage.prototype.addCommand  =
228 function(batchCommand) {

231     var accts = this.getAccounts()._vector.getArray();

233     var settingsObj = appCtxt.getSettings();

266 ZmBackupPage._getAccounts =
267 function() {
268     var savedAccounts = appCtxt.get(ZmSetting.OFFLINE_BACKUP_ACCOUNT_ID);
269     savedAccounts = (savedAccounts && savedAccounts.length) ? savedAccounts.split(",") : [];
270     var accounts = new ZmPrefAccounts();
271     var visAccts = appCtxt.accountList.visibleAccounts;
272     for (var i=0; i< visAccts.length; i++) {
273         var name =  visAccts[i].getDisplayName();
274         var desc = visAccts[i].name;
275         var id = visAccts[i].id;
276 
277         loop1: for (var k=0; k < savedAccounts.length ; k++)  {
278                 var checked = (id == AjxStringUtil.trim(savedAccounts[k]));
279                 if(checked) { break loop1; }
280             }
281         accounts.addPrefAccount(new ZmPrefAccount(name, checked, desc, id));
282     }
283     return accounts;
284 };

292 ZmBackupPage.prototype.isDirty =
293 function() {
294     var allAccountss = this.getAccounts();
295     var r = false;
296     var arr = allAccountss._vector.getArray();
297     for (var i = 0; i < arr.length; i++) {
298         if (arr[i]._origStatus != arr[i].active) {
299             r = true;
300             break;
301         }
302     }
303     return r;

524 ZmBackupPage._getBackups =
525 function(evt) {
526 
527     var soapDoc = AjxSoapDoc.create("AccountBackupEnumerationRequest", "urn:zimbraOffline");
528     var result = appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:false});
529 
530     var _restoreAccts = result.AccountBackupEnumerationResponse.account;
531     var backups = new ZmPrefBackups();
532     for (var i=0; i< _restoreAccts.length; i++) {
533         var acct =  _restoreAccts[i].id;
534         var backupArray = _restoreAccts[i].backup;
535         if(!backupArray) {
536             continue;
537         }
538 
539         for(var k=0; k<backupArray.length; k++) {
540             var fileSize = backupArray[k].fileSize;
541             var timestamp = backupArray[k].time;
542             backups.addPrefBackup(new ZmPrefBackup(timestamp, false, fileSize, acct));
543         }
544 
545     }
546     return backups;
547 };

651         var fSize = item.fileSize;
652         var numFormater = AjxNumberFormat.getInstance();
653         if (fSize != null && fSize >= 0) {
654             if (fSize < 1024) { //" B";
655                 fSize = numFormater.format(fSize) + " "+ZmMsg.b;
656             }
657             else if (fSize < (1024*1024) ) { //" KB";
658                 fSize = numFormater.format(Math.round((fSize / 1024) * 10) / 10) + " "+ZmMsg.kb;
659             }
660             else { //" MB";
661                 fSize = numFormater.format(Math.round((fSize / (1024*1024)) * 10) / 10) + " "+ZmMsg.mb;
662             }
663         } else { fSize = 0+" "+ZmMsg.b; }
664         html[idx++] = fSize;
665         html[idx++] = "</div>";
666     }
667     else if (field == ZmPrefBackupListView.COL_NAME) {
668         html[idx++] = "<div id='";
669         html[idx++] = this._getCellId(item, ZmPrefBackupListView.COL_NAME);
670         html[idx++] = "'>";
671         html[idx++] = AjxDateFormat.getDateTimeInstance(AjxDateFormat.MEDIUM).format(new Date(item.timestamp));
672         html[idx++] = "</div>";
673     }
674     else if (field == ZmPrefBackupListView.COL_ACTION) {
675         html[idx++] = "<div id='";
676         html[idx++] = this._getCellId(item, ZmPrefBackupListView.COL_ACTION);
677         html[idx++] = "'>";
678         html[idx++] = appCtxt.accountList.getAccount(item.acct).name;
679         html[idx++] = "</div>";
680     }
681     return idx;
682 };

684 ZmPrefBackupListView._activeStateChange =
685 function(ev) {
686     var target = DwtUiEvent.getTarget(ev);
687     var bkId = target.getAttribute("_bkId");
688     var bk = AjxCore.objectWithId(bkId);
689     var name = target.getAttribute("_name");
690     var bkups = bk.parent.getBackups();
691     var z = bkups.getPrefBackupByName(name);
692     var bkarray = bkups._vector.getArray();
693     // Make checkbox behave like radio button
694     for (var k=0; k<bkarray.length; k++) {
695         if(bkarray[k].active && (bkarray[k].timestamp != z.timestamp)) {
696             var y = document.getElementById([bkarray[k].timestamp, "_", bkarray[k].acct, "_backupCheckbox"].join(""));
697             y.checked = false;
698             bkarray[k].active = false;
699         }
700     }
701     if (z) {
702         z.active = !z.active;
703     }
704 };
