 11 /**
 12  * This mixin contains the event logic for the tinymce.Editor class.
 13  *
 14  * @mixin tinymce.EditorObservable
 15  * @extends tinymce.util.Observable
 16  */
 17 define("tinymce/EditorObservable", [
 18 	"tinymce/util/Observable",
 19 	"tinymce/dom/DOMUtils",
 20 	"tinymce/util/Tools"
 21 ], function(Observable, DOMUtils, Tools) {
 22 	var DOM = DOMUtils.DOM;
 23 
 24 	function getEventTarget(editor, eventName) {
 25 		if (eventName == 'selectionchange') {
 26 			return editor.getDoc();
 27 		}
 28 
 29 		// Need to bind mousedown/mouseup etc to document not body in iframe mode
 30 		// Since the user might click on the HTML element not the BODY
 31 		if (!editor.inline && /^mouse|click|contextmenu|drop/.test(eventName)) {
 32 			return editor.getDoc();
 33 		}
 34 
 35 		return editor.getBody();
 36 	}
 37 
 38 	function bindEventDelegate(editor, name) {
 39 		var eventRootSelector = editor.settings.event_root, editorManager = editor.editorManager;
 40 		var eventRootElm = editorManager.eventRootElm || getEventTarget(editor, name);
 41 
 42 		if (eventRootSelector) {
 43 			if (!editorManager.rootEvents) {
 44 				editorManager.rootEvents = {};
 45 
 46 				editorManager.on('RemoveEditor', function() {
 47 					if (!editorManager.activeEditor) {
 48 						DOM.unbind(eventRootElm);
 49 						delete editorManager.rootEvents;
 50 					}
 51 				});
 52 			}
 53 
 54 			if (editorManager.rootEvents[name]) {
 55 				return;
 56 			}
 57 
 58 			if (eventRootElm == editor.getBody()) {
 59 				eventRootElm = DOM.select(eventRootSelector)[0];
 60 				editorManager.eventRootElm = eventRootElm;
 61 			}
 62 
 63 			editorManager.rootEvents[name] = true;
 64 
 65 			DOM.bind(eventRootElm, name, function(e) {
 66 				var target = e.target, editors = editorManager.editors, i = editors.length;
 67 
 68 				while (i--) {
 69 					var body = editors[i].getBody();
 70 
 71 					if (body === target || DOM.isChildOf(target, body)) {
 72 						if (!editors[i].hidden) {
 73 							editors[i].fire(name, e);
 74 						}
 75 					}
 76 				}
 77 			});
 78 		} else {
 79 			editor.dom.bind(eventRootElm, name, function(e) {
 80 				if (!editor.hidden) {
 81 					editor.fire(name, e);
 82 				}
 83 			});
 84 		}
 85 	}
 86 
 87 	var EditorObservable = {
 88 		bindPendingEventDelegates: function() {
 89 			var self = this;
 90 
 91 			Tools.each(self._pendingNativeEvents, function(name) {
 92 				bindEventDelegate(self, name);
 93 			});
 94 		},
 95 
 96 		toggleNativeEvent: function(name, state) {
 97 			var self = this;
 98 
 99 			if (self.settings.readonly) {
100 				return;
101 			}
102 
103 			// Never bind focus/blur since the FocusManager fakes those
104 			if (name == "focus" || name == "blur") {
105 				return;
106 			}
107 
108 			if (state) {
109 				if (self.initialized) {
110 					bindEventDelegate(self, name);
111 				} else {
112 					if (!self._pendingNativeEvents) {
113 						self._pendingNativeEvents = [name];
114 					} else {
115 						self._pendingNativeEvents.push(name);
116 					}
117 				}
118 			} else if (self.initialized) {
119 				self.dom.unbind(getEventTarget(self, name), name);
120 			}
121 		}
122 	};
123 
124 	EditorObservable = Tools.extend({}, Observable, EditorObservable);
125 
126 	return EditorObservable;
127 });
