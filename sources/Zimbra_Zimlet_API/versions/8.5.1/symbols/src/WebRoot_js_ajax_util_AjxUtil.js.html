101 /**
102  * Formats a size (in bytes) to the largest whole unit. For example,
103  * AjxUtil.formatSize(302132199) returns "288 MB".
104  *
105  * @param size      The size (in bytes) to be formatted.
106  * @param round     True to round to nearest integer. Default is true.
107  * @param fractions Number of fractional digits to display, if not rounding.
108  *                  Trailing zeros after the decimal point are trimmed.
109  */
110 AjxUtil.formatSize = 
111 function(size, round, fractions) {
112 	if (round == null) round = true;
113 	if (fractions == null) fractions = 20; // max allowed for toFixed is 20
114 
115 	var formattedUnits = AjxMsg.sizeBytes;
116 	var units = AjxMsg.SIZE_BYTES;
117 	if (size >= 1073741824) {
118 		formattedUnits = AjxMsg.sizeGigaBytes;
119 		units = AjxUtil.SIZE_GIGABYTES;
120 	}
121 	else if (size >= 1048576) {
122 		formattedUnits = AjxMsg.sizeMegaBytes;
123 		units = AjxUtil.SIZE_MEGABYTES;
124 	}
125 	else if (size > 1023) {
126 		formattedUnits = AjxMsg.sizeKiloBytes;
127 		units = AjxUtil.SIZE_KILOBYTES;
128 	}
129 
130 
131 	var formattedSize = AjxUtil.formatSizeForUnits(size, units, round, fractions);
132 	return AjxMessageFormat.format(AjxMsg.formatSizeAndUnits, [formattedSize, formattedUnits]);
133 };

135 /**
136  * Formats a size (in bytes) to a specific unit. Since the unit size is
137  * known, the unit is not shown in the returned string. For example,
138  * AjxUtil.formatSizeForUnit(302132199, AjxUtil.SIZE_MEGABYTES, false, 2) 
139  * returns "288.13".
140  *
141  * @param size      The size (in bytes) to be formatted.
142  * @param units     The unit of measure.
143  * @param round     True to round to nearest integer. Default is true.
144  * @param fractions Number of fractional digits to display, if not rounding.
145  *                  Trailing zeros after the decimal point are trimmed.
146  */
147 AjxUtil.formatSizeForUnits = 
148 function(size, units, round, fractions) {
149 	if (units == null) units = AjxUtil.SIZE_BYTES;
150 	if (round == null) round = true;
151 	if (fractions == null) fractions = 20; // max allowed for toFixed is 20
152 
153 	switch (units) {
154 		case AjxUtil.SIZE_GIGABYTES: { size /= 1073741824; break; }
155 		case AjxUtil.SIZE_MEGABYTES: { size /= 1048576; break; }
156 		case AjxUtil.SIZE_KILOBYTES: { size /= 1024; break; }
157 	}
158 	var dot = I18nMsg.numberSeparatorDecimal !='' ? I18nMsg.numberSeparatorDecimal : '.';
159 	var pattern = I18nMsg.formatNumber.replace(/\..*$/, ""); // Strip off decimal, we'll be adding one anyway
160 	pattern = pattern.replace(/,/, "");       // Remove the ,
161 	if (!round && fractions) {
162 		pattern = pattern += dot;
163 		for (var i = 0; i < fractions; i++) {
164 			pattern += "#";
165 		}
166 	}
167 	return AjxNumberFormat.format(pattern, size);
168 };

170 /**
171  * Performs the opposite of AjxUtil.formatSize in that this function takes a 
172  * formatted size.
173  *
174  * @param units Unit constant: "GB", "MB", "KB", "B". Must be specified 
175  *              unless the formatted size ends with the size marker, in
176  *				which case the size marker in the formattedSize param
177  *				overrides this parameter.
178  */
179 AjxUtil.parseSize = 
180 function(formattedSize, units) {
181 	// NOTE: Take advantage of fact that parseFloat ignores bad chars
182 	//       after numbers
183 	if (typeof formattedSize != _STRING_) {
184 		formattedSize = formattedSize.toString() ;
185 	}
186 	var size = parseFloat(formattedSize.replace(/^\s*/,""));
187 
188 	var marker = /[GMK]?B$/i;
189 	var result = marker.exec(formattedSize);
190 	if (result) {
191 		//alert("units: "+units+", result[0]: '"+result[0]+"'");
192 		units = result[0].toUpperCase();
193 	}
194 	
195 	switch (units) {
196 		case AjxUtil.SIZE_GIGABYTES: size *= 1073741824; break;
197 		case AjxUtil.SIZE_MEGABYTES: size *= 1048576; break; 
198 		case AjxUtil.SIZE_KILOBYTES: size *= 1024; break;
199 	}
200 	
201 	//alert("AjxUtil#parseSize: formattedSize="+formattedSize+", size="+size);
202 	return size;
203 };

215 AjxUtil.onassert = 
216 function(aMessage) {
217 	// Create an exception object and set the message
218 	var myException = new Object();
219 	myException.message = aMessage;
220 	
221 	// Compile a stack trace
222 	var myStack = new Array();
223 	if (AjxEnv.isIE5_5up) {
224 		// On IE, the caller chain is on the arguments stack
225 		var myTrace = arguments.callee.caller;
226 		var i = 0; // stop at 20 since there might be somehow an infinite loop here. Maybe in case of a recursion. 
227 		while (myTrace && i++ < 20) {
228 		    myStack[myStack.length] = myTrace;
229 	    	myTrace = myTrace.caller;
230 		}
231 	} else {
232 		try {
233 			var myTrace = arguments.callee.caller;
234 			while (myTrace) {
235 				myStack[myStack.length] = myTrace;
236 				if (myStack.length > 2) break;
237 				myTrace = myTrace.caller;
238 		    }
239 		} catch (e) {
240 		}
241 	}
242 	myException.stack = myStack;
243 	
244 	// Alert with the message and a description of the stack
245 	var stackString = '';
246 	var MAX_LEN = 170;
247 	for (var i = 1; i < myStack.length; i++) {
248 		if (i > 1) stackString += '\n';
249 		if (i < 11) {
250 			var fs = myStack[i].toString();
251 			if (fs.length > MAX_LEN) {
252 				fs = fs.substr(0,MAX_LEN) + '...';
253 				fs = fs.replace(/\n/g, '');
254 			}
255 			stackString += i + ': ' + fs;
256 		} else {
257 			stackString += '(' + (myStack.length - 11) + ' frames follow)';
258 			break;
259 		}
260 	}
261 	alert('assertion:\n\n' + aMessage + '\n\n---- Call Stack ---\n' + stackString);
262 	
263 	// Now throw the exception
264 	throw myException;
265 };

272 AjxUtil.getInnerText = 
273 function(node) {
274  	if (AjxEnv.isIE)
275  		return node.innerText;
276 
277 	function f(n) {
278 		if (n) {
279 			if (n.nodeType == 3 /* TEXT_NODE */)
280 				return n.data;
281 			if (n.nodeType == 1 /* ELEMENT_NODE */) {
282 				if (/^br$/i.test(n.tagName))
283 					return "\r\n";
284 				var str = "";
285 				for (var i = n.firstChild; i; i = i.nextSibling)
286 					str += f(i);
287 				return str;
288 			}
289 		}
290 		return "";
291 	};
292 	return f(node);
293 };

295 /**
296  * This method returns a proxy for the specified object. This is useful when
297  * you want to modify properties of an object and want to keep those values
298  * separate from the values in the original object. You can then iterate
299  * over the proxy's properties and use the <code>hasOwnProperty</code>
300  * method to determine if the property is a new value in the proxy.
301  * <p>
302  * <strong>Note:</strong>
303  * A reference to the original object is stored in the proxy as the "_object_" 
304  * property.
305  *
306  * @param object [object] The object to proxy.
307  * @param level  [number] The number of property levels deep to proxy.
308  *						  Defaults to zero.
309  */
310 AjxUtil.createProxy = 
311 function(object, level) {
312 	var proxy;
313 	var proxyCtor = function(){}; // bug #6517 (Safari doesnt like 'new Function')
314 	proxyCtor.prototype = object;
315 	if (object instanceof Array) {
316 		proxy  = new Array();
317 		var cnt  = object.length;
318 		for(var ix = 0; ix < cnt; ix++) {
319 			proxy[ix] = object[ix];
320 		}
321 	} else {
322 		proxy = new proxyCtor;
323 	}
324 	
325 	if (level) {
326 		for (var prop in object) {
327 			if (typeof object[prop] == "object" && object[prop] !== null)
328 				proxy[prop] = AjxUtil.createProxy(object[prop], level - 1);
329 		}
330 	}	
331 	
332 	proxy._object_ = object;
333 	return proxy;
334 };
335 
336 AjxUtil.unProxy =
337 function(proxy) {
338 	var object = proxy && proxy._object_;
339 	if (object) {
340 		for (var prop in proxy) {
341 			if (proxy.hasOwnProperty(prop) && prop!="_object_") {
342 				object[prop] = proxy[prop];
343 			}
344 		}
345 		return object;
346 	}
347 	return null;
348 }

439 /**
440  * Generate another hash mapping property values to their names. Each value
441  * should be unique; otherwise the results are undefined.
442  *
443  * @param obj                   An object, treated as a hash.
444  * @param func [function]       An optional function for filtering properties.
445  */
446 AjxUtil.valueHash = function(obj, acceptFunc) {
447     // don't rely on the value in the object itself
448     var hasown = Object.prototype.hasOwnProperty.bind(obj);
449 
450     var r = {};
451     for (var k in obj) {
452         var v = obj[k];
453 
454         if (!hasown(k) || (acceptFunc && !acceptFunc(k, obj)))
455             continue;
456         r[v] = k;
457     }
458     return r;
459 };

462 /**
463  * Call a function with the the items in the given object, which special logic
464  * for handling of arrays.
465  *
466  * @param obj                   Array or other object
467  * @param func [function]       Called with index or key and value.
468  */
469 AjxUtil.foreach = function(obj, func) {
470     if (!func) return;
471 
472     if (AjxUtil.isArrayLike(obj)) {
473         var array = obj;
474 
475         for (var i = 0; i < array.length; i++) {
476             func(array[i], i);
477         }
478     } else {
479         // don't rely on the value in the object itself
480         hasown = Object.prototype.hasOwnProperty.bind(obj);
481 
482         for (var k in obj) {
483             if (hasown(k))
484                 func(obj[k], k)
485         }
486     }
487 };

505 /**
506  * Remove duplicates from the given array,
507  * <strong>in-place</strong>. Stable with regards to ordering.
508  *
509  * Please note that this method is O(n^2) if Array is backed by an
510  * array/vector data structure.
511  *
512  * @param array [array]     array to process
513  * @param keyfn [function]  used to extract a comparison key from each
514  *                          list element, default is to compare
515  *                          elements directly. if the comparison key
516  *                          is 'undefined', the element is always
517  *                          retained
518  */
519 AjxUtil.dedup = function(array, keyfn) {
520 	if (!keyfn)
521 		keyfn = function(v) { return v; };
522 
523 	var seen = {};
524 
525 	for (var i = 0; i < array.length; i++) {
526 		var key = keyfn(array[i]);
527 
528 		if (key !== undefined && seen[key]) {
529 			array.splice(i, 1);
530 			i -= 1;
531 		}
532 
533 		seen[key] = true;
534 	}
535 };
