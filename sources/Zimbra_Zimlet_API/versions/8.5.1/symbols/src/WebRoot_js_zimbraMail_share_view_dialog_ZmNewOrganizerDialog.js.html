 26 /**
 27  * Creates a new organizer dialog.
 28  * @class
 29  * This class represents a new organizer dialog.
 30  * 
 31  * @param	{DwtControl}	parent		the parent
 32  * @param	{String}	className		the class name
 33  * @param	{String}	title		the title
 34  * @param	{constant}	type		the organizer type
 35  * 
 36  * @extends		ZmDialog
 37  */
 38 ZmNewOrganizerDialog = function(parent, className, title, type, extraButtons) {
 39 	if (arguments.length == 0) return;
 40 
 41 	this._organizerType = type;
 42 	ZmDialog.call(this, {parent:parent, className:className, title:title, id:"CreateNewFolderDialog", extraButtons: extraButtons});
 43 	this._setupControls();
 44 };

 60 /**
 61  * Pops-up the dialog.
 62  * 
 63  * @param {ZmOrganizer|hash}	params      popup parameters
 64  * @param	{ZmAccount}	account		the account
 65  */
 66 ZmNewOrganizerDialog.prototype.popup =
 67 function(params, account) {
 68 
 69 	params = params || {};
 70     var folder = params instanceof ZmOrganizer ? params : (params && params.organizer);
 71 
 72 	var parentLabelCell = document.getElementById(this._htmlElId + '_parentLabel');
 73 	var parentValueCell = document.getElementById(this._htmlElId + '_parentValue');
 74 	this._parentFolder = null;
 75 
 76 	// if the user has already implicitly selected a parent folder, don't show overview
 77 	if (folder && folder.id != ZmOrganizer.ID_ROOT) {
 78 		this._parentFolder = folder;
 79 		this._makeOverviewVisible();    // hide all overviews
 80 		if (parentLabelCell) {
 81 			parentLabelCell.colSpan = 1;
 82 			parentLabelCell.innerHTML = ZmMsg.parentFolderLabel;
 83 			parentValueCell.innerHTML = folder.getName();
 84 		}
 85 	}
 86 	else {
 87 		if (this._folderTreeCellId) {
 88 			if (parentLabelCell) {
 89 				parentLabelCell.innerHTML = this._folderLocationLabel;
 90 				parentLabelCell.colSpan = 2;
 91 				parentValueCell.innerHTML = '';
 92 			}
 93 			var overviewParams = {
 94 				appName:		params.appName,
 95 				overviewId:		this.toString() + (params.appName || ""),
 96 				treeIds:		this._treeIds,
 97 				omit:			this._omit,
 98 				fieldId:		this._folderTreeCellId,
 99 				overviewTrees:	[this._organizerType],
100 	            treeStyle:      this._treeStyle
101 			};
102 			this._setOverview(overviewParams);
103 
104 			if (this._folderTreeView) {
105 				// bug #18533 - always make sure header item is visible in "New" dialog
106 				this._folderTreeView.getHeaderItem().setVisible(true, true);
107 
108 				if (!folder || this._omit[folder.nId] || folder.nId == ZmOrganizer.ID_ROOT) {
109 					folder = appCtxt.getFolderTree().root; //default to root if no folder passed, the folder is omitted from the overview. (I don't get the last option, but it was there so I keep it - it's already root)
110 				}
111 				var ti = this._folderTreeView.getTreeItemById(folder.id);
112 				if (ti) {
113 					this._folderTreeView.setSelection(ti);
114 				}
115 				if (folder.nId == ZmOrganizer.ID_ROOT) {
116 					var sid = ZmOrganizer.getSystemId(folder.id);
117 					var ti = this._folderTreeView.getTreeItemById(sid);
118 					if (ti) {
119 						ti.setExpanded(true);
120 					}
121 				}
122 			}
123 		}
124 	}
125 
126     if (this._colorSelect) {
127         var defaultColorCode = ZmOrganizer.DEFAULT_COLOR[this._organizerType],
128             defaultColor = ZmOrganizer.COLOR_VALUES[defaultColorCode],
129             colorMenu = this._colorSelect.getMenu(),
130             moreColorMenu;
131         if(colorMenu) {
132             moreColorMenu = (colorMenu.toString() == "ZmMoreColorMenu") ? colorMenu : colorMenu._getMoreColorMenu();
133             if(moreColorMenu) moreColorMenu.setDefaultColor(defaultColor);
134         }
135 
136         var icon = null;
137         var orgType = this._organizerType;
138         var orgClass = ZmOrganizer.ORG_CLASS[orgType];
139         if (orgClass) {
140 			//to fix bug 55320 - got rid of the calling getIcon on the prototype hack - that caused isRemote to set _isRemote on the prototype thus causing every object to have it by default set.
141             //bug 55491: pass tmp. organizer id to make sure this._isRemote is not true by default.
142 			var sample = new window[orgClass]({id:Dwt.getNextId()}); //get a sample object just for the icon
143 			icon = sample.getIcon();
144         }
145 
146         this._colorSelect.setImage(icon);
147         this._colorSelect.setValue(ZmOrganizer.DEFAULT_COLOR[orgType]);
148     }
149 
150 	var ovContainer = appCtxt.multiAccounts && this._opc.getOverviewContainer(this.toString());
151 	if (ovContainer) {
152 		if (!folder || (folder && folder.nId == ZmOrganizer.ID_ROOT)) {
153 			var acct = account || appCtxt.getActiveAccount();
154 			ovContainer.setSelection(ovContainer.getHeaderItem(acct));
155 		} else {
156 			var overviewId = appCtxt.getOverviewId(this.toString(), account);
157 			var overview = ovContainer.getOverview(overviewId);
158 			var treeView = overview && overview.getTreeView(this._organizerType);
159 			if (treeView) {
160 				ovContainer.deselectAll();
161 				var ti = treeView.getTreeItemById(folder.id);
162 				treeView.setSelection(ti);
163 			}
164 		}
165 
166 		ovContainer.expandAccountOnly(account);
167 	}
168 
169 	ZmDialog.prototype.popup.call(this);
170 };

172 /**
173  * Resets the dialog.
174  * 
175  * @param	{ZmAccount}	account		the account
176  */
177 ZmNewOrganizerDialog.prototype.reset =
178 function(account) {
179 	ZmDialog.prototype.reset.apply(this, arguments);
180 
181 	if (this._remoteCheckboxField) {
182 		this._remoteCheckboxField.checked = false;
183 		var urlRow = document.getElementById(this._remoteCheckboxFieldId+"URLrow");
184 		if (urlRow) urlRow.style.display = "none";
185 	}
186 
187 	if (this._urlField) {
188 		this._urlField.value = "";
189 	}
190 
191 	if (appCtxt.multiAccounts) {
192 		this._account = account;
193 	} else {
194 		this._account = null;
195 	}
196 };

210 ZmNewOrganizerDialog.prototype._contentHtml = 
211 function() {

353 /** 
354  * Checks the input for validity and returns the following array of values:
355  * <ul>
356  * <li> parentFolder
357  * <li> name
358  * <li> color
359  * <li> URL
360  * </ul>
361  */
362 ZmNewOrganizerDialog.prototype._getFolderData =
363 function() {
364 	// make sure a parent was selected
365 	var ov = this._getOverviewOrOverviewContainer();
366 
367 	var parentFolder = this._parentFolder || (ov && ov.getSelected()) || appCtxt.getFolderTree(this._account).root;
368 
369 	if (this._isGlobalSearch) {
370 		//special case for global search (only possible if this is ZmNewSearchDialog
371 		parentFolder = appCtxt.getById(ZmOrganizer.ID_ROOT);
372 	}
373 
374 	// check name for presence and validity
375 	var name = AjxStringUtil.trim(this._nameField.value);
376 	var msg = ZmFolder.checkName(name, parentFolder);
377 
378 	// make sure parent doesn't already have a child by this name
379 	if (!msg && parentFolder.hasChild(name)) {
380         var folderType = appCtxt.getFolderTree(appCtxt.getActiveAccount()).getFolderTypeByName(name);
381 		msg = AjxMessageFormat.format(this._folderNameAlreadyExistsMsg, [name,ZmMsg[folderType.toLowerCase()]]);
382 	}
383 
384 	var color = null;
385 	if (!msg && this._colorSelectId) {
386 		color = this._colorSelect.getValue();
387 	}
388 
389 	var url = null;
390 	if (!msg && this._remoteCheckboxField) {
391 		url = this._remoteCheckboxField.checked ? this._urlField.value : null;
392 		if (url || url != null) {
393 			msg = ZmOrganizer.checkUrl(url);
394 		}
395 	}
396 
397 	if (!msg && parentFolder.disallowSubFolder) {
398 		msg = AjxMessageFormat.format(ZmMsg.errorSubFolderNotAllowed, parentFolder.name);
399 	}
400 
401     if (msg) {
402         return this._showError(msg);
403     }
404 
405 	var account = appCtxt.multiAccounts ? parentFolder.getAccount() : null;
406 	var params = {l:parentFolder.id, name:name, color:color, url:url, account:account};
407     if (String(color).match(/^#/)) {
408         params.rgb = color;
409         delete params.color;
410     }
411     return params;
412 };

447 ZmNewOrganizerDialog.prototype._handleCheckbox =
448 function(event) {
449 	event = event || window.event;
450 	var target = DwtUiEvent.getTarget(event);
451 	var urlRow = document.getElementById(target.id+"URLrow");
452 	var urlField= document.getElementById(target.id+"URLfield");	
453 	urlRow.style.display = target.checked ? (AjxEnv.isIE ? "block" : "table-row") : "none";
454 	if (target.checked) {
455 		urlField.focus();
456 	}
457 };
