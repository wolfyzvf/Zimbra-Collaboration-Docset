208 		/**
209 		 * Registers a specific format by name.
210 		 *
211 		 * @method register
212 		 * @param {Object/String} name Name of the format for example "bold".
213 		 * @param {Object/Array} format Optional format object or array of format variants
214 		 * can only be omitted if the first arg is an object.
215 		 */
216 		function register(name, format) {
217 			if (name) {
218 				if (typeof(name) !== 'string') {
219 					each(name, function(format, name) {
220 						register(name, format);
221 					});
222 				} else {
223 					// Force format into array and add it to internal collection
224 					format = format.length ? format : [format];
225 
226 					each(format, function(format) {
227 						// Set deep to false by default on selector formats this to avoid removing
228 						// alignment on images inside paragraphs when alignment is changed on paragraphs
229 						if (format.deep === undef) {
230 							format.deep = !format.selector;
231 						}
232 
233 						// Default to true
234 						if (format.split === undef) {
235 							format.split = !format.selector || format.inline;
236 						}
237 
238 						// Default to true
239 						if (format.remove === undef && format.selector && !format.inline) {
240 							format.remove = 'none';
241 						}
242 
243 						// Mark format as a mixed format inline + block level
244 						if (format.selector && format.inline) {
245 							format.mixed = true;
246 							format.block_expand = true;
247 						}
248 
249 						// Split classes if needed
250 						if (typeof(format.classes) === 'string') {
251 							format.classes = format.classes.split(/\s+/);
252 						}
253 					});
254 
255 					formats[name] = format;
256 				}
257 			}
258 		}

294 			function setElementFormat(elm, fmt) {
295 				fmt = fmt || format;
296 
297 				if (elm) {
298 					if (fmt.onformat) {
299 						fmt.onformat(elm, fmt, vars, node);
300 					}
301 
302 					each(fmt.styles, function(value, name) {
303 						dom.setStyle(elm, name, replaceVars(value, vars));
304 					});
305 
306 					// Needed for the WebKit span spam bug
307 					// TODO: Remove this once WebKit/Blink fixes this
308 					if (fmt.styles) {
309 						var styleVal = dom.getAttrib(elm, 'style');
310 
311 						if (styleVal) {
312 							elm.setAttribute('data-mce-style', styleVal);
313 						}
314 					}
315 
316 					each(fmt.attributes, function(value, name) {
317 						dom.setAttrib(elm, name, replaceVars(value, vars));
318 					});
319 
320 					each(fmt.classes, function(value) {
321 						value = replaceVars(value, vars);
322 
323 						if (!dom.hasClass(elm, value)) {
324 							dom.addClass(elm, value);
325 						}
326 					});
327 				}
328 			}

330 			function adjustSelectionToVisibleSelection() {
331 				function findSelectionEnd(start, end) {
332 					var walker = new TreeWalker(end);
333 					for (node = walker.current(); node; node = walker.prev()) {
334 						if (node.childNodes.length > 1 || node == start || node.tagName == 'BR') {
335 							return node;
336 						}
337 					}
338 				}
339 
340 				// Adjust selection so that a end container with a end offset of zero is not included in the selection
341 				// as this isn't visible to the user.
342 				var rng = ed.selection.getRng();
343 				var start = rng.startContainer;
344 				var end = rng.endContainer;
345 
346 				if (start != end && rng.endOffset === 0) {
347 					var newEnd = findSelectionEnd(start, end);
348 					var endOffset = newEnd.nodeType == 3 ? newEnd.length : newEnd.childNodes.length;
349 
350 					rng.setEnd(newEnd, endOffset);
351 				}
352 
353 				return rng;
354 			}

356 			function applyStyleToList(node, bookmark, wrapElm, newWrappers, process){
357 				var nodes = [], listIndex = -1, list, startIndex = -1, endIndex = -1, currentWrapElm;
358 
359 				// find the index of the first child list.
360 				each(node.childNodes, function(n, index) {
361 					if (n.nodeName === "UL" || n.nodeName === "OL") {
362 						listIndex = index;
363 						list = n;
364 						return false;
365 					}
366 				});
367 
368 				// get the index of the bookmarks
369 				each(node.childNodes, function(n, index) {
370 					if (n.nodeName === "SPAN" && dom.getAttrib(n, "data-mce-type") == "bookmark") {
371 						if (n.id == bookmark.id + "_start") {
372 							startIndex = index;
373 						} else if (n.id == bookmark.id + "_end") {
374 							endIndex = index;
375 						}
376 					}
377 				});
378 
379 				// if the selection spans across an embedded list, or there isn't an embedded list - handle processing normally
380 				if (listIndex <= 0 || (startIndex < listIndex && endIndex > listIndex)) {
381 					each(grep(node.childNodes), process);
382 					return 0;
383 				} else {
384 					currentWrapElm = dom.clone(wrapElm, FALSE);
385 
386 					// create a list of the nodes on the same side of the list as the selection
387 					each(grep(node.childNodes), function(n, index) {
388 						if ((startIndex < listIndex && index < listIndex) || (startIndex > listIndex && index > listIndex)) {
389 							nodes.push(n);
390 							n.parentNode.removeChild(n);
391 						}
392 					});
393 
394 					// insert the wrapping element either before or after the list.
395 					if (startIndex < listIndex) {
396 						node.insertBefore(currentWrapElm, list);
397 					} else if (startIndex > listIndex) {
398 						node.insertBefore(currentWrapElm, list.nextSibling);
399 					}
400 
401 					// add the new nodes to the list.
402 					newWrappers.push(currentWrapElm);
403 
404 					each(nodes, function(node) {
405 						currentWrapElm.appendChild(node);
406 					});
407 
408 					return currentWrapElm;
409 				}
410 			}

412 			function applyRngStyle(rng, bookmark, node_specific) {
413 				var newWrappers = [], wrapName, wrapElm, contentEditable = true;
414 
415 				// Setup wrapper element
416 				wrapName = format.inline || format.block;
417 				wrapElm = dom.create(wrapName);
418 				setElementFormat(wrapElm);
419 
420 				rangeUtils.walk(rng, function(nodes) {
421 					var currentWrapElm;
422 
423 					/**
424 					 * Process a list of nodes wrap them.
425 					 */
426 					function process(node) {
427 						var nodeName, parentName, found, hasContentEditableState, lastContentEditable;
428 
429 						lastContentEditable = contentEditable;
430 						nodeName = node.nodeName.toLowerCase();
431 						parentName = node.parentNode.nodeName.toLowerCase();
432 
433 						// Node has a contentEditable value
434 						if (node.nodeType === 1 && getContentEditable(node)) {
435 							lastContentEditable = contentEditable;
436 							contentEditable = getContentEditable(node) === "true";
437 							hasContentEditableState = true; // We don't want to wrap the container only it's children
438 						}
439 
440 						// Stop wrapping on br elements
441 						if (isEq(nodeName, 'br')) {
442 							currentWrapElm = 0;
443 
444 							// Remove any br elements when we wrap things
445 							if (format.block) {
446 								dom.remove(node);
447 							}
448 
449 							return;
450 						}
451 
452 						// If node is wrapper type
453 						if (format.wrapper && matchNode(node, name, vars)) {
454 							currentWrapElm = 0;
455 							return;
456 						}
457 
458 						// Can we rename the block
459 						// TODO: Break this if up, too complex
460 						if (contentEditable && !hasContentEditableState && format.block &&
461 							!format.wrapper && isTextBlock(nodeName) && isValid(parentName, wrapName)) {
462 							node = dom.rename(node, wrapName);
463 							setElementFormat(node);
464 							newWrappers.push(node);
465 							currentWrapElm = 0;
466 							return;
467 						}
468 
469 						// Handle selector patterns
470 						if (format.selector) {
471 							// Look for matching formats
472 							each(formatList, function(format) {
473 								// Check collapsed state if it exists
474 								if ('collapsed' in format && format.collapsed !== isCollapsed) {
475 									return;
476 								}
477 
478 								if (dom.is(node, format.selector) && !isCaretNode(node)) {
479 									setElementFormat(node, format);
480 									found = true;
481 								}
482 							});
483 
484 							// Continue processing if a selector match wasn't found and a inline element is defined
485 							if (!format.inline || found) {
486 								currentWrapElm = 0;
487 								return;
488 							}
489 						}
490 
491 						// Is it valid to wrap this item
492 						// TODO: Break this if up, too complex
493 						if (contentEditable && !hasContentEditableState && isValid(wrapName, nodeName) && isValid(parentName, wrapName) &&
494 								!(!node_specific && node.nodeType === 3 &&
495 								node.nodeValue.length === 1 &&
496 								node.nodeValue.charCodeAt(0) === 65279) &&
497 								!isCaretNode(node) &&
498 								(!format.inline || !isBlock(node))) {
499 							// Start wrapping
500 							if (!currentWrapElm) {
501 								// Wrap the node
502 								currentWrapElm = dom.clone(wrapElm, FALSE);
503 								node.parentNode.insertBefore(currentWrapElm, node);
504 								newWrappers.push(currentWrapElm);
505 							}
506 
507 							currentWrapElm.appendChild(node);
508 						} else if (nodeName == 'li' && bookmark) {
509 							// Start wrapping - if we are in a list node and have a bookmark, then
510 							// we will always begin by wrapping in a new element.
511 							currentWrapElm = applyStyleToList(node, bookmark, wrapElm, newWrappers, process);
512 						} else {
513 							// Start a new wrapper for possible children
514 							currentWrapElm = 0;
515 
516 							each(grep(node.childNodes), process);
517 
518 							if (hasContentEditableState) {
519 								contentEditable = lastContentEditable; // Restore last contentEditable state from stack
520 							}
521 
522 							// End the last wrapper
523 							currentWrapElm = 0;
524 						}
525 					}
526 
527 					// Process siblings from range
528 					each(nodes, process);
529 				});
530 
531 				// Wrap links inside as well, for example color inside a link when the wrapper is around the link
532 				if (format.wrap_links === false) {
533 					each(newWrappers, function(node) {
534 						function process(node) {
535 							var i, currentWrapElm, children;
536 
537 							if (node.nodeName === 'A') {
538 								currentWrapElm = dom.clone(wrapElm, FALSE);
539 								newWrappers.push(currentWrapElm);
540 
541 								children = grep(node.childNodes);
542 								for (i = 0; i < children.length; i++) {
543 									currentWrapElm.appendChild(children[i]);
544 								}
545 
546 								node.appendChild(currentWrapElm);
547 							}
548 
549 							each(grep(node.childNodes), process);
550 						}
551 
552 						process(node);
553 					});
554 				}
555 
556 				// Cleanup
557 				each(newWrappers, function(node) {
558 					var childCount;
559 
560 					function getChildCount(node) {
561 						var count = 0;
562 
563 						each(node.childNodes, function(node) {
564 							if (!isWhiteSpaceNode(node) && !isBookmarkNode(node)) {
565 								count++;
566 							}
567 						});
568 
569 						return count;
570 					}

572 					function mergeStyles(node) {
573 						var child, clone;
574 
575 						each(node.childNodes, function(node) {
576 							if (node.nodeType == 1 && !isBookmarkNode(node) && !isCaretNode(node)) {
577 								child = node;
578 								return FALSE; // break loop
579 							}
580 						});
581 
582 						// If child was found and of the same type as the current node
583 						if (child && !isBookmarkNode(child) && matchName(child, format)) {
584 							clone = dom.clone(child, FALSE);
585 							setElementFormat(clone);
586 
587 							dom.replace(clone, node, TRUE);
588 							dom.remove(child, 1);
589 						}
590 
591 						return clone || node;
592 					}
593 
594 					childCount = getChildCount(node);
595 
596 					// Remove empty nodes but only if there is multiple wrappers and they are not block
597 					// elements so never remove single <h1></h1> since that would remove the
598 					// currrent empty block element where the caret is at
599 					if ((newWrappers.length > 1 || !isBlock(node)) && childCount === 0) {
600 						dom.remove(node, 1);
601 						return;
602 					}
603 
604 					if (format.inline || format.wrapper) {
605 						// Merges the current node with it's children of similar type to reduce the number of elements
606 						if (!format.exact && childCount === 1) {
607 							node = mergeStyles(node);
608 						}
609 
610 						// Remove/merge children
611 						each(formatList, function(format) {
612 							// Merge all children of similar type will move styles from child to parent
613 							// this: <span style="color:red"><b><span style="color:red; font-size:10px">text</span></b></span>
614 							// will become: <span style="color:red"><b><span style="font-size:10px">text</span></b></span>
615 							each(dom.select(format.inline, node), function(child) {
616 								var parent;
617 
618 								if (isBookmarkNode(child)) {

970 		 * @param {Boolean} similar Match format that has similar properties.
971 		 * @return {Object} Returns the format object it matches or undefined if it doesn't match.

976 			function matchItems(node, format, item_name) {
977 				var key, value, items = format[item_name], i;
978 
979 				// Custom match
980 				if (format.onmatch) {
981 					return format.onmatch(node, format, item_name);
982 				}
983 
984 				// Check all items
985 				if (items) {
986 					// Non indexed object
987 					if (items.length === undef) {
988 						for (key in items) {
989 							if (items.hasOwnProperty(key)) {
990 								if (item_name === 'attributes') {
991 									value = dom.getAttrib(node, key);
992 								} else {
993 									value = getStyle(node, key);
994 								}
995 
996 								if (similar && !value && !format.exact) {
997 									return;
998 								}

1001 									return;
1002 								}
1003 							}
1004 						}
1005 					} else {
1006 						// Only one match needed for indexed arrays
1007 						for (i = 0; i < items.length; i++) {
1008 							if (item_name === 'attributes' ? dom.getAttrib(node, items[i]) : getStyle(node, items[i])) {
1009 								return format;
1010 							}
1011 						}
1012 					}
1013 				}
1014 
1015 				return format;
1016 			}
1017

1019 				// Check each format in list
1020 				for (i = 0; i < formatList.length; i++) {
1021 					format = formatList[i];
1022 
1023 					// Name name, attributes, styles and classes
1024 					if (matchName(node, format) && matchItems(node, format, 'attributes') && matchItems(node, format, 'styles')) {
1025 						// Match classes
1026 						if ((classes = format.classes)) {
1027 							for (i = 0; i < classes.length; i++) {
1028 								if (!dom.hasClass(node, classes[i])) {
1029 									return;
1030 								}
1031 							}
1032 						}
1033 
1034 						return format;
1035 					}
1036 				}
