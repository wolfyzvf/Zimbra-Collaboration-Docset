 24 /**
 25  * Creates a vector.
 26  * @class
 27  * This class represents a vector.
 28  * 
 29  */
 30 AjxVector = function(array) {
 31 	this._array = array || [];
 32 };
 33 
 34 AjxVector.prototype.isAjxVector = true;
 35 
 36 /**
 37  * Returns a string representation of the object.
 38  * 
 39  * @param	{string}	sep		the seperator
 40  * @param	{boolean}	compress	if <code>true</code>, compress
 41  * 
 42  * @return	{string}	a string representation of the object
 43  */
 44 AjxVector.prototype.toString =
 45 function(sep, compress) {
 46 	if (compress !== true)
 47 		return this._array.join(sep);
 48 
 49 	var a = new Array();
 50 	for (var i = 0; i < this._array.length; i++) {
 51 		var x = this._array[i];
 52 		if  (x != undefined && x != null && x != "")
 53 			a.push(x);
 54 	}
 55 	return a.join(sep);
 56 };

 58 /**
 59  * Creates a vector from a given array.
 60  * 
 61  * @param	{array}	list		an array
 62  * @return	{AjxVector}		the vector
 63  */
 64 AjxVector.fromArray =
 65 function(list) {
 66 	var vec = new AjxVector();
 67 	if (AjxUtil.isArray1(list)) {
 68 		vec._array = list;
 69 	}
 70 	return vec;
 71 };

 83 /**
 84  * Adds a object to the vector.
 85  * 
 86  * @param	{Object}	obj		the object
 87  * @param	{number}		index	the index where to add
 88  * @param	{boolean}	noDuplicates	if <code>true</code>, confirm the object is not in vector before adding
 89  */
 90 AjxVector.prototype.add =
 91 function(obj, index, noDuplicates) {
 92 	// if no duplicates, search for the obj in list and return if found.
 93 	if (noDuplicates && this.contains(obj)) {
 94 		return;
 95 	}
 96 
 97 	AjxUtil.arrayAdd(this._array, obj, index);
 98 };

100 /**
101  * Adds the given array.
102  * 
103  * @param	{array}		list		an array
104  */
105 AjxVector.prototype.addList =
106 function(list) {
107 	if (!list) return;
108 
109 	if (list.length) {// array
110 		this._array = this._array.concat(list);
111 	} else if (list.size && list.size()) {// AjxVector
112 		// in new window, IE seems to lose its rtti :(
113 		if (AjxEnv.isIE && (!(list._array instanceof Array))) {
114 			var newList = [];
115 			for (var i = 0; i < list._array.length; i++) {
116 				newList.push(list._array[i]);
117 			}
118 			list._array = newList;
119 		}
120 
121 		this._array = this._array.concat(list._array);
122 	}
123 };

136 /**
137  * Removes the object at the given index.
138  * 
139  * @param	{number}	index		the index
140  * @return	{Object}	the object at the index or <code>null</code> if no object at index
141  */
142 AjxVector.prototype.removeAt =
143 function(index) {
144 	if (index >= this._array.length || index < 0)
145 		return null;
146 
147 	var delArr = this._array.splice(index, 1);
148 	var ret = null;
149 	if (delArr) {
150 		ret = delArr[0];
151 	}
152 	return ret;
153 };

200 /**
201  * Replaces an object.
202  * 
203  * @param	{Object}	obj		the object to replace
204  * @param	{Object}	newObj	the new object
205  * @return	{Object}	the replaced object or <code>null</code> if not replaced
206  */
207 AjxVector.prototype.replaceObject =
208 function(obj, newObj) {
209 	for (var i = 0; i < this._array.length; i++) {
210 		if (this._array[i] == obj) {
211 			this._array[i] = newObj;
212 			return obj;
213 		}
214 	}
215 	return null;
216 };

218 /**
219  * Returns the index of the obj given w/in vector
220  *
221  * @param {Object}	obj			the object being looked for
222  * @return	{number}	the index or -1 if not found
223  */
224 AjxVector.prototype.indexOf =
225 function(obj) {
226 	if (obj == null)
227 		return -1;
228 
229 	for (var i = 0; i < this._array.length; i++) {
230 		if (this._array[i] == obj)
231 			return i;
232 	}
233 	return -1;
234 };

236 /**
237  * Returns the index of the obj given w/in vector
238  *
239  * @param {Object}	obj			the object being looked for
240  * @param {function}	compareFunc	a function for comparing objects
241  * @return	{number}	the index or -1 if not found
242  */
243 AjxVector.prototype.indexOfLike =
244 function(obj, compareFunc) {
245 	var value = compareFunc.call(obj);
246 
247 	for (var i = 0; i < this._array.length; i++) {
248 		var test = compareFunc.call(this._array[i]);
249 		if (test == value)
250 			return i;
251 	}
252 	return -1;
253 };

279 /**
280  * Returns true if the vector contains the given object, using the given
281  * function to compare objects. The comparison function should return a
282  * type for which the equality test (==) is meaningful, such as a string
283  * or a base type.
284  *
285  * @param {Object}	obj			the object being looked for
286  * @param {function}	compareFunc	a function for comparing objects
287  * @return	{boolean}	<code>true</code> if the object is found
288  */
289 AjxVector.prototype.containsLike =
290 function(obj, compareFunc) {
291 	var value = compareFunc.call(obj);
292 	for (var i = 0; i < this._array.length; i++) {
293 		var test = compareFunc.call(this._array[i]);
294 		if (test == value)
295 			return true;
296 	}
297 	return false;
298 };

374 /**
375  * Performs a binary search.
376  * 
377  * @param	{Object}	valueToFind		the value
378  * @param	{function}	sortFunc		the sort function
379  * @return	{number}	the index
380  */
381 AjxVector.prototype.binarySearch =
382 function(valueToFind, sortFunc) {
383 	if (!sortFunc) {
384 		sortFunc = AjxVector._defaultArrayComparator;
385 	}
386 
387 	var l = 0;
388 	var arr = this._array;
389 	var u = arr.length - 1;
390 
391 	while(true) {
392 		if (u < l) {
393 			return -1;
394 		}
395 
396 		var i = Math.floor((l + u)/ 2);
397 		var comparisonResult = sortFunc(valueToFind, arr[i]);
398 
399 		if (comparisonResult < 0) {
400 			u = i - 1;
401 		} else if (comparisonResult > 0) {
402 			l = i + 1;
403 		} else {
404 			return i;
405 		}
406 	}
407 };

409 AjxVector.prototype.merge =
410 function(offset, list) {
411 
412 	if (offset < 0)
413 		return;
414 
415 	var rawList = list instanceof AjxVector ? list.getArray() : list;
416 
417 	var limit = this._array.length < (offset+rawList.length)
418 		? this._array.length
419 		: offset+rawList.length;
420 
421 	if (offset < this._array.length) {
422 		// replace any overlapping items in vector
423 		var count = 0;
424 		for (var i=offset; i<limit; i++)
425 			this._array[i] = rawList[count++];
426 
427 		// and append the rest
428 		if (count < rawList.length)
429 			this._array = this._array.concat(rawList.slice(count));
430 	} else {
431 		// otherwise, just append the raw list to the end
432 		this._array = this._array.concat(rawList);
433 	}
434 };

439 AjxVector._defaultArrayComparator =
440 function(a, b) {
441 	return a < b ? -1 : (a > b ? 1 : 0);
442 };

444 // Apply function f for each element of the array.  Optionally call it
445 // in the context of obj object.  If "f" is a string, then for each
446 // non-null array element call its "f" member function.
447 AjxVector.prototype.foreach = function(f, obj) {
448 	var l = this.size(), i = 0, el;
449 	if (typeof f == "function") {
450 		while (--l >= 0)
451 			f.call(obj, this.get(i), i++);
452 	} else {
453 		while (--l >= 0) {
454 			el = this.get(i++);
455 			if (el != null)
456 				el[f].call(el); // assuming function
457 		}
458 	}
459 };

461 /**
462  * Return a new AjxVector which contains the results of calling f
463  * (optionally in the context obj) for each element of this array.
464  * <ul>
465  * <li>If "f" is a string, then for each element el:
466  * <ul>
467  * <li>if el[f] is a function, call el[f] and push the result in the returned array.</li>
468  * <li>otherwise push el[f]</li>
469  * </ul>
470  * </li>
471  * </ul>
472  * 
473  * @param	{function}	f 	the function
474  * @param	{Object}	obj		the obj context
475  * @return	{AjxVector}		the resulting vector
476  */
477 AjxVector.prototype.map = function(f, obj) {
478 	var a = [], i = this.size(), el;
479 	if (typeof f == "function") {
480 		while (--i >= 0)
481 			a[i] = f.call(obj, this.get(i), i);
482 	} else if (f instanceof AjxCallback) {
483 		while (--i >= 0)
484 			a[i] = f.run(this.get(i), i);
485 	} else {
486 		while (--i >= 0) {
487 			el = this.get(i);
488 			if (el != null) {
489 				if (typeof el[f] == "function")
490 					a.unshift(el[f].call(el));
491 				else
492 					a.unshift(el[f]);
493 			}
494 		}
495 	}
496 	return AjxVector.fromArray(a);
497 };

509 /**
510  * Return a new AjxVector containing the elements from this vector
511  * except those for which f(el) returns true.  Otherwise said,
512  * "SUBtracts" from this vector those elements for which f(el) returns true.
513  *
514  * @param	{function}	f 	the function
515  * @param	{Object}	obj		the obj context
516  * @return	{AjxVector}		the resulting vector
517  */
518 AjxVector.prototype.sub = function(f, obj) {
519 	var a = [], l = this.size(), i = 0, el;
520 	while (--l >= 0) {
521 		el = this.get(i++);
522 		if (!f.call(obj, el, i))
523 			a.push(el);
524 	}
525 	return AjxVector.fromArray(a);
526 };
