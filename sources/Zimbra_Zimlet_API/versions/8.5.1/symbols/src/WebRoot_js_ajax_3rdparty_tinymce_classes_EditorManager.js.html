 12  * This class used as a factory for manager for tinymce.Editor instances.
 13  *
 14  * @example
 15  * tinymce.EditorManager.init({});
 16  *
 17  * @class tinymce.EditorManager
 18  * @mixes tinymce.util.Observable
 19  * @static
 20  */

 22 	"tinymce/Editor",

 31 	var DOM = DOMUtils.DOM;
 32 	var explode = Tools.explode, each = Tools.each, extend = Tools.extend;
 33 	var instanceCounter = 0, beforeUnloadDelegate, EditorManager;

 35 	function removeEditorFromList(editor) {
 36 		var editors = EditorManager.editors, removedFromList;
 37 
 38 		delete editors[editor.id];
 39 
 40 		for (var i = 0; i < editors.length; i++) {
 41 			if (editors[i] == editor) {
 42 				editors.splice(i, 1);
 43 				removedFromList = true;
 44 				break;
 45 			}
 46 		}
 47 
 48 		// Select another editor since the active one was removed
 49 		if (EditorManager.activeEditor == editor) {
 50 			EditorManager.activeEditor = editors[0];
 51 		}
 52 
 53 		// Clear focusedEditor if necessary, so that we don't try to blur the destroyed editor
 54 		if (EditorManager.focusedEditor == editor) {
 55 			EditorManager.focusedEditor = null;
 56 		}
 57 
 58 		return removedFromList;
 59 	}
 60 
 61 	function purgeDestroyedEditor(editor) {
 62 		// User has manually destroyed the editor lets clean up the mess
 63 		if (editor && !(editor.getContainer() || editor.getBody()).parentNode) {
 64 			removeEditorFromList(editor);
 65 			editor.destroy(true);
 66 			editor = null;
 67 		}
 68 
 69 		return editor;
 70 	}

127 		setup: function() {
128 			var self = this, baseURL, documentBaseURL, suffix = "", preInit, src;
129 
130 			// Get base URL for the current document
131 			documentBaseURL = document.location.href.replace(/[\?#].*$/, '').replace(/[\/\\][^\/]+$/, '');
132 			if (!/[\/\\]$/.test(documentBaseURL)) {
133 				documentBaseURL += '/';
134 			}
135 
136 			// If tinymce is defined and has a base use that or use the old tinyMCEPreInit
137 			preInit = window.tinymce || window.tinyMCEPreInit;
138 			if (preInit) {
139 				baseURL = preInit.base || preInit.baseURL;
140 				suffix = preInit.suffix;
141 			} else {
142 				// Get base where the tinymce script is located
143 				var scripts = document.getElementsByTagName('script');
144 				for (var i = 0; i < scripts.length; i++) {
145 					src = scripts[i].src;
146 
147 					// Script types supported:
148 					// tinymce.js tinymce.min.js tinymce.dev.js
149 					// tinymce.jquery.js tinymce.jquery.min.js tinymce.jquery.dev.js
150 					// tinymce.full.js tinymce.full.min.js tinymce.full.dev.js
151 					if (/tinymce(\.full|\.jquery|)(\.min|\.dev|)\.js/.test(src)) {
152 						if (src.indexOf('.min') != -1) {
153 							suffix = '.min';
154 						}
155 
156 						baseURL = src.substring(0, src.lastIndexOf('/'));
157 						break;
158 					}
159 				}
160 
161 				// We didn't find any baseURL by looking at the script elements
162 				// Try to use the document.currentScript as a fallback
163 				if (!baseURL && document.currentScript) {
164 					src = document.currentScript.src;
165 
166 					if (src.indexOf('.min') != -1) {
167 						suffix = '.min';
168 					}
169 
170 					baseURL = src.substring(0, src.lastIndexOf('/'));
171 				}
172 			}
173 
174 			/**
175 			 * Base URL where the root directory if TinyMCE is located.
176 			 *
177 			 * @property baseURL
178 			 * @type String
179 			 */
180 			self.baseURL = new URI(documentBaseURL).toAbsolute(baseURL);
181 
182 			/**
183 			 * Document base URL where the current document is located.
184 			 *
185 			 * @property documentBaseURL
186 			 * @type String
187 			 */
188 			self.documentBaseURL = documentBaseURL;
189 
190 			/**
191 			 * Absolute baseURI for the installation path of TinyMCE.
192 			 *
193 			 * @property baseURI
194 			 * @type tinymce.util.URI
195 			 */
196 			self.baseURI = new URI(self.baseURL);
197 
198 			/**
199 			 * Current suffix to add to each plugin/theme that gets loaded for example ".min".
200 			 *
201 			 * @property suffix
202 			 * @type String
203 			 */
204 			self.suffix = suffix;
205 
206 			self.focusManager = new FocusManager(self);
207 		},

209 		/**
210 		 * Initializes a set of editors. This method will create editors based on various settings.
211 		 *
212 		 * @method init
213 		 * @param {Object} settings Settings object to be passed to each editor instance.
214 		 * @example
215 		 * // Initializes a editor using the longer method
216 		 * tinymce.EditorManager.init({
217 		 *    some_settings : 'some value'
218 		 * });
219 		 *
220 		 * // Initializes a editor instance using the shorter version
221 		 * tinyMCE.init({
222 		 *    some_settings : 'some value'
223 		 * });
224 		 */
225 		init: function(settings) {
226 			var self = this, editors = [], editor;
227 
228 			function createId(elm) {
229 				var id = elm.id;
230 
231 				// Use element id, or unique name or generate a unique id
232 				if (!id) {
233 					id = elm.name;
234 
235 					if (id && !DOM.get(id)) {
236 						id = elm.name;
237 					} else {
238 						// Generate unique name
239 						id = DOM.uniqueId();
240 					}
241 
242 					elm.setAttribute('id', id);
243 				}
244 
245 				return id;
246 			}
247 
248 			function createEditor(id, settings) {
249 				if (!purgeDestroyedEditor(self.get(id))) {
250 					var editor = new Editor(id, settings, self);
251 					editors.push(editor);
252 					editor.render();
253 				}
254 			}
255 
256 			function execCallback(se, n, s) {
257 				var f = se[n];
258 
259 				if (!f) {
260 					return;
261 				}
262 
263 				return f.apply(s || this, Array.prototype.slice.call(arguments, 2));
264 			}
265 
266 			function hasClass(n, c) {
267 				return c.constructor === RegExp ? c.test(n.className) : DOM.hasClass(n, c);
268 			}
269 
270 			function readyHandler() {
271 				var l, co;
272 
273 				DOM.unbind(window, 'ready', readyHandler);
274 
275 				execCallback(settings, 'onpageload');
276 
277 				if (settings.types) {
278 					// Process type specific selector
279 					each(settings.types, function(type) {
280 						each(DOM.select(type.selector), function(elm) {
281 							createEditor(createId(elm), extend({}, settings, type));
282 						});
283 					});
284 
285 					return;
286 				} else if (settings.selector) {
287 					// Process global selector
288 					each(DOM.select(settings.selector), function(elm) {
289 						createEditor(createId(elm), settings);
290 					});
291 
292 					return;
293 				}
294 
295 				// Fallback to old setting
296 				switch (settings.mode) {
297 					case "exact":
298 						l = settings.elements || '';
299 
300 						if (l.length > 0) {
301 							each(explode(l), function(v) {
302 								if (DOM.get(v)) {
303 									editor = new Editor(v, settings, self);
304 									editors.push(editor);
305 									editor.render();
306 								} else {
307 									each(document.forms, function(f) {
308 										each(f.elements, function(e) {
309 											if (e.name === v) {
310 												v = 'mce_editor_' + instanceCounter++;
311 												DOM.setAttrib(e, 'id', v);
312 												createEditor(v, settings);
313 											}
314 										});
315 									});
316 								}
317 							});
318 						}
319 						break;
320 
321 					case "textareas":
322 					case "specific_textareas":
323 						each(DOM.select('textarea'), function(elm) {
324 							if (settings.editor_deselector && hasClass(elm, settings.editor_deselector)) {
325 								return;
326 							}
327 
328 							if (!settings.editor_selector || hasClass(elm, settings.editor_selector)) {
329 								createEditor(createId(elm), settings);
330 							}
331 						});
332 						break;
333 				}
334 
335 				// Call onInit when all editors are initialized
336 				if (settings.oninit) {
337 					l = co = 0;
338 
339 					each(editors, function(ed) {
340 						co++;
341 
342 						if (!ed.initialized) {
343 							// Wait for it
344 							ed.on('init', function() {
345 								l++;
346 
347 								// All done
348 								if (l == co) {
349 									execCallback(settings, 'oninit');
350 								}
351 							});
352 						} else {
353 							l++;
354 						}
355 
356 						// All done
357 						if (l == co) {
358 							execCallback(settings, 'oninit');
359 						}
360 					});
361 				}
362 			}
363 
364 			self.settings = settings;
365 
366 			DOM.bind(window, 'ready', readyHandler);
367 		},

369 		/**
370 		 * Returns a editor instance by id.
371 		 *
372 		 * @method get
373 		 * @param {String/Number} id Editor instance id or index to return.
374 		 * @return {tinymce.Editor} Editor instance to return.
375 		 * @example
376 		 * // Adds an onclick event to an editor by id (shorter version)
377 		 * tinymce.get('mytextbox').on('click', function(e) {
378 		 *    ed.windowManager.alert('Hello world!');
379 		 * });
380 		 *
381 		 * // Adds an onclick event to an editor by id (longer version)
382 		 * tinymce.EditorManager.get('mytextbox').on('click', function(e) {
383 		 *    ed.windowManager.alert('Hello world!');
384 		 * });
385 		 */
386 		get: function(id) {
387 			if (!arguments.length) {
388 				return this.editors;
389 			}
390 
391 			return id in this.editors ? this.editors[id] : null;
392 		},

394 		/**
395 		 * Adds an editor instance to the editor collection. This will also set it as the active editor.
396 		 *
397 		 * @method add
398 		 * @param {tinymce.Editor} editor Editor instance to add to the collection.
399 		 * @return {tinymce.Editor} The same instance that got passed in.
400 		 */
401 		add: function(editor) {
402 			var self = this, editors = self.editors;
403 
404 			// Add named and index editor instance
405 			editors[editor.id] = editor;
406 			editors.push(editor);
407 
408 			self.activeEditor = editor;
409 
410 			/**
411 			 * Fires when an editor is added to the EditorManager collection.
412 			 *
413 			 * @event AddEditor
414 			 * @param {Object} e Event arguments.
415 			 */
416 			self.fire('AddEditor', {editor: editor});
417 
418 			if (!beforeUnloadDelegate) {
419 				beforeUnloadDelegate = function() {
420 					self.fire('BeforeUnload');
421 				};
422 
423 				DOM.bind(window, 'beforeunload', beforeUnloadDelegate);
424 			}
425 
426 			return editor;
427 		},

441 		/**
442 		 * Removes a editor or editors form page.
443 		 *
444 		 * @example
445 		 * // Remove all editors bound to divs
446 		 * tinymce.remove('div');
447 		 *
448 		 * // Remove all editors bound to textareas
449 		 * tinymce.remove('textarea');
450 		 *
451 		 * // Remove all editors
452 		 * tinymce.remove();
453 		 *
454 		 * // Remove specific instance by id
455 		 * tinymce.remove('#id');
456 		 *
457 		 * @method remove
458 		 * @param {tinymce.Editor/String/Object} [selector] CSS selector or editor instance to remove.
459 		 * @return {tinymce.Editor} The editor that got passed in will be return if it was found otherwise null.
460 		 */
461 		remove: function(selector) {
462 			var self = this, i, editors = self.editors, editor;
463 
464 			// Remove all editors
465 			if (!selector) {
466 				for (i = editors.length - 1; i >= 0; i--) {
467 					self.remove(editors[i]);
468 				}
469 
470 				return;
471 			}
472 
473 			// Remove editors by selector
474 			if (typeof(selector) == "string") {
475 				selector = selector.selector || selector;
476 
477 				each(DOM.select(selector), function(elm) {
478 					self.remove(editors[elm.id]);
479 				});
480 
481 				return;
482 			}
483 
484 			// Remove specific editor
485 			editor = selector;
486 
487 			// Not in the collection
488 			if (!editors[editor.id]) {
489 				return null;
490 			}
491 
492 			/**
493 			 * Fires when an editor is removed from EditorManager collection.
494 			 *
495 			 * @event RemoveEditor
496 			 * @param {Object} e Event arguments.
497 			 */
498 			if (removeEditorFromList(editor)) {
499 				self.fire('RemoveEditor', {editor: editor});
500 			}
501 
502 			if (!editors.length) {
503 				DOM.unbind(window, 'beforeunload', beforeUnloadDelegate);
504 			}
505 
506 			editor.remove();
507 
508 			return editor;
509 		},

511 		/**
512 		 * Executes a specific command on the currently active editor.
513 		 *
514 		 * @method execCommand
515 		 * @param {String} c Command to perform for example Bold.
516 		 * @param {Boolean} u Optional boolean state if a UI should be presented for the command or not.
517 		 * @param {String} v Optional value parameter like for example an URL to a link.
518 		 * @return {Boolean} true/false if the command was executed or not.
519 		 */
520 		execCommand: function(cmd, ui, value) {
521 			var self = this, editor = self.get(value);
522 
523 			// Manager commands
524 			switch (cmd) {
525 				case "mceAddEditor":
526 					if (!self.get(value)) {
527 						new Editor(value, self.settings, self).render();
528 					}
529 
530 					return true;
531 
532 				case "mceRemoveEditor":
533 					if (editor) {
534 						editor.remove();
535 					}
536 
537 					return true;
538 
539 				case 'mceToggleEditor':
540 					if (!editor) {
541 						self.execCommand('mceAddEditor', 0, value);
542 						return true;
543 					}
544 
545 					if (editor.isHidden()) {
546 						editor.show();
547 					} else {
548 						editor.hide();
549 					}
550 
551 					return true;
552 			}
553 
554 			// Run command on active editor
555 			if (self.activeEditor) {
556 				return self.activeEditor.execCommand(cmd, ui, value);
557 			}
558 
559 			return false;
560 		},
