 27 /**
 28  * Creates a list view.
 29  * @class
 30  * A list view presents a list of items as rows with fields (columns).
 31  *
 32  * @author Parag Shah
 33  * @author Conrad Damon
 34  *
 35  * @param {Hash}	params		a hash of parameters
 36  * @param {DwtComposite}	params.parent		the parent widget
 37  * @param {String}	params.className		the CSS class
 38  * @param {constant}	params.posStyle		the positioning style
 39  * @param {String}	params.id			the HTML ID for element
 40  * @param {Array}	params.headerList	the list of IDs for columns
 41  * @param {Boolean}	params.noMaximize	if <code>true</code>, all columns are fixed-width (otherwise, one will expand to fill available space)
 42  * @param {constant}	params.view			the ID of view
 43  * @param {constant}	params.type			the type of item displayed
 44  * @param {ZmListController}	params.controller	the owning controller
 45  * @param {DwtDropTarget}	params.dropTgt		the drop target
 46  * @param {Boolean}	params.pageless		if <code>true</code>, enlarge page via scroll rather than pagination
 47  *        
 48  * @extends		DwtListView
 49  */
 50 ZmListView = function(params) {
 51 
 52 	if (arguments.length == 0) { return; }
 53 	
 54 	params.id = params.id || ZmId.getViewId(params.view);
 55 	DwtListView.call(this, params);
 56 
 57 	this.view = params.view;
 58 	this.type = params.type;
 59 	this._controller = params.controller;
 60 	this.setDropTarget(params.dropTgt);
 61 
 62 	// create listeners for changes to the list model, folder tree, and tag list
 63 	this._listChangeListener = new AjxListener(this, this._changeListener);
 64 	this._tagListChangeListener = new AjxListener(this, this._tagChangeListener);
 65 	var tagList = appCtxt.getTagTree();
 66 	if (tagList) {
 67 		tagList.addChangeListener(this._tagListChangeListener);
 68 	}
 69 	var folderTree = appCtxt.getFolderTree();
 70 	if (folderTree) {
 71 		folderTree.addChangeListener(new AjxListener(this, this._folderChangeListener));
 72 	}
 73 
 74 	this._handleEventType = {};
 75 	this._handleEventType[this.type] = true;
 76 	this._disallowSelection = {};
 77 	this._disallowSelection[ZmItem.F_FLAG] = true;
 78 	this._disallowSelection[ZmItem.F_MSG_PRIORITY] = true;
 79 	this._selectAllEnabled = false;
 80 
 81 	if (params.dropTgt) {
 82 		var args = {container:this._parentEl, threshold:15, amount:5, interval:10, id:params.id};
 83 		this._dndScrollCallback = new AjxCallback(null, DwtControl._dndScrollCallback, [args]);
 84 		this._dndScrollId = params.id;
 85 	}
 86 
 87 	this._isPageless = params.pageless;
 88 	if (this._isPageless) {
 89 		Dwt.setHandler(this._getScrollDiv(), DwtEvent.ONSCROLL, ZmListView.handleScroll);
 90 	}
 91 	this._state = {};
 92 };

143 ZmListView.prototype.set =
144 function(list, sortField) {
145 
146 	this._sortByString = this._controller._currentSearch && this._controller._currentSearch.sortBy;
147     //TODO: We need a longer term fix but this is to prevent a sort by that doesn't match our ZmSearch
148 	//constants and lead to notification issues.
149 	if (this._sortByString) {
150     	this._sortByString = this._sortByString.replace("asc", "Asc").replace("desc", "Desc");// bug 75687
151 	}
152 
153 	var settings = appCtxt.getSettings();
154 	if (!appCtxt.isExternalAccount() && this.view) {
155 		appCtxt.set(ZmSetting.SORTING_PREF,
156 					this._sortByString,
157 					this.view,
158 					false, //setDefault
159 					false, //skipNotify
160 					null, //account
161 					settings && !settings.persistImplicitSortPrefs(this.view)); //skipImplicit - do not persist
162 	}
163 
164 	this.setSelectionHdrCbox(false);
165 
166 	// bug fix #28595 - in multi-account, reset tag list change listeners
167 	if (appCtxt.multiAccounts) {
168 		var tagList = appCtxt.getTagTree();
169 		if (tagList) {
170 			tagList.addChangeListener(this._tagListChangeListener);
171 		}
172 	}
173 
174 	if (this._isPageless) {
175 		if (this._itemsToAdd) {
176 			if (this._itemsToAdd.length) {
177 				this.addItems(this._itemsToAdd);
178 				this._itemsToAdd = null;
179 			}
180 		} else {
181 			var lvList = list;
182 			if (list && list.isZmList) {
183 				list.addChangeListener(this._listChangeListener);
184 				lvList = list.getSubList(0, list.size());
185 			}
186 			DwtListView.prototype.set.call(this, lvList, sortField);
187 		}
188 		this._setRowHeight();
189 	} else {
190 		var subList;
191 		if (list && list.isZmList) {
192 			list.addChangeListener(this._listChangeListener);
193 			subList = list.getSubList(this.offset, this.getLimit());
194 		} else {
195 			subList = list;
196 		}
197 		DwtListView.prototype.set.call(this, subList, sortField);
198 	}
199 	this._rendered = true;
200 
201 	// check in case there are more items but no scrollbar
202 	if (this._isPageless) {
203 		AjxTimedAction.scheduleAction(new AjxTimedAction(this, this._checkItemCount), 1000);
204 	}
205 };

262 ZmListView.prototype._changeListener =
263 function(ev) {
264 
265 	var item = this._getItemFromEvent(ev);
266 	if (!item || ev.handled || !this._handleEventType[item.type]) {
267 		return;
268 	}
269 
270 	if (ev.event === ZmEvent.E_TAGS || ev.event === ZmEvent.E_REMOVE_ALL) {
271 		this._replaceTagImage(item, ZmItem.F_TAG, this._getClasses(ZmItem.F_TAG));
272 	}
273 
274 	if (ev.event === ZmEvent.E_FLAGS) {
275 		var flags = ev.getDetail("flags");
276 		for (var j = 0; j < flags.length; j++) {
277 			var flag = flags[j];
278 			var on = item[ZmItem.FLAG_PROP[flag]];
279 			if (flag === ZmItem.FLAG_FLAGGED) {
280 				this._setImage(item, ZmItem.F_FLAG, on ? "FlagRed" : "FlagDis", this._getClasses(ZmItem.F_FLAG));
281 			}
282 			else if (flag === ZmItem.FLAG_ATTACH) {
283 				this._setImage(item, ZmItem.F_ATTACHMENT, on ? "Attachment" : null, this._getClasses(ZmItem.F_ATTACHMENT));
284 			}
285 			else if (flag === ZmItem.FLAG_PRIORITY) {
286 				this._setImage(item, ZmItem.F_MSG_PRIORITY, on ? "Priority" : "PriorityDis", this._getClasses(ZmItem.F_MSG_PRIORITY));
287 			}
288 		}
289 	}
290 
291 	// Note: move and delete support batch notification mode
292 	if (ev.event === ZmEvent.E_DELETE || ev.event === ZmEvent.E_MOVE) {
293 		var items = ev.batchMode ? this._getItemsFromBatchEvent(ev) : [item];
294 		var needsSort = false;
295 		for (var i = 0, len = items.length; i < len; i++) {
296 			var item = items[i];
297             var movedHere = (item.type === ZmId.ITEM_CONV) ? item.folders[this._folderId] : item.folderId === this._folderId;
298 			if (movedHere && ev.event === ZmEvent.E_MOVE) {
299 				// We've moved the item into this folder
300 				if (this._getRowIndex(item) === null) { // Not already here
301 					this.addItem(item);
302 					// TODO: couldn't we just find the sort index and insert it?
303 					needsSort = true;
304 				}
305 			}
306 			else {
307 				// remove the item if the user is working in this view,
308 				// if we know the item no longer matches the search, or if the item was hard-deleted
309 				if (ev.event === ZmEvent.E_DELETE || this.view == appCtxt.getCurrentViewId() || this._controller._currentSearch.matches(item) === false) {
310 					this.removeItem(item, true, ev.batchMode);
311 					// if we've removed it from the view, we should remove it from the reference
312 					// list as well so it doesn't get resurrected via replenishment *unless*
313 					// we're dealing with a canonical list (i.e. contacts)
314 					var itemList = this.getItemList();
315 					if (ev.event !== ZmEvent.E_MOVE || !itemList.isCanonical) {
316 						itemList.remove(item);
317 					}
318 				}
319 			}
320 		}
321 		if (needsSort) {
322 			this._saveState({scroll: true, selection:true, focus: true});
323 			this._redoSearch(this._restoreState.bind(this, this._state));
324 		}
325 		if (ev.batchMode) {
326 			this._fixAlternation(0);
327 		}
328 		this._checkReplenishOnTimer();
329 		this._controller._resetToolbarOperations();
330 	}
331 };

333 ZmListView.prototype._getItemFromEvent =
334 function(ev) {
335 	var item = ev.item;
336 	if (!item) {
337 		var items = ev.getDetail("items");
338 		item = (items && items.length) ? items[0] : null;
339 	}
340 	return item;
341 };

343 ZmListView.prototype._getItemsFromBatchEvent =
344 function(ev) {
345 
346 	if (!ev.batchMode) { return []; }
347 
348 	var items = ev.items;
349 	if (!items) {
350 		items = [];
351 		var notifs = ev.getDetail("notifs");
352 		if (notifs && notifs.length) {
353 			for (var i = 0, len = notifs.length; i < len; i++) {
354 				var mod = notifs[i];
355 				items.push(mod.item || appCtxt.cacheGet(mod.id));
356 			}
357 		}
358 	}
359 
360 	return items;
361 };

377 ZmListView.prototype._checkReplenishOnTimer =
378 function(ev) {
379 	if (!this.allSelected) {
380 		if (!this._isPageless) {

382 		} else {

385 				this._replenishTimedAction = new AjxTimedAction(this, this._handleResponseCheckReplenish);
386 			}
387 			AjxTimedAction.scheduleAction(this._replenishTimedAction, 10);
388 		}
389 	}

410 ZmListView.prototype._folderChangeListener =
411 function(ev) {
412 	// make sure this is current list view
413 	if (appCtxt.getCurrentController() != this._controller) { return; }
414 	// see if it will be handled by app's postNotify()
415 	if (this._controller._app._checkReplenishListView == this) { return; }
416 
417 	var organizers = ev.getDetail("organizers");
418 	var organizer = (organizers && organizers.length) ? organizers[0] : ev.source;
419 
420 	var id = organizer.id;
421 	var fields = ev.getDetail("fields");
422 	if (ev.event == ZmEvent.E_MODIFY) {
423 		if (!fields) { return; }
424 		if (fields[ZmOrganizer.F_TOTAL]) {
425 			this._controller._resetNavToolBarButtons();
426 		}
427 	}
428 };

430 ZmListView.prototype._tagChangeListener =
431 function(ev) {
432 	if (ev.type != ZmEvent.S_TAG) return;
433 
434 	var fields = ev.getDetail("fields");
435 
436 	var divs = this._getChildren();
437 	var tag = ev.getDetail("organizers")[0];
438 	for (var i = 0; i < divs.length; i++) {
439 		var item = this.getItemFromElement(divs[i]);
440 		if (!item || !item.tags || !item.hasTag(tag.name)) {
441 			continue;
442 		}
443 		var updateRequired = false;
444 		if (ev.event == ZmEvent.E_MODIFY && (fields && (fields[ZmOrganizer.F_COLOR] || fields[ZmOrganizer.F_NAME]))) {
445 			//rename could change the color (for remote shared items, from the remote gray icon to local color and vice versa)
446 			updateRequired = item.tags.length == 1;
447 		}
448 		else if (ev.event == ZmEvent.E_DELETE) {
449 			updateRequired = true;
450 		}
451 		else if (ev.event == ZmEvent.E_CREATE) {
452 			//this could affect item if it had a tag not on tag list (remotely created on shared item, either shared by this user or shared to this user)
453 			updateRequired = true;
454 		}
455 		if (updateRequired) {
456 			this._replaceTagImage(item, ZmItem.F_TAG, this._getClasses(ZmItem.F_TAG));
457 		}
458 	}
459 };

492 ZmListView.prototype._getCellContents =
493 function(htmlArr, idx, item, field, colIdx, params, classes) {
494 	if (field == ZmItem.F_SELECTION) {
495 		idx = this._getImageHtml(htmlArr, idx, "CheckboxUnchecked", this._getFieldId(item, field), classes);
496 	} else if (field == ZmItem.F_TYPE) {

498 	} else if (field == ZmItem.F_FLAG) {
499 		idx = this._getImageHtml(htmlArr, idx, this._getFlagIcon(item.isFlagged), this._getFieldId(item, field), classes);
500 	} else if (field == ZmItem.F_TAG) {
501 		idx = this._getImageHtml(htmlArr, idx, item.getTagImageInfo(), this._getFieldId(item, field), classes);
502 	} else if (field == ZmItem.F_ATTACHMENT) {
503 		idx = this._getImageHtml(htmlArr, idx, item.hasAttach ? "Attachment" : null, this._getFieldId(item, field), classes);
504 	} else if (field == ZmItem.F_DATE) {
505 		htmlArr[idx++] = AjxDateUtil.computeDateStr(params.now || new Date(), item.date);
506 	} else if (field == ZmItem.F_PRIORITY) {
507         var priorityImage = null;
508         if (item.isHighPriority) {
509             priorityImage = "PriorityHigh_list";
510         } else if (item.isLowPriority) {
511 			priorityImage = "PriorityLow_list";
512 		}
513 		if (priorityImage) {
514         	idx = this._getImageHtml(htmlArr, idx, priorityImage, this._getFieldId(item, field), classes);
515 		} else {
516 			htmlArr[idx++] = "<div id='" + this._getFieldId(item, field) + "' " + AjxUtil.getClassAttr(classes) + "></div>";
517 		}
518 	} else {
519 		idx = DwtListView.prototype._getCellContents.apply(this, arguments);
520 	}
521 	return idx;
522 };

585 /**
586  * Parse the DOM ID to figure out what got clicked. IDs consist of three to five parts
587  * joined by the "|" character.
588  *
589  *		type		type of ID (zli, zlir, zlic, zlif) - see DwtId.WIDGET_ITEM*)
590  * 		view		view identifier (eg "TV")
591  * 		item ID		usually numeric
592  * 		field		field identifier (eg "fg") - see ZmId.FLG_*
593  * 		participant	index of participant
594  */
595 ZmListView.prototype._parseId =
596 function(id) {

693 ZmListView.prototype._itemClicked =
694 function(clickedEl, ev) {
695 	if (appCtxt.get(ZmSetting.SHOW_SELECTION_CHECKBOX) && ev.button == DwtMouseEvent.LEFT) {
696 		if (!ev.shiftKey && !ev.ctrlKey) {
697 			// get the field being clicked
698 			var target = this._getEventTarget(ev);
699 			var id = (target && target.id && target.id.indexOf("AjxImg") == -1) ? target.id : clickedEl.id;
700 			var m = id ? this._parseId(id) : null;
701 			if (m && (m.field == ZmItem.F_SELECTION || m.field == ZmItem.F_SELECTION_CELL)) {
702 				//user clicked on a checkbox
703 				if (this._selectedItems.size() == 1) {
704 					var sel = this._selectedItems.get(0);
705 					var item = this.getItemFromElement(sel);
706 					var selFieldId = item ? this._getFieldId(item, ZmItem.F_SELECTION) : null;
707 					var selField = selFieldId ? document.getElementById(selFieldId) : null;
708 					if (selField && sel == clickedEl) {
709 						var isChecked = this._getItemData(sel, ZmListView.ITEM_CHECKED_ATT_NAME);
710 						this._setImage(item, ZmItem.F_SELECTION, isChecked ? ZmListView.UNCHECKED_IMAGE : ZmListView.CHECKED_IMAGE);
711 						this._setItemData(sel, ZmListView.ITEM_CHECKED_ATT_NAME, !isChecked);
712 						if (!isChecked) {
713 							return; //nothing else to do. It's already selected, and was the only selected one. Nothing to remove
714 						}
715 					} else {
716 						if (selField && !this._getItemData(sel, ZmListView.ITEM_CHECKED_ATT_NAME)) {
717 							this.deselectAll();
718 							this._markUnselectedViewedItem(true);
719 						}
720 					}
721 				}
722 				var bContained = this._selectedItems.contains(clickedEl);
723 				this.setMultiSelection(clickedEl, bContained);
724 				this._controller._setItemSelectionCountText();
725 				return;	// do not call base class if "selection" field was clicked
726 			}
727 		} else if (ev.shiftKey) {
728 			// uncheck all selected items first
729 			this._checkSelectedItems(false);
730 
731 			// run base class first so we get the finalized list of selected items
732 			DwtListView.prototype._itemClicked.call(this, clickedEl, ev);
733 
734 			// recheck new list of selected items
735 			this._checkSelectedItems(true);
736 
737 			return;
738 		}
739 	}

1428 ZmListView.getRowsNeeded =
1429 function(params) {
1430 
1431 	var div = params.scrollDiv;
1432 	var sh = div.scrollHeight, st = div.scrollTop, rh = params.rowHeight;
1433 
1434 	// view (porthole) height - everything measured relative to its top
1435 	// prefer clientHeight since (like scrollHeight) it doesn't include borders
1436 	var h = div.clientHeight || Dwt.getSize(div).y;
1437 
1438 	// where we'd like bottom of list view to be (with extra hidden items at bottom)
1439 	var target = h + (params.threshold * rh);
1440 
1441 	// where bottom of list view is (including hidden items)
1442 	var bottom = sh - st;
1443 
1444 	if (bottom == h) {
1445 		// handle cases where there's no scrollbar, but we have more items (eg tall browser, or replenishment)
1446 		bottom = (params.listSize * rh) - st;
1447 		if (st == 0 && params.sbCallback) {
1448 			// give list view a chance to fix width since it may be getting a scrollbar
1449 			params.sbCallback.run();
1450 		}
1451 	}
1452 
1453 	var rowsNeeded = 0;
1454 	if (bottom < target) {
1455 		// buffer below visible bottom of list view is not full
1456 		rowsNeeded = Math.max(Math.floor((target - bottom) / rh), params.limit);
1457 	}
1458 	return rowsNeeded;
1459 };
