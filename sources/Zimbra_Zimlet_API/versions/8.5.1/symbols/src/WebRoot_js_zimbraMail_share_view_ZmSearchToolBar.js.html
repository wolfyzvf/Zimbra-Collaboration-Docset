 22 /**
 23  * @class
 24  * This class represent a search toolbar. The components are some set of: an input field,
 25  * a search button, a save button, and a button to choose what type of item to search for.
 26  * 
 27  * @param {hash}			params		a hash of parameters:
 28  * @param {DwtComposite}	parent		the parent widget
 29  * @param {string}			id			an explicit ID to use for the control's HTML element
 30  * 
 31  * @extends		DwtComposite
 32  */
 33 ZmSearchToolBar = function(params) {
 34 
 35 	if (arguments.length == 0) { return; }
 36 
 37 	params.className = params.className || "ZmSearchToolbar";
 38 	DwtComposite.apply(this, arguments);
 39 	
 40 	this._button = {};
 41 	this._origin = ZmId.SEARCH;
 42 	this._searchMenu = null;
 43 	
 44 	this._createHtml();
 45 };

 75 /**
 76  * Defines a menu item to add when the types menu is created. Static so that it can be called before the
 77  * toolbar has been created.
 78  * 
 79  * @param {string}	id			ID of menu item
 80  * @param {hash}	params		menu item properties
 81  */
 82 ZmSearchToolBar.addMenuItem =
 83 function(id, params) {
 84 
 85 	if (params.msgKey)		   { ZmSearchToolBar.MSG_KEY[id]		 = params.msgKey; }
 86 	if (params.tooltipKey)	   { ZmSearchToolBar.TT_MSG_KEY[id]	 	 = params.tooltipKey; }
 87 	if (params.icon)		   { ZmSearchToolBar.ICON[id]			 = params.icon; }
 88 	if (params.shareIcon)	   { ZmSearchToolBar.SHARE_ICON[id]	 	 = params.shareIcon; }
 89 	if (params.setting)		   { ZmSearchToolBar.SETTING[id]		 = params.setting; }
 90 	if (params.id)			   { ZmSearchToolBar.ID[id]				 = params.id; }
 91 	if (params.disableOffline) { ZmSearchToolBar.DISABLE_OFFLINE[id] = params.disableOffline; }
 92 
 93 	if (params.index == null || params.index < 0 || params.index >= ZmSearchToolBar.MENU_ITEMS.length) {
 94 		ZmSearchToolBar.MENU_ITEMS.push(id);
 95 	} else {
 96 		ZmSearchToolBar.MENU_ITEMS.splice(params.index, 0, id);
 97 	}
 98 };

106  * @param {string}	id			ID of menu item

108 ZmSearchToolBar.prototype.removeMenuItem =
109 function(id) {
110 	
111 	var menu = this._searchMenu;
112 	if (menu) {
113 		var mi = menu.getItemById("_menuItemId", id);
114 		if (mi) {
115 			menu.removeChild(mi);
116 			mi.dispose();
117 		}
118 		this._cleanupSeparators(menu);
119 	}

122 // Remove unneeded separators
123 ZmSearchToolBar.prototype._cleanupSeparators =
124 function(menu) {
125 
126 	var button = this._button[ZmSearchToolBar.TYPES_BUTTON];
127 	menu = menu || (button && button.getMenu());
128 	if (!menu) { return; }
129 
130 	var items = menu.getItems();
131 	var toRemove = [];
132 	for (var i = 0; i < items.length; i++) {
133 		var mi = items[i];
134 		if (mi.isSeparator() && (i == 0 || i == items.length - 1 || items[i - 1].isSeparator())) {
135 			toRemove.push(mi);
136 		}
137 	}
138 	for (var i = 0; i < toRemove.length; i++) {
139 		var mi = toRemove[i];
140 		menu.removeChild(mi);
141 		mi.dispose();
142 	}
143 };

218 ZmSearchToolBar.prototype._handleKeyDown =
219 function(ev) {
220 	var key = DwtKeyEvent.getCharCode(ev);
221 	if (key == 3 || key == 13) {
222 		return this._handleEnterKeyPress(ev);
223 	}
224 	return true;
225 };

228 function(ev) {

231 					ev:				ev,
232 					zimletEvent:	"onKeyPressSearchField",
233 					origin:			this._origin
234 				});
235 	}
236 	return false;
237 };

270 ZmSearchToolBar.prototype._getAcLocation =
271 function() {
272 	var el = this._searchField.getInputElement();
273 	if (!el) { return {}; }
274 	
275 	var elLoc = Dwt.getLocation(el);
276 	var elSize = Dwt.getSize(el);
277 	var strWidth = AjxStringUtil.getWidth(el.value);
278 	if (AjxEnv.isWindows && (AjxEnv.isFirefox || AjxEnv.isSafari || AjxEnv.isChrome) ){
279 		// FF/Win: fudge factor since string is longer in INPUT than when measured in SPAN
280 		strWidth = strWidth * 1.2;
281 	}
282 	var x = elLoc.x + strWidth;
283 	var y = elLoc.y + elSize.y;
284 	DwtPoint.tmp.set(x, y);
285 	return DwtPoint.tmp;
286 };

292  * Adds a types button and support for a custom search menu item to a search toolbar
293  * 
294  * @param params
295  */
296 ZmMainSearchToolBar = function(params) {
297 
298 	if (arguments.length == 0) { return; }
299 
300 
301 	ZmSearchToolBar.apply(this, arguments);
302 
303 	// setup "include shared" menu item
304 	var miParams = {
305 		msgKey:			"searchShared",
306 		tooltipKey:		"searchShared",
307 		icon:			"Group",
308 		setting:		ZmSetting.SHARING_ENABLED,
309 		id:				ZmId.getMenuItemId(ZmId.SEARCH, ZmId.SEARCH_SHARED),
310 		disableOffline: true
311 	};

318 			icon:	"Globe",

337 ZmMainSearchToolBar.prototype._createHtml =
338 function() {
339     var isExternalAccount = appCtxt.isExternalAccount()
340 	this.getHtmlElement().innerHTML = AjxTemplate.expand(this.TEMPLATE, {id:this._htmlElId});
341 
342 	// add search input field
343 	var inputFieldId = this._htmlElId + "_inputField";
344 	var inputField = document.getElementById(inputFieldId);
345 	if (inputField) {
346 		this._searchField = new DwtInputField({parent:this, hint:ZmMsg.searchInput, inputId:ZmId.SEARCH_INPUTFIELD});
347 		var inputEl = this._searchField.getInputElement();
348 		inputEl.className = "search_input";
349 		this._searchField.reparentHtmlElement(inputFieldId);
350 		this._searchField._showHint();
351 		this._searchField.addListener(DwtEvent.ONFOCUS, this._onInputFocus.bind(this));
352 		this._searchField.addListener(DwtEvent.ONBLUR, this._onInputBlur.bind(this));
353         if(isExternalAccount) {
354             this._searchField.setEnabled(false);
355         }
356 	}
357 
358 	// add "search types" menu
359 	var searchMenuBtnId = this._htmlElId + "_searchMenuButton";
360 	var searchMenuBtn = document.getElementById(searchMenuBtnId);
361 	if (searchMenuBtn) {
362 		var firstItem = ZmSearchToolBar.MENU_ITEMS[0];
363 		var button = this._button[ZmSearchToolBar.TYPES_BUTTON] = ZmToolBar.addButton({
364 					parent:		this, 
365 					tdId:		"_searchMenuButton",
366 					buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_MENU),
367 					tooltip:	ZmMsg[ZmSearchToolBar.TT_MSG_KEY[firstItem]],
368 					icon:		ZmSearchToolBar.ICON[firstItem]
369 				});
370 		var menu = new AjxCallback(this, this._createSearchMenu);
371 		button.setMenu(menu, false, DwtMenuItem.RADIO_STYLE);
372 		button.reparentHtmlElement(searchMenuBtnId);
373         if(isExternalAccount) {
374             button.setEnabled(false);
375         }
376 	}
377 
378 	// add search button
379 	this._button[ZmSearchToolBar.SEARCH_BUTTON] = ZmToolBar.addButton({
380 				parent:		this, 
381 				tdId:		"_searchButton",
382 				buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_SEARCH),
383 				lbl:		"",
384 				icon:		"Search2",
385 				template: 	"dwt.Widgets#ZImageOnlyButton",
386 				className: 	"ZImageOnlyButton",
387 				tooltip:	ZmMsg.searchTooltip
388 			});
389 
390 	// add save search button if saved-searches enabled
391 	this._button[ZmSearchToolBar.SAVE_BUTTON] = ZmToolBar.addButton({
392 				parent:		this, 
393 				setting:	ZmSetting.SAVED_SEARCHES_ENABLED,
394 				tdId:		"_saveButton",
395 				buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_SAVE),
396 				lbl:		ZmMsg.save,
397 				icon:		"Save",
398 				type:		"toolbar",
399 				tooltip:	ZmMsg.saveSearchTooltip });
400     if(isExternalAccount) {
401         if(this._button[ZmSearchToolBar.SEARCH_BUTTON]) this._button[ZmSearchToolBar.SEARCH_BUTTON].setEnabled(false);
402         if(this._button[ZmSearchToolBar.SAVE_BUTTON]) this._button[ZmSearchToolBar.SAVE_BUTTON].setEnabled(false);
403     }
404 };

406 ZmMainSearchToolBar.prototype._createSearchMenu =
407 function() {
408 
409 	var menu = this._searchMenu = new DwtMenu({
410 				parent:		this._button[ZmSearchToolBar.TYPES_BUTTON],
411 				className:	"ActionMenu",
412 				id:			ZmId.getMenuId(ZmId.SEARCH)
413 			});
414 	var mi;
415 	if (this._customSearchMenuItems) {
416 		for (var i = 0; i < this._customSearchMenuItems.length; i++) {
417 			var csmi = this._customSearchMenuItems[i];
418 			this._createCustomSearchMenuItem(menu, csmi.icon, csmi.text, csmi.listener);
419 		}
420 	}
421 	var params = {parent:menu, enabled:true, radioGroupId:0};
422 	for (var i = 0; i < ZmSearchToolBar.MENU_ITEMS.length; i++) {
423 		var id = ZmSearchToolBar.MENU_ITEMS[i];
424 
425 		// add separator *before* "shared" menu item
426 		if (id == ZmId.SEARCH_SHARED) {
427 			if (ZmSearchToolBar.MENU_ITEMS.length <= 1) { continue; }
428 			mi = new DwtMenuItem({parent:menu, style:DwtMenuItem.SEPARATOR_STYLE});
429 		}
430 
431 		var setting = ZmSearchToolBar.SETTING[id];
432 		if (setting && !appCtxt.get(setting)) { continue; }
433 
434 		params.style = (id == ZmId.SEARCH_SHARED || id == ZmId.SEARCH_ALL_ACCOUNTS)
435 			? DwtMenuItem.CHECK_STYLE : DwtMenuItem.RADIO_STYLE;
436 		params.imageInfo = ZmSearchToolBar.ICON[id];
437 		params.text = ZmMsg[ZmSearchToolBar.MSG_KEY[id]];
438 		params.id = ZmSearchToolBar.ID[id];
439 		mi = DwtMenuItem.create(params);
440 		mi.setData(ZmSearchToolBar.MENUITEM_ID, id);
441 	}
442 	
443 	this._checkSharedMenuItem();
444 	appCtxt.getSettings().getSetting(ZmSetting.SEARCH_INCLUDES_SHARED).addChangeListener(this._checkSharedMenuItem.bind(this));
445 
446 	appCtxt.getSearchController()._addMenuListeners(menu);
447 	this._searchMenuCreated = true;
448 
449 	return menu;
450 };

452 ZmMainSearchToolBar.prototype.setOfflineState = function(offline) {
453 	var button   = this._button[ZmSearchToolBar.TYPES_BUTTON];
454 	var menu     = button && button.getMenu();
455 	var numItems = menu.getItemCount();
456 	for (var i = 0; i < numItems; i++) {
457 	    var item = menu.getItem(i);
458 		if (item) {
459 			var id   = item.getData(ZmSearchToolBar.MENUITEM_ID);
460 			if (id && ZmSearchToolBar.DISABLE_OFFLINE[id])  {
461 				item.setEnabled(!offline);
462 			}
463 		}
464 	}
465 }

477 ZmMainSearchToolBar.prototype.createCustomSearchBtn =
478 function(icon, text, listener, id) {
479 
480 	if (!this._customSearchListener) {
481 		this._customSearchListener = this._customSearchBtnListener.bind(this);
482 	}
483 
484 	// check if custom search should be a button by checking for the Id against the template
485 	var customSearchBtn = document.getElementById(this._htmlElId + "_customSearchButton");
486 	if (customSearchBtn) {
487 		var data = { icon:icon, text:text, listener:listener };
488 		var button = this._button[ZmSearchToolBar.CUSTOM_BUTTON]
489 		if (!button) {
490 			button = this._button[ZmSearchToolBar.CUSTOM_BUTTON] = ZmToolBar.addButton({
491 						parent:		this,
492 						tdId:		"_customSearchButton",
493 						buttonId:	ZmId.getButtonId(ZmId.SEARCH, ZmId.SEARCH_CUSTOM),
494 						lbl:		text,
495 						icon:		icon
496 					});
497 			button.setData(ZmMainSearchToolBar.CUSTOM_ITEM_ID, data);
498 			button.addSelectionListener(this._customSearchListener);
499 
500 			// show the separator now that we've added a custom search button
501 			var sep = document.getElementById(this._htmlElId + "_customSearchButtonSep");
502 			if (sep) {
503 				Dwt.setVisible(sep, true);
504 			}
505 		} else {
506 			var menu = button && button.getMenu();
507 			var item;
508 			var params = {
509 				parent:			menu,
510 				enabled:		true,
511 				style:			DwtMenuItem.RADIO_STYLE,
512 				radioGroupId:	0,
513 				id:				id
514 			};
515 			if (!menu) {
516 				var btnData = button.getData(ZmMainSearchToolBar.CUSTOM_ITEM_ID);
517 				menu = new DwtMenu({
518 							parent:		button,
519 							className:	"ActionMenu",
520 							id:			ZmId.getMenuId(ZmId.SEARCH, ZmId.SEARCH_CUSTOM)
521 						});
522 				button.setMenu(menu, false, DwtMenuItem.RADIO_STYLE);
523 				params.imageInfo = btnData.icon;
524 				params.text = btnData.text;
525 				item = DwtMenuItem.create(params);
526 				item.setData(ZmMainSearchToolBar.CUSTOM_ITEM_ID, btnData);
527 				item.setData(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_CUSTOM);
528 				item.setChecked(true, true);
529 				item.addSelectionListener(this._customSearchListener);
530 			}
531 			params.imageInfo = icon;
532 			params.text = text;
533 			item = DwtMenuItem.create(params);
534 			item.setData(ZmMainSearchToolBar.CUSTOM_ITEM_ID, data);
535 			item.addSelectionListener(this._customSearchListener);
536 		}
537 	} else {
538 		if (this._searchMenuCreated) {
539 			var menu = this._button[ZmSearchToolBar.TYPES_BUTTON].getMenu();
540 			this._createCustomSearchMenuItem(menu, icon, text, listener, id);
541 		} else {
542 			if (!this._customSearchMenuItems) {
543 				this._customSearchMenuItems = [];
544 			}
545 			this._customSearchMenuItems.push({icon:icon, text:text, listener:listener, id:id});
546 		}
547 	}
548 };

550 ZmMainSearchToolBar.prototype._createCustomSearchMenuItem =
551 function(menu, icon, text, listener, id) {
552 	var mi = menu.getItem(0);
553 	var params = {
554 		parent: menu,
555 		imageInfo: icon,
556 		text: text,
557 		enabled: true,
558 		style: DwtMenuItem.RADIO_STYLE,
559 		radioGroupId: 0,
560 		index: 0,
561 		id: id
562 	};
563 	mi = DwtMenuItem.create(params);
564 	var data = { icon:icon, text:text, listener:listener };
565 	mi.setData(ZmMainSearchToolBar.CUSTOM_ITEM_ID, data);
566 	mi.setData(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_CUSTOM);
567 	mi.addSelectionListener(this._customSearchListener);
568 
569 	// only add separator if this is the first custom search menu item
570 	if (!(mi && mi.getData(ZmMainSearchToolBar.CUSTOM_ITEM_ID))) {
571 		mi = new DwtMenuItem({parent:menu, style:DwtMenuItem.SEPARATOR_STYLE, index:1});
572 	}
573 };

575 ZmMainSearchToolBar.prototype._customSearchBtnListener = 
576 function(ev) {
577 	var item = ev.item;
578 	if (!item) { return; }
579 	var data = item.getData(ZmMainSearchToolBar.CUSTOM_ITEM_ID);
580 	if (!data) { return; }
581 	if (this._customSearchBtn) {
582 		if (item.isDwtMenuItem) {
583 			if (ev.detail != DwtMenuItem.CHECKED) { return; }
584 			this._customSearchBtn.setToolTipContent(data[1]);
585 			this._customSearchBtn.setData(ZmMainSearchToolBar.CUSTOM_ITEM_ID, data);
586 		}
587 		data.listener.run(ev); // call original listener
588 	} else {
589 		var button = this._button[ZmSearchToolBar.TYPES_BUTTON];
590 		button.setToolTipContent(data.text);
591 
592 		var menu = item.parent;
593 		var shareMenuItem = menu ? menu.getItemById(ZmSearchToolBar.MENUITEM_ID, ZmId.SEARCH_SHARED) : null;
594 		if (shareMenuItem) {
595 			shareMenuItem.setChecked(false, true);
596 			shareMenuItem.setEnabled(false);
597 		}
598 
599 		button.setImage(data.icon);
600 		button.setText(data.text);
601 	}
602 };

614 	if (focusObj != this._button[ZmSearchToolBar.TYPES_BUTTON] && focusObj != this._button[ZmSearchToolBar.SEARCH_BUTTON]) {
615 		this._setInputExpanded(false);
616 	}

619 ZmMainSearchToolBar.prototype._setInputExpanded =
620 function(expanded) {
621 	var input = this._searchField.getInputElement();
622 	var cls = expanded ? "search_input-expanded" : "search_input";
623 
624 	if (AjxEnv.isIE9) {
625 		// bug 83493: IE9 gets the layout wrong on the first try
626 		setTimeout(function() {
627 			input.className = cls;
628 		}, 0);
629 	} else {
630 		input.className = cls;
631 	}
632 };
