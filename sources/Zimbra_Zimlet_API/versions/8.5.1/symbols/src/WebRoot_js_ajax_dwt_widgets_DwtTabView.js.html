 23 /**
 24  * Creates a tab view.
 25  * @constructor
 26  * @class
 27  * This class represents a tabbed view. {@link DwtTabView} manages the z-index of the contained tabs. 
 28  * 
 29  * @param {hash}			params			a hash of parameters
 30  * @param {DwtComposite}	parent			the parent widget
 31  * @param {string}			className		the CSS class
 32  * @param {constant}		posStyle		the positioning style (see {@link DwtControl})
 33  * @param {string}			id				an explicit ID to use for the control's HTML element
 34  * 
 35  * @author Greg Solovyev
 36  * 
 37  * @extends DwtComposite
 38  */
 39 DwtTabView = function(params) {
 40 	if (arguments.length == 0) return;
 41 	params = Dwt.getParams(arguments, DwtListView.PARAMS);
 42 	params.className = params.className || "ZTabView";
 43 	params.posStyle = params.posStyle || DwtControl.ABSOLUTE_STYLE;
 44 	DwtComposite.call(this, params);
 45 
 46 	this._stateChangeEv = new DwtEvent(true);
 47 	this._stateChangeEv.item = this;
 48 
 49 	this._tabs = [];
 50 	this._tabIx = 1;
 51 	this._createHtml();
 52 
 53 	var tabGroupId = [this.toString(), this._htmlElId].join("-");
 54 	this._tabGroup = new DwtTabGroup(tabGroupId);
 55 	this._tabGroup.addMember(this._tabBar);
 56 };
 57

104 /**
105  * Adds a tab.
106  * 
107  * @param {string}						title					the text for the tab button
108  * @param {DwtTabViewPage|AjxCallback}	tabViewOrCallback		an instance of the tab view page or callback that returns an instance of {@link DwtTabViewPage}
109  *
110  * @return {string}		key for the added tab	This key can be used to retrieve the tab using {@link #getTab}
111  * 
112  * @see		#getTab
113  */
114 DwtTabView.prototype.addTab =
115 function (title, tabViewOrCallback, buttonId, index) {
116 	var tabKey = this._tabIx++;	
117 
118 	// create tab entry
119 	this._tabs[tabKey] = {
120 		title: title,
121 		button: this._tabBar.addButton(tabKey, title, buttonId, index)
122 	};
123 
124 	// add the page
125 	this.setTabView(tabKey, tabViewOrCallback);
126 
127 	// show the first tab
128 	if (tabKey==1) {
129 		if (tabViewOrCallback instanceof AjxCallback) {
130 			tabViewOrCallback = tabViewOrCallback.run(tabKey);
131 		}
132 		if(tabViewOrCallback) {
133 			tabViewOrCallback.showMe();
134 		}
135 		this._currentTabKey = tabKey;		
136 		this.switchToTab(tabKey);
137 	}
138 	// hide all the other tabs
139 	else if (tabViewOrCallback && !(tabViewOrCallback instanceof AjxCallback)) {
140 		tabViewOrCallback.hideMe();
141 		Dwt.setVisible(tabViewOrCallback.getHtmlElement(), false);
142 	}
143 
144 	this._tabBar.addSelectionListener(tabKey, new AjxListener(this, DwtTabView.prototype._tabButtonListener));
145 
146 	return tabKey;
147 };

232 /**
233  * Sets the tab view.
234  * 
235  * @param {string}						tabKey		the tab key
236  * @param {DwtTabViewPage|AjxCallback}	tabView		 an instance of the tab view page or callback that returns an instance of {@link DwtTabViewPage}
237  */
238 DwtTabView.prototype.setTabView =
239 function(tabKey, tabView) {
240 	var tab = this.getTab(tabKey);
241 	tab.view = tabView;
242 	if (tabView && !(tabView instanceof AjxCallback)) {
243 		this._pageEl.appendChild(tabView.getHtmlElement());
244 		tabView._tabKey = tabKey;
245 		if (tabKey == this._currentTabKey) {
246 			var tabGroup = tabView.getTabGroupMember();
247 			this._tabGroup.replaceMember(tab.tabGroup, tabGroup);
248 			tab.tabGroup = tabGroup;
249 		}
250 	}
251 };

253 /**
254  * Gets the tab view.
255  * 
256  * @param {string}		tabKey		the tab key
257  *
258  * @return {DwtTabViewPage}		the tab view page
259  */
260 DwtTabView.prototype.getTabView =
261 function(tabKey) {
262 	var tab = this.getTab(tabKey);
263 	var tabView = tab && tab.view;
264 	if (tabView instanceof AjxCallback) {
265 		var callback = tabView;
266 		tabView = callback.run(tabKey);
267 		this.setTabView(tabKey, tabView);
268 		var size = this._getTabSize();
269 		tabView.setSize(size.x, size.y);
270 	}
271 	return tabView;
272 };

274 /**
275  * Switches to the tab view.
276  * 
277  * @param {string}		tabKey		the tab key
278  */
279 DwtTabView.prototype.switchToTab = 
280 function(tabKey) {
281 	var ntab = this.getTab(tabKey);
282 	if(ntab) {
283 		// remove old tab from tab-group
284 		var otab = this.getTab(this._currentTabKey);
285 		if (otab) {
286 			this._tabGroup.removeMember(otab.tabGroup);
287 		}
288 		// switch tab
289 		this._showTab(tabKey);
290 		this._tabBar.openTab(tabKey);
291 		// add new tab to tab-group
292 		if (!ntab.tabGroup && ntab.view) {
293 			ntab.tabGroup = ntab.view.getTabGroupMember();
294 		}
295 		this._tabGroup.addMember(ntab.tabGroup);
296 		// notify change
297 		if (this._eventMgr.isListenerRegistered(DwtEvent.STATE_CHANGE)) {
298 			this._eventMgr.notifyListeners(DwtEvent.STATE_CHANGE, this._stateChangeEv);
299 		}
300 	}
301 };

319 DwtTabView.prototype.resetKeyBindings =
320 function() {
321 	var kbm = this.shell.getKeyboardMgr();
322 	if (kbm.isEnabled()) {
323 		var kmm = kbm.__keyMapMgr;
324 		if (!kmm) { return; }
325 		var num = this.getNumTabs();
326 		var seqs = kmm.getKeySequences(DwtKeyMap.MAP_TAB_VIEW, "GoToTab");
327 		for (var k = 0; k < seqs.length; k++) {
328 			var ks = seqs[k];
329 			for (var i = 1; i <= num; i++) {
330 				var keycode = 48 + i;
331 				var newKs = ks.replace(/NNN/, keycode);
332 				kmm.setMapping(DwtKeyMap.MAP_TAB_VIEW, newKs, "GoToTab" + i);
333 			}
334 		}
335 		kmm.reloadMap(DwtKeyMap.MAP_TAB_VIEW);
336 	}
337 };

339 DwtTabView.prototype.handleKeyAction =
340 function(actionCode) {
341 	DBG.println(AjxDebug.DBG3, "DwtTabView.handleKeyAction");
342 
343 	switch (actionCode) {
344 
345 		case DwtKeyMap.NEXT_TAB:
346 			var curTab = this.getCurrentTab();
347 			if (curTab < this.getNumTabs()) {
348 				this.switchToTab(curTab + 1);
349 			}
350 			break;
351 			
352 		case DwtKeyMap.PREV_TAB:
353 			var curTab = this.getCurrentTab();
354 			if (curTab > 1) {
355 				this.switchToTab(curTab - 1);
356 			}
357 			break;
358 		
359 		default:
360 			// Handle action code like "GoToTab3"
361 			var m = actionCode.match(DwtKeyMap.GOTO_TAB_RE);
362 			if (m && m.length) {
363 				var idx = m[1];
364 				if ((idx <= this.getNumTabs()) && (idx != this.getCurrentTab())) {
365 					this.switchToTab(idx);
366 				}
367 			} else {
368 				return false;
369 			}
370 	}
371 	return true;
372 };

377 DwtTabView.prototype._resetTabSizes =
378 function (width, height) {
379 	if (this._tabs && this._tabs.length) {
380 		for (var curTabKey in this._tabs) {
381 			var tabView = this._tabs[curTabKey].view;
382 			if (tabView && !(tabView instanceof AjxCallback)) {
383 				var contentHeight;
384 				contentHeight = contentHeight || height - Dwt.getSize(this._tabBarEl).y;
385 				tabView.resetSize(width, contentHeight);
386 			}	
387 		}
388 	}		
389 };

391 DwtTabView.prototype._getTabSize =
392 function() {
393 	var size = this.getSize();
394 	var width = size.x || this.getHtmlElement().clientWidth;
395 	var height = size.y || this.getHtmlElement().clientHeight;
396 	var tabBarSize = this._tabBar.getSize();
397 	var tabBarHeight = tabBarSize.y || this._tabBar.getHtmlElement().clientHeight;
398 
399 	return new DwtPoint(width, (height - tabBarHeight));
400 };

417 DwtTabView.prototype._showTab = 
418 function(tabKey) {
419 	if (this._tabs && this._tabs[tabKey]) {
420 		this._currentTabKey = tabKey;
421 		this._hideAllTabs();						// hide all the tabs
422 		var tabView = this.getTabView(tabKey);		// make this tab visible
423 		if (tabView) {
424 			tabView.setVisible(true);
425 			tabView.showMe();
426 		}
427 	}
428 };

430 DwtTabView.prototype._hideAllTabs = 
431 function() {
432 	if (this._tabs && this._tabs.length) {
433 		for (var curTabKey in this._tabs) {
434 			var tabView = this._tabs[curTabKey].view;
435 			if (tabView && !(tabView instanceof AjxCallback)) {
436 				tabView.hideMe();
437 				//this._tabs[curTabKey]["view"].setZIndex(DwtTabView.Z_HIDDEN_TAB);
438 				Dwt.setVisible(tabView.getHtmlElement(), false);
439 			}	
440 		}
441 	}
442 };

454 /**
455  * Creates a tab view page.
456  * @constructor
457  * @class
458  * DwtTabViewPage abstract class for a page in a tabbed view.
459  * Tab pages are responsible for creating their own HTML and populating/collecting
460  * data to/from any form fields that they display.
461  * 
462  * @param {DwtComposite}	parent			the parent widget
463  * @param {string}			className		the CSS class
464  * @param {constant}		posStyle		the positioning style (see {@link DwtControl})
465  * 
466  * @extends DwtPropertyPage
467  */
468 DwtTabViewPage = function(parent, className, posStyle, id) {
469 	if (arguments.length == 0) return;
470 	var params = Dwt.getParams(arguments, DwtTabViewPage.PARAMS);
471 	params.className = params.className || "ZTabPage";
472 	params.posStyle = params.posStyle || DwtControl.ABSOLUTE_STYLE;
473 	params.id = id || null;
474 	this._rendered = true; // by default UI creation is not lazy
475 
476 	DwtPropertyPage.call(this, params);
477 
478 	this._createHtml();
479 	this.getHtmlElement().style.overflowY = "auto";
480 	this.getHtmlElement().style.overflowX = "visible";
481 	if (params.contentTemplate) {
482 		this.getContentHtmlElement().innerHTML = AjxTemplate.expand(params.contentTemplate, this._htmlElId);
483 	}
484 };

509 /**
510  * Shows the tab view page.
511  * 
512  */
513 DwtTabViewPage.prototype.showMe =
514 function() {
515 	this.setZIndex(DwtTabView.Z_ACTIVE_TAB);
516 	if (this.parent.getHtmlElement().offsetHeight > 80) { 						// if parent visible, use offsetHeight
517 		this._contentEl.style.height=this.parent.getHtmlElement().offsetHeight-80;
518 	} else {
519 		var parentHeight = parseInt(this.parent.getHtmlElement().style.height);	// if parent not visible, resize page to fit parent
520 		var units = AjxStringUtil.getUnitsFromSizeString(this.parent.getHtmlElement().style.height);
521 		if (parentHeight > 80) {
522 			this._contentEl.style.height = (Number(parentHeight-80).toString() + units);
523 		}
524 	}
525 
526 	this._contentEl.style.width = this.parent.getHtmlElement().style.width;	// resize page to fit parent
527 };

685 	if (this._btnImage != null) {
686 		b.setImage(this._btnImage);
687 	}
688 
689 	if (tabTitle != null) {
690 		b.setText(tabTitle);
691 	}
692 
693 	b.setEnabled(true);
694 	b.setData("tabKey", tabKey);
695 
696 	if (parseInt(tabKey) == 1) {
697 		this.openTab(tabKey, true);
698 	}
699 
700 	// make sure that new button is selected properly
701 	var sindex = this.__getButtonIndex(this._currentTabKey);
702 	if (sindex != -1) {
703 		var nindex = this.__getButtonIndex(tabKey);
704 		if (nindex == sindex + 1) {
705 			Dwt.addClass(b.getHtmlElement(), DwtTabBar.SELECTED_NEXT);
706 		}
707 	}
708 
709 	return b;
710 };

726 /**
727  * Opens the tab.
728  *
729  * @param {string}		tabKey			the id used to create tab button in {@link DwtTabBar.addButton}
730  * @param {boolean}		skipNotify		if <code>true</code>, do not notify listeners
731  */
732 DwtTabBar.prototype.openTab = 
733 function(tabK, skipNotify) {
734 	this._currentTabKey = tabK;
735 	var cnt = this._buttons.length;
736 
737 	for (var ix = 0; ix < cnt; ix ++) {
738 		if (ix==tabK) { continue; }
739 
740 		var button = this._buttons[ix];
741 		if (button) {
742 			this.__markPrevNext(ix, false);
743 			button.setClosed();
744 		}
745 	}
746 
747 	var button = this._buttons[tabK];
748 	if (button) {
749 		button.setOpen();
750 		this.__markPrevNext(tabK, true);
751 	}
752 
753 	if (!skipNotify && this._eventMgr.isListenerRegistered(DwtEvent.STATE_CHANGE)) {
754 		this._eventMgr.notifyListeners(DwtEvent.STATE_CHANGE, this._stateChangeEv);
755 	}
756 };

758 /**
759  * Greg Solovyev 1/4/2005 
760  * changed ev.target.offsetParent.offsetParent to
761  * lookup for the table up the elements stack, because the mouse down event may come from the img elements 
762  * as well as from the td elements.
763  * 
764  * @private
765  */
766 DwtTabBar._setActiveTab =
767 function(ev) {
768 	var tabK;
769 	if (ev && ev.item) {
770 		tabK=ev.item.getData("tabKey");
771 	} else if (ev && ev.target) {
772 		var elem = ev.target;
773 		while (elem.tagName != "TABLE" && elem.offsetParent )
774 			elem = elem.offsetParent;
775 
776 		tabK = elem.getAttribute("tabKey");
777 		if (tabK == null)
778 			return false;
779 	} else {
780 		return false;
781 	}
782 	this.openTab(tabK);
783 };
