214 /**
215  * Checks for the creation of a briefcase or a mount point to one, or of an item
216  *
217  * @param {Hash}	creates		a hash of create notifications
218  * 
219  * @private
220  */
221 ZmBriefcaseApp.prototype.createNotify =
222 function(creates, force) {
223 
224 	if (!creates["folder"] && !creates["doc"] && !creates["link"]) { return; }
225 	if (!force && !this._noDefer && this._deferNotifications("create", creates)) { return; }
226 
227 	for (var name in creates) {
228 		var clist = creates[name];
229 		for (var i = 0; (clist != null) && i < clist.length; i++) {
230 			var create = clist[i];
231 			if (appCtxt.cacheGet(create.id)) { continue; }
232 
233 			if (name == "folder") {
234 				this._handleCreateFolder(create, ZmOrganizer.BRIEFCASE);
235 			} else if (name == "link") {
236 				this._handleCreateLink(create, ZmOrganizer.BRIEFCASE);
237 			} else if (name == "doc") {
238 				var bc = AjxDispatcher.run("GetBriefcaseController");
239                 bc.handleCreateNotify(create);
240 			}
241 		}
242 	}
243 };
244 
245 ZmBriefcaseApp.prototype.modifyNotify =
246 function(modifies, force) {
247     if (!modifies["doc"]) { return; }
248 	var bc = AjxDispatcher.run("GetBriefcaseController");
249     bc.handleModifyNotify(modifies);
250 };

252 ZmBriefcaseApp.prototype.handleOp =
253 function(op) {
254 
255 	switch (op) {
256 		case ZmOperation.NEW_FILE: {
257 			var loadCallback = new AjxCallback(this, this._handleNewItem);
258 			AjxDispatcher.require(["BriefcaseCore", "Briefcase"], false, loadCallback, null, true);
259 			break;
260 		}
261 		case ZmOperation.NEW_BRIEFCASE: {
262 			var loadCallback = new AjxCallback(this, this._handleLoadNewBriefcase);
263 			AjxDispatcher.require(["BriefcaseCore", "Briefcase"], false, loadCallback, null, true);
264 			break;
265 		}
266 
267 		case ZmOperation.NEW_DOC: {
268 			var newDocCallback = new AjxCallback(this, this.newDoc, [ZmMimeTable.APP_ZIMBRA_DOC]);
269 			AjxDispatcher.require(["BriefcaseCore", "Briefcase"], true, newDocCallback, null);
270 			break;
271 		}
272 	}
273 };

275 /**
276  * Creates a new document.
277  * 
278  * @param	{String}	contentType		the content type
279  * @param	{String}	new				the document name
280  * @param	{String}	winName			the name of the popup doc window
281  */
282 ZmBriefcaseApp.prototype.newDoc =
283 function(contentType, name, winName) {
284 	var overviewController = appCtxt.getOverviewController();
285 	var treeController = overviewController.getTreeController(ZmOrganizer.BRIEFCASE);
286 	var folderId = ZmOrganizer.ID_BRIEFCASE;
287 	if (treeController) {
288 		var treeView = treeController.getTreeView(this.getOverviewId());
289 		var briefcase = treeView ? treeView.getSelected() : null;
290 		folderId = briefcase ? briefcase.id : ZmOrganizer.ID_BRIEFCASE;
291 	}
292 
293     if (AjxDispatcher.run("GetBriefcaseController").chkFolderPermission(folderId)) {
294         if (contentType == ZmMimeTable.APP_ZIMBRA_DOC) {
295             var win = appCtxt.getNewWindow(false, null, null, winName);
296             win.command = "documentEdit";
297             win.params = { name: name, folderId: folderId };
298         }
299     }
300 };

367 /**
368  * Performs a search.
369  * 
370  * @param {Hash}	params			a hash of parameters
371  * @param {String}	params.folderId			the ID of briefcase folder to search in
372  * @param {String}	[params.query]				the query to send (overrides folderId)
373  * @param {AjxCallback}	[params.callback]			the callback
374  * @param {String}	[params.accountName]		the account name
375  * @param {Boolean}	[params.noRender]			if <code>true</code>, do not display results
376  */
377 ZmBriefcaseApp.prototype.search =
378 function(params) {
379 
380 	params = params || {};
381     var folderId = params.folderId || (appCtxt.isExternalAccount() ? this.getDefaultFolderId() : ZmOrganizer.ID_BRIEFCASE);
382 	var folder = appCtxt.getById(folderId);
383 
384 	var searchParams = {
385 		query:			params.query || folder.createQuery(),
386 		types:			[ZmItem.BRIEFCASE_ITEM],
387 		limit:			this.getLimit(),
388 		searchFor:		ZmId.ITEM_BRIEFCASE,
389 		callback:		params.callback,
390 		accountName:	params.accountName,
391 		noRender:		params.noRender
392 	};
393 	var sc = appCtxt.getSearchController();
394 	sc.searchAllAccounts = false;
395 	sc.search(searchParams);
396 };

477 
478 ZmBriefcaseApp.prototype.initExternalDndUpload = function(files, node, isInline, selectionCallback, folderId) {
479 	var name = "";
480 
481 	if (!AjxEnv.supportsHTML5File) {
482 		// IE, FF 3.5 and lower - use the File browser
483 		if (selectionCallback) {
484 			selectionCallback.run();
485 		}
486 		return;
487 	}
488 
489 	if (!files) {
490 		files = node.files;
491 	}
492 
493 	var size = 0;
494 	if (files) {
495 		var file;
496 		var docFiles = [];
497 		var errors   = {};
498 		var aCtxt    = ZmAppCtxt.handleWindowOpener();
499 		var maxSize  = aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT);
500 		var briefcaseController = AjxDispatcher.run("GetBriefcaseController");
501 
502 		if (!folderId) {
503 			if (briefcaseController) {
504 				folderId = briefcaseController.getFolderId();
505 			}
506 			if(!folderId || folderId == ZmOrganizer.ID_TRASH) {
507 				folderId = ZmOrganizer.ID_BRIEFCASE;
508 			}
509 		}
510 
511 		if(this.chkFolderPermission(folderId)){
512 			var cFolder = appCtxt.getById(folderId);
513 			var uploadManager = appCtxt.getZmUploadManager();
514 
515 			var errors = [];
516 			for (var i = 0; i < files.length; i++){
517 				var newError = uploadManager.getErrors(files[i], maxSize);
518 				if (newError) {
519 					errors.push(newError);
520 				}
521 			}
522 			if (errors.length > 0) {
523 				var errorMsg = uploadManager.createUploadErrorMsg(errors, maxSize, "<br>");
524 				var msgDlg = appCtxt.getMsgDialog();
525 				msgDlg.setMessage(errorMsg, DwtMessageDialog.WARNING_STYLE);
526 				msgDlg.popup();
527 			} else {
528 				var params = {
529 					attachment:              false,
530 					uploadFolder:            cFolder,
531 					files:                   files,
532 					notes:                   "",
533 					allResponses:            null,
534 					start:                   0,
535 					curView:                 null,

563 /**
564  * uploadSaveDocs performs SaveDocument calls, creating a document with an associated uploadId.  If the file
565  * already exists, conflict resolution is performed.
566  *
567  * @param	{object}	params		params to customize the upload flow:
568  *      uploadFolder                Folder to save associated document into
569  *      files:                      raw File object from the external HTML5 drag and drop
570  *      notes:                      Notes associated with each of the files being added
571  *      allResponses:               All the server responses.  Contains the uploadId (guid) for a file
572  *      errorCallback:              Run upon an error
573  *      conflictAction			    If specified, the action used to resolve a file conflict
574  *      preResolveConflictCallback: Standard processing (SaveDocument), Run prior to conflict resolution
575  *      completeDocSaveCallback:    Standard processing (SaveDocument), Run when all documents have been saved
576  *
577  */
578 ZmBriefcaseApp.prototype.uploadSaveDocs = function(allResponses, params, status, guids) {
579 	if (status != AjxPost.SC_OK) {
580 		appCtxt.getAppController().popupUploadErrorDialog(ZmItem.BRIEFCASE, status);
581 	} else {
582 		var docFiles;
583 		if (allResponses) {
584 			// External DnD files
585 		    docFiles = [];
586 			var files    = params.files;
587 			if (allResponses.length === files.length) {
588 				for (var i = 0; i < files.length; i++){
589 					var file = files[i];
590 					var response = allResponses[i];
591 					var aid = (response && response.aid);
592 					docFiles.push(
593 						{name:     file.name,
594 						 fullname: file.name,
595 						 notes:    params.notes,
596 						 guid:     aid});
597 				}
598 				params.docFiles = docFiles;
599 			}
600 		} else {
601 			// AjxPost callback, providing the guids separately
602 			docFiles = params.docFiles;
603 			if (guids) {
604 				guids = guids.split(",");
605 				for (var i = 0; i < docFiles.length; i++) {
606 					DBG.println("guids[" + i + "]: " + guids[i] + ", files[" + i + "]: " + docFiles[i]);
607 					docFiles[i].guid = guids[i];
608 				}
609 			}
610 		}
611 		if (params.uploadFolder) {
612 			this._uploadSaveDocs2(params);
613 		} else {
614 			this._completeUpload(params);
615 		}
616 	}
617 };

625 ZmBriefcaseApp.prototype._uploadSaveDocs2 = function(params) {
626 
627 	// create document wrappers
628 	var request = [];
629 	var foundOne = false;
630 	var docFiles = params.docFiles;
631 	for (var i = 0; i < docFiles.length; i++) {
632 		var file = docFiles[i];
633 		if (file.done) {
634 			continue;
635 		}
636 		foundOne = true;
637 
638 		var SaveDocumentRequest = { _jsns: "urn:zimbraMail", requestId: i, doc: {}}
639 		var doc = SaveDocumentRequest.doc;
640 		if (file.id) {
641 			doc.id = file.id;
642 			doc.ver = file.version;
643 		} else {
644 			doc.l = params.uploadFolder.id;
645 		}
646 		if (file.notes) {
647 			doc.desc = file.notes;
648 		}
649 		doc.upload = {
650 			id: file.guid
651 		}
652 		request.push(SaveDocumentRequest);
653 	}
654 
655 	if (foundOne) {
656 		var json = {
657 			BatchRequest: {
658 				_jsns: "urn:zimbra",
659 				onerror: "continue",
660 				SaveDocumentRequest: ( (request.length == 1) ? request[0] : request )
661 			}
662 		};
663 		var callback = this._uploadSaveDocsResponse.bind(this, params);
664 		var saveDocParams = {
665 			jsonObj:  json,
666 			asyncMode:true,
667 			callback: callback
668 		};
669 		var appController = appCtxt.getAppController();
670 		appController.sendRequest(saveDocParams);
671 	}
672 	else {
673 		// This calls the callback of the client - e.g. ZmHtmlEditor.prototype._imageUploaded since
674 		// _uploadSaveDocsResponse is not called in this case, we still need the client callback since the
675 		// user chose the "old" version of the image
676 		this._completeUpload(params);
677 	}
678 };

680 ZmBriefcaseApp.prototype._uploadSaveDocsResponse = function(params, response) {
681 	var resp = response && response._data && response._data.BatchResponse;
682 	var docFiles = params.docFiles;
683 
684 	// mark successful uploads
685 	if (resp && resp.SaveDocumentResponse) {
686 		for (var i = 0; i < resp.SaveDocumentResponse.length; i++) {
687 			var saveDocResp = resp.SaveDocumentResponse[i];
688 			docFiles[saveDocResp.requestId].done    = true;
689 			docFiles[saveDocResp.requestId].name    = saveDocResp.doc[0].name;
690 			docFiles[saveDocResp.requestId].id      = saveDocResp.doc[0].id;
691 			docFiles[saveDocResp.requestId].ver     = saveDocResp.doc[0].ver;
692 			docFiles[saveDocResp.requestId].version = saveDocResp.doc[0].ver;
693 		}
694 	}
695 
696 	// check for conflicts
697 	var mailboxQuotaExceeded = false;
698 	var uploadRejected = false;
699 	var isItemLocked = false;
700 	var code = 0;
701 	var conflicts = [];
702 	if (resp && resp.Fault) {
703 		var errors = [];
704 		for (var i = 0; i < resp.Fault.length; i++) {
705 			var fault = resp.Fault[i];
706 			var error = fault.Detail.Error;
707 			code = error.Code;
708 			var attrs = error.a;
709 			isItemLocked = (code == ZmCsfeException.LOCKED);
710 			if (code == ZmCsfeException.MAIL_ALREADY_EXISTS ||
711 				code == ZmCsfeException.MODIFY_CONFLICT) {
712 				var file = docFiles[fault.requestId];
713 				for (var p in attrs) {
714 					var attr = attrs[p];
715 					switch (attr.n) {
716 						case "itemId" : { file.id      = attr._content; break }
717 						case "id":      { file.id      = attr._content; break; }
718 						case "ver":     { file.version = attr._content; break; }
719 						case "name":    { file.name    = attr._content; break; }
720 					}
721 				}
722 				file.version = file.version || 1;
723 				conflicts.push(file);
724 			}else {
725 				DBG.println("Unknown error occurred: " + code);
726 				if (code == ZmCsfeException.MAIL_QUOTA_EXCEEDED) {
727 					mailboxQuotaExceeded = true;
728 				}  else if (code === ZmCsfeException.UPLOAD_REJECTED) {
729 					uploadRejected = true;
730 				}
731 
732 				errors[fault.requestId] = fault;
733 			}
734 		}
735 	}
736 
737 	// dismiss dialog/enable the upload button
738 	if (params.preResolveConflictCallback) {
739 		params.preResolveConflictCallback.run();
740 	}
741 
742 	// TODO: What to do about other errors?
743 	// TODO: This should handle reporting several errors at once
744 	if (mailboxQuotaExceeded) {
745 		this._popupErrorDialog(ZmMsg.errorQuotaExceeded);
746 		return;
747 	}
748 	else if (isItemLocked) {
749 		this._popupErrorDialog(ZmMsg.errorItemLocked);
750 		return;
751 	} else if (uploadRejected) {
752 		// Yes, its bogus, but strings are locked.  This allows us to get somewhat close to indicating what happened.
753 		// TODO: Add an appropriate error string
754 		var rejectedMsg = AjxMessageFormat.format(ZmMsg.uploadError, [ ZmMsg.dlReject ] );
755 		this._popupErrorDialog(rejectedMsg);
756 		return;
757 	}
758 	else if (code == ZmCsfeException.SVC_PERM_DENIED) {
759 		this._popupErrorDialog(ZmMsg.errorPermissionDenied);
760 		if (params.errorCallback) {
761 			params.errorCallback.run();
762 		}
763 		return;
764 	}
765 
766 	// resolve conflicts
767 	var conflictCount = conflicts.length;
768 
769 	var action = params.conflictAction || ZmBriefcaseApp.ACTION_KEEP_MINE;
770 	if (conflictCount > 0 && action == ZmBriefcaseApp.ACTION_ASK) {
771 		var dialog = appCtxt.getUploadConflictDialog();
772 		dialog.popup(params.uploadFolder, conflicts, this._uploadSaveDocs2.bind(this, params));
773 	} else if (conflictCount > 0 && action == ZmBriefcaseApp.ACTION_KEEP_MINE) {
774 		if (params.conflictAction) {
775 			this._shieldSaveDocs(params);
776 		} else {
777 			this._uploadSaveDocs2(params);
778 		}
779 	} else {
780 		this._completeUpload(params);
781 	}
782 };
