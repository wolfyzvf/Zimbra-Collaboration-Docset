 27 /**
 28  * Creates an account object containing meta info about the account.
 29  * @class
 30  * This class represents an account. This object is created primarily if a user has added sub-accounts
 31  * to manage (i.e. a family mailbox).
 32  *
 33  * @author Parag Shah
 34  *
 35  * @param {String}		id			the unique ID for this account
 36  * @param {String}		name		the email address
 37  * @param {Boolean}		visible		if <code>true</code>, make this account available in the overview (i.e. child accounts)
 38  *
 39  * @extends	ZmAccount
 40  */
 41 ZmZimbraAccount = function(id, name, visible) {
 42 
 43 	ZmAccount.call(this, null, id, name);
 44 
 45 	this.visible = (visible !== false);
 46 	/**
 47 	 * The account settings.
 48 	 * @type	ZmSettings
 49 	 */
 50 	this.settings = null;
 51 	this.trees = {};
 52 	this.loaded = false;
 53 	/**
 54 	 * The account Access Control List.
 55 	 * @type	ZmAccessControlList
 56 	 */
 57 	this.acl = new ZmAccessControlList();
 58 	this.metaData = new ZmMetaData(this);
 59 };

118  */

120 function() {
121 	var identity = this.getIdentity();
122 	var name = (!identity)
123 		? this.settings.get(ZmSetting.DISPLAY_NAME)
124 		: identity.name;
125 
126 	if (!name) {

129 	return identity.isDefault && name == ZmIdentity.DEFAULT_NAME ? ZmMsg.accountDefault : name;

167 /**
168  * Gets the identity.
169  * 
170  * @return	{ZmIdentity}	the identity
171  */
172 ZmZimbraAccount.prototype.getIdentity =
173 function() {
174 	var defaultIdentity = appCtxt.getIdentityCollection(this).defaultIdentity;
175 	if (!appCtxt.isFamilyMbox || this.isMain) {
176 		return defaultIdentity;
177 	}
178 
179 	// for family mbox, create dummy identities for child accounts
180 	if (!this.dummyIdentity) {
181 		this.dummyIdentity = new ZmIdentity(this.name);
182 		AjxUtil.hashUpdate(this.dummyIdentity, defaultIdentity, true, ["name","isDefault"]);
183 	}
184 	return this.dummyIdentity;
185 };

187 /**
188  * Gets the tool tip.
189  * 
190  * @return	{String}		the tool tip
191  */
192 ZmZimbraAccount.prototype.getToolTip =
193 function() {
194 	if (this.status || this.lastSync || this.isMain) {
195 		var lastSyncDate = (this.lastSync && this.lastSync != 0)
196 			? (new Date(parseInt(this.lastSync))) : null;
197 
198 		var quota = appCtxt.get(ZmSetting.QUOTA_USED, null, this);
199 		var lastSync;
200 		if (!lastSyncDate) {
201 			// this means, we've synced but server lost the last sync timestamp
202 			if (quota > 0 && !this.isMain) {
203 				lastSync = ZmMsg.unknown;
204 			}
205 		} else {
206 			lastSync = AjxDateUtil.computeWordyDateStr(new Date(), lastSyncDate);
207 		}
208 
209 		var params = {
210 			lastSync: lastSync,
211 			hasNotSynced: this.hasNotSynced(),
212 			status: this.getStatusMessage(),
213 			quota: AjxUtil.formatSize(quota, false, 1)
214 		};
215 
216 		return AjxTemplate.expand("share.App#ZimbraAccountTooltip", params);
217 	}
218 	return "";
219 };

276 /**
277  * Gets the default calendar. For CalDAV based accounts, the default calendar is hidden;
278  * therefore, this method returns the first non-default calendar.
279  * 
280  * @return	{Object}		the calendar
281  * @see		ZmZimbraAccount.isCalDavBased
282  */
283 ZmZimbraAccount.prototype.getDefaultCalendar =
284 function() {
285 	var tree = appCtxt.getFolderTree(this);
286 	if (this.isCalDavBased()) {
287 		var calendars = tree.getByType(ZmOrganizer.CALENDAR);
288 		for (var i = 0; i < calendars.length; i++) {
289 			if (calendars[i].nId == ZmOrganizer.ID_CALENDAR) { continue; }
290 			return calendars[i];
291 		}
292 	}
293 	return tree.getById(ZmOrganizer.ID_CALENDAR);
294 };

296 /**
297  * Updates the account status.
298  * 
299  * @private
300  */
301 ZmZimbraAccount.prototype.updateState =
302 function(acctInfo) {
303 	if (this.isMain) { return; } // main account doesn't sync
304 
305 	// update last sync timestamp
306 	var updateTooltip = false;
307 	if (this.lastSync != acctInfo.lastsync) {
308 		this.lastSync = acctInfo.lastsync;
309 		if (this.visible) {
310 			updateTooltip = true;
311 		}
312 	}
313 
314 	// set to update account (offline) status if changed
315 	var updateStatus = false;
316 	if (this.status != acctInfo.status) {
317 		this.status = acctInfo.status;
318 		if (this.visible) {
319 			updateStatus = true;
320 		}
321 	}
322 
323 	// for all overview containers, update status/tooltip
324 	var container = appCtxt.getOverviewController()._overviewContainer;
325 	for (var i in container) {
326 		var c = container[i];
327 		if (updateStatus || updateTooltip) {
328 			c.updateAccountInfo(this, updateStatus, updateTooltip);
329 		}
330 	}
331 
332 	if (this.visible && acctInfo.unread != this.unread) {
333 		this.unread = acctInfo.unread;
334 	}
335 
336 	this.code = acctInfo.code;
337 	if (acctInfo.error) {
338 		var error = acctInfo.error[0];
339 		this.errorDetail = error.exception[0]._content;
340 		this.errorMessage = error.message;
341 	}
342 };

423 /**
424  * Shows an error message.
425  * 
426  * Offline use only.
427  * 
428  * @private
429  */
430 ZmZimbraAccount.prototype.showErrorMessage =
431 function() {
432 	if (!this.isError()) { return; }
433 
434 	var dialog = (this.status == ZmZimbraAccount.STATUS_ERROR)
435 		? appCtxt.getErrorDialog() : appCtxt.getMsgDialog();
436 
437 	// short message
438 	var msg = this.getZdMsg(this.code);
439 	if (msg == "") {
440 		msg = this.getStatusMessage();
441 	}
442 	dialog.setMessage(msg);
443 
444 	if (this.status == ZmZimbraAccount.STATUS_ERROR) {
445 		// detailed message
446 		var html = [];
447 		var i = 0;
448 		if (this.errorMessage) {
449 			html[i++] = "<p><b>";
450 			html[i++] = ZdMsg.DebugMsg;
451 			html[i++] = "</b>: ";
452 			html[i++] = this.errorMessage;
453 			html[i++] = "</p>";
454 		}
455 
456 		if (this.errorDetail) {
457 			html[i++] = "<p><b>";
458 			html[i++] = ZdMsg.DebugStack;
459 			html[i++] = "</b>:</p><p><pre>";
460 			html[i++] = this.errorDetail;
461 			html[i++] = "</pre></p>";
462 		}
463 
464 		html[i++] = "<p><b>";
465 		html[i++] = ZdMsg.DebugActionNote;
466 		html[i++] = "</b></p>";
467 
468 		dialog.setDetailString(html.join(""));
469 	}
470 
471 	dialog.popup(null, true);
472 };

484 /**
485  * Loads the account.
486  * 
487  * @param	{AjxCallback}	callback		the callback
488  */
489 ZmZimbraAccount.prototype.load =
490 function(callback) {
491 	if (!this.loaded) {
492 		// create new ZmSetting for this account
493 		this.settings = new ZmSettings();
494 
495 		// check "{APP}_ENABLED" state against main account's settings
496 		var mainAcct = appCtxt.accountList.mainAccount;
497 
498 		// for all *loaded* apps, add their app-specific settings
499 		for (var i = 0; i < ZmApp.APPS.length; i++) {
500 			var appName = ZmApp.APPS[i];
501 			var setting = ZmApp.SETTING[appName];
502 			if (setting && appCtxt.get(setting, null, mainAcct)) {
503 				var app = appCtxt.getApp(appName);
504 				if (app) {
505 					app._registerSettings(this.settings);
506 				}
507 			}
508 		}
509 
510 		var command = new ZmBatchCommand(null, this.name);
511 
512 		// load user settings retrieved from server now
513 		var loadCallback = new AjxCallback(this, this._handleLoadSettings);
514 		this.settings.loadUserSettings(loadCallback, null, this.name, null, command);
515 
516 		// get tag info for this account *FIRST* - otherwise, root ID get overridden
517 		var tagDoc = AjxSoapDoc.create("GetTagRequest", "urn:zimbraMail");
518 		var tagCallback = new AjxCallback(this, this._handleLoadTags);
519 		command.addNewRequestParams(tagDoc, tagCallback);
520 
521 		// get meta data for this account
522 		this.loadMetaData(null, command);
523 
524 		// get folder info for this account
525 		var folderDoc = AjxSoapDoc.create("GetFolderRequest", "urn:zimbraMail");
526 		folderDoc.getMethod().setAttribute("visible", "1");
527 		var folderCallback = new AjxCallback(this, this._handleLoadFolders);
528 		command.addNewRequestParams(folderDoc, folderCallback);
529 
530 		var respCallback = new AjxCallback(this, this._handleLoadUserInfo, callback);
531 		var errCallback = new AjxCallback(this, this._handleErrorLoad, callback);
532 		command.run(respCallback, errCallback);
533 	}
534 	else if (callback) {
535 		callback.run();
536 	}
537 };

591 /**
592  * Saves implicit prefs. Because it's done onunload, the save is sync.
593  * 
594  * @private
595  */
596 ZmZimbraAccount.prototype.saveImplicitPrefs =
597 function() {
598     var isExternal = this.settings ? this.settings.get(ZmSetting.IS_EXTERNAL) : false;
599     if (isExternal) {
600         return;
601     }
602 	var list = [];
603 	for (var id in ZmSetting.CHANGED_IMPLICIT) {
604 		var setting = this.settings ? this.settings.getSetting(id) : null;
605 		if (ZmSetting.IS_GLOBAL[setting.id] && !this.isMain) { continue; }
606 		if (setting && (setting.getValue(null, true) != setting.getOrigValue(null, true))) {
607 			list.push(setting);
608 		}
609 	}
610 
611 	if (list.length > 0) {
612 		this.settings.save(list, null, null, this);
613 	}
614 };

641 /**
642  * @private
643  */
644 ZmZimbraAccount.prototype._handleLoadSettings =
645 function(result) {
646 	DBG.println(AjxDebug.DBG1, "Account settings successfully loaded for " + this.name);
647 
648 	// set account type
649 	this.type = appCtxt.isOffline
650 		? appCtxt.get(ZmSetting.OFFLINE_ACCOUNT_FLAVOR, null, this)
651 		: ZmAccount.TYPE_ZIMBRA;
652 
653 	this.isZimbraAccount = this.type == ZmAccount.TYPE_ZIMBRA;
654 
655 	// set icon now that we know the type
656 	switch (this.type) {
657 		case ZmAccount.TYPE_AOL:		this.icon = "AccountAOL"; break;
658 		case ZmAccount.TYPE_GMAIL:		this.icon = "AccountGmail"; break;
659 		case ZmAccount.TYPE_IMAP:		this.icon = "AccountIMAP"; break;
660 		case ZmAccount.TYPE_LIVE:		this.icon = "AccountMSN"; break;
661 		case ZmAccount.TYPE_MSE:		this.icon = "AccountExchange"; break;
662 		case ZmAccount.TYPE_EXCHANGE:	this.icon = "AccountExchange"; break;
663 		case ZmAccount.TYPE_POP:		this.icon = "AccountPOP"; break;
664 		case ZmAccount.TYPE_YMP:		this.icon = "AccountYahoo"; break;
665 		case ZmAccount.TYPE_ZIMBRA:		this.icon = "AccountZimbra"; break;
666 	}
667 
668 	// initialize identities/data-sources/signatures for this account
669 	var obj = result.getResponse().GetInfoResponse;
670 	appCtxt.getIdentityCollection(this).initialize(obj.identities);
671 	appCtxt.getDataSourceCollection(this).initialize(obj.dataSources);
672 	appCtxt.getSignatureCollection(this).initialize(obj.signatures);
673 
674 };
