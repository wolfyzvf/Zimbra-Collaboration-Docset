 22 /**
 23  * 
 24  * @private
 25  */
 26 AjxTimedAction = function(obj, func, args) {
 27 	AjxCallback.call(this, obj, func, args);
 28 	this._tid = -1;
 29 	this._id = -1;
 30     this._runResult = null;
 31 }
 32 AjxTimedAction.prototype = new AjxCallback();
 33 AjxTimedAction.prototype.constructor = AjxTimedAction;
 34 
 35 // Setting a timeout of 25 days or more appears to revert it
 36 // to 0 in FF3 and Safari3. There's really no reason to set
 37 // it to anything above a few days, so set a max of 20 days.
 38 AjxTimedAction.MAX_TIMEOUT = 20 * 24 * 60 * 60 * 1000;
 39 
 40 AjxTimedAction.prototype.toString = 
 41 function() {
 42 	return "AjxTimedAction";
 43 };
 44 
 45 AjxTimedAction.prototype.getRunResult =
 46 function() {
 47     return this._runResult;
 48 };
 49 
 50 AjxTimedAction._pendingActions = {};
 51 AjxTimedAction._nextActionId = 1;
 52 
 53 AjxTimedAction.scheduleAction =
 54 function(action, timeout){
 55 	if (!action) { return; }
 56 	// if tid already exists, cancel previous timeout before setting a new one
 57 	if (action._tid && action._tid != -1) {
 58 		AjxTimedAction.cancelAction(action._id);
 59 	}
 60 
 61 	timeout = timeout || 0; // make sure timeout is numeric
 62 	if (timeout > AjxTimedAction.MAX_TIMEOUT) {
 63 		if (window.DBG) {
 64 			DBG.println(AjxDebug.DBG1, "timeout value above maximum: " + timeout);
 65 		}
 66 		timeout = AjxTimedAction.MAX_TIMEOUT;
 67 	}
 68 	var id = action._id = AjxTimedAction._nextActionId++;
 69 	AjxTimedAction._pendingActions[id] = action;
 70 	var actionStr = "AjxTimedAction._exec(" + id + ")";
 71 	action._tid = window.setTimeout(actionStr, timeout);
 72 	return action._id;
 73 };
 74 
 75 AjxTimedAction.cancelAction =
 76 function(actionId) {
 77 	var action = AjxTimedAction._pendingActions[actionId];
 78 	if (action) {
 79 		window.clearTimeout(action._tid);
 80 		delete AjxTimedAction._pendingActions[actionId];
 81 		delete action._tid;
 82 	}
 83 };
 84 
 85 AjxTimedAction._exec =
 86 function(actionId) {
 87 
 88 	try {
 89 
 90 	var action = AjxTimedAction._pendingActions[actionId];
 91 	if (action) {
 92 		delete AjxTimedAction._pendingActions[actionId];
 93 		delete action._tid;
 94 	    action._runResult = action.run();
 95 	}
 96 
 97 	} catch (ex) {
 98 		AjxException.reportScriptError(ex);
 99 	}
100 };
