 23 /**
 24  * Creates and initializes a manager for the given keymap.
 25  * @constructor
 26  * @class
 27  * A keymap manager parses the keymap into a form that is easily used for 
 28  * translating key codes into actions. It also provides some static methods
 29  * that map the available keyboard to key codes, and which qualify certain
 30  * keys as punctuation, etc.
 31  * 
 32  * @author Ross Dargahi
 33  *
 34  * @param {DwtKeyMap}	keyMap the keymap
 35  *
 36  * @private
 37  */
 38 DwtKeyMapMgr = function(keyMap) {
 39 
 40 	var map = this._map = keyMap.getMap();
 41 	this._args = keyMap._args;
 42 	
 43 	// build FSA for each mapping
 44 	this._fsas = {};
 45 	for (var key in map) {
 46 		DBG.println(AjxDebug.DBG3, "building FSA for key: " + key);
 47 		this._fsas[key] = DwtKeyMapMgr.__buildFSA({}, map[key], key);
 48 	}
 49 	DBG.dumpObj(AjxDebug.DBG3, this._fsas);
 50 };

 52 DwtKeyMapMgr.NOT_A_TERMINAL = -999;
 53 DwtKeyMapMgr.TAB_KEYCODE = 9;
 54 
 55 DwtKeyMapMgr.prototype.toString =
 56 function() {
 57 	return "DwtKeyMapMgr";
 58 };

 60 /**
 61  * This method will attempt to look up the action code for a given key sequence in
 62  * a given key map. 
 63  * 
 64  * @param {string}		keySeq				key sequence to lookup
 65  * @param {string}		mappingName			keymap name in which to search
 66  * @param {boolean}		forceActionCode		if <code>true</code>, then if the key sequence contains both
 67  * 											a submap and an action code, then return the action code.
 68  * 											If this parameter is false or omitted, then
 69  * 											{@link DwtKeyMapMgr.NOT_A_TERMINAL} will be returned for
 70  * 											a key sequence that contains both a submap and an action code.
 71  * 
 72  * @return {string|number}	the action code for the provided key map name, null if there is no action code
 73  * 		or {@link DwtKeyMapMgr.NOT_A_TERMINAL} if the key sequence is an intermediate
 74  * 		node in the key map (i.e. has a submap)
 75  * 
 76  */
 77 DwtKeyMapMgr.prototype.getActionCode =
 78 function(keySeq, mappingName, forceActionCode) {
 79 	//DBG.println(AjxDebug.DBG3, "Getting action code for: " + keySeq + " in map: " + mappingName);
 80 	var mapping =  this._fsas[mappingName];
 81 
 82 	if (!mapping) {
 83 		DBG.println(AjxDebug.DBG3, "No keymap for: " + mappingName);
 84 		return null;
 85 	}
 86 
 87 	var keySeqLen = keySeq.length;
 88 	var tmpFsa = mapping;
 89 	var key;
 90 	for (var j = 0; j < keySeqLen && tmpFsa; j++) {
 91 		key = keySeq[j];
 92 
 93 		if (!tmpFsa || !tmpFsa[key]) break;
 94 
 95 		if (j < keySeqLen - 1) {
 96 			tmpFsa = tmpFsa[key].subMap;
 97 		}
 98 	}
 99 
100 	if (tmpFsa && tmpFsa[key]) {
101 		var binding = tmpFsa[key];
102 		/* If the binding does not have a submap, then it must have an action code
103 		 * so return it. Else if the binding does not have an action code (i.e. it
104 		 * has a submap only) or if forceActionCode is false, then return DwtKeyMapMgr.NOT_A_TERMINAL
105 		 * since we are to behave like an intermediate node. Else return the action code. */
106 		if (!binding.subMap || forceActionCode) {
107 			var inherited = this.__getInheritedActionCode(keySeq, mapping, forceActionCode);
108             //if keyMap not available then return the inherited keyMap.
109             return inherited == DwtKeyMapMgr.NOT_A_TERMINAL ? DwtKeyMapMgr.NOT_A_TERMINAL : ( binding.actionCode || inherited );
110 		} else {
111 			return DwtKeyMapMgr.NOT_A_TERMINAL;
112 		}
113 	} else {
114 		return this.__getInheritedActionCode(keySeq, mapping, forceActionCode);
115 	}
116 };

118 /**
119  * Returns the action for the given map and key sequence.
120  * 
121  */
122 DwtKeyMapMgr.prototype.getAction =
123 function(mapName, keySeq) {
124 	return this._map[mapName][keySeq];
125 };

127 /**
128  * Returns the key sequences associated with the given map and action.
129  */
130 DwtKeyMapMgr.prototype.getKeySequences =
131 function(mapName, action) {
132 	var keySeqs = [];
133 	for (var ks in this._map[mapName]) {
134 		if (this._map[mapName][ks] == action) {
135 			keySeqs.push(ks);
136 		}
137 	}
138 	return keySeqs;
139 };

141 /**
142  * Allow the programmatic setting of a key sequence mapping for a given map
143  * 
144  * @param {string} 			mapName map name to affect
145  * @param {string} 			keySeq the key sequence to set
146  * @param {string|number} action the action code for the key sequence
147  */
148 DwtKeyMapMgr.prototype.setMapping =
149 function(mapName, keySeq, action) {
150 	this._map[mapName][keySeq] = action;
151 };
152 
153 /**
154  * Allow the programatting removal of a key sequence mapping for a given map
155  * 
156  * @param {string} mapName map name to affect
157  * @param {string} keySeq the key sequence to remove
158  */
159 DwtKeyMapMgr.prototype.removeMapping =
160 function(mapName, keySeq) {
161 	delete this._map[mapName][keySeq];
162 };
163 
164 /**
165  * Replace the key sequence for a given action in a keymap 
166  * 
167  * @param {string} mapName map name to affect
168  * @param {string} oldKeySeq the key sequence to replace
169  * @param {string} newKeySeq the new key sequence
170  */
171 DwtKeyMapMgr.prototype.replaceMapping =
172 function(mapName, oldKeySeq, newKeySeq) {
173 	var action = this._map[mapName][oldKeySeq];
174 	if (!action) return;
175 	this.removeMapping(mapName, oldKeySeq);
176 	this.setMapping(mapName, newKeySeq, action);
177 };

179 DwtKeyMapMgr.prototype.setArg =
180 function(mapName, action, arg) {
181 	if (!this._args[mapName]) {
182 		this._args[mapName] = {};
183 	}
184 	this._args[mapName][action] = arg;
185 };
186 
187 DwtKeyMapMgr.prototype.removeArg =
188 function(mapName, action) {
189 	delete this._args[mapName][action];
190 };
191 
192 DwtKeyMapMgr.prototype.getArg =
193 function(mapName, action) {
194 	return this._args[mapName] ? this._args[mapName][action] : null;
195 };

207 /**
208  * Returns a list of maps that the given map inherits from.
209  *
210  * @param {string} mapName Name of the keymap to reload
211  */
212 DwtKeyMapMgr.prototype.getAncestors =
213 function(mapName, list) {
214     list = list || [];
215     var subMap = this._fsas[mapName];
216     var parents = subMap && subMap.inherit;
217     if (parents && parents.length) {
218         for (var i = 0; i < parents.length; i++) {
219             list.push(parents[i]);
220             list = this.getAncestors(parents[i], list);
221         }
222     }
223     return list;
224 };

226 /**
227  * Returns true if the given element accepts text input.
228  * 
229  * @param element	[Element]		DOM element
230  */
231 DwtKeyMapMgr.isInputElement =
232 function(element) {
233 	if (!element) { return false; }
234 	// Check designMode in case we're in an HTML editor iframe
235 	var dm = element.ownerDocument ? element.ownerDocument.designMode : null;
236 	if (dm && (dm.toLowerCase() == "on")) { return true; }
237 
238 	var tag = element.tagName.toUpperCase();
239 	return (tag == "INPUT" || tag == "TEXTAREA");
240 };

242 DwtKeyMapMgr.__buildFSA =
243 function(fsa, mapping, mapName) {
244 	for (var i in mapping) {
245 		// check for inheritance from other maps (in CSV list)
246 		if (i == DwtKeyMap.INHERIT) {
247 			fsa.inherit = mapping[i].split(/\s*,\s*/);
248 			continue;
249 		}
250 		 
251 		var keySeq = i.split(DwtKeyMap.SEP);
252 		var keySeqLen = keySeq.length;
253 		var tmpFsa = fsa;
254 		for (var j = 0; j < keySeqLen; j++) {
255 			var key = keySeq[j];
256 			//DBG.println(AjxDebug.DBG3, "Processing: " + key);
257 			
258 			if (!tmpFsa[key]) {
259 				tmpFsa[key] = {};	// first time visiting this key
260 			}
261 
262 			if (j == keySeqLen - 1) {
263 				/* We are at the last key in the sequence so we can bind the
264 				 * action code to it */
265 				//DBG.println(AjxDebug.DBG3, "BINDING: " + mapping[i]);
266 				tmpFsa[key].actionCode = mapping[i];
267 			} else {
268 				/* We have more keys in the sequence. If our subMap is null,
269 				 * then we need to create it to hold the new key sequences */
270 				if (!tmpFsa[key].subMap) {
271 					tmpFsa[key].subMap = {};
272 					//DBG.println(AjxDebug.DBG3, "NEW SUBMAP");
273 				}
274 					
275 				tmpFsa = tmpFsa[key].subMap;
276 			}			
277 		}
278 	}
279 	return fsa;
280 };
281 
282 DwtKeyMapMgr.prototype.__getInheritedActionCode =
283 function(keySeq, mapping, forceActionCode) {
284 	if (mapping.inherit && mapping.inherit.length) {
285 		var actionCode = null;
286 		var len = mapping.inherit.length;
287 		for (var i = 0; i < len; i++) {
288 			DBG.println(AjxDebug.DBG3, "checking inherited map: " + mapping.inherit[i]);
289 			actionCode = this.getActionCode(keySeq, mapping.inherit[i], forceActionCode);
290 			if (actionCode != null) {
291 				return actionCode;
292 			}
293 		}
294 	}
295 	return null;
296 };

298 /**
299  * Returns true if the given key event has a modifier which makes it nonprintable.
300  * 
301  * @param ev	[Event]		key event
302  */
303 DwtKeyMapMgr.hasModifier =
304 function(ev) {
305 	return (ev.altKey || ev.ctrlKey || ev.metaKey);
