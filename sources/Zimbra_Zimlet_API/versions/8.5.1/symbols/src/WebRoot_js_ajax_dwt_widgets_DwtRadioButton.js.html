 22 /**
 23  * Creates a radio button.
 24  * @constructor
 25  * @class
 26  * This class implements a radio button.
 27  * 
 28  * @param {hash}	params	a hash of parameters
 29  * @param  {DwtComposite}     params.parent	the parent widget
 30  * @param  {constant}     params.style 	the text style. May be one of: {@link DwtCheckbox.TEXT_LEFT} or
 31  * 									{@link DwtCheckbox.TEXT_RIGHT} arithimatically or'd (|) with one of:
 32  * 									{@link DwtCheckbox.ALIGN_LEFT}, {@link DwtCheckbox.ALIGN_CENTER}, or
 33  * 									{@link DwtCheckbox.ALIGN_LEFT}.
 34  * 									The first determines were in the checkbox the text will appear
 35  * 									(if set), the second determine how the content of the text will be
 36  * 									aligned. The default value for this parameter is: 
 37  * 									{@link DwtCheckbox.TEXT_LEFT} | {@link DwtCheckbox.ALIGN_CENTER}
 38  * @param  {string}     params.name		the input control name (required for IE)
 39  * @param  {string}     params.value     the input control value.
 40  * @param  {boolean}     params.checked	the input control checked status (required for IE)
 41  * @param  {string}     params.className	the CSS class
 42  * @param  {constant}     params.posStyle	the positioning style (see {@link DwtControl})
 43  * @param  {string}     params.id		an explicit ID to use for the control's HTML element
 44  * @param  {number}     params.index 	the index at which to add this control among parent's children
 45  * 
 46  * @extends	DwtCheckbox
 47  */
 48 DwtRadioButton = function(params) {
 49 	if (arguments.length == 0) { return; }
 50 	params = Dwt.getParams(arguments, DwtRadioButton.PARAMS);
 51 	params.className = params.className || "DwtRadioButton";
 52 	DwtCheckbox.call(this, params);
 53 }
 54 
 55 DwtRadioButton.PARAMS = DwtCheckbox.PARAMS;
 56 
 57 DwtRadioButton.prototype = new DwtCheckbox;
 58 DwtRadioButton.prototype.constructor = DwtRadioButton;
 59 
 60 DwtRadioButton.prototype.isDwtRadioButton = true;
 61 DwtRadioButton.prototype.isInputControl = true;
 62 DwtRadioButton.prototype.toString = function() { return "DwtRadioButton"; };
 63
