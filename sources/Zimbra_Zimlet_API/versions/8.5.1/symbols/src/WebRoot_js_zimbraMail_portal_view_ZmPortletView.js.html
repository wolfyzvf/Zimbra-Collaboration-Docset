 22 /**
 23  * Creates the portlet view.
 24  * @class
 25  * This class represents the portlet view.
 26  * 
 27  * @param	{Element}		parentEl		the parent element
 28  * @param	{ZmPortlet}		portlet			the portlet
 29  * @param	{String}		className		the class name
 30  * 
 31  * @extends		DwtComposite
 32  */
 33 ZmPortletView = function(parentEl, portlet, className) {
 34     className = className || "ZmPortlet";
 35     DwtComposite.call(this, {parent:DwtShell.getShell(window), className:className, posStyle:DwtControl.STATIC_STYLE});
 36 
 37     // save data
 38     this._portlet = portlet;
 39     this._portlet.view = this;
 40 
 41     this._contentsEl = this.getHtmlElement();
 42     if (parentEl) {
 43         parentEl.portlet = "loaded";
 44         parentEl.innerHTML = "";
 45         parentEl.appendChild(this._contentsEl);
 46     }
 47 
 48     // setup display
 49     this.setIcon(portlet.icon);
 50     this.setTitle(portlet.title);
 51     this.setContentUrl(portlet.actionUrl && portlet.actionUrl.target);
 52 }
 53 ZmPortletView.prototype = new DwtComposite;
 54 ZmPortletView.prototype.constructor = ZmPortletView;
 55 
 56 /**
 57  * Returns a string representation of the object.
 58  * 
 59  * @return		{String}		a string representation of the object
 60  */
 61 ZmPortletView.prototype.toString = function() {
 62     return "ZmPortletView";
 63 };

 69 /**
 70  * Sets the icon.
 71  * 
 72  * @param	{String}	icon		the icon
 73  * 
 74  * @see		AjxImg.setImage
 75  */
 76 ZmPortletView.prototype.setIcon = function(icon) {
 77     if (icon == null || !this._iconEl) return;
 78     AjxImg.setImage(this._iconEl, icon);
 79 };
 80 
 81 /**
 82  * Sets the title.
 83  * 
 84  * @param	{String}		title		the title
 85  */
 86 ZmPortletView.prototype.setTitle = function(title) {
 87     if (title == null || !this._titleEl) return;
 88     this._titleEl.innerHTML = title;
 89 };
 90 
 91 ZmPortletView.prototype.setContent = function(content) {
 92     if (AjxUtil.isString(content)) {
 93         this._contentsEl.innerHTML = content;
 94     }
 95     else if (AjxUtil.ELEMENT_NODE) {
 96         this._contentsEl.innerHTML = "";
 97         this._contentsEl.appendChild(content);
 98     }
 99     else {
100         this._contentsEl.innerHTML = AjxStringUtil.htmlEncode(String(content));
101     }
102 };
103 
104 /**
105  * Sets the content URL as an iframe.
106  * 
107  * @param	{String}	url		the url
108  */
109 ZmPortletView.prototype.setContentUrl = function(url) {
110     if (!url) return;
111 
112     var props = this._portlet.properties;
113     var func = AjxCallback.simpleClosure(ZmPortletView.__replaceProp, null, props);
114     url = url.replace(ZmZimletContext.RE_SCAN_PROP, func);
115     url = this._portlet.zimletCtxt ? this._portlet.zimletCtxt.makeURL({ target: url }, null, props) : url;
116     var html = [
117         "<iframe style='border:none;width:100%;height:100%' ",
118             "src='",url,"'>",
119         "</iframe>"
120     ].join("");
121     this.setContent(html);
122 };
