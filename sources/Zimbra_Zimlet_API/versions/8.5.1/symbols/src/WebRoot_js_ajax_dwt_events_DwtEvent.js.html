 22 /**
 23  * @class
 24  * This class is the base class of all DWT events.
 25  * 
 26  * @param {boolean} 	__init 	a dummy parameter used for class initialization
 27  * 
 28  * @author Ross Dargahi
 29  * @author Conrad Damon
 30  * 
 31  */
 32 DwtEvent = function(__init) {
 33 	if (arguments.length == 0) return;
 34 	/**
 35 	 * The Dwt object that generated the event
 36 	 * @type DwtControl
 37 	 */
 38 	this.dwtObj = null;
 39 }
 40 
 41 /**
 42  * Returns a string representation of the object.
 43  * 
 44  * @return		{string}		a string representation of the object
 45  */
 46 DwtEvent.prototype.toString = 
 47 function() {
 48 	return "DwtEvent";
 49 }

158 // semantic events

160 /**
161  * Action event. An example is right-clicking on a list item or tree item
162  * generally brings up a context menu.
163  */
164 DwtEvent.ACTION	= "ACTION";

166 /**
167  * Control event. Control events are fired by resizing or repositioning {@link DwtControl}s.
168  */
169 DwtEvent.CONTROL = "CONTROL";		// resize

171 /**
172  * Date Range events are fired by the {@link DwtCalendar} widget. This event is

174  */
175 DwtEvent.DATE_RANGE	= "DATE_RANGE";

177 /**
178  * The dispose event is fired when the {@link DwtControl#dispose} method of a control is called.
179  */

184  * @private
185  */
186 DwtEvent.ENTER = "ENTER";			// enter/return key

188 /**
189  * This event is fired when the mouse hovers over a control for a certain period of time.
190  */
191 DwtEvent.HOVEROVER = "HOVEROVER";
192 
193 /**
194  * This event is fired when the mouse stops hovering over a control.
195  */
196 DwtEvent.HOVEROUT = "HOVEROUT";

198 /**
199  * The popdown event is fired when a item (such as a {@link DwtMenu}) is popped down.
200  */
201 DwtEvent.POPDOWN = "POPDOWN";
202 
203 /**
204  * The popup event is fired when a item (such as a {@link DwtMenu}) is popped up.
205  */
206 DwtEvent.POPUP = "POPUP";
207 
208 /**
209  * The selection event is fired when controls are selected. This generally means
210  * that there has been a "left mouse button click" in the control (for example: a button, or

212  */
213 DwtEvent.SELECTION = "SELECTION";		// left-click

215 /**

217  */
218 DwtEvent.TREE = "TREE";
