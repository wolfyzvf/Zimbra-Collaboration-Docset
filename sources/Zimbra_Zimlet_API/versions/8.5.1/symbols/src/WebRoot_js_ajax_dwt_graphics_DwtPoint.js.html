 22 /**
 23  * Creates a point.
 24  * @constructor
 25  * @class
 26  * This class represents a point. A point has an x-coordinate and y-coordinate.
 27  * 
 28  * @author Ross Dargahi
 29  * 
 30  * @param {number} x 	the x coordinate
 31  * @param {number} y 	the y coordinate
 32  * 
 33  */
 34 DwtPoint = function(x, y) {
 35 	/**
 36 	 * The x-coordinate.
 37 	 * @type	number
 38 	 */
 39 	this.x = x || 0;
 40 	/**
 41 	 * The y-coordinate.
 42 	 * @type	number
 43 	 */
 44 	this.y = y || 0;
 45 }
 46 
 47 DwtPoint.tmp = new DwtPoint(0, 0);
 48 
 49 /**
 50  * Returns a string representation of the object.
 51  * 
 52  * @return		{string}		a string representation of the object
 53  */
 54 DwtPoint.prototype.toString = 
 55 function() {
 56 	return "DwtPoint";
 57 }
 58 
 59 /**
 60  * Sets the values of a point
 61  * 
 62  * @param {number} x 	the x coordinate
 63  * @param {number} y 	the y coordinate
 64  */
 65  DwtPoint.prototype.set =
 66  function(x, y) {
 67  	this.x = x;
 68  	this.y = y;
 69  }
