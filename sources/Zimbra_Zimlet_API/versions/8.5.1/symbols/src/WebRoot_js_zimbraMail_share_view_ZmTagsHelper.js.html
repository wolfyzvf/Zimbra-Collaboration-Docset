 22 /**
 23  * @overview
 24  * A static helper class used to do most of the stuff related to an item's tags - get the bubbles HTML, add listeners for changes, and inform the view to redraw the new html whenever needed.
 25  * The parent view itself will set  * the html into the view - this also allows the parent to do special things - for example for empty list of tags in mail message view - hide the "Tags:" row altogether.
 26  */
 27 
 28 ZmTagsHelper = {};
 29 
 30 ZmTagsHelper.setupListeners = function(view) {
 31 
 32 	if (appCtxt.isChildWindow) {
 33 		return;
 34 	}
 35 	// Add change listener to taglist to track changes in tag color, tag name, etc.
 36 	view._tagList = appCtxt.getTagTree();
 37 	if (!view._tagList) {
 38 		return;
 39 	}
 40 	view._tagChangeHandler = ZmTagsHelper._tagChangeListener.bind(view);
 41 	view._tagList.addChangeListener(view._tagChangeHandler);
 42 };
 43 
 44 ZmTagsHelper.disposeListeners =
 45 function(view) {
 46 	view._tagList && view._tagList.removeChangeListener(view._tagChangeHandler);
 47 };
 48 
 49 /**
 50  * gets the tags HTML (the bubbles html) for the item's tags
 51  * @param item
 52  */
 53 ZmTagsHelper.getTagsHtml =
 54 function(item, view) {
 55 
 56 	if (!appCtxt.get(ZmSetting.TAGGING_ENABLED)) {
 57 		return "";
 58 	}
 59 
 60 	var tags = item && item.getSortedTags();
 61 	if (!(tags && tags.length)) {
 62 		return "";
 63 	}
 64 
 65 	var html = [], i = 0;
 66 	for (var j = 0; j < tags.length; j++) {
 67 		var tag = tags[j];
 68 		if (!tag) {
 69 			continue;
 70 		}
 71 		var tagParams = {
 72 			tag:		tag,
 73 			html:		html,
 74 			i:			i,
 75 			view:		view,
 76 			readOnly:	item.isReadOnly()
 77 		};
 78 		i = ZmTagsHelper._getTagHtml(tagParams);
 79 	}
 80 	return html.join("");
 81 };
 82 
 83 ZmTagsHelper._getTagHtml =
 84 function(params) {
 85 	params = params || {};
 86 	var tag			= params.tag;
 87 	var html		= params.html;
 88 	var i			= params.i;
 89 	var view		= params.view;
 90 	var readOnly	= params.readOnly;
 91 
 92 	var tagClick = ['ZmTagsHelper._tagClick("', view._htmlElId, '","', AjxStringUtil.encodeQuotes(tag.name), '");'].join("");
 93 	var removeClick = ['ZmTagsHelper._removeTagClick("', view._htmlElId, '","', AjxStringUtil.encodeQuotes(tag.name), '");'].join("");
 94 
 95 	html[i++] = "<span class='addrBubble TagBubble' notoggle=1 >";
 96 
 97 	html[i++] = "<span class='TagImage' onclick='";
 98 	html[i++] = tagClick;
 99 	html[i++] = "'>";
100 	html[i++] = AjxImg.getImageHtml(tag.getIconWithColor(), null);
101 	html[i++] = "</span>";
102 
103 	html[i++] = "<span class='TagName' onclick='";
104 	html[i++] = tagClick;
105 	html[i++] = "'>";
106 	html[i++] = AjxStringUtil.htmlEncodeSpace(tag.name);
107 	html[i++] = " </span>";
108 
109     if (!readOnly) {
110         html[i++] = "<span class='ImgBubbleDelete' onclick='";
111         html[i++] = removeClick;
112         html[i++] = "'>";
113         html[i++] = "</span>";
114     }
115 	html[i++] = "</span>";
116 
117 	return i;
118 };
119 
120 
121 ZmTagsHelper._tagClick =
122 function(parentId, tagName) {
123 	var tag = ZmTagsHelper._getTagClicked(tagName);
124 	var view = DwtControl.fromElementId(parentId);
125 	appCtxt.getSearchController().search({query: tag.createQuery(), inclSharedItems: true});
126 };
127 
128 ZmTagsHelper._removeTagClick =
129 function(parentId, tagName) {
130 	var tag = ZmTagsHelper._getTagClicked(tagName);
131 	var view = DwtControl.fromElementId(parentId);
132 	ZmListController.prototype._doTag.call(view._controller, view._item, tag, false);
133 };
134 
135 ZmTagsHelper._getTagClicked =
136 function(tagName) {
137 	var tagList = appCtxt.getAccountTagList();
138 	return tagList.getByNameOrRemote(tagName);
139 };
140 
141 ZmTagsHelper._tagChangeListener =
142 function(ev) {
143 	//note - "this" is bound to the view.
144 	if (ev.type != ZmEvent.S_TAG) {	return; }
145 	if (this._disposed) { return; }
146 
147 	if (ev.event == ZmEvent.E_DELETE || ev.event == ZmEvent.E_MODIFY || ev.event == ZmEvent.E_CREATE) {
148 		//note - create is needed in case of a tag that was not in local tag list (due to sharing) that now is.
149 		this._setTags();
150 	}
151 };
