 26 /**
 27  * Creates a timezone dialog.
 28  * @class
 29  * This class represents a timezone dialog.
 30  * 
 31  * @param	{DwtComposite}	parent		the parent
 32  * @param	{String}	className		the class name
 33  *  
 34  * @extends		DwtDialog
 35  */
 36 ZmTimezonePicker = function(parent, className) {
 37 
 38     var buttons = [ ZmTimezonePicker.SAVE_BUTTON ];
 39     var saveButton = new DwtDialog_ButtonDescriptor(ZmTimezonePicker.SAVE_BUTTON, ZmMsg.save, DwtDialog.ALIGN_RIGHT);
 40 	DwtDialog.call(this, {parent:parent, className:className,
 41         title:ZmMsg.selectTimezoneTitle, standardButtons:DwtDialog.NO_BUTTONS,
 42         extraButtons: [saveButton]});
 43 
 44     this.setButtonListener(ZmTimezonePicker.SAVE_BUTTON, new AjxListener(this, this._handleSaveButton));
 45     this.setContent(this._contentHtml());
 46 	this._setTimezoneMenu();
 47 };
 48 
 49 ZmTimezonePicker.prototype = new DwtDialog;
 50 ZmTimezonePicker.prototype.constructor = ZmTimezonePicker;

 57 };

 64 };

 71 ZmTimezonePicker.prototype._setTimezoneMenu =
 72 function() {
 73 	var timezoneListener = new AjxListener(this, this._timezoneListener);
 74 	this._tzoneSelect = new DwtSelect({parent:this, parentElement: (this._htmlElId + "_tzSelect"), layout: DwtMenu.LAYOUT_SCROLL, maxRows:7});
 75 	this._tzoneSelect.addChangeListener(timezoneListener);
 76 
 77 	this._tzoneShowAll = new DwtCheckbox({parent:this, parentElement:(this._htmlElId+"_tzShowAll")});
 78 	this._tzoneShowAll.setText(ZmMsg.selectTimezoneIShowAll);
 79 	this._tzoneShowAll.addSelectionListener(new AjxListener(this, this._handleShowAllChange));
 80 };
 81 
 82 ZmTimezonePicker.prototype._initTzSelect =
 83 function(force) {
 84     var showAll = this._tzoneShowAll.isSelected();
 85 	var options = showAll ? AjxTimezone.getAbbreviatedZoneChoices() : AjxTimezone.getMatchingTimezoneChoices();
 86     var serverIdMap = {};
 87     var serverId;
 88 	if (force || options.length != this._tzCount) {
 89 		this._tzCount = options.length;
 90 		this._tzoneSelect.clearOptions();
 91 		for (var i = 0; i < options.length; i++) {
 92             if(options[i].autoDetected) continue;
 93 
 94             serverId = options[i].value;
 95             //avoid duplicate entries
 96             if(!showAll && serverIdMap[serverId]) continue;
 97             serverIdMap[serverId] = true;
 98 
 99             this._tzoneSelect.addOption(options[i]);
100 		}
101 	}
102 };
103 
104 ZmTimezonePicker.prototype.autoSelectTimezone =
105 function() {
106     if(AjxTimezone.DEFAULT_RULE.autoDetected) {
107 
108         var cRule = AjxTimezone.DEFAULT_RULE;
109         var standardOffsetMatch, daylightOffsetMatch, transMatch;
110 
111         for(var i in AjxTimezone.MATCHING_RULES) {
112             var rule = AjxTimezone.MATCHING_RULES[i];
113             if(rule.autoDetected) continue;
114             if(rule.standard.offset == cRule.standard.offset) {
115 
116                 if(!standardOffsetMatch) standardOffsetMatch = rule.serverId;
117 
118                 var isDayLightOffsetMatching = (cRule.daylight && rule.daylight && (rule.daylight.offset == cRule.daylight.offset));
119 
120                 if(isDayLightOffsetMatching) {
121 
122                     if(!daylightOffsetMatch) daylightOffsetMatch = rule.serverId;
123 
124                     var isTransYearMatching = (rule.daylight.trans[0] == cRule.daylight.trans[0]);
125                     var isTransMonthMatching = (rule.daylight.trans[1] == cRule.daylight.trans[1]);
126 
127                     if(isTransYearMatching && isTransMonthMatching && !transMatch) {
128                         transMatch = rule.serverId;
129                     }
130                 }
131             }
132         }
133         //select closest matching timezone
134         var serverId = transMatch ? transMatch : (daylightOffsetMatch || standardOffsetMatch);
135         if(serverId) this._tzoneSelect.setSelectedValue(serverId);
136     }else {
137         var tz = AjxTimezone.getServerId(AjxTimezone.DEFAULT);
138         this._tzoneSelect.setSelectedValue(tz);
139     }
140 };

142 /**
143  * Updates the selected timezone.
144  * 
145  * @param	{Hash}	dateInfo		a hash of date information
146  * @param	{ZmTimezone}	timezone		the timezone
147  */
148 ZmTimezonePicker.prototype.updateTimezone =
149 function(dateInfo) {
150 	this._tzoneSelect.setSelectedValue(dateInfo.timezone);
151 };

163 ZmTimezonePicker.prototype._handleShowAllChange = function(evt) {
164     var value = this._tzoneSelect.getValue();
165     this._initTzSelect(true);
166     this._tzoneSelect.setSelectedValue(value);
167 };

188 ZmTimezonePicker.prototype._handleSaveButton =
189 function(event) {
190     var timezone = this._tzoneSelect.getValue();
191     this.popdown();
192     if(this._callback) {
193         this._callback.run(timezone);
194     }
195 };
196 
197 ZmTimezonePicker.prototype.handleKeyAction =
198 function(actionCode, ev) {
199 	switch (actionCode) {
200 		case DwtKeyMap.CANCEL:
201 			this._runCallbackForButtonId(DwtDialog.CANCEL_BUTTON);
202 			break;
203 		default:
204 			DwtDialog.prototype.handleKeyAction.call(this, actionCode, ev);
205 			break;
206 	}
207 	return true;
208 };
