 28 /**
 29  * Creates a component that lets the user search for and select one or more contacts.
 30  * @constructor
 31  * @class
 32  * This class creates and manages a component that lets the user search for and
 33  * select one or more contacts. It is intended to be plugged into a larger
 34  * component such as a dialog or a tab view.
 35  *
 36  * @author Conrad Damon
 37  *
 38  * @param {hash}			params				hash of parameters:
 39  * @param {DwtComposite}		parent			the containing widget
 40  * @param {string}				className		the CSS class
 41  * @param {hash}				options			hash of options:
 42  * @param {string}					preamble	explanatory text
 43  * @param {array}					searchFor	list of ZmContactsApp.SEARCHFOR_*
 44  * @param {boolean}					showEmails	if true, show each email in results
 45  *
 46  * @extends		DwtComposite
 47  *
 48  * TODO: scroll-based paging
 49  * TODO: adapt for contact picker and attachcontacts zimlet
 50  */
 51 ZmContactSearch = function(params) {
 52 
 53 	params = params || {};
 54 	params.parent = params.parent || appCtxt.getShell();
 55 	params.className = params.className || "ZmContactSearch";
 56 	DwtComposite.call(this, params);
 57 
 58 	this._options = params.options;
 59 	this._initialized = false;
 60 	this._searchErrorCallback = new AjxCallback(this, this._handleErrorSearch);
 61 	if (!ZmContactSearch._controller) {
 62 		ZmContactSearch._controller = new ZmContactSearchController();
 63 	}
 64 	this._controller = ZmContactSearch._controller;
 65 	this._initialize();
 66 };
 67

 89 /**
 90  * Performs a search.
 91  *
 92  */
 93 ZmContactSearch.prototype.search =
 94 function(ascending, firstTime, lastId, lastSortVal) {
 95 
 96 	if (!AjxUtil.isSpecified(ascending)) {
 97 		ascending = true;
 98 	}
 99 
100 	var query = this._searchCleared ? AjxStringUtil.trim(this._searchField.value) : "";
101 
102 	var queryHint;
103 	if (this._selectDiv) {
104 		var searchFor = this._selectDiv.getValue();
105 		this._contactSource = (searchFor == ZmContactsApp.SEARCHFOR_CONTACTS || searchFor == ZmContactsApp.SEARCHFOR_PAS)
106 			? ZmItem.CONTACT
107 			: ZmId.SEARCH_GAL;
108 
109 		if (searchFor == ZmContactsApp.SEARCHFOR_PAS) {
110 			queryHint = ZmSearchController.generateQueryForShares(ZmId.ITEM_CONTACT) || "is:local";
111 		} else if (searchFor == ZmContactsApp.SEARCHFOR_CONTACTS) {
112 			queryHint = "is:local";
113 		} else if (searchFor == ZmContactsApp.SEARCHFOR_GAL) {
114             ascending = true;
115         }
116 	} else {
117 		this._contactSource = appCtxt.get(ZmSetting.CONTACTS_ENABLED, null, this._account)
118 			? ZmItem.CONTACT
119 			: ZmId.SEARCH_GAL;
120 
121 		if (this._contactSource == ZmItem.CONTACT) {
122 			queryHint = "is:local";
123 		}
124 	}
125 
126 	this._searchIcon.className = "DwtWait16Icon";
127 
128 	// XXX: line below doesn't have intended effect (turn off column sorting for GAL search)
129 //	this._chooser.sourceListView.sortingEnabled = (this._contactSource == ZmItem.CONTACT);
130 
131 	var params = {
132 		obj: this,
133 		ascending: ascending,
134 		query: query,
135 		queryHint: queryHint,
136 		offset: this._list.size(),
137 		lastId: lastId,
138 		lastSortVal: lastSortVal,
139 		respCallback: (new AjxCallback(this, this._handleResponseSearch, [firstTime])),
140 		errorCallback: this._searchErrorCallback,
141 		accountName: (this._account && this._account.name)
142 	};
143 	ZmContactsHelper.search(params);
144 };
145 
146 ZmContactSearch.prototype._handleResponseSearch =
147 function(firstTime, result) {
148 
149 	this._controller.show(result.getResponse(), firstTime);
150 	var list = this._controller._list;
151 	if (list) {
152 		this._list = list.getVector();
153 		if (list && list.size() == 1) {
154 			this._listView.setSelection(list.get(0));
155 		}
156 	}
157 
158 	this._searchIcon.className = "ImgSearch";
159 	this._searchButton.setEnabled(true);
160 };
161 
162 ZmContactSearch.prototype.getContacts =
163 function() {
164 	return this._listView.getSelection();
165 };
166 
167 ZmContactSearch.prototype.setAccount =
168 function(account) {
169 	if (this._account != account) {
170 		this._account = account;
171 		this._resetSelectDiv();
172 	}
173 };
174 
175 ZmContactSearch.prototype.reset =
176 function(query, account) {
177 
178 	this._offset = 0;
179 	if (this._list) {
180 		this._list.removeAll();
181 	}
182 	this._list = new AjxVector();
183 
184 	// reset search field
185 	this._searchField.disabled = false;
186 	this._searchField.focus();
187 	query = query || this._searchField.value;
188 	if (query) {
189 		this._searchField.className = "";
190 		this._searchField.value = query;
191 		this._searchCleared = true;
192 	} else {
193 		this._searchField.className = "searchFieldHint";
194 		this._searchField.value = ZmMsg.contactPickerHint;
195 		this._searchCleared = false;
196 	}
197 
198 	this.setAccount(account || this._account);
199 };

204 ZmContactSearch.prototype._initialize =
205 function() {
206 
207 	this._searchForHash = this._options.searchFor ? AjxUtil.arrayAsHash(this._options.searchFor) : {};
208 
209 	this.getHtmlElement().innerHTML = this._contentHtml();
210 
211 	if (this._options.preamble) {
212 		var div = document.getElementById(this._htmlElId + "_preamble");
213 		div.innerHTML = this._options.preamble;
214 	}
215 
216 	this._searchIcon = document.getElementById(this._htmlElId + "_searchIcon");
217 
218 	// add search button
219 	this._searchButton = new DwtButton({parent:this, parentElement:(this._htmlElId + "_searchButton")});
220 	this._searchButton.setText(ZmMsg.search);
221 	this._searchButton.addSelectionListener(new AjxListener(this, this._searchButtonListener));
222 
223 	// add select menu, if needed
224 	var selectCellId = this._htmlElId + "_folders";
225 	var selectCell = document.getElementById(selectCellId);
226 	if (selectCell) {
227 		this._selectDiv = new DwtSelect({parent:this, parentElement:selectCellId});
228 		this._resetSelectDiv();
229 		this._selectDiv.addChangeListener(new AjxListener(this, this._searchTypeListener));
230 	}
231 
232 	this._searchField = document.getElementById(this._htmlElId + "_searchField");
233 	Dwt.setHandler(this._searchField, DwtEvent.ONKEYUP, ZmContactSearch._keyPressHdlr);
234 	Dwt.setHandler(this._searchField, DwtEvent.ONCLICK, ZmContactSearch._onclickHdlr);
235 	this._keyPressCallback = new AjxCallback(this, this._searchButtonListener);
236 
237 	var listDiv = document.getElementById(this._htmlElId + "_results");
238 	if (listDiv) {
239 		params = {parent:this, parentElement:listDiv, options:this._options};
240 		this._listView = this._controller._listView = new ZmContactSearchListView(params);
241 	}
242 
243 	this._initialized = true;
244 };

246 /**
247  * @private
248  */
249 ZmContactSearch.prototype._contentHtml =
250 function() {
251 
252 	var showSelect;
253 	if (appCtxt.multiAccounts) {
254 		var list = appCtxt.accountList.visibleAccounts;
255 		for (var i = 0; i < list.length; i++) {
256 			this._setSearchFor(list[i]);
257 			if (this._searchFor.length > 1) {
258 				showSelect = true;
259 				break;
260 			}
261 		}
262 	} else {
263 		this._setSearchFor();
264 		showSelect = (this._searchFor.length > 1);
265 	}
266 
267 	var subs = {
268 		id: this._htmlElId,
269 		showSelect: showSelect
270 	};
271 
272 	return (AjxTemplate.expand("abook.Contacts#ZmContactSearch", subs));
273 };
274 
275 ZmContactSearch.prototype._setSearchFor =
276 function(account) {
277 
278 	account = account || this._account;
279 	this._searchFor = [];
280 	if (this._options.searchFor && this._options.searchFor.length) {
281 		for (var i = 0; i < this._options.searchFor.length; i++) {
282 			var searchFor = this._options.searchFor[i];
283 			if (appCtxt.get(ZmContactSearch.SEARCHFOR_SETTING[searchFor], null, account)) {
284 				this._searchFor.push(searchFor);
285 			}
286 		}
287 	}
288 	this._searchForHash = AjxUtil.arrayAsHash(this._searchFor);
289 };
290 
291 /**
292  * @private
293  */
294 ZmContactSearch.prototype._resetSelectDiv =
295 function() {
296 
297 	if (!this._selectDiv) { return; }
298 	
299 	this._selectDiv.clearOptions();
300 	this._setSearchFor();
301 
302 	var sfh = this._searchForHash;
303 	if (sfh[ZmContactsApp.SEARCHFOR_CONTACTS]) {
304 		this._selectDiv.addOption(ZmMsg.contacts, false, ZmContactsApp.SEARCHFOR_CONTACTS);
305 
306 		if (sfh[ZmContactsApp.SEARCHFOR_PAS]) {
307 			this._selectDiv.addOption(ZmMsg.searchPersonalSharedContacts, false, ZmContactsApp.SEARCHFOR_PAS);
308 		}
309 	}
310 
311 	if (sfh[ZmContactsApp.SEARCHFOR_GAL]) {
312 		this._selectDiv.addOption(ZmMsg.GAL, true, ZmContactsApp.SEARCHFOR_GAL);
313 	}
314 
315 	if (!appCtxt.get(ZmSetting.INITIALLY_SEARCH_GAL, null, this._account) ||
316 		!appCtxt.get(ZmSetting.GAL_ENABLED, null, this._account))
317 	{
318 		this._selectDiv.setSelectedValue(ZmContactsApp.SEARCHFOR_CONTACTS);
319 	}
320 
321 	// TODO
322 //	if (sfh[ZmContactsApp.SEARCHFOR_FOLDERS]) {
323 //	}
324 };

336 /**
337  * @private
338  */
339 ZmContactSearch._keyPressHdlr =
340 function(ev) {
341 	var stb = DwtControl.getTargetControl(ev);
342 	var charCode = DwtKeyEvent.getCharCode(ev);
343 	if (!stb._searchCleared) {
344 		stb._searchField.className = stb._searchField.value = "";
345 		stb._searchCleared = true;
346 	}
347 	if (stb._keyPressCallback && (charCode == 13 || charCode == 3)) {
348 		stb._keyPressCallback.run();
349 		return false;
350 	}
351 	return true;
352 };
353 
354 /**
355  * @private
356  */
357 ZmContactSearch._onclickHdlr =
358 function(ev) {
359 	var stb = DwtControl.getTargetControl(ev);
360 	if (!stb._searchCleared) {
361 		stb._searchField.className = stb._searchField.value = "";
362 		stb._searchCleared = true;
363 	}
364 };

387 ZmContactSearchController.prototype.show =
388 function(searchResult, firstTime) {
389 
390 	var more = searchResult.getAttribute("more");
391 	var list = this._list = searchResult.getResults(ZmItem.CONTACT);
392 	if (list.size() == 0 && firstTime) {
393 		this._listView._setNoResultsHtml();
394 	}
395 
396 	more = more || (this._offset + ZmContactsApp.SEARCHFOR_MAX) < this._list.size();
397 	this._listView.set(list);
398 };

425 ZmContactSearchListView.prototype._getHeaderList =
426 function() {
427 	var headerList = [];
428 	headerList.push(new DwtListHeaderItem({field:ZmItem.F_TYPE, width:ZmMsg.COLUMN_WIDTH_FOLDER_CN}));
429 	headerList.push(new DwtListHeaderItem({field:ZmItem.F_NAME, text:ZmMsg._name, width:ZmMsg.COLUMN_WIDTH_NAME_CN}));
430 	headerList.push(new DwtListHeaderItem({field:ZmItem.F_EMAIL, text:ZmMsg.email}));
431 
432 	return headerList;
433 };

435 /**
436  * @private
437  */
438 ZmContactSearchListView.prototype._getCellContents =
439 function(htmlArr, idx, contact, field, colIdx, params) {
440 	if (field == ZmItem.F_TYPE) {
441 		htmlArr[idx++] = AjxImg.getImageHtml(contact.getIcon());
442 	} else if (field == ZmItem.F_NAME) {
443 		htmlArr[idx++] = AjxStringUtil.htmlEncode(contact.getFileAs());
444 	} else if (field == ZmItem.F_EMAIL) {
445 		htmlArr[idx++] = AjxStringUtil.htmlEncode(contact.getEmail());
446 	}
447 	return idx;
448 };
