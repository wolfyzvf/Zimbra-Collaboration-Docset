 26 /**
 27  * Creates a folder properties view for the folder dialog
 28  * @class
 29  * This class represents a dialog tab view displayed by a tabToolbar
 30  * 
 31  * @param	{DwtControl}	parent		    the parent (dialog)
 32  * @param	{String}	    className		the class name
 33  * 
 34  * @extends		DwtComposite
 35  */
 36 ZmFolderDialogTabView = function(parent, className) {
 37     if (arguments.length == 0) return;
 38 
 39     DwtTabViewPage.call(this, parent, className, Dwt.RELATIVE_STYLE);
 40 
 41     this._createView();
 42 };
 43 
 44 ZmFolderDialogTabView.prototype = new DwtTabViewPage;
 45 ZmFolderDialogTabView.prototype.constructor = ZmFolderDialogTabView;

 78 ZmFolderDialogTabView.prototype._createCheckboxItem =
 79 function(name, label) {
 80     var checkboxName  = "_" + name + "Checkbox"
 81     var containerName = "_" + name + "El"
 82 
 83     this[checkboxName] = document.createElement("INPUT");
 84     this[checkboxName].type = "checkbox";
 85     this[checkboxName]._dialog = this;
 86     this[checkboxName].id = checkboxName;
 87 
 88     this[containerName] = document.createElement("DIV");
 89     this[containerName].style.display = "none";
 90     this[containerName].appendChild(this[checkboxName]);
 91     var lbl = document.createElement("label");
 92     lbl.innerHTML = label;
 93     lbl.htmlFor = checkboxName;
 94     this[containerName].appendChild(lbl);
 95 
 96     return this[containerName];
 97 }

 99 ZmFolderDialogTabView.prototype._createBusyOverlay =
100 function(htmlElement) {
101     this._busyOverlay = document.createElement("div");
102     this._busyOverlay.className = "ZmDialogTabViewBusy";
103     this._busyOverlay.style.position = "absolute";
104     Dwt.setBounds(this._busyOverlay, 0, 0, "100%", "100%")
105     Dwt.setZIndex(this._busyOverlay, Dwt.Z_VEIL);
106     this._busyOverlay.innerHTML = "<table cellspacing=0 cellpadding=0 style='width:100%; height:100%'><tr><td> </td></tr></table>";
107     htmlElement.appendChild(this._busyOverlay);
108 	Dwt.setVisible(this._busyOverlay, false);
109 
110     this._setBusyFlag = false;
111 }
112 
113 ZmFolderDialogTabView.prototype._setBusy =
114 function(busy) {
115     if (!this._setBusyFlag) {
116 		// transition from non-busy to busy state
117 		Dwt.setCursor(this._busyOverlay, "wait");
118     	Dwt.setVisible(this._busyOverlay, true);
119     	this._setBusyFlag = this._blockInput = true;
120     } else if (this._setBusy) {
121 		// transition from busy to non-busy state
122 	    Dwt.setCursor(this._busyOverlay, "default");
123 	    Dwt.setVisible(this._busyOverlay, false);
124 	    this._setBusyFlag = this._blockInput = false;
125 	}
126 }
