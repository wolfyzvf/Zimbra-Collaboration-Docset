 28 /**
 29  * Creates a toolbar.
 30  * @constructor
 31  * @class
 32  * Creates a toolbar. Components must be added via the <code>add*()</code> functions.
 33  * A toolbar is a horizontal or vertical strip of widgets (usually buttons).
 34  *
 35  * @author Ross Dargahi
 36  * 
 37  * @param {hash}	params		a hash of parameters
 38  * @param	{DwtComposite}	params.parent	the parent widget
 39  * @param	{string}	params.className				the CSS class
 40  * @param	{DwtToolBar.HORIZ_STYLE|DwtToolBar.VERT_STYLE}	params.posStyle		the positioning style
 41  * @param	{constant}	params.style					the menu style
 42  * @param	{number}	params.index 				the index at which to add this control among parent's children
 43  * 
 44  * @extends	DwtComposite
 45  */
 46 DwtToolBar = function(params) {
 47 	if (arguments.length == 0) { return; }
 48 	params = Dwt.getParams(arguments, DwtToolBar.PARAMS);
 49 
 50 	params.className = params.className || "ZToolbar";
 51 	DwtComposite.call(this, params);
 52 
 53 	// since we attach event handlers at the toolbar level, make sure we don't double up on
 54 	// handlers when we have a toolbar within a toolbar
 55 	if (params.parent instanceof DwtToolBar) {
 56 		this._hasSetMouseEvents = params.parent._hasSetMouseEvents;
 57 	}
 58 	if (params.handleMouse !== false && !this._hasSetMouseEvents) {
 59 		var events = AjxEnv.isIE ? [DwtEvent.ONMOUSEDOWN, DwtEvent.ONMOUSEUP] :
 60 								   [DwtEvent.ONMOUSEDOWN, DwtEvent.ONMOUSEUP, DwtEvent.ONMOUSEOVER, DwtEvent.ONMOUSEOUT];
 61 		this._setEventHdlrs(events);
 62 		this._hasSetMouseEvents = true;
 63 	}
 64 
 65 	this._style = params.style || DwtToolBar.HORIZ_STYLE;
 66     this._items = [];
 67     this._createHtml();
 68 
 69     this._numFillers = 0;
 70 	this._curFocusIndex = 0;
 71 
 72 	this._keyMapName = (this._style == DwtToolBar.HORIZ_STYLE) ? DwtKeyMap.MAP_TOOLBAR_HORIZ : DwtKeyMap.MAP_TOOLBAR_VERT;
 73 };
 74

259 DwtToolBar.prototype.handleKeyAction =
260 function(actionCode, ev) {
261 
262 	var item = this.getItem(this._curFocusIndex);
263 	var numItems = this.getItemCount();
264 	if (numItems < 2) {
265 		return true;
266 	}
267 
268 	switch (actionCode) {
269 
270 		case DwtKeyMap.PREV:
271 			if (this._curFocusIndex > 0) {
272 				this._moveFocus(true);
273 			}
274 			break;
275 
276 		case DwtKeyMap.NEXT:
277 			if (this._curFocusIndex < (numItems - 1)) {
278 				this._moveFocus();
279 			}
280 			break;
281 
282 		default:
283 			// pass everything else to currently focused item
284 			if (item) {
285 				return item.handleKeyAction(actionCode, ev);
286 			}
287 	}
288 	return true;
289 };

365 /**
366  * @private
367  */
368 DwtToolBar.prototype._addItem =
369 function(type, element, index) {
370 
371     // get the reference element for insertion
372     var placeEl = this._items[index] || this._suffixEl;
373 
374     // insert item
375 	var spliceIndex = index || (typeof index == "number") ? index : this._items.length;
376 	this._items.splice(spliceIndex, 0, element);
377     this._itemsEl.insertBefore(element, placeEl);
378 
379     // append spacer
380     // TODO!
381 };

383 /**
384  * @private
385  */
386 DwtToolBar.prototype._removeItem =
387 function(item) {
388 	for (var i = 0; i < this._items.length; i++) {
389 		if (this._items[i] == item) {
390 			this._items.splice(i,1);
391 			this._itemsEl.removeChild(item);
392 			break;
393 		}
394 	}
395 };

397 /**
398  * transfer focus to the current item
399  * @private
400  */
401 DwtToolBar.prototype._focus =
402 function(item) {
403 	DBG.println(AjxDebug.DBG3, "DwtToolBar: FOCUS");
404 	// make sure the key for expanding a button submenu matches our style
405 	if (!this._submenuKeySet) {
406 		var kbm = this.shell.getKeyboardMgr();
407 		if (kbm.isEnabled()) {
408 			var kmm = kbm.__keyMapMgr;
409 			if (kmm) {
410 				if (this._style == DwtToolBar.HORIZ_STYLE) {
411 					kmm.removeMapping(DwtKeyMap.MAP_BUTTON, "ArrowRight");
412 					kmm.setMapping(DwtKeyMap.MAP_BUTTON, "ArrowDown", DwtKeyMap.SUBMENU);
413 				} else {
414 					kmm.removeMapping(DwtKeyMap.MAP_BUTTON, "ArrowDown");
415 					kmm.setMapping(DwtKeyMap.MAP_BUTTON, "ArrowRight", DwtKeyMap.SUBMENU);
416 				}
417 				kmm.reloadMap(DwtKeyMap.MAP_BUTTON);
418 			}
419 		}
420 		this._submenuKeySet = true;
421 	}
422 
423 	item = item ? item : this._getFocusItem(this._curFocusIndex);
424 	if (item) {
425 		item._hasFocus = true;	// so that focus class is set
426 		item._focus();
427 	} else {
428 		// if current item isn't focusable, find first one that is
429 		this._moveFocus();
430 	}
431 };

448 /**
449  * Returns the item at the given index, as long as it can accept focus.
450  * For now, we only move focus to simple components like buttons. Also,
451  * the item must be enabled and visible.
452  *
453  * @param {number}	index		the index of item within toolbar
454  * @return	{Object}	the item
455  * 
456  * @private
457  */
458 DwtToolBar.prototype._getFocusItem =
459 function(index) {
460 	var item = this.getItem(index);
461 	if (!item || (item instanceof DwtToolBar))	{ return null; }
462 	if (item._noFocus)							{ return null; }
463 	if (item.getEnabled && !item.getEnabled())	{ return null; }
464 	if (item.getVisible && !item.getVisible())	{ return null; }
465 	return item;
466 };
467 
468 /**
469  * Moves focus to next or previous item that can take focus.
470  *
471  * @param {boolean}	back		if <code>true</code>, move focus to previous item
472  * 
473  * @private
474  */
475 DwtToolBar.prototype._moveFocus =
476 function(back) {
477 	var index = this._curFocusIndex;
478 	var maxIndex = this.getItemCount() - 1;
479 	var item = null;
480 	while (!item && index >= 0 && index <= maxIndex) {
481 		index = back ? index - 1 : index + 1;
482 		item = this._getFocusItem(index);
483 	}
484 	if (item) {
485 		this._blur();
486 		this._curFocusIndex = index;
487 		this._focus(item);
488 	}
489 };

491 /**
492  * @private
493  */
494 DwtToolBar.prototype.__markPrevNext =
495 function(id, opened) {
496     var index = this.__getButtonIndex(id);
497     var prev = this.__getButtonAt(index - 1);
498     var next = this.__getButtonAt(index + 1);
499     if (opened) {
500         if (prev) Dwt.delClass(prev.getHtmlElement(), DwtToolBar._NEXT_PREV_RE, DwtToolBar.SELECTED_PREV);
501         if (next) Dwt.delClass(next.getHtmlElement(), DwtToolBar._NEXT_PREV_RE, DwtToolBar.SELECTED_NEXT);
502     }
503     else {
504         if (prev) Dwt.delClass(prev.getHtmlElement(), DwtToolBar._NEXT_PREV_RE);
505         if (next) Dwt.delClass(next.getHtmlElement(), DwtToolBar._NEXT_PREV_RE);
506     }
507 
508     // hack: mark the first and last items so we can style them specially
509     //	MOW note: this should really not be here, as it only needs to be done once,
510     //				but I'm not sure where to put it otherwise
511     var first = this.__getButtonAt(0);
512     if (first) Dwt.addClass(first.getHtmlElement(), DwtToolBar.FIRST_ITEM);
513 
514     var last = this.__getButtonAt(this.getItemCount()-1);
515     if (last) Dwt.addClass(last.getHtmlElement(), DwtToolBar.LAST_ITEM);
516 };

518 /**
519  * Find the array index of a toolbar button by id.
520  *
521  * Works only if descendent classes implement the _buttons property as a
522  * native Array.
523  * @param id {number} Index to check and see if exists in the array.
524  * @return {number} Index of the id in the array, or -1 if the id does not
525  * exist.
526  * @private
527  */
528 DwtToolBar.prototype.__getButtonIndex =
529 function(id) {
530     var toolBarButtons = this.getChildren();
531     var button = this._buttons[id];
532     if (toolBarButtons && toolBarButtons.length && button)
533         return AjxUtil.indexOf(toolBarButtons, button);
534     return -1;
535 };
536 
537 /**
538  * Find a toolbar button by id.
539  *
540  * Works only if descendent classes implement the _buttons property as a
541  * native Array.
542  * @param index {number} The integer index of the button to retrieve.
543  * @return {DwtButton} The DWT button at the current index, or null if the
544  * buton does not exist.
545  * @private
546  */
547 DwtToolBar.prototype.__getButtonAt =
548 function(index) {
549     var i = 0;
550     // NOTE: _buttons seems to always be implemented as an Array.
551     // This code should not be needed because:
552     // * If we're working with Objects-as-associative-arrays, we don't want
553     //   numeric indexes.
554     // * If we're working with arrays, id will ALWAYS be i if found, or
555     //   undefined if not found. This function could be done via a simple
556     //   return this._buttons && this._buttons[index];
557     //   or something close to that if falsey values might be valid.
558     for (var name in this._buttons) {
559         // NOTE: Protect from native Array.prototype extensions
560         if (this._buttons.hasOwnProperty(name)) {
561             if (i == index) {
562                 return this._buttons[name];
563             }
564             i++;
565         }
566     }
567     return null;
568 };
