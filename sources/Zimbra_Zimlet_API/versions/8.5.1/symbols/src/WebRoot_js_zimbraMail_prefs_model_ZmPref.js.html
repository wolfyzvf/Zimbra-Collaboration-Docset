 74 ZmPref.loadCsvFormats =
 75 function(setup){
 76     var formats = appCtxt.get(ZmSetting.AVAILABLE_CSVFORMATS);
 77 	if (!formats._options) {
 78 		var options = formats._options = [];
 79 		var displayOptions = formats._displayOptions = [];
 80 		for(var i=0; i<formats.length; i++){
 81 			options.push(formats[i]);
 82 		}
 83 		options.sort(ZmPref.__BY_CSVFORMAT);
 84 		for(var i=0; i < options.length; i++){
 85 			displayOptions.push((ZmMsg[options[i]] || options[i]));
 86 		}
 87 	}
 88 	setup.options = formats._options;
 89 	setup.displayOptions = formats._displayOptions;
 90 };

 94 	if (a.match(/^yahoo/))  return -1;
 95 	if (b.match(/^yahoo/))  return  1;

211 ZmPref.approximateValue =
212 function(sortedValues, value) {
213 	// find closest value
214 	for (var i = 0; i < sortedValues.length + 1; i++) {
215 		var a = sortedValues[i];
216 		var b = sortedValues[i+1];
217 		if (value < b) {
218 			var da = value - a;
219 			var db = b - value;
220 			return da < db ? a : b;
221 		}
222 	}
223 	return sortedValues[sortedValues.length - 1];
224 };

272 ZmPref.approximateLifetime =
273 function(prefId, duration, validateFunc1/*, ..., validateFuncN*/) {
274 	// convert durations to seconds
275 	var values = [].concat(ZmPref.SETUP[prefId].options);
276 	for (var i = 0; i < values.length; i++) {
277 		var value = values[i];
278 		values[i] = ZmPref.__DUR2SECS(value != "0" ? value+"d" : value);
279 	}
280 	values.sort(ZmPref.__BY_NUMBER);
281 
282 	// remove invalid options
283 	valuesLoop: for (var i = values.length - 1; i >= 0; i--) {
284 		for (var j = 2; j < arguments.length; j++) {
285 			var validateFunc = arguments[j];
286 			var value = ZmPref.__SECS2DUR(values[i]);
287 			if (!validateFunc(value)) {
288 				values.pop();
289 				continue valuesLoop;
290 			}
291 		}
292 		break;
293 	}
294 
295 	// if zero, the closest match is the greatest
296 	var seconds;
297 	if (duration == "0") {
298 		seconds = values[values.length - 1];
299 	}
300 
301 	// approximate to closest number of seconds
302 	else {
303 		seconds = ZmPref.approximateValue(values, ZmPref.__DUR2SECS(duration+"d"));
304 	}
305 
306 	// convert back to duration
307 	duration = ZmPref.__SECS2DUR(seconds);
308 	return duration != "0"  ? parseInt(duration, 10) : 0;
309 };

448 ZmPref.__BY_NUMBER =
449 function(a, b) {
450 	if (a == b) return 0;
451 	if (a == Math.POSITIVE_INFINITY || b == Math.NEGATIVE_INFINITY) return 1;
452 	if (b == Math.POSITIVE_INFINITY || a == Math.NEGATIVE_INFINITY) return -1;
453 	return Number(a) - Number(b);
454 };
455 
456 ZmPref.__BY_DURATION =
457 function(a, b) {
458 	if (a == b) return 0;
459 	if (a == "0") return 1;
460 	if (b == "0") return -1;
461 	var asecs = ZmPref.__DUR2SECS(a);
462 	var bsecs = ZmPref.__DUR2SECS(b);
463 	return asecs - bsecs;
464 };

468 ZmPref.__DURMULT = { "s": 1, "m": 60, "h": 3600, "d": 86400/*, "w": 604800*/ };
469 ZmPref.__DURDIV = { /*604800: "w",*/ 86400: "d", 3600: "h", 60: "m", 1: "s" };
470 
471 ZmPref.__DUR2SECS =
472 function(duration) {
473 	if (duration == "0") return Number.POSITIVE_INFINITY;
474 
475 	var type = duration.substring(duration.length - 1).toLowerCase();
476 	return parseInt(duration, 10) * ZmPref.__DURMULT[type];
477 };
478 
479 ZmPref.__SECS2DUR =
480 function(seconds, type) {
481 	if (seconds == Number.POSITIVE_INFINITY) return "0";
482 
483 	var divisors = ZmPref.__DURDIV;
484 	if (type) {
485 		type = {};
486 		type[ ZmPref.__DURMULT[type] ] = type;
487 	}
488 	for (var divisor in divisors) {
489 		var result = Math.floor(seconds / divisor);
490 		if (result > 0) {
491 			return [ result, divisors[divisor] ].join("");
492 		}
493 	}
494 
495 	return "0"+type;
496 };

537 /**
538  * Registers a preferences section ("tab").
539  * <p>
540  * The <code>params</code> argument can have the following properties:
541  *
542  * @param title         [string]    The section title.
543  * @param templateId    [string]    (Optional) The template associated to this
544  *                                  section. If not specified, the id is used.
545  *                                  Note: The default template base is:
546  *                                  "prefs.Pages#".
547  * @param priority      [int]       The section priority used when determining
548  *                                  the order of the sections.
549  * @param precondition  [any]       (Optional) Specifies the precondition
550  *                                  under which this section is shown. If this
551  *                                  parameter is a function, it returns a
552  *                                  boolean signifying that this section is
553  *                                  present -- true to show this section. If
554  *                                  this parameter is <em>not</em> a function,
555  *                                  then it is considered a setting identifier
556  *                                  that determines the presence of this
557  *                                  section.
558  * @param prefs         [Array]     List of preferences that appear in this
559  *                                  section.
560  * @param manageChanges [boolean]   Determines whether this section manages
561  *                                  its own pref changes. If true, then
562  *                                  ZmPrefView#getChangedPrefs will not query
563  *                                  the section for changes.
564  * @param manageDirty   [boolean]   Determines whether this section manages
565  *                                  its "dirty" state (i.e. whether any prefs
566  *                                  have changed values). If true, then
567  *                                  ZmPrefView#getChangedPrefs will call
568  *                                  isDirty() on the section view.
569  */
570 ZmPref.registerPrefSection =
571 function(id, params) {
572 	if (!id || !params) { return; }
573 
574 	// set template for section
575 	var templateId = params.templateId || id;
576 	if (!templateId.match(/#/)) {
577 		templateId = ["prefs.Pages",templateId].join("#");
578 	}
579 	params.templateId = templateId;
580 	params.id = id;
581 
582 	// save section
583 	appCtxt.set(ZmSetting.PREF_SECTIONS, params, id);
584 	ZmPref._prefSectionArray = null;
585 };

599 /** Returns a sorted array of pref sections (based on priority). */
600 ZmPref.getPrefSectionArray =
601 function() {
602 	if (!ZmPref._prefSectionArray) {
603 		ZmPref._prefSectionArray = [];
604 		var prefSectionMap = appCtxt.get(ZmSetting.PREF_SECTIONS);
605 		for (var id in prefSectionMap) {
606 			ZmPref._prefSectionArray.push(prefSectionMap[id]);
607 		}

672 ZmPref._normalizeFontId = function(id, dontFallback) {
673 	var oldid = id;
674 	id = id.replace(/,\s/g,",").replace(/'/g,"").toLowerCase(); // Make sure all ids that are supposed to be found in ZmPref.FONT_FAMILY are actually found
675 	if (!dontFallback) {
676 		var map = ZmPref.FONT_FAMILY;
677 		if (map && !map[id]) {
678 			var keys = AjxUtil.keys(map);
679 			if (keys.length) {
680 				var splitId = id.split(","); // e.g. ["times new roman","helvetica"]
681 				for (var i=0; i<splitId.length; i++) { // Loop over input font names
682 					for (var j=0; j<keys.length; j++) { // Loop over candidate styles, e.g. ["arial,sans-serif","times new roman,serif"]
683 						if (keys[j].indexOf(splitId[i]) != -1) {
684 							return keys[j];
685 						}
686 					}
687 				}
688 				return keys[0];
689 			}
690 		}
691 	}
692 	return id;
693 };
694 ZmPref._normalizeFontName = function(fontId) {
695 	return ZmPref.FONT_FAMILY[ZmPref._normalizeFontId(fontId)].name;
696 };
697 ZmPref._normalizeFontValue = function(fontId) {
698 	return ZmPref.FONT_FAMILY[ZmPref._normalizeFontId(fontId)].value;
699 };
