 27 /**
 28  * Creates an empty contact list controller.
 29  * @class
 30  * This class manages list views of contacts. So far there are two different list
 31  * views, one that shows the contacts in a traditional list format, and the other
 32  * which shows them as business cards. Since there are two views, we need to keep
 33  * track of which is the current view.
 34  *
 35  * @author Roland Schemers
 36  * @author Conrad Damon
 37  * 
 38  * @param {DwtControl}					container					the containing shell
 39  * @param {ZmApp}						app							the containing application
 40  * @param {constant}					type						type of controller
 41  * @param {string}						sessionId					the session id
 42  * @param {ZmSearchResultsController}	searchResultsController		containing controller
 43  * 
 44  * @extends	ZmListController
 45  */
 46 ZmContactListController = function(container, contactsApp, type, sessionId, searchResultsController) {
 47 
 48 	if (arguments.length == 0) { return; }
 49 	ZmListController.apply(this, arguments);
 50 
 51 	this._viewFactory = {};
 52 	this._viewFactory[ZmId.VIEW_CONTACT_SIMPLE] = ZmContactSplitView;
 53 
 54 	if (this.supportsDnD()) {
 55 		this._dragSrc = new DwtDragSource(Dwt.DND_DROP_MOVE);
 56 		this._dragSrc.addDragListener(this._dragListener.bind(this));
 57 	}
 58 
 59 	this._listChangeListener = this._handleListChange.bind(this);
 60 
 61 	this._listeners[ZmOperation.EDIT]			= this._editListener.bind(this);
 62 	this._listeners[ZmOperation.PRINT]			= null; // override base class to do nothing
 63 	this._listeners[ZmOperation.PRINT_CONTACT]	= this._printListener.bind(this);
 64 	this._listeners[ZmOperation.PRINT_ADDRBOOK]	= this._printAddrBookListener.bind(this);
 65 	this._listeners[ZmOperation.NEW_GROUP]		= this._groupListener.bind(this);
 66 
 67 	this._parentView = {};
 68 };

 91 /**
 92  * Shows the search results.
 93  * 
 94  * @param	{Object}	searchResult		the search results
 95  * @param	{Boolean}	isGalSearch		<code>true</code> if results from GAL search
 96  * @param	{String}	folderId		the folder id
 97  */
 98 ZmContactListController.prototype.show =
 99 function(searchResult, isGalSearch, folderId) {
100 
101 	this._searchType = isGalSearch
102 		? ZmContactListController.SEARCH_TYPE_GAL
103 		: ZmContactListController.SEARCH_TYPE_CANONICAL;
104 
105 	this._folderId = folderId;
106 	var selectedContacts;
107 	
108 	if (searchResult.isZmContactList) {
109 		this.setList(searchResult);			// set as canonical list of contacts
110 		this._list._isShared = false;		// this list is not a search of shared items
111 		selectedContacts = this._listView[this._currentViewId] && this._listView[this._currentViewId].getSelection();
112 		this._contactSearchResults = false;
113     }
114 	else if (searchResult.isZmSearchResult) {
115 		this._searchType |= ZmContactListController.SEARCH_TYPE_NEW;
116 		this.setList(searchResult.getResults(ZmItem.CONTACT));
117 
118 		// HACK - find out if user did a "is:anywhere" search (for printing)
119 		if (searchResult.search && searchResult.search.isAnywhere()) {
120 			this._searchType |= ZmContactListController.SEARCH_TYPE_ANYWHERE;
121 		}
122 
123 		if (searchResult.search && searchResult.search.userText && this.getCurrentView()) {
124 			this.getCurrentView().getAlphabetBar().reset();
125 		}
126 
127 		if (isGalSearch) {
128 			this._list = this._list || new ZmContactList(searchResult.search, true);
129 			this._list._isShared = false;
130 			this._list.isGalPagingSupported = AjxUtil.isSpecified(searchResult.getAttribute("offset"));
131 		} else {
132 			// find out if we just searched for a shared address book
133 			var addrbook = folderId ? appCtxt.getById(folderId) : null;
134 			this._list._isShared = addrbook ? addrbook.link : false;
135 		}
136 
137 		this._list.setHasMore(searchResult.getAttribute("more"));
138 
139 		selectedContacts = this._listView[this._currentViewId] && this._listView[this._currentViewId].getSelection();
140 		ZmListController.prototype.show.apply(this, [searchResult, this._currentViewId]);
141 		this._contactSearchResults = true;
142 	}
143 
144 	// reset offset if list view has been created
145 	var view = this._currentViewId;
146 	if (this._listView[view]) {
147 		this._listView[view].offset = 0;
148 	}
149 	this.switchView(view, true);
150 
151 	if (selectedContacts && selectedContacts.length && this._listView[view]) {
152 		this._listView[view].setSelection(selectedContacts[0]);
153 	}
154 };

158 /**
159  * Change how contacts are displayed. There are two views: the "simple" view
160  * shows a list of contacts on the left and the selected contact on the right;
161  * the "cards" view shows contacts as business cards.
162  * 
163  * @param {constant}	view			the view to show
164  * @param {Boolean}	force			if <code>true</code>, render view even if it's the current view
165  * @param {Boolean}	initialized		if <code>true</code>, app has been initialized
166  * @param {Boolean}	stageView		if <code>true</code>, stage the view but don't push it
167  */
168 ZmContactListController.prototype.switchView =
169 function(view, force, initialized, stageView) {
170 	if (view && ((view != this._currentViewId) || force)) {
171 		this._currentViewId = view;
172 		DBG.timePt("setting up view", true);
173 		this._setup(view);
174 		DBG.timePt("done setting up view");
175 
176 		var elements = this.getViewElements(view, this._parentView[view]);
177 
178 		// call initialize before _setView since we havent set the new view yet
179 		if (!initialized) {
180 			this._initializeAlphabetBar(view);
181 		}
182 
183 		this._setView({ view:		view,
184 						viewType:	this._currentViewType,
185 						noPush:		this.isSearchResults,
186 						elements:	elements,
187 						isAppView:	true,
188 						stageView:	stageView});
189 		if (this.isSearchResults) {
190 			// if we are switching views, make sure app view mgr is up to date on search view's components
191 			appCtxt.getAppViewMgr().setViewComponents(this.searchResultsController.getCurrentViewId(), elements, true);
192 		}
193 		this._resetNavToolBarButtons();
194 
195 		// HACK: reset search toolbar icon (its a hack we're willing to live with)
196 		if (this.isGalSearch() && !this._list.isGalPagingSupported) {
197 			appCtxt.getSearchController().setDefaultSearchType(ZmId.SEARCH_GAL);
198 			if (this._list.hasMore()) {
199 				var d = appCtxt.getMsgDialog();
200 				d.setMessage(ZmMsg.errorSearchNotExpanded);
201 				d.popup();
202 			}
203 		}
204 
205 		this._setTabGroup(this._tabGroups[view]);
206 
207 		if (!initialized) {
208 			var list = this._listView[view].getList();
209 			if (list) {
210 				this._listView[view].setSelection(list.get(0));
211 			}
212 		}
213 	}
214 };

247 /**
248  * Search the alphabet.
249  * 
250  * @param	{String}	letter		the letter
251  * @param	{String}	endLetter	the end letter
252  */
253 ZmContactListController.prototype.searchAlphabet =
254 function(letter, endLetter) {
255 	var folderId = this._folderId || ZmFolder.ID_CONTACTS;
256 	var folder = appCtxt.getById(folderId);
257 	var query = folder ? folder.createQuery() : null;
258 
259 	if (query) {
260 		var params = {
261 			query: query,
262 			types: [ZmItem.CONTACT],
263 			offset: 0,
264 			limit: (this._listView[this._currentViewId].getLimit()),
265 			lastId: 0,
266 			lastSortVal: letter,
267 			endSortVal: endLetter
268 		};
269 		appCtxt.getSearchController().search(params);
270 	}
271 };

286 ZmContactListController.prototype.handleKeyAction =
287 function(actionCode) {
288 	DBG.println(AjxDebug.DBG3, "ZmContactListController.handleKeyAction");
289     var isExternalAccount = appCtxt.isExternalAccount();
290 	var isWebClientOffline = appCtxt.isWebClientOffline();
291 	switch (actionCode) {
292 
293 		case ZmKeyMap.EDIT:
294             if (isExternalAccount || isWebClientOffline) { break; }
295 			this._editListener();
296 			break;
297 
298 		case ZmKeyMap.PRINT:
299 			if (appCtxt.get(ZmSetting.PRINT_ENABLED) && !isWebClientOffline) {
300 				this._printListener();
301 			}
302 			break;
303 
304 		case ZmKeyMap.PRINT_ALL:
305 			if (appCtxt.get(ZmSetting.PRINT_ENABLED) && !isWebClientOffline) {
306 				this._printAddrBookListener();
307 			}
308 			break;
309 
310 		case ZmKeyMap.NEW_MESSAGE:
311 			if (isExternalAccount) { break; }
312 			this._composeListener();
313 			break;
314 
315 		default:
316 			return ZmListController.prototype.handleKeyAction.call(this, actionCode);
317 	}
318 	return true;
319 };

575 ZmContactListController.prototype._handleListChange =
576 function(ev) {
577 	if (ev.event == ZmEvent.E_MODIFY || ev.event == ZmEvent.E_CREATE) {
578 		if (!ev.getDetail("visible")) {
579 			return;
580 		}
581 		var items = ev.getDetail("items");
582 		var item = items && items.length && items[0];
583 		if (item instanceof ZmContact && this._currentViewType == ZmId.VIEW_CONTACT_SIMPLE && item.folderId == this._folderId) {
584 			var alphaBar = this._parentView[this._currentViewId].getAlphabetBar();
585 			//only set the view if the contact is in the list
586 			if(!alphaBar || alphaBar.isItemInAlphabetLetter(item)) {
587 				this._parentView[this._currentViewId].setContact(item, this.isGalSearch());
588 			}
589 		}
590 	}
591 };

593 /**
594  * Create menu for View button and add listeners.
595  * 
596  * @private
597  */
598 ZmContactListController.prototype._setupViewMenu =
599 function(view, firstTime) {
600 	var btn;
601 
602 	if (firstTime) {
603 		btn = this._toolbar[view].getButton(ZmOperation.VIEW_MENU);
604 		var menu = btn.getMenu();
605 		if (!menu) {
606 			menu = new ZmPopupMenu(btn);
607 			btn.setMenu(menu);
608 			for (var i = 0; i < ZmContactListController.VIEWS.length; i++) {
609 				var id = ZmContactListController.VIEWS[i];
610 				var mi = menu.createMenuItem(id, {image:ZmContactListController.ICON[id],
611 													text:ZmMsg[ZmContactListController.MSG_KEY[id]],
612 													style:DwtMenuItem.RADIO_STYLE});
613 				mi.setData(ZmOperation.MENUITEM_ID, id);
614 				mi.addSelectionListener(this._listeners[ZmOperation.VIEW]);
615 				if (id == view)
616 					mi.setChecked(true, true);
617 			}
618 		}
619 	} else {
620 		// always set the switched view to be the checked menu item
621 		btn = this._toolbar[view].getButton(ZmOperation.VIEW_MENU);
622 		var menu = btn ? btn.getMenu() : null;
623 		var mi = menu ? menu.getItemById(ZmOperation.MENUITEM_ID, view) : null;
624 		if (mi) { mi.setChecked(true, true); }
625 	}
626 
627 	// always reset the view menu button icon to reflect the current view
628 	btn.setImage(ZmContactListController.ICON[view]);
629 };

1076 
1077 	items = AjxUtil.toArray(items);
1078 
1079 	var move = [];
1080 	var copy = [];
1081 	var moveFromGal = [];
1082 	for (var i = 0; i < items.length; i++) {
1083 		var item = items[i];
1084 		if (item.isGal) {

1086 		} else if (!item.folderId || item.folderId != folder.id) {
1087 			if (!this._isItemMovable(item, isShiftKey, folder)) {

1091 			}
1092 		}
1093 	}
1094

1096 	var outOfTrash = (moveOutFolder && moveOutFolder.isInTrash() && !folder.isInTrash());
1097 
1098     var allDoneCallback = this._getAllDoneCallback();
1099 	if (move.length) {
1100         var params = {items:move, folder:folder, attrs:attrs, outOfTrash:outOfTrash};
1101 		var list = params.list = this._getList(params.items);
1102         this._setupContinuation(this._doMove, [folder, attrs, isShiftKey], params, allDoneCallback);
1103         list = outOfTrash ? this._list : list;
1104 		list.moveItems(params);
1105 	}
1106 
1107 	if (copy.length) {
1108         var params = {items:copy, folder:folder, attrs:attrs};
1109 		var list = params.list = this._getList(params.items);
1110         this._setupContinuation(this._doMove, [folder, attrs, isShiftKey], params, allDoneCallback);
1111         list = outOfTrash ? this._list : list;
1112 		list.copyItems(params);
1113 	}
1114 
1115 	if (moveFromGal.length) {
1116 		var batchCmd = new ZmBatchCommand(true, null, true);
1117 		for (var j = 0; j < moveFromGal.length; j++) {
1118 			var contact = moveFromGal[j];
1119 			contact.attr[ZmContact.F_folderId] = folder.id;
1120 			batchCmd.add(new AjxCallback(contact, contact.create, [contact.attr]));
1121 		}
1122 		batchCmd.run(new AjxCallback(this, this._handleMoveFromGal));
1123 	}
1124 };

1285 	var memberType;

1332 		//handle potential duplicates

1378 ZmContactListController.prototype._sortContactGroups =
1379 function(contactGroups) {
1380 	var sortByNickname = function(a, b) {
1381 		var aNickname = ZmContact.getAttr(a, "nickname");
1382 		var bNickname = ZmContact.getAttr(b, "nickname");
1383 
1384 		if (!aNickname || !bNickname) {
1385 			return 0;
1386 		}
1387 
1388 		if (aNickname.toLowerCase() > bNickname.toLowerCase())
1389 			return 1;
1390 		if (aNickname.toLowerCase() < bNickname.toLowerCase())
1391 			return -1;
1392 
1393 		return 0;
1394 	};
1395 
1396 	return contactGroups.sort(sortByNickname);
