 16 ZmAddressObjectHandler = function() {
 17 
 18 	ZmObjectHandler.call(this, ZmAddressObjectHandler.TYPE);
 19 }
 20 
 21 ZmAddressObjectHandler.prototype = new ZmObjectHandler;
 22 ZmAddressObjectHandler.prototype.constructor = ZmAddressObjectHandler;
 23 
 24 // Consts
 25 ZmAddressObjectHandler.TYPE = "address";
 26 // TODO This regex is very very simple.  It only matches single line simple addresses like:
 27 // 1234 Main St City CA 99999
 28 ZmAddressObjectHandler.ADDRESS_RE = /[\w]{3,}([A-Za-z]\.)?([ \w]*\#\d+)?(\r\n| )[ \w]{3,}\x20[A-Za-z]{2}\x20\d{5}(-\d{4})?\b/ig;
 29 
 30 // Y! maps geocoder, since Google doesn't have one.
 31 ZmAddressObjectHandler.URL = "http://api.local.yahoo.com/MapsService/V1/geocode?appid=ZimbraMail&location=";
 32 
 33 // Make DOM safe id's
 34 ZmAddressObjectHandler.encodeId = function(s) {
 35 	return s.replace(/[^A-Za-z0-9]/g, "");
 36 };
 37 
 38 ZmAddressObjectHandler.CACHE = new Array();
 39 
 40 // Public methods
 41 
 42 ZmAddressObjectHandler.prototype.match =
 43 function(line, startIndex) {
 44 	ZmAddressObjectHandler.ADDRESS_RE.lastIndex = startIndex;
 45 	return ZmAddressObjectHandler.ADDRESS_RE.exec(line);
 46 };
 47 
 48 ZmAddressObjectHandler.prototype._getHtmlContent =
 49 function(html, idx, address, context) {
 50 	var	url = "http://maps.google.com/maps?q="+AjxStringUtil.htmlEncode(address);
 51     html[idx++] = '<a target="_blank" href="'+url+'">'+AjxStringUtil.htmlEncode(address)+'</a>';	
 52 	return idx;
 53 };
 54 
 55 ZmAddressObjectHandler.prototype.getToolTipText =
 56 function(obj, context) {
 57 	return '<div class="AddressContent" id="'+ZmAddressObjectHandler.encodeId(obj)+'"> </div>';
 58 };
 59 
 60 ZmAddressObjectHandler.prototype.populateToolTip =
 61 function(obj, context) {
 62 	if(ZmAddressObjectHandler.CACHE[obj+"lng"] && ZmAddressObjectHandler.CACHE[obj+"lat"]) {
 63 		ZmAddressObjectHandler.displayMap(ZmAddressObjectHandler.CACHE[obj+"lng"], 
 64 		                                  ZmAddressObjectHandler.CACHE[obj+"lat"], obj);
 65 	} else {
 66 		var url = "/service/proxy?target=" + AjxStringUtil.urlEncode(ZmAddressObjectHandler.URL + AjxStringUtil.htmlEncode(obj));
 67 		AjxRpc.invoke(null, url, null, new AjxCallback(this, this._callback, obj), true);
 68 	}
 69 };
 70 
 71 ZmAddressObjectHandler.prototype._callback = 
 72 function(arg0, arg1) {
 73 	// Quick RDF parser
 74 	var r = arg1.text;
 75 	var lg_s = r.indexOf("<Longitude>");
 76 	var lg_e = r.indexOf("</Longitude>");
 77 	var lt_s = r.indexOf("<Latitude>");
 78 	var lt_e = r.indexOf("</Latitude>");
 79 	ZmAddressObjectHandler.displayMap(r.substring(lg_s+11,lg_e), r.substring(lt_s+10,lt_e), arg0);
 80 };
 81 
 82 ZmAddressObjectHandler.displayMap = function(lng, lat, obj) {
 83 	DBG.println(AjxDebug.DBG2, "gMAPS: lat: " + lat + " long: " + lng + " obj: '" + obj + "'");
 84 	var element = document.getElementById(ZmAddressObjectHandler.encodeId(obj));
 85     var map = new GMap(element);
 86     var point = new GPoint(lng, lat);
 87     map.centerAndZoom(point, 4);
 88 	var marker = new GMarker(point);
 89 	map.addOverlay(marker);
 90 	marker.openInfoWindowHtml("<div id=\""+ZmAddressObjectHandler.encodeId(obj)+"tip\" style=\"width:255px\"><b>"+obj+"</b></div>");
 91     if(!ZmAddressObjectHandler.CACHE[obj+"lng"] || !ZmAddressObjectHandler.CACHE[obj+"lat"]) {
 92 		DBG.println(AjxDebug.DBG2, "gMAPS: Adding to cache");
 93 		ZmAddressObjectHandler.CACHE[obj+"lng"] = lng;
 94 		ZmAddressObjectHandler.CACHE[obj+"lat"] = lat;
 95 	}
 96 };
