 17 /**
 18  * Creates an input field.
 19  * @constructor
 20  * @class
 21  * This class represents an input field..
 22  * <p>
 23  * <h3>CSS Elements</h3>
 24  * <ul>
 25  * <li><code><className></code> input 			specifies the look of the input field during normal editing
 26  * <li><code><className>-error</code> input		specifies the look of the input field in an error state
 27  * </ul>
 28  *
 29  * @author Ross Dargahi
 30  *
 31  * @param {hash}	params			a hash of parameters
 32  * @param {DwtComposite}      params.parent			the parent widget
 33  * @param {constant}      params.type				the data type of the input field
 34  * @param {string}      params.initialValue		the initial value of the field
 35  * @param {number}      params.size				size of the input field (in characters)
 36  * @param {number}      params.rows				the number of rows (more than 1 means textarea)
 37  * @param {boolean}      params.forceMultiRow		if <code>true</code>, forces use of textarea even if rows == 1
 38  * @param {number}      params.maxLen			the maximum length (in characters) of the input
 39  * @param {constant}      params.errorIconStyle		the error icon style
 40  * @param {constant}      params.validationStyle	the validation type
 41  * @param  {function}     params.validator			the custom validation function
 42  * @param {Object}      params.validatorCtxtObj		the object context for validation function
 43  * @param {string}      params.className			the CSS class
 44  * @param {constant}      params.posStyle			the positioning style (see {@link DwtControl})
 45  * @param {boolean}      params.required          if <code>true</code>, mark as required.
 46  * @param {string}      params.hint				a hint to display in the input field when the value is empty.
 47  * @param {string}      params.id				an explicit ID to use for the control's DIV element
 48  * @param {string}      params.inputId			an explicit ID to use for the control's INPUT element
 49  * 
 50  * @extends		DwtComposite
 51  * 
 52  * TODO: Use HTML5 feature of placeholder text for inputs (not supported by IE):
 53  * http://www.whatwg.org/specs/web-apps/current-work/multipage/common-input-element-attributes.html#the-placeholder-attribute
 54  */
 55 DwtInputField = function(params) {
 56 
 57 	if (arguments.length == 0) return;
 58 	this._origClassName = params.className ? params.className : "DwtInputField";
 59 	this._errorClassName = this._origClassName + "-Error";
 60 	this._hintClassName = this._origClassName + "-hint";
 61 	this._disabledClassName = this._origClassName + "-disabled";
 62 	this._errorHintClassName = this._origClassName + "-errorhint";
 63 	DwtComposite.call(this, params);
 64 
 65     this._inputEventHandlers = {};
 66 
 67     this._type = params.type ? params.type : DwtInputField.STRING;
 68     this._rows = params.rows ? params.rows : 1;
 69     this._size = params.size;
 70 
 71 	this._errorIconStyle = params.errorIconStyle ? params.errorIconStyle :
 72 							params.validator ? DwtInputField.ERROR_ICON_RIGHT : DwtInputField.ERROR_ICON_NONE;
 73 	this._validationStyle = params.validationStyle ? params.validationStyle : DwtInputField.ONEXIT_VALIDATION;
 74 
 75 	this._hasError = false;
 76 	this._hintIsVisible = false;
 77 	this._hint = params.hint;
 78 	
 79 	var inputFieldId = params.inputId || Dwt.getNextId();
 80 	var errorIconId = Dwt.getNextId();
 81 	var htmlEl = this.getHtmlElement();
 82 	if (this._errorIconStyle == DwtInputField.ERROR_ICON_NONE) {
 83 		if (params.forceMultiRow || (params.rows && params.rows > 1)) {
 84 			var htmlArr = ["<textarea id='", inputFieldId, "' rows=", params.rows];
 85 			var i = htmlArr.length;
 86 			if (params.forceMultiRow || params.size) {
 87 				htmlArr[i++] = " cols=";
 88 				htmlArr[i++] = params.size || 1;
 89 			}
 90 			if (params.wrap) {
 91 				htmlArr[i++] = " wrap=";
 92 				htmlArr[i++] = params.wrap;
 93 			}
 94 			htmlArr[i++] = "></textarea>"
 95 			htmlEl.innerHTML = htmlArr.join("");
 96 		} else {
 97 			htmlEl.innerHTML = ["<input id='",inputFieldId,"'>"].join("");
 98 		}
 99 
100 	} else {
101 		var htmlArr = ["<table cellspacing='0' cellpadding='0'><tr>"];
102 		var i = 1;
103 		if (this._errorIconStyle == DwtInputField.ERROR_ICON_LEFT)
104 			htmlArr[i++] = ["<td style='padding-right:2px;'id='", errorIconId, "'></td>"].join("");
105 
106 		htmlArr[i++] = ["<td>", "<input id='", inputFieldId, "'>", "</td>"].join("");
107 
108 		if (this._errorIconStyle == DwtInputField.ERROR_ICON_RIGHT)
109 			htmlArr[i++] = ["<td style='padding-left:2px;' id='", errorIconId, "'></td>"].join("");
110 
111 		htmlArr[i++] = "</tr></table>";
112 		htmlEl.innerHTML = htmlArr.join("");
113 
114 		if (this._errorIconStyle != DwtInputField.ERROR_ICON_NONE) {
115 			this._errorIconTd = document.getElementById(errorIconId);
116 			this._errorIconTd.vAlign = "middle";
117 			this._errorIconTd.innerHTML = DwtInputField._NOERROR_ICON_HTML;
118 		}
119 	}
120 
121 	this._tabGroup = new DwtTabGroup(this._htmlElId);
122 	if (params.forceMultiRow || this._rows > 1) {
123         this._inputField = document.getElementById(inputFieldId);
124         this._inputField.onkeyup = DwtInputField._keyUpHdlr;
125         this._inputField.onblur = DwtInputField._blurHdlr;
126 		this._inputField.onfocus = DwtInputField._focusHdlr;
127         this._inputField.onkeydown = DwtInputField._keyDownHdlr;
128 
129         if (params.size)
130             this._inputField.size = params.size;
131         if (params.maxLen)
132             this._inputField.maxLength = this._maxLen = params.maxLen;
133 
134         //MOW:  this.setCursor("default");
135 
136         this._inputField.value = params.initialValue || "";
137 		this._tabGroup.addMember(this._inputField);
138 	}
139     else {
140         var oinput = document.getElementById(inputFieldId);
141         var ninput = this.__createInputEl(params);
142 		// bug fix #
143 		if (AjxEnv.isCamino) {
144 			oinput.parentNode.style.overflow = "hidden";
145 		}
146 		oinput.parentNode.replaceChild(ninput, oinput);
147 	}
148 
149     this.setValidatorFunction(params.validatorCtxtObj, params.validator);
150 	this._setMouseEventHdlrs(false);
151 	this._setKeyPressEventHdlr(false);
152 	
153     if (params.required != null) {
154         this.setRequired(params.required);
155     }
156 };

249 /**
250  * Sets the input type.
251  * 
252  * @param	{constant}	type		the input type
253  */
254 DwtInputField.prototype.setInputType = function(type) {
255     if (type != this._type && this._rows == 1) {
256         this._type = type;
257         if (AjxEnv.isIE) {
258             var oinput = this._inputField;
259             var ninput = this.__createInputEl();
260             oinput.parentNode.replaceChild(ninput, oinput);
261         }
262         else {
263             this._inputField.type = this._type != DwtInputField.PASSWORD ? "text" : "password";
264         }
265     }

268 /**
269 * Sets the validator function. This function is executed during validation.
270 *
271 * @param {Object}	obj 		if present, the validator function is executed within
272 *		the context of this object
273 * @param {function}	validator 	the validator function
274 */
275 DwtInputField.prototype.setValidatorFunction =
276 function(obj, validator) {
277 	if (validator) {
278 		this._validator = validator;
279 		this._validatorObj = obj;
280 	} else {
281 		switch (this._type) {
282 			case DwtInputField.NUMBER:	this._validator = DwtInputField.validateNumber; break;
283 		    case DwtInputField.INTEGER:	this._validator = DwtInputField.validateInteger; break;
284 		    case DwtInputField.FLOAT:	this._validator = DwtInputField.validateFloat; break;
285 		    case DwtInputField.STRING:
286 		    case DwtInputField.PASSWORD:this._validator = DwtInputField.validateString;	break;
287 		    case DwtInputField.DATE: 	this._validator = DwtInputField.validateDate; break;
288 		    default: 					this._validator = DwtInputField.validateAny;
289 		}
290 	}
291 };

339 /**
340  * Sets the value for the input field.
341  *
342  * @param	{string}	value	the value
343  * @param	{boolean}	noValidate		if <code>true</code>, do not validate
344  */
345 DwtInputField.prototype.setValue =
346 function(value, noValidate) {
347 	// XXX: if we're disabled, the validation step messes up the style
348 	value = value || "";
349 	this._inputField.value = value;
350 	if(!noValidate) {
351 		value = this._validateInput(value);
352 		if (value != null) {
353 			this._inputField.value = value;
354 		}
355 	}
356 	if (this._hintIsVisible && value) {
357 		this._hideHint(value);
358 	} else if (!value) {
359 		this._showHint();
360 	}
361 };
362 
363 DwtInputField.prototype.clear =
364 function() {
365 	this.setValue("");
366 };

368 /**
369  * Sets the hint for the input field.
370  *
371  * @param {string}	hint 	the hint
372  */
373 DwtInputField.prototype.setHint =
374 function(hint) {
375 	var oldHint = this._hint;
376 	this._hint = hint;
377 	if (this._hintIsVisible) {
378 		this.getInputElement().value = hint;
379 		if (!hint) {
380 			this._hintIsVisible = false;
381 			this._updateClassName();
382 		}
383 	} else if (this._inputField.value == '') {
384 		this._showHint();
385 	}
386 };

502 /**
503  * Checks the validity of the input field value.
504  *
505  * @return {string}	a canonical value if valid or <code>null</code> if the field value is not valid
506  */
507 DwtInputField.prototype.isValid =
508 function() {
509 	if (!this.getEnabled()) {
510 		return this.getValue();
511 	}
512 	try {
513 		if (typeof this._validator == "function") {
514 			return this._validatorObj
515 				? this._validator.call(this._validatorObj, this.getValue(), this)
516 				: this._validator(this.getValue());
517 		} else {
518 			return this._validator.test(this._inputField.value);
519 		}
520 	} catch(ex) {
521 		if (typeof ex == "string")
522 			return null;
523 		else
524 			throw ex;
525 	}
526 };

528 /**
529  * Validates the current input in the field. This method should be called
530  * if the validation style has been set to DwtInputField.MANUAL_VALIDATION
531  * and it is time for the field to be validated
532  *
533  * @return {boolean}	<code>true</code> if the field is valid
534  */
535 DwtInputField.prototype.validate =
536 function() {
537 	var value = this._validateInput(this.getValue());
538 	if (value != null) {
539 		this.setValue(value);
540 		return true;
541 	} else {
542 		return false;
543 	}
544 };

548 /**
549  * Validates a number.
550  * 
551  * @param	{string}	value		the value
552  * @return	{boolean}	<code>true</code> if valid
553  */
554 DwtInputField.validateNumber =
555 function(value) {
556 	var n = new Number(value);
557 	if (isNaN(n) || (Math.round(n) != n))
558 		throw AjxMsg.notAnInteger;
559 	return DwtInputField.validateFloat.call(this, value);
560 };
561 
562 /**
563  * Validates an integer.
564  * 
565  * @param	{string}	value		the value
566  * @return	{boolean}	<code>true</code> if valid
567  */
568 DwtInputField.validateInteger =
569 function(value) {
570 	var n = new Number(value);
571 	if (isNaN(n) || (Math.round(n) != n) || (n.toString() != value))
572 		throw AjxMsg.notAnInteger;
573 	if (this._minNumVal && value < this._minNumVal)
574 		throw AjxMessageFormat.format(AjxMsg.numberLessThanMin, this._minNumVal);
575 	if (this._maxNumVal && value > this._maxNumVal)
576 		throw AjxMessageFormat.format(AjxMsg.numberMoreThanMax, this._maxNumVal);
577 	return value;
578 };
579 
580 /**
581  * Validates a float.
582  * 
583  * @param	{string}	value		the value
584  * @return	{boolean}	<code>true</code> if valid
585  */
586 DwtInputField.validateFloat =
587 function(value) {
588 	if (this._required && value == "")
589 		throw AjxMsg.valueIsRequired;
590 	var n = new Number(value);
591 	if (isNaN(n))
592 		throw AjxMsg.notANumber;
593 	if (this._minNumVal && value < this._minNumVal)
594 		throw AjxMessageFormat.format(AjxMsg.numberLessThanMin, this._minNumVal);
595 	if (this._maxNumVal && value > this._maxNumVal)
596 		throw AjxMessageFormat.format(AjxMsg.numberMoreThanMax, this._maxNumVal);
597 
598 	// make canonical value
599 	if (this._decimals != null) {
600 		var str = n.toString();
601 		var pos = str.indexOf(".");
602 		if (pos == -1)
603 			pos = str.length;
604 		value = n.toPrecision(pos + this._decimals);
605 	} else {
606 		value = n.toString();
607 	}
608 
609 	return value;
610 };
611 
612 /**
613  * Validates a string.
614  * 
615  * @param	{string}	value		the value
616  * @return	{boolean}	<code>true</code> if valid
617  */
618 DwtInputField.validateString =
619 function(value) {
620 	if (this._required && value == "")
621 		throw AjxMsg.valueIsRequired;
622 	if (this._minLen != null && value.length < this._minLen)
623 		throw AjxMessageFormat.format(AjxMsg.stringTooShort, this._minLen);
624 	if (this._maxLen != null && value.length > this._maxLen)
625 		throw AjxMessageFormat.format(AjxMsg.stringTooLong, this._maxLen);
626 	return value;
627 };
628 
629 /**
630  * Validates a date.
631  * 
632  * @param	{string}	value		the value
633  * @return	{boolean}	<code>true</code> if valid
634  */
635 DwtInputField.validateDate = 
636 function(value) {
637 	if (this._required && value == "")
638 		throw AjxMsg.valueIsRequired;
639 	
640 	if (AjxDateUtil.simpleParseDateStr(value) == null) {
641 		throw AjxMsg.invalidDatetimeString;
642 	}
643 
644 	return value;
645 };
646 
647 /**
648  * Validates an email.
649  * 
650  * @param	{string}	value		the value
651  * @return	{boolean}	<code>true</code> if valid
652  */
653 DwtInputField.validateEmail = function(value) {
654 	if (this._required && value == "")
655 		throw AjxMsg.valueIsRequired;
656 	if (!AjxEmailAddress.isValid(value))
657 		throw AjxMsg.invalidEmailAddr;
658 	return value;
659 };
660 
661 DwtInputField.validateAny =
662 function(value) {
663 	if (this._required && value == "")
664 		throw AjxMsg.valueIsRequired;
665 	// note that null will always be regarded as invalid. :-) I guess this
666 	// is OK.  An input field never has a null value.
667 	return value;
668 };

674 DwtInputField.prototype._validateRegExp =
675 function(value) {
676 	if (this._required && value == "")
677 		throw AjxMsg.valueIsRequired;
678 	if (this._regExp && !this._regExp.test(value)) {
679 		throw this._errorString;
680 	}
681 	return value;
682 };

684 DwtInputField._keyUpHdlr =
685 function(ev) {
686 	var keyEv = DwtShell.keyEvent;
687 	keyEv.setFromDhtmlEvent(ev, true);
688 
689 	var obj = keyEv.dwtObj;
690 	var keyCode = keyEv.keyCode;
691     if (obj.notifyListeners(DwtEvent.ONKEYUP, keyEv)) {
692         return true;
693     }
694 
695 	// ENTER || TAB
696 	var val = null;
697 	if ((keyCode == 0x0D || keyCode == 0x09)
698 	    && obj._validationStyle == DwtInputField.ONEXIT_VALIDATION)
699 		val = obj._validateInput(obj.getValue());
700 	else if (obj._validationStyle == DwtInputField.CONTINUAL_VALIDATION)
701 		val = obj._validateInput(obj.getValue());
702 
703 	if (val != null && val != obj.getValue())
704 		obj.setValue(val);
705 
706 	return true;
707 };

709 DwtInputField._blurHdlr =
710 function(ev) {
711 	var obj = DwtControl.getTargetControl(ev);
712 	if (obj) {
713 		obj._hasFocus = false;
714 		if (obj._validationStyle == DwtInputField.ONEXIT_VALIDATION) {
715 			var val = obj._validateInput(obj.getValue());
716 			if (val != null) {
717 				obj.setValue(val);
718 			}
719 		}
720 		if (!obj._hintIsVisible && obj._hint) {
721 			obj._showHint();
722 		}
723 	}
724 	obj.notifyListeners(DwtEvent.ONBLUR, ev)
725 };

771 DwtInputField.prototype._updateClassName = 
772 function() {
773 	var className;
774 	if (!this.getEnabled()) {
775 		className = this._disabledClassName;
776 	} else if (this._hasError) {
777 		if (this._hintIsVisible && !this._hasFocus) {
778 			className = this._errorHintClassName;
779 		} else {
780 			className = this._errorClassName;
781 		}
782 	} else if (this._hintIsVisible && !this._hasFocus) {
783 		className = this._hintClassName;
784 	} else {
785 		className = this._origClassName;
786 	}
787 	this.getHtmlElement().className = className;
788 };

790 DwtInputField.prototype._validateInput =
791 function(value) {
792 	var isValid = true;
793 	var retVal;
794 	var errorStr;
795 
796 	if (!this.getEnabled()) {
797 		retVal = this.getValue();
798 	} else {
799 		try {
800 			if (typeof this._validator == "function") {
801 				retVal = value = this._validatorObj
802 					? this._validator.call(this._validatorObj, value, this)
803 					: this._validator(value);
804 			} else if (!this._validator.test(value)) {
805 				errorStr = this._errorString;
806 			}
807 		} catch(ex) {
808 			if (typeof ex == "string")
809 				errorStr = ex;
810 			else
811 				throw ex;
812 		}
813 	}
814 	
815 	if (errorStr) {
816 		this._hasError = true;
817 		if (this._errorIconTd)
818 			this._errorIconTd.innerHTML = DwtInputField._ERROR_ICON_HTML;
819 		this.setToolTipContent(errorStr);
820 		isValid = false;
821 		retVal = null;
822 	} else {
823 		this._hasError = false;
824 		if (this._errorIconTd)
825 			this._errorIconTd.innerHTML = DwtInputField._NOERROR_ICON_HTML;
826 		this.setToolTipContent(null);
827 		isValid = true;
828 	}
829 	this._updateClassName();
830 
831 	if (this._validationCallback)
832 		this._validationCallback.run(this, isValid, value);
833 
834 	return retVal;
835 };

874 DwtInputField.prototype.__createInputEl =
875 function(params) {
876 	// clean up old input field if present
877 	var oinput = this._inputField;
878 	if (oinput) {
879 		for (var eventType in this._inputEventHandlers) {
880 			oinput.removeAttribute(eventType);
881 		}
882 	}
883 
884 	// create new input field
885 	var ninput;
886 	var type = this._type != DwtInputField.PASSWORD ? "text" : "password";
887 	if (AjxEnv.isIE) {
888 		try {
889 			var ninput = document.createElement(["<INPUT type='",type,"'>"].join(""));
890 		} catch(e) {
891 			ninput = document.createElement("INPUT");
892 			ninput.type = type;
893 		}
894 	} else {
895 		ninput = document.createElement("INPUT");
896 		ninput.type = type;
897 	}
898 	this._inputField = ninput;
899 
900 	// set common values
901 	var size = params ? params.size : oinput.size;
902 	var maxLen = params ? params.maxLen : oinput.maxLength;
903 
904 	ninput.autocomplete = "off";
905 	if (size) {
906 		ninput.size = size;
907 	}
908 	if (maxLen) {
909 		ninput.maxLength = maxLen;
910 	}
911 	ninput.value = (params ? params.initialValue : oinput.value) || "";
912 	ninput.readonly = oinput ? oinput.readonly : false;
913     if (params && params.inputId) {
914         ninput.id = params.inputId;
915     }
916 
917 	// add event handlers
918 	ninput.onkeyup = DwtInputField._keyUpHdlr;
919 	ninput.onblur = DwtInputField._blurHdlr;
920 	ninput.onfocus = DwtInputField._focusHdlr;
921     ninput.onkeydown = DwtInputField._keyDownHdlr;
922 	for (var eventType in this._inputEventHandlers) {
923 		ninput[eventType] = this._inputEventHandlers[eventType];
924 	}
925 
926 	this._tabGroup.removeAllMembers();
927 	this._tabGroup.addMember(ninput);
928 	return ninput;
929 };
