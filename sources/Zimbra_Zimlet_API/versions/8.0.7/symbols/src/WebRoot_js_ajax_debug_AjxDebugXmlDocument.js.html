 25 // used to find the Automation server name
 26 getDomDocumentPrefix = function() {
 27 	if (getDomDocumentPrefix.prefix)
 28 		return getDomDocumentPrefix.prefix;
 29 	
 30 	var prefixes = ["MSXML2", "Microsoft", "MSXML", "MSXML3"];
 31 	var o;
 32 	var len = prefixes.length;
 33 	for (var i = 0; i < len; i++) {
 34 		try {
 35 			// try to create the objects
 36 			o = new ActiveXObject(prefixes[i] + ".DomDocument");
 37 			return getDomDocumentPrefix.prefix = prefixes[i];
 38 		}
 39 		catch (ex) {};
 40 	}
 41 	
 42 	throw new Error("Could not find an installed XML parser");
 43 }
 44 
 45 AjxDebugXmlDocument.prototype.create = 
 46 function () {
 47 	try {
 48 		// DOM2
 49 		if (document.implementation && document.implementation.createDocument) {
 50 			var doc = document.implementation.createDocument("", "", null);
 51 			
 52 			// some versions of Moz do not support the readyState property
 53 			// and the onreadystate event so we patch it!
 54 			if (doc.readyState == null) {
 55 				doc.readyState = 1;
 56 				doc.addEventListener("load", function () {
 57 					doc.readyState = 4;
 58 					if (typeof doc.onreadystatechange == "function")
 59 						doc.onreadystatechange();
 60 				}, false);
 61 			}
 62 			
 63 			return doc;
 64 		}
 65 		if (window.ActiveXObject)
 66 			return new ActiveXObject(getDomDocumentPrefix() + ".DomDocument");
 67 	}
 68 	catch (ex) {}
 69 	throw new Error("Your browser does not support XmlDocument objects");
 70 }

 72 // Create the loadXML method and xml getter for Mozilla
 73 if (window.DOMParser &&
 74 	window.XMLSerializer &&
 75 	window.Node && Node.prototype && Node.prototype.__defineGetter__)
 76 {
 77 	if (AjxEnv.isSafari) {
 78 		Document.prototype.loadXML = function(s) {
 79 			// parse the string to a new doc
 80 			var doc2 = (new DOMParser()).parseFromString(s, "text/xml");
 81 
 82 			// remove all initial children
 83 			while (this.hasChildNodes()) {
 84 				this.removeChild(this.lastChild);
 85 			}
 86 
 87 			// insert and import nodes
 88 			var len = doc2.childNodes.length;
 89 			for (var i = 0; i < len; i++) {
 90 				this.appendChild(this.importNode(doc2.childNodes[i], true));
 91 			}
 92 		};
 93 
 94 		// This serializes the DOM tree to an XML String
 95 		// Usage: var sXml = oNode.xml
 96 		Document.prototype.__defineGetter__("xml", function () {
 97 			return (new XMLSerializer()).serializeToString(this);
 98 		});
 99 	}
100 	//
101 	// XMLDocument did not extend Document interface in some versions of Mozilla
102 	// so explicitly define it here.
103 	//
104 	else {
105 		AjxDebugXmlDocument.prototype.loadXML = function(s) {
106 			// parse the string to a new doc
107 			var doc2 = (new DOMParser()).parseFromString(s, "text/xml");
108 		
109 			// remove all initial children
110 			while (this.hasChildNodes())
111 				this.removeChild(this.lastChild);
112 
113 			// insert and import nodes
114 			var len = doc2.childNodes.length;
115 			for (var i = 0; i < len; i++)
116 				this.appendChild(this.importNode(doc2.childNodes[i], true));
117 		};
118 
119 		// This serializes the DOM tree to an XML String
120 		// Usage: var sXml = oNode.xml
121 		AjxDebugXmlDocument.prototype.__defineGetter__("xml", function () {
122 			return (new XMLSerializer()).serializeToString(this);
123 		});
124 	}
125 };
