 17 /**
 18  * Creates a Header Tree Item.
 19  * @constructor
 20  * @class
 21  * This class implements a tree item widget.
 22  *
 23  * @author Dave Comfort
 24  *
 25  * @param {hash}	params				a hash of parameters
 26  * @param {DwtComposite}      params.parent				the parent widget
 27  * @param {number}      params.index 				the index at which to add this control among parent's children
 28  * @param {string}      params.text 					the label text for the tree item
 29  * @param {string}      params.imageInfo				the icon for the tree item
 30  * @param {boolean}      params.deferred				if <code>true</code>, postpone initialization until needed.
 31  * @param {string}      params.className				the CSS class
 32  * @param  {constant}	params.posStyle				the positioning style
 33  * @param {boolean}      params.forceNotifySelection	force notify selection even if checked style
 34  * @param {boolean}      params.forceNotifyAction		force notify action even if checked style
 35  * @param {hash}		  params.optButton				a hash of data for showing a options button in the item: image, tooltip, callback
 36  * @param {boolean}      params.selectable			if <code>true</code>, this item is selectable
 37  *        
 38  * @extend		DwtTreeItem
 39  */
 40 DwtHeaderTreeItem = function(params) {
 41 	this.overview = params.overview;
 42 	this._optButton = params.optButton;
 43 	this._noNodeCell = params.noNodeCell;
 44 	DwtTreeItem.call(this, params);
 45 };
 46 
 47 DwtHeaderTreeItem.prototype = new DwtTreeItem;
 48 DwtHeaderTreeItem.prototype.constructor = DwtHeaderTreeItem;
 49 
 50 DwtHeaderTreeItem.prototype.TEMPLATE = "dwt.Widgets#ZHeaderTreeItem";
 51 
 52 DwtHeaderTreeItem.prototype.toString =
 53 function() {
 54 	return "DwtHeaderTreeItem";
 55 };
 56 
 57 DwtHeaderTreeItem.prototype._createHtmlFromTemplate =
 58 function(template, data) {
 59 	data.noNodeCell = this._noNodeCell;
 60 	DwtTreeItem.prototype._createHtmlFromTemplate.apply(this, arguments);
 61 };
 62 
 63 DwtHeaderTreeItem.prototype._initialize =
 64 function() {
 65 	DwtTreeItem.prototype._initialize.apply(this, arguments);
 66 	if (this._optButton) {
 67 		this._optButtonId = this._htmlElId + "_optButton";
 68 		var optButtonEl = document.getElementById(this._optButtonId);
 69 		if (optButtonEl) {
 70 			this._optButtonItem = new DwtBorderlessButton({parent:this, style:DwtLabel.IMAGE_LEFT});
 71 			this._optButtonItem.setToolTipContent(this._optButton.tooltip);
 72 			this._optButtonItem.callback = this._optButton.callback;
 73 			this._optButtonItem.addSelectionListener(new AjxListener(this, this._onclickHandler));
 74 			this._optButtonItem.replaceElement(this._optButtonId);
 75 			this._optButtonItem.setImage("ContextMenu");
 76 			this._optButtonItem.setIconEl(this._optButtonItem.getHtmlElement()); // image container is button
 77 		}
 78 	}
 79 };
 80 
 81 DwtHeaderTreeItem.prototype._onclickHandler =
 82 function(ev) {
 83     this._tree._itemActioned(this, ev);
 84 };
 85 
 86 
 87 DwtHeaderTreeItem.prototype._focusByMouseUpEvent =
 88 function(ev)  {
 89 	var targetId = ev.target && ev.target.id;
 90 	if (targetId && (targetId == this._headerButtonId)) { return; }
 91 	DwtTreeItem.prototype._focusByMouseUpEvent.apply(this, arguments);
 92 };
