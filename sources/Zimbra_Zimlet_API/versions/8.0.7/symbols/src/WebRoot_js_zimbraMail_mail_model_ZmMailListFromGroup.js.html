1 /**
  2  * From Group divides messages into sort by sender.
  3  */
  4 ZmMailListFromGroup = function() {
  5     this.id = ZmId.GROUPBY_FROM;
  6 	this.field = ZmItem.F_FROM;
  7     ZmMailListGroup.call(this);
  8 
  9 };
 10 
 11 ZmMailListFromGroup.prototype = new ZmMailListGroup;
 12 ZmMailListFromGroup.prototype.constructor =  ZmMailListFromGroup;
 13 
 14 /**
 15  *  returns HTML string for all sections.
 16  *  @param {boolean} sortAsc    true/false if sort ascending
 17  *  @return {String} HTML for all sections including section header
 18  * @param sortAsc
 19  */
 20 ZmMailListFromGroup.prototype.getAllSections =
 21 function(sortAsc) {
 22     var htmlArr = [];
 23     var keys = this._sortKeys(sortAsc);
 24 
 25     for(var i=0; i<keys.length; i++) {
 26         var key = keys[i].addr;
 27         if (this._section[key].length > 0) {
 28             var sectionHeader = keys[i].title;
 29             htmlArr.push(this.getSectionHeader(sectionHeader));
 30             htmlArr.push(this._section[key].join(""));
 31         }
 32     }
 33 
 34     return htmlArr.join("");
 35 };

 37 /**
 38  * Adds item to section
 39  * @param {ZmMailMsg} msg   mail message
 40  * @param {String} item  HTML to add to section
 41  * @return {String} section returns section if successfully added, else returns null
 42  */
 43 ZmMailListFromGroup.prototype.addMsgToSection =
 44 function(msg, item){
 45    var emails =  msg.getEmails();
 46    var participants = msg.participants ? msg.participants.getArray() : null;
 47    if (emails) {
 48 	    emails = emails.getArray();
 49 	    var email = emails[0];
 50 		if (this._section.hasOwnProperty(email)) {
 51 			this._section[email].push(item);
 52 			return email;
 53 		}
 54 		else {
 55 			this._section[email] = [];
 56 			this._section[email].push(item);
 57 			var title = participants ? this._getSectionTitle(participants[0]) : email;
 58 			this._sectionTitle[email] = {addr: email, title:title};
 59 			return email;
 60 		}
 61    }
 62    return false;
 63 };

 65 /**
 66  * Determines if message is in group
 67  * @param {String} section ID of section
 68  * @param {ZmMailMsg} msg
 69  * @return {boolean} true/false
 70  */
 71 ZmMailListFromGroup.prototype.isMsgInSection =
 72 function(section, msg) {
 73 
 74     var addr;
 75     var participants = msg.participants;
 76     if (participants) {
 77         var arr = participants.getArray();
 78         for (var i=0; i<arr.length; i++){
 79             if (arr[i].getType() == "FROM") {
 80               addr = arr[i].getAddress();
 81               break;
 82             }
 83         }
 84     }
 85 
 86     if (addr && addr == section) {
 87        return true;
 88     }
 89 
 90     return false;
 91 };

 93 /**
 94  * Returns the sort by (ZmSearch.NAME_ASC or ZmSearch.NAME_DESC)
 95  * @param {boolean} sortAsc
 96  * @return {String} sortBy
 97  */
 98 ZmMailListFromGroup.prototype.getSortBy =
 99 function(sortAsc) {
100     if (sortAsc) {
101         return ZmSearch.NAME_ASC;
102     }
103     return ZmSearch.NAME_DESC;
104 };

106 ZmMailListFromGroup.prototype._init =
107 function() {
108     this._section = {};
109     this._sectionTitle = {};
110 };

112 ZmMailListFromGroup.prototype._sortKeys =
113 function(sortAsc) {
114   var keys = [];
115   var i=0;
116   for (var name in this._sectionTitle) {
117       keys[i++] = {addr: this._sectionTitle[name].addr, title: this._sectionTitle[name].title};
118   }
119 
120   var sortAscIgnoreCase = function(a, b) {
121     a = a.title.toLowerCase();
122     b = b.title.toLowerCase();
123     if (a > b)
124         return 1;
125     if (a < b)
126         return -1;
127     return 0;
128   };
129 
130   var sortDescIgnoreCase = function(a, b) {
131     a = a.title.toLowerCase();
132     b = b.title.toLowerCase();
133     if (a < b)
134         return 1;
135     if (a > b)
136         return -1;
137     return 0;
138   };
139 
140   keys.sort(sortAscIgnoreCase);
141   if (!sortAsc) {
142       keys.reverse(sortDescIgnoreCase);
143   }
144   return keys;
145 };

147 ZmMailListFromGroup.prototype._getSectionTitle =
148 function(participant) {
149     var sectionHeader = participant.getAddress();
150     var name = participant.getName();
151     if (name) {
152         sectionHeader = name;
153     }
154     return sectionHeader;
155 };
156 
157 ZmMailListFromGroup.prototype._getSectionHeaderTitle =
158 function(section) {
159     return this._sectionTitle[section].title;
160 };
