 16 /**
 17  * Creates a preferences section. This is a "pseudo" organizer for
 18  * the preferences application tree view.
 19  * @constructor
 20  * @class
 21  * This class represents the preference page in the preferences application.
 22  * 
 23  * @param {Hash}	params    a hash of parameters
 24  * @param	{int}	     params.id			the numeric ID
 25  * @param	{String}	params.name		the name
 26  * @param	{ZmOrganizer}	params.parent		the parent folder
 27  * @param	{ZmTree}	params.tree		the tree model that contains this folder
 28  * @param	{String}	params.pageId		the ID of pref page
 29  * @param	{String}	params.icon		the icon name
 30  * @param	{String}	params.tooltip		the tool tip text
 31  * 
 32  * @extends		ZmOrganizer
 33  */
 34 ZmPrefPage = function(params) {
 35 	if (arguments.length == 0) { return; }
 36 	params.type = params.type || ZmOrganizer.PREF_PAGE;
 37 	ZmOrganizer.call(this, params);
 38 	this.pageId = params.pageId;
 39 	this.icon = params.icon;
 40 	this.tooltip = params.tooltip;
 41 };
 42 
 43 ZmPrefPage.prototype = new ZmOrganizer;
 44 ZmPrefPage.prototype.constructor = ZmPrefPage;
 45 
 46 ZmPrefPage.prototype.toString = function() {
 47 	return "ZmPrefPage";
 48 };

 60 ZmPrefPage.createFromSection = function(section) {
 61 	var overviewController = appCtxt.getOverviewController(); 
 62 	var treeController = overviewController.getTreeController(ZmOrganizer.PREF_PAGE);
 63 	var params = {
 64 		id: ZmId.getPrefPageId(section.id),
 65 		name: section.title,
 66 		parent: null,
 67 		tree: treeController.getDataTree(),
 68 		icon: section.icon,
 69 		tooltip: section.description
 70 	};
 71 	return new ZmPrefPage(params);
 72 };
