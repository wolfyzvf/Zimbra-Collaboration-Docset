1 /**
  2  * Priority Group divides a mail list into sections: Important & Unread, Important, Flagged, Everything else
  3  * This implemenation is experimental and based on Tag which will change in the future.
  4  * TODO:  Search needs to be batch request based. (implementation outside the scope of this object, but noting here)
  5  * @param group
  6  */
  7 ZmMailListPriorityGroup = function(group) {
  8     this.id = ZmId.GROUPBY_PRIORITY;
  9     ZmMailListGroup.call(this);
 10 };
 11 
 12 ZmMailListPriorityGroup.prototype = new ZmMailListGroup;
 13 ZmMailListPriorityGroup.prototype.constructor =  ZmMailListPriorityGroup;
 14 
 15 ZmMailListPriorityGroup.IMPORTANT_UNREAD = "IMPORTANT_UNREAD";
 16 ZmMailListPriorityGroup.IMPORTANT_READ = "IMPORTANT_READ";
 17 ZmMailListPriorityGroup.FLAGGED = "FLAGGED";
 18 ZmMailListPriorityGroup.EVERYTHING_ELSE = "EVERYTHING_ELSE";
 19 
 20 ZmMailListPriorityGroup.TAG = "Important";
 21 
 22 ZmMailListPriorityGroup.GROUP = [ZmMailListPriorityGroup.IMPORTANT_UNREAD, ZmMailListPriorityGroup.FLAGGED,
 23                                  ZmMailListPriorityGroup.IMPORTANT_READ, ZmMailListPriorityGroup.EVERYTHING_ELSE];
 24 
 25 ZmMailListPriorityGroup.SECTION_TITLE = {};
 26 ZmMailListPriorityGroup.SECTION_TITLE[ZmMailListPriorityGroup.IMPORTANT_READ] = ZmMsg.mailPriorityImportantRead;
 27 ZmMailListPriorityGroup.SECTION_TITLE[ZmMailListPriorityGroup.IMPORTANT_UNREAD] = ZmMsg.mailPriorityImportantUnread;
 28 ZmMailListPriorityGroup.SECTION_TITLE[ZmMailListPriorityGroup.FLAGGED] = ZmMsg.mailPriorityFlagged;
 29 ZmMailListPriorityGroup.SECTION_TITLE[ZmMailListPriorityGroup.EVERYTHING_ELSE] = ZmMsg.mailPriorityEverythingElse;
 30 
 31 ZmMailListPriorityGroup.prototype.getAllSections =
 32 function() {
 33     var htmlArr = [];
 34     for (var i = 0; i<ZmMailListPriorityGroup.GROUP.length; i++) {
 35         if (this._section[ZmMailListPriorityGroup.GROUP[i]].length > 0) {
 36             htmlArr.push(this.getSectionHeader(ZmMailListPriorityGroup.SECTION_TITLE[ZmMailListPriorityGroup.GROUP[i]]));
 37             htmlArr.push(this._section[ZmMailListPriorityGroup.GROUP[i]].join(""));
 38         }
 39         else if (this._showEmptySectionHeader) {
 40             htmlArr.push(this.getSectionHeader(ZmMailListPriorityGroup.SECTION_TITLE[ZmMailListPriorityGroup.GROUP[i]]));
 41         }
 42     }
 43     return htmlArr.join("");
 44 };
 45 
 46 ZmMailListPriorityGroup.prototype.addMsgToSection =
 47 function(msg, item){
 48    for (var i = 0; i<ZmMailListPriorityGroup.GROUP.length; i++) {
 49        if (this.isMsgInSection(ZmMailListPriorityGroup.GROUP[i], msg)) {
 50            this._section[ZmMailListPriorityGroup.GROUP[i]].push(item);
 51            return true;
 52        }
 53    }
 54    return false;
 55 };
 56 
 57 ZmMailListPriorityGroup.prototype.isMsgInSection =
 58 function(section, msg) {
 59     switch(section) {
 60         case ZmMailListPriorityGroup.IMPORTANT_UNREAD:
 61             return this._isImportantAndUnread(msg);
 62 
 63         case ZmMailListPriorityGroup.IMPORTANT_READ:
 64             return this._isImportantAndRead(msg);
 65 
 66         case ZmMailListPriorityGroup.FLAGGED:
 67             return this._isFlagged(msg);
 68 
 69         case ZmMailListPriorityGroup.EVERYTHING_ELSE:
 70            return this._noMatchingGroup(msg);
 71 
 72         default:
 73             return false;
 74     }
 75 };
 76 
 77 ZmMailListPriorityGroup.prototype._init =
 78 function() {
 79 	this._section = {};
 80     this._section[ZmMailListPriorityGroup.IMPORTANT_UNREAD] = [];
 81     this._section[ZmMailListPriorityGroup.IMPORTANT_READ] = [];
 82     this._section[ZmMailListPriorityGroup.FLAGGED] = [];
 83     this._section[ZmMailListPriorityGroup.EVERYTHING_ELSE] = [];
 84 	if (!this._importantTag) {
 85 		this._importantTag = this._getImportantTag();
 86 	}
 87 };
 88 
 89 ZmMailListPriorityGroup.prototype._isImportantAndUnread =
 90 function(msg){
 91    if (msg && this._importantTag) {
 92        if (msg.hasTag(this._importantTag.id) && msg.isUnread) {
 93            return true;
 94        }
 95    }
 96    return false;
 97 };
 98 
 99 ZmMailListPriorityGroup.prototype._isImportantAndRead =
100 function(msg) {
101   if (msg && this._importantTag) {
102       if (msg.hasTag(this._importantTag.id) && !msg.isUnread) {
103           return true;
104       }
105   }
106   return false;
107 };
108 
109 ZmMailListPriorityGroup.prototype._isFlagged =
110 function(msg) {
111    if (msg) {
112        return msg.isFlagged;
113    }
114    return false;
115 };
116 
117 
118 ZmMailListPriorityGroup.prototype._noMatchingGroup =
119 function(msg) {
120     if (!this._isImportantAndUnread(msg) && !this._isImportantAndRead(msg) &&
121         !this._isFlagged(msg)) {
122         return true;
123     }
124     return false;
125 };
126 
127 ZmMailListPriorityGroup.prototype._getImportantTag =
128 function() {
129 	var tagList = appCtxt.getTagTree();
130 	if (tagList) {
131 	   return tagList.getByName("Important");
132 	}
133 	return null;
134 };
