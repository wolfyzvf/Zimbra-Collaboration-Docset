 21 /**
 22  * @class
 23  * This class provides the implementation for portlet for use in the portal application.
 24  * 
 25  * @param	{ZmList}	list	the list that contains this item (may be <code>null</code>)
 26  * @param	{String}	id		the portlet id
 27  * @param	{Object}	def		the portlet definition
 28  *
 29  * @extends	ZmItem
 30  * 
 31  * @see		ZmPortalApp
 32  */
 33 ZmPortlet = function(list, id, def) {
 34     ZmItem.call(this, ZmItem.PORTLET, id, list);
 35 
 36     // save zimlet
 37     var zimletMgr = appCtxt.getZimletMgr();
 38     this.zimletName = def.zimlet;
 39     this.zimletCtxt = zimletMgr.getZimletsHash()[this.zimletName];
 40     this.zimlet = this.zimletCtxt && this.zimletCtxt.handlerObject;
 41 
 42     // save data
 43 	this.global = /^true|on|yes$/i.test(def.global);
 44     this.icon = def.icon;
 45     this.title = def.title;
 46     if (this.title) {
 47         this.title = this.zimletCtxt ? this.zimletCtxt.processMessage(def.title) : def.zimlet;
 48     }
 49     var portlet = this.zimletCtxt && this.zimletCtxt.portlet;
 50     this.actionUrl = portlet && portlet.actionUrl;
 51 
 52     // merge default and specified properties
 53     this.properties = {};
 54     var defaultProps = portlet && portlet.portletProperties;
 55     for (var i in defaultProps) {
 56         var prop = defaultProps[i];
 57         this.properties[prop.name] = prop.value;
 58     }
 59     if (def.properties) {
 60         for (var i = 0; i < def.properties.length; i++) {
 61             var prop = def.properties[i];
 62             this.properties[prop.name] = prop._content;
 63         }
 64     }
 65 
 66 	// string replacement
 67 	if (this.zimletCtxt) {
 68 		for (var pname in this.properties) {
 69 			this.properties[pname] = this.zimletCtxt.replaceObj(ZmZimletContext.RE_SCAN_APP, this.properties[pname], ZmZimletContext.APP);
 70 			// TODO: replace msg,obj,etc...
 71 		}
 72 	}
 73 
 74     // setup refresh interval
 75     if (this.actionUrl) {
 76         this.setRefreshInterval(this.actionUrl.refresh);
 77     }
 78 }
 79 ZmPortlet.prototype = new ZmItem;
 80 ZmPortlet.prototype.constructor = ZmPortlet;
 81 
 82 ZmPortlet.prototype.toString = function() { return "ZmPortlet"; }

104 /**
105  * Refreshes the portlet.
106  * 
107  */
108 ZmPortlet.prototype.refresh = function() {
109     if (this.view) {
110         this._refreshTime = new Date().getTime();
111         if (this.actionUrl) {
112             this.view.setContentUrl(this.actionUrl.target);
113         }
114         else if (this.zimlet instanceof ZmZimletBase) {
115             this.zimlet.portletRefreshed(this);
116         }
117         else if (this.zimlet) {
118             var text = AjxMessageFormat.format(ZmMsg.zimletNotLoaded, this.zimletName);
119             this.setContent(text);
120         }
121         else {
122             var text = AjxMessageFormat.format(ZmMsg.zimletUnknown, this.zimletName);
123             this.setContent(text);
124         }
125     }
126 };
127 
128 /**
129  * Sets the refresh interval.
130  * 
131  * @param	{int}		interval		the refresh interval (in milliseconds)
132  */
133 ZmPortlet.prototype.setRefreshInterval = function(interval) {
134     if (this._refreshActionId != -1) {
135         clearInterval(this._refreshActionId);
136         this._refreshActionId = -1;
137     }
138     this._refreshInterval = interval;
139     if (interval) {
140         if (!this._refreshAction) {
141             this._refreshAction = AjxCallback.simpleClosure(this.refresh, this);
142         }
143         this._refreshActionId = setInterval(this._refreshAction, interval);
144     }
145 };
146 
147 /**
148  * Sets the content.
149  * 
150  * @param	{String}	content		the content
151  */
152 ZmPortlet.prototype.setContent = function(content) {
153     if (this.view) {
154         this.view.setContent(content);
155     }
156     else {
157         DBG.println("no view to set content ("+this.id+")");
158     }
159 };
160 
161 /**
162  * Sets the content url.
163  * 
164  * @param	{String}	url		the content url
165  */
166 ZmPortlet.prototype.setContentUrl = function(url) {
167     if (this.view) {
168         this.view.setContentUrl(url);
169     }
170     else {
171         DBG.println("no view to set content url ("+this.id+")");
172     }
173 };
174 
175 /**
176  * Sets the portlet to "paused".
177  * 
178  * @param	{Boolean}	paused		if <code>true</code>, pause the portlet
179  */
180 ZmPortlet.prototype.setPaused = function(paused) {
181     if (this._refreshActionId != -1 && paused) {
182         this._pauseTime = new Date().getTime();
183         clearInterval(this._refreshActionId);
184         this._refreshActionId = -1;
185     }
186     else if (this._refreshInterval && !paused) {
187         var delta = this._refreshInterval - (this._pauseTime - this._refreshTime);
188         var delay = delta < this._refreshInterval ? delta : 0;
189         var resumeAction = AjxCallback.simpleClosure(this._resumeRefresh, this);
190         setTimeout(resumeAction, delay);
191     }
192 };

194 //
195 // Protected methods

197 
198 /**
199  * @private
200  */
201 ZmPortlet.prototype._resumeRefresh = function() {
202     this.refresh();
203     this._refreshActionId = setInterval(this._refreshAction, this._refreshInterval);
204 };
