 15 /**
 16  * Creates a spinner control.
 17  * @constructor
 18  * @class
 19  * Represents a entry field for entering numeric values.  Has 2 arrow buttons
 20  * that can be used to increment or decrement the current value with a step
 21  * that can be specified.
 22  *
 23  * <h4>CSS</h4>
 24  * <ul>
 25  * <li><code>DwtSpinner</code>              -- a table that contains the spinner elements
 26  * <li><code>DwtSpinner-inputCell</code>    -- the TD that holds the input field
 27  * <li><code>DwtSpinner-btnCell</code>      -- a DIV holding the 2 arrow buttons
 28  * <li><code>DwtSpinner-upBtn</code>        -- the DIV button for increment operation
 29  * <li><code>DwtSpinner-downBtn</code>      -- the DIV button for decrement operation
 30  * <li><code>DwtSpinner-up-pressed</code>   -- upBtn while pressed
 31  * <li><code>DwtSpinner-down-pressed</code> -- downBtn while pressed
 32  * <li><code>DwtSpinner-disabled</code>     -- the table gets this class added when the widget is disabled
 33  * </ul>
 34  * 
 35  * @param	{hash}	params		a hash of parameters
 36  * @param {DwtComposite} params.parent 	the parent widget
 37  * @param {string} params.className the class name for the containing DIV (see {@link DwtControl})
 38  * @param {string} params.posStyle 	the positioning style (see {@link DwtControl})
 39  * @param {number} params.max 	the maximum value
 40  * @param {number} params.min 	the minimum value
 41  * @param {number} params.size 	size of the input field, as in <code><input size="X"></code>
 42  * @param {number} params.value the original value of the input field
 43  * @param {number} params.maxLen the maximum length of the text in the input field
 44  * @param {number} params.step 	the amount to add or substract when the arrow buttons are pressed
 45  * @param {number} [params.decimals=0] Number of decimal digits.  Specify 0 to allow only
 46  *                 integers (default). Pass <code>null</code> to allow float numbers but
 47  *                 not enforce decimals.
 48  * @param {string} [params.align="right"] 	the align of the input field text (see <code>dwt.css</code>)
 49  *
 50  * @author Mihai Bazon
 51  * 
 52  * @extends DwtControl
 53  */
 54 DwtSpinner = function(params) {
 55 	if (arguments.length == 0) return;
 56 	DwtControl.call(this, { parent        : params.parent,
 57                                 className     : params.className,
 58                                 posStyle      : params.posStyle,
 59                                 parentElement : params.parentElement
 60                               });
 61 
 62 	// setup arguments
 63 	this._maxValue      = params.max  != null ? params.max  : null;
 64 	this._minValue      = params.min  != null ? params.min  : null;
 65 	this._fieldSize     = params.size != null ? params.size : 3;
 66 	this._origValue     = params.value     || 0;
 67 	this._maxLen        = params.maxLen    || null;
 68 	this._step          = params.step      || 1;
 69 	this._decimals      = 'decimals' in params ? params.decimals : 0;
 70 	this._align         = params.align     || null;
 71 
 72 	// timerFunc is a closure that gets called upon timeout when the user
 73 	// presses and holds the mouse button
 74 	this._timerFunc = AjxCallback.simpleClosure(this._timerFunc, this);
 75 
 76 	// upon click and hold we capture mouse events
 77 	this._btnPressCapture = new DwtMouseEventCapture({
 78 		targetObj:this,
 79 		id:"DwtSpinner",
 80 		mouseUpHdlr:AjxCallback.simpleClosure(this._stopCapture, this)
 81 	});
 82 
 83 	this._createElements();
 84 };

 86 DwtSpinner.prototype = new DwtControl;
 87 DwtSpinner.prototype.constructor = DwtSpinner;

 89 DwtSpinner.INIT_TIMER = 250;
 90 DwtSpinner.SLOW_TIMER = 125;
 91 DwtSpinner.FAST_TIMER = 33;

145 DwtSpinner.prototype._getValidValue = function(val) {
146 	var n = parseFloat(val);
147 	if (isNaN(n) || n == null)
148 		n = this._lastValidValue; // note that this may be string
149 	if (n == null)
150 		n = this._minValue || 0;
151 	if (this._minValue != null && n < this._minValue)
152 		n = this._minValue;
153 	if (this._maxValue != null && n > this._maxValue)
154 		n = this._maxValue;
155 	// make sure it's a number
156 	n = parseFloat(n);
157 	if (this._decimals != null)
158 		n = n.toFixed(this._decimals);
159 	this._lastValidValue = n;
160 	return n;
161 };

163 /**
164  * Gets the input element.
165  * 
166  * @return	{Element}	the element
167  */
168 DwtSpinner.prototype.getInputElement = function() {
169 	return document.getElementById(this._idField);
170 };
171 
172 DwtSpinner.prototype._getUpButton = function() {
173 	return document.getElementById(this._idUpButton);
174 };
175 
176 DwtSpinner.prototype._getDownButton = function() {
177 	return document.getElementById(this._idDownButton);
178 };
179 
180 DwtSpinner.prototype._getButton = function(direction) {
181 	switch (direction) {
182 	    case "Up"   : return this._getUpButton();
183 	    case "Down" : return this._getDownButton();
184 	}
185 };
186 
187 DwtSpinner.prototype._setBtnState = function(dir, disabled) {
188 	var btn = this._getButton(dir);
189 	if (disabled) {
190 		Dwt.addClass(btn, "DwtSpinner-" + dir + "-disabled");
191 		btn.firstChild.className = "Img" + dir + "ArrowSmallDis";
192 	} else {
193 		Dwt.delClass(btn, "DwtSpinner-" + dir + "-disabled");
194 		btn.firstChild.className = "Img" + dir + "ArrowSmall";
195 	}
196 };

198 /**
199  * Gets the value.
200  * 
201  * @return	{number}	the value
202  */
203 DwtSpinner.prototype.getValue = function() {
204 	return parseFloat(this._getValidValue(this.getInputElement().value));
205 };
206 
207 /**
208  * Sets the value.
209  * 
210  * @param	{number}	val		the value
211  */
212 DwtSpinner.prototype.setValue = function(val) {
213 	if (val == null)
214 		val = this.getInputElement().value;
215 	val = this._getValidValue(val);
216 	this.getInputElement().value = val;
217 	val = parseFloat(val);
218 	this._setBtnState("Down", this._minValue != null && this._minValue == val);
219 	this._setBtnState("Up", this._maxValue != null && this._maxValue == val);
220 };
221 
222 DwtSpinner.prototype.setEnabled = function(enabled) {
223 	DwtControl.prototype.setEnabled.call(this, enabled);
224 	this.getInputElement().disabled = !enabled;
225 	var table = this.getHtmlElement().firstChild;
226 	if (!enabled)
227 		Dwt.addClass(table, "DwtSpinner-disabled");
228 	else
229 		Dwt.delClass(table, "DwtSpinner-disabled");
230 };
231 
232 DwtSpinner.prototype._rotateVal = function(direction) {
233 	var val = this.getValue();
234 	switch (direction) {
235 	    case "Up"   : val += this._step; break;
236 	    case "Down" : val -= this._step; break;
237 	}
238 	this.setValue(val);
239 };
240 
241 DwtSpinner.prototype._btnPressed = function(direction) {
242 	if (!this.getEnabled())
243 		return;
244 	Dwt.addClass(this._getButton(direction), "DwtSpinner-" + direction + "-pressed");
245 	this._direction = direction;
246 	this._rotateVal(direction);
247 	this._btnPressCapture.capture();
248 	this._timerSteps = 0;
249 	this._timer = setTimeout(this._timerFunc, DwtSpinner.INIT_TIMER);
250 };
251 
252 DwtSpinner.prototype._timerFunc = function() {
253 	var v1 = this.getValue();
254 	this._rotateVal(this._direction);
255 	var v2 = this.getValue();
256 	this._timerSteps++;
257 	var timeout = this._timerSteps > 4 ? DwtSpinner.FAST_TIMER : DwtSpinner.SLOW_TIMER;
258 	if (v1 != v2)
259 		this._timer = setTimeout(this._timerFunc, timeout);
260 	else
261 		this._stopCapture();
262 };
263 
264 DwtSpinner.prototype._stopCapture = function() {
265 	if (this._timer)
266 		clearTimeout(this._timer);
267 	this._timer = null;
268 	this._timerSteps = null;
269 	var direction = this._direction;
270 	Dwt.delClass(this._getButton(direction), "DwtSpinner-" + direction + "-pressed");
271 	this._direction = null;
272 	this._btnPressCapture.release();
273 	var input = this.getInputElement();
274 	input.focus();
275 	Dwt.setSelectionRange(input, 0, input.value.length);
276 };
277 
278 DwtSpinner.prototype.__onKeyPress = function(ev) {
279 	if (AjxEnv.isIE)
280 		ev = window.event;
281 	var dir = null;
282 	switch (ev.keyCode) {
283 	    case 38:
284 		dir = "Up";
285 		break;
286 	    case 40:
287 		dir = "Down";
288 		break;
289 	}
290 	if (dir) {
291 		this._rotateVal(dir);
292 		var input = this.getInputElement();
293 		Dwt.setSelectionRange(input, 0, input.value.length);
294 	}
295 };
296 
297 DwtSpinner.prototype.focus = function() {
298 	this.getInputElement().focus();
299 };
300 
301 DwtSpinner.prototype.select = function() {
302 	var input = this.getInputElement();
303 	input.focus();
304 	Dwt.setSelectionRange(input, 0, input.value.length);
305 };
