 57 ZmTaskMgr.prototype.getApptSummaries =
 58 function(params) {
 59 	var apptVec = this.setSearchParams(params);
 60 
 61 	if (apptVec != null && (apptVec instanceof AjxVector)) {
 62 		return apptVec;
 63 	}
 64 
 65 	// this array will hold a list of tasks as we collect them from the server
 66 	this._rawTasks = [];
 67 
 68 	if (params.callback) {
 69 		this._search(params);
 70 	} else {
 71 		return this._search(params);
 72 	}
 73 };
 74 
 75 ZmTaskMgr.prototype.setSearchParams =
 76 function(params) {
 77 	if (!(params.folderIds instanceof Array)) {
 78 		params.folderIds = [params.folderIds];
 79 	} else if (params.folderIds.length == 0) {
 80 		var newVec = new AjxVector();
 81 		if (params.callback) {
 82 			params.callback.run(newVec);
 83 		}
 84 		return newVec;
 85 	}
 86 
 87 	var folderIdMapper = {};
 88 	var query = "";
 89 	for (var i=0; i < params.folderIds.length; i++) {
 90 		var fid = params.folderIds[i];
 91 		var systemFolderId = appCtxt.getActiveAccount().isMain
 92 			? fid : ZmOrganizer.getSystemId(fid);
 93 
 94 		// map remote folder ids into local ones while processing search since
 95 		// server wont do it for us (see bug 7083)
 96 		var folder = appCtxt.getById(systemFolderId);
 97 		var rid = folder ? folder.getRemoteId() : systemFolderId;
 98 		folderIdMapper[rid] = systemFolderId;
 99 
100 		if (query.length) {
101 			query += " OR ";
102 		}
103 		var idText = AjxUtil.isNumeric(fid) ? fid : ['"', fid, '"'].join("");
104 		query += "inid:" + idText;
105 
106 	}
107 	params.queryHint = query;
108 	params.folderIdMapper = folderIdMapper;
109 	params.offset = 0;
110 };

112 ZmTaskMgr.prototype._search =
113 function(params) {
114 	var jsonObj = {SearchRequest:{_jsns:"urn:zimbraMail"}};
115 	var request = jsonObj.SearchRequest;
116 
117 	this._setSoapParams(request, params);
118 
119 	var accountName = appCtxt.multiAccounts ? appCtxt.accountList.mainAccount.name : null;
120 	if (params.callback) {
121 		appCtxt.getAppController().sendRequest({
122 			jsonObj: jsonObj,
123 			asyncMode: true,
124 			callback: (new AjxCallback(this, this._getApptSummariesResponse, [params])),
125 			noBusyOverlay: params.noBusyOverlay,
126 			accountName: accountName
127 		});
128 	} else {
129 		var response = appCtxt.getAppController().sendRequest({jsonObj: jsonObj, accountName: accountName});
130 		var result = new ZmCsfeResult(response, false);
131 		return this._getApptSummariesResponse(params, result);
132 	}
133 };
134 
135 ZmTaskMgr.prototype._setSoapParams =
136 function(request, params) {
137 	request.sortBy = "none";
138 	request.limit = "500";
139 	//request.calExpandInstStart = params.start;
140 	//request.calExpandInstEnd = params.end;
141 	request.types = ZmSearch.TYPE[ZmItem.TASK];
142 	request.offset = params.offset;
143 
144 	var query = params.query;
145 	if (params.queryHint) {
146 		query = (query != null)
147 			? (query + " (" + params.queryHint + ")")
148 			: params.queryHint;
149 	}
150 	request.query = {_content:query};
151 };

154 ZmTaskMgr.prototype._getApptSummariesResponse =
155 function(params, result) {
156 	// TODO: mark both as needing refresh?
157 	if (!result) { return; }
158 
159 	var callback = params.callback;
160 	var resp;
161 	try {
162 		resp = result.getResponse();
163 	} catch (ex) {
164 		if (callback) {
165 			callback.run(new AjxVector());
166 		}
167 		return;
168 	}
169 
170 	var searchResp = resp.SearchResponse;
171 	var newList = this.processSearchResponse(searchResp, params);
172 	if (newList == null) { return; }
173 
174 	if (callback) {
175 		callback.run(newList, params.query);
176 	} else {
177 		return newList;
178 	}
179 };

181 ZmTaskMgr.prototype.processSearchResponse =
182 function(searchResp, params) {
183 	if(!searchResp) { return; }
184 
185 	if (searchResp && searchResp.task && searchResp.task.length) {
186 		this._rawTasks = this._rawTasks != null
187 			? this._rawTasks.concat(searchResp.task)
188 			: searchResp.task;
189 
190 		// if "more" flag set, keep requesting more appts
191 		if (searchResp.more) {
192 			var lastAppt = searchResp.task[searchResp.task.length-1];
193 			if (lastAppt) {
194 				params.offset += 500;
195 				this._search(params);
196 				return;
197 			}
198 		}
199 	}
200 
201 	var newList = new AjxVector();
202 	if (this._rawTasks && this._rawTasks.length) {
203 		this._list = new ZmList(ZmItem.TASK);
204 		for (var i = 0; i < this._rawTasks.length; i++) {
205 			DBG.println(AjxDebug.DBG2, "task[j]:" + this._rawTasks[i].name);
206 			var taskNode = this._rawTasks[i];
207 			//var instances = taskNode ? taskNode.inst : null;
208 			//if (instances) {
209 				var args = {list:this._list};
210 				//for (var j = 0; j < instances.length; j++) {
211 					var task = ZmTask.createFromDom(taskNode, args, null);
212 					DBG.println(AjxDebug.DBG2, "lite task :" + task);
213 					if (task) newList.add(task);
214 				//}
215 			//}
216 		}
217 
218 	}
219 	return newList;
220 };

235 ZmTaskMgr.prototype._handleError =
236 function(ex) {
237 	if (ex.code == 'mail.INVITE_OUT_OF_DATE' ||	ex.code == 'mail.NO_SUCH_APPT') {
238 		var msgDialog = appCtxt.getMsgDialog();
239 		msgDialog.setMessage(ZmMsg.apptOutOfDate, DwtMessageDialog.INFO_STYLE);
240 		msgDialog.popup();
241 		return true;
242 	}
243 	return false;
244 };

247 ZmTaskMgr.prototype.getQuickReminderSearchTimeRange =
248 function() {
249 	var endOfDay = new Date();
250 	endOfDay.setHours(23,59,59,999);
251 
252 	var end = new Date(endOfDay.getTime());
253 
254 	var start = endOfDay;
255 	start.setHours(0,0,0, 0);
256 
257 	return { start: start.getTime(), end: end.getTime() };
258 };

280 ZmTaskMgr.prototype._quickReminderCallback =
281 function(list) {
282     var newList = new AjxVector();
283     this._cacheMap = {};
284     var size = list.size();
285 
286     var currentTime  = (new Date()).getTime();
287 
288     for (var i = 0; i < size; i++) {
289         var appt = list.get(i);
290         var id = appt.id;
291         if (!this._cacheMap[id]) {
292             this._cacheMap[id] = appt;
293             if(appt.isAllDayEvent()) continue;
294             var diff = appt.getStartTime() - currentTime;
295             var isUpcomingEvent = (diff >= 0 && diff <= AjxDateUtil.MSEC_PER_HOUR)
296             if((currentTime >= appt.getStartTime() && currentTime <= appt.getEndTime()) || isUpcomingEvent) {
297                 appt.isUpcomingEvent = isUpcomingEvent;
298                 newList.add(appt);
299             }
300         }
301     }
302 
303     var qDlg = this.getQuickReminderDialog();
304     qDlg.initialize(newList);
305     qDlg.popup();
306 };
