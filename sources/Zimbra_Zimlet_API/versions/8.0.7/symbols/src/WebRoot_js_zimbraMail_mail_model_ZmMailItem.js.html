 16 /**
 17  * Creates a mail item.
 18  * @constructor
 19  * @class
 20  * This class represents a mail item, which may be a conversation or a mail
 21  * message.
 22  *
 23  * @param {constant}	type		the type of object (conv or msg)
 24  * @param {int}	id		the unique ID
 25  * @param {ZmMailList}	list		the list that contains this mail item
 26  * @param {Boolean}	noCache		if <code>true</code>, do not cache this item
 27  * 
 28  * @extends		ZmItem
 29  */
 30 ZmMailItem = function(type, id, list, noCache) {
 31 
 32 	if (arguments.length == 0) { return; }
 33 	ZmItem.call(this, type, id, list, noCache);
 34 
 35 	this._loaded = false;
 36 	this._initializeParticipants();
 37 };
 38 
 39 ZmMailItem.prototype = new ZmItem;
 40 ZmMailItem.prototype.constructor = ZmMailItem;
 41 
 42 ZmMailItem.sortBy = ZmSearch.DATE_DESC;
 43 ZmMailItem.sortCompare =
 44 function(itemA, itemB) {
 45 	var sortBy = ZmMailItem.sortBy;
 46 	if (!sortBy || (sortBy != ZmSearch.DATE_DESC && sortBy != ZmSearch.DATE_ASC)) { return 0; }
 47 
 48 	var itemDateA = parseInt(itemA.date);
 49 	var itemDateB = parseInt(itemB.date);
 50 	if (sortBy == ZmSearch.DATE_DESC) {
 51 		return (itemDateA > itemDateB) ? -1 : (itemDateA < itemDateB) ? 1 : 0;
 52 	}
 53 	if (sortBy == ZmSearch.DATE_ASC) {
 54 		return (itemDateA > itemDateB) ? 1 : (itemDateA < itemDateB) ? -1 : 0;
 55 	}
 56 };

 84 ZmMailItem.prototype.getColor =
 85 function() {
 86 	if (!this.tags || this.tags.length !== 1) {
 87 		return null;
 88 	}
 89 	var tagList = appCtxt.getAccountTagList(this);
 90 
 91 	var tag = tagList.getByNameOrRemote(this.tags[0]);
 92 
 93 	return tag.getColor();
 94 };

112 ZmMailItem.prototype.notifyModify =
113 function(obj, batchMode) {
114 	var fields = {};
115 	if (obj.e && obj.e.length) {
116 		this._clearParticipants();
117 		this._initializeParticipants();
118 		for (var i = 0; i < obj.e.length; i++) {
119 			this._parseParticipantNode(obj.e[i]);
120 		}
121 		fields[ZmItem.F_FROM] = true;
122 		this._notify(ZmEvent.E_MODIFY, {fields:fields});
123 	}
124 
125 	return ZmItem.prototype.notifyModify.apply(this, arguments);
126 };
127 
128 ZmMailItem.prototype._initializeParticipants =
129 function() {
130 	this.participants = new AjxVector();
131 	this.participantsElided = false;
132 };
133 
134 ZmMailItem.prototype._clearParticipants =
135 function() {
136 	if (this.participants) {
137 		this.participants.removeAll();
138 		this.participants = null;
139 		this.participantsElided = false;
140 	}
141 };

143 ZmMailItem.prototype._getFlags =
144 function() {
145 	var list = ZmItem.prototype._getFlags.call(this);
146 	list.push(ZmItem.FLAG_UNREAD, ZmItem.FLAG_MUTE, ZmItem.FLAG_REPLIED, ZmItem.FLAG_FORWARDED, ZmItem.FLAG_READ_RECEIPT_SENT, ZmItem.FLAG_PRIORITY);
147 	return list;
148 };

150 ZmMailItem.prototype._markReadLocal =
151 function(on) {
152 	this.isUnread = !on;
153 	this._notify(ZmEvent.E_FLAGS, {flags:[ZmItem.FLAG_UNREAD]});
154 };

156 ZmMailItem.prototype._parseParticipantNode =
157 function(node) {
158 	var type = AjxEmailAddress.fromSoapType[node.t];
159 	if (type == AjxEmailAddress.READ_RECEIPT) {
160 		this.readReceiptRequested = true;
161 	} else {
162 		var addr = new AjxEmailAddress(node.a, type, node.p, node.d);
163 		addr.isGroup = node.isGroup;
164 		addr.canExpand = node.isGroup && node.exp;
165 		var ac = window.parentAppCtxt || window.appCtxt;
166 		ac.setIsExpandableDL(node.a, addr.canExpand);
167 		this.participants.add(addr);
168 	}
169 };

181 /**
182  * Checks if this item is in Junk or Trash and the user is not including
183  * those in search results.
184  * 
185  * @return	{Boolean}	<code>true</code> if this item is in the Junk or Trash folder
186  */
187 ZmMailItem.prototype.ignoreJunkTrash =
188 function() {
189 	return Boolean((this.folderId == ZmFolder.ID_SPAM && !appCtxt.get(ZmSetting.SEARCH_INCLUDES_SPAM)) ||
190 				   (this.folderId == ZmFolder.ID_TRASH && !appCtxt.get(ZmSetting.SEARCH_INCLUDES_TRASH)));
191 };

193 ZmMailItem.prototype.setAutoSendTime =
194 function(autoSendTime) {
195 	var wasScheduled = this.isScheduled;
196 	var isDate = AjxUtil.isDate(autoSendTime);
197 	this.flagLocal(ZmItem.FLAG_ISSCHEDULED, isDate);
198 	var autoSendTime = isDate ? autoSendTime : null;
199 	if (autoSendTime != this.autoSendTime) {
200 		this.autoSendTime = autoSendTime;
201 		this._notify(ZmEvent.E_MODIFY);
202 	}
203 	if (wasScheduled != this.isScheduled) {
204 		this._notify(ZmEvent.E_FLAGS, {flags: ZmItem.FLAG_ISSCHEDULED});
205 	}
206 };
