 16 /**
 17  * @constructor
 18  * @class
 19  * This static class provides an interface for send requests to a server. It
 20  * essentially wraps {@link AjxRpcRequest}. This {@link AjxRpc} link maintains a cache of
 21  * {@link AjxRpcRequest} objects which it attempts to reuse before allocating additional
 22  * objects. It also has a mechanism whereby if an {@link AjxRpcRequest} object is 
 23  * in a "busy" state for a extended period of time, it will reap it appropriately.
 24  *
 25  * @author Ross Dargahi
 26  * @author Conrad Damon
 27  * 
 28  * @see AjxRpcRequest
 29  */
 30 AjxRpc = function() {
 31 };
 32 
 33 AjxRpc.__rpcCache		= [];		// The pool of RPC contexts available
 34 AjxRpc.__rpcOutstanding	= {};		// The pool of RPC contexts in use
 35 
 36 AjxRpc.__RPC_CACHE_MAX		= 50;		// maximum number of busy contexts we can have
 37 AjxRpc.__RPC_ID				= 0;		// used for context IDs
 38 AjxRpc.__RPC_IN_USE			= 0;		// number of contexts that are busy
 39 AjxRpc.__RPC_HIGH_WATER		= 0;		// high water mark for busy contexts
 40 AjxRpc.__RPC_REAP_AGE		= 300000;	// 5 minutes; mark any context older than this (in ms) as free
 41 AjxRpc.__RPC_REAP_INTERVAL	= 1800000;	// 30 minutes; run the reaper this often
 42 
 43 /**
 44  * Submits a request to a URL. The request is handled through a pool of request
 45  * contexts (each a wrapped XmlHttpRequest). The context does the real work.
 46  *
 47  * @param {string} [requestStr] 	the HTTP request string/document
 48  * @param {string} serverUrl 	the request target 
 49  * @param {array} [requestHeaders] an array of HTTP request headers
 50  * @param {AjxCallback} callback 	the callback for asynchronous requests. This callback 
 51  * 		will be invoked when the requests completes. It will be passed the same
 52  * 		values as when this method is invoked synchronously (see the return values
 53  * 		below) with the exception that if the call times out (see timeout param 
 54  * 		below), then the object passed to the callback will be the same as in the 
 55  * 		error case with the exception that the status will be set to 
 56  * 		{@link AjxRpcRequest.TIMEDOUT}.
 57  * @param {Constant} [method] 		the HTTP method -- GET, POST, PUT, DELETE. if <code>true</code>, use get method for backward compatibility
 58  * @param {number} [timeout] 		the timeout (in milliseconds) after which the request is canceled
 59  * 
 60  * @return {object|hash}	if invoking in asynchronous mode, then it will return the id of the 
 61  * 		underlying {@link AjxRpcRequest} object. Else if invoked synchronously, if
 62  * 		there is no error (i.e. we get a HTTP result code of 200 from the server),
 63  * 		an object with the following attributes is returned
 64  * 		<ul>
 65  * 		<li>text - the string response text</li>
 66  * 		<li>xml - the string response xml</li>
 67  * 		<li>success - boolean set to true</li>
 68  * 		</ul>
 69  * 		If there is an error, then the following will be returned
 70  * 		<ul>
 71  * 		<li>text - the string response text<li>
 72  * 		<li>xml - the string response xml </li>
 73  * 		<li>success - boolean set to <code>false</code></li>
 74  * 		<li>status - HTTP status</li>
 75  * 		</ul>
 76  * 
 77  * @throws	{AjxException.NETWORK_ERROR}	a network error occurs
 78  * @throws	{AjxException.UNKNOWN_ERROR}	an unknown error occurs
 79  * 
 80  * @see	AjxRpcRequest#invoke
 81  * 
 82  */
 83 AjxRpc.invoke =
 84 function(requestStr, serverUrl, requestHeaders, callback, method, timeout) {
 85 
 86 	var asyncMode = (callback != null);
 87 	var rpcCtxt = AjxRpc.__getFreeRpcCtxt();
 88 
 89 	try {
 90 		var response = rpcCtxt.invoke(requestStr, serverUrl, requestHeaders, callback, method, timeout);
 91 	} catch (ex) {
 92 		var newEx = new AjxException();
 93 		newEx.method = "AjxRpc.prototype._invoke";
 94 		if (ex instanceof Error) {
 95 			newEx.detail = ex.message;
 96 			newEx.code = AjxException.NETWORK_ERROR;
 97 			newEx.msg = "Network error";
 98 		} else if (ex.code == 101){
 99 			// Chrome 
100 			newEx.detail = ex.message;
101 			newEx.code = AjxException.NETWORK_ERROR;
102 			newEx.msg = "Network error";
103 		} else {
104 			newEx.detail = ex.toString();
105 			newEx.code = AjxException.UNKNOWN_ERROR;
106 			newEx.msg = "Unknown Error";
107 		}
108 		// exception hit: we're done whether sync or async, free the context
109 		AjxRpc.freeRpcCtxt(rpcCtxt);
110 		throw newEx;
111 	}
112 	if (!asyncMode) {
113 		// we've returned from a sync request, free the context
114 		AjxRpc.freeRpcCtxt(rpcCtxt);
115 	}
116 	return response;
117 };
118 
119 /**
120  * @private
121  */
122 AjxRpc.freeRpcCtxt =
123 function(rpcCtxt) {
124 	// we're done using this rpcCtxt. Add it back to the pool
125 	if (AjxRpc.__rpcOutstanding[rpcCtxt.id]) {
126 		AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " --- freeing rpcCtxt " + rpcCtxt.id);
127 		AjxRpc.__rpcCache.push(rpcCtxt);
128 		delete AjxRpc.__rpcOutstanding[rpcCtxt.id];
129 		AjxRpc.__RPC_IN_USE--;
130 	}
131 };
132 
133 AjxRpc.removeRpcCtxt =
134 function(rpcCtxt) {
135 	AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " REMOVE rpcCtxt " + rpcCtxt.id);
136 	if (AjxRpc.__rpcOutstanding[rpcCtxt.id]) {
137 		delete AjxRpc.__rpcOutstanding[rpcCtxt.id];
138 		AjxRpc.__RPC_IN_USE--;
139 	}
140 	AjxUtil.arrayRemove(AjxRpc.__rpcCache, rpcCtxt);
141 };

143 /**
144  * Returns the request from the RPC context with the given ID.
145  *
146  * @param {String} id RPC context ID
147  * 
148  * @return The <i>AjxRpcRequest</i> object associated with <code>id</code> or null
149  * 		if no object exists for the supplied id
150  * @type AjxRpcRequest
151  * 
152  * @private
153  */
154 AjxRpc.getRpcRequestById = 
155 function(id) {
156 	return (AjxRpc.__rpcOutstanding[id]);
157 };

159 /**
160  * Factory method for getting context objects.
161  * 
162  * @private
163  */
164 AjxRpc.__getFreeRpcCtxt = 
165 function() {
166 	var rpcCtxt;
167 
168 	if (AjxRpc.__rpcCache.length > 0) {
169 		rpcCtxt = AjxRpc.__rpcCache.pop();
170 		AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " reusing RPC ID " + rpcCtxt.id);
171 	} else {
172 		if (AjxRpc.__RPC_IN_USE < AjxRpc.__RPC_CACHE_MAX) {
173 			// we haven't reached our limit, so create a new AjxRpcRequest
174 			var id = "__RpcCtxt_" + AjxRpc.__RPC_ID;
175 			rpcCtxt = new AjxRpcRequest(id);
176 			AjxRpc.__RPC_ID++;
177 			AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " Created RPC " + id);
178 		} else {
179 			// yikes, we're out of rpc's! Look for an old one to kill.
180 			rpcCtxt = AjxRpc.__reap();
181 
182 			// if reap didn't find one either, bail.
183 			if (!rpcCtxt) {
184 				var text = [];
185 				for (var i in AjxRpc.__rpcOutstanding) {
186 					var rpcCtxt = AjxRpc.__rpcOutstanding[i];
187 					text.push(rpcCtxt.methodName);
188 				}
189 				var detail = text.join("<br>") + "<br>";
190 				AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " Out of RPC cache!!! Outstanding requests: " + detail);
191 				throw new AjxException("Out of RPC cache", AjxException.OUT_OF_RPC_CACHE, "AjxRpc.__getFreeRpcCtxt", detail);
192 			}
193 		}
194 	}
195 
196 	AjxRpc.__rpcOutstanding[rpcCtxt.id] = rpcCtxt;
197 	AjxRpc.__RPC_IN_USE++;
198 	if (AjxRpc.__RPC_IN_USE > AjxRpc.__RPC_HIGH_WATER) {
199 		AjxRpc.__RPC_HIGH_WATER = AjxRpc.__RPC_IN_USE;
200 		AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " High water mark: " + AjxRpc.__RPC_HIGH_WATER);
201 	}
202 
203 	// always reset timestamp before returning rpcCtxt
204 	rpcCtxt.timestamp = (new Date()).getTime();
205 	return rpcCtxt;
206 };

208 /**
209  * Frees expired contexts.
210  * 
211  * @param {boolean}	all		if true, frees all expired contexts; otherwise, returns the first one it finds
212  * @private
213  */
214 AjxRpc.__reap =
215 function(all) {
216 	var rpcCtxt;
217 	var time = (new Date()).getTime();
218 	AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " Running RPC context reaper");
219 	for (var i in AjxRpc.__rpcOutstanding) {
220 		rpcCtxt = AjxRpc.__rpcOutstanding[i];
221 		if ((rpcCtxt.timestamp + AjxRpc.__RPC_REAP_AGE) < time) {
222 			DBG.println(AjxDebug.DBG1, "AjxRpc.__reap: cleared RPC context " + rpcCtxt.id);
223 			AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " AjxRpc.__reap: cleared RPC context " + rpcCtxt.id);
224 			rpcCtxt.cancel();
225 			delete AjxRpc.__rpcOutstanding[i];
226 			AjxRpc.__RPC_IN_USE--;
227 			if (!all) {
228 				return rpcCtxt;
229 			}
230 		}
231 	}
232 	return null;
233 };
234 
235 window.setInterval(AjxRpc.__reap.bind(null, true), AjxRpc.__RPC_REAP_INTERVAL);
