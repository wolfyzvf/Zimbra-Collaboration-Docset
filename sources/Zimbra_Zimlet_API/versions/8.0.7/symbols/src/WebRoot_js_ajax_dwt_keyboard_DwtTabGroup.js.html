 17 /**
 18  * Creates an empty tab group.
 19  * @constructor
 20  * @class
 21  * A tab group is used to manage keyboard focus among a group of related visual 
 22  * elements. It is a tree structure consisting of elements and other tab groups.
 23  * <p>
 24  * The root tab group is the only one without a parent tab group, and is the one
 25  * that the application interacts with. Focus listeners register with the root
 26  * tab group. The root tab group tracks where focus is.
 27  * 
 28  * @param {string}	name					the name of this tab group
 29  * @param {boolean}	blockDefaultHandling	if <code>true</code>, do not fall back to default key
 30  * 											handler for this tab group
 31  * 
 32  * @author Ross Dargahi
 33  */
 34 DwtTabGroup = function(name, blockDefaultHandling) {
 35 
 36 	this.__members = new AjxVector();
 37 	this.__parent = null;
 38 	this.__name = name;
 39 	this.__blockApplicationHandling = blockDefaultHandling;
 40 	this.__currFocusMember = null;
 41 	this.__evtMgr = new AjxEventMgr();
 42 };

100 /**
101  * Adds a member to the tab group.
102  * 
103  * @param {Array|DwtControl|DwtTabGroup|HTMLElement} member	the member(s) to be added
104  * @param {number} [index] 		the index at which to add the member. If omitted, the member
105  * 		will be added to the end of the tab group
106  */
107 DwtTabGroup.prototype.addMember =
108 function(member, index) {
109 	if (!member) {return;}
110 	var members = (member instanceof Array) ? member : [member];
111 
112 	for (var i = 0, len = members.length; i < len; i++) {
113 		this.__members.add(members[i], index);
114 
115 		// If adding a tab group, register me as its parent
116 		if (members[i] instanceof DwtTabGroup) {
117 			members[i].newParent(this);
118 		}
119 	}
120 };

167 /**
168  * This method replaces a member in the tab group with a new member. If the member being
169  * replaced is currently the focus member, then we will try to set focus to the
170  * previous member. If that fails, we will try the next member.
171  * 
172  * @param {DwtControl|DwtTabGroup|HTMLElement} oldMember 	the member to be replaced
173  * @param {DwtControl|DwtTabGroup|HTMLElement} newMember 	the replacing member
174  * 		If this parameter is <code>null</code>, then this method effectively removes <code>oldMember</code>
175  * @param {boolean} [checkEnabled] 	if <code>true</code>, then make sure that if we have a newly focused
176  * 		member it is enabled
177  * @param {boolean} [skipNotify] if <code>true</code>, notification is not fired. This flag is
178  * 		typically set by the tab management framework when it is calling into this method
179  * @return {DwtControl|DwtTabGroup|HTMLElement}	replaced member or <code>null></code> if <code>oldMember</code> is not in the tab group
180  */
181 DwtTabGroup.prototype.replaceMember =
182 function(oldMember, newMember, checkEnabled, skipNotify, focusItem, noFocus) {
183 	var tg = this.__getTabGroupForMember(oldMember);
184 	if (!tg) {
185 		this.addMember(newMember);
186 		return null;
187 	}
188 
189 	/* If we are removing the current focus member, then we need to adjust the focus
190 	 * member index. If the tab group is empty as a result of the removal
191 	 */
192 	var root = this.__getRootTabGroup();
193 	var newFocusMember;
194 	if (focusItem) {
195 		newFocusMember = focusItem;
196 	} else if (root.__currFocusMember == oldMember ||
197 		((oldMember instanceof DwtTabGroup) && oldMember.contains(root.__currFocusMember))) {
198 
199 		if (newMember) {
200 			newFocusMember = (newMember instanceof DwtTabGroup) ? newMember.getFirstMember() : newMember;
201 		} else {
202 			newFocusMember = this.__getPrevMember(oldMember, checkEnabled);
203 			if (!newFocusMember) {
204 				newFocusMember =  this.__getNextMember(oldMember, checkEnabled);
205 			}
206 		}
207 	}
208 	if (newFocusMember && !noFocus) {
209 		root.__currFocusMember = newFocusMember;
210 //		DBG.println("kbnav", "DwtTabGroup.replaceMember: current focus member is now " + root.__currFocusMember);
211 		if (!skipNotify) {
212 			this.__notifyListeners(newFocusMember);
213 		}
214 	}
215 		
216 	return newMember ? this.__members.replaceObject(oldMember, newMember) : this.__members.remove(oldMember);
217 };

293 /**
294  * Sets the current focus member. 
295  * 
296  * @param {DwtControl|HTMLElement} member 		the member to which to set focus
297  * @param {boolean} [checkEnabled] 	if <code>true</code>, then make sure the member is enabled
298  * @param {boolean} [skipNotify] if <code>true</code>, notification is not fired. This flag
299  * 		typically set by Dwt tab management framework when it is calling into this method
300  * 
301  * @return {boolean}	<code>true</code> if member was part of the tab group hierarchy, else false
302  *
303  * @throws DwtTabGroup.NOT_ROOT_TABGROUP
304  */
305 DwtTabGroup.prototype.setFocusMember =
306 function(member, checkEnabled, skipNotify) {
307 	this.__checkRoot();	
308 	if (!this.__checkEnabled(member, checkEnabled)) {
309 		return false;
310 	}
311 
312 	var tg = this.__getTabGroupForMember(member);
313 	if (tg) {
314 		this.__currFocusMember = member;
315 //		DBG.println("kbnav", "DwtTabGroup.setFocusMember: current focus member is now " + this.__currFocusMember);
316 		if (!skipNotify) {
317 			this.__notifyListeners(this.__currFocusMember);
318 		}
319 		return true;	
320 	}
321 	return false;
322 };

362 /**
363  * Resets the the focus member to the first element in the tab group.
364  * 
365  * @param {boolean} [checkEnabled] 	if <code>true</code>, then pick a enabled member to which to set focus
366  * @param {boolean} [skipNotify] if <code>true</code>, notification is not fired. This flag
367  * 		typically set by Dwt tab management framework when it is calling into this method
368  * 
369  * @return {DwtControl|HTMLElement}	the new focus member
370  *
371  * @throws DwtTabGroup.NOT_ROOT_TABGROUP
372  */
373 DwtTabGroup.prototype.resetFocusMember =
374 function(checkEnabled, skipNotify) {
375 	this.__checkRoot();
376 	var focusMember = this.__getLeftMostMember(checkEnabled);
377 	if ((focusMember != this.__currFocusMember) && !skipNotify) {
378 		this.__notifyListeners(this.__currFocusMember);
379 	}
380 //	DBG.println("kbnav", "DwtTabGroup.resetFocusMember: current focus member is now " + this.__currFocusMember);
381 	this.__currFocusMember = focusMember;
382 	
383 	return this.__currFocusMember;
384 };

417 /**
418  * Returns the previous member in the tag group.
419  * 
420  * @private
421  */
422 DwtTabGroup.prototype.__getPrevMember =
423 function(member, checkEnabled) {
424 	var a = this.__members.getArray();
425 	// Start working from the member to the immediate left, then keep going left
426 	for (var i = this.__members.indexOf(member) - 1; i > -1; i--) {
427 		var prevMember = a[i];
428 		/* if sibling is not a tab group, then it is the previous child. If the
429 		 * sibling is a tab group, get its rightmost member.*/
430 		if (!(prevMember instanceof DwtTabGroup)) {
431 			if (this.__checkEnabled(prevMember, checkEnabled)) {
432 				return prevMember;
433 			}
434 		} else {
435 			prevMember = prevMember.__getRightMostMember(checkEnabled);
436 			if (prevMember && this.__checkEnabled(prevMember, checkEnabled)) {
437 				return prevMember;
438 			}
439 		}
440 	}
441 	/* If we have fallen through to here it is because the tab group only has 
442 	 * one member. So we roll up to the parent, unless we are at the root in 
443 	 * which case we return null. */
444 	return this.__parent ? this.__parent.__getPrevMember(this, checkEnabled) : null;
445 };

447 /**
448  * Returns true if the given member can accept focus, or if there is no need to check.
449  * If we are checking, the member must be enabled and visible if it is a control, and
450  * enabled otherwise. A member may also set the "noTab" flag to take itself out of the
451  * tab hierarchy.
452  * 
453  * @private
454  */
455 DwtTabGroup.prototype.__checkEnabled =
456 function(member, checkEnabled) {
457 	if (!checkEnabled) return true;
458 	if (!member || member.noTab) return false;
459 	if (member instanceof DwtControl) {
460 		return (member.getEnabled() && member.getVisible());
461 	} else {
462 		return !member.disabled && Dwt.getVisible(member);
463 	}
464 };

466 /**
467  * Sets and returns the next member in the tag group.
468  * 
469  * @private
470  */
471 DwtTabGroup.prototype.__getNextMember =
472 function(member, checkEnabled) {
473 	var a = this.__members.getArray();
474 	var sz = this.__members.size();
475 
476 	// Start working from the member to the immediate left of <member> rightwards
477 	for (var i = this.__members.indexOf(member) + 1; i < sz; i++) {
478 		var nextMember = a[i];
479 		/* if sibling is not a tab group, then it is the next child. If the
480 		 * sibling is a tab group, get its leftmost member.*/
481 		if (!(nextMember instanceof DwtTabGroup)) {
482 			if (this.__checkEnabled(nextMember, checkEnabled)) {
483 				return nextMember;
484 			}
485 		} else {
486 			nextMember = nextMember.__getLeftMostMember(checkEnabled);
487 			if (nextMember && this.__checkEnabled(nextMember, checkEnabled)) {
488 				return nextMember;
489 			}
490 		}
491 	}
492 
493 	/* If we have fallen through to here it is because the tab group only has 
494 	 * one member or we are at the end of the list. So we roll up to the parent, 
495 	 * unless we are at the root in which case we return null. */
496 	return this.__parent ? this.__parent.__getNextMember(this, checkEnabled) : null;
497 };

499 /**
500  * Finds the rightmost member of the tab group. Will recurse down
501  * into contained tab groups if necessary.
502  * @private
503  */
504 DwtTabGroup.prototype.__getRightMostMember =
505 function(checkEnabled) {
506 	var a = this.__members.getArray();
507 	var member = null;
508 	
509 	/* Work backwards from the rightmost member. If the member is a tab group, then
510 	 * recurse into it. If member is not a tab group, return it as it is the 
511 	 * rightmost element. */
512 	for (var i = this.__members.size() - 1; i >= 0; i--) {
513 		member = a[i]
514 		if (!(member instanceof DwtTabGroup)) {
515 			if (this.__checkEnabled(member, checkEnabled)) break;
516 		} else {
517 			member = member.__getRightMostMember(checkEnabled);
518 			if (member && this.__checkEnabled(member, checkEnabled)) break;
519 		}
520 	}
521 
522 	return (member && this.__checkEnabled(member, checkEnabled)) ? member : null;
523 };
524 
525 /**
526  *  Finds the rightmost member of the tab group. Will recurse down
527  * into contained tab groups if necessary.
528  * @private
529  */
530 DwtTabGroup.prototype.__getLeftMostMember =
531 function(checkEnabled) {
532 	var sz = this.__members.size();
533 	var a = this.__members.getArray();
534 	var member = null;
535 
536 	/* Work forwards from the leftmost member. If the member is a tabgroup, then
537 	 * recurse into it. If member is not a tabgroup, return it as it is the 
538 	 * rightmost element */
539 	for (var i = 0; i < sz; i++) {
540 		member = a[i]
541 		if (!(member instanceof DwtTabGroup)) {
542 			if  (this.__checkEnabled(member, checkEnabled)) break;
543 		} else {
544 			member = member.__getLeftMostMember(checkEnabled);
545 			if (member && this.__checkEnabled(member, checkEnabled)) break;
546 		}
547 	}
548 
549 	return (member && this.__checkEnabled(member, checkEnabled)) ? member : null;
550 };

552 /**
553  * Notifies focus change listeners.
554  * @private
555  */
556 DwtTabGroup.prototype.__notifyListeners =
557 function(newFocusMember) {
558 	// Only the root tab group will issue notifications
559 	var rootTg = this.__getRootTabGroup();
560 	if (rootTg.__evtMgr) {
561 		var evt = DwtTabGroup.__changeEvt;
562 		evt.reset();
563 		evt.tabGroup = this;
564 		evt.newFocusMember = newFocusMember;
565 		rootTg.__evtMgr.notifyListeners(DwtEvent.STATE_CHANGE, evt);
566 	}
567 };

569 /**
570  * @private
571  */
572 DwtTabGroup.prototype.__getRootTabGroup =
573 function() {
574 	var root = this;
575 	while (root.__parent) {
576 		root = root.__parent;
577 	}
578 	
579 	return root;
580 }

582 /**
583  * @private
584  */
585 DwtTabGroup.prototype.__dump =
586 function(tg, debugLevel, level) {
587 	level = level || 0;
588 	var levelIndent = "";
589 	for (var i = 0; i < level; i++) {
590 		levelIndent += "    ";
591 	}
592 	
593 	debugLevel = debugLevel || AjxDebug.DBG1;
594 	DBG.println(debugLevel, levelIndent + " TABGROUP: " + tg.__name);
595 	levelIndent += "    ";
596 	
597 	var sz = tg.__members.size();
598 	var a = tg.__members.getArray();
599 	for (var i = 0; i < sz; i++) {
600 		if (a[i] instanceof DwtTabGroup) {
601 			tg.__dump(a[i], debugLevel, level + 1);
602 		} else if (a[i].toString) {
603 			DBG.println(debugLevel, levelIndent + "   " + a[i].toString());
604 		} else {
605 			DBG.println(debugLevel, levelIndent + "   " + a[i].tagName);
606 		}
607 	}
608 };

610 /**
611  * Sets the next or previous focus member.
612  * @private
613  */
614 DwtTabGroup.prototype.__setFocusMember =
615 function(next, checkEnabled, skipNotify) {
616 	// If there is currently no focus member, then reset to the first member
617 	// and return
618 	if (!this.__currFocusMember) {
619 		return this.resetFocusMember(checkEnabled, skipNotify);
620 	}
621 	
622 	var tabGroup = this.__getTabGroupForMember(this.__currFocusMember);
623 	if (!tabGroup) {
624 		DBG.println(AjxDebug.DBG1, "tab group not found for focus member");
625 		return null;
626 	}
627 	var m = (next) ? tabGroup.__getNextMember(this.__currFocusMember, checkEnabled) 
628 				   : tabGroup.__getPrevMember(this.__currFocusMember, checkEnabled);
629 
630 	if (!m) {
631 		m = (next) ? this.__getLeftMostMember(checkEnabled)
632 				   : this.__getRightMostMember(checkEnabled);
633 
634 		// Test for the case where there is only one member in the tabgroup
635 		if (m == this.__currFocusMember) {
636 			return null;
637 		}
638 	}
639 
640 	this.__currFocusMember = m;
641 	
642 //	DBG.println("kbnav", "DwtTabGroup._setFocusMember: current focus member is now " + this.__currFocusMember);
643 	if (!skipNotify) {
644 		this.__notifyListeners(this.__currFocusMember);
645 	}
646 	
647 	return this.__currFocusMember;
648 };

653 DwtTabGroup.prototype.__getTabGroupForMember =
654 function(member) {
655 	if (!member) return null;
656 	var sz = this.__members.size();
657 	var a = this.__members.getArray();
658 	var m;
659 	for (var i = 0; i < sz; i++) {
660 		m = a[i];
661 		if (m == member) {
662 			return this;
663 		} else if (m instanceof DwtTabGroup && (m = m.__getTabGroupForMember(member))) {
664 			return m;
665 		}
666 	}
667 	return null;
668 };
