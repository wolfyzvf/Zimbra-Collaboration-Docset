 20 /**
 21  * Creates a folder properties dialog.
 22  * @class
 23  * This class represents a folder properties dialog.
 24  * 
 25  * @param	{DwtControl}	parent		the parent
 26  * @param	{String}	className		the class name
 27  * 
 28  * @extends		DwtDialog
 29  */
 30 ZmFolderPropsDialog = function(parent, className) {
 31 	className = className || "ZmFolderPropsDialog";
 32 	var extraButtons;
 33 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
 34 		extraButtons = [
 35 			new DwtDialog_ButtonDescriptor(ZmFolderPropsDialog.ADD_SHARE_BUTTON, ZmMsg.addShare, DwtDialog.ALIGN_LEFT)
 36 		];
 37 	}
 38 
 39 
 40 	DwtDialog.call(this, {parent:parent, className:className, title:ZmMsg.folderProperties, extraButtons:extraButtons, id:"FolderProperties"});
 41 
 42 	this._tabViews  = [];
 43 	this._tabKeys   = [];
 44 	this._tabInUse  = [];
 45 	this._tabKeyMap = {};
 46 
 47 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
 48 		this.registerCallback(ZmFolderPropsDialog.ADD_SHARE_BUTTON, this._handleAddShareButton, this);
 49 	}
 50 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 51 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._handleCancelButton));
 52 
 53 	this._folderChangeListener = new AjxListener(this, this._handleFolderChange);
 54 
 55 	this._createView();
 56 };

 85 /**
 86  * Pops-up the properties dialog.
 87  * 
 88  * @param	{ZmOrganizer}	organizer		the organizer
 89  */
 90 ZmFolderPropsDialog.prototype.popup =
 91 function(organizer) {
 92 	this._organizer = organizer;
 93     for (var i = 0; i < this._tabViews.length; i++) {
 94         this._tabViews[i].setOrganizer(organizer);
 95     }
 96     // On popup, make the property view visible
 97 	var tabKey = this.getTabKey(ZmFolderPropsDialog.TABKEY_PROPERTIES);
 98 	this._tabContainer.switchToTab(tabKey, true);
 99 
100 	organizer.addChangeListener(this._folderChangeListener);
101 
102 	this._handleFolderChange();
103 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
104 		var isShareable = ZmOrganizer.SHAREABLE[organizer.type];
105 
106 		var isVisible = (!organizer.link || organizer.isAdmin());
107 		this.setButtonVisible(ZmFolderPropsDialog.ADD_SHARE_BUTTON, isVisible && isShareable);
108 	}
109 
110 	DwtDialog.prototype.popup.call(this);
111 };

113 ZmFolderPropsDialog.prototype.popdown =
114 function() {
115 	if (this._organizer) {
116 		this._organizer.removeChangeListener(this._folderChangeListener);
117 		this._organizer = null;
118 	}
119 	DwtDialog.prototype.popdown.call(this);
120 };

145 ZmFolderPropsDialog.prototype._handleResendShare =
146 function(event, share) {
147 
148 	AjxDispatcher.require("Share");
149 	share = share || Dwt.getObjectFromElement(DwtUiEvent.getTarget(event));
150 
151 	// create share info
152 	var tmpShare = new ZmShare({object:share.object});
153 	tmpShare.grantee.id = share.grantee.id;
154 	tmpShare.grantee.email = (share.grantee.type == "guest") ? share.grantee.id : share.grantee.name;
155 	tmpShare.grantee.name = share.grantee.name;
156     if (tmpShare.object.isRemote()) {
157 		tmpShare.grantor.id = tmpShare.object.zid;
158 		tmpShare.grantor.email = tmpShare.object.owner;
159 		tmpShare.grantor.name = tmpShare.grantor.email;
160 		tmpShare.link.id = tmpShare.object.rid;
161 	} else {
162 		tmpShare.grantor.id = appCtxt.get(ZmSetting.USERID);
163 		tmpShare.grantor.email = appCtxt.get(ZmSetting.USERNAME);
164 		tmpShare.grantor.name = appCtxt.get(ZmSetting.DISPLAY_NAME) || tmpShare.grantor.email;
165 		tmpShare.link.id = tmpShare.object.id;
166 	}
167 
168 	tmpShare.link.name = share.object.name;
169 	tmpShare.link.view = ZmOrganizer.getViewName(share.object.type);
170 	tmpShare.link.perm = share.link.perm;
171 
172 	if (share.grantee.type == "guest") {
173         // Pass action as ZmShare.NEW even for resend for external user
174         tmpShare._sendShareNotification(tmpShare.grantee.email, tmpShare.link.id, "", ZmShare.NEW);
175 	}
176     else {
177 	    tmpShare.sendMessage(ZmShare.NEW);
178     }
179 	appCtxt.setStatusMsg(ZmMsg.resentShareMessage);
180 
181 	return false;
182 };

190 ZmFolderPropsDialog.prototype._handleOkButton =
191 function(event) {
192 
193 	// New batch command, stop on error
194 	var batchCommand = new ZmBatchCommand(null, null, true);
195     var saveState = {
196         commandCount: 0,
197         errorMessage: []
198     };
199     for (var i = 0; i < this._tabViews.length; i++) {
200         if (this._tabInUse[i]) {
201             // Save all in use tabs
202             this._tabViews[i].doSave(batchCommand, saveState);
203         }
204     }
205 
206     if (saveState.errorMessage.length > 0) {
207         var msg = saveState.errorMessage.join("<br>");
208         var dialog = appCtxt.getMsgDialog();
209         dialog.reset();
210         dialog.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
211         dialog.popup();
212     }  else if (saveState.commandCount > 0) {
213         var callback = new AjxCallback(this, this.popdown);
214         batchCommand.run(callback);
215     } else {
216         this.popdown();
217     }
218 };

235 ZmFolderPropsDialog.prototype._handleFolderChange =
236 function(event) {
237 	var organizer = this._organizer;
238 
239     // Get the components that will be hidden or displayed
240     var tabBar = this._tabContainer.getTabBar();
241     var tabKey = this.getTabKey(ZmFolderPropsDialog.TABKEY_RETENTION);
242     var retentionTabButton = this._tabContainer.getTabButton(tabKey);
243     var retentionIndex = this._tabKeyMap[ZmFolderPropsDialog.TABKEY_RETENTION];
244 
245     if ((organizer.type != ZmOrganizer.FOLDER) || organizer.link) {
246         // Not a folder, or shared - hide the retention view (and possibly the toolbar)
247         this._tabInUse[retentionIndex] = false;
248         if (this._tabViews.length > 2) {
249             // More than two tabs, hide the retention tab, leave the toolbar intact
250             tabBar.setVisible(true);
251             retentionTabButton.setVisible(false);
252         } else {
253             // Two or fewer tabs.  Hide the toolbar, just let the properties view display standalone
254             // (On popup, the display defaults to the property view)
255             tabBar.setVisible(false);
256         }
257     } else {
258         // Using the retention tab view - show the toolbar and all tabs
259         this._tabInUse[retentionIndex] = true;
260         retentionTabButton.setVisible(true);
261         tabBar.setVisible(true);
262     }
263 
264     for (var i = 0; i < this._tabViews.length; i++) {
265         if (this._tabInUse[i]) {
266             // Update all in use tabs to use the specified folder
267             this._tabViews[i]._handleFolderChange(event);
268         }
269     }
270 
271 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
272 		this._populateShares(organizer);
273 	}
274 
275 };

277 ZmFolderPropsDialog.prototype._populateShares =
278 function(organizer) {
279 
280 	this._sharesGroup.setContent("");
281 
282 	var displayShares = this._getDisplayShares(organizer);
283 
284 	var getFolder = false;
285 	if (displayShares.length) {
286 		for (var i = 0; i < displayShares.length; i++) {
287 			var share = displayShares[i];
288 			if (!(share.grantee && share.grantee.name)) {
289 				getFolder = true;
290 			}
291 		}
292 	}
293 
294 	if (getFolder) {
295 		var respCallback = new AjxCallback(this, this._handleResponseGetFolder, [displayShares]);
296 		organizer.getFolder(respCallback);
297 	} else {
298 		this._handleResponseGetFolder(displayShares);
299 	}
300 
301 
302 	this._sharesGroup.setVisible(displayShares.length > 0);
303 };

305 ZmFolderPropsDialog.prototype._getDisplayShares =
306 function(organizer) {
307 
308 	var shares = organizer.shares;
309 	var displayShares = [];
310 	if ((!organizer.link || organizer.isAdmin()) && shares && shares.length > 0) {
311 		AjxDispatcher.require("Share");
312 		var userZid = appCtxt.accountList.mainAccount.id;
313 		// if a folder was shared with us with admin rights, a share is created since we could share it;
314 		// don't show any share that's for us in the list
315 		for (var i = 0; i < shares.length; i++) {
316 			var share = shares[i];
317 			if (share.grantee) {
318 				var granteeId = share.grantee.id;
319 				if ((share.grantee.type != ZmShare.TYPE_USER) || (share.grantee.id != userZid)) {
320 					displayShares.push(share);
321 				}
322 			}
323 		}
324 	}
325 
326 	return displayShares;
327 };

329 ZmFolderPropsDialog.prototype._handleResponseGetFolder =
330 function(displayShares, organizer) {
331 
332 	if (organizer) {
333 		displayShares = this._getDisplayShares(organizer);
334 	}
335 
336 	if (displayShares.length) {
337 		var table = document.createElement("TABLE");
338 		table.className = "ZPropertySheet";
339 		table.cellSpacing = "6";
340 		for (var i = 0; i < displayShares.length; i++) {
341 			var share = displayShares[i];
342 			var row = table.insertRow(-1);
343 			var nameEl = row.insertCell(-1);
344 			nameEl.style.paddingRight = "15px";
345 			var nameText = share.grantee && share.grantee.name;
346 			if (share.isAll()) {
347 				nameText = ZmMsg.shareWithAll;
348 			} else if (share.isPublic()) {
349 				nameText = ZmMsg.shareWithPublic;
350 			} else if (share.isGuest()){
351 				nameText = nameText || (share.grantee && share.grantee.id);
352 			}
353 			nameEl.innerHTML = AjxStringUtil.htmlEncode(nameText);
354 
355 			var roleEl = row.insertCell(-1);
356 			roleEl.style.paddingRight = "15px";
357 			roleEl.innerHTML = ZmShare.getRoleName(share.link.role);
358 
359 			this.__createCmdCells(row, share);
360 		}
361 		this._sharesGroup.setElement(table);
362 
363 		var width = Dwt.DEFAULT;
364 		var height = displayShares.length > 5 ? ZmFolderPropsDialog.SHARES_HEIGHT : Dwt.CLEAR;
365 
366 		var insetElement = this._sharesGroup.getInsetHtmlElement();
367 		Dwt.setScrollStyle(insetElement, Dwt.SCROLL);
368 		Dwt.setSize(insetElement, width, height);
369 	}
370 
371 	this._sharesGroup.setVisible(displayShares.length > 0);
372 };

374 ZmFolderPropsDialog.prototype.__createCmdCells =
375 function(row, share) {
376 	var type = share.grantee.type;
377 	if (type == ZmShare.TYPE_DOMAIN || !share.link.role) {
378 		var cell = row.insertCell(-1);
379 		cell.colSpan = 3;
380 		cell.innerHTML = ZmMsg.configureWithAdmin;
381 		return;
382 	}
383 
384 	var actions = [ZmShare.EDIT, ZmShare.REVOKE, ZmShare.RESEND];
385 	var handlers = [this._handleEditShare, this._handleRevokeShare, this._handleResendShare];
386 
387 	for (var i = 0; i < actions.length; i++) {
388 		var action = actions[i];
389 		var cell = row.insertCell(-1);
390 
391 		// public shares have no editable fields, and sent no mail
392 		var isAllShare = share.grantee && (share.grantee.type == ZmShare.TYPE_ALL);
393 		if (((isAllShare || share.isPublic() || share.isGuest()) && (action == ZmShare.EDIT)) ||
394             ((isAllShare || share.isPublic()) && action == ZmShare.RESEND)) { continue; }
395 
396 		var link = document.createElement("A");
397 		link.href = "#";
398 		link.innerHTML = ZmShare.ACTION_LABEL[action];
399 
400 		Dwt.setHandler(link, DwtEvent.ONCLICK, handlers[i]);
401 		Dwt.associateElementWithObject(link, share);
402 
403 		cell.appendChild(link);
404 	}
405 };

407 ZmFolderPropsDialog.prototype.addTab =
408 function(index, id, tabViewPage) {
409 	if (!this._tabContainer || !tabViewPage) { return null; }
410 
411 	this._tabViews[index] = tabViewPage;
412 	this._tabKeys[index]  = this._tabContainer.addTab(tabViewPage.getTitle(), tabViewPage);
413 	this._tabInUse[index] = true;
414 	this._tabKeyMap[id] = index;
415 	return this._tabKeys[index];
416 };

418 ZmFolderPropsDialog.prototype._initializeTabView =
419 function(view) {
420     this._tabContainer = new DwtTabView(view, null, Dwt.STATIC_STYLE);
421 
422 	//ZmFolderPropertyView handle things such as color and type. (in case you're searching for "color" and can't find in this file. I know I did)
423     this.addTab(0, ZmFolderPropsDialog.TABKEY_PROPERTIES, new ZmFolderPropertyView(this, this._tabContainer));
424     this.addTab(1, ZmFolderPropsDialog.TABKEY_RETENTION,  new ZmFolderRetentionView(this, this._tabContainer));
425 
426 	// setup shares group
427 	if (appCtxt.get(ZmSetting.SHARING_ENABLED))	{
428         this._sharesGroup = new DwtGrouper(view, "DwtGrouper ZmFolderPropSharing");
429 		this._sharesGroup.setLabel(ZmMsg.folderSharing);
430 		this._sharesGroup.setVisible(false);
431 		this._sharesGroup.setScrollStyle(Dwt.SCROLL);
432         view.getHtmlElement().appendChild(this._sharesGroup.getHtmlElement());
433 	}
434 };
