 80 /**
 81  * Loads the zimlets.
 82  * 
 83  * @param	{Array}	zimletArray		an array of {@link ZmZimlet} objects
 84  * @param	{Array}	userProps		an array of properties
 85  * @param	{String}	target		the target
 86  * @param	{AjxCallback}	callback	the callback
 87  * @param	{Boolean}	sync		<code>true</code> for synchronous
 88  * 
 89  * @private
 90  */
 91 ZmZimletMgr.prototype.loadZimlets =
 92 function(zimletArray, userProps, target, callback, sync) {
 93 	var href = window.location.href.toLowerCase();
 94 	if(href.indexOf("zimlets=none") > 0) {
 95 		return;
 96 	} else if(href.indexOf("zimlets=core") > 0) {
 97 		zimletArray = this._getCoreZimlets(zimletArray);
 98 	}
 99 	var isHttp = document.location.protocol == ZmSetting.PROTO_HTTP;
100 	var isMixedMode = appCtxt.get(ZmSetting.PROTOCOL_MODE) == ZmSetting.PROTO_MIXED;
101 	var showAllZimlets = href.indexOf("zimlets=all") > 0;
102 	if(isMixedMode && !appCtxt.isOffline && !showAllZimlets && isHttp
103 			&& appCtxt.get(ZmSetting.DISABLE_SENSITIVE_ZIMLETS_IN_MIXED_MODE) == "TRUE") {
104 		zimletArray = this._getNonSensitiveZimlets(zimletArray);
105 	}
106 	if (!zimletArray || !zimletArray.length) {
107 		this.loaded = true;
108 		this._resetOverviewTree();
109 		return;
110 	}
111 	var packageCallback = callback ? new AjxCallback(this, this._loadZimlets, [zimletArray, userProps, target, callback, sync]) : null;
112 	AjxPackage.require({ name: "Zimlet", callback: packageCallback });
113 	if (!callback) {
114 		this._loadZimlets(zimletArray, userProps, target, callback, sync);
115 	}
116 };

119 /**
120  * Returns non-sensitive Zimlets whose config_template.xml file does not contain "hasSensitiveData=true"
121  * @param	{Array}	zimletArray	an array of {@link ZmZimlet} objects
122  *
123  * @private
124  */
125 ZmZimletMgr.prototype._getNonSensitiveZimlets =
126 function(zimletArray) {
127 	if (!zimletArray || !zimletArray.length) {
128 		return;
129 	}
130 	var nonSensitiveZimlets = [];
131 	var len = zimletArray.length;
132 	for(var i = 0; i < len; i++) {
133 		var configProps = [];
134 		var zimletObj = zimletArray[i];
135 		var isSensitiveZimlet = false;
136 		var zimletName = zimletObj.zimlet && zimletObj.zimlet[0] ? zimletObj.zimlet[0].name : "";
137 		var zimletConfig = zimletObj.zimletConfig;
138 
139 		if(zimletConfig)  {
140 			if(zimletConfig[0]
141 					&& zimletConfig[0].global
142 					&& zimletConfig[0].global[0]
143 					&& zimletConfig[0].global[0].property) {
144 
145 				configProps = zimletConfig[0].global[0].property;
146 				for(var j = 0; j < configProps.length; j++) {
147 					var property = configProps[j];
148 					if(property.name == ZmZimletMgr.HAS_SENSITIVE_DATA_CONFIG_NAME && property._content == "true") {
149 						isSensitiveZimlet = true;
150 						break;
151 					}
152 				}
153 			}
154 		}
155 		if(!isSensitiveZimlet) {
156 			nonSensitiveZimlets.push(zimletObj);
157 		}
158 	}
159 	return nonSensitiveZimlets;
160 };

185 /**
186  * @private
187  */
188 ZmZimletMgr.prototype._loadZimlets =
189 function(zimletArray, userProps, target, callback, sync) {
190 	var z;
191 	var loadZimletArray = [];
192 	var targetRe = new RegExp("\\b"+(target || "main")+"\\b");
193 	for (var i=0; i < zimletArray.length; i++) {
194 		var zimletObj = zimletArray[i];
195 		var zimlet0 = zimletObj.zimlet[0];
196 		// NOTE: Only instantiate zimlet context for specified target
197 		if (!targetRe.test(zimlet0.target || "main")) { continue; }
198 		z = new ZmZimletContext(i, zimletObj);
199 		this._ZIMLETS_BY_ID[z.name] = z;
200 		this._ZIMLETS.push(z);
201 		loadZimletArray.push(zimletObj);
202 	}
203 	if (userProps) {
204 		for (i = 0; i < userProps.length; ++i) {
205 			var p = userProps[i];
206 			z = this._ZIMLETS_BY_ID[p.zimlet];
207 			if (z) {
208 				z.setPropValue(p.name, p._content);
209 			}
210 		}
211 	}
212 	if (!appCtxt.isChildWindow) {
213 		var panelZimlets = this.getPanelZimlets();
214 		if (panelZimlets && panelZimlets.length > 0) {
215 			var zimletTree = appCtxt.getZimletTree();
216 			if (!zimletTree) {
217 				zimletTree = new ZmFolderTree(ZmOrganizer.ZIMLET);
218 				var account = appCtxt.multiAccounts && appCtxt.accountList.mainAccount;
219 				appCtxt.setTree(ZmOrganizer.ZIMLET, zimletTree, account);
220 			}
221 			zimletTree.reset();
222 			zimletTree.loadFromJs(panelZimlets, "zimlet");
223 		} else { // reset overview tree accordingly
224 			this._resetOverviewTree();
225 		}
226 	}
227 
228 	// load zimlet code/CSS
229 	var zimletNames = this._getZimletNames(loadZimletArray);
230 	this._loadIncludes(loadZimletArray, zimletNames, (sync ? callback : null) );
231 	this._loadStyles(loadZimletArray, zimletNames);
232 
233 	if (callback && !sync) {
234 		callback.run();
235 	}
236 };

243 	var zimletTree = appCtxt.getZimletTree();

252 			}
253 		}
254 	}

400 /**
401  * Handles zimlet notification.
402  * 
403  * @param	{Object}	event	the event
404  * @param	{Object}	args	the arguments
405  * 
406  * @return true if any zimlet handled the notification
407  * 
408  * @private
409  */
410 ZmZimletMgr.prototype.notifyZimlets =
411 function(event, args) {
412 	
413 	args = AjxUtil.toArray(args);
414 
415 	var handled = false;
416 	for (var i = 0; i < this._ZIMLETS.length; ++i) {
417 		var z = this._ZIMLETS[i].handlerObject;
418 		if (z && (z instanceof ZmZimletBase) && z.getEnabled() && (typeof z[event] == "function")) {
419 			var result = args ? z[event].apply(z, args) : z[event].apply(z);	// IE cannot handle empty args
420 			handled = handled || result;
421 		}
422 	}
423 	
424 	return handled;
425 };
426 
427 ZmZimletMgr.prototype.notifyZimlet =
428 function(zimletName, event, args) {
429 	var zimlet = this.getZimletByName(zimletName);
430 	var z = zimlet && zimlet.handlerObject;
431 	if (z && (z instanceof ZmZimletBase) && z.getEnabled() && (typeof z[event] == "function")) {
432 		return (args ? z[event].apply(z, args) : z[event].apply(z));	// IE cannot handle empty args
433 	}
434 };

436 /**
437  * Processes a request (from core-zcs to zimlets) and returns value of the
438  * first zimlet that serves the request.
439  * PS: 
440  * - Requestor must handle 'null' value
441  * - stores/caches the zimlet for a given request to improve performance.
442  * - also stores _requestNotHandledByAnyZimlet if no zimlet handles this
443  *	request(in the current session), again to improve performance.
444  * e.g: appCtxt.getZimletMgr().processARequest("getMailCellStyle", item, field)
445  * 
446  * @private
447  */
448 ZmZimletMgr.prototype.processARequest =
449 function(request) {
450 	if (this._requestNotHandledByAnyZimlet[request]) { return null; }
451 
452 	var args = new Array(arguments.length - 1);
453 	for (var i = 0; i < args.length;) {
454 		args[i] = arguments[++i];
455 	}
456 	var sz = this._serviceZimlets[request];
457 	if (sz) { // if we already know a zimlet that serves this request, use it.
458 		return sz[request].apply(sz, args);
459 	}
460 
461 	var a = this._ZIMLETS;
462 	for (var i = 0; i < a.length; ++i) {
463 		var z = a[i].handlerObject;
464 		if (z && (z instanceof ZmZimletBase) && z.getEnabled() &&
465 			(typeof z[request] == "function"))
466 		{
467 			 this._serviceZimlets[request] = z;//store 
468 			 return z[request].apply(z, args);
469 		}
470 	}
471 	if (this.isLoaded()) { // add to an array to indicate that no zimlet implements this request
472 		this._requestNotHandledByAnyZimlet[request]=request;
473 	}
474 	return null;
475 };

504 /**
505  * @private
506  */
507 ZmZimletMgr.prototype._finished_loadIncludes =
508 function(zimletNames, callback) {
509 	if (!appCtxt.isChildWindow) {
510 		this.renameZimletsLabel();
511 	}
512 	this.loaded = true;
513 	var zimlets = this.getZimletsHash();
514 	for (var i = 0; i < zimletNames.length; i++) {
515 		var showedDialog = false;
516 		var name = zimletNames[i];
517 		try {
518 			zimlets[name]._finished_loadIncludes();
519 		} catch (e) {
520 			if (!showedDialog) {
521 				var dialog = appCtxt.getErrorDialog();
522 				var message = AjxMessageFormat.format(ZmMsg.zimletInitError, name);
523 				dialog.setMessage(message, e.toString(), DwtMessageDialog.CRITICAL_STYLE);
524 				dialog.popup();
525 				showedDialog = true;
526 			}
527 			DBG.println(AjxDebug.DBG1, "Error initializing zimlet '" + name + "': " + e);
528 		}
529 	}
530 	if (appCtxt.get(ZmSetting.PORTAL_ENABLED) && !appCtxt.isChildWindow) {
531 		var params = {
532 			name: "Portal",
533 			callback: (new AjxCallback(this, this._finished_loadIncludes2, [callback]))
534 		};
535 		AjxPackage.require(params);
536 	} else {
537 		this._finished_loadIncludes2(callback);
538 	}
539 };

578 /**
579  * @private
580  */
581 ZmZimletMgr.prototype.__getIncludes =
582 function(zimletArray, zimletNames, isJS) {
583     // get language info
584     var languageId = null;
585     var countryId = null;
586     if (appCtxt.get(ZmSetting.LOCALE_NAME)) {
587         var locale = appCtxt.get(ZmSetting.LOCALE_NAME) || "";
588         var parts = locale.split("_");
589         languageId = parts[0];
590         countryId = parts[1];
591     }
592     var locid = "";
593     if (languageId) locid += "&language="+languageId;
594     if (countryId) locid += "&country="+countryId;
595 
596     // add cache killer to each url
597     var query = [
598         "?v=", window.cacheKillerVersion
599 //        window.appDevMode ? new Date().getTime() : window.cacheKillerVersion
600     ].join("");
601 
602     // add messages for all zimlets
603     var includes = [];
604     if (window.appDevMode && isJS) {
605         var zimlets = appCtxt.get(ZmSetting.ZIMLETS) || [];
606         if(appCtxt.isChildWindow) {
607             var winOpener = window.opener || window;
608             zimlets = winOpener.appCtxt.get(ZmSetting.ZIMLETS) || []
609         }
610         for (var i = 0; i < zimlets.length; i++) {
611             var zimlet = zimlets[i].zimlet[0];
612             includes.push([appContextPath, "/res/", zimlet.name, ".js", query, locid].join(""));
613         }
614     }
615 
616 	// add remote urls
617 	for (var i = 0; i < zimletArray.length; i++) {
618 		var zimlet = zimletArray[i].zimlet[0];
619 		var baseUrl = zimletArray[i].zimletContext[0].baseUrl;
620 		var isDevZimlet = baseUrl.match("/_dev/");
621 
622 		// include links
623 		var links = (isJS ? zimlet.include : zimlet.includeCSS) || [];
624 		for (var j = 0; j < links.length; j++) {
625 			var url = links[j]._content;
626 			if (ZmZimletMgr._RE_REMOTE.test(url)) {
627 				var fullurl = [ ZmZimletBase.PROXY, AjxStringUtil.urlComponentEncode(url) ].join("");
628 				includes.push(fullurl);
629 				continue;
630 			}
631 			if (window.appDevMode || isDevZimlet) {
632                 var debug = isDevZimlet ? "&debug=1" : "";
633 				includes.push([baseUrl, url, query, locid, debug].join(""));
634 			}
635 		}
636 	}
637 
638 	// add link to aggregated files
639 	if (!window.appDevMode) {
640 		var cosId = null;
641 		if (appCtxt.getSettings() && appCtxt.getSettings().getInfoResponse && appCtxt.getSettings().getInfoResponse.cos) {
642 			cosId = appCtxt.getSettings().getInfoResponse.cos.id;
643 		}
644 		var extension = (!AjxEnv.isIE || (!AjxEnv.isIE6 && AjxEnv.isIE6up)) ? appExtension : "";
645 		includes.unshift([
646 			"/service/zimlet/res/Zimlets-nodev_all",
647 			(isJS ? (".js" + extension) : ".css"),
648 			(languageId ? "?language=" + languageId : ""),
649 			(countryId ? "&country=" + countryId : ""),
650 			(cosId ? "&cosId=" + cosId : "")  // For an explanation of why we add cosId here, please see bug #58979
651 		].join(""));
652 	}
653 
654 	return includes;
655 };

675 /**
676  * Changes the zimlet label.
677  * 
678  * @param	{Object}	item		the item
679  */
680 ZmZimletMgr.prototype.changeZimletLabel =
681 function(item) {
682 	var zimlet = item.getData(Dwt.KEY_OBJECT);
683 	if (zimlet) {
684 		var currentLabel = zimlet.getName();
685 		var regEx = /\$/;
686 		if (currentLabel.match(regEx)) {
687 			var replaceLabel = currentLabel.replace(/\${msg./,'').replace(/}/,'');
688 			var zimletContextName = zimlet.getZimletContext().name;
689 			if (window[zimletContextName]) {
690 				var str = window[zimletContextName][replaceLabel];
691 				if (str) {
692 					item.setText(str);
693 					zimlet.setName(str);
694 				}
695 			}
696 		}
697 	}
698 };
