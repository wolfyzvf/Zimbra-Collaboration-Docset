 23 /**
 24  * Creates the object handler.
 25  * @class
 26  * This class defines the default implementation for a zimlet object handler.
 27  * <br />
 28  * <br />
 29  * To write a zimlet, see {@link ZmZimletBase}. 
 30  * 
 31  * @param	{string}	typeName	the type name
 32  * @param	{string}	className	the class name
 33  */
 34 ZmObjectHandler = function(typeName, className) {
 35 	if (arguments.length > 0) {
 36 		this.init(typeName, className);
 37 	}
 38 }
 39 
 40 ZmObjectHandler.prototype.constructor = ZmObjectHandler;
 41 
 42 /**
 43  * This method is called by the Zimlet framework to initialize the object.
 44  * 
 45  * @param	{string}	typeName	the type name
 46  * @param	{string}	className	the class name; if <code>null</code>, "Object" will be used
 47  */
 48 ZmObjectHandler.prototype.init =
 49 function(typeName, className) {
 50 	this._typeName = typeName;
 51 	this._className = className ? className : "Object";
 52 };
 53 
 54 /**
 55  * Returns a string representation of the object.
 56  * 
 57  * @return		{string}		a string representation of the object
 58  */
 59 ZmObjectHandler.prototype.toString = 
 60 function() {
 61 	// If you can find a cleaner way to get the name of 
 62 	// a sub-class without hard coding each instance
 63 	// in a toString() method feel free to change.
 64 	if(!this._toString) {
 65 		var ctor = "" + this.constructor;
 66 		ctor = ctor.substring(0,ctor.indexOf("("));
 67 		this._toString = ctor.substring("function ".length);
 68 	}
 69 	return this._toString;
 70 };

 82 /**
 83  * Gets the class name for a given object.
 84  * 
 85  * @param	{Object}		obj			the object
 86  * @param	{Object}		context		the content
 87  * @param	{string}		spanId		ID of the SPAN
 88  *
 89  * @return	{string}		the class name
 90  */
 91 ZmObjectHandler.prototype.getClassName =
 92 function(obj, context, spanId) {
 93 	return this._className;
 94 };

 96 /**
 97  * Gets the hovered class name for the given object.
 98  * 
 99  * @param	{Object}		obj			the object
100  * @param	{Object}		context		the content
101  * @param	{string}		spanId		ID of hovered SPAN
102  *
103  * @return	{string}		the hovered class name
104  */
105 ZmObjectHandler.prototype.getHoveredClassName =
106 function(obj, context, spanId) {
107 	var cname = this.getClassName(obj, context, spanId);
108 	if (this._cachedClassNameForHovered !== cname) {
109 		this._cachedClassNameForHovered = cname;
110 		this._classNameHovered = cname + "-" + DwtCssStyle.HOVER;
111 	}
112 	return this._classNameHovered;
113 };
114 
115 /**
116  * Gets the active class name for a given object.
117  * 
118  * @param	{Object}		obj		the object
119  * @param	{Object}		context		the content
120  * @param	{string}		spanId		ID of the SPAN
121  *
122  * @return	{string}		the active class name
123  */
124 ZmObjectHandler.prototype.getActiveClassName =
125 function(obj, context, spanId) {
126 	var cname = this.getClassName(obj, context, spanId);
127 	if (this._cachedClassNameForActive !== cname) {
128 		this._cachedClassNameForActive = cname;
129 		this._classNameActive = cname + "-" + DwtCssStyle.ACTIVE;
130 	}
131 	return this._classNameActive;
132 };

134 /**
135  * @private
136  */
137 ZmObjectHandler.prototype.findObject =
138 function(content, startIndex) {
139 	if (startIndex === 0) {
140 		this._lastMatch = null;
141 		this._noMatch = false;
142 	}
143 	if (this._noMatch) {return null;}
144 	if (this._lastMatch && this._lastMatch.index >= startIndex) {
145 		return this._lastMatch;
146 	}
147 	this._lastMatch = this.match(content, startIndex);
148 	this._noMatch = (this._lastMatch === null);
149 	return this._lastMatch;
150 };

153 /**
154  * This method is used to match content for a zimlet. Zimlet implementations should
155  * override this method. Usage should return a non-null result in the format of
156  * <code>String.match</code> if text on the line matched the handler regular expression.
157  * 
158  * <pre>
159  * var result = handler.match(line);
160  * result[0] // should be matched string
161  * result.index // should be location within line match occurred
162  * </pre>
163  * 
164  * Handlers can also set result.context which will be passed back to
165  * them during the various method calls ({@link #getToolTipText}, etc). Handlers should set
166  * regex.lastIndex to startIndex and then use <code>regex.exec(content)</code>. Handlers should
167  * also use the "g" option when constructing their regex.
168  */
169 ZmObjectHandler.prototype.match =
170 function(content, startIndex) {
171 	return null;
172 };

174 /**
175  * Generates content inside the <code><span></code> tag.
176  * 
177  * @return	{number}	the content index
178  * @private
179  * */
180 ZmObjectHandler.prototype._getHtmlContent =
181 function(html, idx, obj, context, spanId) {
182 	html[idx++] = AjxStringUtil.htmlEncode(obj, true);
183 	return idx;
184 };
185 
186 /**
187  * Generates the <code><span></code> tag.
188  * 
189  * @return	{number}	the content index
190  * @private
191  */
192 ZmObjectHandler.prototype.generateSpan = 
193 function(html, idx, obj, spanId, context, options) {
194 	html[idx++] = "<span class='";
195 	html[idx++] = this.getClassName(obj);
196 	html[idx++] = "' id='";
197 	html[idx++] = spanId;
198 	html[idx++] = "'>";
199 	idx = this._getHtmlContent(html, idx, obj, context, spanId, options);
200 	html[idx++] = "</span>";
201 	return idx;
202 };

279 /**
280  * This method is called when the handler is hovered-over.
281  * 
282  * @private
283  */
284 ZmObjectHandler.prototype.hoverOver = function(object, context, x, y) {
285 	var shell = DwtShell.getShell(window);
286 	var tooltip = shell.getToolTip();
287 	tooltip.setContent(this.getToolTipText(object, context));
288 	tooltip.popup(x, y);
289 	this.populateToolTip(object, context);
290 };

298 	var shell = DwtShell.getShell(window);
299 	var tooltip = shell.getToolTip();
300 	tooltip.popdown();
