 85 /**
 86  * Propose new time for an appointment
 87  *
 88  * @param	{ZmAppt}	    appt		            the appointment
 89  * @param	{AjxCallback}	proposeTimeCallback		callback executed  after proposing time
 90  * @return	{Boolean}	    <code>true</code>       indicates that propose time is executed
 91  */
 92 ZmApptComposeController.prototype.sendCounterAppointmentRequest =
 93 function(appt, proposeTimeCallback) {
 94 	var callback = new AjxCallback(this, this._handleCounterAppointmentRequest, proposeTimeCallback);
 95     var apptEditView = this._composeView ? this._composeView.getApptEditView() : null;
 96     var viewMode = apptEditView ? apptEditView.getMode() : null;
 97 	appt.sendCounterAppointmentRequest(callback, null, viewMode);
 98 	return true;
 99 };

184 ZmApptComposeController.prototype.saveCalItem =
185 function(attId) {
186 	this._composeView.cancelLocationRequest();
187 	var appt = this._composeView.getAppt(attId);
188     var numRecurrence = this._composeView.getNumLocationConflictRecurrence ?
189         this._composeView.getNumLocationConflictRecurrence() :
190         ZmTimeSuggestionPrefDialog.DEFAULT_NUM_RECURRENCE;
191 
192 	if (appt) {
193 
194 		if (!appt.isValidDuration()) {
195 			this._composeView.showInvalidDurationMsg();
196 			this.enableToolbar(true);
197 			return false;
198 		}
199 		if (!appt.isValidDurationRecurrence()) {
200 			this._composeView.showInvalidDurationRecurrenceMsg();
201 			this.enableToolbar(true);
202 			return false;
203 		}
204 
205         if (appCtxt.get(ZmSetting.GROUP_CALENDAR_ENABLED)) {
206             if (this._requestResponses)
207             	appt.setRsvp(this._requestResponses.getChecked());
208             appt.setMailNotificationOption(true);
209         }
210 
211         if(appt.isProposeTime && !appt.isOrganizer()) {
212             return this.sendCounterAppointmentRequest(appt);
213         }
214 
215 		if (appt.isForward) {
216 			var addrs = this._composeView.getForwardAddress();
217 
218 			// validate empty forward address
219 			if (!addrs.gotAddress) {
220 				var msgDialog = appCtxt.getMsgDialog();
221 				msgDialog.setMessage(ZmMsg.noForwardAddresses, DwtMessageDialog.CRITICAL_STYLE);
222 				msgDialog.popup();
223                 this.enableToolbar(true);
224 				return false;
225 			}
226 
227 			if (addrs[ZmApptEditView.BAD] && addrs[ZmApptEditView.BAD].size()) {
228 				var cd = appCtxt.getOkCancelMsgDialog();
229 				cd.reset();
230 				var bad = AjxStringUtil.htmlEncode(addrs[ZmApptEditView.BAD].toString(AjxEmailAddress.SEPARATOR));
231 				var msg = AjxMessageFormat.format(ZmMsg.compBadAddresses, bad);
232 				cd.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
233 				cd.registerCallback(DwtDialog.OK_BUTTON, this._badAddrsOkCallback, this, [cd,appt]);
234 				cd.setVisible(true); // per fix for bug 3209
235 				cd.popup();
236                 this.enableToolbar(true);
237 				return false;
238 			}
239 
240             //attendee forwarding an appt
241             /* if(!appt.isOrganizer()) */ return this.forwardCalItem(appt);
242 		}
243 
244 		if (!this._attendeeValidated && this._invalidAttendees && this._invalidAttendees.length > 0) {
245 			var dlg = appCtxt.getYesNoMsgDialog();
246 			dlg.registerCallback(DwtDialog.YES_BUTTON, this._clearInvalidAttendeesCallback, this, [appt, attId, dlg]);
247 			var msg = "";
248             if(this._action == ZmCalItemComposeController.SAVE){
249                msg = AjxMessageFormat.format(ZmMsg.compSaveBadAttendees, AjxStringUtil.htmlEncode(this._invalidAttendees.join(",")));
250             }
251             else{
252                 msg = AjxMessageFormat.format(ZmMsg.compBadAttendees, AjxStringUtil.htmlEncode(this._invalidAttendees.join(",")));
253             }
254 			dlg.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
255 			dlg.popup();
256             this.enableToolbar(true);
257             this._attendeeValidated = true;
258 			return false;
259 		}
260 
261         //Validation Check for Significant / Insignificant / Local changes
262         if(this._action == ZmCalItemComposeController.SAVE && !appt.inviteNeverSent){
263             //Check for Significant Changes
264             if(this._checkIsDirty(ZmApptEditView.CHANGES_SIGNIFICANT)){
265                 this._getChangesDialog().popup();
266                 this.enableToolbar(true);
267                 return false;
268             }
269         }
270 
271 		var origAttendees = appt.origAttendees;						// bug fix #4160
272 		if (origAttendees && origAttendees.length > 0 && 			// make sure we're not u/l'ing a file
273 			attId == null) 											// make sure we are editing an existing appt w/ attendees
274 		{
275 			if (!appt.inviteNeverSent && !this._composeView.getApptEditView().isDirty(true)) {	// make sure other fields (besides attendees field) have not changed
276 				var attendees = appt.getAttendees(ZmCalBaseItem.PERSON);
277 				if (attendees.length > 0) {
278 					// check whether organizer has added/removed any attendees
279 					if (this._action == ZmCalItemComposeController.SEND && this._attendeesUpdated(appt, attId, attendees, origAttendees))
280 						return false;
281 				}
282 			}
283 
284 			// check whether moving appt from local to remote folder with attendees
285 			var cc = AjxDispatcher.run("GetCalController");
286 			if (cc.isMovingBetwAccounts(appt, appt.__newFolderId)) {
287 				var dlg = appCtxt.getMsgDialog();
288                 dlg.setMessage(ZmMsg.orgChange, DwtMessageDialog.WARNING_STYLE);
289                 dlg.popup();
290                 this.enableToolbar(true);
291                 return false;
292 			}
293 		}
294 
295         var ret = this._initiateSaveWithChecks(appt, attId, numRecurrence);
296 		return ret;
297 	}
298 
299 	return false;
300 };

302 ZmApptComposeController.prototype._initiateSaveWithChecks =
303 function(appt, attId, numRecurrence) {
304     var resources = appt.getAttendees(ZmCalBaseItem.EQUIPMENT);
305     var locations = appt.getAttendees(ZmCalBaseItem.LOCATION);
306     var attendees = appt.getAttendees(ZmCalBaseItem.PERSON);
307 
308     var notifyList;
309 
310     var needsPermissionCheck = (attendees && attendees.length > 0) ||
311                                (resources && resources.length > 0) ||
312                                (locations && locations.length > 0);
313 
314     var needsConflictCheck = !appt.isForward &&
315          ((resources && resources.length > 0) ||
316          // If alteredLocations specified, it implies the user
317          // has already examined and modified the location conflicts
318          // that they want - so issue no further warnings.
319 
320          // NOTE: appt.alteredLocations is disabled (and hence undefined)
321          //       for now.  It will be set once CreateAppt/ModifyAppt
322          //       SOAP API changes are completed (Bug 56464)
323           (!appt.alteredLocations && locations && locations.length > 0));
324 
325     if (needsConflictCheck) {
326         this.checkConflicts(appt, numRecurrence, attId, notifyList);
327         return false;
328     } else if (needsPermissionCheck) {
329         this.checkAttendeePermissions(appt, attId, notifyList);
330         return false;
331     } else {
332         this._saveCalItemFoRealz(appt, attId, notifyList);
333     }
334     return true;
335 };

337 ZmApptComposeController.prototype.updateToolbarOps =
338 function(mode, appt) {
339 
340     var saveButton = this._toolbar.getButton(ZmOperation.SAVE);
341     var sendButton = this._toolbar.getButton(ZmOperation.SEND_INVITE);
342 
343     if (mode == ZmCalItemComposeController.APPT_MODE) {
344         saveButton.setText(ZmMsg.saveClose);
345         saveButton.setVisible(true);
346         sendButton.setVisible(false);
347     } else {
348         sendButton.setVisible(true);
349         saveButton.setVisible(true);
350         saveButton.setText(ZmMsg.save);
351 
352         //change cancel button's text/icon to close
353         var cancelButton = this._toolbar.getButton(ZmOperation.CANCEL);
354         cancelButton.setText(ZmMsg.close);
355     }
356 	if (this._requestResponses) {
357 		this._requestResponses.setEnabled(mode !== ZmCalItemComposeController.APPT_MODE);
358 	}
359 
360     if ((this._mode == ZmCalItem.MODE_PROPOSE_TIME) || ZmCalItem.FORWARD_MAPPING[this._mode]) {
361         sendButton.setVisible(true);
362         saveButton.setVisible(false);
363         // Enable the RequestResponse when Forwarding
364 		if (this._requestResponses) {
365 			this._requestResponses.setEnabled(this._mode !== ZmCalItem.MODE_PROPOSE_TIME);
366 		}
367     }
368 
369 };

400

411 ZmApptComposeController.prototype._sendAfterExceptionCheck =
412 function(){
413      var appt = this._composeView.getApptEditView()._calItem;
414      var isExceptionAllowed = appCtxt.get(ZmSetting.CAL_EXCEPTION_ON_SERIES_TIME_CHANGE);
415      var isEditingSeries = (this._mode == ZmCalItem.MODE_EDIT_SERIES);
416      var showWarning = appt.isRecurring() && appt.hasEx && isEditingSeries && appt.getAttendees(ZmCalBaseItem.PERSON) && !isExceptionAllowed && this._checkIsDirty(ZmApptEditView.CHANGES_TIME_RECURRENCE);
417      if(showWarning){
418           var dialog = appCtxt.getYesNoCancelMsgDialog();
419 		  dialog.setMessage(ZmMsg.recurrenceUpdateWarning, DwtMessageDialog.WARNING_STYLE);
420           dialog.registerCallback(DwtDialog.YES_BUTTON, this._sendContinue, this,[dialog]);
421           dialog.registerCallback(DwtDialog.NO_BUTTON, this._dontSend,this,[dialog]);
422           dialog.getButton(DwtDialog.CANCEL_BUTTON).setText(ZmMsg.discard);
423 		  dialog.registerCallback(DwtDialog.CANCEL_BUTTON, this._dontSendAndClose,this,[dialog]);
424 		  dialog.popup();
425     }
426     else{
427         this._sendContinue();
428     }
429 }

478 ZmApptComposeController.prototype._saveListener =
479 function(ev, force) {
480     var isMeeting = !this._composeView.isAttendeesEmpty();
481 
482     this._action = isMeeting ? ZmCalItemComposeController.SAVE : ZmCalItemComposeController.SAVE_CLOSE;
483 
484     //attendee should not have send/save option
485     if(!this._composeView.isOrganizer()) {
486         this._action = ZmCalItemComposeController.SAVE_CLOSE;
487     }
488     this.enableToolbar(false);
489 
490     var dlg = appCtxt.getOkCancelMsgDialog();
491     if(dlg.isPoppedUp()){
492         dlg.popdown();
493     }
494 
495     if(!force && this._action == ZmCalItemComposeController.SAVE){
496         var appt = this._composeView.getApptEditView()._calItem;
497         var inviteNeverSent = (appt && appt.inviteNeverSent);
498         var showDlg = true;
499         if(appt.isDraft){
500             showDlg = false;
501         }
502         if(showDlg && !inviteNeverSent && (this._checkIsDirty(ZmApptEditView.CHANGES_SIGNIFICANT)
503                 ||  this._checkIsDirty(ZmApptEditView.CHANGES_LOCAL))){
504             showDlg = false;
505         }
506         if(showDlg){
507             dlg.setMessage(ZmMsg.saveApptInfoMsg);
508             dlg.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._saveListener, [ev, true]));
509             dlg.popup();
510             this.enableToolbar(true);
511             return;
512         }
513     }
514 
515 	if (this._doSave() === false) {
516 		return;
517     }
518 };

597 ZmApptComposeController.prototype.checkConflicts =
598 function(appt, numRecurrence, attId, notifyList) {
599 	var resources = appt.getAttendees(ZmCalBaseItem.EQUIPMENT);

629 		}

634 		for (var i = 0; i < resources.length; i++) {

637 	}

693 	var mode = appt.viewMode,
694 	    jsonObj = {
695             CheckRecurConflictsRequest: {

719

723     if(recurInfo) {

726         recurrence.parseExcludeInfo(recur);
727     }
728 
729     if(mode != ZmCalItem.MODE_EDIT_SINGLE_INSTANCE) {

792 // Use the (numRecurrences * the recurrence period * repeat.customCount)
793 // time interval to determine the endDate of the resourceConflict check
794 ZmApptComposeController.getCheckResourceConflictEndTime =
795 function(appt, originalStartDate, numRecurrence) {
796     var startDate = new Date(originalStartDate.getTime());
797     var recurrence = appt.getRecurrence();
798     var endDate;
799     var range = recurrence.repeatCustomCount * numRecurrence;
800     if (recurrence.repeatType == ZmRecurrence.NONE) {
801         endDate = appt.endDate;
802     } else if (recurrence.repeatType == ZmRecurrence.DAILY) {
803         endDate = AjxDateUtil.roll(startDate, AjxDateUtil.DAY, range);
804     } else if (recurrence.repeatType == ZmRecurrence.WEEKLY) {
805         endDate = AjxDateUtil.roll(startDate, AjxDateUtil.WEEK, range);
806     } else if (recurrence.repeatType == ZmRecurrence.MONTHLY) {
807         endDate = AjxDateUtil.roll(startDate, AjxDateUtil.MONTH, range);
808     } else if (recurrence.repeatType == ZmRecurrence.YEARLY) {
809         endDate = AjxDateUtil.roll(startDate, AjxDateUtil.YEAR, range);
810     }
811     var endTime = endDate.getTime();
812     if (recurrence.repeatEndDate) {
813         var repeatEndTime = recurrence.repeatEndDate.getTime();
814         if (endTime > repeatEndTime) {
815             endTime = repeatEndTime;
816         }
817     }
818     return endTime;
819 }

1022 ZmApptComposeController.prototype._attendeesUpdated =
1023 function(appt, attId, attendees, origAttendees) {
1024 	// create hashes of emails for comparison
1025 	var origEmails = {};
1026 	for (var i = 0; i < origAttendees.length; i++) {
1027 		var email = origAttendees[i].getEmail();
1028 		origEmails[email] = true;
1029 	}
1030 	var fwdEmails = {};
1031 	var fwdAddrs = appt.getForwardAddress();
1032 	for(var i=0;i<fwdAddrs.length;i++) {
1033 		var email = fwdAddrs[i].getAddress();
1034 		fwdEmails[email] = true;
1035 	}
1036 	var curEmails = {};
1037 	for (var i = 0; i < attendees.length; i++) {
1038 		var email = attendees[i].getEmail();
1039 		curEmails[email] = true;
1040 	}
1041 
1042 	// walk the current list of attendees and check if there any new ones
1043 	for (var i = 0 ; i < attendees.length; i++) {
1044 		var email = attendees[i].getEmail();
1045 		if (!origEmails[email] && !fwdEmails[email]) {
1046 			this._addedAttendees.push(email);
1047 		}
1048 	}
1049     
1050 	for (var i = 0 ; i < origAttendees.length; i++) {
1051 		var email = origAttendees[i].getEmail();
1052 		if (!curEmails[email]) {
1053 			this._removedAttendees.push(email);
1054 		}
1055 	}
1056 
1057 	if (this._addedAttendees.length > 0 || this._removedAttendees.length > 0) {
1058 		if (!this._notifyDialog) {
1059 			this._notifyDialog = new ZmApptNotifyDialog(this._shell);
1060 			this._notifyDialog.addSelectionListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._notifyDlgOkListener));
1061 			this._notifyDialog.addSelectionListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._notifyDlgCancelListener));
1062 		}
1063 		appt.setMailNotificationOption(true);
1064 		this._notifyDialog.initialize(appt, attId, this._addedAttendees, this._removedAttendees);
1065 		this._notifyDialog.popup();
1066         this.enableToolbar(true);
1067 		return true;
1068 	}
1069 
1070 	return false;
1071 };

1194     if(viewMode == ZmCalItem.MODE_NEW || viewMode == ZmCalItem.MODE_NEW_FROM_QUICKADD || viewMode == ZmAppt.MODE_DRAG_OR_SASH) {
1195         isNewAppt = true;
1196     }
1197 
1198     if(this.isCloseAction()) {
1199         calItem.handlePostSaveCallbacks();
1200         this.closeView();	    
1201     }else {
1202         this.enableToolbar(true);
1203         if(isNewAppt) {
1204             viewMode = calItem.isRecurring() ? ZmCalItem.MODE_EDIT_SERIES : ZmCalItem.MODE_EDIT;
1205         }
1206         calItem.setFromSavedResponse(result);
1207         if(this._action == ZmCalItemComposeController.SAVE){
1208             calItem.isDraft = this._draftFlag;
1209             calItem.draftUpdated = true;
1210         }
1211         this._composeView.set(calItem, viewMode);        
1212     }
1213 
1214     var msg = isNewAppt ? ZmMsg.apptCreated : ZmMsg.apptSaved;
1215     if(calItem.hasAttendees()){
1216         if(this._action == ZmCalItemComposeController.SAVE || this._action == ZmCalItemComposeController.SAVE_CLOSE){
1217             msg = ZmMsg.apptSaved;
1218         }else{
1219             if(viewMode != ZmCalItem.MODE_NEW){
1220                 msg = ZmMsg.apptSent;
1221             }
1222         }              
1223     }
1224     appCtxt.setStatusMsg(msg);
1225     
1226     appCtxt.notifyZimlets("onSaveApptSuccess", [this, calItem, result]);//notify Zimlets on success
1227 };
