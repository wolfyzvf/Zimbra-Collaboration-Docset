 21 /**
 22  * Creates an empty accordion widget.
 23  * @constructor
 24  * @class
 25  * This class implements an accordion widget, which is a stack of expandable
 26  * accordion headers. Clicking on an accordion header's button expands it in
 27  * place.
 28  *
 29  * @author Parag Shah
 30  *
 31  * @param {DwtControl}	parent	the parent widget
 32  * @param {string}	className	the CSS class
 33  * @param {Dwt.STATIC_STYLE|Dwt.ABSOLUTE_STYLE|Dwt.RELATIVE_STYLE}	posStyle	the positioning style
 34  * 
 35  * @extends		DwtComposite
 36  */
 37 DwtAccordion = function(parent, className, posStyle) {
 38 
 39 	if (arguments.length == 0) return;
 40 	DwtComposite.call(this, {parent:parent, className:className, posStyle:(posStyle || Dwt.ABSOLUTE_STYLE)});
 41 
 42 	this._initialize(className);
 43 };
 44 
 45 DwtAccordion.prototype = new DwtComposite;
 46 DwtAccordion.prototype.constructor = DwtAccordion;

 61 /**
 62  * Adds an item to the accordion, in the form of a table row.
 63  *
 64  * @param {hash}	params		a hash of parameters
 65  * @param {string}      params.title			the text for accordion header
 66  * @param {hash}      params.data			the item data
 67  * @param {string}      params.icon			the icon
 68  * @param {boolean}      params.hideHeader	if <code>true</code>, do not show header (ideal when there's only one visible header item)
 69  */
 70 DwtAccordion.prototype.addAccordionItem =
 71 function(params) {
 72 
 73 	if (!this.isListenerRegistered(DwtEvent.CONTROL)) {
 74 		this.addControlListener(new AjxListener(this, this._controlListener));
 75 	}
 76 
 77 	var itemNum = this.__ITEMCOUNT++;
 78 	var item = new DwtAccordionItem(itemNum, params.title, params.data, this);
 79 	var subs = {
 80 		id: this._htmlElId,
 81 		itemNum: itemNum,
 82 		title: params.title,
 83 		icon: params.icon
 84 	};
 85 
 86 	// append new accordion item
 87 	var row = this._table.insertRow(-1);
 88 	var cell = row.insertCell(-1);
 89 	cell.id = this._htmlElId + "_cell_" + itemNum;
 90 	cell.className = "ZAccordionCell";
 91 	cell.innerHTML = AjxTemplate.expand("dwt.Widgets#ZAccordionItem", subs);
 92 
 93 	// add onclick event handler to header DIV
 94 	var headerDiv = document.getElementById(this._htmlElId + "_header_" + itemNum);
 95 	if (params.hideHeader) {
 96 		Dwt.setVisible(headerDiv, false);
 97 	} else {
 98 		headerDiv.onclick = AjxCallback.simpleClosure(this._handleOnClickHeader, this, item);
 99 		headerDiv.oncontextmenu = AjxCallback.simpleClosure(this._handleOnRightClickHeader, this, item);
100 		headerDiv.onmouseover = AjxCallback.simpleClosure(this._handleOnMouseoverHeader, this, item);
101 		headerDiv.onmouseout = AjxCallback.simpleClosure(this._handleOnMouseoutHeader, this, item);
102 	}
103 	this._items.push(item);
104 
105 	return item;
106 };

118 /**
119  * Gets the accordion item with the given ID.
120  *
121  * @param {number}	id	the accordion item ID
122  * @return	{DwtAccordionItem}		the item or <code>null</code> if not found

124 DwtAccordion.prototype.getItem =
125 function(id) {
126 	for (var i = 0; i < this._items.length; i++) {
127 		if (this._items[i].id == id) {
128 			return this._items[i];

130 	}
131 	return null;

171 /**
172  * Allows the accordion items to be clickable or not. If disabled, the label of
173  * each accordion item will be grayed out.
174  *
175  * @param {boolean}	enabled			if <code>true</code>, enabled.
176  */
177 DwtAccordion.prototype.setEnabled =
178 function(enabled) {
179 	if (enabled == this._enabled) { return; }
180 
181 	this._enabled = enabled;
182 
183 	for (var i in this._items) {
184 		var item = this._items[i];
185 		if (this._currentItemId != item.id) {
186 			item._setEnabled(enabled);
187 		}
188 	}
189 };

191 /**
192  * Resizes the accordion. This override applies accordion size changes to accordion items as well.
193  *
194  * @param {number}	width		the new width for accordion
195  * @param {number}	height		the new height for accordion
196  */
197 DwtAccordion.prototype.resize =
198 function(width, height) {
199 	if (width) {
200 		// if width changed, resize all header items
201 		for (var i = 0; i < this._items.length; i++) {
202 			var id = this._items[i].id;
203 			var title = document.getElementById(this._htmlElId + "_title_" + id);
204 			var fudge = 30;
205 
206 			var iconCell = document.getElementById(this._htmlElId + "_icon_" + id);
207 			if (iconCell && iconCell.className && iconCell.className != "") {
208 				fudge += 16; // the default width of an icon
209 			}
210 			Dwt.setSize(title, width - fudge);
211 
212 		}
213 	}
214 
215 	var newHeight;
216 	if (height) {
217 		var hdr = document.getElementById(this._htmlElId + "_header_" + this._currentItemId);
218 		if (hdr) {
219 			var hdrHeightSum = Dwt.getSize(hdr).y * this._getVisibleHeaderCount();
220 			newHeight = (height - hdrHeightSum); // force min. height of 100px?
221 		}
222 	}
223 
224 	var body = document.getElementById(this._htmlElId + "_body_" + this._currentItemId);
225 	if (body) {
226 		Dwt.setSize(body, width, newHeight);
227 		if (body.firstChild) {
228 			Dwt.setSize(body.firstChild, width, newHeight);
229 		}
230 	}
231 };

233 DwtAccordion.prototype.setBounds =
234 function(x, y, width, height) {
235 	DwtComposite.prototype.setBounds.call(this, x, y, width, height);
236 
237 	this.resize(width, height);

250 /**
251  * Expands the accordion item with the given ID by making its body visible. The bodies of
252  * other items are hidden.
253  *
254  * @param {number}	id	the accordion item ID
255  * @param {boolean}	notify	if <code>true</code>, selection listeners are to be notified
256  */
257 DwtAccordion.prototype.expandItem =
258 function(id, notify) {
259 	var selectedItem;
260 	for (var i = 0; i < this._items.length; i++) {
261 		var itemId = this._items[i].id;
262 		var header = document.getElementById(this._htmlElId + "_header_" + itemId);
263 		var body = document.getElementById(this._htmlElId + "_body_" + itemId);
264 		var cell = document.getElementById(this._htmlElId + "_cell_" + itemId);
265 		var status = document.getElementById(this._htmlElId + "_status_" + itemId);
266 
267 		if (id == itemId) {
268 			Dwt.setVisible(body, true);
269 			header.className = "ZAccordionHeader ZWidget ZSelected";
270 			status.className = "ImgAccordionOpened";
271 			cell.style.height = "100%";
272 			this._currentItemId = id;
273 			selectedItem = this._items[i];
274 		} else {
275 			Dwt.setVisible(body, false);
276 			header.className = "ZAccordionHeader ZWidget";
277 			status.className = "ImgAccordionClosed";
278 			cell.style.height = "0px";
279 		}
280 	}
281 
282 	if (selectedItem && notify && this.isListenerRegistered(DwtEvent.SELECTION)) {
283 		this.notifySelectionListeners(selectedItem);
284 	}
285 };
286 
287 DwtAccordion.prototype.notifySelectionListeners =
288 function(selectedItem) {
289 	var selEv = DwtShell.selectionEvent;
290 	selEv.item = this;
291 	selEv.detail = selectedItem;
292 	this.notifyListeners(DwtEvent.SELECTION, selEv);
293 };

295 /**
296  * Attaches the HTML content of the given control to the accordion item with
297  * the given ID.
298  *
299  * @param {number}	id				the accordion item ID
300  * @param {DwtControl}	contentObject		the control that contains this item's content
301  */
302 DwtAccordion.prototype.setItemContent =
303 function(id, contentObject) {
304 	var aiBody = this.getBody(id);
305 	if (aiBody) {
306 		this._items[id].control = contentObject;
307 		contentObject.reparentHtmlElement(aiBody);
308 		var size = contentObject.getSize();
309 		this.resize(size.x, size.y);
310 	}
311 };

376 /**
377  * Creates the HTML skeleton for the accordion.
378  * 
379  * @private
380  */
381 DwtAccordion.prototype._initialize =
382 function() {
383 	this._items = [];
384 	this.__ITEMCOUNT = 0;
385 
386 	this.getHtmlElement().innerHTML = AjxTemplate.expand("dwt.Widgets#ZAccordion", {id: this._htmlElId});
387 	this._table = document.getElementById(this._htmlElId + "_accordion_table");
388 
389 	this._setMouseEventHdlrs();
390 };

392 /**
393  * Returns the number of accordion items which have visible headers.
394  * 
395  * @private
396  */
397 DwtAccordion.prototype._getVisibleHeaderCount =
398 function() {
399 	var count = 0;
400 	for (var i = 0; i < this._items.length; i++) {
401 		var hdr = document.getElementById(this._htmlElId + "_header_" + this._items[i].id);
402 		if (hdr && Dwt.getVisible(hdr)) {
403 			count++;
404 		}
405 	}
406 	return count;
407 };

412 /**
413  * When a header button is clicked, the item is expanded. Also, any listeners
414  * are notified.
415  *
416  * @param {DwtAccordionItem}	item		the accordion item whose header was clicked
417  * @param {DwtUiEvent}	ev		the click event
418  * 
419  * @private
420  */
421 DwtAccordion.prototype._handleOnClickHeader =
422 function(item, ev) {
423 	if (!this._enabled) { return; }
424 
425 	this.expandItem(item.id, true);
426 };
427 
428 /**
429  * When a header button is right-clicked, any listeners will be notified so a
430  * context menu can be shown, for example.
431  *
432  * @param {DwtAccordionItem}	item		the accordion item whose header was clicked
433  * @param {DwtUiEvent}	ev		the click event
434  * 
435  * @private
436  */
437 DwtAccordion.prototype._handleOnRightClickHeader =
438 function(item, ev) {
439 	ev = ev || window.event;
440 
441 	if (this.isListenerRegistered(DwtEvent.ONCONTEXTMENU)) {
442 		var selEv = DwtShell.selectionEvent;
443 		DwtUiEvent.copy(selEv, ev);
444 		selEv.item = this;
445 		selEv.detail = item;
446 		this.notifyListeners(DwtEvent.ONCONTEXTMENU, selEv);
447 	}
448 };
449 
450 DwtAccordion.prototype._handleOnMouseoverHeader =
451 function(item, ev) {
452 	ev = ev || window.event;
453 
454 	if (this.isListenerRegistered(DwtEvent.ONMOUSEOVER)) {
455 		var selEv = DwtShell.selectionEvent;
456 		DwtUiEvent.copy(selEv, ev);
457 		selEv.item = this;
458 		selEv.detail = item;
459 		this.notifyListeners(DwtEvent.ONMOUSEOVER, selEv);
460 	}
461 };
462 
463 DwtAccordion.prototype._handleOnMouseoutHeader =
464 function(ev) {
465 	this.setToolTipContent(null);
466 };

468 /**
469  * Handles a resize event.
470  *
471  * @param {DwtEvent}	ev	the control event
472  * 
473  * @private
474  */
475 DwtAccordion.prototype._controlListener =
476 function(ev) {
477 	if (this.getScrollStyle() != Dwt.CLIP) { return; }
478 
479 	var newWidth = (ev.oldWidth != ev.newWidth) ? ev.newWidth : null;
480 	var newHeight = (ev.oldHeight != ev.newHeight) ? ev.newHeight : null;
481 
482 	if ((!newWidth && !newHeight) || ev.newWidth < 0 || ev.newHeight < 0) { return;	}
483 
484 	this.resize(newWidth, newHeight);
485 };

550 /**
551  * @private
552  */
553 DwtAccordionItem.prototype._setEnabled =
554 function(enabled) {
555 	var titleCell = document.getElementById(this.accordion._htmlElId + "_title_" + this.id);
556 	if (titleCell) {
557 		if (enabled) {
558 			Dwt.delClass(titleCell, "ZDisabled");
559 		} else {
560 			Dwt.addClass(titleCell, "ZDisabled");
561 		}
562 	}
563 
564 	var status = document.getElementById(this.accordion._htmlElId + "_status_" + this.id);
565 	if (status) {
566 		if (enabled) {
567 			Dwt.delClass(status, "ZDisabledImage ZDisabled");
568 		} else {
569 			Dwt.addClass(status, "ZDisabledImage ZDisabled");
570 		}
571 	}
