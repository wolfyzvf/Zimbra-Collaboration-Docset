 28 /**
 29  * Creates a label.
 30  * @constructor
 31  * @class
 32  * This class represents a label, which consists of an image and/or text. It is used
 33  * both as a concrete class and as the base class for {@link DwtButton}. The label
 34  * components are managed within a table. The label can be enabled or disabled, which are reflected in 
 35  * its display. A disabled label looks greyed out.
 36  * 
 37  * <h4>CSS</h4>
 38  * <ul>
 39  * <li><code>.className table</code> - the label table</li>
 40  * <li><code>.className .Icon</code> - class name for the icon image cell</li>
 41  * <li><code>.className .Text</code> - enabled text cell</li>
 42  * <li><code>.className .DisabledText</code> - disabled text cell</li>
 43  * </ul>
 44  * 
 45  * <h4>Keyboard Actions</h4>
 46  * None
 47  * 
 48  * <h4>Events</h4>
 49  * None
 50  * 
 51  * @author Ross Dargahi
 52  * 
 53  * @param {hash}		params		the hash of parameters
 54  * @param	{DwtComposite}	params.parent	the parent widget
 55  * @param	{constant}	params.style		the label style: May be one of: {@link DwtLabel.IMAGE_LEFT} 
 56  * 											or {@link DwtLabel.IMAGE_RIGHT} arithmetically or'd (|) with  one of:
 57  * 											{@link DwtLabel.ALIGN_LEFT}, {@link DwtLabel.ALIGN_CENTER}, or {@link DwtLabel.ALIGN_LEFT}
 58  * 											The first determines were in the label the icon will appear (if one is set), the second
 59  * 											determine how the content of the label will be aligned. The default value for
 60  * 											this parameter is: {@link DwtLabel.IMAGE_LEFT} | {@link DwtLabel.ALIGN_CENTER}
 61  * @param	{string}	params.className	the CSS class
 62  * @param	{constant}	params.posStyle		the positioning style (see {@link DwtControl})
 63  * @param	{string}	params.id			the to use for the control HTML element
 64  * @param	{number}	params.index 		the index at which to add this control among parent's children
 65  *        
 66  * @extends DwtComposite
 67  */
 68 DwtLabel = function(params) {
 69 	if (arguments.length == 0) { return; }
 70 	params = Dwt.getParams(arguments, DwtLabel.PARAMS);
 71 	
 72 	params.className = params.className || "DwtLabel";
 73 	DwtComposite.call(this, params);
 74 
 75 	/**
 76 	 * The label style. See the constructor for more info.
 77 	 */
 78 	this._style = params.style || (DwtLabel.IMAGE_LEFT | DwtLabel.ALIGN_CENTER);
 79 	
 80 	/**
 81 	 * The label text background color.
 82 	 */
 83 	this._textBackground = null;
 84 	
 85 	/**
 86 	 * The label text foreground color.
 87 	 */
 88 	this._textForeground = null;
 89 
 90     this._createHtml(params.template);
 91     //MOW:  this.setCursor("default");
 92 }
 93 
 94 DwtLabel.PARAMS = ["parent", "style", "className", "posStyle", "id", "index"];
 95

 97 DwtLabel.prototype.constructor = DwtLabel;

172 /**

175  */
176 DwtLabel.prototype.dispose =
177 function() {
178 	delete this._dropDownEl;
179 	delete this._iconEl;
180 	delete this._textEl;
181 	DwtControl.prototype.dispose.call(this);
182 };

184 /**
185  * Sets the enabled/disabled state of the label. A disabled label may have a different
186  * image, and greyed out text. This method overrides {@link DwtControl#setEnabled}.
187  *
188  * @param {boolean} enabled 		if <code>true</code>, set the label as enabled
189  */
190 DwtLabel.prototype.setEnabled =
191 function(enabled) {
192 	if (enabled != this._enabled) {
193 		DwtControl.prototype.setEnabled.call(this, enabled);
194 		var direction = this._style & DwtLabel.IMAGE_RIGHT ? DwtLabel.RIGHT : DwtLabel.LEFT;
195 		this.__imageInfo = this.__imageInfo || {};
196 		this.__setImage(this.__imageInfo[direction]);
197 	}
198 }

200 /**
201  * Gets the current image info.
202  *
203  * @param	{string}	direction		position of the image
204  *
205  * @return	{string}	the image info
206  */
207 DwtLabel.prototype.getImage =
208 function(direction) {
209 	direction = direction || (this._style & DwtLabel.IMAGE_RIGHT ? DwtLabel.RIGHT : DwtLabel.LEFT);
210 	return this.__imageInfo[direction];
211 }

213 /**
214  * Sets the main (enabled) image. If the label is currently enabled, the image is updated.
215  *
216  * @param	{string}	imageInfo		the image
217  * @param	{string}	direction		position of the image
218  */
219 DwtLabel.prototype.setImage =
220 function(imageInfo, direction) {
221 	direction = direction || (this._style & DwtLabel.IMAGE_RIGHT ? DwtLabel.RIGHT : DwtLabel.LEFT);
222 	this.__imageInfo = this.__imageInfo || {};
223 	this.__imageInfo[direction] = imageInfo;
224 	this.__setImage(imageInfo, direction);
225 }

254 /**
255  * Gets the label text.
256  * 
257  * @return	{string}	the text or <code>null</code> if not set
258  */
259 DwtLabel.prototype.getText =
260 function() {
261 	return (this.__text != null) ? this.__text : null;
262 }
263 
264 /**
265 * Sets the label text, and manages the placement and display.
266 *
267 * @param {string}	text	the new label text
268 */
269 DwtLabel.prototype.setText = function(text) {
270 
271     if (!this._textEl) {
272 	    return;
273     }
274 
275     if (text == null || text == "") {
276         this.__text = null;
277         this._textEl.innerHTML = "";
278     }
279     else {
280 		this.__text = text;
281         this._textEl.innerHTML = text;
282     }
283 
284 	this._textSet(text);
285 };

292 DwtLabel.prototype.setTextBackground =

296         this._textEl.style.backgroundColor = color;
297     }
298 }

303  * @param	{string}	color	the foreground color

313 /**
314  * Sets the align style.
315  * 
316  * @param		{constant}		alignStyle		the align style (see {@link DwtControl})
317  */
318 DwtLabel.prototype.setAlign =
319 function(alignStyle) {
320 	this._style = alignStyle;
321 
322 	// reset dom since alignment style may have changed
323 	var direction = this._style & DwtLabel.IMAGE_RIGHT ? DwtLabel.RIGHT : DwtLabel.LEFT;
324 	this.__imageInfo = this.__imageInfo || {};
325     this.__setImage(this.__imageInfo[direction]);
326 }

342 /**
343  * @private
344  */
345 DwtLabel.prototype._createHtml = function(templateId) {
346     var data = { id: this._htmlElId };
347     this._createHtmlFromTemplate(templateId || this.TEMPLATE, data);
348 };
349 
350 /**
351  * @private
352  */
353 DwtLabel.prototype._createHtmlFromTemplate = function(templateId, data) {
354     DwtControl.prototype._createHtmlFromTemplate.call(this, templateId, data);
355     this._textEl = document.getElementById(data.id+"_title");
356 };
357 
358 /**
359  * @private
360  *
361  * @param	{string}	direction		position of the image
362  */
363 DwtLabel.prototype._getIconEl = function(direction) {
364     var _dir = this._style & DwtLabel.IMAGE_RIGHT ? DwtLabel.RIGHT : DwtLabel.LEFT;
365     direction = typeof direction === 'boolean' ? _dir : (direction || _dir);    // fix for Bug 90130
366 	// MOW: getting the proper icon element on demand rather than all the time for speed
367 	this._iconEl = this._iconEl || {};
368 	return this._iconEl[direction] ||
369 		(this._iconEl[direction] = document.getElementById(this._htmlElId+"_"+direction+"_icon"));
370 };

376 /**
377  * Set the label's image, and manage its placement.
378  *
379  * @private
380  *
381  * @param	{string}	imageInfo		the image
382  * @param	{string}	direction		position of the image
383  */
384 DwtLabel.prototype.__setImage =
385 function(imageInfo, direction) {
386 
387 	var iconEl = this._getIconEl(direction);
388 	if (iconEl) {
389 		if (imageInfo) {
390 			AjxImg.setImage(iconEl, imageInfo, null, !this._enabled);
391 
392 			// set a ZHasRightIcon or ZHasLeftIcon on the outer element, depending on which we set
393 			var elementClass = (this._style & DwtLabel.IMAGE_RIGHT ? "ZHasRightIcon" : "ZHasLeftIcon");
394 			Dwt.addClass(this.getHtmlElement(), elementClass);
395 		} else {
396 			iconEl.innerHTML = "";
397 		}
398 	}
399 };
400 
401 // Accessibility
402 DwtLabel.prototype._textSet = function(text) {
403 
404 	// assign the ARIA label directly; we want it to override the tooltip, if any
405 	if (!this.hasAttribute('aria-labelledby')) {
406 		if (text) {
407 			this.setAttribute('aria-label', text);
408 		} else {
409 			this.removeAttribute('aria-label');
410 		}
411 	}
412 };
