 22 /**
 23  * Creates an empty mobile device object.
 24  * @constructor
 25  * @class
 26  * This class represents  mobile device.
 27  * 
 28  * @author Parag Shah
 29  *
 30  * @param	{Hash}		params		a hash of parameters
 31  * @param	{Date}		params.lastPolicyUpdate		the last policy update time
 32  * @param	{Date}		params.firstReqReceived		the first request received time
 33  * @param	{Date}		params.remoteWipeAckTime		the remote wipe acknowledged time
 34  * @param	{Date}		params.remoteWipeReqTime		the remote wipe requested time
 35  * @param	{constant}	params.status			the status (see <code>ZmMobileDevice.STATUS_</code>)
 36  */
 37 ZmMobileDevice = function(params) {
 38 	this.lastPolicyUpdate = params.lastPolicyUpdate;
 39 	this.firstReqReceived = params.firstReqReceived;
 40 	this.remoteWipeAckTime = params.remoteWipeAckTime;
 41 	this.remoteWipeReqTime = params.remoteWipeReqTime;
 42 	this.status = params.status;
 43 	this.provisionable = params.provisionable;
 44 	this.protocol = params.protocol;
 45 	this.ua = params.ua;
 46 	this.type = params.type;
 47 	this.id = params.id;
 48 };

 82 /**
 83  * Gets the status.
 84  * 
 85  * @return	{constant}	the status (see <code>ZmMobileDevice.STATUS_</code> constants)
 86  */
 87 ZmMobileDevice.prototype.getStatus =
 88 function() {
 89 	return (!this.provisionable && this.status == ZmMobileDevice.STATUS_NEED_PROVISION)
 90 		? ZmMobileDevice.STATUS_OK : this.status;
 91 };
 92 
 93 /**
 94  * Gets the status string.
 95  * 
 96  * @return	{String}	the status string
 97  */
 98 ZmMobileDevice.prototype.getStatusString =
 99 function() {
100 	var status = this.getStatus();
101 
102 	switch (status) {
103 		case ZmMobileDevice.STATUS_NEED_PROVISION:			return ZmMsg.mobileStatusNeedProvision;
104 		case ZmMobileDevice.STATUS_OK:						return ZmMsg.mobileStatusOk;
105 		case ZmMobileDevice.STATUS_SUSPENDED:				return ZmMsg.mobileStatusSuspended;
106 		case ZmMobileDevice.STATUS_REMOTE_WIPE_REQUESTED:	return ZmMsg.mobileStatusWipe;
107 		case ZmMobileDevice.STATUS_REMOTE_WIPE_COMPLETE:	return ZmMsg.mobileStatusWipeComplete;
108 	}
109 	return "";
110 };

152 ZmMobileDevice.prototype.doAction =
153 function(id, callback) {
154 	var request;
155 	switch (id) {
156 		case ZmOperation.MOBILE_REMOVE: 		request = "RemoveDeviceRequest"; break;
157 		case ZmOperation.MOBILE_RESUME_SYNC:	request = "ResumeDeviceRequest"; break;
158 		case ZmOperation.MOBILE_SUSPEND_SYNC:	request = "SuspendDeviceRequest"; break;
159 		case ZmOperation.MOBILE_WIPE:			request = "RemoteWipeRequest"; break;
160 		case ZmOperation.MOBILE_CANCEL_WIPE:	request = "CancelPendingRemoteWipeRequest"; break;
161 	}
162 
163 	if (request) {
164 		var soapDoc = AjxSoapDoc.create(request, "urn:zimbraSync");
165 		var node = soapDoc.set("device");
166 		node.setAttribute("id", this.id);
167 
168 		var respCallback = new AjxCallback(this, this._handleDoAction, callback);
169 		appCtxt.getAppController().sendRequest({soapDoc:soapDoc, asyncMode:true, callback:respCallback});
170 	}
171 };
172 
173 ZmMobileDevice.prototype._handleDoAction =
174 function(callback, results) {
175 	var resp = results.getResponse();
176 	for (var i in resp) {
177 		var device = resp[i].device && resp[i].device[0];
178 		if (device && device.id == this.id) {
179 			this.status = device.status;
180 			if (device.lastPolicyUpdate) {
181 				this.lastPolicyUpdate = device.lastPolicyUpdate;
182 			}
183 			if (device.firstReqReceived) {
184 				this.firstReqReceived = device.firstReqReceived;
185 			}
186 			if (device.remoteWipeAckTime) {
187 				this.remoteWipeAckTime = device.remoteWipeAckTime;
188 			}
189 			if (device.remoteWipeReqTime) {
190 				this.remoteWipeReqTime = device.remoteWipeReqTime;
191 			}
192 		}
193 	}
194 
195 	if (callback) {
196 		callback.run();
197 	}
198 };
