 22 /**
 23  * Creates a confirmation dialog.
 24  * @class
 25  * This class represents a confirmation dialog.
 26  * 
 27  * @param {DwtComposite}	parent  the parent widget (the shell)
 28  * @param {string}		className  the CSS class
 29  * 
 30  * @extends		DwtDialog
 31  */
 32 DwtConfirmDialog = function(parent, className, id, buttons, extraButtons) {
 33 	if (arguments.length == 0) return;
 34 
 35     if(!buttons){
 36 	buttons = [ DwtDialog.YES_BUTTON, DwtDialog.NO_BUTTON, DwtDialog.CANCEL_BUTTON ];
 37     }
 38 	DwtDialog.call(this, {parent:parent, className:className || "DwtConfirmDialog", title:AjxMsg.confirmTitle, standardButtons:buttons, id:id, extraButtons:extraButtons});
 39 	
 40 	this._questionDiv = document.createElement("DIV");
 41 	this._questionDiv.className = "DwtConfirmDialogQuestion";
 42 	this._getContentDiv().appendChild(this._questionDiv);
 43 	
 44 	this.registerCallback(DwtDialog.YES_BUTTON, this._handleYesButton, this);
 45 	this.registerCallback(DwtDialog.NO_BUTTON, this._handleNoButton, this);
 46 	this.registerCallback(DwtDialog.CANCEL_BUTTON, this._handleCancelButton, this);
 47 }
 48 DwtConfirmDialog.prototype = new DwtDialog;
 49 DwtConfirmDialog.prototype.constructor = DwtConfirmDialog;
 50 
 51 DwtConfirmDialog.prototype.toString =
 52 function() {
 53 	return "DwtConfirmDialog";
 54 };
 55

 57 // Public methods
 58 
 59 /**
 60  * Pops up the confirmation dialog. The caller passes in the confirmation
 61  * question and callbacks for the Yes, No, and Cancel buttons.
 62  * <p>
 63  * <strong>Note:</strong>
 64  * If the callback for the No button is not specified, the confirmation
 65  * dialog assumes that the caller is only concerned with a Yes response
 66  * and hides the (presumably) extraneous Cancel button.
 67  * 
 68  * @param	{string}	questionHtml		the question HTML
 69  * @param	{AjxCallback}	yesCallback		the "yes" button callback
 70  * @param	{AjxCallback}	noCallback		the "no" button callback
 71  * @param	{AjxCallback}	cancelCallback		the "cancel" button callback
 72  * @param	{DwtPoint}	loc			the location
 73  */
 74 DwtConfirmDialog.prototype.popup = 
 75 function(questionHtml, yesCallback, noCallback, cancelCallback, loc) {
 76 	this._questionDiv.innerHTML = questionHtml || "";
 77 	
 78 	this._yesCallback = yesCallback;
 79 	this._noCallback = noCallback;
 80 	this._cancelCallback = cancelCallback;
 81 	
 82 	this.setButtonVisible(DwtDialog.CANCEL_BUTTON, Boolean(noCallback));
 83 	
 84 	DwtDialog.prototype.popup.call(this, loc);
 85 };
 86 
 87 DwtConfirmDialog.prototype.popdown =
 88 function() {
 89 	this._yesCallback = this._noCallback = this._cancelCallback = null;
 90 	DwtDialog.prototype.popdown.call(this);
 91 };
 92 
 93 // Protected methods
 94 
 95 DwtConfirmDialog.prototype._handleYesButton =
 96 function(ev) {
 97 	if (this._yesCallback) this._yesCallback.run(ev);
 98 	this.popdown();
 99 };
100 
101 DwtConfirmDialog.prototype._handleNoButton =
102 function(ev) {
103 	if (this._noCallback) this._noCallback.run(ev);
104 	this.popdown();
105 };
106 
107 DwtConfirmDialog.prototype._handleCancelButton =
108 function(ev) {
109 	if (this._cancelCallback) this._cancelCallback.run(ev);
110 	this.popdown();
111 };
112 
113 DwtConfirmDialog.prototype._getSeparatorTemplate =
114 function() {
115 	return "";
116 };
