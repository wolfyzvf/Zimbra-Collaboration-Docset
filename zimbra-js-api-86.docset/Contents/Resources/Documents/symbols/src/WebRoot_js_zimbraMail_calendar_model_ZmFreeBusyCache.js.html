 59 //filter free busy slots for given time from compressed/accumulated free busy response that got cached already
 60 ZmFreeBusyCache.prototype.getFreeBusySlot =
 61 function(startTime, endTime, id, excludeTimeSlots) {
 62     var slotDate = new Date(startTime);
 63     slotDate.setHours(0, 0, 0, 0);
 64 
 65     var fbSlots = this._schedule[id] || [];
 66     var fbResult = {id: id};
 67 
 68     //free busy response is always merged
 69     var usr, searchRange, newSearchIsInRange;
 70     for(var i= fbSlots.length; --i >= 0;) {
 71         usr = fbSlots[i];
 72         searchRange = usr.searchRange;
 73 
 74         if(searchRange) {
 75             newSearchIsInRange = (startTime >= searchRange.startTime && endTime <= searchRange.endTime);
 76             if(!newSearchIsInRange) continue;
 77         }
 78 
 79         if (usr.n) this._addFBInfo(usr.n, id, ZmFreeBusyCache.STATUS_UNKNOWN, startTime, endTime, fbResult, excludeTimeSlots);
 80         if (usr.t) this._addFBInfo(usr.t, id, ZmFreeBusyCache.STATUS_TENTATIVE, startTime, endTime, fbResult, excludeTimeSlots);
 81         if (usr.b) this._addFBInfo(usr.b, id, ZmFreeBusyCache.STATUS_BUSY, startTime, endTime, fbResult, excludeTimeSlots);
 82         if (usr.u) this._addFBInfo(usr.u, id, ZmFreeBusyCache.STATUS_OUT, startTime, endTime, fbResult, excludeTimeSlots);
 83         if (usr.f) this._addFBInfo(usr.f, id, ZmFreeBusyCache.STATUS_FREE, startTime, endTime, fbResult, excludeTimeSlots);
 84     }
 85 
 86     return fbResult;
 87 };

 89 ZmFreeBusyCache.prototype._addFBInfo =
 90 function(slots, id, status, startTime, endTime, fbResult, excludeTimeSlots) {
 91 
 92     if(!fbResult[status]) fbResult[status] = [];
 93 
 94     for (var i = 0; i < slots.length; i++) {
 95         var fbSlot;
 96         if(slots[i].s >= startTime && slots[i].e  <= endTime) {
 97             fbSlot = {s: slots[i].s, e: slots[i].e};
 98         }else if(startTime >= slots[i].s && endTime  <= slots[i].e) {
 99             fbSlot = {s: startTime, e: endTime};
100         }else if(startTime >= slots[i].s && startTime  <= slots[i].e) {
101             fbSlot = {s: startTime, e: slots[i].e};
102         }else if(endTime >= slots[i].s && endTime  <= slots[i].e) {
103             fbSlot = {s: slots[i].s, e: endTime};
104         }
105 
106         if(fbSlot) {
107             if(excludeTimeSlots && status != ZmFreeBusyCache.STATUS_FREE && status != ZmFreeBusyCache.STATUS_UNKNOWN) {
108                 this._addByExcludingTime(excludeTimeSlots, fbSlot, fbResult, status);
109             }else {
110                 fbResult[status].push(fbSlot);
111             }
112         }
113     };
114 
115     if(fbResult[status].length == 0) fbResult[status] = null;
116 };
117 
118 ZmFreeBusyCache.prototype._addByExcludingTime =
119 function(excludeTimeSlots, fbSlot, fbResult, status) {
120     var startTime =  excludeTimeSlots.s;
121     var endTime =  excludeTimeSlots.e;
122     var newFBSlot;
123 
124     if(fbSlot.s == startTime && fbSlot.e == endTime) {
125         return;
126     }
127 
128     if(fbSlot.s < startTime && fbSlot.e > endTime) {
129         fbResult[status].push({s: fbSlot.s, e: startTime});
130         newFBSlot = {s: endTime, e: fbSlot.e};
131     }else if(fbSlot.s < startTime && fbSlot.e >= startTime) {
132         newFBSlot = {s: fbSlot.s, e: startTime};
133     }else if(fbSlot.s <= endTime && fbSlot.e > endTime) {
134         newFBSlot = {s: endTime, e: fbSlot.e};
135     }else if(fbSlot.s <= startTime && fbSlot.e <= startTime) {
136         newFBSlot = {s: fbSlot.s, e: fbSlot.e};
137     }else if(fbSlot.s >= endTime && fbSlot.e >= endTime) {
138         newFBSlot = {s: fbSlot.s, e: fbSlot.e};
139     }
140 
141     if(newFBSlot) {
142         fbResult[status].push(newFBSlot);
143     }
144 };

146 ZmFreeBusyCache.prototype.getFreeBusyInfo =
147 function(params) {
148 
149     var requiredEmails = [], freeBusyKey, emails = params.emails, fbSlot;
150     for (var i = emails.length; --i >= 0;) {
151         freeBusyKey = params.startTime + "";
152         //check local cache
153         var entryExists = false
154         if(this._schedule[emails[i]]) {
155             var fbSlots = this.getFreeBusySlot(params.startTime, params.endTime, emails[i]);
156             if(fbSlots.f || fbSlots.u || fbSlots.b || fbSlots.t || fbSlots.n) entryExists = true;            
157         };
158         if(!entryExists) requiredEmails.push(emails[i]);
159     }
160 
161     var fbCallback = new AjxCallback(this, this._handleResponseFreeBusy, [params]);
162     var fbErrorCallback = new AjxCallback(this, this._handleErrorFreeBusy, [params]);
163 
164     if(requiredEmails.length) {
165 	    return this._getFreeBusyInfo(params.startTime,
166                                      params.endTime,
167                                      requiredEmails.join(","),
168                                      fbCallback,
169                                      fbErrorCallback,
170                                      params.noBusyOverlay,
171                                      params.account,
172                                      params.excludedId);
173     }else {
174         if(params.callback) {
175             params.callback.run();
176         }
177         return null;
178     }
179 
180 };

182 //cache free busy response in user-id -> slots hash map
183 ZmFreeBusyCache.prototype. _handleResponseFreeBusy =
184 function(params, result) {
185 
186     var freeBusyKey;
187 	var args = result.getResponse().GetFreeBusyResponse.usr || [];
188     for (var i = 0; i < args.length; i++) {
189 		var usr = args[i];
190         var id = usr.id;
191         if (!id) {
192             continue;
193         }
194         if(!this._schedule[id]) {
195             this._schedule[id] = [];
196         }
197 
198         usr.searchRange = {startTime: params.startTime,  endTime: params.endTime};
199         this._schedule[id].push(usr);
200     };
201 
202     if(params.callback) {
203         params.callback.run(result);
204     }
205 };

207 ZmFreeBusyCache.prototype._handleErrorFreeBusy =
208 function(params, result) {
209     if(params.errorCallback) {
210         params.errorCallback.run(result);
211     }
212 };

214 ZmFreeBusyCache.prototype._getFreeBusyInfo =
215 function(startTime, endTime, emailList, callback, errorCallback, noBusyOverlay, acct, excludedId) {
216 	var soapDoc = AjxSoapDoc.create("GetFreeBusyRequest", "urn:zimbraMail");
217 	soapDoc.setMethodAttribute("s", startTime);
218 	soapDoc.setMethodAttribute("e", endTime);
219 	soapDoc.setMethodAttribute("uid", emailList);
220 	if (excludedId) {
221 		soapDoc.setMethodAttribute("excludeUid", excludedId);
222 	}
223 
224 	return appCtxt.getAppController().sendRequest({
225 		soapDoc: soapDoc,
226 		asyncMode: true,
227 		callback: callback,
228 		errorCallback: errorCallback,
229 		noBusyOverlay: noBusyOverlay,
230 		accountName: (acct ? acct.name : null)
231 	});
232 };

234 //working hrs related code
235 ZmFreeBusyCache.prototype.getWorkingHours =
236 function(params) {
237 
238     var requiredEmails = [], whKey, emails = params.emails || [];
239     for (var i = emails.length; --i >= 0;) {
240         whKey = this.getWorkingHoursKey(emails[i], (new Date(params.startTime)).getDay());
241         //check local cache
242         if(!this._workingHrs[whKey]) requiredEmails.push(emails[i]);
243     }
244 
245     var fbCallback = new AjxCallback(this, this._handleResponseWorkingHrs, [params]);
246     var fbErrorCallback = new AjxCallback(this, this._handleErrorWorkingHrs, [params]);
247 
248     if(requiredEmails.length) {
249 	    return this._getWorkingHours(params.startTime,
250                                      params.endTime,
251                                      requiredEmails.join(","),
252                                      fbCallback,
253                                      fbErrorCallback,
254                                      params.noBusyOverlay,
255                                      params.account);
256     }else {
257         if(params.callback) {
258             params.callback.run();
259         }
260         return null;
261     }
262 
263 };

265 ZmFreeBusyCache.prototype._handleResponseWorkingHrs =
266 function(params, result) {
267 
268     var freeBusyKey;
269 	var args = result.getResponse().GetWorkingHoursResponse.usr || [];
270     for (var i = 0; i < args.length; i++) {
271 		var usr = args[i];
272         var id = usr.id;
273         if (!id) {
274             continue;
275         }
276         this._addWorkingHrInfo(params.startTime, params.endTime, usr);
277     };
278 
279     if(params.callback) {
280         params.callback.run(result);
281     }
282 };

284 ZmFreeBusyCache.prototype._addWorkingHrInfo =
285 function(startTime, endTime, usr) {
286     var id = usr.id;
287     if (usr.f) this._addWorkingHrSlot(usr.f, id, ZmFreeBusyCache.STATUS_WORKING_HOURS);
288     if (usr.u) this._addWorkingHrSlot(usr.u, id, ZmFreeBusyCache.STATUS_NON_WORKING_HOURS);
289     if (usr.n) this._addWorkingHrSlot(usr.n, id, ZmFreeBusyCache.STATUS_UNKNOWN);
290 };

292 ZmFreeBusyCache.prototype._addWorkingHrSlot =
293 function(slots, id, status) {
294     var slotTime, slotDate, whKey, whSlots;
295     for (var i = 0; i < slots.length; i++) {
296         slotTime = slots[i].s;
297         slotDate = new Date(slotTime);
298         whKey = this.getWorkingHoursKey(id, slotDate.getDay());
299         whSlots = this._workingHrs[whKey];
300         if(!whSlots) {
301             this._workingHrs[whKey] = whSlots = {id: id};
302         }
303 
304         if(!whSlots[status]) {
305             whSlots[status] = [];
306         }
307         whSlots[status].push(slots[i]);
308 
309         //unknown working hours slots will be compressed on server response (will send one accumulated slots)
310         if(status == ZmFreeBusyCache.STATUS_UNKNOWN) {
311             this._workingHrs[id] = whSlots;
312         }
313     };
314 };

341 ZmFreeBusyCache.prototype.getWorkingHrsSlot =
342 function(startTime, endTime, id) {
343     var whKey = this.getWorkingHoursKey(id, (new Date(startTime)).getDay());
344     var whSlots = this._workingHrs[whKey] || {};
345     var whResult = {id: id};
346 
347     //handle the case where the working hours are not available and slot dates are accumulated
348     var unknownSlots = this._workingHrs[id];
349     if(unknownSlots) {
350         return unknownSlots;
351     }
352 
353     if(whSlots[ZmFreeBusyCache.STATUS_WORKING_HOURS]) whResult[ZmFreeBusyCache.STATUS_WORKING_HOURS] = whSlots[ZmFreeBusyCache.STATUS_WORKING_HOURS];
354     if(whSlots[ZmFreeBusyCache.STATUS_NON_WORKING_HOURS]) whResult[ZmFreeBusyCache.STATUS_NON_WORKING_HOURS] = whSlots[ZmFreeBusyCache.STATUS_NON_WORKING_HOURS];
355     if(whSlots[ZmFreeBusyCache.STATUS_UNKNOWN]) whResult[ZmFreeBusyCache.STATUS_UNKNOWN] = whSlots[ZmFreeBusyCache.STATUS_UNKNOWN];
356 
357     if(!whResult[ZmFreeBusyCache.STATUS_WORKING_HOURS] && !whResult[ZmFreeBusyCache.STATUS_NON_WORKING_HOURS] && !whResult[ZmFreeBusyCache.STATUS_UNKNOWN]) return null;
358     return whResult;        
359 };

361 ZmFreeBusyCache.prototype._addWHInfo =
362 function(slots, id, status, startTime, endTime, whResult) {
363 
364     if(!whResult[status]) whResult[status] = [];
365 
366     for (var i = 0; i < slots.length; i++) {
367         if(startTime >= slots[i].s && endTime <= slots[i].e) {
368             whResult[status].push({s: startTime, e: endTime});
369         }else if(slots[i].s >= startTime && slots[i].e <= endTime) {
370             whResult[status].push({s: slots[i].s, e: slots[i].e});
371         }else if(slots[i].s >= startTime && slots[i].s  <= endTime) {
372             whResult[status].push({s: slots[i].s, e: endTime});
373         }else if(slots[i].e >= startTime && slots[i].e  <= endTime) {
374             whResult[status].push({s: startTime, e: slots[i].e});
375         }
376     };
377 
378     if(whResult[status].length == 0) whResult[status] = null;
379 };

381 /**
382  * converts working hours in different time base to required or current time base
383  * this is done due to the fact that working hrs pattern repeat everyweek and
384  * working hours are not fetched for every date change to optimize client code
385  * @param slot  {object} working hrs slot with start and end time in milliseconds
386  * @param relativeDate {date} optional date object relative to which the slot timings are converted
387  */
388 ZmFreeBusyCache.prototype.convertWorkingHours =
389 function(slot, relativeDate) {
390     relativeDate = relativeDate || new Date();
391     var slotStartDate = new Date(slot.s);
392     var slotEndDate = new Date(slot.e);
393     var dur = slot.e - slot.s;
394     slot.s = (new Date(relativeDate.getTime())).setHours(slotStartDate.getHours(), slotStartDate.getMinutes(), 0, 0);
395     slot.e = slot.s + dur;
396 };
