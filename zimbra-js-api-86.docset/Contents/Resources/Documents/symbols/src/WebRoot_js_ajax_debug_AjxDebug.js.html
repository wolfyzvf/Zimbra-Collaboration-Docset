 23 /**
 24  * Creates a new debug window. The document inside is not kept open.  All the
 25  * output goes into a single <div> element.
 26  * @constructor
 27  * @class
 28  * This class pops up a debug window and provides functions to send output there
 29  * in various ways. The output is continuously appended to the bottom of the
 30  * window. The document is left unopened so that the browser doesn't think it's
 31  * continuously loading and keep its little icon flailing forever. Also, the DOM
 32  * tree can't be manipulated on an open document. All the output is added to the
 33  * window by appending it the DOM tree. Another method of appending output is to
 34  * open the document and use document.write(), but then the document is left open.
 35  * <p>
 36  * Any client that uses this class can turn off debugging by changing the first
 37  * argument to the constructor to {@link AjxDebug.NONE}.
 38  *
 39  * @author Conrad Damon
 40  * @author Ross Dargahi
 41  *
 42  * @param {constant}	level	 	debug level for the current debugger (no window will be displayed for a level of NONE)
 43  * @param {string}		name 		the name of the window (deprecated)
 44  * @param {boolean}		showTime	if <code>true</code>, display timestamps before debug messages
 45  * @param {constant}	target		output target (AjxDebug.TGT_WINDOW | AjxDebug.TGT_CONSOLE)
 46  *
 47  * @private
 48  */
 49 AjxDebug = function(params) {
 50 
 51 	if (arguments.length == 0) {
 52 		params = {};
 53 	}
 54 	else if (typeof arguments[0] == "number") {
 55 		params = {level:arguments[0], name:arguments[1], showTime:arguments[2]};
 56 	}
 57 
 58 	this._showTime = params.showTime;
 59 	this._target = params.target || AjxDebug.TGT_WINDOW;
 60 	this._showTiming = false;
 61 	this._startTimePt = this._lastTimePt = 0;
 62 	this._dbgWindowInited = false;
 63 
 64 	this._msgQueue = [];
 65 	this._isPrevWinOpen = false;
 66 	this.setDebugLevel(params.level);
 67 };

183 /**
184  * Prints a debug message. Any HTML will be rendered, and a line break is added.
185  *
186  * @param {constant}	level	 	debug level for the current debugger
187  * @param {string}	msg		the text to display
188  */
189 AjxDebug.prototype.println =
190 function(level, msg, linkName) {
191 	
192 	if (!this._isWriteable()) { return; }
193 
194 	try {
195 		var result = this._handleArgs(arguments);
196 		if (!result) { return; }
197 
198 		msg = result.args.join("");
199 		var eol = (this._target != AjxDebug.TGT_CONSOLE) ? "<br>" : "";
200 		this._add({msg:this._timestamp() + msg + eol, linkName:result.linkName, level:level});
201 	} catch (ex) {
202 		// do nothing
203 	}
204 };

216 /**
217  * Prints an object into a table, with a column for properties and a column for values. Above the table is a header with the object
218  * class and the CSS class (if any). The properties are sorted (numerically if they're all numbers). Creating and appending table
219  * elements worked in Mozilla but not IE. Using the insert* methods works for both. Properties that are function
220  * definitions are skipped.
221  *
222  * @param {constant}	level	 	debug level for the current debugger
223  * @param {object}	obj		the object to be printed
224  * @param {boolean}	showFuncs		if <code>true</code>, show props that are functions
225  */
226 AjxDebug.prototype.dumpObj =
227 function(level, obj, showFuncs, linkName) {
228 	if (!this._isWriteable()) { return; }
229 
230 	var result = this._handleArgs(arguments);
231 	if (!result) { return; }
232 
233 	obj = result.args[0];
234 	if (!obj) { return; }
235 
236 	showFuncs = result.args[1];
237 	this._add({obj:obj, linkName:result.linkName, showFuncs:showFuncs, level:level});
238 };

240 /**
241  * Dumps a bunch of text into a <textarea>, so that it is wrapped and scrollable. HTML will not be rendered.
242  *
243  * @param {constant}	level	 	debug level for the current debugger
244  * @param {string}	text		the text to output as is
245  */
246 AjxDebug.prototype.printRaw =
247 function(level, text, linkName) {
248 	if (!this._isWriteable()) { return; }
249 
250 	var result = this._handleArgs(arguments);
251 	if (!result) { return; }
252 
253 	this._add({obj:result.args[0], isRaw:true, linkName:result.linkName, level:level});

256 /**
257  * Pretty-prints a chunk of XML, doing color highlighting for different types of nodes.
258  *
259  * @param {constant}	level	 	debug level for the current debugger
260  * @param {string}	text		some XML
261  * 
262  * TODO: fix for printing to console
263  */
264 AjxDebug.prototype.printXML =
265 function(level, text, linkName) {
266 	if (!this._isWriteable()) { return; }
267 
268 	var result = this._handleArgs(arguments);
269 	if (!result) { return; }
270 
271 	text = result.args[0];
272 	if (!text) { return; }
273 
274 	// skip generating pretty xml if theres too much data
275 	if (text.length > AjxDebug.MAX_OUT) {
276 		this.printRaw(text);
277 		return;
278 	}
279 	this._add({obj:text, isXml:true, linkName:result.linkName, level:level});
280 };

282 /**
283  * Reveals white space in text by replacing it with tags.
284  *
285  * @param {constant}	level	 	debug level for the current debugger
286  * @param {string}	text		the text to be displayed
287  */
288 AjxDebug.prototype.display =
289 function(level, text, linkName) {
290 	if (!this._isWriteable()) { return; }
291 
292 	var result = this._handleArgs(arguments);
293 	if (!result) { return; }
294 
295 	text = result.args[0];
296 	text = text.replace(/\r?\n/g, '[crlf]');
297 	text = text.replace(/ /g, '[space]');
298 	text = text.replace(/\t/g, '[tab]');
299 	this.printRaw(level, text, linkName);
300 };

302 /**
303  * Turn the display of timing statements on/off.
304  *
305  * @param {boolean}	on			if <code>true</code>, display timing statements
306  * @param {string}	msg		the message to show when timing is turned on
307  */
308 AjxDebug.prototype.showTiming =
309 function(on, msg) {
310 	this._showTiming = on;
311 	if (on) {
312 		this._enable(true);
313 	}
314 	var state = on ? "on" : "off";
315 	var text = "Turning timing info " + state;
316 	if (msg) {
317 		text = text + ": " + msg;
318 	}
319 
320 	var debugMsg = new DebugMessage({msg:text});
321 	this._addMessage(debugMsg);
322 	this._startTimePt = this._lastTimePt = new Date().getTime();
323 };
324 
325 /**
326  * Displays time elapsed since last time point.
327  *
328  * @param {string}	msg		the text to display with timing info
329  * @param {boolean}	restart	if <code>true</code>, set timer back to zero
330  */
331 AjxDebug.prototype.timePt =
332 function(msg, restart) {
333 	if (!this._showTiming || !this._isWriteable()) { return; }
334 
335 	if (restart) {
336 		this._startTimePt = this._lastTimePt = new Date().getTime();
337 	}
338 	var now = new Date().getTime();
339 	var elapsed = now - this._startTimePt;
340 	var interval = now - this._lastTimePt;
341 	this._lastTimePt = now;
342 
343 	var spacer = restart ? "<br/>" : "";
344 	msg = msg ? " " + msg : "";
345 	var text = [spacer, "[", elapsed, " / ", interval, "]", msg].join("");
346 	var html = "<div>" + text + "</div>";
347 
348 	// Add the message to our stack
349 	this._addMessage(new DebugMessage({msg:html}));
350 	return interval;
351 };

381 AjxDebug.prototype._enable =
382 function(enabled) {
383 
384 	this._enabled = enabled;
385 	if (this._target == AjxDebug.TGT_WINDOW) {
386 		if (enabled) {
387 			if (!this._dbgName) {
388 				this._dbgName = "AjxDebugWin_" + location.hostname.replace(/\./g,'_');
389 			}
390 			if (this._debugWindow == null || this._debugWindow.closed) {
391 				this._openDebugWindow();
392 			}
393 		} else {
394 			if (this._debugWindow) {
395 				this._debugWindow.close();
396 				this._debugWindow = null;
397 			}
398 		}
399 	}
400 };

402 AjxDebug.prototype._isWriteable =
403 function() {
404 	if (this.isDisabled()) {
405 		return false;
406 	}
407 	if (this._target == AjxDebug.TGT_WINDOW) {
408 		return (!this._isPaused && this._debugWindow && !this._debugWindow.closed);
409 	}
410 	return true;

413 AjxDebug.prototype._getHtmlForObject =
414 function(obj, params) {
415 
416 	params = params || {};
417 	var html = [];
418 	var idx = 0;
419 
420 	if (obj === undefined) {
421 		html[idx++] = "<span>Undefined</span>";
422 	} else if (obj === null) {
423 		html[idx++] = "<span>NULL</span>";
424 	} else if (AjxUtil.isBoolean(obj)) {
425 		html[idx++] = "<span>" + obj + "</span>";
426 	} else if (AjxUtil.isNumber(obj)) {
427 		html[idx++] = "<span>" + obj +"</span>";
428 	} else {
429 		if (params.isRaw) {
430 			html[idx++] = this._timestamp();
431 			html[idx++] = "<textarea rows='25' style='width:100%' readonly='true'>";
432 			html[idx++] = obj;
433 			html[idx++] = "</textarea><p></p>";
434 		} else if (params.isXml) {
435 			var xmldoc = new AjxDebugXmlDocument;
436 			var doc = xmldoc.create();
437 			// IE bizarrely throws error if we use doc.loadXML here (bug 40451)
438 			if (doc && ("loadXML" in doc)) {
439 				doc.loadXML(obj);
440 				html[idx++] = "<div style='border-width:2px; border-style:inset; width:100%; height:300px; overflow:auto'>";
441 				html[idx++] = this._createXmlTree(doc, 0, {"authToken":true});
442 				html[idx++] = "</div>";
443 			} else {
444 				html[idx++] = "<span>Unable to create XmlDocument to show XML</span>";
445 			}
446 		} else {
447 			html[idx++] = "<div style='border-width:2px; border-style:inset; width:100%; height:300px; overflow:auto'><pre>";
448 			html[idx++] = this._dump(obj, true, params.showFuncs, {"ZmAppCtxt":true, "authToken":true});
449 			html[idx++] = "</div></pre>";
450 		}
451 	}
452 	return html.join("");
453 };

462 /**
463  * Marshals args to public debug functions. In general, the debug level is an optional
464  * first arg. If the first arg is a debug level, check it and then strip it from the args.
465  * The last argument is an optional name for the link from the left panel.
466  *
467  * Returns an object with the link name and a list of the arguments (other than level and
468  * link name).
469  *
470  * @param {array}	args				an arguments list
471  *
472  * @private
473  */
474 AjxDebug.prototype._handleArgs =
475 function(args) {
476 
477 	// don't output anything if debugging is off, or timing is on
478 	if (this._level == AjxDebug.NONE || this._showTiming || args.length == 0) { return; }
479 
480 	// convert args to a true Array so they're easier to deal with
481 	var argsArray = new Array(args.length);
482 	for (var i = 0; i < args.length; i++) {
483 		argsArray[i] = args[i];
484 	}
485 
486 	var result = {args:null, linkName:null};
487 
488 	// remove link name from arg list if present - check if last arg is *Request or *Response
489 	var origLen = argsArray.length;
490 	if (argsArray.length > 1) {
491 		var lastArg = argsArray[argsArray.length - 1];
492 		if (lastArg && lastArg.indexOf && (lastArg.indexOf("DebugWarn") != -1 || ((lastArg.indexOf(" ") == -1) && (/Request|Response$/.test(lastArg))))) {
493 			result.linkName = lastArg;
494 			argsArray.pop();
495 		}
496 	}
497 
498 	// check level if provided, strip it from args; level is either a number, or 1-8 lowercase letters/numbers
499 	var userLevel = null;
500 	var firstArg = argsArray[0];
501 	var gotUserLevel = (typeof firstArg == "number" || ((origLen > 1) && firstArg.length <= 8 && /^[a-z0-9]+$/.test(firstArg)));
502 	if (gotUserLevel) {
503 		userLevel = firstArg;
504 		argsArray.shift();
505 	}
506 	if (userLevel && (AjxDebug.BUFFER_MAX[userLevel] == null)) {
507 		if (typeof this._level == "number") {
508 			if (typeof userLevel != "number" || (userLevel > this._level)) { return; }
509 		} else {
510 			if (userLevel != this._level) { return; }
511 		}
512 	}
513 	result.args = argsArray;
514 
515 	return result;
516 };

518 AjxDebug.prototype._openDebugWindow =
519 function(force) {
520 	var name = AjxEnv.isIE ? "_blank" : this._dbgName;
521 	this._debugWindow = window.open("", name, "width=600,height=400,resizable=yes,scrollbars=yes");
522 
523 	if (this._debugWindow == null) {
524 		this._enabled = false;
525 		return;
526 	}
527 
528 	this._enabled = true;
529 	this._isPrevWinOpen = this._debugWindow.debug;
530 	this._debugWindow.debug = true;
531 
532 	try {
533 		this._document = this._debugWindow.document;
534 		this.setTitle("Debug");
535 
536 		if (!this._isPrevWinOpen) {
537 			this._document.write(
538 				"<html>",
539 					"<head>",
540 						"<script>",
541 							"function blank() {return [",
542 								"'<html><head><style type=\"text/css\">',",
543 									"'P, TD, DIV, SPAN, SELECT, INPUT, TEXTAREA, BUTTON {',",
544 											"'font-family: Tahoma, Arial, Helvetica, sans-serif;',",
545 											"'font-size:11px;}',",
546 									"'.Content {display:block;margin:0.25em 0em;}',",
547 									"'.Link {cursor: pointer;color:blue;text-decoration:underline;white-space:nowrap;width:100%;}',",
548 									"'.DebugWarn {color:red;font-weight:bold;}',",
549 									"'.Run {color:black; background-color:red;width:100%;font-size:18px;font-weight:bold;}',",
550 									"'.RunLink {display:block;color:black;background-color:red;font-weight:bold;white-space:nowrap;width:100%;}',",
551 								"'</style></head><body></body></html>'].join(\"\");}",
552 						"</script>",
553 					"</head>",
554 					"<frameset cols='125, *'>",
555 						"<frameset rows='*,40'>",
556 							"<frame name='", AjxDebug._LINK_FRAME_ID, "' id='", AjxDebug._LINK_FRAME_ID, "' src='javascript:parent.parent.blank();'>",
557 							"<frame name='", AjxDebug._BOTTOM_FRAME_NAME, "' id='", AjxDebug._BOTTOM_FRAME_ID, "' src='javascript:parent.parent.blank();' scrolling=no frameborder=0>",
558 						"</frameset>",
559 						"<frame name='", AjxDebug._CONTENT_FRAME_ID, "' id='", AjxDebug._CONTENT_FRAME_ID, "' src='javascript:parent.blank();'>",
560 					"</frameset>",
561 				"</html>"
562 			);
563 			this._document.close();
564 			
565 			var ta = new AjxTimedAction(this, AjxDebug.prototype._finishInitWindow);
566 			AjxTimedAction.scheduleAction(ta, 2500);
567 		} else {
568 			this._finishInitWindow();
569 
570 			this._contentFrame = this._document.getElementById(AjxDebug._CONTENT_FRAME_ID);
571 			this._linkFrame = this._document.getElementById(AjxDebug._LINK_FRAME_ID);
572 			this._createLinkNContent("RunLink", "NEW RUN", "Run", "NEW RUN");
573 
574 			this._attachHandlers();
575 
576 			this._dbgWindowInited = true;
577 			// show any messages that have been queued up, while the window loaded.
578 			this._showMessages();
579 		}
580 	} catch (ex) {
581 		if (this._debugWindow) {
582 			this._debugWindow.close();
583 		}
584 		this._openDebugWindow(true);
585 	}
586 };

588 AjxDebug.prototype._finishInitWindow =
589 function() {
590 	try {
591 		this._contentFrame = this._debugWindow.document.getElementById(AjxDebug._CONTENT_FRAME_ID);
592 		this._linkFrame = this._debugWindow.document.getElementById(AjxDebug._LINK_FRAME_ID);
593 
594 		var frame = this._debugWindow.document.getElementById(AjxDebug._BOTTOM_FRAME_ID);
595 		var doc = frame.contentWindow.document;
596 		var html = [];
597 		var i = 0;
598 		html[i++] = "<table><tr><td><button id='";
599 		html[i++] = AjxDebug._BOTTOM_FRAME_ID;
600 		html[i++] = "_clear'>Clear</button></td><td><button id='";
601 		html[i++] = AjxDebug._BOTTOM_FRAME_ID;
602 		html[i++] = "_pause'>Pause</button></td></tr></table>";
603 		if (doc.body) {
604 			doc.body.innerHTML = html.join("");
605 		}
606 	}
607 	catch (ex) {
608 		// IE chokes on the popup window on cold start-up (when IE is started
609 		// for the first time after system reboot). This should not prevent the
610 		// app from running and should not bother the user
611 	}
612 
613 	if (doc) {
614 		this._clearBtn = doc.getElementById(AjxDebug._BOTTOM_FRAME_ID + "_clear");
615 		this._pauseBtn = doc.getElementById(AjxDebug._BOTTOM_FRAME_ID + "_pause");
616 	}
617 
618 	this._attachHandlers();
619 	this._dbgWindowInited = true;
620 	this._showMessages();
621 };

623 AjxDebug.prototype._attachHandlers =
624 function() {
625 	// Firefox allows us to attach an event listener, and runs it even though
626 	// the window with the code is gone ... odd, but nice. IE, though will not
627 	// run the handler, so we make sure, even if we're  coming back to the
628 	// window, to attach the onunload handler. In general reattach all handlers
629 	// for IE
630 	var unloadHandler = AjxCallback.simpleClosure(this._unloadHandler, this);
631 	if (AjxEnv.isIE) {
632 		this._unloadHandler = unloadHandler;
633 		this._debugWindow.attachEvent('onunload', unloadHandler);
634 	}
635 	else {
636 		this._debugWindow.onunload = unloadHandler;
637 	}
638 
639 	if (this._clearBtn) {
640 		this._clearBtn.onclick = AjxCallback.simpleClosure(this._clear, this);
641 	}
642 	if (this._pauseBtn) {
643 		this._pauseBtn.onclick = AjxCallback.simpleClosure(this._pause, this);
644 	}
645 };

679 /**
680  * This function takes an XML node and returns an HTML string that displays that node
681  * the indent argument is used to describe what depth the node is at so that
682  * the HTML code can create a nice indentation.
683  * 
684  * @private
685  */
686 AjxDebug.prototype._createXmlTree =
687 function (node, indent, omit) {
688 	if (node == null) { return ""; }
689 
690 	var str = "";
691 	var len;
692 	switch (node.nodeType) {
693 		case 1:	// Element
694 			str += "<div style='color: blue; padding-left: 16px;'><<span style='color: DarkRed;'>" + node.nodeName + "</span>";
695 
696 			if (omit && omit[node.nodeName]) {
697 				return str + "/></div>";
698 			}
699 
700 			var attrs = node.attributes;
701 			len = attrs.length;
702 			for (var i = 0; i < len; i++) {
703 				str += this._createXmlAttribute(attrs[i]);
704 			}
705 
706 			if (!node.hasChildNodes()) {
707 				return str + "/></div>";
708 			}
709 			str += "><br />";
710 
711 			var cs = node.childNodes;
712 			len = cs.length;
713 			for (var i = 0; i < len; i++) {
714 				str += this._createXmlTree(cs[i], indent + 3, omit);
715 			}
716 			str += "</<span style='color: DarkRed;'>" + node.nodeName + "</span>></div>";
717 			break;
718 
719 		case 9:	// Document
720 			var cs = node.childNodes;
721 			len = cs.length;
722 			for (var i = 0; i < len; i++) {
723 				str += this._createXmlTree(cs[i], indent, omit);
724 			}
725 			break;
726 
727 		case 3:	// Text
728 			if (!/^\s*$/.test(node.nodeValue)) {
729 				var val = node.nodeValue.replace(/</g, "<").replace(/>/g, ">");
730 				str += "<span style='color: WindowText; padding-left: 16px;'>" + val + "</span><br />";
731 			}
732 			break;
733 
734 		case 7:	// ProcessInstruction
735 			str += "<?" + node.nodeName;
736 
737 			var attrs = node.attributes;
738 			len = attrs.length;
739 			for (var i = 0; i < len; i++) {
740 				str += this._createXmlAttribute(attrs[i]);
741 			}
742 			str+= "?><br />"
743 			break;
744 
745 		case 4:	// CDATA
746 			str = "<div style=''><![CDATA[<span style='color: WindowText; font-family: \"Courier New\"; white-space: pre; display: block; border-left: 1px solid Gray; padding-left: 16px;'>" +
747 				  node.nodeValue +
748 				  "</span>]" + "]></div>";
749 			break;
750 
751 		case 8:	// Comment
752 			str = "<div style='color: blue; padding-left: 16px;'><!--<span style='white-space: pre; font-family: \"Courier New\"; color: Gray; display: block;'>" +
753 				  node.nodeValue +
754 				  "</span>--></div>";
755 			break;
756 
757 		case 10:
758 				str = "<div style='color: blue; padding-left: 16px'><!DOCTYPE " + node.name;
759 				if (node.publicId) {
760 					str += " PUBLIC \"" + node.publicId + "\"";
761 					if (node.systemId)
762 						str += " \"" + node.systemId + "\"";
763 				}
764 				else if (node.systemId) {
765 					str += " SYSTEM \"" + node.systemId + "\"";
766 				}
767 				str += "></div>";
768 
769 				// TODO: Handle custom DOCTYPE declarations (ELEMENT, ATTRIBUTE, ENTITY)
770 				break;
771 
772 		default:
773 			this._inspect(node);
774 	}
775 
776 	return str;
777 };

822 AjxDebug.prototype._showMessagesInWindow =
823 function() {
824 
825 	if (!this._dbgWindowInited) {
826 		// For now, don't show the messages-- assuming that this case only
827 		// happens at startup, and many messages will be written
828 		return;
829 	}
830 	try {
831 		if (this._msgQueue.length > 0) {
832 			var contentFrame = this.getContentFrame();
833 			var linkFrame = this.getLinkFrame();
834 			if (!contentFrame || !linkFrame) { return; }
835 
836 			var contentFrameDoc = contentFrame.contentWindow.document;
837 			var linkFrameDoc = linkFrame.contentWindow.document;
838 			var now = new Date();
839 			for (var i = 0, len = this._msgQueue.length; i < len; ++i ) {
840 				var msg = this._msgQueue[i];
841 				var linkLabel = msg.linkName;
842 				var contentLabel = [msg.message, msg.extraHtml].join("");
843 				this._createLinkNContent("Link", linkLabel, "Content", contentLabel, now);
844 			}
845 		}
846 
847 		this._scrollToBottom();
848 	} catch (ex) {}
849 };

851 AjxDebug.prototype._addMessagesToBuffer =
852 function() {
853 
854 	var eol = (this._target == AjxDebug.TGT_CONSOLE) ? "<br>" : "";
855 	for (var i = 0, len = this._msgQueue.length; i < len; ++i ) {
856 		var msg = this._msgQueue[i];
857 		AjxDebug._addMessageToBuffer(msg.type, msg.message + msg.extraHtml + eol);
858 	}

861 AjxDebug._addMessageToBuffer =
862 function(type, msg) {
863 
864 	type = type || AjxDebug.DEFAULT_TYPE;
865 	var max = AjxDebug.BUFFER_MAX[type];
866 	if (max > 0) {
867 		var buffer = AjxDebug.BUFFER[type] = AjxDebug.BUFFER[type] || [];
868 		while (buffer.length >= max) {
869 			buffer.shift();
870 		}
871 		buffer.push(msg);
872 	}
873 };

898 AjxDebug.prototype._createLinkNContent =
899 function(linkClass, linkLabel, contentClass, contentLabel, now) {
900 
901 	var linkFrame = this.getLinkFrame();
902 	if (!linkFrame) { return; }
903 
904 	now = now || new Date();
905 	var timeStamp = ["[", this._getTimeStamp(now), "]"].join("");
906 	var id = "Lnk_" + now.getTime();
907 
908 	// create link
909 	if (linkLabel) {
910 		var linkFrameDoc = linkFrame.contentWindow.document;
911 		var linkEl = linkFrameDoc.createElement("DIV");
912 		linkEl.className = linkClass;
913 		linkEl.innerHTML = [linkLabel, timeStamp].join(" - ");
914 		linkEl._targetId = id;
915 		linkEl._dbg = this;
916 		linkEl.onclick = AjxDebug._linkClicked;
917 
918 		var linkBody = linkFrameDoc.body;
919 		linkBody.appendChild(linkEl);
920 	}
921 
922 	// create content
923 	var contentFrameDoc = this.getContentFrame().contentWindow.document;
924 	var contentEl = contentFrameDoc.createElement("DIV");
925 	contentEl.className = contentClass;
926 	contentEl.id = id;
927 	contentEl.innerHTML = contentLabel;
928 
929 	contentFrameDoc.body.appendChild(contentEl);
930 
931 	// always show latest
932 	this._scrollToBottom();
933 };

936 function() {
937 	var contentFrame = this._dbg.getContentFrame();

941 		y += el.offsetTop;
942 		el = el.offsetParent;
943 	}
944

946 };

982 /**
983  *
984  * @param {hash}	params			hash of params:
985  * @param {string}	methodNameStr	SOAP method, eg SearchRequest or SearchResponse
986  * @param {boolean}	asyncMode		true if request made asynchronously
987  */
988 AjxDebug.logSoapMessage =
989 function(params) {
990 
991 	if (params.methodNameStr == "NoOpRequest" || params.methodNameStr == "NoOpResponse") { return; }
992 
993 	var ts = AjxDebug._getTimeStamp();
994 	var msg = ["<b>", params.methodNameStr, params.asyncMode ? "" : " (SYNCHRONOUS)" , " - ", ts, "</b>"].join("");
995 	for (var type in AjxDebug.BUFFER) {
996 		if (type == AjxDebug.DEFAULT_TYPE) { continue; }
997 		AjxDebug.println(type, msg);
998 	}
999 	if (window.DBG) {
1000         // Link is written here:
1001         var linkName = params.methodNameStr;
1002         if (!params.asyncMode) {
1003             linkName = "<span class='DebugWarn'>SYNCHRONOUS </span>" + linkName;
1004         }
1005         window.DBG.println(window.DBG._level, msg, linkName);
1006 	}
1007 };
