 22 /**
 23  * Dialog containing the ZmResolveLocationView list.
 24  * @constructor
 25  * @class
 26  *
 27  *  @author Vince Bellows
 28  *
 29  * @param controller		[ZmApptComposeController]	the appt compose controller
 30  * @param composeView		[ZmApptEditView]	        the appt edit view
 31  * @param okCallback		[function]	                callback upon OK
 32  * @param assistantView		[ZmScheduleAssistantView]	Assistant that provides location FB info
 33  *
 34  */
 35 
 36 ZmResolveLocationConflictDialog = function(controller, composeView, okCallback, assistantView) {
 37     this._controller = controller;
 38     this._composeView = composeView;
 39 
 40     this._okCallback = okCallback;
 41     // The location assistant view associated with the parent appt view will
 42     // provide the set of viable locations for each conflict date
 43     this._assistantView = assistantView;
 44 
 45 	DwtDialog.call(this, {parent:appCtxt.getShell(),
 46         title:ZmMsg.resolveLocationConflicts});
 47 
 48 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._handleOkButton));
 49 	this.setButtonListener(DwtDialog.CANCEL_BUTTON, new AjxListener(this, this._handleCancelButton));
 50 };
 51 
 52 ZmResolveLocationConflictDialog.prototype = new DwtDialog;
 53 ZmResolveLocationConflictDialog.prototype.constructor = ZmResolveLocationConflictDialog;
 54 
 55 
 56 ZmResolveLocationConflictDialog.prototype.toString =
 57 function() {
 58 	return "ZmResolveLocationConflictDialog";
 59 };

 61 // For each conflict dateTime, find the set of available locations.  These will be used to
 62 // populate the alternate location pulldowns
 63 ZmResolveLocationConflictDialog.prototype._determineAlternateLocations =
 64 function(params) {
 65     var addr;
 66     var durationInfo;
 67     var locInfo;
 68     var resolveLocInfo;
 69 
 70     this._resolveLocList = new AjxVector();
 71 
 72     for (var i = 0; i < this._inst.length; i++) {
 73         if (this._inst[i].usr) {
 74             // A Conflict exists - add an entry to the resolution list
 75             resolveLocInfo = {};
 76             resolveLocInfo.originalLocation = ZmResolveLocationView.NO_SELECTION;
 77             resolveLocInfo.enabled = true;
 78             var exceptionLoc = this._locationExceptions[this._inst[i].s];
 79             if (exceptionLoc) {
 80                 // Stored value (from either an existing exception in the DB or a
 81                 // previous use of this dialog).  Save it to use for indicating the
 82                 // selected location in the alternate locations pulldown.
 83                 var locationEmails = [];
 84                 for (var j = 0; j < exceptionLoc.length; j++) {
 85                     locationEmails.push(exceptionLoc[j].getEmail());
 86                 }
 87                 if (locationEmails.length > 1) {
 88                     // Multi select not currently supported.  If an exception was created
 89                     // outside this dialog specifying multiple locations, just display it
 90                     // (i.e. do not create a pulldown, just create a label)
 91                     resolveLocInfo.originalLocation  = locationEmails.join(',');
 92                     resolveLocInfo.enabled = false;
 93                 } else if (locationEmails.length == 1) {
 94                     resolveLocInfo.originalLocation = locationEmails[0];
 95                 }
 96              }
 97 
 98             durationInfo = {};
 99             durationInfo.startTime = this._inst[i].s;
100             durationInfo.endTime   = this._inst[i].s + this._inst[i].dur;
101             durationInfo.duration  = this._inst[i].dur;
102 
103             params.duration = durationInfo
104             resolveLocInfo.inst = this._inst[i];
105 
106             // Get locations that are available for this conflict's startTime to endTime
107             locInfo = this._assistantView.computeLocationAvailability(durationInfo, params);
108             resolveLocInfo.alternateLocationInfo = locInfo.locations;
109             // Add to the list for display by the ZmResolveLocationView
110             this._resolveLocList.add(resolveLocInfo);
111         }
112     }
113     params.list = this._resolveLocList;
114     this._resolveLocationView.set(params);
115 
116 };
117 
118 ZmResolveLocationConflictDialog.prototype._handleOkButton =
119 function(event) {
120     var alteredLocations   = {};
121     var locationExceptions = {};
122     var location;
123     var newLocation;
124     var resolveInfo;
125 
126     for (var i = 0; i < this._resolveLocList.size(); i++) {
127         resolveInfo = this._resolveLocList.get(i);
128         if (resolveInfo.enabled) {
129             // Get the selected alternate (if any)
130             newLocation = this._resolveLocationView.getAlternateLocation(i);
131         } else {
132             // Pulldown was not used due to multiple locations already set for entry
133             newLocation = resolveInfo.originalLocation;
134         }
135         if (newLocation != ZmResolveLocationView.NO_SELECTION) {
136             location = this._composeView.getAttendeesFromString(ZmCalBaseItem.LOCATION, newLocation, false);
137             if (location) {
138                 locationExceptions[resolveInfo.inst.s] = location.getArray();
139             }
140             if (newLocation != resolveInfo.originalLocation) {
141                 // Location changed - pass to caller to apply upon Save
142                 alteredLocations[resolveInfo.inst.s] = locationExceptions[resolveInfo.inst.s];
143             }
144         }
145 
146     }
147 
148     this.popdown();
149     if(this._okCallback) this._okCallback.run(locationExceptions, alteredLocations);
150 
151 };
152 
153 ZmResolveLocationConflictDialog.prototype._handleCancelButton =
154 function(event) {
155 	this.popdown();
156 };
157 
158 ZmResolveLocationConflictDialog.prototype.popup =
159 function(appt, inst, locationExceptions) {
160     this._appt = appt;
161     this._inst = inst ? inst : [];
162 
163     // Existing set of location exceptions - either persisted to the DB, or specified
164     // by a previous use of this dialog.
165     this._locationExceptions = locationExceptions ? locationExceptions : {};
166 
167     DwtDialog.prototype.popup.call(this);
168     this._resolveLocationView.setLoadingHtml();
169 
170     // Use the assistantView to get sets of locations for each conflict date.
171     // These will be used to populate the alternative location dropdown.
172     var fbEndTime = 0;
173     if (this._inst.length > 0) {
174         var inst = this._inst[this._inst.length-1];
175         fbEndTime = inst.s + inst.dur;
176         this._assistantView.getLocationFBInfo(
177             this._determineAlternateLocations, this, fbEndTime);
178     }
179 };
180 
181 ZmResolveLocationConflictDialog.prototype.setContent =
182 function(text) {
183 	var contentDiv = this._getContentDiv();
184 
185     this._resolveLocationView = new ZmResolveLocationView(
186         this, this._controller, this._apptView);
187     this._resolveLocationView.reparentHtmlElement(contentDiv);
188 
189 
190 };
191 
192 ZmResolveLocationConflictDialog.prototype.cleanup =
193 function() {
194     if(this._resolveLocationView) this._resolveLocationView.removeAll();
195 };
