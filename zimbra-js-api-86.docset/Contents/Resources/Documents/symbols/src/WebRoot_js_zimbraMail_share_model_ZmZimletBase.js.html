 68 /**
 69  * @private
 70  */
 71 ZmZimletBase.prototype._init =
 72 function(zimletContext, shell) {
 73 	DBG.println(AjxDebug.ZIMLET, "Creating zimlet " + zimletContext.name);
 74 	this._passRpcErrors = false;
 75 	this._zimletContext = zimletContext;
 76 	this._dwtShell = shell;
 77 	this._origIcon = this.xmlObj().icon;
 78 	this.__zimletEnabled = true;
 79 	this.name = this.xmlObj().name;
 80 
 81 	var contentObj = this.xmlObj("contentObject");
 82 	if (contentObj && contentObj.matchOn) {
 83 		var regExInfo = contentObj.matchOn.regex;
 84 		if(!regExInfo.attrs) {regExInfo.attrs = "ig";}
 85 		this.RE = new RegExp(regExInfo._content, regExInfo.attrs);
 86 		if (contentObj.type) {
 87 			this.type = contentObj.type;
 88 		}
 89 		ZmObjectHandler.prototype.init.call(this, this.type, contentObj["class"]);
 90 	}
 91 };

202 /**
203  * @private
204  */
205 ZmZimletBase.prototype._dispatch =
206 function(handlerName) {
207 	var params = [];
208 	var obj;
209 	var url;
210 	for (var i = 1; i < arguments.length; ++i) {
211 		params[i-1] = arguments[i];
212 	}
213 	// create a canvas if so was specified
214 	var canvas;
215 	switch (handlerName) {
216 	    case "singleClicked":
217 	    case "doubleClicked":
218 		// the panel item was clicked
219 		obj = this.xmlObj("zimletPanelItem")
220 			[handlerName == "singleClicked" ? "onClick" : "onDoubleClick"];
221 		if (!obj) {
222 			break;
223 		}
224 		url = obj.actionUrl;
225 		if (url) {
226 			url = this.xmlObj().makeURL(url);
227 		}
228 		if (obj && (obj.canvas || url)) {
229 			canvas = this.makeCanvas(obj.canvas, url);
230 		}
231 		break;
232 
233 	    case "doDrop":
234 		obj = params[1]; // the dragSrc that matched
235 		if (!obj)
236 			break;
237 		if (obj.canvas) {
238 			canvas = obj.canvas[0];
239 		}
240 		url = obj.actionUrl;
241 		if (url && canvas) {
242 			// params[0] is the dropped object
243 			url = this.xmlObj().makeURL(url[0], params[0]);
244 			canvas = this.makeCanvas(canvas, url);
245 			return "";
246 		}
247 		break;
248 	}
249 	if (canvas) {
250 		params.push(canvas);
251 	}
252 	return this.xmlObj().callHandler(handlerName, params);
253 };

475 /**
476  * This method is called when content (e.g. a mail message) is being parsed.
477  * The match method may be called multiple times for a given piece of content and
478  * should apply the pattern matching as defined for a given zimlet <code><regex></code>.
479  * Zimlets should also use the "g" option when constructing their <code><regex></code>.
480  *
481  * <p>
482  * The return should be an array in the form:
483  *  
484  * <pre>
485  * result[0...n] // should be matched string(s)
486  * result.index // should be location within line where match occurred
487  * result.input // should be the input parameter content
488  * </pre>
489  * </p>
490  * 
491  * @param	{string}	content		the content line to perform a match against
492  * @param	{number}	startIndex	the start index (i.e. where to begin the search)
493  * @return	{array}	the matching content object from the <code>startIndex</code> if the content matched the specified zimlet handler regular expression; otherwise <code>null</code>
494  */
495 ZmZimletBase.prototype.match =
496 function(content, startIndex) {
497 	if(!this.RE) {return null;}
498 	this.RE.lastIndex = startIndex;
499 	var ret = this.RE.exec(content);
500 	if (ret) {
501 		ret.context = ret;
502 	}
503 	return ret;
504 };

525 /**
526  * This method is called when the tool tip is popping-up.
527  *
528  * @param	{Object}	spanElement		the enclosing span element
529  * @param	{string}	contentObjText	the content object text
530  * @param	{array}		matchContent	the matched content
531  * @param	{Object}	canvas			the canvas
532  */
533 ZmZimletBase.prototype.toolTipPoppedUp =
534 function(spanElement, contentObjText, matchContext, canvas) {
535 	var c = this.xmlObj("contentObject");
536 	if (c && c.toolTip) {
537 		var obj = this._createContentObj(contentObjText, matchContext);
538 
539 		var txt;
540 		if (c.toolTip instanceof Object && c.toolTip.actionUrl) {
541 		    this.xmlObj().handleActionUrl(c.toolTip.actionUrl, [{type:"tooltip"}], obj, canvas);
542 		    // XXX the tooltip needs "some" text on it initially, otherwise it wouldn't resize afterwards.
543 		    txt = ZmMsg.zimletFetchingTooltipData;
544 		} else {
545 			// If it's an email address just use the address value.
546 			if (obj.objectContent instanceof AjxEmailAddress) {obj.objectContent = obj.objectContent.address;}
547 			txt = this.xmlObj().process(c.toolTip, obj);
548 		}
549 		canvas.innerHTML = txt;
550 
551 		Dwt.setSize(canvas, parseInt(c.toolTip.width) || Dwt.DEFAULT, parseInt(c.toolTip.height) || Dwt.DEFAULT);
552 		
553 		if (this._isTooltipSticky()) {
554 			Dwt.setHandler(canvas, DwtEvent.ONCLICK, AjxCallback.simpleClosure(this.setTooltipSticky, this, [true]));
555 
556 			var omem = DwtOutsideMouseEventMgr.INSTANCE;
557 			omem.startListening({
558 				id: "ZimletTooltip",
559 				elementId: canvas.id,
560 				outsideListener: new AjxListener(this, this.setTooltipSticky, [false, true])
561 			});
562 
563 		}
564 	}
565 };

641  * This method creates the property editor for the set of <code><property></code> elements defined
642  * in the <code><userProperties></code> element. The default implementation of this
643  * method will auto-create a property editor based on the attributes of the user properties.
644  * </p>
645  * <p>
646  * Override this method if a custom property editor is required.
647  * </p>
648  * 
649  * @param	{AjxCallback}	callback	the callback method for saving user properties
650  */
651 ZmZimletBase.prototype.createPropertyEditor =
652 function(callback) {
653 	var userprop = this.xmlObj().userProperties;
654 
655 	if (!userprop || !userprop.length) {return;}
656 
657     for (var i = 0; i < userprop.length; ++i) {
658         userprop[i].label = this._zimletContext.processMessage(userprop[i].label);
659         if (userprop[i].type == "enum") {
660         	var items = userprop[i].item;
661         	for (var j=0; items != null && j < items.length; j++) {
662         		if (items[j] == null)
663         			continue;
664         		var item = items[j];
665         		item.label = this._zimletContext.processMessage(item.label);
666         	}
667         }
668 	}
669 
670 	if (!this._dlg_propertyEditor) {
671 		var view = new DwtComposite(this.getShell());
672 		var pe = this._propertyEditor = new DwtPropertyEditor(view, true);
673 		pe.initProperties(userprop);
674 		var dialog_args = {
675 			title : this._zimletContext.processMessage(this.xmlObj("description")) + " preferences",
676 			view  : view
677 		};

679 		pe.setFixedLabelWidth();
680 		pe.setFixedFieldWidth();
681 		dlg.setButtonListener(DwtDialog.OK_BUTTON,
682 				      new AjxListener(this, function() {
683 					      this.saveUserProperties(callback);
684 				      }));
685 	}

931 /**
932  * Saves the user properties.
933  * 
934  * @param	{AjxCallback}	callback		the callback to invoke after the save
935  * @return	{string}		an empty string or an error message
936  */
937 ZmZimletBase.prototype.saveUserProperties =
938 function(callback) {
939 	var soapDoc = AjxSoapDoc.create("ModifyPropertiesRequest", "urn:zimbraAccount");
940 
941 	var props = this.xmlObj().userProperties;
942 	var check = this.checkProperties(props);
943 
944 	if (!check)
945 		return "";
946 	if (typeof check == "string")
947 		return this.displayErrorMessage(check);
948 
949 	if (this._propertyEditor)
950 		if (!this._propertyEditor.validateData())
951 			return "";
952 
953 	// note that DwtPropertyEditor actually works on the original
954 	// properties object, which means that we already have the edited data
955 	// in the xmlObj :-) However, the props. dialog will be dismissed if
956 	// present.
957 	for (var i = 0; i < props.length; ++i) {
958 		var p = soapDoc.set("prop", props[i].value);
959 		p.setAttribute("zimlet", this.xmlObj("name"));
960 		p.setAttribute("name", props[i].name);
961 	}
962 
963 	var ajxcallback = null;
964 	if (callback)
965 		ajxcallback = new AjxCallback(this, function(result) {
966 			// TODO: handle errors
967 			callback.run();
968 		});
969 	var params = { soapDoc: soapDoc, callback: ajxcallback, asyncMode: true, sensitive:true};
970 	appCtxt.getAppController().sendRequest(params);
971 	if (this._dlg_propertyEditor) {
972 		this._dlg_propertyEditor.popdown();
973 		// force the dialog to be reconstructed next time
974 		this._dlg_propertyEditor.dispose();
975 		this._propertyEditor = null;
976 		this._dlg_propertyEditor = null;
977 	}
978 	return "";
979 };

1121 ZmZimletBase.prototype.makeCanvas =
1122 function(canvasData, url, x, y) {
1123 	if(canvasData && canvasData.length)
1124         canvasData = canvasData[0];    
1125     var canvas = null;
1126 	var div;
1127 
1128 	div = document.createElement("div");
1129 	div.id = "zimletCanvasDiv";
1130 
1131 	// HACK #1: if an actionUrl was specified and there's no <canvas>, we

1133 	if (!canvasData && url)
1134 		canvasData = { type: "window" };
1135 
1136 	// HACK #2: some folks insist on using "style" instead of "type". ;-)
1137 	if (canvasData.style && !canvasData.type)
1138 		canvasData.type = canvasData.style;
1139 
1140 	switch (canvasData.type) {
1141 	    case "window":
1142 		var browserUrl = url;
1143 		if (browserUrl == null)
1144 			browserUrl = appContextPath+"/public/blank.html";
1145 		var contentObject = this.xmlObj("contentObject");
1146         if(contentObject && !canvasData.width && contentObject.onClick ) {
1147             if(contentObject.onClick.canvas.props == "")
1148                 canvas = window.open(browserUrl);
1149             else if(contentObject.onClick.canvas.props != "")
1150                 canvas = window.open(browserUrl, this.xmlObj("name"), contentObject.onClick.canvas.props);
1151         }
1152         else{
1153             var props = canvasData.props ? [ canvasData.props ] : [ "toolbar=yes,location=yes,status=yes,menubar=yes,scrollbars=yes,resizable=yes"];
1154             if (canvasData.width)
1155                 props.push("width=" + canvasData.width);
1156             if (canvasData.height)
1157                 props.push("height=" + canvasData.height);
1158             props = props.join(",");
1159             canvas = window.open(browserUrl, this.xmlObj("name"), props);
1160         }
1161         if (!url) {
1162 			// TODO: add div element in the window.
1163 			//canvas.document.getHtmlElement().appendChild(div);
1164 		}
1165 		break;
1166 
1167 	    case "dialog":
1168 		var view = new DwtComposite(this.getShell());
1169 		if (canvasData.width)
1170 			view.setSize(canvasData.width, Dwt.DEFAULT);
1171 		if (canvasData.height)
1172 			view.setSize(Dwt.DEFAULT, canvasData.height);

1174 		title = this._zimletContext.processMessage(title);

1177 		if (url) {

1190 			canvas.iframe = el;
1191 		} else {
1192 			view.getHtmlElement().appendChild(div);

1196 
1197         case "tooltip":

1363 ZmZimletBase.prototype.getMsgsForConv =
1364 function(callback, convObj){
1365
