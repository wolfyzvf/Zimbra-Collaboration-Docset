 22 /**
 23  * @class
 24  * Simple manager for "idle" events. Add a handler like this:
 25  *
 26  * <pre>
 27  *    var idleTimer = new DwtIdleTimer(10000, new AjxCallback(obj, obj.handler));
 28  *
 29  *    obj.handler = function(idle) {
 30  *       if (idle) {
 31  *          // do idle stuff here
 32  *       } else {
 33  *          // user is back
 34  *       }
 35  *    }
 36  * </pre>
 37  * 
 38  * With this code, when the user is idle for 10 seconds obj.handler(true) will
 39  * be called.  When the user gets back from idle, obj.handler(false) will be
 40  * called and the timer restarted.
 41  * </p>
 42  * <p>
 43  * To cancel a timer, call <code>idleTimer.kill()</code>. To restart it later, you can
 44  * <code>idleTimer.resurrect(timeout)</code>. The timeout parameter is optional, pass it only if you
 45  * want to modify it.
 46  * </p>
 47  * <p>
 48  * You can create multiple handlers, each with its own callback and timeout.  A
 49  * new {@link DwtIdleTimer} will start running right away and will continue to do so
 50  * until you <code>kill()</code> it.
 51  * </p>
 52  * 
 53  * @param	{number}	[timeout]		the timeout 
 54  * @param	{AjxCallback}	handler		the callback
 55  * 
 56  * @private
 57  */
 58 DwtIdleTimer = function(timeout, handler) {
 59 	DwtIdleTimer._initEvents();
 60 	this.timeout = timeout;
 61 	this.handler = handler;
 62 	this.idle = false;
 63 	this._onIdle = AjxCallback.simpleClosure(this.setIdle, this);
 64 	this._startTimer();
 65 	DwtIdleTimer.getHandlers().add(this);
 66 };
 67 
 68 DwtIdleTimer.idleHandlers = 0;
 69 
 70 DwtIdleTimer.prototype.toString =
 71 function() {
 72 	return "DwtIdleTimer";
 73 };
 74 
 75 DwtIdleTimer.prototype.kill =
 76 function() {
 77 	this._stopTimer();
 78 	this.idle = false;
 79 	DwtIdleTimer.getHandlers().remove(this);
 80 };
 81 
 82 DwtIdleTimer.prototype.resurrect =
 83 function(timeout) {
 84 	this.idle = false; // make sure we start "unidle"
 85 	DwtIdleTimer.getHandlers().add(this, null, true);
 86 	if (timeout != null) {
 87 		this.timeout = timeout;
 88 	}
 89 	this._startTimer();
 90 };
 91 
 92 DwtIdleTimer.prototype.setIdle =
 93 function() {
 94 	if (!this.idle) {
 95 		DwtIdleTimer.idleHandlers++;
 96 		this.idle = true;
 97 		this.handler.run(true);
 98 	}
 99 };
100 
101 DwtIdleTimer.prototype.resume =
102 function() {
103 	if (this.idle) {
104 		this.idle = false;
105 		this.handler.run(false);
106 		DwtIdleTimer.idleHandlers--;
107 	}
108 };
109 
110 DwtIdleTimer.prototype._startTimer =
111 function() {
112 	this._stopTimer();
113 	this._timer = setTimeout(this._onIdle, this.timeout);
114 };
115 
116 DwtIdleTimer.prototype._stopTimer =
117 function() {
118 	if (this._timer) {
119 		clearTimeout(this._timer);
120 		this._timer = null;
121 	}
122 };
123 
124 DwtIdleTimer._initEvents =
125 function() {
126 	// execute only once per session
127 	if (!DwtIdleTimer._initialized) {
128 		if (!AjxEnv.isIE) {
129 			window.addEventListener("keydown", DwtIdleTimer.resetIdle, true);
130 			window.addEventListener("mousemove", DwtIdleTimer.resetIdle, true);
131 			window.addEventListener("mousedown", DwtIdleTimer.resetIdle, true);
132 			window.addEventListener("focus", DwtIdleTimer.resetIdle, true);
133 		} else {
134 			document.body.attachEvent("onkeydown", DwtIdleTimer.resetIdle);
135 			document.body.attachEvent("onkeyup", DwtIdleTimer.resetIdle);
136 			document.body.attachEvent("onmousedown", DwtIdleTimer.resetIdle);
137 			document.body.attachEvent("onmousemove", DwtIdleTimer.resetIdle);
138 			document.body.attachEvent("onmouseover", DwtIdleTimer.resetIdle);
139 			document.body.attachEvent("onmouseout", DwtIdleTimer.resetIdle);
140 			window.attachEvent("onfocus", DwtIdleTimer.resetIdle);
141 		}
142 		DwtIdleTimer._initialized = true;
143 	}
144 };
145 
146 DwtIdleTimer.getHandlers =
147 function() {
148 	var a = DwtIdleTimer.HANDLERS;
149 	if (!a) {
150 		a = DwtIdleTimer.HANDLERS = new AjxVector();
151 	}
152 	return a;
153 };
154 
155 DwtIdleTimer.resetIdle =
156 function() {
157 	var a = DwtIdleTimer.getHandlers();
158 	a.foreach("_startTimer"); // we need to restart timers anyway...
159 	if (DwtIdleTimer.idleHandlers > 0) {
160 		a.foreach("resume");
161 	}
162 };
