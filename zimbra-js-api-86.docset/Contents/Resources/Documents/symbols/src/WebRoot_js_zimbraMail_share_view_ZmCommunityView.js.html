 22 /**
 23  * Sets up an iframe which displays content from Community (activity stream, notifications, chat).
 24  * @class
 25  * This class displays Community content in an iframe.
 26  *
 27  * @extends		ZmAppIframeView
 28  *
 29  * @author Conrad Damon
 30  */
 31 ZmCommunityView = function(params) {
 32 	ZmAppIframeView.apply(this, arguments);
 33 	this._createFrame(params);
 34 	this._setupMessageHandling();
 35 };
 36 
 37 ZmCommunityView.prototype = new ZmAppIframeView;
 38 ZmCommunityView.prototype.constructor = ZmCommunityView;
 39 
 40 ZmCommunityView.prototype.isZmCommunityView = true;
 41 ZmCommunityView.prototype.toString = function() { return "ZmCommunityView"; };
 42 
 43 ZmCommunityView.prototype._setupMessageHandling = function(params) {
 44 
 45 	// Set up to handle messages sent to us via postMessage()
 46 	var iframe = document.getElementById(this._iframeId);
 47 	if (iframe) {
 48 		var callback = ZmCommunityView.handleMessage.bind(null, this);
 49 		if (window.addEventListener) {
 50 			window.addEventListener('message', callback, false);
 51 		}
 52 		else if (window.attachEvent) {
 53 			window.attachEvent('onmessage', callback);
 54 		}
 55 	}
 56 };
 57 
 58 ZmCommunityView.prototype._getIframeId = function() {
 59 	return 'fragment-41812_iframe';     // this is what Community is expecting
 60 };
 61 
 62 /**
 63  * If Community tells us there is new content, turn the tab orange if it's not the current tab,
 64  * and refresh the content.
 65  *
 66  * @param view
 67  * @param event
 68  */
 69 ZmCommunityView.handleMessage = function(view, event) {
 70 
 71 	var iframe = document.getElementById(view._iframeId);
 72 	if (iframe && event.source === iframe.contentWindow) {
 73 		var data = AjxStringUtil.parseQueryString(event.data || '');
 74 		var isUnread = (data.unread && data.unread.toLowerCase() === 'true');
 75 		if (data.type === 'community-notification' && isUnread) {
 76 			appCtxt.getApp(view._appName).startAlert();
 77 			view.getUpdates();
 78 		}
 79 	}
 80 };
 81 
 82 /**
 83  * Sends a message to Community to refresh the content.
 84  */
 85 ZmCommunityView.prototype.getUpdates = function() {
 86 
 87 	var iframe = document.getElementById(this._iframeId)
 88 	if (iframe) {
 89 		iframe.contentWindow.postMessage('type=community-update', '*');
 90 	}
 91 };
 92 
 93 // Called when user switches to this tab.
 94 ZmCommunityView.prototype.activate = function(active) {
 95 	if (active) {
 96 		this.getUpdates();
 97 	}
 98 };
 99 
100 // Code to run when the user clicks the refresh (circle-arrow) button. Shouldn't really be
101 // needed, but doesn't hurt to have it.
102 ZmCommunityView.prototype.runRefresh = function() {
103 	this.getUpdates();
104 };
