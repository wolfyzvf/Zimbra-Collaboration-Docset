 28 /**
 29  * Creates a new task controller to manage task creation/editing.
 30  * @class
 31  * This class manages task creation/editing.
 32  *
 33  * @author Parag Shah
 34  *
 35  * @param {DwtShell}	container	the containing shell
 36  * @param {ZmApp}		app			the containing app
 37  * @param {constant}	type		controller type
 38  * @param {string}		sessionId	the session id
 39  * 
 40  * @extends		ZmCalItemComposeController
 41  */
 42 ZmTaskController = function(container, app, type, sessionId) {
 43 	if (arguments.length == 0) { return; }
 44 	ZmCalItemComposeController.apply(this, arguments);
 45 };

 64 ZmTaskController.prototype.saveCalItem =
 65 function(attId) {
 66 	var calItem = this._composeView.getCalItem(attId);
 67 	if (calItem) {
 68 		this._saveCalItemFoRealz(calItem, attId);
 69 		return true;
 70 	}
 71 	return false;
 72 };

 74 ZmTaskController.prototype.isCloseAction =
 75 function() {
 76     return this._action == ZmCalItemComposeController.SAVE;
 77 };
 78 
 79 ZmTaskController.prototype._handleResponseSave =
 80 function(calItem, result) {
 81 	ZmCalItemComposeController.prototype._handleResponseSave.call(this, calItem);
 82 	if(this._action == ZmCalItemComposeController.SAVE) {
 83 		this.closeView();	
 84 	}
 85 	// XXX: null out message so we re-fetch task next time its opened
 86 	// To optimize, we should save the modified contents into cache'd task item
 87 	if (calItem && calItem._orig)
 88 		calItem._orig.message = null;
 89 
 90     //Cache the item for further processing
 91     calItem.cache();
 92     //need to set rev,ms for next soap request
 93     calItem.setFromSavedResponse(result);
 94     
 95 	appCtxt.setStatusMsg(ZmMsg.taskSaved);
 96     if(calItem.alarm == true || calItem.isAlarmModified()) {
 97         this._app.getReminderController().refresh();
 98     }    
 99 };

111 ZmTaskController.prototype._getButtonOverrides =
112 function(buttons) {
113 
114 	if (!(buttons && buttons.length)) { return; }
115 
116 	var overrides = {};
117 	var idParams = {
118 		skinComponent:  ZmId.SKIN_APP_TOP_TOOLBAR,
119 		componentType:  ZmId.WIDGET_BUTTON,
120 		app:            ZmId.APP_TASKS,
121 		containingView: ZmId.VIEW_TASKEDIT
122 	};
123 	for (var i = 0; i < buttons.length; i++) {
124 		var buttonId = buttons[i];
125 		overrides[buttonId] = {};
126 		idParams.componentName = buttonId;
127 		var item = (buttonId === ZmOperation.SEP) ? "Separator" : buttonId + " button";
128 		var description = item + " on top toolbar for task edit view";
129 		overrides[buttonId].domId = ZmId.create(idParams, description);
130 	}
131 	return overrides;
132 };

134 ZmTaskController.prototype._setComposeTabGroup =
135 function(setFocus) {
136 	var tg = this._createTabGroup();
137 	var rootTg = appCtxt.getRootTabGroup();
138 	tg.newParent(rootTg);
139 	tg.addMember(this._toolbar);
140 	this._composeView._addTabGroupMembers(tg);
141 
142 	var focusItem = this._composeView || this._composeView._getDefaultFocusItem() || tg.getFirstMember(true);
143 	var ta = new AjxTimedAction(this, this._setFocus, [focusItem, !setFocus]);
144 	AjxTimedAction.scheduleAction(ta, 10);
145 };

152 /**
153  * Checks if the tasks is moving from local to remote folder (or vice versa).
154  * 
155  * @param	{ZmTask}	task			the task
156  * @param	{String}	newFolderId		the folder id
157  * @return	{Boolean}	<code>true</code> if moving from local to remote folder
158  */
159 ZmTaskController.prototype.isMovingBetwAccounts =
160 function(task, newFolderId) {
161 	var isMovingBetw = false;
162 	if (task._orig) {
163 		var origFolder = task._orig.getFolder();
164 		var newFolder = appCtxt.getById(newFolderId);
165 		if (origFolder && newFolder) {
166 			if ((origFolder.id != newFolderId) &&
167 				((origFolder.link && !newFolder.link) || (!origFolder.link && newFolder.link)))
168 			{
169 				isMovingBetw = true;
170 			}
171 		}
172 	}
173 	return isMovingBetw;
174 };

186 ZmTaskController.prototype._printListener =
187 function() {
188 	var url = ["/h/printtasks?id=", this._composeView._calItem.invId];
189     
190     if (appCtxt.isOffline) {
191         var acctName = this._composeView._calItem.getAccount().name;
192         url.push("&acct=", acctName);
193     }
194 	window.open([appContextPath, url.join(""), "&tz=", AjxTimezone.getServerId(AjxTimezone.DEFAULT)].join(""), "_blank");
195 };
