 22 /**
 23  * 
 24  * @private
 25  */
 26 DwtGrouper = function(parent, className, posStyle) {
 27 	if (arguments.length == 0) return;
 28 	className = className || "DwtGrouper";
 29 	posStyle = posStyle || DwtControl.STATIC_STYLE;
 30 	DwtComposite.call(this, {parent:parent, posStyle:posStyle});
 31 
 32 	this._labelEl = document.createElement("LEGEND");
 33 	this._labelEl.id = Dwt.getNextId();
 34 	this._insetEl = document.createElement("DIV");
 35 	this._borderEl = document.createElement("FIELDSET");
 36 	this._borderEl.appendChild(this._labelEl);
 37 	this._borderEl.appendChild(this._insetEl);
 38 
 39 	this._tabGroup = new DwtTabGroup(this.toString());
 40 
 41 	var element = this.getHtmlElement();
 42 	element.appendChild(this._borderEl);
 43 }
 44 
 45 DwtGrouper.prototype = new DwtComposite;
 46 DwtGrouper.prototype.constructor = DwtGrouper;
 47 
 48 // Public methods
 49 
 50 DwtGrouper.prototype.setLabel = function(htmlContent) {
 51 	Dwt.setVisible(this._labelEl, Boolean(htmlContent));
 52 	// HACK: undo block display set by Dwt.setVisible
 53 	this._labelEl.style.display = "";
 54 	this._labelEl.innerHTML = htmlContent ? htmlContent : "";
 55 };
 56 
 57 DwtGrouper.prototype.setContent = function(htmlContent) {
 58 	var element = this._insetEl;
 59 	element.innerHTML = htmlContent;
 60 	var inputElements = element.getElementsByTagName('input');
 61 	for (var i=0; i < inputElements.length; i++) {
 62 		this._tabGroup.addMember(inputElements[i]);
 63 	}
 64 };
 65 
 66 DwtGrouper.prototype.setElement = function(htmlElement) {
 67 	var element = this._insetEl;
 68 	Dwt.removeChildren(element);
 69 	element.appendChild(htmlElement);
 70 };
 71 
 72 DwtGrouper.prototype.setView = function(control) {
 73 	this.setElement(control.getHtmlElement());
 74 };
 75 
 76 DwtGrouper.prototype.getInsetHtmlElement = function() {
 77 	return this._insetEl;
 78 };
 79 
 80 DwtGrouper.prototype.getTabGroupMember = function(){
 81 	return this._tabGroup;
 82 };
