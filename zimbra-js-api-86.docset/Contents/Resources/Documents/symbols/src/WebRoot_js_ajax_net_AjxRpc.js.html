 22 /**
 23  * @constructor
 24  * @class
 25  * This static class provides an interface for send requests to a server. It
 26  * essentially wraps {@link AjxRpcRequest}. This {@link AjxRpc} link maintains a cache of
 27  * {@link AjxRpcRequest} objects which it attempts to reuse before allocating additional
 28  * objects. It also has a mechanism whereby if an {@link AjxRpcRequest} object is 
 29  * in a "busy" state for a extended period of time, it will reap it appropriately.
 30  *
 31  * @author Ross Dargahi
 32  * @author Conrad Damon
 33  * 
 34  * @see AjxRpcRequest
 35  */
 36 AjxRpc = function() {
 37 };
 38 
 39 AjxRpc.__rpcCache		= [];		// The pool of RPC contexts available
 40 AjxRpc.__rpcOutstanding	= {};		// The pool of RPC contexts in use
 41 
 42 AjxRpc.__RPC_CACHE_MAX		= 50;		// maximum number of busy contexts we can have
 43 AjxRpc.__RPC_ID				= 0;		// used for context IDs
 44 AjxRpc.__RPC_IN_USE			= 0;		// number of contexts that are busy
 45 AjxRpc.__RPC_HIGH_WATER		= 0;		// high water mark for busy contexts
 46 AjxRpc.__RPC_REAP_AGE		= 300000;	// 5 minutes; mark any context older than this (in ms) as free
 47 AjxRpc.__RPC_REAP_INTERVAL	= 1800000;	// 30 minutes; run the reaper this often
 48 
 49 /**
 50  * Submits a request to a URL. The request is handled through a pool of request
 51  * contexts (each a wrapped XmlHttpRequest). The context does the real work.
 52  *
 53  * @param {string} [requestStr] 	the HTTP request string/document
 54  * @param {string} serverUrl 	the request target 
 55  * @param {array} [requestHeaders] an array of HTTP request headers
 56  * @param {AjxCallback} callback 	the callback for asynchronous requests. This callback 
 57  * 		will be invoked when the requests completes. It will be passed the same
 58  * 		values as when this method is invoked synchronously (see the return values
 59  * 		below) with the exception that if the call times out (see timeout param 
 60  * 		below), then the object passed to the callback will be the same as in the 
 61  * 		error case with the exception that the status will be set to 
 62  * 		{@link AjxRpcRequest.TIMEDOUT}.
 63  * @param {Constant} [method] 		the HTTP method -- GET, POST, PUT, DELETE. if <code>true</code>, use get method for backward compatibility
 64  * @param {number} [timeout] 		the timeout (in milliseconds) after which the request is canceled
 65  * 
 66  * @return {object|hash}	if invoking in asynchronous mode, then it will return the id of the 
 67  * 		underlying {@link AjxRpcRequest} object. Else if invoked synchronously, if
 68  * 		there is no error (i.e. we get a HTTP result code of 200 from the server),
 69  * 		an object with the following attributes is returned
 70  * 		<ul>
 71  * 		<li>text - the string response text</li>
 72  * 		<li>xml - the string response xml</li>
 73  * 		<li>success - boolean set to true</li>
 74  * 		</ul>
 75  * 		If there is an error, then the following will be returned
 76  * 		<ul>
 77  * 		<li>text - the string response text<li>
 78  * 		<li>xml - the string response xml </li>
 79  * 		<li>success - boolean set to <code>false</code></li>
 80  * 		<li>status - HTTP status</li>
 81  * 		</ul>
 82  * 
 83  * @throws	{AjxException.NETWORK_ERROR}	a network error occurs
 84  * @throws	{AjxException.UNKNOWN_ERROR}	an unknown error occurs
 85  * 
 86  * @see	AjxRpcRequest#invoke
 87  * 
 88  */
 89 AjxRpc.invoke =
 90 function(requestStr, serverUrl, requestHeaders, callback, method, timeout) {
 91 
 92 	var asyncMode = (callback != null);
 93 	var rpcCtxt = AjxRpc.__getFreeRpcCtxt();
 94 
 95 	try {
 96 		var response = rpcCtxt.invoke(requestStr, serverUrl, requestHeaders, callback, method, timeout);
 97 	} catch (ex) {
 98 		var newEx = new AjxException();
 99 		newEx.method = "AjxRpc.prototype._invoke";
100 		if (ex instanceof Error) {
101 			newEx.detail = ex.message;
102 			newEx.code = AjxException.NETWORK_ERROR;
103 			newEx.msg = "Network error";
104 		} else if (ex.code == 101){
105 			// Chrome 
106 			newEx.detail = ex.message;
107 			newEx.code = AjxException.NETWORK_ERROR;
108 			newEx.msg = "Network error";
109 		} else {
110 			newEx.detail = ex.toString();
111 			newEx.code = AjxException.UNKNOWN_ERROR;
112 			newEx.msg = "Unknown Error";
113 		}
114 		// exception hit: we're done whether sync or async, free the context
115 		AjxRpc.freeRpcCtxt(rpcCtxt);
116 		throw newEx;
117 	}
118 	if (!asyncMode) {
119 		// we've returned from a sync request, free the context
120 		AjxRpc.freeRpcCtxt(rpcCtxt);
121 	}
122 	return response;
123 };
124 
125 /**
126  * @private
127  */
128 AjxRpc.freeRpcCtxt =
129 function(rpcCtxt) {
130 	// we're done using this rpcCtxt. Add it back to the pool
131 	if (AjxRpc.__rpcOutstanding[rpcCtxt.id]) {
132 		AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " --- freeing rpcCtxt " + rpcCtxt.id);
133 		AjxRpc.__rpcCache.push(rpcCtxt);
134 		delete AjxRpc.__rpcOutstanding[rpcCtxt.id];
135 		AjxRpc.__RPC_IN_USE--;
136 	}
137 };
138 
139 AjxRpc.removeRpcCtxt =
140 function(rpcCtxt) {
141 	AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " REMOVE rpcCtxt " + rpcCtxt.id);
142 	if (AjxRpc.__rpcOutstanding[rpcCtxt.id]) {
143 		delete AjxRpc.__rpcOutstanding[rpcCtxt.id];
144 		AjxRpc.__RPC_IN_USE--;
145 	}
146 	AjxUtil.arrayRemove(AjxRpc.__rpcCache, rpcCtxt);
147 };

149 /**
150  * Returns the request from the RPC context with the given ID.
151  *
152  * @param {String} id RPC context ID
153  * 
154  * @return The <i>AjxRpcRequest</i> object associated with <code>id</code> or null
155  * 		if no object exists for the supplied id
156  * @type AjxRpcRequest
157  * 
158  * @private
159  */
160 AjxRpc.getRpcRequestById = 
161 function(id) {
162 	return (AjxRpc.__rpcOutstanding[id]);
163 };

165 /**
166  * Factory method for getting context objects.
167  * 
168  * @private
169  */
170 AjxRpc.__getFreeRpcCtxt = 
171 function() {
172 	var rpcCtxt;
173 
174 	if (AjxRpc.__rpcCache.length > 0) {
175 		rpcCtxt = AjxRpc.__rpcCache.pop();
176 		AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " reusing RPC ID " + rpcCtxt.id);
177 	} else {
178 		if (AjxRpc.__RPC_IN_USE < AjxRpc.__RPC_CACHE_MAX) {
179 			// we haven't reached our limit, so create a new AjxRpcRequest
180 			var id = "__RpcCtxt_" + AjxRpc.__RPC_ID;
181 			rpcCtxt = new AjxRpcRequest(id);
182 			AjxRpc.__RPC_ID++;
183 			AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " Created RPC " + id);
184 		} else {
185 			// yikes, we're out of rpc's! Look for an old one to kill.
186 			rpcCtxt = AjxRpc.__reap();
187 
188 			// if reap didn't find one either, bail.
189 			if (!rpcCtxt) {
190 				var text = [];
191 				for (var i in AjxRpc.__rpcOutstanding) {
192 					var rpcCtxt = AjxRpc.__rpcOutstanding[i];
193 					text.push(rpcCtxt.methodName);
194 				}
195 				var detail = text.join("<br>") + "<br>";
196 				AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " Out of RPC cache!!! Outstanding requests: " + detail);
197 				throw new AjxException("Out of RPC cache", AjxException.OUT_OF_RPC_CACHE, "AjxRpc.__getFreeRpcCtxt", detail);
198 			}
199 		}
200 	}
201 
202 	AjxRpc.__rpcOutstanding[rpcCtxt.id] = rpcCtxt;
203 	AjxRpc.__RPC_IN_USE++;
204 	if (AjxRpc.__RPC_IN_USE > AjxRpc.__RPC_HIGH_WATER) {
205 		AjxRpc.__RPC_HIGH_WATER = AjxRpc.__RPC_IN_USE;
206 		AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " High water mark: " + AjxRpc.__RPC_HIGH_WATER);
207 	}
208 
209 	// always reset timestamp before returning rpcCtxt
210 	rpcCtxt.timestamp = (new Date()).getTime();
211 	return rpcCtxt;
212 };

214 /**
215  * Frees expired contexts.
216  * 
217  * @param {boolean}	all		if true, frees all expired contexts; otherwise, returns the first one it finds
218  * @private
219  */
220 AjxRpc.__reap =
221 function(all) {
222 	var rpcCtxt;
223 	var time = (new Date()).getTime();
224 	AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " Running RPC context reaper");
225 	for (var i in AjxRpc.__rpcOutstanding) {
226 		rpcCtxt = AjxRpc.__rpcOutstanding[i];
227 		if ((rpcCtxt.timestamp + AjxRpc.__RPC_REAP_AGE) < time) {
228 			DBG.println(AjxDebug.DBG1, "AjxRpc.__reap: cleared RPC context " + rpcCtxt.id);
229 			AjxDebug.println(AjxDebug.RPC, AjxDebug._getTimeStamp() + " AjxRpc.__reap: cleared RPC context " + rpcCtxt.id);
230 			rpcCtxt.cancel();
231 			delete AjxRpc.__rpcOutstanding[i];
232 			AjxRpc.__RPC_IN_USE--;
233 			if (!all) {
234 				return rpcCtxt;
235 			}
236 		}
237 	}
238 	return null;
239 };
240 
241 window.setInterval(AjxRpc.__reap.bind(null, true), AjxRpc.__RPC_REAP_INTERVAL);
