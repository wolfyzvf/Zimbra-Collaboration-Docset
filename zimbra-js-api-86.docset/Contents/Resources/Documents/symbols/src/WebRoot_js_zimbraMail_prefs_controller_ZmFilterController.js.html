 22 /**
 23  * Creates a new, empty filter rules controller.
 24  * @class
 25  * This class represents the filter rules controller. This controller manages
 26  * the filter rules page, which has a button toolbar and a list view of the rules.
 27  *
 28  * @author Conrad Damon
 29  *
 30  * @param {DwtShell}		container		the shell
 31  * @param {ZmPreferencesApp}	prefsApp		the preferences application
 32  * 
 33  * @extends		ZmController
 34  */
 35 ZmFilterController = function(container, prefsApp, prefsView, section, parent) {
 36 
 37 	ZmPrefController.call(this, container, prefsApp);
 38 
 39 	this._prefsView = prefsView;
 40 
 41 	this._parent = parent;
 42 	this._toolbar = parent._toolbar;
 43 
 44 	this._filterView = new ZmFilterPage(prefsView, section, this);
 45 	this._incomingFilterRulesController = new ZmFilterRulesController(container, prefsApp, this._filterView, this, false);
 46 	this._outgoingFilterRulesController = new ZmFilterRulesController(container, prefsApp, this._filterView, this, true);
 47 };
 48 
 49 ZmFilterController.prototype = new ZmPrefController;
 50 ZmFilterController.prototype.constructor = ZmFilterController;
 51

 53 function() {
 54 	return "ZmFilterController";
 55 };

 92 ZmFilterController.prototype.hasOutgoingFiltersActive =
 93 function(callback) {
 94 	var rules = this._outgoingFilterRulesController.getRules();
 95 	if (!rules._initialized) {
 96 		rules.loadRules(false, new AjxCallback(this, this._handleLoadFilters, [callback]));
 97 	} else {
 98 		var outgoingActive = rules.getActiveRules().size() > 0;
 99 		if (callback)
100 			callback.run(outgoingActive);
101 		return outgoingActive;
102 	}
103 };
104 
105 ZmFilterController.prototype._handleLoadFilters =
106 function(callback) {
107 	var outgoingActive = this._outgoingFilterRulesController.getRules().getActiveRules().size() > 0;
108 	if (callback)
109 		callback.run(outgoingActive);
110 	return outgoingActive;
111 };
112 
113 ZmFilterController.prototype._stateChangeListener =
114 function (ev) {
115    var index, sel, rules = null;
116 
117    var listView = this._incomingFilterRulesController.getListView();
118    if (listView) {
119        sel = listView.getSelection()[0];
120        rules = this._incomingFilterRulesController.getRules();
121        index = sel ? rules.getIndexOfRule(sel) : null;
122        this._incomingFilterRulesController.resetListView(index);
123    }
124 
125    var outListView = this._outgoingFilterRulesController.getListView();
126    if (outListView) {
127        sel = outListView.getSelection()[0];
128        rules = this._outgoingFilterRulesController.getRules();
129        index = sel ? rules.getIndexOfRule(sel) : null;
130        this._outgoingFilterRulesController.resetListView(index);
131    }
132 
133 	ZmPrefController.prototype._stateChangeListener.apply(this, arguments);
134 };
