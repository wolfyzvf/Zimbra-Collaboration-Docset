 28 /**
 29  * 
 30  * @class
 31  * This class represents an IM address and is used to centralize the representation of IM addresses,
 32  * since they will be used in both contacts, IM and probably in a Zimlet that recognizes them in email.
 33  * 
 34  */
 35 ZmImAddress = {
 36 
 37     IM_SERVICES : [
 38             { label: ZmMsg.zimbraTitle , value: "local" },
 39             { label: ZmMsg.yahoo       , value: "yahoo" },
 40             { label: ZmMsg.msn         , value: "msn" },
 41             { label: ZmMsg.aol         , value: "aol" },
 42             { label: ZmMsg.other       , value: "other" }
 43     ],
 44 
 45     REGEXP : [], // it's built at load-time below
 46 
 47     parse : function(addr) {
 48         var m = ZmImAddress.REGEXP.exec(addr);
 49         if (m) {
 50             return { service    : m[1],
 51                 screenName : m[2] };
 52         }
 53                 // undef if unknown
 54     },
 55 
 56     make : function(service, screenName) {
 57         var addr = service + "://" + screenName;
 58                 // check if it's acceptable
 59         if (ZmImAddress.parse(addr)) {
 60             return addr;
 61         }
 62         return "";
 63     },
 64 
 65     display : function(id) {
 66         var addr = ZmImAddress.parse(id);
 67         if (addr) {
 68             var a = ZmImAddress.IM_SERVICES, i = 0, s;
 69             while (s = a[i++]) {
 70                 if (s.value == addr.service)
 71                     break;
 72             }
 73             if (s)
 74                 return addr.screenName + " (" + s.label + ")";
 75         }
 76         return id;
 77     }
 78 
 79 };
