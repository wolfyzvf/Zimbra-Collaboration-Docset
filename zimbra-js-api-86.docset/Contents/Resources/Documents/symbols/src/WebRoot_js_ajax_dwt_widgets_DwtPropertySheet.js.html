 22 /**
 23  * 
 24  * @private
 25  */
 26 DwtPropertySheet = function(parent, className, posStyle, labelSide) {
 27 	if (arguments.length == 0) return;
 28 	className = className || "DwtPropertySheet";
 29 	DwtComposite.call(this, {parent:parent, className:className, posStyle:posStyle});
 30 
 31 	this._labelSide = labelSide || DwtPropertySheet.DEFAULT;
 32 
 33 	this._propertyIdCount = 0;
 34 	this._propertyList = [];
 35 	this._propertyMap = {};
 36 
 37 	this._tabGroup = new DwtTabGroup(this.toString());
 38 
 39 	this._tableEl = document.createElement("TABLE");
 40 	// Cellspacing needed for IE in quirks mode
 41 	this._tableEl.cellSpacing = 6;
 42 
 43 	var element = this.getHtmlElement();
 44 	element.appendChild(this._tableEl);
 45 	this._setAllowSelection();
 46 }
 47 
 48 DwtPropertySheet.prototype = new DwtComposite;
 49 DwtPropertySheet.prototype.constructor = DwtPropertySheet;
 50 
 51 DwtPropertySheet.prototype.toString = 
 52 function() {
 53 	return "DwtPropertySheet";
 54 }
 55 
 56 // Constants
 57 
 58 DwtPropertySheet.RIGHT = "right";
 59 DwtPropertySheet.LEFT = "left";
 60 DwtPropertySheet.DEFAULT = DwtPropertySheet.LEFT;
 61 
 62 // Data
 63 
 64 DwtPropertySheet.prototype._labelCssClass = "Label";
 65 DwtPropertySheet.prototype._valueCssClass = "Field";
 66 
 67 // Public methods
 68 
 69 /**
 70  * Adds a property.
 71  *
 72  * @param label [string] The property label. The value is used to set the
 73  *				inner HTML of the property label cell.
 74  * @param value The property value. If the value is an instance of DwtControl
 75  *				the element returned by <code>getHtmlElement</code> is used;
 76  *				if the value is an instance of Element, it is added directly;
 77  *				anything else is set as the inner HTML of the property value
 78  *				cell.
 79  * @param required [boolean] Determines if the property should be marked as
 80  *				   required. This is denoted by an asterisk next to the label.
 81  */
 82 DwtPropertySheet.prototype.addProperty = function(label, value, required) {
 83 
 84 	var index = this._tableEl.rows.length;
 85 
 86 	var row = this._tableEl.insertRow(-1);
 87 	row.vAlign = this._vAlign ? this._vAlign : "top";
 88 
 89 	var labelId = Dwt.getNextId(),
 90 		valueId = Dwt.getNextId(),
 91 		tabValue;   // element or control that can be a tab group member
 92 
 93 	if (this._labelSide == DwtPropertySheet.LEFT) {
 94 		this._insertLabel(row, label, required, labelId, valueId);
 95 		tabValue = this._insertValue(row, value, required, labelId, valueId);
 96 	}
 97 	else {
 98 		this._insertValue(row, value, required, labelId, valueId);
 99 		tabValue = this._insertLabel(row, label, required, labelId, valueId);
100 	}
101 
102 	var id = this._propertyIdCount++;
103 	var property = {
104 		id:         id,
105 		index:      index,
106 		row:        row,
107 		visible:    true,
108 		labelId:    labelId,
109 		valueId:    valueId,
110 		tabValue:   tabValue
111 	};
112 	this._propertyList.push(property);
113 	this._propertyMap[id] = property;
114 	return id;
115 };
116 
117 DwtPropertySheet.prototype._insertLabel = function(row, label, required, labelId, valueId) {
118 
119 	var labelCell = row.insertCell(-1);
120 	labelCell.className = this._labelCssClass;
121 	labelCell.id = labelId;
122 	labelCell.setAttribute("for", valueId);
123 	if (this._labelSide != DwtPropertySheet.LEFT) {
124 		labelCell.width = "100%";
125 		labelCell.style.textAlign = "left";
126 	}
127 	labelCell.innerHTML = label;
128 	if (required) {
129 		var asterisk = this._tableEl.ownerDocument.createElement("SUP");
130 		asterisk.innerHTML = "*";
131 		labelCell.insertBefore(asterisk, labelCell.firstChild);
132 	}
133 };
134 
135 DwtPropertySheet.prototype._insertValue = function(row, value, required, labelId, valueId) {
136 
137 	var valueCell = row.insertCell(-1);
138 	valueCell.className = this._valueCssClass;
139 	valueCell.id = valueId;
140 
141 	if (!value) {
142 		valueCell.innerHTML = " ";
143 	}
144 	else if (value.isDwtControl) {
145 		valueCell.appendChild(value.getHtmlElement());
146 		this._tabGroup.addMember(value);
147 		var input = value.getInputElement && value.getInputElement();
148 		if (input) {
149 			input.setAttribute('aria-labelledby', labelId);
150 		}
151 	}
152 	else if (value.nodeType == AjxUtil.ELEMENT_NODE) {
153 		valueCell.appendChild(value);
154 		this._addTabGroupMemberEl(valueCell);
155 		value.setAttribute('aria-labelledby', labelId);
156 	}
157 	else {
158 		valueCell.innerHTML = String(value);
159 		this._addTabGroupMemberEl(valueCell);
160 		valueCell.setAttribute('aria-labelledby', labelId);
161 		value = valueCell;
162 	}
163 
164 	return value;
165 };
166 
167 /**
168  * Add element's leaf children to tabgroup.
169  *
170  * @param element HTML element.
171  */
172 DwtPropertySheet.prototype._addTabGroupMemberEl = function(element, isTabStop) {
173 
174 	var obj = this;
175 
176 	// recursive function to add leaf nodes
177 	function addChildren(el) {
178 		if (el.children.length > 0) {
179 			AjxUtil.foreach(el.children, function(child){
180 				addChildren(child);
181 			});
182 		}
183 		else {
184 			if (AjxUtil.isBoolean(isTabStop)) {
185 				obj.noTab = !isTabStop;
186 			}
187 			else {
188 				// add leaf to tabgroup
189 				obj._makeFocusable(el);
190 				obj._tabGroup.addMember(el);
191 			}
192 		}
193 	}
194 
195 	addChildren(element, isTabStop);
196 };
197 
198 DwtPropertySheet.prototype.getTabGroupMember = function() {
199 	return this._tabGroup;
200 };
201 
202 DwtPropertySheet.prototype.getProperty = function(id) {
203 	return this._propertyMap[id];
204 };
205 
206 DwtPropertySheet.prototype.removeProperty = function(id) {
207 	var prop = this._propertyMap[id];
208 	if (prop) {
209 		var propIndex = prop.index;
210 		if (prop.visible) {
211 			var tableIndex = this.__getTableIndex(propIndex);
212 			var row = this._tableEl.rows[tableIndex];
213 			row.parentNode.removeChild(row);
214 		}
215 		prop.row = null;
216 		for (var i = propIndex + 1; i < this._propertyList.length; i++) {
217 			this._propertyList[i].index--;
218 		}
219 		this._propertyList.splice(propIndex, 1);
220 		delete this._propertyMap[id];
221 	}
222 };
223 
224 DwtPropertySheet.prototype.setPropertyVisible = function(id, visible) {
225 
226 	var prop = this._propertyMap[id];
227 	if (prop && prop.visible !== visible) {
228 		prop.visible = visible;
229 		Dwt.setVisible(this._tableEl.rows[prop.index], visible);
230 		var tabValue = prop.tabValue;
231 		if (tabValue.isDwtControl) {
232 			tabValue.noTab = !visible;
233 		}
234 		else {
235 			this._addTabGroupMemberEl(tabValue, !visible)
236 		}
237 	}
238 };
