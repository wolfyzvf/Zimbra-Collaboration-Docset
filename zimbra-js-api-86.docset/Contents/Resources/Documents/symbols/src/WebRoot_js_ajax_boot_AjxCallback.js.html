 22 /**
 23  * Creates a callback which consists of at least a function reference, and possibly also
 24  * an object to call it from.
 25  * @constructor
 26  * @class
 27  * This class represents a callback function which can be called standalone, or from a
 28  * given object. What the callback takes as arguments and what it returns are left to the
 29  * client.
 30  *
 31  * @author Conrad Damon
 32  * 
 33  * @param {object}	obj		the object to call the function from
 34  * @param {function}	func	the callback function
 35  * @param {array}	args   the default arguments
 36  * 
 37  */
 38 AjxCallback = function(obj, func, args) {
 39 	if (arguments.length == 0) return;
 40 
 41     if (typeof arguments[0] == "function") {
 42         this.obj = null;
 43         this.func = arguments[0];
 44         this.args = arguments[1];
 45     }
 46     else {
 47         this.obj = obj;
 48         this.func = func;
 49         this.args = args;
 50     }
 51 };
 52 
 53 AjxCallback.prototype.isAjxCallback = true;
 54 AjxCallback.prototype.toString = function() { return "AjxCallback"; }
 55 
 56 AjxCallback.NOP = new AjxCallback(function(){});

 58 /**
 59  * Runs the callback function, from within the object if there is one. The
 60  * called function passed arguments are the concatenation of the argument
 61  * array passed to this object's constructor and the argument array passed
 62  * to the <code>run</code> method. Whatever the called function returns is
 63  * returned to the caller.
 64  *
 65  * @param {array}	[arg1..argN]	the first argument which will be appended to the argument
 66  *				array passed to this object's constructor. Any number of
 67  *				arguments may be passed to the <code>run</code> method.
 68  */
 69 AjxCallback.prototype.run =
 70 function(/* arg1 ... argN */) {
 71 	// combine original args with new ones
 72 	var args = [];
 73 
 74 	// sometimes we want to pass a null or false argument, so simply
 75 	// checking for if (this.args) won't do.
 76 	if (typeof this.args != "undefined") {
 77 		if (this.args!==null && this.args instanceof Array) { // IE mysteriously screws up the instanceof test if this.args is null
 78 			// NOTE: We must NOT use this.args directly if this method's
 79 			//       params are gonna be pushed onto the array because it
 80 			//       will change the original args!
 81 			args = arguments.length > 0 ? args.concat(this.args) : this.args;
 82 		} else {
 83 			args.push(this.args);
 84 		}
 85 	}
 86 
 87 	for (var i = 0; i < arguments.length; ++i) {
 88 		args.push(arguments[i]);
 89 	}
 90 
 91 	// invoke function
 92 	if (this.func) {
 93         return this.func.apply(this.obj || window, args);
 94 	}
 95 };

 97 /**
 98  * This version of {@link AjxCallback.run} is here for {@link AjxDispatcher}, because it has a <code>run()</code>
 99  * method in which it marshals arguments into an array. That leads to a problem
100  * in which the arguments are marshalled twice, so that by the time AjxDispatcher
101  * calls <code>callback.run(args)</code>, the args have already been collected into an array.
102  * Then when the function is invoked, it gets passed an actual array instead of the
103  * intended arg list. Calling <code>callback.run.apply(callback, args)</code> works on Firefox,
104  * but IE throws the error "Object expected", so we do this instead.
105  *
106  * @param	{array}	argList	 an array of arguments and treats them as an argument list, instead of as a single argument
107  * 
108  * @private
109  */
110 AjxCallback.prototype.run1 =
111 function(argList) {
112 	// combine original args with new ones
113 	var args = [];
114 
115 	// sometimes we want to pass a null or false argument, so simply
116 	// checking for if (this.args) won't do.
117 	if (typeof this.args != "undefined") {
118 		if (this.args!==null && this.args instanceof Array) {
119 			// NOTE: We must NOT use this.args directly if this method's
120 			//       params are gonna be pushed onto the array because it
121 			//       will change the original args!
122 			args = arguments.length > 0 ? args.concat(this.args) : this.args;
123 		} else {
124 			args.push(this.args);
125 		}
126 	}
127 
128 	if (argList && argList.length) {
129 		for (var i = 0; i < argList.length; ++i) {
130 			args.push(argList[i]);
131 		}
132 	}
133 
134 	// invoke function
135 	if (this.func) {
136 		return this.func.apply(this.obj || window, args);
137 	}
138 };

140 /**
141  * This method returns a plain function that will call your supplied "func" in the context
142  * of "obj" and pass to it, in this order, any additional arguments that you
143  * pass to <code>simpleClosure</code> and the arguments that were passed to it at the call
144  * time.
145  *
146  * <p>
147  * An example should do:
148  *
149  * <pre>
150  *   div.onclick = AjxCallback.simpleClosure(this.handler, this, "some data");
151  *   ...
152  *   this.handler = function(data, event) {
153  *      // event will be passed for DOM2 compliant browsers
154  *      // and data is "some data"
155  *   };
156  * </pre>
157  * 
158  * @param	{function}	func		the function
159  * @param	{object}	obj			the object to call the function from
160  * @param	{array}	[arg1...argN]		any number of arguments
161  */
162 AjxCallback.simpleClosure = function(func, obj) {
163 	var args = [];
164 	for (var i = 2; i < arguments.length; ++i)
165 		args.push(arguments[i]);
166 	return function() {
167 		var args2 = [];
168 		for (var i = 0; i < arguments.length; ++i)
169 			args2.push(arguments[i]);
170 		return func.apply(obj || this, args.concat(args2));
171 	};
172 };

174 AjxCallback.returnFalse = function() { return false; };
175 
176 AjxCallback.isNull = function(x) { return x == null; };
