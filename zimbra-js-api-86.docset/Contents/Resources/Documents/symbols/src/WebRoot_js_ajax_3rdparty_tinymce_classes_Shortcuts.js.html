 11 /**
 12  * Contains all logic for handling of keyboard shortcuts.
 13  */
 14 define("tinymce/Shortcuts", [
 15 	"tinymce/util/Tools",
 16 	"tinymce/Env"
 17 ], function(Tools, Env) {
 18 	var each = Tools.each, explode = Tools.explode;
 19 
 20 	var keyCodeLookup = {
 21 		"f9": 120,
 22 		"f10": 121,
 23 		"f11": 122
 24 	};
 25 
 26 	return function(editor) {
 27 		var self = this, shortcuts = {};
 28 
 29 		editor.on('keyup keypress keydown', function(e) {
 30 			if ((e.altKey || e.ctrlKey || e.metaKey) && !e.isDefaultPrevented()) {
 31 				each(shortcuts, function(shortcut) {
 32 					var ctrlKey = Env.mac ? e.metaKey : e.ctrlKey;
 33 
 34 					if (shortcut.ctrl != ctrlKey || shortcut.alt != e.altKey || shortcut.shift != e.shiftKey) {
 35 						return;
 36 					}
 37 
 38 					if (e.keyCode == shortcut.keyCode || (e.charCode && e.charCode == shortcut.charCode)) {
 39 						e.preventDefault();
 40 
 41 						if (e.type == "keydown") {
 42 							shortcut.func.call(shortcut.scope);
 43 						}
 44 
 45 						return true;
 46 					}
 47 				});
 48 			}
 49 		});
 50 
 51 		/**
 52 		 * Adds a keyboard shortcut for some command or function.
 53 		 *
 54 		 * @method addShortcut
 55 		 * @param {String} pattern Shortcut pattern. Like for example: ctrl+alt+o.
 56 		 * @param {String} desc Text description for the command.
 57 		 * @param {String/Function} cmdFunc Command name string or function to execute when the key is pressed.
 58 		 * @param {Object} sc Optional scope to execute the function in.
 59 		 * @return {Boolean} true/false state if the shortcut was added or not.
 60 		 */
 61 		self.add = function(pattern, desc, cmdFunc, scope) {
 62 			var cmd;
 63 
 64 			cmd = cmdFunc;
 65 
 66 			if (typeof(cmdFunc) === 'string') {
 67 				cmdFunc = function() {
 68 					editor.execCommand(cmd, false, null);
 69 				};
 70 			} else if (Tools.isArray(cmd)) {
 71 				cmdFunc = function() {
 72 					editor.execCommand(cmd[0], cmd[1], cmd[2]);
 73 				};
 74 			}
 75 
 76 			each(explode(pattern.toLowerCase()), function(pattern) {
 77 				var shortcut = {
 78 					func: cmdFunc,
 79 					scope: scope || editor,
 80 					desc: editor.translate(desc),
 81 					alt: false,
 82 					ctrl: false,
 83 					shift: false
 84 				};
 85 
 86 				each(explode(pattern, '+'), function(value) {
 87 					switch (value) {
 88 						case 'alt':
 89 						case 'ctrl':
 90 						case 'shift':
 91 							shortcut[value] = true;
 92 							break;
 93 
 94 						default:
 95 							// Allow numeric keycodes like ctrl+219 for ctrl+[
 96 							if (/^[0-9]{2,}$/.test(value)) {
 97 								shortcut.keyCode = parseInt(value, 10);
 98 							} else {
 99 								shortcut.charCode = value.charCodeAt(0);
100 								shortcut.keyCode = keyCodeLookup[value] || value.toUpperCase().charCodeAt(0);
101 							}
102 					}
103 				});
104 
105 				shortcuts[
106 					(shortcut.ctrl ? 'ctrl' : '') + ',' +
107 					(shortcut.alt ? 'alt' : '') + ',' +
108 					(shortcut.shift ? 'shift' : '') + ',' +
109 					shortcut.keyCode
110 				] = shortcut;
111 			});
112 
113 			return true;
114 		};
115 	};
116 });
