 22 /**
 23  * @class
 24  * This static class serves as a central location for registering and calling
 25  * public API methods. For example, the registration of a public API method for
 26  * pulling up the compose form might look something like this:
 27  * 
 28  * <pre>
 29  * AjxDispatcher.register("Compose", "Mail", new AjxCallback(this, this.doAction));
 30  * </pre>
 31  * 
 32  * and a client call of it like this:
 33  * 
 34  * <pre>
 35  * AjxDispatcher.run("Compose", {action:ZmOperation.NEW_MESSAGE, inNewWindow:false});
 36  * </pre>
 37  * 
 38  * Registration will most likely need to happen in a constructor, when 'this' is
 39  * available, since you'll most likely want the function call to happen in that
 40  * object's context.
 41  * <p>
 42  * A package can also register a callback to be run once the package has loaded. One use case for
 43  * that is to register newly-defined classes as drop targets. There is a wrapper method around
 44  * AjxPackage.require() that will call that method and then run the post-load callback (if any):
 45  *
 46  * <pre>
 47  * AjxDispatcher.require("Calendar");
 48  * </pre>
 49  * 
 50  * @author Conrad Damon
 51  */
 52 AjxDispatcher = function() {}
 53 
 54 // Table of API names, packages, and associated function calls
 55 AjxDispatcher._registry = {};
 56 
 57 // Table of package names and callbacks to run after loading (optional)
 58 AjxDispatcher._package = {};
 59 
 60 AjxDispatcher._preLoad	= [];
 61 AjxDispatcher._postLoad	= [];
 62 AjxDispatcher._loadFunctionsEnabled	= false;
 63 AjxDispatcher._timedAction = null;
 64 
 65 /**
 66  * Adds a function to be called after the given package has been loaded.
 67  * 
 68  * @param {string}	pkg		the name of package
 69  * @param {AjxCallback}	callback	the callback to run after package has loaded
 70  */
 71 AjxDispatcher.addPackageLoadFunction =
 72 function(pkg, callback) {
 73 	var pkgData = AjxDispatcher._getPackageData(pkg);
 74 	if (!pkgData._loaded && !AjxPackage.isDefined(pkg)) {
 75 		pkgData.callback.push(callback);
 76 	} else {
 77 		AjxTimedAction.scheduleAction(new AjxTimedAction(callback, callback.run), 0);
 78 	}
 79 };

 81 /**
 82  * Adds a function to be called while a package is being loaded. A typical use
 83  * is to display a "Loading..." screen.
 84  * 
 85  * @param {AjxCallback}	callback	the callback to run after package has loaded
 86  */
 87 AjxDispatcher.addPreLoadFunction =
 88 function(callback) {
 89 	AjxDispatcher._preLoad.push(callback);

 92 /**

 97  */
 98 AjxDispatcher.addPostLoadFunction =
 99 function(callback) {
100 	AjxDispatcher._postLoad.push(callback);
101 };

128 /**
129  * Checks if the given package has been loaded.
130  * 
131  * @param	{string}	pkg		the name of package
132  * @return	{boolean}	<code>true</code> if the package is loaded
133  */
134 AjxDispatcher.loaded =
135 function(pkg) {
136     var pkgData = AjxDispatcher._getPackageData(pkg);
137 	return (pkgData && pkgData._loaded) || AjxPackage.isDefined(pkg);
138 };
139 
140 /**
141  * Programmatically sets whether the given packages has been loaded. Use with care!
142  * 
143  * @param {string}	pkg		the name of package
144  * @return	{boolean}	if <code>true</code>, the package is loaded
145  */
146 AjxDispatcher.setLoaded =
147 function(pkg, loaded) {
148 	var pkgData = AjxDispatcher._getPackageData(pkg);
149 	pkgData._loaded = loaded;
150 	if (loaded) {
151 		var callbacks = pkgData.callback || [];
152 		for (var i = 0; i < callbacks.length; i++) {
153 			callbacks[i].run();
154 		}
155 	}
156 };

158 /**
159  * Registers an API method so that it may be called.
160  * 
161  * @param {string}	method	the name of the API method
162  * @param {string}	pkg		the name of required package(s)
163  * @param {AjxCallback}	callback	the callback to run for this API call
164  */
165 AjxDispatcher.registerMethod =
166 function(method, pkg, callback) {
167 	AjxDispatcher._registry[method] = {pkg:pkg, callback:callback};
168 };

170 /**
171  * Calls the given API method with the given arguments. It can be passed any
172  * number of arguments (provided after the API name), and they will be passed
173  * to the function that gets called.
174  * 
175  * PS: You are in a maze of twisty callbacks, all alike.
176  * 
177  * @param {string}	method		the name of the API method
178  * @param {boolean}	async			if <code>true</code>, load package asynchronously
179  * @param {AjxCallback}	callback		the callback to run with results (must be present
180  * 										if there are pre- or post-load functions)
181  * @param {boolean}		preLoadOk		if <code>true</code>, okay to run registered pre-load function
182  * 
183  * @private
184  */
185 AjxDispatcher.run =
186 function(params /*, arg1 ... argN */) {
187 	if (!params) { return; }
188 	var method, noLoad, async, callback, preLoadOk;
189 	if (typeof(params) == "string") {
190 		method = params;
191         async = false;
192 		preLoadOk = false;
193 	} else {
194 		method = params.method;
195 		noLoad = params.noLoad;
196 		callback = params.callback;
197         async = params.async != null ? params.async : Boolean(callback);
198 		preLoadOk = params.preLoadOk != null ? params.preLoadOk : (callback != null);
199 	}
200 	var item = AjxDispatcher._registry[method];
201 	if (!item) {
202 		// method hasn't been registered
203 		AjxPackage.__log("API method '" + method + "' not found");
204 		return;
205 	}
206 	AjxPackage.__log("Run method: " + method);
207 	var pkg = item.pkg;
208 	var args = [];
209 	for (var i = 1; i < arguments.length; ++i) {
210 		args.push(arguments[i]);
211 	}
212 	if (callback) {
213 		args.push(callback);
214 	}
215 	
216 	return AjxDispatcher.require(pkg, async, item.callback, args, preLoadOk);
217 };

219 /**
220  * Loads the given package, and runs its requested post-load callback. Clients should
221  * be careful not to mix async and sync calls for the same package, in order to avoid
222  * race conditions.
223  * 
224  * @param {string}	pkg				the name of the API method
225  * @param {boolean}	async				if <code>true</code>, load package asynchronously
226  * @param {AjxCallback}	callback			the callback to run after pkg load
227  * @param {array}	args				the args to pass to callback
228  * @param {boolean}	preLoadOk			if <code>true</code>, okay to run registered pre-load function
229  */
230 AjxDispatcher.require =
231 function(pkg, async, callback, args, preLoadOk) {
232 	if (!pkg) { return; }
233 	
234 	if (typeof(pkg) == "string") {
235 		pkg = [pkg];
236 	}
237 	var unloaded = [];
238 	for (var i = 0; i < pkg.length; i++) {
239 		var p = pkg[i];
240 		if (!AjxDispatcher._getPackageData(p)._loaded) {
241 			unloaded.push(p);
242 		}
243 	}
244 	if (unloaded.length == 0) {
245 		return AjxDispatcher._postLoadCallback(pkg, false, callback, args);
246 	} else {
247 		// need callback in order to run pre-load function
248 		var preLoad = AjxDispatcher._preLoad;
249 		if (preLoadOk && AjxDispatcher._loadFunctionsEnabled && preLoad.length) {
250 			AjxPackage.__log("pre-load function");
251 			AjxDispatcher._timedAction = new AjxCallback(null, AjxDispatcher._continueRequire, [unloaded, async, callback, args]);
252 			for (var i = 0; i < preLoad.length; i++) {
253 				preLoad[i].run();
254 			}
255 			window.setTimeout('AjxDispatcher._timedAction.run()', 0);
256 		} else {
257 			return AjxDispatcher._continueRequire(unloaded, async, callback, args);
258 		}
259 	}
260 };
261 
262 AjxDispatcher._continueRequire =
263 function(pkg, async, callback, args) {
264 	var pkgString = pkg.join(", ");
265 	AjxPackage.__log("------------------------------------- Loading package: " + pkgString);
266 	if (window.console && window.console.log) { console.log("------------------------------------- Loading package: " + pkgString); }
267 	if (async && callback) {
268 		var postLoadCallback = new AjxCallback(null, AjxDispatcher._postLoadCallback, [pkg, true, callback, args]);
269 		AjxPackage.require({name:pkg, callback:postLoadCallback});
270 	} else {
271 		var _st = new Date();
272 		for (var i = 0; i < pkg.length; i++) {
273 			AjxPackage.require(pkg[i]);
274 		}
275 		var _en = new Date();
276 		var t = _en.getTime() - _st.getTime();
277 		AjxPackage.__log("LOAD TIME for " + pkgString + ": " + t);
278 
279 		return AjxDispatcher._postLoadCallback(pkg, true, callback, args);
280 	}
281 };
282 
283 AjxDispatcher._postLoadCallback =
284 function(pkg, pkgWasLoaded, callback, args) {
285     for (var i = 0; i < pkg.length; i++) {
286         AjxDispatcher._getPackageData(pkg[i])._loaded = true;
287     }
288     for (var i = 0; i < pkg.length; i++) {
289 		var pkgData = AjxDispatcher._getPackageData(pkg[i]);
290 		if (pkgWasLoaded && pkgData.callback.length && !pkgData.callbackDone) {
291 			pkgData.callbackDone = true;
292 			AjxPackage.__log("Running post-load package function for " + pkg[i]);
293 			var callbacks = pkgData.callback;
294 			for (var j = 0; j < callbacks.length; j++) {
295 				callbacks[j].run();
296 			}
297 			pkgData.callback.length = 0;
298 		}
299 	}
300 	if (pkgWasLoaded) {
301 		var postLoad = AjxDispatcher._postLoad;
302 		if (AjxDispatcher._loadFunctionsEnabled && postLoad.length) {
303 			for (var i = 0; i < postLoad.length; i++) {
304 				postLoad[i].run();
305 			}
306 		}
307 	}
308 	
309 	if (callback) {
310  		return callback.isAjxCallback ? callback.run1(args) : callback(args);
311 	}
312 };
313 
314 AjxDispatcher._getPackageData = function(pkg) {
315 	if (!AjxDispatcher._package[pkg]) {
316 		AjxDispatcher._package[pkg] = { callback: [] };
317 	}
318 	return AjxDispatcher._package[pkg];
319 };
