 28 /**
 29  * Creates a dialog.
 30  * @class
 31  * This class is a base class for miscellaneous organizer-related dialogs. An instance
 32  * of this class can be re-used to show different overviews.
 33  *
 34  * @author Conrad Damon
 35  *
 36  * @param {Hash}	params		a hash of parameters
 37  * @param	{DwtControl}	params.parent		the parent widget
 38  * @param	{DwtMsgDialog}	params.msgDialog		the message dialog
 39  * @param	{String}	params.className		the CSS class name
 40  * @param	{String}	params.title		the dialog title
 41  * @param	{Array|constant}	params.standardButtons		an array of standard buttons to include. Defaults to {@link DwtDialog.OK_BUTTON} and {@link DwtDialog.CANCEL_BUTTON}.
 42  * @param	{Array}	params.extraButtons		a list of {@link DwtDialog_ButtonDescriptor} objects describing custom buttons to add to the dialog
 43  * @param	{DwtControl}	params.view				the dialog contents
 44  * 
 45  * @extends	DwtDialog
 46  */
 47 ZmDialog = function(params) {
 48 	if (arguments.length == 0) { return; }
 49 
 50 	DwtDialog.call(this, params);
 51 
 52 	if (!params.view) {
 53 		this.setContent(this._contentHtml());
 54 	}
 55 
 56 	if (this._button[DwtDialog.OK_BUTTON]) {
 57 		this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._okButtonListener));
 58 	}
 59 
 60 	this._overview = {};
 61 	this._opc = appCtxt.getOverviewController();
 62 
 63 	this._baseTabGroupSize = this._tabGroup.size();
 64 };
 65 
 66 ZmDialog.prototype = new DwtDialog;
 67 ZmDialog.prototype.constructor = ZmDialog;
 68 
 69 ZmDialog.prototype.isZmDialog = true;
 70 ZmDialog.prototype.toString = function() { return "ZmDialog"; };
 71 
 72 /**
 73  * @private
 74  */
 75 ZmDialog.prototype._contentHtml =
 76 function() {
 77 	return "";
 78 };
 79 
 80 /**
 81  * Sets the view for this dialog.
 82  * 
 83  * @param	{DwtComposite}		newView		the view
 84  * @param	{Boolean}		noReset		if <code>true</code>, do not reset the dialog; <code>false</code> otherwise
 85  */
 86 ZmDialog.prototype.setView =
 87 function(newView, noReset) {
 88 	this.reset();
 89 	if (newView) {
 90         var contentDiv = this._getContentDiv();
 91         var el = newView.getHtmlElement();
 92 		contentDiv.parentNode.replaceChild(el, contentDiv);
 93 		this._contentDiv = el;
 94 	}
 95 };
 96 
 97 ZmDialog.prototype.popup =
 98 function() {
 99 	// Bug 38281: for multiuse dialogs, we need to re-add the discretionary
100 	// tab stops to the base list (the dialog buttons)
101 	this._tabGroup.__members._array.splice(this._baseTabGroupSize);
102 	this._tabGroup.addMember(this._getTabGroupMembers());
103 
104 	DwtDialog.prototype.popup.call(this);
105 	if (this._focusElement) {
106 		appCtxt.getKeyboardMgr().grabFocus(this._focusElement);	
107 	}
108 };
109 
110 ZmDialog.prototype.reset =
111 function() {
112 	if (this._nameField) {
113 		this._nameField.value = "";
114 	}
115 	DwtDialog.prototype.reset.call(this);
116 };

137 /**
138  * Displays the given list of tree views in an overview, creating it if
139  * necessary, and appends the overview to an element in the dialog. Since
140  * dialogs may be reused, it is possible that it will display different
141  * overviews. That is handled by making sure that only the current overview is
142  * visible.
143  * 
144  * @param params		[hash]				hash of params:
145  *        treeIds		[array]				list of tree views to show
146  *        omit			[hash]				IDs of organizers to exclude
147  *        fieldId		[string]			DOM ID of element that contains overview
148  *        overviewId	[string]*			ID for the overview
149  *        noRootSelect	[boolean]*			if true, don't make root tree item(s) selectable
150  * @param forceSingle	[boolean]*			if true, don't make multi-account overviews
151  * @private
152  */
153 ZmDialog.prototype._setOverview =
154 function(params, forceSingle) {
155 
156 	// when in multi-account mode, hide the old overview since we don't know
157 	// whether we want to show an overview container or just a single-account overview.
158 	if (appCtxt.multiAccounts) {
159 		var oldOverview = this._opc.getOverviewContainer(this._curOverviewId) ||
160 						  this._opc.getOverview(this._curOverviewId);
161 
162 		if (oldOverview) {
163 			oldOverview.setVisible(false);
164 		}
165 	}
166 
167 	// multi-account uses overview container
168 	if (appCtxt.multiAccounts && !forceSingle) {
169 		// use overviewId as the containerId; container will assign overviewId's
170 		var containerId = this._curOverviewId = params.overviewId;
171 		var ovContainer = this._opc.getOverviewContainer(containerId);
172 		if (!ovContainer) {
173 			var overviewParams = {
174 				overviewClass:	"dialogOverviewContainer",
175 				headerClass:	"DwtTreeItem",
176 				noTooltips:		true,
177 				treeStyle:		params.treeStyle,
178 				treeIds:		params.treeIds,
179 				overviewTrees:	params.overviewTrees,
180 				omit:			params.omit,
181 				omitPerAcct:	params.omitPerAcct,
182 				selectable:		params.selectable
183 			};
184 			var containerParams = {
185 				appName: params.appName,
186 				containerId: containerId
187 			};
188 			ovContainer = this._opc.createOverviewContainer(containerParams, overviewParams);
189 			ovContainer.setSize(Dwt.DEFAULT, "200");
190 			document.getElementById(params.fieldId).appendChild(ovContainer.getHtmlElement());
191 		}
192 
193 		// make overview container visible
194 		ovContainer.setVisible(true);
195 
196 		return ovContainer;
197 	}
198 
199 	// single-account overview handling
200 	var overviewId = this._curOverviewId = params.overviewId;
201 	var overview = this._opc.getOverview(overviewId);
202 	if (!overview) {
203 		var ovParams = {
204 			overviewId:		overviewId,
205 			overviewClass:	params.overviewClass || "dialogOverview",
206 			headerClass:	"DwtTreeItem",
207 			noTooltips:		true,
208 			treeStyle:		params.treeStyle,
209 			dynamicWidth:	params.dynamicWidth,
210 			treeIds:		params.treeIds,
211 			account:		((appCtxt.multiAccounts && params.forceSingle) ? appCtxt.getActiveAccount() : (params.account || appCtxt.getActiveAccount())),
212 			skipImplicit: 	true,
213 			appName:        params.appName
214 		};
215 		overview = this._overview[overviewId] = this._opc.createOverview(ovParams);
216 		this._renderOverview(overview, params.treeIds, params.omit, params.noRootSelect);
217 		document.getElementById(params.fieldId).appendChild(overview.getHtmlElement());
218 	}
219 	else {
220 		//this might change between clients so have to update this.
221 		this._setRootSelection(overview, params.treeIds, params.noRootSelect);
222 	}
223 
224 	this._makeOverviewVisible(overviewId);
225 
226 	return overview;
227 };
228 
229 /**
230  * @private
231  */
232 ZmDialog.prototype._makeOverviewVisible =
233 function(overviewId) {
234 	for (var id in this._overview) {
235 		this._overview[id].setVisible(id == overviewId);
236 	}
237 };

239 /**
240  * Renders the tree views in the overview, and makes the header items
241  * selectable (since they can generally be targets of whatever action the dialog
242  * is facilitating).
243  * 
244  * @param overview		[ZmOverview]		the overview
245  * @param treeIds		[array]				list of tree views to show
246  * @param omit			[hash]*				IDs of organizers to exclude
247  * @param noRootSelect	[boolean]*			if true, don't make root tree item(s) selectable
248  * @private
249  */
250 ZmDialog.prototype._renderOverview =
251 function(overview, treeIds, omit, noRootSelect) {
252 	overview.set(treeIds, omit);
253 	this._setRootSelection(overview, treeIds, noRootSelect);
254 };
255 
256 ZmDialog.prototype._setRootSelection =
257 function(overview, treeIds, noRootSelect) {
258 	for (var i = 0; i < treeIds.length; i++) {
259 		var treeView = overview.getTreeView(treeIds[i]);
260 		var hi = treeView && treeView.getHeaderItem();
261 		if (hi) {
262 			hi.enableSelection(!noRootSelect);
263 		}
264 	}
265 };

284 /**
285  * @private
286  */
287 ZmDialog.prototype._showError =
288 function(msg, loc) {
289 	var nLoc = loc || (new DwtPoint(this.getLocation().x + 50, this.getLocation().y + 100));
290 	var msgDialog = appCtxt.getMsgDialog();
291 
292 	msgDialog.reset();
293     msgDialog.setMessage(AjxStringUtil.htmlEncode(msg), DwtMessageDialog.CRITICAL_STYLE);
294 	msgDialog.popup(nLoc);
295 };

297 /**
298  * @private
299  */
300 ZmDialog.prototype._getTabGroupMembers =
301 function() {
302 	return this._nameField ? [ this._nameField ] : [];
303 };
