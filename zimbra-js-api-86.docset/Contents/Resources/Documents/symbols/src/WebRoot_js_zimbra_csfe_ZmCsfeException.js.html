 23  * @overview
 24  * This file contains the exception class.

 26 /**
 27  * Creates an exception.
 28  * @class
 29  * This class represents an exception returned by the server as a response, generally as a fault. The fault
 30  * data is converted to properties of the exception.
 31  *
 32  * @param {Hash}	params	a hash of parameters
 33  * @param {String}      params.msg		the explanation (Fault.Reason.Text)
 34  * @param {String}      params.code		the error code (Fault.Detail.Error.Code)
 35  * @param {String}      params.method	the request name
 36  * @param {String}      params.detail	the Fault.Code.Value
 37  * @param {Object}      [params.data]		an optional structured fault data (Fault.Detail.Error.a)
 38  * @param {String}      params.trace		the trace info (Fault.Detail.Error.Trace)
 39  * @param {String}       params.request	the SOAP or JSON that represents the request
 40  * 
 41  * @extends		AjxException
 42  */
 43 ZmCsfeException = function(params) {
 44 
 45 	params = Dwt.getParams(arguments, ZmCsfeException.PARAMS);
 46 
 47 	AjxException.call(this, params.msg, params.code, params.method, params.detail);
 48 	
 49 	if (params.data) {
 50 		this.data = {};
 51 		for (var i = 0; i < params.data.length; i++) {
 52 			var item = params.data[i];
 53 			var key = item.n;
 54 			if (!this.data[key]) {
 55 				this.data[key] = [];
 56 			}
 57 			this.data[key].push(item._content);
 58 		}
 59 	}
 60 	
 61 	this.trace = params.trace;
 62 	this.request = params.request;
 63 };
 64 
 65 ZmCsfeException.PARAMS = ["msg", "code", "method", "detail", "data", "trace"];
 66 
 67 ZmCsfeException.prototype = new AjxException;
 68 ZmCsfeException.prototype.constructor = ZmCsfeException;
 69 ZmCsfeException.prototype.isZmCsfeException = true;
 70

 78 	return "ZmCsfeException";
 79 };

 93 /**
 94  * Gets the error messages.
 95  * 
 96  * @param	{String}	code	the code
 97  * @param	{Array}	args		the message format args
 98  * 
 99  * @return	{String}	the message
100  */
101 ZmCsfeException.getErrorMsg =
102 function(code, args) {
103 	var msg = ZMsg[code];
104 	if (!msg) {
105 		ZmCsfeException._unknownFormat = ZmCsfeException._unknownFormat || new AjxMessageFormat(ZMsg.unknownError);
106 		return ZmCsfeException._unknownFormat.format(code);
107 	}
108 	this.msg = this.msg || msg;
109 	return args ? AjxMessageFormat.format(msg, args) : msg;
110 };
111 
112 //
113 // Public methods
114 //
115 
116 /**
117  * Gets the error message.
118  * 
119  * @param	{Array}	args		the message format args
120  * @return	{String}	the message
121  */
122 ZmCsfeException.prototype.getErrorMsg =
123 function(args) {
124 	return ZmCsfeException.getErrorMsg(this.code, args);
125 };
126 
127 /**
128  * Gets the data.
129  * 
130  * @param	{Object}	key		the key
131  * 
132  * @return	{Object}	the data
133  */
134 ZmCsfeException.prototype.getData =
135 function(key) {
136 	return this.data && this.data[key];
137 };
