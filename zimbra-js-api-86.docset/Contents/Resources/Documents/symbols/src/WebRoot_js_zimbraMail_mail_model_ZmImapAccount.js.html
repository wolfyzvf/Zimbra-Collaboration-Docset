 22 /**
 23  * Creates an IMAP account.
 24  * @class
 25  * This class represents an IMAP account.
 26  * 
 27  * @param	{String}	id		the id
 28  * 
 29  * @extends		ZmDataSource
 30  */
 31 ZmImapAccount = function(id) {
 32 	ZmDataSource.call(this, ZmAccount.TYPE_IMAP, id);
 33 };
 34 
 35 ZmImapAccount.prototype = new ZmDataSource;
 36 ZmImapAccount.prototype.constructor = ZmImapAccount;
 37 
 38 
 39 // Constants
 40 /**
 41  * Defines the "cleartext" port.
 42  * 
 43  * @type	int
 44  */
 45 ZmImapAccount.PORT_CLEAR	= 143;
 46 /**
 47  * Defines the "ssl" port.
 48  * 
 49  * @type	int
 50  */
 51 ZmImapAccount.PORT_SSL		= 993;
 52 ZmImapAccount.PORT_DEFAULT	= ZmImapAccount.PORT_CLEAR;

 55 // advanced settings
 56 ZmImapAccount.prototype.ELEMENT_NAME = "imap";
 57 ZmImapAccount.prototype.port = ZmImapAccount.PORT_DEFAULT;

 63 function() {
 64 	return "ZmImapAccount";
 65 };

 67 /**
 68  * Gets the default port.
 69  * 
 70  * @return	{int}		the port
 71  */
 72 ZmImapAccount.prototype.getDefaultPort =
 73 function() {
 74 	return (this.connectionType == ZmDataSource.CONNECT_SSL)
 75 		? ZmImapAccount.PORT_SSL : ZmImapAccount.PORT_DEFAULT;
 76 };

 78 /**
 79  * Comparison function for *IMAP* folders. Since IMAP folderId's are *not* well-
 80  * known, we have to compare their names instead of their ID's.
 81  * 
 82  * @param	{ZmFolder}	folderA
 83  * @param	{ZmFolder}	folderB
 84  * @return	{int}	0 if the folders are the same; 1 if "a" is before "b"; -1 if "b" is before "a"
 85  */
 86 ZmImapAccount.sortCompare =
 87 function(folderA, folderB) {
 88 	var check = ZmOrganizer.checkSortArgs(folderA, folderB);
 89 	if (check != null) { return check; }
 90 
 91 	var aId = ZmFolder.getIdForName(folderA.name);
 92 	var bId = ZmFolder.getIdForName(folderB.name);
 93 
 94 	if (ZmFolder.SORT_ORDER[aId] && ZmFolder.SORT_ORDER[bId]) {
 95 		return (ZmFolder.SORT_ORDER[aId] - ZmFolder.SORT_ORDER[bId]);
 96 	}
 97 	if (!ZmFolder.SORT_ORDER[aId] && ZmFolder.SORT_ORDER[bId]) { return 1; }
 98 	if (ZmFolder.SORT_ORDER[aId] && !ZmFolder.SORT_ORDER[bId]) { return -1; }
 99 	if (folderA.name.toLowerCase() > folderB.name.toLowerCase()) { return 1; }
100 	if (folderA.name.toLowerCase() < folderB.name.toLowerCase()) { return -1; }
101 	return 0;
102 };
