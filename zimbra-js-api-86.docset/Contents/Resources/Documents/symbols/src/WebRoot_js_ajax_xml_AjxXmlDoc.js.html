 41 /**
 42  * <strong>Note:</strong>
 43  * Anybody that uses these regular expressions MUST reset the <code>lastIndex</code>
 44  * property to zero or else the results are not guaranteed to be correct. You should
 45  * use {@link AjxXmlDoc.replaceInvalidChars} instead.
 46  * 
 47  * @private
 48  */
 49 AjxXmlDoc.INVALID_CHARS_RE = /[\u0000-\u0008\u000B-\u000C\u000E-\u001F\uD800-\uDFFF\uFFFE-\uFFFF]/g;
 50 AjxXmlDoc.REC_AVOID_CHARS_RE = /[\u007F-\u0084\u0086-\u009F\uFDD0-\uFDDF]/g;

 56 AjxXmlDoc._inited = false;
 57 AjxXmlDoc._msxmlVers = null;
 58 AjxXmlDoc._useDOM =
 59     Boolean(document.implementation && document.implementation.createDocument);
 60 AjxXmlDoc._useActiveX =
 61     !AjxXmlDoc._useDOM && Boolean(window.ActiveXObject);

 63 /**
 64  * Creates an XML doc.
 65  * 
 66  * @return	{AjxXmlDoc}	the XML doc
 67  */
 68 AjxXmlDoc.create =
 69 function() {
 70 	var xmlDoc = new AjxXmlDoc();
 71 	var newDoc = null;
 72 	if (AjxXmlDoc._useActiveX) {
 73 		newDoc = new ActiveXObject(AjxXmlDoc._msxmlVers);
 74 		newDoc.async = true; // Force Async loading
 75 		if (AjxXmlDoc._msxmlVers == "MSXML2.DOMDocument.4.0") {
 76 			newDoc.setProperty("SelectionLanguage", "XPath");
 77 			newDoc.setProperty("SelectionNamespaces", "xmlns:zimbra='urn:zimbra' xmlns:mail='urn:zimbraMail' xmlns:account='urn:zimbraAccount'");
 78 		}
 79 	} else if (AjxXmlDoc._useDOM) {
 80 		newDoc = document.implementation.createDocument("", "", null);
 81 	} else {
 82 		throw new AjxException("Unable to create new Doc", AjxException.INTERNAL_ERROR, "AjxXmlDoc.create");
 83 	}
 84 	xmlDoc._doc = newDoc;
 85 	return xmlDoc;
 86 }

101 /**
102  * Creates an XML doc from an XML string.
103  * 
104  * @param	{string}		xml		the XML string
105  * @return	{AjxXmlDoc}		the XML doc
106  */
107 AjxXmlDoc.createFromXml =
108 function(xml) {
109 	var xmlDoc = AjxXmlDoc.create();

111 	return xmlDoc;

114 /**
115  * Replaces invalid characters in the given string.
116  * 
117  * @param	{string}	s	the string
118  * @return	{string}	the resulting string
119  */
120 AjxXmlDoc.replaceInvalidChars = function(s) {
121 	AjxXmlDoc.INVALID_CHARS_RE.lastIndex = 0;
122 	return s.replace(AjxXmlDoc.INVALID_CHARS_RE, "?");
123 };

125 AjxXmlDoc.getXml =
126 function(node) {
127 	var ser = new XMLSerializer();
128 	return AjxXmlDoc.replaceInvalidChars(ser.serializeToString(node));
129 }

141 AjxXmlDoc.prototype.loadFromString =
142 function(str) {
143 	var doc = this._doc;
144 	doc.loadXML(str);
145 	if (AjxXmlDoc._useActiveX) {
146 		if (doc.parseError.errorCode != 0)
147 			throw new AjxException(doc.parseError.reason, AjxException.INVALID_PARAM, "AjxXmlDoc.loadFromString");
148 	}
149 }

151 AjxXmlDoc.prototype.loadFromUrl =
152 function(url) {
153 	if(AjxEnv.isChrome || AjxEnv.isSafari) {
154 		var xmlhttp = new window.XMLHttpRequest();
155 		xmlhttp.open("GET", url, false);
156 		xmlhttp.send(null);
157 		var xmlDoc = xmlhttp.responseXML;
158 		this._doc = xmlDoc;
159 	} else {
160 		this._doc.load(url);
161 	}
162 }

164 /**
165  * This function tries to create a JavaScript representation of the DOM. In some cases,
166  * it is easier to work with JS objects rather than do DOM lookups.
167  *
168  * <p>
169  * Rules:
170  * <ol>
171  * <li>The top-level tag gets lost; only it's content is seen important.</li>
172  * <li>Each node will be represented as a JS object.  It's textual content
173  *      will be saved in node.__msh_content (returned by <code>toString()</code>).</li>
174  * <li>Attributes get discarded.</li>
175  * <li>Each subnode will map to a property with its tagName in the parent
176  *      node <code>parent[subnode.tagName] == subnode</code></li>
177  * <li>If multiple nodes with the same tagName have the same parent node, then
178  *      <code>parent[tagName]</code> will be an array containing the objects, rather than a
179  *      single object.</li>
180  * </ol>
181  * 
182  * So what this function allows us to do is for instance this, starting with this XML doc:
183  *
184  * <pre>
185  * <error>
186  *   <code>404</code>
187  *   <name>Not Found</name>
188  *   <description>Page wasn't found on this server.</description>
189  * </error>
190  * </pre>
191  * 
192  * <pre>
193  * var obj = AjxXmlDoc.createFromXml(XML).toJSObject();
194  * alert(obj.code + " " + obj.name + " " + obj.description);
195  * </pre>
196  * 
197  * Here's an array example:
198  * <pre>
199  * <return>
200  *   <item>
201  *     <name>John Doe</name>
202  *     <email>foo@bar.com</email>
203  *   </item>
204  *   <item>
205  *     <name>Johnny Bravo</name>
206  *     <email>bravo@cartoonnetwork.com</email>
207  *   </item>
208  * </return>
209  * </pre>
210  * 
211  * <pre>
212  * var obj = AjxXmlDoc.createFromXml(XML).toJSObject();
213  * for (var i = 0; i < obj.item.length; ++i) {
214  *   alert(obj.item[i].name + " / " + obj.item[i].email);
215  * }
216  * </pre>
217  *
218  * Note that if there's only one <item> tag, then obj.item will be an object
219  * rather than an array.  And if there is no <item> tag, then obj.item will be
220  * undefined.  These are cases that the calling application must take care of.
221  */
222 AjxXmlDoc.prototype.toJSObject = 
223 function(dropns, lowercase, withAttrs) {
224 	_node = function() { this.__msh_content = ''; };
225 	_node.prototype.toString = function() { return this.__msh_content; };
226 	rec = function(i, o) {
227 		var tags = {}, t, n;
228 		for (i = i.firstChild; i; i = i.nextSibling) {
229 			if (i.nodeType == 1) {
230 				t = i.tagName;
231 				if (dropns)      t = t.replace(/^.*?:/, "");
232 				if (lowercase)   t = t.toLowerCase();
233 				n = new _node();
234 				if (tags[t]) {
235 					if (tags[t] == 1) {
236 						o[t] = [ o[t] ];
237 						tags[t] = 2;
238 					}
239 					o[t].push(n);
240 				} else {
241 					o[t] = n;
242 					tags[t] = 1;
243 				}
244 				//do attributes
245 				if(withAttrs) {
246 					if(i.attributes && i.attributes.length) {
247 						for(var ix = 0;ix<i.attributes.length;ix++) {
248 							attr = i.attributes[ix];
249 							n[attr.name] = AjxUtil.isNumeric(attr.value) ? attr.value : String(attr.value);
250 						}
251 					}
252 				}
253 				rec(i, n);
254 			} else if (i.nodeType == 3)
255 				o.__msh_content += i.nodeValue;
256 		}
257 	};
258 	var o = new _node();
259 	rec(this._doc.documentElement, o);
260 	return o;
261 };

263 AjxXmlDoc.prototype.getElementsByTagNameNS = 
264 function(ns, tag) {
265 	var doc = this.getDoc();
266 	return !doc.getElementsByTagNameNS
267 		? doc.getElementsByTagName(ns + ":" + tag)
268 		: doc.getElementsByTagNameNS(ns, tag);
269 };

272 function(ns, tag) {

274 };

282 AjxXmlDoc._init =
283 function() {
284 	if (AjxXmlDoc._useActiveX) {
285 		var msxmlVers = ["MSXML4.DOMDocument", "MSXML3.DOMDocument", "MSXML2.DOMDocument.4.0",
286 				 "MSXML2.DOMDocument.3.0", "MSXML2.DOMDocument", "MSXML.DOMDocument",
287 				 "Microsoft.XmlDom"];
288 		for (var i = 0; i < msxmlVers.length; i++) {
289 			try {
290 				new ActiveXObject(msxmlVers[i]);
291 				AjxXmlDoc._msxmlVers = msxmlVers[i];
292 				break;
293 			} catch (ex) {
294 			}
295 		}
296 		if (!AjxXmlDoc._msxmlVers) {
297 			throw new AjxException("MSXML not installed", AjxException.INTERNAL_ERROR, "AjxXmlDoc._init");
298 		}
299 	} else if (!Document.prototype.loadXML) {
300 		// add loadXML to Document's API
301 		Document.prototype.loadXML = function(str) {
302 			var domParser = new DOMParser();
303 			var domObj = domParser.parseFromString(str, "text/xml");
304 			// remove old child nodes since we recycle DOMParser and append new
305 			while (this.hasChildNodes()) {
306 				this.removeChild(this.lastChild);
307 			}
308 			var len = domObj.childNodes.length;
309             for (var i = 0; i < len; i++) {
310                 var importedNode = this.importNode(domObj.childNodes[i], true);
311                 this.appendChild(importedNode);
312             }
313 		}
314 		
315 		if (AjxEnv.isNav) {
316 			_NodeGetXml = function() {
317 				var ser = new XMLSerializer();
318 				return ser.serializeToString(this);
319 			}
320 			Node.prototype.__defineGetter__("xml", _NodeGetXml);
321 		}
322 	}
323 	
324 	AjxXmlDoc._inited = true;
325 };

327 AjxXmlDoc.prototype.set =
328 function(name, value, element) {
329    var p = this._doc.createElement(name);
330       if (value != null) {
331          var cdata = this._doc.createTextNode("");
332          p.appendChild(cdata);
333          cdata.nodeValue = value;
334       }
335       if (element == null) {
336          this.root.appendChild(p);
337       } else {
338          element.appendChild(p);
339       }
340    return p;
341 };

343 AjxXmlDoc.prototype.getDocXml =
344 function() {
345    if (AjxEnv.isSafari)
346       return AjxXmlDoc.getXml(this.getDoc());
347    else
348       return AjxXmlDoc.replaceInvalidChars(this.getDoc().xml);
349 };

351 /**
352  * Creates an XML document with a root element.
353  * 
354  * @param	{string}	rootName	the root name
355  * @return	{AjxXmlDoc}	the XML document
356  */
357 AjxXmlDoc.createRoot =
358 function(rootName) {
359    var xmldoc = AjxXmlDoc.create();
360    var d = xmldoc.getDoc();
361    xmldoc.root = d.createElement(rootName);
362 
363    d.appendChild(xmldoc.root);
364    return xmldoc;
365 };
366 
367 /**
368  * Creates an XML document with the element.
369  * 
370  * @param	{string}	name	the element name
371  * @param	{string}	value	the element value
372  * @return	{AjxXmlDoc}	the XML document
373  */
374 AjxXmlDoc.createElement =
375 function(name, value) {
376 	
377    var xmldoc = AjxXmlDoc.create();
378    var d = xmldoc.getDoc();
379    xmldoc.root = d.createElement(name);
380    if (value != null) {
381    		//xmldoc.root.nodeValue = value;
382    	 	var cdata = d.createTextNode("");
383         xmldoc.root.appendChild(cdata);
384         cdata.nodeValue = value;
385    }
386    
387    d.appendChild(xmldoc.root);
388    return xmldoc;
389    
390 };

393 AjxXmlDoc.prototype.appendChild =
394 function(xmldoc){
395    //Security Exception WRONG_DOCUMENT_ERR thrown when we append nodes created of diff. documents
396    //Chrome/Safari does not like it.
397    if(this._doc != xmldoc._doc && ( AjxEnv.isChrome || AjxEnv.isSafari )){
398         this.root.appendChild(this.getDoc().importNode(xmldoc.root, true));
399    }else{
400         this.root.appendChild(xmldoc.root);
401    }
402 };
