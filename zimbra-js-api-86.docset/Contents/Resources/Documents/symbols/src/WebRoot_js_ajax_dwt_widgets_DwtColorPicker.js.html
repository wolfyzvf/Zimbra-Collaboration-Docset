 23 /**
 24  * Creates a color picker displaying "Web safe" colors.
 25  * @constructor
 26  * @class
 27  * Instances of this class may be
 28  * used with {@link DwtMenu} to create a {@link DwtColorPicker} menu. Clicking on a color cell generates a
 29  * DwtSelectionEvent the detail attribute of which contains the color string associated
 30  * the cell on which the user clicked
 31  *
 32  *
 33  * @author Ross Dargahi
 34  * 
 35  * @param {hash}	params		a hash of parameters
 36  * @param {DwtComposite} params.parent		the parent widget
 37  * @param {string}       params.className	a CSS class
 38  * @param {constant}     params.posStyle	the positioning style
 39  * @param {boolean}      params.hideNoFill  True to hide the no-fill/use-default option
 40  * @param {string}       params.noFillLabel			the no-fill label
 41  * @param {boolean}      params.allowColorInput		if <code>true</code>, allow a text field to allow user to input their customized RGB value
 42  * @param {string}       params.defaultColor Default color.
 43  * 
 44  * @extends		DwtControl
 45  */
 46 DwtColorPicker = function(params) {
 47 	if (arguments.length == 0) return;
 48     params = Dwt.getParams(arguments, DwtColorPicker.PARAMS);
 49 
 50 	params.className = params.className || "DwtColorPicker";
 51 	DwtComposite.call(this, params);
 52 
 53     this._hideNoFill = params.hideNoFill;
 54 	this._noFillLabel = params.noFillLabel;
 55     this._allowColorInput = params.allowColorInput;
 56     this._defaultColor = params.defaultColor || "#000000";
 57     this._createHtml();
 58 };
 59 
 60 DwtColorPicker.prototype = new DwtComposite;
 61 DwtColorPicker.prototype.constructor = DwtColorPicker;

 63 DwtColorPicker.prototype.toString = function() {
 64 	return "DwtColorPicker";

 73 // RE to parse out components out of a "rgb(r, g, b);" string
 74 DwtColorPicker._RGB_RE = /^rgb\(([0-9]{1,3}),\s*([0-9]{1,3}),\s*([0-9]{1,3})\)$/;
 75 DwtColorPicker._HEX_RE = /^\#([0-9a-f]{2})([0-9a-f]{2})([0-9a-f]{2})$/i;

113 DwtColorPicker.prototype._createHtml = function(templateId) {
114     this._createHtmlFromTemplate(templateId||this.TEMPLATE, {id:this.getHtmlElement().id});
115 };

117 DwtColorPicker.prototype._createHtmlFromTemplate = function(templateId, data) {
118     data.allowColorInput = this._allowColorInput;
119     data.hideNoFill = this._hideNoFill;
120     data.noFillLabel = this._noFillLabel;
121     DwtComposite.prototype._createHtmlFromTemplate.apply(this, arguments);
122 
123     // create controls
124     if (data.allowColorInput) {
125         var inputEl = document.getElementById(data.id+"_input");
126 	var inputParams = {
127 		parent: this,
128 		validationStyle: DwtInputField.CONTINUAL_VALIDATION, //update the preview for each key up 
129 		errorIconStyle: DwtInputField.ERROR_ICON_RIGHT, 
130 		validator: DwtColorPicker.__isValidInputValue
131 	};
132         var input = this._colorInput = new DwtInputField(inputParams);
133         input.replaceElement(inputEl);
134 	// Add  callback for update the preview when the input value is validated.
135 	var updateCallback = new AjxCallback(this, this._updatePreview);
136 	input.setValidationCallback(updateCallback);
137 	
138 	var error = this._error = new DwtLabel({parent:this});
139 	var errorEl = document.getElementById(data.id+"_error");
140         error.replaceElement(errorEl);
141         error.setVisible(false);
142         
143 	this._preview = document.getElementById(data.id+"_preview");
144         
145 	var buttonEl = document.getElementById(data.id+"_button");
146         var button = new DwtButton({parent:this});
147         button.setText(AjxMsg.setColor);
148         button.replaceElement(buttonEl);
149         button.addSelectionListener(new AjxListener(this, this._handleSetColor));
150     }
151 
152     var buttonEl = document.getElementById(data.id+"_default");
153     if (buttonEl) {
154         if (!DwtColorPicker.Button) {
155             DwtColorPicker.__defineClasses();
156         }
157         var button = this._defaultColorButton = new DwtColorPicker.Button({parent:this});
158         button.setText(data.noFillLabel || AjxMsg.colorsUseDefault);
159         button.replaceElement(buttonEl);
160         button.addSelectionListener(new AjxListener(this, this._handleColorSelect, [0]));
161     }
162 
163     // set color handlers
164     var colorsEl = document.getElementById(data.id+"_colors");
165     var mouseOver = AjxEnv.isIE ? DwtEvent.ONMOUSEENTER : DwtEvent.ONMOUSEOVER;
166     var mouseOut  = AjxEnv.isIE ? DwtEvent.ONMOUSELEAVE : DwtEvent.ONMOUSEOUT;
167 
168     Dwt.setHandler(colorsEl, DwtEvent.ONMOUSEDOWN, AjxCallback.simpleClosure(this._handleMouseDown, this));
169     Dwt.setHandler(colorsEl, DwtEvent.ONMOUSEUP, AjxCallback.simpleClosure(this._handleMouseUp, this));
170     Dwt.setHandler(colorsEl, mouseOver, AjxCallback.simpleClosure(this._handleMouseOver, this));
171     Dwt.setHandler(colorsEl, mouseOut, AjxCallback.simpleClosure(this._handleMouseOut, this));
172 };
173 
174 DwtColorPicker.prototype._handleMouseOver = function(htmlEvent) {
175     var event = DwtUiEvent.getEvent(htmlEvent);
176     var target = DwtUiEvent.getTarget(event);
177     if (!Dwt.hasClass(target, "Color")) return;
178 
179     this._handleMouseOut(htmlEvent);
180     Dwt.addClass(target, DwtControl.HOVER);
181     this._mouseOverEl = target;
182 };
183 
184 DwtColorPicker.prototype._handleMouseOut = function(htmlEvent) {
185     if (this._mouseOverEl) {
186         Dwt.delClass(this._mouseOverEl, DwtControl.HOVER);
187     }
188     this._mouseOverEl = null;
189 };
190 
191 DwtColorPicker.prototype._handleMouseDown = function(htmlEvent) {
192     var event = DwtUiEvent.getEvent(htmlEvent);
193     var target = DwtUiEvent.getTarget(event);
194     this._mouseDownEl = Dwt.hasClass(target, "Color") ? target : null;
195 };
196 DwtColorPicker.prototype._handleMouseUp = function(htmlEvent) {
197     var event = DwtUiEvent.getEvent(htmlEvent);
198     var target = DwtUiEvent.getTarget(event);
199     if (this._mouseDownEl != target) return;
200 
201     var cssColor = DwtCssStyle.getProperty(target, "background-color");
202     this._handleColorSelect(DwtColorPicker.__color2hex(cssColor));
203 };
204 
205 DwtColorPicker.prototype._handleSetColor = function(evt) {
206     var color = this._colorInput.getValue();
207     if (color) {
208 	color = DwtColorPicker.__color2hex(color);
209 	if(!color) 
210 		return; 
211     	this._handleColorSelect(color);
212     }
213 };
214 
215 DwtColorPicker.prototype._handleColorSelect = function(color) {
216     this._inputColor = color;
217 
218     // If our parent is a menu then we need to have it close
219     if (this.parent instanceof DwtMenu) {
220         DwtMenu.closeActiveMenu();
221     }
222 
223     // Call Listeners on mouseEv.target.id
224     if (this.isListenerRegistered(DwtEvent.SELECTION)) {
225         var selEvent = DwtShell.selectionEvent;
226 //        DwtUiEvent.copy(selEvent, htmlEvent);
227         selEvent.item = this;
228         selEvent.detail = this._inputColor;
229         this.notifyListeners(DwtEvent.SELECTION, selEvent);
230     }
231 };

248 DwtColorPicker.__color2hex = function(s) {
249 	//in IE we can't get the calculated value so for white/black we get white/black (of course it could be set the the hex value in the markup but this is more bulletproof to make sure here)
250 	if (s == "white") {
251 		return "#FFFFFF";
252 	}
253 	if (s == "black") {
254 		return "#000000";
255 	}
256 
257     var m = s && s.match(DwtColorPicker._RGB_RE);
258     if (m) {
259 	// each component should be in range of (0 - 255)
260 	for( var i = 1; i <= 3; i++ ) {
261 		if(parseInt(m[i]) > 255)
262 			return "";
263 	}
264         return AjxColor.color(m[1], m[2], m[3]);
265     }
266     m = s && s.match(DwtColorPicker._HEX_RE);
267     if (m) {
268         return s;
269     }
270     return "";
271 };
272 
273 DwtColorPicker.__isValidInputValue = function(s) {
274    // null is valid for we consider the condition
275    // the user delete all the word it has been input
276    if (!s)
277 	return s;
278    var r = DwtColorPicker.__color2hex(s);
279    if (!r) { 
280 	throw AjxMsg.colorFormatError;	
281    }
282    return s;	
283 };

285 DwtColorPicker.prototype._updatePreview = function(inputelement, isValid, value){
286    if (isValid) {
287 	value = DwtColorPicker.__color2hex(value);
288 	Dwt.setVisible(this._preview, true);
289 	this._preview.style.backgroundColor = value;
290 	this._error.setVisible(false);
291    }
292    else {
293 	Dwt.setVisible(this._preview, false);
294 	this._error.setVisible(true);
295 	this._error.setText(AjxMsg.colorFormatError);
296    }
297 };
