 44 ZmCalMgr.prototype._createMiniCalendar =
 45 function(date) {
 46 	date = date ? date : new Date();
 47 
 48 	var firstDayOfWeek = appCtxt.get(ZmSetting.CAL_FIRST_DAY_OF_WEEK) || 0;
 49 
 50     //todo: need to use server setting to decide the weekno standard
 51     var serverId = AjxTimezone.getServerId(AjxTimezone.DEFAULT);
 52     var useISO8601WeekNo = (serverId && serverId.indexOf("Europe")==0 && serverId != "Europe/London");
 53 
 54 	this._miniCalendar = new DwtCalendar({parent: this._container, posStyle:DwtControl.ABSOLUTE_STYLE,
 55 										  firstDayOfWeek: firstDayOfWeek, showWeekNumber: appCtxt.get(ZmSetting.CAL_SHOW_CALENDAR_WEEK), useISO8601WeekNo: useISO8601WeekNo});
 56 	this._miniCalendar.setDate(date);
 57 	this._miniCalendar.setScrollStyle(Dwt.CLIP);
 58 	this._miniCalendar.addSelectionListener(new AjxListener(this, this._miniCalSelectionListener));
 59 	this._miniCalendar.addActionListener(new AjxListener(this, this._miniCalActionListener));
 60 	this._miniCalendar.addDateRangeListener(new AjxListener(this, this._miniCalDateRangeListener));
 61 	this._miniCalendar.setMouseOverDayCallback(new AjxCallback(this, this._miniCalMouseOverDayCallback));
 62 	this._miniCalendar.setMouseOutDayCallback(new AjxCallback(this, this._miniCalMouseOutDayCallback));
 63 
 64 	var list = [];
 65 	if (appCtxt.get(ZmSetting.MAIL_ENABLED)) {
 66 		list.push("ZmMailMsg");
 67 		list.push("ZmConv");
 68 	}
 69 	if (appCtxt.get(ZmSetting.CONTACTS_ENABLED)) {
 70 		list.push("ZmContact");
 71 	}
 72 	this._miniCalDropTarget = new DwtDropTarget(list);
 73 	this._miniCalDropTarget.addDropListener(new AjxListener(this, this._miniCalDropTargetListener));
 74 	this._miniCalendar.setDropTarget(this._miniCalDropTarget);
 75 
 76 	var workingWeek = [];
 77 	for (var i = 0; i < 7; i++) {
 78 		var d = (i + firstDayOfWeek) % 7;
 79 		workingWeek[i] = (d > 0 && d < 6);
 80 	}
 81 	this._miniCalendar.setWorkingWeek(workingWeek);
 82 
 83 	// add mini-calendar to skin
 84 	var components = {};
 85 	components[ZmAppViewMgr.C_TREE_FOOTER] = this._miniCalendar;
 86 	appCtxt.getAppViewMgr().setViewComponents(ZmAppViewMgr.GLOBAL, components, true);
 87 	
 88 	var app = appCtxt.getApp(ZmApp.CALENDAR);
 89 	var show = app._active || appCtxt.get(ZmSetting.CAL_ALWAYS_SHOW_MINI_CAL);
 90 	this._miniCalendar.setSkipNotifyOnPage(show && !app._active);
 91 	if (!app._active) {
 92 		this._miniCalendar.setSelectionMode(DwtCalendar.DAY);
 93 	}
 94 
 95     this._dayRollTimer = null;
 96 };

100 ZmCalMgr.prototype.startDayRollTimer =
101 function(){
102   if(!this._dayRollTimer){
103     var curTime = new Date();
104     var rollTime = new Date();
105     rollTime.setHours(23,59,59,999);
106     var interval = rollTime.getTime() - curTime.getTime();
107     var dayRollAction = new AjxTimedAction(this,this._rollDay);
108     AjxTimedAction.scheduleAction(dayRollAction,interval);
109   }
110 }

164 ZmCalMgr.prototype._getMiniCalActionMenu =
165 function() {
166 	if (this._minicalMenu == null) {
167 
168 		this.postInitListeners();
169 
170 		var list = [ZmOperation.NEW_APPT, ZmOperation.NEW_ALLDAY_APPT, ZmOperation.SEP, ZmOperation.SEARCH_MAIL];
171 		//Zimlet hack
172 		var zimletOps = ZmZimlet.actionMenus ? ZmZimlet.actionMenus["ZmCalViewController"] : null;
173 		if (zimletOps && zimletOps.length) {
174 			for (var i = 0; i < zimletOps.length; i++) {
175 				var op = zimletOps[i];
176 				ZmOperation.defineOperation(null, op);
177 				list.push(op.id);
178 			}
179 		}
180 		var params = {parent: appCtxt.getShell(), menuItems:list};
181 		this._minicalMenu = new ZmActionMenu(params);
182 		list = this._minicalMenu.opList;
183 		var cnt = list.length;
184 		for(var ix=0; ix < cnt; ix++) {
185 			if(this._listeners[list[ix]]) {
186 				this._minicalMenu.addSelectionListener(list[ix], this._listeners[list[ix]]);
187 			}
188 		}
189 	}
190 	return this._minicalMenu;
191 };

193 // Zimlet hack
194 ZmCalMgr.prototype.postInitListeners =
195 function () {
196 	if (ZmZimlet.listeners && ZmZimlet.listeners["ZmCalViewController"]) {
197 		for (var ix in ZmZimlet.listeners["ZmCalViewController"]) {
198 			if (ZmZimlet.listeners["ZmCalViewController"][ix] instanceof AjxListener) {
199 				this._listeners[ix] = ZmZimlet.listeners["ZmCalViewController"][ix];
200 			} else {
201 				this._listeners[ix] = new AjxListener(this, this._proxyListeners, [ZmZimlet.listeners["ZmCalViewController"][ix]]);
202 			}
203 		}
204 	}
205 };

251 ZmCalMgr.prototype._getDayToolTipOnDelay =
252 function(control, day) {
253 	if (!this._currentMouseOverDay) { return; }
254 	if ((this._currentMouseOverDay.getDate() == day.getDate()) &&
255 		(this._currentMouseOverDay.getMonth() == day.getMonth()))
256 	{
257 		this._currentMouseOverDay = null;
258         var mouseEv = DwtShell.mouseEvent;
259         if(mouseEv && mouseEv.docX > 0 && mouseEv.docY > 0) {
260             var callback = new AjxCallback(this, this.showTooltip, [control, mouseEv.docX, mouseEv.docY]);
261             this.getCalViewController().getDayToolTipText(day, false, callback, true);
262         }
263 	}
264 };

278 ZmCalMgr.prototype.getApptSummaries =
279 function(params) {
280 	var apptVec = this.setSearchParams(params);
281 
282 	if (apptVec != null && (apptVec instanceof AjxVector)) {
283 		return apptVec;
284 	}
285 
286 	// this array will hold a list of appts as we collect them from the server
287 	this._rawAppts = [];
288 
289 	if (params.callback) {
290 		this._search(params);
291 	} else {
292 		return this._search(params);
293 	}
294 };

296 ZmCalMgr.prototype.setSearchParams =
297 function(params) {
298 	if (!(params.folderIds instanceof Array)) {
299 		params.folderIds = [params.folderIds];
300 	} else if (params.folderIds.length == 0) {
301 		var newVec = new AjxVector();
302 		if (params.callback) {
303 			params.callback.run(newVec);
304 		}
305 		return newVec;
306 	}
307 
308 	var folderIdMapper = {};
309 	var query = "";
310 	for (var i=0; i < params.folderIds.length; i++) {
311 		var fid = params.folderIds[i];
312 		var systemFolderId = appCtxt.getActiveAccount().isMain
313 			? fid : ZmOrganizer.getSystemId(fid);
314 
315 		// map remote folder ids into local ones while processing search since
316 		// server wont do it for us (see bug 7083)
317 		var folder = appCtxt.getById(systemFolderId);
318 		var rid = folder ? folder.getRemoteId() : systemFolderId;
319 		folderIdMapper[rid] = systemFolderId;
320 
321 		if (query.length) {
322 			query += " OR ";
323 		}
324 		var idText = AjxUtil.isNumeric(fid) ? fid : ['"', fid, '"'].join("");
325 		query += "inid:" + idText;
326 		
327 	}
328 	params.queryHint = query;
329     params.needToFetch = params.folderIds;
330 	params.folderIdMapper = folderIdMapper;
331 	params.offset = 0;
332 };

334 ZmCalMgr.prototype._search =
335 function(params) {
336 	var jsonObj = {SearchRequest:{_jsns:"urn:zimbraMail"}};
337 	var request = jsonObj.SearchRequest;
338 
339 	this._setSoapParams(request, params);
340 
341     var calController = this.getCalViewController();
342     var apptCache     = calController.getApptCache();
343 
344     var accountName = appCtxt.multiAccounts ? appCtxt.accountList.mainAccount.name : null;
345 	if (params.callback) {
346 		appCtxt.getAppController().sendRequest({
347 			jsonObj: jsonObj,
348 			asyncMode: true,
349 			callback: (new AjxCallback(this, this._getApptSummariesResponse, [params])),
350             offlineCallback: apptCache.offlineSearchAppts(null, null, params),
351 			noBusyOverlay: params.noBusyOverlay,
352 			accountName: accountName
353 		});
354 	} else {
355 		var response = appCtxt.getAppController().sendRequest({jsonObj: jsonObj, accountName: accountName});
356 		var result = new ZmCsfeResult(response, false);
357 		return this._getApptSummariesResponse(params, result);
358 	}
359 };

361 ZmCalMgr.prototype._setSoapParams =
362 function(request, params) {	
363 	request.sortBy = "none";
364 	request.limit = "500";
365 	request.calExpandInstStart = params.start;
366 	request.calExpandInstEnd = params.end;
367 	request.types = ZmSearch.TYPE[ZmItem.APPT];
368 	request.offset = params.offset;
369 
370 	var query = params.query;
371 	if (params.queryHint) {
372 		query = (query != null)
373 			? (query + " (" + params.queryHint + ")")
374 			: params.queryHint;
375 	}
376 	request.query = {_content:query};
377 };

380 ZmCalMgr.prototype._getApptSummariesResponse =
381 function(params, result) {
382 	// TODO: mark both as needing refresh?
383 	if (!result) { return; }
384 
385 	var callback = params.callback;
386 	var resp;
387 	try {
388 		resp = result.getResponse();
389 	} catch (ex) {
390 		if (callback) {
391 			callback.run(new AjxVector());
392 		}
393 		return;
394 	}
395 
396 	var searchResp = resp.SearchResponse;
397 	var newList = this.processSearchResponse(searchResp, params);
398 	if (newList == null) { return; }
399 
400 	if (callback) {
401 		callback.run(newList, params.query);
402 	} else {
403 		return newList;
404 	}
405 };

407 ZmCalMgr.prototype.processSearchResponse = 
408 function(searchResp, params) {
409 	if(!searchResp) { return; }
410 
411 	if (searchResp && searchResp.appt && searchResp.appt.length) {
412 		this._rawAppts = this._rawAppts != null 
413 			? this._rawAppts.concat(searchResp.appt)
414 			: searchResp.appt;
415 
416 		// if "more" flag set, keep requesting more appts
417 		if (searchResp.more) {
418 			var lastAppt = searchResp.appt[searchResp.appt.length-1];
419 			if (lastAppt) {
420 				params.offset += 500;
421 				this._search(params);
422 				return;
423 			}
424 		}
425 	}
426 
427 	var newList = new AjxVector();
428 	if (this._rawAppts && this._rawAppts.length) {
429 		this._list = new ZmList(ZmItem.APPT);
430 		for (var i = 0; i < this._rawAppts.length; i++) {
431 			DBG.println(AjxDebug.DBG2, "appt[j]:" + this._rawAppts[i].name);
432 			var apptNode = this._rawAppts[i];
433 			var instances = apptNode ? apptNode.inst : null;
434 			if (instances) {
435 				var args = {list:this._list};
436 				for (var j = 0; j < instances.length; j++) {
437 					var appt = ZmCalBaseItem.createFromDom(apptNode, args, instances[j]);
438 					DBG.println(AjxDebug.DBG2, "lite appt :" + appt);
439 					if (appt) newList.add(appt);
440 				}
441 			}
442 		}
443 
444 	}
445 	return newList;
446 };

556 ZmCalMgr.prototype._quickReminderCallback =
557 function(list) {
558     var newList = new AjxVector();
559     this._cacheMap = {};
560     var size = list.size();
561 
562     var currentTime  = (new Date()).getTime();
563 
564     for (var i = 0; i < size; i++) {
565         var appt = list.get(i);
566         var id = appt.id;
567         if (!this._cacheMap[id]) {
568             this._cacheMap[id] = appt;
569             if(appt.isAllDayEvent()) continue;
570             var diff = appt.getStartTime() - currentTime;
571             var isUpcomingEvent = (diff >= 0 && diff <= AjxDateUtil.MSEC_PER_HOUR)
572             if((currentTime >= appt.getStartTime() && currentTime <= appt.getEndTime()) || isUpcomingEvent) {
573                 appt.isUpcomingEvent = isUpcomingEvent;
574                 newList.add(appt);
575             }
576         }
577     }
578 
579     var qDlg = this.getQuickReminderDialog();
580     qDlg.initialize(newList);
581     qDlg.popup();
582 };
