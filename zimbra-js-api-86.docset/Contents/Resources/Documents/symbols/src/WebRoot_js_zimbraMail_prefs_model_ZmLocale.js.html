 22 /**
 23  * Default constructor.
 24  * @class
 25  * This class represents a locale.
 26  * 
 27  * @param	{String}	id		the id
 28  * @param	{String}	name	the name
 29  * @param	{String}	image	the image
 30  * @param	{String}	localName	the name in user's locale
 31  *
 32  * @see		ZmLocale.create
 33  */
 34 ZmLocale = function(id, name, image, localName) {
 35 	this.id = id;
 36 	this.name = name;
 37 	this.localName = localName;
 38 	this._image = image;
 39 };
 40 
 41 //List of RTL supporting languages
 42 ZmLocale.RTLLANGUAGES = {
 43     ar:"Arabic",
 44     iw:"Hebrew"
 45 };
 46 
 47 ZmLocale.localeMap = {};
 48 ZmLocale.languageMap = {};
 49 
 50 /**
 51  * Creates the locale.
 52  * 
 53  * @param	{String}	id		the locale id (for example, <code>en_US</code>)
 54  * @param	{String}	name	the locale name
 55  * @param	{String}	localName	the name in user's locale
 56  */
 57 ZmLocale.create =
 58 function(id, name, localName) {
 59 	var index = id.indexOf("_");
 60 	var languageId;
 61 	var country = null;
 62 	if (index == -1) {
 63 		languageId = id;
 64 	}
 65 	else {
 66 		languageId = id.substr(0, index);
 67 		country = id.substring(id.length - 2);
 68 	}
 69 
 70 	var languageObj = ZmLocale.languageMap[languageId];
 71 	if (!languageObj) {
 72 		languageObj = new ZmLocale(languageId, name, null, localName);
 73 		ZmLocale.languageMap[languageId] = languageObj;
 74 		ZmLocale.localeMap[id] = languageObj;
 75 	}
 76 	if (country) {
 77 		var localeObj = new ZmLocale(id, name, null, localName);
 78 		languageObj._add(localeObj);
 79 		ZmLocale.localeMap[id] = localeObj;
 80 		return localeObj;
 81 	}
 82 	else {
 83 		languageObj.name = name;
 84 		return languageObj;
 85 	}
 86 };

 88 /**
 89  * Checks if there are more than one selectable locale.
 90  * 
 91  * @return	{Boolean}	<code>true</code> if there are more than one selectable locale
 92  */
 93 ZmLocale.hasChoices =
 94 function() {
 95 	var count = 0;
 96 	for (var id in ZmLocale.localeMap) {
 97 		var locale = ZmLocale.localeMap[id]; 
 98 		if (!locale.locales) {
 99 			count++;
100 		}
101 		if (count >= 2) {
102 			return true;
103 		}
104 	}
105 	return false;
106 };

108 /**
109  * Gets the image.
110  * 
111  * @return	{String}	the image
112  */
113 ZmLocale.prototype.getImage =
114 function() {
115 	return this._image;
116 };

118 /**
119  * Gets the name in both the locale itself, and in the local (user) locale. 
120  *
121  * @return	{String}	the name
122  */
123 ZmLocale.prototype.getNativeAndLocalName =
124 function() {
125 	if (this.name == this.localName) {
126 		/* don't show both if they are the same - it looks extremely funny */
127 		return this.name;
128 	}
129 	return [this.localName, " - ", this.name].join("");
130 };
131 
132 ZmLocale.prototype._add =
133 function(locale) {
134 	(this.locales = this.locales || []).push(locale);
135 };
