 63 ZmApptController.prototype._createToolBar =
 64 function() {
 65 
 66 	var buttons = [ ZmOperation.SEND_INVITE, ZmOperation.SAVE, ZmOperation.CANCEL, ZmOperation.SEP,
 67                     ZmOperation.TAG_MENU
 68                     ];
 69     var secondaryButtons = [ZmOperation.EDIT, ZmOperation.DUPLICATE_APPT, ZmOperation.SEP,
 70                             ZmOperation.REPLY, ZmOperation.REPLY_ALL, ZmOperation.FORWARD_APPT, ZmOperation.PROPOSE_NEW_TIME, ZmOperation.DELETE, ZmOperation.SEP,
 71                             ZmOperation.SHOW_ORIG

 77 	this._toolbar = new ZmButtonToolBar({parent:this._container, buttons:buttons, context:this._currentViewId, controller:this, secondaryButtons:secondaryButtons});

 80 	this._toolbar.addSelectionListener(ZmOperation.REPLY, this._replyListener.bind(this));
 81 	this._toolbar.addSelectionListener(ZmOperation.REPLY_ALL, this._replyAllListener.bind(this));
 82 	this._toolbar.addSelectionListener(ZmOperation.FORWARD_APPT, this._forwardListener.bind(this));
 83 	this._toolbar.addSelectionListener(ZmOperation.EDIT, this._editListener.bind(this));
 84 	this._toolbar.addSelectionListener(ZmOperation.PROPOSE_NEW_TIME, this._proposeTimeListener.bind(this));
 85 	this._toolbar.addSelectionListener(ZmOperation.DELETE, this._deleteListener.bind(this));
 86 	this._toolbar.addSelectionListener(ZmOperation.DUPLICATE_APPT, this._duplicateApptListener.bind(this));
 87 	this._toolbar.addSelectionListener(ZmOperation.SHOW_ORIG, this._showOrigListener.bind(this));

 89 	if (appCtxt.get(ZmSetting.PRINT_ENABLED)) {
 90 		this._toolbar.addSelectionListener(ZmOperation.PRINT, this._printListener.bind(this));

 92 
 93     var sendButton = this._toolbar.getButton(ZmOperation.SEND_INVITE);

 95

104 	if (spellCheckButton) {
105 		spellCheckButton.setAlign(DwtLabel.IMAGE_LEFT | DwtButton.TOGGLE_STYLE);

110 ZmApptController.prototype._initToolbar =
111 function(mode) {
112     ZmCalItemComposeController.prototype._initToolbar.call(this, mode);
113     var saveButton = this._toolbar.getButton(ZmOperation.SAVE);
114     saveButton.setEnabled(false);
115 
116     var editButton = this._toolbar.getButton(ZmOperation.EDIT),
117         forwardApptButton,
118         deleteButton;
119 
120     if (editButton) {
121         if (mode === ZmCalItem.MODE_EDIT_SINGLE_INSTANCE) {
122             if (editButton.getText() !== ZmMsg.editInstance) {
123                 editButton.setText(ZmMsg.editInstance);
124                 forwardApptButton = this._toolbar.getButton(ZmOperation.FORWARD_APPT);
125                 deleteButton = this._toolbar.getButton(ZmOperation.DELETE);
126                 forwardApptButton && forwardApptButton.setText(ZmMsg.forwardInstance);
127                 deleteButton && deleteButton.setText(ZmMsg.deleteApptInstance);
128             }
129         }
130         else if (mode === ZmCalItem.MODE_EDIT_SERIES) {
131             if (editButton.getText() !== ZmMsg.editSeries) {
132                 editButton.setText(ZmMsg.editSeries);
133                 forwardApptButton = this._toolbar.getButton(ZmOperation.FORWARD_APPT);
134                 deleteButton = this._toolbar.getButton(ZmOperation.DELETE);
135                 forwardApptButton && forwardApptButton.setText(ZmMsg.forwardSeries);
136                 deleteButton && deleteButton.setText(ZmMsg.deleteApptSeries);
137             }
138         }
139         else {
140             if (editButton.getText() !== ZmMsg.edit) {
141                 editButton.setText(ZmMsg.edit);
142                 forwardApptButton = this._toolbar.getButton(ZmOperation.FORWARD_APPT);
143                 deleteButton = this._toolbar.getButton(ZmOperation.DELETE);
144                 forwardApptButton && forwardApptButton.setText(ZmMsg.forward);
145                 deleteButton && deleteButton.setText(ZmMsg.del);
146             }
147         }
148     }
149 
150     // bug 68451: disabling edit options for trashed appointments
151     var calItem = this.getCalItem();
152     var calendar = calItem && calItem.getFolder();
153     var isTrash = calendar && calendar.nId==ZmOrganizer.ID_TRASH;
154     var isReadOnly = calendar && calendar.isReadOnly();
155 
156     if(isTrash){
157         this._disableEditForTrashedItems();
158     }
159 
160     if (appCtxt.isWebClientOffline()) {
161         this._disableEditForOffline();
162     }
163     if (isReadOnly) {
164         this._disableActionsForReadOnlyAppt();
165     }
166     if (appCtxt.isExternalAccount()) {
167         this._disableActionsForExternalAccount();
168     }
169 };

253 ZmApptController.prototype._deleteListener =
254 function(ev) {
255 	var op = this.getMode();
256     var calItem = this.getCalItem();
257     if(!calItem) {
258         return;
259     }
260 
261     if (calItem.isRecurring()) {
262         var mode = (op == ZmCalItem.MODE_EDIT_SINGLE_INSTANCE)
263             ? ZmCalItem.MODE_DELETE_INSTANCE
264             : ZmCalItem.MODE_DELETE_SERIES;
265         this._app.getCalController()._promptDeleteAppt(calItem, mode);
266     }
267     else {
268         this._app.getCalController()._deleteAppointment(calItem);
269     }
270 };

305 ZmApptController.prototype._saveListener =
306 function(ev) {
307     if(!this.isDirty() || !this.getOpValue()) {
308         return;
309     }
310 	var op = (ev && ev.item instanceof DwtMenuItem)
311 		? ev.item.getData(ZmOperation.KEY_ID) : null;
312 
313 
314     var calItem = this.getCalItem();
315     if(!calItem) {
316         return;
317     }
318     if (calItem.isRecurring() && !op) {
319         var mode = this.getMode();
320         op = (mode == ZmCalItem.MODE_EDIT_SINGLE_INSTANCE) ? ZmOperation.VIEW_APPT_INSTANCE : ZmOperation.VIEW_APPT_SERIES;
321     }
322 
323     var saveCallback = new AjxCallback(this, this._handleSaveResponse);
324     var calViewCtrl = this._app.getCalController();
325 	// This will trigger a call to  ZmMailMsg.sendInviteReply, which updates the offline appointment ptst field
326 	// and the invite mail msg.
327     var respCallback = new AjxCallback(calViewCtrl, calViewCtrl._handleResponseHandleApptRespondAction, [calItem, this.getOpValue(), op, saveCallback]);
328 	calItem.getDetails(null, respCallback, this._errorCallback);
329 
330     //this._app.getCalController()._replyAppointment(calItem, true);
331 };

350 ZmApptController.prototype._handleSaveResponse =
351 function(result, value) {
352     if (appCtxt.isWebClientOffline()) {
353         // Set the value of the appt stored in-memory in the list.  Normally, this would be updated
354         // by a notification, but not offline.
355         //var appt = this.getCalItem();
356         //appt.ptst = value;
357 
358         // Update the version currently in use.  It may get updated again below, but it doesn't matter
359         this.getCurrentView().setOrigPtst(value);
360     }
361     if (this.isCloseAction()) {
362         this._closeView();
363     } else {
364         this.getCurrentView().setOrigPtst(value);
365     }
366 };

373 ZmApptController.prototype._forwardListener =
374 function(ev) {
375 	var op = this.getMode();
376     var calItem = this.getCalItem();
377     if(!calItem) {
378         return;
379     }
380 
381     var mode = ZmCalItem.MODE_FORWARD;
382     if (calItem.isRecurring()) {
383 		mode = (op == ZmCalItem.MODE_EDIT_SINGLE_INSTANCE)
384 			? ZmCalItem.MODE_FORWARD_SINGLE_INSTANCE
385 			: ZmCalItem.MODE_FORWARD_SERIES;
386 	}
387 
388     this._app.getCalController()._forwardAppointment(calItem, mode);
389 };

393 	var calItem = this.getCalItem();
394     if(!calItem) {

396     }
397 	var url = ["/h/printappointments?id=", calItem.invId, "&tz=", AjxTimezone.getServerId(AjxTimezone.DEFAULT)]; //bug:53493

427 ZmApptController.prototype._proposeTimeListener =
428 function(ev) {
429 	var calItem = this.getCalItem();
430     if(!calItem) {
431         return;
432     }
433     //Pass mode edit to open the appt in edit mode. The mode 'propose new time' will be added later.
434 	var mode = ZmCalItem.MODE_EDIT;
435 	if (calItem.isRecurring()) {
436 		mode = this.getMode();
437 	}
438 	var appt = calItem;
439 	var clone = ZmAppt.quickClone(appt);
440 	clone.setProposeTimeMode(true);
441 	clone.getDetails(mode, new AjxCallback(this, this._proposeTimeContinue, [clone, mode]));
442 };

450 ZmApptController.prototype._doTag =
451 function(items, tag, doTag) {
452 
453 	var list = this._getTaggableItems(items);
454 
455 	if (doTag) {
456 		if (list.length > 0 && list.length == items.length) {
457 			// there are items to tag, and all are taggable
458 			ZmBaseController.prototype._doTag.call(this, list, tag, doTag);
459 		} else {
460 			var msg;
461 			var dlg = appCtxt.getMsgDialog();
462 			if (list.length > 0 && list.length < items.length) {
463 				// there are taggable and nontaggable items
464 				var listener = new AjxListener(this, this._handleDoTag, [dlg, list, tag, doTag]);
465 				dlg.setButtonListener(DwtDialog.OK_BUTTON, listener);
466 				msg = ZmMsg.tagReadonly;
467 			} else if (list.length == 0) {
468 				// no taggable items
469 				msg = ZmMsg.nothingToTag;
470 			}
471 			dlg.setMessage(msg);
472 			dlg.popup();
473 		}
474 	} else if (list.length > 0) {
475 		ZmBaseController.prototype._doTag.call(this, list, tag, doTag);
476 	}
477 };

544 ZmApptController.prototype.saveCalItem =
545 function(attId) {
546     var done = true;
547     if (this.isDirty()) {
548         var calItem = this.getCalItem();
549         if(calItem) {
550             var saveCallback = new AjxCallback(this, this._handleSaveResponse);
551             var calViewCtrl = this._app.getCalController();
552             var respCallback =
553                 new AjxCallback(calViewCtrl, calViewCtrl._handleResponseHandleApptRespondAction,
554                     [calItem, this.getOpValue(), null, saveCallback]);
555             calItem.getDetails(null, respCallback, this._errorCallback);
556             done = false;
557         }
558     }
559     if(done && this.isCloseAction()) {
560         this._closeView();
561     }
562 
563 };
