 22 /**
 23  * @overview
 24  * 
 25  * This file defines authentication.
 26  *
 27  */
 28 
 29 /**
 30  * Constructor. Use {@link execute} to construct the authentication.
 31  * @class
 32  * This class represents in-app authentication following the expiration of the session.
 33  * 
 34  * @see		#execute
 35  */
 36 ZmAuthenticate = function() {}
 37 
 38 ZmAuthenticate.prototype.isZmAuthenticate = true;
 39 ZmAuthenticate.prototype.toString = function() { return "ZmAuthenticate"; };
 40 
 41 
 42 ZmAuthenticate._isAdmin = false;
 43 
 44 /**
 45  * Sets the authentication as "admin".
 46  * 
 47  * @param	{Boolean}	isAdmin		<code>true</code> if admin
 48  */
 49 ZmAuthenticate.setAdmin =
 50 function(isAdmin) {
 51 	ZmAuthenticate._isAdmin = isAdmin;
 52 };
 53 
 54 /**
 55  * Executes an authentication.
 56  * 
 57  * @param	{String}	uname		the username
 58  * @param	{String}	pword		the password
 59  * @param	{AjxCallback}	callback	the callback
 60  */
 61 ZmAuthenticate.prototype.execute =
 62 function(uname, pword, callback) {
 63 	var command = new ZmCsfeCommand();
 64 	var soapDoc;
 65 	if (!ZmAuthenticate._isAdmin) {
 66 		soapDoc = AjxSoapDoc.create("AuthRequest", "urn:zimbraAccount");
 67 		var el = soapDoc.set("account", uname);
 68 		el.setAttribute("by", "name");
 69 	} else {
 70 		soapDoc = AjxSoapDoc.create("AuthRequest", "urn:zimbraAdmin", null);
 71 		soapDoc.set("name", uname);
 72 	}
 73 	soapDoc.set("virtualHost", location.hostname);	
 74 	soapDoc.set("password", pword);
 75 	var respCallback = new AjxCallback(this, this._handleResponseExecute, callback);
 76 	command.invoke({soapDoc: soapDoc, noAuthToken: true, noSession: true, asyncMode: true, callback: respCallback})
 77 };
 78 
 79 /**
 80  * @private
 81  */
 82 ZmAuthenticate.prototype._handleResponseExecute =
 83 function(callback, result) {
 84 	if (!result.isException()) {
 85 		ZmCsfeCommand.noAuth = false;
 86 	}
 87 
 88 	if (callback) {
 89 		callback.run(result);
 90 	}
 91 };
