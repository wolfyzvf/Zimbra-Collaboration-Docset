 28 /**
 29  * Creates an address book tree controller.
 30  * @class
 31  * This class is a controller for the tree view used by the address book 
 32  * application. This class uses the support provided by {@link ZmOperation}. 
 33  *
 34  * @author Parag Shah
 35  * 
 36  * @extends		ZmFolderTreeController
 37  */
 38 ZmAddrBookTreeController = function() {
 39 
 40 	ZmFolderTreeController.call(this, ZmOrganizer.ADDRBOOK);
 41 
 42 	this._listeners[ZmOperation.NEW_ADDRBOOK]	= this._newListener.bind(this);
 43 	this._listeners[ZmOperation.SHARE_ADDRBOOK]	= this._shareAddrBookListener.bind(this);
 44 
 45 	this._app = appCtxt.getApp(ZmApp.CONTACTS);
 46 };
 47 
 48 ZmAddrBookTreeController.prototype = new ZmFolderTreeController;
 49 ZmAddrBookTreeController.prototype.constructor = ZmAddrBookTreeController;
 50 
 51 ZmAddrBookTreeController.prototype.isZmAddrBookTreeController = true;
 52 ZmAddrBookTreeController.prototype.toString = function() { return "ZmAddrBookTreeController"; };

 56 /**
 57  * Shows the controller and returns the resulting tree view.
 58  * 
 59  * @param	{Hash}	params		 a hash of parameters
 60  * @return	{ZmTreeView}	the tree view
 61  */
 62 ZmAddrBookTreeController.prototype.show =
 63 function(params) {
 64 	params.include = {};
 65 	params.include[ZmFolder.ID_TRASH] = true;
 66     params.showUnread = false;
 67     var treeView = ZmFolderTreeController.prototype.show.call(this, params);
 68 
 69 	// contacts app has its own Trash folder so listen for change events
 70 	var trash = this.getDataTree().getById(ZmFolder.ID_TRASH);
 71 	if (trash) {
 72 		trash.addChangeListener(new AjxListener(this, this._trashChangeListener, treeView));
 73 	}
 74 
 75 	return treeView;
 76 };
 77 
 78 /**
 79  * @private
 80  */
 81 ZmAddrBookTreeController.prototype._trashChangeListener =
 82 function(treeView, ev) {
 83 	var organizers = ev.getDetail("organizers");
 84 	if (!organizers && ev.source) {
 85 		organizers = [ev.source];
 86 	}
 87 
 88 	// handle one organizer at a time
 89 	for (var i = 0; i < organizers.length; i++) {
 90 		var organizer = organizers[i];
 91 
 92 		if (organizer.id == ZmFolder.ID_TRASH &&
 93 			ev.event == ZmEvent.E_MODIFY)
 94 		{
 95 			var fields = ev.getDetail("fields");
 96 			if (fields && (fields[ZmOrganizer.F_TOTAL] || fields[ZmOrganizer.F_SIZE])) {
 97 				var ti = treeView.getTreeItemById(organizer.id);
 98 				if (ti) ti.setToolTipContent(organizer.getToolTip(true));
 99 			}
100 		}
101 	}
102 };

104 /**
105  * Enables/disables operations based on the given organizer ID.
106  * 
107  * @private
108  */
109 ZmAddrBookTreeController.prototype.resetOperations =
110 function(parent, type, id) {
111 	var deleteText = ZmMsg.del;
112 	var addrBook = appCtxt.getById(id);
113 	var nId = addrBook ? addrBook.nId : ZmOrganizer.normalizeId(id);
114 	var isTrash = (nId == ZmFolder.ID_TRASH);
115 
116 	var isDLs = (nId == ZmFolder.ID_DLS);
117 
118 	this.setVisibleIfExists(parent, ZmOperation.EMPTY_FOLDER, nId == ZmFolder.ID_TRASH);
119 
120 	if (isTrash) {
121 		parent.enableAll(false);
122 		parent.enable(ZmOperation.DELETE_WITHOUT_SHORTCUT, false);
123 		var hasContent = ((addrBook.numTotal > 0) || (addrBook.children && (addrBook.children.size() > 0)));
124 		parent.enable(ZmOperation.EMPTY_FOLDER,hasContent);
125 		parent.getOp(ZmOperation.EMPTY_FOLDER).setText(ZmMsg.emptyTrash);
126 	}
127 	else if (isDLs) {
128 		parent.enableAll(false);
129 	}
130 	else {
131 		parent.enableAll(true);        
132 		if (addrBook) {
133 
134 			parent.enable([ZmOperation.NEW_ADDRBOOK], !addrBook.isReadOnly());
135 
136 			if (addrBook.isSystem() || appCtxt.isExternalAccount()) {
137 				parent.enable([ZmOperation.DELETE_WITHOUT_SHORTCUT, ZmOperation.RENAME_FOLDER], false);
138 			} else if (addrBook.link) {
139 				parent.enable([ZmOperation.SHARE_ADDRBOOK], !addrBook.link || addrBook.isAdmin());
140 			}
141 			if (appCtxt.isOffline) {
142 				var acct = addrBook.getAccount();
143 				parent.enable([ZmOperation.SHARE_ADDRBOOK], !acct.isMain && acct.isZimbraAccount);
144 			}
145 		}
146 	}
147 
148 	if (addrBook) {
149 		parent.enable(ZmOperation.EXPAND_ALL, (addrBook.size() > 0));
150 	}
151 
152 	var op = parent.getOp(ZmOperation.DELETE_WITHOUT_SHORTCUT);
153 	if (op) {
154 		op.setText(deleteText);
155 	}
156 	this._enableRecoverDeleted(parent, isTrash);
157 
158 	// we always enable sharing in case we're in multi-mbox mode
159 	this._resetButtonPerSetting(parent, ZmOperation.SHARE_ADDRBOOK, appCtxt.get(ZmSetting.SHARING_ENABLED));
160 };

162 /**
163  * override to take care of not allowing dropping DLs do folders
164  * @param ev
165  * @private
166  */
167 ZmAddrBookTreeController.prototype._dropListener =
168 function(ev) {
169 	var items = AjxUtil.toArray(ev.srcData.data);
170 	for (var i = 0; i < items.length; i++) {
171 		var item = items[i];
172 		if (!item.isZmContact) {
173 			continue;
174 		}
175 		if (item.isDistributionList()) {
176 			ev.doIt = false;
177 			return;
178 		}
179 	}
180 	// perform default action
181 	ZmFolderTreeController.prototype._dropListener.apply(this, arguments);
182 };

271 ZmAddrBookTreeController.dlFolderClicked =
272 function() {
273 	var request = {
274 		_jsns: "urn:zimbraAccount",
275 		"ownerOf": 1,
276 		attrs: "zimbraDistributionListUnsubscriptionPolicy,zimbraDistributionListSubscriptionPolicy,zimbraHideInGal"
277 	};
278 
279 	var jsonObj = {GetAccountDistributionListsRequest: request};
280 	var respCallback = ZmAddrBookTreeController._handleAccountDistributionListResponse;
281 	appCtxt.getAppController().sendRequest({jsonObj: jsonObj, asyncMode: true, callback: respCallback});
282 };

284 /**
285  * Called when a left click occurs (by the tree view listener). The folder that
286  * was clicked may be a search, since those can appear in the folder tree. The
287  * appropriate search will be performed.
288  *
289  * @param {ZmOrganizer}	folder		the folder or search that was clicked
290  * 
291  * @private
292  */
293 ZmAddrBookTreeController.prototype._itemClicked =
294 function(folder) {
295 	if (folder.id == ZmFolder.ID_DLS) {
296 		ZmAddrBookTreeController.dlFolderClicked();
297 	}
298 	else if (folder.type == ZmOrganizer.SEARCH) {
299 		// if the clicked item is a search (within the folder tree), hand
300 		// it off to the search tree controller
301 		var stc = this._opc.getTreeController(ZmOrganizer.SEARCH);
302 		stc._itemClicked(folder);
303 	}
304 	else {
305 		var capp = appCtxt.getApp(ZmApp.CONTACTS);
306 		capp.currentSearch = null;
307 		var query = capp.currentQuery = folder.createQuery();
308 		var sc = appCtxt.getSearchController();
309 		sc.setDefaultSearchType(ZmItem.CONTACT);
310 		var acct = folder.getAccount();
311 		var params = {
312 			query: query,
313 			searchFor: ZmItem.CONTACT,
314 			fetch: true,
315 			sortBy: ZmSearch.NAME_ASC,
316 			callback: new AjxCallback(this, this._handleSearchResponse, [folder]),
317 			accountName: (acct && acct.name)
318 		};
319 		sc.search(params);
320 
321 		if (folder.id != ZmFolder.ID_TRASH) {
322 			var clc = AjxDispatcher.run("GetContactListController");
323 			var view = clc.getCurrentView();
324 			if (view) {
325 				view.getAlphabetBar().reset();
326 			}
327 		}
328 	}
329 };

331 /**
332  * @private
333  */
334 ZmAddrBookTreeController.prototype._handleSearchResponse =
335 function(folder, result) {
336 	// bug fix #19307 - Trash is special when in Contacts app since it
337 	// is a FOLDER type in ADDRBOOK tree. So reset selection if clicked
338 	if (folder.nId == ZmFolder.ID_TRASH) {
339 		this._treeView[this._app.getOverviewId()].setSelected(folder, true);
340 	}
341 };

343 /**
344  * @private
345  */
346 ZmAddrBookTreeController._handleAccountDistributionListResponse =
347 function(result) {
348 
349 	var contactList = new ZmContactList(null, true, ZmItem.CONTACT);
350 	var dls = result._data.GetAccountDistributionListsResponse.dl;
351 	if (dls) {
352 		for (var i = 0; i < dls.length; i++) {
353 			var dl = dls[i];
354 			var attrs = dl._attrs;
355 			attrs.email = dl.name; // in this case the email comes in the "name" property.
356 			attrs.type = "group";
357 			
358 			attrs[ZmContact.F_dlDisplayName] = dl.d || dl.name;
359 			contactList.addFromDom(dl);
360 		}
361 	}
362 	var clc = AjxDispatcher.run("GetContactListController");
363 	clc.show(contactList, true, ZmFolder.ID_DLS);
364 };
