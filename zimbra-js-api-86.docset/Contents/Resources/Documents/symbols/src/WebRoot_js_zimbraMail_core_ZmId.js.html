 28 /**
 29  * Constructor
 30  * @class
 31  * This class is responsible for providing unique, predictable IDs for HTML elements.
 32  * That way, code outside the client can easily locate particular elements.
 33  * <br/>
 34  * <br/>
 35  * Not every element that has an associated JS object will have a known ID. Those are
 36  * allocated only for elements it would be useful to locate: major components of the UI,
 37  * toolbars, buttons, views, menus, some menu items, some selects, and miscellaneous other
 38  * components.
 39  * <br/>
 40  * <br/>
 41  * Unless noted otherwise, a getElementById() on any of the non-skin IDs will return a DIV. One exception
 42  * is input fields. The ID is given to the DwtInputField's actual INPUT, rather than to the
 43  * DIV that contains it. Most other exceptions are table-related: TABLE, TR, and TD.
 44  * <br/>
 45  * <br/>
 46  * There is a simple naming scheme for the IDs themselves. Each ID starts with a "z" followed
 47  * by one to a few letters that indicate the type of object (widget) represented by the element:
 48  * 
 49  * <ul>
 50  * <li><b>z</b> a component that is not a special-purpose widget listed below</li>
 51  * <li><b>ztb</b> 		a toolbar (see {@link ZmId.WIDGET})</li>
 52  * <li><b>zb</b> 		a button (see {@link ZmId.WIDGET_TOOLBAR})</li>
 53  * <li><b>zi</b> 		an input field (see {@link ZmId.WIDGET_INPUT})</li>
 54  * <li><b>zm</b> 		a menu (see {@link ZmId.WIDGET_MENU})</li>
 55  * <li><b>zmi</b> 		a menu item (see {@link ZmId.WIDGET_MENU_ITEM})</li>
 56  * <li><b>zs</b> 		a select (see {@link ZmId.WIDGET_SELECT})</li>
 57  * <li><b>zov</b> 		an overview (see {@link ZmId.WIDGET_OVERVIEW})</li>
 58  * <li><b>zt</b> 		a tree (see {@link ZmId.WIDGET_TREE})</li>
 59  * <li><b>zti</b> 		a tree item (see {@link ZmId.WIDGET_TREE_ITEM})</li>
 60  * <li><b>ztih</b> 	a tree item header (see {@link ZmId.WIDGET_TREE_ITEM_HDR})</li>
 61  *</ul>
 62  *
 63  * The skin defines its own container IDs starting with "skin_", which we provide constants for here.
 64  * 
 65  * @author Conrad Damon
 66  */

289 /**
290  * Generates a tree item ID based on the underlying organizer and the overview ID (since the same

292  * 
293  * <ul>
294  * <li><b>zti|Mail|2</b> Inbox</li>
295  * <li><b>zti|Mail|172</b>			user-created item in mail overview</li>
296  * <li><b>zti|Contacts|7</b>			system Contacts folder</li>
297  * <li><b>zti|Calendar|304</b>		user-created item in calendar overview</li>
298  * <li><b>ztih|Mail|FOLDER</b>		Folders header in mail overview</li>

301  * Constants for some system folder tree items have been provided as a convenience.
302  * 
303  * @param {String}	overviewId	the unique ID for overview
304  * @param {ZmOrganizer}	organizerId	the ID of the data object backing tree item
305  * @param {constant}	type			the organizer type (for headers only)
306  * @return	{String}	the id
307  */
308 ZmId.getTreeItemId =
309 function(overviewId, organizerId, type) {
310 	if (!organizerId && !type) { return; }
311 	if (type) {
312 		return DwtId.makeId(ZmId.WIDGET_TREE_ITEM_HDR, overviewId, type);
313 	} else {
314 		return DwtId.makeId(ZmId.WIDGET_TREE_ITEM, overviewId, organizerId);
315 	}
316 };

321  * message view may appear within several double-pane views). The component name

333 ZmId.getViewId =
334 function(viewId, component, context) {
335 	var id = DwtId.makeId(ZmId.WIDGET_VIEW, context, viewId);
336 	return component ? [id, component].join("") : id;
337 };

484 /*

486  * Overviews
487  * ---------
488  * 
489  * An overview is a collection of trees. The primary place that the user will see an overview is
490  * at the left side of the client. Note that each app has its own overview, since it may want to
491  * show a different set of trees. For example, the mail app shows trees for folders, searches, tags,
492  * and zimlets by default. Overviews also appear when the user needs to choose something from a tree,
493  * for example selecting a folder within a move dialog when moving a message.
494  * 
495  * A tree is a collection of tree items, each of which may have its own tree items.
496  * 
497  * The overview IDs for the main overviews that show up at the left are just app names (ZmId.APP_*).

499  *

503  * 
504  * 		ZmId.getOverviewId(ZmId.APP_MAIL)

507  * 
508  * 		ZmId.getTreeId(ZmId.APP_MAIL, ZmId.ORG_FOLDER)

510  * Tree items:
511  * 
512  * 		ZmId.getTreeItemId(ZmId.APP_MAIL, ZmFolder.ID_INBOX)
513  * 		ZmId.getTreeItemId(ZmId.APP_MAIL, 2)
514  * 		ZmId.TREEITEM_INBOX
515  * 
516  * TODO: come up with simpler names for other (non-app) overviews

563 /*
564  * ------------
565  * Message view
566  * ------------
567  * 
568  * A message view displays an email message. There are several different instances of message views, which
569  * makes it necessary to include a context (owning view) to be able to identify each one of them.
570  * 
571  * The function to use is:
572  * 
573  * 		ZmId.getViewId(ZmId.VIEW_MSG, component, context)
574  * 
575  * Since message views are not singletons, a context is always necessary. Omit the component only when getting
576  * an ID for a message view itself.
577  * 
578  * To get the ID for a message view, pass the constant for the message view as well as the context, which can be
579  * ZmId.VIEW_CONVLIST, ZmId.VIEW_CONV, ZmId.VIEW_MSG, or ZmId.VIEW_TRAD:
580  * 
581  * 		ZmId.getViewId(ZmId.VIEW_MSG, null, ZmId.VIEW_TRAD)
582  * 
583  * There are also many components within a message view which are useful to retrieve. To get the ID for a
584  * message view component, pass the component ID (ZmId.MV_*):
585  * 
586  * 		ZmId.getViewId(ZmId.VIEW_MSG, ZmId.MV_HDR_TABLE_TOP_ROW, ZmId.VIEW_TRAD)
587  * 		ZmId.getViewId(ZmId.VIEW_MSG, ZmId.MV_ATT_LINKS, ZmId.VIEW_TRAD)
588  * 
589  * 		var bodyId = ZmId.getViewId(ZmId.VIEW_MSG, ZmId.MV_MSG_BODY, ZmId.VIEW_TRAD)
590  * 
591  * will return the ID for the DIV containing the msg body iframe. To get the ID of the IFRAME element
592  * itself, pass that ID as the context for the IFRAME:
593  * 
594  * 		var iframeId = DwtId.getIframeId(bodyId);
595  * 
596  * For buttons within msg view, pass the context and operation as usual, and add the identifier for
597  * message view (which distinguishes its buttons from, say, those on the VIEW_TRAD toolbar).
598  * 
599  * 		ZmId.getButtonId(ZmId.VIEW_MSG, ZmId.OP_CLOSE, ZmId.VIEW_CONV)

1168 /*
1169  * Experimental ID code below. The main idea is to make easier for a third party (such as QA) to find what they're
1170  * looking for. A secondary goal is to ensure that we always use unique IDs. The systematic approach above is prone
1171  * to failure in that regard, since the same set of inputs will produce the same ID.
1172  *
1173  * The new approach is to introduce a level of indirection between the fields and the ID. IDs will go back to being
1174  * unique and opaque, based on an incrementing number. A hash will be maintained which maps a collection of fields
1175  * to the actual ID used in the DOM.
1176  *
1177  * The core of the new system is related closely to the old system: a set of params which, taken together, should
1178  * uniquely identify an element. The possible values for each param are typically constants defined in this class.
1179  *
1180  * To make it easy for clients of this ID system to successfully look up IDs, creators of IDs should provide as many
1181  * parameters as possible. For example, providing both skinComponent and componentType may be redundant, but then
1182  * the ID can be looked up using either parameter.
1183  *
1184  * The parameters and their values:
1185  *
1186  * skinComponent
1187  *
1188  *      The HTML skin for ZCS defines a number of components and provides containers for them. This param identifies
1189  *      which skin component contains the element. Note that the IDs for the skin containers themselves (as well as
1190  *      a few elements within those containers) are defined by the skin (in skin.html), and are not part of this set
1191  *      of IDs. May often be omitted when looking up an ID.
1192  *
1193  *      ZmId.SKIN_*
1194  *
1195  * componentType
1196  *
1197  *      The general category of view component that maps to the element. It may be a type of widget such as a button
1198  *      or a menu, something more general like a view, or even a subcomponent like a list view header.
1199  *
1200  *      DwtId.WIDGET_*
1201  *      ZmId.WIDGET_*
1202  *
1203  * componentName
1204  *
1205  *      The component name identifies the component among components of a similar type. It may be the name of an
1206  *      operation. For example, a search button would have the type "button" and the name "search". (There may be
1207  *      more than one search button, so other params may be necessary to uniquely identify each one.)
1208  *
1209  *      ZmId.VIEW_*
1210  *      ZmId.OP_*
1211  *      ZmId.TB_*
1212  *      ZmId.SEARCH_*
1213  *      ZmId.MV_*
1214  *      ZmId.CMP_*
1215  *      ZmId.GROUPBY_*
1216  *
1217  * app
1218  *
1219  *      The name of the application that contains the element. Some elements are global and do not have an associated
1220  *      application. Elements associated with an app appear within the main content area.
1221  *
1222  *      ZmId.APP_*
1223  *
1224  * containingView
1225  *
1226  *      For an element within the main content area, the identifier of the view that contains it.
1227  *
1228  *      ZmId.VIEW_*
1229  *
1230  * sessionId
1231  *
1232  *      A number identifying a session for a view which can appear in more than one tab at a time. For example, there
1233  *      may be multiple compose sessions if the user replies to several different messages without sending the replies.
1234  *
1235  * itemType
1236  *
1237  *      The type of item, such as message, contact, or appointment.
1238  *
1239  *      ZmId.ITEM_*
1240  *
1241  * itemId
1242  *
1243  *      The ID of the item (for example, a mail message) that the element is tied to. For local items, it's a number.
1244  *      For shared items, it's a compound string comprising an account ID and a local numeric ID.
1245  *
1246  * organizerType
1247  *
1248  *      The type of organizer, such as folder, tag, or zimlet. Organizers generally appear in the overview.
1249  *
1250  *      ZmId.ORG_*
1251  *
1252  * organizerId
1253  *
1254  *      The ID of the organizer (for example, a mail folder) that the element is tied to. For local organizers, it's a
1255  *      number. For shared folders, it's a compound string comprising an account ID and a local numeric ID.
1256  *
1257  * field
1258  *
1259  *      A field identifies a specific sub-part of a component. It might be something that helps make up a widget,
1260  *      such as the "left icon" in a button, or it might be something ZCS-specific like the "subject" field in a list
1261  *      view that displays mail messages. The line between componentName and field can be a bit blurry. Generally a
1262  *      componentName refers to a container of some sort, like a list row or header.
1263  *
1264  *      ZmId.FLD_*
1265  *
1266  * tagName

1271  *

1273  *
1274  * parentId

1277  *

1289 /**
1290  * Returns a unique ID that can later be looked up. As many params as possible should be provided, in order to
1291  * make lookup easier. If one or more IDs is found to already have been created with the given set of params,
1292  * a sequence number is added as a parameter.
1293  *
1294  * @param {hash}        params          set of fields describing the ID's element
1295  * @param {string}      description     (optional) a brief description of the purpose of the ID
1296  */
1297 ZmId.create = function(params, description) {
1298 
1299 	var idParams = AjxUtil.hashCopy(params);
1300 	var ids = ZmId.lookup(params);
1301 	if (ids) {
1302 		idParams.sequence = (typeof ids === "string") ? 1 : ids.length;
1303 	}
1304 	idParams.description = description || "";
1305 	var newId = ZmId.BASE + ZmId.SEQ++;
1306 	idParams.id = newId;
1307 	ZmId._idHash[newId] = idParams;
1308 	ZmId._idList.push(idParams);
1309 
1310 	for (var key in params) {
1311 		ZmId._valueToParam[params[key]] = key;
1312 	}
1313 
1314 	return newId;
1315 };

1317 /**
1318  * Returns the DOM ID that matches the given set of params. If more than one ID matches, a list is returned.
1319  * A partial set of params may be provided. The more params provided, the better the chance of finding just one ID.
1320  * The best approach is to provide the minimal set of params that will uniquely differentiate the element. If no
1321  * params are provided, returns all IDs.
1322  *
1323  * Optionally, a list of values can be given. An attempt will be made to reverse-engineer the params by figuring
1324  * out the appropriate key for each value. This method will never be as reliable as providing a hash in the first place.
1325  *
1326  * @param {hash|array}  params    set of fields describing the ID(s) being sought
1327  */
1328 ZmId.lookup = function(params) {
1329 
1330 	if (!params) {
1331 		return ZmId._idList;
1332 	}
1333 
1334 	if (AjxUtil.isArray(params)) {
1335 		params = ZmId._convertValues(params);
1336 	}
1337 
1338 	var ids = [];
1339 	for (var i = 0, len = ZmId._idList.length; i < len; i++) {
1340 		var idParams = ZmId._idList[i];
1341 		var add = true;
1342 		for (var param in params) {

1410 // Convert a list of values of ID parameters back into a hash by figuring out the matching key for each value.
1411 // View names (such as "CLV") are a bit tricky since they can be either a componentName (for a view widget), or
1412 // a containingView. A small number might be an organizer ID (eg Inbox is 2), or a session ID.
1413 ZmId._convertValues = function(values) {
1414 
1415 	var params = {},
1416 		viewValue, numValue;
1417 
1418 	for (var i = 0; i < values.length; i++) {
1419 		var value = values[i];
1420 		if (ZmId._isViewType[value]) {
1421 			viewValue = value;
1422 		}
1423 		else if (AjxUtil.isNumber(value) || AjxUtil.isNumeric(value)) {
1424 			var num = parseInt(value);
1425 			if (num < 10) {
1426 				numValue = num;
1427 			}
1428 		}
1429 		else {
1430 			var param = ZmId._valueToParam[value];
1431 			params[param] = value;
1432 		}
1433 	}
1434 
1435 	// A view value is a componentName only if the componentType is a view.
1436 	if (viewValue) {
1437 		var viewParam = (params.componentType === ZmId.WIDGET_VIEW) ? "componentName" : "containingView";
1438 		params[viewParam] = viewValue;
1439 	}
1440 
1441 	// A single-digit number is probably an organizer ID or a session ID.
1442 	if (numValue) {
1443 		var viewParam = params.organizerType ? "organizerId" : "sessionId";
1444 		params[viewParam] = viewValue;
1445 	}
1446 
1447 	return params;
1448 };
