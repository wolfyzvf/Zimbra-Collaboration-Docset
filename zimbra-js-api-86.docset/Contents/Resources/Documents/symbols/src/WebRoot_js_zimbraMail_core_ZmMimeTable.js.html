133 ZmMimeTable.getInfo =
134 function(type, createIfUndefined) {
135 	var entry = ZmMimeTable._table[type];
136 	if (!entry && createIfUndefined) {
137 		entry = ZmMimeTable._table[type] = {desc: type, image: "UnknownDoc", imageLarge: "UnknownDoc_48"};
138 	}
139 	if (entry) {
140 		if (!entry.type) {
141 			entry.type = type;
142 		}
143 	} else {
144 		// otherwise, check if main category is in table
145 		var baseType = type.split("/")[0];
146 		if (baseType) {
147 			entry = ZmMimeTable._table[baseType];
148 		}
149 	}
150 	return entry;
151 };

153 /**
154  * Checks if the type is ignored.
155  * 
156  * @param	{constant}	type		the type
157  * @return	{Boolean}	<code>true</code> if the type is ignored
158  */
159 ZmMimeTable.isIgnored = 
160 function(type) {
161 	return (type == ZmMimeTable.MULTI_ALT ||
162 			type == ZmMimeTable.MULTI_MIXED ||
163 			type == ZmMimeTable.MULTI_RELATED ||
164 			type == ZmMimeTable.MULTI_APPLE_DBL ||
165 			type == ZmMimeTable.APP_MS_TNEF ||
166 			type == ZmMimeTable.APP_MS_TNEF2 ||
167             type == ZmMimeTable.APP_SIGNATURE);
168 };
169 
170 /**
171  * Checks if the type is renderable.
172  * 
173  * @param	{constant}	type		the type
174  * @return	{Boolean}	<code>true</code> if the type is renderable
175  */
176 ZmMimeTable.isRenderable =
177 function(type, textOnly) {
178 	return (type === ZmMimeTable.TEXT_HTML ||
179 			type === ZmMimeTable.TEXT_PLAIN ||
180 			(!textOnly && ZmMimeTable.isRenderableImage(type)) ||
181 			(!textOnly && type === ZmMimeTable.APP_ADOBE_PDF));
182 };
183 
184 ZmMimeTable.isTextType =
185 function(type){
186     return (type.match(/^text\/.*/) &&
187             type != ZmMimeTable.TEXT_HTML &&
188             type != ZmMimeTable.TEXT_CAL);  
189 };
190 
191 /**
192  * Checks if the type is a renderable image.
193  * 
194  * @param	{constant}	type		the type
195  * @return	{Boolean}	<code>true</code> if the type is a renderable image
196  */
197 ZmMimeTable.isRenderableImage =
198 function(type) {
199 	return (type == ZmMimeTable.IMG_JPEG ||
200 			type == ZmMimeTable.IMG_GIF ||
201 			type == ZmMimeTable.IMG_BMP ||
202 			type == ZmMimeTable.IMG_PNG);
203 };
204 
205 /**
206  * Checks if the type has an HTML version.
207  * 
208  * @param	{constant}	type		the type
209  * @return	{Boolean}	<code>true</code> if the type has an HTML version
210  */
211 ZmMimeTable.hasHtmlVersion =
212 function(type) {
213 	return (!(ZmMimeTable.isIgnored(type) ||
214 			type.match(/^text\/plain/) ||
215 			type.match(/^image/) ||
216 			type.match(/^audio/) ||
217 			type.match(/^video/)));
218 };
219 
220 ZmMimeTable.isMultiMedia =
221 function(type){
222     return (type.match(/^audio/) ||
223 			type.match(/^video/));
224 };
225 
226 ZmMimeTable.isWebDoc =
227 function(type) {
228     return (type == ZmMimeTable.APP_ZIMBRA_DOC);
229 };
