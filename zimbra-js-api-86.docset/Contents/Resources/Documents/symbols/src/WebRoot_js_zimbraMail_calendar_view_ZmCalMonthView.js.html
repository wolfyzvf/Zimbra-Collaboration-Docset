136 ZmCalMonthView.prototype._reserveRow = 
137 function(day, apptSet, appt) {
138 	var appts = day.allDayAppts;
139     var row = -1;
140 
141     if (apptSet.rows[day.week] === undefined) {
142         // New apptSet, or another week in the set.  Find a free slot or add to the end
143         row = this._allocateRow(day, apptSet);
144     } else {
145         // Use the existing row for the apptSet
146         row = apptSet.rows[day.week];
147     }
148     var apptToMove = appts[row];
149     appts[row] = appt;
150 
151     if (apptToMove) {
152         // The row was in use, need to move to free slot or end
153         var uniqId = this._getApptUniqueId(apptToMove);
154         var apptSetToMove = this._apptSets[uniqId];
155         row = this._allocateRow(day, apptSetToMove);
156         appts[row] = apptToMove;
157         apptSetToMove.rows[day.week] = row;
158     }
159 };

161 ZmCalMonthView.prototype._allocateRow =
162 function(day, apptSet) {
163     var appts = day.allDayAppts;
164     apptSet.rows[day.week] = appts.length;
165     for (var i=0; i < appts.length; i++) {
166         if (appts[i] == null) {
167             apptSet.rows[day.week] = i;
168             break;
169         }
170     }
171     return apptSet.rows[day.week];
172 }

174 ZmCalMonthView.prototype.addAppt = 
175 function(appt) {
176     var day = this._getDayForAppt(appt);
177 
178     if (appt._orig.isAllDayEvent() || appt._orig.isMultiDay()) {
179         var uniqueId = this._getApptUniqueId(appt);
180         var apptSet = this._apptSets[uniqueId];
181         if (apptSet == null) {
182             apptSet = this._createApptSet(appt, uniqueId);
183         }
184         apptSet.appts.push(appt);
185     }
186 
187     if (day) {
188         if (appt._orig.isAllDayEvent(appt)) {
189             // make sure multi-day all day appts line up
190             // Assuming sliced up appts passed in chronological order (first is start) - verify
191             if (!day.allDayAppts) {
192                 day.allDayAppts = [];
193             }  else {
194 
195             }
196             // Reserve a row if its onscreen
197             this._reserveRow(day, apptSet, appt);
198         } else {
199             if (!day.appts) {
200                 day.appts = [];
201             }
202             day.appts.push(appt);
203         }
204     }
205 
206 };

209 // Multi-day appts have been sliced into a set of single day appts.  Accumulate
210 // the appts into an apptSet, so that when DnD is performed we can update the
211 // position of each appt slice that comprises the full appt.
212 ZmCalMonthView.prototype._createApptSet =
213 function(appt, uniqueId, day) {
214     var dow;
215     var apptSet = null;
216     if (day) {
217         dow = day.dow;
218     } else {
219         // DayIndex should be < 0 (no corresponding day, and assuming sliced appts
220         // passed in from earliest to last)
221         var dayIndex = this._createDayIndexFromDate(appt.startDate);
222         var zeroDay = this._days[0]
223         dow = (zeroDay.dow + dayIndex + 7) % 7;
224     }
225     // The set tracks the starting dow of the full appt, whether or not it is
226     // an all-day event (note that there can be multi-day non-all-day appts),
227     // the appt slices (one per day) that comprise the full appt, and the row
228     // position of each slice  (in case the full appt spans multiple weeks).
229     apptSet = { appts: [], rows: {}, dow: dow, allDay: appt.isAllDayEvent()};
230     this._apptSets[uniqueId] = apptSet;
231     return apptSet;
232 }

234 ZmCalMonthView.prototype._postSet = 
235 function() {
236 	// now go through each day and create appts in correct order to line things up
237 	var day;
238 	if(this._expandedDayInfo) {
239 		var row = this._expandedDayInfo.week;
240 		var col = this._expandedDayInfo.dow;
241 		var day = this._days[row*7+col];
242 		var d = new Date(this._date.getTime());
243 		d.setHours(0,0,0);
244 		if(d.getTime() == day.date.getTime()) {
245 			this.setDayView(day);
246 		}else {
247 			this.createApptItems();
248 		}
249 	}else {
250 		this.createApptItems();
251 	}
252 };

255 ZmCalMonthView.prototype.createApptItems =
256 function() {
257 	var allDayParent = document.getElementById( this._daysId);
258     var day;
259     // Create the all-day divs
260     this._apptAllDayDiv = {};
261     for (var uniqueId in this._apptSets) {
262         var apptSet = this._apptSets[uniqueId];
263         if (!apptSet.allDay) continue;
264 
265         var currentWeek = -1;
266         var first = true;
267         var last = false;
268         var startDayIndex = this._createDayIndexFromDate(apptSet.appts[0]._orig.startDate);
269         // Set first (header div) if offscreen and first div, or first on the grid,
270         //   starting a new week.
271         // Set last if in last day (41) or last in appt sequence
272         for (var iAppt = 0; iAppt < apptSet.appts.length; iAppt++) {
273             var appt = apptSet.appts[iAppt];
274             day = this._getDayForAppt(appt);
275             first = (iAppt == 0);
276             last = (iAppt == (apptSet.appts.length - 1));
277             if (day) {
278                 if (((startDayIndex + iAppt) == (this.numDays -1))) {
279                     last = true;
280                 }
281                 if (day.week != currentWeek) {
282                     // Catches 1st on-screen appt (0th or not)
283                     first = true;
284                     currentWeek = day.week;
285                 }
286             }
287             var allDayDiv = this._createAllDayApptDiv(allDayParent, appt, iAppt, first, last);
288             if (!day) {
289                 Dwt.setVisible(allDayDiv, false);
290             }
291             first = false;
292             last = false;
293         }
294     }
295 
296 	for (var i=0; i < 6; i++)	 {
297 		for (var j=0; j < 7; j++)	 {
298             var dayIndex = i*7+j;
299 			day = this._days[dayIndex];
300 			if (day.allDayAppts) {
301 				for (var k=0; k < day.allDayAppts.length; k++) {
302 					var appt = day.allDayAppts[k];			
303 					var div = this._attachAllDayFillerHtml(appt, dayIndex);
304 					this._fillers.push(div);
305 				}
306 			}
307 			if (day.appts) {
308 				for (var k=0; k < day.appts.length; k++)
309 					var div = this._createItemHtml(day.appts[k], null);
310 			}
311 		}
312 	}
313 	
314 	if (!this._needFirstLayout)
315 		this._layout();
316 		
317 	if(this._dayView) {
318 		this._dayView.setVisible(false);
319 		this.clearExpandedDay();
320 	}
321 };

323 ZmCalMonthView.prototype._preSet = 
324 function() {
325     // reset all layout data
326 	// cleanup any filler
327 	if (this._fillers.length > 0) {
328 		for (var i=0; i < this._fillers.length; i++) {
329 			var f = 	this._fillers[i];
330 			this._fillers[i] = null;
331 			f.parentNode.removeChild(f);
332 		}
333 		this._fillers = [];
334 	}
335     this._apptSets = new Object();;
336 	for (var i=0; i < 6; i++)	 {
337 		for (var j=0; j < 7; j++)	 {
338 			day = this._days[i*7+j];
339 			if (day.allDayAppts)	delete day.allDayAppts;
340 			if (day.appts) delete day.appts;
341 		}
342 	}
343 };

455 ZmCalMonthView.prototype._createItemHtmlContents =
456 function(appt, tr) {

466     var bodyColors   = ZmApptViewHelper.getApptColor(needsAction, calendar, tagNames, "body");
467     var bodyStyle    = ZmCalBaseView._toColorsCss(bodyColors.appt);
468 
469 
470     var data = {
471         id: this._getItemId(appt),
472         appt: appt,
473         duration: appt.getShortStartHour(),
474         headerStyle: headerStyle,
475         bodyStyle: bodyStyle,
476         multiday: appt._fanoutFirst != null,
477         first: appt._fanoutFirst,
478         last: appt._fanoutLast,
479         showAsColor : ZmApptViewHelper._getShowAsColorFromId(fba),
480         tagIcon: tagIcon
481     };
482     ZmApptViewHelper.setupCalendarColor(true, bodyColors, tagNames, data, "headerStyle", null, 0, 0);

487     // Hack for IE - it doesn't display the tag and peel unless you  alter a containing className.
488     // The month template div does not have any classNames, so this is safe.
489     cell.firstChild.className = "";
490 }

1793 ZmCalMonthView.prototype._reattachApptDnDHtml =
1794 function(data, startIndex, deselect) {
1795      for (var i = 0; i < data.trEl.length; i++) {
1796          var day = this._days[startIndex + i];
1797          if (data.apptDiv[i]) {
1798              // Detach the Appt DnD Proxy Div from the allDayParent
1799              data.apptDiv[i].parentNode.removeChild(data.apptDiv[i]);
1800          }
1801 
1802          if (day && data.trEl[i]) {
1803              // TD that originally contained the appt
1804              var td = data.trEl[i].firstChild;
1805              if (td.saveHTML) {
1806                  if (startIndex == data.startDayIndex) {
1807                      // Remove the filler
1808                      td.removeChild(td.firstChild);
1809                  } else {
1810                      // Dropped in a new cell - find the correct position within the appts of the current day
1811                      var tBody = document.getElementById(day.dayId);
1812                      var insertIndex = 0;
1813                      for (insertIndex = 0; insertIndex < tBody.childNodes.length; insertIndex++) {
1814                          var targetTR = tBody.childNodes[insertIndex];
1815                          if ((targetTR.apptStartTimeOffset !== undefined) && (targetTR.apptStartTimeOffset > data.timeOffset[i])) {
1816                               break;
1817                         }
1818                      }
1819                      // Remove the original TR from its day div
1820                      if (data.trEl[i].parentNode) {
1821                          data.trEl[i].parentNode.removeChild(data.trEl[i]);
1822                      }
1823                      data.trEl[i].removeChild(td);
1824 
1825                      // Create a new TR in the new day div
1826                      var tr = tBody.insertRow(insertIndex);
1827                      tr.appendChild(td);
1828                      tr.id = data.trEl[i].id;
1829                      tr.className = data.trEl[i].className;
1830                  }
1831                  // Set the td with the original appt content.  Do via innerHTML since IE
1832                  // does not handle the gradient coloring properly if the appt's div is simply moved
1833                  td.innerHTML = td.saveHTML;
1834                  // Hack for IE - it doesn't display the tag and peel unless you  alter a containing className.
1835                  // The month template div does not have any classNames, so this is safe.
1836                  td.firstChild.className = "";
1837              }
1838          }
1839      }
1840      data.apptDiv = {};
1841 }

1915         "    headingBaseY: " + headingBaseY +  "    sizeY:" + size.y);
1916 
1917 	var upper = docY < headingBaseY + headingHeight + 4;;
1918 	var lower = docY > originPt.y + size.y - 8; // - 8;
1919 
1920 	if (upper || lower) {
1921 		var sTop = div.scrollTop;
1922 		if (upper && sTop > 0) {
1923             DBG.println(AjxDebug.DBG3,"_handleApptScrollRegion sTop: " + sTop);
1924 			offset = -(sTop > incr ? incr : sTop);
1925 		} else if (lower) {
1926             var fullSize = Dwt.getSize(fullDiv);
1927             var sVisibleTop = fullSize.y - size.y;
1928             DBG.println(AjxDebug.DBG3,"_handleApptScrollRegion sTop: " + sTop + ", sVisibleTop: " + sVisibleTop);
1929 			if (sTop < sVisibleTop) {
1930 				var spaceLeft = sVisibleTop - sTop;
1931 				offset = spaceLeft  > incr ?incr : spaceLeft;
1932                 DBG.println(AjxDebug.DBG3,"_handleApptScrollRegion spaceLeft: " + spaceLeft);
1933 			}
1934 		}
1935 		if (offset != 0) {
1936 			div.scrollTop += offset;
1937             DBG.println(AjxDebug.DBG3,"_handleApptScrollRegion offset: " + offset);
1938             this._containerRect.set(this._containerRect.x, this._containerRect.y - offset);
1939             data.originY -= offset;
1940             //DBG.println(AjxDebug.DBG3,"_handleApptScrollRegion new containerRect.y = " + this._containerRect.y + ",   originY = " + data.originY);
1941 		}
1942 
1943 	}
