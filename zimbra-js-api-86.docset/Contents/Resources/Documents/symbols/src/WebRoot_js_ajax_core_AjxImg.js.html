 44 /**
 45  * This method will set the image for <i>parentEl</i>. <i>parentEl</i> should 
 46  * only contain this image and no other children
 47  *
 48  * @param parentEl 		the parent element for the image
 49  * @param imageName 		the name of the image.  The CSS class for the image will be "Img<imageName>".
 50  * @param useParenEl 	if <code>true</code> will use the parent element as the root for the image and will not create an intermediate DIV
 51  * @param _disabled		if <code>true</code>, will append " ZDisabledImage" to the CSS class for the image,
 52  * @param {array}       classes             array of class names to be applied to this image
 53  *							which will make the image partly transparent
 54  */
 55 AjxImg.setImage =
 56 function(parentEl, imageName, useParentEl, _disabled, classes) {
 57 	
 58 	if (!parentEl) { return; }
 59 	
 60 	classes = classes || [];
 61 	var origImageName = imageName;
 62     var color, m = imageName && imageName.match(AjxImg.RE_COLOR);
 63 	if (m) {
 64 		imageName = m && m[1];
 65 		color = m && m[2];
 66 	}
 67 
 68 	var className = AjxImg.getClassForImage(imageName, _disabled);
 69 	if (useParentEl) {
 70 		classes.push(className);
 71 		parentEl.className = classes.join(" ");
 72 		return;
 73 	}
 74 	var id = parentEl.firstChild && parentEl.firstChild.id;
 75         
 76 	var overlayName = className+"Overlay";
 77 	var maskName = className+"Mask";
 78 	if (color && window.AjxImgData && AjxImgData[overlayName] && AjxImgData[maskName]) {
 79 		color = (color.match(/^\d$/) ? ZmOrganizer.COLOR_VALUES[color] : color) ||
 80 				ZmOrganizer.COLOR_VALUES[ZmOrganizer.ORG_DEFAULT_COLOR];
 81 		parentEl.innerHTML = AjxImg.getImageHtml({
 82 			imageName: origImageName,
 83 			attrStr: id ? "id='"+id+"'" : null,
 84 			disabled: _disabled
 85 		});
 86 		return;
 87 	}
 88 
 89 	if (parentEl.firstChild == null || parentEl.firstChild.nodeName.toLowerCase() != "div") {
 90 		var html = [], i = 0;
 91 		html[i++] = "<div ";
 92 		if (id) {
 93 			html[i++] = " id='";
 94 			html[i++] = id;
 95 			html[i++] = "' ";
 96 		}
 97 		if (className) {
 98 			classes.push(className);
 99 		}
100 		html[i++] = AjxUtil.getClassAttr(classes);
101 		html[i++] = "></div>";
102 		parentEl.innerHTML = html.join("");
103 		return;
104 	} else if (AjxEnv.isIE && !AjxEnv.isIE9up) {
105 		parentEl.firstChild.innerHTML = "";
106 	}
107 	if (className) {
108 		classes.push(className);
109 	}
110 	parentEl.firstChild.className = classes.join(" ");
111 };
112 
113 AjxImg.setDisabledImage = function(parentEl, imageName, useParentEl, classes) {
114 	return AjxImg.setImage(parentEl, imageName, useParentEl, true, classes);
115 };

117 AjxImg.getClassForImage =
118 function(imageName, disabled) {
119 	var className = imageName ? "Img" + imageName : "";
120 	if (disabled) className += " ZDisabledImage";
121 	return className;
122 };

149 /**
150  * Returns the HTML needed to display the given image.
151  *
152  * @param {object}		params		hash of params:
153  * @param {string}		imageName		the image you want to render
154  * @param {string}		styles			optional style info (for example, "display:inline")
155  * @param {string}		attrStr			optional attributes (for example, "id=X748")
156  * @param {boolean}		wrapInTable		if true, wrap the HTML in a TABLE
157  * @param {boolean}		disabled		if true, show image as disabled
158  * @param {array}		classes			array of class names to be applied to this image
159  * @param {string}		altText			alternative text for non-visual users
160  * 
161  * @return	{string}	the image string
162  */
163 AjxImg.getImageHtml = 
164 function() {
165 	var params = Dwt.getParams(arguments, AjxImg.GET_IMAGE_HTML_PARAMS);
166 
167 	var imageName = params.imageName;
168 	var styles = params.styles || "";
169 	var styleStr = styles ? " style='" + styles + "'" : "";
170 	var attrStr = params.attrStr ? " " + params.attrStr : "";
171 	var disabled = params.disabled;
172 	var classes = params.classes || [];
173 	var altText = params.altText;
174 
175 	var pre = params.wrapInTable ? "<table style='display:inline' cellpadding=0 cellspacing=0 border=0><tr><td align=center valign=bottom>" : "";
176     var html = "";
177 	var post = params.wrapInTable ? "</td></tr></table>" : "";
178 
179 	if (imageName) {
180         var color, m = imageName.match(AjxImg.RE_COLOR);
181         if (m) {
182             imageName = m && m[1];
183             color = m && m[2];
184         }
185 
186         var className = AjxImg.getClassForImage(imageName, disabled);
187         var overlayName = className + "Overlay";
188         var maskName = className + "Mask";
189         if (color && window.AjxImgData && AjxImgData[overlayName] && AjxImgData[maskName]) {
190             color = (color.match(/^\d$/) ? ZmOrganizer.COLOR_VALUES[color] : color) ||
191                     ZmOrganizer.COLOR_VALUES[ZmOrganizer.ORG_DEFAULT_COLOR];
192 
193             var overlay = AjxImgData[overlayName], mask = AjxImgData[maskName];
194 
195             // we're creating IMG elements here, so we can use the alt attribute
196             if (altText) {
197                 attrStr += " alt='" + AjxStringUtil.encodeQuotes(altText) + "'";
198             }
199 
200             if (AjxEnv.isIE && !AjxEnv.isIE9up) {
201                 var size = [
202                     "width:", overlay.w, "px;",
203                     "height:", overlay.h, "px;"
204                 ].join("");
205                 var location = [
206                     "top:", mask.t, ";",
207                     "left:", mask.l, ";"
208                 ].join("");
209                 var clip = [
210                     'clip:rect(',
211                     (-1 * mask.t) - 1, 'px,',
212                     overlay.w - 1, 'px,',
213                     (mask.t * -1) + overlay.h - 1, 'px,',
214                     overlay.l, 'px);'
215                 ].join('');
216                 var filter = 'filter:mask(color=' + color + ');';
217                 html = [
218                     // NOTE: Keep in sync with output of ImageMerger.java.
219                     "<div class='IEImage' style='display:inline-block;zoom:1;position:relative;overflow:hidden;", size, styles, "' ", attrStr,">",
220                         "<div class='IEImageMask' style='position:absolute;top:0px;left:0px;", size, "'>",
221                             "<img src='", mask.ief, "?v=", window.cacheKillerVersion, "' border=0 style='position:absolute;", location, filter, "'>",
222                         "</div>",
223                         "<div class='IEImageOverlay ", overlayName, "' style='", size, "position:absolute;top:0;left:0;'></div>",
224                     "</div>"
225                 ].join("");
226             }
227             else {
228                 if (!overlay[color]) {
229                     var width = overlay.w, height = overlay.h;
230 
231                     var canvas = document.createElement("CANVAS");
232                     canvas.width = width;
233                     canvas.height = height;
234 
235                     var ctx = canvas.getContext("2d");
236 
237                     ctx.save();
238                     ctx.clearRect(0,0,width,height);
239 
240                     ctx.save();
241 	                var imgId = attrStr;
242 	                if (!imgId) {
243 		                imgId = Dwt.getNextId("CANVAS_IMG_");  //create an imgId in case we need to update the img.src for an element without an id
244 		                attrStr = " id='" + imgId + "'";
245 	                }
246 	                else {
247 		                var match = attrStr.match(/id=[\"\']([^\"\']+)[\"\']+/);
248 		                if (match && match.length > 1) {
249 			                imgId = match[1]; //extract the ID value
250 		                }
251 		                AjxDebug.println(AjxDebug.TAG_ICON, "imgId = " + imgId);
252 	                }
253 	                var maskElement = document.getElementById(maskName);
254 	                var overlayElement = document.getElementById(overlayName);
255 	                if (!maskElement.complete || !overlayElement.complete) {
256 		                AjxDebug.println(AjxDebug.TAG_ICON, "mask status = " + maskElement.complete + " for " + imgId);
257 		                AjxDebug.println(AjxDebug.TAG_ICON, "overlay status = " + overlayElement.complete + " for " + imgId);
258 						var maskImg = new Image();
259 						maskImg.onload = function() {
260 							AjxDebug.println(AjxDebug.TAG_ICON, "mask image loaded");
261 							var overlayImg = new Image();
262 							overlayImg.onload = function() {
263 								AjxImg._drawCanvasImage(ctx, maskImg, overlayImg, mask, overlay, color, width, height)
264 								AjxDebug.println(AjxDebug.TAG_ICON, "overlay image loaded");
265 								var el = document.getElementById(imgId);
266 								if (el) {
267 									AjxDebug.println(AjxDebug.TAG_ICON, "element found for id = " + imgId);
268 									el.src = canvas.toDataURL();
269 									overlay[color] = canvas.toDataURL(); //only save if successful
270 								}
271 								else {
272 									AjxDebug.println(AjxDebug.TAG_ICON, "no element found for id = " + imgId);
273 								}
274 							}
275 							overlayImg.src = document.getElementById(overlayName).src;
276 	                    }
277 	                    maskImg.src = document.getElementById(maskName).src;
278 	                }
279 	                else {
280 		                //image already downloaded
281 		                AjxImg._drawCanvasImage(ctx, maskElement, overlayElement, mask, overlay, color, width, height);
282 		                overlay[color] = canvas.toDataURL();
283 	                }
284                 }
285 
286                 html = [
287                     "<img src='", overlay[color], "'"," border=0 ", AjxUtil.getClassAttr(classes), styleStr, attrStr, ">"
288                 ].join("");
289             }
290         }
291         else {
292 	        classes.push("Img" + imageName);
293             html = [
294                 "<div ", AjxUtil.getClassAttr(classes), styleStr, attrStr, ">"
295             ];
296             if (altText) {
297                 // alt is invalid on DIVs, so use a hidden element
298                 html.push(
299                     "<span class='ScreenReaderOnly'>",
300                     AjxStringUtil.htmlEncode(altText),
301                     "</span>"
302                 );
303             };
304             html.push("</div>");
305 
306             html = html.join("");
307         }
308 	}
309     else {
310         html = [
311             "<div", styleStr, attrStr, ">"
312         ];
313         if (altText) {
314             // alt is invalid on DIVs, so use a hidden element
315             html.push(
316                 "<span class='ScreenReaderOnly'>",
317                 AjxStringUtil.htmlEncode(altText),
318                 "</span>"
319             );
320         };
321         html.push("</div>");
322         html = html.join("");
323     }
324 	return pre || post ? [pre,html,post].join("") : html;
325 };

327 /**
328  * Gets the "image" as an HTML string.
329  *
330  * @param imageName		     the image you want to render
331  * @param imageStyleStr      optional style info (for example, "display:inline")
332  * @param attrStr		     optional attributes (for example, "id=X748")
333  * @param label			     the text that follows this image
334  * @param containerClassName class to use instead of the default inlineIcon class
335  * @return	{string}	     the image string
336  */
337 AjxImg.getImageSpanHtml =
338 function(imageName, imageStyleStr, attrStr, label, containerClassName) {
339     containerClassName = containerClassName || "inlineIcon";
340 	var html = [
341         "<span style='white-space:nowrap'>",
342         "<span class='",
343         containerClassName,
344         "'>",
345         AjxImg.getImageHtml(imageName, imageStyleStr, attrStr),
346         (label || ""),
347         "</span>",
348         "</span>"
349     ];
350 
351 	return html.join("");
352 };

354 /**
355  * Helper method to draw the image using both the mask image and the overlay image
356  * 
357  * @param ctx  {Object} canvas context
358  * @param maskImg   {HtmlElement} mask image object
359  * @param overlayImg {HtmlElement} overlay image object
360  * @param mask  {Object} mask object
361  * @param overlay {Object} overlay object
362  * @param color {String} color for fill
363  * @param width {int} width
364  * @param height {int} height
365  * 
366  * @private
367  */
368 AjxImg._drawCanvasImage = 
369 function(ctx, maskImg, overlayImg, mask, overlay, color, width, height) {
370 	ctx.drawImage(maskImg, mask.l, mask.t);
371 	ctx.globalCompositeOperation = "source-out";
372 	ctx.fillStyle = color;
373 	ctx.fillRect(0, 0, width, height);
374 	ctx.restore();
375 	ctx.drawImage(overlayImg, overlay.l, overlay.t);
376 	ctx.restore();	
377 };
