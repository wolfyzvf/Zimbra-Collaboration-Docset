 24 * @param parent			the element that created this view

 27 ZmResourceConflictDialog = function(parent) {
 28 	var selectId = Dwt.getNextId();
 29 
 30     var saveButton = new DwtDialog_ButtonDescriptor(ZmResourceConflictDialog.SAVE_BUTTON, ZmMsg.save, DwtDialog.ALIGN_RIGHT, null);
 31     var cancelButton = new DwtDialog_ButtonDescriptor(ZmResourceConflictDialog.CANCEL_BUTTON, ZmMsg.cancel, DwtDialog.ALIGN_RIGHT, null);
 32 
 33     //Bug fix # 80490 - Added an ID handler to the dialog
 34 	DwtDialog.call(this, {parent:parent, id:"RESC_CONFLICT_DLG", standardButtons: DwtDialog.NO_BUTTONS, extraButtons: [saveButton, cancelButton]});
 35 
 36 	this.setContent(this._contentHtml(selectId));
 37 	this.setTitle(ZmMsg.resourceConflictLabel);
 38 	
 39 	this._freeBusyStatusMap = {
 40 		"F" : "free",
 41 		"B" : "busy",
 42 		"T" : "tentative",
 43 		"O" : "outOfOffice",
 44 		"U" : "unknown"
 45 	};
 46 
 47     this.registerCallback(ZmResourceConflictDialog.SAVE_BUTTON, this._handleSaveButton, this);
 48     this.registerCallback(ZmResourceConflictDialog.CANCEL_BUTTON, this._handleCancelButton, this);
 49 };

164 ZmResourceConflictDialog.prototype.getDurationText =
165 function(inst) {
166     var hourMinOffset = 0,
167         start = new Date(),
168         userTimeZone = appCtxt.get(ZmSetting.DEFAULT_TIMEZONE),
169         currentTimeZone = AjxTimezone.getServerId(AjxTimezone.DEFAULT);
170     if(this._appt.isAllDayEvent() && userTimeZone != currentTimeZone) {
171         var offset1 = AjxTimezone.getOffset(AjxTimezone.getClientId(currentTimeZone), start);
172         var offset2 = AjxTimezone.getOffset(AjxTimezone.getClientId(userTimeZone), start);
173         hourMinOffset = (offset2 - offset1) * 60 * 1000;    //offset is in minutes convert to miliseconds
174     }
175     start = new Date(inst.s + hourMinOffset);
176 	var endTime = start.getTime() + inst.dur;
177 	var end = new Date(endTime);
178 
179     var pattern =  ZmMsg.apptTimeInstance;
180     if(this._appt.isAllDayEvent()) {
181         pattern =  ZmMsg.apptTimeAllDay;
182         if(end.getDate() != start.getDate()) {
183             pattern =  ZmMsg.apptTimeAllDayMulti;
184         }
185     }
186 
187 	return AjxMessageFormat.format(pattern, [start, end, ""]);
188 };

190 ZmResourceConflictDialog.prototype.initialize =
191 function(list, appt, callback, cancelCallback) {
192 	this._list = list;
193 	this._appt = appt;
194 	this._instData = {};
195     this._callback = callback;
196     this._cancelCallback = cancelCallback;
197 	this._canceledInstanceCount = 0;
198 	
199 	var attendeeMap = {};
200 	var types = [ZmCalBaseItem.PERSON, ZmCalBaseItem.LOCATION, ZmCalBaseItem.EQUIPMENT];
201 	
202 	for(var i = 0; i < types.length; i++) {
203 		var attendees = appt.getAttendees(types[i]);
204 		for(var j = 0; j < attendees.length; j++) {
205 			var at = attendees[j];
206 			var email = at ? at.getEmail() : null;
207 			if(email) {
208 				attendeeMap[email] = at;
209 			}
210 		}
211 	}
212 	
213 	
214 	var html = new AjxBuffer();
215 
216 	var formatter = AjxDateFormat.getDateTimeInstance(AjxDateFormat.SHORT, AjxDateFormat.MEDIUM);
217 	
218 	var size = this._conflictSize = list.length;
219 
220     var dlgC = document.getElementById(this._listId);
221     Dwt.setSize(dlgC, Dwt.DEFAULT, size > 5 ? ZmResourceConflictDialog.MAX_HEIGHT : ZmResourceConflictDialog.HEIGHT);
222 
223 	html.append("<table cellpadding=2 cellspacing=0 border=0 width=100%>");
224 	for (var i=0; i < size; i++) {
225 		var inst = list[i];
226 		var data = this._instData[i] = {inst: inst};
227 		this._addConflictInst(html, inst, data, attendeeMap, i > 0);
228 	}
229 	html.append("</table>");
230 
231 	if (this._cancelButtons) {
232 		for (var buttonId in this._cancelButtons) {
233 			this._cancelButtons[buttonId].dispose();
234 		}
235 	}
236 	this._cancelButtons = {};
237     
238 	var div = document.getElementById(this._listId);
239 	div.innerHTML = html.toString();
240 
241     if(appt.getRecurType() == ZmRecurrence.NONE && size==1) {
242         return;
243     }
244 
245     var recurrence = appt.getRecurrence();
246 
247     for (var i = 0; i < size; i++) {
248         var data = this._instData[i];
249         var cancelButtonContainer = document.getElementById(data.cancelButtonId);
250         cancelButtonContainer.innerHTML = this.getCancelHTML(recurrence.isInstanceCanceled(data.inst.ridZ));
251         Dwt.setHandler(cancelButtonContainer, DwtEvent.ONCLICK, AjxCallback.simpleClosure(this._handleCancelInstance, this, data.inst.ridZ, data.cancelButtonId, data.deltaId));
252     }
253 };

265 ZmResourceConflictDialog.prototype._handleCancelInstance =
266 function(ridZ, cancelButtonId, deltaId) {
267     var instEl = document.getElementById(ridZ + "_conflictInstTxt");
268     var deltaEl = document.getElementById(deltaId);
269     var cancelEl = document.getElementById(cancelButtonId);
270     if(instEl) {
271         var appt = this._appt;
272         var recurrence = appt.getRecurrence();
273         if(recurrence) {
274             var cancelInstance = !recurrence.isInstanceCanceled(ridZ);
275             if(cancelInstance) {
276                 recurrence.addCancelRecurId(ridZ);
277                 this._canceledInstanceCount++;
278             }else {
279                 recurrence.removeCancelRecurId(ridZ);
280                 this._canceledInstanceCount--;
281             }
282             if(cancelEl) {
283                 cancelEl.innerHTML =  this.getCancelHTML(cancelInstance);
284             }            
285         }
286     }
287 };

294 ZmResourceConflictDialog.prototype.popup =
295 function() {
296 	DwtDialog.prototype.popup.call(this);
297 	var dblBookingAllowed = appCtxt.get(ZmSetting.CAL_RESOURCE_DBL_BOOKING_ALLOWED);
298 	this._button[ZmResourceConflictDialog.SAVE_BUTTON].setEnabled(dblBookingAllowed);

301 ZmResourceConflictDialog.prototype._handleSaveButton =
302 function() {
303     if(this._callback) this._callback.run();
304     this.popdown();
305 };
306 
307 ZmResourceConflictDialog.prototype._handleCancelButton =
308 function() {
309     if(this._cancelCallback) this._cancelCallback.run();
310     if(this._appt) this._appt.getRecurrence().resetCancelRecurIds();
311     this.popdown();
312 };
313 
314 ZmResourceConflictDialog.prototype._handleIgnoreAllAndSaveButton =
315 function() {
316     var size = this._list ? this._list.length : 0;
317     var appt = this._appt;
318     for (var i = 0; i < size; i++) {
319         var data = this._instData[i];
320         if(appt && data.inst) {
321             appt._recurrence.addCancelRecurId(data.inst.ridZ);
322         }
323     }
324     this._handleSaveButton();
325 };
