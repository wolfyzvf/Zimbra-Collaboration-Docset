 29 /**
 30  * Creates an attachment type list
 31  * @class
 32  * This class represents attachment types.
 33  * 
 34  * @extends	ZmModel
 35  */
 36 ZmAttachmentTypeList = function() {
 37 	ZmModel.call(this, ZmEvent.S_ATT);
 38 };
 39 
 40 ZmAttachmentTypeList.prototype = new ZmModel;
 41 ZmAttachmentTypeList.prototype.constructor = ZmAttachmentTypeList;
 42 
 43 ZmAttachmentTypeList.prototype.isZmAttachmentTypeList = true;
 44 ZmAttachmentTypeList.prototype.toString = function() { return "ZmAttachmentTypeList"; };
 45 
 46 /**
 47  * Gets the attachments.
 48  * 
 49  * @return	{Array}	an array of attachments
 50  */
 51 ZmAttachmentTypeList.prototype.getAttachments =
 52 function() {
 53 	return this._attachments;
 54 };
 55 
 56 /**
 57  * Compares attachment type lists by description.
 58  * 
 59  * @param	{ZmAttachmentTypeList}	a			the first entry
 60  * @param	{ZmAttachmentTypeList}	b			the first entry
 61  * @return	{int}	0 if the entries match; 1 if "a" is before "b"; -1 if "b" is before "a"
 62  */
 63 ZmAttachmentTypeList.compareEntry = 
 64 function(a,b) {
 65 	if (a.desc.toLowerCase() < b.desc.toLowerCase())	{ return -1; }
 66 	if (a.desc.toLowerCase() > b.desc.toLowerCase())	{ return 1; }
 67 	return 0;
 68 };

 70 /**
 71  * Loads the attachments.
 72  * 
 73  * @param	{AjxCallback}	callback		the callback to call after load
 74  */
 75 ZmAttachmentTypeList.prototype.load =
 76 function(callback) {
 77 
 78 	this._attachments = [];
 79 
 80 	var jsonObj = {BrowseRequest:{_jsns:"urn:zimbraMail"}};
 81 	var request = jsonObj.BrowseRequest;
 82 	request.browseBy = "attachments";
 83 
 84 	var respCallback = this._handleResponseLoad.bind(this, callback);
 85 	appCtxt.getAppController().sendRequest({jsonObj: jsonObj, asyncMode: true, callback: respCallback});
 86 };
 87 
 88 /**
 89  * @private
 90  */
 91 ZmAttachmentTypeList.prototype._handleResponseLoad =
 92 function(callback, result) {
 93 	var att = this._organizeTypes(result.getResponse().BrowseResponse.bd);
 94     var isZipFileIncluded = false;
 95 	if (att) {
 96 		for (var i = 0; i < att.length; i++) {
 97 			var type = att[i]._content;
 98 			if (!ZmMimeTable.isIgnored(type)) {
 99 				this._attachments.push(ZmMimeTable.getInfo(type, true));
100 			}
101 		}
102 		this._attachments.sort(ZmAttachmentTypeList.compareEntry);
103 	}
104 
105 	if (callback) {
106 		callback.run(this._attachments);
107 	}
108 };

110 /**
111  * Check whether type is from the following list
112  * Adobe PDF
113  * Microsoft Word (doc, docx)
114  * Microsoft Powerpoint
115  * Microsoft Excel
116  * Email Message
117  * HTML
118  * Calendar (ical)
119  *
120  *
121  * @param	{String}			attachment type
122  * @return	{Boolean}           true if the type in the above list, otherwise false
123  *
124  * @private
125  */
126 ZmAttachmentTypeList.prototype._isSupportedType  =
127 function(type){
128 	var supportedTypes =  [ZmMimeTable.APP_ADOBE_PDF, ZmMimeTable.APP_MS_WORD,ZmMimeTable.APP_MS_EXCEL,
129     	                   ZmMimeTable.APP_MS_PPT, ZmMimeTable.APP_ZIP,ZmMimeTable.APP_ZIP2, ZmMimeTable.MSG_RFC822,
130         	               ZmMimeTable.TEXT_HTML, ZmMimeTable.TEXT_CAL];
131 
132     return AjxUtil.arrayContains(supportedTypes, type);
133 };
134 
135 /**
136  * Returns group type if type belongs to following group:
137  *  Text (vcard, csv)
138  *  Video (mpeg, mov)
139  *  Audio (wav, mp3, etc)
140  *  Archive (zip, etc)
141  *  Application (any)
142  *  Image (bmp, png, gif, tiff, jpg, psd, ai, jpeg)
143  *
144  * @param	{String}	    attachment type
145  * @return	{String}	    attachment group if it exits in the above list, otherwise null
146  *
147  * @private
148  */
149 ZmAttachmentTypeList.prototype._isSupportedGroup =
150 function(type){
151     var supportedGroups = [ZmMimeTable.APP,ZmMimeTable.AUDIO,ZmMimeTable.IMG,ZmMimeTable.TEXT,ZmMimeTable.VIDEO ];
152     var regExp =new RegExp("^" + supportedGroups.join("|^") ,"ig");
153     var groupType = type.match(regExp)
154     return groupType && groupType[0];
155 
156 };
157 
158 /**
159  * Returns set of supported type/group of attachments
160  *
161  * @param	{Array} list of attachment types
162  * @return	{Array} Set of types which is an intersection of att and supported types/groups
163  * @private
164  */
165 ZmAttachmentTypeList.prototype._organizeTypes =
166 function(att) {
167 
168 	var res = [];
169 	if (!(att && att.length)) {
170 		return res;
171 	}
172 
173 	for (var i = 0; i < att.length; i++) {
174 		var type = att[i]._content;
175 		var freq = att[i].freq;
176 		var skip = true;
177 		var groupType = null;
178 		if (this._isSupportedType(type)) {
179 			skip = false;
180 		} else if (type = this._isSupportedGroup(type)) {
181 			skip = false;
182 			// Check if group is already in result
183 			for (var j=0; j < res.length; j++) {
184 				if (res[j]. _content === type ) {
185 					res[j].freq += freq;
186 					skip = true;
187 					break;
188 				}
189 			}
190 		}
191 	
192 		if (!skip) {
193 			res.push({_content: type, freq: freq});
194 		}
195 
196 	}
197 	return res;
198 };
