 28 /**
 29  * Creates an object manager.
 30  * @class
 31  * This class is used to high-light objects within a given view.
 32  * 
 33  * @author Kevin Henrikson
 34  *
 35  * @param {DwtComposite}	view			the view this manager is going to high-light for
 36  * @param {AjxCallback}	selectCallback  the callback triggered when user clicks on high-light object (provide if you want to do something before the clicked on object opens its corresponding view)
 37  * @param {Boolean}	skipHandlers 	<code>true</code> to avoid adding the standard handlers
 38  */
 39 ZmObjectManager = function(view, selectCallback, skipHandlers) {
 40 
 41 	this._selectCallback = selectCallback;
 42 	this._uuid = Dwt.getNextId();
 43 	this._objectIdPrefix = "OBJ_PREFIX_";
 44 	this._objectHandlers = {};
 45 
 46 	// don't include when looking for objects. only used to provide tool tips for images
 47 	if (appCtxt.get(ZmSetting.MAIL_ENABLED) && window["ZmImageAttachmentObjectHandler"]) {
 48 		this._imageAttachmentHandler = new ZmImageAttachmentObjectHandler();
 49 	}
 50 
 51 	// create handlers (see registerHandler below)
 52 	if (!skipHandlers) {
 53         this.initialized = false;
 54         this._addAutoHandlers();
 55 	} else {
 56         this.initialized = true;
 57     }
 58 
 59     this.sortHandlers();
 60 	this.reset();
 61 	this.setView(view);
 62 };

 78 /**
 79  * Registers the handler.
 80  * 
 81  * @param	{Object}	obj		the object
 82  * @param	{constant}	type	the type
 83  * @param	{constant}	priority	the priority
 84  */
 85 ZmObjectManager.registerHandler =
 86 function(obj, type, priority) {
 87 	if (typeof obj == "string") {
 88 		obj = eval(obj);
 89 	}
 90 	var c = ZmObjectManager._autohandlers;
 91 	if (!obj.__registered) {
 92 		var id = c.push(obj);
 93 		var i = id - 1;
 94 		if(type) {
 95 			c[i].useType = type;
 96 		}
 97 		if(priority) {
 98 			c[i].usePrio = priority;
 99 		}
100 		obj.__registered = true;
101 	}
102 };
103 
104 /**
105  * @private
106  */
107 ZmObjectManager.unregisterHandler =
108 function(obj) {
109 	if (typeof obj == "string") {
110 		obj = eval(obj);
111 	}
112  	var c = ZmObjectManager._autohandlers, i;
113 	for (i = c.length; --i >= 0;) {
114 		if (c[i] === obj) {
115 			c.splice(i, 1);
116 			break;
117 		}
118 	}
119 };

131 /**
132  * Gets the handlers.
133  * 
134  * @return	{Array}	an array of {@link ZmObjectHandler} objects
135  */
136 ZmObjectManager.prototype.getHandlers =
137 function() {
138 	if (!this.initialized && appCtxt.zimletsPresent()) {
139 		var zimletMgr = appCtxt.getZimletMgr();
140 		if (zimletMgr.isLoaded()) {
141 			this.initialized = true;
142 			var zimlets = zimletMgr.getContentZimlets();
143 			for (var i = 0; i < zimlets.length; i++) {
144 				this.addHandler(zimlets[i], zimlets[i].type, zimlets[i].prio);
145 			}
146 		}
147 	}
148 	return this._objectHandlers;
149 };

151 /**
152  * Adds the handler.
153  * 
154  * @param	{ZmObjectHandler}	h		the handler
155  * @param	{constant}			type	the type
156  * @param	{constant}		priority	the priority
157  */
158 ZmObjectManager.prototype.addHandler =
159 function(h, type, priority) {
160 	type = type || (h.getTypeName() ? h.getTypeName() : "none");
161 	priority = priority ? priority : -1;
162 	h._prio = priority;
163 	//DBG.println(AjxDebug.DBG3, "addHandler " + h + " type: " + type + " prio: " + priority);
164 	var oh = this.getHandlers();
165 	if (!oh[type]) {oh[type] = [];}
166 	oh[type].push(h);
167 };
168 
169 /**
170  * Removes the handler.
171  * 
172  * @param	{ZmObjectHandler}	h		the handler
173  * @param	{constant}			type	the type
174  */
175 ZmObjectManager.prototype.removeHandler =
176 function(h, type) {
177 	type = type || (h.getTypeName() ? h.getTypeName() : "none");
178 	var oh = this.getHandlers();
179 	if (oh[type]) {
180 		for (var i = 0, count = oh[type].length; i < count; i++) {
181 			if (oh[type][i] == h) {
182 				oh[type].splice(i, 1);
183 				break;
184 			}
185 		}
186 	}
187 };

189 /**
190  * Sorts the handlers.
191  * 
192  */
193 ZmObjectManager.prototype.sortHandlers =
194 function() {
195 	this._allObjectHandlers = [];
196     var objectHandlers = this.getHandlers();
197     for (var type in objectHandlers) {
198 		// Object handlers grouped by Type
199 		objectHandlers[type].sort(ZmObjectManager.__byPriority);
200 
201 		// Copy each array to a single array of all Object Handlers
202 		for (var k = 0; k < objectHandlers[type].length; k++) {
203 			this._allObjectHandlers.push(objectHandlers[type][k]);
204 		}
205 	}
206 	this._allObjectHandlers.sort(ZmObjectManager.__byPriority);
207 };

209 /**
210  * @private
211  */
212 ZmObjectManager.prototype._addAutoHandlers =
213 function() {
214 	var c = ZmObjectManager._autohandlers, i, obj, prio;
215 	for (i = 0; i < c.length; ++i) {
216 		obj = c[i];
217 		var	handler = obj;
218 		var type = obj.TYPE;
219 		// TODO: adding internal handler should not rely on having zimlets (here and elsewhere)
220 		if (appCtxt.zimletsPresent()) {
221 			if (!(obj.isZmObjectHandler)) {
222 				handler = new obj();
223 			}
224 		}
225 		if (obj.useType) {
226 			type = obj.useType;
227 		}
228 		if (obj.usePrio) {
229 			prio = obj.usePrio;
230 		}
231 		this.addHandler(handler, type, prio);
232 	}
233 };

299 /**
300  * Finds objects.
301  * 
302  * @param	{String}	content		the content
303  * @param	{Boolean}	htmlEncode	<code>true</code> to HTML encode the content
304  * @param	{constant}	type		the type
305  * @param	{Boolean}	isTextMsg	<code>true</code> if is text msg
306  * @param	{hash}		options		arbitrary options to pass to handler
307  *
308  * @return	{String}	the object
309  */
310 ZmObjectManager.prototype.findObjects =
311 function(content, htmlEncode, type, isTextMsg, options) {
312 	if  (!content) {return "";}
313 	var html = [];
314 	var idx = 0;
315 
316 	var maxIndex = content.length;
317 	var lastIndex = 0;
318 
319     var objectHandlers = this.getHandlers();
320     while (true) {
321 		var lowestResult = null;
322 		var lowestIndex = maxIndex;
323 		var lowestHandler = null;
324 
325 		// if given a type, just go thru the handler defined for that type.
326 		// otherwise, go thru every handler we have. Regardless, ask each handler
327 		// to find us a match >= to lastIndex. Handlers that didn't match last
328 		// time will simply return, handlers that matched last time that we didn't
329 		// use (because we found a closer match) will simply return that match again.
330 		//
331 		// when we are done, we take the handler with the lowest index.
332 		var i;
333 		var handlers;
334 		var chunk;
335 		var result = null;
336 		if (type) {
337 			//DBG.println(AjxDebug.DBG3, "findObjects type [" + type + "]");
338 			handlers = objectHandlers[type];
339 			if (handlers) {
340 				for (i = 0; i < handlers.length; i++) {
341 					//DBG.println(AjxDebug.DBG3, "findObjects by TYPE (" + handlers[i] + ")");
342 					result = handlers[i].findObject(content, lastIndex, this);
343 					// No match keep trying.
344 					if(!result) {continue;}
345 					// Got a match let's handle it.
346 					if (result.index >= lowestIndex) {break;}
347 					lowestResult = result;
348 					lowestIndex = result.index;
349 					lowestHandler = handlers[i];
350 				}
351 			}
352 			// If it's an email address just handle it and return the result.
353 			if (type == "email" || content instanceof AjxEmailAddress) {
354 				if (lowestHandler) {
355                     content = this._getAjxEmailAddress(content);
356 					this.generateSpan(lowestHandler, html, idx, content, lowestResult, options);
357 				} else {
358 					html[idx++] = AjxStringUtil.htmlEncode(content.toString());
359 				}
360 				return html.join("");
361 			}
362 		} else {
363 			for (var j = 0; j < this._allObjectHandlers.length; j++) {
364 				var handler = this._allObjectHandlers[j];
365 				//DBG.println(AjxDebug.DBG3, "findObjects trying (" + handler + ")");
366 				result = handler.findObject(content, lastIndex, this);
367 				if (result && result.index < lowestIndex) {
368 					lowestResult = result;
369 					lowestIndex = result.index;
370 					lowestHandler = handler;
371 				}
372 			}
373 		}
374 
375 		if (!lowestResult) {
376 			// all done
377 			// do last chunk
378 			chunk = content.substring(lastIndex, maxIndex);
379 			if (htmlEncode) {
380 				html[idx++] = AjxStringUtil.htmlEncode(chunk, !!isTextMsg);
381 			} else {
382 				html[idx++] = chunk;
383 			}
384 			break;
385 		}
386 
387 		//  add anything before the match
388 		if (lowestIndex > lastIndex) {
389 			chunk = content.substring(lastIndex, lowestIndex);
390 			if (htmlEncode) {
391 				html[idx++] = AjxStringUtil.htmlEncode(chunk, !!isTextMsg);
392 			} else {
393 				html[idx++] = chunk;
394 			}
395 		}
396 
397 		// add the match
398 		if(lowestHandler) {
399 			idx = this.generateSpan(lowestHandler, html, idx, lowestResult[0], lowestResult.context);
400 		} else {
401 			html[idx++] = lowestResult[0];
402 		}
403 
404 		// update the index
405 		lastIndex = lowestResult.index + (lowestResult.matchLength || lowestResult[0].length);
406 	}
407 
408 	return html.join("");
409 };

412 /**
413  * Added this customized method for the sake of ZmMailMsgView performance.
414  * 
415  * TODO: Integrate this method to findObjectsInNode()
416  * 
417  * @private
418  */
419 ZmObjectManager.prototype.processObjectsInNode = function(doc, node){
420 
421     var objectManager = this;
422 	doc = doc || node.ownerDocument;
423     var tmpdiv = doc.createElement("div");
424 
425     var recurse = function(node, handlers) {
426 		var tmp, i, val, next;
427 		switch (node.nodeType) {
428 		    case 1:	// ELEMENT_NODE
429 			node.normalize();
430 			tmp = node.tagName.toLowerCase();
431 
432 			if (next == null) {
433 				if (/^(img|a)$/.test(tmp)) {
434                     var href;
435                     try {
436                         // IE can throw an "Invalid Argument" error depending on value of href
437                         // e.g: http://0:0:0:0:0:0:0:1%0:7070/service/soap/ContactActionRequest:1331608015326:9c4f5868c5b0b4f2
438                         href = node.href;
439                     }
440                     catch(e) {
441                         //do nothing
442                     }
443 
444                     var isMailToLink = tmp === "a" && ZmMailMsgView._MAILTO_RE.test(href),
445                         isUrlLink = tmp === "a" && ZmMailMsgView._URL_RE.test(href);
446 
447                     if ((isMailToLink || isUrlLink) && node.target){
448 						// tricky.
449 						var txt = isMailToLink ? href :RegExp.$1 ;
450 						tmp = doc.createElement("div");
451 						tmp.innerHTML = objectManager.findObjects(AjxStringUtil.trim(txt));
452 						tmp = tmp.firstChild;
453 						if (tmp.nodeType == 3 /* Node.TEXT_NODE */) {
454 							// probably no objects were found.  A warning would be OK here
455 							// since the regexps guarantee that objects _should_ be found.
456 							return tmp.nextSibling;
457 						}
458 						// here, tmp is an object span, but it
459 						// contains the URL (href) instead of
460 						// the original link text.
461 						node.parentNode.insertBefore(tmp, node); // add it to DOM
462 						tmp.innerHTML = "";
463 						tmp.appendChild(node); // we have the original link now
464 						return tmp.nextSibling;	// move on
465 					}
466 					handlers = false;
467 				}
468 			} else {
469 				// consider processed
470 				node = next;
471 			}
472 
473 			// bug 28264: the only workaround possible seems to be
474 			// to remove textIndent styles that have a negative value:
475 			if (parseFloat(node.style.textIndent) < 0) {
476 				node.style.textIndent = "";
477 			}
478             for (i = node.firstChild; i; i = recurse(i, handlers)) {}
479 			return node.nextSibling;
480 
481 		    case 3:	// TEXT_NODE
482 		    case 4:	// CDATA_SECTION_NODE (just in case)
483 			// generate ObjectHandler-s
484 			if (handlers && /[^\s\xA0]/.test(node.data)) try {
485  				var a = null, b = null;
486 
487 				if (!AjxEnv.isIE) {
488 					// this block of code is supposed to free the object handlers from
489 					// dealing with whitespace.  However, IE sometimes crashes here, for
490 					// reasons that weren't possible to determine--hence we avoid this
491 					// step for IE.  (bug #5345)
492 					var results = /^[\s\xA0]+/.exec(node.data);
493 					if (results) {
494 						a = node;
495 						node = node.splitText(results[0].length);
496 					}
497 					results = /[\s\xA0]+$/.exec(node.data);
498 					if (results)
499 						b = node.splitText(node.data.length - results[0].length);
500 				}
501 
502 				tmp = tmpdiv;
503 				var code = objectManager.findObjects(node.data, true, null, false);
504 				var disembowel = false;
505 				if (AjxEnv.isIE) {
506 					// Bug #6481, #4498: innerHTML in IE massacrates whitespace
507 					//            unless it sees a <pre> in the code.
508 					tmp.innerHTML = [ "<pre>", code, "</pre>" ].join("");
509 					disembowel = true;
510 				} else {
511 					tmp.innerHTML = code;
512 				}
513 
514 				if (a)
515 					tmp.insertBefore(a, tmp.firstChild);
516 				if (b)
517 					tmp.appendChild(b);
518 
519 				a = node.parentNode;
520 				if (disembowel)
521 					tmp = tmp.firstChild;
522 				while (tmp.firstChild)
523 					a.insertBefore(tmp.firstChild, node);
524 				tmp = node.nextSibling;
525 				a.removeChild(node);
526 				return tmp;
527 			} catch(ex) {};
528 		}
529 		return node.nextSibling;
530 	};
531 
532     // Parse through the DOM directly and find objects.
533 	if (node && node.childNodes && node.childNodes.length) {
534 		for (var i = 0; i < node.childNodes.length; i++){
535 			recurse(node.childNodes[i], true);
536 		}
537 	}
538 };

540 /**
541  * @private
542  */
543 ZmObjectManager.prototype.findObjectsInNode =
544 function(node, re_discard, re_allow, callbacks) {
545 	var objectManager = this, doc = node.ownerDocument, tmpdiv = doc.createElement("div");
546 
547 	if (!re_discard)
548 		re_discard = /^(script|link|object|iframe|applet)$/i;
549 
550 	// This inner function does the actual work.  BEWARE that it return-s
551 	// in various places, not only at the end.
552 	var recurse = function(node, handlers) {
553 		var tmp, i, val, next;
554 		switch (node.nodeType) {
555 		    case 1:	// ELEMENT_NODE
556 			node.normalize();
557 			tmp = node.tagName.toLowerCase();
558 			if (callbacks && callbacks.foreachElement) {
559 				next = callbacks.foreachElement(node, tmp, re_discard, re_allow);
560 			}
561 			if (next == null) {
562 				if (/^(img|a)$/.test(tmp)) {
563 					if (tmp == "a" && node.target
564 					    && (ZmMailMsgView._URL_RE.test(node.href)
565 						|| ZmMailMsgView._MAILTO_RE.test(node.href)))
566 					{
567 						// tricky.
568 						var txt = RegExp.$1;
569 						tmp = doc.createElement("div");
570 						tmp.innerHTML = objectManager.findObjects(AjxStringUtil.trim(RegExp.$1));
571 						tmp = tmp.firstChild;
572 						if (tmp.nodeType == 3 /* Node.TEXT_NODE */) {
573 							// probably no objects were found.  A warning would be OK here
574 							// since the regexps guarantee that objects _should_ be found.
575 							return tmp.nextSibling;
576 						}
577 						// here, tmp is an object span, but it
578 						// contains the URL (href) instead of
579 						// the original link text.
580 						node.parentNode.insertBefore(tmp, node); // add it to DOM
581 						tmp.innerHTML = "";
582 						tmp.appendChild(node); // we have the original link now
583 						return tmp.nextSibling;	// move on
584 					}
585 					handlers = false;
586 				} else if (re_discard.test(tmp) || (re_allow && !re_allow.test(tmp))) {
587 					tmp = node.nextSibling;
588 					node.parentNode.removeChild(node);
589 					return tmp;
590 				}
591 			} else {
592 				// consider processed
593 				node = next;
594 			}
595 
596 			if (AjxEnv.isIE) {
597 				// strips expression()-s, bwuahahaha!
598 				// granted, they get lost on the server-side anyway, but assuming some get through...
599 				// the line below exterminates them.
600 				node.style.cssText = node.style.cssText;
601 			}
602 
603 			for (i = node.firstChild; i; i = recurse(i, handlers)) {}
604 			return node.nextSibling;
605 
606 		    case 3:	// TEXT_NODE
607 		    case 4:	// CDATA_SECTION_NODE (just in case)
608 			// generate ObjectHandler-s
609 			if (handlers && /[^\s\xA0]/.test(node.data)) try {
610  				var a = null, b = null;
611 
612 				if (!AjxEnv.isIE) {
613 					// this block of code is supposed to free the object handlers from
614 					// dealing with whitespace.  However, IE sometimes crashes here, for
615 					// reasons that weren't possible to determine--hence we avoid this
616 					// step for IE.  (bug #5345)
617 					var results = /^[\s\xA0]+/.exec(node.data);
618 					if (results) {
619 						a = node;
620 						node = node.splitText(results[0].length);
621 					}
622 					results = /[\s\xA0]+$/.exec(node.data);
623 					if (results)
624 						b = node.splitText(node.data.length - results[0].length);
625 				}
626 
627 				tmp = tmpdiv;
628 				var code = objectManager.findObjects(node.data, true, null, false);
629 				var disembowel = false;
630 				if (AjxEnv.isIE) {
631 					// Bug #6481, #4498: innerHTML in IE massacrates whitespace
632 					//            unless it sees a <pre> in the code.
633 					tmp.innerHTML = [ "<pre>", code, "</pre>" ].join("");
634 					disembowel = true;
635 				} else {
636 					tmp.innerHTML = code;
637 				}
638 
639 				if (a)
640 					tmp.insertBefore(a, tmp.firstChild);
641 				if (b)
642 					tmp.appendChild(b);
643 
644 				a = node.parentNode;
645 				if (disembowel)
646 					tmp = tmp.firstChild;
647 				while (tmp.firstChild)
648 					a.insertBefore(tmp.firstChild, node);
649 				tmp = node.nextSibling;
650 				a.removeChild(node);
651 				return tmp;
652 			} catch(ex) {};
653 		}
654 		return node.nextSibling;
655 	};
656 	var df = doc.createDocumentFragment();
657 	while (node.firstChild) {
658 		df.appendChild(node.firstChild); // NODE now out of the displayable DOM
659 		recurse(df.lastChild, true, this);	 // parse tree and findObjects()
660 	}
661 	node.appendChild(df);	// put nodes back in the document
662 };

664 /**
665  * Sets handler attribute.
666  * 
667  * @param	{String}	type		the type
668  * @param	{String}	name		the attribute name
669  * @param	{Object}	value		the value
670  */
671 ZmObjectManager.prototype.setHandlerAttr =
672 function(type, name, value) {
673     var handlers = this.getHandlers()[type];
674 	if (handlers) {
675 		for (var i = 0; i < handlers.length; i++) {
676 			handlers[i][name] = value;
677 		}
678 	}
679 };

682  * Generates the span.
683  * 
684  * @private
685  */
686 ZmObjectManager.prototype.generateSpan =
687 function(handler, html, idx, obj, context, options) {
688 	var id = this._objectIdPrefix + Dwt.getNextId();
689     if (handler && handler.name) {
690         id = id + "_" + handler.name;
691     }
692 	this._objects[id] = {object: obj, handler: handler, id: id, context: context };
693 	return handler.generateSpan(html, idx, obj, id, context, options);
694 };

698  */
699 ZmObjectManager.prototype._findObjectSpan =
700 function(e) {
701 	while (e && (!e.id || e.id.indexOf(this._objectIdPrefix) !== 0)) {

707 /**
708  * @private
709  */
710 ZmObjectManager.prototype._mouseOverListener =
711 function(ev) {
712 	var span = this._findObjectSpan(ev.target);
713 	if (!span) {return false;}
714 	var object = this._objects[span.id];
715 	if (!object) {return false;}
716 
717 	span.className = object.handler.getHoveredClassName(object.object, object.context, span.id);
718 	if (object.handler.hasToolTipText()) {
719 		var shell = DwtShell.getShell(window);
720 		var manager = shell.getHoverMgr();
721 		if ((!manager.isHovering() || manager.getHoverObject() != object) && !DwtMenu.menuShowing()) {
722 			manager.reset();
723 			manager.setHoverOverDelay(ZmObjectManager._TOOLTIP_DELAY);
724 			manager.setHoverObject(object);
725 			manager.setHoverOverData(object);
726 			manager.setHoverOverListener(this._hoverOverListener);
727 			manager.hoverOver(ev.docX, ev.docY);
728 			ev.hoverStarted = true;
729 		}
730 	}
731 
732 	ev._returnValue = true;
733 	ev._dontCallPreventDefault = true;
734 	return false;
735 };

790 /**
791  * @private
792  */
793 ZmObjectManager.prototype._mouseDownListener =
794 function(ev) {
795 
796 	// "authoritative" means a previous listener doesn't want propagation to get reset
797 	if (!ev._authoritative) {
798 		ev._dontCallPreventDefault = true;
799 		ev._returnValue = true;
800 		ev._stopPropagation = false;
801 	}
802 
803 	var span = this._findObjectSpan(ev.target);
804 	if (!span) {
805 		return true;
806 	}
807 	var object = this._objects[span.id];
808 	if (!object) {
809 		return true;
810 	}
811 
812 	ev._stopPropagation = true;
813 
814 	var shell = DwtShell.getShell(window);
815 	var manager = shell.getHoverMgr();
816 	manager.setHoverOutDelay(0);
817 	manager.setHoverOutData(object);
818 	manager.setHoverOutListener(this._hoverOutListener);
819 	manager.hoverOut();
820 
821 	span.className = object.handler.getActiveClassName(object.object, object.context, span.id);
822 	if (ev.button == DwtMouseEvent.RIGHT) {
823 		var menu = object.handler.getActionMenu(object.object, span, object.context, ev);
824 		if (menu) {
825 			menu.popup(0, ev.docX, ev.docY);
826 			// if we have an action menu, don't let the browser show its context menu too
827 			ev._dontCallPreventDefault = false;
828 			ev._returnValue = false;
829 			ev._stopPropagation = true;
830 			return true;
831 		}
832 	} else if (ev.button == DwtMouseEvent.LEFT) {
833 		if (this._selectCallback) {
834 			this._selectCallback.run();
835 		}
836 		object.handler.selected(object.object, span, ev, object.context);
837 		return true;
838 	}
839 	return false;
840 };
