 26 /**
 27  * Creates a new tag dialog.
 28  * @class
 29  * This class represents a new tag dialog.
 30  * 
 31  * @param	{DwtControl}	parent		the parent
 32  * @param	{String}	className		the class name
 33  * 
 34  * @extends		ZmDialog
 35  */
 36 ZmNewTagDialog = function(parent, className) {
 37 	ZmDialog.call(this, {parent:parent, className:className, title:ZmMsg.createNewTag, id:"CreateTagDialog"});
 38 
 39 	this._setNameField(this._htmlElId+"_name");
 40 	this._setTagColorMenu();
 41 	this._setAccountMenu();
 42 	DBG.timePt("set content");
 43 };
 44 
 45 ZmNewTagDialog.prototype = new ZmDialog;
 46 ZmNewTagDialog.prototype.constructor = ZmNewTagDialog;
 47 
 48 ZmNewTagDialog.prototype.toString = 
 49 function() {
 50 	return "ZmNewTagDialog";
 51 };
 52 
 53 /**
 54  * Pops-up the dialog.
 55  * 
 56  * @param	{ZmOrganizer}	org		the organizer
 57  * @param	{ZmAccount}		account	the account
 58  */
 59 ZmNewTagDialog.prototype.popup =
 60 function(org, account) {
 61 	if (this._accountSelect) {
 62 		var acct = account || appCtxt.getActiveAccount();
 63 		this._accountSelect.setSelectedValue(acct.id);
 64 	}
 65 
 66 	ZmDialog.prototype.popup.call(this);
 67 };
 68 
 69 ZmNewTagDialog.prototype.cleanup =
 70 function(bPoppedUp) {
 71 	DwtDialog.prototype.cleanup.call(this, bPoppedUp);
 72 	var color = this._getNextColor();
 73 	this._setColorButton(color, ZmOrganizer.COLOR_TEXT[color], ZmTag.getIcon(color));
 74 };
 75 
 76 ZmNewTagDialog.prototype._colorListener = 
 77 function(ev) {
 78 	var color = ev.item.getData(ZmOperation.MENUITEM_ID);
 79 	this._setColorButton(color, ZmOrganizer.COLOR_TEXT[color], ZmTag.getIcon(color));
 80 };
 81 
 82 ZmNewTagDialog.prototype._setTagColorMenu =
 83 function() {
 84 	var fieldId = this._htmlElId + "_tagColor";
 85 	this._colorButton = new DwtButton({parent:this, parentElement:fieldId, id:"ZmTagColorMenu"});
 86 	this._colorButton.noMenuBar = true;
 87 
 88 	var menu = ZmOperation.addColorMenu(this._colorButton, true);
 89 
 90 	var color = ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
 91 	this._setColorButton(color, ZmOrganizer.COLOR_TEXT[color], ZmTag.getIcon(color));
 92 
 93 	this._tagColorListener = new AjxListener(this, this._colorListener);
 94     menu.addSelectionListener(this._tagColorListener);
 95 };
 96 
 97 ZmNewTagDialog.prototype._setAccountMenu =
 98 function() {
 99 	if (!appCtxt.multiAccounts) { return; }
100 
101 	var fieldId = this._htmlElId + "_account";
102 	this._accountSelect = new DwtSelect({parent:this, parentElement:fieldId});
103 
104 	var accounts = appCtxt.accountList.visibleAccounts;
105 	for (var i = 0; i < accounts.length; i++) {
106 		var acct = accounts[i];
107 		if (appCtxt.get(ZmSetting.TAGGING_ENABLED, null, acct)) {
108 			var o = new DwtSelectOption(acct.id, null, acct.getDisplayName(), null, null, acct.getIcon());
109 			this._accountSelect.addOption(o);
110 		}
111 	}
112 };
113 
114 ZmNewTagDialog.prototype._setColorButton =
115 function(color, text, image) {
116 	this._colorButton.setData(ZmOperation.MENUITEM_ID, color);
117 	this._colorButton.setText(text || ZmMsg.custom);
118 	this._colorButton.setImage(image);
119 };

124 };

126 ZmNewTagDialog.prototype._okButtonListener =
127 function(ev) {
128 	var results = this._getTagData();
129 	if (results) {
130 		DwtDialog.prototype._buttonListener.call(this, ev, results);
131 	}
132 };
133 
134 ZmNewTagDialog.prototype._getTagData =
135 function() {
136 	var acctId = this._accountSelect && this._accountSelect.getValue();
137 	var account = acctId && appCtxt.accountList.getAccount(acctId);
138 
139 	// check name for presence and validity
140 	var name = AjxStringUtil.trim(this._nameField.value);
141 	var msg = ZmTag.checkName(name);
142 
143 	// make sure tag doesn't already exist
144 	var tagTree = appCtxt.getTagTree(account);
145 	if (!msg && tagTree && tagTree.getByName(name)) {
146 		msg = ZmMsg.tagNameExists;
147 	}
148 
149 	if (msg) return this._showError(msg);
150     var color = this._colorButton.getData(ZmOperation.MENUITEM_ID);
151     var data = {name:name, color:color, accountName:(account && account.name)};
152     if (String(color).match(/^#/)) {
153         data.rgb = color;
154         delete data.color;
155     }
156     return data;
157 };
158 
159 ZmNewTagDialog.prototype._enterListener =
160 function(ev) {
161 	var results = this._getTagData();
162 	if (results) {
163 		this._runEnterCallback(results);
164 	}
165 };
166 
167 ZmNewTagDialog.prototype._getNextColor =
168 function() {
169 	var colorUsed = {};
170 	var tagTree = appCtxt.getTagTree();
171 	if (!tagTree) {
172 		return ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
173 	}
174 
175 	var tags = tagTree.root.children.getArray();
176 	if (!(tags && tags.length)) {
177 		return ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
178 	}
179 
180 	for (var i = 0; i < tags.length; i++) {
181 		colorUsed[tags[i].color] = true;
182 	}
183 	for (var i = 0; i < ZmTagTree.COLOR_LIST.length; i++) {
184 		var color = ZmTagTree.COLOR_LIST[i];
185 		if (!colorUsed[color]) {
186 			return color;
187 		}
188 	}
189 
190 	return ZmOrganizer.DEFAULT_COLOR[ZmOrganizer.TAG];
191 };
192 
193 ZmNewTagDialog.prototype._getTabGroupMembers =
194 function() {
195 	return [this._nameField, this._colorButton];
196 };
