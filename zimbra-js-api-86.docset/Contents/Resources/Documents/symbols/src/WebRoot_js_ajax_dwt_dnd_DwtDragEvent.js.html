 22 /**
 23  * @constructor
 24  * @class
 25  * DwtDragEvent is generated by the Drag and Drop framework when a drag operation is
 26  * in process. The drag event is dispatched to the registered {@link DwtDragSource} instance.
 27  * 
 28  * @author Ross Dargahi
 29  * 
 30  * @see DwtDragSource
 31  */
 32 DwtDragEvent = function() {
 33 	/**
 34 	 * Type of drag operation. One of:
 35 	 * <ul>
 36 	 * <li>{@link DwtDragEvent.DRAG_START}</li>
 37 	 * <li>{@link DwtDragEvent.SET_DATA}</li>
 38 	 * <li>{@link DwtDragEvent.DRAG_END}</li>
 39 	 * </ul>
 40 	 */
 41 	this.operation = null;
 42 	
 43 	/**
 44 	 * Drag source control
 45 	 * @type DwtControl
 46 	 */
 47 	this.srcControl = null;
 48 	
 49 	/**
 50 	 * Action being performed. One of:
 51 	 * <ul>
 52 	 * <li>{@link Dwt.DND_DROP_NONE}</li>
 53 	 * <li>{@link Dwt.DND_DROP_COPY}</li>
 54 	 * <li>{@link Dwt.DND_DROP_MOVE}</li>
 55 	 * </ul>
 56 	 */
 57 	this.action = null;
 58 	
 59 	/**
 60 	 * Whether the DnD framework should perform the operation. The application is
 61 	 * responsible for setting this value based on whatever business logic it is
 62 	 * implementing
 63 	 * @type boolean
 64 	 */
 65 	this.doIt = false;
 66 	
 67 	/**
 68 	 * Drag source data. This is the application data associated with the item being dragged.
 69 	 */
 70 	this.srcData = null;
 71 };
 72 
 73 /**
 74  * Drag initialization.
 75  */
 76 DwtDragEvent.DRAG_INIT = "INIT";
 77 
 78 /**
 79  * Drag is starting.
 80  */
 81 DwtDragEvent.DRAG_START = "START";
 82 
 83 /**
 84  * Set the <code>srcData</code> field of the event.
 85  */
 86 DwtDragEvent.SET_DATA = "SET_DATA";
 87 
 88 /**
 89  * Drag movement has occurred.
 90  */
 91 DwtDragEvent.DRAG_MOVE = "MOVE";
 92 
 93 /**
 94  * Drag has ended.
 95  */
 96 DwtDragEvent.DRAG_END = "END";
 97 
 98 /**
 99  * Drag canceled (i.e. dropped on invalid target).
100  */
101 DwtDragEvent.DRAG_CANCEL = "CANCEL";
