 22 /**
 23  * Creates a new controller to show mail send confirmation.
 24  * @constructor
 25  * @class
 26  * This class represents the mail confirmation controller.
 27  * 
 28  * @param {DwtShell}	container	the containing shell
 29  * @param {ZmApp}		mailApp		the containing app
 30  * @param {constant}	type		controller type
 31  * @param {string}		sessionId	the session id
 32  * 
 33  * @extends		ZmController
 34  */
 35 ZmMailConfirmController = function(container, mailApp, type, sessionId) {
 36 
 37 	ZmController.apply(this, arguments);
 38 	this._elementsToHide = ZmAppViewMgr.LEFT_NAV;
 39 };
 40 
 41 ZmMailConfirmController.prototype = new ZmController();
 42 ZmMailConfirmController.prototype.constructor = ZmMailConfirmController;
 43 
 44 ZmMailConfirmController.prototype.isZmMailConfirmController = true;
 45 ZmMailConfirmController.prototype.toString = function() { return "ZmMailConfirmController"; };
 46 
 47 ZmMailConfirmController.getDefaultViewType =
 48 function() {
 49 	return ZmId.VIEW_MAIL_CONFIRM;
 50 };
 51 ZmMailConfirmController.prototype.getDefaultViewType = ZmMailConfirmController.getDefaultViewType;
 52 
 53 /**
 54  * Shows the confirmation that the message was sent.
 55  *
 56  * @param	{ZmMailMsg}				msg					the message that was sent
 57  * @param	{constant}				composeViewId		the compose view id
 58  * @param	{constant}				composeTabId		the compose tab id
 59  * @param	{ZmComposeController}	controller			compose controller
 60  */
 61 ZmMailConfirmController.prototype.showConfirmation =
 62 function(msg, composeViewId, composeTabId, controller) {
 63 
 64 	this._composeViewId = composeViewId;
 65 	this._composeTabId = composeTabId;
 66 	this._composeController = controller;
 67 
 68 	if (!this._view) {
 69 		this._initView();
 70 	}
 71 
 72     this._initializeToolBar();
 73 	this.resetToolbarOperations(this._toolbar);
 74 	this._view.showConfirmation(msg);
 75 
 76 	if (appCtxt.isChildWindow) {
 77 		appCtxt.getAppViewMgr()._setViewVisible(ZmId.VIEW_LOADING, false);
 78 	}
 79 
 80 	var avm = appCtxt.getAppViewMgr();
 81 	avm.pushView(this._currentViewId);
 82 };
 83 
 84 ZmMailConfirmController.prototype.resetToolbarOperations =
 85 function() {
 86 	this._toolbar.enableAll(true);
 87 };
 88 
 89 ZmMailConfirmController.prototype.getKeyMapName =
 90 function() {
 91 	return ZmKeyMap.MAP_GLOBAL;
 92 };
 93 
 94 ZmMailConfirmController.prototype.handleKeyAction =
 95 function(actionCode) {
 96 	switch (actionCode) {
 97 		case ZmKeyMap.CANCEL:
 98 			this._closeListener();
 99 			break;
100 
101 		default:
102 			return ZmController.prototype.handleKeyAction.call(this, actionCode);
103 			break;
104 	}
105 	return true;
106 };
107 
108 ZmMailConfirmController.prototype._initView =
109 function() {
110 	this._view = new ZmMailConfirmView(this._container, this);
111 	this._view.addNewContactsListener(new AjxListener(this, this._addNewContactsListener));
112 
113 	var tg = this._createTabGroup();
114 	var rootTg = appCtxt.getRootTabGroup();
115 	tg.newParent(rootTg);
116 	tg.addMember(this._view.getTabGroupMember());
117 
118 	this._initializeToolBar();
119 	var elements = this.getViewElements(null, this._view, this._toolbar);
120 
121 	var callbacks = {};
122 	callbacks[ZmAppViewMgr.CB_PRE_HIDE] = this._preHideCallback.bind(this);
123 	callbacks[ZmAppViewMgr.CB_POST_SHOW] = this._postShowCallback.bind(this);
124     this._app.createView({	viewId:		this._currentViewId,
125 							viewType:	this._currentViewType,
126 							elements:	elements,
127 							hide:		this._elementsToHide,
128 							controller:	this,
129 							callbacks:	callbacks,
130 							tabParams:	{ id:this._composeTabId }});
131 };
132 
133 ZmMailConfirmController.prototype._initializeToolBar =
134 function() {
135 	if (this._toolbar) return;
136 
137 	var buttons = [ZmOperation.CLOSE];
138 
139 	var className = appCtxt.isChildWindow ? "ZmAppToolBar_cw" : "ZmAppToolBar";
140 	this._toolbar = new ZmButtonToolBar({parent:this._container, buttons:buttons, className:className+" ImgSkin_Toolbar",
141 										 context:ZmId.VIEW_MAIL_CONFIRM});
142 	this._toolbar.addSelectionListener(ZmOperation.CLOSE, new AjxListener(this, this._closeListener));
143 };
144 
145 ZmMailConfirmController.prototype._getDefaultFocusItem =
146 function() {
147 	return this._view.getDefaultFocusItem();
148 };
149 
150 ZmMailConfirmController.prototype._closeListener =
151 function() {
152 	this._doClose();
153 };
154 
155 ZmMailConfirmController.prototype._addNewContactsListener =
156 function(attrs) {
157 	if (!attrs.length) {
158 		this._doClose();
159 		return;
160 	}
161 	
162 	var batchCommand = new ZmBatchCommand(false, null, true);
163 	for (var i = 0, count = attrs.length; i < count; i++) {
164 		var contact = new ZmContact();
165 		batchCommand.add(new AjxCallback(contact, contact.create, [attrs[i]]));
166 	}
167 	batchCommand.run(new AjxCallback(this, this._handleResponseCreateContacts));
168 };
169 
170 ZmMailConfirmController.prototype._handleResponseCreateContacts =
171 function() {
172 	this._doClose();
173 };
174 
175 ZmMailConfirmController.prototype._doClose =
176 function() {
177 	appCtxt.getAppViewMgr().popView(true);
178 };
