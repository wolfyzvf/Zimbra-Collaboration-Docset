 22 /**
 23  * Creates a combo box.
 24  * @constructor
 25  * @class
 26  * This class represents a combo box.
 27  *
 28  * @author Dave Comfort
 29  * 
 30  * @param {hash}	params		a hash of parameters
 31  * @param {DwtComposite}      parent		the parent widget
 32  * @param {boolean}	useLabel		Set to true if the value should be shown in a DwtLabel. Defaults to false, showing it in a DwtInputField.
 33  * @param {hash}	inputParams		params for the input (see {@link DwtInputField} or {@link DwtLabel})
 34  * @param {string}      className		the CSS class
 35  * @param {constant}      posStyle		the positioning style (see {@link DwtControl})
 36  * @param {int}     maxRows         The number of maxRows needed in drop down(see {@link DwtMenu})
 37  * @param {constant} layout         The layout of the drop down(see {@link DwtMenu})
 38  * @param {boolean} autoScroll    Set to true if auto scroll to the input text is needed. Defaults to false.
 39  * 
 40  * @extends		DwtComposite
 41  */
 42 DwtComboBox = function(params) {
 43     if (arguments.length == 0) { return; }
 44 	params = Dwt.getParams(arguments, DwtComboBox.PARAMS);
 45     params.className = params.className || "DwtComboBox";
 46     DwtComposite.call(this, params);
 47     
 48     this.input = null;
 49 	this._menu = null;
 50     this._button = null;
 51     
 52     this._textToValue = {}; // Map of text strings to their values.
 53     this._valueToText = {};
 54     this._valueToItem = {};
 55 	this._size = 0;
 56 
 57     this._hasMenuCallback = true;
 58 	this._menuItemListenerObj = new AjxListener(this, this._menuItemListener);
 59 
 60     this._inputParams = params.inputParams;
 61 	this._useLabel = Boolean(params.useLabel);
 62 	this._maxRows = params.maxRows;
 63 	this._layout = params.layout;
 64 	this._autoScroll = params.autoScroll || false;
 65     this._createHtml();
 66 };
 67

108 /**
109  * Adds an entry to the combo box list.
110  * 
111  * @param {string}	text		the user-visible text for the entry
112  * @param {string}	value		the value for the entry
113  * @param {boolean}	selected	if <code>true</code>, the entry is selected
114  */
115 DwtComboBox.prototype.add =
116 function(text, value, selected) {
117 	this._textToValue[text] = value;
118     this._valueToText[value] = text;
119     if (!this._hasMenuCallback) {
120 		var menu = this._button.getMenu();
121     	this._createMenuItem(menu, text);
122 	}
123 	if (selected) {
124 		this.setText(text);
125 	}
126 	this._size++;
127 	this._updateButton();
128 };
129 
130 /**
131  * Removes the specified value from the list.
132  *
133  * @param	{string}	value		the value
134  */
135 DwtComboBox.prototype.remove = function(value) {
136     var item = this._valueToItem[value];
137     if (item) {
138         this._button.getMenu().removeChild(item);
139         var text = this._valueToText[value];
140         delete this._textToValue[text];
141         delete this._valueToText[value];
142         delete this._valueToItem[value];
143         if (this.getText() == text) {
144             this.setText("");
145         }
146 		this._size--;
147 		this._updateButton();
148 	}
149 };
150 
151 /**
152  * Removes all the items in the list.
153  * 
154  */
155 DwtComboBox.prototype.removeAll = function() {
156     this._button.setMenu(new AjxCallback(this, this._createMenu), true);
157     this._hasMenuCallback = true;
158 
159     this._textToValue = {};
160     this._valueToText = {};
161     this._valueToItem = {};
162 	this._size = 0;
163 	this._updateButton();
164 };

166 /**
167  * Gets the value of the currently selected entry. If the entry
168  * is one that was not added via the add method (that is, if it was
169  * typed in by the user) then <code>null</code> is returned.
170  * 
171  * @return	{string}	the value
172  */
173 DwtComboBox.prototype.getValue =
174 function() {
175 	var text = this.getText();
176 	return this._textToValue[text];
177 };
178 
179 /**
180  * Sets the value.
181  * 
182  * @param	{string}	value		the value
183  */
184 DwtComboBox.prototype.setValue = function(value) {
185 	var text = this._valueToText[value];
186 	this.setText(text || value);
187 };

189 /**
190  * Gets the text of the currently selected entry.
191  * 
192  * @return	{string}	the text
193  */
194 DwtComboBox.prototype.getText =
195 function() {
196 	return this._useLabel ? this.input.getText() : this.input.getValue();
197 };
198 
199 /**
200  * Sets the selected text.
201  * 
202  * @param	{string}	text		the text
203  */
204 DwtComboBox.prototype.setText =
205 function(text) {
206 	if (this._useLabel)
207 		this.input.setText(text);
208 	else
209 		this.input.setValue(text);
210 };

213 function(enabled) {

216 		this.input.setEnabled(enabled);
217 		this._button.setEnabled(enabled);
218     }

234 DwtComboBox.prototype._createMenu =
235 function() {
236 	var params = {parent:this};
237 	if (this._maxRows) {
238 		params.maxRows = this._maxRows;
239 	}
240 	if (this._layout) {
241 		params.layout = this._layout;
242 	}
243     var menu = this._menu = new DwtMenu(params);
244     for (var i in this._textToValue) {
245     	var item = this._createMenuItem(menu, i);
246         var value = this._textToValue[i];
247         this._valueToItem[value] = item;
248     }
249 	this._hasMenuCallback = false;
250 	return menu;
251 };
252 
253 DwtComboBox.prototype._createMenuItem =
254 function(menu, text) {
255 	var item = new DwtMenuItem({parent:menu});
256 	item.setText(text);
257 	item.addSelectionListener(this._menuItemListenerObj);
258 	if (!this._menuWidth) {
259 		this._menuWidth = this.getW() - 10; // 10 is some fudge factor that lines up the menu right.
260 	}
261     item.getHtmlElement().style.minWidth = this._menuWidth;
262     return item;
263 };
264 
265 DwtComboBox.prototype._menuItemListener =
266 function(ev) {
267 	var menuItem = ev.dwtObj;
268 	var ovalue = this.getText();
269 	var nvalue = menuItem.getText();
270 	this.setText(nvalue);
271 	this._menu.popdown();
272 
273 	// notify our listeners
274 	var event = DwtUiEvent.getEvent(ev);
275 	event._args = { selectObj: this, newValue: nvalue, oldValue: ovalue };
276 	this.notifyListeners(DwtEvent.ONCHANGE, event);
277 
278 	if (!this._useLabel) {
279 	    var input = this.input.getInputElement();
280 	    input.focus();
281 	    input.select();
282 	}
283 };

285 DwtComboBox.prototype._handleKeyDown = function(ev) {
286 	var keycode = DwtKeyEvent.getCharCode(ev);
287 
288 	this.__ovalue = this.getText();
289 
290 	return true;
291 };
292 
293 DwtComboBox.prototype._handleKeyUp = function(ev) {
294 	// propagate event to DwtInputField
295 	DwtInputField._keyUpHdlr(ev);
296 	// notify our listeners
297 	var event = DwtUiEvent.getEvent(ev);
298 	var newValue = this.getText();
299 	event._args = { selectObj: this, newValue: newValue, oldValue: this.__ovalue };
300 	this.notifyListeners(DwtEvent.ONCHANGE, event);
301 	if (this._menu && this._autoScroll && newValue != this.__ovalue) {
302 		//if auto scroll is on then scroll to the index which starts with
303 		//the value in input field
304 		var index = 0;
305 		for (var text in this._textToValue) {
306 			if (text.indexOf(newValue) == 0) {
307 				this._menu.scrollToIndex(index);
308 				break;
309 			}
310 			index++;
311 		}
312 	}
313 	return true;
314 };

316 DwtComboBox.prototype._updateButton =
317 function() {
318 	this._button.setVisible(this._size > 0);
319 };
320 
321 DwtComboBox.prototype._createHtml = function(templateId) {
322     var data = { id: this._htmlElId };
323     this._createHtmlFromTemplate(templateId || this.TEMPLATE, data);
324 };
325 
326 DwtComboBox.prototype._createHtmlFromTemplate = function(templateId, data) {
327     DwtComposite.prototype._createHtmlFromTemplate.call(this, templateId, data);
328 
329 	var inputParams = this._inputParams || {};
330 	inputParams.parent = this;
331 	inputParams.size = inputParams.size || 40;
332 	delete this._inputParams;
333     
334 	this.input = (this._useLabel ?
335 	              new DwtLabel(inputParams) : new DwtInputField(inputParams));
336     this.input.replaceElement(data.id + "_input");
337 	this.input.setHandler(DwtEvent.ONKEYDOWN, AjxCallback.simpleClosure(this._handleKeyDown, this));
338 	this.input.setHandler(DwtEvent.ONKEYUP, AjxCallback.simpleClosure(this._handleKeyUp, this));
339 
340     this._button = new DwtComboBoxButton({parent:this});
341 	this._button.setMenu(new AjxListener(this, this._createMenu), true);
342     this._button.replaceElement(data.id + "_button");
343 	this._updateButton();
344 
345 	this._tabGroup = new DwtTabGroup(this._htmlElId);
346 	this._tabGroup.addMember(this.input);
347 	this._tabGroup.addMember(this._button);
348 };

350 /**
351  * The input field inherits the id for accessibility purposes.
352  * 
353  * @private
354  */
355 DwtComboBox.prototype._replaceElementHook =
356 function(oel, nel, inheritClass, inheritStyle) {
357 	DwtComposite.prototype._replaceElementHook.apply(this, arguments);
358 	// set input settings
359 	if (oel.size) {
360 		var el = (this._useLabel ?
361 		          this.input.getHtmlElement() :
362 		          this.input.getInputElement());
363 		el.size = oel.size;
364 	}
365 	if (oel.title) {
366 		this.input.setHint(oel.title);
367 	}
368 };

376  * 
377  * @param {hash}	params		a hash of parameters
378  * @param {DwtComposite}       params.parent		the parent widget
379  * @param	{string}       params.className		the CSS class
