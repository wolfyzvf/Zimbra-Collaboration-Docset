 22 /**
 23  * @overview
 24  * 
 25  * This file defines an application event.
 26  *
 27  */
 28 
 29 /**
 30  * Creates an empty application event.
 31  * @class
 32  * This class represents an event related to a change of state for an individual
 33  * application or for ZCS as a whole.
 34  * 
 35  * @param {Object}	the application to which this event applies; if <code>null</code>, the event applies to ZCS
 36  * 
 37  * @extends		ZmEvent
 38  */
 39 ZmAppEvent = function(app) {
 40 	ZmEvent.call(this);
 41 };
 42 
 43 ZmAppEvent.prototype = new ZmEvent;
 44 ZmAppEvent.prototype.constructor = ZmAppEvent;
 45 
 46 /**
 47  * Event used to notify listeners before startup (i.e. before the first
 48  * app is activated). This is a bit of a misnomer because this event occurs
 49  * after the apps are initialized but before the first app is shown. This
 50  * allows code to be executed after the apps have registered settings
 51  * but before the app actually acts on those settings.
 52  *
 53  * @see ZmAppEvent.POST_STARTUP
 54  */
 55 ZmAppEvent.PRE_STARTUP	= "PRESTARTUP";
 56 
 57 /**
 58  * Defines the event used to notify listeners post-startup.
 59  */
 60 ZmAppEvent.POST_STARTUP	= "POSTSTARTUP";
 61 /**
 62  * Defines the event used to notify listeners pre-launch.
 63  */
 64 ZmAppEvent.PRE_LAUNCH	= "PRELAUNCH";
 65 /**
 66  * Defines the event used to notify listeners post-launch.
 67  */
 68 ZmAppEvent.POST_LAUNCH	= "POSTLAUNCH";
 69 /**
 70  * Defines the event used to notify listeners post-render.
 71  */
 72 ZmAppEvent.POST_RENDER	= "POSTRENDER";
 73 
 74 ZmAppEvent.ACTIVATE	= "ACTIVATE";
 75 
 76 // Triggered after processing of an async response finishes
 77 ZmAppEvent.RESPONSE = "RESPONSE";
 78 
 79 /**
 80  * Returns a string representation of the object.
 81  * 
 82  * @return		{String}		a string representation of the object
 83  */
 84 ZmAppEvent.prototype.toString =
 85 function() {
 86 	return "ZmAppEvent";
 87 };
