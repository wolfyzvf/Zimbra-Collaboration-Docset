 23 /**
 24  * XSLT engine <a href="http://www.w3.org/TR/xslt">http://www.w3.org/TR/xslt</a>
 25  * @class
 26  * Supports IE and Firefox. Use the following static methods to create instance.
 27  *
 28  * <pre>
 29  * xslt = AjxXslt.createFromUrl(url of the stylesheet)
 30  * xslt = AjxXslt.createFromString(stylesheet in string)
 31  * </pre>
 32  *
 33  * Then apply the transformation on a document.  Two methods are available depending on the needs.
 34  *
 35  * <pre>
 36  * dom = xslt.transformToDom(doc);
 37  * xml = xslt.transformToString(doc);
 38  * </pre>
 39  * 
 40  * @private
 41  */
 42 AjxXslt = function() {
 43 	var doc = AjxXmlDoc.create();
 44 	if (AjxEnv.isIE) {
 45 		var msdoc = null;
 46 		var vers = ["MSXML2.FreeThreadedDOMDocument.5.0", "MSXML2.FreeThreadedDOMDocument.3.0"];
 47 		for (var i = 0; i < vers.length; i++) {
 48 			try {
 49 				msdoc = new ActiveXObject(vers[i]);
 50 				break;
 51 			} catch (ex) {
 52 			}
 53 		}
 54 		if (!msdoc) {
 55 			throw new AjxException("FreeThreadedDOMDocument", AjxException.UNSUPPORTED, "AjxXslt");
 56 		}
 57 		msdoc.async = false;
 58 		doc._doc = msdoc;
 59 	}
 60 	this._doc = doc;
 61 };
 62 
 63 AjxXslt.prototype.toString =
 64 function() {
 65 	return "AjxXslt";
 66 };
 67 
 68 AjxXslt.createFromUrl =
 69 function(url) {
 70 	var xslt = new AjxXslt();
 71 
 72 	xslt.loadUrl(url);
 73 
 74 	return xslt;
 75 };
 76 
 77 AjxXslt.createFromString =
 78 function(str) {
 79 	var xslt = new AjxXslt();
 80 	
 81 	xslt._doc.loadFromString(str);
 82 	xslt.createProcessor();
 83 	
 84 	return xslt;
 85 };
 86 
 87 AjxXslt.prototype.createProcessor =
 88 function() {
 89 	var doc = this._doc.getDoc();
 90 	if (AjxEnv.isNav || AjxEnv.isChrome || AjxEnv.isSafari) {
 91 		this._processor = new XSLTProcessor();
 92 		if(this._processor) {
 93 			this._processor.importStylesheet(doc);
 94 		}
 95 	} else if (AjxEnv.isIE) {
 96 		var err = doc.parseError;
 97 	    if (err.errorCode != 0) {
 98 			DBG.println(AjxDebug.DBG1, "Parse error (" + err.reason + ") at line " + err.line + ", character " + err.linepos + "\n" + err.srcText);
 99 			throw new AjxException(err.reason, AjxException.INVALID_PARAM, "AjxXslt.createProcessor");
100 		}
101 
102 		var proc = null;
103 		var vers = ["MSXML2.XSLTemplate.5.0", "MSXML2.XSLTemplate.3.0"];
104 		for (var i = 0; i < vers.length; i++) {
105 			try {
106 				proc = new ActiveXObject(vers[i]);
107 				break;
108 			} catch (ex) {
109 			}
110 		}
111 		if (!proc) {
112 			throw new AjxException("XSLTemplate", AjxException.UNSUPPORTED, "AjxXslt.createProcessor");
113 		}
114         this._processor = proc;
115 		if(this._processor) {
116 			this._processor.stylesheet = doc;
117 		}
118 	}
119 };
120 
121 AjxXslt._finishedLoading =
122 function() {
123 	var xslt = this._xslt;  // "this" is the document which xsl is being loaded to.
124 	xslt.createProcessor();
125 };
126 
127 AjxXslt.prototype.loadUrl =
128 function(url) {
129 	var doc = this._doc;
130 	
131 	if (AjxEnv.isNav) {
132 		var docImpl = doc.getDoc();
133 		docImpl._xslt = this;  // for callback
134 		docImpl.addEventListener("load", AjxXslt._finishedLoading, false);
135 	}
136 
137 	doc.loadFromUrl(url);
138 
139 	if (AjxEnv.isIE || AjxEnv.isChrome || AjxEnv.isSafari) {
140 		this.createProcessor();
141 	}
142 };

144 AjxXslt.prototype.transformToDom =
145 function(dom) {
146 	var ret;
147 	if (AjxEnv.isIE) {
148 		ret = this.transformIE(dom);
149 	} else if (AjxEnv.isNav || AjxEnv.isChrome || AjxEnv.isSafari) {
150 		return this.transformNav(dom);  // already in dom
151 	} else {
152 		DBG.println(AjxDebug.DBG1, "No XSL transformation due to browser incompatibility.");
153 		return dom;
154 	}
155 	var doc = AjxXmlDoc.createFromXml(ret);
156 	return doc.getDoc();
157 };
158 
159 AjxXslt.prototype.transformToString =
160 function(dom) {
161 	var ret;
162 	if (AjxEnv.isIE) {
163 		return this.transformIE(dom);  // already in str
164 	} else if (AjxEnv.isNav || AjxEnv.isChrome || AjxEnv.isSafari) {
165 		ret = this.transformNav(dom);
166 	} else {
167 		DBG.println(AjxDebug.DBG1, "No XSL transformation due to browser incompatibility.");
168 		return dom.documentElement.innerHTML;
169 	}
170 	
171 	if (!ret || !ret.documentElement) {
172 		throw new AjxException("XSL transformation failed.", AjxException.INVALID_PARAM, "AjxXslt.transformToString");
173 	}
174 	
175 	var elem = ret.documentElement;
176 	if ((elem instanceof HTMLElement) ||
177 		(elem instanceof HTMLHtmlElement)) {
178 		// good.
179 		return elem.innerHTML;
180 	} else if (elem instanceof Element) {
181 		// ok.
182 		return AjxXmlDoc.replaceInvalidChars(elem.xml);
183 	}
184 	DBG.println(AjxDebug.DBG1, "Transformation resulted in non-element.");
185 	return dom.documentElement.innerHTML;
186 };
187 
188 /**
189 * IE returns html text.
190 */
191 AjxXslt.prototype.transformIE =
192 function(dom) {
193 	try {
194 		var xsltProc = this._processor.createProcessor();
195         xsltProc.input = dom;
196         xsltProc.transform();
197 		return xsltProc.output;
198 	} catch (exception) {
199 		DBG.println(AjxDebug.DBG1, "Exception in XSL transformation: "+exception.description);
200 		throw new AjxException(exception.description, AjxException.INVALID_PARAM, "AjxXslt.transformIE");
201 	}
202 };
203 
204 /**
205 * Returns either HTMLDocument or XMLDocument, depending on the transformation.
206 */
207 AjxXslt.prototype.transformNav =
208 function(dom) {
209 	if(!this._processor) {
210 		return "";
211 	}
212 	return this._processor.transformToDocument(dom);
213 };
214 
215 /**
216 * Returns DocumentFragment
217 */
218 AjxXslt.prototype.transformNav2 =
219 function(dom) {
220 	this._fragment = document.implementation.createDocument("", "", null);
221 	if(!this._processor) {
222 		return "";
223 	}
224 	return this._processor.transformToFragment(dom, this._fragment);
225 };
