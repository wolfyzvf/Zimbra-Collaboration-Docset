 27 /**
 28  * Creates the zimlet
 29  * @class
 30  * This class represents a zimlet.
 31  * 
 32  * @param	{String}	id		the id
 33  * @param	{String}	name	the name
 34  * @param	{Object}	parent	the parent
 35  * @param	{ZmTree}	tree	the tree
 36  * @param	{String}	color	the color
 37  * @extends		ZmOrganizer
 38  */
 39 ZmZimlet = function(id, name, parent, tree, color) {
 40 	ZmOrganizer.call(this, {type: ZmOrganizer.ZIMLET, id: id, name: name, parent: parent, tree: tree});
 41 };

 76 // Static methods
 77 /**
 78  * @private
 79  */
 80 ZmZimlet.createFromJs =
 81 function(parent, obj, tree, link) {
 82 	if (!obj && obj.length < 1) {return null;}
 83 
 84 	// create zimlet root
 85 	var zimletRoot = new ZmZimlet(ZmZimlet.ID_ZIMLET_ROOT, ZmMsg.zimlets, parent, tree, null, null);
 86 	if (obj && obj.length) {
 87 		var id = ZmZimlet.ID_ZIMLET;
 88 		for (var i = 0; i < obj.length; i++) {
 89 			var lbl = obj[i].processMessage(obj[i].zimletPanelItem.label);
 90 			// bug fix #23860 - unique-ify zimlet ID's so they dont conflict!
 91 			var zimletId = (++id) + "_z";
 92 			var childZimlet = new ZmZimlet(zimletId, lbl, zimletRoot, tree, null, null);
 93 			zimletRoot.children.add(childZimlet);
 94 			// WARNING: it's a bit unorthodox to do this linkage
 95 			// here, but we really do need these objects know about
 96 			// each other.
 97 			childZimlet._zimletContext = obj[i];
 98 			childZimlet._zimletContext._id = zimletId;
 99 			childZimlet._toolTip = obj[i].zimletPanelItem.toolTipText;
100 			obj[i]._organizer = childZimlet;
101 		}
102 	}
103 	return zimletRoot;
104 };

106 /**
107  * Compares and sorts the zimlets by name (case-insensitive).
108  * 
109  * @param	{ZmZimlet}	zimletA		the zimlet
110  * @param	{ZmZimlet}	zimletB		the zimlet
111  * @return	{int}	0 if the zimlets match; 1 if "a" is before "b"; -1 if "b" is before "a"
112  */
113 ZmZimlet.sortCompare =
114 function(zimletA, zimletB) {
115 	var check = ZmOrganizer.checkSortArgs(zimletA, zimletB);
116 	if (!check) {return check;}
117 
118 	// sort by name
119 	var zimletAName = zimletA.name.toLowerCase();
120 	var zimletBName = zimletB.name.toLowerCase();
121 	if (zimletAName < zimletBName) {return -1;}
122 	if (zimletAName > zimletBName) {return 1;}
123 	return 0;
124 };

138 // Public methods
139 /**
140  * Resets the names.
141  * 
142  */
143 ZmZimlet.prototype.resetNames =
144 function() {
145 	var oldName = this.name;
146 	var oldToolTip = this._toolTip;
147 	if(this._zimletContext && this._toolTip) {
148 		this._toolTip = this._zimletContext.processMessage(this._toolTip);
149 	}
150 	if(this._zimletContext && this.name) {
151 		this.name = this._zimletContext.processMessage(this.name);
152 	}
153 	// Update only if there was a change
154 	if((oldName != this.name) || (oldToolTip != this._toolTip)) {
155 		var fields = {};
156 		fields[ZmOrganizer.F_NAME] = true;
157 		var details = {};
158 		details.fields = fields;
159 		this._notify(ZmEvent.E_MODIFY, details);
160 	}
161 };
