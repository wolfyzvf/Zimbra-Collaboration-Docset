 22 /**
 23  * Creates the filters page, with tabs for incoming and outgoing filters.
 24  * @class
 25  * This class represents the filters page.
 26  * 
 27  * @param {DwtControl}	            parent			the containing widget
 28  * @param {Object}	                section			the page
 29  * @param {ZmFilterController}	    controller		the filter controller
 30  * 
 31  * @extends	ZmPreferencesPage
 32  * 
 33  * @private
 34  */
 35 ZmFilterPage = function(parent, section, controller) {
 36 	ZmPreferencesPage.apply(this, arguments);
 37 };
 38 
 39 ZmFilterPage.prototype = new ZmPreferencesPage;
 40 ZmFilterPage.prototype.constructor = ZmFilterPage;
 41 
 42 ZmFilterPage.prototype.isZmFilterPage = true;
 43 ZmFilterPage.prototype.toString = function () { return "ZmFilterPage"; };
 44 
 45 ZmFilterPage.prototype._createControls =
 46 function() {
 47 	if (appCtxt.get(ZmSetting.PRIORITY_INBOX_ENABLED)) {
 48 		this._activityStreamsButton = new DwtButton({parent:this, parentElement: this._htmlElId+"_ACTIVITY_STREAM_BUTTON" });
 49 		this._activityStreamsButton.setText(ZmMsg.activityStreamSettings);
 50 		this._activityStreamsButton.addSelectionListener(new AjxListener(this, this._activityStreamDialog));
 51 	}
 52 	this._tabView = new DwtTabView({parent:this, posStyle:Dwt.STATIC_STYLE});
 53 	this._tabView.reparentHtmlElement(this._htmlElId+"_tabview");
 54 	var incomingController = this._controller.getIncomingFilterRulesController();
 55 	this._tabView.addTab(ZmMsg.incomingMessageFilters, incomingController.getFilterRulesView());
 56 	var outgoingController = this._controller.getOutgoingFilterRulesController();
 57 	this._tabView.addTab(ZmMsg.outgoingMessageFilters, outgoingController.getFilterRulesView());
 58 	this.setVisible(true);
 59 	
 60 	this.hasRendered = true;
 61 };
 62 
 63 ZmFilterPage.prototype.reset =
 64 function() {
 65 	ZmPreferencesPage.prototype.reset.apply(this, arguments);
 66 	this._controller._stateChangeListener();
 67 };
 68 
 69 ZmFilterPage.prototype.getTabView =
 70 function () {
 71 	return this._tabView;
 72 };
 73 
 74 ZmFilterPage.prototype.hasResetButton =
 75 function() {
 76 	return false;
 77 };
 78 
 79 
 80 //
 81 // Protected methods
 82 //
 83 
 84 ZmFilterPage.prototype._setupCustom = function(id, setup, value) {
 85 	if (id == "FILTER_TABS") {
 86 		return this.getTabView();
 87 	}
 88 	return ZmPreferencesPage.prototype._setupCustom.apply(this, arguments);
 89 };
 90 
 91 ZmFilterPage.prototype._activityStreamDialog = function() {
 92 	var priorityFilterDialog = appCtxt.getPriorityMessageFilterDialog();
 93 	ZmController.showDialog(priorityFilterDialog);
 94 };
