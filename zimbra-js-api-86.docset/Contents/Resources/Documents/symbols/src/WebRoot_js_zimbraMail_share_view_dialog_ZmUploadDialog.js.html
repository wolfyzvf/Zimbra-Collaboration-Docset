 26 /**
 27  * Creates an upload dialog.
 28  * @class
 29  * This class represents an upload dialog.
 30  * 
 31  * @param	{DwtComposite}	shell		the parent
 32  * @param	{String}	className		the class name
 33  *  
 34  * @extends		DwtDialog
 35  */
 36 ZmUploadDialog = function(shell, className) {
 37 	className = className || "ZmUploadDialog";
 38 	var title = ZmMsg.uploadDocs;
 39 	DwtDialog.call(this, {parent:shell, className:className, title:title});
 40 	this.setButtonListener(DwtDialog.OK_BUTTON, new AjxListener(this, this._upload));
 41 	this._createUploadHtml();
 42     this._showLinkTitleText = false;
 43     this._linkText = {};
 44 	this._controller = null;
 45 
 46 	this._inprogress = false;
 47 }

110 ZmUploadDialog.prototype.popup =
111 function(controller, folder, callback, title, loc, oneFileOnly, noResolveAction, showNotes, isImage, conflictAction) {
112 	if (this._inprogress) {
113 		this.warnInProgress();
114 		return;
115 	}
116 
117 	this._controller     = controller;
118 	this._uploadFolder   = folder;
119 	this._uploadCallback = callback;
120 	this._conflictAction = conflictAction;
121 	var aCtxt = ZmAppCtxt.handleWindowOpener();
122 
123     this._supportsHTML5 = AjxEnv.supportsHTML5File && !this._showLinkTitleText && (aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT) != null);
124 
125 	this.setTitle(title || ZmMsg.uploadDocs);
126 
127 	// reset input fields
128 	var table = this._tableEl;
129 	var rows = table.rows;
130 	while (rows.length) {
131 		table.deleteRow(rows.length - 1);
132 	}
133 	this._addFileInputRow(oneFileOnly);
134 
135 	// hide/show elements
136     var id = this._htmlElId;
137     var labelEl = document.getElementById(id+"_label");
138     if (labelEl) {
139         if(oneFileOnly && isImage){
140             labelEl.innerHTML = ZmMsg.uploadChooseImage;
141             Dwt.setVisible(labelEl, true);
142         }
143         else{
144             labelEl.innerHTML = ZmMsg.uploadChoose;
145             Dwt.setVisible(labelEl, !oneFileOnly);
146         }
147     }
148     var actionRowEl = document.getElementById(id+"_actionRow");
149 	if (actionRowEl) {
150 		Dwt.setVisible(actionRowEl, !noResolveAction);
151 	}
152 
153     var notesEl = document.getElementById(id+"_notesTD");
154 	if (notesEl) {
155 		Dwt.setVisible(notesEl, showNotes);
156 	}
157     // In case of a single file upload show proper info message
158 
159     var docSizeInfo = document.getElementById((id+"_info"));
160     var attSize = AjxUtil.formatSize(aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT) || 0, true);
161         if(docSizeInfo){
162             if(oneFileOnly){
163                 docSizeInfo.innerHTML = AjxMessageFormat.format(ZmMsg.attachmentLimitMsgSingleFile, attSize);
164             }
165             else{
166                 docSizeInfo.innerHTML = AjxMessageFormat.format(ZmMsg.attachmentLimitMsg, attSize);
167             }
168         }
169 
170 
171 	// show
172 	DwtDialog.prototype.popup.call(this, loc);
173 };

175 ZmUploadDialog.prototype.popdown =
176 function() {
177 	/***
178 	// NOTE: Do NOT set these values to null! The conflict dialog will
179 	//       call back to this dialog after it's hidden to process the
180 	//       files that should be replaced.
181 	this._uploadFolder = null;
182 	this._uploadCallback = null;
183 	/***/
184 
185     this._extensions = null;
186 
187     //Cleanup
188     this._enableStatus = false;
189 
190     this._notes.removeAttribute("disabled");
191     this.setNotes("");
192     this._msgInfo.innerHTML = "";
193 	this._conflictAction = null;
194     
195 	DwtDialog.prototype.popdown.call(this);
196 };

204 //to give explicitly the uploadForm, files to upload and folderId used for briefcase
205 ZmUploadDialog.prototype.uploadFiles =
206 function(files, uploadForm, folder) {
207 	if (this._inprogress) {
208 		this.warnInProgress();
209 		return;
210 	}
211 
212     if (files.length == 0) {
213 		return;
214 	}
215     this._uploadFolder = folder;
216 
217     var popDownCallback = this.popdown.bind(this);
218     var uploadParams = {
219         uploadFolder: folder,
220         preResolveConflictCallback: popDownCallback,
221         errorCallback: popDownCallback,
222         finalCallback: this._finishUpload.bind(this),
223         docFiles: files
224     }
225 
226 	var aCtxt = appCtxt.isChildWindow ? parentAppCtxt : appCtxt;
227 	var briefcaseApp = aCtxt.getApp(ZmApp.BRIEFCASE);
228     var callback =  briefcaseApp.uploadSaveDocs.bind(briefcaseApp, null, uploadParams);
229 
230     var uploadMgr = appCtxt.getUploadManager();
231 	  window._uploadManager = uploadMgr;
232 
233     try {
234 		uploadMgr.execute(callback, uploadForm);
235 	} catch (ex) {
236 		if (ex.msg) {
237 			this._popupErrorDialog(ex.msg);
238 		} else {
239 			this._popupErrorDialog(ZmMsg.unknownError);
240 		}
241 	}
242 };

244 // Protected methods
245 ZmUploadDialog.prototype._upload = function(){
246     var form         	= this._uploadForm;
247     var uploadFiles  	= [];
248     var errors       	= {};
249     this._linkText   	= {};
250     var aCtxt        	= ZmAppCtxt.handleWindowOpener();
251     var maxSize      	=  aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT);
252     var elements     	= form.elements;
253     var notes           = this.getNotes();
254 	var fileObj 		= [];
255 	var zmUploadManager = appCtxt.getZmUploadManager();
256 	var file;
257     var msgFormat;
258     var errorFilenames;
259 	var newError;
260     for (var i = 0; i < elements.length; i++) {
261         var element = form.elements[i];
262         if ((element.name != ZmUploadDialog.UPLOAD_FIELD_NAME) || !element.value)  continue;
263 
264         this._msgInfo.innerHTML = "";
265 		var errors = [];
266         if(this._supportsHTML5){
267             var files = element.files;
268 			var errors = [];
269             for (var j = 0; j < files.length; j++){
270                 file = files[j];
271                 fileObj.push(file);
272 				newError = zmUploadManager.getErrors(file, maxSize);
273 				if (newError) {
274 					errors.push(newError);
275 				} else {
276                     uploadFiles.push({name: file.name, fullname: file.name, notes: notes});
277                 }
278             }
279         } else {
280 			var fileName = element.value.replace(/^.*[\\\/:]/, "");
281             file = { name: fileName };
282 			newError = zmUploadManager.getErrors(file, maxSize);
283 			if (newError) {
284 				errors.push(newError);
285 			} else {
286                 uploadFiles.push({ fullname: element.value, name: fileName, notes: notes});
287 			}
288         }
289 		if(this._showLinkTitleText) {
290 			var id = element.id;
291 			id = id.replace("_input", "") + "_titleinput";
292 			var txtElement = document.getElementById(id);
293 			if(txtElement) {
294 				this._linkText[file.name] = txtElement.value;
295 			}
296 		}
297 
298     }
299 
300 	if (errors.length > 0) {
301 		this._msgInfo.innerHTML = zmUploadManager.createUploadErrorMsg(errors, maxSize, "<br>");
302 	} else if (uploadFiles.length > 0) {
303 		var briefcaseApp    = aCtxt.getApp(ZmApp.BRIEFCASE);
304 		var shutDownCallback = null;
305 		var uploadButton     = null;
306 		if (this._controller == null) {
307 			shutDownCallback = this.popdown.bind(this);
308 		} else {
309 			var toolbar = this._controller.getCurrentToolbar();
310 			if (toolbar) {
311 				uploadButton = toolbar.getOp(ZmOperation.NEW_FILE);
312 			}
313 			this.popdown();
314 			shutDownCallback = this._enableUpload.bind(this, uploadButton);
315 		}
316         var uploadParams = {
317 			attachment:                 false,
318             files:                      fileObj,
319             notes:                      notes,
320             allResponses:               null,
321             start:                      0,
322             uploadFolder:               this._uploadFolder,
323 			completeAllCallback:        briefcaseApp.uploadSaveDocs.bind(briefcaseApp),
324 			conflictAction:				this._conflictAction || this._selector.getValue(),
325             preResolveConflictCallback: shutDownCallback,
326             errorCallback:              shutDownCallback,
327             completeDocSaveCallback:    this._finishUpload.bind(this, uploadButton),
328             docFiles:                   uploadFiles
329         }
330 		uploadParams.progressCallback = this._uploadFileProgress.bind(this, uploadButton, uploadParams);
331 
332         try {
333             if (this._supportsHTML5) {
334                 zmUploadManager.upload(uploadParams);
335             } else {
336 				var callback  = briefcaseApp.uploadSaveDocs.bind(briefcaseApp, null, uploadParams);
337                 var uploadMgr = appCtxt.getUploadManager();
338                 window._uploadManager = uploadMgr;
339                 uploadMgr.execute(callback, this._uploadForm);
340             }
341 
342 			if (uploadButton) {
343 				// The 16x16 upload image has ImgUpload0 (no fill) .. ImgUpload12 (completely filled) variants to give a
344 				// gross idea of the progress.
345 				ZmToolBar._setButtonStyle(uploadButton, null, ZmMsg.uploading, "Upload0");
346 				this._inprogress = true;
347 			}
348         } catch (ex) {
349 			this._enableUpload(uploadButton);
350             if (ex.msg) {
351                 this._popupErrorDialog(ex.msg);
352             } else {
353                 this._popupErrorDialog(ZmMsg.unknownError);
354             }
355         }
356     }
357 };

359 ZmUploadDialog.prototype._uploadFileProgress =
360 	function(uploadButton, params, progress) {
361 		if (!uploadButton || !params || !progress.lengthComputable || !params.totalSize) return;
362 
363 		// The 16x16 upload image has ImgUpload0 (no fill) .. ImgUpload12 (completely filled) variants to give a
364 		// gross idea of the progress.  A tooltip indicating the progress will be added too.
365 		var progressFraction = (progress.loaded / progress.total);
366 		var uploadedSize = params.uploadedSize + (params.currentFileSize * progressFraction);
367 		var fractionUploaded = uploadedSize/params.totalSize;
368 		if (fractionUploaded > 1) {
369 			fractionUploaded = 1;
370 		}
371 		var progressBucket = Math.round(fractionUploaded * 12);
372 		DBG.println(AjxDebug.DBG3,"Upload Progress: total=" + params.totalSize + ", uploadedSize=" + params.uploadedSize +
373 					", currentSize="+ params.currentFileSize + ", progressFraction=" +
374 					progressFraction + ", fractionUploaded=" + fractionUploaded + ", progressBucket=" + progressBucket);
375 		ZmToolBar._setButtonStyle(uploadButton, null, ZmMsg.uploadNewFile, "Upload" + progressBucket.toString());
376 
377 		var tooltip = AjxMessageFormat.format(ZmMsg.uploadPercentComplete, [ Math.round(fractionUploaded * 100) ] )
378 		uploadButton.setToolTipContent(tooltip, true);
379 	};

381 ZmUploadDialog.prototype._enableUpload = function(uploadButton) {
382 	if (!uploadButton) return;
383 
384 	ZmToolBar._setButtonStyle(uploadButton, null, ZmMsg.uploadNewFile, null);
385 	uploadButton.setToolTipContent(ZmMsg.uploadNewFile, true);
386 	this._inprogress = false;
387 };

389 ZmUploadDialog.prototype._finishUpload = function(uploadButton, docFiles, uploadFolder) {
390     var filenames = [];
391     for (var i in docFiles) {
392         var name = docFiles[i].name;
393         if(this._linkText[name]) {
394             docFiles[i].linkText = this._linkText[name];
395         }
396         filenames.push(name);
397     }
398 	this._enableUpload(uploadButton);
399 
400     this._uploadCallback.run(uploadFolder, filenames, docFiles);
401 };

403 ZmUploadDialog.prototype._addFileInputRow = function(oneInputOnly) {
404 	var id = Dwt.getNextId();
405 	var inputId = id + "_input";
406 	var removeId = id + "_remove";
407 	var addId = id + "_add";
408     var sizeId = id + "_size";
409 
410 	var table = this._tableEl;
411 	var row = table.insertRow(-1);
412 
413     var cellLabel = row.insertCell(-1);
414     cellLabel.innerHTML = ZmMsg.fileLabel;
415 
416 	var cell = row.insertCell(-1);
417 	// bug:53841 allow only one file upload when oneInputOnly is set
418 	cell.innerHTML = [
419 		"<input id='",inputId,"' type='file' name='",ZmUploadDialog.UPLOAD_FIELD_NAME,"' size=30 ",(this._supportsHTML5 ? (oneInputOnly ? "" : "multiple") : ""),">"
420 	].join("");
421 
422 	var cell = row.insertCell(-1);
423     cell.id = sizeId;
424 	cell.innerHTML = " ";
425 
426     //HTML5
427     if(this._supportsHTML5){
428         var inputEl = document.getElementById(inputId);
429         var sizeEl = cell;
430         Dwt.setHandler(inputEl, "onchange", AjxCallback.simpleClosure(this._handleFileSize, this, inputEl, sizeEl));
431     }
432 
433     if(oneInputOnly){
434         cell.colSpan = 3;
435     }else{    
436         var cell = row.insertCell(-1);
437         cell.innerHTML = [
438             "<span ",
439             "id='",removeId,"' ",
440             "onmouseover='this.style.cursor=\"pointer\"' ",
441             "onmouseout='this.style.cursor=\"default\"' ",
442             "style='color:blue;text-decoration:underline;'",
443             ">", ZmMsg.remove, "</span>"
444         ].join("");
445         var removeSpan = document.getElementById(removeId);
446         Dwt.setHandler(removeSpan, DwtEvent.ONCLICK, ZmUploadDialog._removeHandler);
447 
448         var cell = row.insertCell(-1);
449         cell.innerHTML = " ";
450         var cell = row.insertCell(-1);
451         cell.innerHTML = [
452             "<span ",
453             "id='",addId,"' ",
454             "onmouseover='this.style.cursor=\"pointer\"' ",
455             "onmouseout='this.style.cursor=\"default\"' ",
456             "style='color:blue;text-decoration:underline;'",
457             ">", ZmMsg.add, "</span>"
458         ].join("");
459         var addSpan = document.getElementById(addId);
460         Dwt.setHandler(addSpan, DwtEvent.ONCLICK, ZmUploadDialog._addHandler);
461     }
462 
463 
464     if(this._showLinkTitleText) {
465         var txtInputId = id + "_titleinput";
466         var txtRow = table.insertRow(-1);
467         var txtCell = txtRow.insertCell(-1);
468         txtCell.innerHTML = [
469     		ZmMsg.linkTitleOptionalLabel
470     	].join("");
471 
472         txtCell = txtRow.insertCell(-1);
473 	    txtCell.innerHTML = [
474     		"<input id='",txtInputId,"' type='text' name='",ZmUploadDialog.UPLOAD_TITLE_FIELD_NAME,"' size=40>"
475     	].join("");
476         txtCell.colSpan = 3;
477     }
478 };

480 ZmUploadDialog.prototype._handleFileSize =
481 function(inputEl, sizeEl){
482 
483     var files = inputEl.files;
484     if(!files) return;
485 
486     var sizeStr = [], className, totalSize =0;
487     for(var i=0; i<files.length;i++){
488         var file = files[i];
489         var size = file.size || file.fileSize /*Safari*/ || 0;
490 	    var aCtxt = ZmAppCtxt.handleWindowOpener();
491         if(size > aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT))
492             className = "RedC";
493         totalSize += size;
494     }
495 
496     if(sizeEl) {
497         sizeEl.innerHTML = "  ("+AjxUtil.formatSize(totalSize, true)+")";
498         if(className)
499             Dwt.addClass(sizeEl, "RedC");
500         else
501             Dwt.delClass(sizeEl, "RedC");
502     }
503     
504 };

506 ZmUploadDialog._removeHandler = function(event) {
507 	var span = DwtUiEvent.getTarget(event || window.event);
508 	var cell = span.parentNode;
509 	var row = cell.parentNode;
510 
511     var endRow = row;
512 
513     if(span.id) {
514        var id = span.id;
515        id = id.replace("_remove", "") + "_titleinput";
516        var txtInput = document.getElementById(id);
517        if(txtInput) {
518            var txtCell = txtInput.parentNode;
519            var txtRow = txtCell.parentNode;
520            endRow = txtRow;
521        }
522     }
523     
524 	if (row.previousSibling == null && endRow.nextSibling == null) {
525 		var comp = DwtControl.findControl(span);
526 		comp._addFileInputRow();
527 	}
528 
529     if(endRow != row) {
530         endRow.parentNode.removeChild(endRow);
531     }
532 
533 	row.parentNode.removeChild(row);
534 };

536 ZmUploadDialog._addHandler = function(event) {
537 	var span = DwtUiEvent.getTarget(event || window.event);
538 	var comp = DwtControl.findControl(span);
539 	comp._addFileInputRow();
540 };

542 ZmUploadDialog.prototype._createUploadHtml = function() {
543 	var id = this._htmlElId;
544 	var aCtxt = ZmAppCtxt.handleWindowOpener();
545     var uri = aCtxt.get(ZmSetting.CSFE_UPLOAD_URI);
546 
547     var subs = {
548         id: id,
549         uri: uri
550     };
551     this.setContent(AjxTemplate.expand("share.Dialogs#UploadDialog", subs));
552 
553     //variables
554     this._uploadForm = document.getElementById((id+"_form"));
555     this._tableEl = document.getElementById((id + "_table"));
556     this._msgInfo = document.getElementById((id+"_msg"));
557     this._notes = document.getElementById((id+"_notes"));
558 
559     //Conflict Selector
560 	this._selector = new DwtSelect({parent:this});
561 	this._selector.addOption(ZmMsg.uploadActionKeepMine, false, ZmBriefcaseApp.ACTION_KEEP_MINE);
562 	this._selector.addOption(ZmMsg.uploadActionKeepTheirs, false, ZmBriefcaseApp.ACTION_KEEP_THEIRS);
563 	this._selector.addOption(ZmMsg.uploadActionAsk, true, ZmBriefcaseApp.ACTION_ASK);
564 	this._selector.reparentHtmlElement((id+"_conflict"));
565     
566     //Info Section
567     var docSizeInfo = document.getElementById((id+"_info"));
568     if(docSizeInfo){
569         var attSize = AjxUtil.formatSize(aCtxt.get(ZmSetting.DOCUMENT_SIZE_LIMIT) || 0, true)
570         docSizeInfo.innerHTML = AjxMessageFormat.format(ZmMsg.attachmentLimitMsg, attSize);
571     }
572     	
573 };
