 22 /**
 23 * show history of the status window
 24 * @param parent			the element that created this view
 25  * @private
 26 */
 27 ZmQuickReminderDialog = function(parent, reminderController, calController) {
 28 
 29 	// init custom buttons
 30 	var selectId = Dwt.getNextId();
 31 	// call base class
 32 	DwtDialog.call(this, {parent:parent, standardButtons: [DwtDialog.OK_BUTTON]});
 33 
 34 	this._reminderController = reminderController;
 35 	this._calController = calController;
 36 
 37 	this.setContent(this._contentHtml(selectId));
 38 	this.setTitle(ZmMsg.currentMeetings);
 39 };
 40 
 41 ZmQuickReminderDialog.prototype = new DwtDialog;
 42 ZmQuickReminderDialog.prototype.constructor = ZmQuickReminderDialog;
 43 
 44 
 45 ZmQuickReminderDialog.SOON = -AjxDateUtil.MSEC_PER_FIFTEEN_MINUTES;
 46

 48 // Public methods

 50 ZmQuickReminderDialog.prototype.toString = 
 51 function() {
 52 	return "ZmQuickReminderDialog";
 53 };

 55 ZmQuickReminderDialog.prototype.popup =
 56 function() {
 57 	DwtDialog.prototype.popup.call(this);
 58 };

 60 ZmQuickReminderDialog.prototype.initialize =
 61 function(list) {
 62 	this._list = list.clone();
 63 	this._apptData = {};
 64 
 65 	var html = [];
 66 	var idx = 0;
 67 	var size = list.size();
 68 
 69 	html[idx++] = "<table cellpadding=0 cellspacing=0 border=0 width=100%>";
 70 	for (var i = 0; i < size; i++) {
 71 		var appt = list.get(i);
 72 		var uid = appt.getUniqueId(true);
 73 		var data = this._apptData[uid] = {appt:appt};
 74 		idx = this._addAppt(html, idx, appt, data, (i === size - 1));
 75 	}
 76     if(size == 0) {
 77         html[idx++] = '<tr name="rdsep">';
 78 		html[idx++] = '<td colspan=3><div>' + ZmMsg.noMeetingsFound + '</div></td>';
 79 		html[idx++] = '</tr>';
 80     }
 81 	html[idx++] = "</table>";
 82 
 83 	
 84 	if (this._openButtons) {
 85 		for (var id in this._openButtons) {
 86 			this._openButtons[id].dispose();
 87 		}
 88 	}
 89 	this._openButtons = {};
 90 
 91 	var openListener = new AjxListener(this, this._openButtonListener);
 92 	var div = document.getElementById(this._listId);
 93 	div.innerHTML = html.join("");
 94 
 95 	for (var i = 0; i < size; i++) {
 96 		var appt = list.get(i);
 97 		var uid = appt.getUniqueId(true);
 98 		var data = this._apptData[uid];
 99 
100 		// open button
101 		var openBtn = this._openButtons[data.openBtnId] = new DwtLinkButton({id: "openBtn_" + id, parent: this, parentElement: data.openLinkId, noDropDown: true});
102 		openBtn.setText(appt.getName());
103 		openBtn.addSelectionListener(openListener);
104 		openBtn.apptUid = uid;
105 
106 		this._updateDelta(data);
107 	}
108 };
109 
110 ZmQuickReminderDialog.prototype._contentHtml =
111 function(selectId) {
112 	this._listId = Dwt.getNextId();
113 	return ["<div class='ZmQuickReminderDialog' id='", this._listId, "'>"].join("");
114 };
115 
116 ZmQuickReminderDialog.prototype._updateDelta = 
117 function(data) {
118 	var td = document.getElementById(data.deltaId);
119 	if (td) {
120 		var startDelta = this._computeDelta(data.appt);
121 
122 		if (startDelta >= 0) 							td.className = 'ZmReminderOverdue';
123 		else if (startDelta > ZmQuickReminderDialog.SOON)	td.className = 'ZmReminderSoon';
124 		else											td.className = 'ZmReminderFuture';
125 
126 		td.innerHTML = this._formatDeltaString(startDelta);
127 	}
128 };

130 ZmQuickReminderDialog.prototype._addAppt =
131 function(html, idx, appt, data, noSep) {
132 
133 	data.openLinkId = Dwt.getNextId();
134 	data.deltaId = Dwt.getNextId();
135 	data.rowId = Dwt.getNextId();
136 
137 	var calName = (appt.folderId != ZmOrganizer.ID_CALENDAR && this._calController)
138 		? this._calController.getCalendarName(appt.folderId) : null;
139 
140 	var calendar = appCtxt.getById(appt.folderId);
141 
142     var apptLabel = appt.isUpcomingEvent ? " (" + ZmMsg.upcoming + ")" : ""
143 
144 	var params = {
145 		noSep: noSep,
146 		rowId: data.rowId,
147 		calName: calName,
148 		accountName: (appCtxt.multiAccounts && calendar && calendar.getAccount().getDisplayName()),
149 		location: appt.getLocation(),
150 		apptIconHtml: (AjxImg.getImageHtml(appt.otherAttendees ? "ApptMeeting" : "Appointment")),
151 		organizer: appt.otherAtt ? appt.organizer : null,
152 		reminderName: (AjxStringUtil.htmlEncode(appt.name + apptLabel)),
153 		durationText: (AjxStringUtil.trim(this._getDurationText(appt))),
154 		deltaId: data.deltaId,
155 		openLinkId: data.openLinkId
156 	};
157 	html[idx++] = AjxTemplate.expand("calendar.Calendar#ReminderDialogRow", params);
158 	return idx;
159 };

161 ZmQuickReminderDialog.prototype._openButtonListener =
162 function(ev) {
163 
164 	var obj = DwtControl.getTargetControl(ev);
165 	var data = this._apptData[obj.apptUid];
166 	var appt = data ? data.appt : null;
167 	if (appt) {
168 		AjxDispatcher.require(["MailCore", "CalendarCore", "Calendar"]);
169 
170 		var cc = AjxDispatcher.run("GetCalController");
171 
172 		// the give appt object is a ZmCalBaseItem. We need a ZmAppt
173 		var newAppt = new ZmAppt();
174 		for (var i in appt) {
175 			if (!AjxUtil.isFunction(appt[i])) {
176 				newAppt[i] = appt[i];
177 			}
178 		}
179 		var callback = new AjxCallback(cc, cc._showAppointmentDetails, newAppt);
180 		newAppt.getDetails(null, callback, null, null, true);
181         this.popdown();
182 	}
183 };

185 ZmQuickReminderDialog.prototype._getDurationText =
186 function(appt) {
187 	var isMultiDay = appt.isMultiDay();
188 	var start = appt.startDate;
189 	var endTime = appt.getEndTime();
190 	var end = new Date(endTime);
191 
192 	if (appt.isAllDayEvent()) {
193 		end = new Date(endTime - (isMultiDay ? 2 * AjxDateUtil.MSEC_PER_HOUR : 0));
194 		var pattern = isMultiDay ? ZmMsg.apptTimeAllDayMulti : ZmMsg.apptTimeAllDay;
195 		return AjxMessageFormat.format(pattern, [start, end]);
196 	}
197 	var pattern = isMultiDay ? ZmMsg.apptTimeInstanceMulti : ZmMsg.apptTimeInstance;
198 	return AjxMessageFormat.format(pattern, [start, end, ""]);
199 };

201 ZmQuickReminderDialog.prototype._computeDelta =
202 function(appt) {
203 	return ((new Date()).getTime() - appt.getStartTime());
204 };

206 ZmQuickReminderDialog.prototype._formatDeltaString =
207 function(deltaMSec) {
208 	var prefix = deltaMSec < 0 ? "In" : "OverdueBy";
209 	deltaMSec = Math.abs(deltaMSec);
210 
211 	// calculate parts
212 	var years =  Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 365));
213 	if (years != 0)
214 		deltaMSec -= years * AjxDateUtil.MSEC_PER_DAY * 365;
215 	var months = Math.floor(deltaMSec / (AjxDateUtil.MSEC_PER_DAY * 30.42));
216 	if (months > 0)
217 		deltaMSec -= Math.floor(months * AjxDateUtil.MSEC_PER_DAY * 30.42);
218 	var days = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_DAY);
219 	if (days > 0)
220 		deltaMSec -= days * AjxDateUtil.MSEC_PER_DAY;
221 	var hours = Math.floor(deltaMSec / AjxDateUtil.MSEC_PER_HOUR);
222 	if (hours > 0)
223 		deltaMSec -= hours * AjxDateUtil.MSEC_PER_HOUR;
224 	var mins = Math.floor(deltaMSec / 60000);
225 	if (mins > 0)
226 		deltaMSec -= mins * 60000;
227 	var secs = Math.floor(deltaMSec / 1000);
228 	if (secs > 30 && mins < 59) mins++;
229 
230 	var secs = 0;
231 
232 	// determine message
233 	var amount;
234 	if (years > 0) {
235 		amount = "Years";
236 		if (years <= 3 && months > 0) {
237 			amount = "YearsMonths";
238 		}
239 	} else if (months > 0) {
240 		amount = "Months";
241 		if (months <= 3 && days > 0) {
242 			amount = "MonthsDays";
243 		}
244 	} else if (days > 0) {
245 		amount = "Days";
246 		if (days <= 2 && hours > 0) {
247 			amount = "DaysHours";
248 		}
249 	} else if (hours > 0) {
250 		amount = "Hours";
251 		if (hours < 5 && mins > 0) {
252 			amount = "HoursMinutes";
253 		}
254 	} else {
255 		amount = "Minutes";
256 	}
257 
258 	// format message
259 	var key = ["reminder",prefix,amount].join("");
260 	var args = [deltaMSec, years, months, days, hours, mins, secs];
261 	return AjxMessageFormat.format(ZmMsg[key], args);
262 };
