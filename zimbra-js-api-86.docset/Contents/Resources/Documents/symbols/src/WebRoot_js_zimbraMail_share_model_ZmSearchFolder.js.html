 27 /**
 28  * Creates the search folder.
 29  * @class
 30  * This class represents a search folder.
 31  * 
 32  * @param	{Hash}	params		a hash of parameters
 33  * 
 34  * @extends	ZmFolder
 35  */
 36 ZmSearchFolder = function(params) {
 37 	params.type = ZmOrganizer.SEARCH;
 38 	ZmFolder.call(this, params);
 39 	
 40 	if (params.query) {
 41 		var searchParams = {
 42 			query:			params.query,
 43 			types:			params.types,
 44 			checkTypes:		true,
 45 			sortBy:			params.sortBy,
 46 			searchId:		params.id,
 47 			accountName:	(params.account && params.account.name)
 48 		};
 49 		this.search = new ZmSearch(searchParams);
 50 	}
 51 };
 52 
 53 ZmSearchFolder.prototype = new ZmFolder;
 54 ZmSearchFolder.prototype.constructor = ZmSearchFolder;

 67 /**
 68  * Creates a search folder.
 69  * 
 70  * @param	{Hash}	params		a hash of parameters
 71  */
 72 ZmSearchFolder.create = function(params) {
 73 
 74 	params = params || {};
 75 
 76 	var search = params.search,
 77 		jsonObj = { CreateSearchFolderRequest: { _jsns:"urn:zimbraMail" } },
 78 		searchNode = jsonObj.CreateSearchFolderRequest.search = {};
 79 
 80 	searchNode.name = params.name;
 81 	searchNode.query = search.query;
 82 	searchNode.l = params.l;
 83 	if (params.sortBy) {
 84 		searchNode.sortBy = params.sortBy;
 85 	}
 86 
 87 	searchNode.types = ZmSearchFolder._getSearchTypes(search);
 88 
 89 	if (params.rgb) {
 90 		searchNode.rgb = params.rgb;
 91 	}
 92 	else if (params.color) {
 93 		var color = ZmOrganizer.getColorValue(params.color, params.type);
 94 		if (color) {
 95 			searchNode.color = color;
 96 		}
 97 	}
 98 
 99 	var accountName;
100 	if (params.isGlobal) {
101 		searchNode.f = 'g';
102 		accountName = appCtxt.accountList.mainAccount.name;
103 	}
104 
105 	return appCtxt.getAppController().sendRequest({
106 		jsonObj:        jsonObj,
107 		asyncMode:      params.asyncMode !== false,
108 		accountName:    accountName,
109 		callback:       ZmSearchFolder._handleCreate,
110 		errorCallback:  params.errorCallback || ZmOrganizer._handleErrorCreate.bind(null)
111 	});
112 };
113 
114 // converts a vector of types to a string the server can understand
115 ZmSearchFolder._getSearchTypes = function(search) {
116 
117 	var typeStr = "";
118 	if (search && search.types) {
119 		var a = search.types.getArray();
120 		if (a.length) {
121 			var typeStr = [];
122 			for (var i = 0; i < a.length; i++) {
123 				typeStr.push(ZmSearch.TYPE[a[i]]);
124 			}
125 			typeStr = typeStr.join(",");
126 		}
127 	}
128 	return typeStr;
129 };
130 
131 ZmSearchFolder._handleCreate =
132 function(params) {
133 	appCtxt.setStatusMsg(ZmMsg.searchSaved);
134 };

136 /**
137  * Sets the underlying search query.
138  *
139  * @param	{String}	    query		    search query
140  * @param	{AjxCallback}	callback		the callback
141  * @param	{AjxCallback}	errorCallback		the error callback
142  * @param	{ZmBatchCommand}	batchCmd		the batch command
143  */
144 ZmSearchFolder.prototype.setQuery = function(query, callback, errorCallback, batchCmd) {
145 
146 	if (query === this.search.query) {
147 		return;
148 	}
149 
150 	var params = {
151 		callback:       callback,
152 		errorCallback:  errorCallback,
153 		batchCmd:       batchCmd
154 	};
155 
156 	var cmd = "ModifySearchFolderRequest";
157 	var request = {
158 		_jsns: "urn:zimbraMail",
159 		search: {
160 			query:  query,
161 			id:     params.id || this.id,
162 			types:  ZmSearchFolder._getSearchTypes(this.search)
163 		}
164 	};
165 	var jsonObj = {};
166 	jsonObj[cmd] = request;
167 
168 	var respCallback = this._handleResponseOrganizerAction.bind(this, params);
169 	if (params.batchCmd) {
170 		params.batchCmd.addRequestParams(jsonObj, respCallback, params.errorCallback);
171 	}
172 	else {
173 		var accountName;
174 		if (appCtxt.multiAccounts) {
175 			accountName = this.account ? this.account.name : appCtxt.accountList.mainAccount.name;
176 		}
177 		appCtxt.getAppController().sendRequest({
178 			jsonObj:        jsonObj,
179 			asyncMode:      true,
180 			accountName:    accountName,
181 			callback:       respCallback,
182 			errorCallback:  params.errorCallback
183 		});
184 	}
185 };

187 /**
188  * Gets the icon.
189  * 
190  * @return	{String}	the icon

192 ZmSearchFolder.prototype.getIcon = 
193 function() {
194 	return (this.nId == ZmOrganizer.ID_ROOT)

196 		: (this.isOfflineGlobalSearch ? "GlobalSearchFolder" : "SearchFolder");
197 };

205 /**
206  * Returns the organizer with the given ID. Looks in this organizer's tree first.
207  * Since a search folder may have either a regular folder or another search folder
208  * as its parent, we may need to get the parent folder from another type of tree.
209  *
210  * @param {int}	parentId	the ID of the organizer to find
211  * 
212  * @private
213  */
214 ZmSearchFolder.prototype._getNewParent =
215 function(parentId) {
216 	var parent = appCtxt.getById(parentId);
217 	if (parent) {
218 		return parent;
219 	}
220 	
221 	return appCtxt.getById(parentId);
222 };

224 // Handle a change to the underlying search query
225 ZmSearchFolder.prototype.notifyModify =	function(obj) {
226 
227 	if (obj.query && obj.query !== this.search.query && obj.id === this.id) {
228 		this.search.query = obj.query;
229 		var fields = {};
230 		fields[ZmOrganizer.F_QUERY] = true;
231 		this._notify(ZmEvent.E_MODIFY, {
232 			fields: fields
233 		});
234 		obj.query = null;
235 	}
236 	ZmFolder.prototype.notifyModify.apply(this, [obj]);
237 };
