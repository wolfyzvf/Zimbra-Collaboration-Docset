 29 /**
 30  * Creates the account list.
 31  * @class
 32  * This class is used to store and manage a list of accounts for a mailbox.
 33  *
 34  * @author Parag Shah
 35  */
 36 ZmAccountList = function() {
 37 	this._accounts = {};
 38 	this._count = 0;
 39 	this.visibleAccounts = [];
 40 	this.mainAccount = null;
 41 	this.activeAccount = null;
 42 	this.defaultAccount = null; // the first non-main account.
 43 
 44 	this._evtMgr = new AjxEventMgr();
 45 };

 67 /**
 68  * Gets the number of visible accounts for this mailbox.
 69  *
 70  * @param	{Boolean}	includeInvisible	if <code>true</code>, include the number of invisible accounts for this mailbox

 78 /**
 79  * Adds the account.
 80  * 
 81  * @param	{ZmAccount}	account		the account
 82  */
 83 ZmAccountList.prototype.add =
 84 function(account) {
 85 	this._accounts[account.id] = account;
 86 	this._count++;
 87 
 88 	if (account.visible || account.id == ZmAccountList.DEFAULT_ID) {
 89 		this.visibleAccounts.push(account);
 90 	}
 91 
 92 	if (account.id == ZmAccountList.DEFAULT_ID) {
 93 		this.mainAccount = account;
 94 	}
 95 };

150 /**
151  * Gets the cumulative item count of all accounts for the given folder ID.
152  *
153  * @param {String}	folderId		the folder id
154  * @param {Boolean}	checkUnread		if <code>true</code>, checks the unread count instead of item count
155  * @return	{int}	the item count
156  */
157 ZmAccountList.prototype.getItemCount =
158 function(folderId, checkUnread) {
159 	var count = 0;
160 	for (var i = 0; i < this.visibleAccounts.length; i++) {
161 		var acct = this.visibleAccounts[i];
162 		if (acct.isMain) { continue; } // local account should never have drafts
163 
164 		var fid = ZmOrganizer.getSystemId(folderId, acct);
165 		var folder = appCtxt.getById(fid);
166 		if (folder) {
167 			count += (checkUnread ? folder.numUnread : folder.numTotal);
168 		}
169 	}
170 
171 	return count;
172 };

174 /**
175  * Generates a query.
176  * 
177  * @param {String}	folderId		the folder id
178  * @param	{Array}	types		the types
179  * @return	{String}	the query
180  */
181 ZmAccountList.prototype.generateQuery =
182 function(folderId, types) {
183 	// XXX: for now, let's just search for *one* type at a time
184 	var type = types && types.get(0);
185 	var query = [];
186 	var list = this.visibleAccounts;
187 	var fid = folderId || ZmOrganizer.ID_ROOT;
188 	var syntax = folderId ? "inid" : "underid";
189 	for (var i = 0; i < list.length; i++) {
190 		var acct = list[i];
191 
192 		// dont add any apps not supported by this account
193 		if ((type && !acct.isAppEnabled(ZmItem.APP[type])) || acct.isMain) { continue; }
194 
195 		var part = [syntax, ':"', ZmOrganizer.getSystemId(fid, acct, true), '"'];
196 		query.push(part.join(""));
197 	}
198     if(fid == ZmOrganizer.ID_ROOT) {
199         query.push([syntax, ':"', appCtxt.accountList.mainAccount.id, ':', fid, '"'].join(""));
200     }
201 	DBG.println(AjxDebug.DBG2, "query = " + query.join(" OR "));
202 	return (query.join(" OR "));
203 };

205 /**
206  * Loads each visible account serially by requesting the following requests from
207  * the server in a batch request:
208  * 
209  * <ul>
210  * <li><code><GetInfoRequest></code></li>
211  * <li><code><GetTafReqyuest></code></li>
212  * <li><code><GetFolderRequest></code></li>
213  * </ul>
214  * 
215  * @param {AjxCallback}	callback		the callback to trigger once all accounts have been loaded
216  */
217 ZmAccountList.prototype.loadAccounts =
218 function(callback) {
219 	var list = (new Array()).concat(this.visibleAccounts);
220 	this._loadAccount(list, callback);
221 };
222 
223 /**
224  * @private
225  */
226 ZmAccountList.prototype._loadAccount =
227 function(accounts, callback) {
228 	var acct = accounts.shift();
229 	if (acct) {
230 		var respCallback = new AjxCallback(this, this._loadAccount, [accounts, callback]);
231 		acct.load(respCallback);
232 	} else {
233 		// do any post account load initialization
234 		ZmOrganizer.HIDE_EMPTY[ZmOrganizer.TAG] = true;
235 		ZmOrganizer.HIDE_EMPTY[ZmOrganizer.SEARCH] = true;
236 
237 		// enable compose based on whether at least one account supports smtp
238 		for (var i = 0; i < this.visibleAccounts.length; i++) {
239 			if (appCtxt.get(ZmSetting.OFFLINE_SMTP_ENABLED, null, this.visibleAccounts[i])) {
240 				appCtxt.set(ZmSetting.OFFLINE_COMPOSE_ENABLED, true, null, null, true);
241 				break;
242 			}
243 		}
244 
245 		if (callback) {
246 			callback.run();
247 		}
248 	}
249 };

251 /**
252  * Sets the given account as the active one, which will then be used when fetching
253  * any account-specific data such as settings or folder tree.
254  *
255  * @param {ZmZimbraAccount}	account		the account to make active
256  * @param {Boolean}	skipNotify		if <code>true</code>, skip notify
257  */
258 ZmAccountList.prototype.setActiveAccount =
259 function(account, skipNotify) {
260 	this.activeAccount = account;
261 
262 	this._evt = this._evt || new ZmEvent();
263 	this._evt.account = account;
264 
265 	if (!skipNotify) {
266 		this._evtMgr.notifyListeners("ACCOUNT", this._evt);
267 	}
268 };

281 /**
282  * Checks if any of the non-main, visible accounts is currently doing an initial sync.
283  * 
284  * @return	{Boolean}	<code>true</code> if any of the non-main accounts are doing initial sync
285  */
286 ZmAccountList.prototype.isInitialSyncing =
287 function() {
288 	for (var i = 0; i < this.visibleAccounts.length; i++) {
289 		var acct = this.visibleAccounts[i];
290 		if (acct.isMain) { continue; }
291 
292 		if (acct.isOfflineInitialSync()) {
293 			return true;
294 		}
295 	}
296 
297 	return false;
298 };

300 /**
301  * Returns true if any of the visible accounts have the given status
302  *
303  * @param 	{String}		status 		Status to check for
304  */
305 ZmAccountList.prototype.isSyncStatus =
306 function(status) {
307 	for (var i = 0; i < this.visibleAccounts.length; i++) {
308 		var acct = this.visibleAccounts[i];
309 		if (acct.isMain) { continue; }
310 
311 		if (acct.status == status) {
312 			return true;
313 		}
314 	}
315 
316 	return false;
317 };

319 /**
320  * Checks if there is at least one of the given account types in the
321  * account list. Note: if the given account type is ZCS, the local parent
322  * account is NOT included when searching the account list.
323  *
324  * @param {String}	type	the type of account to check
325  * @return	{Boolean}	<code>true</code> if the account exists
326  */
327 ZmAccountList.prototype.accountTypeExists =
328 function(type) {
329 	for (var i = 0; i < this.visibleAccounts.length; i++) {
330 		var acct = this.visibleAccounts[i];
331 		if (type == ZmAccount.TYPE_ZIMBRA && acct.isMain) { continue; }
332 		if (acct.type == type) { return true; }
333 	}
334 
335 	return false;
336 };

339  * Syncs all visible accounts.

348 
349 /**
350  * @private
351  */
352 ZmAccountList.prototype._sendSync =
353 function(accounts, callback) {
354 	var acct = accounts.shift();
355 	if (acct) {
356 		if (!acct.isMain) { // skip the main account
357 			acct.sync();
358 		}
359 		AjxTimedAction.scheduleAction(new AjxTimedAction(this, this._sendSync, [accounts, callback]), 500);
360 	} else {
361 		if (callback) {
362 			callback.run();
363 		}
364 	}
365 };

367 /**
368  * Creates the main account and all its children. In the normal case, the "main"
369  * account is the only account, and represents the user who logged in. If family
370  * mailbox is enabled, that account is a parent account with dominion over child
371  * accounts. If offline, the main account is the "local" account.
372  *
373  * @param {ZmSettings}	settings	the settings for the main account
374  * @param {Object}	obj		the JSON obj containing meta info about the main account and its children
375  */
376 ZmAccountList.prototype.createAccounts =
377 function(settings, obj) {
378 	// first, replace the dummy main account with real information
379 	var account = appCtxt.accountList.mainAccount;
380 	account.id = obj.id;
381 	account.name = obj.name;
382 	account.isMain = true;
383 	account.isZimbraAccount = true;
384 	account.loaded = true;
385 	account.visible = true;
386 	account.settings = settings;
387 	account.type = ZmAccount.TYPE_ZIMBRA;
388 	account.icon = "AccountZimbra";
389 	account.active = true; // always set active for main/parent account
390 
391 	this._accounts[account.id] = account;
392 	delete this._accounts[ZmAccountList.DEFAULT_ID];
393 
394 	this.setActiveAccount(account);
395 
396 	if (appCtxt.isOffline) {
397 		account.displayName = ZmMsg.localFolders;
398 	}
399 
400 	// second, create all child accounts if applicable
401 	var childAccounts = obj.childAccounts && obj.childAccounts.childAccount;
402 	if (childAccounts) {
403 		for (var i = 0; i < childAccounts.length; i++) {
404 			this.add(ZmZimbraAccount.createFromDom(childAccounts[i]));
405 		}
406 
407 		// set global vars per number of child accounts
408 		appCtxt.multiAccounts = this.size() > 1;
409 		appCtxt.isFamilyMbox = appCtxt.multiAccounts && !appCtxt.isOffline;
410 
411 		this.defaultAccount = appCtxt.isFamilyMbox ? this.mainAccount : this.visibleAccounts[1];
412 	}
413 };

442 /**
443  * Gets the tool tip for the folder.
444  * 
445  * @param	{String}	folderId	the folder id
446  * @return	{String}	the tool tip
447  */
448 ZmAccountList.prototype.getTooltipForVirtualFolder =
449 function(folderId) {
450 	var numTotal = 0;
451 	var sizeTotal = 0;
452 
453 	for (var i = 0; i < this.visibleAccounts.length; i++) {
454 		var acct = this.visibleAccounts[i];
455 		var fid = ZmOrganizer.getSystemId(folderId, acct);
456 		var folder = appCtxt.getById(fid);
457 		if (folder) {
458 			numTotal += folder.numTotal;
459 			sizeTotal += folder.sizeTotal;
460 		}
461 	}
462 
463 	var subs = {
464 		itemText: ZmMsg.messages,
465 		numTotal: numTotal,
466 		sizeTotal: sizeTotal
467 	};
468 
469 	return AjxTemplate.expand("share.App#FolderTooltip", subs);
470 };
