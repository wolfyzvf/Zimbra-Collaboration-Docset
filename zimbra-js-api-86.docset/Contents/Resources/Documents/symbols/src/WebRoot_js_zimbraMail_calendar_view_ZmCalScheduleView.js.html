156 ZmCalNewScheduleView.prototype._mouseDownAction =
157 function(ev, div, isDblClick) {
158     DBG.println(AjxDebug.DBG2,  "mouse down action");
159     var target = DwtUiEvent.getTarget(ev),
160         targetId,
161         tmp,
162         index = {},
163         apptDate,
164         duration = 30,
165         folderId = null,
166         isAllDay = false;
167     isDblClick = isDblClick || false;
168     if(target && target.className.indexOf("ZmSchedulerGridDiv") != -1) {
169         targetId = target.id;
170         tmp = targetId.split("__");
171         index.start = tmp[1] - 1;
172         index.end = tmp[1];
173 
174         if(this._scheduleView) {
175             this._scheduleView.setDateBorder(index);
176             this._scheduleView._outlineAppt();
177             if(!this._date) {
178                 this._date = new Date();
179             }
180             apptDate = new Date(this._date);
181             apptDate.setHours(0, index.end*30, 0);
182 
183             this._timeSelectionEvent(apptDate, duration, isDblClick, isAllDay, folderId, ev.shiftKey);
184         }
185     }
186 };

248 ZmCalNewScheduleView.prototype.renderAppts =
249 function(list) {
250     var timeRange = this.getTimeRange();
251 	if (list) {
252 		var size = list.size();
253 		DBG.println(AjxDebug.DBG2,"list.size:"+size);
254 		if (size != 0) {
255             var showDeclined = appCtxt.get(ZmSetting.CAL_SHOW_DECLINED_MEETINGS);
256             this._computeApptLayout();
257 			for (var i=0; i < size; i++) {
258 				var ao = list.get(i);
259 				if (ao && ao.isInRange(timeRange.start, timeRange.end) &&
260 				    (showDeclined || (ao.ptst != ZmCalBaseItem.PSTATUS_DECLINED))) {
261                     this.addAppt(ao);
262 				}
263 			}
264 		}
265 	}
266 };

338 ZmCalNewScheduleView.prototype.processMetadataAttendees =
339 function(metadataResponse) {
340     var objAttendees = metadataResponse.getResponse().BatchResponse.GetMailboxMetadataResponse[0].meta[0]._attrs,
341         emails = [],
342         email,
343         acct,
344         i;
345 
346     for (email in objAttendees) {
347         if(email && objAttendees[email]) {
348             emails.push(objAttendees[email]);
349         }
350     }
351     this._mdAttendees = AjxVector.fromArray(emails);
352     for (i=0; i<this._mdAttendees.size(); i++) {
353         acct = ZmApptViewHelper.getAttendeeFromItem(this._mdAttendees.get(i), ZmCalBaseItem.PERSON);
354         this._attendees[ZmCalBaseItem.PERSON].add(acct, null, true);
355     }
356 
357     AjxDispatcher.require(["MailCore", "CalendarCore", "Calendar", "CalendarAppt"]);
358     this._scheduleView = new ZmFreeBusySchedulerView(this, this._attendees, this._controller, this.getDateInfo());
359     this._scheduleView.setComposeMode(false);
360     this._scheduleView.setVisible(true);
361     this._scheduleView.showMe();
362     this._scheduleView.reparentHtmlElement(this._apptBodyDivId);
363 
364     //Called to handle the sync issue
365     this.resetListItems(this._calNotRenderedList);
366     this.renderAppts(this._calNotRenderedList);
367     delete this._calNotRenderedList;
368 };

370 ZmCalNewScheduleView.prototype.setMetadataAttendees =
371 function(organizer, email) {
372     if(!organizer) {
373         organizer = this.getOrganizer();
374     }
375     if (email instanceof Array) {
376         email = email.join(',');
377     }
378     var md = new ZmMetaData(organizer.getAccount());
379     this._mdAttendees.add(email, null, true);
380     return md.set(ZmCalNewScheduleView.ATTENDEES_METADATA, this._mdAttendees.getArray());
381 };

396 ZmCalNewScheduleView.prototype._resetCalendarData =
397 function() {
398     var i,
399         tb,
400         acct,
401         acctEmail,
402         strAttendees,
403         mdAttendees,
404         organizer = this.getOrganizer();
405 	this._calendars = this._controller.getCheckedCalendars();
406 	this._calendars.sort(ZmFolder.sortCompareNonMail);
407 	this._folderIdToColIndex = {};
408 	this._columns = [];
409 	this._numCalendars = this._calendars.length;
410     this._attendees[ZmCalBaseItem.PERSON] = new AjxVector();
411 
412     for (i=0; i<this._numCalendars; i++) {
413         acctEmail = this._calendars[i].getOwner();
414         if(organizer.getEmail() != acctEmail) {
415             //if not organizer add to the attendee list
416             acct = ZmApptViewHelper.getAttendeeFromItem(acctEmail, ZmCalBaseItem.PERSON);
417             this._attendees[ZmCalBaseItem.PERSON].add(acct, null, true);
418         }
419     }
420 
421     if(this._scheduleView) {
422         //this._attendees[ZmCalBaseItem.PERSON] = this._scheduleView.getAttendees();
423 
424         for (i=0; i<this._mdAttendees.size(); i++) {
425             acct = ZmApptViewHelper.getAttendeeFromItem(this._mdAttendees.get(i), ZmCalBaseItem.PERSON);
426             this._attendees[ZmCalBaseItem.PERSON].add(acct, null, true);
427         }
428 
429         this._scheduleView.cleanup();
430         this._scheduleView.setComposeMode(false);
431         this._scheduleView.set(this.getDate(), organizer, this._attendees);
432         this._scheduleView.enablePartcipantStatusColumn(true);
433     }
434     else {
435         this._mdAttendees = new AjxVector();
436         this.getMetadataAttendees(organizer);        
437     }
