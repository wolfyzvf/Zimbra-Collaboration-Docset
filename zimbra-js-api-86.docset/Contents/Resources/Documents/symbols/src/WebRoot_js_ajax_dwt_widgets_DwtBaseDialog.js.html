 28 /**
 29  * @class
 30  * This is a base class for dialogs. Given content, this class will take care of 
 31  * showing and hiding the dialog, as well as dragging it.
 32  * <p>
 33  * Dialogs always hang off the main shell since their stacking order is managed through z-index.
 34  *
 35  * @author Ross Dargahi
 36  * @author Conrad Damon
 37  * 
 38  * @param {hash}	params		a hash of parameters
 39  * @param	{DwtComposite}	params.parent	the parent widget (the shell)
 40  * @param	{string}	params.className		the CSS class
 41  * @param	{string}	params.title		the title
 42  * @param	{number}	[params.zIndex=Dwt.Z_DIALOG]		the z-index to set for this dialog when it is visible
 43  * @param	{DwtBaseDialog.MODAL|DwtBaseDialog.MODELESS}	[params.mode=DwtBaseDialog.MODAL] 		the modality of the dialog
 44  * @param	{DwtPoint}	params.loc			the location at which to popup the dialog. Defaults to being centered within its parent
 45  * @param	{boolean}		params.disposeOnPopDown		    destroy the content of dialog on popdown, Defaults to false
 46  * @param	{DwtControl}	params.view 		the the control whose element is to be re-parented
 47  * @param	{string}	params.dragHandleId 		the the ID of element used as drag handle
 48  * 
 49  * @extends	DwtComposite
 50  */
 51 DwtBaseDialog = function(params) {
 52 	if (arguments.length == 0) { return; }
 53 
 54 	params = Dwt.getParams(arguments, DwtBaseDialog.PARAMS);
 55 	var parent = params.parent;
 56 	if (!(parent instanceof DwtShell)) {
 57 		throw new DwtException("DwtBaseDialog parent must be a DwtShell", DwtException.INVALIDPARENT, "DwtDialog");
 58 	}
 59 	params.className = params.className || "DwtBaseDialog";
 60 	params.posStyle = DwtControl.ABSOLUTE_STYLE;
 61 	this._title = params.title || "";
 62 
 63 	DwtComposite.call(this, params);
 64     this._disposeOnPopDown = params.disposeOnPopDown || false;
 65 	this._shell = parent;
 66 	this._zIndex = params.zIndex || Dwt.Z_DIALOG;
 67 	this._mode = params.mode || DwtBaseDialog.MODAL;
 68 	
 69 	this._loc = new DwtPoint();
 70 	if (params.loc) {
 71 		this._loc.x = params.loc.x;
 72 		this._loc.y = params.loc.y
 73 	} else {
 74 		this._loc.x = this._loc.y = Dwt.LOC_NOWHERE;
 75 	}
 76 
 77 	// Default dialog tab group.
 78 	this._tabGroup = new DwtTabGroup(this.toString());
 79 
 80     this._dragHandleId = params.dragHandleId || this._htmlElId + "_handle";
 81 	this._createHtml();
 82     this._initializeDragging(this._dragHandleId);
 83 
 84 	if (params.view) {
 85 		this.setView(params.view);
 86     }
 87 
 88 	// reset tab index
 89     this.setZIndex(Dwt.Z_HIDDEN); // not displayed until popup() called
 90 	this._position(DwtBaseDialog.__nowhereLoc);
 91 
 92 	// Make sure mouse clicks propagate to the DwtDraggable handler (document.onMouseMove and onMouseUp)
 93 	this._propagateEvent[DwtEvent.ONMOUSEUP] = true;
 94 
 95 }

201 /**
202  * Pops-up the dialog, makes the dialog visible in places. Everything under the dialog will
203  * become veiled if we are modal. Note: popping up a dialog will block
204  * keyboard actions from being delivered to the global key action handler (if one
205  * is registered).
206  *
207  * @param {DwtPoint}		loc		the desired location
208  */
209 DwtBaseDialog.prototype.popup =
210 function(loc) {
211 	if (this._poppedUp) { return; }
212 
213 	this.cleanup(true);
214 	var thisZ = this._zIndex;
215 
216 	// if we're modal, setup the veil effect, and track which dialogs are open
217 	if (this._mode == DwtBaseDialog.MODAL) {
218 		thisZ = this._setModalEffect(thisZ);
219 	}
220 
221 	this._shell._veilOverlay.activeDialogs.push(this);
222 	
223 	// use whichever has a value, local has precedence
224 	if (loc) {
225 		this._loc.x = loc.x;
226 		this._loc.y = loc.y;
227 	}
228 	this._position(loc);
229 
230     //reset TAB focus before popup of dialog.
231     //method be over-written to focus a different member.
232     this._resetTabFocus();
233 
234 	this.setZIndex(thisZ);
235 	this._poppedUp = true;
236 
237 	// Push our tab group
238 	var kbMgr = this._shell.getKeyboardMgr();
239 	kbMgr.pushTabGroup(this._tabGroup);
240 	kbMgr.pushDefaultHandler(this);
241 
242 	this.notifyListeners(DwtEvent.POPUP, this);
243 };

253 DwtBaseDialog.prototype.focus = 
254 function () {
255 	// if someone is listening for the focus to happen, give control to them,
256 	// otherwise focus on this dialog.
257 	if (this.isListenerRegistered(DwtEvent.ONFOCUS)) {
258 		this.notifyListeners(DwtEvent.ONFOCUS);
259 	} else if (this._focusElementId){
260 		var focEl = document.getElementById(this._focusElementId);
261 		if (focEl) {
262 			focEl.focus();
263 		}
264 	}
265 };

277 /**
278  * Pops-down and hides the dialog.
279  * 
280  */
281 DwtBaseDialog.prototype.popdown =
282 function() {
283 
284 	if (this._poppedUp) {
285 		this._poppedUp = false;
286 		this.cleanup(false);
287 	
288 		//var myZIndex = this.getZIndex();
289 	    var myZIndex = this._zIndex;
290 		this.setZIndex(Dwt.Z_HIDDEN);
291 		//TODO we should not create an object everytime we popdown a dialog (ditto w/popup)
292 		this._position(DwtBaseDialog.__nowhereLoc);
293 		if (this._mode == DwtBaseDialog.MODAL) {
294 			this._undoModality(myZIndex);
295 		} else {
296 			this._shell._veilOverlay.activeDialogs.pop();
297 		}
298 		//this.removeKeyListeners();
299 
300         //Dispose the dialog if _disposeOnPopDown is set to true
301 		if (this._disposeOnPopDown === true) {
302             this.dispose();
303         }
304 
305 		// Pop our tab group
306 		var kbMgr = this._shell.getKeyboardMgr();
307 		kbMgr.popTabGroup(this._tabGroup);
308 		kbMgr.popDefaultHandler();
309 		
310 		this.notifyListeners(DwtEvent.POPDOWN, this);
311 	}
312 };

314 /**
315  * Sets the content of the dialog to a new view. Essentially re-parents
316  * the supplied control's HTML element to the dialogs HTML element
317  * 
318  * @param {DwtControl} newView		the control whose element is to be re-parented
319  */
320 DwtBaseDialog.prototype.setView =
321 function(newView) {
322 	this.reset();
323 	if (newView) {
324 		this._getContentDiv().appendChild(newView.getHtmlElement());
325 	}
326 };

338 /**
339 * Cleans up the dialog so it can be used again later.
340 * 
341 * @param	{boolean}		bPoppedUp		if <code>true</code>, the dialog is popped-up; <code>false</code> otherwise
342 */
343 DwtBaseDialog.prototype.cleanup =
344 function(bPoppedUp) {
345 	//TODO handle different types of input fields e.g. checkboxes etc
346 	var inputFields = this._getInputFields();
347 	
348 	if (inputFields) {
349 		var len = inputFields.length;
350 		for (var i = 0; i < len; i++) {
351 			inputFields[i].disabled = !bPoppedUp;
352 			if (bPoppedUp)
353 				inputFields[i].value = "";
354 		}
355 	}
356 }

403 /**
404  * Gets the active dialog.
405  * 
406  * @return	{DwtBaseDialog}		the active dialog
407  */
408 DwtBaseDialog.getActiveDialog = 
409 function() {
410 	var dialog = null;
411 	var shellObj = DwtShell.getShell(window);
412 	if (shellObj) {
413 		var len = shellObj._veilOverlay.activeDialogs.length;
414 		if (len > 0) {
415 			dialog = shellObj._veilOverlay.activeDialogs[len - 1];
416 		}
417 	}
418 	return dialog;
419 };

425 /**
426  * @private
427  */
428 DwtBaseDialog.prototype._initializeDragging =
429 function(dragHandleId) {
430 	var dragHandle = document.getElementById(dragHandleId);
431 	if (dragHandle) {
432 		var control = DwtControl.fromElementId(window._dwtShellId);
433 		if (control) {
434 			var p = Dwt.getSize(control.getHtmlElement());
435 			var dragObj = document.getElementById(this._htmlElId);
436 			var size = this.getSize();
437 			var dragEndCb = new AjxCallback(this, this._dragEnd);
438 			var dragCb = new AjxCallback(this, this._duringDrag);
439 			var dragStartCb = new AjxCallback(this, this._dragStart);
440 
441 			DwtDraggable.init(dragHandle, dragObj, 0,
442 							  document.body.offsetWidth - 10, 0, document.body.offsetHeight - 10, dragStartCb, dragCb, dragEndCb);
443 		}
444 	}
445 };

463 /**
464  * @private
465  */
466 DwtBaseDialog.prototype._createHtmlFromTemplate = function(templateId, data) {
467     // set default params
468     data.dragId = this._dragHandleId;
469     data.title = this._title;
470     data.icon = "";
471     data.closeIcon1 = "";
472     data.closeIcon2 = "";
473     data.controlsTemplateId = this.CONTROLS_TEMPLATE;
474 
475     // expand template
476     DwtComposite.prototype._createHtmlFromTemplate.call(this, templateId, data);
477 
478     // remember elements
479     this._titleBarEl = document.getElementById(data.id+"_titlebar");
480     this._titleEl = document.getElementById(data.id+"_title");
481     this._contentEl = document.getElementById(data.id+"_content");
482 
483     this.setAttribute('aria-describedby', this._contentEl.id);
484 	this._contentEl.setAttribute('role', 'document');
485 	if (this._titleEl) {
486 		this.setAttribute('aria-labelledby', this._titleEl.id);
487 		this._titleEl.setAttribute('role', 'heading');
488 		this._titleEl.setAttribute('aria-level', '2');
489 	}
490 
491     // NOTE: This is for backwards compatibility. There are just
492     //       too many sub-classes of dialog that expect to return
493     //       the dialog contents via the _getContentHtml method.
494     this.setContent(this._getContentHtml());
495 };

497 /**
498  * @private
499  */
500 DwtBaseDialog.prototype._setModalEffect =
501 function() {
502 	// place veil under this dialog
503 	var dialogZ = this._shell._veilOverlay.dialogZ;
504 	var currentDialogZ = null;
505 	var thisZ, veilZ;
506 	if (dialogZ.length)
507 		currentDialogZ = dialogZ[dialogZ.length - 1];
508 	if (currentDialogZ) {
509 		thisZ = currentDialogZ + 2;
510 		veilZ = currentDialogZ + 1;
511 	} else {
512 		thisZ = this._zIndex;
513 		veilZ = Dwt.Z_VEIL;
514 	}
515 	this._shell._veilOverlay.veilZ.push(veilZ);
516 	this._shell._veilOverlay.dialogZ.push(thisZ);
517 	Dwt.setZIndex(this._shell._veilOverlay, veilZ);
518 	return thisZ;
519 };
520 
521 /**
522  * @private
523  */
524 DwtBaseDialog.prototype._undoModality =
525 function (myZIndex) {
526 	var veilZ = this._shell._veilOverlay.veilZ;
527 	veilZ.pop();
528 	var newVeilZ = veilZ[veilZ.length - 1];
529 	Dwt.setZIndex(this._shell._veilOverlay, newVeilZ);
530 	this._shell._veilOverlay.dialogZ.pop();
531 	this._shell._veilOverlay.activeDialogs.pop();
532 	if (this._shell._veilOverlay.activeDialogs.length > 0 ) {
533 		this._shell._veilOverlay.activeDialogs[0].focus();
534 	}
535 };
