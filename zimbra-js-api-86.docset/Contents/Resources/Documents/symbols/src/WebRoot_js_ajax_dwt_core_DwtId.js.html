 22 /**
 23  * @class
 24  * This class is responsible for providing unique, predictable IDs for HTML elements.
 25  * That way, code outside the client can locate particular elements.
 26  * <p>
 27  * Not every element that has an associated JS object will have a known ID. Those are
 28  * allocated only for elements it would be useful to locate: major components of the UI,
 29  * toolbars, buttons, views, menus, some menu items, and some selects.
 30  * <p>
 31  * There is a simple naming scheme for the IDs themselves. Each ID starts with a "z" followed
 32  * by one to a few letters that indicate the type of object (widget) represented by the element.
 33  * 
 34  * @author Conrad Damon
 35  */
 36  
 37 DwtId = function() {}
 38 
 39 // separator for parts used in constructing IDs - need to pick one that
 40 // doesn't show up in any of the parts
 41 DwtId.SEP = "__";

103 /**
104  * Joins the given arguments into an ID, excluding empty ones.
105  * 
106  * @private
107  */
108 DwtId.makeId =
109 function() {
110 	var list = [];
111 	for (var i = 0; i < arguments.length; i++) {
112 		var arg = arguments[i];
113 		if (arg != null && arg != "") {
114 			list.push(arg);
115 		}
116 	}
117 	return list.join(DwtId.SEP);
118 };
119 DwtId._makeId = DwtId.makeId;	// back-compatibility
120 
121 /**
122  * Gets an ID for a list view.
123  * 
124  * @param {constant}	context		the owning view identifier
125  * @param {DwtId.LIST_VIEW_HEADERS|DwtId.LIST_VIEW_ROWS}	modifier	indicates element within list view (see <code>DwtId.LIST_VIEW*</code> constants)	
126  * @return	{string}	the ID
127  */
128 DwtId.getListViewId =
129 function(context, modifier) {
130 	return DwtId.makeId(DwtId.WIDGET_LIST_VIEW, context, modifier);
131 };
132 
133 /**
134  * Gets an ID for an element within a list view header.
135  * 
136  * @param {constant}	type		the type of hdr element (see <code>DwtId.WIDGET_HDR*</code> constants)
137  * @param {constant}	context	the the ID of owning view
138  * @param {constant}	hdr		the header ID
139  * @return	{string}	the ID
140  */
141 DwtId.getListViewHdrId =
142 function(type, context, hdr) {
143 	return DwtId.makeId(type, context, hdr);
144 };
145 
146 /**
147  * Gets an ID for an element associated with the display of an item in a list view.
148  * 
149  * @param {constant}	type		the type of item element (see <code>DwtId.WIDGET_ITEM*</code> constants)
150  * @param {constant}	context		the ID of owning view
151  * @param {string}	itemId	the item ID (typically numeric)
152  * @param {constant}	field		the field identifier (for example, "su" for subject)
153  * @return	{string}	the ID
154  */
155 DwtId.getListViewItemId =
156 function(type, context, itemId, field) {
157 	return DwtId.makeId(type, context, itemId, field);
158 };
159 
160 /**
161  * Gets an ID for an IFRAME.
162  * 
163  * @param {constant}	context	the ID of owning {@link DwtIframe}
164  * @return	{string}	the ID
165  */
166 DwtId.getIframeId =
167 function(context) {
168 	return DwtId.makeId(context, DwtId.IFRAME);
169 };
