 26 /**
 27  * Creates an overview container for a multi-account mailbox.
 28  * @class
 29  * Creates a header tree item for an account if mailbox has multiple accounts
 30  * configured. For each account header, a {@link ZmOverview} is added a child. If mailbox
 31  * only has one account configured, no account header is created and the
 32  * {@link ZmOverview} is added as a child of the container.
 33  *
 34  * @param	{Hash}	params		a hash of parameters
 35  * 
 36  * @author Parag Shah
 37  */
 38 ZmAccountOverviewContainer = function(params) {
 39 	if (arguments.length == 0) { return; }
 40 
 41 	ZmOverviewContainer.call(this, params);
 42 
 43 	this._vFolderTreeItemMap = {};
 44 	this._settingChangeListener = new AjxListener(this, this._handleSettingChange);
 45 
 46 	var mouseListener = new AjxListener(this, this._mouseListener);
 47 	this.addListener(DwtEvent.ONMOUSEDOWN, mouseListener);
 48 	this.addListener(DwtEvent.ONMOUSEUP, mouseListener);
 49 };

 83 /**
 84  * Expands the given account only (collapses all other accounts).
 85  *
 86  * @param {ZmAccount}	account		the account to expand
 87  */
 88 ZmAccountOverviewContainer.prototype.expandAccountOnly =
 89 function(account) {
 90 	if (!account) {
 91 		account = appCtxt.getActiveAccount();
 92 	}
 93 
 94 	for (var i in this._headerItems) {
 95 		this._headerItems[i].setExpanded((i == account.id), false, false);
 96 	}

 99 ZmAccountOverviewContainer.prototype.getSelected =
100 function() {
101 	var selected = ZmOverviewContainer.prototype.getSelected.call(this);
102 
103 	if (!selected) {
104 		selected = this.getSelection()[0];
105 		var account = selected && appCtxt.accountList.getAccount(selected.getData(Dwt.KEY_ID));
106 		var tree = account && appCtxt.getFolderTree(account);
107 		return tree && tree.root;
108 	}
109 
110 	return selected;
111 };

113 /**
114  * Initializes the account overview.
115  * 
116  * @param	{Hash}	params		a hash of parameters
117  * 
118  */
119 ZmAccountOverviewContainer.prototype.initialize =
120 function(params) {
121 
122 	var header, acct;
123 	var accounts = appCtxt.accountList.visibleAccounts;
124 	var showAllMailboxes = (appCtxt.isOffline && this._appName == ZmApp.MAIL && (accounts.length > 2));
125 	var showBackgroundColor = showAllMailboxes;
126 	var mainAcct = appCtxt.accountList.mainAccount;
127 	var origOmit = params.omit;
128 
129 	for (var i = 0; i < accounts.length; i++) {
130 		acct = accounts[i];
131 		// skip the main account in offline mode since we'll add it at the end
132 		if (appCtxt.isOffline && acct.isMain && this._appName != ZmApp.PREFERENCES) { continue; }
133 		if (!acct.active) { continue; }
134 
135 		 params.omit = {};
136 
137 		if (acct.type == ZmAccount.TYPE_POP) {
138 			params.omit[ZmFolder.ID_SPAM]   = true;
139 			params.omit[ZmFolder.ID_OUTBOX] = true;
140 		}
141 
142         if(params.overviewId && !params.isAppOverview) {
143             var overviewId = params.overviewId.split(":")[1];
144             if(overviewId && (overviewId == "ZmNewOrganizerDialog")){
145                 params.omit[ZmFolder.ID_DRAFTS] = true;
146             }
147         }
148 
149 		this._addAccount(params, acct, showBackgroundColor, null, "account" + i);
150 
151 		header = this.getHeaderItem(acct);
152 		if (header) {
153 			this._setupHeader(header, acct);
154 		}
155 
156 		this.updateAccountInfo(acct, true, true);
157 
158 		showBackgroundColor = !showBackgroundColor;
159 	}
160 
161 	// add "All Mailboxes"
162 	skip = origOmit && origOmit[ZmOrganizer.ID_ALL_MAILBOXES];
163 	if (showAllMailboxes && !skip) {
164 		var text = ZmMsg[ZmFolder.MSG_KEY[ZmOrganizer.ID_ALL_MAILBOXES]];
165 		var hdrText = appCtxt.get(ZmSetting.OFFLINE_ALL_MAILBOXES_TREE_OPEN)
166 			? text : this._getFolderLabel(ZmOrganizer.ID_INBOX, text);
167 		var params1 = {
168 			parent: this,
169 			text: hdrText,
170 			imageInfo: "AccountAll"
171 		};
172 		var showAllMboxes = appCtxt.get(ZmSetting.OFFLINE_SHOW_ALL_MAILBOXES);
173 		var allTi = this._allMailboxesTreeHeader = new DwtTreeItem(params1);
174 		allTi.setData(Dwt.KEY_ID, ZmOrganizer.ID_ALL_MAILBOXES);
175 		allTi.addClassName("ZmOverviewGlobalInbox");
176 		allTi._initialize(0, true);
177 		allTi.setVisible(showAllMboxes);
178 		allTi.__origText = text;
179 		if (showAllMboxes) {
180 			this.highlightAllMboxes();
181 		}
182 
183 		var setting = appCtxt.getSettings(mainAcct).getSetting(ZmSetting.OFFLINE_SHOW_ALL_MAILBOXES);
184 		setting.addChangeListener(this._settingChangeListener);
185 
186 		var folders = ZmAccountOverviewContainer.VIRTUAL_FOLDERS;
187 		for (var i = 0; i < folders.length; i++) {
188 			var folderId = folders[i];
189 
190 			// add virtual system folders
191 			params1 = {
192 				parent: allTi,
193 				text: this._getFolderLabel(folderId, ZmMsg[ZmFolder.MSG_KEY[folderId]]),
194 				imageInfo: ZmFolder.ICON[folderId]
195 			};
196 			var ti = this._vFolderTreeItemMap[folderId] = new DwtTreeItem(params1);
197 			ti.setData(Dwt.KEY_ID, folderId);
198 			ti.addClassName("DwtTreeItemChildDiv");
199 			ti._initialize(null, true);
200 			ti.setToolTipContent(appCtxt.accountList.getTooltipForVirtualFolder(folderId));
201 		}
202 
203 		// add global searches
204 		params1 = {
205 			parent: allTi,
206 			text: ZmMsg.globalSearches,
207 			imageInfo: "SearchFolder"
208 		};
209 		var searchTi = this._searchTreeHeader = new DwtTreeItem(params1);
210 		searchTi.addClassName("DwtTreeItemChildDiv");
211 		searchTi._initialize(null, true);
212 		searchTi.__isSearch = true;
213 
214 		var root = appCtxt.getById(ZmOrganizer.ID_ROOT, mainAcct);
215 		var searchFolders = root.getByType(ZmOrganizer.SEARCH);
216 		for (var i = 0; i < searchFolders.length; i++) {
217 			var folder = searchFolders[i];
218 			if (folder.id != ZmOrganizer.ID_ALL_MAILBOXES &&
219 				folder.isOfflineGlobalSearch)
220 			{
221 				this.addSearchFolder(folder);
222 			}
223 		}
224 		searchTi.setVisible(searchTi.getItemCount() > 0);
225 
226 		// bug #43734 - set expand/collapse state based on user's pref
227 		if (appCtxt.get(ZmSetting.OFFLINE_SAVED_SEARCHES_TREE_OPEN)) {
228 			searchTi.setExpanded(true, null, true);
229 		}
230 		if (appCtxt.get(ZmSetting.OFFLINE_ALL_MAILBOXES_TREE_OPEN)) {
231 			allTi.setExpanded(true, null, true);
232 		}
233 	}
234 
235 	// add the "local" account last
236 	if (appCtxt.isOffline) {
237 		var params2 = AjxUtil.hashCopy(params);
238 		params2.omit = {};
239 
240 		if (this._appName != ZmApp.PREFERENCES) {
241 			this._addAccount(params2, mainAcct, showBackgroundColor, "ZmOverviewLocalHeader", "LocalFolders");
242 
243 			header = this.getHeaderItem(mainAcct);
244 			header.setExpanded(appCtxt.get(ZmSetting.ACCOUNT_TREE_OPEN, null, mainAcct));
245 
246 			this.updateAccountInfo(mainAcct, false, true);
247 		}
248 		else {
249 			var params3 = {
250 				parent: this,
251 				text: mainAcct.getDisplayName(),
252 				imageInfo: mainAcct.getIcon()
253 			};
254 			var localPrefTi = new DwtTreeItem(params3);
255 			localPrefTi._initialize(null, true);
256 
257 			var globalPrefOverviewId = appCtxt.getOverviewId(this.containerId, mainAcct);
258 			var tv = this._overview[globalPrefOverviewId].getTreeView(ZmOrganizer.PREF_PAGE);
259 			var importExportTi = tv.getTreeItemById("PREF_PAGE_IMPORT_EXPORT");
260 
261 			tv.getHeaderItem().removeChild(importExportTi);
262 			localPrefTi._addItem(importExportTi);
263 			importExportTi.addClassName("DwtTreeItemChildDiv");
264 			localPrefTi.setExpanded(true, null, true);
265 		}
266 	}
267 
268 	// add zimlets at the end of all overviews
269 	var skip = params.omit && params.omit[ZmOrganizer.ID_ZIMLET];
270 
271 	if (!appCtxt.inStartup && !skip && appCtxt.getZimletMgr().getPanelZimlets().length == 0) {
272 		skip = true;
273 	}
274 
275 	if (!skip) {
276 		AjxDispatcher.require("Zimlet");
277 	}
278 
279 	if (!skip && window[ZmOverviewController.CONTROLLER[ZmOrganizer.ZIMLET]] &&
280 		this._appName != ZmApp.PREFERENCES)
281 	{
282 		var headerLabel = ZmOrganizer.LABEL[ZmOrganizer.ZIMLET];
283 		var headerDataId = params.overviewId = appCtxt.getOverviewId([this.containerId, headerLabel], null);
284 		var headerParams = {
285 			label: ZmMsg[headerLabel],
286 			icon: "Zimlet",
287 			dataId: headerDataId,
288 			className: "ZmOverviewZimletHeader"
289 		};
290 		params.overviewTrees = [ZmOrganizer.ZIMLET];
291 
292 		this._addSection(headerParams, null, params);
293 
294 		var header = this._headerItems[headerDataId];
295 		if (header) {
296 			header.__isZimlet = true;
297 			header.setExpanded(appCtxt.get(ZmSetting.ZIMLET_TREE_OPEN, null, mainAcct));
298 		}
299 	}
300 
301 	this._initializeActionMenu();
302 };

304 /**
305  * Adds a search folder.
306  * 
307  * @param	{ZmFolder}		folder		the folder
308  */
309 ZmAccountOverviewContainer.prototype.addSearchFolder =
310 function(folder) {
311 	if (!this._searchTreeHeader) { return; }
312 
313 	var params = {
314 		parent: this._searchTreeHeader,
315 		text: folder.getName(),
316 		imageInfo: folder.getIcon()
317 	};
318 	var ti = new DwtTreeItem(params);
319 	ti.setData(Dwt.KEY_ID, folder);
320 	ti._initialize(null, true);
321 
322 	if (!this._searchTreeHeader.getVisible()) {
323 		this._searchTreeHeader.setVisible(true);
324 	}
325 };

327 /**
328  * Sets/updates the account-level status icon next to account name tree header.
329  * This only applies to app-based overview containers (i.e. not dialogs). Also resets the
330  * tooltip for the account header tree item.
331  *
332  * @param {ZmZimbraAccount}	account		the account to update status icon for
333  * @param	{Boolean}	updateStatus	if <code>true</code>, update the status
334  * @param	{Boolean}	updateTooltip	if <code>true</code>, update the tool tip
335  */
336 ZmAccountOverviewContainer.prototype.updateAccountInfo =
337 function(account, updateStatus, updateTooltip) {
338 	// check if appName is a real app (and not a dialog) before setting the account status
339 	var hi = appCtxt.getApp(this._appName) && this.getHeaderItem(account);
340 	if (hi) {
341 		if (updateStatus) {
342 			var html = (account.status == ZmZimbraAccount.STATUS_RUNNING)
343 				? ("<img src='/img/animated/ImgSpinner.gif' width=16 height=16 border=0>")
344 				: (AjxImg.getImageHtml(account.getStatusIcon()));
345 
346 			if (hi._extraCell) {
347 				hi._extraCell.innerHTML = (html || "");
348 			}
349             if (appCtxt.isOffline && account.status == ZmZimbraAccount.STATUS_AUTHFAIL) {
350                 var dialog = appCtxt.getPasswordChangeDialog();
351                 dialog.popup(account);
352             }
353 		}
354 
355 		if (updateTooltip || updateStatus) {
356 			hi.setToolTipContent(account.getToolTip());
357 		}
358 	}
359 };

361 /**
362  * Updates the label.
363  * 
364  * @param	{ZmOrganizer}	organizer		the organizer
365  */
366 ZmAccountOverviewContainer.prototype.updateLabel =
367 function(organizer) {
368 	// update account header if necessary
369 	if (organizer.nId == ZmOrganizer.ID_INBOX) {
370 		var acct = organizer.getAccount();
371 		var header = this.getHeaderItem(acct);
372 		if (header && !header.getExpanded()) {
373 			header.setText(this._getAccountHeaderLabel(acct));
374 		}
375 	}
376 
377 	// update virtual folder label
378 	var ti = this._vFolderTreeItemMap[organizer.nId];
379 	if (ti) {
380 		ti.setText(this._getFolderLabel(organizer.nId, organizer.name));
381 		if (organizer.nId == ZmOrganizer.ID_INBOX &&
382 			!this._allMailboxesTreeHeader.getExpanded())
383 		{
384 			var text = this._getFolderLabel(organizer.nId, this._allMailboxesTreeHeader.__origText);
385 			this._allMailboxesTreeHeader.setText(text);
386 		}
387 	}
388 };

404 ZmAccountOverviewContainer.prototype.resetOperations =
405 function(parent, data) {
406 
407 	var emptyFolderOp = parent.getOp(ZmOperation.EMPTY_FOLDER);
408 
409 	if (data instanceof ZmSearchFolder) {
410 		parent.getOp(ZmOperation.MARK_ALL_READ).setVisible(false);
411 		emptyFolderOp.setVisible(false);
412 		parent.getOp(this._newOp).setVisible(false);
413 		if (appCtxt.isOffline) {
414 			parent.getOp(ZmOperation.SYNC).setVisible(false);
415 		}
416 		parent.getOp(ZmOperation.DELETE).setVisible(true);
417 		return;
418 	}
419 
420 	var acct = appCtxt.accountList.getAccount(data);
421 	var isAcctType = (acct || data == ZmOrganizer.ID_ALL_MAILBOXES);
422 
423 	parent.getOp(ZmOperation.MARK_ALL_READ).setVisible(!isAcctType);
424 	emptyFolderOp.setVisible(false);
425 	parent.getOp(this._newOp).setVisible(isAcctType && data != ZmOrganizer.ID_ALL_MAILBOXES);
426 	if (appCtxt.isOffline) {
427 		parent.getOp(ZmOperation.SYNC).setVisible(isAcctType && (!acct || (acct && !acct.isMain)));
428 	}
429 	parent.getOp(ZmOperation.DELETE).setVisible(false);
430 
431 	if (isAcctType) {
432 		parent.enable(this._newOp, true); 
433 		parent.enable(ZmOperation.SYNC, (!acct || (acct && !acct.isMain)));
434 	} else {
435 		// reset mark all based on a "friendly" hack ;)
436 		var markAllEnabled = false;
437 		if (data != ZmOrganizer.ID_OUTBOX && data != ZmFolder.ID_DRAFTS &&
438 			this._actionedHeaderItem.getText().indexOf("bold") != -1)
439 		{
440 			markAllEnabled = true;
441 		}
442 		parent.enable(ZmOperation.MARK_ALL_READ, markAllEnabled);
443 
444 		// reset empty "folder" based on Trash/Junk
445 		if (data == ZmOrganizer.ID_TRASH || data == ZmOrganizer.ID_SPAM) {
446 			var text = (data == ZmOrganizer.ID_TRASH) ? ZmMsg.emptyTrash : ZmMsg.emptyJunk;
447 			emptyFolderOp.setText(text);
448 			emptyFolderOp.setVisible(true);
449 			parent.enable(ZmOperation.EMPTY_FOLDER, !this._isFolderEmpty(data));
450 		}
451 	}
452 };

473 ZmAccountOverviewContainer.prototype._addAccount =
474 function(params, account, showBackgroundColor, headerClassName, predictableId) {
475 	params.overviewId = appCtxt.getOverviewId(this.containerId, account);
476 	params.account = account;	// tree controller might need reference to account
477 
478 	// only show sections for apps that are supported unless this is prefs app
479 	var app = appCtxt.getApp(this._appName);
480 	var isSupported = (!app || (app && appCtxt.get(ZmApp.SETTING[this._appName], null, account)));
481 
482 	if (this._appName == ZmApp.PREFERENCES || isSupported) {
483 		var omit = params.omitPerAcct
484 			? params.omitPerAcct[account.id] : params.omit;
485 
486 		var headerLabel, headerIcon;
487 		if (this._appName == ZmApp.PREFERENCES && account.isMain && appCtxt.isOffline) {
488 			headerLabel = ZmMsg.allAccounts;
489 			headerIcon = "AccountAll";
490 		} else {
491 			headerLabel = account.getDisplayName();
492 			if (!appCtxt.isFamilyMbox) {
493 				headerIcon = account.getIcon()
494 			}
495 		}
496 
497 		var headerParams = {
498 			label: headerLabel,
499 			icon: headerIcon,
500 			dataId: account.id,
501 			className: headerClassName,
502 			predictableId: predictableId
503 		};
504 
505 		this._addSection(headerParams, omit, params, showBackgroundColor);
506 	}
507 
508 	var setting = appCtxt.getSettings(account).getSetting(ZmSetting.QUOTA_USED);
509 	setting.addChangeListener(this._settingChangeListener);
510 };

512 ZmAccountOverviewContainer.prototype._addSection =
513 function(headerParams, omit, overviewParams, showBackgroundColor) {
514 	// create a top-level section header
515 	var params = {
516 		parent: this,
517 		text:			headerParams.label,
518 		imageInfo:		headerParams.icon,
519 		selectable:		overviewParams.selectable,
520 		className:		headerParams.className,
521 		id:				this.getHTMLElId() + "__" + (headerParams.predictableId || headerParams.dataId) + "__SECTION"
522 	};
523 	var header = this._headerItems[headerParams.dataId] = new DwtTreeItem(params);
524 	header.setData(Dwt.KEY_ID, headerParams.dataId);
525 	header.setScrollStyle(Dwt.CLIP);
526 	header._initialize(null, true, true);
527 	header.addClassName(showBackgroundColor ? "ZmOverviewSectionHilite" : "ZmOverviewSection");
528 
529 	// reset some params for child overviews
530 	overviewParams.id = ZmId.getOverviewId(overviewParams.overviewId);
531 	overviewParams.parent = header;
532 	overviewParams.scroll = Dwt.CLIP;
533 	overviewParams.posStyle = Dwt.STATIC_STYLE;
534 
535 	// next, create an overview for this account and add it to the account header
536 	var ov = this._controller._overview[overviewParams.overviewId] = this._overview[overviewParams.overviewId] = new ZmOverview(overviewParams, this._controller);
537 	header._dndScrollCallback = this._overview._dndScrollCallback,
538 	header._dndScrollId = this._overview._scrollableContainerId,
539 	header._addItem(ov, null, true);
540 
541 	// finally set treeviews for this overview
542 	var treeIds = overviewParams.overviewTrees || overviewParams.treeIds;
543 	ov.set(treeIds, omit);
544 };

546 ZmAccountOverviewContainer.prototype._setupHeader =
547 function(header, acct) {
548 	// always expand header in prefs app, otherwise follow implicit user pref
549 	if (this._appName == ZmApp.PREFERENCES) {
550 		header.setExpanded(true, false, true);
551 		header.enableSelection(false);
552 	} else {
553 		var isExpanded = appCtxt.get(ZmSetting.ACCOUNT_TREE_OPEN, null, acct);
554 		header.setExpanded(isExpanded);
555 		if (!isExpanded) {
556 			header.setText(this._getAccountHeaderLabel(acct));
557 		}
558 	}
559 
560 	// add onclick support
561 	if (header._extraCell) {
562 		header._extraCell.onclick = AjxCallback.simpleClosure(this._handleStatusClick, this, acct);
563 	}
564 
565 	// add DnD support
566 	var dropTgt = this._controller.getTreeController(ZmOrganizer.FOLDER).getDropTarget();
567 	var root = ZmOrganizer.getSystemId(ZmOrganizer.ID_ROOT, acct);
568 	header.setDropTarget(dropTgt);
569 	header.setData(Dwt.KEY_OBJECT, appCtxt.getById(root));
570 };

572 /**
573  * Iterates all visible account and checks whether folder for each account is
574  * empty. Used by "All Mailboxes" to determine whether a virtual folder needs
575  * to be disabled or not when right-clicked.
576  *
577  * @param folderId		[String]		Normalized folder ID (should not be fully qualified)
578  */
579 ZmAccountOverviewContainer.prototype._isFolderEmpty =
580 function(folderId) {
581 	var accounts = appCtxt.accountList.visibleAccounts;
582 	for (var i = 0; i < accounts.length; i++) {
583 		var folder = appCtxt.getById(ZmOrganizer.getSystemId(folderId, accounts[i]));
584 		if (folder && folder.numTotal > 0) {
585 			return false;
586 		}
587 	}
588 
589 	return true;
590 };

598 ZmAccountOverviewContainer.prototype._treeViewListener =
599 function(ev) {
600 	if (ev.detail != DwtTree.ITEM_ACTIONED &&
601 		ev.detail != DwtTree.ITEM_SELECTED &&
602 		ev.detail != DwtTree.ITEM_DBL_CLICKED)
603 	{
604 		return;
605 	}
606 
607 	var item = this._actionedHeaderItem = ev.item;
608 
609 	// do nothing if zimlet/search is clicked
610 	if (item && (item.__isZimlet || item.__isSearch)) { return; }
611 
612 	var data = item && item.getData(Dwt.KEY_ID);
613 
614 	if (ev.detail == DwtTree.ITEM_ACTIONED && appCtxt.getApp(this._appName)) {	// right click
615 		var actionMenu = this._getActionMenu(data);
616 		if (actionMenu) {
617 			this.resetOperations(actionMenu, data);
618 			actionMenu.popup(0, ev.docX, ev.docY);
619 		}
620 	}
621 	else if ((ev.detail == DwtTree.ITEM_SELECTED) && item) {					// left click
622 		// if calendar/prefs app, do nothing
623 		if (this._appName == ZmApp.CALENDAR ||
624 			this._appName == ZmApp.PREFERENCES)
625 		{
626 			return;
627 		}
628 
629 		this._deselectAllTreeViews();
630 
631 		// this avoids processing clicks in dialogs etc.
632 		if (!ZmApp.NAME[this._appName]) { return; }
633 
634 		// if an account header item was clicked, run the default search for it
635 		if (data) {
636 			var sc = appCtxt.getSearchController();
637 
638 			var account = appCtxt.accountList.getAccount(data);
639 			if (account) {
640 				// bug 41196 - turn off new mail notifier if inactive account header clicked
641 				if (appCtxt.isOffline && account.inNewMailMode) {
642 					account.inNewMailMode = false;
643 					var allContainers = appCtxt.getOverviewController()._overviewContainer;
644 					for (var i in allContainers) {
645 						allContainers[i].updateAccountInfo(account, true, true);
646 					}
647 				}
648 
649 				// don't process click if user clicked on error status icon
650 				if ((ev.target.parentNode == ev.item._extraCell) && account.isError()) {
651 					return;
652 				}
653 
654 				sc.searchAllAccounts = false;
655 				appCtxt.accountList.setActiveAccount(account);
656 
657 				if (appCtxt.isOffline && account.hasNotSynced() && !account.__syncAsked) {
658 					account.__syncAsked = true;
659 
660 					var dialog = appCtxt.getYesNoMsgDialog();
661 					dialog.registerCallback(DwtDialog.YES_BUTTON, this._syncAccount, this, [dialog, account]);
662 					dialog.setMessage(ZmMsg.neverSyncedAsk, DwtMessageDialog.INFO_STYLE);
663 					dialog.popup();
664 				}
665 
666 				var fid = ZmOrganizer.DEFAULT_FOLDER[ZmApp.ORGANIZER[this._appName]];
667 				var folder = appCtxt.getById(ZmOrganizer.getSystemId(fid, account));
668 
669 				// briefcase is not a ZmFolder so let's skip for now
670 				if (!(folder instanceof ZmFolder)) { return; }
671 
672 				var defaultSortBy = (this._appName == ZmApp.TASKS)
673 					? ZmSearch.DUE_DATE_DESC : ZmSearch.DATE_DESC;
674 
675 				var params = {
676 					query: folder.createQuery(),
677 					getHtml: appCtxt.get(ZmSetting.VIEW_AS_HTML),
678 					searchFor: (ZmApp.DEFAULT_SEARCH[this._appName]),
679 					sortBy: ((sc.currentSearch && folder.nId == sc.currentSearch.folderId) ? null : defaultSortBy),
680 					accountName: (account && account.name),
681 					noUpdateOverview: true
682 				};
683 			} else {
684 				var main = appCtxt.accountList.mainAccount;
685 				sc.resetSearchAllAccounts();
686 				sc.searchAllAccounts = true;
687 
688 				if (data instanceof ZmSearchFolder) {
689 					params = {
690 						searchAllAccounts: true,
691 						accountName: main.name,
692 						getHtml: appCtxt.get(ZmSetting.VIEW_AS_HTML),
693 						noUpdateOverview: true
694 					};
695 					sc.redoSearch(data.search, false, params);
696 					return;
697 				}
698 				if (data == ZmOrganizer.ID_ALL_MAILBOXES) {
699 					data = ZmFolder.ID_INBOX;
700 				}
701 
702 				params = {
703 					queryHint: appCtxt.accountList.generateQuery(data),
704 					folderId: null,
705 					getHtml: appCtxt.get(ZmSetting.VIEW_AS_HTML, null, main),
706 					searchFor: (ZmApp.DEFAULT_SEARCH[this._appName]),
707 					sortBy: ZmSearch.DATE_DESC,
708 					accountName: main.name,
709 					noUpdateOverview: true
710 				};
711 			}
712 
713 			sc.search(params);
714 		}
715 	} else {																	// double click
716 		// handle double click?
717 	}
718 };

720 ZmAccountOverviewContainer.prototype._treeListener =
721 function(ev) {
722 	if (ev.detail != DwtTree.ITEM_COLLAPSED &&
723 		ev.detail != DwtTree.ITEM_EXPANDED)
724 	{
725 		return;
726 	}
727 
728 	var header = ev.item;
729 	var expanded = ev.detail == DwtTree.ITEM_EXPANDED;
730 
731 	var acct;
732 	if (header) {
733 		if (header.__isSearch) {
734 			appCtxt.set(ZmSetting.OFFLINE_SAVED_SEARCHES_TREE_OPEN, expanded);
735 			return;
736 		}
737 
738 		var data = header.getData(Dwt.KEY_ID);
739 		if (data == ZmOrganizer.ID_ALL_MAILBOXES) {
740 			var text = expanded
741 				? header.__origText
742 				: this._getFolderLabel(ZmOrganizer.ID_INBOX, header.__origText);
743 			header.setText(text);
744 
745 			// bug #43734 - remember expand/collapse state
746 			appCtxt.set(ZmSetting.OFFLINE_ALL_MAILBOXES_TREE_OPEN, expanded);
747 
748 			return;
749 		}
750 
751 		acct = header.__isZimlet
752 			? appCtxt.accountList.mainAccount
753 			: appCtxt.accountList.getAccount(data);
754 	}
755 
756 	if (acct && appCtxt.getCurrentAppName() != ZmApp.PREFERENCES) {
757 		if (!appCtxt.inStartup) {
758             // set account as active when account tree is opened/closed
759             appCtxt.accountList.setActiveAccount(acct);
760 			appCtxt.set(ZmSetting.ACCOUNT_TREE_OPEN, expanded, null, null, null, acct);
761 		}
762 
763 		if (!header.__isZimlet) {
764 			var text = expanded
765 				? acct.getDisplayName()
766 				: this._getAccountHeaderLabel(acct);
767 			header.setText(text);
768 		}
769 	}
770 };

777 ZmAccountOverviewContainer.prototype._handleSettingChange =
778 function(ev) {
779 	if (ev.type != ZmEvent.S_SETTING) { return; }
780 
781 	var setting = ev.source;
782 
783 	if (setting.id == ZmSetting.OFFLINE_SHOW_ALL_MAILBOXES) {
784 		var isVisible = setting.getValue();
785 		this._allMailboxesTreeHeader.setVisible(isVisible);
786 		if (!isVisible) {
787 			if (appCtxt.getActiveAccount().isMain) {
788 				appCtxt.accountList.setActiveAccount(appCtxt.accountList.defaultAccount);
789 			}
790 			appCtxt.getSearchController().searchAllAccounts = false;
791 			appCtxt.getApp(ZmApp.MAIL).mailSearch();
792 		} else {
793 			this._deselect(this._allMailboxesTreeHeader);
794 		}
795 	}
796 	else if (setting.id == ZmSetting.QUOTA_USED) {
797 		this.updateAccountInfo(ev.getDetails().account, false, true);
798 	}
799 };

801 ZmAccountOverviewContainer.prototype._getAccountHeaderLabel =
802 function(acct, header) {
803 	var inboxId = (this._appName == ZmApp.MAIL)
804 		? ZmOrganizer.getSystemId(ZmOrganizer.ID_INBOX, acct, true) : null;
805 	var inbox = inboxId && appCtxt.getById(inboxId);
806 	if (inbox && inbox.numUnread > 0) {
807 		var name = AjxMessageFormat.format(ZmMsg.folderUnread, [acct.getDisplayName(), inbox.numUnread]);
808 		return (["<span style='font-weight:bold;'>", name, "</span>"].join(""));
809 	}
810 
811 	return acct.getDisplayName();
812 };

814 ZmAccountOverviewContainer.prototype._getFolderLabel =

820 		return (["<span style='font-weight:bold;'>", name, "</span>"].join(""));
821 	}
822 
823 	return label;

826 ZmAccountOverviewContainer.prototype._initializeActionMenu =
827 function() {
828 	if (!this._actionMenu) {
829 		var orgType = ZmApp.ORGANIZER[this._appName];
830 		this._newOp = ZmOrganizer.NEW_OP[orgType];
831 
832 		var ops = [this._newOp];
833 		if (appCtxt.isOffline) {
834 			ops.push(ZmOperation.SYNC);
835 		}
836 		ops.push(ZmOperation.MARK_ALL_READ,
837 				ZmOperation.EMPTY_FOLDER,
838 				ZmOperation.DELETE);
839 
840 		this._actionMenu = new AjxCallback(this, this._createActionMenu, [ops]);
841 	}
842 };

844 ZmAccountOverviewContainer.prototype._createActionMenu =
845 function(menuItems) {
846 	var listener = new AjxListener(this, this._actionMenuListener);
847 	var actionMenu = new ZmActionMenu({parent:appCtxt.getShell(), menuItems:menuItems});
848 	menuItems = actionMenu.opList;
849 	for (var i = 0; i < menuItems.length; i++) {
850 		var mi = actionMenu.getItem(i);
851 		var op = menuItems[i];
852 		if (op == ZmOperation.SYNC) {
853 			mi.setText(ZmMsg.sendReceive);
854 		}
855 		actionMenu.addSelectionListener(op, listener);
856 	}
857 
858 	actionMenu.addPopdownListener(new AjxListener(this, this._menuPopdownActionListener));
859 
860 	return actionMenu;
861 };

868 ZmAccountOverviewContainer.prototype._actionMenuListener =
869 function(ev) {
870 	var opId = ev.item.getData(ZmOperation.KEY_ID);
871 	var data = this._actionedHeaderItem.getData(Dwt.KEY_ID);
872 
873 	if (opId == this._newOp) {
874 		var treeId = ZmApp.ORGANIZER[this._appName];
875 		var tc = this._controller.getTreeController(treeId, true);
876 		if (tc) {
877 			tc._actionedOrganizer = null;
878 			var account = appCtxt.accountList.getAccount(data);
879 			tc._actionedOrganizer = appCtxt.getFolderTree(account).root;
880 			tc._newListener(ev, account);
881 		}
882 	}
883 	else if (opId == ZmOperation.SYNC) {
884 		if (data == ZmOrganizer.ID_ALL_MAILBOXES) {
885 			appCtxt.accountList.syncAll();
886 		} else {
887 			var account = appCtxt.accountList.getAccount(data);
888 			if (account) {
889 				account.sync();
890 			}
891 		}
892 	}
893 	else if (opId == ZmOperation.MARK_ALL_READ) {
894 		this._doAction(data, opId);
895 	}
896 	else if (opId == ZmOperation.EMPTY_FOLDER) {
897 		this._confirmEmptyAction(data, opId);
898 	}
899 	else if (opId == ZmOperation.DELETE) {
900 		data._delete();
901 		var parent = this._actionedHeaderItem.parent;
902 		parent.removeChild(this._actionedHeaderItem); // HACK: just nuke it
903 		parent.setVisible(parent.getItemCount() > 0);
904 	}
905 };

914 		? ZmMsg.confirmEmptyTrashFolder

918 	dialog.popup();

927 ZmAccountOverviewContainer.prototype._doAction =
928 function(folderId, opId) {
929 	var bc = new ZmBatchCommand(true, appCtxt.accountList.mainAccount.name);
930 
931 	var accounts = appCtxt.accountList.visibleAccounts;
932 	for (var i = 0; i < accounts.length; i++) {
933 		var account = accounts[i];
934 		if (account.isMain) { continue; }
935 
936 		var fid = ZmOrganizer.getSystemId(folderId, account);
937 		var folder = appCtxt.getById(fid);
938 		if (folder) {
939 			if (opId == ZmOperation.MARK_ALL_READ) {
940 				folder.markAllRead(bc);
941 			} else {
942 				folder.empty(null, bc);
943 			}
944 			bc.curId++;
945 		}
946 	}
947 
948 	bc.run();
949 	if (appCtxt.isOffline) {
950 		appCtxt.getApp(ZmApp.MAIL).clearNewMailBadge();
951 	}
952 };
