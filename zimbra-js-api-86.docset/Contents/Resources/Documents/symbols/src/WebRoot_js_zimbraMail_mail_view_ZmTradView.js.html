 22 ZmTradView = function(params) {
 23 
 24 	params.className = params.className || "ZmTradView";
 25 	params.mode = ZmId.VIEW_TRAD;
 26 	ZmDoublePaneView.call(this, params);
 27 }
 28 
 29 ZmTradView.prototype = new ZmDoublePaneView;
 30 ZmTradView.prototype.constructor = ZmTradView;
 31 
 32 ZmTradView.prototype.isZmTradView = true;
 33 ZmTradView.prototype.toString = function() { return "ZmTradView"; };
 34 
 35 ZmTradView.prototype._createMailListView =
 36 function(params) {
 37 	return new ZmMailMsgListView(params);
 38 };
 39 
 40 ZmTradView.prototype._createMailItemView =
 41 function(params) {
 42 	params.id = ZmId.getViewId(ZmId.VIEW_MSG, null, params.view);
 43 	return new ZmMailMsgView(params);
 44 };
