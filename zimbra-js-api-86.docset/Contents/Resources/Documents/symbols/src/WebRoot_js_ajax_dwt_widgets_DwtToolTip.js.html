 22 /**
 23  * Singleton tooltip class.
 24  */
 25 DwtToolTip = function(shell, className, dialog) {
 26 
 27 	if (arguments.length == 0) { return; }
 28 
 29 	this.shell = shell;
 30 	this._dialog = dialog;
 31 	this._poppedUp = false;
 32 	this._div = document.createElement("div");
 33 	this._div.className = className || "DwtToolTip";
 34 	this._div.style.position = DwtControl.ABSOLUTE_STYLE;
 35 	this.shell.getHtmlElement().appendChild(this._div);
 36 	Dwt.setZIndex(this._div, Dwt.Z_HIDDEN);
 37 	Dwt.setLocation(this._div, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
 38 
 39 	this._eventMgr = new AjxEventMgr();
 40 
 41     // create html
 42     // NOTE: This id is ok because there's only ever one instance of a tooltip
 43     var templateId = "dwt.Widgets#DwtToolTip";
 44     this._div.innerHTML = AjxTemplate.expand(templateId);
 45 
 46     var params = AjxTemplate.getParams(templateId);
 47     this._offsetX = (params.width != null) ? Number(params.width) : DwtToolTip.POPUP_OFFSET_X;
 48     this._offsetY = (params.height != null) ? Number(params.height) : DwtToolTip.POPUP_OFFSET_Y;
 49 
 50     // save reference to content div
 51     this._contentDiv = document.getElementById("tooltipContents");
 52 
 53     Dwt.setHandler(this._div, DwtEvent.ONMOUSEOVER, AjxCallback.simpleClosure(this._mouseOverListener, this));
 54     Dwt.setHandler(this._div, DwtEvent.ONMOUSEOUT, AjxCallback.simpleClosure(this._mouseOutListener, this));
 55 
 56 	var events = [DwtEvent.ONCLICK,DwtEvent.ONDBLCLICK,DwtEvent.ONMOUSEDOWN,DwtEvent.ONMOUSEENTER,DwtEvent.ONMOUSELEAVE,DwtEvent.ONMOUSEMOVE,DwtEvent.ONMOUSEUP,DwtEvent.ONMOUSEWHEEL,DwtEvent.ONSCROLL];
 57 	for (var i=0; i<events.length; i++) {
 58 		var event = events[i];
 59     	Dwt.setHandler(this._div, event, AjxCallback.simpleClosure(this.notifyListeners, this, [event]));
 60 	}
 61 };

 97 /**
 98  * Shows the tooltip. By default, its position will be relative to the location of the
 99  * cursor when the mouseover event happened. Alternatively, the control that generated
100  * the tooltip can be passed in, and the tooltip will be positioned relative to it. If
101  * the control is a large composite control (eg a DwtListView), the hover event can be
102  * passed so that the actual target of the event can be found.
103  * 
104  * @param {number}			x					X-coordinate of cursor
105  * @param {number}			y					Y-coordinate of cursor
106  * @param {boolean}			skipInnerHTML		if true, do not copy content to DOM
107  * @param {boolean}			popdownOnMouseOver	if true, hide tooltip on mouseover
108  * @param {DwtControl}		obj					control that tooltip is for (optional)
109  * @param {DwtHoverEvent}	hoverEv				hover event (optional)
110  * @param {AjxCallback}		popdownListener		callback to run when tooltip pops down
111  */
112 DwtToolTip.prototype.popup = 
113 function(x, y, skipInnerHTML, popdownOnMouseOver, obj, hoverEv, popdownListener) {
114 	this._hovered = false;
115     if (this._popupAction) {
116         AjxTimedAction.cancelAction(this._popupAction);
117         this._popupAction = null;
118     }
119 	// popdownOnMouseOver may be true to pop down the tooltip if the mouse hovers over the tooltip. Optionally,
120 	// it can be an AjxCallback that will be called after popping the tooltip down.
121     this._popdownOnMouseOver = popdownOnMouseOver;
122 	// popdownListener is always called after popping the tooltip down, regardless of what called the popdown
123     this._popdownListener = popdownListener;
124     if (this._content != null) {
125 		if(!skipInnerHTML) {
126             this._contentDiv.innerHTML = this._content;
127         }
128 
129 		this._popupAction = new AjxTimedAction(this, this._positionElement, [x, y, obj, hoverEv]);
130 		AjxTimedAction.scheduleAction(this._popupAction, 5);
131 	}
132 };

134 /*
135 * setSticky allows making the tooltip not to popdown. 
136 * IMPORTANT: Tooltip is singleton inside Zimbra i.e. only one instance of tooltip is reused by all objects. 
137 * So, it is very important for the code setting tooltip to sticky to have some mechanism to close the tooltip by itself. 
138 * Like have a close-button inside tooltip and when clicked, should set the setSticky(false) and then close the tooltip.
139 *
140 * If setSticky(true) is called, _poppedUp is set to false, which is essentially pretending the tooltip is not
141 * up. In that case, a call to popdown will not close the tooltip. And that means tooltip will stay up even if some other
142 * code path calls popdown on the singleton tooltip.
143 *
144 */
145 DwtToolTip.prototype.setSticky = 
146 function(bool) {
147 	this._poppedUp = !bool;
148 };
149 
150 DwtToolTip.prototype.popdown = 
151 function() {
152     this._popdownOnMouseOver = false;
153 	this._hovered = false;
154     if (this._popupAction) {
155         AjxTimedAction.cancelAction(this._popupAction);
156         this._popupAction = null;
157     }
158 	if (this._content != null && this._poppedUp) {
159 		Dwt.setLocation(this._div, Dwt.LOC_NOWHERE, Dwt.LOC_NOWHERE);
160 		this._poppedUp = false;
161 		if (this._popdownListener instanceof AjxCallback) {
162 			this._popdownListener.run();
163 		}
164 		this._popdownListener = null;
165 	}
166 };

172 // Positions the tooltip relative to the base element based on vertical and horizontal offsets.
173 DwtToolTip.prototype._positionElement = 
174 function(startX, startY, obj, hoverEv) {
175 	
176     this._popupAction = null;
177 	
178 	var wdSize = DwtShell.getShell(window).getSize();
179 	var wdWidth = wdSize.x, wdHeight = wdSize.y;
180 
181 	var tooltipX, tooltipY, baseLoc;
182 	var baseEl = obj && obj.getTooltipBase(hoverEv);
183 	if (baseEl) {
184 		baseLoc = Dwt.toWindow(baseEl);
185 		var baseSz = Dwt.getSize(baseEl);
186 		tooltipX = baseLoc.x + this._offsetX;
187 		tooltipY = baseLoc.y + baseSz.y + this._offsetY;
188 	}
189 	else {
190 		tooltipX = startX + this._offsetX;
191 		tooltipY = startY + this._offsetY;
192 	}
193 
194 	var popupSize = Dwt.getSize(this._div);
195 	var popupWidth = popupSize.x, popupHeight = popupSize.y;
196 
197 	// check for sufficient room to the right
198 	if (tooltipX + popupWidth > wdWidth - DwtToolTip.WINDOW_GUTTER) {
199 		tooltipX = wdWidth - DwtToolTip.WINDOW_GUTTER - popupWidth;
200 	}
201 	// check for sufficient room below
202 	if (tooltipY + popupHeight > wdHeight - DwtToolTip.WINDOW_GUTTER) {
203 		tooltipY = (baseLoc ? baseLoc.y : tooltipY) - this._offsetY - popupHeight;
204 	}
205 
206 	Dwt.setLocation(this._div, tooltipX, tooltipY);
207 	var zIndex = this._dialog ? this._dialog.getZIndex() + Dwt._Z_INC : Dwt.Z_TOOLTIP;
208 	Dwt.setZIndex(this._div, zIndex);
209     this._poppedUp = true;
210 };
211 
212 DwtToolTip.prototype._mouseOverListener = 
213 function(ev) {
214 	this._hovered = true;
215     if (this._popdownOnMouseOver && this._poppedUp) {
216         var callback = (this._popdownOnMouseOver.isAjxCallback || AjxUtil.isFunction(this._popdownOnMouseOver)) ? this._popdownOnMouseOver : null;
217         this.popdown();
218         if (callback) {
219             callback.run();
220 		}
221     }
222 	this.notifyListeners(DwtEvent.ONMOUSEOVER);
223 };
224 
225 DwtToolTip.prototype._mouseOutListener = 
226 function(ev) {
227 	ev = DwtUiEvent.getEvent(ev, this._div)
228 	var location = Dwt.toWindow(this._div);
229 	var size = Dwt.getSize(this._div);
230 	// We sometimes get mouseover events even though the cursor is inside the tooltip, so double-check before popping down
231 	if (ev.clientX <= location.x || ev.clientX >= (location.x + size.x) || ev.clientY <= location.y || ev.clientY >= (location.y + size.y)) {
232 		this.popdown();
233 		this.notifyListeners(DwtEvent.ONMOUSEOUT);
234 	}
235 };
236 
237 DwtToolTip.prototype.getHovered = 
238 function() {
239 	return this._hovered;
240 };

243 // The com_zimbra_email zimlet wants to put a listener on our mouseout event, but overwriting the existing handler is a no-no
244 // and we actually only want that event when the double-check above succeeds. Let API users add event listeners in a more clean way.
245 DwtToolTip.prototype.addListener =
246 function(eventType, listener, index) {
247 	return this._eventMgr.addListener(eventType, listener, index);
248 };
249 
250 DwtToolTip.prototype.setListener =
251 function(eventType, listener, index) {
252 	this.removeAllListeners(eventType);
253 	return this._eventMgr.addListener(eventType, listener, index);
254 };
255 
256 DwtToolTip.prototype.removeListener =
257 function(eventType, listener) {
258 	return this._eventMgr.removeListener(eventType, listener);
259 };
260 
261 DwtToolTip.prototype.removeAllListeners =
262 function(eventType) {
263 	return this._eventMgr.removeAll(eventType);
264 };
265 
266 DwtToolTip.prototype.isListenerRegistered =
267 function(eventType) {
268 	return this._eventMgr.isListenerRegistered(eventType);
269 };
270 
271 DwtToolTip.prototype.notifyListeners =
272 function(eventType, event) {
273 	return this._eventMgr.notifyListeners(eventType, event);
274 };
