 27 /**
 28  * Creates a timezone
 29  * @class
 30  * This class represents a timezone.
 31  * 
 32  */
 33 ZmTimezone = function() {}
 34 
 35 // Static methods
 36 
 37 /**
 38  * Gets the default timezone.
 39  * 
 40  * @return	{AjxTimezone}	the timezone
 41  */
 42 ZmTimezone.getDefault =
 43 function() {
 44 	var shell = DwtShell.getShell(window);
 45 	var serverId = appCtxt.get(ZmSetting.DEFAULT_TIMEZONE);
 46 	return (serverId) ? AjxTimezone.getClientId(serverId) : AjxTimezone.DEFAULT;
 47 };
 48 
 49 /**
 50  * Gets the default rule.
 51  * 
 52  * @return	{String}	the rule
 53  */
 54 ZmTimezone.getDefaultRule =
 55 function() {
 56 	return AjxTimezone.getRule(ZmTimezone.getDefault());
 57 };
 58 
 59 /**
 60  * This function mirrors the <code>AjxSoapDoc#set</code> method
 61  * to add a timezone element at the specific place within the
 62  * given SOAP document. The added element takes the form of the
 63  * <code><tz></code> element as defined for <code><SearchRequest></code>.
 64  *
 65  * @param {object|AjxSoapDoc}	request			the JSON request object or SOAP document
 66  * @param {String}	timezoneClientId	the client identifier
 67  * @param {Node}	parentNode		(optional) the parent node at which to add
 68  * @param {Boolean}	skipKnownTimezone	(optional) if <code>true</code>, does not add the "tz" element if it's one of the known set
 69  */
 70 ZmTimezone.set =
 71 function(request, timezoneClientId, parentNode, skipKnownTimezone) {
 72 	var timezone = AjxTimezone.getRule(timezoneClientId);
 73 	if (!timezone) { return; }
 74 
 75 	if (timezone.autoDetected || !skipKnownTimezone) {
 76 		if (request instanceof AjxSoapDoc) {
 77 			ZmTimezone._setSoap(request, timezoneClientId, parentNode, timezone);
 78 		} else {
 79 			ZmTimezone._setJson(request, timezoneClientId, timezone);
 80 		}
 81 	}
 82 };
 83 
 84 /**
 85  * @private
 86  */
 87 ZmTimezone._setSoap =
 88 function(soapDoc, timezoneClientId, parentNode, timezone) {
 89 	var tz = soapDoc.set("tz", null, parentNode);
 90 	var id = AjxTimezone.getServerId(timezoneClientId);
 91 	tz.setAttribute("id", id);
 92 	if (timezone.autoDetected) {
 93 		tz.setAttribute("stdoff", timezone.standard.offset);
 94 		if (timezone.daylight) {
 95 			tz.setAttribute("dayoff", timezone.daylight.offset);
 96             var enames = [ "standard", "daylight" ];
 97             var pnames = [ "mon", "mday", "week", "wkday", "hour", "min", "sec" ];
 98             for (var i = 0; i < enames.length; i++) {
 99                 var ename = enames[i];
100                 var onset = timezone[ename];
101                 
102                 var el = soapDoc.set(ename, null, tz);
103                 for (var j = 0; j < pnames.length; j++) {
104                     var pname = pnames[j];
105                     if (pname in onset) {
106                         el.setAttribute(pname, onset[pname]);
107                     }
108                 }
109             }
110         }
111 	}
112 };
113 
114 /**
115  * @private
116  */
117 ZmTimezone._setJson =
118 function(request, timezoneClientId, timezone) {
119 	var id = AjxTimezone.getServerId(timezoneClientId);
120 	var tz = request.tz = {id:id};
121 	if (timezone.autoDetected) {
122 		tz.stdoff = timezone.standard.offset;
123 		if (timezone.daylight) {
124 			tz.dayoff = timezone.daylight.offset;
125             var enames = [ "standard", "daylight" ];
126             var pnames = [ "mon", "mday", "week", "wkday", "hour", "min", "sec" ];
127             for (var i = 0; i < enames.length; i++) {
128                 var ename = enames[i];
129                 var onset = timezone[ename];
130                 tz[ename] = {};
131                 for (var j = 0; j < pnames.length; j++) {
132                     var pname = pnames[j];
133                     if (pname in onset) {
134                     	tz[ename][pname] = onset[pname];
135                     }
136                 }
137             }
138         }
139 	}
140 };
