 22 /**
 23 * Creates an empty task view.
 24 * @constructor
 25 * @class
 26 * Simple read-only view of a task. It looks more or less like a message -
 27 * the notes have their own area at the bottom, and everything else goes into a
 28 * header section at the top.
 29 *
 30 * @author Parag Shah
 31 *
 32 * @param parent		[DwtComposite]	parent widget
 33 * @param posStyle	[constant]		positioning style
 34 * @param controller	[ZmController]	owning controller
 35 */
 36 ZmTaskView = function(parent, posStyle, controller) {
 37 
 38 	var id = ZmId.getViewId(ZmId.VIEW_TASK, null, parent._htmlElId);
 39 	ZmCalItemView.call(this, parent, posStyle, controller, id);
 40 };
 41 
 42 ZmTaskView.prototype = new ZmCalItemView;
 43 ZmTaskView.prototype.constructor = ZmTaskView;
 44 ZmTaskView.prototype.isZmTaskView = true;

 65 ZmTaskView.prototype._getSubs =
 66 function(calItem) {
 67 	var subject = calItem.getName();
 68 	var location = calItem.location;
 69 	var isException = calItem._orig ? calItem._orig.isException : calItem.isException;
 70 	var startDate = calItem.startDate ? AjxDateFormat.getDateInstance().format(calItem.startDate) : null;
 71 	var dueDate = calItem.endDate ? AjxDateFormat.getDateInstance().format(calItem.endDate) : null;
 72 	var priority = calItem.priority ? ZmCalItem.getLabelForPriority(calItem.priority) : null;
 73 	var status = calItem.status ? ZmCalItem.getLabelForStatus(calItem.status) : null;
 74 	var pComplete = calItem.pComplete;
 75 	var recurStr = calItem.isRecurring() ? calItem.getRecurBlurb() : null;
 76 	var attachStr = this._getAttachString(calItem);
 77     var alarm = calItem.alarm;
 78     var remindDate = calItem.remindDate ? AjxDateFormat.getDateInstance().format(calItem.remindDate) : null;
 79     var remindTime = calItem.remindDate ? AjxDateFormat.getTimeInstance(AjxDateFormat.SHORT).format(calItem.remindDate) : "";
 80 
 81 	if (this._objectManager) {
 82 		this._objectManager.setHandlerAttr(ZmObjectManager.DATE,
 83 											ZmObjectManager.ATTR_CURRENT_DATE,
 84 											calItem.startDate);
 85 
 86 		subject = this._objectManager.findObjects(subject, true);
 87 		if (location) location = this._objectManager.findObjects(location, true);
 88 		if (startDate) startDate = this._objectManager.findObjects(startDate, true);
 89 		if (dueDate) dueDate = this._objectManager.findObjects(dueDate, true);
 90 	}
 91 
 92 	return {
 93 		id: this._htmlElId,
 94 		subject: subject,
 95 		location: location,
 96 		isException: isException,
 97 		startDate: startDate,
 98 		dueDate: dueDate,
 99 		priority: priority,
100 		status: status,
101 		pComplete: pComplete,
102 		recurStr: recurStr,
103 		attachStr: attachStr,
104         remindDate: remindDate,
105         remindTime: remindTime,
106         alarm: alarm,
107 		folder: appCtxt.getTree(ZmOrganizer.TASKS).getById(calItem.folderId),
108 		folderLabel: ZmMsg.folder,
109         isTask:true,
110         _infoBarId:this._infoBarId
111 	};
112 };

116 ZmTaskView.prototype._renderCalItem =
117 function(calItem) {
118 
119    if(this._controller.isReadingPaneOn() && !this._newTab) {
120 	this._lazyCreateObjectManager();
121 
122 	var subs = this._getSubs(calItem);
123 	var editBtnCellId = this._htmlElId + "_editBtnCell";
124 	this._hdrTableId = this._htmlElId + "_hdrTable";
125 
126     var el = this.getHtmlElement();
127 	el.innerHTML = AjxTemplate.expand("tasks.Tasks#ReadOnlyView", subs);
128     this._setTags(calItem);
129 
130 	// content/body
131 	var hasHtmlPart = (calItem.notesTopPart && calItem.notesTopPart.getContentType() == ZmMimeTable.MULTI_ALT);
132 	var mode = (hasHtmlPart && appCtxt.get(ZmSetting.VIEW_AS_HTML))
133 		? ZmMimeTable.TEXT_HTML : ZmMimeTable.TEXT_PLAIN;
134 
135 	var bodyPart = calItem.getNotesPart(mode);
136 
137     if (!bodyPart && calItem.message){
138         bodyPart = calItem.message.getInviteDescriptionContentValue(ZmMimeTable.TEXT_PLAIN);
139     }
140 
141 	if (bodyPart) {
142 		this._msg = this._msg || this._calItem._currentlyLoaded;
143         if (mode === ZmMimeTable.TEXT_PLAIN) {
144             bodyPart = AjxStringUtil.convertToHtml(bodyPart);
145         }
146         this._makeIframeProxy({container: el, html:bodyPart, isTextMsg:(mode == ZmMimeTable.TEXT_PLAIN)});
147 	}
148    } else {
149      ZmCalItemView.prototype._renderCalItem.call(this, calItem);
150    }
151    Dwt.setLoadedTime("ZmTaskItem");
152    calItem.addChangeListener(this._taskChangeListener.bind(this));
153 
154 };
155 
156 ZmTaskView.prototype._taskChangeListener =
157 function(ev){
158     if(ev.event == ZmEvent.E_TAGS || ev.type == ZmEvent.S_TAG) {
159         this._setTags(this._calItem);
160     }
161 };
