 21 /**
 22  * From Group divides messages into sort by sender.
 23  */
 24 ZmMailListFromGroup = function() {
 25     this.id = ZmId.GROUPBY_FROM;
 26 	this.field = ZmItem.F_FROM;
 27     ZmMailListGroup.call(this);
 28 
 29 };
 30 
 31 ZmMailListFromGroup.prototype = new ZmMailListGroup;
 32 ZmMailListFromGroup.prototype.constructor =  ZmMailListFromGroup;
 33 
 34 /**
 35  *  returns HTML string for all sections.
 36  *  @param {boolean} sortAsc    true/false if sort ascending
 37  *  @return {String} HTML for all sections including section header
 38  * @param sortAsc
 39  */
 40 ZmMailListFromGroup.prototype.getAllSections =
 41 function(sortAsc) {
 42     var htmlArr = [];
 43     var keys = this._sortKeys(sortAsc);
 44 
 45     for(var i=0; i<keys.length; i++) {
 46         var key = keys[i].addr;
 47         if (this._section[key].length > 0) {
 48             var sectionHeader = keys[i].title;
 49             htmlArr.push(this.getSectionHeader(sectionHeader));
 50             htmlArr.push(this._section[key].join(""));
 51         }
 52     }
 53 
 54     return htmlArr.join("");
 55 };
 56 
 57 /**
 58  * Adds item to section
 59  * @param {ZmMailMsg} msg   mail message
 60  * @param {String} item  HTML to add to section
 61  * @return {String} section returns section if successfully added, else returns null
 62  */
 63 ZmMailListFromGroup.prototype.addMsgToSection =
 64 function(msg, item){
 65     var email = null;
 66     var fromParticipant =  msg.getAddress(AjxEmailAddress.FROM);
 67     if (fromParticipant) {
 68         email = fromParticipant.getAddress();
 69         if (this._section.hasOwnProperty(email)) {
 70             this._section[email].push(item);
 71         } else {
 72             this._section[email] = [];
 73             this._section[email].push(item);
 74             var title = this._getSectionTitle(fromParticipant);
 75             this._sectionTitle[email] = {addr: email, title:title};
 76         }
 77     }
 78     return email;
 79 };

 82 /**
 83  * Determines if message is in group
 84  * @param {String} section ID of section
 85  * @param {ZmMailMsg} msg
 86  * @return {boolean} true/false
 87  */
 88 ZmMailListFromGroup.prototype.isMsgInSection =
 89 function(section, msg) {
 90 
 91     var addr;
 92     var participants = msg.participants;
 93     if (participants) {
 94         var arr = participants.getArray();
 95         for (var i=0; i<arr.length; i++){
 96             if (arr[i].getType() == "FROM") {
 97               addr = arr[i].getAddress();
 98               break;
 99             }
100         }
101     }
102 
103     if (addr && addr == section) {
104        return true;
105     }
106 
107     return false;
108 };

110 /**
111  * Returns the sort by (ZmSearch.NAME_ASC or ZmSearch.NAME_DESC)
112  * @param {boolean} sortAsc
113  * @return {String} sortBy
114  */
115 ZmMailListFromGroup.prototype.getSortBy =
116 function(sortAsc) {
117     if (sortAsc) {
118         return ZmSearch.NAME_ASC;
119     }
120     return ZmSearch.NAME_DESC;
121 };

123 ZmMailListFromGroup.prototype._init =
124 function() {
125     this._section = {};
126     this._sectionTitle = {};
127 };
128 
129 ZmMailListFromGroup.prototype._sortKeys =
130 function(sortAsc) {
131   var keys = [];
132   var i=0;
133   for (var name in this._sectionTitle) {
134       keys[i++] = {addr: this._sectionTitle[name].addr, title: this._sectionTitle[name].title};
135   }
136 
137   var sortAscIgnoreCase = function(a, b) {
138     a = a.title.toLowerCase();
139     b = b.title.toLowerCase();
140     if (a > b)
141         return 1;
142     if (a < b)
143         return -1;
144     return 0;
145   };
146 
147   var sortDescIgnoreCase = function(a, b) {
148     a = a.title.toLowerCase();
149     b = b.title.toLowerCase();
150     if (a < b)
151         return 1;
152     if (a > b)
153         return -1;
154     return 0;
155   };
156 
157   keys.sort(sortAscIgnoreCase);
158   if (!sortAsc) {
159       keys.reverse(sortDescIgnoreCase);
160   }
161   return keys;
162 };
163 
164 ZmMailListFromGroup.prototype._getSectionTitle =
165 function(participant) {
166     var sectionHeader = participant.getAddress();
167     var name = participant.getName();
168     if (name) {
169         sectionHeader = name;
170     }
171     return sectionHeader;
172 };
173 
174 ZmMailListFromGroup.prototype._getSectionHeaderTitle =
175 function(section) {
176     return this._sectionTitle[section].title;
177 };
