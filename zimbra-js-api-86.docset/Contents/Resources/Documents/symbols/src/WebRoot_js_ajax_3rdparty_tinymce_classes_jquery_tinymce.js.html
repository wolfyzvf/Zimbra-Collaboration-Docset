 11 /*global tinymce:true, jQuery */

 13 (function($) {
 14 	var undef,
 15 		lazyLoading,
 16 		delayedInits = [],
 17 		win = window;
 18 
 19 	$.fn.tinymce = function(settings) {
 20 		var self = this, url, base, lang, suffix = "", patchApplied;
 21 
 22 		// No match then just ignore the call
 23 		if (!self.length) {
 24 			return self;
 25 		}
 26 
 27 		// Get editor instance
 28 		if (!settings) {
 29 			return window.tinymce ? tinymce.get(self[0].id) : null;
 30 		}
 31 
 32 		self.css('visibility', 'hidden'); // Hide textarea to avoid flicker
 33 
 34 		function init() {
 35 			var editors = [], initCount = 0;
 36 
 37 			// Apply patches to the jQuery object, only once
 38 			if (!patchApplied) {
 39 				applyPatch();
 40 				patchApplied = true;
 41 			}
 42 
 43 			// Create an editor instance for each matched node
 44 			self.each(function(i, node) {
 45 				var ed, id = node.id, oninit = settings.oninit;
 46 
 47 				// Generate unique id for target element if needed
 48 				if (!id) {
 49 					node.id = id = tinymce.DOM.uniqueId();
 50 				}
 51 
 52 				// Only init the editor once
 53 				if (tinymce.get(id)) {
 54 					return;
 55 				}
 56 
 57 				// Create editor instance and render it
 58 				ed = new tinymce.Editor(id, settings, tinymce.EditorManager);
 59 				editors.push(ed);
 60 
 61 				ed.on('init', function() {
 62 					var scope, func = oninit;
 63 
 64 					self.css('visibility', '');
 65 
 66 					// Run this if the oninit setting is defined
 67 					// this logic will fire the oninit callback ones each
 68 					// matched editor instance is initialized
 69 					if (oninit) {
 70 						// Fire the oninit event ones each editor instance is initialized
 71 						if (++initCount == editors.length) {
 72 							if (typeof(func) === "string") {
 73 								scope = (func.indexOf(".") === -1) ? null : tinymce.resolve(func.replace(/\.\w+$/, ""));
 74 								func = tinymce.resolve(func);
 75 							}
 76 
 77 							// Call the oninit function with the object
 78 							func.apply(scope || tinymce, editors);
 79 						}
 80 					}
 81 				});
 82 			});
 83 
 84 			// Render the editor instances in a separate loop since we
 85 			// need to have the full editors array used in the onInit calls
 86 			$.each(editors, function(i, ed) {
 87 				ed.render();
 88 			});
 89 		}
 90 
 91 		// Load TinyMCE on demand, if we need to
 92 		if (!win.tinymce && !lazyLoading && (url = settings.script_url)) {
 93 			lazyLoading = 1;
 94 			base = url.substring(0, url.lastIndexOf("/"));
 95 
 96 			// Check if it's a dev/src version they want to load then
 97 			// make sure that all plugins, themes etc are loaded in source mode as well
 98 			if (url.indexOf('.min') != -1) {
 99 				suffix = ".min";
100 			}
101 
102 			// Setup tinyMCEPreInit object this will later be used by the TinyMCE
103 			// core script to locate other resources like CSS files, dialogs etc
104 			// You can also predefined a tinyMCEPreInit object and then it will use that instead
105 			win.tinymce = win.tinyMCEPreInit || {
106 				base: base,
107 				suffix: suffix
108 			};
109 
110 			// url contains gzip then we assume it's a compressor
111 			if (url.indexOf('gzip') != -1) {
112 				lang = settings.language || "en";
113 				url = url + (/\?/.test(url) ? '&' : '?') + "js=true&core=true&suffix=" + escape(suffix) +
114 					"&themes=" + escape(settings.theme || 'modern') + "&plugins=" +
115 					escape(settings.plugins || '') + "&languages=" + (lang || '');
116 
117 				// Check if compressor script is already loaded otherwise setup a basic one
118 				if (!win.tinyMCE_GZ) {
119 					win.tinyMCE_GZ = {
120 						start: function() {
121 							function load(url) {
122 								tinymce.ScriptLoader.markDone(tinymce.baseURI.toAbsolute(url));
123 							}
124 
125 							// Add core languages
126 							load("langs/" + lang + ".js");
127 
128 							// Add themes with languages
129 							load("themes/" + settings.theme + "/theme" + suffix + ".js");
130 							load("themes/" + settings.theme + "/langs/" + lang + ".js");
131 
132 							// Add plugins with languages
133 							$.each(settings.plugins.split(","), function(i, name) {
134 								if (name) {
135 									load("plugins/" + name + "/plugin" + suffix + ".js");
136 									load("plugins/" + name + "/langs/" + lang + ".js");
137 								}
138 							});
139 						},
140 
141 						end: function() {
142 						}
143 					};
144 				}
145 			}
146 
147 			var script = document.createElement('script');
148 			script.type = 'text/javascript';
149 			script.onload = script.onreadystatechange = function(e) {
150 				e = e || window.event;
151 
152 				if (lazyLoading !== 2 && (e.type == 'load' || /complete|loaded/.test(script.readyState))) {
153 					tinymce.dom.Event.domLoaded = 1;
154 					lazyLoading = 2;
155 
156 					// Execute callback after mainscript has been loaded and before the initialization occurs
157 					if (settings.script_loaded) {
158 						settings.script_loaded();
159 					}
160 
161 					init();
162 
163 					$.each(delayedInits, function(i, init) {
164 						init();
165 					});
166 				}
167 			};
168 			script.src = url;
169 			document.body.appendChild(script);
170 		} else {
171 			// Delay the init call until tinymce is loaded
172 			if (lazyLoading === 1) {
173 				delayedInits.push(init);
174 			} else {
175 				init();
176 			}
177 		}
178 
179 		return self;
180 	};
181 
182 	// Add :tinymce psuedo selector this will select elements that has been converted into editor instances
183 	// it's now possible to use things like $('*:tinymce') to get all TinyMCE bound elements.
184 	$.extend($.expr[":"], {
185 		tinymce: function(e) {
186 			return !!(e.id && "tinymce" in window && tinymce.get(e.id));
187 		}
188 	});
189 
190 	// This function patches internal jQuery functions so that if
191 	// you for example remove an div element containing an editor it's
192 	// automatically destroyed by the TinyMCE API
193 	function applyPatch() {
194 		// Removes any child editor instances by looking for editor wrapper elements
195 		function removeEditors(name) {
196 			// If the function is remove
197 			if (name === "remove") {
198 				this.each(function(i, node) {
199 					var ed = tinyMCEInstance(node);
200 
201 					if (ed) {
202 						ed.remove();
203 					}
204 				});
205 			}
206 
207 			this.find("span.mceEditor,div.mceEditor").each(function(i, node) {
208 				var ed = tinymce.get(node.id.replace(/_parent$/, ""));
209 
210 				if (ed) {
211 					ed.remove();
212 				}
213 			});
214 		}
215 
216 		// Loads or saves contents from/to textarea if the value
217 		// argument is defined it will set the TinyMCE internal contents
218 		function loadOrSave(value) {
219 			var self = this, ed;
220 
221 			// Handle set value
222 			/*jshint eqnull:true */
223 			if (value != null) {
224 				removeEditors.call(self);
225 
226 				// Saves the contents before get/set value of textarea/div
227 				self.each(function(i, node) {
228 					var ed;
229 
230 					if ((ed = tinymce.get(node.id))) {
231 						ed.setContent(value);
232 					}
233 				});
234 			} else if (self.length > 0) {
235 				// Handle get value
236 				if ((ed = tinymce.get(self[0].id))) {
237 					return ed.getContent();
238 				}
239 			}
240 		}
241 
242 		// Returns tinymce instance for the specified element or null if it wasn't found
243 		function tinyMCEInstance(element) {
244 			var ed = null;
245 
246 			if (element && element.id && win.tinymce) {
247 				ed = tinymce.get(element.id);
248 			}
249 
250 			return ed;
251 		}
252 
253 		// Checks if the specified set contains tinymce instances
254 		function containsTinyMCE(matchedSet) {
255 			return !!((matchedSet) && (matchedSet.length) && (win.tinymce) && (matchedSet.is(":tinymce")));
256 		}
257 
258 		// Patch various jQuery functions
259 		var jQueryFn = {};
260 
261 		// Patch some setter/getter functions these will
262 		// now be able to set/get the contents of editor instances for
263 		// example $('#editorid').html('Content'); will update the TinyMCE iframe instance
264 		$.each(["text", "html", "val"], function(i, name) {
265 			var origFn = jQueryFn[name] = $.fn[name],
266 				textProc = (name === "text");
267 
268 			$.fn[name] = function(value) {
269 				var self = this;
270 
271 				if (!containsTinyMCE(self)) {
272 					return origFn.apply(self, arguments);
273 				}
274 
275 				if (value !== undef) {
276 					loadOrSave.call(self.filter(":tinymce"), value);
277 					origFn.apply(self.not(":tinymce"), arguments);
278 
279 					return self; // return original set for chaining
280 				} else {
281 					var ret = "";
282 					var args = arguments;
283 
284 					(textProc ? self : self.eq(0)).each(function(i, node) {
285 						var ed = tinyMCEInstance(node);
286 
287 						if (ed) {
288 							ret += textProc ? ed.getContent().replace(/<(?:"[^"]*"|'[^']*'|[^'">])*>/g, "") : ed.getContent({save: true});
289 						} else {
290 							ret += origFn.apply($(node), args);
291 						}
292 					});
293 
294 					return ret;
295 				}
296 			};
297 		});
298 
299 		// Makes it possible to use $('#id').append("content"); to append contents to the TinyMCE editor iframe
300 		$.each(["append", "prepend"], function(i, name) {
301 			var origFn = jQueryFn[name] = $.fn[name],
302 				prepend = (name === "prepend");
303 
304 			$.fn[name] = function(value) {
305 				var self = this;
306 
307 				if (!containsTinyMCE(self)) {
308 					return origFn.apply(self, arguments);
309 				}
310 
311 				if (value !== undef) {
312 					self.filter(":tinymce").each(function(i, node) {
313 						var ed = tinyMCEInstance(node);
314 
315 						if (ed) {
316 							ed.setContent(prepend ? value + ed.getContent() : ed.getContent() + value);
317 						}
318 					});
319 
320 					origFn.apply(self.not(":tinymce"), arguments);
321 
322 					return self; // return original set for chaining
323 				}
324 			};
325 		});
326 
327 		// Makes sure that the editor instance gets properly destroyed when the parent element is removed
328 		$.each(["remove", "replaceWith", "replaceAll", "empty"], function(i, name) {
329 			var origFn = jQueryFn[name] = $.fn[name];
330 
331 			$.fn[name] = function() {
332 				removeEditors.call(this, name);
333 
334 				return origFn.apply(this, arguments);
335 			};
336 		});
337 
338 		jQueryFn.attr = $.fn.attr;
339 
340 		// Makes sure that $('#tinymce_id').attr('value') gets the editors current HTML contents
341 		$.fn.attr = function(name, value) {
342 			var self = this, args = arguments;
343 
344 			if ((!name) || (name !== "value") || (!containsTinyMCE(self))) {
345 				if (value !== undef) {
346 					return jQueryFn.attr.apply(self, args);
347 				} else {
348 					return jQueryFn.attr.apply(self, args);
349 				}
350 			}
351 
352 			if (value !== undef) {
353 				loadOrSave.call(self.filter(":tinymce"), value);
354 				jQueryFn.attr.apply(self.not(":tinymce"), args);
355 
356 				return self; // return original set for chaining
357 			} else {
358 				var node = self[0], ed = tinyMCEInstance(node);
359 
360 				return ed ? ed.getContent({save: true}) : jQueryFn.attr.apply($(node), args);
361 			}
362 		};
363 	}
364 })(jQuery);
