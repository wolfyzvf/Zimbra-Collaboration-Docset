 27 /**
 28  * Creates a request manager.
 29  * @class
 30  * This class manages the sending of requests to the server, and handles the
 31  * responses, including refresh blocks and notifications.
 32  *
 33  * @author Conrad Damon
 34  * 
 35  * @param {ZmController}	controller	the main controller
 36  */
 37 ZmRequestMgr = function(controller) {
 38 
 39 	this._controller = controller;
 40 	
 41 	appCtxt.setRequestMgr(this);
 42 
 43 	ZmCsfeCommand.setServerUri(appCtxt.get(ZmSetting.CSFE_SERVER_URI));
 44 	var cv = appCtxt.get(ZmSetting.CLIENT_VERSION);
 45 	ZmCsfeCommand.clientVersion = (!cv || cv.indexOf('@') == 0) ? "dev build" : cv;
 46 	
 47 	this._shell = appCtxt.getShell();
 48 
 49     this._highestNotifySeen = 0;
 50 
 51 	this._cancelActionId = {};
 52 	this._pendingRequests = {};
 53 
 54 	this._useXml = appCtxt.get(ZmSetting.USE_XML);
 55 	this._logRequest = appCtxt.get(ZmSetting.LOG_REQUEST);
 56 	this._stdTimeout = appCtxt.get(ZmSetting.TIMEOUT);
 57 
 58 	this._unreadListener = new AjxListener(this, this._unreadChangeListener);
 59 };
 60 
 61 // request states
 62 ZmRequestMgr._SENT		= 1;
 63 ZmRequestMgr._RESPONSE	= 2;
 64 ZmRequestMgr._CANCEL	= 3;
 65 
 66 // retry settings
 67 ZmRequestMgr.RETRY_MAX			= 2;	// number of times to retry before throwing exception
 68 ZmRequestMgr.RETRY_DELAY		= 5;	// seconds to delay between retries
 69 ZmRequestMgr.RETRY_ON_EXCEPTION = {};	// which exceptions to retry on
 70 ZmRequestMgr.RETRY_ON_EXCEPTION[ZmCsfeException.EMPTY_RESPONSE] = true;
 71 
 72 ZmRequestMgr._nextReqId = 1;
 73 
 74 ZmRequestMgr.OFFLINE_HEAP_DUMP          = "heapdump_upload";
 75 ZmRequestMgr.OFFLINE_MUST_RESYNC        = "resync";
 76 ZmRequestMgr.OFFLINE_MUST_GAL_RESYNC        = "gal_resync";
 77 ZmRequestMgr.OFFLINE_FOLDER_MOVE_FAILED = "foldermove_failed";

 79  * Returns a string representation of the object.

 81  * @return		{String}		a string representation of the object
 82  */
 83 ZmRequestMgr.prototype.toString =
 84 function() {
 85 	return "ZmRequestMgr";

 88 /**
 89  * Sends a request to the CSFE and processes the response. Notifications and
 90  * refresh blocks that come in the response header are handled. Also handles
 91  * exceptions by default, though the caller can pass in a special callback to
 92  * run for exceptions. The error callback should return true if it has
 93  * handled the exception, and false if standard exception handling should still
 94  * be performed.
 95  *
 96  * @param {Hash}		params				a hash of parameters
 97  * @param {AjxSoapDoc}	soapDoc				the SOAP document that represents the request
 98  * @param {Object}		jsonObj				the JSON object that represents the request (alternative to soapDoc)
 99  * @param {Boolean}		asyncMode			if <code>true</code>, request will be made asynchronously
100  * @param {AjxCallback}	callback			the next callback in chain for async request
101  * @param {AjxCallback}	errorCallback		the callback to run if there is an exception
102  * @param {AjxCallback}	continueCallback	the callback to run after user re-auths
103  * @param {AjxCallback}	offlineCallback	    the callback to run if the user is offline
104  * @param {int}			timeout				the timeout value (in seconds)
105  * @param {Boolean}		noBusyOverlay		if <code>true</code>, don't use the busy overlay
106  * @param {String}		accountName			the name of account to execute on behalf of
107  * @param {Object}		response			the pre-determined response (no request will be made)
108  * @param {Boolean}		skipAuthCheck		if <code>true</code>, do not check if auth token has changed
109  * @param {constant}	resend				the reason for resending request
110  * @param {Boolean}		sensitive			if <code>true</code>, attempt to use secure conn to protect data
111  * @param {Boolean}		noSession			if <code>true</code>, no session info is included
112  * @param {String}		restUri				the REST URI to send the request to
113  * @param {boolean}		emptyResponseOkay	if true, empty or no response from server is not an erro
114  * @param {boolean}		offlineRequest	    if true, request will not be send to server
115  * @param {boolean}		useChangeToken	    if true, request will try to use change token in header
116  */
117 ZmRequestMgr.prototype.sendRequest =
118 function(params) {
119 	var response = params.response;
120 	if (response) {
121 		if (params.reqId) {
122 			params = this._pendingRequests[params.reqId] || params;
123 			params.response = response;
124 		}
125 		params.asyncMode = true;	// canned response set up async style
126 		return this._handleResponseSendRequest(params, new ZmCsfeResult(response));
127 	}
128 	if (params.offlineRequest || appCtxt.isWebClientOffline()) {
129 		if (params.offlineCallback) {
130 			params.offlineCallback.run(params);
131 		}
132 		return;
133 	}
134 	
135 	var reqId = params.reqId = ("Req_"+ZmRequestMgr._nextReqId++);
136 	DBG.println("req", "assign req ID: " + reqId);
137 	var timeout = params.timeout = (params.timeout != null) ? params.timeout : this._stdTimeout;
138 	if (timeout) {
139 		timeout = timeout * 1000; // convert seconds to ms
140 	}
141 	var asyncCallback = params.asyncMode ? new AjxCallback(this, this._handleResponseSendRequest, [params]) : null;
142 
143 	if (params.sensitive) {
144 		DBG.println(AjxDebug.DBG2, "request contains sensitive data");
145 		// NOTE: If only http mode is available, there's nothing we can
146 		//       do. And if we're already using https mode, then there's
147 		//       nothing we need to do. We only attempt to send the
148 		//       request securely if mixed mode is enabled and the app
149 		//       was loaded using http.
150 		var isHttp = document.location.protocol == ZmSetting.PROTO_HTTP;
151 		var isMixedMode = appCtxt.get(ZmSetting.PROTOCOL_MODE) == ZmSetting.PROTO_MIXED;
152 		if(isHttp && isMixedMode) {
153 			return this._sensitiveRequest(params, reqId);
154 		}
155 	}
156 
157 	var command = new ZmCsfeCommand();
158 	// bug fix #10652, 82704 - dont set change token if accountName is not main account or is not specified
159 	// (since we're executing on someone else's mbox)
160 	var accountName = params.accountName;
161 	if (!accountName) {
162 		var acct = appCtxt.getActiveAccount();
163 		accountName = (acct && acct.id != ZmAccountList.DEFAULT_ID) ? acct.name : null;
164 	}
165 	var changeToken = null;
166 	if (params.useChangeToken && (!accountName || (accountName === appCtxt.accountList.mainAccount.name))) {
167 		changeToken = this._changeToken;
168 	}
169 	var cmdParams, methodName;
170 
171 	if (params.restUri) {
172 		cmdParams =	{	restUri:			params.restUri,
173 						asyncMode:			params.asyncMode,
174 						callback:			asyncCallback
175 					};
176 	} else {
177 		cmdParams = {	jsonObj:			params.jsonObj,
178 						soapDoc:			params.soapDoc,
179 						accountName:		accountName,
180 						useXml:				this._useXml,
181 						changeToken:		changeToken,
182 						asyncMode:			params.asyncMode,
183 						callback:			asyncCallback,
184 						logRequest:			this._logRequest,
185 						highestNotifySeen:	this._highestNotifySeen,
186 						noAuthToken:		true,	// browser will handle auth token cookie
187 						skipAuthCheck:		params.skipAuthCheck,
188 						resend:				params.resend,
189 						noSession:			params.noSession,
190 						useStringify1:		(AjxEnv.isIE || AjxEnv.isModernIE) && params.fromChildWindow,
191 						emptyResponseOkay:	params.emptyResponseOkay
192 					};
193 		methodName = params.methodName = ZmCsfeCommand.getMethodName(cmdParams.jsonObj || cmdParams.soapDoc);
194 	}
195 
196 	appCtxt.currentRequestParams = params;
197 	DBG.println("req", "send request " + reqId + ": " + methodName);
198 	var cancelParams = timeout ? [reqId, params.errorCallback, params.noBusyOverlay] : null;
199 	if (!params.noBusyOverlay) {
200 		var cancelCallback = null;
201 		var showBusyDialog = false;
202 		if (timeout) {
203 			DBG.println("req", "ZmRequestMgr.sendRequest: timeout for " + reqId + " is " + timeout);
204 			cancelCallback = new AjxCallback(this, this.cancelRequest, cancelParams);
205 			this._shell.setBusyDialogText(ZmMsg.askCancel);
206 			showBusyDialog = true;
207 		}
208 		// put up busy overlay to block user input
209 		this._shell.setBusy(true, reqId, showBusyDialog, timeout, cancelCallback);
210 	} else if (timeout) {
211 		var action = new AjxTimedAction(this, this.cancelRequest, cancelParams);
212 		this._cancelActionId[reqId] = AjxTimedAction.scheduleAction(action, timeout);
213 		DBG.println("req", "schedule cancel action for reqId " + reqId + ": " + this._cancelActionId[reqId]);
214 	}
215 
216 	this._pendingRequests[reqId] = command;
217 
218 	try {
219 		DBG.println("req", "invoke req: " + params.reqId);
220 		var response = params.restUri ? command.invokeRest(cmdParams) : command.invoke(cmdParams);
221 		command.state = ZmRequestMgr._SENT;
222 	} catch (ex) {
223 		DBG.println("req", "caught exception on invoke of req: " + params.reqId);
224 		this._handleResponseSendRequest(params, new ZmCsfeResult(ex, true));
225 		return;
226 	}
227 
228 	return (params.asyncMode) ? reqId : (this._handleResponseSendRequest(params, response));
229 };

231 /**
232  * @private
233  * @param {Array}	params.ignoreErrs	list of error codes that can be ignored, when params.errorCallback does not exists.
234  */
235 ZmRequestMgr.prototype._handleResponseSendRequest =
236 function(params, result) {
237 	DBG.println("req", "ZmRequestMgr.handleResponseSendRequest for req: " + params.reqId);
238 	var isCannedResponse = (params.response != null);
239 	if (!isCannedResponse && !appCtxt.isWebClientOffline()) {
240 		if (!this._pendingRequests[params.reqId]) {
241 			DBG.println("req", "ZmRequestMgr.handleResponseSendRequest no pending request for " + params.reqId);
242 			return;
243 		}
244 		if (this._pendingRequests[params.reqId].state == ZmRequestMgr._CANCEL) {
245 			DBG.println("req", "ZmRequestMgr.handleResponseSendRequest state=CANCEL for " + params.reqId);
246 			return;
247 		}
248 	
249 		this._pendingRequests[params.reqId].state = ZmRequestMgr._RESPONSE;
250 	
251 		if (!params.noBusyOverlay) {
252 			this._shell.setBusy(false, params.reqId); // remove busy overlay
253 		}
254 	}
255 
256 	var response, refreshBlock;
257 	try {
258 		if (params.asyncMode && !params.restUri) {
259 			response = result.getResponse(); // may throw exception
260 		} else {
261 			// for sync responses, manually throw exception if necessary
262 			if (result._isException) {
263 				throw result._data;
264 			} else {
265 				response = result;
266 			}
267 		}
268 		if (response.Header) {
269 			refreshBlock = this._handleHeader(response.Header);
270 		}
271 	} catch (ex) {
272 		DBG.println("req", "Request " + params.reqId + " got an exception");
273 		var ecb = params.errorCallback;
274 		if (ecb) {
275             var handled = ecb.run(ex, params);
276 			if (!handled) {
277 				this._handleException(ex, params);
278 			}
279 		} else {
280 			var ignore = function(ignoreErrs, errCode){
281 			/*
282 				Checks errCode exits in ignoreErrs
283 			*/
284 				if (ignoreErrs && (ignoreErrs.length > 0)){
285 					for (var val in ignoreErrs)
286 						if (ignoreErrs[val] == errCode) 
287 							return true;
288 				}
289 				return false;
290 			}(params.ignoreErrs, ex.code)
291 
292             if (ex.code === ZmCsfeException.EMPTY_RESPONSE && params.offlineCallback) {
293                 params.offlineCallback(params);
294 				if (appCtxt.isWebClientOffline() && !params.noBusyOverlay) {
295 					this._shell.setBusy(false, params.reqId); // remove busy overlay
296 				}
297                 ignore = true;
298             }
299 			if (!ignore)
300 				this._handleException(ex, params);
301 		}
302 		var hdr = result.getHeader();
303 		if (hdr) {
304 			this._handleHeader(hdr);
305 			this._handleNotifications(hdr);
306 		}
307 		this._clearPendingRequest(params.reqId);
308 		return;
309 	}
310 
311     if (params.asyncMode && !params.restUri) {
312 	    result.set(response.Body);
313 	}
314 
315     // if we didn't get an exception, then we should make sure that the
316     // poll timer is running (just in case it got an exception and stopped)
317 	if (!appCtxt.isOffline && !isCannedResponse) {
318 		this._controller._kickPolling(true);
319 	}
320 
321 	var methodName = ZmCsfeCommand.getMethodName(params.jsonObj || params.soapDoc);
322 	if (params.asyncMode && params.callback) {
323 		DBG.println(AjxDebug.DBG1, "------------------------- Running response callback for " + methodName);
324 		params.callback.run(result);
325 	}
326 
327 	DBG.println(AjxDebug.DBG1, "------------------------- Processing notifications for " + methodName);
328 	this._handleNotifications(response.Header, methodName);
329 
330 	this._clearPendingRequest(params.reqId);
331 
332 	if (refreshBlock && (!appCtxt.isOffline || !appCtxt.multiAccounts) && !params.more) {
333 		this._refreshHandler(refreshBlock);
334 	}
335 	
336 	if (!params.asyncMode) {
337 		return response.Body;
338 	}
339 	
340 	var ctlr = this._controller;
341 	if (ctlr._evtMgr && ctlr._evtMgr.isListenerRegistered(ZmAppEvent.RESPONSE)) {
342 		ctlr._evt.request = methodName;
343 		ctlr.notify(ZmAppEvent.RESPONSE);
344 	}
345 };

347 /**
348  * Cancels the request.
349  * 
350  * @param	{String}	reqId		the request id
351  * @param	{AjxCallback}	errorCallback		the callback
352  * @param	{Boolean}	noBusyOverlay	if <code>true</code>, do not show busy overlay
353  */
354 ZmRequestMgr.prototype.cancelRequest =
355 function(reqId, errorCallback, noBusyOverlay) {
356 	DBG.println("req", "ZmRequestMgr.cancelRequest: " + reqId);
357 	if (!this._pendingRequests[reqId]) { return; }
358 	if (this._pendingRequests[reqId].state == ZmRequestMgr._RESPONSE) { return; }
359 
360 	this._pendingRequests[reqId].state = ZmRequestMgr._CANCEL;
361 	if (!noBusyOverlay) {
362 		this._shell.setBusy(false, reqId);
363 	}
364 	DBG.println("req", "canceling the XHR");
365 	this._pendingRequests[reqId].cancel();
366 	if (errorCallback) {
367 		DBG.println("req", "calling the error callback");
368 		var ex = new AjxException("Request canceled", AjxException.CANCELED, "ZmRequestMgr.prototype.cancelRequest");
369 		errorCallback.isAjxCallback ? errorCallback.run(ex) : errorCallback(ex);
370 	}
371 	this._clearPendingRequest(reqId);
372 };
373 
374 /**
375  * @private
376  */
377 ZmRequestMgr.prototype._clearPendingRequest =
378 function(reqId) {
379 	var request = this._pendingRequests[reqId];
380 	if (request) {
381 		if (request.iframeId) {
382 			var iframe = document.getElementById(request.iframeId);
383 			if (iframe) {
384 				iframe.parentNode.removeChild(iframe);
385 			}
386 		}
387 		delete this._pendingRequests[reqId];
388 	}
389 	var cancelId = this._cancelActionId[reqId];
390 	if (cancelId && cancelId != -1) {
391 		DBG.println("req", "unschedule cancel action for reqId " + reqId + ": " + cancelId);
392 		AjxTimedAction.cancelAction(cancelId);
393 		this._cancelActionId[reqId] = -1;
394 	}
395 };

397 /**
398  * Handles a response's SOAP header, except for notifications. Updates our
399  * change token, and processes a <code><refresh></code> block if there is one (happens
400  * when a new session is created on the server).
401  *
402  * @param {Object}	hdr	a SOAP header
403  * 
404  * @private
405  */
406 ZmRequestMgr.prototype._handleHeader =
407 function(hdr) {
408 
409 	var ctxt = hdr && hdr.context;
410 	if (!ctxt) { return; }
411 
412 	// update change token if we got one
413 	if (ctxt.change) {
414 		this._changeToken = ctxt.change.token;
415 	}
416 
417 	// offline/zdesktop only
418 	if (ctxt.zdsync && ctxt.zdsync.account) {
419 		var acctList = ctxt.zdsync.account;
420 		for (var i = 0; i < acctList.length; i++) {
421             var acct = appCtxt.accountList.getAccount(acctList[i].id);
422 			if (acct) {
423                 //server is sending info to get user's consent on something.
424                 var dialog = acctList[i].dialog;
425                 if(dialog) {
426                     this._handleOfflineInfoDialog(dialog[0], acct)
427                 }
428 				acct.updateState(acctList[i]);
429 			}
430 		}
431 	}
432 
433 	if (ctxt.refresh) {
434 		this._controller.runAppFunction("_clearDeferredFolders");
435 		this._loadTrees(ctxt.refresh);
436 		this._controller.runAppFunction("_createVirtualFolders");
437 		this._highestNotifySeen = 0;
438 	}
439 
440 	return ctxt.refresh;
441 };
442 /**
443  * Handles server's notification to get user's consent on something
444  *
445  * @param {Object}	dlg is json object
446  * @param {Object}	account object
447  *
448  * @private
449  */
450 ZmRequestMgr.prototype._handleOfflineInfoDialog =
451 function(dlg, acct) {
452 
453     if(!dlg.type) {
454         return;
455     }
456     var cont;
457     switch(dlg.type) {
458         case ZmRequestMgr.OFFLINE_HEAP_DUMP: {
459             cont = ZmMsg.offlineHeapDump;
460             break;
461         }
462         case ZmRequestMgr.OFFLINE_MUST_RESYNC: {
463             cont = AjxMessageFormat.format(ZmMsg.offlineMustReSync, acct.name);
464             break;
465         }
466 		case ZmRequestMgr.OFFLINE_MUST_GAL_RESYNC: {
467 			cont = AjxMessageFormat.format(ZmMsg.offlineMustGalReSync, acct.name);
468 			break;
469 		}
470         case ZmRequestMgr.OFFLINE_FOLDER_MOVE_FAILED: {
471             appCtxt.setStatusMsg(ZmMsg.offlineMoveFolderError);
472             break;
473         }
474         default:
475     }
476     if (!cont) {
477         return;
478     }
479     var dialog = appCtxt.getOkCancelMsgDialog();
480     dialog.setMessage(cont);
481     dialog.registerCallback(DwtDialog.OK_BUTTON, this._handleOfflineDialogAction, this, [dialog, dlg.type, acct.id, true]);
482     dialog.registerCallback(DwtDialog.CANCEL_BUTTON, this._handleOfflineDialogAction, this, [dialog, dlg.type, acct.id, false]);
483     dialog.popup();
484 };
485 /**
486  * Sends DialogActionRequest with user's consent YES/NO
487  * @param {object} dlg is getOkCancelMsgDialog
488  * @param {string} type
489  * @param {string} acctId Account ID
490  * @param {boolean} action
491  */
492 ZmRequestMgr.prototype._handleOfflineDialogAction =
493 function(dlg, type, acctId, action) {
494     var args = {
495 			jsonObj: { DialogActionRequest: { _jsns: "urn:zimbraOffline", type: type, id:acctId, action: action ? "yes" : "no" } },
496             callback: new AjxCallback(this, this._handleOfflineDialogActionResp, dlg),
497 			errorCallback: new AjxCallback(this, this._handleOfflineDialogActionResp, dlg),
498 			asyncMode: true
499 		};
500     this.sendRequest(args);
501 };

508 ZmRequestMgr.prototype._handleOfflineDialogActionResp =

511         dlg.popdown();
512     }

515 /**
516  * For transient network exceptions, retry the request after a small delay.
517  * We will only retry a limited number of times.
518  * 
519  * @param {AjxException}	ex			the exception
520  * @param {Hash}	params		a hash of the original request params
521  */
522 ZmRequestMgr.prototype._handleException =
523 function(ex, params) {
524 	var handled = false;
525 	if (ZmRequestMgr.RETRY_ON_EXCEPTION[ex.code]) {
526 		params.retryCount = params.retryCount || 0;
527 		if (params.retryCount < ZmRequestMgr.RETRY_MAX) {
528 			DBG.println(AjxDebug.DBG1, "RETRY " + ex.method + " due to " + ex.code);
529 			params.resend = ZmCsfeCommand.RETRY;
530 			params.retryCount++;
531 			AjxTimedAction.scheduleAction(new AjxTimedAction(this, 
532 				function() {
533 					this.sendRequest(params);
534 				}), ZmRequestMgr.RETRY_DELAY * 1000);
535 			handled = true;
536 		}
537 	}
538 	
539 	if (!handled) {
540 		this._controller._handleException(ex, params);
541 	}
542 };

544 /**
545  * Handles the <code><notify></code> block of a response's SOAP header.
546  *
547  * @param {Object}	hdr	a SOAP header
548  * 
549  * @private
550  */
551 ZmRequestMgr.prototype._handleNotifications =
552 function(hdr, methodName) {
553 
554 	if (hdr && hdr.context && hdr.context.notify) {
555         for (var i = 0; i < hdr.context.notify.length; i++) {
556         	var notify = hdr.context.notify[i];
557         	var seq = notify.seq;
558             // BUG?  What if the array isn't in sequence-order?  Could we miss notifications?
559 			var sid = hdr.context && ZmCsfeCommand.extractSessionId(hdr.context.session);
560             if (notify.seq > this._highestNotifySeen && !(sid && ZmCsfeCommand._staleSession[sid])) {
561                 DBG.println(AjxDebug.DBG1, "Handling notification[" + i + "] seq=" + seq);
562                 this._highestNotifySeen = seq;
563                 this._notifyHandler(notify, methodName);
564             } else {
565             	DBG.println(AjxDebug.DBG1, "SKIPPING notification[" + i + "] seq=" + seq + " highestNotifySeen=" + this._highestNotifySeen);
566 	      	}
567     	}
568 	}
569 };

571 /**
572  * A <code><refresh></code> block is returned in a SOAP response any time the session ID has 
573  * changed. It always happens on the first SOAP command (GetInfoRequest).
574  * After that, it happens after a session timeout. 
575  * 
576  * @param {Object}	refresh	the refresh block (JSON)
577  * @private
578  */
579 ZmRequestMgr.prototype._refreshHandler =
580 function(refresh) {
581 
582 	DBG.println(AjxDebug.DBG1, "Handling REFRESH");
583 	AjxDebug.println(AjxDebug.NOTIFY, "REFRESH block received");
584 	if (!appCtxt.inStartup) {
585 		this._controller._execPoll();
586 	}
587 
588 	if (refresh.version) {
589 		if (!this._canceledReload) {
590 			var curVersion = appCtxt.get(ZmSetting.SERVER_VERSION);
591 			if (curVersion != refresh.version) {
592 				appCtxt.set(ZmSetting.SERVER_VERSION, refresh.version);
593 				if (curVersion) {
594 					var dlg = appCtxt.getYesNoMsgDialog();
595 					dlg.reset();
596 					dlg.registerCallback(DwtDialog.YES_BUTTON, this._reloadYesCallback, this, [dlg, curVersion, refresh.version]);
597 					dlg.registerCallback(DwtDialog.NO_BUTTON, this._reloadNoCallback, this, [dlg, refresh]);
598 					var msg = AjxMessageFormat.format(ZmMsg.versionChangeRestart, [curVersion, refresh.version]);
599 					dlg.setMessage(msg, DwtMessageDialog.WARNING_STYLE);
600 					dlg.popup();
601 					appCtxt.reloadAppCache(true);
602 					return;
603 				}
604 			}
605 		}
606 	}
607 
608 	if (!this._recentlyRefreshed) {
609 		// Run any app-requested refresh routines
610 		this._controller.runAppFunction("refresh", false, refresh);
611 		this._recentlyRefreshed = true;
612 		this._lastSkippedRefresh = null;
613 	} else {
614 		this._lastSkippedRefresh = refresh;
615 	}
616 
617 	if (!this._refreshTimer)
618 		this._refreshTimer = new AjxTimedAction(this, this._refreshTimeout);
619 
620 	AjxTimedAction.scheduleAction(this._refreshTimer, 5000);
621 };
622 
623 ZmRequestMgr.prototype._refreshTimeout =
624 function() {
625 	if (this._lastSkippedRefresh) {
626 		this._controller.runAppFunction("refresh", false, this._lastSkippedRefresh);
627 		this._lastSkippedRefresh = null;
628 	}
629 	this._recentlyRefreshed = false;
630 };

666 /**
667  * @private
668  */
669 ZmRequestMgr.prototype._loadTree =
670 function(type, unread, obj, objType, account) {
671 	var isTag = (type == ZmOrganizer.TAG);
672 	var tree = appCtxt.getTree(type, account);
673 	if (tree) {
674 		tree.reset();
675 	} else {
676 		tree = isTag ? new ZmTagTree(account) : new ZmFolderTree();
677 	}
678 	appCtxt.setTree(type, tree, account);
679 	tree.addChangeListener(this._unreadListener);
680 	tree.getUnreadHash(unread);
681 	tree.loadFromJs(obj, objType, account);
682 };

684 /**
685  * To handle notifications, we keep track of all the models in use. A model could
686  * be an item, a list of items, or an organizer tree. Currently we never get an
687  * organizer by itself.
688  * 
689  * @private
690  */
691 ZmRequestMgr.prototype._notifyHandler =
692 function(notify, methodName) {
693 	DBG.println(AjxDebug.DBG1, "Handling NOTIFY");
694 	AjxDebug.println(AjxDebug.NOTIFY, "Notification block:");
695 	AjxDebug.dumpObj(AjxDebug.NOTIFY, notify);
696 	this._controller.runAppFunction("preNotify", false, notify);
697 	if (notify.deleted && notify.deleted.id) {
698 		this._handleDeletes(notify.deleted);
699 	}
700 	if (notify.created) {
701 		this._handleCreates(notify.created);
702 	}
703 	if (notify.modified) {
704 		this._handleModifies(notify.modified);
705 	}
706 
707 	if (ZmOffline.isOnlineMode() && (notify.deleted || notify.created || notify.modified)) {
708 		appCtxt.webClientOfflineHandler.scheduleSyncRequest(notify, methodName);
709 	}
710 	this._controller.runAppFunction("postNotify", false, notify);
711 };

713 /**
714  * A delete notification hands us a list of IDs which could be anything. First, we
715  * run any app delete handlers. Any IDs which have been handled by an app will
716  * be nulled out. The generic handling here will be applied to the rest - the item is
717  * retrieved from the item cache and told it has been deleted.
718  *
719  * @param {Object}	deletes	the node containing all 'deleted' notifications
720  * 
721  * @private
722  */
723 ZmRequestMgr.prototype._handleDeletes =
724 function(deletes) {
725 	var ids = deletes.id.split(",");
726 	this._controller.runAppFunction("deleteNotify", false, ids);
727 
728 	for (var i = 0; i < ids.length; i++) {
729 		var id = ids[i];
730 		if (!id) { continue; }
731 		var item = appCtxt.cacheGet(id);
732 		DBG.println(AjxDebug.DBG2, "ZmRequestMgr: handling delete notif for ID " + id);
733 		if (item && item.notifyDelete) {
734 			item.notifyDelete();
735 			appCtxt.cacheRemove(id);
736 			item = null;
737 		}
738 	}
739 };

741 /**
742  * Create notifications hand us full XML nodes. First, we run any app
743  * create handlers, which will mark any create nodes that they handle. Remaining
744  * creates are handled here.
745  * 
746  * @param {Object}	creates	the node containing all 'created' notifications
747  * 
748  * @private
749  */
750 ZmRequestMgr.prototype._handleCreates =
751 function(creates) {
752 	this._controller.runAppFunction("createNotify", false, creates);
753 
754 	for (var name in creates) {
755         if (creates.hasOwnProperty(name)) {
756             var list = creates[name];
757             for (var i = 0; i < list.length; i++) {
758                 var create = list[i];
759                 if (create._handled) { continue; }
760                 // ignore create notif for item we already have (except tags, which can reuse IDs)
761                 if (appCtxt.cacheGet(create.id) && name != "tag") { continue; }
762 
763                 DBG.println(AjxDebug.DBG1, "ZmRequestMgr: handling CREATE for node: " + name);
764                 if (name == "tag") {
765                     var account = appCtxt.multiAccounts && ZmOrganizer.parseId(create.id).account;
766                     var tagTree = appCtxt.getTagTree(account);
767                     if (tagTree) {
768                         tagTree.root.notifyCreate(create);
769                     }
770                 } else if (name == "folder" || name == "search" || name == "link") {
771                     var parentId = create.l;
772                     var parent = appCtxt.getById(parentId);
773                     if (parent && parent.notifyCreate && parent.type != ZmOrganizer.TAG) { // bug #37148
774                         parent.notifyCreate(create, name);
775                     }
776                 }
777             }
778         }
779 	}
780 };

782 /**
783  * First, we run any app modify handlers, which will mark any nodes that
784  * they handle. Remaining modify notifications are handled here.
785  * 
786  * @param {Object}	modifies	the node containing all 'modified' notifications
787  * 
788  * @private
789  */
790 ZmRequestMgr.prototype._handleModifies =
791 function(modifies) {
792 
793 	this._controller.runAppFunction("modifyNotify", false, modifies);
794 
795 	for (var name in modifies) {
796 		if (name == "mbx") {
797 			var mboxes = modifies[name];
798 			for (var i = 0; i < mboxes.length; i++) {
799 				var mbox = mboxes[i];
800 				var acctId = mbox.acct;
801 				var account = acctId && appCtxt.accountList.getAccount(acctId);
802 				var setting = appCtxt.getSettings(account).getSetting(ZmSetting.QUOTA_USED);
803 				setting.notifyModify({_name:name, s:mbox.s, account:account});
804 			}
805 			continue;
806 		}
807 
808 		var list = modifies[name];
809 		for (var i = 0; i < list.length; i++) {
810 			var mod = list[i];
811 			if (mod._handled) { continue; }
812 			DBG.println(AjxDebug.DBG2, "ZmRequestMgr: handling modified notif for ID " + mod.id + ", node type = " + name);
813 			var item = appCtxt.cacheGet(mod.id);
814 
815 			// bug fix #31991 - for contact modifies, check the contact list
816 			// Since we lazily create ZmContact items, it wont be in the global cache.
817 			// TODO: move to contacts app
818 			if (!item && name == "cn" && AjxDispatcher.loaded("ContactsCore")) {
819 				var capp = appCtxt.getApp(ZmApp.CONTACTS);
820 				if (capp.isContactListLoaded()) {
821 					item = capp.getContactList().getById(mod.id);
822 				}
823 			}
824 
825 			if (item && item.notifyModify) {
826 				mod._isRemote = (name == "folder" && item.link);	// remote subfolder
827 				item.notifyModify(mod);
828 			}
829 		}
830 	}
831 };

833 /**
834  * Changes browser title if it's a folder or tag whose unread count has changed.
835  *
836  * @param ev	the event
837  * 
838  * @private
839  */
840 ZmRequestMgr.prototype._unreadChangeListener =
841 function(ev) {
842 	if (ev.event == ZmEvent.E_MODIFY) {
843 		var fields = ev.getDetail("fields");
844 		if (fields && fields[ZmOrganizer.F_UNREAD]) {
845 			var organizers = ev.getDetail("organizers");
846 			var organizer = organizers ? organizers[0] : null;
847 			var id = organizer ? (organizer.isSystem() ? organizer.nId : organizer.id) : null;
848 			var search = appCtxt.getCurrentSearch();
849 			if (search) {
850 				var searchFolder = appCtxt.multiAccounts && appCtxt.getById(search.folderId);
851 				var searchFolderId = (searchFolder && searchFolder.getAccount() == appCtxt.getActiveAccount())
852 					? searchFolder.nId : search.folderId;
853 
854 				if (id && (id == searchFolderId || id == search.tagId)) {
855 					Dwt.setTitle(search.getTitle());
856 				}
857 			}
858 			var mailApp = appCtxt.getApp(ZmApp.MAIL);
859 			if (mailApp) {
860 				mailApp.setNewMailNotice(organizer);
861 			}
862 		}
863 	}
864 };

866 ZmRequestMgr.prototype._sensitiveRequest =
867 function(params, reqId) {
868 	DBG.println(AjxDebug.DBG2, "sending request securely");
869 	// adjust command parameters
870 	// TODO: Because of timing issues, should we not use session info?
871 	// TODO: But for batch commands, some updates would not be seen immediately.
872 	// TODO: To avoid security warning, send response in URL; so limit length
873 	params.noSession = true;
874 
875 	// information
876 	var requestStr = ZmCsfeCommand.getRequestStr(params);
877 	var loc = document.location;
878 	var port = appCtxt.get(ZmSetting.HTTPS_PORT);
879 	if (port && port != ZmSetting.DEFAULT_HTTPS_PORT) {
880 		port = ":"+port;
881 	}
882 
883 	// create iframe
884 	var iframe = document.createElement("IFRAME");
885 	iframe.style.display = "none";
886 	iframe.id = Dwt.getNextId();
887 	document.body.appendChild(iframe);
888 
889 	// set contents
890 	var iframeDoc = Dwt.getIframeDoc(iframe);
891 	iframeDoc.write(
892 		"<form ",
893 			"id=",iframe.id,"-form ",
894 			"target=",iframe.id,"-iframe ",
895 			"method=POST ",
896 			"action='https://",loc.hostname,port,appContextPath,"/public/secureRequest.jsp'",
897 		">",
898 			"<input type=hidden name=reqId value='",reqId,"'>",
899 			"<textarea name=data>",
900 				AjxStringUtil.htmlEncode(requestStr),
901 			"</textarea>",
902 		"</form>",
903 		"<iframe name=",iframe.id,"-iframe></iframe>"
904 	);
905 	iframeDoc.close();
906 
907 	// save the params for the response
908 	params.iframeId = iframe.id;
909 	this._pendingRequests[reqId] = params;
910 
911 	// submit form
912 	var form = iframeDoc.getElementById(iframe.id+"-form");
913 	form.submit();
914 };
